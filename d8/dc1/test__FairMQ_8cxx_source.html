36 {

38  using std::move;

40  auto headerMessage = o2::pmr::getMessage(move(inputStack), targetResource);

43  parts.AddPart(move(headerMessage));

70 template <typename I, typename F>
71 auto forEach(I begin, I end, F&& function)
72 {
73 
74  using span = gsl::span<const std::byte>;
75  using SPAN_SIZE_TYPE = span::size_type;
76  using gsl::narrow_cast;
77  for (auto it = begin; it != end; ++it) {
78  std::byte* headerBuffer{nullptr};
79  SPAN_SIZE_TYPE headerBufferSize{0};
80  if (*it != nullptr) {
81  headerBuffer = reinterpret_cast<std::byte*>((*it)->GetData());
82  headerBufferSize = narrow_cast<SPAN_SIZE_TYPE>((*it)->GetSize());
83  }
84  ++it;
85  std::byte* dataBuffer{nullptr};
86  SPAN_SIZE_TYPE dataBufferSize{0};
87  if (*it != nullptr) {
88  dataBuffer = reinterpret_cast<std::byte*>((*it)->GetData());
89  dataBufferSize = narrow_cast<SPAN_SIZE_TYPE>((*it)->GetSize());
90  }
91 
92  // call the user provided function
93  function(span{headerBuffer, headerBufferSize}, span{dataBuffer, dataBufferSize});
94  }
95  return std::move(function);
96 }
