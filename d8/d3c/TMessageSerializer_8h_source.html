112 template <typename T>
113 inline void TMessageSerializer::serialize(FairTMessage& tm, const T* input, //
114  const TClass* cl, CacheStreamers streamers, //
115  CompressionLevel compressionLevel)
116 {
117  if (streamers == CacheStreamers::yes) {
118  tm.EnableSchemaEvolution(true);
119  }
120 
121  if (compressionLevel >= 0) {
122  // if negative, skip to use ROOT default
123  tm.SetCompressionLevel(compressionLevel);
124  }
125 
126  // TODO: check what WriateObject and WriteObjectAny are doing
127  if (cl == nullptr) {
128  tm.WriteObject(input);
129  } else {
130  tm.WriteObjectAny(input, cl);
131  }
132 
133  if (streamers == CacheStreamers::yes) {
134  updateStreamers(tm, sStreamers);
135  }
136 }

138 template <typename T>
139 inline std::unique_ptr<T> TMessageSerializer::deserialize(gsl::span<std::byte> buffer)
140 {
141  TClass* tgtClass = TClass::GetClass(typeid(T));
142  if (tgtClass == nullptr) {
143  throw runtime_error_f("class is not ROOT-serializable: %s", typeid(T).name());
144  }
145  // FIXME: we need to add consistency check for buffer data to be serialized
146  // at the moment, TMessage might simply crash if an invalid or inconsistent
147  // buffer is provided
148  FairTMessage tm(buffer);
149  TClass* serializedClass = tm.GetClass();
150  if (serializedClass == nullptr) {
151  throw runtime_error_f("can not read class info from buffer");
152  }
153  if (tgtClass != serializedClass && serializedClass->GetBaseClass(tgtClass) == nullptr) {
154  throw runtime_error_f("can not convert serialized class %s into target class %s",
155  tm.GetClass()->GetName(),
156  tgtClass->GetName());
157  }
158  return std::unique_ptr<T>(reinterpret_cast<T*>(tm.ReadObjectAny(serializedClass)));
159 }

173 inline void TMessageSerializer::Serialize(FairMQMessage& msg, const TObject* input,
174  TMessageSerializer::CacheStreamers streamers,
175  TMessageSerializer::CompressionLevel compressionLevel)
176 {
177  std::unique_ptr<FairTMessage> tm = std::make_unique<FairTMessage>(kMESS_OBJECT);

181  msg.Rebuild(tm->Buffer(), tm->BufferSize(), FairTMessage::free, tm.get());
