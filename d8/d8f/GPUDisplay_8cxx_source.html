202 void GPUDisplay::mAnimationCloseAngle(float& newangle, float lastAngle)
203 {
204  const float delta = lastAngle > newangle ? (2 * CAMath::Pi()) : (-2 * CAMath::Pi());
205  while (fabsf(newangle + delta - lastAngle) < fabsf(newangle - lastAngle)) {
206  newangle += delta;
207  }

209 void GPUDisplay::mAnimateCloseQuaternion(float* v, float lastx, float lasty, float lastz, float lastw)
210 {
211  float distPos2 = (lastx - v[0]) * (lastx - v[0]) + (lasty - v[1]) * (lasty - v[1]) + (lastz - v[2]) * (lastz - v[2]) + (lastw - v[3]) * (lastw - v[3]);
212  float distNeg2 = (lastx + v[0]) * (lastx + v[0]) + (lasty + v[1]) * (lasty + v[1]) + (lastz + v[2]) * (lastz + v[2]) + (lastw + v[3]) * (lastw + v[3]);
213  if (distPos2 > distNeg2) {
214  for (int i = 0; i < 4; i++) {
215  v[i] = -v[i];
216  }
217  }

219 void GPUDisplay::setAnimationPoint()
220 {
221  if (mCfgL.animationMode & 4) // Spherical
222  {
223  float rxy = sqrtf(mXYZ[0] * mXYZ[0] + mXYZ[2] * mXYZ[2]);
224  float anglePhi = atan2f(mXYZ[0], mXYZ[2]);
225  float angleTheta = atan2f(mXYZ[1], rxy);
226  if (mAnimateVectors[0].size()) {
227  mAnimationCloseAngle(anglePhi, mAnimateVectors[2].back());
228  }
229  if (mAnimateVectors[0].size()) {
230  mAnimationCloseAngle(angleTheta, mAnimateVectors[3].back());
231  }
232  mAnimateVectors[1].emplace_back(0);
233  mAnimateVectors[2].emplace_back(anglePhi);
234  mAnimateVectors[3].emplace_back(angleTheta);
235  } else {
236  for (int i = 0; i < 3; i++) {
237  mAnimateVectors[i + 1].emplace_back(mXYZ[i]);
238  }
239  // Cartesian
240  }
241  float r = sqrtf(mXYZ[0] * mXYZ[0] + mXYZ[1] * mXYZ[1] + mXYZ[2] * mXYZ[2]);
242  mAnimateVectors[4].emplace_back(r);
243  if (mCfgL.animationMode & 1) // Euler-angles
244  {
245  for (int i = 0; i < 3; i++) {
246  float newangle = mAngle[i];
247  if (mAnimateVectors[0].size()) {
248  mAnimationCloseAngle(newangle, mAnimateVectors[i + 5].back());
249  }
250  mAnimateVectors[i + 5].emplace_back(newangle);
251  }
252  mAnimateVectors[8].emplace_back(0);
253  } else { // Quaternions
254  float v[4];
255  createQuaternionFromMatrix(v, mViewMatrixP);
256  if (mAnimateVectors[0].size()) {
257  mAnimateCloseQuaternion(v, mAnimateVectors[5].back(), mAnimateVectors[6].back(), mAnimateVectors[7].back(), mAnimateVectors[8].back());
258  }
259  for (int i = 0; i < 4; i++) {
260  mAnimateVectors[i + 5].emplace_back(v[i]);
261  }
262  }
263  float delay = 0.f;
264  if (mAnimateVectors[0].size()) {
265  delay = mAnimateVectors[0].back() + ((int)(mAnimationDelay * 20)) / 20.f;
266  }
267  mAnimateVectors[0].emplace_back(delay);
268  mAnimateConfig.emplace_back(mCfgL);
269 }
