54 template <typename T>
55 bool readFromTree(TTree& tree, const std::string brname, T& dest, int ev = 0)
56 {
57  auto* br = tree.GetBranch(brname.c_str());
58  if (br && br->GetEntries() > ev) {
59  auto* ptr = &dest;
60  br->SetAddress(&ptr);
61  br->GetEntry(ev);
62  br->ResetAddress();
63  return true;
64  }
65  return false;
66 }

115 void CTFReaderSpec::stopReader()
116 {
117  if (!mFileFetcher) {
118  return;
119  }
120  LOGP(info, "CTFReader stops processing, {} files read, {} files failed", mFilesRead - mNFailedFiles, mNFailedFiles);
121  LOGP(info, "CTF reading total timing: Cpu: {:.3f} Real: {:.3f} s for {} TFs in {} loops",
122  mTimer.CpuTime(), mTimer.RealTime(), mCTFCounter, mFileFetcher->getNLoops());
123  mRunning = false;
124  mFileFetcher->stop();
125  mFileFetcher.reset();
126  mCTFTree.reset();
127  if (mCTFFile) {
128  mCTFFile->Close();
129  }
130  mCTFFile.reset();
131 }

145 void CTFReaderSpec::openCTFFile(const std::string& flname)
146 {
147  try {
148  mFilesRead++;
149  mCTFFile.reset(TFile::Open(flname.c_str()));
150  if (!mCTFFile || !mCTFFile->IsOpen() || mCTFFile->IsZombie()) {
151  throw std::runtime_error("failed to open CTF file");
152  }
153  mCTFTree.reset((TTree*)mCTFFile->Get(std::string(o2::base::NameConf::CTFTREENAME).c_str()));
154  if (!mCTFTree) {
155  throw std::runtime_error("failed to load CTF tree from");
156  }
157  } catch (const std::exception& e) {
158  LOG(error) << "Cannot process " << flname << ", reason: " << e.what();
159  mCTFTree.reset();
160  mCTFFile.reset();
161  mNFailedFiles++;
162  if (mFileFetcher) {
163  mFileFetcher->popFromQueue(mInput.maxLoops < 1);
164  }
165  }
166  mCurrTreeEntry = 0;
167 }

170 void CTFReaderSpec::run(ProcessingContext& pc)
171 {
172 
173  if (!mCTFCounter) { // RS FIXME: this is a temporary hack to avoid late-starting devices to lose the input
174  LOG(warning) << "This is a hack, sleeping 5 s at startup";
175  usleep(1000000);
176  }
177 
178  std::string tfFileName;
179  if (mCTFCounter >= mInput.maxTFs || (!mInput.ctfIDs.empty() && mSelIDEntry >= mInput.ctfIDs.size())) { // done
180  LOG(info) << "All CTFs from selected range were injected, stopping";
181  mRunning = false;
182  }
183 
184  while (mRunning) {
185  if (mCTFTree) { // there is a tree open with multiple CTF
186  if (mInput.ctfIDs.empty() || mInput.ctfIDs[mSelIDEntry] == mCTFCounter) { // no selection requested or matching CTF ID is found
187  LOG(debug) << "TF " << mCTFCounter << " of " << mInput.maxTFs << " loop " << mFileFetcher->getNLoops();
188  mSelIDEntry++;
189  processTF(pc);
190  break;
191  } else { // explict CTF ID selection list was provided and current entry is not selected
192  LOGP(info, "Skipping CTF${} ({} of {} in {})", mCTFCounter, mCurrTreeEntry, mCTFTree->GetEntries(), mCTFFile->GetName());
193  checkTreeEntries();
194  mCTFCounter++;
195  continue;
196  }
197  }
198  //
199  tfFileName = mFileFetcher->getNextFileInQueue();
200  if (tfFileName.empty()) {
201  if (!mFileFetcher->isRunning()) { // nothing expected in the queue
202  mRunning = false;
203  break;
204  }
205  usleep(5000); // wait 5ms for the files cache to be filled
206  continue;
207  }
208  LOG(info) << "Reading CTF input " << ' ' << tfFileName;
209  openCTFFile(tfFileName);
210  }
211 
212  if (!mRunning) {
213  pc.services().get<ControlService>().endOfStream();
214  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
215  stopReader();
216  }
217 }

220 void CTFReaderSpec::processTF(ProcessingContext& pc)
221 {
222  auto cput = mTimer.CpuTime();
223  mTimer.Start(false);
224 
225  CTFHeader ctfHeader;
226  if (!readFromTree(*(mCTFTree.get()), "CTFHeader", ctfHeader, mCurrTreeEntry)) {
227  throw std::runtime_error("did not find CTFHeader");
228  }
229  if (ctfHeader.creationTime == 0) { // try to repair header with ad hoc data
230  tryToFixCTFHeader(ctfHeader);
231  }
232 
233  LOG(info) << ctfHeader;
234 
235  // send CTF Header
236  pc.outputs().snapshot({"header"}, ctfHeader);
237  setMessageHeader(ctfHeader, "header", pc);
238 
239  processDetector<o2::itsmft::CTF>(DetID::ITS, ctfHeader, pc);
240  processDetector<o2::itsmft::CTF>(DetID::MFT, ctfHeader, pc);
241  processDetector<o2::emcal::CTF>(DetID::EMC, ctfHeader, pc);
242  processDetector<o2::hmpid::CTF>(DetID::HMP, ctfHeader, pc);
243  processDetector<o2::phos::CTF>(DetID::PHS, ctfHeader, pc);
244  processDetector<o2::tpc::CTF>(DetID::TPC, ctfHeader, pc);
245  processDetector<o2::trd::CTF>(DetID::TRD, ctfHeader, pc);
246  processDetector<o2::ft0::CTF>(DetID::FT0, ctfHeader, pc);
247  processDetector<o2::fv0::CTF>(DetID::FV0, ctfHeader, pc);
248  processDetector<o2::fdd::CTF>(DetID::FDD, ctfHeader, pc);
249  processDetector<o2::tof::CTF>(DetID::TOF, ctfHeader, pc);
250  processDetector<o2::mid::CTF>(DetID::MID, ctfHeader, pc);
251  processDetector<o2::mch::CTF>(DetID::MCH, ctfHeader, pc);
252  processDetector<o2::cpv::CTF>(DetID::CPV, ctfHeader, pc);
253  processDetector<o2::zdc::CTF>(DetID::ZDC, ctfHeader, pc);
254  processDetector<o2::ctp::CTF>(DetID::CTP, ctfHeader, pc);
255 
256  // send sTF acknowledge message
257  {
258  auto& stfDist = pc.outputs().make<o2::header::STFHeader>({"STFDist"});
259  stfDist.id = uint64_t(mCurrTreeEntry);
260  stfDist.firstOrbit = ctfHeader.firstTForbit;
261  stfDist.runNumber = uint32_t(ctfHeader.run);
262  setMessageHeader(ctfHeader, "STFDist", pc);
263  }
264 
265  auto entryStr = fmt::format("({} of {} in {})", mCurrTreeEntry, mCTFTree->GetEntries(), mCTFFile->GetName());
266  checkTreeEntries();
267  mTimer.Stop();
268  // do we need to way to respect the delay ?
269  long tNow = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
270  auto tDiff = tNow - mLastSendTime;
271  if (mCTFCounter) {
272  if (tDiff < mInput.delay_us) {
273  usleep(mInput.delay_us - tDiff); // respect requested delay before sending
274  }
275  } else {
276  mLastSendTime = tNow;
277  }
278  tNow = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
279  LOGP(info, "Read CTF#{} {} in {:.3f} s, {:.4f} s elapsed from previous CTF", mCTFCounter, entryStr, mTimer.CpuTime() - cput, 1e-6 * (tNow - mLastSendTime));
280  mLastSendTime = tNow;
281  mCTFCounter++;
282 }

299 void CTFReaderSpec::setMessageHeader(const CTFHeader& ctfHeader, const std::string& lbl, ProcessingContext& pc) const
300 {
301  auto* stack = pc.outputs().findMessageHeaderStack({lbl});
302  if (!stack) {
303  throw std::runtime_error(fmt::format("failed to find output message header stack for {}", lbl));
304  }
305  auto dh = const_cast<o2::header::DataHeader*>(o2::header::get<o2::header::DataHeader*>(stack));
306  dh->firstTForbit = ctfHeader.firstTForbit;
307  dh->tfCounter = mCTFCounter;
308  dh->runNumber = uint32_t(ctfHeader.run);
309  auto dph = const_cast<o2::framework::DataProcessingHeader*>(o2::header::get<o2::framework::DataProcessingHeader*>(stack));
310  dph->creation = ctfHeader.creationTime;
311 }

314 template <typename C>
315 void CTFReaderSpec::processDetector(DetID det, const CTFHeader& ctfHeader, ProcessingContext& pc) const
316 {
317  if (mInput.detMask[det]) {
318  const auto lbl = det.getName();
319  auto& bufVec = pc.outputs().make<std::vector<o2::ctf::BufferType>>({lbl}, ctfHeader.detectors[det] ? sizeof(C) : 0);
320  if (ctfHeader.detectors[det]) {
321  C::readFromTree(bufVec, *(mCTFTree.get()), lbl, mCurrTreeEntry);
322  } else if (!mInput.allowMissingDetectors) {
323  throw std::runtime_error(fmt::format("Requested detector {} is missing in the CTF", lbl));
324  }
325  setMessageHeader(ctfHeader, lbl, pc);
326  }

377 DataProcessorSpec getCTFReaderSpec(const CTFReaderInp& inp)
378 {
379  std::vector<OutputSpec> outputs;
380 
381  outputs.emplace_back(OutputLabel{"header"}, "CTF", "HEADER", 0, Lifetime::Timeframe);
382  for (auto id = DetID::First; id <= DetID::Last; id++) {
383  if (inp.detMask[id]) {
384  DetID det(id);
385  outputs.emplace_back(OutputLabel{det.getName()}, det.getDataOrigin(), "CTFDATA", 0, Lifetime::Timeframe);
386  }
387  }
388  outputs.emplace_back(OutputSpec{{"STFDist"}, o2::header::gDataOriginFLP, o2::header::gDataDescriptionDISTSTF, 0});
389 
390  return DataProcessorSpec{
391  "ctf-reader",
392  Inputs{},
393  outputs,
394  AlgorithmSpec{adaptFromTask<CTFReaderSpec>(inp)},
395  Options{{"select-ctf-ids", VariantType::String, "", {"comma-separated list CTF IDs to inject (from cumulative counter of CTFs seen)"}}}};
396 }
