57 bool SimConfig::resetFromParsedMap(boost::program_options::variables_map const& vm)
58 {
59  using o2::detectors::DetID;
60  mConfigData.mMCEngine = vm["mcEngine"].as<std::string>();
61  mConfigData.mActiveDetectors = vm["modules"].as<std::vector<std::string>>();
62  auto& active = mConfigData.mActiveDetectors;
63  if (active.size() == 1 && active[0] == "all") {
64  active.clear();
65  for (int d = DetID::First; d <= DetID::Last; ++d) {
66 #ifdef ENABLE_UPGRADES
67  if (d != DetID::IT3 && d != DetID::TRK && d != DetID::FT3) {
68  active.emplace_back(DetID::getName(d));
69  }
70 #else
71  active.emplace_back(DetID::getName(d));
72 #endif
73  }
74  // add passive components manually (make a PassiveDetID for them!)
75  active.emplace_back("HALL");
76  active.emplace_back("MAG");
77  active.emplace_back("DIPO");
78  active.emplace_back("COMP");
79  active.emplace_back("PIPE");
80  active.emplace_back("ABSO");
81  active.emplace_back("SHIL");
82  }
83  // now we take out detectors listed as skipped
84  auto& skipped = vm["skipModules"].as<std::vector<std::string>>();
85  for (auto& s : skipped) {
86  auto iter = std::find(active.begin(), active.end(), s);
87  if (iter != active.end()) {
88  // take it out
89  active.erase(iter);
90  }
91  }
92 
93  mConfigData.mGenerator = vm["generator"].as<std::string>();
94  mConfigData.mTrigger = vm["trigger"].as<std::string>();
95  mConfigData.mNEvents = vm["nEvents"].as<unsigned int>();
96  mConfigData.mExtKinFileName = vm["extKinFile"].as<std::string>();
97  mConfigData.mEmbedIntoFileName = vm["embedIntoFile"].as<std::string>();
98  mConfigData.mStartEvent = vm["startEvent"].as<unsigned int>();
99  mConfigData.mBMax = vm["bMax"].as<float>();
100  mConfigData.mIsMT = vm["isMT"].as<bool>();
101  mConfigData.mOutputPrefix = vm["outPrefix"].as<std::string>();
102  mConfigData.mLogSeverity = vm["logseverity"].as<std::string>();
103  mConfigData.mLogVerbosity = vm["logverbosity"].as<std::string>();
104  mConfigData.mKeyValueTokens = vm["configKeyValues"].as<std::string>();
105  mConfigData.mConfigFile = vm["configFile"].as<std::string>();
106  mConfigData.mPrimaryChunkSize = vm["chunkSize"].as<unsigned int>();
107  mConfigData.mInternalChunkSize = vm["chunkSizeI"].as<int>();
108  mConfigData.mStartSeed = vm["seed"].as<int>();
109  mConfigData.mSimWorkers = vm["nworkers"].as<int>();
110  if (vm.count("timestamp")) {
111  mConfigData.mTimestamp = vm["timestamp"].as<uint64_t>();
112  } else {
113  mConfigData.mTimestamp = std::chrono::time_point_cast<std::chrono::milliseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
114  }
115  mConfigData.mCCDBUrl = vm["CCDBUrl"].as<std::string>();
116  mConfigData.mAsService = vm["asservice"].as<bool>();
117  if (vm.count("noemptyevents")) {
118  mConfigData.mFilterNoHitEvents = true;
119  }
120  mConfigData.mField = std::stoi((vm["field"].as<std::string>()).substr(0, (vm["field"].as<std::string>()).rfind("U")));
121  mConfigData.mUniformField = (vm["field"].as<std::string>()).find("U") != std::string::npos;
122  return true;
123 }

125 bool SimConfig::resetFromArguments(int argc, char* argv[])
126 {
127  namespace bpo = boost::program_options;
128 
129  // Arguments parsing
130  bpo::variables_map vm;
131  bpo::options_description desc("Allowed options");
132  desc.add_options()("help,h", "Produce help message.");
133  initOptions(desc);
134 
135  try {
136  bpo::store(parse_command_line(argc, argv, desc), vm);
137 
138  // help
139  if (vm.count("help")) {
140  std::cout << desc << std::endl;
141  return false;
142  }
143 
144  bpo::notify(vm);
145  } catch (const bpo::error& e) {
146  std::cerr << e.what() << "\n\n";
147  std::cerr << "Error parsing command line arguments; Available options:\n";
148 
149  std::cerr << desc << std::endl;
150  return false;
151  }
152 
153  return resetFromParsedMap(vm);
154 }

175  "outPrefix,o", bpo::value<std::string>(&data.outputPrefix)->default_value("o2sim"), "prefix of output files")(
176  "outDir,d", bpo::value<std::string>(&data.outputDir), "directory where to put simulation output (created when non-existant")(
177  "configKeyValues", bpo::value<std::string>(&data.keyValueTokens)->default_value(""), "semicolon separated key=value strings (e.g.: 'TPC.gasDensity=1;...")(
178  "configFile", bpo::value<std::string>(&data.configFile)->default_value(""), "Path to an INI or JSON configuration file")(

186  .options(options)

189  bpo::notify(vm);
190  } catch (const bpo::error& e) {
191  std::cerr << e.what() << "\n\n";
192  std::cerr << "Error parsing ReConfig data; Available options:\n";
193  std::cerr << options << std::endl;
194  return false;
195  }
196  return true;
