36 inline void TrackletsParser::swapByteOrder(unsigned int& ui)
37 {
38  ui = (ui >> 24) |
39  ((ui << 8) & 0x00FF0000) |
40  ((ui >> 8) & 0x0000FF00) |
41  (ui << 24);

72 void TrackletsParser::OutputIncomingData()
73 {
74  LOG(info) << "Data to parse for Tracklets from " << std::hex << mStartParse << " to " << mEndParse;
75  int wordcount = 0;
76  std::stringstream outputstring;
77  auto word = mStartParse;
78  outputstring << "tracklet 0x" << std::hex << std::setfill('0') << std::setw(6) << 0 << " :: ";
79  while (word <= mEndParse) { // loop over the entire data buffer (a complete link of tracklets and digits)
80 
81  if (wordcount != 0 && (wordcount % 8 == 0 || word == mEndParse)) {
82  LOG(info) << outputstring.str();
83  outputstring.str("");
84  outputstring << "tracklet 0x" << std::hex << std::setfill('0') << std::setw(6) << wordcount << " :: ";
85  }
86  if (wordcount == 0) {
87  outputstring << " 0x" << std::hex << std::setfill('0') << std::setw(8) << HelperMethods::swapByteOrderreturn(*word);
88  } else {
89  outputstring << " 0x" << std::hex << std::setfill('0') << std::setw(8) << HelperMethods::swapByteOrderreturn(*word);
90  }
91  word++;
92  wordcount++;
93  }
94  LOG(info) << "Data buffer to parse for Tracklets end";
95 }

97 int TrackletsParser::Parse()
98 {
99  //we are handed the buffer payload of an rdh and need to parse its contents.
100  //producing a vector of digits.
101 
102  mTrackletParsingBad = false;
103  if (mHeaderVerbose) {
104  OutputIncomingData();
105  }
106  //mData holds a buffer containing tracklets parse placing tracklets in the output vector.
107  //mData holds 2048 digits.
108  mCurrentLink = 0;
109  mWordsRead = 0;
110  mTrackletsFound = 0;
111  if (mTrackletHCHeaderState == 0) { //TODO move this to the reader as done for the digithcheader
112  // tracklet hc header is never present
113  mState = StateTrackletMCMHeader;
114  } else {
115  if (mTrackletHCHeaderState == 1) {
116  auto nextword = std::next(mStartParse);
117  if (*nextword != constants::TRACKLETENDMARKER) {
118  //we have tracklet data so no TracletHCHeader
119  mState = StateTrackletHCHeader;
120  } else {
121  //we have no tracklet data so no TracletHCHeader
122  mState = StateTrackletMCMHeader;
123  }
124  } else {
125  if (mTrackletHCHeaderState != 2) {
126  LOG(warn) << "unknwon TrackletHCHeaderState of " << mIgnoreTrackletHCHeader;
127  }
128  // tracklet hc header is always present
129  mState = StateTrackletHCHeader; // we start with a trackletMCMHeader
130  }
131  }
132 
133  int currentLinkStart = 0;
134  int mcmtrackletcount = 0;
135  int trackletloopcount = 0;
136  int headertrackletcount = 0;
137  bool ignoreDataTillTrackletEndMarker = false; // used for when we need to dump the rest of the tracklet data.
138  for (auto word = mStartParse; word < mEndParse; ++word) { // loop over the entire data buffer (a complete link of tracklets and digits)
139 
140  if (mState == StateFinished) {
141  return mWordsRead;
142  }
143  //loop over all the words ...
144  //check for tracklet end marker 0x1000 0x1000
145  int index = std::distance(mStartParse, word);
146  int indexend = std::distance(word, mEndParse);
147  std::array<uint32_t, o2::trd::constants::HBFBUFFERMAX>::iterator nextword = word;
148  std::advance(nextword, 1);
149  uint32_t nextwordcopy = *nextword;
150 
151  if (mByteOrderFix) {
152  swapByteOrder(*word);
153  swapByteOrder(nextwordcopy);
154  }
155 
156  if (*word == 0x10001000 && nextwordcopy == 0x10001000) {
157  if (!StateTrackletEndMarker && !StateTrackletHCHeader) {
158  LOG(warn) << "State should be trackletend marker current ?= end marker ?? " << mState << " ?=" << StateTrackletEndMarker;
159  }
160 
161  mWordsRead += 2;
162 
163  if (mHeaderVerbose) {
164  LOG(info) << "***TrackletEndMarker : 0x" << std::hex << *word << " and 0x" << nextwordcopy << " at offset " << std::distance(mStartParse, word);
165  }
166 
167  mState = StateFinished;
168  return mWordsRead;
169  }
170  if (*word == o2::trd::constants::CRUPADDING32) {
171  //padding word first as it clashes with the hcheader.
172  mState = StatePadding;
173  incParsingError(TRDParsingTrackletCRUPaddingWhileParsingTracklets);
174  mWordsDumped = std::distance(word, mEndParse);
175  ignoreDataTillTrackletEndMarker = true;
176  word = mEndParse;
177  //TODO remove tracklets already added erroneously
178  continue; // bail out
179 
180  } else {
181  if (ignoreDataTillTrackletEndMarker) {
182  mWordsDumped++;
183  incParsingError(TRDParsingTrackletBit11NotSetInTrackletHCHeader);
184  continue; //go back to the start of loop, walk the data till the above code of the tracklet end marker is hit, padding is hit or we get to the end of the data.
185  }
186  //fix to missing bit on supermodule 16 and 17, to set the uniquely identifying bit.
187  if (mState == StateTrackletHCHeader) {
188  if ((mFEEID.supermodule > 15) && mOptions[TRDFixSM1617Bit] && mTrackletHCHeaderState == 2) {
189  *word |= 1 << 11; //flip bit eleven for the tracklethcheader for the last 2 supermodules (bug/misconfiguration/broken/other) not sure why its like this yet, but it is.
190  }
191  }
192  //now for Tracklet hc header
193  if ((((*word) & (0x1 << 11)) != 0) && !mIgnoreTrackletHCHeader && mState == StateTrackletHCHeader) { //TrackletHCHeader has bit 11 set to 1 always. Check for state because raw data can have bit 11 set!
194  if (mState != StateTrackletHCHeader) {
195  incParsingError(TRDParsingTrackletBit11NotSetInTrackletHCHeader);
196  }
197  //read the header
198  if (mHeaderVerbose) {
199  LOG(info) << "*** TrackletHCHeader : 0x" << std::hex << *word << " at offset :0x" << std::distance(mStartParse, word);
200  }
201  //we actually have a header word.
202  mTrackletHCHeader = (TrackletHCHeader*)&word;
203  //sanity check of trackletheader ??
204  if (!trackletHCHeaderSanityCheck(*mTrackletHCHeader)) {
205  incParsingError(TRDParsingTrackletHCHeaderSanityCheckFailure);
206  }
207  mWordsRead++;
208  mState = StateTrackletMCMHeader; // now we should read a MCMHeader next time through loop
209  } else { //not TrackletHCHeader
210  if (((*word) & 0x80000001) == 0x80000001 && mState == StateTrackletMCMHeader) { //TrackletMCMHeader has the bits on either end always 1
211  //mcmheader
212  mTrackletMCMHeader = (TrackletMCMHeader*)&(*word);
213  if (mHeaderVerbose) {
214  LOG(info) << "***TrackletMCMHeader : 0x" << std::hex << *word << " at offset: 0x" << std::distance(mStartParse, word);
215  TrackletMCMHeader a;
216  a.word = *word;
217  printTrackletMCMHeader(a);
218  }
219  if (!trackletMCMHeaderSanityCheck(*mTrackletMCMHeader)) {
220  incParsingError(TRDParsingTrackletMCMHeaderSanityCheckFailure);
221  }
222  headertrackletcount = getNumberofTracklets(*mTrackletMCMHeader);
223  if (headertrackletcount > 0) {
224  mState = StateTrackletMCMData; // afrter reading a header we should then have data for next round through the loop
225  } else {
226  mState = StateTrackletMCMHeader;
227  }
228 
229  mcmtrackletcount = 0;
230  mWordsRead++;
231  } else {
232  if (mState == StateTrackletMCMHeader || (mState == StateTrackletHCHeader && !mOptions[mIgnoreTrackletHCHeader])) {
233  // if we are here something is wrong, dump the data. The else of line 227 should imply we are in StateTrackletMCMData;
234  ignoreDataTillTrackletEndMarker = true;
235  incParsingError(TRDParsingTrackletStateMCMHeaderButParsingMCMData);
236  mWordsDumped = std::distance(word, mEndParse);
237  ignoreDataTillTrackletEndMarker = true;
238  word = mEndParse;
239  continue;
240  }
241  mState = StateTrackletMCMData;
242  //tracklet data;
243  mTrackletMCMData = (TrackletMCMData*)&(*word);
244  mWordsRead++;
245  if (mHeaderVerbose) {
246  LOG(info) << "*** TrackletMCMData : 0x" << std::hex << *word << " at offset :0x" << std::distance(mStartParse, word);
247  printTrackletMCMData(*mTrackletMCMData);
248  }
249  // do we have more tracklets than the header allows?
250  if (headertrackletcount < mcmtrackletcount) {
251  ignoreDataTillTrackletEndMarker = true;
252  //dump the rest of the data ... undo any tracklets already written?
253  //cant dump till mEndParse and digits are after the tracklets
254  //we can assume the mcmtrackletcountth (n from the end) last tracklets in the vector are to be removed.
255  incParsingError(TRDParsingTrackletTrackletCountGTThatDeclaredInMCMHeader);
256  mEventRecord->popTracklets(mcmtrackletcount); // our format is always 4
257  //TODO count remove warning
258  }
259  // take the header and this data word and build the underlying 64bit tracklet.
260  int q0, q1, q2;
261  int qa, qb;
262  switch (mcmtrackletcount) {
263  case 0:
264  qa = mTrackletMCMHeader->pid0;
265  break;
266  case 1:
267  qa = mTrackletMCMHeader->pid1;
268  break;
269  case 2:
270  qa = mTrackletMCMHeader->pid2;
271  break;
272  default:
273  LOG(alarm) << "mcmtrackletcount is not in [0:2] count=" << mcmtrackletcount << " headertrackletcount=" << headertrackletcount << " something very wrong parsing the TrackletMCMData fields with data of : 0x" << std::hex << mTrackletMCMData->word;
274  incParsingError(TRDParsingTrackletInvalidTrackletCount);
275  //this should have been caught above by the headertrackletcount to mcmtrackletcount
276  ignoreDataTillTrackletEndMarker = true;
277  break;
278  }
279  if (!ignoreDataTillTrackletEndMarker) {
280  q0 = getQFromRaw(mTrackletMCMHeader, mTrackletMCMData, 0, mcmtrackletcount);
281  q1 = getQFromRaw(mTrackletMCMHeader, mTrackletMCMData, 1, mcmtrackletcount);
282  q2 = getQFromRaw(mTrackletMCMHeader, mTrackletMCMData, 2, mcmtrackletcount);
283  int padrow = mTrackletMCMHeader->padrow;
284  int col = mTrackletMCMHeader->col;
285  int pos = mTrackletMCMData->pos;
286  int slope = mTrackletMCMData->slope;
287  int hcid = mDetector * 2 + mHalfChamberSide;
288  if (mHeaderVerbose) {
289  if (mTrackletHCHeaderState) {
290  LOG(info) << "Tracklet HCID : " << hcid << " mDetector:" << mDetector << " robside:" << mHalfChamberSide << " " << mTrackletMCMHeader->padrow << ":" << mTrackletMCMHeader->col << " ---- " << mTrackletHCHeader->supermodule << ":" << mTrackletHCHeader->stack << ":" << mTrackletHCHeader->layer << ":" << mTrackletHCHeader->side << " rawhcheader : 0x" << std::hex << std::hex << mTrackletHCHeader->word;
291  } else {
292  LOG(info) << "Tracklet HCID : " << hcid << " mDetector:" << mDetector << " robside:" << mHalfChamberSide << " " << mTrackletMCMHeader->padrow << ":" << mTrackletMCMHeader->col;
293  }
294  }
295  //TODO cross reference hcid to somewhere for a check. mDetector is assigned at the time of parser init.
296  mEventRecord->getTracklets().emplace_back(4, hcid, padrow, col, pos, slope, q0, q1, q2); // our format is always
297  mEventRecord->incTrackletsFound(1);
298  mTrackletsFound++;
299  mcmtrackletcount++;
300  if (mcmtrackletcount == headertrackletcount) { // headertrackletcount and mcmtrackletcount are not zero based counting
301  // at the end of the tracklet output of this mcm
302  // next to come can either be an mcmheaderword or a trackletendmarker.
303  // check next word if its a trackletendmarker
304  auto nextdataword = std::next(word, 1);
305  // the check is unambigous between trackletendmarker and mcmheader
306  if ((*nextdataword) == constants::TRACKLETENDMARKER) {
307  mState = StateTrackletEndMarker;
308  } else {
309  mState = StateTrackletMCMHeader;
310  }
311  }
312  }
313  }
314  }
315  } // else
316  trackletloopcount++;
317  } //end of for loop
318  //sanity check
319  //LOG(warn) << " end of Trackelt parsing but we are exiting with out a tracklet end marker with " << mWordsRead << " 32bit words read";
320  //mEventRecord.ErrorStats[TRDParsingTrackletExitingNoTrackletEndMarker]++;
321  incParsingError(TRDParsingTrackletExitingNoTrackletEndMarker);
322 
323  mTrackletParsingBad = true;
324  return mWordsRead;
325 }
