82 FIT_CALIBRATOR_TEMPLATES
83 void FIT_CALIBRATOR_TYPE::finalizeSlot(Slot& slot)
84 {
85  static std::map<std::string, std::string> md;
86  auto* container = slot.getContainer();
87  static const double TFlength = 1E-3 * o2::raw::HBFUtils::Instance().getNOrbitsPerTF() * o2::constants::lhc::LHCOrbitMUS; // in ms
88  uint64_t starting;
89  if constexpr (std::is_same_v<decltype(*container), TimeSlotStorageType&>) {
90  starting = container->getFirstCreation();
91  }
92  uint64_t stopping = starting + (slot.getTFEnd() - slot.getTFStart()) * TFlength;
93  LOG(info) << "!!!! TFstart " << slot.getTFStart() << " TFend " << slot.getTFEnd() << "starting = " << starting << " - stopping = " << stopping;
94 
95  auto calibrationObject = FITCalibrationObjectProducer::generateCalibrationObject<CalibrationObjectType>(*container);
96  auto preparedCalibObjects = FITCalibrationApi::prepareCalibrationObjectToSend(calibrationObject, starting, stopping);
97 
98  mStoredCalibrationObjects.insert(mStoredCalibrationObjects.end(),
99  std::make_move_iterator(preparedCalibObjects.begin()),
100  std::make_move_iterator(preparedCalibObjects.end()));
101 }
