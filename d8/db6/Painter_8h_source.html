53 struct PadCoordinates {

57  void rotate(float angDeg)
58  {
59  const auto ang = 0.017453292519943295 * angDeg;
60  const auto cs = std::cos(ang);
61  const auto sn = std::sin(ang);
62  for (size_t i = 0; i < xVals.size(); ++i) {
63  const auto x = xVals[i] * cs - yVals[i] * sn;
64  const auto y = xVals[i] * sn + yVals[i] * cs;
65  xVals[i] = x;
66  yVals[i] = y;
67  }
68  }
69 };
