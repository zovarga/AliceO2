206  const IrregularSpline1D& gridV = getGridV();

214 template <typename T>
215 GPUd() void IrregularSpline2D3D::correctEdges(T* data) const
216 {
217  const IrregularSpline1D& gridU = getGridU();
218  const IrregularSpline1D& gridV = getGridV();
219  int nu = gridU.getNumberOfKnots();
220  int nv = gridV.getNumberOfKnots();
221 
222  { // left edge of U
223  int iu = 0;
224  const IrregularSpline1D::Knot* s = gridU.getKnots() + iu;
225  double c0, c1, c2, c3;
226  gridU.getEdgeCorrectionCoefficients(s[0].u, s[1].u, s[2].u, s[3].u, c0, c1, c2, c3);
227  for (int iv = 0; iv < nv; iv++) {
228  T* f0 = data + (nu * (iv) + iu) * 3;
229  T* f1 = f0 + 3;
230  T* f2 = f0 + 6;
231  T* f3 = f0 + 9;
232  for (int idim = 0; idim < 3; idim++) {
233  f0[idim] = (T)(c0 * f0[idim] + c1 * f1[idim] + c2 * f2[idim] + c3 * f3[idim]);
234  }
235  }
236  }
237 
238  { // right edge of U
239  int iu = nu - 4;
240  const IrregularSpline1D::Knot* s = gridU.getKnots() + iu;
241  double c0, c1, c2, c3;
242  gridU.getEdgeCorrectionCoefficients(s[3].u, s[2].u, s[1].u, s[0].u, c3, c2, c1, c0);
243  for (int iv = 0; iv < nv; iv++) {
244  T* f0 = data + (nu * (iv) + iu) * 3;
245  T* f1 = f0 + 3;
246  T* f2 = f0 + 6;
247  T* f3 = f0 + 9;
248  for (int idim = 0; idim < 3; idim++) {
249  f3[idim] = (T)(c0 * f0[idim] + c1 * f1[idim] + c2 * f2[idim] + c3 * f3[idim]);
250  }
251  }
252  }
253 
254  { // low edge of V
255  int iv = 0;
256  const IrregularSpline1D::Knot* s = gridV.getKnots() + iv;
257  double c0, c1, c2, c3;
258  gridV.getEdgeCorrectionCoefficients(s[0].u, s[1].u, s[2].u, s[3].u, c0, c1, c2, c3);
259  for (int iu = 0; iu < nu; iu++) {
260  T* f0 = data + (nu * iv + iu) * 3;
261  T* f1 = f0 + nu * 3;
262  T* f2 = f0 + nu * 6;
263  T* f3 = f0 + nu * 9;
264  for (int idim = 0; idim < 3; idim++) {
265  f0[idim] = (T)(c0 * f0[idim] + c1 * f1[idim] + c2 * f2[idim] + c3 * f3[idim]);
266  }
267  }
268  }
269 
270  { // high edge of V
271  int iv = nv - 4;
272  const IrregularSpline1D::Knot* s = gridV.getKnots() + iv;
273  double c0, c1, c2, c3;
274  gridV.getEdgeCorrectionCoefficients(s[3].u, s[2].u, s[1].u, s[0].u, c3, c2, c1, c0);
275  for (int iu = 0; iu < nu; iu++) {
276  T* f0 = data + (nu * iv + iu) * 3;
277  T* f1 = f0 + nu * 3;
278  T* f2 = f0 + nu * 6;
279  T* f3 = f0 + nu * 9;
280  for (int idim = 0; idim < 3; idim++) {
281  f3[idim] = (T)(c0 * f0[idim] + c1 * f1[idim] + c2 * f2[idim] + c3 * f3[idim]);
282  }
283  }
284  }

288 GPUdi() void IrregularSpline2D3D::getSpline(GPUgeneric() const T* correctedData, float u, float v, GPUgeneric() T& x, GPUgeneric() T& y, GPUgeneric() T& z) const
289 {
290  // Get interpolated value for f(u,v) using data array correctedData[getNumberOfKnots()] with corrected edges
291 
292  const IrregularSpline1D& gridU = getGridU();
293  const IrregularSpline1D& gridV = getGridV();
294  int nu = gridU.getNumberOfKnots();
295  int iu = gridU.getKnotIndex(u);
296  int iv = gridV.getKnotIndex(v);
297 
298  const IrregularSpline1D::Knot& knotU = gridU.getKnot(iu);
299  const IrregularSpline1D::Knot& knotV = gridV.getKnot(iv);
300 
301  const T* dataV0 = correctedData + (nu * (iv - 1) + iu - 1) * 3;
302  const T* dataV1 = dataV0 + 3 * nu;
303  const T* dataV2 = dataV0 + 6 * nu;
304  const T* dataV3 = dataV0 + 9 * nu;
305 
306  T dataV[12];
307  for (int i = 0; i < 12; i++) {
308  dataV[i] = gridV.getSpline(knotV, dataV0[i], dataV1[i], dataV2[i], dataV3[i], v);
309  }
310 
311  T* dataU0 = dataV + 0;
312  T* dataU1 = dataV + 3;
313  T* dataU2 = dataV + 6;
314  T* dataU3 = dataV + 9;
315 
316  T res[3];
317  for (int i = 0; i < 3; i++) {
318  res[i] = gridU.getSpline(knotU, dataU0[i], dataU1[i], dataU2[i], dataU3[i], u);
319  }
320  x = res[0];
321  y = res[1];
322  z = res[2];
323 }

326 {
327  // Same as getSpline, but using vectorized calculation.
328  // \param correctedData should be at least 128-bit aligned

330 #if !defined(__CINT__) && !defined(__ROOTCINT__) && !defined(__ROOTCLING__) && !defined(GPUCA_GPUCODE) && !defined(GPUCA_NO_VC) && defined(__cplusplus) && __cplusplus >= 201703L

336  const IrregularSpline1D::Knot& knotU = gridU.getKnot(iu);
337  const IrregularSpline1D::Knot& knotV = gridV.getKnot(iv);
338 
339  const float* dataV0 = correctedData + (nu * (iv - 1) + iu - 1) * 3;
340  const float* dataV1 = dataV0 + 3 * nu;
341  const float* dataV2 = dataV0 + 6 * nu;
342  const float* dataV3 = dataV0 + 9 * nu;
343 
344  // calculate F values at V==v and U == Ui of knots
345 
346  Vc::SimdArray<float, 12> dataV0vec(dataV0), dataV1vec(dataV1), dataV2vec(dataV2), dataV3vec(dataV3), dataVvec = gridV.getSpline(knotV, dataV0vec, dataV1vec, dataV2vec, dataV3vec, v);
347 
348  using V = std::conditional_t<(Vc::float_v::size() >= 4),
349  Vc::float_v,
350  Vc::SimdArray<float, 3>>;
351 
352  float dataV[9 + V::size()];
353  // dataVvec.scatter( dataV, Vc::SimdArray<float,12>::IndexType::IndexesFromZero() );
354  //dataVvec.scatter(dataV, Vc::SimdArray<uint, 12>(Vc::IndexesFromZero));
355  dataVvec.store(dataV, Vc::Unaligned);
356 
357  for (unsigned int i = 12; i < 9 + V::size(); i++) { // fill not used part of the vector with 0
358  dataV[i] = 0.f;
359  }
360  // calculate F values at V==v and U == u
361  V dataU0vec(dataV), dataU1vec(dataV + 3), dataU2vec(dataV + 6), dataU3vec(dataV + 9);
362 
363  V dataUVvec = gridU.getSpline(knotU, dataU0vec, dataU1vec, dataU2vec, dataU3vec, u);
364 
365  x = dataUVvec[0];
366  y = dataUVvec[1];
367  z = dataUVvec[2];
368 #else
369  getSpline(correctedData, u, v, x, y, z);
370 #endif
