47 void AlignableDetectorITS::defineVolumes()
48 {
49  // define ITS volumes
50  //
51  auto geom = o2::its::GeometryTGeo::Instance();
52 
53  AlignableVolume *volITS = nullptr, *volLr = nullptr, *volHB = nullptr, *volSt = nullptr, *volHSt = nullptr, *volMod = nullptr;
54  AlignableSensorITS* sens = nullptr;
55  //
56  std::unordered_map<std::string, AlignableVolume*> sym2vol;
57  addVolume(volITS = new AlignableVolume(geom->composeSymNameITS(), getDetLabel(), mController));
58  sym2vol[volITS->getSymName()] = volITS;
59  //
60  int nonSensCnt = 0;
61  for (int ilr = 0; ilr < geom->getNumberOfLayers(); ilr++) {
62  addVolume(volLr = new AlignableVolume(geom->composeSymNameLayer(ilr), getNonSensLabel(nonSensCnt++), mController));
63  sym2vol[volLr->getSymName()] = volLr;
64  volLr->setParent(volITS);
65  for (int ihb = 0; ihb < 2; ihb++) {
66  addVolume(volHB = new AlignableVolume(geom->composeSymNameHalfBarrel(ilr, ihb), getNonSensLabel(nonSensCnt++), mController));
67  volHB->setParent(volLr);
68  int nstavesHB = geom->getNumberOfStaves(ilr) / 2;
69  for (int ist = 0; ist < nstavesHB; ist++) {
70  addVolume(volSt = new AlignableVolume(geom->composeSymNameStave(ilr, ihb, ist), getNonSensLabel(nonSensCnt++), mController));
71  sym2vol[volSt->getSymName()] = volSt;
72  volSt->setParent(volLr);
73  for (int ihst = 0; ihst < geom->getNumberOfHalfStaves(ilr); ihst++) {
74  addVolume(volHSt = new AlignableVolume(geom->composeSymNameHalfStave(ilr, ihb, ist, ihst), getNonSensLabel(nonSensCnt++), mController));
75  sym2vol[volHSt->getSymName()] = volHSt;
76  volHSt->setParent(volSt);
77  for (int imd = 0; imd < geom->getNumberOfModules(ilr); imd++) {
78  addVolume(volMod = new AlignableVolume(geom->composeSymNameModule(ilr, ihb, ist, ihst, imd), getNonSensLabel(nonSensCnt++), mController));
79  sym2vol[volMod->getSymName()] = volMod;
80  volMod->setParent(volHSt);
81  } // module
82  } //halfstave
83  } // stave
84  } // halfBarrel
85  } // layer
86 
87  for (int ich = 0; ich < geom->getNumberOfChips(); ich++) {
88  int chID = o2::base::GeometryManager::getSensID(mDetID, ich);
89  if (ich != chID) {
90  throw std::runtime_error(fmt::format("mismatch between counter {} and composed {} chip IDs", ich, chID));
91  }
92  addVolume(sens = new AlignableSensorITS(o2::base::GeometryManager::getSymbolicName(mDetID, ich), chID, getSensLabel(chID), mController));
93  int lay = 0, hba, sta = 0, ssta = 0, modd = 0, chip = 0;
94  geom->getChipId(chID, lay, hba, sta, ssta, modd, chip);
95  AlignableVolume* parVol = sym2vol[modd < 0 ? geom->composeSymNameStave(lay, hba, sta) : geom->composeSymNameModule(lay, hba, sta, ssta, modd)];
96  if (!parVol) {
97  throw std::runtime_error(fmt::format("did not find parent for chip {}", chID));
98  }
99  sens->setParent(parVol);
100  }
101  //
102 }

131 {
132  // set syst. errors for specific layer
133  auto geom = o2::its::GeometryTGeo::Instance();
134  int chMin = geom->getFirstChipIndex(ilr), chMax = geom->getLastChipIndex(ilr);
135  for (int isn = chMin; isn <= chMax; isn++) {
