175  void registerService(hash_type typeHash, void* service, ServiceKind kind, uint64_t threadId, char const* name = nullptr) const;
176 
177  // Lookup a given @a typeHash for a given @a threadId at
178  // a unique (per typeHash) location. There might
179  // be other typeHash which sit in the same place, but
180  // the if statement will rule them out. As long as
181  // only one thread writes in a given i + id location
182  // as guaranteed by the atomic, mServicesKey[i + id] will
183  // either be 0 or the final value.
184  // This method should NEVER register a new service, event when requested.
185  int getPos(uint32_t typeHash, uint64_t threadId) const
186  {
187  auto threadHashId = (typeHash ^ threadId) & MAX_SERVICES_MASK;
188  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
189  if (mServicesKey[i + threadHashId].load() == typeHash) {
190  return i + threadHashId;
191  }
192  }
193  return -1;
194  }
195 
196  // Basic, untemplated API. This will require explicitly
197  // providing the @a typeHash for the Service type,
198  // there @a threadId of the thread asking the service
199  // and the @a kind of service. This
200  // method might trigger the registering of a new service
201  // if the service is not a stream service and the global
202  // zero service is available.
203  // Use this API only if you know what you are doing.
204  void* get(uint32_t typeHash, uint64_t threadId, ServiceKind kind, char const* name = nullptr) const
205  {
206  // Look for the service. If found, return it.
207  // Notice how due to threading issues, we might
208  // find it with getPos, but the value can still
209  // be nullptr.
210  auto pos = getPos(typeHash, threadId);
211  if (pos != -1 && mServicesMeta[pos].kind == ServiceKind::Stream && mServicesMeta[pos].threadId != threadId) {
212  throwError(runtime_error_f("Inconsistent registry for thread %d. Expected %d", threadId, mServicesMeta[pos].threadId));
213  O2_BUILTIN_UNREACHABLE();
214  }
215 
216  if (pos != -1) {
217  mServicesKey[pos].load();
218  std::atomic_thread_fence(std::memory_order_acquire);
219  void* ptr = mServicesValue[pos];
220  if (ptr) {
221  return ptr;
222  }
223  }
224  // We are looking up a service which is not of
225  // stream kind and was not looked up by this thread
226  // before.
227  if (threadId != 0) {
228  int pos = getPos(typeHash, 0);
229  if (pos != -1 && kind != ServiceKind::Stream) {
230  mServicesKey[pos].load();
231  std::atomic_thread_fence(std::memory_order_acquire);
232  registerService(typeHash, mServicesValue[pos], kind, threadId, name);
233  }
234  if (pos != -1) {
235  mServicesKey[pos].load();
236  std::atomic_thread_fence(std::memory_order_acquire);
237  return mServicesValue[pos];
238  } else {
239  throwError(runtime_error_f("Unable to find requested service %s", name));
240  }
241  }
242  // If we are here it means we never registered a
243  // service for the 0 thread (i.e. the main thread).
244  return nullptr;
245  }

293  template <typename T>
294  std::enable_if_t<std::is_const_v<T> == false, bool> active() const
295  {
296  constexpr auto typeHash = TypeIdHelpers::uniqueId<T>();
297  auto tid = std::this_thread::get_id();
298  std::hash<std::thread::id> hasher;
299  if (this->getPos(typeHash, 0) != -1) {
300  return true;
301  }
302  auto result = this->getPos(typeHash, hasher(tid)) != -1;
303  return result;
304  }

309  template <typename T>
310  T& get() const
311  {
312  constexpr auto typeHash = TypeIdHelpers::uniqueId<T>();
313  auto tid = std::this_thread::get_id();
314  std::hash<std::thread::id> hasher;
315  auto ptr = this->get(typeHash, hasher(tid), ServiceKind::Serial, typeid(T).name());
316  if (O2_BUILTIN_LIKELY(ptr != nullptr)) {
317  if constexpr (std::is_const_v<T>) {
318  return *reinterpret_cast<T const*>(ptr);
319  } else {
320  return *reinterpret_cast<T*>(ptr);
321  }
322  }
323  throwError(runtime_error_f("Unable to find service of kind %s. Make sure you use const / non-const correctly.", typeid(T).name()));
324  O2_BUILTIN_UNREACHABLE();
325  }
