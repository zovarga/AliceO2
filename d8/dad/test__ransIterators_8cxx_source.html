25 class ShiftFunctor
26 {
27  public:
28  ShiftFunctor(size_t shift) : mShift{shift} {};
29 
30  template <typename iterA_T, typename iterB_T>
31  inline uint32_t operator()(iterA_T iterA, iterB_T iterB) const
32  {
33  return *iterB + (static_cast<uint32_t>(*iterA) << mShift);
34  };
35 
36  template <typename iterA_T, typename iterB_T>
37  inline void operator()(iterA_T iterA, iterB_T iterB, uint32_t value) const
38  {
39  *iterA = value >> mShift;
40  *iterB = value & ((1 << mShift) - 1);
41  };
42 
43  private:
44  size_t mShift;
45 };
