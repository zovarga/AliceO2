25 o2::fdd::RecPoint Reconstructor::process(o2::fdd::Digit const& digitBC,
26  gsl::span<const o2::fdd::ChannelData> inChData,
27  gsl::span<o2::fdd::ChannelDataFloat> outChData)
28 {
29  //Compute charge weighted average time
30  Double_t timeFDA = 0, timeFDC = 0;
31  Double_t weightFDA = 0.0, weightFDC = 0.0;
32 
33  int nch = inChData.size();
34  for (int ich = 0; ich < nch; ich++) {
35  outChData[ich] = o2::fdd::ChannelDataFloat{inChData[ich].mPMNumber,
36  (inChData[ich].mTime) * TimePerTDC,
37  (double)inChData[ich].mChargeADC,
38  0}; // Fill with ADC number once implemented
39  Float_t adc = outChData[ich].mChargeADC;
40  Float_t time = outChData[ich].mTime;
41  if (time == o2::InteractionRecord::DummyTime) {
42  continue;
43  }
44  Float_t timeErr = 1;
45  if (adc > 1) {
46  timeErr = 1 / adc;
47  }
48  if (outChData[ich].mPMNumber < 8) {
49  timeFDC += time / (timeErr * timeErr);
50  weightFDC += 1. / (timeErr * timeErr);
51  } else {
52  timeFDA += time / (timeErr * timeErr);
53  weightFDA += 1. / (timeErr * timeErr);
54  }
55  }
56  const int nsToPs = 1e3;
57  std::array<int, 2> mCollisionTime = {o2::fdd::RecPoint::sDummyCollissionTime, o2::fdd::RecPoint::sDummyCollissionTime};
58 
59  mCollisionTime[o2::fdd::RecPoint::TimeA] = (weightFDA > 1) ? round(timeFDA / weightFDA * nsToPs) : o2::fdd::RecPoint::sDummyCollissionTime;
60  mCollisionTime[o2::fdd::RecPoint::TimeC] = (weightFDC > 1) ? round(timeFDC / weightFDC * nsToPs) : o2::fdd::RecPoint::sDummyCollissionTime;
61 
62  return RecPoint{mCollisionTime, digitBC.ref.getFirstEntry(), digitBC.ref.getEntries(), digitBC.getIntRecord(), digitBC.mTriggers};
63 }
