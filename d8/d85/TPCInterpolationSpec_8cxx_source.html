52 void TPCInterpolationDPL::run(ProcessingContext& pc)
53 {
54  LOG(info) << "TPC Interpolation Workflow initialized. Start processing...";
55  mTimer.Start(false);
56  RecoContainer recoData;
57  recoData.collectData(pc, *mDataRequest.get());
58 
59  // load the input tracks
60  std::vector<o2::globaltracking::RecoContainer::GlobalIDSet> gidTables;
61  std::vector<o2::track::TrackParCov> seeds;
62  std::vector<float> trkTimes;
63  std::vector<GTrackID> gids;
64  bool processITSTPConly = mProcessITSTPConly; // so that the flag can be used inside the lambda
65  // the creator goes from most complete track (ITS-TPC-TRD-TOF) to least complete one (ITS-TPC)
66  auto creator = [&gidTables, &seeds, &trkTimes, &recoData, &processITSTPConly, &gids](auto& _tr, GTrackID _origID, float t0, float tErr) {
67  if constexpr (std::is_base_of_v<o2::track::TrackParCov, std::decay_t<decltype(_tr)>>) {
68  bool trackGood = true;
69  bool hasOuterPoint = false;
70  auto gidTable = recoData.getSingleDetectorRefs(_origID);
71  if (!gidTable[GTrackID::ITS].isIndexSet() || !gidTable[GTrackID::TPC].isIndexSet()) {
72  // ITS and TPC track is always needed. At this stage ITS afterburner tracks are also rejected
73  return true;
74  }
75  if (gidTable[GTrackID::TRD].isIndexSet() || gidTable[GTrackID::TOF].isIndexSet()) {
76  hasOuterPoint = true;
77  }
78  const auto itstpcTrk = &recoData.getTPCITSTrack(gidTable[GTrackID::ITSTPC]);
79  const auto itsTrk = &recoData.getITSTrack(gidTable[GTrackID::ITS]);
80  const auto tpcTrk = &recoData.getTPCTrack(gidTable[GTrackID::TPC]);
81  // apply track quality cuts
82  if (itsTrk->getChi2() / itsTrk->getNumberOfClusters() > param::MaxITSChi2 || tpcTrk->getChi2() / tpcTrk->getNClusterReferences() > param::MaxTPCChi2) {
83  // reduced chi2 cut is the same for all track types
84  trackGood = false;
85  }
86  if (!hasOuterPoint) {
87  // ITS-TPC track (does not have outer points in TRD or TOF)
88  if (!processITSTPConly) {
89  return true;
90  }
91  if (itsTrk->getNumberOfClusters() < param::MinITSNClsNoOuterPoint || tpcTrk->getNClusterReferences() < param::MinTPCNClsNoOuterPoint) {
92  trackGood = false;
93  }
94  } else {
95  if (itsTrk->getNumberOfClusters() < param::MinITSNCls || tpcTrk->getNClusterReferences() < param::MinTPCNCls) {
96  trackGood = false;
97  }
98  }
99  if (trackGood) {
100  trkTimes.push_back(t0);
101  seeds.emplace_back(itsTrk->getParamOut()); // FIXME: should this not be a refit of the ITS track?
102  gidTables.emplace_back(gidTable);
103  gids.push_back(_origID);
104  }
105  return true;
106  } else {
107  return false;
108  }
109  };
110  recoData.createTracksVariadic(creator); // create track sample considered for interpolation
111  LOG(info) << "Created " << seeds.size() << " seeds.";
112 
113  if (mUseMC) {
114  // possibly MC labels will be used to check filtering procedure performance before interpolation
115  // not yet implemented
116  }
117 
118  mInterpolation.process(recoData, gids, gidTables, seeds, trkTimes);
119  mTimer.Stop();
120  LOGF(info, "TPC insterpolation timing: Cpu: %.3e Real: %.3e s", mTimer.CpuTime(), mTimer.RealTime());
121  mTimer.Start(true);
122  mResidualProcessor.setInputData(mInterpolation.getReferenceTracks(), mInterpolation.getClusterResiduals());
123  // in the following step one can create one output file per TPC sector with local residuals and statistics
124  mResidualProcessor.doOutlierRejection(false); // if parameter is 'true', then the unbinned filtered residuals will be written to a debug file
125  mTimer.Stop();
126  LOGF(info, "TPC outlier filtering and local residual extraction timing: Cpu: %.3e Real: %.3e s", mTimer.CpuTime(), mTimer.RealTime());
127 
128  if (mWriteResiduals) {
129  pc.outputs().snapshot(Output{"GLO", "TPCINT_TRK", 0, Lifetime::Timeframe}, mInterpolation.getReferenceTracks());
130  pc.outputs().snapshot(Output{"GLO", "TPCINT_RES", 0, Lifetime::Timeframe}, mInterpolation.getClusterResiduals());
131  }
132  pc.outputs().snapshot(Output{"GLO", "UNBINNEDRES", 0, Lifetime::Timeframe}, mResidualProcessor.getUnbinnedResiduals());
133  mTimer.Stop();
134 }

142 DataProcessorSpec getTPCInterpolationSpec(GTrackID::mask_t src, bool useMC, bool processITSTPConly, bool writeResiduals)
143 {
144  auto dataRequest = std::make_shared<DataRequest>();
145  std::vector<OutputSpec> outputs;
146 
147  if (useMC) {
148  LOG(fatal) << "MC usage must be disabled for this workflow, since it is not yet implemented";
149  }
150 
151  dataRequest->requestTracks(src, useMC);
152  dataRequest->requestClusters(src, useMC);
153 
154  if (writeResiduals) {
155  outputs.emplace_back("GLO", "TPCINT_TRK", 0, Lifetime::Timeframe);
156  outputs.emplace_back("GLO", "TPCINT_RES", 0, Lifetime::Timeframe);
157  }
158  outputs.emplace_back("GLO", "UNBINNEDRES", 0, Lifetime::Timeframe);
159 
160  return DataProcessorSpec{
161  "tpc-track-interpolation",
162  dataRequest->inputs,
163  outputs,
164  AlgorithmSpec{adaptFromTask<TPCInterpolationDPL>(dataRequest, useMC, processITSTPConly, writeResiduals)},
165  Options{}};
166 }
