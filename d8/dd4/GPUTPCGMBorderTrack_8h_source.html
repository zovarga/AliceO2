51  GPUd() static bool CheckChi2(float x1, float y1, float cx1, float cxy1, float cy1, float x2, float y2, float cx2, float cxy2, float cy2, float chi2cut)
52  {
53  //* Calculate Chi2/ndf deviation
54  float dx = x1 - x2;
55  float dy = y1 - y2;
56  float cx = cx1 + cx2;
57  float cxy = cxy1 + cxy2;
58  float cy = cy1 + cy2;
59  float det = cx * cy - cxy * cxy;
60  // printf("Res %f Det %f Cut %f %s - ", ( cy*dx - (cxy+cxy)*dy )*dx + cx*dy*dy, det, (det + det) * chi2cut, (CAMath::Abs(( cy*dx - (cxy+cxy)*dy )*dx + cx*dy*dy) < CAMath::Abs((det+det)*chi2cut)) ? "OK" : "Fail");
61  return (CAMath::Abs((cy * dx - (cxy + cxy) * dy) * dx + cx * dy * dy) < CAMath::Abs((det + det) * chi2cut));
62  }

76  GPUd() bool CheckChi2QPt(const GPUTPCGMBorderTrack& t, float chi2cut) const
77  {
78  float d = mP[4] - t.mP[4];
79  if (CAMath::Abs(d) > 0.3f && CAMath::Abs(d) > 0.5f * CAMath::Min(CAMath::Abs(mP[4]), CAMath::Abs(t.mP[4]))) {
80  return false; // Crude cut to avoid some bogus matches, TODO: recheck
