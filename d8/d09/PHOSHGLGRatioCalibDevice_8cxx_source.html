39 void PHOSHGLGRatioCalibDevice::run(o2::framework::ProcessingContext& ctx)
40 {
41 
42  // scan Cells stream, collect HG/LG pairs
43  if (mRunStartTime == 0) {
44  mRunStartTime = o2::header::get<o2::framework::DataProcessingHeader*>(ctx.inputs().get("cellTriggerRecords").header)->startTime;
45  }
46 
47  auto cells = ctx.inputs().get<gsl::span<o2::phos::Cell>>("cells");
48  LOG(debug) << "[PHOSHGLGRatioCalibDevice - run] Received " << cells.size() << " cells, running calibration ...";
49  auto cellsTR = ctx.inputs().get<gsl::span<o2::phos::TriggerRecord>>("cellTriggerRecords");
50  for (const auto& tr : cellsTR) {
51  int firstCellInEvent = tr.getFirstEntry();
52  int lastCellInEvent = firstCellInEvent + tr.getNumberOfObjects();
53  mMapPairs.clear();
54 
55  for (int i = firstCellInEvent; i < lastCellInEvent; i++) {
56  const Cell c = cells[i];
57 
58  auto search = mMapPairs.find(c.getAbsId());
59  if (search != mMapPairs.end()) { //exist
60  if (c.getHighGain()) {
61  search->second.mHGAmp = c.getEnergy();
62  }
63  if (c.getLowGain()) {
64  search->second.mLGAmp = c.getEnergy();
65  }
66  } else {
67  PairAmp pa = {0};
68  if (c.getHighGain()) {
69  pa.mHGAmp = c.getEnergy();
70  }
71  if (c.getLowGain()) {
72  pa.mLGAmp = c.getEnergy();
73  }
74  mMapPairs.insert({c.getAbsId(), pa});
75  }
76  }
77  fillRatios();
78  }
79 }

91 void PHOSHGLGRatioCalibDevice::fillRatios()
92 {
93  // scan collected map and fill ratio histogram if both channels are filled
94  for (auto& it : mMapPairs) {
95  if (it.second.mHGAmp > 0 && it.second.mLGAmp > mMinLG) {
96  mhRatio->Fill(it.first, float(it.second.mHGAmp / it.second.mLGAmp));
97  }
98  }
99  mMapPairs.clear(); //to avoid double counting
100 }

102 void PHOSHGLGRatioCalibDevice::calculateRatios()
103 {
104  // Calculate mean of the ratio
105  short n = o2::phos::Mapping::NCHANNELS - 1792;
106  if (mhRatio->Integral() > 2 * minimalStatistics * n) { //average per channel
107 
108  TF1* fitFunc = new TF1("fitFunc", "gaus", 0., 4000.);
109  fitFunc->SetParameters(1., 200., 60.);
110  fitFunc->SetParLimits(1, 10., 2000.);
111  for (short i = n; i > 0; i--) {
112  TH1D* tmp = mhRatio->ProjectionY(Form("channel%d", i), i, i);
113  fitFunc->SetParameters(tmp->Integral(), tmp->GetMean(), tmp->GetRMS());
114  if (tmp->Integral() < minimalStatistics) {
115  tmp->Delete();
116  continue;
117  }
118  tmp->Fit(fitFunc, "QL0", "", 0., 20.);
119  float a = fitFunc->GetParameter(1);
120  mCalibParams->setHGLGRatio(i + 1792, a); //absId starts from 0
121  tmp->Delete();
122  }
123  }
124 }

126 void PHOSHGLGRatioCalibDevice::checkRatios()
127 {
128  //Compare ratios to current ones stored in CCDB
129  if (!mUseCCDB) {
130  mUpdateCCDB = true;
131  //Set default values for gain and time
132  for (short i = o2::phos::Mapping::NCHANNELS; i > 1792; i--) {
133  mCalibParams->setGain(i, 0.005);
134  mCalibParams->setHGTimeCalib(i, 0.);
135  mCalibParams->setLGTimeCalib(i, 0.);
136  }
137  return;
138  }
139  LOG(info) << "Retrieving current HG/LG ratio from CCDB";
140  //Read current padestals for comarison
141  o2::ccdb::CcdbApi ccdb;
142  ccdb.init(mCCDBPath); // or http://localhost:8080 for a local installation
143  std::map<std::string, std::string> metadata;
144  auto* currentCalibParams = ccdb.retrieveFromTFileAny<CalibParams>("PHS/Calib/CalibParams", metadata, mRunStartTime);
145 
146  if (!currentCalibParams) { //was not read from CCDB, but expected
147  mUpdateCCDB = true;
148  LOG(error) << "Can not read current CalibParams from ccdb";
149  return;
150  }
151 
152  LOG(info) << "Got current calibration from CCDB";
153  //Compare to current
154  int nChanged = 0;
155  for (short i = o2::phos::Mapping::NCHANNELS; i > 1792; i--) {
156  short dp = 2;
157  if (currentCalibParams->getHGLGRatio(i) > 0) {
158  dp = mCalibParams->getHGLGRatio(i) / currentCalibParams->getHGLGRatio(i);
159  } else {
160  if (mCalibParams->getHGLGRatio(i) == 0 && currentCalibParams->getHGLGRatio(i) == 0) {
161  dp = 1.;
162  }
163  }
164  mRatioDiff[i] = dp;
165  if (abs(dp - 1.) > 0.1) { //not a fluctuation
166  nChanged++;
167  }
168  //Copy other stuff from the current CalibParams to new one
169  mCalibParams->setGain(i, currentCalibParams->getGain(i));
170  mCalibParams->setHGTimeCalib(i, currentCalibParams->getHGTimeCalib(i));
171  mCalibParams->setLGTimeCalib(i, currentCalibParams->getLGTimeCalib(i));
172  }
173  LOG(info) << nChanged << "channels changed more than 10 %";
174  if (nChanged > kMinorChange) { //serious change, do not update CCDB automatically, use "force" option to overwrite
175  LOG(error) << "too many channels changed: " << nChanged << " (threshold " << kMinorChange << ")";
176  if (!mForceUpdate) {
177  LOG(error) << "you may use --forceupdate option to force updating ccdb";
178  }
179  mUpdateCCDB = false;
180  } else {
181  mUpdateCCDB = true;
182  }
183 }

185 void PHOSHGLGRatioCalibDevice::sendOutput(DataAllocator& output)
186 {
187 
188  // extract CCDB infos and calibration objects, convert it to TMemFile and send them to the output
189  if (mUpdateCCDB || mForceUpdate) {
190  // prepare all info to be sent to CCDB
191  auto flName = o2::ccdb::CcdbApi::generateFileName("CalibParams");
192  std::map<std::string, std::string> md;
193  o2::ccdb::CcdbObjectInfo info("PHS/Calib/CalibParams", "CalibParams", flName, md, mRunStartTime, 99999999999999);
194  info.setMetaData(md);
195  auto image = o2::ccdb::CcdbApi::createObjectImage(mCalibParams.get(), &info);
196 
197  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName()
198  << " of size " << image->size()
199  << " bytes, valid for " << info.getStartValidityTimestamp()
200  << " : " << info.getEndValidityTimestamp();
201 
202  header::DataHeader::SubSpecificationType subSpec{(header::DataHeader::SubSpecificationType)0};
203  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "PHOS_HGLGratio", subSpec}, *image.get());
204  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "PHOS_HGLGratio", subSpec}, info);
205  }
206  //Anyway send change to QC
207  LOG(info) << "[PHOSHGLGRatioCalibDevice - sendOutput] Sending QC ";
208  output.snapshot(o2::framework::Output{"PHS", "CALIBDIFF", 0, o2::framework::Lifetime::Timeframe}, mRatioDiff);
209 }
