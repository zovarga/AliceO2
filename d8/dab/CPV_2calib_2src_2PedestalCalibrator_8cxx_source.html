54 void PedestalSpectrum::analyze()
55 {
56  if (mIsAnalyzed) {
57  return; // already analyzed, nothing to do
58  }
59 
60  mNPeaks = 0;
61  if (mNEntries == 0) { // no statistics to analyze
62  mPedestalValue = 0;
63  mPedestalRMS = 0.;
64  mIsAnalyzed = true;
65  return;
66  }
67  // (A)typical amplitude spectrum from 1 channel in pedestal run
68  // ^ counts peak2
69  // | |
70  // | peak1 |
71  // | | |
72  // | | || peak3
73  // | || || |
74  // | ||| |||| |<----non-tolerated gap---->||
75  // -------------------------------------------------------------------------->
76  // 0 10 ^ 20 30 ADC amplitude
77  // tolerated gap
78  // we want to find all the peaks, determine their mean and rms
79  // and mean and rms of all the distribution
80  std::vector<uint16_t> peakLowEdge, peakHighEdge;
81  peakLowEdge.push_back(mSpectrumContainer.begin()->first);
82  peakHighEdge.push_back((--mSpectrumContainer.end())->first);
83  uint32_t peakCounts(0), totalCounts(0);
84  float peakSumA(0.), peakSumA2(0.), totalSumA(0.), totalSumA2(0.);
85  mNPeaks = 0;
86 
87  auto iNextAmpl = mSpectrumContainer.begin();
88  iNextAmpl++;
89  for (auto iAmpl = mSpectrumContainer.begin(); iAmpl != mSpectrumContainer.end(); iAmpl++, iNextAmpl++) {
90  peakCounts += iAmpl->second;
91  totalCounts += iAmpl->second;
92  peakSumA += iAmpl->first * iAmpl->second; // mean = sum [A_i * w_i], where A_i is ADC amplitude, w_i is weight = (binCount/totalCount)
93  peakSumA2 += (iAmpl->first * iAmpl->first) * iAmpl->second; // rms = sum [(A_i)^2 * w_i] - mean^2
94  totalSumA += iAmpl->first * iAmpl->second;
95  totalSumA2 += (iAmpl->first * iAmpl->first) * iAmpl->second;
96 
97  if (iNextAmpl != mSpectrumContainer.end()) { //is iAmpl not the last bin?
98  if ((iNextAmpl->first - iAmpl->first) > mToleratedGapWidth) { // let's consider |bin1-bin2|<=5 belong to same peak
99  // firts, save peak low and high edge (just for the future cases)
100  peakHighEdge.push_back(iAmpl->first);
101  peakLowEdge.push_back(iNextAmpl->first);
102  mNPeaks++;
103  mMeanOfPeaks.push_back(peakSumA / peakCounts);
104  mRMSOfPeaks.push_back(sqrt(peakSumA2 / peakCounts - mMeanOfPeaks.back() * mMeanOfPeaks.back()));
105  mPeakCounts.push_back(peakCounts);
106  peakSumA = 0.;
107  peakSumA2 = 0.;
108  peakCounts = 0;
109  }
110  } else { //this is last bin
111  peakHighEdge.push_back(iAmpl->first);
112  mMeanOfPeaks.push_back(peakSumA / peakCounts);
113  mRMSOfPeaks.push_back(sqrt(peakSumA2 / peakCounts - mMeanOfPeaks.back() * mMeanOfPeaks.back()));
114  mPeakCounts.push_back(peakCounts);
115  mNPeaks++;
116  }
117  }
118  // last element of mPeakCounts, mMeanOfPeaks and mRMSOfPeaks is total count, mean and rms
119  mMeanOfPeaks.push_back(totalSumA / totalCounts);
120  mRMSOfPeaks.push_back(sqrt(totalSumA2 / totalCounts - mMeanOfPeaks.back() * mMeanOfPeaks.back()));
121  mPeakCounts.push_back(totalCounts);
122 
123  //final decision on pedestal value and RMS
124  if (mNPeaks == 1) { //everything seems to be good
125  mPedestalValue = mMeanOfPeaks.back();
126  mPedestalRMS = mRMSOfPeaks.back();
127  if ((mPedestalRMS > mSuspiciousPedestalRMS) && ((mPedestalValue + mPedestalRMS * mZSnSigmas) < peakHighEdge.back())) {
128  mPedestalRMS = (peakHighEdge.back() - mPedestalValue) / mZSnSigmas;
129  }
130  } else if (mNPeaks > 1) { // there are some problems with several pedestal peaks
131  mPedestalValue = mMeanOfPeaks.at(mNPeaks - 1); // mean of last peak
132  //mPedestalValue = mMeanOfPeaks.back();
133  mPedestalRMS = mRMSOfPeaks.back(); // total RMS of distribution
134  if ((mPedestalValue + mPedestalRMS * mZSnSigmas) < peakHighEdge.back()) {
135  mPedestalRMS = (peakHighEdge.back() - mPedestalValue) / mZSnSigmas;
136  }
137  }
138  mIsAnalyzed = true;
139 }

245 void PedestalCalibrator::finalizeSlot(TimeSlot& slot)
246 {
247  auto& cpvParams = o2::cpv::CPVSimParams::Instance();
248  auto& toleratedChannelEfficiencyLow = cpvParams.mPedClbToleratedChannelEfficiencyLow;
249  auto& toleratedChannelEfficiencyHigh = cpvParams.mPedClbToleratedChannelEfficiencyHigh;
250  auto& nSigmasZS = cpvParams.mZSnSigmas;
251 
252  PedestalCalibData* calibData = slot.getContainer();
253  LOG(info) << "PedestalCalibrator::finalizeSlot() : finalizing slot "
254  << slot.getTFStart() << " <= TF <= " << slot.getTFEnd() << " with " << calibData->mNEvents << " events.";
255 
256  o2::cpv::Geometry geo; // CPV geometry object
257 
258  // o2::cpv::Pedestals - calibration object used at reconstruction
259  // and efficiencies vector
260  // and dead channels list
261  // and thresholds for FEE
262  // and channels with high thresholds (> 511)
263  o2::cpv::Pedestals* peds = new o2::cpv::Pedestals();
264  std::vector<float> efficiencies;
265  std::vector<int> deadChannels;
266  std::vector<int> thresholdsFEE;
267  std::vector<int> highPedChannels;
268 
269  short ccId, dil, gas, pad, ped, threshold;
270  int addr, adrThr;
271  float sigma, efficiency;
272 
273  for (int i = 0; i < geo.kNCHANNELS; i++) {
274  // Pedestals
275  ped = std::floor(calibData->mPedestalSpectra[i].getPedestalValue()) + 1;
276  sigma = calibData->mPedestalSpectra[i].getPedestalRMS();
277  peds->setPedestal(i, ped);
278  peds->setPedSigma(i, sigma);
279 
280  // efficiencies
281  float efficiency = 1. * calibData->mPedestalSpectra[i].getNEntries() / calibData->mNEvents;
282  efficiencies.push_back(efficiency);
283 
284  // dead channels
285  if (efficiency == 0.0) {
286  deadChannels.push_back(i);
287  }
288 
289  // FEE Thresholds
290  threshold = ped + std::floor(sigma * nSigmasZS) + 1;
291  if (threshold > 511) {
292  threshold = 511; // set maximum threshold for suspisious channels
293  highPedChannels.push_back(i);
294  }
295  geo.absIdToHWaddress(i, ccId, dil, gas, pad);
296  addr = ccId * 4 * 5 * 64 + dil * 5 * 64 + gas * 64 + pad;
297  adrThr = (addr << 16) + threshold;
298  // to read back: addr = (adrThr >> 16); threshold = (adrThr & 0xffff)
299  thresholdsFEE.push_back(adrThr);
300  }
301 
302  mPedestalsVec.push_back(*peds);
303  mPedEfficienciesVec.push_back(efficiencies);
304  mDeadChannelsVec.push_back(deadChannels);
305  mThresholdsFEEVec.push_back(thresholdsFEE);
306  mHighPedChannelsVec.push_back(highPedChannels);
307 
308  // metadata for o2::cpv::Pedestals
309  std::map<std::string, std::string> metaData;
310  auto className = o2::utils::MemFileHelper::getClassName(peds);
311  auto fileName = o2::ccdb::CcdbApi::generateFileName(className);
312  auto timeStamp = o2::ccdb::getCurrentTimestamp();
313  mCcdbInfoPedestalsVec.emplace_back("CPV/Calib/Pedestals", className, fileName, metaData, timeStamp, timeStamp + 31536000000); // one year validity time (in milliseconds!)
314 
315  // metadata for efficiencies
316  className = o2::utils::MemFileHelper::getClassName(efficiencies);
317  fileName = o2::ccdb::CcdbApi::generateFileName(className);
318  mCcdbInfoPedEfficienciesVec.emplace_back("CPV/PedestalRun/ChannelEfficiencies", className, fileName, metaData, timeStamp, timeStamp + 31536000000); // one year validity time (in milliseconds!)
319 
320  // metadata for dead channels
321  className = o2::utils::MemFileHelper::getClassName(deadChannels);
322  fileName = o2::ccdb::CcdbApi::generateFileName(className);
323  mCcdbInfoDeadChannelsVec.emplace_back("CPV/PedestalRun/DeadChannels", className, fileName, metaData, timeStamp, timeStamp + 31536000000); // one year validity time (in milliseconds!)
324 
325  // metadata for ThreasholdsVec
326  className = o2::utils::MemFileHelper::getClassName(thresholdsFEE);
327  fileName = o2::ccdb::CcdbApi::generateFileName(className);
328  mCcdbInfoThresholdsFEEVec.emplace_back("CPV/PedestalRun/FEEThresholds", className, fileName, metaData, timeStamp, timeStamp + 31536000000); // one year validity time (in milliseconds!)
329 
330  // metadata for high pedestal (> 511) channels
331  className = o2::utils::MemFileHelper::getClassName(highPedChannels);
332  fileName = o2::ccdb::CcdbApi::generateFileName(className);
333  mCcdbInfoHighPedChannelsVec.emplace_back("CPV/PedestalRun/HighPedChannels", className, fileName, metaData, timeStamp, timeStamp + 31536000000); // one year validity time (in milliseconds!)
334 }
