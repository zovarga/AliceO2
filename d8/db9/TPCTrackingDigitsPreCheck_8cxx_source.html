49 TPCTrackingDigitsPreCheck::precheckModifiedData TPCTrackingDigitsPreCheck::runPrecheck(o2::gpu::GPUTrackingInOutPointers* ptrs, o2::gpu::GPUO2InterfaceConfiguration* config)
50 {
51  if (ptrs->tpcPackedDigits) {
52  std::unique_ptr<precheckModifiedDataInternal> retVal = std::make_unique<precheckModifiedDataInternal>();
53  retVal->tpcDigitsMap = *ptrs->tpcPackedDigits;
54  const float zsThreshold = config->configReconstruction.tpc.zsThreshold;
55  const int maxContTimeBin = config->configGRP.continuousMaxTimeBin;
56  bool updateDigits = zsThreshold > 0 && ptrs->tpcZS == nullptr;
57  const auto& d = ptrs->tpcPackedDigits;
58  for (int i = 0; i < Sector::MAXSECTOR; i++) {
59  if (updateDigits) {
60  retVal->gpuDigits[i].reserve(d->nTPCDigits[i]);
61  }
62  for (int j = 0; j < d->nTPCDigits[i]; j++) {
63  if (maxContTimeBin && d->tpcDigits[i][j].getTimeStamp() >= maxContTimeBin) {
64  throw std::runtime_error("Digit time bin exceeds time frame length");
65  }
66  if (updateDigits) {
67  if (d->tpcDigits[i][j].getChargeFloat() >= zsThreshold) {
68  if (d->tpcDigitsMC) {
69  for (const auto& element : d->tpcDigitsMC->v[i]->getLabels(j)) {
70  retVal->gpuDigitsMC[i].addElement(retVal->gpuDigits[i].size(), element);
71  }
72  }
73  retVal->gpuDigits[i].emplace_back(d->tpcDigits[i][j]);
74  }
75  }
76  }
77  if (updateDigits) {
78  retVal->tpcDigitsMap.tpcDigits[i] = retVal->gpuDigits[i].data();
79  retVal->tpcDigitsMap.nTPCDigits[i] = retVal->gpuDigits[i].size();
80  if (ptrs->tpcPackedDigits->tpcDigitsMC) {
81  retVal->gpuDigitsMC[i].flatten_to(retVal->gpuDigitsMCConst[i].first);
82  retVal->gpuDigitsMCConst[i].second = retVal->gpuDigitsMCConst[i].first;
83  retVal->tpcDigitsMapMC.v[i] = &retVal->gpuDigitsMCConst[i].second;
84  }
85  }
86  }
87  if (updateDigits) {
88  if (ptrs->tpcPackedDigits->tpcDigitsMC) {
89  retVal->tpcDigitsMap.tpcDigitsMC = &retVal->tpcDigitsMapMC;
90  }
91  ptrs->tpcPackedDigits = &retVal->tpcDigitsMap;
92  }
93  return precheckModifiedData(std::move(retVal));
94  }
95  return precheckModifiedData();
96 }
