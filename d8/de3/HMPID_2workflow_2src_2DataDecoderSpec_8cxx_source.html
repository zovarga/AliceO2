96 void DataDecoderTask::endOfStream(framework::EndOfStreamContext& ec)
97 {
98  // Records the statistics
99  float avgEventSize; //[o2::hmpid::Geo::MAXEQUIPMENTS];
100  float avgBusyTime; //[o2::hmpid::Geo::MAXEQUIPMENTS];
101  float numOfSamples; //[o2::hmpid::Geo::N_MODULES][o2::hmpid::Geo::N_YCOLS][o2::hmpid::Geo::N_XROWS];
102  float sumOfCharges; //[o2::hmpid::Geo::N_MODULES][o2::hmpid::Geo::N_YCOLS][o2::hmpid::Geo::N_XROWS];
103  float squareOfCharges; //[o2::hmpid::Geo::N_MODULES][o2::hmpid::Geo::N_YCOLS][o2::hmpid::Geo::N_XROWS];
104  float xb;
105  float yb;
106 
107  TString filename = TString::Format("%s_stat.root", mRootStatFile.c_str());
108  LOG(info) << "Create the stat file " << filename.Data();
109  TFile mfileOut(TString::Format("%s", filename.Data()), "RECREATE");
110  TTree* theObj[Geo::N_MODULES + 1];
111  for (int i = 0; i < Geo::N_MODULES; i++) { // Create the TTree array
112  TString tit = TString::Format("HMPID Data Decoding Statistic results Mod=%d", i);
113  theObj[i] = new TTree("o2hmp", tit);
114  theObj[i]->Branch("x", &xb, "s");
115  theObj[i]->Branch("y", &yb, "s");
116  theObj[i]->Branch("Samples", &numOfSamples, "i");
117  theObj[i]->Branch("Sum_of_charges", &sumOfCharges, "l");
118  theObj[i]->Branch("Sum_of_square", &squareOfCharges, "l");
119  }
120  theObj[Geo::N_MODULES] = new TTree("o2hmp", "HMPID Data Decoding Statistic results");
121  theObj[Geo::N_MODULES]->Branch("Average_Event_Size", &avgEventSize, "F");
122  theObj[Geo::N_MODULES]->Branch("Average_Busy_Time", &avgBusyTime, "F");
123 
124  // Update the Stat for the Decoding
125  int numEqui = mDeco->getNumberOfEquipments();
126  // cycle in order to update info for the last event
127  for (int i = 0; i < numEqui; i++) {
128  if (mDeco->mTheEquipments[i]->mNumberOfEvents > 0) {
129  mDeco->updateStatistics(mDeco->mTheEquipments[i]);
130  }
131  }
132  char summaryFileName[254];
133  sprintf(summaryFileName, "%s_stat.txt", mRootStatFile.c_str());
134  mDeco->writeSummaryFile(summaryFileName);
135  for (int e = 0; e < numEqui; e++) {
136  avgEventSize = mDeco->getAverageEventSize(e);
137  avgBusyTime = mDeco->getAverageBusyTime(e);
138  theObj[Geo::N_MODULES]->Fill();
139  }
140  for (int m = 0; m < o2::hmpid::Geo::N_MODULES; m++) {
141  for (int y = 0; y < o2::hmpid::Geo::N_YCOLS; y++) {
142  for (int x = 0; x < o2::hmpid::Geo::N_XROWS; x++) {
143  xb = x;
144  yb = y;
145  numOfSamples = mDeco->getPadSamples(m, x, y);
146  sumOfCharges = mDeco->getPadSum(m, x, y);
147  squareOfCharges = mDeco->getPadSquares(m, x, y);
148  theObj[m]->Fill();
149  }
150  }
151  }
152  for (int i = 0; i <= Geo::N_MODULES; i++) {
153  theObj[i]->Write();
154  }
155 
156  mExTimer.logMes("End the Decoding ! Digits decoded = " + std::to_string(mTotalDigits) + " Frames received = " + std::to_string(mTotalFrames));
157  mExTimer.stop();
158  return;
159 }

161 // the decodeTF() function processes the the messages generated by the (sub)TimeFrame builder
162 void DataDecoderTask::decodeTF(framework::ProcessingContext& pc)
163 {
164  LOG(debug) << "*********** In decodeTF **************";
165 
166  // get the input buffer
167  auto& inputs = pc.inputs();
168 
169  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload this means that the "delayed message"
170  // mechanism created it in absence of real data from upstream. Processor should send empty output to not block the workflow
171  {
172  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
173  std::vector<InputSpec> dummy{InputSpec{"dummy", ConcreteDataMatcher{"HMP", "RAWDATA", 0xDEADBEEF}}};
174  for (const auto& ref : InputRecordWalker(inputs, dummy)) {
175  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
176  if (dh->payloadSize == 0) {
177  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
178  if (++contDeadBeef <= maxWarn) {
179  LOGP(warning, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
180  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize,
181  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
182  }
183  return;
184  }
185  }
186  contDeadBeef = 0; // if good data, reset the counter
187  }
188 
189  DPLRawParser parser(inputs, o2::framework::select("TF:HMP/RAWDATA"));
190  mDeco->mDigits.clear();
191  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
192  uint32_t* theBuffer = (uint32_t*)it.raw();
193  mDeco->setUpStream(theBuffer, it.size() + it.offset());
194  try {
195  if (mFastAlgorithm) {
196  mDeco->decodePageFast(&theBuffer);
197  } else {
198  mDeco->decodePage(&theBuffer);
199  }
200  } catch (int e) {
201  // The stream end !
202  LOG(debug) << "End Page decoding !";
203  }
204  mTotalFrames++;
205  }
206  mTotalDigits += mDeco->mDigits.size();
207 }

240 void DataDecoderTask::decodeRawFile(framework::ProcessingContext& pc)
241 {
242  LOG(info) << "*********** In decode rawfile **************";
243 
244  for (auto&& input : pc.inputs()) {
245  if (input.spec->binding == "file") {
246  const header::DataHeader* header = o2::header::get<header::DataHeader*>(input.header);
247  if (!header) {
248  return;
249  }
250 
251  auto const* raw = input.payload;
252  size_t payloadSize = header->payloadSize;
253 
254  LOG(info) << " payloadSize=" << payloadSize;
255  if (payloadSize == 0) {
256  return;
257  }
258 
259  uint32_t* theBuffer = (uint32_t*)input.payload;
260  int pagesize = header->payloadSize;
261  mDeco->setUpStream(theBuffer, pagesize);
262  try {
263  if (mFastAlgorithm) {
264  mDeco->decodePageFast(&theBuffer);
265  } else {
266  mDeco->decodePage(&theBuffer);
267  }
268  } catch (int e) {
269  // The stream end !
270  LOG(debug) << "End Page decoding !";
271  }
272  }
273  }
274  return;

278 o2::framework::DataProcessorSpec getDecodingSpec(bool askDISTSTF)
279 {
280  std::vector<o2::framework::InputSpec> inputs;
281  inputs.emplace_back("TF", o2::framework::ConcreteDataTypeMatcher{"HMP", "RAWDATA"}, o2::framework::Lifetime::Optional);
282  if (askDISTSTF) {
283  inputs.emplace_back("stdDist", "FLP", "DISTSUBTIMEFRAME", 0, Lifetime::Timeframe);
284  }
285 
286  std::vector<o2::framework::OutputSpec> outputs;
287  outputs.emplace_back("HMP", "DIGITS", 0, o2::framework::Lifetime::Timeframe);
288  outputs.emplace_back("HMP", "INTRECORDS", 0, o2::framework::Lifetime::Timeframe);
289 
290  return DataProcessorSpec{
291  "HMP-RawStreamDecoder",
292  inputs,
293  outputs,
294  AlgorithmSpec{adaptFromTask<DataDecoderTask>()},
295  Options{{"result-file", VariantType::String, "/tmp/hmpRawDecodeResults", {"Base name of the decoding results files."}},
296  {"fast-decode", VariantType::Bool, false, {"Use the fast algorithm. (error 0.8%)"}}}};
297 }
