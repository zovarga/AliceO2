50 void EntropyDecoderSpec::run(ProcessingContext& pc)
51 {
52  auto cput = mTimer.CpuTime();
53  mTimer.Start(false);
54  updateTimeDependentParams(pc);
55 
56  auto buff = pc.inputs().get<gsl::span<o2::ctf::BufferType>>("ctf");
57  // since the buff is const, we cannot use EncodedBlocks::relocate directly, instead we wrap its data to another flat object
58  // const auto ctfImage = o2::itsmft::CTF::getImage(buff.data());
59 
60  auto& rofs = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(OutputRef{"ROframes"});
61  if (mGetDigits) {
62  auto& digits = pc.outputs().make<std::vector<o2::itsmft::Digit>>(OutputRef{"Digits"});
63  if (buff.size()) {
64  mCTFCoder.decode(o2::itsmft::CTF::getImage(buff.data()), rofs, digits, mNoiseMap.get(), mPattIdConverter);
65  }
66  mTimer.Stop();
67  LOG(info) << "Decoded " << digits.size() << " digits in " << rofs.size() << " RO frames in " << mTimer.CpuTime() - cput << " s";
68  } else {
69  auto& compcl = pc.outputs().make<std::vector<o2::itsmft::CompClusterExt>>(OutputRef{"compClusters"});
70  auto& patterns = pc.outputs().make<std::vector<unsigned char>>(OutputRef{"patterns"});
71  if (buff.size()) {
72  mCTFCoder.decode(o2::itsmft::CTF::getImage(buff.data()), rofs, compcl, patterns, mNoiseMap.get(), mPattIdConverter);
73  }
74  mTimer.Stop();
75  LOG(info) << "Decoded " << compcl.size() << " clusters in " << rofs.size() << " RO frames in " << mTimer.CpuTime() - cput << " s";
76  }
77 }

85 void EntropyDecoderSpec::updateTimeDependentParams(ProcessingContext& pc)
86 {
87  static bool coderUpdated = false; // with dicts loaded from CCDB one should check also the validity of current object
88  if (!coderUpdated) {
89  coderUpdated = true;
90  if (!mCTFDictPath.empty() && mCTFDictPath != "none") {
91  mCTFCoder.createCodersFromFile<CTF>(mCTFDictPath, o2::ctf::CTFCoderBase::OpType::Decoder);
92  }
93 
94  if (mMaskNoise) {
95  if (o2::utils::Str::pathExists(mNoiseFilePath)) {
96  TFile* f = TFile::Open(mNoiseFilePath.data(), "old");
97  mNoiseMap.reset((NoiseMap*)f->Get("ccdb_object"));
98  LOG(info) << "Loaded noise map from " << mNoiseFilePath;
99  }
100  if (!mNoiseMap) {
101  throw std::runtime_error("Noise masking was requested but noise mask was not provided");
102  }
103  }
104 
105  if (mGetDigits || mMaskNoise) {
106  if (o2::utils::Str::pathExists(mClusDictPath)) {
107  mPattIdConverter.loadDictionary(mClusDictPath);
108  LOG(info) << "Loaded cluster topology dictionary from " << mClusDictPath;
109  } else {
110  LOG(info) << "Cluster topology dictionary is absent, all cluster patterns expected to be stored explicitly";
111  }
112  }
113  }
114 }

116 DataProcessorSpec getEntropyDecoderSpec(o2::header::DataOrigin orig, int verbosity, bool getDigits)
117 {
118  std::vector<OutputSpec> outputs;
119  if (getDigits) {
120  outputs.emplace_back(OutputSpec{{"Digits"}, orig, "DIGITS", 0, Lifetime::Timeframe});
121  outputs.emplace_back(OutputSpec{{"ROframes"}, orig, "DIGITSROF", 0, Lifetime::Timeframe});
122  } else {
123  outputs.emplace_back(OutputSpec{{"compClusters"}, orig, "COMPCLUSTERS", 0, Lifetime::Timeframe});
124  outputs.emplace_back(OutputSpec{{"ROframes"}, orig, "CLUSTERSROF", 0, Lifetime::Timeframe});
125  outputs.emplace_back(OutputSpec{{"patterns"}, orig, "PATTERNS", 0, Lifetime::Timeframe});
126  }
127 
128  return DataProcessorSpec{
129  EntropyDecoderSpec::getName(orig),
130  Inputs{InputSpec{"ctf", orig, "CTFDATA", 0, Lifetime::Timeframe}},
131  outputs,
132  AlgorithmSpec{adaptFromTask<EntropyDecoderSpec>(orig, verbosity, getDigits)},
133  Options{
134  {"ctf-dict", VariantType::String, o2::base::NameConf::getCTFDictFileName(), {"File of CTF decoding dictionary"}},
135  {"mask-noise", VariantType::Bool, false, {"apply noise mask to digits or clusters (involves reclusterization)"}}}};
136 }
