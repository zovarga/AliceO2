64 void GlobalFwdMatchingDPL::init(InitContext& ic)
65 {
66  //-------- init geometry and field --------//
67  o2::base::GeometryManager::loadGeometry();
68  std::unique_ptr<o2::parameters::GRPObject> grp{o2::parameters::GRPObject::loadFrom()};
69  mMatching.setMFTTriggered(!grp->isDetContinuousReadOut(o2::detectors::DetID::MFT));
70  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::MFT>::Instance();
71  if (mMatching.isMFTTriggered()) {
72  mMatching.setMFTROFrameLengthMUS(alpParams.roFrameLengthTrig / 1.e3); // MFT ROFrame duration in \mus
73  } else {
74  mMatching.setMFTROFrameLengthInBC(alpParams.roFrameLengthInBC); // MFT ROFrame duration in \mus
75  }
76  mMatching.setMCTruthOn(mUseMC);
77 
78  // set bunch filling. Eventually, this should come from CCDB
79  const auto* digctx = o2::steer::DigitizationContext::loadFromFile();
80  const auto& bcfill = digctx->getBunchFilling();
81  mMatching.setBunchFilling(bcfill);
82 
83  std::string dictPath = o2::itsmft::ClustererParam<o2::detectors::DetID::MFT>::Instance().dictFilePath;
84  std::string dictFile = o2::base::DetectorNameConf::getAlpideClusterDictionaryFileName(o2::detectors::DetID::MFT, dictPath);
85  if (o2::utils::Str::pathExists(dictFile)) {
86  mMFTDict.readFromFile(dictFile);
87  LOG(info) << "Forward track-matching is running with a provided MFT dictionary: " << dictFile;
88  } else {
89  LOG(info) << "Dictionary " << dictFile << " is absent, Matching expects MFT cluster patterns";
90  }
91  mMatching.setMFTDictionary(&mMFTDict);
92 
93  const auto& matchingParam = GlobalFwdMatchingParam::Instance();
94 
95  if (matchingParam.isMatchUpstream() && mMatchRootOutput) {
96  LOG(fatal) << "Invalid MFTMCH matching configuration: matchUpstream and enable-match-output";
97  }
98 
99  mMatching.init();
100 }

102 void GlobalFwdMatchingDPL::run(ProcessingContext& pc)
103 {
104  const auto* dh = o2::header::get<o2::header::DataHeader*>(pc.inputs().getFirstValid(true).header);
105  LOG(info) << " startOrbit: " << dh->firstTForbit;
106  mTimer.Start(false);
107 
108  RecoContainer recoData;
109  recoData.collectData(pc, *mDataRequest.get());
110 
111  mMatching.run(recoData);
112 
113  const auto& matchingParam = GlobalFwdMatchingParam::Instance();
114 
115  if (matchingParam.saveMode == kSaveTrainingData) {
116  pc.outputs().snapshot(Output{"GLO", "GLFWDMFT", 0, Lifetime::Timeframe}, mMatching.getMFTMatchingPlaneParams());
117  pc.outputs().snapshot(Output{"GLO", "GLFWDMCH", 0, Lifetime::Timeframe}, mMatching.getMCHMatchingPlaneParams());
118  pc.outputs().snapshot(Output{"GLO", "GLFWDINF", 0, Lifetime::Timeframe}, mMatching.getMFTMCHMatchInfo());
119  } else {
120  pc.outputs().snapshot(Output{"GLO", "GLFWD", 0, Lifetime::Timeframe}, mMatching.getMatchedFwdTracks());
121  }
122 
123  if (mUseMC) {
124  pc.outputs().snapshot(Output{"GLO", "GLFWD_MC", 0, Lifetime::Timeframe}, mMatching.getMatchLabels());
125  }
126  if (mMatchRootOutput) {
127  pc.outputs().snapshot(Output{"GLO", "MTC_MFTMCH", 0, Lifetime::Timeframe}, mMatching.getMFTMCHMatchInfo());
128  }
129  mTimer.Stop();
130 }

138 DataProcessorSpec getGlobalFwdMatchingSpec(bool useMC, bool matchRootOutput)
139 {
140 
141  const auto& matchingParam = GlobalFwdMatchingParam::Instance();
142 
143  std::vector<OutputSpec> outputs;
144  auto dataRequest = std::make_shared<DataRequest>();
145 
146  o2::dataformats::GlobalTrackID::mask_t src = o2::dataformats::GlobalTrackID::getSourcesMask("MFT,MCH");
147 
148  dataRequest->requestMFTClusters(false); // MFT clusters labels are not used
149  dataRequest->requestTracks(src, useMC);
150 
151  if (matchingParam.isMatchUpstream()) {
152  dataRequest->requestMFTMCHMatches(useMC); // Request MFTMCH Matches
153  }
154 
155  if (matchingParam.useMIDMatch) {
156  dataRequest->requestMCHMIDMatches(useMC); // Request MCHMID Matches
157  }
158 
159  if (matchingParam.saveMode == kSaveTrainingData) {
160  outputs.emplace_back("GLO", "GLFWDMFT", 0, Lifetime::Timeframe);
161  outputs.emplace_back("GLO", "GLFWDMCH", 0, Lifetime::Timeframe);
162  outputs.emplace_back("GLO", "GLFWDINF", 0, Lifetime::Timeframe);
163  } else {
164  outputs.emplace_back("GLO", "GLFWD", 0, Lifetime::Timeframe);
165  }
166 
167  if (useMC) {
168  outputs.emplace_back("GLO", "GLFWD_MC", 0, Lifetime::Timeframe);
169  }
170 
171  if (matchRootOutput) {
172  outputs.emplace_back("GLO", "MTC_MFTMCH", 0, Lifetime::Timeframe);
173  }
174 
175  return DataProcessorSpec{
176  "globalfwd-track-matcher",
177  dataRequest->inputs,
178  outputs,
179  AlgorithmSpec{adaptFromTask<GlobalFwdMatchingDPL>(dataRequest, useMC, matchRootOutput)},
180  Options{}};
181 }
