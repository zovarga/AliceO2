44 // just need to implement 2 special methods init + run (there is no need to inherit from anything)
45 class TOFDPLRecoWorkflowTask
46 {
47  using evIdx = o2::dataformats::EvIndex<int, int>;
48  using MatchOutputType = std::vector<o2::dataformats::MatchInfoTOF>;
49 
50  bool mUseMC = true;
51  bool mUseFIT = false;

53  public:
54  explicit TOFDPLRecoWorkflowTask(bool useMC, bool useFIT) : mUseMC(useMC), mUseFIT(useFIT) {}
55 
56  void init(framework::InitContext& ic)
57  {
58  // nothing special to be set up
59  o2::base::GeometryManager::loadGeometry();
60  o2::base::Propagator::initFieldFromGRP();
61  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
62  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
63  if (o2::utils::Str::pathExists(matLUTFile)) {
64  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
65  o2::base::Propagator::Instance()->setMatLUT(lut);
66  LOG(info) << "Loaded material LUT from " << matLUTFile;
67  } else {
68  LOG(info) << "Material LUT " << matLUTFile << " file is absent, only TGeo can be used";
69  }
70 
71  mTimer.Stop();
72  mTimer.Reset();
73  }
74 
75  void run(framework::ProcessingContext& pc)
76  {
77  mTimer.Start(false);
78  //>>>---------- attach input data --------------->>>
79  const auto clustersRO = pc.inputs().get<gsl::span<o2::tof::Cluster>>("tofcluster");
80  const auto tracksRO = pc.inputs().get<gsl::span<o2::dataformats::TrackTPCITS>>("globaltrack");
81 
82  if (mUseFIT) {
83  // Note: the particular variable will go out of scope, but the span is passed by copy to the
84  // worker and the underlying memory is valid throughout the whole computation
85  auto recPoints = std::move(pc.inputs().get<gsl::span<o2::ft0::RecPoints>>("fitrecpoints"));
86  mMatcher.setFITRecPoints(recPoints);
87  LOG(info) << "TOF Reco Workflow pulled " << recPoints.size() << " FIT RecPoints";
88  }
89 
90  // we do a copy of the input but we are looking for a way to avoid it (current problem in conversion form unique_ptr to *)
91 
92  gsl::span<const o2::MCCompLabel> itstpclab;
93  o2::dataformats::MCTruthContainer<o2::MCCompLabel> toflab;
94  if (mUseMC) {
95  const auto toflabel = pc.inputs().get<o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("tofclusterlabel");
96  itstpclab = pc.inputs().get<gsl::span<o2::MCCompLabel>>("itstpclabel");
97  toflab = std::move(*toflabel);
98  }
99 
100  mMatcher.run(tracksRO, clustersRO, toflab, itstpclab);
101 
102  // in run_match_tof aggiugnere esplicitamente la chiamata a fill del tree (nella classe MatchTOF) e il metodo per leggere i vettori di output
103 
104  //...
105  // LOG(info) << "TOF CLUSTERER : TRANSFORMED " << digits->size()
106  // << " DIGITS TO " << mClustersArray.size() << " CLUSTERS";
107 
108  // send matching-info
109  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "MTC_ITSTPC", 0, Lifetime::Timeframe}, mMatcher.getMatchedTrackVector());
110  if (mUseMC) {
111  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "MCMATCHTOF", 0, Lifetime::Timeframe}, mMatcher.getMatchedTOFLabelsVector());
112  }
113  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "CALIBDATA", 0, Lifetime::Timeframe}, mMatcher.getCalibVector());
114  mTimer.Stop();
115  }
116 
117  void endOfStream(EndOfStreamContext& ec)
118  {
119  LOGF(info, "TOF Matching total timing: Cpu: %.3e Real: %.3e s in %d slots",
120  mTimer.CpuTime(), mTimer.RealTime(), mTimer.Counter() - 1);
121  }
122 
123  private:
124  o2::globaltracking::MatchTOF mMatcher;
125  TStopwatch mTimer;
126 };

128 o2::framework::DataProcessorSpec getTOFRecoWorkflowSpec(bool useMC, bool useFIT)
129 {
130  std::vector<InputSpec> inputs;
131  std::vector<OutputSpec> outputs;
132  inputs.emplace_back("tofcluster", o2::header::gDataOriginTOF, "CLUSTERS", 0, Lifetime::Timeframe);
133  inputs.emplace_back("globaltrack", "GLO", "TPCITS", 0, Lifetime::Timeframe);
134  if (useMC) {
135  inputs.emplace_back("tofclusterlabel", o2::header::gDataOriginTOF, "CLUSTERSMCTR", 0, Lifetime::Timeframe);
136  inputs.emplace_back("itstpclabel", "GLO", "TPCITS_MC", 0, Lifetime::Timeframe);
137  }
138 
139  if (useFIT) {
140  inputs.emplace_back("fitrecpoints", o2::header::gDataOriginFT0, "RECPOINTS", 0, Lifetime::Timeframe);
141  }
142 
143  outputs.emplace_back(o2::header::gDataOriginTOF, "MTC_ITSTPC", 0, Lifetime::Timeframe);
144  if (useMC) {
145  outputs.emplace_back(o2::header::gDataOriginTOF, "MCMATCHTOF", 0, Lifetime::Timeframe);
146  }
147  outputs.emplace_back(o2::header::gDataOriginTOF, "CALIBDATA", 0, Lifetime::Timeframe);
148 
149  return DataProcessorSpec{
150  "TOFRecoWorkflow",
151  inputs,
152  outputs,
153  AlgorithmSpec{adaptFromTask<TOFDPLRecoWorkflowTask>(useMC, useFIT)},
154  Options{
155  {"material-lut-path", VariantType::String, "", {"Path of the material LUT file"}}}};
156 }
