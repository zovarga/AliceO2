130 bool ConfigParamsHelper::dpl2BoostOptions(const std::vector<ConfigParamSpec>& spec,
131  boost::program_options::options_description& options,
132  boost::program_options::options_description vetos)
133 {
134  bool haveOption = false;
135  for (const auto& configSpec : spec) {
136  // skip everything found in the veto definition
137  try {
138  if (vetos.find_nothrow(configSpec.name, false) != nullptr) {
139  continue;
140  }
141  } catch (boost::program_options::ambiguous_option& e) {
142  for (auto const& alternative : e.alternatives()) {
143  std::cerr << alternative << std::endl;
144  }
145  throw;
146  }
147 
148  haveOption = true;
149  std::stringstream defaultValue;
150  defaultValue << configSpec.defaultValue;
151  if (configSpec.type != VariantType::Bool) {
152  if (configSpec.defaultValue.type() != VariantType::Empty) {
153  options.add_options()(configSpec.name.c_str(),
154  bpo::value<std::string>()->default_value(defaultValue.str()),
155  configSpec.help.c_str());
156  } else {
157  options.add_options()(configSpec.name.c_str(),
158  bpo::value<std::string>(),
159  configSpec.help.c_str());
160  }
161  } else {
162  if (configSpec.defaultValue.type() != VariantType::Empty) {
163  options.add_options()(configSpec.name.c_str(),
164  bpo::value<bool>()->zero_tokens()->default_value(configSpec.defaultValue.get<bool>()),
165  configSpec.help.c_str());
166  } else {
167  options.add_options()(configSpec.name.c_str(),
168  bpo::value<bool>()->zero_tokens(),
169  configSpec.help.c_str());
170  }
171  }
172  }
173 
174  return haveOption;
175 }
