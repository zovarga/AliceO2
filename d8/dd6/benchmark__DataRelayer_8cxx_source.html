53 // A simple test where an input is provided
54 // and the subsequent InputRecord is immediately requested.
55 static void BM_RelaySingleSlot(benchmark::State& state)
56 {
57  Monitoring metrics;
58  InputSpec spec{"clusters", "TPC", "CLUSTERS"};
59 
60  std::vector<InputRoute> inputs = {
61  InputRoute{spec, 0, "Fake", 0}};
62 
63  std::vector<ForwardRoute> forwards;
64  TimesliceIndex index{1};
65 
66  auto policy = CompletionPolicyHelpers::consumeWhenAny();
67  DataRelayer relayer(policy, inputs, metrics, index);
68  relayer.setPipelineLength(4);
69 
70  // Let's create a dummy O2 Message with two headers in the stack:
71  // - DataHeader matching the one provided in the input
72  DataHeader dh;
73  dh.dataDescription = "CLUSTERS";
74  dh.dataOrigin = "TPC";
75  dh.subSpecification = 0;
76 
77  DataProcessingHeader dph{0, 1};
78  Stack stack{dh, dph};
79  auto transport = FairMQTransportFactory::CreateTransportFactory("zeromq");
80  // we are creating the inflight messages once outside the benchmark
81  // loop and make sure that they are moved back to the original vector
82  // when processed by the relayer
83  std::vector<FairMQMessagePtr> inflightMessages;
84  inflightMessages.emplace_back(transport->CreateMessage(stack.size()));
85  inflightMessages.emplace_back(transport->CreateMessage(1000));
86  memcpy(inflightMessages[0]->GetData(), stack.data(), stack.size());
87 
88  for (auto _ : state) {
89  relayer.relay(inflightMessages[0]->GetData(), inflightMessages.data(), inflightMessages.size());
90  std::vector<RecordAction> ready;
91  relayer.getReadyToProcess(ready);
92  assert(ready.size() == 1);
93  assert(ready[0].slot.index == 0);
94  assert(ready[0].op == CompletionPolicy::CompletionOp::Consume);
95  auto result = relayer.consumeAllInputsForTimeslice(ready[0].slot);
96  assert(result.size() == 1);
97  assert(result.at(0).size() == 1);
98  inflightMessages = std::move(result[0].messages);
99  }
100 }
