37 uint64_t calib_processing_helper::processRawData(o2::framework::InputRecord& inputs, std::unique_ptr<RawReaderCRU>& reader, bool useOldSubspec, const std::vector<int>& sectors, size_t* nerrors)
38 {
39  std::vector<InputSpec> filter = {{"check", ConcreteDataTypeMatcher{o2::header::gDataOriginTPC, "RAWDATA"}, Lifetime::Timeframe}};
40  size_t errorCount = 0;
41  // TODO: check if presence of data sampling can be checked in another way
42  bool sampledData = true;
43  for ([[maybe_unused]] auto const& ref : InputRecordWalker(inputs, filter)) {
44  sampledData = false;
45  break;
46  }
47  // used for online monitor
48  if (sampledData) {
49  filter = {{"sampled-rawdata", ConcreteDataTypeMatcher{"DS2", "RAWDATA"}, Lifetime::Timeframe}};
50  for ([[maybe_unused]] auto const& ref : InputRecordWalker(inputs, filter)) {
51  sampledData = false;
52  break;
53  }
54  }
55  // used for QC
56  if (sampledData) {
57  filter = {{"sampled-rawdata", ConcreteDataTypeMatcher{"DS", "RAWDATA"}, Lifetime::Timeframe}};
58  LOGP(info, "Using sampled data");
59  }
60 
61  uint64_t activeSectors = 0;
62  bool isLinkZS = false;
63  bool readFirst = false;
64  uint32_t firstOrbit = 0;
65 
66  // for LinkZS data the maximum sync offset is needed to align the data properly.
67  // getBCsyncOffsetReference only works, if the full TF is seen. Alternatively, this value could be set
68  // fixed to e.g. 144 or 152 which is the maximum sync delay expected
69  // this is less precise and might lead to more time bins which have to be removed at the beginnig
70  // or end of the TF
71  // uint32_t syncOffsetReference = getBCsyncOffsetReference(inputs, filter);
72  uint32_t syncOffsetReference = 144;
73 
74  for (auto const& ref : InputRecordWalker(inputs, filter)) {
75  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
76  // skip empty HBF
77  if (dh->payloadSize == 2 * sizeof(o2::header::RAWDataHeader)) {
78  continue;
79  }
80 
81  firstOrbit = dh->firstTForbit;
82 
83  // ---| extract hardware information to do the processing |---
84  const auto subSpecification = dh->subSpecification;
85  rdh_utils::FEEIDType feeID = (rdh_utils::FEEIDType)dh->subSpecification;
86 
87  if (useOldSubspec) {
88  //---| old definition by Gvozden |---
89  // TODO: make auto detect from firt RDH?
90  const auto cruID = (rdh_utils::FEEIDType)(subSpecification >> 16);
91  const auto linkID = (rdh_utils::FEEIDType)((subSpecification + (subSpecification >> 8)) & 0xFF) - 1;
92  const auto endPoint = (rdh_utils::FEEIDType)((subSpecification >> 8) & 0xFF) > 0;
93  feeID = rdh_utils::getFEEID(cruID, endPoint, linkID);
94  }
95 
96  const uint64_t sector = rdh_utils::getCRU(feeID) / 10;
97 
98  // sector selection should be better done by directly subscribing to a range of subspecs. But this might not be that simple
99  if (sectors.size() && (std::find(sectors.begin(), sectors.end(), int(sector)) == sectors.end())) {
100  continue;
101  }
102 
103  activeSectors |= (0x1 << sector);
104 
105  // ===| for debugging only |===
106  // remove later
107  rdh_utils::FEEIDType cruID, linkID, endPoint;
108  rdh_utils::getMapping(feeID, cruID, endPoint, linkID);
109  const auto globalLinkID = linkID + endPoint * 12;
110  LOGP(debug, "Specifier: {}/{}/{} Part {} of {}", dh->dataOrigin, dh->dataDescription, subSpecification, dh->splitPayloadIndex, dh->splitPayloadParts);
111  LOGP(debug, "Payload size: {}", dh->payloadSize);
112  LOGP(debug, "CRU: {}; linkID: {}; endPoint: {}; globalLinkID: {}", cruID, linkID, endPoint, globalLinkID);
113  // ^^^^^^
114 
115  // TODO: exception handling needed?
116  const gsl::span<const char> raw = inputs.get<gsl::span<char>>(ref);
117  std::unique_ptr<o2::framework::RawParser<8192>> rawparserPtr;
118  try {
119 
120  o2::framework::RawParser parser(raw.data(), raw.size());
121  // detect decoder type by analysing first RDH
122  if (!readFirst) {
123  auto it = parser.begin();
124  auto* rdhPtr = it.get_if<o2::header::RAWDataHeaderV6>();
125  if (!rdhPtr) {
126  throw std::runtime_error("could not get RDH from packet");
127  }
128  const auto link = RDHUtils::getLinkID(*rdhPtr);
129  const auto detField = RDHUtils::getDetectorField(*rdhPtr);
130  if ((link == rdh_utils::UserLogicLinkID) || (detField == 1)) {
131  LOGP(info, "Detected Link-based zero suppression");
132  isLinkZS = true;
133  if (!reader->getManager() || !reader->getManager()->getLinkZSCallback()) {
134  LOGP(fatal, "LinkZSCallback must be set in RawReaderCRUManager");
135  }
136  }
137 
138  // firstOrbit = RDHUtils::getHeartBeatOrbit(*rdhPtr);
139  LOGP(info, "First orbit in present TF: {}", firstOrbit);
140  readFirst = true;
141  }
142 
143  if (isLinkZS) {
144  processLinkZS(parser, reader, firstOrbit, syncOffsetReference);
145  } else {
146  processGBT(parser, reader, feeID);
147  }
148 
149  } catch (const std::exception& e) {
150  LOGP(alarm, "EXCEPTIION in processRawData: {} -> skipping part:{}/{} of spec:{}/{}/{}, size:{}", e.what(), dh->splitPayloadIndex, dh->splitPayloadParts,
151  dh->dataOrigin, dh->dataDescription, subSpecification, dh->payloadSize);
152  errorCount++;
153  continue;
154  }
155  }
156  if (nerrors) {
157  *nerrors += errorCount;
158  }
159  return activeSectors;
160 }

162 void processGBT(o2::framework::RawParser<>& parser, std::unique_ptr<RawReaderCRU>& reader, const rdh_utils::FEEIDType feeID)
163 {
164  // TODO: currently this will only work for HBa1, since the sync is in the first packet and
165  // the decoder expects all packets of one link to be processed at once
166  rdh_utils::FEEIDType cruID, linkID, endPoint;
167  rdh_utils::getMapping(feeID, cruID, endPoint, linkID);
168  const auto globalLinkID = linkID + endPoint * 12;
169 
170  // ---| update hardware information in the reader |---
171  reader->forceCRU(cruID);
172  reader->setLink(globalLinkID);
173 
174  rawreader::ADCRawData rawData;
175  rawreader::GBTFrame gFrame;
176 
177  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
178 
179  const auto size = it.size();
180  auto data = it.data();
181  // LOGP(info, "Data size: {}", size);
182 
183  int iFrame = 0;
184  for (int i = 0; i < size; i += 16) {
185  gFrame.setFrameNumber(iFrame);
186  gFrame.setPacketNumber(iFrame / 508);
187  gFrame.readFromMemory(gsl::span<const std::byte>((std::byte*)data + i, 16));
188 
189  // extract the half words from the 4 32-bit words
190  gFrame.getFrameHalfWords();
191 
192  gFrame.getAdcValues(rawData);
193  gFrame.updateSyncCheck(false);
194 
195  ++iFrame;
196  }
197  }
198 
199  reader->runADCDataCallback(rawData);
200 }

202 void processLinkZS(o2::framework::RawParser<>& parser, std::unique_ptr<RawReaderCRU>& reader, uint32_t firstOrbit, uint32_t syncOffsetReference)
203 {
204  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
205  auto* rdhPtr = it.get_if<o2::header::RAWDataHeaderV6>();
206  if (!rdhPtr) {
207  throw std::runtime_error("could not get RDH from packet");
208  }
209  // workaround for MW2 data
210  // const bool useTimeBins = true;
211  // const auto cru = RDHUtils::getCRUID(*rdhPtr);
212  // const auto feeID = (RDHUtils::getFEEID(*rdhPtr) & 0x7f) | (cru << 7);
213 
214  // skip all data that is not Link-base zero suppression
215  const auto link = RDHUtils::getLinkID(*rdhPtr);
216  const auto detField = RDHUtils::getDetectorField(*rdhPtr);
217  if (!((detField == 1) || ((detField == 0 || detField == 0xdeadbeef) && link == rdh_utils::UserLogicLinkID))) {
218  continue;
219  }
220 
221  const bool useTimeBins = false;
222  const auto feeID = RDHUtils::getFEEID(*rdhPtr);
223  const auto orbit = RDHUtils::getHeartBeatOrbit(*rdhPtr);
224  const auto data = (const char*)it.data();
225  const auto size = it.size();
226  raw_processing_helpers::processZSdata(data, size, feeID, orbit, firstOrbit, syncOffsetReference, reader->getManager()->getLinkZSCallback(), useTimeBins);
227  }
228 }

230 // find the global sync offset reference, using the large sync offset to avoid negative time bins
231 uint32_t getBCsyncOffsetReference(InputRecord& inputs, const std::vector<InputSpec>& filter)
232 {
233  uint32_t syncOffsetReference = 144;
234 
235  for (auto const& ref : InputRecordWalker(inputs, filter)) {
236  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
237  // skip empty HBF
238  if (dh->payloadSize == 2 * sizeof(o2::header::RAWDataHeader)) {
239  continue;
240  }
241 
242  const gsl::span<const char> raw = inputs.get<gsl::span<char>>(ref);
243  o2::framework::RawParser parser(raw.data(), raw.size());
244 
245  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
246  auto* rdhPtr = it.get_if<o2::header::RAWDataHeaderV6>();
247  if (!rdhPtr) {
248  throw std::runtime_error("could not get RDH from packet"); // RS: this method is not used at the moment
249  }
250 
251  // only process LinkZSdata, only supported for data where this is already set in the UL
252  const auto link = RDHUtils::getLinkID(*rdhPtr);
253  const auto detField = RDHUtils::getDetectorField(*rdhPtr);
254  if (!((detField == 1) || ((detField == 0 || detField == 0xdeadbeef) && link == rdh_utils::UserLogicLinkID))) {
255  continue;
256  }
257 
258  const auto data = (const char*)it.data();
259  const auto size = it.size();
260 
261  zerosupp_link_based::ContainerZS* zsdata = (zerosupp_link_based::ContainerZS*)data;
262  const zerosupp_link_based::ContainerZS* const zsdataEnd = (zerosupp_link_based::ContainerZS*)(data + size);
263 
264  while (zsdata < zsdataEnd) {
265  const auto& header = zsdata->cont.header;
266  // align to header word if needed
267  if (!header.hasCorrectMagicWord()) {
268  zsdata = (zerosupp_link_based::ContainerZS*)((const char*)zsdata + sizeof(zerosupp_link_based::Header));
269  continue;
270  }
271 
272  // skip trigger info
273  if (header.isTriggerInfo()) {
274  zsdata = zsdata->next();
275  continue;
276  }
277 
278  syncOffsetReference = std::max(header.syncOffsetBC, syncOffsetReference);
279 
280  // only read first time bin for each link
281  break;
282  }
283  }
284  }
285 
286  LOGP(info, "syncOffsetReference in this TF: {}", syncOffsetReference);
287  return syncOffsetReference;
