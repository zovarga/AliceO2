131 void CompressedDecodingTask::run(ProcessingContext& pc)
132 {
133  mTimer.Start(false);
134 
135  mCreationTime = std::chrono::high_resolution_clock::now().time_since_epoch().count() / 1000000;
136 
137  //RS set the 1st orbit of the TF from the O2 header, relying on rdhHandler is not good (in fact, the RDH might be eliminated in the derived data)
138  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true));
139  mInitOrbit = dh->firstTForbit;
140  if (!mConetMode) {
141  mDecoder.setFirstIR({0, mInitOrbit});
142  }
143 
144  decodeTF(pc);
145 
146  if (!mConetMode) {
147  mHasToBePosted = true;
148  } else if (mNCrateOpenTF == mNCrateCloseTF) {
149  mHasToBePosted = true;
150  }
151 
152  if (mHasToBePosted) {
153  postData(pc);
154  }
155  mTimer.Stop();
156 }

164 void CompressedDecodingTask::decodeTF(ProcessingContext& pc)
165 {
166  auto& inputs = pc.inputs();
167 
168  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload this means that the "delayed message"
169  // mechanism created it in absence of real data from upstream. Processor should send empty output to not block the workflow
170  {
171  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
172  std::vector<InputSpec> dummy{InputSpec{"dummy", ConcreteDataMatcher{"TOF", mDataDesc, 0xDEADBEEF}}};
173  for (const auto& ref : InputRecordWalker(inputs, dummy)) {
174  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
175  if (dh->payloadSize == 0) {
176  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
177  if (++contDeadBeef <= maxWarn) {
178  LOGP(warning, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
179  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize,
180  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
181  }
182  return;
183  }
184  }
185  contDeadBeef = 0; // if good data, reset the counter
186  }
187 
189  std::vector<InputSpec> sel{InputSpec{"filter", ConcreteDataTypeMatcher{"TOF", "CRAWDATA"}}};
190  for (const auto& ref : InputRecordWalker(pc.inputs(), sel)) {
191  // for (auto iit = pc.inputs().begin(), iend = pc.inputs().end(); iit != iend; ++iit) {

423  outputs.emplace_back(o2::header::gDataOriginTOF, "ERRORS", 0, Lifetime::Timeframe);
424  outputs.emplace_back(o2::header::gDataOriginTOF, "DIAFREQ", 0, Lifetime::Timeframe);

427  "tof-compressed-decoder",
428  inputs,
429  // select(std::string("x:TOF/" + inputDesc).c_str()),
430  outputs,
