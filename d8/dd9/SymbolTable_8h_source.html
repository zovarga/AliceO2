66 template <typename T>
67 SymbolTable<T>::SymbolTable(const RenormedFrequencyTable& frequencyTable) : mOffset{frequencyTable.getMinSymbol()}, mPrecision{frequencyTable.getRenormingBits()}
68 {
69  LOG(trace) << "start building symbol table";
70 
71  mIndex.reserve(frequencyTable.size() + 1); // +1 for incompressible symbol
72  mSymbols.reserve(frequencyTable.getNUsedAlphabetSymbols());
73 
74  mEscapeSymbol = [&]() -> std::unique_ptr<T> {
75  const count_t symbolFrequency = frequencyTable.getIncompressibleSymbolFrequency();
76  const count_t cumulatedFrequency = frequencyTable.getNumSamples() - symbolFrequency;
77  return std::make_unique<T>(symbolFrequency, cumulatedFrequency, this->getPrecision());
78  }();
79 
80  count_t cumulatedFrequency = 0;
81  for (const auto symbolFrequency : frequencyTable) {
82  if (symbolFrequency) {
83  mSymbols.emplace_back(symbolFrequency, cumulatedFrequency, this->getPrecision());
84  mIndex.emplace_back(&mSymbols.back());
85  cumulatedFrequency += symbolFrequency;
86  } else {
87  mIndex.emplace_back(mEscapeSymbol.get());
88  }
89  }
90 
91  // escape symbol
92  mIndex.emplace_back(mEscapeSymbol.get());
93 
94 // advanced diagnostics for debug builds
95 #if !defined(NDEBUG)
96  LOG(debug2) << "SymbolTableProperties: {"
97  << "entries:" << mSymbols.size() << ", "
98  << "sizeB: " << mSymbols.size() * sizeof(T) + mIndex.size() * sizeof(T*) << "}";
99 #endif
100 
101 #ifdef O2_RANS_PRINT_PROCESSED_DATA
102  JSONArrayLogger<count_t> arrayLogger;
103  for (auto f : frequencyTable) {
104  arrayLogger << f;
105  }
106  arrayLogger << frequencyTable.getIncompressibleSymbolFrequency();
107  LOG(info) << "symbolTableFrequencies:" << arrayLogger;
108 #endif
109 
110  LOG(trace) << "done building symbol table";
111 }

115 {

117  // static cast to unsigned: idx < 0 => (uint)idx > MAX_INT => idx > mIndex.size()
