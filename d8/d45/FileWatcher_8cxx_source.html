131 bool FileWatcher::refresh()
132 {
133  string previous = this->currentItem();
134  LOG(info) << "previous:" << previous;
135  LOG(info) << "currentFile:" << this->mCurrentFile;
136 
137  this->mFiles = load(this->mDataFolder);
138  std::sort(this->mFiles.begin(), this->mFiles.end());
139  if (this->mCurrentFile != mEndGuard) {
140  if (this->mFiles.empty()) {
141  this->mCurrentFile = mEndGuard; // list empty - stick to last element
142  } else if (this->mCurrentFile < *(this->mFiles.begin())) {
143  this->mCurrentFile = mLowGuard; // lower then first => go to first
144  } else {
145  auto it = std::find(mFiles.begin(), mFiles.end(), this->mCurrentFile);
146  if (it == this->mFiles.end()) {
147  this->mCurrentFile = mEndGuard; // not on the list -> go to last element
148  }
149  }
150  }
151  //for (auto it = this->mFiles.begin(); it != this->mFiles.end(); ++it) {
152  // LOG(info) << *it;
153  //}
154  this->mFiles.push_front(mLowGuard);
155  this->mFiles.push_back(mEndGuard);
156 
157  LOG(info) << "this->mFiles.size() = " << this->mFiles.size();
158  LOG(info) << "this->mCurrentFile = " << this->mCurrentFile;
159  LOG(info) << "current:" << this->currentItem();
160  return previous != this->currentItem();
161 }

181 void FileWatcher::saveCurrentFileToFolder(const string& destinationFolder)
182 {
