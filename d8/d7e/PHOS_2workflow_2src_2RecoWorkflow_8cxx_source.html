48  {"raw", InputType::Raw},
49  {"digits", InputType::Digits},

56 o2::framework::WorkflowSpec getWorkflow(bool disableRootInp,
57  bool disableRootOut,
58  bool propagateMC,
59  std::string const& cfgInput,
60  std::string const& cfgOutput,
61  bool fullCluOut,
62  int ddlId)
63 {
64  InputType inputType;
65 
66  try {
67  inputType = InputMap.at(cfgInput);
68  } catch (std::out_of_range&) {
69  throw std::invalid_argument(std::string("invalid input type: ") + cfgInput);
70  }
71  std::vector<OutputType> outputTypes;
72  try {
73  outputTypes = RangeTokenizer::tokenize<OutputType>(cfgOutput, [](std::string const& token) { return OutputMap.at(token); });
74  } catch (std::out_of_range&) {
75  throw std::invalid_argument(std::string("invalid output type: ") + cfgOutput);
76  }
77  auto isEnabled = [&outputTypes](OutputType type) {
78  return std::find(outputTypes.begin(), outputTypes.end(), type) != outputTypes.end();
79  };
80 
81  o2::framework::WorkflowSpec specs;
82 
83  //Raw to ....
84  if (inputType == InputType::Raw) {
85  //no explicit raw reader ??
86 
87  if (isEnabled(OutputType::Cells)) {
88  specs.emplace_back(o2::phos::reco_workflow::getRawToCellConverterSpec(ddlId));
89  if (!disableRootOut) {
90  specs.emplace_back(o2::phos::getCellWriterSpec(false));
91  }
92  }
93  if (isEnabled(OutputType::Clusters)) {
94  specs.emplace_back(o2::phos::reco_workflow::getRawToCellConverterSpec(ddlId));
95  specs.emplace_back(o2::phos::reco_workflow::getCellClusterizerSpec(false, fullCluOut)); //no MC propagation
96  if (!disableRootOut) {
97  specs.emplace_back(o2::phos::getClusterWriterSpec(false));
98  }
99  }
100  }
101 
102  // Digits to ....
103  if (inputType == InputType::Digits) {
104  if (!disableRootInp) {
105  specs.emplace_back(o2::phos::getDigitsReaderSpec(propagateMC));
106  }
107  if (isEnabled(OutputType::Cells)) {
108  // add converter for cells
109  specs.emplace_back(o2::phos::reco_workflow::getCellConverterSpec(propagateMC));
110  if (!disableRootOut) {
111  specs.emplace_back(o2::phos::getCellWriterSpec(propagateMC));
112  }
113  } else {
114  if (isEnabled(OutputType::Clusters)) {
115  specs.emplace_back(o2::phos::reco_workflow::getClusterizerSpec(propagateMC, fullCluOut));
116  if (!disableRootOut) {
117  specs.emplace_back(o2::phos::getClusterWriterSpec(propagateMC));
118  }
119  }
120  }
121  }
122 
123  //Cells to
124  if (inputType == InputType::Cells) {
125  if (!disableRootInp) {
126  specs.emplace_back(o2::phos::getCellReaderSpec(propagateMC));
127  }
128  if (isEnabled(OutputType::Clusters)) {
129  // add clusterizer
130  specs.emplace_back(o2::phos::reco_workflow::getCellClusterizerSpec(propagateMC, fullCluOut));
131  if (!disableRootOut) {
132  specs.emplace_back(o2::phos::getClusterWriterSpec(propagateMC));
133  }
134  }
135  }
136 
137  return std::move(specs);
138 }
