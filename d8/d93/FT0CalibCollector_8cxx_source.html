27 void FT0CalibInfoSlot::fill(const gsl::span<const o2::ft0::FT0CalibrationInfoObject> data)
28 {
29  // fill container
30  // we first order the data that arrived, to improve speed when filling
31  int nd = data.size();
32  LOG(info) << "FT0CalibInfoSlot::fill entries in incoming data = " << nd;
33  std::vector<int> ord(nd);
34  std::iota(ord.begin(), ord.end(), 0);
35  std::sort(ord.begin(), ord.end(), [&data](int i, int j) { return data[i].getChannelIndex() < data[j].getChannelIndex(); });
36  int chPrev = 0, offsPrev = 0;
37  for (int i = 0; i < nd; i++) {
38  if (std::abs(data[ord[i]].getTime()) > HISTO_RANGE) {
39  continue;
40  }
41  const auto& dti = data[ord[i]];
42  auto ch = dti.getChannelIndex();
43  auto offset = offsPrev;
44  if (ch > chPrev) {
45  offset += std::accumulate(mEntriesSlot.begin() + chPrev, mEntriesSlot.begin() + ch, 0);
46  }
47  offsPrev = offset;
48  chPrev = ch;
49  auto it = mFT0CollectedCalibInfoSlot.emplace(mFT0CollectedCalibInfoSlot.begin() + offset, data[ord[i]].getChannelIndex(), data[ord[i]].getTime(), data[ord[i]].getAmp(), data[ord[i]].getTimeStamp());
50  mEntriesSlot[ch]++;
51  }
52 }

54 void FT0CalibInfoSlot::merge(const FT0CalibInfoSlot* prev)
55 {
56  // merge data of 2 slots
57 
58  LOG(info) << "Merging two slots with entries: current slot -> " << mFT0CollectedCalibInfoSlot.size() << " , previous slot -> " << prev->mFT0CollectedCalibInfoSlot.size();
59 
60  int offset = 0, offsetPrev = 0;
61  std::vector<o2::ft0::FT0CalibrationInfoObject> tmpVector;
62  for (int ch = 0; ch < NCHANNELS; ch++) {
63  if (mEntriesSlot[ch] != 0) {
64  for (int i = offset; i < offset + mEntriesSlot[ch]; i++) {
65  tmpVector.emplace_back(mFT0CollectedCalibInfoSlot[i]);
66  }
67  offset += mEntriesSlot[ch];
68  }
69  if (prev->mEntriesSlot[ch] != 0) {
70  for (int i = offsetPrev; i < offsetPrev + prev->mEntriesSlot[ch]; i++) {
71  tmpVector.emplace_back(prev->mFT0CollectedCalibInfoSlot[i]);
72  }
73  offsetPrev += prev->mEntriesSlot[ch];
74  mEntriesSlot[ch] += prev->mEntriesSlot[ch];
75  }
76  }
77  mFT0CollectedCalibInfoSlot.swap(tmpVector);
78  LOG(debug) << "After merging the size is " << mFT0CollectedCalibInfoSlot.size();
79  return;
80 }

123 bool FT0CalibCollector::hasEnoughData(const Slot& slot) const
124 {
125  // We define that we have enough data if the tree is big enough.
126  // each FT0CalibrationInfoObject is composed of two int8 and one int16 --> 32 bytes
127  // E.g. supposing that we have 500000 entries per channel --> 500 eneries per one amplitude bin
128  // we can check if we have 500000*o2::ft0::Geometry::NCHANNELS entries in the vector
129 
130  const o2::ft0::FT0CalibInfoSlot* c = slot.getContainer();
131  LOG(info) << "we have " << c->getCollectedCalibInfoSlot().size() << " entries";
132  int maxNumberOfHits = mAbsMaxNumOfHits ? mMaxNumOfHits : mMaxNumOfHits * NCHANNELS;
133  if (mTFsendingPolicy || c->getCollectedCalibInfoSlot().size() > maxNumberOfHits) {
134  return true;
135  }
136  return false;
137 }
