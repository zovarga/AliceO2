268 GPUdi() int TPCFastSpaceChargeCorrection::getCorrection(int slice, int row, float u, float v, float& dx, float& du, float& dv) const
269 {
270  const SplineType& spline = getSpline(slice, row);
271  const float* splineData = getSplineData(slice, row);
272  float su = 0, sv = 0;
273  mGeo.convUVtoScaledUV(slice, row, u, v, su, sv);
274  su *= spline.getGridX1().getUmax();
275  sv *= spline.getGridX2().getUmax();
276  float dxuv[3];
277  spline.interpolateU(splineData, su, sv, dxuv);
278  dx = dxuv[0];
279  du = dxuv[1];
280  dv = dxuv[2];
281  return 0;
282 }

284 GPUdi() void TPCFastSpaceChargeCorrection::getCorrectionInvCorrectedX(
285  int slice, int row, float cu, float cv, float& x) const
286 {
287  //const RowInfo& rowInfo = getRowInfo(row);
288  const SliceRowInfo& sliceRowInfo = getSliceRowInfo(slice, row);
289  const Spline2D<float, 1>& spline = reinterpret_cast<const Spline2D<float, 1>&>(getSpline(slice, row));
290  const float* splineData = getSplineData(slice, row, 1);
291  float gridU = (cu - sliceRowInfo.CorrU0) * sliceRowInfo.scaleCorrUtoGrid;
292  float gridV = cv * sliceRowInfo.scaleCorrVtoGrid;
293  float dx = 0;
294  spline.interpolateU(splineData, gridU, gridV, &dx);
295  x = mGeo.getRowInfo(row).x + dx;
296 }

298 GPUdi() void TPCFastSpaceChargeCorrection::getCorrectionInvUV(
299  int slice, int row, float corrU, float corrV, float& nomU, float& nomV) const
300 {
301  //const RowInfo& rowInfo = getRowInfo(row);
302  const SliceRowInfo& sliceRowInfo = getSliceRowInfo(slice, row);
303  const Spline2D<float, 2>& spline = reinterpret_cast<const Spline2D<float, 2>&>(getSpline(slice, row));
304  const float* splineData = getSplineData(slice, row, 2);
305  float gridU = (corrU - sliceRowInfo.CorrU0) * sliceRowInfo.scaleCorrUtoGrid;
306  float gridV = corrV * sliceRowInfo.scaleCorrVtoGrid;
307  float duv[2];
308  spline.interpolateU(splineData, gridU, gridV, duv);
309  nomU = corrU - duv[0];
310  nomV = corrV - duv[1];
311 }

313 GPUdi() float TPCFastSpaceChargeCorrection::getMaxDriftLength(int slice, int row, float pad) const
314 {
315  const RowActiveArea& area = getSliceRowInfo(slice, row).activeArea;
316  const float* c = area.maxDriftLengthCheb;
317  float x = -1.f + 2.f * pad / mGeo.getRowInfo(row).maxPad;
318  float y = c[0] + c[1] * x;
319  float f0 = 1.;
320  float f1 = x;
321  x *= 2.f;
322  for (int i = 2; i < 5; i++) {
323  double f = x * f1 - f0;
324  y += c[i] * f;
325  f0 = f1;
326  f1 = f;
327  }
328  return y;
329 }
