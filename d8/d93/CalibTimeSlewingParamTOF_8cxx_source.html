61 float CalibTimeSlewingParamTOF::evalTimeSlewing(int channel, float totIn) const
62 {
63 
64  // totIn is in ns
65  // the correction is returned in ps
66 
67  int sector = channel / NCHANNELXSECTOR;
68  channel = channel % NCHANNELXSECTOR;
69 
70  if (sector >= NSECTORS) {
71  return 0.; // something went wrong!
72  }
73 
74  int n = (*(mChannelStart[sector]))[channel];
75  if (n < 0) {
76  return 0.;
77  }
78 
79  int nstop = mTimeSlewing[sector]->size();
80  if (channel < NCHANNELXSECTOR - 1) {
81  nstop = (*(mChannelStart[sector]))[channel + 1];
82  }
83 
84  if (n >= nstop) {
85  return 0.; // something went wrong!
86  }
87 
88  if (totIn == 0) {
89  return (float)((*(mTimeSlewing[sector]))[n].second + (*(mGlobalOffset[sector]))[channel]);
90  }
91 
92  // we convert tot from ns to ps and to unsigned short
93  unsigned short tot = (unsigned short)(totIn * 1000);
94 
95  while (n < nstop && tot > (*(mTimeSlewing[sector]))[n].first) {
96  n++;
97  }
98  n--;
99 
100  if (n < 0) { // tot is lower than the first available value
101  return 0;
102  }
103 
104  if (n == nstop - 1) {
105  return (float)((*(mTimeSlewing[sector]))[n].second + (*(mGlobalOffset[sector]))[channel]); // use the last value stored for that channel
106  }
107 
108  float w1 = (float)(tot - (*(mTimeSlewing[sector]))[n].first);
109  float w2 = (float)((*(mTimeSlewing[sector]))[n + 1].first - tot);
110 
111  return (float)((*(mGlobalOffset[sector]))[channel] + (((*(mTimeSlewing[sector]))[n].second * w2 + (*(mTimeSlewing[sector]))[n + 1].second * w1) / ((*(mTimeSlewing[sector]))[n + 1].first - (*(mTimeSlewing[sector]))[n].first)));
112 }

115 void CalibTimeSlewingParamTOF::addTimeSlewingInfo(int channel, float tot, float time)
116 {
117  // WE ARE ASSUMING THAT:
118  // channels have to be filled in increasing order (within the sector)
119  // tots have to be filled in increasing order (within the channel)
120 
121  // tot here is provided in ns, time in ps;
122  // tot will have to be converted into ps;
123  // type will have to be converted to unsigned short (tot) and short (time)
124 
125  int sector = channel / NCHANNELXSECTOR;
126  channel = channel % NCHANNELXSECTOR;
127 
128  // printf("DBG: addTimeSlewinginfo sec=%i\n",sector);
129 
130  if (sector >= NSECTORS) {
131  return; // something went wrong!
132  }
133 
134  int currentch = channel;
135  while (currentch > -1 && (*(mChannelStart[sector]))[currentch] == -1) {
136  // printf("DBG: fill channel %i\n",currentch);
137  // set also all the previous ones which were not filled
138  (*(mChannelStart[sector]))[currentch] = mTimeSlewing[sector]->size();
139  (*(mGlobalOffset[sector]))[currentch] = time;
140  currentch--;
141  }
142  // printf("DBG: emplace back (%f,%f)\n",tot,time);
143  (*(mTimeSlewing[sector])).emplace_back((unsigned short)(tot * 1000), (short)(time - (*(mGlobalOffset[sector]))[channel]));
144 }
