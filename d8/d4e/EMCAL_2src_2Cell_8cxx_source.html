39 void Cell::setTimeStamp(float timestamp)
40 {
41  // truncate:
42  const float TIME_MIN = -1. * TIME_SHIFT,
43  TIME_MAX = TIME_RANGE - TIME_SHIFT;
44  if (timestamp < TIME_MIN) {
45  timestamp = TIME_MIN;
46  } else if (timestamp > TIME_MAX) {
47  timestamp = TIME_MAX;
48  }
49  getDataRepresentation()->mTime = static_cast<uint16_t>((timestamp + TIME_SHIFT) / TIME_RESOLUTION);
50 }
