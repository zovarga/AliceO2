30 void Encoder::init(std::string_view outDir, std::string_view fileFor, int verbosity, std::vector<ROBoardConfig> configurations)
31 {
33 
34  auto linkUniqueIds = mFEEIdConfig.getConfiguredLinkUniqueIDs();
35 
36  // Initialises the GBT link encoders
37  for (auto& linkUniqueId : linkUniqueIds) {
38  auto gbtUniqueId = mFEEIdConfig.getGBTUniqueId(linkUniqueId);
39  std::vector<ROBoardConfig> gbtConfigs;
40  for (auto& cfg : configurations) {
41  if (gbtmapper::isBoardInGBT(cfg.boardId, gbtUniqueId)) {
42  gbtConfigs.emplace_back(cfg);
43  }
44  }
45  mGBTEncoders[gbtUniqueId].setConfig(gbtUniqueId, gbtConfigs);
46  }
47 
48  // Initializes the output link
49  auto ir = getOrbitIR(0);
50  mRawWriter.setVerbosity(verbosity);
51  for (uint16_t cruId = 0; cruId < 2; ++cruId) {
52  for (uint8_t epId = 0; epId < 2; ++epId) {
53  uint16_t feeId = 2 * cruId + epId;
54  std::string outFileLink = fmt::format("{}/MID", outDir);
55  if (fileFor != "all") { // single file for all links
56  outFileLink += "_alio2-cr1-flp159";
57  if (fileFor != "flp") {
58  outFileLink += fmt::format("_cru{}_{}", cruId, epId);
59  if (fileFor != "cru") {
60  outFileLink += fmt::format("_lnk{}_feeid{}", raw::sUserLogicLinkID, feeId);
61  if (fileFor != "link") {
62  throw std::runtime_error("invalid option provided for file grouping");
63  }
64  }
65  }
66  }
67  outFileLink += ".raw";
68  mRawWriter.registerLink(feeId, cruId, raw::sUserLogicLinkID, epId, outFileLink);
69 
70  for (auto gbtUniqueId : mFEEIdConfig.getGBTUniqueIdsInLink(feeId)) {
71  // Initializes the trigger response to be added to the empty HBs
72  mGBTEncoders[gbtUniqueId].processTrigger(ir, raw::sORB);
73  mGBTEncoders[gbtUniqueId].flush(mOrbitResponse[feeId], ir);
74  }
75 
76  mOrbitResponseWord[feeId] = mOrbitResponse[feeId];
77  completeWord(mOrbitResponseWord[feeId]);
78  }
79  }
80 
81  mRawWriter.setEmptyPageCallBack(this);
82 }

100 void Encoder::completeWord(std::vector<char>& buffer)
101 {
103  size_t dataSize = buffer.size();
104  size_t cruWord = 2 * o2::raw::RDHUtils::GBTWord;
105  size_t modulo = dataSize % cruWord;
106  if (modulo) {
107  dataSize += cruWord - modulo;
108  buffer.resize(dataSize, static_cast<char>(0));
109  }
110 }

112 void Encoder::writePayload(uint16_t feeId, const InteractionRecord& ir)
113 {
115 
116  std::vector<char> buf;
117  for (auto& gbtUniqueId : mFEEIdConfig.getGBTUniqueIdsInLink(feeId)) {
118  if (!mGBTEncoders[gbtUniqueId].isEmpty()) {
119  mGBTEncoders[gbtUniqueId].flush(buf, ir);
120  }
121  }
122  if (buf.empty()) {
123  return;
124  }
125 
126  // Add the orbit response
127  buf.insert(buf.begin(), mOrbitResponse[feeId].begin(), mOrbitResponse[feeId].end());
128  completeWord(buf);
129  mRawWriter.addData(feeId, feeId / 2, raw::sUserLogicLinkID, feeId % 2, ir, buf);
130 }

132 void Encoder::finalize(bool closeFile)
133 {
135  if (mLastIR.isDummy()) {
136  mLastIR.bc = 0;
137  mLastIR.orbit = mRawWriter.getHBFUtils().orbitFirst;
138  }
139  auto ir = getOrbitIR(mLastIR.orbit);
140  auto nextIr = getOrbitIR(mLastIR.orbit + 1);
141  for (uint16_t feeId = 0; feeId < 4; ++feeId) {
142  auto ir = getOrbitIR(mLastIR.orbit);
143  // Write the last payload
144  writePayload(feeId, ir);
145  // Since the regional response comes after few clocks,
146  // we might have the corresponding regional cards in the next orbit.
147  // If this is the case, we flush all data of the next orbit
148  writePayload(feeId, nextIr);
149  }
150  if (closeFile) {
151  mRawWriter.close();
152  }
153 }

155 void Encoder::process(gsl::span<const ColumnData> data, const InteractionRecord& ir, EventType eventType)
156 {
158 
159  if (ir.orbit != mLastIR.orbit) {
160  onOrbitChange(mLastIR.orbit);
161  }
162 
163  // Converts ColumnData to ROBoards
164  mConverter.process(data);
165 
166  mGBTMap.clear();
167 
168  // Group local boards according to the GBT link they belong
169  for (auto& item : mConverter.getDataMap()) {
170  auto feeId = gbtmapper::getGBTIdFromUniqueLocId(item.first);
171  mGBTMap[feeId].emplace_back(item.second);
172  }
173 
174  // Process the GBT links
175  for (auto& item : mGBTMap) {
176  mGBTEncoders[item.first].process(item.second, ir);
177  }
178  mLastIR = ir;
179 }
