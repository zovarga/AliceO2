278 GPUdi() float GPUTPCGMTrackParam::GetMirroredY(float Bz) const
279 {
280  // get Y of the point which has the same X, but located on the other side of trajectory
281  float qptBz = GetQPt() * Bz;
282  float cosPhi2 = 1.f - GetSinPhi() * GetSinPhi();
283  if (CAMath::Abs(qptBz) < 1.e-8f) {
284  qptBz = 1.e-8f;
285  }
286  if (cosPhi2 < 0.f) {
287  cosPhi2 = 0.f;
288  }
289  return GetY() - 2.f * sqrt(cosPhi2) / qptBz;
290 }
