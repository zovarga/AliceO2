79 GPUd() void ClusterAccumulator::finalize(const ChargePos& pos, Charge q, TPCTime timeOffset)
80 {
81  mQtot += q;
82 
83  mPadMean /= mQtot;
84  mTimeMean /= mQtot;
85  mPadSigma /= mQtot;
86  mTimeSigma /= mQtot;
87 
88  mPadSigma = CAMath::Sqrt(mPadSigma - mPadMean * mPadMean);
89  mTimeSigma = CAMath::Sqrt(mTimeSigma - mTimeMean * mTimeMean);
90

93  mTimeMean += timeOffset + pos.time();

96  bool leftEdge = (pad < 2);
97  bool correct = (leftEdge) ? (pad < mPadMean) : (pad > mPadMean);
