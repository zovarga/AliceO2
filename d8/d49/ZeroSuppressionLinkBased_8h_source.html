32 struct CommonHeader {
33  static constexpr uint32_t MagicWordLinkZS = 0xFC;
34  static constexpr uint32_t MagicWordTrigger = 0xAA;
35 
36  union {
37  uint64_t word0 = 0;
38  struct {
39  uint64_t bitMaskLow : 64;
40  };
41  };
42  union {
44  uint64_t word1 = 0;
45  struct {
46  uint64_t bitMaskHigh : 16;
47  uint32_t bunchCrossing : 12;
48  uint32_t numWordsPayload : 4;
49  uint32_t syncOffsetBC : 8;
50  uint32_t syncOffsetCRUCycles : 16;
51  uint32_t magicWord : 8;
52  };
53  };
54 
55  bool hasCorrectMagicWord() const { return (magicWord == MagicWordLinkZS) || (magicWord == MagicWordTrigger); }
56  bool isLinkZS() const { return (magicWord == MagicWordLinkZS); }
57  bool isTriggerInfo() const { return (magicWord == MagicWordTrigger); }
58 };

61 struct Header final : public CommonHeader {
62 
63  std::bitset<80> getChannelBits() const
64  {
65  return std::bitset<80>((std::bitset<80>(bitMaskHigh) << 64) | std::bitset<80>(bitMaskLow));
66  }
67 
68  bool isFillWord() const { return (word0 == 0xffffffffffffffff) && (word1 == 0xffffffffffffffff); }
69 };

81 template <uint32_t DataBitSizeT = 12, uint32_t SignificantBitsT = 2>
82 struct Data {
83  static constexpr uint32_t ChannelsPerWord = DataWordSizeBits / DataBitSizeT;
84  static constexpr uint32_t ChannelsPerHalfWord = ChannelsPerWord / 2;
85  static constexpr uint32_t DataBitSize = DataBitSizeT;
86  static constexpr uint32_t SignificantBits = SignificantBitsT;
87  static constexpr uint64_t BitMask = ((uint64_t(1) << DataBitSize) - 1);
88  static constexpr float FloatConversion = 1.f / float(1 << SignificantBits);
89 
90  uint64_t adcValues[2]{};
91 
93  void setADCValue(uint32_t pos, uint64_t value)
94  {
95  const uint32_t word = pos / ChannelsPerHalfWord;
96  const uint32_t posInWord = pos % ChannelsPerHalfWord;
97 
98  const uint64_t set = (value & BitMask) << (posInWord * DataBitSize);
99  const uint64_t mask = (0xFFFFFFFFFFFFFFFF ^ (BitMask << (posInWord * DataBitSize)));
100 
101  adcValues[word] &= mask;
102  adcValues[word] |= set;
103  }
104 
106  void setADCValueFloat(uint32_t pos, float value)
107  {
108  setADCValue(pos, uint64_t((value + 0.5f * FloatConversion) / FloatConversion));
109  }
110 
112  uint32_t getADCValue(uint32_t pos) const
113  {
114  const uint32_t word = pos / ChannelsPerHalfWord;
115  const uint32_t posInWord = pos % ChannelsPerHalfWord;
116 
117  return (adcValues[word] >> (posInWord * DataBitSize)) & BitMask;
118  }
119 
121  float getADCValueFloat(uint32_t pos) const
122  {
123  return float(getADCValue(pos)) * FloatConversion;
124  }
125 
127  void reset()
128  {
129  adcValues[0] = 0;
130  adcValues[1] = 0;
131  }
132 };

175 template <uint32_t DataBitSizeT = 12, uint32_t SignificantBitsT = 2, bool HasHeaderT = true>
176 struct Container {
177  static constexpr uint32_t ChannelsPerWord = DataWordSizeBits / DataBitSizeT;
178 
179  ContainerT<DataBitSizeT, SignificantBitsT, HasHeaderT> cont;
180 
182  uint32_t getADCValue(uint32_t word) { return cont.data[word / ChannelsPerWord].getADCValue(word % ChannelsPerWord); }
183 
185  float getADCValueFloat(uint32_t word) { return cont.data[word / ChannelsPerWord].getADCValueFloat(word % ChannelsPerWord); }
186 
188  void setADCValue(uint32_t word, uint64_t value) { cont.data[word / ChannelsPerWord].setADCValue(word % ChannelsPerWord, value); }
189 
191  void setADCValueFloat(uint32_t word, float value) { cont.data[word / ChannelsPerWord].setADCValueFloat(word % ChannelsPerWord, value); }
192 
194  void reset()
195  {
196  for (int i = 0; i < cont.getDataWords(); ++i) {
197  cont.data[i].reset();
198  }
199  }
200 
201  uint32_t getBunchCrossing() const { return cont.getBunchCrossing(); }
202  uint32_t getDataWords() const { return cont.getDataWords(); }
203  uint32_t getTotalWords() const { return cont.getTotalWords(); }
204  std::bitset<80> getChannelBits() const { return cont.getChannelBits(); }
205 
207  size_t getTotalSizeBytes() const { return getTotalWords() * DataWordSizeBytes; }
208 
210  Container* next() const
211  {
212  return (Container*)((const char*)this + getTotalWords() * DataWordSizeBytes);
213  }
