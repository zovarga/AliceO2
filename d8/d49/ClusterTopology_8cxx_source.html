52 unsigned int ClusterTopology::hashFunction(const void* key, int len)
53 {
54  //
55  // Developed from https://github.com/rurban/smhasher , function MurMur2
56  //
57  // 'm' and 'r' are mixing constants generated offline.
58  const unsigned int m = 0x5bd1e995;
59  const int r = 24;
60  // Initialize the hash
61  unsigned int h = len ^ 0xdeadbeef;
62  // Mix 4 bytes at a time into the hash
63  const unsigned char* data = (const unsigned char*)key;
64  // int recIndex=0;
65  while (len >= 4) {
66  unsigned int k = *(unsigned int*)data;
67  k *= m;
68  k ^= k >> r;
69  k *= m;
70  h *= m;
71  h ^= k;
72  data += 4;
73  len -= 4;
74  }
75  // Handle the last few bytes of the input array
76  switch (len) {
77  case 3:
78  h ^= data[2] << 16;
79  case 2:
80  h ^= data[1] << 8;
81  case 1:
82  h ^= data[0];
83  h *= m;
84  };
85  // Do a few final mixes of the hash to ensure the last few
86  // bytes are well-incorporated.
87  h ^= h >> 13;
88  h *= m;
89  h ^= h >> 15;
90  return h;
91 }

93 unsigned long ClusterTopology::getCompleteHash(int nRow, int nCol,
94  const unsigned char patt[ClusterPattern::MaxPatternBytes])
95 {
96  unsigned char extended_pattern[ClusterPattern::kExtendedPatternBytes] = {0};
97  extended_pattern[0] = (unsigned char)nRow;
98  extended_pattern[1] = (unsigned char)nCol;
99  int nBits = nRow * nCol;
100  int nBytes = nBits / 8;
101  if (nBits % 8 != 0) {
102  nBytes++;
103  }
104  memcpy(&extended_pattern[2], patt, nBytes);
105 
106  unsigned long partialHash = (unsigned long)hashFunction(extended_pattern, nBytes);
107  // The first four bytes are directly taken from partialHash
108  unsigned long completeHash = partialHash << 32;
109  // The last four bytes of the hash are the first 32 pixels of the topology.
110  // The bits reserved for the pattern that are not used are set to 0.
111  if (nBytes == 1) {
112  completeHash += ((((unsigned long)extended_pattern[2]) << 24));
113  } else if (nBytes == 2) {
114  completeHash += ((((unsigned long)extended_pattern[2]) << 24) + (((unsigned long)extended_pattern[3]) << 16));
115  } else if (nBytes == 3) {
116  completeHash += ((((unsigned long)extended_pattern[2]) << 24) + (((unsigned long)extended_pattern[3]) << 16) +
117  (((unsigned long)extended_pattern[4]) << 8));
118  } else if (nBytes >= 4) {
119  completeHash += ((((unsigned long)extended_pattern[2]) << 24) + (((unsigned long)extended_pattern[3]) << 16) +
120  (((unsigned long)extended_pattern[4]) << 8) + ((unsigned long)extended_pattern[5]));
121  } else {
122  LOG(error) << "No fired pixels in small topology";
123  throw std::runtime_error("No fired pixels in small topology");
124  }
125  return completeHash;
126 }
