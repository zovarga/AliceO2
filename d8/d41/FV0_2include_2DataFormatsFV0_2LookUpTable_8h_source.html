58  explicit LookUpTable(bool fillLinearly)
59  : mTopoVector(Constants::nPms * Constants::nChannelsPerPm, {0, 0}),
60  mInvTopo(mTopoVector.size(), 0)
61  {
62  if (fillLinearly) {
63  LOG(info) << "Mapping of global channel and (PM, PM channel) pair";
64  for (int link = 0; link < Constants::nPms; ++link) {
65  for (int ch = 0; ch < Constants::nChannelsPerPm; ++ch) {
66  mTopoVector[link * Constants::nChannelsPerPm + ch] = o2::fv0::Topo{link, ch};
67  }
68  }
69  } else {
70  // TODO: If needed: implement more realistic splitting: 1 ring -> 1 PM instead of linear
71  LOG(warning) << "Don't use it - not implemented yet.";
72  }
73 
74  // Fill inverted LUT - matters only if LUT is not linear
75  for (size_t channel = 0; channel < mTopoVector.size(); ++channel) {
76  mInvTopo[getIdx(mTopoVector[channel].pmLink, mTopoVector[channel].pmCh)] = channel;
77  }
78  }
79 
80  int getChannel(int link, int mcp) const { return mInvTopo[getIdx(link, mcp)]; }
81  int getLink(int channel) const { return mTopoVector[channel].pmLink; }
82  int getPmChannel(int channel) const { return mTopoVector[channel].pmCh; }
83  int getTcmLink() const { return mLinkTCM; }
84  bool isTCM(int link, int ep) const { return link == getTcmLink() && ep == 0; }
85  Topo getTopoPM(int globalChannelID) const { return mTopoVector[globalChannelID]; }
86  Topo getTopoTCM() const { return Topo{getTcmLink(), 0}; }
87  std::size_t getNchannels() const { return mTopoVector.size(); } //get number of global PM channels
88  void printFullMap() const
89  {
90  LOG(info) << "o2::fv0::LookUpTable::printFullMap(): mTopoVector: [globalCh link pmCh]";
91  for (size_t channel = 0; channel < mTopoVector.size(); ++channel) {
92  LOG(info) << channel << " " << mTopoVector[channel].pmLink << " " << mTopoVector[channel].pmCh;
93  }
94  LOG(info) << "o2::fv0::LookUpTable::printFullMap(): mInvTopo: [idx globalCh link pmCh]";
95  for (size_t idx = 0; idx < mInvTopo.size(); ++idx) {
96  LOG(info) << idx << " " << mInvTopo[idx] << " " << getLinkFromIdx(mInvTopo[idx]) << " " << getPmChannelFromIdx(mInvTopo[idx]);
97  }
98  }
99 
100  private:
101  std::vector<Topo> mTopoVector; // iterator of each vector element gives the global channel number
102  std::vector<int> mInvTopo; // each element is an iterator of mTopoVector
103  static constexpr int mLinkTCM = Constants::nPms;
104 
105  static int getIdx(int link, int pmCh)
106  {
107  assert(pmCh < Constants::nChannelsPerPm);
108  return link * Constants::nChannelsPerPm + pmCh;
109  }
110  static int getLinkFromIdx(int idx) { return idx / Constants::nChannelsPerPm; }
111  static int getPmChannelFromIdx(int idx) { return idx % Constants::nChannelsPerPm; }
112 
113  ClassDefNV(LookUpTable, 1);
114 };

145  auto makeMapFEEmetadata() -> std::map<Topo_t, RDHtype>
146  {
147  std::map<Topo_t, RDHtype> mapResult;
148  const uint16_t cruID = 0; //constant
149  const uint32_t endPointID = 0; //constant
150  uint64_t feeID = 0; //increments
151  //PM
152  for (int iCh = 0; iCh < Instance().getNchannels(); iCh++) {
153  auto pairInserted = mapResult.insert({makeGlobalTopo(Instance().getTopoPM(iCh)), RDHtype{}});
154  if (pairInserted.second) {
155  auto& rdhObj = pairInserted.first->second;
156  const auto& topoObj = pairInserted.first->first;
157  if constexpr (std::is_same<RDHhelper, void>::value) {
158  rdhObj.linkID = topoObj.pmLink;
159  rdhObj.endPointID = endPointID;
160  rdhObj.feeId = feeID;
161  rdhObj.cruID = cruID;
162  } else //Using RDHUtils
163  {
164  RDHhelper::setLinkID(&rdhObj, topoObj.pmLink);
165  RDHhelper::setEndPointID(&rdhObj, endPointID);
166  RDHhelper::setFEEID(&rdhObj, feeID);
167  RDHhelper::setCRUID(&rdhObj, cruID);
168  }
169  feeID++;
170  }
171  }
172  //TCM
173  {
174  auto pairInserted = mapResult.insert({makeGlobalTopo(Instance().getTopoTCM()), RDHtype{}});
175  if (pairInserted.second) {
176  auto& rdhObj = pairInserted.first->second;
177  const auto& topoObj = pairInserted.first->first;
178  if constexpr (std::is_same<RDHhelper, void>::value) {
179  rdhObj.linkID = topoObj.pmLink;
180  rdhObj.endPointID = endPointID;

183  } else //Using RDHUtils

187  RDHhelper::setFEEID(&rdhObj, feeID);

189  }

191  LOG(info) << "WARNING! CHECK LUT! TCM METADATA IS INCORRECT!";
192  }
193  }
