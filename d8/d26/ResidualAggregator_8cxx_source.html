60 void ResidualsContainer::init(const TrackResiduals* residualsEngine)
61 {
62  trackResiduals = residualsEngine;
63  fileName += std::to_string(std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count());
64  fileName += ".root";
65  fileOut = std::make_unique<TFile>(fileName.c_str(), "recreate");
66  treeOutResiduals = std::make_unique<TTree>(treeNameResiduals.c_str(), "TPC binned residuals");
67  treeOutStats = std::make_unique<TTree>(treeNameStats.c_str(), "Voxel statistics mean position and nEntries");
68  for (int iSec = 0; iSec < SECTORSPERSIDE * SIDES; ++iSec) {
69  residualsPtr[iSec] = &residuals[iSec];
70  statsPtr[iSec] = &stats[iSec];
71  stats[iSec].resize(trackResiduals->getNVoxelsPerSector());
72  for (int ix = 0; ix < trackResiduals->getNXBins(); ++ix) {
73  for (int ip = 0; ip < trackResiduals->getNY2XBins(); ++ip) {
74  for (int iz = 0; iz < trackResiduals->getNZ2XBins(); ++iz) {
75  auto& statsVoxel = stats[iSec][trackResiduals->getGlbVoxBin(ix, ip, iz)];
76  // COG estimates are set to the bin center by default
77  trackResiduals->getVoxelCoordinates(iSec, ix, ip, iz, statsVoxel.meanPos[TrackResiduals::VoxX], statsVoxel.meanPos[TrackResiduals::VoxF], statsVoxel.meanPos[TrackResiduals::VoxZ]);
78  }
79  }
80  }
81  treeOutResiduals->Branch(Form("sec%d", iSec), &residualsPtr[iSec]);
82  treeOutStats->Branch(Form("sec%d", iSec), &statsPtr[iSec]);
83  }
84 }

94 void ResidualsContainer::fill(const gsl::span<const TrackResiduals::UnbinnedResid> data)
95 {
96  // receives large vector of unbinned residuals and fills the sector-wise vectors
97  // with binned residuals and statistics
98  LOG(debug) << "Filling ResidualsContainer with vector of size " << data.size();
99  for (const auto& residIn : data) {
100  int sec = residIn.sec;
101  auto& residVecOut = residuals[sec];
102  auto& statVecOut = stats[sec];
103  std::array<unsigned char, TrackResiduals::VoxDim> bvox;
104  float yPos = residIn.y * param::MaxY / 0x7fff;
105  float zPos = residIn.z * param::MaxZ / 0x7fff;
106  if (!trackResiduals->findVoxelBin(sec, param::RowX[residIn.row], yPos, zPos, bvox)) {
107  // we are not inside any voxel
108  LOGF(debug, "Dropping residual in sec(%i), x(%f), y(%f), z(%f)", sec, param::RowX[residIn.row], yPos, zPos);
109  continue;
110  }
111  residVecOut.emplace_back(residIn.dy, residIn.dz, residIn.tgSlp, bvox);
112  auto& stat = statVecOut[trackResiduals->getGlbVoxBin(bvox)];
113  float& binEntries = stat.nEntries;
114  float oldEntries = binEntries++;
115  float norm = 1.f / binEntries;
116  // update COG for voxel bvox (don't need to update X here, it stays at the pad row radius)
117  stat.meanPos[TrackResiduals::VoxF] = (stat.meanPos[TrackResiduals::VoxF] * oldEntries + yPos / param::RowX[residIn.row]) * norm;
118  stat.meanPos[TrackResiduals::VoxZ] = (stat.meanPos[TrackResiduals::VoxZ] * oldEntries + zPos / param::RowX[residIn.row]) * norm;
119  ++nResidualsTotal;
120  }
121  treeOutResiduals->Fill();
122  for (auto& residVecOut : residuals) {
123  residVecOut.clear();
124  }
125 }

127 void ResidualsContainer::merge(ResidualsContainer* prev)
128 {
129  // the previous slot is merged to this one and afterwards
130  // the previous one will be deleted
131  for (int iSec = 0; iSec < SECTORSPERSIDE * SIDES; ++iSec) {
132  // merge statistics
133  const auto& statVecPrev = prev->stats[iSec];
134  auto& statVec = stats[iSec];
135  for (int iVox = 0; iVox < trackResiduals->getNVoxelsPerSector(); ++iVox) {
136  const auto& statPrev = statVecPrev[iVox];
137  auto& stat = statVec[iVox];
138  float norm = 1.f;
139  if (statPrev.nEntries + stat.nEntries > 0.1) {
140  // if there is at least a single entry in either of the containers we need the proper norm
141  norm /= (statPrev.nEntries + stat.nEntries);
142  }
143  stat.meanPos[TrackResiduals::VoxF] = (stat.meanPos[TrackResiduals::VoxF] * stat.nEntries + statPrev.meanPos[TrackResiduals::VoxF] * statPrev.nEntries) * norm;
144  stat.meanPos[TrackResiduals::VoxZ] = (stat.meanPos[TrackResiduals::VoxZ] * stat.nEntries + statPrev.meanPos[TrackResiduals::VoxZ] * statPrev.nEntries) * norm;
145  stat.nEntries += statPrev.nEntries;
146  }
147  // prepare merging of residuals
148  prev->treeOutResiduals->SetBranchAddress(Form("sec%d", iSec), &residualsPtr);
149  }
150  // We append the entries of the tree of the following slot to the
151  // previous slot and afterwards move the merged tree to this slot.
152  // This way the order of the entries is preserved
153  for (int i = 0; i < treeOutResiduals->GetEntries(); ++i) {
154  treeOutResiduals->GetEntry(i);
155  prev->treeOutResiduals->Fill();
156  }
157 
158  treeOutResiduals = std::move(prev->treeOutResiduals);
159 
160  nResidualsTotal += prev->nResidualsTotal;
161 }

176 bool ResidualAggregator::hasEnoughData(const Slot& slot) const
177 {
178  LOG(debug) << "There are " << slot.getContainer()->getNEntries() << " entries currently. Min entries prt voxel: " << mMinEntries;
179  auto entriesPerVoxel = slot.getContainer()->getNEntries() / (mTrackResiduals.getNVoxelsPerSector() * SECTORSPERSIDE * SIDES);
180  return entriesPerVoxel >= mMinEntries;

188 void ResidualAggregator::finalizeSlot(Slot& slot)
189 {
190  auto cont = slot.getContainer();
191  cont->print();
192  cont->fillStatisticsBranches();
193  cont->fileOut->cd();
194  cont->treeOutResiduals->Write();
195  cont->treeOutResiduals.reset();
196  cont->treeOutStats->Write();
197  cont->treeOutStats.reset();
198  cont->fileOut->Close();
199  cont->fileOut.reset();
200 }
