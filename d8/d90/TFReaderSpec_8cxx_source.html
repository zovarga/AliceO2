99 void TFReaderSpec::run(o2f::ProcessingContext& ctx)
100 {
101  if (!mTFCounter) { // RS FIXME: this is a temporary hack to avoid late-starting devices to lose the input
102  LOG(warning) << "This is a hack, sleeping 2 s at startup";
103  usleep(2000000);
104  }
105  if (!mDevice) {
106  mDevice = ctx.services().get<o2f::RawDeviceService>().device();
107  mOutputRoutes = ctx.services().get<o2f::RawDeviceService>().spec().outputs; // copy!!!
108  // start TFBuilder thread
109  mRunning = true;
110  mTFBuilderThread = std::thread(&TFReaderSpec::TFBuilder, this);
111  }
112  static auto tLastTF = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
113  static long deltaSending = 0; // time correction for sending
114  auto device = ctx.services().get<o2f::RawDeviceService>().device();
115  assert(device);
116  if (device != mDevice) {
117  throw std::runtime_error(fmt::format("FMQDevice has changed, old={} new={}", fmt::ptr(mDevice), fmt::ptr(device)));
118  }
119 
120  auto acknowledgeOutput = [this](FairMQParts& parts) {
121  int np = parts.Size();
122  for (int ip = 0; ip < np; ip += 2) {
123  const auto& msgh = parts[ip];
124  const auto* hd = o2h::get<o2h::DataHeader*>(msgh.GetData());
125  if (hd->splitPayloadIndex == 0) { // check the 1st one only
126  auto& entry = this->mSeenOutputMap[{hd->dataDescription.str, hd->dataOrigin.str}];
127  if (entry.count != this->mTFCounter) {
128  entry.count = this->mTFCounter; // acknowledge identifier seen in the data
129  LOG(debug) << "Found a part " << ip << " of " << np << " | " << hd->dataOrigin.as<std::string>() << "/" << hd->dataDescription.as<std::string>()
130  << "/" << hd->subSpecification << " part " << hd->splitPayloadIndex << " of " << hd->splitPayloadParts << " for TF " << this->mTFCounter;
131  }
132  }
133  }
134  };
135 
136  auto findOutputChannel = [&ctx, this](o2h::DataHeader& h) {
137  if (!this->mInput.rawChannelConfig.empty()) {
138  return std::string{this->mInput.rawChannelConfig};
139  } else {
140  auto outputRoutes = ctx.services().get<o2f::RawDeviceService>().spec().outputs;
141  for (auto& oroute : outputRoutes) {
142  LOG(debug) << "comparing with matcher to route " << oroute.matcher << " TSlice:" << oroute.timeslice;
143  if (o2f::DataSpecUtils::match(oroute.matcher, h.dataOrigin, h.dataDescription, h.subSpecification) && ((h.tfCounter % oroute.maxTimeslices) == oroute.timeslice)) {
144  LOG(debug) << "picking the route:" << o2f::DataSpecUtils::describe(oroute.matcher) << " channel " << oroute.channel;
145  return std::string{oroute.channel};
146  }
147  }
148  }
149  LOGP(error, "Failed to find output channel for {}/{}/{} @ timeslice {}", h.dataOrigin.str, h.dataDescription.str, h.subSpecification, h.tfCounter);
150  return std::string{};
151  };
152 
153  auto addMissingParts = [this, &findOutputChannel](TFMap& msgMap) {
154  // at least the 1st header is guaranteed to be filled by the reader, use it for extra info
155  const auto* dataptr = (*msgMap.begin()->second.get())[0].GetData();
156  const auto* hd0 = o2h::get<o2h::DataHeader*>(dataptr);
157  const auto* dph = o2h::get<o2f::DataProcessingHeader*>(dataptr);
158  for (auto& out : this->mSeenOutputMap) {
159  if (out.second.count == this->mTFCounter) { // was seen in the data
160  continue;
161  }
162  LOG(debug) << "Adding dummy output for " << out.first.dataOrigin.as<std::string>() << "/" << out.first.dataDescription.as<std::string>()
163  << "/" << out.second.defSubSpec << " for TF " << this->mTFCounter;
164  o2h::DataHeader outHeader(out.first.dataDescription, out.first.dataOrigin, out.second.defSubSpec, 0);
165  outHeader.payloadSerializationMethod = o2h::gSerializationMethodNone;
166  outHeader.firstTForbit = hd0->firstTForbit;
167  outHeader.tfCounter = hd0->tfCounter;
168  const auto fmqChannel = findOutputChannel(outHeader);
169  if (fmqChannel.empty()) { // no output channel
170  continue;
171  }
172  auto fmqFactory = this->mDevice->GetChannel(fmqChannel, 0).Transport();
173  o2h::Stack headerStack{outHeader, *dph};
174  auto hdMessage = fmqFactory->CreateMessage(headerStack.size(), fair::mq::Alignment{64});
175  auto plMessage = fmqFactory->CreateMessage(0, fair::mq::Alignment{64});
176  memcpy(hdMessage->GetData(), headerStack.data(), headerStack.size());
177  FairMQParts* parts = msgMap[fmqChannel].get();
178  if (!parts) {
179  msgMap[fmqChannel] = std::make_unique<FairMQParts>();
180  parts = msgMap[fmqChannel].get();
181  }
182  parts->AddPart(std::move(hdMessage));
183  parts->AddPart(std::move(plMessage));
184  }
185  };
186 
187  while (1) {
188  if (mTFCounter >= mInput.maxTFs) { // done
189  stopProcessing(ctx);
190  break;
191  }
192  if (mTFQueue.size()) {
193  auto tfPtr = std::move(mTFQueue.front());
194  mTFQueue.pop();
195  if (!tfPtr) {
196  LOG(error) << "Builder provided nullptr TF pointer";
197  continue;
198  }
199  if (mInput.sendDummyForMissing) {
200  for (auto& msgIt : *tfPtr.get()) { // complete with empty output for the specs which were requested but not seen in the data
201  acknowledgeOutput(*msgIt.second.get());
202  }
203  addMissingParts(*tfPtr.get());
204  }
205 
206  auto tNow = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
207  auto tDiff = tNow - tLastTF + 2 * deltaSending;
208  if (mTFCounter && tDiff < mInput.delay_us) {
209  usleep(mInput.delay_us - tDiff); // respect requested delay before sending
210  }
211  size_t nparts = 0;
212  auto tSend = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
213  for (auto& msgIt : *tfPtr.get()) {
214  acknowledgeOutput(*msgIt.second.get());
215  nparts += msgIt.second->Size() / 2;
216  device->Send(*msgIt.second.get(), msgIt.first);
217  }
218  tNow = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
219  deltaSending = mTFCounter ? tNow - tLastTF : 0;
220  LOGP(info, "Sent TF {} of {} parts, {:.4f} s elapsed from previous TF.", mTFCounter, nparts, double(deltaSending) * 1e-6);
221  deltaSending -= mInput.delay_us;
222  if (!mTFCounter || deltaSending < 0) {
223  deltaSending = 0; // correction for next delay
224  }
225  tLastTF = tNow;
226  ++mTFCounter;
227  break;
228  }
229  if (!mRunning) { // no more TFs will be provided
230  stopProcessing(ctx);
231  break;
232  }
233  usleep(5000); // wait 5ms for new TF to be built
234  }
235 }

250 void TFReaderSpec::stopProcessing(o2f::ProcessingContext& ctx)
251 {
252  LOG(info) << mTFCounter << " TFs in " << mFileFetcher->getNLoops() << " loops were sent";
253  mRunning = false;
254  mFileFetcher->stop();
255  mFileFetcher.reset();
256  if (mTFBuilderThread.joinable()) {
257  mTFBuilderThread.join();
258  }
259  if (!mInput.rawChannelConfig.empty()) {
260  auto device = ctx.services().get<o2f::RawDeviceService>().device();
261  o2f::SourceInfoHeader exitHdr;
262  exitHdr.state = o2f::InputChannelState::Completed;
263  const auto exitStack = o2h::Stack(o2h::DataHeader(o2h::gDataDescriptionInfo, o2h::gDataOriginAny, 0, 0), o2f::DataProcessingHeader(), exitHdr);
264  auto fmqFactory = device->GetChannel(mInput.rawChannelConfig, 0).Transport();
265  auto hdEOSMessage = fmqFactory->CreateMessage(exitStack.size(), fair::mq::Alignment{64});
266  auto plEOSMessage = fmqFactory->CreateMessage(0, fair::mq::Alignment{64});
267  memcpy(hdEOSMessage->GetData(), exitStack.data(), exitStack.size());
268  FairMQParts eosMsg;
269  eosMsg.AddPart(std::move(hdEOSMessage));
270  eosMsg.AddPart(std::move(plEOSMessage));
271  device->Send(eosMsg, mInput.rawChannelConfig);
272  LOG(info) << "Sent EoS message to " << mInput.rawChannelConfig;
273  } else {
274  ctx.services().get<o2f::ControlService>().endOfStream();
275  }
276  ctx.services().get<o2f::ControlService>().readyToQuit(o2f::QuitRequest::Me);
277 }

280 void TFReaderSpec::TFBuilder()
281 {
282  // build TFs and add to the queue
283  std::string tfFileName;
284  auto sleepTime = std::chrono::microseconds(mInput.delay_us > 10000 ? mInput.delay_us : 10000);
285  while (mRunning && mDevice) {
286  if (mTFQueue.size() >= size_t(mInput.maxTFCache)) {
287  std::this_thread::sleep_for(sleepTime);
288  continue;
289  }
290  tfFileName = mFileFetcher ? mFileFetcher->getNextFileInQueue() : "";
291  if (!mRunning || (tfFileName.empty() && !mFileFetcher->isRunning()) || mTFBuilderCounter >= mInput.maxTFs) {
292  // stopped or no more files in the queue is expected or needed
293  LOG(info) << "TFBuilder stops processing";
294  if (mFileFetcher) {
295  mFileFetcher->stop();
296  }
297  mRunning = false;
298  break;
299  }
300  if (tfFileName.empty()) {
301  std::this_thread::sleep_for(10ms); // fait for the files cache to be filled
302  continue;
303  }
304  LOG(info) << "Processing file " << tfFileName;
305  SubTimeFrameFileReader reader(tfFileName, mInput.detMask);
306  size_t locID = 0;
307  //try
308  {
309  while (mRunning && mTFBuilderCounter < mInput.maxTFs) {
310  if (mTFQueue.size() >= size_t(mInput.maxTFCache)) {
311  std::this_thread::sleep_for(sleepTime);
312  continue;
313  }
314  auto tf = reader.read(mDevice, mOutputRoutes, mInput.rawChannelConfig, mInput.verbosity);
315  if (tf) {
316  mTFBuilderCounter++;
317  }
318  if (mRunning && tf) {
319  mTFQueue.push(std::move(tf));
320  locID++;
321  } else {
322  break;
323  }
324  }
325  // remove already processed file from the queue, unless they are needed for further looping
326  if (mFileFetcher) {
327  mFileFetcher->popFromQueue(mFileFetcher->getNLoops() >= mInput.maxLoops);
328  }
329  } /*catch (...) {
330  LOGP(error, "Error when building {}-th TF from file {}", locID, tfFileName);
331  mFileFetcher->popFromQueue(mFileFetcher->getNLoops() >= mInput.maxLoops); // remove faile TF file
332  } */
333  }
334 }

337 o2f::DataProcessorSpec o2::rawdd::getTFReaderSpec(o2::rawdd::TFReaderInp& rinp)
338 {
339  // check which inputs are present in files to read
340  o2f::DataProcessorSpec spec;
341  spec.name = "tf-reader";
342  const DetID::mask_t DEFMask = DetID::getMask("ITS,TPC,TRD,TOF,PHS,CPV,EMC,HMP,MFT,MCH,MID,ZDC,FT0,FV0,FDD,CTP");
343  rinp.detMask = DetID::getMask(rinp.detList) & DEFMask;
344  rinp.detMaskRawOnly = DetID::getMask(rinp.detListRawOnly) & DEFMask;
345  rinp.detMaskNonRawOnly = DetID::getMask(rinp.detListNonRawOnly) & DEFMask;
346  if (rinp.rawChannelConfig.empty()) {
347  // we don't know a priori what will be the content of the TF data, so we create all possible outputs
348  for (DetID::ID id = DetID::First; id <= DetID::Last; id++) {
349  if (rinp.detMask[id]) {
350  if (!rinp.detMaskNonRawOnly[id]) {
351  spec.outputs.emplace_back(o2f::OutputSpec{o2f::ConcreteDataTypeMatcher{DetID::getDataOrigin(id), "RAWDATA"}});
352  rinp.hdVec.emplace_back(o2h::DataHeader{"RAWDATA", DetID::getDataOrigin(id), 0xDEADBEEF, 0}); // in abcence of real data this will be sent
353  }
354  //
355  if (rinp.detMaskRawOnly[id]) { // used asked to not open non-raw channels
356  continue;
357  }
358  // in case detectors were processed on FLP
359  if (id == DetID::TOF) {
360  spec.outputs.emplace_back(o2f::OutputSpec{o2f::ConcreteDataTypeMatcher{DetID::getDataOrigin(DetID::TOF), "CRAWDATA"}});
361  rinp.hdVec.emplace_back(o2h::DataHeader{"CRAWDATA", DetID::getDataOrigin(DetID::TOF), 0xDEADBEEF, 0}); // in abcence of real data this will be sent
362  } else if (id == DetID::FT0 || id == DetID::FV0 || id == DetID::FDD) {
363  spec.outputs.emplace_back(o2f::OutputSpec{DetID::getDataOrigin(id), "DIGITSBC", 0});
364  spec.outputs.emplace_back(o2f::OutputSpec{DetID::getDataOrigin(id), "DIGITSCH", 0});
365  rinp.hdVec.emplace_back(o2h::DataHeader{"DIGITSBC", DetID::getDataOrigin(id), 0, 0}); // in abcence of real data this will be sent
366  rinp.hdVec.emplace_back(o2h::DataHeader{"DIGITSCH", DetID::getDataOrigin(id), 0, 0}); // in abcence of real data this will be sent
367  } else if (id == DetID::PHS) {
368  spec.outputs.emplace_back(o2f::OutputSpec{DetID::getDataOrigin(id), "CELLS", 0});
369  spec.outputs.emplace_back(o2f::OutputSpec{DetID::getDataOrigin(id), "CELLTRIGREC", 0});
370  rinp.hdVec.emplace_back(o2h::DataHeader{"CELLS", DetID::getDataOrigin(id), 0, 0}); // in abcence of real data this will be sent
371  rinp.hdVec.emplace_back(o2h::DataHeader{"CELLTRIGREC", DetID::getDataOrigin(id), 0, 0}); // in abcence of real data this will be sent
372  } else if (id == DetID::CPV) {
373  spec.outputs.emplace_back(o2f::OutputSpec{DetID::getDataOrigin(id), "DIGITS", 0});
374  spec.outputs.emplace_back(o2f::OutputSpec{DetID::getDataOrigin(id), "DIGITTRIGREC", 0});
375  spec.outputs.emplace_back(o2f::OutputSpec{DetID::getDataOrigin(id), "RAWHWERRORS", 0});
376  rinp.hdVec.emplace_back(o2h::DataHeader{"DIGITS", DetID::getDataOrigin(id), 0, 0}); // in abcence of real data this will be sent
377  rinp.hdVec.emplace_back(o2h::DataHeader{"DIGITTRIGREC", DetID::getDataOrigin(id), 0, 0}); // in abcence of real data this will be sent
378  rinp.hdVec.emplace_back(o2h::DataHeader{"RAWHWERRORS", DetID::getDataOrigin(id), 0, 0}); // in abcence of real data this will be sent
379  } else if (id == DetID::EMC) {
380  spec.outputs.emplace_back(o2f::OutputSpec{o2f::ConcreteDataTypeMatcher{DetID::getDataOrigin(id), "CELLS"}});
381  spec.outputs.emplace_back(o2f::OutputSpec{o2f::ConcreteDataTypeMatcher{DetID::getDataOrigin(id), "CELLSTRGR"}});
382  spec.outputs.emplace_back(o2f::OutputSpec{o2f::ConcreteDataTypeMatcher{DetID::getDataOrigin(id), "DECODERERR"}});
383  rinp.hdVec.emplace_back(o2h::DataHeader{"CELLS", DetID::getDataOrigin(id), 0, 0}); // in abcence of real data this will be sent
384  rinp.hdVec.emplace_back(o2h::DataHeader{"CELLSTRGR", DetID::getDataOrigin(id), 0, 0}); // in abcence of real data this will be sent
385  rinp.hdVec.emplace_back(o2h::DataHeader{"DECODERERR", DetID::getDataOrigin(id), 0, 0}); // in abcence of real data this will be sent
386  }
387  }
388  }
389 
390  spec.outputs.emplace_back(o2f::OutputSpec{{"stfDist"}, o2h::gDataOriginFLP, o2h::gDataDescriptionDISTSTF, 0});
391  } else {
392  auto nameStart = rinp.rawChannelConfig.find("name=");
393  if (nameStart == std::string::npos) {
394  throw std::runtime_error("raw channel name is not provided");
395  }
396  nameStart += strlen("name=");
397  auto nameEnd = rinp.rawChannelConfig.find(",", nameStart + 1);
398  if (nameEnd == std::string::npos) {
399  nameEnd = rinp.rawChannelConfig.size();
400  }
401  spec.options = {o2f::ConfigParamSpec{"channel-config", o2f::VariantType::String, rinp.rawChannelConfig, {"Out-of-band channel config"}}};
402  rinp.rawChannelConfig = rinp.rawChannelConfig.substr(nameStart, nameEnd - nameStart);
403  }
404 
405  spec.algorithm = o2f::adaptFromTask<TFReaderSpec>(rinp);
406 
407  return spec;
408 }
