22 long getFutureTimestamp(int secondsInFuture)
23 {
24  std::chrono::seconds sec(secondsInFuture);
25  auto future = std::chrono::system_clock::now() + sec;
26  auto future_ms = std::chrono::time_point_cast<std::chrono::milliseconds>(future);
27  auto epoch = future_ms.time_since_epoch();
28  auto value = std::chrono::duration_cast<std::chrono::milliseconds>(epoch);
29  return value.count();
30 }
