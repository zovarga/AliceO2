36 float MSangle(float mass, float p, float xX0)
37 {
38  float beta = p / std::hypot(mass, p);
39  return 0.0136f * std::sqrt(xX0) * (1.f + 0.038f * std::log(xX0)) / (beta * p);
40 }

71 void TimeFrame::addPrimaryVertices(const std::vector<Vertex>& vertices)
72 {
73  for (const auto& vertex : vertices) {
74  mPrimaryVertices.emplace_back(vertex);
75  const int w{vertex.getNContributors()};
76  mBeamPos[0] = (mBeamPos[0] * mBeamPosWeight + vertex.getX() * w) / (mBeamPosWeight + w);
77  mBeamPos[1] = (mBeamPos[1] * mBeamPosWeight + vertex.getY() * w) / (mBeamPosWeight + w);
78  mBeamPosWeight += w;
79  }
80  mROframesPV.push_back(mPrimaryVertices.size());
81 }

83 int TimeFrame::loadROFrameData(const o2::itsmft::ROFRecord& rof, gsl::span<const itsmft::Cluster> clusters,
84  const dataformats::MCTruthContainer<MCCompLabel>* mcLabels)
85 {
86  GeometryTGeo* geom = GeometryTGeo::Instance();
87  geom->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L, o2::math_utils::TransformType::L2G));
88  int clusterId{0};
89 
90  auto first = rof.getFirstEntry();
91  auto clusters_in_frame = rof.getROFData(clusters);
92  for (auto& c : clusters_in_frame) {
93  int layer = geom->getLayer(c.getSensorID());
94 
96  auto xyz = c.getXYZGloRot(*geom);
97  addTrackingFrameInfoToLayer(layer, xyz.x(), xyz.y(), xyz.z(), c.getX(), geom->getSensorRefAlpha(c.getSensorID()),
98  std::array<float, 2>{c.getY(), c.getZ()},
99  std::array<float, 3>{c.getSigmaY2(), c.getSigmaYZ(), c.getSigmaZ2()});
100 
102  addClusterToLayer(layer, xyz.x(), xyz.y(), xyz.z(), mUnsortedClusters[layer].size());
103  addClusterExternalIndexToLayer(layer, first + clusterId);
104  clusterId++;
105  }
106 
107  for (unsigned int iL{0}; iL < mUnsortedClusters.size(); ++iL) {
108  mROframesClusters[iL].push_back(mUnsortedClusters[iL].size());
109  }
110  if (mcLabels) {
111  mClusterLabels = mcLabels;
112  }
113  mNrof++;
114  return clusters_in_frame.size();

117 int TimeFrame::loadROFrameData(gsl::span<o2::itsmft::ROFRecord> rofs, gsl::span<const itsmft::CompClusterExt> clusters, gsl::span<const unsigned char>::iterator& pattIt, const itsmft::TopologyDictionary& dict, const dataformats::MCTruthContainer<MCCompLabel>* mcLabels)
118 {
119  GeometryTGeo* geom = GeometryTGeo::Instance();
120  geom->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L, o2::math_utils::TransformType::L2G));
121 
122  mNrof = 0;
123  for (int clusterId{0}; clusterId < clusters.size() && mNrof < rofs.size(); ++clusterId) {
124  auto& c = clusters[clusterId];
125 
126  int layer = geom->getLayer(c.getSensorID());
127 
128  auto pattID = c.getPatternID();
129  o2::math_utils::Point3D<float> locXYZ;
130  float sigmaY2 = DefClusError2Row, sigmaZ2 = DefClusError2Col, sigmaYZ = 0; //Dummy COG errors (about half pixel size)
131  if (pattID != itsmft::CompCluster::InvalidPatternID) {
132  sigmaY2 = dict.getErr2X(pattID);
133  sigmaZ2 = dict.getErr2Z(pattID);
134  if (!dict.isGroup(pattID)) {
135  locXYZ = dict.getClusterCoordinates(c);
136  } else {
137  o2::itsmft::ClusterPattern patt(pattIt);
138  locXYZ = dict.getClusterCoordinates(c, patt);
139  }
140  } else {
141  o2::itsmft::ClusterPattern patt(pattIt);
142  locXYZ = dict.getClusterCoordinates(c, patt, false);
143  }
144  auto sensorID = c.getSensorID();
145  // Inverse transformation to the local --> tracking
146  auto trkXYZ = geom->getMatrixT2L(sensorID) ^ locXYZ;
147  // Transformation to the local --> global
148  auto gloXYZ = geom->getMatrixL2G(sensorID) * locXYZ;
149 
150  addTrackingFrameInfoToLayer(layer, gloXYZ.x(), gloXYZ.y(), gloXYZ.z(), trkXYZ.x(), geom->getSensorRefAlpha(sensorID),
151  std::array<float, 2>{trkXYZ.y(), trkXYZ.z()},
152  std::array<float, 3>{sigmaY2, sigmaYZ, sigmaZ2});
153 
155  addClusterToLayer(layer, gloXYZ.x(), gloXYZ.y(), gloXYZ.z(), mUnsortedClusters[layer].size());
156  addClusterExternalIndexToLayer(layer, clusterId);
157 
158  while (mNrof < rofs.size() && clusterId >= rofs[mNrof].getFirstEntry() + rofs[mNrof].getNEntries() - 1) {
159  for (unsigned int iL{0}; iL < mUnsortedClusters.size(); ++iL) {
160  mROframesClusters[iL].push_back(mUnsortedClusters[iL].size());
161  }
162  mNrof++;
163  }
164  }
165  if (mcLabels) {
166  mClusterLabels = mcLabels;
167  }
168  return mNrof;
169 }

180 void TimeFrame::initialise(const int iteration, const MemoryParameters& memParam, const TrackingParameters& trkParam)
181 {
182  if (iteration == 0) {
183  mTracks.clear();
184  mTracksLabel.clear();
185  mTracks.resize(mNrof);
186  mTracksLabel.resize(mNrof);
187  mCells.resize(trkParam.CellsPerRoad());
188  mCellsLookupTable.resize(trkParam.CellsPerRoad() - 1);
189  mCellsNeighbours.resize(trkParam.CellsPerRoad() - 1);
190  mCellLabels.resize(trkParam.CellsPerRoad());
191  mTracklets.resize(trkParam.TrackletsPerRoad());
192  mTrackletLabels.resize(trkParam.TrackletsPerRoad());
193  mTrackletsLookupTable.resize(trkParam.CellsPerRoad());
194  mIndexTables.clear();
195  mIndexTableUtils.setTrackingParameters(trkParam);
196  mPositionResolution.resize(trkParam.NLayers);
197  mBogusClusters.resize(trkParam.NLayers, 0);
198 
199  for (unsigned int iLayer{0}; iLayer < mClusters.size(); ++iLayer) {
200  if (mClusters[iLayer].size()) {
201  continue;
202  }
203  mClusters[iLayer].clear();
204  mClusters[iLayer].resize(mUnsortedClusters[iLayer].size());
205  mUsedClusters[iLayer].clear();
206  mUsedClusters[iLayer].resize(mUnsortedClusters[iLayer].size(), false);
207  mPositionResolution[iLayer] = std::hypot(trkParam.LayerMisalignment[iLayer], trkParam.LayerResolution[iLayer]);
208  }
209 
210  mIndexTables.resize(mNrof);
211  std::vector<ClusterHelper> cHelper;
212  std::vector<int> clsPerBin(trkParam.PhiBins * trkParam.ZBins, 0);
213  for (int rof{0}; rof < mNrof; ++rof) {
214  mIndexTables[rof].resize(trkParam.TrackletsPerRoad(), std::vector<int>(trkParam.ZBins * trkParam.PhiBins + 1, 0));
215  if (mMultiplicityCutMask.size() == mNrof && !mMultiplicityCutMask[rof]) {
216  continue;
217  }
218  for (int iLayer{0}; iLayer < trkParam.NLayers; ++iLayer) {
219  std::fill(clsPerBin.begin(), clsPerBin.end(), 0);
220  const auto unsortedClusters{getUnsortedClustersOnLayer(rof, iLayer)};
221  const int clustersNum{static_cast<int>(unsortedClusters.size())};
222 
223  cHelper.clear();
224  cHelper.resize(clustersNum);
225 
226  for (int iCluster{0}; iCluster < clustersNum; ++iCluster) {
227  const Cluster& c = unsortedClusters[iCluster];
228  ClusterHelper& h = cHelper[iCluster];
229  float x = c.xCoordinate - mBeamPos[0];
230  float y = c.yCoordinate - mBeamPos[1];
231  const float& z = c.zCoordinate;
232  float phi = math_utils::computePhi(x, y);
233  int zBin{mIndexTableUtils.getZBinIndex(iLayer, z)};
234  if (zBin < 0) {
235  zBin = 0;
236  mBogusClusters[iLayer]++;
237  } else if (zBin >= trkParam.ZBins) {
238  zBin = trkParam.ZBins - 1;
239  mBogusClusters[iLayer]++;
240  }
241  int bin = mIndexTableUtils.getBinIndex(zBin, mIndexTableUtils.getPhiBinIndex(phi));
242  h.phi = phi;
243  h.r = math_utils::hypot(x, y);
244  mMinR[iLayer] = o2::gpu::GPUCommonMath::Min(h.r, mMinR[iLayer]);
245  mMaxR[iLayer] = o2::gpu::GPUCommonMath::Max(h.r, mMaxR[iLayer]);
246  h.bin = bin;
247  h.ind = clsPerBin[bin]++;
248  }
249  std::vector<int> lutPerBin(clsPerBin.size());
250  lutPerBin[0] = 0;
251  for (unsigned int iB{1}; iB < lutPerBin.size(); ++iB) {
252  lutPerBin[iB] = lutPerBin[iB - 1] + clsPerBin[iB - 1];
253  }
254 
255  auto clusters2beSorted{getClustersOnLayer(rof, iLayer)};
256  for (int iCluster{0}; iCluster < clustersNum; ++iCluster) {
257  const ClusterHelper& h = cHelper[iCluster];
258 
259  Cluster& c = clusters2beSorted[lutPerBin[h.bin] + h.ind];
260  c = unsortedClusters[iCluster];
261  c.phi = h.phi;
262  c.radius = h.r;
263  c.indexTableBinIndex = h.bin;
264  }
265 
266  if (iLayer > 0) {
267  for (unsigned int iB{0}; iB < clsPerBin.size(); ++iB) {
268  mIndexTables[rof][iLayer - 1][iB] = lutPerBin[iB];
269  }
270  for (auto iB{clsPerBin.size()}; iB < (int)mIndexTables[rof][iLayer - 1].size(); iB++) {
271  mIndexTables[rof][iLayer - 1][iB] = clustersNum;
272  }
273  }
274  }
275  }
276  }
277 
278  mRoads.clear();
279  mRoadLabels.clear();
280 
281  mMSangles.resize(trkParam.NLayers);
282  mPhiCuts.resize(mClusters.size() - 1, 0.f);
283 
284  float oneOverR{0.001f * 0.3f * std::abs(mBz) / trkParam.TrackletMinPt};
285  for (unsigned int iLayer{0}; iLayer < mClusters.size(); ++iLayer) {
286  mMSangles[iLayer] = MSangle(0.14f, trkParam.TrackletMinPt, trkParam.LayerxX0[iLayer]);
287  mPositionResolution[iLayer] = std::hypot(trkParam.LayerMisalignment[iLayer], trkParam.LayerResolution[iLayer]);
288 
289  if (iLayer < mClusters.size() - 1) {
290  const float& r1 = trkParam.LayerRadii[iLayer];
291  const float& r2 = trkParam.LayerRadii[iLayer + 1];
292  const float res1 = std::hypot(trkParam.PVres, mPositionResolution[iLayer]);
293  const float res2 = std::hypot(trkParam.PVres, mPositionResolution[iLayer + 1]);
294  const float cosTheta1half = std::sqrt(1.f - Sq(0.5f * r1 * oneOverR));
295  const float cosTheta2half = std::sqrt(1.f - Sq(0.5f * r2 * oneOverR));
296  float x = r2 * cosTheta1half - r1 * cosTheta2half;
297  float delta = std::sqrt(1. / (1.f - 0.25f * Sq(x * oneOverR)) * (Sq(0.25f * r1 * r2 * Sq(oneOverR) / cosTheta2half + cosTheta1half) * Sq(res1) + Sq(0.25f * r1 * r2 * Sq(oneOverR) / cosTheta1half + cosTheta2half) * Sq(res2)));
298  mPhiCuts[iLayer] = std::min(std::asin(0.5f * x * oneOverR) + 2.f * mMSangles[iLayer] + delta, constants::math::Pi * 0.5f);
299  }
300  }
301 
302  for (unsigned int iLayer{0}; iLayer < mTracklets.size(); ++iLayer) {
303  mTracklets[iLayer].clear();
304  mTrackletLabels[iLayer].clear();
305  if (iLayer < mCells.size()) {
306  mCells[iLayer].clear();
307  mTrackletsLookupTable[iLayer].clear();
308  mTrackletsLookupTable[iLayer].resize(mClusters[iLayer + 1].size(), 0);
309  mCellLabels[iLayer].clear();
310  }
311 
312  if (iLayer < mCells.size() - 1) {
313  mCellsLookupTable[iLayer].clear();
314  mCellsNeighbours[iLayer].clear();
315  }
316  }
317 }

319 void TimeFrame::checkTrackletLUTs()
320 {
321  for (uint32_t iLayer{0}; iLayer < getTracklets().size(); ++iLayer) {
322  int prev{-1};
323  int count{0};
324  for (uint32_t iTracklet{0}; iTracklet < getTracklets()[iLayer].size(); ++iTracklet) {
325  auto& trk = getTracklets()[iLayer][iTracklet];
326  int currentId{trk.firstClusterIndex};
327  if (currentId < prev) {
328  std::cout << "First Cluster Index not increasing monotonically on L:T:ID:Prev " << iLayer << "\t" << iTracklet << "\t" << currentId << "\t" << prev << std::endl;
329  } else if (currentId == prev) {
330  count++;
331  } else {
332  if (iLayer > 0) {
333  auto& lut{getTrackletsLookupTable()[iLayer - 1]};
334  if (count != lut[prev + 1] - lut[prev]) {
335  std::cout << "LUT count broken " << iLayer - 1 << "\t" << prev << "\t" << count << "\t" << lut[prev + 1] << "\t" << lut[prev] << std::endl;
336  }
337  }
338  count = 1;
339  }
340  prev = currentId;
341  if (iLayer > 0) {
342  auto& lut{getTrackletsLookupTable()[iLayer - 1]};
343  if (iTracklet >= lut[currentId + 1] || iTracklet < lut[currentId]) {
344  std::cout << "LUT broken: " << iLayer - 1 << "\t" << currentId << "\t" << iTracklet << std::endl;
345  }
346  }
347  }
348  }
