57  void run(o2::framework::ProcessingContext& pc) final
58  {
59  // set the min range of TFs for first TF
60  if (mProcessedTFs == 0) {
61  mTFRange[0] = getCurrentTF(pc);
62  }
63 
64  for (int i = 0; i < mCRUs.size(); ++i) {
65  const DataRef ref = pc.inputs().getByPos(i);
66  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
67  const o2::tpc::CRU cruTmp(tpcCRUHeader->subSpecification);
68  mOneDIDCAggregator.aggregate1DIDCs(cruTmp.side(), pc.inputs().get<std::vector<float>>(ref), mProcessedTFs, cruTmp.region());
69  }
70  ++mProcessedTFs;
71 
72  if (!(mProcessedTFs % ((mTimeFrames + 5) / 5))) {
73  LOGP(info, "aggregated TFs: {}", mProcessedTFs);
74  }
75 
76  if (mProcessedTFs == mTimeFrames) {
77  mTFRange[1] = getCurrentTF(pc) + 1; // set the TF for last aggregated TF
78  mProcessedTFs = 0; // reset processed TFs for next aggregation interval
79 
80  // perform fourier transform of 1D-IDCs
81  auto intervals = mOneDIDCAggregator.getIntegrationIntervalsPerTF();
82  mIDCFourierTransform.setIDCs(std::move(mOneDIDCAggregator).getAggregated1DIDCs(), std::move(intervals));
83  mIDCFourierTransform.calcFourierCoefficients();
84 
85  if (mDebug) {
86  LOGP(info, "dumping FT to file");
87  mIDCFourierTransform.dumpToFile(fmt::format("FourierAGG_{:02}.root", getCurrentTF(pc)).data());
88  }
89 
90  // storing to CCDB
91  sendOutput(pc.outputs());
92  }
93  }
