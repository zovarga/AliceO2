112 template <class T>
113 inline const T CalDet<T>::getValue(const int sector, const int globalPadInSector) const
114 {
115  // This shold be a temporary speedup, a proper restructuring of Mapper and CalDet/CalArray is needed.
116  // The default granularity for the moment should be ROC, for the assumptions below this should be assured
117  assert(mPadSubset == PadSubset::ROC);
118  int roc = sector;
119  int padInROC = globalPadInSector;
120  const int padsInIROC = Mapper::getPadsInIROC();
121  if (globalPadInSector >= padsInIROC) {
122  roc += Mapper::getNumberOfIROCs();
123  padInROC -= padsInIROC;
124  }
125  return mData[roc].getValue(padInROC);
126 }

129 template <class T>
130 inline const T CalDet<T>::getValue(const ROC roc, const size_t row, const size_t pad) const
131 {
132  // TODO: might need speedup and beautification
133  static const Mapper& mapper = Mapper::instance();
134 
135  // bind row and pad to the maximum rows and pads in the requested region
136  const size_t nRows = mapper.getNumberOfRowsROC(roc);
137  const size_t mappedRow = row % nRows;
138  const size_t nPads = mapper.getNumberOfPadsInRowROC(roc, row);
139  const size_t mappedPad = pad % nPads;
140 
141  // TODO: implement missing cases
142  switch (mPadSubset) {
143  case PadSubset::ROC: {
144  return mData[roc].getValue(mappedRow, mappedPad);
145  break;
146  }
147  case PadSubset::Partition: {
148  return T{};
149  break;
150  }
151  case PadSubset::Region: {
152  const auto globalRow = roc.isOROC() ? mappedRow + mapper.getNumberOfRowsROC(ROC(0)) : mappedRow;
153  return mData[Mapper::REGION[globalRow] + roc.getSector() * Mapper::NREGIONS].getValue(Mapper::OFFSETCRUGLOBAL[globalRow] + mappedPad);
154  break;
155  }
156  }
157  return T{};
158 }

161 template <class T>
162 inline const T CalDet<T>::getValue(const CRU cru, const size_t row, const size_t pad) const
163 {
164  // TODO: might need speedup and beautification
165  static const Mapper& mapper = Mapper::instance();
166  const auto& info = mapper.getPadRegionInfo(cru.region());
167 
168  // bind row and pad to the maximum rows and pads in the requested region
169  const size_t nRows = info.getNumberOfPadRows();
170  const size_t mappedRow = row % nRows;
171  const size_t nPads = info.getPadsInRowRegion(mappedRow);
172  const size_t mappedPad = pad % nPads;
173 
174  switch (mPadSubset) {
175  case PadSubset::ROC: {
176  const ROC roc = cru.roc();
177  const size_t irocOffset = (cru.rocType() == RocType::IROC) ? 0 : mapper.getNumberOfRowsROC(0);
178  const size_t rowROC = mappedRow + info.getGlobalRowOffset() - irocOffset;
179  const size_t channel = mapper.getPadNumberInROC(PadROCPos(roc, rowROC, mappedPad));
180  // printf("roc %2d, row %2zu, pad %3zu, channel: %3zu\n", roc.getRoc(), rowROC, mappedPad, channel);
181  return mData[roc].getValue(channel);
182  break;
183  }
184  case PadSubset::Partition: {
185  break;
186  }
187  case PadSubset::Region: {
188  return mData[cru].getValue(mappedRow, mappedPad);
189  break;
190  }
191  }
192  return T{};
193 }

375 template <class T>
376 void CalDet<T>::initData()
377 {
378  const auto& mapper = Mapper::instance();
379 
380  // ---| Define number of sub pad regions |------------------------------------
381  size_t size = 0;
382  std::string frmt;
383  switch (mPadSubset) {
384  case PadSubset::ROC: {
385  size = ROC::MaxROC;
386  frmt = "%1%_ROC_%2$02d";
387  break;
388  }
389  case PadSubset::Partition: {
390  size = Sector::MAXSECTOR * mapper.getNumberOfPartitions();
391  frmt = "%1%_Partition_%2$02d";
392  break;
393  }
394  case PadSubset::Region: {
395  size = Sector::MAXSECTOR * mapper.getNumberOfPadRegions();
396  frmt = "%1%_Region_%2$02d";
397  break;
398  }
399  }
400 
401  for (size_t i = 0; i < size; ++i) {
402  mData.push_back(CalType(mPadSubset, i));
403  mData.back().setName(boost::str(boost::format(frmt) % mName % i));
404  }
405 }
