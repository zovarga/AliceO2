78  mPt = new TH1F("mPt", "Pt distribution of matched tracks; Pt [GeV/c]; dNdPt", 100, 0.f, 20.f);
79  mPhiTPC = new TH1F("mPhiTPC", "Phi distribution of TPC tracks; Phi [rad]; dNdPhi", 100, 0.f, 2 * TMath::Pi());
80  mFractionITSTPCmatchPhi = new TEfficiency("mFractionITSTPCmatchPhi", "Fraction of ITSTPC matched tracks vs Phi; Phi [rad]; Eff", 100, 0.f, 2 * TMath::Pi());
81  mPhi = new TH1F("mPhi", "Phi distribution of matched tracks; Phi [rad]; dNdPhi", 100, 0.f, 2 * TMath::Pi());
82  // These will be empty in case of no MC info...
83  mPhiTPCPhysPrim = new TH1F("mPhiTPCPhysPrim", "Phi distribution of TPC tracks (physical primary); Phi [rad]; dNdPhi", 100, 0.f, 2 * TMath::Pi());
84  mFractionITSTPCmatchPhiPhysPrim = new TEfficiency("mFractionITSTPCmatchPhiPhysPrim", "Fraction of ITSTPC matched tracks vs Phi (physical primary); Phi [rad]; Eff", 100, 0.f, 2 * TMath::Pi());
85  mPhiPhysPrim = new TH1F("mPhiPhysPrim", "Phi distribution of matched tracks (physical primary); Phi [rad]; dNdPhi", 100, 0.f, 2 * TMath::Pi());
86  // ...till here
87  mEta = new TH1F("mEta", "Eta distribution of matched tracks; Eta; dNdEta", 100, -2.f, 2.f);
88  mChi2Matching = new TH1F("mChi2Matching", "Chi2 of matching; chi2", 100, 0, 30);
89  mChi2Refit = new TH1F("mChi2Refit", "Chi2 of refit; chi2", 200, 0, 100);
90 
91  // log binning for pT
92  const Int_t nbinsPt = 100;
93  const Double_t xminPt = 0.01;
94  const Double_t xmaxPt = 20;
95  Double_t* xbinsPt = new Double_t[nbinsPt + 1];
96  Double_t xlogminPt = TMath::Log10(xminPt);
97  Double_t xlogmaxPt = TMath::Log10(xmaxPt);
98  Double_t dlogxPt = (xlogmaxPt - xlogminPt) / nbinsPt;
99  Double_t yPt = (xlogmaxPt - xlogminPt) / nbinsPt;
100  for (int i = 0; i <= nbinsPt; i++) {
101  Double_t xlogPt = xlogminPt + i * dlogxPt;
102  xbinsPt[i] = TMath::Exp(TMath::Log(10) * xlogPt);
103  }
104  mTimeResVsPt = new TH2F("mTimeResVsPt", "Time resolution vs Pt; Pt [GeV/c]; time res [us]", nbinsPt, xbinsPt, 100, 0.f, 2.f);
105  mPtTPCPhysPrim = new TH1F("mPtTPPhysPrimC", "Pt distribution of TPC tracks (physical primary); Pt [GeV/c]; dNdPt", nbinsPt, xbinsPt);
106  mFractionITSTPCmatchPhysPrim = new TEfficiency("mFractionITSTPCmatchPhysPrim", "Fraction of ITSTPC matched tracks vs Pt (physical primary); Pt [GeV/c]; Eff", nbinsPt, xbinsPt);
107  mPtPhysPrim = new TH1F("mPtPhysPrim", "Pt distribution of matched tracks (physical primary); Pt [GeV/c]; dNdPt", nbinsPt, xbinsPt);
108

162 void MatchITSTPCQC::run(o2::framework::ProcessingContext& ctx)
163 {
164  static int evCount = 0;
165  mSelectedTPCtracks.clear();
166  mRecoCont.collectData(ctx, *mDataRequest.get());
167  mTPCTracks = mRecoCont.getTPCTracks();
168  mITSTPCTracks = mRecoCont.getTPCITSTracks();
169 
170  LOG(debug) << "****** Number of found ITSTPC tracks = " << mITSTPCTracks.size();
171  LOG(debug) << "****** Number of found TPC tracks = " << mTPCTracks.size();
172 
173  // cache selection for TPC tracks
174  for (auto itrk = 0; itrk < mTPCTracks.size(); ++itrk) {
175  auto const& trkTpc = mTPCTracks[itrk];
176  if (selectTrack(trkTpc)) {
177  mSelectedTPCtracks.push_back(itrk);
178  }
179  }
180 
181  // numerator + eta, chi2...
182  if (mUseMC) {
183  mMapLabels.clear();
184  for (auto itrk = 0; itrk < mITSTPCTracks.size(); ++itrk) {
185  auto const& trk = mITSTPCTracks[itrk];
186  auto idxTrkTpc = trk.getRefTPC().getIndex();
187  if (std::any_of(mSelectedTPCtracks.begin(), mSelectedTPCtracks.end(), [idxTrkTpc](int el) { return el == idxTrkTpc; })) {
188  auto lbl = mRecoCont.getTrackMCLabel({(unsigned int)(itrk), GID::Source::ITSTPC});
189  if (mMapLabels.find(lbl) == mMapLabels.end()) {
190  auto const* mcParticle = mcReader.getTrack(lbl);
191  if (isPhysicalPrimary(mcParticle)) {
192  mMapLabels.insert({lbl, {itrk, true}});
193  } else {
194  mMapLabels.insert({lbl, {itrk, false}});
195  }
196  } else {
197  // winner (if more tracks have the same label) has the highest pt
198  if (mITSTPCTracks[mMapLabels.at(lbl).mIdx].getPt() < trk.getPt()) {
199  mMapLabels.at(lbl).mIdx = itrk;
200  }
201  }
202  }
203  }
204  LOG(info) << "number of entries in map for nominator (without duplicates) = " << mMapLabels.size();
205  // now we use only the tracks in the map to fill the histograms (--> tracks have passed the
206  // track selection and there are no duplicated tracks wrt the same MC label)
207  for (auto const& el : mMapLabels) {
208  auto const& trk = mITSTPCTracks[el.second.mIdx];
209  auto const& trkTpc = mTPCTracks[trk.getRefTPC()];
210  mPt->Fill(trkTpc.getPt());
211  mPhi->Fill(trkTpc.getPhi());
212  // we fill also the denominator
213  mPtTPC->Fill(trkTpc.getPt());
214  mPhiTPC->Fill(trkTpc.getPhi());
215  if (el.second.mIsPhysicalPrimary) {
216  mPtPhysPrim->Fill(trkTpc.getPt());
217  mPhiPhysPrim->Fill(trkTpc.getPhi());
218  // we fill also the denominator
219  mPtTPCPhysPrim->Fill(trkTpc.getPt());
220  mPhiTPCPhysPrim->Fill(trkTpc.getPhi());
221  }
222  ++mNITSTPCSelectedTracks;
223  }
224  }
225 
226  for (auto const& trk : mITSTPCTracks) {
227  if (trk.getRefTPC().getIndex() >= mTPCTracks.size()) {
228  LOG(fatal) << "******************** ATTENTION! idx = " << trk.getRefTPC().getIndex() << ", size of container = " << mTPCTracks.size() << " in TF " << evCount;
229  continue;
230  }
231  auto const& trkTpc = mTPCTracks[trk.getRefTPC()];
232  auto idxTrkTpc = trk.getRefTPC().getIndex();
233  if (std::any_of(mSelectedTPCtracks.begin(), mSelectedTPCtracks.end(), [idxTrkTpc](int el) { return el == idxTrkTpc; })) {
234  if (!mUseMC) {
235  mPt->Fill(trkTpc.getPt());
236  mPhi->Fill(trkTpc.getPhi());
237  }
238  mEta->Fill(trkTpc.getEta());
239  mChi2Matching->Fill(trk.getChi2Match());
240  mChi2Refit->Fill(trk.getChi2Refit());
241  mTimeResVsPt->Fill(trkTpc.getPt(), trk.getTimeMUS().getTimeStampError());
242  LOG(debug) << "*** chi2Matching = " << trk.getChi2Match() << ", chi2refit = " << trk.getChi2Refit() << ", timeResolution = " << trk.getTimeMUS().getTimeStampError();
243  ++mNITSTPCSelectedTracks;
244  }
245  }
246 
247  // now filling the denominator for the efficiency calculation
248  if (mUseMC) {
249  mMapTPCLabels.clear();
250  // filling the map where we store for each MC label, the track id of the reconstructed
251  // track with the highest number of TPC clusters
252  for (auto itrk = 0; itrk < mTPCTracks.size(); ++itrk) {
253  auto const& trk = mTPCTracks[itrk];
254  if (std::any_of(mSelectedTPCtracks.begin(), mSelectedTPCtracks.end(), [itrk](int el) { return el == itrk; })) {
255  auto lbl = mRecoCont.getTrackMCLabel({(unsigned int)(itrk), GID::Source::TPC});
256  if (mMapLabels.find(lbl) != mMapLabels.end()) {
257  // the track was already added to the denominator
258  continue;
259  }
260  if (mMapTPCLabels.find(lbl) == mMapTPCLabels.end()) {
261  o2::MCTrack const* mcParticle = mcReader.getTrack(lbl);
262  if (isPhysicalPrimary(mcParticle)) {
263  mMapTPCLabels.insert({lbl, {itrk, true}});
264  } else {
265  mMapTPCLabels.insert({lbl, {itrk, false}});
266  }
267  } else {
268  // winner (if more tracks have the same label) has the highest number of TPC clusters
269  if (mTPCTracks[mMapTPCLabels.at(lbl).mIdx].getNClusters() < trk.getNClusters()) {
270  mMapTPCLabels.at(lbl).mIdx = itrk;
271  }
272  }
273  }
274  }
275  LOG(info) << "number of entries in map for denominator (without duplicates) = " << mMapTPCLabels.size() + mMapLabels.size();
276  // now we use only the tracks in the map to fill the histograms (--> tracks have passed the
277  // track selection and there are no duplicated tracks wrt the same MC label)
278  for (auto const& el : mMapTPCLabels) {
279  auto const& trk = mTPCTracks[el.second.mIdx];
280  mPtTPC->Fill(trk.getPt());
281  mPhiTPC->Fill(trk.getPhi());
282  if (el.second.mIsPhysicalPrimary) {
283  mPtTPCPhysPrim->Fill(trk.getPt());
284  mPhiTPCPhysPrim->Fill(trk.getPhi());
285  }
286  ++mNTPCSelectedTracks;
287  }
288  } else {
289  // if we are in data, we loop over all tracks (no check on the label)
290  for (int itrk = 0; itrk < mTPCTracks.size(); ++itrk) {
291  auto const& trk = mTPCTracks[itrk];
292  if (std::any_of(mSelectedTPCtracks.begin(), mSelectedTPCtracks.end(), [itrk](int el) { return el == itrk; })) {
293  mPtTPC->Fill(trk.getPt());
294  mPhiTPC->Fill(trk.getPhi());
295  ++mNTPCSelectedTracks;
296  }
297  }
298  }
299  evCount++;
300 }

304 bool MatchITSTPCQC::selectTrack(o2::tpc::TrackTPC const& track)
305 {
306 
307  if (track.getPt() < mPtCut) {
308  return false;
309  }
310  if (std::abs(track.getEta()) > mEtaCut) {
311  return false;
312  }
313  if (track.getNClusters() < mNTPCClustersCut) {
314  return false;
315  }
316 
317  math_utils::Point3D<float> v{};
318  std::array<float, 2> dca;
319  if (!(const_cast<o2::tpc::TrackTPC&>(track).propagateParamToDCA(v, mBz, &dca, mDCACut)) || std::abs(dca[0]) > mDCACutY) {
320  return false;
321  }
322 
323  return true;
324 }

328 void MatchITSTPCQC::finalize()
329 {
330 
331  // first we use denominators and nominators to set the TEfficiency; later they are scaled
332 
333  for (int i = 0; i < mPtTPC->GetNbinsX(); ++i) {
334  if (mPtTPC->GetBinContent(i + 1) < mPt->GetBinContent(i + 1)) {
335  LOG(error) << "bin " << i + 1 << ": mPtTPC[i] = " << mPtTPC->GetBinContent(i + 1) << ", mPt[i] = " << mPt->GetBinContent(i + 1);
336  }
337  }
338  for (int i = 0; i < mPhiTPC->GetNbinsX(); ++i) {
339  if (mPhiTPC->GetBinContent(i + 1) < mPhi->GetBinContent(i + 1)) {
340  LOG(error) << "bin " << i + 1 << ": mPhiTPC[i] = " << mPhiTPC->GetBinContent(i + 1) << ", mPhi[i] = " << mPhi->GetBinContent(i + 1);
341  }
342  }
343 
344  // we need to force to replace the total histogram, otherwise it will compare it to the previous passed one, and it might get an error of inconsistency in the bin contents
345  if (!mFractionITSTPCmatch->SetTotalHistogram(*mPtTPC, "f") ||
346  !mFractionITSTPCmatch->SetPassedHistogram(*mPt, "")) {
347  LOG(fatal) << "Something went wrong when defining the efficiency histograms vs Pt!";
348  }
349  if (!mFractionITSTPCmatchPhi->SetTotalHistogram(*mPhiTPC, "f") ||
350  !mFractionITSTPCmatchPhi->SetPassedHistogram(*mPhi, "")) {
351  LOG(fatal) << "Something went wrong when defining the efficiency histograms vs Phi!";
352  }
353 
354  if (mUseMC) {
355  if (!mFractionITSTPCmatchPhysPrim->SetTotalHistogram(*mPtTPCPhysPrim, "f") ||
356  !mFractionITSTPCmatchPhysPrim->SetPassedHistogram(*mPtPhysPrim, "")) {
357  LOG(fatal) << "Something went wrong when defining the efficiency histograms vs Pt (PhysPrim)!";
358  }
359  if (!mFractionITSTPCmatchPhiPhysPrim->SetTotalHistogram(*mPhiTPCPhysPrim, "f") ||
360  !mFractionITSTPCmatchPhiPhysPrim->SetPassedHistogram(*mPhiPhysPrim, "")) {
361  LOG(fatal) << "Something went wrong when defining the efficiency histograms vs Phi (PhysPrim)!";
362  }
363  }
364 
365  float scaleFactTPC = 1. / mNTPCSelectedTracks;
366  float scaleFactITSTPC = 1. / mNITSTPCSelectedTracks;
367  /*
368  mPtTPC->Scale(scaleFactTPC);
369  mPt->Scale(scaleFactITSTPC);
370  mPhiTPC->Scale(scaleFactTPC);
371  mPhi->Scale(scaleFactITSTPC);
372  if (mUseMC) {
373  mPtTPCPhysPrim->Scale(scaleFactTPC);
374  mPtPhysPrim->Scale(scaleFactITSTPC);
375  mPhiTPCPhysPrim->Scale(scaleFactTPC);
376  mPhiPhysPrim->Scale(scaleFactITSTPC);
377  }
378  mEta->Scale(scaleFactITSTPC);
379  mChi2Matching->Scale(scaleFactITSTPC);
380  mChi2Refit->Scale(scaleFactITSTPC);
381  //mTimeResVsPt->Scale(scaleFactITSTPC); // if to few entries, one sees nothing after normalization --> let's not normalize
382  */
383 }
