33 o2::header::RDHAny RawReaderMemory::decodeRawHeader(const void* payloadwords)
34 {
35  auto headerversion = RDHDecoder::getVersion(payloadwords);
36  if (headerversion == 4) {
37  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV4*>(payloadwords));
38  } else if (headerversion == 5) {
39  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV5*>(payloadwords));
40  } else if (headerversion == 6) {
41  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV6*>(payloadwords));
42  }
43  LOG(error) << "RawReaderMemory::decodeRawHeader() : Unknown RDH version";
44  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV6*>(payloadwords));
45 }

57 //Read the next pages until the stop bit is found or new HBF reached
58 //it means we read 1 HBF per next() call
59 RawErrorType_t RawReaderMemory::next()
60 {
61  mRawPayload.clear();
62  bool isStopBitFound = false;
63  do {
64  RawErrorType_t e = nextPage();
65  if (e == RawErrorType_t::kPAGE_NOTFOUND || // nothing left to read...
66  e == RawErrorType_t::kRDH_DECODING || // incorrect rdh -> fatal error
67  e == RawErrorType_t::kPAYLOAD_INCOMPLETE || // we reached end of mRawMemoryBuffer but payload size from rdh tells to read more
68  e == RawErrorType_t::kSTOPBIT_NOTFOUND) { //new HBF orbit started but no stop bit found, need to return
69  return e; //some principal error occured -> stop reading.
70  }
71  isStopBitFound = RDHDecoder::getStop(mRawHeader);
72  } while (!isStopBitFound);
73 
74  return RawErrorType_t::kOK;
75 }

77 //Read the next ONLY ONE page from the stream (single DMA page)
78 //note: 1 raw header per page
79 RawErrorType_t RawReaderMemory::nextPage()
80 {
81  if (!hasNext()) {
82  return RawErrorType_t::kPAGE_NOTFOUND;
83  }
84  mRawHeaderInitialized = false;
85  mPayloadInitialized = false;
86 
87  // Read RDH header
88  o2::header::RDHAny rawHeader;
89  try {
90  rawHeader = decodeRawHeader(mRawMemoryBuffer.data() + mCurrentPosition);
91  } catch (...) {
92  return RawErrorType_t::kRDH_DECODING; //this is fatal error
93  }
94  if (RDHDecoder::getSourceID(rawHeader) != 0x8) {
95  // Not a CPV RDH
96  mCurrentPosition += RDHDecoder::getOffsetToNext(rawHeader); //moving on
97  return RawErrorType_t::kNOT_CPV_RDH;
98  }
99  if (mIsJustInited || mStopBitWasNotFound) { //reading first time after init() or stopbit was not found
100  mCurrentHBFOrbit = RDHDecoder::getHeartBeatOrbit(rawHeader);
101  mRawHeader = rawHeader; //save RDH of first page as mRawHeader
102  mRawHeaderInitialized = true;
103  mStopBitWasNotFound = false; //reset this flag as we start to read again
104  mIsJustInited = false;
105  } else if (mCurrentHBFOrbit != RDHDecoder::getHeartBeatOrbit(rawHeader)) {
106  //next HBF started but we didn't find stop bit.
107  mStopBitWasNotFound = true;
108  return RawErrorType_t::kSTOPBIT_NOTFOUND; //Stop reading, this will be read again by calling next()
109  }
110  mRawHeader = rawHeader; //save RDH of current page as mRawHeader
111  mRawHeaderInitialized = true;
112 
113  auto tmp = mRawMemoryBuffer.data();
114  int start = (mCurrentPosition + RDHDecoder::getHeaderSize(mRawHeader));
115  int end = (mCurrentPosition + RDHDecoder::getMemorySize(mRawHeader));
116  bool isPayloadIncomplete = false;
117  if (mCurrentPosition + RDHDecoder::getMemorySize(mRawHeader) > mRawMemoryBuffer.size()) {
118  // Payload incomplete
119  end = mRawMemoryBuffer.size(); //OK, lets read it anyway. Maybe there still are some completed events...
120  }
121  for (auto iword = start; iword < end; iword++) {
122  mRawPayload.push_back(tmp[iword]);
123  }
124  mPayloadInitialized = true;
125 
126  mCurrentPosition += RDHDecoder::getOffsetToNext(mRawHeader);
127  if (isPayloadIncomplete) {
128  return RawErrorType_t::kPAYLOAD_INCOMPLETE; //return error so we can it handle later
129  }
130  return RawErrorType_t::kOK;
131 }
