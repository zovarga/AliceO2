82  // masks for records components

121  template <class T, typename CG>
122  static int decodeChip(ChipPixelData& chipData, T& buffer, CG cidGetter)
123  {
124  // read record for single non-empty chip, updating on change module and cycle.
125  // return number of records filled (>0), EOFFlag or Error
126  //
127  bool needSorting = false; // if DColumns order is wrong, do explicit reordering
128  auto roErrHandler = [&chipData](uint8_t roErr) {
129 #ifdef ALPIDE_DECODING_STAT
130  if (roErr == MaskErrBusyViolation) {
131  chipData.setError(ChipStat::BusyViolation);
132  } else if (roErr == MaskErrDataOverrun) {
133  chipData.setError(ChipStat::DataOverrun);
134  } else if (roErr == MaskErrFatal) {
135  chipData.setError(ChipStat::Fatal);
136  }
137 #endif
138  };
139 
140  uint8_t dataC = 0, timestamp = 0;
141  uint16_t dataS = 0, region = 0;
142 #ifdef ALPIDE_DECODING_STAT
143  uint16_t rowPrev = 0xffff;
144 #endif
145  //
146  int nRightCHits = 0; // counter for the hits in the right column of the current double column
147  std::uint16_t rightColHits[NRows]; // buffer for the accumulation of hits in the right column
148  std::uint16_t colDPrev = 0xffff; // previously processed double column (to dected change of the double column)
149 
150  uint32_t expectInp = ExpectChipHeader | ExpectChipEmpty; // data must always start with chip header or chip empty flag
151 
152  chipData.clear();
153  while (buffer.next(dataC)) {
154  //
155  // ---------- chip info ?
156  uint8_t dataCM = dataC & (~MaskChipID);
157  //
158  if ((expectInp & ExpectChipEmpty) && dataCM == CHIPEMPTY) { // empty chip was expected
159  chipData.setChipID(cidGetter(dataC & MaskChipID)); // here we set the global chip ID
160  if (!buffer.next(timestamp)) {
161 #ifdef ALPIDE_DECODING_STAT
162  chipData.setError(ChipStat::TruncatedChipEmpty);
163 #endif
164  return unexpectedEOF("CHIP_EMPTY:Timestamp");
165  }
166  chipData.resetChipID();
167  expectInp = ExpectChipHeader | ExpectChipEmpty;
168  continue;
169  }
170 
171  if ((expectInp & ExpectChipHeader) && dataCM == CHIPHEADER) { // chip header was expected
172  chipData.setChipID(cidGetter(dataC & MaskChipID)); // here we set the global chip ID
173  if (!buffer.next(timestamp)) {
174 #ifdef ALPIDE_DECODING_STAT
175  chipData.setError(ChipStat::TruncatedChipHeader);
176 #endif
177  return unexpectedEOF("CHIP_HEADER");
178  }
179  expectInp = ExpectRegion; // now expect region info
180  continue;
181  }
182 
183  // region info ?
184  if ((expectInp & ExpectRegion) && (dataC & REGION) == REGION) { // chip header was seen, or hit data read
185  region = dataC & MaskRegion;
186  expectInp = ExpectData;
187  continue;
188  }
189 
190  if ((expectInp & ExpectChipTrailer) && dataCM == CHIPTRAILER) { // chip trailer was expected
191  expectInp = ExpectChipHeader | ExpectChipEmpty;
192  chipData.setROFlags(dataC & MaskROFlags);
193 #ifdef ALPIDE_DECODING_STAT
194  uint8_t roErr = dataC & MaskROFlags;
195  if (roErr) {
196  roErrHandler(roErr);
197  }
198 #endif
199  // in case there are entries in the "right" columns buffer, add them to the container
200  if (nRightCHits) {
201  colDPrev++;
202  for (int ihr = 0; ihr < nRightCHits; ihr++) {
203  addHit(chipData, rightColHits[ihr], colDPrev);
204  }
205  }
206 
207  if (!chipData.getData().size() && !chipData.isErrorSet()) {
208  nRightCHits = 0;
209  colDPrev = 0xffff;
210  chipData.clear();
211  continue;
212  }
213  break;
214  }
215 
216  // hit info ?
217  if ((expectInp & ExpectData)) {
218  if (isData(dataC)) { // region header was seen, expect data
219  // note that here we are checking on the byte rather than the short, need complete to ushort
220  dataS = dataC << 8;
221  if (!buffer.next(dataC)) {
222 #ifdef ALPIDE_DECODING_STAT
223  chipData.setError(ChipStat::TruncatedRegion);
224 #endif
225  return unexpectedEOF("CHIPDATA");
226  }
227  dataS |= dataC;
228  // we are decoding the pixel addres, if this is a DATALONG, we will fetch the mask later
229  uint16_t dColID = (dataS & MaskEncoder) >> 10;
230  uint16_t pixID = dataS & MaskPixID;
231 
232  // convert data to usual row/pixel format
233  uint16_t row = pixID >> 1;
234  // abs id of left column in double column
235  uint16_t colD = (region * NDColInReg + dColID) << 1; // TODO consider <<4 instead of *NDColInReg?
236  bool rightC = (row & 0x1) ? !(pixID & 0x1) : (pixID & 0x1); // true for right column / lalse for left
237  // if we start new double column, transfer the hits accumulated in the right column buffer of prev. double column
238  if (colD != colDPrev) {
239  if (colD < colDPrev && colDPrev != 0xffff) {
240  needSorting = true;
241 #ifdef ALPIDE_DECODING_STAT
242  chipData.setError(ChipStat::WrongDColOrder);
243 #endif
244  }
245  colDPrev++;
246  for (int ihr = 0; ihr < nRightCHits; ihr++) {
247  addHit(chipData, rightColHits[ihr], colDPrev);
248  }
249  colDPrev = colD;
250  nRightCHits = 0; // reset the buffer
251 #ifdef ALPIDE_DECODING_STAT
252  rowPrev = 0xffff;
253  }
254  // this is a special test to exclude repeated data of the same pixel fired
255  else if (row == rowPrev) { // same row/column fired repeatedly, hope this check is temporary
256  chipData.setError(ChipStat::RepeatingPixel);
257  chipData.addErrorInfo((uint64_t(colD + rightC) << 16) | uint64_t(row));
258  if ((dataS & (~MaskDColID)) == DATALONG) { // skip pattern w/o decoding
259  uint8_t hitsPattern = 0;
260  if (!buffer.next(hitsPattern)) {
261  chipData.setError(ChipStat::TruncatedLondData);
262  return unexpectedEOF("CHIP_DATA_LONG:Pattern");
263  }
264  if (hitsPattern & (~MaskHitMap)) {
265  return unexpectedEOF("CHIP_DATA_LONG:Pattern");
266  }
267  }
268  expectInp = ExpectChipTrailer | ExpectData | ExpectRegion;
269  continue; // end of DATA(SHORT or LONG) processing
270  } else {
271  rowPrev = row;
272 #endif
273  }
274 
275  // we want to have hits sorted in column/row, so the hits in right column of given double column
276  // are first collected in the temporary buffer
277  // real columnt id is col = colD + 1;
278  if (rightC) {
279  rightColHits[nRightCHits++] = row; // col = colD+1
280  } else {
281  addHit(chipData, row, colD); // col = colD, left column hits are added directly to the container
282  }
283 
284  if ((dataS & (~MaskDColID)) == DATALONG) { // multiple hits ?
285  uint8_t hitsPattern = 0;
286  if (!buffer.next(hitsPattern)) {
287 #ifdef ALPIDE_DECODING_STAT
288  chipData.setError(ChipStat::TruncatedLondData);
289 #endif
290  return unexpectedEOF("CHIP_DATA_LONG:Pattern");
291  }
292  if (hitsPattern & (~MaskHitMap)) {
293 #ifdef ALPIDE_DECODING_STAT
294  chipData.setError(ChipStat::WrongDataLongPattern);
295 #endif
296  return unexpectedEOF("CHIP_DATA_LONG:Pattern");
297  }
298  for (int ip = 0; ip < HitMapSize; ip++) {
299  if (hitsPattern & (0x1 << ip)) {
300  uint16_t addr = pixID + ip + 1, rowE = addr >> 1;
301  if (addr & ~MaskPixID) {
302 #ifdef ALPIDE_DECODING_STAT
303  chipData.setError(ChipStat::WrongRow);
304 #endif
305  return unexpectedEOF(fmt::format("Non-existing encoder {} decoded, DataLong was {:x}", pixID, dataS));
306  }
307  rightC = ((rowE & 0x1) ? !(addr & 0x1) : (addr & 0x1)); // true for right column / lalse for left
308  // the real columnt is int colE = colD + rightC;
309  if (rightC) { // same as above
310  rightColHits[nRightCHits++] = rowE;
311  } else {
312  addHit(chipData, rowE, colD + rightC); // left column hits are added directly to the container
313  }
314  }
315  }
316  }
317  } else {
318 #ifdef ALPIDE_DECODING_STAT
319  chipData.setError(ChipStat::NoDataFound);
320 #endif
321  return unexpectedEOF(fmt::format("Expected DataShort or DataLong mask, got {:x}", dataS));
322  }
323  expectInp = ExpectChipTrailer | ExpectData | ExpectRegion;
324  continue; // end of DATA(SHORT or LONG) processing
325  }
326 
327  if (dataC == BUSYON) {
328 #ifdef ALPIDE_DECODING_STAT
329  chipData.setError(ChipStat::BusyOn);
330 #endif
331  continue;
332  }
333  if (dataC == BUSYOFF) {
334 #ifdef ALPIDE_DECODING_STAT
335  chipData.setError(ChipStat::BusyOff);
336 #endif
337  continue;
338  }
339 
340  if (!dataC) {
341  buffer.clear(); // 0 padding reached (end of the cable data), no point in continuing
342  break;
343  }
344 
345  // in case of BUSY VIOLATION the Trailer may come directly after the Header
346  if ((expectInp & ExpectRegion) && (dataCM == CHIPTRAILER) && (dataC & MaskROFlags)) {
347  expectInp = ExpectChipHeader | ExpectChipEmpty;
348  chipData.setROFlags(dataC & MaskROFlags);
349  roErrHandler(dataC & MaskROFlags);
350  break;
351  }
352 
353  // check for APE errors, see https://alice.its.cern.ch/jira/browse/O2-1717?focusedCommentId=274714&page=com.atlassian.jira.plugin.system.issuetabpanels:comment-tabpanel#comment-274714
354  bool fatalAPE = false;
355  auto codeAPE = ChipStat::getAPECode(dataC, fatalAPE);
356  if (codeAPE >= 0) {
357 #ifdef ALPIDE_DECODING_STAT
358  chipData.setError(ChipStat::DecErrors(codeAPE));
359 #endif
360  if (fatalAPE) {
361  buffer.clear(); // no point in contiunuing with this cable data
362  } else { // skip eventual padding
363  while (buffer.next(dataC)) {
364  if (dataC) { // padding is over, make 1 step back in the buffer
365  auto currPtr = buffer.getPtr();
366  buffer.setPtr(--currPtr);
367  break;
368  }
369  }
370  }
371  return unexpectedEOF(fmt::format("APE error {:#02x} [expectation = 0x{:#02x}]", int(dataC), int(expectInp)));
372  }
373 #ifdef ALPIDE_DECODING_STAT
374  chipData.setError(ChipStat::UnknownWord);
375  // fill the error buffer with a few bytes of wrong data
376  const uint8_t* begPtr = buffer.data();
377  const uint8_t* endPtr = buffer.getEnd();
378  const uint8_t* curPtr = buffer.getPtr();
379  size_t offsBack = std::min(ChipPixelData::MAXDATAERRBYTES - ChipPixelData::MAXDATAERRBYTES_AFTER, size_t(curPtr - begPtr));
380  size_t offsAfter = std::min(ChipPixelData::MAXDATAERRBYTES_AFTER, size_t(endPtr - curPtr));
381  std::memcpy(chipData.getRawErrBuff().data(), curPtr - offsBack, offsBack + offsAfter);
382  chipData.setNBytesInRawBuff(offsBack + offsAfter);
383 #endif
384  return unexpectedEOF(fmt::format("Unknown word 0x{:x} [expectation = 0x{:x}]", int(dataC), int(expectInp))); // error
385  }
386 
387  if (needSorting && chipData.getData().size()) { // d.columns were in a wrong order, need to sort the data
388  auto& pixData = chipData.getData();
389  std::sort(pixData.begin(), pixData.end(),
390  [](PixelData& a, PixelData& b) { return a.getCol() < b.getCol() || (a.getCol() == b.getCol() && a.getRowDirect() < b.getRowDirect()); });
391  // if the columns ordering was wrong, detection of same pixel fired twice might have failed, make sure there are no duplicates
392  auto currPix = pixData.begin(), prevPix = currPix++;
393  while (currPix != pixData.end()) {
394  if (prevPix->getCol() == currPix->getCol() && prevPix->getRowDirect() == currPix->getRowDirect()) {
395  pixData.erase(prevPix);
396  }
397  prevPix = currPix++;
398  }
399  }
400  return chipData.getData().size();
401  }

429  {
430  if (mNoisyPixels) {

441  void addPixel(short row, short col)
442  {
443  int last = mPix2Encode.size();
444  mPix2Encode.emplace_back(row, col);
445  if (last && row == mPix2Encode[last - 1].row) { // extend current row
446  mPix2Encode[last - 1].nextInRow = last; // refer to new link in the same row
447  } else { // create new row
