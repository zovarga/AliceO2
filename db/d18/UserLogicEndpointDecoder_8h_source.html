94 template <typename CHARGESUM, int VERSION>
95 size_t UserLogicEndpointDecoder<CHARGESUM, VERSION>::append(Payload buffer)
96 {
97  if (buffer.size() % 8) {
98  throw std::invalid_argument("buffer size should be a multiple of 8");
99  }
100  size_t n{0};
101 
102  for (size_t i = 0; i < buffer.size(); i += 8) {
103  uint64_t word = (static_cast<uint64_t>(buffer[i + 0])) |
104  (static_cast<uint64_t>(buffer[i + 1]) << 8) |
105  (static_cast<uint64_t>(buffer[i + 2]) << 16) |
106  (static_cast<uint64_t>(buffer[i + 3]) << 24) |
107  (static_cast<uint64_t>(buffer[i + 4]) << 32) |
108  (static_cast<uint64_t>(buffer[i + 5]) << 40) |
109  (static_cast<uint64_t>(buffer[i + 6]) << 48) |
110  (static_cast<uint64_t>(buffer[i + 7]) << 56);
111 
112  if (word == 0) {
113  continue;
114  }
115  if (word == 0xFEEDDEEDFEEDDEED) {
116  continue;
117  }
118 
119  ULHeaderWord<VERSION> ulword{word};
120 
121  int gbt = ulword.linkID;
122 
123  // The User Logic uses the condition gbt=15 to identify special control and diagnostics words
124  // that are generated internally and do not contain data coming from the front-end electronics.
125  if (gbt == 15) {
126  // TODO: the exact format of the UL control words is still being defined and tested,
127  // hence proper decoding will be implemented once the format is finalized.
128  // For the moment we simply avoid throwing an exception when linkID is equal to 15
129  continue;
130  } else {
131  if (gbt < 0 || gbt > 11) {
132  SampaErrorHandler handler = mDecodedDataHandlers.sampaErrorHandler;
133  if (handler) {
134  DsElecId dsId{static_cast<uint16_t>(0), static_cast<uint8_t>(0), static_cast<uint8_t>(0)};
135  handler(dsId, -1, ErrorBadLinkID);
136  continue;
137  } else {
138  throw fmt::format("warning : out-of-range gbt {} word={:08X}\n", gbt, word);
139  }
140  }
141  }
142 
143  // Get the corresponding decoders array, or allocate it if does not exist yet
144  auto d = mElinkDecoders.find(gbt);
145  if (d == mElinkDecoders.end()) {
146 
147  // Compute the (feeId, linkId) pair...
148  FeeLinkId feeLinkId(mFeeId, gbt);
149  // ... and get the solar from it
150  auto solarId = mFee2SolarMapper(feeLinkId);
151 
152  if (!solarId.has_value()) {
153  SampaErrorHandler handler = mDecodedDataHandlers.sampaErrorHandler;
154  if (handler) {
155  DsElecId dsId{static_cast<uint16_t>(0), static_cast<uint8_t>(0), static_cast<uint8_t>(0)};
156  handler(dsId, -1, ErrorUnknownLinkID);
157  continue;
158  } else {
159  throw std::logic_error(fmt::format("{} Could not get solarId from feeLinkId={}\n", __PRETTY_FUNCTION__, asString(feeLinkId)));
160  }
161  }
162 
163  mElinkDecoders.emplace(static_cast<uint16_t>(gbt),
164  impl::makeArray<40>([=](size_t i) {
165  DsElecId dselec{solarId.value(), static_cast<uint8_t>(i / 5), static_cast<uint8_t>(i % 5)};
166  return ElinkDecoder(dselec, mDecodedDataHandlers);
167  }));
168  d = mElinkDecoders.find(gbt);
169  }
170 
171  uint16_t dsid = ulword.dsID;
172  if (dsid > 39) {
173  SampaErrorHandler handler = mDecodedDataHandlers.sampaErrorHandler;
174  if (handler) {
175  DsElecId dsId{static_cast<uint16_t>(0), static_cast<uint8_t>(0), static_cast<uint8_t>(0)};
176  handler(dsId, -1, ErrorBadELinkID);
177  continue;
178  } else {
179  throw fmt::format("warning : out-of-range DS ID {} word={:08X}\n", dsid, word);
180  }
181  }
182 
183  int8_t error = ulword.error;
184  bool incomplete = ulword.incomplete > 0;
185  uint64_t data50 = ulword.data;
186 
187  d->second.at(dsid).append(data50, error, incomplete);
188  n += 8;
189  }
190  return n;
191 }
