42 void PHOSRunbyrunCalibDevice::endOfStream(o2::framework::EndOfStreamContext& ec)
43 {
44  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
45  mCalibrator->checkSlotsToFinalize(INFINITE_TF);
46  mCalibrator->endOfStream();
47  mRunByRun = mCalibrator->getCalibration();
48  if (checkFitResult()) {
49  LOG(info) << "End of stream reached, sending output to CCDB";
50  // prepare all info to be sent to CCDB
51  auto flName = o2::ccdb::CcdbApi::generateFileName("Runbyrun");
52  std::map<std::string, std::string> md;
53  o2::ccdb::CcdbObjectInfo info("PHS/Calib/Runbyrun", "Runbyrun", flName, md, mRunStartTime, INFINITE_TF);
54  info.setMetaData(md);
55  auto image = o2::ccdb::CcdbApi::createObjectImage(&mRunByRun, &info);
56 
57  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName()
58  << " of size " << image->size()
59  << " bytes, valid for " << info.getStartValidityTimestamp()
60  << " : " << info.getEndValidityTimestamp();
61 
62  header::DataHeader::SubSpecificationType subSpec{(header::DataHeader::SubSpecificationType)0};
63  ec.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "PHOS_Runbyrun", subSpec}, *image.get());
64  ec.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "PHOS_Runbyrun", subSpec}, info);
65  } else {
66  LOG(error) << "Incorrect fit results: " << mRunByRun[0] << "+-" << mRunByRun[1] << ", "
67  << mRunByRun[2] << "+-" << mRunByRun[3] << ", "
68  << mRunByRun[4] << "+-" << mRunByRun[5] << ", "
69  << mRunByRun[6] << "+-" << mRunByRun[7];
70  }
71  //TODO! Send mRunByRun for QC and trending plots
72  //
73 
74  //Get ready for next run
75  mCalibrator->initOutput(); // reset the outputs once they are already sent
76 }
77 bool PHOSRunbyrunCalibDevice::checkFitResult()
78 {
79  bool res = true;
80  const float massmin = 0.125;
81  const float massmax = 0.155;
82  for (int mod = 0; mod < 4; mod++) {
83  res &= mRunByRun[2 * mod] < massmax && mRunByRun[2 * mod] > massmin;
84  }
85  return res;
86 }
