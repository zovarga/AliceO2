69  void run(framework::ProcessingContext& pc)
70  {
71  static bool finished = false;
72  if (finished) {
73  return;
74  }
75  float noiseProba = DigitizerParam::Instance().noiseProba;
76  auto tStart = std::chrono::high_resolution_clock::now();
77  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
78  context->initSimChains(o2::detectors::DetID::MCH, mSimChains);
79  const auto& eventRecords = context->getEventRecords();
80  const auto& eventParts = context->getEventParts();
81  std::vector<o2::mch::Digit> digits;
82  std::vector<o2::mch::ROFRecord> rofs;
83  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labels;
84  size_t firstIdx = 0;
85  auto mchRecords = groupIR(eventRecords);
86  for (auto mrec : mchRecords) {
87  auto collisionIndices = mrec.second;
88  const auto& mchIR = mrec.first;
89  mDigitizer->startCollision(mchIR);
90  for (auto collisionIndex : collisionIndices) {
91  for (const auto& part : eventParts[collisionIndex]) {
92  std::vector<o2::mch::Hit> hits;
93  context->retrieveHits(mSimChains, "MCHHit", part.sourceID, part.entryID, &hits);
94  mDigitizer->processHits(hits, part.entryID, part.sourceID);
95  }
96  }
97  mDigitizer->addNoise(noiseProba);
98  mDigitizer->extractDigitsAndLabels(digits, labels);
99  auto nEntries = digits.size() - firstIdx;
100  rofs.emplace_back(ROFRecord(mchIR, firstIdx, nEntries));
101  firstIdx = digits.size();
102  }
103  pc.outputs().snapshot(Output{"MCH", "DIGITS", 0, Lifetime::Timeframe}, digits);
104  pc.outputs().snapshot(Output{"MCH", "DIGITROFS", 0, Lifetime::Timeframe}, rofs);
105  if (pc.outputs().isAllowed({"MCH", "DIGITSLABELS", 0})) {
106  pc.outputs().snapshot(Output{"MCH", "DIGITSLABELS", 0, Lifetime::Timeframe}, labels);
107  }
108  pc.outputs().snapshot(Output{"MCH", "ROMode", 0, Lifetime::Timeframe},
109  DigitizerParam::Instance().continuous ? o2::parameters::GRPObject::CONTINUOUS : o2::parameters::GRPObject::TRIGGERING);
110 
111  // we should be only called once; tell DPL that this process is ready to exit
112  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
113  finished = true;
114 
115  logStatus(digits, rofs, labels, tStart);
116  }
117 
118  private:
119  std::unique_ptr<Digitizer> mDigitizer;
120  std::vector<TChain*> mSimChains;
121 };
