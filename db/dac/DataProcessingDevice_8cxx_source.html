242 void on_socket_polled(uv_poll_t* poller, int status, int events)
243 {
244  PollerContext* context = (PollerContext*)poller->data;
245  context->state->loopReason |= DeviceState::DATA_SOCKET_POLLED;
246  switch (events) {
247  case UV_READABLE: {
248  ZoneScopedN("socket readable event");
249  LOG(debug) << "socket polled UV_READABLE: " << context->name;
250  context->state->loopReason |= DeviceState::DATA_INCOMING;
251  } break;
252  case UV_WRITABLE: {
253  ZoneScopedN("socket writeable");
254  LOG(debug) << "socket polled UV_WRITEABLE";
255  context->state->loopReason |= DeviceState::DATA_OUTGOING;
256  } break;
257  case UV_DISCONNECT: {
258  ZoneScopedN("socket disconnect");
259  LOG(debug) << "socket polled UV_DISCONNECT";
260  } break;
261  case UV_PRIORITIZED: {
262  ZoneScopedN("socket prioritized");
263  LOG(debug) << "socket polled UV_PRIORITIZED";
264  } break;
265  }
266  // We do nothing, all the logic for now stays in DataProcessingDevice::doRun()

334 void on_signal_callback(uv_signal_t* handle, int signum)
335 {
336  ZoneScopedN("Signal callaback");
337  LOG(debug) << "Signal " << signum << " received.";
338  DeviceContext* context = (DeviceContext*)handle->data;
339  context->state->loopReason |= DeviceState::SIGNAL_ARRIVED;
340  size_t ri = 0;
341  while (ri != context->quotaEvaluator->mOffers.size()) {
342  auto& offer = context->quotaEvaluator->mOffers[ri];
343  // We were already offered some sharedMemory, so we
344  // do not consider the offer.
345  // FIXME: in principle this should account for memory
346  // available and being offered, however we
347  // want to get out of the woods for now.
348  if (offer.valid && offer.sharedMemory != 0) {
349  return;
350  }
351  ri++;
352  }
353  // Find the first empty offer and have 1GB of shared memory there
354  for (size_t i = 0; i < context->quotaEvaluator->mOffers.size(); ++i) {
355  auto& offer = context->quotaEvaluator->mOffers[i];
356  if (offer.valid == false) {
357  offer.cpu = 0;
358  offer.memory = 0;
359  offer.sharedMemory = 1000000000;
360  offer.valid = true;
361  offer.user = -1;
362  break;
363  }
364  }
365  context->stats->totalSigusr1 += 1;
366 }

389 void DataProcessingDevice::initPollers()
390 {
391  // We add a timer only in case a channel poller is not there.
392  if ((mStatefulProcess != nullptr) || (mStatelessProcess != nullptr)) {
393  for (auto& x : fChannels) {
394  if ((x.first.rfind("from_internal-dpl", 0) == 0) &&
395  (x.first.rfind("from_internal-dpl-aod", 0) != 0) &&
396  (x.first.rfind("from_internal-dpl-ccdb-backend", 0) != 0) &&
397  (x.first.rfind("from_internal-dpl-injected", 0)) != 0) {
398  LOG(debug) << x.first << " is an internal channel. Skipping as no input will come from there." << std::endl;
399  continue;
400  }
401  // We only watch receiving sockets.
402  if (x.first.rfind("from_" + mSpec.name + "_", 0) == 0) {
403  LOG(debug) << x.first << " is to send data. Not polling." << std::endl;
404  continue;
405  }
406  // We assume there is always a ZeroMQ socket behind.
407  int zmq_fd = 0;
408  size_t zmq_fd_len = sizeof(zmq_fd);
409  // FIXME: I should probably save those somewhere... ;-)
410  auto* poller = (uv_poll_t*)malloc(sizeof(uv_poll_t));
411  x.second[0].GetSocket().GetOption("fd", &zmq_fd, &zmq_fd_len);
412  if (zmq_fd == 0) {
413  LOG(error) << "Cannot get file descriptor for channel." << x.first;
414  continue;
415  }
416  LOG(debug) << "Polling socket for " << x.second[0].GetName();
417  // FIXME: leak
418  auto* pCtx = (PollerContext*)malloc(sizeof(PollerContext));
419  pCtx->name = strdup(x.first.c_str());
420  pCtx->loop = mState.loop;
421  pCtx->device = this;
422  pCtx->state = &mState;
423  pCtx->fd = zmq_fd;
424  poller->data = pCtx;
425  uv_poll_init(mState.loop, poller, zmq_fd);
426  mState.activeInputPollers.push_back(poller);
427  }
428  // In case we do not have any input channel and we do not have
429  // any timers or signal watchers we still wake up whenever we can send data to downstream
430  // devices to allow for enumerations.
431  if (mState.activeInputPollers.empty() && mState.activeTimers.empty() && mState.activeSignals.empty()) {
432  for (auto& x : fChannels) {
433  if (x.first.rfind(mSpec.channelPrefix + "from_internal-dpl", 0) == 0) {
434  LOG(debug) << x.first << " is an internal channel. Not polling." << std::endl;
435  continue;
436  }
437  assert(x.first.rfind(mSpec.channelPrefix + "from_" + mSpec.name + "_", 0) == 0);
438  // We assume there is always a ZeroMQ socket behind.
439  int zmq_fd = 0;
440  size_t zmq_fd_len = sizeof(zmq_fd);
441  // FIXME: I should probably save those somewhere... ;-)
442  auto* poller = (uv_poll_t*)malloc(sizeof(uv_poll_t));
443  x.second[0].GetSocket().GetOption("fd", &zmq_fd, &zmq_fd_len);
444  if (zmq_fd == 0) {
445  LOG(error) << "Cannot get file descriptor for channel." << x.first;
446  continue;
447  }
448  LOG(debug) << "Polling socket for " << x.second[0].GetName();
449  // FIXME: leak
450  auto* pCtx = (PollerContext*)malloc(sizeof(PollerContext));
451  pCtx->name = strdup(x.first.c_str());
452  pCtx->loop = mState.loop;
453  pCtx->device = this;
454  pCtx->state = &mState;
455  pCtx->fd = zmq_fd;
456  poller->data = pCtx;
457  uv_poll_init(mState.loop, poller, zmq_fd);
458  mState.activeOutputPollers.push_back(poller);
459  }
460  }
461  } else {
462  // This is a fake device, so we can request to exit immediately
463  mServiceRegistry.get<ControlService>().readyToQuit(QuitRequest::Me);
464  // A two second timer to stop internal devices which do not want to
465  auto* timer = (uv_timer_t*)malloc(sizeof(uv_timer_t));
466  uv_timer_init(mState.loop, timer);
467  timer->data = &mState;
468  uv_timer_start(timer, on_idle_timer, 2000, 2000);
469  mState.activeTimers.push_back(timer);
470  }

493 void DataProcessingDevice::InitTask()
494 {
495  auto distinct = DataRelayerHelpers::createDistinctRouteIndex(mSpec.inputs);
496  int i = 0;
497  for (auto& di : distinct) {
498  auto& route = mSpec.inputs[di];
499  if (route.configurator.has_value() == false) {
500  i++;
501  continue;
502  }
503  ExpirationHandler handler{
504  RouteIndex{i++},
505  route.matcher.lifetime,
506  route.configurator->creatorConfigurator(mState, mServiceRegistry, *mConfigRegistry),
507  route.configurator->danglingConfigurator(mState, *mConfigRegistry),
508  route.configurator->expirationConfigurator(mState, *mConfigRegistry)};
509  mExpirationHandlers.emplace_back(std::move(handler));
510  }
511 
512  if (mState.awakeMainThread == nullptr) {
513  mState.awakeMainThread = (uv_async_t*)malloc(sizeof(uv_async_t));
514  mState.awakeMainThread->data = &mState;
515  uv_async_init(mState.loop, mState.awakeMainThread, on_awake_main_thread);
516  }
517 
518  mDeviceContext.expectedRegionCallbacks = std::stoi(fConfig->GetValue<std::string>("expected-region-callbacks"));
519  mDeviceContext.exitTransitionTimeout = std::stoi(fConfig->GetValue<std::string>("exit-transition-timeout"));
520 
521  for (auto& channel : fChannels) {
522  channel.second.at(0).Transport()->SubscribeToRegionEvents([this,
523  &context = mDeviceContext,
524  &registry = mServiceRegistry,
525  &pendingRegionInfos = mPendingRegionInfos,
526  &regionInfoMutex = mRegionInfoMutex](FairMQRegionInfo info) {
527  std::lock_guard<std::mutex> lock(regionInfoMutex);
528  LOG(debug) << ">>> Region info event" << info.event;
529  LOG(debug) << "id: " << info.id;
530  LOG(debug) << "ptr: " << info.ptr;
531  LOG(debug) << "size: " << info.size;
532  LOG(debug) << "flags: " << info.flags;
533  context.expectedRegionCallbacks -= 1;
534  pendingRegionInfos.push_back(info);
535  // We always want to handle these on the main loop
536  uv_async_send(registry.get<DeviceState>().awakeMainThread);
537  });
538  }
539 
540  // Add a signal manager for SIGUSR1 so that we can force
541  // an event from the outside, making sure that the event loop can
542  // be unblocked (e.g. by a quitting DPL driver) even when there
543  // is no data pending to be processed.
544  uv_signal_t* sigusr1Handle = (uv_signal_t*)malloc(sizeof(uv_signal_t));
545  uv_signal_init(mState.loop, sigusr1Handle);
546  sigusr1Handle->data = &mDeviceContext;
547  uv_signal_start(sigusr1Handle, on_signal_callback, SIGUSR1);
548 
550  DataProcessingDevice::initPollers();
551 
552  // Whenever we InitTask, we consider as if the previous iteration
553  // was successful, so that even if there is no timer or receiving
554  // channel, we can still start an enumeration.
555  mWasActive = true;
556 
557  // We should be ready to run here. Therefore we copy all the
558  // required parts in the DataProcessorContext. Eventually we should
559  // do so on a per thread basis, with fine grained locks.
560  mDataProcessorContexes.resize(1);
561  this->fillContext(mDataProcessorContexes.at(0), mDeviceContext);
562 
567  while (mDeviceContext.expectedRegionCallbacks > 0 && uv_run(mState.loop, UV_RUN_ONCE)) {
568  // Handle callbacks if any
569  {
570  std::lock_guard<std::mutex> lock(mRegionInfoMutex);
571  handleRegionCallbacks(mServiceRegistry, mPendingRegionInfos);
572  }
573  }

576 void DataProcessingDevice::fillContext(DataProcessorContext& context, DeviceContext& deviceContext)
577 {
578  context.wasActive = &mWasActive;
579 
580  deviceContext.device = this;
581  deviceContext.spec = &mSpec;
582  deviceContext.state = &mState;
583  deviceContext.quotaEvaluator = &mQuotaEvaluator;
584  deviceContext.stats = &mStats;
585  context.isSink = false;
586  // If nothing is a sink, the rate limiting simply does not trigger.
587  bool enableRateLimiting = std::stoi(fConfig->GetValue<std::string>("timeframes-rate-limit"));

647 void DataProcessingDevice::Run()
648 {
649  while (mState.transitionHandling != TransitionHandlingState::Expired) {
650  if (mState.nextFairMQState.empty() == false) {
651  this->ChangeState(mState.nextFairMQState.back());
652  mState.nextFairMQState.pop_back();
653  }
654  // Notify on the main thread the new region callbacks, making sure
655  // no callback is issued if there is something still processing.
656  {
657  std::lock_guard<std::mutex> lock(mRegionInfoMutex);
658  handleRegionCallbacks(mServiceRegistry, mPendingRegionInfos);
659  }
660  // This will block for the correct delay (or until we get data
661  // on a socket). We also do not block on the first iteration
662  // so that devices which do not have a timer can still start an
663  // enumeration.
664  {
665  ZoneScopedN("uv idle");
666  TracyPlot("past activity", (int64_t)mWasActive);
667  mServiceRegistry.get<DriverClient>().flushPending();
668  auto shouldNotWait = (mWasActive &&
669  (mState.streaming != StreamingState::Idle) && (mState.activeSignals.empty())) ||
670  (mState.streaming == StreamingState::EndOfStreaming);
671  if (NewStatePending()) {
672  shouldNotWait = true;
673  }
674  if (mState.transitionHandling == TransitionHandlingState::NoTransition && NewStatePending()) {
675  mState.transitionHandling = TransitionHandlingState::Requested;
676  auto timeout = mDeviceContext.exitTransitionTimeout;
677  if (timeout != 0) {
678  auto* timer = (uv_timer_t*)malloc(sizeof(uv_timer_t));
679  timer->data = &mState;
680  mState.activeTimers.push_back(timer);
681  uv_timer_init(mState.loop, timer);
682  mState.transitionHandling = TransitionHandlingState::Requested;
683  uv_timer_start(timer, on_transition_requested_expired, timeout * 1000, 0);
684  if (mProcessingPolicies.termination == TerminationPolicy::QUIT) {
685  LOGP(info, "New state requested. Waiting for {} seconds before quitting.");
686  } else {
687  LOGP(info, "New state requested. Waiting for {} seconds before switching to READY state.", timeout);
688  }
689  } else {
690  mState.transitionHandling = TransitionHandlingState::Expired;
691  if (mProcessingPolicies.termination == TerminationPolicy::QUIT) {
692  LOGP(info, "New state requested. No timeout set, quitting immediately as per --completion-policy");
693  } else {
694  LOGP(info, "New state requested. No timeout set, switching to READY state immediately");
695  }
696  }
697  }
698  TracyPlot("shouldNotWait", (int)shouldNotWait);
699  uv_run(mState.loop, shouldNotWait ? UV_RUN_NOWAIT : UV_RUN_ONCE);
700  TracyPlot("loopReason", (int64_t)(uint64_t)mState.loopReason);
701 
702  mState.loopReason = DeviceState::NO_REASON;
703  if (!mState.pendingOffers.empty()) {
704  mQuotaEvaluator.updateOffers(mState.pendingOffers, uv_now(mState.loop));
705  }
706  }
707 
708  // Notify on the main thread the new region callbacks, making sure
709  // no callback is issued if there is something still processing.
710  // Notice that we still need to perform callbacks also after
711  // the socket epolled, because otherwise we would end up serving
712  // the callback after the first data arrives is the system is too
713  // fast to transition from Init to Run.
714  {
715  std::lock_guard<std::mutex> lock(mRegionInfoMutex);
716  handleRegionCallbacks(mServiceRegistry, mPendingRegionInfos);
717  }
718 
719  assert(mStreams.size() == mHandles.size());
721  TaskStreamRef streamRef{-1};
722  for (int ti = 0; ti < mStreams.size(); ti++) {
723  auto& taskInfo = mStreams[ti];
724  if (taskInfo.running) {
725  continue;
726  }
727  streamRef.index = ti;
728  }
729  using o2::monitoring::Metric;
730  using o2::monitoring::Monitoring;
731  using o2::monitoring::tags::Key;
732  using o2::monitoring::tags::Value;
733  // We have an empty stream, let's check if we have enough
734  // resources for it to run something
735  if (streamRef.index != -1) {
736  // Synchronous execution of the callbacks. This will be moved in the
737  // moved in the on_socket_polled once we have threading in place.
738  auto& handle = mHandles[streamRef.index];
739  auto& stream = mStreams[streamRef.index];
740  handle.data = &mStreams[streamRef.index];
741 
742  static std::function<void(ComputingQuotaOffer const&, ComputingQuotaStats const& stats)> reportExpiredOffer = [&monitoring = mServiceRegistry.get<o2::monitoring::Monitoring>()](ComputingQuotaOffer const& offer, ComputingQuotaStats const& stats) {
743  monitoring.send(Metric{(uint64_t)stats.totalExpiredOffers, "resource-offer-expired"}.addTag(Key::Subsystem, Value::DPL));
744  monitoring.send(Metric{(uint64_t)stats.totalExpiredBytes, "arrow-bytes-expired"}.addTag(Key::Subsystem, Value::DPL));
745  monitoring.flushBuffer();
746  };
747 
748  // Deciding wether to run or not can be done by passing a request to
749  // the evaluator. In this case, the request is always satisfied and
750  // we run on whatever resource is available.
751  bool enough = mQuotaEvaluator.selectOffer(streamRef.index, mSpec.resourcePolicy.request, uv_now(mState.loop));
752 
753  if (enough) {
754  stream.id = streamRef;
755  stream.running = true;
756  stream.context = &mDataProcessorContexes.at(0);
757 #ifdef DPL_ENABLE_THREADING
758  stream.task.data = &handle;
759  uv_queue_work(mState.loop, &stream.task, run_callback, run_completion);
760 #else
761  run_callback(&handle);
762  run_completion(&handle, 0);
763 #endif
764  } else {
765  mDataProcessorContexes.at(0).deviceContext->quotaEvaluator->handleExpired(reportExpiredOffer);
766  mWasActive = false;
767  }
768  } else {
769  mWasActive = false;
770  }
771  FrameMark;
772  }
774  for (size_t ci = 0; ci < mDeviceContext.spec->inputChannels.size(); ++ci) {
775  auto& info = mDeviceContext.state->inputChannelInfos[ci];
776  info.parts.fParts.clear();
777  }
778  mState.transitionHandling = TransitionHandlingState::NoTransition;

783 void DataProcessingDevice::doPrepare(DataProcessorContext& context)
784 {
785  ZoneScopedN("DataProcessingDevice::doPrepare");
786  context.registry->get<DataProcessingStats>().beginIterationTimestamp = uv_hrtime() / 1000000;
787 
788  *context.wasActive = false;
789  {
790  ZoneScopedN("CallbackService::Id::ClockTick");
791  context.registry->get<CallbackService>()(CallbackService::Id::ClockTick);
792  }
793  // Whether or not we had something to do.
794 
795  // Notice that fake input channels (InputChannelState::Pull) cannot possibly
796  // expect to receive an EndOfStream signal. Thus we do not wait for these
797  // to be completed. In the case of data source devices, as they do not have
798  // real data input channels, they have to signal EndOfStream themselves.
799  context.allDone = std::any_of(context.deviceContext->state->inputChannelInfos.begin(), context.deviceContext->state->inputChannelInfos.end(), [](const auto& info) {
800  return info.parts.fParts.empty() == true && info.state != InputChannelState::Pull;
801  });
802 
803  // Whether or not all the channels are completed
804  for (size_t ci = 0; ci < context.deviceContext->spec->inputChannels.size(); ++ci) {
805  auto& channel = context.deviceContext->spec->inputChannels[ci];
806  auto& info = context.deviceContext->state->inputChannelInfos[ci];
807 
808  if (info.state != InputChannelState::Completed && info.state != InputChannelState::Pull) {
809  context.allDone = false;
810  }
811  if (info.state != InputChannelState::Running) {
812  // Remember to flush data if we are not running
813  // and there is some message pending.
814  if (info.parts.Size()) {
815  DataProcessingDevice::handleData(context, info);
816  }
817  continue;
818  }
819  int64_t result = -2;
820  if (info.channel == nullptr) {
821  info.channel = &context.deviceContext->device->GetChannel(channel.name, 0);
822  }
823  auto& socket = info.channel->GetSocket();
824  // If we have pending events from a previous iteration,
825  // we do receive in any case.
826  // Otherwise we check if there is any pending event and skip
827  // this channel in case there is none.
828  if (info.hasPendingEvents == 0) {
829  socket.Events(&info.hasPendingEvents);
830  // If we do not read, we can continue.
831  if ((info.hasPendingEvents & 1) == 0 && (info.parts.Size() == 0)) {
832  continue;
833  }
834  }
835  // Notice that there seems to be a difference between the documentation
836  // of zeromq and the observed behavior. The fact that ZMQ_POLLIN
837  // is raised does not mean that a message is immediately available to
838  // read, just that it will be available soon, so the receive can
839  // still return -2. To avoid this we keep receiving on the socket until
840  // we get a message. In order not to overflow the DPL queue we process
841  // one message at the time and we keep track of wether there were more
842  // to process.
843  bool newMessages = false;
844  while (true) {
845  if (info.parts.Size() < 64) {
846  FairMQParts parts;
847  info.channel->Receive(parts, 0);
848  for (auto&& part : parts) {
849  info.parts.fParts.emplace_back(std::move(part));
850  }
851  newMessages |= true;
852  }
853 
854  if (info.parts.Size() >= 0) {
855  DataProcessingDevice::handleData(context, info);
856  // Receiving data counts as activity now, so that
857  // We can make sure we process all the pending
858  // messages without hanging on the uv_run.
859  break;
860  }
861  }
862  // We check once again for pending events, keeping track if this was the
863  // case so that we can immediately repeat this loop and avoid remaining
864  // stuck in uv_run. This is because we will not get notified on the socket
865  // if more events are pending due to zeromq level triggered approach.
866  socket.Events(&info.hasPendingEvents);
867  if (info.hasPendingEvents) {
868  *context.wasActive |= newMessages;
869  }
870  }

873 void DataProcessingDevice::doRun(DataProcessorContext& context)
874 {
875  auto switchState = [&registry = context.registry,
876  &state = context.deviceContext->state](StreamingState newState) {
877  LOG(debug) << "New state " << (int)newState << " old state " << (int)state->streaming;
878  state->streaming = newState;
879  registry->get<ControlService>().notifyStreamingState(state->streaming);
880  };
881 
882  if (context.deviceContext->state->streaming == StreamingState::Idle) {
883  *context.wasActive = false;
884  return;
885  }
886 
887  context.completed->clear();
888  context.completed->reserve(16);
889  *context.wasActive |= DataProcessingDevice::tryDispatchComputation(context, *context.completed);
890  DanglingContext danglingContext{*context.registry};
891 
892  context.registry->preDanglingCallbacks(danglingContext);
893  if (*context.wasActive == false) {
894  context.registry->get<CallbackService>()(CallbackService::Id::Idle);
895  }
896  auto activity = context.relayer->processDanglingInputs(*context.expirationHandlers, *context.registry, true);
897  *context.wasActive |= activity.expiredSlots > 0;
898 
899  context.completed->clear();
900  *context.wasActive |= DataProcessingDevice::tryDispatchComputation(context, *context.completed);
901 
902  context.registry->postDanglingCallbacks(danglingContext);
903 
904  // If we got notified that all the sources are done, we call the EndOfStream
905  // callback and return false. Notice that what happens next is actually
906  // dependent on the callback, not something which is controlled by the
907  // framework itself.
908  if (context.allDone == true && context.deviceContext->state->streaming == StreamingState::Streaming) {
909  switchState(StreamingState::EndOfStreaming);
910  *context.wasActive = true;
911  }
912 
913  if (context.deviceContext->state->streaming == StreamingState::EndOfStreaming) {
914  context.registry->get<DriverClient>().flushPending();
915  // We keep processing data until we are Idle.
916  // FIXME: not sure this is the correct way to drain the queues, but
917  // I guess we will see.
920  bool hasOnlyGenerated = (context.deviceContext->spec->inputChannels.size() == 1) && (context.deviceContext->spec->inputs[0].matcher.lifetime == Lifetime::Timer || context.deviceContext->spec->inputs[0].matcher.lifetime == Lifetime::Enumeration);
921  while (DataProcessingDevice::tryDispatchComputation(context, *context.completed) && hasOnlyGenerated == false) {
922  context.relayer->processDanglingInputs(*context.expirationHandlers, *context.registry, false);
923  }
924  EndOfStreamContext eosContext{*context.registry, *context.allocator};
925 
926  context.registry->preEOSCallbacks(eosContext);
927  context.registry->get<CallbackService>()(CallbackService::Id::EndOfStream, eosContext);
928  context.registry->postEOSCallbacks(eosContext);
929 
930  for (auto& channel : context.deviceContext->spec->outputChannels) {
931  DataProcessingHelpers::sendEndOfStream(*context.deviceContext->device, channel);
932  }
933  // This is needed because the transport is deleted before the device.
934  context.relayer->clear();
935  switchState(StreamingState::Idle);
936  if (hasOnlyGenerated) {
937  *context.wasActive = false;
938  } else {
939  *context.wasActive = true;
940  }
941  // On end of stream we shut down all output pollers.
942  for (auto& poller : context.deviceContext->state->activeOutputPollers) {
943  uv_poll_stop(poller);
944  }
945  context.deviceContext->state->activeOutputPollers.clear();
946  return;
947  }
948 
949  if (context.deviceContext->state->streaming == StreamingState::Idle) {
950  // On end of stream we shut down all output pollers.
951  for (auto& poller : context.deviceContext->state->activeOutputPollers) {
952  uv_poll_stop(poller);
953  }
954  context.deviceContext->state->activeOutputPollers.clear();
955  }
956 
957  return;

975 void DataProcessingDevice::handleData(DataProcessorContext& context, InputChannelInfo& info)
976 {
977  ZoneScopedN("DataProcessingDevice::handleData");
978  assert(context.deviceContext->spec->inputChannels.empty() == false);
979  assert(info.parts.Size() > 0);
980 
981  // Initial part. Let's hide all the unnecessary and have
982  // simple lambdas for each of the steps I am planning to have.
983  assert(!context.deviceContext->spec->inputs.empty());
984 
985  enum struct InputType {
986  Invalid,
987  Data,
988  SourceInfo
989  };
990 
991  struct InputInfo {
992  InputInfo(size_t p, size_t s, InputType t)
993  : position(p), size(s), type(t)
994  {
995  }
996  size_t position;
997  size_t size;
998  InputType type;
999  };
1000 
1001  // This is how we validate inputs. I.e. we try to enforce the O2 Data model
1002  // and we do a few stats. We bind parts as a lambda captured variable, rather
1003  // than an input, because we do not want the outer loop actually be exposed
1004  // to the implementation details of the messaging layer.
1005  auto getInputTypes = [&stats = context.registry->get<DataProcessingStats>(),
1006  &info, &context]() -> std::optional<std::vector<InputInfo>> {
1007  auto& parts = info.parts;
1008  stats.inputParts = parts.Size();
1009 
1010  TracyPlot("messages received", (int64_t)parts.Size());
1011  std::vector<InputInfo> results;
1012  // we can reserve the upper limit
1013  results.reserve(parts.Size() / 2);
1014  size_t nTotalPayloads = 0;
1015 
1016  auto insertInputInfo = [&results, &nTotalPayloads](size_t position, size_t length, InputType type) {
1017  results.emplace_back(position, length, type);
1018  if (type != InputType::Invalid && length > 1) {
1019  nTotalPayloads += length - 1;
1020  }
1021  };
1022 
1023  for (size_t pi = 0; pi < parts.Size(); pi += 2) {
1024  auto sih = o2::header::get<SourceInfoHeader*>(parts.At(pi)->GetData());
1025  if (sih) {
1026  info.state = sih->state;
1027  insertInputInfo(pi, 2, InputType::SourceInfo);
1028  *context.wasActive = true;
1029  continue;
1030  }
1031  auto dh = o2::header::get<DataHeader*>(parts.At(pi)->GetData());
1032  if (!dh) {
1033  insertInputInfo(pi, 0, InputType::Invalid);
1034  LOGP(error, "Header is not a DataHeader?");
1035  continue;
1036  }
1037  if (dh->payloadSize > parts.At(pi + 1)->GetSize()) {
1038  insertInputInfo(pi, 0, InputType::Invalid);
1039  LOGP(error, "DataHeader payloadSize mismatch");
1040  continue;
1041  }
1042  TracyPlot("payload size", (int64_t)dh->payloadSize);
1043  auto dph = o2::header::get<DataProcessingHeader*>(parts.At(pi)->GetData());
1044  TracyAlloc(parts.At(pi + 1)->GetData(), parts.At(pi + 1)->GetSize());
1045  if (!dph) {
1046  insertInputInfo(pi, 2, InputType::Invalid);
1047  LOGP(error, "Header stack does not contain DataProcessingHeader");
1048  continue;
1049  }
1050  if (dh->splitPayloadParts > 0 && dh->splitPayloadParts == dh->splitPayloadIndex) {
1051  // this is indicating a sequence of payloads following the header
1052  // FIXME: we will probably also set the DataHeader version
1053  insertInputInfo(pi, dh->splitPayloadParts + 1, InputType::Data);
1054  pi += dh->splitPayloadParts - 1;
1055  } else {
1056  // We can set the type for the next splitPayloadParts
1057  // because we are guaranteed they are all the same.
1058  // If splitPayloadParts = 0, we assume that means there is only one (header, payload)
1059  // pair.
1060  size_t finalSplitPayloadIndex = pi + (dh->splitPayloadParts > 0 ? dh->splitPayloadParts : 1) * 2;
1061  if (finalSplitPayloadIndex > parts.Size()) {
1062  LOGP(error, "DataHeader::splitPayloadParts invalid");
1063  insertInputInfo(pi, 0, InputType::Invalid);
1064  continue;
1065  }
1066  insertInputInfo(pi, 2, InputType::Data);
1067  for (; pi + 2 < finalSplitPayloadIndex; pi += 2) {
1068  insertInputInfo(pi + 2, 2, InputType::Data);
1069  }
1070  }
1071  }
1072  assert(std::accumulate(results.begin(), results.end(), 0, [](size_t const& count, auto const& element) -> size_t { return count + element.size; }));
1073  if (results.size() + nTotalPayloads != parts.Size()) {
1074  LOG(error) << "inconsistent number of inputs extracted";
1075  return std::nullopt;
1076  }
1077  return results;
1078  };

1082  };

1084  auto handleValidMessages = [&info, &context = context, &relayer = *context.relayer, &reportError](std::vector<InputInfo> const& inputInfos) {
1085  static WaitBackpressurePolicy policy;
1086  auto& parts = info.parts;
1087  // We relay execution to make sure we have a complete set of parts
1088  // available.
1089  for (auto ii = 0; ii < inputInfos.size(); ++ii) {
1090  auto const& input = inputInfos[ii];
1091  switch (input.type) {
1092  case InputType::Data: {
1093  auto headerIndex = input.position;
1094  auto nMessages = 0;
1095  auto nPayloadsPerHeader = 0;
1096  if (input.size > 2) {
1097  // header and multiple payload sequence
1098  nMessages = input.size;
1099  nPayloadsPerHeader = nMessages - 1;
1100  } else {
1101  // multiple header-payload pairs
1102  auto dh = o2::header::get<DataHeader*>(parts.At(headerIndex)->GetData());
1103  nMessages = dh->splitPayloadParts > 0 ? dh->splitPayloadParts * 2 : 2;
1104  nPayloadsPerHeader = 1;
1105  ii += (nMessages / 2) - 1;
1106  }
1107  auto relayed = relayer.relay(parts.At(headerIndex)->GetData(),
1108  &parts.At(headerIndex),
1109  nMessages,
1110  nPayloadsPerHeader);
1111  switch (relayed) {
1112  case DataRelayer::Backpressured:
1113  if (info.normalOpsNotified == true && info.backpressureNotified == false) {
1114  LOGP(alarm, "Backpressure on channel {}. Waiting.", info.channel->GetName());
1115  info.backpressureNotified = true;
1116  info.normalOpsNotified = false;
1117  }
1118  policy.backpressure(info);
1119  break;
1120  case DataRelayer::Dropped:
1121  case DataRelayer::Invalid:
1122  case DataRelayer::WillRelay:
1123  if (info.normalOpsNotified == false && info.backpressureNotified == true) {
1124  LOGP(info, "Back to normal on channel {}.", info.channel->GetName());
1125  info.normalOpsNotified = true;
1126  info.backpressureNotified = false;
1127  }
1128  break;
1129  }
1130  } break;
1131  case InputType::SourceInfo: {
1132  *context.wasActive = true;
1133  auto headerIndex = input.position;
1134  auto payloadIndex = input.position + 1;
1135  assert(payloadIndex < parts.Size());
1136  auto dh = o2::header::get<DataHeader*>(parts.At(headerIndex)->GetData());
1137  // FIXME: the message with the end of stream cannot contain
1138  // split parts.
1139  parts.At(headerIndex).reset(nullptr);
1140  parts.At(payloadIndex).reset(nullptr);
1141  // for (size_t i = 0; i < dh->splitPayloadParts > 0 ? dh->splitPayloadParts * 2 - 1 : 1; ++i) {
1142  // parts.At(headerIndex + 1 + i).reset(nullptr);
1143  // }
1144  // pi += dh->splitPayloadParts > 0 ? dh->splitPayloadParts - 1 : 0;
1145 
1146  } break;
1147  case InputType::Invalid: {
1148  reportError("Invalid part found.");
1149  } break;
1150  }
1151  }
1152  auto it = std::remove_if(parts.fParts.begin(), parts.fParts.end(), [](auto& msg) -> bool { return msg.get() == nullptr; });
1153  auto r = std::distance(it, parts.fParts.end());
1154  parts.fParts.erase(it, parts.end());
1155  if (parts.fParts.size()) {
1156  LOG(debug) << parts.fParts.size() << " messages backpressured";
1157  }
1158  };
1159 
1160  // Second part. This is the actual outer loop we want to obtain, with
1161  // implementation details which can be read. Notice how most of the state
1162  // is actually hidden. For example we do not expose what "input" is. This
1163  // will allow us to keep the same toplevel logic even if the actual meaning
