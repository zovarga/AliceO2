52 void Tracking::checkTrack(const TrackTRD& trkTrd, bool isTPCTRD)
53 {
54  auto propagator = o2::base::Propagator::Instance();
55  TrackQC qcStruct;
56  qcStruct.type = isTPCTRD ? 0 : 1;
57  qcStruct.nTracklets = trkTrd.getNtracklets();
58  qcStruct.chi2 = trkTrd.getChi2();
59  qcStruct.reducedChi2 = trkTrd.getReducedChi2();
60  qcStruct.pt = trkTrd.getPt();
61 
62  LOGF(debug, "Got track with %i tracklets and ID %i", trkTrd.getNtracklets(), trkTrd.getRefGlobalTrackId());
63  const auto& trkSeed = isTPCTRD ? mTracksTPC[trkTrd.getRefGlobalTrackId()].getParamOut() : mTracksITSTPC[trkTrd.getRefGlobalTrackId()].getParamOut();
64  auto trk = trkSeed;
65  for (int iLayer = 0; iLayer < NLAYER; ++iLayer) {
66  qcStruct.findable[iLayer] = trkTrd.getIsFindable(iLayer);
67  int trkltId = trkTrd.getTrackletIndex(iLayer);
68  if (trkltId < 0) {
69  continue;
70  }
71  int trkltDet = mTrackletsRaw[trkltId].getDetector();
72  int trkltSec = trkltDet / (NLAYER * NSTACK);
73  if (trkltSec != o2::math_utils::angle2Sector(trk.getAlpha())) {
74  if (!trk.rotate(o2::math_utils::sector2Angle(trkltSec))) {
75  LOGF(debug, "Track could not be rotated in tracklet coordinate system");
76  break;
77  }
78  }
79  if (!propagator->PropagateToXBxByBz(trk, mTrackletsCalib[trkltId].getX(), mMaxSnp, mMaxStep, mMatCorr)) {
80  LOGF(debug, "Track propagation failed in layer %i (pt=%f, xTrk=%f, xToGo=%f)", iLayer, trk.getPt(), trk.getX(), mTrackletsCalib[trkltId].getX());
81  break;
82  }
83 
84  const PadPlane* pad = Geometry::instance()->getPadPlane(trkltDet);
85  float tilt = tan(TMath::DegToRad() * pad->getTiltingAngle()); // tilt is signed! and returned in degrees
86  float tiltCorrUp = tilt * (mTrackletsCalib[trkltId].getZ() - trk.getZ());
87  float zPosCorrUp = mTrackletsCalib[trkltId].getZ() + mRecoParam.getZCorrCoeffNRC() * trk.getTgl();
88  float padLength = pad->getRowSize(mTrackletsRaw[trkltId].getPadRow());
89  if (!((trk.getSigmaZ2() < (padLength * padLength / 12.f)) && (std::fabs(mTrackletsCalib[trkltId].getZ() - trk.getZ()) < padLength))) {
90  tiltCorrUp = 0.f;
91  }
92 
93  std::array<float, 2> trkltPosUp{mTrackletsCalib[trkltId].getY() - tiltCorrUp, zPosCorrUp};
94  std::array<float, 3> trkltCovUp;
95  mRecoParam.recalcTrkltCov(tilt, trk.getSnp(), pad->getRowSize(mTrackletsRaw[trkltId].getPadRow()), trkltCovUp);
96  auto chi2trklt = trk.getPredictedChi2(trkltPosUp, trkltCovUp);
97 
98  qcStruct.trackX[iLayer] = trk.getX();
99  qcStruct.trackY[iLayer] = trk.getY();
100  qcStruct.trackZ[iLayer] = trk.getZ();
101  qcStruct.trackSnp[iLayer] = trk.getSnp();
102  qcStruct.trackTgl[iLayer] = trk.getTgl();
103  qcStruct.trackQpt[iLayer] = trk.getQ2Pt();
104  qcStruct.trackletYraw[iLayer] = mTrackletsCalib[trkltId].getY();
105  qcStruct.trackletZraw[iLayer] = mTrackletsCalib[trkltId].getZ();
106  qcStruct.trackletY[iLayer] = trkltPosUp[0];
107  qcStruct.trackletZ[iLayer] = trkltPosUp[1];
108  qcStruct.trackletDy[iLayer] = mTrackletsCalib[trkltId].getDy();
109  qcStruct.trackletSlope[iLayer] = mTrackletsRaw[trkltId].getSlopeBinSigned();
110  qcStruct.trackletDet[iLayer] = trkltDet;
111  qcStruct.trackletHCId[iLayer] = mTrackletsRaw[trkltId].getHCID();
112  qcStruct.trackletRob[iLayer] = mTrackletsRaw[trkltId].getROB();
113  qcStruct.trackletMcm[iLayer] = mTrackletsRaw[trkltId].getMCM();
114  qcStruct.trackletChi2[iLayer] = chi2trklt;
115  }
116  mTrackQC.push_back(qcStruct);
117 }
