83 bool Tracks::processTrack(const o2::tpc::TrackTPC& track)
84 {
85  // ===| variables required for cutting and filling |===
86  const auto eta = track.getEta();
87  const auto phi = track.getPhi();
88  const auto pt = track.getPt();
89  const auto sign = track.getSign();
90  const auto nCls = track.getNClusterReferences();
91  const auto dEdxTot = track.getdEdx().dEdxTotTPC;
92 
93  double absEta = TMath::Abs(eta);
94 
95  // ===| histogram filling before cuts |===
96  mHist1D[0].Fill(nCls);
97  mHist1D[15].Fill(eta);
98  mHist1D[16].Fill(pt);
99  mHist2D[7].Fill(eta, nCls);
100  mHist2D[8].Fill(pt, nCls);
101  mHist2D[9].Fill(phi, eta);
102 
103  // ===| histogram filling including cuts |===
104  if (absEta < 1. && nCls > 60 && dEdxTot > 20) {
105 
106  // ===| 1D histogram filling |===
107  mHist1D[1].Fill(nCls);
108  mHist1D[2].Fill(eta);
109 
110  if (eta > 0.) {
111  mHist1D[3].Fill(phi);
112  } else {
113  mHist1D[4].Fill(phi);
114  }
115 
116  mHist1D[5].Fill(pt);
117  mHist1D[6].Fill(sign);
118 
119  if (sign < 0.) {
120  mHist1D[7].Fill(eta);
121  mHist1D[13].Fill(pt);
122  if (eta > 0.) {
123  mHist1D[9].Fill(phi);
124  } else {
125  mHist1D[11].Fill(phi);
126  }
127  } else {
128  mHist1D[8].Fill(eta);
129  mHist1D[14].Fill(pt);
130  if (eta > 0.) {
131  mHist1D[10].Fill(phi);
132  } else {
133  mHist1D[12].Fill(phi);
134  }
135  }
136 
137  // ===| 2D histogram filling |===
138  mHist2D[0].Fill(eta, nCls);
139 
140  if (eta > 0.) {
141  mHist2D[1].Fill(phi, nCls);
142  } else {
143  mHist2D[2].Fill(phi, nCls);
144  }
145 
146  mHist2D[3].Fill(pt, nCls);
147  mHist2D[4].Fill(phi, eta);
148 
149  if (sign < 0.) {
150  mHist2D[5].Fill(phi, eta);
151  } else {
152  mHist2D[6].Fill(phi, eta);
153  }
154  }
155 
156  return true;
157 }
