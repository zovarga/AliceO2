25 // Return a vector of file names in the current dir matching the search
26 // pattern. If searchpattern is empty, all files will be returned. Otherwise
27 // searchpattern will be treated/parsed as proper regular expression.
28 std::vector<std::string> listFiles(std::string const& dir, std::string const& searchpattern)
29 {
30  std::vector<std::string> filenames;
31  std::string rs = searchpattern.empty() ? ".*" : searchpattern;
32  std::regex str_expr(rs);
33 
34  for (auto& p : std::filesystem::directory_iterator(dir)) {
35  if (!p.is_directory()) {
36  auto fn = p.path().filename().string();
37  if (regex_match(fn, str_expr)) {
38  filenames.push_back(p.path().string());
39  }
40  }
41  }
42  return filenames;
43 }
