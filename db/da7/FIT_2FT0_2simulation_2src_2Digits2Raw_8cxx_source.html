11 /*
12  Digits to RAW data coding. RAW data format - class DataFormat/Detectors/FIT/FT0/RawEventData
13  18 PMs (GBT links) 12 MCPs each and 1 TCM, each stream transmit separately
14 Event header - 80bits
15  uint startDescriptor : 4;
16  uint nGBTWords : 4;
17  uint reservedField : 32;
18  uint orbit : 32;
19  uint bc : 12;
20 
21  Event data 40bits
22  short int time : 12;
23  short int charge : 12;
24  unsigned short int numberADC : 1;
25  bool isDoubleEvent : 1;
26  bool is1TimeLostEvent : 1;
27  bool is2TimeLostEvent : 1;
28  bool isADCinGate : 1;
29  bool isTimeInfoLate : 1;
30  bool isAmpHigh : 1;
31  bool isEventInTVDC : 1;
32  bool isTimeInfoLost : 1;
33  uint channelID : 4;
34 GBT packet:
35 RDH + Event header + event data, 2 channels per 1 GBT word;
36 if no data for this PM - only headers.
37 
38 Trigger mode : detector sends data to FLP at each trigger;
39 Continueous mode : for only bunches with data at least in 1 channel.
40 */

76 void Digits2Raw::readDigits(const std::string& outDir, const std::string& fileDigitsName)
77 {
78  LOG(info) << "**********Digits2Raw::convertDigits" << std::endl;
79  mWriter.setCarryOverCallBack(this);
80 
81  o2::ft0::LookUpTable lut{o2::ft0::LookUpTable::readTable()};
82  mLinkTCM = lut.getLink(lut.getTCMchannel());
83  LOG(info) << " ##### LookUp set, TCM " << mLinkTCM;
84  std::string outd = outDir;
85  if (outd.back() != '/') {
86  outd += '/';
87  }
88  using namespace o2::raw;
89  for (int ilink = 0; ilink < NPMs - 1; ++ilink) {
90  if (ilink < 8) {
91  mLinkID = uint32_t(ilink);
92  mCruID = uint16_t(0);
93  mEndPointID = uint32_t(0);
94  mFeeID = uint64_t(ilink);
95  } else {
96  mLinkID = uint32_t(ilink) - 8;
97  mCruID = uint16_t(0);
98  mEndPointID = uint32_t(1);
99  mFeeID = uint64_t(ilink);
100  }
101  std::string outFileLink = mOutputPerLink ? o2::utils::Str::concat_string(outDir, "ft0_link", std::to_string(ilink), ".raw") : o2::utils::Str::concat_string(outDir, "ft0.raw");
102  mWriter.registerLink(mFeeID, mCruID, mLinkID, mEndPointID, outFileLink);
103  LOG(info) << " registered links " << mLinkID << " endpoint " << mEndPointID;
104  }
105  //TCM
106  std::string outFileLink = mOutputPerLink ? o2::utils::Str::concat_string(outDir, "ft0_link", std::to_string(NPMs - 1), ".raw") : o2::utils::Str::concat_string(outDir, "ft0.raw");
107  mWriter.registerLink(mLinkTCM + 8, mCruID, mLinkTCM, 0, outFileLink);
108  LOG(info) << " registered link TCM " << mLinkTCM;
109 
110  TFile* fdig = TFile::Open(fileDigitsName.data());
111  assert(fdig != nullptr);
112  LOG(info) << " Open digits file " << fileDigitsName.data();
113  TTree* digTree = (TTree*)fdig->Get("o2sim");
114 
115  std::vector<o2::ft0::Digit> digitsBC, *ft0BCDataPtr = &digitsBC;
116  std::vector<o2::ft0::ChannelData> digitsCh, *ft0ChDataPtr = &digitsCh;
117 
118  digTree->SetBranchAddress("FT0DIGITSBC", &ft0BCDataPtr);
119  digTree->SetBranchAddress("FT0DIGITSCH", &ft0ChDataPtr);
120 
121  uint32_t old_orbit = ~0;
122  o2::InteractionRecord intRecord;
123  o2::InteractionRecord lastIR = mSampler.getFirstIR();
124  std::vector<o2::InteractionRecord> HBIRVec;
125 
126  for (int ient = 0; ient < digTree->GetEntries(); ient++) {
127  digTree->GetEntry(ient);
128 
129  int nbc = digitsBC.size();
130  LOG(info) << "Entry " << ient << " : " << nbc << " BCs stored";
131  for (int ibc = 0; ibc < nbc; ibc++) {
132  auto& bcd = digitsBC[ibc];
133  intRecord = bcd.getIntRecord();
134  auto channels = bcd.getBunchChannelData(digitsCh);
135  if (!channels.empty()) {
136  convertDigits(bcd, channels, lut, intRecord);
137  }
138  }
139  }

142 void Digits2Raw::convertDigits(o2::ft0::Digit bcdigits,
143  gsl::span<const ChannelData> pmchannels,
144  const o2::ft0::LookUpTable& lut,
145  o2::InteractionRecord const& intRecord)
146 {
147 
148  // check empty event
149  int oldlink = -1;
150  int oldendpoint = -1;
151  int nchannels = 0;
152  int nch = pmchannels.size();
153  for (int ich = 0; ich < nch; ich++) {
154  int nlink = lut.getLink(pmchannels[ich].ChId);
155  int ep = lut.getEP(pmchannels[ich].ChId);
156  if (nlink != oldlink || ep != oldendpoint) {
157  if (oldlink >= 0) {
158  uint nGBTWords = uint((nchannels + 1) / 2);
159  LOG(debug) << " oldlink " << oldlink << " old EP " << oldendpoint << " nGBTWords " << nGBTWords << " new link " << nlink << " ep " << ep;
160  if ((nchannels % 2) == 1) {
161  mRawEventData.mEventData[nchannels] = {};
162  }
163  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
164  auto data = mRawEventData.to_vector(false);
165  mLinkID = uint32_t(oldlink);
166  mFeeID = uint64_t(oldlink);
167  mEndPointID = uint32_t(oldendpoint);
168  if (mEndPointID == 1) {
169  mFeeID += 8;
170  }
171  LOG(debug) << " new link start " << mFeeID << " " << mCruID << " " << mLinkID << " " << mEndPointID;
172  mWriter.addData(mFeeID, mCruID, mLinkID, mEndPointID, intRecord, data);
173  }
174  oldlink = nlink;
175  oldendpoint = ep;
176  mRawEventData.mEventHeader = makeGBTHeader(nlink, intRecord);
177  nchannels = 0;
178  LOG(debug) << " switch to new link " << nlink << " EP " << ep;
179  }
180  auto& newData = mRawEventData.mEventData[nchannels];
181  bool isAside = (pmchannels[ich].ChId < 96);
182  newData.charge = pmchannels[ich].QTCAmpl;
183  newData.time = pmchannels[ich].CFDTime;
184  newData.generateFlags();
185  newData.channelID = lut.getMCP(pmchannels[ich].ChId);
186  LOG(debug) << " ID " << int(pmchannels[ich].ChId) << " packed GBT " << nlink << " channelID " << (int)newData.channelID << " charge " << newData.charge << " time " << newData.time << " chain " << int(newData.numberADC)
187  << " size " << sizeof(newData) << " mEndPointID " << ep;
188  nchannels++;
189  }
190  // fill mEventData[nchannels] with 0s to flag that this is a dummy data
191  uint nGBTWords = uint((nchannels + 1) / 2);
192  if ((nchannels % 2) == 1) {
193  mRawEventData.mEventData[nchannels] = {};
194  }
195  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
196  auto datalast = mRawEventData.to_vector(false);
197  mLinkID = uint32_t(oldlink);
198  mFeeID = uint64_t(oldlink);
199  mEndPointID = uint32_t(oldendpoint);
200  if (mEndPointID == 1) {
201  mFeeID += 8;
202  }
203  mWriter.addData(mFeeID, mCruID, mLinkID, mEndPointID, intRecord, datalast);
204  LOG(debug) << " last " << mFeeID << " " << mCruID << " " << mLinkID << " " << mEndPointID;
205  //TCM
206  mRawEventData.mEventHeader = makeGBTHeader(mLinkTCM, intRecord); //TCM
207  mRawEventData.mEventHeader.nGBTWords = 1;
208  auto& tcmdata = mRawEventData.mTCMdata;
209  mTriggers = bcdigits.getTriggers();
210 
211  float ampA = mTriggers.amplA;
212  float ampC = mTriggers.amplC;
213  if (ampA > 131071) {
214  ampA = 131071; //2^17
215  }
216  if (ampC > 131071) {
217  ampC = 131071; //2^17
218  }
219  tcmdata.vertex = mTriggers.getVertex();
220  tcmdata.orA = mTriggers.getOrA();
221  tcmdata.orC = mTriggers.getOrC();
222  tcmdata.sCen = mTriggers.getSCen();
223  tcmdata.cen = mTriggers.getCen();
224  tcmdata.nChanA = mTriggers.nChanA;
225  tcmdata.nChanC = mTriggers.nChanC;
226  tcmdata.amplA = ampA;
227  tcmdata.amplC = ampC;
228  tcmdata.timeA = mTriggers.timeA;
229  tcmdata.timeC = mTriggers.timeC;
230  LOG(debug) << " TCM triggers read "
231  << " time A " << mTriggers.timeA << " time C " << mTriggers.timeC
232  << " amp A " << ampA << " amp C " << ampC
233  << " N A " << int(mTriggers.nChanA) << " N C " << int(mTriggers.nChanC)
234  << " trig "
235  << " ver " << mTriggers.getVertex() << " A " << mTriggers.getOrA() << " C " << mTriggers.getOrC();
236 
237  LOG(debug) << "TCMdata"
238  << " time A " << tcmdata.timeA << " time C " << tcmdata.timeC
239  << " amp A " << tcmdata.amplA << " amp C " << tcmdata.amplC
240  << " N A " << int(tcmdata.nChanA) << " N C " << int(tcmdata.nChanC)
241  << " trig "
242  << " ver " << tcmdata.vertex << " A " << tcmdata.orA << " C " << tcmdata.orC
243  << " size " << sizeof(tcmdata);
244 
245  auto data = mRawEventData.to_vector(1);
246  mLinkID = uint32_t(mLinkTCM);
247  mFeeID = uint64_t(mLinkTCM) + 8;
248  mEndPointID = 0;
249  mWriter.addData(mFeeID, mCruID, mLinkID, mEndPointID, intRecord, data);
250  LOG(debug) << " TCM " << mFeeID << " " << mCruID << " " << mLinkID << " " << mEndPointID;
251 }
