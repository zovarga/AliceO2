29 ObjectStore extractObjectFrom(const framework::DataRef& ref)
30 {
31  // We do extraction on the low level to efficiently determine if the message
32  // contains an object inheriting MergeInterface or TObject. If we did it the
33  // the following way and catch an exception:
34  // framework::DataRefUtils::as<MergeInterface>(ref)
35  // it could cause a memory leak if `ref` contained a non-owning TCollection.
36  // This way we also avoid doing most of the checks twice.
37  const static std::string errorPrefix = "Could not extract object to be merged: ";
38 
39  using DataHeader = o2::header::DataHeader;
40  auto header = framework::DataRefUtils::getHeader<const DataHeader*>(ref);
41  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
42  throw std::runtime_error(errorPrefix + "It is not ROOT-serialized");
43  }
44 
45  o2::framework::FairTMessage ftm(const_cast<char*>(ref.payload), header->payloadSize);
46  auto* storedClass = ftm.GetClass();
47  if (storedClass == nullptr) {
48  throw std::runtime_error(errorPrefix + "Unknown stored class");
49  }
50 
51  auto* mergeInterfaceClass = TClass::GetClass(typeid(MergeInterface));
52  auto* tObjectClass = TClass::GetClass(typeid(TObject));
53 
54  bool inheritsFromMergeInterface = storedClass->InheritsFrom(mergeInterfaceClass);
55  bool inheritsFromTObject = storedClass->InheritsFrom(tObjectClass);
56 
57  if (!inheritsFromMergeInterface && !inheritsFromTObject) {
58  throw std::runtime_error(
59  errorPrefix + "Class '" + storedClass->GetName() + "'does not inherit from MergeInterface nor TObject");
60  }
61 
62  auto* object = ftm.ReadObjectAny(storedClass);
63  if (object == nullptr) {
64  throw std::runtime_error(
65  errorPrefix + "Failed to read object with name '" + storedClass->GetName() + "' from message using ROOT serialization.");
66  }
67 
68  if (inheritsFromMergeInterface) {
69  MergeInterface* objectAsMergeInterface = inheritsFromTObject ? dynamic_cast<MergeInterface*>(static_cast<TObject*>(object)) : static_cast<MergeInterface*>(object);
70  if (objectAsMergeInterface == nullptr) {
71  throw std::runtime_error(errorPrefix + "Could not cast '" + storedClass->GetName() + "' to MergeInterface");
72  }
73  objectAsMergeInterface->postDeserialization();
74  return MergeInterfacePtr(objectAsMergeInterface);
75  } else {
76  return TObjectPtr(static_cast<TObject*>(object), algorithm::deleteTCollections);
77  }
78 }
