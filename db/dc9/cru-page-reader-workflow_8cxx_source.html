68  void computePayloadSize()
69  {
70  payloadSize = 0;
71  if (buf == nullptr) {
72  return;
73  }
74 
75  size_t offset = 0;
76  while (offset < totalSize) {
77  char* ptr = buf + offset;
78  RDH* rdh = (RDH*)ptr;
79 
80  auto rdhVersion = o2::raw::RDHUtils::getVersion(rdh);
81  auto rdhHeaderSize = o2::raw::RDHUtils::getHeaderSize(rdh);
82  auto memorySize = o2::raw::RDHUtils::getMemorySize(rdh);
83  auto pageSize = o2::raw::RDHUtils::getOffsetToNext(rdh);
84 
85  payloadSize += memorySize - rdhHeaderSize;
86 
87  offset += pageSize;
88  }
89  }

226  bool appendHBF(TimeFrame& tf, char* framePtr, size_t frameSize, bool addHBF)
227  {
228  // new size of the TimeFrame buffer after appending the HBFrame
229  size_t newSize = tf.totalSize + frameSize;
230  // increase the size of the memory buffer
231  tf.buf = (char*)realloc(tf.buf, newSize);
232  if (tf.buf == nullptr) {
233  std::cout << "failed to allocate TimeFrame buffer" << std::endl;
234  return false;
235  }
236 
237  // copy the HBFrame into the TimeFrame buffer
238  char* bufPtr = tf.buf + tf.totalSize;
239  memcpy(bufPtr, framePtr, frameSize);
240 
241  if (addHBF) {
242  // add the offset and size of the HBFrame to the vector
243  tf.hbframes.emplace_back(std::make_pair(tf.totalSize, frameSize));
244  // increase the TimeFrame sizes
245  tf.tfSize += frameSize;
246  }
247 
248  // increase the total buffer sizes
249  tf.totalSize += frameSize;
250 
251  return true;
252  }

255  void sendTF(framework::ProcessingContext& pc)
256  {
258  RDH rdh;
259  char* buf{nullptr};
260  size_t frameSize{0};
261 
262  static int TFid = 0;
263 
264  if (mTimeFrameMax > 0 && TFid == mTimeFrameMax) {
265  pc.services().get<ControlService>().endOfStream();
266  return;
267  }
268 
269  while (true) {
270 
271  // stop if the required number of frames has been reached
272  if (mFrameMax == 0) {
273  pc.services().get<ControlService>().endOfStream();
274  return;
275  }
276 
277  if (mPrint && false) {
278  printf("mFrameMax: %d\n", mFrameMax);
279  }
280  if (mFrameMax > 0) {
281  mFrameMax -= 1;
282  }
283 
284  // read the next RDH, stop if no more data is available
285  if (mPrint) {
286  std::cout << "Reading " << sizeof(RDH) << " for RDH from input file\n";
287  }
288  mInputFile.read((char*)(&rdh), sizeof(RDH));
289  if (mInputFile.fail()) {
290  if (mPrint) {
291  std::cout << "end of file reached" << std::endl;
292  }
293  pc.services().get<ControlService>().endOfStream();
294  return; // probably reached eof
295  }
296 
297  // check that the RDH version is ok (only RDH versions from 4 to 6 are supported at the moment)
298  auto rdhVersion = o2::raw::RDHUtils::getVersion(rdh);
299  auto rdhHeaderSize = o2::raw::RDHUtils::getHeaderSize(rdh);
300  uint16_t cruID = o2::raw::RDHUtils::getCRUID(rdh) & 0x3F;
301  uint8_t endPointID = o2::raw::RDHUtils::getEndPointID(rdh);
302  uint8_t linkID = o2::raw::RDHUtils::getLinkID(rdh);
303  uint16_t feeID = cruID * 2 + endPointID;
304  auto stopBit = o2::raw::RDHUtils::getStop(rdh);
305  auto triggerType = o2::raw::RDHUtils::getTriggerType(rdh);
306  auto pageSize = o2::raw::RDHUtils::getOffsetToNext(rdh);
307  auto pageCounter = RDHUtils::getPageCounter(rdh);
308 
309  if (mPrint) {
310  printf("%6d: V %X offset %4d packet %3d srcID %d cruID %2d dp %d link %2d orbit %u bc %4d trig 0x%08X p %d s %d",
311  (int)0, (int)rdhVersion, (int)pageSize,
312  (int)RDHUtils::getPacketCounter(rdh), (int)RDHUtils::getSourceID(rdh),
313  (int)cruID, (int)endPointID, (int)linkID,
314  (uint32_t)RDHUtils::getHeartBeatOrbit(rdh), (int)RDHUtils::getTriggerBC(rdh),
315  (int)triggerType, (int)pageCounter, (int)stopBit);
316  if ((triggerType & 0x800) != 0) {
317  printf(" <===");
318  }
319  printf("\n");
320  }
321  if (rdhVersion < 4 || rdhVersion > 6 || rdhHeaderSize != 64) {
322  return;
323  }
324 
325  TFQueue& tfQueue = tfQueues[feeID][linkID];
326 
327  // get the frame size from the RDH offsetToNext field
328  if (mPrint && false) {
329  std::cout << "pageSize=" << pageSize << std::endl;
330  }
331 
332  // stop if the frame size is too small
333  if (pageSize < rdhHeaderSize) {
334  std::cout << mFrameMax << " - pageSize too small: " << pageSize << std::endl;
335  pc.services().get<ControlService>().endOfStream();
336  return;
337  }
338 
339  // allocate or extend the output buffer
340  buf = (char*)realloc(buf, frameSize + pageSize);
341  if (buf == nullptr) {
342  std::cout << mFrameMax << " - failed to allocate buffer" << std::endl;
343  pc.services().get<ControlService>().endOfStream();
344  return;
345  }
346 
347  // copy the RDH into the output buffer
348  memcpy(buf + frameSize, &rdh, rdhHeaderSize);
349 
350  // read the frame payload into the output buffer
351  if (mPrint) {
352  std::cout << "Reading " << pageSize - rdhHeaderSize << " for payload from input file\n";
353  }
354  mInputFile.read(buf + frameSize + rdhHeaderSize, pageSize - rdhHeaderSize);
355 
356  // stop if data cannot be read completely
357  if (mInputFile.fail()) {
358  if (mPrint) {
359  std::cout << "end of file reached" << std::endl;
360  }
361  free(buf);
362  pc.services().get<ControlService>().endOfStream();
363  return; // probably reached eof
364  }
365 
366  // increment the total buffer size
367  frameSize += pageSize;
368 
369  if ((triggerType & 0x800) != 0 && stopBit == 0 && pageCounter == 0) {
370  // This is the start of a new TimeFrame, so we need to take some actions:
371  // - push a new TimeFrame in the queue
372  // - append the last N HBFrames of the previous TimeFrame at the beginning of the current one
373  // - set the initial total size of the TimeFrame buffer
374  char* prevTFptr{nullptr};
375  size_t prevTFsize{0};
376 
377  if (mPrint) {
378  std::cout << "tfQueue.size(): " << tfQueue.size() << std::endl;
379  }
380  if (!tfQueue.empty()) {
381  TimeFrame& prevTF = tfQueue.back();
382  size_t nhbf = prevTF.hbframes.size();
383  if ((mOverlap > 0) && (nhbf >= mOverlap)) {
384  size_t hbfID = nhbf - mOverlap;
385  prevTFptr = prevTF.buf + prevTF.hbframes[hbfID].first;
386  for (size_t i = hbfID; i < nhbf; i++) {
387  prevTFsize += prevTF.hbframes[i].second;
388  }
389  }
390  }
391 
392  tfQueue.emplace();
393 
394  if (prevTFsize > 0) {
395  tfQueue.back().buf = (char*)malloc(prevTFsize);
396  if (tfQueue.back().buf == nullptr) {
397  std::cout << mFrameMax << " - failed to allocate TimeFrame buffer" << std::endl;
398  pc.services().get<ControlService>().endOfStream();
399  return;
400  }
401 
402  memcpy(tfQueue.back().buf, prevTFptr, prevTFsize);
403  }
404 
405  tfQueue.back().totalSize = prevTFsize;
406  }
407 
408  if (stopBit && tfQueue.size() > 0) {
409  // we reached the end of the current HBFrame, we need to append it to the TimeFrame
410 
411  if (mPrint) {
412  std::cout << "Appending HBF to TF #" << tfQueue.size() << std::endl;
413  printHBF(buf, frameSize);
414  }
415  if (!appendHBF(tfQueue.back(), buf, frameSize, true)) {
416  std::cout << mFrameMax << " - failed to append HBframe" << std::endl;
417  pc.services().get<ControlService>().endOfStream();
418  return;
419  }
420 
421  if (tfQueue.size() == 2) {
422  // we have two TimeFrames in the queue, we also append mOverlap HBFrames to the first one
423  if (tfQueue.back().hbframes.size() <= mOverlap) {
424  if (mPrint) {
425  std::cout << "Appending HBF to TF #1" << std::endl;
426  printHBF(buf, frameSize);
427  }
428  if (!appendHBF(tfQueue.front(), buf, frameSize, false)) {
429  std::cout << mFrameMax << " - failed to append HBframe" << std::endl;
430  pc.services().get<ControlService>().endOfStream();
431  return;
432  }
433  }
434  }
435 
436  // free the HBFrame buffer
437  free(buf);
438  buf = nullptr;
439  frameSize = 0;
440 
441  if (tfQueue.size() == 2 && tfQueue.back().hbframes.size() >= mOverlap) {
442  // we collected enough HBFrames after the last fully recorded TimeFrame, so we can send it
443  tfQueue.front().computePayloadSize();
444  if (mPrint) {
445  tfQueue.front().print();
446  sleep(1);
447  }
448  if (tfQueue.front().payloadSize > 0) {
449  if (mSaveTF && TFid < 100) {
450  char fname[500];
451  snprintf(fname, 499, "tf-%03d.raw", TFid);
452  FILE* fout = fopen(fname, "wb");
453  if (fout) {
454  fwrite(tfQueue.front().buf, tfQueue.front().totalSize, 1, fout);
455  fclose(fout);
456  }
457  }
458 
459  auto freefct = [](void* data, void* /*hint*/) { free(data); };
460  pc.outputs().adoptChunk(Output{"RDT", "RAWDATA"}, tfQueue.front().buf, tfQueue.front().totalSize, freefct, nullptr);
461  TFid += 1;
462  }
463  tfQueue.pop();
464  break;
465  }
466  }
467  }
468  }

471  void run(framework::ProcessingContext& pc)
472  {
473  if (mFullTF) {
474  sendTF(pc);
475  //pc.services().get<ControlService>().endOfStream();
476  return;
477  }

480  RDH rdh;
481  char* buf{nullptr};
482  size_t bufSize{0};

484  while (true) {
485 
486  // stop if the required number of frames has been reached
487  if (mFrameMax == 0) {
488  pc.services().get<ControlService>().endOfStream();
489  return;
490  }
491 
492  if (mPrint) {
493  printf("mFrameMax: %d\n", mFrameMax);
494  }
495  if (mFrameMax > 0) {
496  mFrameMax -= 1;
497  }
498

501  if (mInputFile.fail()) {
502  if (mPrint) {
503  std::cout << "end of file reached" << std::endl;
504  }
505  pc.services().get<ControlService>().endOfStream();
506  return; // probably reached eof
507  }
508 
509  // check that the RDH version is ok (only RDH versions from 4 to 6 are supported at the moment)
510  auto rdhVersion = o2::raw::RDHUtils::getVersion(rdh);
511  auto rdhHeaderSize = o2::raw::RDHUtils::getHeaderSize(rdh);
512  if (mPrint) {
513  std::cout << "header_version=" << (int)rdhVersion << std::endl;
514  }
515  if (rdhVersion < 4 || rdhVersion > 6 || rdhHeaderSize != 64) {
516  return;
517  }
518

520  auto frameSize = o2::raw::RDHUtils::getOffsetToNext(rdh);
521  if (mPrint) {
522  std::cout << "frameSize=" << frameSize << std::endl;
523  }
524 
525  // stop if the frame size is too small
526  if (frameSize < rdhHeaderSize) {
527  std::cout << mFrameMax << " - frameSize too small: " << frameSize << std::endl;
528  pc.services().get<ControlService>().endOfStream();
529  return;
530  }
531 
532  // allocate the output buffer
533  buf = (char*)realloc(buf, bufSize + frameSize);
534  if (buf == nullptr) {
535  std::cout << mFrameMax << " - failed to allocate buffer" << std::endl;
536  pc.services().get<ControlService>().endOfStream();
537  return;
538  }
539 
540  // copy the RDH into the output buffer
541  memcpy(buf + bufSize, &rdh, rdhHeaderSize);
542 
543  // read the frame payload into the output buffer
544  mInputFile.read(buf + bufSize + rdhHeaderSize, frameSize - rdhHeaderSize);
545 
546  // stop if data cannot be read completely
547  if (mInputFile.fail()) {
548  if (mPrint) {
549  std::cout << "end of file reached" << std::endl;
550  }
551  free(buf);
552  pc.services().get<ControlService>().endOfStream();
553  return; // probably reached eof
554  }
555 
556  // increment the total buffer size
557  bufSize += frameSize;
558 
559  auto stopBit = o2::raw::RDHUtils::getStop(rdh);
560 
561  // when requesting full HBframes, the output message is sent only when the stop RDH is reached
562  // otherwise we send one message for each CRU page
563  if ((stopBit != 0) || (mFullHBF == false)) {
564  // create the output message
565  auto freefct = [](void* data, void* /*hint*/) { free(data); };
566  pc.outputs().adoptChunk(Output{"RDT", "RAWDATA"}, buf, bufSize, freefct, nullptr);
567 
568  // stop the readout loop
569  break;
570  }
571  } // while (true)
572  }
