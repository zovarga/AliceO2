142 int GPUReconstruction::Init()
143 {
144  if (mMaster) {
145  throw std::runtime_error("Must not call init on slave!");
146  }
147  int retVal = InitPhaseBeforeDevice();
148  if (retVal) {
149  return retVal;
150  }
151  for (unsigned int i = 0; i < mSlaves.size(); i++) {
152  retVal = mSlaves[i]->InitPhaseBeforeDevice();
153  if (retVal) {
154  GPUError("Error initialization slave (before deviceinit)");
155  return retVal;
156  }
157  mNStreams = std::max(mNStreams, mSlaves[i]->mNStreams);
158  mHostMemorySize = std::max(mHostMemorySize, mSlaves[i]->mHostMemorySize);
159  mDeviceMemorySize = std::max(mDeviceMemorySize, mSlaves[i]->mDeviceMemorySize);
160  }
161  if (InitDevice()) {
162  return 1;
163  }
164  mHostMemoryPoolEnd = (char*)mHostMemoryBase + mHostMemorySize;
165  mDeviceMemoryPoolEnd = (char*)mDeviceMemoryBase + mDeviceMemorySize;
166  if (InitPhasePermanentMemory()) {
167  return 1;
168  }
169  for (unsigned int i = 0; i < mSlaves.size(); i++) {
170  mSlaves[i]->mDeviceMemoryBase = mDeviceMemoryPermanent;
171  mSlaves[i]->mHostMemoryBase = mHostMemoryPermanent;
172  mSlaves[i]->mDeviceMemorySize = mDeviceMemorySize - ((char*)mSlaves[i]->mDeviceMemoryBase - (char*)mDeviceMemoryBase);
173  mSlaves[i]->mHostMemorySize = mHostMemorySize - ((char*)mSlaves[i]->mHostMemoryBase - (char*)mHostMemoryBase);
174  mSlaves[i]->mHostMemoryPoolEnd = mHostMemoryPoolEnd;
175  mSlaves[i]->mDeviceMemoryPoolEnd = mDeviceMemoryPoolEnd;
176  if (mSlaves[i]->InitDevice()) {
177  GPUError("Error initialization slave (deviceinit)");
178  return 1;
179  }
180  if (mSlaves[i]->InitPhasePermanentMemory()) {
181  GPUError("Error initialization slave (permanent memory)");
182  return 1;
183  }
184  mDeviceMemoryPermanent = mSlaves[i]->mDeviceMemoryPermanent;
185  mHostMemoryPermanent = mSlaves[i]->mHostMemoryPermanent;
186  }
187  retVal = InitPhaseAfterDevice();
188  if (retVal) {
189  return retVal;
190  }
191  ClearAllocatedMemory();
192  for (unsigned int i = 0; i < mSlaves.size(); i++) {
193  mSlaves[i]->mDeviceMemoryPermanent = mDeviceMemoryPermanent;
194  mSlaves[i]->mHostMemoryPermanent = mHostMemoryPermanent;
195  retVal = mSlaves[i]->InitPhaseAfterDevice();
196  if (retVal) {
197  GPUError("Error initialization slave (after device init)");
198  return retVal;
199  }
200  mSlaves[i]->ClearAllocatedMemory();
201  }
202  return 0;
203 }

275  mProcessingSettings.fullMergerOnGPU = false;
276  }

537  throw std::bad_alloc();

541  }

545 void GPUReconstruction::AllocateRegisteredMemoryInternal(GPUMemoryResource* res, GPUOutputControl* control, GPUReconstruction* recPool)
546 {
547  if (mProcessingSettings.memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL && (control == nullptr || control->useInternal())) {
548  if (!(res->mType & GPUMemoryResource::MEMORY_EXTERNAL)) {
549  if (res->mPtrDevice && res->mReuse < 0) {
550  operator delete(res->mPtrDevice GPUCA_OPERATOR_NEW_ALIGNMENT);
551  }
552  res->mSize = std::max((size_t)res->SetPointers((void*)1) - 1, res->mOverrideSize);
553  if (res->mReuse >= 0) {
554  if (res->mSize > mMemoryResources[res->mReuse].mSize) {
555  GPUError("Invalid reuse, insufficient size: %lld < %lld", (long long int)mMemoryResources[res->mReuse].mSize, (long long int)res->mSize);
556  throw std::bad_alloc();
557  }
558  res->mPtrDevice = mMemoryResources[res->mReuse].mPtrDevice;
559  } else {
560  res->mPtrDevice = operator new(res->mSize + GPUCA_BUFFER_ALIGNMENT GPUCA_OPERATOR_NEW_ALIGNMENT);
561  }
562  res->mPtr = GPUProcessor::alignPointer<GPUCA_BUFFER_ALIGNMENT>(res->mPtrDevice);
563  res->SetPointers(res->mPtr);
564  if (mProcessingSettings.allocDebugLevel >= 2) {
565  std::cout << (res->mReuse >= 0 ? "Reused " : "Allocated ") << res->mName << ": " << res->mSize << "\n";
566  }
567  if (res->mType & GPUMemoryResource::MEMORY_STACK) {
568  mNonPersistentIndividualAllocations.emplace_back(res);
569  }
570  if ((size_t)res->mPtr % GPUCA_BUFFER_ALIGNMENT) {
571  GPUError("Got buffer with insufficient alignment");
572  throw std::bad_alloc();
573  }
574  }
575  } else {
576  if (res->mPtr != nullptr) {

579  }

581  res->mOverrideSize = GPUCA_BUFFER_ALIGNMENT;
582  }
583  if ((!IsGPU() || (res->mType & GPUMemoryResource::MEMORY_HOST) || mProcessingSettings.keepDisplayMemory) && !(res->mType & GPUMemoryResource::MEMORY_EXTERNAL)) { // keepAllMemory --> keepDisplayMemory
584  if (control && control->useExternal()) {
585  if (control->allocator) {
586  res->mSize = std::max((size_t)res->SetPointers((void*)1) - 1, res->mOverrideSize);
587  res->mPtr = control->allocator(CAMath::nextMultipleOf<GPUCA_BUFFER_ALIGNMENT>(res->mSize));
588  res->mSize = std::max<size_t>((char*)res->SetPointers(res->mPtr) - (char*)res->mPtr, res->mOverrideSize);
589  if (mProcessingSettings.allocDebugLevel >= 2) {
590  std::cout << "Allocated (from callback) " << res->mName << ": " << res->mSize << "\n";
591  }
592  } else {
593  void* dummy = nullptr;
594  res->mSize = AllocateRegisteredMemoryHelper(res, res->mPtr, control->ptrCurrent, control->ptrBase, control->size, &GPUMemoryResource::SetPointers, dummy, "host");
595  }
596  } else {
597  res->mSize = AllocateRegisteredMemoryHelper(res, res->mPtr, recPool->mHostMemoryPool, recPool->mHostMemoryBase, recPool->mHostMemorySize, &GPUMemoryResource::SetPointers, recPool->mHostMemoryPoolEnd, "host");
598  }
599  if ((size_t)res->mPtr % GPUCA_BUFFER_ALIGNMENT) {
600  GPUError("Got buffer with insufficient alignment");
601  throw std::bad_alloc();
602  }
603  }
604  if (IsGPU() && (res->mType & GPUMemoryResource::MEMORY_GPU)) {
605  if (res->mProcessor->mLinkedProcessor == nullptr) {
606  GPUError("Device Processor not set (%s)", res->mName);
607  throw std::bad_alloc();
608  }
609  size_t size = AllocateRegisteredMemoryHelper(res, res->mPtrDevice, recPool->mDeviceMemoryPool, recPool->mDeviceMemoryBase, recPool->mDeviceMemorySize, &GPUMemoryResource::SetDevicePointers, recPool->mDeviceMemoryPoolEnd, " gpu");
610 
611  if (!(res->mType & GPUMemoryResource::MEMORY_HOST) || (res->mType & GPUMemoryResource::MEMORY_EXTERNAL)) {
612  res->mSize = size;
613  } else if (size != res->mSize) {
614  GPUError("Inconsistent device memory allocation (%s: device %lu vs %lu)", res->mName, size, res->mSize);
615  throw std::bad_alloc();
616  }
617  if ((size_t)res->mPtrDevice % GPUCA_BUFFER_ALIGNMENT) {
618  GPUError("Got buffer with insufficient alignment");
619  throw std::bad_alloc();
620  }
621  }
622  UpdateMaxMemoryUsed();
623  }

900 void GPUReconstruction::RunPipelineWorker()
901 {
902  if (!mInitialized || !mProcessingSettings.doublePipeline || mMaster != nullptr || !mSlaves.size()) {
903  throw std::invalid_argument("Cannot start double pipeline mode");
904  }
905  if (mProcessingSettings.debugLevel >= 3) {
906  GPUInfo("Pipeline worker started");
907  }
908  bool terminate = false;
909  while (!terminate) {
910  {
911  std::unique_lock<std::mutex> lk(mPipelineContext->mutex);
912  mPipelineContext->cond.wait(lk, [this] { return this->mPipelineContext->queue.size() > 0; });
913  }
914  GPUReconstructionPipelineQueue* q;
915  {
916  std::lock_guard<std::mutex> lk(mPipelineContext->mutex);
917  q = mPipelineContext->queue.front();
918  mPipelineContext->queue.pop();
919  }
920  if (q->op == 1) {
921  terminate = 1;
922  } else {
923  q->retVal = q->chain->RunChain();
924  }
925  std::lock_guard<std::mutex> lk(q->m);
926  q->done = true;
927  q->c.notify_one();
928  }
929  if (mProcessingSettings.debugLevel >= 3) {
930  GPUInfo("Pipeline worker ended");
931  }

939 int GPUReconstruction::EnqueuePipeline(bool terminate)
940 {
941  ClearAllocatedMemory(true);
942  GPUReconstruction* rec = mMaster ? mMaster : this;
943  std::unique_ptr<GPUReconstructionPipelineQueue> qu(new GPUReconstructionPipelineQueue);

945  q->chain = terminate ? nullptr : mChains[0].get();
946  q->op = terminate ? 1 : 0;
947  std::unique_lock<std::mutex> lkdone(q->m);
948  {

950  if (rec->mPipelineContext->terminate) {

953  rec->mPipelineContext->queue.push(q);
954  rec->mPipelineContext->terminate = terminate;
955  rec->mPipelineContext->cond.notify_one();
956  }
957  q->c.wait(lkdone, [&q]() { return q->done; });
958  if (q->retVal) {
