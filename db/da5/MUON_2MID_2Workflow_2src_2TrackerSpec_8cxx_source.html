38 class TrackerDeviceDPL
39 {
40  public:
41  void init(o2::framework::InitContext& ic)
42  {
43 
44  if (!gGeoManager) {
45  o2::base::GeometryManager::loadGeometry();
46  }
47 
48  mTracker = std::make_unique<Tracker>(createTransformationFromManager(gGeoManager));
49 
50  if (!mTracker->init(true)) {
51  LOG(error) << "Initialization of MID tracker device failed";
52  }
53 
54  auto stop = [this]() {
55  LOG(info) << "Capacities: ROFRecords: " << mTracker->getTrackROFRecords().capacity() << " tracks: " << mTracker->getTracks().capacity() << " clusters: " << mTracker->getClusters().capacity();
56  double scaleFactor = 1.e6 / mNROFs;
57  LOG(info) << "Processing time / " << mNROFs << " ROFs: full: " << mTimer.count() * scaleFactor << " us tracking: " << mTimerAlgo.count() * scaleFactor << " us";
58  };
59  ic.services().get<of::CallbackService>().set(of::CallbackService::Id::Stop, stop);
60  }

62  void run(o2::framework::ProcessingContext& pc)
63  {
64  auto tStart = std::chrono::high_resolution_clock::now();
65 
66  auto msg = pc.inputs().get("mid_clusters");
67  gsl::span<const Cluster> clusters = of::DataRefUtils::as<const Cluster>(msg);
68 
69  auto msgROF = pc.inputs().get("mid_clusters_rof");
70  gsl::span<const ROFRecord> inROFRecords = of::DataRefUtils::as<const ROFRecord>(msgROF);
71 
72  auto tAlgoStart = std::chrono::high_resolution_clock::now();
73  mTracker->process(clusters, inROFRecords);
74  mTimerAlgo += std::chrono::high_resolution_clock::now() - tAlgoStart;
75 
76  pc.outputs().snapshot(of::Output{"MID", "TRACKS", 0, of::Lifetime::Timeframe}, mTracker->getTracks());
77  LOG(debug) << "Sent " << mTracker->getTracks().size() << " tracks.";
78  pc.outputs().snapshot(of::Output{"MID", "TRACKCLUSTERS", 0, of::Lifetime::Timeframe}, mTracker->getClusters());
79  LOG(debug) << "Sent " << mTracker->getClusters().size() << " track clusters.";
80 
81  pc.outputs().snapshot(of::Output{"MID", "TRACKROFS", 0, of::Lifetime::Timeframe}, mTracker->getTrackROFRecords());
82  LOG(debug) << "Sent " << mTracker->getTrackROFRecords().size() << " ROFs.";
83  pc.outputs().snapshot(of::Output{"MID", "TRCLUSROFS", 0, of::Lifetime::Timeframe}, mTracker->getClusterROFRecords());
84  LOG(debug) << "Sent " << mTracker->getClusterROFRecords().size() << " ROFs.";
85 
86  mTimer += std::chrono::high_resolution_clock::now() - tStart;
87  mNROFs += inROFRecords.size();
88  }
89 
90  private:
91  std::unique_ptr<Tracker> mTracker{nullptr};
92  std::chrono::duration<double> mTimer{0};
93  std::chrono::duration<double> mTimerAlgo{0};
94  unsigned int mNROFs{0};
95 };
