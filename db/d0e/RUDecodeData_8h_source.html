70 template <class Mapping>
71 int RUDecodeData::decodeROF(const Mapping& mp)
72 {
73  nChipsFired = 0;
74  lastChipChecked = 0;
75  int ntot = 0;
76  auto* chipData = &chipsData[0];
77  for (int icab = 0; icab < nCables; icab++) { // cableData is ordered in such a way to have chipIDs in increasing order
78  if (!cableData[icab].getSize()) {
79  continue;
80  }
81  auto cabHW = cableHWID[icab];
82  auto chIdGetter = [this, &mp, cabHW](int cid) {
83  //return mp.getGlobalChipID(cid, cabHW, *this->ruInfo);
84  auto chip = mp.getGlobalChipID(cid, cabHW, *this->ruInfo);
85  return chip;
86  };
87  int ret = 0;
88  while ((ret = AlpideCoder::decodeChip(*chipData, cableData[icab], chIdGetter)) || chipData->isErrorSet()) { // we register only chips with hits or errors flags set
89  setROFInfo(chipData, cableLinkPtr[icab]);
90 #ifdef ALPIDE_DECODING_STAT
91  fillChipStatistics(icab, chipData);
92 #endif
93  if (ret >= 0 && chipData->getChipID() < Mapping::getNChips()) { // make sure there was no error | upd: why? if there was a non fatal error, we use chip data
94  if (chipData->isErrorSet() && nChipsFired && chipData->getChipID() == chipsData[nChipsFired - 1].getChipID()) {
95  // we are still in the chip data whose decoding was aborted due to the error, reuse this chip data
96  LOGP(debug, "re-entry into the data of the chip {} after previously detector error", chipData->getChipID());
97  continue;
98  }
99  ntot += chipData->getData().size();
100  if (++nChipsFired < chipsData.size()) { // fetch next free chip
101  chipData = &chipsData[nChipsFired];
102  } else {
103  break; // last chip decoded
104  }
105  }
106  }
107  }
108 
109  return ntot;
110 }
