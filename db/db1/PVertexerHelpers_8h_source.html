99 struct TrackVF {
105  enum { kUsed,
106  kNoVtx = -1,
107  kDiscarded = kNoVtx - 1 };
108  float x;
109  float y;
110  float z;
111  float sig2YI;
112  float sig2ZI;
113  float sigYZI;
114  float tgP;
115  float tgL;
116  float cosAlp;
117  float sinAlp;
118 
119  TimeEst timeEst;
120  float wgh = 0.;
121  float wghHisto = 0.; // weight based on track errors, used for histogramming
122  int entry;
123  int32_t bin = -1; // seeds histo bin
124  GTrackID gid{};
125  int vtxID = kNoVtx;
126  //
127  bool canAssign() const { return wgh > 0. && vtxID == kNoVtx; }
128  bool canUse() const { return vtxID == kNoVtx; }
129  bool canUse(float zmin, float zmax) const
130  {
131  return canUse() && (z > zmin && z < zmax);
132  }
133  bool operator<(const TrackVF& trc) const { return z < trc.z; }
134 
135  float getZForXY(float vx, float vy) const
136  {
137  return z + tgL * (vx * cosAlp + vy * sinAlp - x);
138  }
139 
140  // weighted distance^2 to other track (accounting for own errors only)
141  float getDist2(const TrackVF& o) const
142  {
143  auto dt = timeEst.getTimeStamp() - o.timeEst.getTimeStamp();
144  auto dte2 = timeEst.getTimeStampError() * timeEst.getTimeStampError() + o.timeEst.getTimeStampError() * o.timeEst.getTimeStampError();
145  auto dtnorm2 = dt * dt / dte2;
146  auto dz = z - o.z;
147  return dtnorm2 + dz * dz * sig2ZI;
148  }
149 
150  // weighted distance^2 to other track (accounting for both track errors errors only)
151  float getDist2Sym(const TrackVF& o) const
152  {
153  auto dt = timeEst.getTimeStamp() - o.timeEst.getTimeStamp();
154  auto dz = z - o.z;
155  float dte2 = o.timeEst.getTimeStampError() * o.timeEst.getTimeStampError() + timeEst.getTimeStampError() * timeEst.getTimeStampError();
156  return dt / dte2 + dz * dz / (1. / sig2ZI + 1. / o.sig2ZI);
157  }
158 
159  float getResiduals(const PVertex& vtx, float& dy, float& dz) const
160  {
161  // get residuals (Y and Z DCA in track frame) and calculate chi2
162  float dx = vtx.getX() * cosAlp + vtx.getY() * sinAlp - x; // VX rotated to track frame - trackX
163  dy = y + tgP * dx - (-vtx.getX() * sinAlp + vtx.getY() * cosAlp);
164  dz = z + tgL * dx - vtx.getZ();
165  return (dy * dy * sig2YI + dz * dz * sig2ZI) + 2. * dy * dz * sigYZI;
166  }
167 
168  float getResiduals(const PVertex& vtx) const
169  {
170  // get residuals (Y and Z DCA in track frame) and calculate chi2
171  float dx = vtx.getX() * cosAlp + vtx.getY() * sinAlp - x; // VX rotated to track frame - trackX
172  auto dy = y + tgP * dx - (-vtx.getX() * sinAlp + vtx.getY() * cosAlp);
173  auto dz = z + tgL * dx - vtx.getZ();
174  return (dy * dy * sig2YI + dz * dz * sig2ZI) + 2. * dy * dz * sigYZI;
175  }
176 
177  float evalChi2ToVertex(const PVertex& vtx, bool useTime)
178  {
179  constexpr float NDOF2I = 1. / 2, NDOF3I = 1. / 3;
180  float chi2T = getResiduals(vtx); // track-vertex residuals and chi2
181  if (useTime) {
182  float dt = timeEst.getTimeStamp() - vtx.getTimeStamp().getTimeStamp();
183  chi2T += dt * dt / (timeEst.getTimeStampError() * timeEst.getTimeStampError());
184  chi2T *= NDOF3I;
185  } else {
186  chi2T *= NDOF2I;
187  }
188  return chi2T;
189  }
190 
191  TrackVF() = default;
192  TrackVF(const o2::track::TrackParCov& src, const TimeEst& t_est, int _entry, GTrackID _gid, float addHTErr2 = 0., float addHZErr2 = 0.)
193  : x(src.getX()), y(src.getY()), z(src.getZ()), tgL(src.getTgl()), tgP(src.getSnp() / std::sqrt(1. - src.getSnp()) * (1. + src.getSnp())), timeEst(t_est), entry(_entry), gid(_gid)
194  {
195  o2::math_utils::sincos(src.getAlpha(), sinAlp, cosAlp);
196  auto det = src.getSigmaY2() * src.getSigmaZ2() - src.getSigmaZY() * src.getSigmaZY();
197  auto detI = 1. / det;
198  sig2YI = src.getSigmaZ2() * detI;
199  sig2ZI = src.getSigmaY2() * detI;
200  sigYZI = -src.getSigmaZY() * detI;
201  wghHisto = 1. / ((src.getSigmaZ2() + addHZErr2) * (t_est.getTimeStampError() * t_est.getTimeStampError() + addHTErr2));
202  }
