100  void run(o2::framework::ProcessingContext& pc) final
101  {
102  // set the min range of TFs for first TF
103  if (mProcessedTFs == 0) {
104  mTFRange[0] = getCurrentTF(pc);
105  }
106 
107  for (int i = 0; i < mCRUs.size(); ++i) {
108  const DataRef ref = pc.inputs().getByPos(i);
109  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
110  const int cru = tpcCRUHeader->subSpecification - mLaneId * CRU::MaxCRU;
111  mIDCFactorization.setIDCs(pc.inputs().get<std::vector<float>>(ref), cru, mProcessedTFs); // aggregate IDCs
112  }
113  ++mProcessedTFs;
114 
115  if (!(mProcessedTFs % ((mIDCFactorization.getNTimeframes() + 5) / 5))) {
116  LOGP(info, "aggregated TFs: {}", mProcessedTFs);
117  }
118 
119  if (mProcessedTFs == mIDCFactorization.getNTimeframes()) {
120  mTFRange[1] = getCurrentTF(pc) + 1; // set the TF for last aggregated TF
121  mProcessedTFs = 0; // reset processed TFs for next aggregation interval
122  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
123  mIDCFactorization.factorizeIDCs(true); // calculate DeltaIDC, 0D-IDC, 1D-IDC
124  } else {
125  mIDCFactorization.factorizeIDCs(false); // calculate DeltaIDC, 0D-IDC, 1D-IDC
126  }
127 
128  if (mDebug) {
129  LOGP(info, "dumping aggregated and factorized IDCs and FT to file");
130  mIDCFactorization.dumpToFile(fmt::format("IDCFactorized_{:02}.root", getCurrentTF(pc)).data());
131  }
132 
133  // storing to CCDB
134  sendOutput(pc.outputs());
135  }
136  }

187  void sendOutput(DataAllocator& output)
188  {
189  if (mSendOutDebug) {
190  sendOutputDebug(output);
191  }
192 
193  if (mWriteToDB) {
194  const long timeStampStart = getFirstTF();
195  const long timeStampEnd = getLastTF();
196  mDBapi.storeAsTFileAny<o2::tpc::IDCZero>(&mIDCFactorization.getIDCZero(), "TPC/Calib/IDC/IDC0", mMetadata, timeStampStart, timeStampEnd);
197  mDBapi.storeAsTFileAny<o2::tpc::IDCOne>(&mIDCFactorization.getIDCOne(), "TPC/Calib/IDC/IDC1", mMetadata, timeStampStart, timeStampEnd);
198 
199  for (unsigned int iChunk = 0; iChunk < mIDCFactorization.getNChunks(); ++iChunk) {
200  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
201  // perform grouping of IDC Delta if necessary
202  mIDCStruct.mIDCs.setIDCs(std::move(mIDCFactorization).getIDCDeltaUncompressed(iChunk));
203  LOGP(info, "averaging and grouping DeltaIDCs for TFs {} - {} for CRUs {} to {} using {} threads", getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk), mCRUs.front(), mCRUs.back(), mIDCStruct.mIDCs.getNThreads());
204  mIDCStruct.mIDCs.processIDCs();
205  if (mDebug) {
206  mIDCStruct.mIDCs.dumpToFile(fmt::format("IDCDeltaAveraged_chunk{:02}_{:02}.root", iChunk, getFirstTFDeltaIDC(iChunk)).data());
207  }
208  }
209 
210  switch (mCompressionDeltaIDC) {
211  case IDCDeltaCompression::MEDIUM:
212  default: {
213  using compType = unsigned short;
214  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
215  auto idcDeltaMediumCompressed = IDCDeltaCompressionHelper<compType>::getCompressedIDCs(mIDCStruct.mIDCs.getIDCGroupData());
216  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<compType>>(&idcDeltaMediumCompressed, "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk));
217  } else {
218  auto idcDeltaMediumCompressed = mIDCFactorization.getIDCDeltaMediumCompressed(iChunk);
219  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<compType>>(&idcDeltaMediumCompressed, "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk));
220  }
221 
222  break;
223  }
224  case IDCDeltaCompression::HIGH: {
225  using compType = unsigned char;
226  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
227  auto idcDeltaMediumCompressed = IDCDeltaCompressionHelper<compType>::getCompressedIDCs(mIDCStruct.mIDCs.getIDCGroupData());
228  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<compType>>(&idcDeltaMediumCompressed, "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk));
229  } else {
230  auto idcDeltaHighCompressed = mIDCFactorization.getIDCDeltaHighCompressed(iChunk);
231  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<compType>>(&idcDeltaHighCompressed, "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk));
232  }
233  break;
234  }
235  case IDCDeltaCompression::NO:
236  if constexpr (std::is_same_v<Type, TPCFactorizeIDCSpecGroup>) {
237  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<float>>(&mIDCStruct.mIDCs.getIDCGroupData(), "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk));
238  } else {
239  mDBapi.storeAsTFileAny<o2::tpc::IDCDelta<float>>(&mIDCFactorization.getIDCDeltaUncompressed(iChunk), "TPC/Calib/IDC/IDCDELTA", mMetadata, getFirstTFDeltaIDC(iChunk), getLastTFDeltaIDC(iChunk));
240  }
241  break;
242  }
243  }
244  }
245  // reseting aggregated IDCs. This is done for safety, but if all data is received in the next aggregation interval it isnt necessary... remove it?
246  mIDCFactorization.reset();
247  }
