59 void EntropyDecoderSpec::run(ProcessingContext& pc)
60 {
61  auto cput = mTimer.CpuTime();
62  mTimer.Start(false);
63 
64  auto buff = pc.inputs().get<gsl::span<o2::ctf::BufferType>>("ctf");
65 
66  auto& rofs = pc.outputs().make<std::vector<o2::mch::ROFRecord>>(OutputRef{"rofs"});
67  auto& digits = pc.outputs().make<std::vector<o2::mch::Digit>>(OutputRef{"digits"});
68 
69  // since the buff is const, we cannot use EncodedBlocks::relocate directly, instead we wrap its data to another flat object
70  const auto ctfImage = o2::mch::CTF::getImage(buff.data());
71  mCTFCoder.decode(ctfImage, rofs, digits);
72 
73  mTimer.Stop();
74  LOG(info) << "Decoded " << digits.size() << " MCH digits in " << rofs.size() << " ROFRecords in " << mTimer.CpuTime() - cput << " s.";
75 }
