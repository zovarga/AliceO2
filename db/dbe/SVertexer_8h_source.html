133 template <typename V0CONT, typename V0REFCONT, typename CASCCONT, typename CASCREFCONT>
134 void SVertexer::extractSecondaryVertices(V0CONT& v0s, V0REFCONT& vtx2V0Refs, CASCCONT& cascades, CASCREFCONT& vtx2CascRefs)
135 {
136  v0s.clear();
137  vtx2V0Refs.clear();
138  vtx2V0Refs.resize(mPVertices.size());
139  cascades.clear();
140  vtx2CascRefs.clear();
141  vtx2CascRefs.resize(mPVertices.size());
142 
143  auto& tmpV0s = mV0sTmp[0];
144  auto& tmpCascs = mCascadesTmp[0];
145  int nv0 = tmpV0s.size(), nCasc = tmpCascs.size();
146  std::vector<int> v0SortID(nv0), v0NewInd(nv0), cascSortID(nCasc);
147  std::iota(v0SortID.begin(), v0SortID.end(), 0);
148  std::sort(v0SortID.begin(), v0SortID.end(), [&](int i, int j) { return tmpV0s[i].getVertexID() < tmpV0s[j].getVertexID(); });
149  std::iota(cascSortID.begin(), cascSortID.end(), 0);
150  std::sort(cascSortID.begin(), cascSortID.end(), [&](int i, int j) { return tmpCascs[i].getVertexID() < tmpCascs[j].getVertexID(); });
151 
152  // relate V0s to primary vertices
153  int pvID = -1, nForPV = 0;
154  for (int iv = 0; iv < nv0; iv++) {
155  const auto& v0 = tmpV0s[v0SortID[iv]];
156  if (pvID < v0.getVertexID()) {
157  if (pvID > -1) {
158  vtx2V0Refs[pvID].setEntries(nForPV);
159  }
160  pvID = v0.getVertexID();
161  vtx2V0Refs[pvID].setFirstEntry(v0s.size());
162  nForPV = 0;
163  }
164  v0NewInd[v0SortID[iv]] = v0s.size(); // memorise updated v0 id to fix its reference in the cascade
165  v0s.push_back(v0);
166  nForPV++;
167  }
168  if (pvID != -1) { // finalize
169  vtx2V0Refs[pvID].setEntries(nForPV);
170  // fill empty slots
171  int ent = v0s.size();
172  for (int ip = vtx2V0Refs.size(); ip--;) {
173  if (vtx2V0Refs[ip].getEntries()) {
174  ent = vtx2V0Refs[ip].getFirstEntry();
175  } else {
176  vtx2V0Refs[ip].setFirstEntry(ent);
177  }
178  }
179  }
180  // update V0s references in cascades
181  for (auto& casc : tmpCascs) {
182  casc.setV0ID(v0NewInd[casc.getV0ID()]);
183  }
184 
185  // relate Cascades to primary vertices
186  pvID = -1;
187  nForPV = 0;
188  for (int iv = 0; iv < nCasc; iv++) {
189  const auto& casc = tmpCascs[cascSortID[iv]];
190  if (pvID < casc.getVertexID()) {
191  if (pvID > -1) {
192  vtx2CascRefs[pvID].setEntries(nForPV);
193  }
194  pvID = casc.getVertexID();
195  vtx2CascRefs[pvID].setFirstEntry(cascades.size());
196  nForPV = 0;
197  }
198  cascades.push_back(casc);
199  nForPV++;
200  }
201  if (pvID != -1) { // finalize
202  vtx2CascRefs[pvID].setEntries(nForPV);
203  // fill empty slots
204  int ent = cascades.size();
205  for (int ip = vtx2CascRefs.size(); ip--;) {
206  if (vtx2CascRefs[ip].getEntries()) {
207  ent = vtx2CascRefs[ip].getFirstEntry();
208  } else {
209  vtx2CascRefs[ip].setFirstEntry(ent);
210  }
211  }
212  }
213 }
