32 void GlobalOffsetsContainer::fill(const gsl::span<const GlobalOffsetsInfoObject>& data)
33 {
34  // fill container
35  for (auto& entry : data) {
36  if (std::abs(entry.getT0AC()) < RANGE) {
37  updateFirstCreation(entry.getTimeStamp());
38  auto time = entry.getT0AC();
39  time += RANGE;
40  mHisto[(time)]++;
41  mEntries++;
42  }
43  }
44 }
45 void GlobalOffsetsContainer::merge(GlobalOffsetsContainer* prev)
46 {

49  mHisto[i] += prev->mHisto[i];

51  mEntries += prev->mEntries;

55 int GlobalOffsetsContainer::getMeanGaussianFitValue() const
56 {
57  if (!hasEnoughEntries()) {
58  return 0;
59  }
60  if (mEntries == 0) {
61  return 0;
62  }
63  float sum = 0;
64  for (int ic = 0; ic < NBINS; ic++) {
65  sum += float(ic - RANGE) * float(mHisto[ic]);
66  // LOG(info)<<" histo "<<ic<<" "<< float(mHisto[ic]);
67  }
68  std::array<double, 3> fitValues;
69  double fitres = fitGaus(NBINS, mHisto.data(), -RANGE, RANGE, fitValues);
70  if (fitres >= 0) {
71  LOG(info) << "Fit result " << fitres << " Mean = " << fitValues[1] << " Sigma = " << fitValues[2] << " size " << mEntries << " mean " << sum / mEntries;
72  return (static_cast<int>(std::round(fitValues[1])));
73  } else {
74  LOG(error) << "Fit failed with result = " << fitres;
75  return 0;
76  }
77 }
