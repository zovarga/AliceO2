61 void DigitReader::run(ProcessingContext& pc)
62 {
63  auto ent = mTree->GetReadEntry() + 1;
64  assert(ent < mTree->GetEntries()); // this should not happen
65 
66  o2::dataformats::IOMCTruthContainerView* plabels = nullptr;
67  if (mUseMC) {
68  mTree->SetBranchAddress(mDigtMCTruthBranchName.c_str(), &plabels);
69  }
70  mTree->GetEntry(ent);
71  LOG(info) << mDetName << "DigitReader pushes " << mDigROFRec.size() << " ROFRecords, "
72  << mDigits.size() << " digits at entry " << ent;
73 
74  // This is a very ugly way of providing DataDescription, which anyway does not need to contain detector name.
75  // To be fixed once the names-definition class is ready
76  pc.outputs().snapshot(Output{mOrigin, "DIGITSROF", 0, Lifetime::Timeframe}, mDigROFRec);
77  pc.outputs().snapshot(Output{mOrigin, "DIGITS", 0, Lifetime::Timeframe}, mDigits);
78  if (mUseCalib) {
79  pc.outputs().snapshot(Output{mOrigin, "GBTCALIB", 0, Lifetime::Timeframe}, mCalib);
80  }
81 
82  if (mUseMC) {
83  auto& sharedlabels = pc.outputs().make<o2::dataformats::ConstMCTruthContainer<o2::MCCompLabel>>(Output{mOrigin, "DIGITSMCTR", 0, Lifetime::Timeframe});
84  plabels->copyandflatten(sharedlabels);
85  delete plabels;
86  pc.outputs().snapshot(Output{mOrigin, "DIGITSMC2ROF", 0, Lifetime::Timeframe}, mDigMC2ROFs);
87  }
88 
89  if (mTree->GetReadEntry() + 1 >= mTree->GetEntries()) {
90  pc.services().get<ControlService>().endOfStream();
91  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
92  }
93 }

95 void DigitReader::connectTree(const std::string& filename)
96 {
97  mTree.reset(nullptr); // in case it was already loaded
98  mFile.reset(TFile::Open(filename.c_str()));
99  assert(mFile && !mFile->IsZombie());
100  mTree.reset((TTree*)mFile->Get(mDigTreeName.c_str()));
101  assert(mTree);
102 
103  mTree->SetBranchAddress(mDigROFBranchName.c_str(), &mDigROFRecPtr);
104  mTree->SetBranchAddress(mDigitBranchName.c_str(), &mDigitsPtr);
105  if (mUseCalib) {
106  if (!mTree->GetBranch(mCalibBranchName.c_str())) {
107  throw std::runtime_error("GBT calibration data requested but not found in the tree");
108  }
109  mTree->SetBranchAddress(mCalibBranchName.c_str(), &mCalibPtr);
110  }
111  if (mUseMC) {
112  if (!mTree->GetBranch(mDigtMC2ROFBranchName.c_str()) || !mTree->GetBranch(mDigtMCTruthBranchName.c_str())) {
113  throw std::runtime_error("MC data requested but not found in the tree");
114  }
115  mTree->SetBranchAddress(mDigtMC2ROFBranchName.c_str(), &mDigMC2ROFsPtr);
116  }
117  LOG(info) << "Loaded tree from " << filename << " with " << mTree->GetEntries() << " entries";
118 }

120 DataProcessorSpec getITSDigitReaderSpec(bool useMC, bool useCalib, std::string defname)
121 {
122  std::vector<OutputSpec> outputSpec;
123  outputSpec.emplace_back("ITS", "DIGITS", 0, Lifetime::Timeframe);
124  outputSpec.emplace_back("ITS", "DIGITSROF", 0, Lifetime::Timeframe);
125  if (useCalib) {
126  outputSpec.emplace_back("ITS", "GBTCALIB", 0, Lifetime::Timeframe);
127  }
128  if (useMC) {
129  outputSpec.emplace_back("ITS", "DIGITSMCTR", 0, Lifetime::Timeframe);
130  outputSpec.emplace_back("ITS", "DIGITSMC2ROF", 0, Lifetime::Timeframe);
131  }
132 
133  return DataProcessorSpec{
134  "its-digit-reader",
135  Inputs{},
136  outputSpec,
137  AlgorithmSpec{adaptFromTask<ITSDigitReader>(useMC, useCalib)},
138  Options{
139  {"its-digit-infile", VariantType::String, defname, {"Name of the input digit file"}},
140  {"input-dir", VariantType::String, "none", {"Input directory"}}}};
141 }
