29 LabeledDigit& LabeledDigit::operator+=(const LabeledDigit& other)
30 {
31  if (canAdd(other)) {
32  Int_t a1 = getAmplitudeADC();
33  Int_t a2 = other.getAmplitudeADC();
34  Double_t r = ((a1 + a2) != 0) ? 1.0 / (a1 + a2) : 0.0;
35  mDigit += other.getDigit();
36 
37  for (int j = 0; j < mLabels.size(); j++) {
38  mLabels.at(j).setAmplitudeFraction(mLabels.at(j).getAmplitudeFraction() * a1 * r);
39  }
40 
41  for (auto label : other.getLabels()) {
42  label.setAmplitudeFraction(label.getAmplitudeFraction() * a2 * r);
43  mLabels.push_back(label);
44  }
45  }
46  // Does nothing if the digits are in different towers or have incompatible times.
47  return *this;
48 }
