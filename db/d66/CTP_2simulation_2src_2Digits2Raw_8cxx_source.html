52 void Digits2Raw::processDigits(const std::string& fileDigitsName)
53 {
54  std::unique_ptr<TFile> digiFile(TFile::Open(fileDigitsName.c_str()));
55  if (!digiFile || digiFile->IsZombie()) {
56  LOG(fatal) << "Failed to open input digits file " << fileDigitsName;
57  return;
58  }
59  LOG(info) << "Processing digits to raw";
60  TTree* digiTree = (TTree*)digiFile->Get("o2sim");
61  if (!digiTree) {
62  LOG(fatal) << "Failed to get digits tree";
63  return;
64  }
65  std::vector<o2::ctp::CTPDigit> CTPDigits, *fCTPDigitsPtr = &CTPDigits;
66  if (digiTree->GetBranch("CTPDigits")) {
67  digiTree->SetBranchAddress("CTPDigits", &fCTPDigitsPtr);
68  } else {
69  LOG(fatal) << "Branch CTPDigits is missing";
70  return;
71  }
72  o2::InteractionRecord intRec = {0, 0};
73  // Get first orbit
74  uint32_t orbit0 = 0;
75  bool firstorbit = 1;
76  // Add all CTPdigits for given orbit
77  LOG(info) << "Number of entries: " << digiTree->GetEntries();
78  for (int ient = 0; ient < digiTree->GetEntries(); ient++) {
79  digiTree->GetEntry(ient);
80  int nbc = CTPDigits.size();
81  LOG(info) << "Entry " << ient << " : " << nbc << " BCs stored";
82  std::vector<gbtword80_t> hbfIR;
83  std::vector<gbtword80_t> hbfTC;
84  for (auto const& ctpdig : CTPDigits) {
85  LOG(debug) << ctpdig.intRecord.bc << " bc all orbit " << ctpdig.intRecord.orbit;
86  if ((orbit0 == ctpdig.intRecord.orbit) || firstorbit) {
87  if (firstorbit == true) {
88  firstorbit = false;
89  orbit0 = ctpdig.intRecord.orbit;
90  LOG(info) << "First orbit:" << orbit0;
91  }
92  LOG(debug) << ctpdig.intRecord.orbit << " orbit bc " << ctpdig.intRecord.bc;
93  gbtword80_t gbtdigIR;
94  gbtword80_t gbtdigTC;
95  digit2GBTdigit(gbtdigIR, gbtdigTC, ctpdig);
96  LOG(debug) << "ir:" << gbtdigIR;
97  LOG(debug) << "tr:" << gbtdigTC;
98  hbfIR.push_back(gbtdigIR);
99  hbfTC.push_back(gbtdigTC);
100  } else {
101  std::vector<char> buffer;
102  LOG(info) << "Packing orbit:" << orbit0;
103  intRec.orbit = orbit0;
104  LOG(info) << "hbfIR:" << hbfIR.size() << " hbfTC:" << hbfTC.size();
105  if (mZeroSuppressedIntRec == true) {
106  buffer = digits2HBTPayload(hbfIR, NIntRecPayload);
107  } else {
108  std::vector<gbtword80_t> hbfIRnonZS = addEmptyBC(hbfIR);
109  buffer = digits2HBTPayload(hbfIRnonZS, NIntRecPayload);
110  }
111  // add data for IR
112  LOG(debug) << "IR buffer size:" << buffer.size() << ":";
113  mWriter.addData(getFEEIDIR(), mCruID, GBTLinkIDIntRec, mEndPointID, intRec, buffer);
114  // add data for Trigger Class Record
115  buffer.clear();
116  buffer = digits2HBTPayload(hbfTC, NClassPayload);
117  LOG(debug) << "TC buffer size:" << buffer.size() << ":";
118  mWriter.addData(getFEEIDTC(), mCruID, GBTLinkIDClassRec, mEndPointID, intRec, buffer);
119  //
120  orbit0 = ctpdig.intRecord.orbit;
121  hbfIR.clear();
122  hbfTC.clear();
123  LOG(debug) << ctpdig.intRecord.orbit << " orbit bc " << ctpdig.intRecord.bc;
124  gbtword80_t gbtdigIR;
125  gbtword80_t gbtdigTC;
126  digit2GBTdigit(gbtdigIR, gbtdigTC, ctpdig);
127  LOG(debug) << "ir:" << gbtdigIR;
128  LOG(debug) << "tr:" << gbtdigTC;
129  hbfIR.push_back(gbtdigIR);
130  hbfTC.push_back(gbtdigTC);
131  }
132  }
133  // Last orbit in record
134  std::vector<char> buffer;
135  LOG(info) << "Packing orbit:" << orbit0;
136  intRec.orbit = orbit0;
137  if (mZeroSuppressedIntRec == true) {
138  buffer = digits2HBTPayload(hbfIR, NIntRecPayload);
139  } else {
140  std::vector<gbtword80_t> hbfIRnonZS = addEmptyBC(hbfIR);
141  buffer = digits2HBTPayload(hbfIRnonZS, NIntRecPayload);
142  }
143  // add data for IR
144  LOG(debug) << "IR buffer size:" << buffer.size() << ":";
145  mWriter.addData(getFEEIDIR(), mCruID, GBTLinkIDIntRec, mEndPointID, intRec, buffer);
146  // add data for Trigger Class Record
147  buffer.clear();
148  buffer = digits2HBTPayload(hbfTC, NClassPayload);
149  LOG(debug) << "TC buffer size:" << buffer.size() << ":";
150  mWriter.addData(getFEEIDTC(), mCruID, GBTLinkIDClassRec, mEndPointID, intRec, buffer);
151  //
152  //orbit0 = ctpdig.intRecord.orbit;
153  firstorbit = true;
154  hbfIR.clear();
155  hbfTC.clear();
156  }

174 std::vector<char> Digits2Raw::digits2HBTPayload(const gsl::span<gbtword80_t> digits, uint32_t Npld) const
175 {
176  std::vector<char> toAdd;
177  uint32_t size_gbt = 0;
178  gbtword80_t gbtword;
179  gbtword80_t gbtsend;
180  bool valid;
181  for (auto const& dig : digits) {
182  valid = makeGBTWord(dig, gbtword, size_gbt, Npld, gbtsend);
183  if (valid == true) {
184  for (uint32_t i = 0; i < NGBT; i += 8) {
185  uint32_t w = 0;
186  for (uint32_t j = 0; j < 8; j++) {
187  w += (1 << j) * gbtsend[i + j];
188  }
189  char c = w;
190  toAdd.push_back(c);
191  }
192  // Pad zeros up to 128 bits
193  uint32_t NZeros = (o2::raw::RDHUtils::GBTWord * 8 - NGBT) / 8;
194  for (uint32_t i = 0; i < NZeros; i++) {
195  char c = 0;
196  toAdd.push_back(c);
197  }
198  }
199  }
200  // add what is left: maybe never left anything - tbc
201  //LOG(info) << size_gbt << " size valid " << valid;
202  //LOG(info) << "gbtword:" << gbtword;
203  //LOG(info) << "gbtsend:" << gbtsend;
204  if (size_gbt > 0) {
205  LOG(debug) << "Adding left over.";
206  gbtword80_t gbtsend = gbtword;
207  for (uint32_t i = 0; i < NGBT; i += 8) {
208  uint32_t w = 0;
209  for (uint32_t j = 0; j < 8; j++) {
210  w += (1 << j) * gbtsend[i + j];
211  }
212  char c = w;
213  toAdd.push_back(c);
214  }
215  // Pad zeros up to 128 bits
216  uint32_t NZeros = (o2::raw::RDHUtils::GBTWord * 8 - NGBT) / 8;
217  for (uint32_t i = 0; i < NZeros; i++) {
218  char c = 0;
219  toAdd.push_back(c);
220  }
221  }
222  return std::move(toAdd);
223 }

267 std::vector<gbtword80_t> Digits2Raw::addEmptyBC(std::vector<gbtword80_t>& hbfIRZS)
268 {
269  std::vector<gbtword80_t> hbfIRnonZS;
270  if (hbfIRZS.size() == 0) {
271  LOG(error) << "Int record with zero size not expected here.";
272  return hbfIRnonZS;
273  }
274  uint32_t bcnonzero = 0;
275  if (hbfIRZS[0] != 0) {
276  gbtword80_t bs = 0;
277  hbfIRnonZS.push_back(bs);
278  }
279  for (auto const& item : hbfIRZS) {
280  uint32_t bcnonzeroNext = (item.to_ulong()) & 0xfff;
281  for (int i = (bcnonzero + 1); i < bcnonzeroNext; i++) {
282  gbtword80_t bs = i;
283  hbfIRnonZS.push_back(bs);
284  }
285  bcnonzero = bcnonzeroNext;
286  hbfIRnonZS.push_back(item);
287  }
288  for (int i = (bcnonzero + 1); i < 3564; i++) {
289  gbtword80_t bs = i;
290  hbfIRnonZS.push_back(bs);
291  }
292  return hbfIRnonZS;
