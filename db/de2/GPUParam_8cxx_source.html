123 void GPUParam::UpdateGRPSettings(const GPUSettingsGRP* g, const GPUSettingsProcessing* p)
124 {
125  if (g) {
126  par.assumeConstantBz = g->constBz;
127  par.toyMCEventsFlag = g->homemadeEvents;
128  par.continuousTracking = g->continuousMaxTimeBin != 0;
129  par.continuousMaxTimeBin = g->continuousMaxTimeBin == -1 ? GPUSettings::TPC_MAX_TF_TIME_BIN : g->continuousMaxTimeBin;
130  polynomialField.Reset();
131  if (par.assumeConstantBz) {
132  GPUTPCGMPolynomialFieldManager::GetPolynomialField(GPUTPCGMPolynomialFieldManager::kUniform, par.bzkG, polynomialField);
133  } else {
134  GPUTPCGMPolynomialFieldManager::GetPolynomialField(par.bzkG, polynomialField);
135  }
136  }
137  par.earlyTpcTransform = rec.tpc.forceEarlyTransform == -1 ? (!par.continuousTracking) : rec.tpc.forceEarlyTransform;
138  par.qptB5Scaler = CAMath::Abs(par.bzkG) > 0.1 ? CAMath::Abs(par.bzkG) / 5.006680f : 1.f;
139  if (p) {
140  par.debugLevel = p->debugLevel;
141  par.resetTimers = p->resetTimers;
142  if (p->automaticQPtThresholds) {
143  rec.maxTrackQPt = 1.f / GPUCA_MIN_TRACK_PTB5_DEFAULT / par.qptB5Scaler;
144  rec.tpc.rejectQPt = 1.f / GPUCA_MIN_TRACK_PTB5_REJECT / par.qptB5Scaler;
145  }
146  }
