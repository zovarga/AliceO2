25 void checkResults(const DCAFitter& fitter, const std::array<float, 3> xyz)
26 {
27  int nCand = fitter.getNCandidates();
28  BOOST_CHECK(nCand > 0);
29  for (int ic = 0; ic < nCand; ic++) {
30  const auto& vtx = fitter.getPCACandidate(ic);
31  float dx = vtx.x - xyz[0], dy = vtx.y - xyz[1], dz = vtx.z - xyz[2];
32  float dst = TMath::Sqrt(dx * dx + dy * dy + dz * dz);
33 
34  const auto &trc0 = fitter.getTrack0(ic), &trc1 = fitter.getTrack1(ic); // track parameters at V0
35  printf("Candidate %d: DCA:%+e Vtx: %+e %+e %+e [Diff to true: %+e %+e %+e -> %+e]\n",
36  ic, fitter.getChi2AtPCACandidate(ic), vtx.x, vtx.y, vtx.z, dx, dy, dz, dst);
37  printf("Track X-parameters at PCA: %+e %+e\n", trc0.getX(), trc1.getX());
38  trc0.print();
39  trc1.print();
40  BOOST_CHECK(dst < 5e-2);
41  }
42 }
