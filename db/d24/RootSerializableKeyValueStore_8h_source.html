43  SerializedInfo(SerializedInfo const& other)
44  {
45  // we do a deep copy
46  N = other.N;

150  // generic implementation for put relying on TClass
151  template <typename T>
152  void put_impl(std::string const& key, T const& value, GetState& state, ...)
153  {
154  // make sure we have a TClass for this
155  // if there is a TClass, we'll use ROOT serialization to encode into the buffer
156  auto ptr = new T(value);
157  auto cl = TClass::GetClass(typeid(value));
158  if (!cl) {
159  state = GetState::kNOTCLASS;
160  return;
161  }
162  char* bufferptr = nullptr;
163  TBufferFile buff(TBuffer::kWrite);
164  buff.WriteObjectAny(ptr, cl);
165  int N = buff.Length();
166  bufferptr = new char[N];
167  memcpy(bufferptr, buff.Buffer(), sizeof(char) * N);
168 
169  auto name = std::type_index(typeid(value)).name();
170  mStore.insert(std::pair<std::string, SerializedInfo>(key, SerializedInfo((void*)ptr, N, (char*)bufferptr, cl, name)));
171  }
172 
173  // implementation for put for trivial types
174  template <typename T, typename std::enable_if<std::is_trivial<T>::value, T>::type* = nullptr>
175  void put_impl(std::string const& key, T const& value, GetState& state, int)
176  {
177  // we forbid pointers
178  static_assert(!std::is_pointer<T>::value);
179  // serialization of trivial types is easy (not based on ROOT)
180  auto ptr = new T(value);
181  int N = sizeof(T);
182  auto bufferptr = new char[N];
183  memcpy(bufferptr, (char*)ptr, sizeof(char) * N);
184 
185  auto name = std::type_index(typeid(value)).name();
186  mStore.insert(std::pair<std::string, SerializedInfo>(key, SerializedInfo((void*)ptr, N, (char*)bufferptr, nullptr, name)));
187  }

189  // generic implementation for get relying on TClass
190  template <typename T>
191  const T* get_impl(std::string const& key, GetState& state, ...) const
192  {
193  state = GetState::kOK;
194  auto iter = mStore.find(key);
195  if (iter != mStore.end()) {
196  auto& info = const_cast<SerializedInfo&>(iter->second);
197  auto cl = TClass::GetClass(typeid(T));
198  if (!cl) {
199  state = GetState::kNOTCLASS;
200  return nullptr;
201  }
202  if (info.cl && strcmp(cl->GetName(), info.cl->GetName()) == 0) {
203  // if there is a (cached) object pointer ... we return it
204  if (info.objptr) {
205  return (T*)info.objptr;
206  }
207  // do this only once and cache instance into info.objptr
208  TBufferFile buff(TBuffer::kRead, info.N, info.bufferptr, false, nullptr);
209  buff.Reset();
210  auto instance = (T*)buff.ReadObjectAny(cl);
211  info.objptr = instance;
212  return (T*)info.objptr;
213  } else {
214  state = GetState::kWRONGTYPE;
215  return nullptr;
216  }
217  }
218  state = GetState::kNOSUCHKEY;
219  return nullptr;
220  }
221 
222  // implementation for standard POD types
223  template <typename T, typename std::enable_if<std::is_trivial<T>::value, T>::type* = nullptr>
224  const T* get_impl(std::string const& key, GetState& state, int) const
225  {
226  state = GetState::kOK;
227  auto iter = mStore.find(key);
228  if (iter != mStore.end()) {
229  auto& info = const_cast<SerializedInfo&>(iter->second);
230  if (strcmp(std::type_index(typeid(T)).name(), info.typeinfo_name.c_str()) == 0) {
231  // if there is a (cached) object pointer ... we return it
232  if (info.objptr) {
233  return (T*)info.objptr;
234  }
235  info.objptr = (T*)info.bufferptr;
236  return (T*)info.objptr;
237  } else {
238  state = GetState::kWRONGTYPE;
239  return nullptr;
240  }
241  }
242  state = GetState::kNOSUCHKEY;
243  return nullptr;
244  }
245 
246  // removes a previous entry
247  void remove_entry(std::string const& key)
248  {
249  auto iter = mStore.find(key);
250  if (iter != mStore.end()) {
251  mStore.erase(iter);
252  }
253  }
