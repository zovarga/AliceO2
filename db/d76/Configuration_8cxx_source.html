93 int CTPConfiguration::processConfigurationLine(std::string& line, int& level)
94 {
95  if (line.size() == 0) {
96  return 0;
97  }
98  if (line.at(0) == '#') {
99  return 0;
100  }
101  size_t first;
102  if ((first = line.find("PARTITION:")) != std::string::npos) {
103  mName = o2::utils::Str::trim_copy(line.erase(first, 10));
104  return 0;
105  }
106  if ((first = line.find("VERSION:")) != std::string::npos) {
107  mVersion = o2::utils::Str::trim_copy(line.erase(first, 8));
108  return 0;
109  }
110  if ((first = line.find("INPUTS:")) != std::string::npos) {
111  level = 1;
112  return 0;
113  }
114  if ((first = line.find("DESCRIPTORS:")) != std::string::npos) {
115  level = 3;
116  return 0;
117  }
118  if ((first = line.find("CLUSTERS:")) != std::string::npos) {
119  level = 4;
120  return 0;
121  }
122  if ((first = line.find("CLASSES:")) != std::string::npos) {
123  level = 7;
124  return 0;
125  }
127  std::vector<std::string> tokens = o2::utils::Str::tokenize(line, ' ');
128  size_t ntokens = tokens.size();
129  if (ntokens == 0) {
130  return 0;
131  }
132  switch (level) {
133  case 1:
135  {
136  if (ntokens != 4) {
137  LOG(fatal) << "INPUTS syntax error in, wrong number of items, expected 4:" << line;
138  return level;
139  }
140  CTPInput inp;
141  inp.name = tokens[0];
142  std::string detName = tokens[1];
143  inp.level = tokens[2];
144  o2::detectors::DetID det(detName.c_str());
145  isDetector(det);
146  inp.detID = det.getID();
147  //std::cout << "id:" << det.getID() << " ndets:" << det.getNDetectors() << std::endl;
148  try {
149  inp.inputMask = std::stoull(tokens[3], nullptr, 0);
150  } catch (...) {
151  LOG(fatal) << "INPUTS syntax error in mask:" << line;
152  return level;
153  }
154  mInputs.push_back(inp);
155  break;
156  }
157  case 3:
159  {
160  CTPDescriptor desc;
161  desc.name = tokens[0];
162  tokens.erase(tokens.begin());
163  for (auto& item : tokens) {
164  //CTPInput *inp = const_cast<CTPInput*> (isInputInConfig(item));
165  CTPInput* inp = isInputInConfig(item);
166  if (inp == nullptr) {
167  LOG(fatal) << "DESCRIPTOR: input not in INPUTS:" << item << " LINE:" << line;
168  } else {
169  desc.inputs.push_back(inp);
170  }
171  }
172  // Create inputs and mask
173  mDescriptors.push_back(desc);
174  break;
175  }
176  case 4:
178  {
179  CTPCluster cluster;
180  cluster.name = tokens[0];
181  tokens.erase(tokens.begin());
182  o2::detectors::DetID::mask_t mask;
183  for (auto& item : tokens) {
184  o2::detectors::DetID det(item.c_str());
185  isDetector(det);
186  mask |= det.getMask();
187  }
188  cluster.maskCluster = mask;
189  mClusters.push_back(cluster);
190  break;
191  }
192  case 7:
194  {
195  CTPClass cls;
196  if (ntokens != 4) {
197  LOG(fatal) << "CLASSES syntax error, wrong number of items, expected 4:" << line;
198  return level;
199  }
200  cls.name = tokens[0];
201  try {
202  cls.classMask = std::stoull(tokens[1]);
203  } catch (...) {
204  LOG(fatal) << "CLASSES syntax error in mask:" << line;
205  return level;
206  }
207  std::string token = tokens[2];
208  auto it = std::find_if(mDescriptors.begin(), mDescriptors.end(), [&token](const CTPDescriptor& obj) { return obj.name == token; });
209  if (it != mDescriptors.end()) {
210  cls.descriptor = &*it;
211  } else {
213  LOG(fatal) << "CLASSES syntax error, descriptor not found:" << token;
214  }
216  token = tokens[3];
217  auto it2 = std::find_if(mClusters.begin(), mClusters.end(), [&token](const CTPCluster& obj) { return obj.name == token; });
218  if (it2 != mClusters.end()) {
219  cls.cluster = &*it2;
220  } else {
222  LOG(fatal) << "CLASSES syntax error, cluster not found:" << token;
223  }
224  mCTPClasses.push_back(cls);
225  break;
226  }
227  default: {
228  LOG(fatal) << "CTP Config parser Unknown level:" << level;
229  }
230  }
231  return 0;
232 }
