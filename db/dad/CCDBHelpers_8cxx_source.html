59 CCDBHelpers::ParserResult CCDBHelpers::parseRemappings(char const* str)
60 {
61  std::unordered_map<std::string, std::string> remappings;
62  std::string currentUrl = "";
63 
64  enum ParsingStates {
65  IN_BEGIN,
66  IN_BEGIN_URL,
67  IN_BEGIN_TARGET,
68  IN_END_TARGET,
69  IN_END_URL
70  };
71  ParsingStates state = IN_BEGIN;
72 
73  while (true) {
74  switch (state) {
75  case IN_BEGIN: {
76  if (*str == 0) {
77  return {remappings, ""};
78  }
79  state = IN_BEGIN_URL;
80  }
81  case IN_BEGIN_URL: {
82  if (*str != '/' && (strncmp("http://", str, 6) != 0) && (strncmp("https://", str, 7) != 0)) {
83  return {remappings, "URL should start with either / or http:// / https://"};
84  }
85  state = IN_END_URL;
86  } break;
87  case IN_END_URL: {
88  char const* c = strchr(str, '=');
89  if (c == nullptr) {
90  return {remappings, "Expecting at least one target path, missing `='?"};
91  }
92  if ((c - str) == 0) {
93  return {remappings, "Empty url"};
94  }
95  currentUrl = std::string_view(str, c - str);
96  state = IN_BEGIN_TARGET;
97  str = c + 1;
98  } break;
99  case IN_BEGIN_TARGET: {
100  if (*str == 0) {
101  return {remappings, "Empty target"};
102  }
103  state = IN_END_TARGET;
104  } break;
105  case IN_END_TARGET: {
106  char const* c = strpbrk(str, ",;");
107  if (c == nullptr) {
108  if (remappings.count(str)) {
109  return {remappings, fmt::format("Path {} requested more than once.", str)};
110  }
111  remappings[std::string(str)] = currentUrl;
112  return {remappings, ""};
113  }
114  if ((c - str) == 0) {
115  return {remappings, "Empty target"};
116  }
117  auto key = std::string(str, c - str);
118  if (remappings.count(str)) {
119  return {remappings, fmt::format("Path {} requested more than once.", key)};
120  }
121  remappings[key] = currentUrl;
122  if (*c == ';') {
123  state = IN_BEGIN_URL;
124  } else {
125  state = IN_BEGIN_TARGET;
126  }
127  str = c + 1;
128  } break;
129  }
130  }
131 }

133 AlgorithmSpec CCDBHelpers::fetchFromCCDB()

135  return adaptStateful([](ConfigParamRegistry const& options, DeviceSpec const& spec) {
136  std::shared_ptr<CCDBFetcherHelper> helper = std::make_shared<CCDBFetcherHelper>();
137  auto backend = options.get<std::string>("condition-backend");
138  LOGP(info, "CCDB Backend at: {}", backend);
139  helper->api.init(options.get<std::string>("condition-backend"));
140  helper->createdNotBefore = std::to_string(options.get<int64_t>("condition-not-before"));
141  helper->createdNotAfter = std::to_string(options.get<int64_t>("condition-not-after"));
142  auto remapString = options.get<std::string>("condition-remap");
143  ParserResult result = CCDBHelpers::parseRemappings(remapString.c_str());
144  if (!result.error.empty()) {
145  throw runtime_error_f("Error while parsing remapping string %s", result.error.c_str());
146  }
147 
148  helper->remappings = result.remappings;
149 
150  for (auto &route : spec.outputs) {
151  if (route.matcher.lifetime != Lifetime::Condition) {
152  continue;
153  }
154  helper->routes.push_back(route);
155  LOGP(info, "The following route is a condition {}", route.matcher);
156  for (auto& metadata : route.matcher.metadata) {
157  if (metadata.type == VariantType::String) {
158  LOGP(info, "- {}: {}", metadata.name, metadata.defaultValue);
159  }
160  }
161  }
162 
163  auto getOrbitResetTime = [](o2::pmr::vector<char> const& v) -> Long64_t {
164  Int_t previousErrorLevel = gErrorIgnoreLevel;
165  gErrorIgnoreLevel = kFatal;
166  TMemFile memFile("name", const_cast<char*>(v.data()), v.size(), "READ");
167  gErrorIgnoreLevel = previousErrorLevel;
168  if (memFile.IsZombie()) {
169  throw runtime_error("CTP is Zombie");
170  }
171  TClass* tcl = TClass::GetClass(typeid(std::vector<Long64_t>));
172  void* result = ccdb::CcdbApi::extractFromTFile(memFile, tcl);
173  if (!result) {
174  throw runtime_error_f("Couldn't retrieve object corresponding to %s from TFile", tcl->GetName());
175  }
176  memFile.Close();
177  std::vector<Long64_t>* ctp = (std::vector<Long64_t>*)result;
178  return (*ctp)[0];
179  };

181  return adaptStateless([helper, &getOrbitResetTime](DataTakingContext& dtc, DataAllocator& allocator, TimingInfo& timingInfo) {

191  std::string etag;

193  if (url2uuid != helper->mapURL2UUID.end()) {
194  etag = url2uuid->second;
195  }
196  Output output{"CTP", "OrbitReset", 0, Lifetime::Condition};
197  auto&& v = allocator.makeVector<char>(output);
198  helper->api.loadFileToMemory(v, path, metadata, timingInfo.creation,
199  &headers, etag,
200  helper->createdNotAfter,
201  helper->createdNotBefore);
202 
203  if ((headers.count("Error") != 0) || (etag.empty() && v.empty())) {
204  LOGP(error, "Unable to find object {}/{}", path, timingInfo.creation);
205  //FIXME: I should send a dummy message.
206  return;
207  }
208  Long64_t newOrbitResetTime = orbitResetTime;
209  if (etag.empty()) {
210  helper->mapURL2UUID[path] = headers["ETag"]; // update uuid
211  newOrbitResetTime = getOrbitResetTime(v);
212  auto cacheId = allocator.adoptContainer(output, std::move(v), true, header::gSerializationMethodNone);

214  LOGP(debug, "Caching {} for {} (DPL id {})", path, headers["ETag"], cacheId.value);
215  } else if (v.size()) { // but should be overridden by fresh object
216  // somewhere here pruneFromCache should be called
217  helper->mapURL2UUID[path] = headers["ETag"]; // update uuid
218  newOrbitResetTime = getOrbitResetTime(v);
219  auto cacheId = allocator.adoptContainer(output, std::move(v), true, header::gSerializationMethodNone);
220  helper->mapURL2DPLCache[path] = cacheId;
221  LOGP(debug, "Caching {} for {} (DPL id {})", path, headers["ETag"], cacheId.value);
222  // one could modify the adoptContainer to take optional old cacheID to clean:
223  // mapURL2DPLCache[URL] = ctx.outputs().adoptContainer(output, std::move(outputBuffer), true, mapURL2DPLCache[URL]);

266  } else if (isPrefix(ccdbMetadataPrefix, meta.name)) {
