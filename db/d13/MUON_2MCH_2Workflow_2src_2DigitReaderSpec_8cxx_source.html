52  mReader = std::make_unique<RootTreeReader>("o2sim", filename.c_str(), -1,
53  RootTreeReader::PublishingMode::Single,
54  RootTreeReader::BranchDefinition<std::vector<Digit>>{
55  Output{header::gDataOriginMCH, mDescriptions[0], 0, Lifetime::Timeframe}, "MCHDigit"},
56  RootTreeReader::BranchDefinition<std::vector<ROFRecord>>{
57  Output{header::gDataOriginMCH, mDescriptions[1], 0, Lifetime::Timeframe}, "MCHROFRecords"},
58  RootTreeReader::BranchDefinition<dataformats::MCTruthContainer<MCCompLabel>>{
59  Output{header::gDataOriginMCH, mDescriptions[2], 0, Lifetime::Timeframe}, "MCHMCLabels"});
60  } else {
61  mReader = std::make_unique<RootTreeReader>("o2sim", filename.c_str(), -1,
62  RootTreeReader::PublishingMode::Single,
63  RootTreeReader::BranchDefinition<std::vector<Digit>>{
64  Output{header::gDataOriginMCH, mDescriptions[0], 0, Lifetime::Timeframe}, "MCHDigit"},
65  RootTreeReader::BranchDefinition<std::vector<ROFRecord>>{
66  Output{header::gDataOriginMCH, mDescriptions[1], 0, Lifetime::Timeframe}, "MCHROFRecords"});
67  }
68  }

83 framework::DataProcessorSpec getDigitReaderSpec(bool useMC, const char* specName)
84 {
85  std::vector<OutputSpec> outputs;
86  std::vector<header::DataDescription> descriptions;
87  std::stringstream ss;
88  ss << "A:" << header::gDataOriginMCH.as<std::string>() << "/DIGITS/0";
89  ss << ";B:" << header::gDataOriginMCH.as<std::string>() << "/DIGITROFS/0";
90  if (useMC) {
91  ss << ";C:" << header::gDataOriginMCH.as<std::string>() << "/DIGITLABELS/0";
92  }
93  auto matchers = select(ss.str().c_str());
94  for (auto& matcher : matchers) {
95  outputs.emplace_back(DataSpecUtils::asOutputSpec(matcher));
96  descriptions.emplace_back(DataSpecUtils::asConcreteDataDescription(matcher));
97  }
98 
99  return DataProcessorSpec{
100  specName,
101  Inputs{},
102  outputs,
103  AlgorithmSpec{adaptFromTask<DigitsReaderDeviceDPL>(useMC, descriptions)},
104  Options{{"mch-digit-infile", VariantType::String, "mchdigits.root", {"Name of the input file"}}}};
105 }
