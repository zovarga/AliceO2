209 struct IndexExclusive {
211  template <typename... Cs, typename Key, typename T1, typename... T>
212  static auto indexBuilder(const char* label, framework::pack<Cs...>, Key const&, std::tuple<T1, T...>&& tables)
213  {
214  static_assert(sizeof...(Cs) == sizeof...(T) + 1, "Number of columns does not coincide with number of supplied tables");
215  using tables_t = framework::pack<T...>;
216  using first_t = T1;
217  auto tail = tuple_tail(tables);
218  TableBuilder builder;
219  auto cursor = framework::FFL(builder.cursor<o2::soa::Table<Cs...>>());
220 
221  std::array<int32_t, sizeof...(T)> values;
222  iterator_tuple_t<std::decay_t<T>...> iterators = std::apply(
223  [](auto&&... x) {
224  return std::make_tuple(x.begin()...);
225  },
226  tail);
227 
228  using rest_it_t = decltype(pack_from_tuple(iterators));
229 
230  auto setValue = [&](auto& x, int idx) -> bool {
231  using type = std::decay_t<decltype(x)>;
232  constexpr auto position = framework::has_type_at_v<type>(rest_it_t{});
233 
234  if constexpr (std::is_same_v<framework::pack_element_t<position, framework::pack<std::decay_t<T>...>>, Key>) {
235  values[position] = idx;
236  return true;
237  } else {
238  lowerBound<Key>(idx, x);
239  if (x == soa::RowViewSentinel{static_cast<uint64_t>(x.mMaxRow)}) {
240  return false;
241  } else if (x.template getId<Key>() != idx) {
242  return false;
243  } else {
244  values[position] = x.globalIndex();
245  ++x;
246  return true;
247  }
248  }
249  };
250 
251  auto first = std::get<first_t>(tables);
252  for (auto& row : first) {
253  auto idx = -1;
254  if constexpr (std::is_same_v<first_t, Key>) {
255  idx = row.globalIndex();
256  } else {
257  idx = row.template getId<Key>();
258  }
259  if (std::apply(
260  [](auto&... x) {
261  return ((x == soa::RowViewSentinel{static_cast<uint64_t>(x.mMaxRow)}) && ...);
262  },
263  iterators)) {
264  break;
265  }
266 
267  auto result = std::apply(
268  [&](auto&... x) {
269  std::array<bool, sizeof...(T)> results{setValue(x, idx)...};
270  return (results[framework::has_type_at_v<std::decay_t<decltype(x)>>(rest_it_t{})] && ...);
271  },
272  iterators);
273 
274  if (result) {
275  cursor(0, row.globalIndex(), values[framework::has_type_at_v<T>(tables_t{})]...);
276  }
277  }
278  builder.setLabel(label);
279  return builder.finalize();
280  }

323  } else {

326  values[position] = -1;
327  return false;
328  } else if (x.template getId<Key>() != idx) {
329  values[position] = -1;
330  return false;
331  } else {
332  values[position] = x.globalIndex();

334  return true;

336  }
337  };
338 
339  auto first = std::get<first_t>(tables);
340  for (auto& row : first) {
341  auto idx = -1;
342  if constexpr (std::is_same_v<first_t, Key>) {
343  idx = row.globalIndex();
344  } else {
345  idx = row.template getId<Key>();
346  }
347  std::apply(
348  [&](auto&... x) {
349  (setValue(x, idx), ...);
350  },
351  iterators);
352 
353  cursor(0, row.globalIndex(), values[framework::has_type_at_v<T>(tables_t{})]...);
