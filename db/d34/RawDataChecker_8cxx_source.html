36 bool RawDataChecker::process(gsl::span<const ROBoard> localBoards, gsl::span<const ROFRecord> rofRecords, gsl::span<const ROFRecord> pageRecords)
37 {
39 
40  bool isOk = true;
41  mDebugMsg.clear();
42  std::unordered_map<uint16_t, std::vector<ROFRecord>> rofs;
43  for (auto& rof : rofRecords) {
44  auto& loc = localBoards[rof.firstEntry];
45  auto crateId = raw::getCrateId(loc.boardId);
46  auto linkId = crateparams::getGBTIdFromBoardInCrate(raw::getLocId(loc.boardId));
47  auto feeId = crateparams::makeGBTUniqueId(crateId, linkId);
48  rofs[feeId].emplace_back(rof);
49  }
50 
51  for (auto& item : rofs) {
52  isOk &= mCheckers[item.first].process(localBoards, item.second, pageRecords);
53  mDebugMsg += mCheckers[item.first].getDebugMessage();
54  }
55 
56  return isOk;
57 }
