146 void someDataProducerAlgorithm(ProcessingContext& ctx)
147 {
148  size_t index = ctx.services().get<ParallelContext>().index1D();
149  std::this_thread::sleep_for(std::chrono::seconds(1));
150  // Creates a new message of size collectionChunkSize which
151  // has "TPC" as data origin and "CLUSTERS" as data description.
152  auto& tpcClusters = ctx.outputs().make<FakeCluster>(
153  Output{"TPC", "CLUSTERS", static_cast<o2::header::DataHeader::SubSpecificationType>(index)}, collectionChunkSize);
154  int i = 0;
155 
156  for (auto& cluster : tpcClusters) {
157  assert(i < collectionChunkSize);
158  cluster.x = index;
159  cluster.y = i;
160  cluster.z = i;
161  cluster.q = rand() % 1000;
162  i++;
163  }
164 }

166 void someProcessingStageAlgorithm(ProcessingContext& ctx)
167 {
168  size_t index = ctx.services().get<ParallelContext>().index1D();
169 
170  const FakeCluster* inputDataTpc = reinterpret_cast<const FakeCluster*>(ctx.inputs().get("dataTPC").payload);
171 
172  auto& processedTpcClusters = ctx.outputs().make<FakeCluster>(
173  Output{"TPC", "CLUSTERS_P", static_cast<o2::header::DataHeader::SubSpecificationType>(index)},
174  collectionChunkSize);
175 
176  int i = 0;
177  for (auto& cluster : processedTpcClusters) {
178  assert(i < collectionChunkSize);
179  cluster.x = -inputDataTpc[i].x;
180  cluster.y = 2 * inputDataTpc[i].y;
181  cluster.z = inputDataTpc[i].z * inputDataTpc[i].q;
182  cluster.q = inputDataTpc[i].q;
183  i++;
184  }
185 };
