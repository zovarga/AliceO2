78  void fillData(bool finalFill = false)
79  {
80  for (auto it = mCollectedData.begin(); it != mCollectedData.end();) {
81  auto& dataStore = it->second;
82  if (!finalFill && (dataStore.received.to_ulong() != mSectorMask)) {
83  ++it;
84  continue;
85  }
86  const auto oldTF = mPresentTF;
87  mPresentTF = it->first;
88  const auto oldTForbit = mFirstTForbit;
89  mFirstTForbit = dataStore.firstOrbit;
90  mTFOrbits.emplace_back(mFirstTForbit);
91 
92  LOGP(info, "Filling tree entry {} for run {}, tf {}, orbit {}, final {}, sectors {}, (expected: {})", mNTFs, mRun, mPresentTF, mFirstTForbit, finalFill, dataStore.received.to_string(), std::bitset<Sector::MAXSECTOR>(mSectorMask).to_string());
93  auto& data = dataStore.data;
94  std::array<std::vector<T>*, Sector::MAXSECTOR> dataPtr;
95  for (size_t sector = 0; sector < data.size(); ++sector) {
96  auto& inData = data[sector];
97  dataPtr[sector] = &inData;
98 
99  if (!mDataBranches[sector]) {
100  mDataBranches[sector] = mTreeOut->Branch(fmt::format("{}_{}", BranchName.at(mBranchType), sector).data(), &inData);
101  } else {
102  mDataBranches[sector]->SetAddress(&dataPtr[sector]);
103  }
104  }
105 
106  mTreeOut->Fill();
107  for (size_t sector = 0; sector < data.size(); ++sector) {
108  mDataBranches[sector]->ResetAddress();
109  }
110 
111  mPresentTF = oldTF;
112  mFirstTForbit = oldTForbit;
113  ++mNTFs;
114 
115  it = mCollectedData.erase(it);
116  }
117  }

119  void run(ProcessingContext& pc) final
120  {
121  const std::string NAStr = "NA";
122 
123  const auto dh = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true));
124  auto oldRun = mRun;
125  auto oldTF = mPresentTF;
126  mRun = processing_helpers::getRunNumber(pc);
127  mPresentTF = dh->tfCounter;
128  mFirstTForbit = dh->firstTForbit;
129 
130  auto oldEnv = mEnvironmentID;
131  {
132  auto envN = pc.services().get<RawDeviceService>().device()->fConfig->GetProperty<std::string>("environment_id", NAStr);
133  if (envN != NAStr) {
134  mEnvironmentID = envN;
135  }
136  }
137  if ((oldRun != 0 && oldRun != mRun) || (!oldEnv.empty() && oldEnv != mEnvironmentID)) {
138  LOGP(warning, "RunNumber/Environment changed from {}/{} to {}/{}", oldRun, oldEnv, mRun, mEnvironmentID);
139  closeTreeAndFile();
140  }
141 
142  // check for the LHCPeriod
143  if (mLHCPeriod.empty()) {
144  auto LHCPeriodStr = pc.services().get<RawDeviceService>().device()->fConfig->GetProperty<std::string>("LHCPeriod", NAStr);
145  if (LHCPeriodStr != NAStr) {
146  mLHCPeriod = LHCPeriodStr;
147  } else {
148  const char* months[12] = {"JAN", "FEB", "MAR", "APR", "MAY", "JUN", "JUL", "AUG", "SEP", "OCT", "NOV", "DEC"};
149  time_t now = time(nullptr);
150  auto ltm = gmtime(&now);
151  mLHCPeriod = months[ltm->tm_mon];
152  LOG(warning) << "LHCPeriod is not available, using current month " << mLHCPeriod;
153  }
154  mLHCPeriod += fmt::format("_{}", DetID::getName(DetID::TPC));
155  }
156 
157  if (mWrite) {
158  if (!mCollectedData.size() || mCollectedData.find(mPresentTF) == mCollectedData.end()) {
159  prepareTreeAndFile(dh);
160  }
161 
162  fillData();
163  }
164 
165  for (auto const& inputRef : InputRecordWalker(pc.inputs())) {
166  auto& dataStore = mCollectedData[mPresentTF];
167  auto const* sectorHeader = DataRefUtils::getHeader<TPCSectorHeader*>(inputRef);
168  if (sectorHeader == nullptr) {
169  LOGP(error, "sector header missing on header stack for input on ", inputRef.spec->binding);
170  continue;
171  }
172 
173  const int sector = sectorHeader->sector();
174  // check if data was already received. Should never happen!
175  if (dataStore.received.test(sector)) {
176  LOGP(fatal, "data for sector {} and TF {} already received", sector, mPresentTF);
177  }
178  auto data = pc.inputs().get<std::vector<T>>(inputRef);
179  // LOGP(info, "Received data for sector {} with {} entries in TF {}, orbit {}", sector, data.size(), mPresentTF, mFirstTForbit);
180  dataStore.data[sector] = data;
181  dataStore.received.set(sector);
182  dataStore.firstOrbit = mFirstTForbit;
183  }
184  }

257  // if (sz < mChkSize) {

260  // ctfDir = mDirFallBack;

263  if (mCreateRunEnvDir && !mEnvironmentID.empty()) {
264  ctfDir += fmt::format("{}_{}/", mEnvironmentID, mRun);
265  if (!std::filesystem::exists(ctfDir)) {
266  if (!std::filesystem::create_directories(ctfDir)) {
267  throw std::runtime_error(fmt::format("Failed to create {} directory", ctfDir));

272  }
273  mCurrentFileName = o2::base::NameConf::getCTFFileName(mRun, dh->firstTForbit, dh->tfCounter, "tpc_krypton");
274  mCurrentFileNameFull = fmt::format("{}{}", ctfDir, mCurrentFileName);
275  mFileOut.reset(TFile::Open(fmt::format("{}{}", mCurrentFileNameFull, TMPFileEnding).c_str(), "recreate")); // to prevent premature external usage, use temporary name
276  mTreeOut = std::make_unique<TTree>(TreeName.at(mBranchType).data(), "O2 tree");
277  mInfoBranches.emplace_back(mTreeOut->Branch("run", &mRun));
278  mInfoBranches.emplace_back(mTreeOut->Branch("tfCounter", &mPresentTF));

284 
285  mNFiles++;

289 template <typename T>
290 void FileWriterDevice<T>::closeTreeAndFile()
291 {
292  if (!mTreeOut) {
293  return;
294  }
295 
296  LOGP(info, "closing file {}", mCurrentFileName);
297  try {
298  mFileOut->cd();
299  mTreeOut->SetEntries();
300  mTreeOut->Write();
301  mTreeOut.reset();
302  mFileOut->Close();
303  mFileOut.reset();
304  if (!TMPFileEnding.empty()) {
305  std::filesystem::rename(o2::utils::Str::concat_string(mCurrentFileNameFull, TMPFileEnding), mCurrentFileNameFull);
306  }
307  // write file metaFile data
308  if (mStoreMetaFile) {
309  mFileMetaData->fillFileData(mCurrentFileNameFull);
310  mFileMetaData->run = mRun;
311  mFileMetaData->LHCPeriod = mLHCPeriod;
312  mFileMetaData->type = "raw";
313  auto metaFileNameTmp = fmt::format("{}{}.tmp", mMetaFileDir, mCurrentFileName);
314  auto metaFileName = fmt::format("{}{}.done", mMetaFileDir, mCurrentFileName);
315  try {
316  std::ofstream metaFileOut(metaFileNameTmp);
317  metaFileOut << *mFileMetaData.get();
318  metaFileOut << "TFOrbits: ";
319  for (size_t i = 0; i < mTFOrbits.size(); i++) {
320  metaFileOut << fmt::format("{}{}", i ? ", " : "", mTFOrbits[i]);
321  }
322  metaFileOut << '\n';
323  metaFileOut.close();
324  std::filesystem::rename(metaFileNameTmp, metaFileName);
325  } catch (std::exception const& e) {
326  LOG(error) << "Failed to store meta data file " << metaFileName << ", reason: " << e.what();
327  }
328  mFileMetaData.reset();
329  }
330  } catch (std::exception const& e) {
331  LOG(error) << "Failed to finalize file " << mCurrentFileNameFull << ", reason: " << e.what();
332  }
333  mTFOrbits.clear();
334  mInfoBranches.clear();
335  std::fill(mDataBranches.begin(), mDataBranches.end(), nullptr);
336  mNTFs = 0;
337  // mAccSize = 0;
338  // removeLockFile();
339 }
