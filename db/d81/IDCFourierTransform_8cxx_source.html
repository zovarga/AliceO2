61 template <class Type>
62 void o2::tpc::IDCFourierTransform<Type>::calcFourierCoefficientsNaive(const o2::tpc::Side side)
63 {
64  // check if IDCs are present for current side
65  if (this->getNIDCs(side) == 0) {
66  LOGP(warning, "no 1D-IDCs found!");
67  mFourierCoefficients.reset(side);
68  return;
69  }
70 
71  const std::vector<unsigned int> offsetIndex = this->getLastIntervals(side);
72 
73  // see: https://en.wikipedia.org/wiki/Discrete_Fourier_transform#Definitiona
74  const bool add = mFourierCoefficients.getNCoefficientsPerTF() % 2;
75  const unsigned int lastCoeff = mFourierCoefficients.getNCoefficientsPerTF() / 2;
76 
77 #pragma omp parallel for num_threads(sNThreads)
78  for (unsigned int interval = 0; interval < this->getNIntervals(); ++interval) {
79  const std::vector<float>& idcOneExpanded{this->getExpandedIDCOne(side)}; // 1D-IDC values which will be used for the FFT
80  for (unsigned int coeff = 0; coeff < lastCoeff; ++coeff) {
81  const unsigned int indexDataReal = mFourierCoefficients.getIndex(interval, 2 * coeff); // index for storing real fourier coefficient
82  const unsigned int indexDataImag = indexDataReal + 1; // index for storing complex fourier coefficient
83  const float term0 = o2::constants::math::TwoPI * coeff / this->mRangeIDC;
84  for (unsigned int index = 0; index < this->mRangeIDC; ++index) {
85  const float term = term0 * index;
86  const float idc0 = idcOneExpanded[index + offsetIndex[interval]];
87  mFourierCoefficients(side, indexDataReal) += idc0 * std::cos(term);
88  mFourierCoefficients(side, indexDataImag) -= idc0 * std::sin(term);
89  }
90  }
91  if (add) {
92  const unsigned int indexDataReal = mFourierCoefficients.getIndex(interval, 2 * lastCoeff); // index for storing real fourier coefficient
93  const float term0 = o2::constants::math::TwoPI * lastCoeff / this->mRangeIDC;
94  for (unsigned int index = 0; index < this->mRangeIDC; ++index) {
95  const float term = term0 * index;
96  const float idc0 = idcOneExpanded[index + offsetIndex[interval]];
97  mFourierCoefficients(side, indexDataReal) += idc0 * std::cos(term);
98  }
99  }
100  }
101  // normalize coefficient to number of used points
102  normalizeCoefficients(side);
103 }

105 template <class Type>
106 void o2::tpc::IDCFourierTransform<Type>::calcFourierCoefficientsFFTW3(const o2::tpc::Side side)
107 {
108  // for FFTW and OMP see: https://stackoverflow.com/questions/15012054/fftw-plan-creation-using-openmp
109  // check if IDCs are present for current side
110  if (this->getNIDCs(side) == 0) {
111  LOGP(warning, "no 1D-IDCs found!");
112  mFourierCoefficients.reset(side);
113  return;
114  }
115 
116  const std::vector<unsigned int> offsetIndex = this->getLastIntervals(side);
117  const std::vector<float>& idcOneExpanded{this->getExpandedIDCOne(side)}; // 1D-IDC values which will be used for the FFT
118 
119  if constexpr (std::is_same_v<Type, IDCFourierTransformBaseAggregator>) {
120 #pragma omp parallel for num_threads(sNThreads)
121  for (unsigned int interval = 0; interval < this->getNIntervals(); ++interval) {
122  fftwLoop(idcOneExpanded, offsetIndex, interval, side, omp_get_thread_num());
123  }
124  } else {
125  fftwLoop(idcOneExpanded, offsetIndex, 0, side, 0);
126  }
127 
128  normalizeCoefficients(side);
129 }

131 template <class Type>
132 inline void o2::tpc::IDCFourierTransform<Type>::fftwLoop(const std::vector<float>& idcOneExpanded, const std::vector<unsigned int>& offsetIndex, const unsigned int interval, const o2::tpc::Side side, const unsigned int thread)
133 {
134  std::memcpy(mVal1DIDCs[thread], &idcOneExpanded[offsetIndex[interval]], this->mRangeIDC * sizeof(float)); // copy IDCs to avoid seg fault when using SIMD instructions
135  fftwf_execute_dft_r2c(mFFTWPlan, mVal1DIDCs[thread], mCoefficients[thread]); // perform ft
136  std::memcpy(&(*(mFourierCoefficients.mFourierCoefficients[side].begin() + mFourierCoefficients.getIndex(interval, 0))), mCoefficients[thread], mFourierCoefficients.getNCoefficientsPerTF() * sizeof(float)); // store coefficients
137 }

139 template <class Type>
140 std::vector<std::vector<float>> o2::tpc::IDCFourierTransform<Type>::inverseFourierTransformNaive(const o2::tpc::Side side) const
141 {
142  if (this->mRangeIDC % 2) {
143  LOGP(info, "number of specified fourier coefficients is {}, but should be an even number! FFTW3 method is used instead!", mFourierCoefficients.getNCoefficientsPerTF());
144  return inverseFourierTransformFFTW3(side);
145  }
146 
147  // vector containing for each intervall the inverse fourier IDCs
148  std::vector<std::vector<float>> inverse(this->getNIntervals());
149  const float factor = o2::constants::math::TwoPI / this->mRangeIDC;
150 
151  // loop over all the intervals. For each interval the coefficients are calculated
152  for (unsigned int interval = 0; interval < this->getNIntervals(); ++interval) {
153  inverse[interval].resize(this->mRangeIDC);
154  for (unsigned int index = 0; index < this->mRangeIDC; ++index) {
155  const float term0 = factor * index;
156  unsigned int coeffTmp = 0;
157  int fac = 1; // if input data is real (and it is) the coefficients are mirrored https://dsp.stackexchange.com/questions/4825/why-is-the-fft-mirrored
158  for (unsigned int coeff = 0; coeff < this->mRangeIDC; ++coeff) {
159  const unsigned int indexDataReal = mFourierCoefficients.getIndex(interval, 2 * coeffTmp); // index for storing real fourier coefficient
160  const unsigned int indexDataImag = indexDataReal + 1; // index for storing complex fourier coefficient
161  const float term = term0 * coeff;
162  inverse[interval][index] += mFourierCoefficients(side, indexDataReal) * std::cos(term) - fac * mFourierCoefficients(side, indexDataImag) * std::sin(term);
163  if (coeff < getNMaxCoefficients() - 1) {
164  ++coeffTmp;
165  } else {
166  --coeffTmp;
167  fac = -1;
168  };
169  }
170  }
171  }
172  return inverse;
173 }

175 template <class Type>
176 std::vector<std::vector<float>> o2::tpc::IDCFourierTransform<Type>::inverseFourierTransformFFTW3(const o2::tpc::Side side) const
177 {
178  // vector containing for each intervall the inverse fourier IDCs
179  std::vector<std::vector<float>> inverse(this->getNIntervals());
180 
181  // loop over all the intervals. For each interval the coefficients are calculated
182  // this loop and execution of FFTW is not optimized as it is used only for debugging
183  for (unsigned int interval = 0; interval < this->getNIntervals(); ++interval) {
184  inverse[interval].resize(this->mRangeIDC);
185  std::vector<std::array<float, 2>> val1DIDCs;
186  val1DIDCs.reserve(this->mRangeIDC);
187  for (unsigned int index = 0; index < getNMaxCoefficients(); ++index) {
188  const unsigned int indexDataReal = mFourierCoefficients.getIndex(interval, 2 * index); // index for storing real fourier coefficient
189  const unsigned int indexDataImag = indexDataReal + 1; // index for storing complex fourier coefficient
190  val1DIDCs.emplace_back(std::array<float, 2>{mFourierCoefficients(side, indexDataReal), mFourierCoefficients(side, indexDataImag)});
191  }
192  const fftwf_plan fftwPlan = fftwf_plan_dft_c2r_1d(this->mRangeIDC, reinterpret_cast<fftwf_complex*>(val1DIDCs.data()), inverse[interval].data(), FFTW_ESTIMATE);
193  fftwf_execute(fftwPlan);
194  fftwf_destroy_plan(fftwPlan);
195  }
196  return inverse;
197 }

207 template <class Type>
208 void o2::tpc::IDCFourierTransform<Type>::dumpToTree(const char* outFileName) const
209 {
210  o2::utils::TreeStreamRedirector pcstream(outFileName, "RECREATE");
211  pcstream.GetFile()->cd();
212  for (unsigned int iSide = 0; iSide < o2::tpc::SIDES; ++iSide) {
213  const o2::tpc::Side side = iSide == 0 ? Side::A : Side::C;
214  const std::vector<unsigned int> offsetIndex = this->getLastIntervals(side);
215  const auto idcOneExpanded = this->getExpandedIDCOne(side);
216  const auto inverseFourier = inverseFourierTransformNaive(side);
217  const auto inverseFourierFFTW3 = inverseFourierTransformFFTW3(side);
218 
219  for (unsigned int interval = 0; interval < this->getNIntervals(); ++interval) {
220  std::vector<float> oneDIDCInverse = inverseFourier[interval];
221  std::vector<float> oneDIDCInverseFFTW3 = inverseFourierFFTW3[interval];
222 
223  // get 1D-IDC values used for calculation of the fourier coefficients
224  std::vector<float> oneDIDC;
225  oneDIDC.reserve(this->mRangeIDC);
226  for (unsigned int index = 0; index < this->mRangeIDC; ++index) {
227  oneDIDC.emplace_back(idcOneExpanded[index + offsetIndex[interval]]);
228  }
229 
230  for (unsigned int coeff = 0; coeff < mFourierCoefficients.getNCoefficientsPerTF(); ++coeff) {
231  float coefficient = mFourierCoefficients(side, mFourierCoefficients.getIndex(interval, coeff));
232 
233  pcstream << "tree"
234  << "side=" << iSide
235  << "interval=" << interval
236  << "icoefficient=" << coeff // index of ith coefficient
237  << "coefficient=" << coefficient // value for ith coefficient
238  << "1DIDC.=" << oneDIDC
239  << "1DIDCiDFT.=" << oneDIDCInverse
240  << "1DIDCiDFTFFTW3.=" << oneDIDCInverseFFTW3
241  << "\n";
242  }
243  }
244  }

247 template <class Type>
248 void o2::tpc::IDCFourierTransform<Type>::printFFTWPlan() const
249 {
250  float* val1DIDCs = fftwf_alloc_real(this->mRangeIDC);
251  fftwf_complex* coefficients = fftwf_alloc_complex(getNMaxCoefficients());
252  fftwf_plan fftwPlan = fftwf_plan_dft_r2c_1d(this->mRangeIDC, val1DIDCs, coefficients, FFTW_ESTIMATE);
253  char* splan = fftwf_sprint_plan(fftwPlan);
254 
255  LOGP(info, "========= printing FFTW plan ========= \n {}", splan);
256  double add = 0;
257  double mul = 0;
258  double fusedMultAdd = 0;
259  fftwf_flops(fftwPlan, &add, &mul, &fusedMultAdd);
260  LOGP(info, "additions: {} multiplications: {} fused multiply-add: {} sum: {}", add, mul, fusedMultAdd, add + mul + fusedMultAdd);
261 
262  // free memory
263  free(splan);
264  fftwf_free(coefficients);
265  fftwf_free(val1DIDCs);
266  fftwf_destroy_plan(fftwPlan);
267 }
