31 // a label container with both contiguous and non-contiguous modes
32 // for label storage
33 // NOTE: the current specialization for contiguous storage is not
34 // yet on par with MCTruthContainer ... but eventually we might unify
35 // everything and keep only the more generic and configurable version
36 template <typename LabelType, bool isContiguousStorage = true>
37 class LabelContainer
38 {
39  public:
40  struct HeaderElementContinuous {
41  HeaderElementContinuous() = default; // for ROOT IO
42  HeaderElementContinuous(unsigned short s, unsigned int i) : size(s), index(i) {}
43  unsigned int index = 0; // index of first label in the actual label storage
44  unsigned short size = 0; // total number of labels
45  ClassDefNV(HeaderElementContinuous, 1);
46  };
47 
48  struct HeaderElementLinked {
49  HeaderElementLinked() = default; // for ROOT IO
50  HeaderElementLinked(unsigned int i, int l, unsigned short s) : index(i), lastindex(l), size(s) {}
51  unsigned int index = 0; // index of first label in the actual label storage
52  unsigned int lastindex = 0; // index of last label in the actual label storage
53  unsigned short size = 0; // total number of labels
54  ClassDefNV(HeaderElementLinked, 1);
55  };
56 
57  using HeaderElement = typename std::conditional<isContiguousStorage, HeaderElementContinuous, HeaderElementLinked>::type;
58  using StoredLabelType = typename std::conditional<isContiguousStorage, LabelType, std::pair<LabelType, int>>::type;

60  // internal functions allowing the iterator implementation to be completely generic
61  static unsigned int getNextIndex(unsigned int index, std::vector<LabelType> const& /*labels*/)
62  {
63  return index + 1;
64  }

66  {
67  return labels[index].second;

69  static LabelType& dereference(std::vector<LabelType>& v, int index)

76  }
77  static int lastIndex(HeaderElementContinuous const& h)
78  {
79  return h.index + h.size;
80  }
81  static int lastIndex(HeaderElementLinked const& h)
82  {
83  // -1 since this is indication of end of linked list
84  return -1;

87  // an iterator class to iterate over truthelements
88  class Iterator : public std::iterator<std::input_iterator_tag, LabelType>
89  {
90  private:
91  std::vector<StoredLabelType>& mLabelsRef; // reference to labels vector
92  int index; // startindex
93  public:
94  Iterator(std::vector<StoredLabelType>& v, int i) : mLabelsRef(v), index(i) {}
95  Iterator(const Iterator& it) : mLabelsRef(it.mLabelsRef), index(it.index) {}
96  Iterator& operator=(const Iterator& it)
97  {
98  mLabelsRef = it.mLabelsRef;
99  index = it.index;
100  return *this;
101  }
102 
103  // go to the next element as indicated by second entry of StoredLabelType
104  Iterator& operator++()
105  {
106  index = getNextIndex(index, mLabelsRef);
107  return *this;
108  }
109 
110  bool operator==(const Iterator& rhs) const { return index == rhs.index; }
111  bool operator!=(const Iterator& rhs) const { return index != rhs.index; }
112  LabelType& operator*() { return dereference(mLabelsRef, index); }
113  };

115  // a proxy class offering a (non-owning) container view on labels of a given data index
116  // container offers basic forward iterator functionality
117  class LabelView
118  {
119  private:
120  int dataindex;
121  std::vector<HeaderElement>& mHeaderArrayRef;
122  std::vector<StoredLabelType>& mLabelArrayRef;
123 
124  public:
125  constexpr LabelView(int i, std::vector<HeaderElement>& v1, std::vector<StoredLabelType>& v2) : dataindex(i), mHeaderArrayRef(v1), mLabelArrayRef(v2) {}
126 
127  // begin + end iterators to loop over the labels
128  Iterator begin()
129  {
130  return dataindex < mHeaderArrayRef.size() ? Iterator(mLabelArrayRef, mHeaderArrayRef[dataindex].index) : Iterator(mLabelArrayRef, 0);
131  }
132 
133  Iterator end()
134  {
135  return dataindex < mHeaderArrayRef.size() ? Iterator(mLabelArrayRef, lastIndex(mHeaderArrayRef[dataindex])) : Iterator(mLabelArrayRef, 0);
136  }
137 
138  // get number of labels
139  size_t size() const { return dataindex < mHeaderArrayRef.size() ? mHeaderArrayRef[dataindex].size : 0; }
140  };

142  static void addLabelImpl(int dataindex, std::vector<HeaderElementContinuous>& headerv, std::vector<LabelType>& labelv, LabelType const& label)
143  {
144  if (dataindex < headerv.size()) {
145  // look if we have something for this dataindex already
146  // must currently be the last one
147  if (dataindex != (headerv.size() - 1)) {
148  throw std::runtime_error("LabelContainer: unsupported code path");
149  }
150  } else {
151  assert(dataindex == headerv.size());
152  // add a new one
153  headerv.emplace_back(0, labelv.size());
154  }
155  auto& header = headerv[dataindex];
156  header.size++;
157  labelv.emplace_back(label);
158  }
159 
160  static void addLabelImpl(int dataindex, std::vector<HeaderElementLinked>& headerv, std::vector<std::pair<LabelType, int>>& labelv, LabelType const& label)
161  {
162  labelv.emplace_back(std::make_pair(label, -1));
163  // something exists for this dataindex already
164  if (dataindex < headerv.size()) {
165  auto& header = headerv[dataindex];
166  // increase size
167  header.size++;
168  const auto lastindex = labelv.size() - 1;
169  // fix link at previous last index
170  labelv[header.lastindex].second = lastindex;
171  // new last index
172  header.lastindex = lastindex;
173  } else {
174  // we support only appending in order?
175  assert(dataindex == headerv.size());
176  // add a new header element; pointing to last slot in mTruthArray
177  const auto lastpos = labelv.size() - 1;
178  headerv.emplace_back(lastpos, lastpos, 1);
179  }
180  }
