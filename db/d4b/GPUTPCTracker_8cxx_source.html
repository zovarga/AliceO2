107 void GPUTPCTracker::RegisterMemoryAllocation()
108 {
109  AllocateAndInitializeLate();
110  bool reuseCondition = !mRec->GetProcessingSettings().keepDisplayMemory && mRec->GetProcessingSettings().trackletSelectorInPipeline && ((mRec->GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCSliceTracking) || mRec->GetProcessingSettings().ompKernels == 1 || mRec->GetProcessingSettings().ompThreads == 1);
111  GPUMemoryReuse reLinks{reuseCondition, GPUMemoryReuse::REUSE_1TO1, GPUMemoryReuse::TrackerDataLinks, (unsigned short)(mISlice % mRec->GetProcessingSettings().nStreams)};
112  mMemoryResLinks = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataLinks, GPUMemoryResource::MEMORY_SCRATCH | GPUMemoryResource::MEMORY_STACK, "TPCSliceLinks", reLinks);
113  mMemoryResSliceScratch = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataScratch, GPUMemoryResource::MEMORY_SCRATCH | GPUMemoryResource::MEMORY_STACK | GPUMemoryResource::MEMORY_CUSTOM, "TPCSliceScratch");
114  mMemoryResSliceInput = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataInput, GPUMemoryResource::MEMORY_INPUT | GPUMemoryResource::MEMORY_STACK | GPUMemoryResource::MEMORY_CUSTOM, "TPCSliceInput");
115  GPUMemoryReuse reWeights{reuseCondition, GPUMemoryReuse::REUSE_1TO1, GPUMemoryReuse::TrackerDataWeights, (unsigned short)(mISlice % mRec->GetProcessingSettings().nStreams)};
116  mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataWeights, GPUMemoryResource::MEMORY_SCRATCH | GPUMemoryResource::MEMORY_STACK, "TPCSliceWeights", reWeights);
117  GPUMemoryReuse reScratch{reuseCondition, GPUMemoryReuse::REUSE_1TO1, GPUMemoryReuse::TrackerScratch, (unsigned short)(mISlice % mRec->GetProcessingSettings().nStreams)};
118  mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersScratch, GPUMemoryResource::MEMORY_SCRATCH | GPUMemoryResource::MEMORY_STACK, "TPCTrackerScratch", reScratch);
119  mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersScratchHost, GPUMemoryResource::MEMORY_SCRATCH_HOST, "TPCTrackerHost");
120  mMemoryResCommon = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersCommon, GPUMemoryResource::MEMORY_PERMANENT, "TPCTrackerCommon");
121  mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersDataRows, GPUMemoryResource::MEMORY_PERMANENT, "TPCSliceRows");
122 
123  unsigned int type = mRec->GetProcessingSettings().fullMergerOnGPU ? GPUMemoryResource::MEMORY_SCRATCH : GPUMemoryResource::MEMORY_OUTPUT;
124  if (mRec->GetProcessingSettings().memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) { // For individual scheme, we allocate tracklets separately, and change the type for the following allocations to custom
125  type |= GPUMemoryResource::MEMORY_CUSTOM;
126  mMemoryResTracklets = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersTracklets, type, "TPCTrackerTracklets");
127  }
128  mMemoryResOutput = mRec->RegisterMemoryAllocation(this, &GPUTPCTracker::SetPointersOutput, type, "TPCTrackerTracks");
129 }

199 GPUh() void GPUTPCTracker::WriteOutput()
200 {
201  mOutput->SetNTracks(0);
202  mOutput->SetNLocalTracks(0);
203  mOutput->SetNTrackClusters(0);
204 
205  if (mCommonMem->nTracks == 0) {
206  return;
207  }
208  if (mCommonMem->nTracks > GPUCA_MAX_SLICE_NTRACK) {
209  GPUError("Maximum number of tracks exceeded, cannot store");
210  return;
211  }
212 
213  int nStoredHits = 0;
214  int nStoredTracks = 0;
215  int nStoredLocalTracks = 0;
216 
217  GPUTPCTrack* out = mOutput->FirstTrack();
218 
219  trackSortData* trackOrder = new trackSortData[mCommonMem->nTracks];
220  for (unsigned int i = 0; i < mCommonMem->nTracks; i++) {
221  trackOrder[i].fTtrack = i;
222  trackOrder[i].fSortVal = mTracks[trackOrder[i].fTtrack].NHits() / 1000.f + mTracks[trackOrder[i].fTtrack].Param().GetZ() * 100.f + mTracks[trackOrder[i].fTtrack].Param().GetY();
223  }
224  std::sort(trackOrder, trackOrder + mCommonMem->nLocalTracks, SortComparison<trackSortData>); // TODO: Check why this sorting affects the merging efficiency!
225  std::sort(trackOrder + mCommonMem->nLocalTracks, trackOrder + mCommonMem->nTracks, SortComparison<trackSortData>);
226 
227  for (unsigned int iTrTmp = 0; iTrTmp < mCommonMem->nTracks; iTrTmp++) {
228  const int iTr = trackOrder[iTrTmp].fTtrack;
229  GPUTPCTrack& iTrack = mTracks[iTr];
230 
231  *out = iTrack;
232  int nClu = 0;
233  int iID = iTrack.FirstHitID();
234 
235  for (int ith = 0; ith < iTrack.NHits(); ith++) {
236  const GPUTPCHitId& ic = mTrackHits[iID + ith];
237  int iRow = ic.RowIndex();
238  int ih = ic.HitIndex();
239 
240  const GPUTPCRow& row = mData.Row(iRow);
241  int clusterIndex = mData.ClusterDataIndex(row, ih);
242 #ifdef GPUCA_ARRAY_BOUNDS_CHECKS
243  if (ih >= row.NHits() || ih < 0) {
244  GPUError("Array out of bounds access (Sector Row) (Hit %d / %d - NumC %d): Sector %d Row %d Index %d", ith, iTrack.NHits(), NHitsTotal(), mISlice, iRow, ih);
245  fflush(stdout);
246  continue;
247  }
248  if (clusterIndex >= NHitsTotal() || clusterIndex < 0) {
249  GPUError("Array out of bounds access (Cluster Data) (Hit %d / %d - NumC %d): Sector %d Row %d Hit %d, Clusterdata Index %d", ith, iTrack.NHits(), NHitsTotal(), mISlice, iRow, ih, clusterIndex);
250  fflush(stdout);
251  continue;
252  }
253 #endif
254 
255  float origX, origY, origZ;
256  unsigned char flags;
257  unsigned short amp;
258  int id;
259  if (Param().par.earlyTpcTransform) {
260  origX = mData.ClusterData()[clusterIndex].x;
261  origY = mData.ClusterData()[clusterIndex].y;
262  origZ = mData.ClusterData()[clusterIndex].z;
263  flags = mData.ClusterData()[clusterIndex].flags;
264  amp = mData.ClusterData()[clusterIndex].amp;
265  id = mData.ClusterData()[clusterIndex].id;
266  } else {
267  const ClusterNativeAccess& cls = *mConstantMem->ioPtrs.clustersNative;
268  id = clusterIndex + cls.clusterOffset[mISlice][0];
269  GPUTPCConvertImpl::convert(*mConstantMem, mISlice, iRow, cls.clustersLinear[id].getPad(), cls.clustersLinear[id].getTime(), origX, origY, origZ);
270  flags = cls.clustersLinear[id].getFlags();
271  amp = cls.clustersLinear[id].qTot;
272  }
273  GPUTPCSliceOutCluster c;
274  c.Set(id, iRow, flags, amp, origX, origY, origZ);
275 #ifdef GPUCA_TPC_RAW_PROPAGATE_PAD_ROW_TIME
276  c.mPad = mData.ClusterData()[clusterIndex].pad;
277  c.mTime = mData.ClusterData()[clusterIndex].time;
278 #endif
279  out->SetOutTrackCluster(nClu, c);
280  nClu++;
281  }
282 
283  nStoredTracks++;
284  if (iTr < mCommonMem->nLocalTracks) {
285  nStoredLocalTracks++;
286  }
287  nStoredHits += nClu;
288  out->SetNHits(nClu);
289  out = out->NextTrack();
290  }
291  delete[] trackOrder;
292 
293  mOutput->SetNTracks(nStoredTracks);
294  mOutput->SetNLocalTracks(nStoredLocalTracks);
295  mOutput->SetNTrackClusters(nStoredHits);
296  if (Param().par.debugLevel >= 3) {
297  GPUInfo("Slice %d, Output: Tracks %d, local tracks %d, hits %d", mISlice, nStoredTracks, nStoredLocalTracks, nStoredHits);
298  }
299 }
