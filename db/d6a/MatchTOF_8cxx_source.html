183 bool MatchTOF::prepareTPCData()
184 {
185  mNotPropagatedToTOF[trkType::UNCONS] = 0;
186  mNotPropagatedToTOF[trkType::CONSTR] = 0;
187 
188  auto creator = [this](auto& trk, GTrackID gid, float time0, float terr) {
189  const int nclustersMin = 0;
190  if constexpr (isTPCTrack<decltype(trk)>()) {
191  if (trk.getNClusters() < nclustersMin) {
192  return true;
193  }
194 
195  if (std::abs(trk.getQ2Pt()) > mMaxInvPt) {
196  return true;
197  }
198  this->addTPCSeed(trk, gid);
199  }
200  if constexpr (isTPCITSTrack<decltype(trk)>()) {
201  if (trk.getParamOut().getX() < o2::constants::geom::XTPCOuterRef - 1.) {
202  return true;
203  }
204  this->addITSTPCSeed(trk, gid);
205  }
206  if constexpr (isTRDTrack<decltype(trk)>()) {
207  this->addTRDSeed(trk, gid, time0, terr);
208  }
209  return true;
210  };
211  mRecoCont->createTracksVariadic(creator);
212 
213  for (int it = 0; it < trkType::SIZE; it++) {
214  mMatchedTracksIndex[it].resize(mTracksWork[it].size());
215  std::fill(mMatchedTracksIndex[it].begin(), mMatchedTracksIndex[it].end(), -1); // initializing all to -1
216  }
217 
218  if (mIsTPCused) {
219  LOG(debug) << "Number of UNCONSTRAINED tracks that failed to be propagated to TOF = " << mNotPropagatedToTOF[trkType::UNCONS];
220 
221  // sort tracks in each sector according to their time (increasing in time)
222  for (int sec = o2::constants::math::NSectors; sec--;) {
223  auto& indexCache = mTracksSectIndexCache[trkType::UNCONS][sec];
224  LOG(debug) << "Sorting sector" << sec << " | " << indexCache.size() << " tracks";
225  if (!indexCache.size()) {
226  continue;
227  }
228  std::sort(indexCache.begin(), indexCache.end(), [this](int a, int b) {
229  auto& trcA = mTracksWork[trkType::UNCONS][a].second;
230  auto& trcB = mTracksWork[trkType::UNCONS][b].second;
231  return ((trcA.getTimeStamp() - trcA.getTimeStampError()) - (trcB.getTimeStamp() - trcB.getTimeStampError()) < 0.);
232  });
233  } // loop over tracks of single sector
234  }
235  if (mIsITSTPCused || mIsTPCTRDused || mIsITSTPCTRDused) {
236  LOG(debug) << "Number of CONSTRAINED tracks that failed to be propagated to TOF = " << mNotPropagatedToTOF[trkType::CONSTR];
237 
238  // sort tracks in each sector according to their time (increasing in time)
239  for (int sec = o2::constants::math::NSectors; sec--;) {
240  auto& indexCache = mTracksSectIndexCache[trkType::CONSTR][sec];
241  LOG(debug) << "Sorting sector" << sec << " | " << indexCache.size() << " tracks";
242  if (!indexCache.size()) {
243  continue;
244  }
245  std::sort(indexCache.begin(), indexCache.end(), [this](int a, int b) {
246  auto& trcA = mTracksWork[trkType::CONSTR][a].second;
247  auto& trcB = mTracksWork[trkType::CONSTR][b].second;
248  return ((trcA.getTimeStamp() - mSigmaTimeCut * trcA.getTimeStampError()) - (trcB.getTimeStamp() - mSigmaTimeCut * trcB.getTimeStampError()) < 0.);
249  });
250  } // loop over tracks of single sector
251  }
252 
253  return true;
254 }

448 void MatchTOF::doMatching(int sec)
449 {
450  trkType type = trkType::CONSTR;
451 
453  auto& cacheTOF = mTOFClusSectIndexCache[sec]; // array of cached TOF cluster indices for this sector; reminder: they are ordered in time!
454  auto& cacheTrk = mTracksSectIndexCache[type][sec]; // array of cached tracks indices for this sector; reminder: they are ordered in time!
455  int nTracks = cacheTrk.size(), nTOFCls = cacheTOF.size();
456  LOG(debug) << "Matching sector " << sec << ": number of tracks: " << nTracks << ", number of TOF clusters: " << nTOFCls;
457  if (!nTracks || !nTOFCls) {
458  return;
459  }
460  int itof0 = 0; // starting index in TOF clusters for matching of the track
461  int detId[2][5]; // at maximum one track can fall in 2 strips during the propagation; the second dimention of the array is the TOF det index
462  float deltaPos[2][3]; // at maximum one track can fall in 2 strips during the propagation; the second dimention of the array is the residuals
463  o2::track::TrackLTIntegral trkLTInt[2]; // Here we store the integrated track length and time for the (max 2) matched strips
464  int nStepsInsideSameStrip[2] = {0, 0}; // number of propagation steps in the same strip (since we have maximum 2 strips, it has dimention = 2)
465  float deltaPosTemp[3];
466  std::array<float, 3> pos;
467  std::array<float, 3> posBeforeProp;
468  float posFloat[3];
469 
470  // prematching for TPC only tracks (identify BC candidate to correct z for TPC track accordingly to v_drift)
471 
472  LOG(debug) << "Trying to match %d tracks" << cacheTrk.size();
473  for (int itrk = 0; itrk < cacheTrk.size(); itrk++) {
474  for (int ii = 0; ii < 2; ii++) {
475  detId[ii][2] = -1; // before trying to match, we need to inizialize the detId corresponding to the strip number to -1; this is the array that we will use to save the det id of the maximum 2 strips matched
476  nStepsInsideSameStrip[ii] = 0;
477  }
478  int nStripsCrossedInPropagation = 0; // how many strips were hit during the propagation
479  auto& trackWork = mTracksWork[type][cacheTrk[itrk]];
480  auto& trefTrk = trackWork.first;
481  auto& intLT = mLTinfos[type][cacheTrk[itrk]];
482 
483  // Printf("intLT (before doing anything): length = %f, time (Pion) = %f", intLT.getL(), intLT.getTOF(o2::track::PID::Pion));
484  float minTrkTime = (trackWork.second.getTimeStamp() - mSigmaTimeCut * trackWork.second.getTimeStampError()) * 1.E6; // minimum time in ps
485  float maxTrkTime = (trackWork.second.getTimeStamp() + mSigmaTimeCut * trackWork.second.getTimeStampError()) * 1.E6; // maximum time in ps
486  int istep = 1; // number of steps
487  float step = 1.0; // step size in cm
488 
489  //uncomment for local debug
490  /*

908  // compare the times of the track and the TOF clusters - remember that they both are ordered in time!
909

911  continue;
912  }

935  // this because in case of multiple-hit cluster position is averaged on all pads contributing to the cluster (then error position matrix can be used for Chi2 if nedeed)

937  float posCorr[3] = {0, 0, 0};
938

940  posCorr[0] += Geo::XPAD, ndigits++;

944  }

947  }

950  }
951  if (trefTOF.isBitSet(Cluster::kDown)) {
952  posCorr[2] += Geo::ZPAD, ndigits++;
953  }

956  }
957  if (trefTOF.isBitSet(Cluster::kUpRight)) {
958  posCorr[0] -= Geo::XPAD, posCorr[2] -= Geo::ZPAD, ndigits++;
959  }
960  if (trefTOF.isBitSet(Cluster::kDownRight)) {
961  posCorr[0] -= Geo::XPAD, posCorr[2] += Geo::ZPAD, ndigits++;
962  }

971  int trackIdTOF;
972  int eventIdTOF;

992  }
