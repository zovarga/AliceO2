46  void run(o2::framework::ProcessingContext& pc) final
47  {
48  for (auto const& ref : InputRecordWalker(pc.inputs(), mFilter)) {
49  ++mReceivedCRUs;
50  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
51  const int cru = tpcCRUHeader->subSpecification >> 7;
52  const o2::tpc::CRU cruTmp(cru);
53  mOneDIDCAggregator.aggregate1DIDCs(cruTmp.side(), pc.inputs().get<std::vector<float>>(ref), 0, cruTmp.region());
54  }
55 
56  if (mReceivedCRUs != mCRUs.size()) {
57  return;
58  }
59 
60  mReceivedCRUs = 0;
61 
62  // perform fourier transform of 1D-IDCs
63  mIDCFourierTransform.setIDCs(std::move(mOneDIDCAggregator).getAggregated1DIDCs());
64  mIDCFourierTransform.calcFourierCoefficients();
65 
66  if (mDebug) {
67  LOGP(info, "dumping FT to file");
68  mIDCFourierTransform.dumpToFile(fmt::format("FourierEPN_{:02}.root", o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true))->tfCounter).data());
69  }
70 
71  sendOutput(pc.outputs());
72  }
73 
74  void endOfStream(o2::framework::EndOfStreamContext& ec) final
75  {
76  ec.services().get<ControlService>().readyToQuit(QuitRequest::Me);
