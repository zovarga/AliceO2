51 void RawWriter::run(ProcessingContext& pc)
52 {
53  auto digits = pc.inputs().get<std::vector<o2::tof::Digit>*>("tofdigits");
54  auto row = pc.inputs().get<std::vector<o2::tof::ReadoutWindowData>*>("readoutwin");
55  int nwindow = row->size();
56  LOG(debug) << "Encoding " << nwindow << " TOF readout windows";
57 
58  int cache = 1024 * 1024; // 1 MB
59  int verbosity = 0;
60 
61  o2::tof::raw::Encoder encoder;
62  encoder.setVerbose(verbosity);
63 
64  encoder.open(mOutFileName, mOutDirName, mFileFor);
65  encoder.alloc(cache);
66 
67  int nwindowperorbit = Geo::NWINDOW_IN_ORBIT;
68  int nwindowintimeframe = o2::raw::HBFUtils::Instance().getNOrbitsPerTF() * nwindowperorbit;
69  int nwindowFilled = nwindow;
70  if (nwindowFilled % nwindowintimeframe) {
71  nwindowFilled = (nwindowFilled / nwindowintimeframe + 1) * nwindowintimeframe;
72  }
73 
74  std::vector<o2::tof::Digit> emptyWindow;
75 
76  std::vector<o2::tof::Digit> digitRO;
77 
78  std::vector<std::vector<o2::tof::Digit>> digitWindows;
79 
80  for (int i = 0; i < nwindow; i += nwindowperorbit) { // encode 3 tof windows (1 orbit)
81  if (verbosity) {
82  printf("----------\nwindow = %d - %d\n----------\n", i, i + nwindowperorbit - 1);
83  }
84 
85  digitWindows.clear();
86 
87  // push all windows in the current orbit in the structure
88  for (int j = i; j < i + nwindowperorbit; j++) {
89  if (j < nwindow) {
90  digitRO.clear();
91  for (int id = 0; id < row->at(j).size(); id++) {
92  digitRO.push_back((*digits)[row->at(j).first() + id]);
93  }
94  digitWindows.push_back(digitRO);
95  } else {
96  digitWindows.push_back(emptyWindow);
97  }
98  }
99 
100  encoder.encode(digitWindows, i);
101  }
102 
103  // create configuration file for rawreader
104  encoder.getWriter().writeConfFile("TOF", "RAWDATA", o2::utils::Str::concat_string(mOutDirName, '/', "TOFraw.cfg"));
105  encoder.close();
106 }
