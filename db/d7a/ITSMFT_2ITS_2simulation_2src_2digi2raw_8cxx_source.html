49 {

56

61  add_option("input-file,i", bpo::value<std::string>()->default_value("itsdigits.root"), "input ITS digits file");

66  add_option("hbfutils-config,u", bpo::value<std::string>()->default_value(std::string(o2::base::NameConf::DIGITIZATIONCONFIGFILE)), "config file for HBFUtils (or none)");
67  add_option("configKeyValues", bpo::value<std::string>()->default_value(""), "comma-separated configKeyValues");
68 
69  opt_all.add(opt_general).add(opt_hidden);
70  bpo::store(bpo::command_line_parser(argc, argv).options(opt_all).positional(opt_pos).run(), vm);
71 
72  if (vm.count("help")) {
73  std::cout << opt_general << std::endl;
74  exit(0);
75  }
76 
77  bpo::notify(vm);
78  } catch (bpo::error& e) {
79  std::cerr << "ERROR: " << e.what() << std::endl
80  << std::endl;
81  std::cerr << opt_general << std::endl;
82  exit(1);
83  } catch (std::exception& e) {
84  std::cerr << e.what() << ", application will now exit" << std::endl;
85  exit(2);
86  }
87 
88  std::string confDig = vm["hbfutils-config"].as<std::string>();
89  if (!confDig.empty() && confDig != "none") {
90  o2::conf::ConfigurableParam::updateFromFile(confDig, "HBFUtils");
91  }
92  o2::conf::ConfigurableParam::updateFromString(vm["configKeyValues"].as<std::string>());
93  digi2raw(vm["input-file"].as<std::string>(),
94  vm["output-dir"].as<std::string>(),
95  vm["file-for"].as<std::string>(),
96  vm["verbosity"].as<uint32_t>(),
97  vm["rdh-version"].as<uint32_t>(),
98  vm["no-empty-hbf"].as<bool>());
99  LOG(info) << "HBFUtils settings used for conversion:";
100 
101  o2::raw::HBFUtils::Instance().print();
102 
103  return 0;
104 }

106 void digi2raw(std::string_view inpName, std::string_view outDir, std::string_view fileFor, int verbosity, uint32_t rdhV, bool noEmptyHBF, int superPageSizeInB)
107 {
108  TStopwatch swTot;
109  swTot.Start();
110  using ROFR = o2::itsmft::ROFRecord;
111  using ROFRVEC = std::vector<o2::itsmft::ROFRecord>;
112  const uint8_t ruSWMin = 0, ruSWMax = 0xff; // seq.ID of 1st and last RU (stave) to convert
113 
114  LOG(info) << "HBFUtil settings:";
115  o2::raw::HBFUtils::Instance().print();
116 
117  // if needed, create output directory
118  if (!std::filesystem::exists(outDir)) {
119  if (!std::filesystem::create_directories(outDir)) {
120  LOG(fatal) << "could not create output directory " << outDir;
121  } else {
122  LOG(info) << "created output directory " << outDir;
123  }
124  }
125 
127  std::string digTreeName{o2::base::NameConf::MCTTREENAME.data()};
128  TChain digTree(digTreeName.c_str());
129  digTree.AddFile(inpName.data());
130  digTree.SetBranchStatus("*MCTruth*", 0); // ignore MC info
131 
132  std::vector<o2::itsmft::Digit> digiVec, *digiVecP = &digiVec;
133  std::string digBranchName = o2::utils::Str::concat_string(MAP::getName(), "Digit");
134  if (!digTree.GetBranch(digBranchName.c_str())) {
135  LOG(fatal) << "Failed to find the branch " << digBranchName << " in the tree " << digTreeName;
136  }
137  digTree.SetBranchAddress(digBranchName.c_str(), &digiVecP);
138 
139  // ROF record entries in the digit tree
140  ROFRVEC rofRecVec, *rofRecVecP = &rofRecVec;
141  std::string rofRecName = o2::utils::Str::concat_string(MAP::getName(), "DigitROF");
142  if (!digTree.GetBranch(rofRecName.c_str())) {
143  LOG(fatal) << "Failed to find the branch " << rofRecName << " in the tree " << digTreeName;
144  }
145  digTree.SetBranchAddress(rofRecName.c_str(), &rofRecVecP);
147  std::string inputGRP = o2::base::NameConf::getGRPFileName();
148  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
149 
150  o2::itsmft::MC2RawEncoder<MAP> m2r;
151  m2r.setVerbosity(verbosity);
152  m2r.setContinuousReadout(grp->isDetContinuousReadOut(MAP::getDetID())); // must be set explicitly
153  m2r.setDefaultSinkName(o2::utils::Str::concat_string(MAP::getName(), ".raw"));
154  m2r.setMinMaxRUSW(ruSWMin, ruSWMax);
155  m2r.getWriter().setSuperPageSize(superPageSizeInB);
156  m2r.getWriter().useRDHVersion(rdhV);
157  m2r.getWriter().setDontFillEmptyHBF(noEmptyHBF);
158 
159  m2r.setVerbosity(verbosity);
160  setupLinks(m2r, outDir, MAP::getName(), fileFor);
161  //-------------------------------------------------------------------------------<<<<
162  int lastTreeID = -1;
163  long offs = 0, nEntProc = 0;
164  for (int i = 0; i < digTree.GetEntries(); i++) {
165  digTree.GetEntry(i);
166  for (const auto& rofRec : rofRecVec) {
167  int nDigROF = rofRec.getNEntries();
168  if (verbosity) {
169  LOG(info) << "Processing ROF:" << rofRec.getROFrame() << " with " << nDigROF << " entries";
170  rofRec.print();
171  }
172  if (!nDigROF) {
173  if (verbosity) {
174  LOG(info) << "Frame is empty"; // ??
175  }
176  continue;
177  }
178  nEntProc++;
179  auto dgs = nDigROF ? gsl::span<const o2::itsmft::Digit>(&digiVec[rofRec.getFirstEntry()], nDigROF) : gsl::span<const o2::itsmft::Digit>();
180  m2r.digits2raw(dgs, rofRec.getBCData());
181  }
182  } // loop over multiple ROFvectors (in case of chaining)
183 
184  m2r.getWriter().writeConfFile(MAP::getName(), "RAWDATA", o2::utils::Str::concat_string(outDir, '/', MAP::getName(), "raw.cfg"));
185  m2r.finalize(); // finish TF and flush data
186  //
187  swTot.Stop();
188  swTot.Print();
189 }

396 void setupLinks(o2::itsmft::MC2RawEncoder<MAP>& m2r, std::string_view outDir, std::string_view outPrefix, std::string_view fileFor)
397 {
398  //------------------------------------------------------------------------------->>>>
399  // just as an example, we require here that the staves are read via 3 links, with partitioning according to lnkXB below
400  // while OB staves use only 1 link.
401  // Note, that if the RU container is not defined, it will be created automatically
402  // during encoding.
403  // If the links of the container are not defined, a single link readout will be assigned
404 
405  const auto& mp = m2r.getMapping();
406  constexpr int MaxLinksPerRU = 3;
407  auto getNLinks = [](int lr) { return lr < 3 ? MaxLinksPerRU : MaxLinksPerRU - 1; };
408  int lnkAssign[3][MaxLinksPerRU] = {
409  // requested link cabling for IB, MB and OB
410  {3, 3, 3}, // IB
411  {14, 14, 0}, // MB
412  {14, 14, 0} // OB
413  };
414  std::unordered_map<int, int> cruMaxRU, cruNRU;
415  std::array<int, o2::itsmft::ChipMappingITS::getNRUs()> ruSWEntry{};
416  ruSWEntry.fill(-1);
417  int ntab = sizeof(itsHWMap) / sizeof(ITSRUMapping);
418  for (int ir = 0; ir < ntab; ir++) {
419  const auto& ru = itsHWMap[ir];
420  cruMaxRU[ru.cruHWID]++;
421  ruSWEntry[mp.getRUIDSW(ru.layer, ru.ruInLayer)] = ir;
422  }
423  std::string outFileLink;
424  for (int ruID = 0; ruID < o2::itsmft::ChipMappingITS::getNRUs(); ruID++) {
425  const auto& ruhw = itsHWMap[ruSWEntry[ruID]];
426  int nRUsOnCRU = cruMaxRU[ruhw.cruHWID], ruOnCRU = cruNRU[ruhw.cruHWID]++;
427  if (ruID < m2r.getRUSWMin() || ruID > m2r.getRUSWMax()) { // ignored RUs ?
428  continue;
429  }
430  m2r.getCreateRUDecode(ruID); // create RU container
431  int* lnkAs = lnkAssign[mp.getRUType(ruID)];
432  int accL = 0;
433  for (int il = 0; il < getNLinks(ruhw.layer); il++) { // create links
434  auto& ru = *m2r.getRUDecode(ruID);
435  ru.links[il] = m2r.addGBTLink();
436  uint32_t lanes = mp.getCablesOnRUType(mp.getRUType(ruID)); // lanes pattern of this RU
437  auto link = m2r.getGBTLink(ru.links[il]);
438  link->lanes = lanes & (((0x1 << lnkAs[il]) - 1) << (accL));
439 
440  link->idInCRU = ruOnCRU + il * nRUsOnCRU; // linkID
441  link->cruID = ruhw.cruHWID;
442  link->feeID = mp.RUSW2FEEId(ruID, il);
443  link->endPointID = link->idInCRU > 11 ? 1 : 0;
444  accL += lnkAs[il];
445  outFileLink = o2::utils::Str::concat_string(outDir, "/", outPrefix);
446  if (fileFor != "all") { // single file for all links
447  outFileLink += fmt::format("_{}", ruhw.flp);
448  if (fileFor != "flp") {
449  outFileLink += fmt::format("_cru{}_{}", ruhw.cruHWID, link->endPointID);
450  if (fileFor != "cru") {
451  outFileLink += fmt::format("_lnk{}_feeid{}", link->idInCRU, link->feeID);
452  if (fileFor != "link") {
453  throw std::runtime_error("invalid option provided for file grouping");
454  }
455  }
456  }
457  }
458  outFileLink += ".raw";
459  m2r.getWriter().registerLink(link->feeID, link->cruID, link->idInCRU, link->endPointID, outFileLink);
460  if (m2r.getVerbosity()) {
461  LOG(info) << "RU" << ruID << '(' << ruhw.ruInLayer << " on lr " << ruhw.layer << ") " << link->describe()
462  << " -> " << outFileLink;
463  }
464  }
465  }
