162  static double FuncDeltaOffset(double* x, double* params)
163  {
164  int i1 = int(x[0]) % 96;
165  int i2 = int(x[0]) / 96 ? (i1 + 48) : (i1 + 1);
166 
167  if (i1 < 0) {
168  return 0;
169  }
170  if (i2 >= Geo::NPADS) {
171  return 0;
172  }
173 
174  return (params[i1] - params[i2]);
175  }

183  {

258  void setStripFunction()
259  {
260  mStripOffsetFunction.clear();
261  for (int i = 0; i < 96; i++) {
262  int irow = i % 48;
263  int icolumn = i / 48;
264  if (i > 0) {
265  mStripOffsetFunction.append(Form("++ ("));
266  } else {
267  mStripOffsetFunction.append(Form("("));
268  }
269  bool kadd = false;
270  if (irow < 47) {
271  mStripOffsetFunction.append(Form("(x > %d && x < %d)", irow + icolumn * 48, irow + icolumn * 48 + 1));
272  kadd = true;
273  }
274  if (irow > 0) {
275  mStripOffsetFunction.append("-");
276  mStripOffsetFunction.append(Form("(x > %d && x < %d)", irow + icolumn * 48 - 1, irow + icolumn * 48));
277  kadd = true;
278  }
279  if (icolumn < 1) {
280  if (kadd) {
281  mStripOffsetFunction.append("+");
282  }
283  mStripOffsetFunction.append(Form("(x > %d && x < %d)", irow + 96, irow + 96 + 1));
284  } else {
285  mStripOffsetFunction.append("-");
286  mStripOffsetFunction.append(Form("(x > %d && x < %d)", irow + 96, irow + 96 + 1));
287  }
288  mStripOffsetFunction.append(Form(") "));
289  }
290  TLinearFitter localFitter2(1, mStripOffsetFunction.c_str()); // this is a workaround: when we define the TLinearFitter,
291  // the function is sort of added in some global namespace
292  // or somthing analogous; when then we use more than 1 thread,
293  // the threads conflict with each other in this operation, and
294  // it does not work. So we do the operation of adding the
295  // function to this global namespece in advance.
296  }
