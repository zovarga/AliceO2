64 void PrimaryVertexingSpec::init(InitContext& ic)
65 {
66  //-------- init geometry and field --------//
67  o2::base::GeometryManager::loadGeometry();
68  o2::base::Propagator::initFieldFromGRP();
69 
70  std::unique_ptr<o2::parameters::GRPObject> grp{o2::parameters::GRPObject::loadFrom()};
71  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
72  if (!grp->isDetContinuousReadOut(DetID::ITS)) {
73  mITSROFrameLengthMUS = alpParams.roFrameLengthTrig / 1.e3; // ITS ROFrame duration in \mus
74  } else {
75  mITSROFrameLengthMUS = alpParams.roFrameLengthInBC * o2::constants::lhc::LHCBunchSpacingNS * 1e-3; // ITS ROFrame duration in \mus
76  }
77  mVertexer.setITSROFrameLength(mITSROFrameLengthMUS);
78 
79  // this is a hack to provide Mat.LUT from the local file, in general will be provided by the framework from CCDB
80  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
81  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
82  if (o2::utils::Str::pathExists(matLUTFile)) {
83  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
84  o2::base::Propagator::Instance()->setMatLUT(lut);
85  LOG(info) << "Loaded material LUT from " << matLUTFile;
86  } else {
87  LOG(info) << "Material LUT " << matLUTFile << " file is absent, only TGeo can be used";
88  }
89  mTimer.Stop();
90  mTimer.Reset();
91  mVertexer.setValidateWithIR(mValidateWithIR);
92 
93  // set bunch filling. Eventually, this should come from CCDB
94  const auto* digctx = o2::steer::DigitizationContext::loadFromFile();
95  const auto& bcfill = digctx->getBunchFilling();
96  mVertexer.setBunchFilling(bcfill);
97  mVertexer.init();
98 }

100 void PrimaryVertexingSpec::run(ProcessingContext& pc)
101 {
102  double timeCPU0 = mTimer.CpuTime(), timeReal0 = mTimer.RealTime();
103  mTimer.Start(false);
104 
105  o2::globaltracking::RecoContainer recoData;
106  recoData.collectData(pc, *mDataRequest.get());
107  // select tracks of needed type, with minimal cuts, the real selected will be done in the vertexer
108 
109  std::vector<TrackWithTimeStamp> tracks;
110  std::vector<o2::MCCompLabel> tracksMCInfo;
111  std::vector<o2d::GlobalTrackID> gids;
112  auto maxTrackTimeError = PVertexerParams::Instance().maxTimeErrorMUS;
113  auto halfROFITS = 0.5 * mITSROFrameLengthMUS;
114  auto hw2ErrITS = 2.f / std::sqrt(12.f) * mITSROFrameLengthMUS; // conversion from half-width to error for ITS
115 
116  auto creator = [maxTrackTimeError, hw2ErrITS, halfROFITS, &tracks, &gids](auto& _tr, GTrackID _origID, float t0, float terr) {
117  if (!_origID.includesDet(DetID::ITS)) {
118  return true; // just in case this selection was not done on RecoContainer filling level
119  }
120  if constexpr (isITSTrack<decltype(_tr)>()) {
121  t0 += halfROFITS; // ITS time is supplied in \mus as beginning of ROF
122  terr *= hw2ErrITS; // error is supplied as a half-ROF duration, convert to \mus
123  }
124  // for all other tracks the time is in \mus with gaussian error
125  if constexpr (std::is_base_of_v<o2::track::TrackParCov, std::decay_t<decltype(_tr)>>) {
126  if (terr < maxTrackTimeError) {
127  tracks.emplace_back(TrackWithTimeStamp{_tr, {t0, terr}});
128  gids.emplace_back(_origID);
129  }
130  }
131 
132  return true;
133  };
134 
135  recoData.createTracksVariadic(creator); // create track sample considered for vertexing
136 
137  if (mUseMC) {
138  recoData.fillTrackMCLabels(gids, tracksMCInfo);
139  }
140  std::vector<PVertex> vertices;
141  std::vector<GIndex> vertexTrackIDs;
142  std::vector<V2TRef> v2tRefs;
143  std::vector<o2::MCEventLabel> lblVtx;
144 
145  // RS FIXME this will not have effect until the 1st orbit is propagated, until that will work only for TF starting at orbit 0
146  mVertexer.setStartIR({0, DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true))->firstTForbit});
147 
148  std::vector<o2::InteractionRecord> ft0Data;
149  if (mValidateWithIR) { // select BCs for validation
150  const o2::ft0::InteractionTag& ft0Params = o2::ft0::InteractionTag::Instance();
151  auto ft0all = recoData.getFT0RecPoints();
152  for (const auto& ftRP : ft0all) {
153  if (ft0Params.isSelected(ftRP)) {
154  ft0Data.push_back(ftRP.getInteractionRecord());
155  }
156  }
157  }
158 
159  mVertexer.process(tracks, gids, ft0Data, vertices, vertexTrackIDs, v2tRefs, tracksMCInfo, lblVtx);
160  pc.outputs().snapshot(Output{"GLO", "PVTX", 0, Lifetime::Timeframe}, vertices);
161  pc.outputs().snapshot(Output{"GLO", "PVTX_CONTIDREFS", 0, Lifetime::Timeframe}, v2tRefs);
162  pc.outputs().snapshot(Output{"GLO", "PVTX_CONTID", 0, Lifetime::Timeframe}, vertexTrackIDs);
163 
164  if (mUseMC) {
165  pc.outputs().snapshot(Output{"GLO", "PVTX_MCTR", 0, Lifetime::Timeframe}, lblVtx);
166  }
167 
168  mTimer.Stop();
169  LOG(info) << "Found " << vertices.size() << " primary vertices, timing: CPU: "
170  << mTimer.CpuTime() - timeCPU0 << " Real: " << mTimer.RealTime() - timeReal0 << " s";
171 }

173 void PrimaryVertexingSpec::endOfStream(EndOfStreamContext& ec)
174 {
175  mVertexer.end();
176  LOGF(info, "Primary vertexing total timing: Cpu: %.3e Real: %.3e s in %d slots",
177  mTimer.CpuTime(), mTimer.RealTime(), mTimer.Counter() - 1);
178 }

180 DataProcessorSpec getPrimaryVertexingSpec(GTrackID::mask_t src, bool validateWithFT0, bool useMC)
181 {
182  std::vector<OutputSpec> outputs;
183  auto dataRequest = std::make_shared<DataRequest>();
184 
185  dataRequest->requestTracks(src, useMC);
186  if (validateWithFT0 && src[GTrackID::FT0]) {
187  dataRequest->requestFT0RecPoints(false);
188  }
189 
190  outputs.emplace_back("GLO", "PVTX", 0, Lifetime::Timeframe);
191  outputs.emplace_back("GLO", "PVTX_CONTID", 0, Lifetime::Timeframe);
192  outputs.emplace_back("GLO", "PVTX_CONTIDREFS", 0, Lifetime::Timeframe);
193 
194  if (useMC) {
195  outputs.emplace_back("GLO", "PVTX_MCTR", 0, Lifetime::Timeframe);
196  }
197 
198  return DataProcessorSpec{
199  "primary-vertexing",
200  dataRequest->inputs,
201  outputs,
202  AlgorithmSpec{adaptFromTask<PrimaryVertexingSpec>(dataRequest, validateWithFT0, useMC)},
203  Options{{"material-lut-path", VariantType::String, "", {"Path of the material LUT file"}}}};
204 }
