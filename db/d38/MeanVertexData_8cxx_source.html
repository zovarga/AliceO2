42 void MeanVertexData::fill(const gsl::span<const PVertex> data)
43 {
44  // fill container
45 
46  LOG(info) << "input size = " << data.size();
47  for (int i = data.size(); i--;) {
48  // filling the histogram in binned mode
49  auto x = data[i].getX();
50  auto y = data[i].getY();
51  auto z = data[i].getZ();
52  LOG(debug) << "i = " << i << " --> x = " << x << ", y = " << y << ", z = " << z;
53  auto dx = x + rangeX;
54  uint32_t binx = dx < 0 ? 0xffffffff : (x + rangeX) * v2BinX;
55  auto dy = y + rangeY;
56  uint32_t biny = dy < 0 ? 0xffffffff : (y + rangeY) * v2BinY;
57  auto dz = z + rangeZ;
58  uint32_t binz = dz < 0 ? 0xffffffff : (z + rangeZ) * v2BinZ;
59  if (binx < nbinsX || biny < nbinsY || binz < nbinsZ) { // accounts also for z<-rangeZ
60  histoX[binx]++;
61  histoY[biny]++;
62  histoZ[binz]++;
63  entries++;
64  }
65  }
66 
67  for (int i = 0; i < histoX.size(); i++) {
68  LOG(debug) << "histoX, bin " << i << ": entries = " << histoX[i];
69  }
70  for (int i = 0; i < histoY.size(); i++) {
71  LOG(debug) << "histoY, bin " << i << ": entries = " << histoY[i];
72  }
73  for (int i = 0; i < histoZ.size(); i++) {
74  LOG(debug) << "histoZ, bin " << i << ": entries = " << histoZ[i];
75  }
76 }

79 void MeanVertexData::subtract(const MeanVertexData* prev)
80 {
81  // remove entries from prev
82 
83  assert(histoX.size() == prev->histoX.size());
84  assert(histoY.size() == prev->histoY.size());
85  assert(histoZ.size() == prev->histoZ.size());
86 
87  for (int i = histoX.size(); i--;) {
88  histoX[i] -= prev->histoX[i];
89  }
90  for (int i = histoY.size(); i--;) {
91  histoY[i] -= prev->histoY[i];
92  }
93  for (int i = histoZ.size(); i--;) {
94  histoZ[i] -= prev->histoZ[i];
95  }
96  entries -= prev->entries;
97 }
