63 void FullHistoryMerger::run(framework::ProcessingContext& ctx)
64 {
65  // we have to avoid mistaking the timer input with data inputs.
66  auto* timerHeader = ctx.inputs().get("timer-publish").header;
67 
68  for (const DataRef& ref : InputRecordWalker(ctx.inputs())) {
69  if (ref.header != timerHeader) {
70  updateCache(ref);
71  mUpdatesReceived++;
72  }
73  }
74 
75  if (ctx.inputs().isValid("timer-publish") && !mFirstObjectSerialized.first.empty()) {
76  mCyclesSinceReset++;
77  mergeCache();
78  publish(ctx.outputs());
79 
80  if (mConfig.mergedObjectTimespan.value == MergedObjectTimespan::LastDifference ||
81  mConfig.mergedObjectTimespan.value == MergedObjectTimespan::NCycles && mConfig.mergedObjectTimespan.param == mCyclesSinceReset) {
82  clear();
83  }
84  }
85 }

106 void FullHistoryMerger::updateCache(const DataRef& ref)
107 {
108  auto* dh = get<DataHeader*>(ref.header);
109  std::string sourceID = std::string(dh->dataOrigin.str) + "/" + std::string(dh->dataDescription.str) + "/" + std::to_string(dh->subSpecification);
110 
111  // I am not sure if ref.spec is always a concrete spec and not a broader matcher. Comparing it this way should be safer.
112  if (mFirstObjectSerialized.first.empty() || mFirstObjectSerialized.first == sourceID) {
113 
114  // We store one object in the serialized form, so we can take it as the first object to be merged (multiple times).
115  // If we kept it deserialized, we would need to require implementing a clone() method in MergeInterface.
116 
117  delete mFirstObjectSerialized.second.spec;
118  delete mFirstObjectSerialized.second.header;
119  delete mFirstObjectSerialized.second.payload;
120 
121  mFirstObjectSerialized.first = sourceID;
122  mFirstObjectSerialized.second.spec = new InputSpec(*ref.spec);
123  mFirstObjectSerialized.second.header = new char[Stack::headerStackSize(reinterpret_cast<std::byte const*>(dh))];
124  memcpy((void*)mFirstObjectSerialized.second.header, ref.header, dh->headerSize);
125  mFirstObjectSerialized.second.payload = new char[dh->payloadSize];
126  memcpy((void*)mFirstObjectSerialized.second.payload, ref.payload, dh->payloadSize);
127 
128  } else {
129  mCache[sourceID] = object_store_helpers::extractObjectFrom(ref);
130  }
131 }

133 void FullHistoryMerger::mergeCache()
134 {
135  LOG(debug) << "Merging " << mCache.size() + 1 << " objects.";
136 
137  mMergedObject = object_store_helpers::extractObjectFrom(mFirstObjectSerialized.second);
138  assert(!std::holds_alternative<std::monostate>(mMergedObject));
139  mObjectsMerged++;
140 
141  // We expect that all the objects use the same kind of interface
142  if (std::holds_alternative<TObjectPtr>(mMergedObject)) {
143 
144  auto target = std::get<TObjectPtr>(mMergedObject);
145  for (auto& [name, entry] : mCache) {
146  (void)name;
147  auto other = std::get<TObjectPtr>(entry);
148  algorithm::merge(target.get(), other.get());
149  mObjectsMerged++;
150  }
151 
152  } else if (std::holds_alternative<MergeInterfacePtr>(mMergedObject)) {
153  auto target = std::get<MergeInterfacePtr>(mMergedObject);
154  for (auto& [name, entry] : mCache) {
155  (void)name;
156  auto other = std::get<MergeInterfacePtr>(entry);
157  target->merge(other.get());
158  mObjectsMerged++;
159  }
160  }
161 }

163 void FullHistoryMerger::publish(framework::DataAllocator& allocator)
164 {
165  // todo see if std::visit is faster here
166  if (std::holds_alternative<std::monostate>(mMergedObject)) {
167  LOG(info) << "No objects received since start or reset, nothing to publish";
168  } else if (std::holds_alternative<MergeInterfacePtr>(mMergedObject)) {
169  allocator.snapshot(framework::OutputRef{MergerBuilder::mergerOutputBinding(), mSubSpec},
170  *std::get<MergeInterfacePtr>(mMergedObject));
171  LOG(info) << "Published the merged object containing " << mCache.size() + 1 << " incomplete objects. "
172  << mUpdatesReceived << " updates were received during the last cycle.";
173  } else if (std::holds_alternative<TObjectPtr>(mMergedObject)) {
174  allocator.snapshot(framework::OutputRef{MergerBuilder::mergerOutputBinding(), mSubSpec},
175  *std::get<TObjectPtr>(mMergedObject));
176  LOG(info) << "Published the merged object containing " << mCache.size() + 1 << " incomplete objects. "
177  << mUpdatesReceived << " updates were received during the last cycle.";
178  } else {
179  throw std::runtime_error("mMergedObject' variant has no value.");
180  }
181 
182  mTotalObjectsMerged += mObjectsMerged;
183  mTotalUpdatesReceived += mUpdatesReceived;
184  mCollector->send({mTotalObjectsMerged, "total_objects_merged"}, monitoring::DerivedMetricMode::RATE);
185  mCollector->send({mObjectsMerged, "objects_merged_since_last_publication"});
186  mCollector->send({mTotalUpdatesReceived, "total_updates_received"}, monitoring::DerivedMetricMode::RATE);
187  mCollector->send({mUpdatesReceived, "updates_received_since_last_publication"});
188  mCollector->send({mCyclesSinceReset, "cycles_since_reset"});
189  mObjectsMerged = 0;
190  mUpdatesReceived = 0;
191 }
