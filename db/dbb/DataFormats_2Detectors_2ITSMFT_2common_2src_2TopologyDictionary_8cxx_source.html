75 int TopologyDictionary::readFromFile(const std::string& fname)

77  if (o2::utils::Str::endsWith(fname, ".root")) {

88 int TopologyDictionary::readBinaryFile(const std::string& fname)
89 {
90  mVectorOfIDs.clear();
91  mCommonMap.clear();
92  for (auto& p : mSmallTopologiesLUT) {
93  p = -1;
94  }
95  std::ifstream in(fname.data(), std::ios::in | std::ios::binary);
96  GroupStruct gr;
97  int groupID = 0;
98  if (!in.is_open()) {
99  LOG(error) << "The file " << fname << " coud not be opened";
100  throw std::runtime_error("The file coud not be opened");
101  } else {
102  while (in.read(reinterpret_cast<char*>(&gr.mHash), sizeof(unsigned long))) {
103  in.read(reinterpret_cast<char*>(&gr.mErrX), sizeof(float));
104  in.read(reinterpret_cast<char*>(&gr.mErrZ), sizeof(float));
105  in.read(reinterpret_cast<char*>(&gr.mErr2X), sizeof(float));
106  in.read(reinterpret_cast<char*>(&gr.mErr2Z), sizeof(float));
107  in.read(reinterpret_cast<char*>(&gr.mXCOG), sizeof(float));
108  in.read(reinterpret_cast<char*>(&gr.mZCOG), sizeof(float));
109  in.read(reinterpret_cast<char*>(&gr.mNpixels), sizeof(int));
110  in.read(reinterpret_cast<char*>(&gr.mFrequency), sizeof(double));
111  in.read(reinterpret_cast<char*>(&gr.mIsGroup), sizeof(bool));
112  in.read(reinterpret_cast<char*>(&gr.mPattern.mBitmap), sizeof(unsigned char) * (ClusterPattern::kExtendedPatternBytes));
113  mVectorOfIDs.push_back(gr);
114  if (!gr.mIsGroup) {
115  mCommonMap.insert(std::make_pair(gr.mHash, groupID));
116  if (gr.mPattern.getUsedBytes() == 1) {
117  mSmallTopologiesLUT[(gr.mPattern.getColumnSpan() - 1) * 255 + (int)gr.mPattern.mBitmap[2]] = groupID;
118  }
119  } else {
120  mGroupMap.insert(std::make_pair((int)(gr.mHash >> 32) & 0x00000000ffffffff, groupID));
121  }
122  groupID++;
123  }
124  }
125  in.close();
126  return 0;
127 }

144 math_utils::Point3D<float> TopologyDictionary::getClusterCoordinates(const CompCluster& cl) const
145 {

147  o2::itsmft::SegmentationAlpide::detectorToLocalUnchecked(cl.getRow(), cl.getCol(), locCl);

153 math_utils::Point3D<float> TopologyDictionary::getClusterCoordinates(const CompCluster& cl, const ClusterPattern& patt, bool isGroup)
154 {
155  auto refRow = cl.getRow();
156  auto refCol = cl.getCol();
157  float xCOG = 0, zCOG = 0;
158  patt.getCOG(xCOG, zCOG);
159  if (isGroup) {
160  refRow -= round(xCOG);
161  refCol -= round(zCOG);
162  }
163  math_utils::Point3D<float> locCl;
164  o2::itsmft::SegmentationAlpide::detectorToLocalUnchecked(refRow + xCOG, refCol + zCOG, locCl);
165  return locCl;
166 }

169 TopologyDictionary* TopologyDictionary::loadFrom(const std::string& fname, const std::string& objName)
170 {
171  // load object from file
172  TFile fl(fname.c_str());
173  if (fl.IsZombie()) {
174  throw std::runtime_error(fmt::format("Failed to open {} file", fname));
175  }
176  auto dict = reinterpret_cast<o2::itsmft::TopologyDictionary*>(fl.GetObjectChecked(objName.c_str(), o2::itsmft::TopologyDictionary::Class()));
177  if (!dict) {
178  throw std::runtime_error(fmt::format("Failed to load {} from {}", objName, fname));
179  }
180  return dict;
181 }
