34 FileFetcher::FileFetcher(const std::string& input, const std::string& selRegex, const std::string& remRegex,
35  const std::string& copyCmd, const std::string& copyDir)
36  : mCopyCmd(copyCmd), mCopyDirName(copyDir)
37 {
38  if (!selRegex.empty()) {
39  mSelRegex = std::make_unique<std::regex>(selRegex.c_str());
40  }
41  if (!remRegex.empty()) {
42  mRemRegex = std::make_unique<std::regex>(remRegex);
43  }
44  mNoRemoteCopy = mCopyCmd == "no-copy";
45 
46  // parse input list
47  mCopyDirName = o2::utils::Str::create_unique_path(mCopyDirName, 8);
48  processInput(input);
49  LOGP(info, "Input contains {} files, {} remote", getNFiles(), mNRemote);
50  if (mNRemote) {
51  if (mNoRemoteCopy) { // make sure the copy command is provided, unless copy was explicitly forbidden
52  LOGP(info, "... but their local copying is explicitly forbidden");
53  } else {
54  if (mCopyCmd.find("?src") == std::string::npos || mCopyCmd.find("?dst") == std::string::npos) {
55  throw std::runtime_error(fmt::format("remote files asked but copy cmd \"{}\" is not valid", mCopyCmd));
56  }
57  try {
58  fs::create_directories(mCopyDirName);
59  } catch (...) {
60  throw std::runtime_error(fmt::format("failed to create scratch directory {}", mCopyDirName));
61  }
62  mCopyCmdLogFile = fmt::format("{}/{}", mCopyDirName, "copy-cmd.log");
63  LOGP(info, "FileFetcher tmp scratch directory is set to {}", mCopyDirName);
64  }
65  }
66 }

83 void FileFetcher::processInput(const std::vector<std::string>& input)
84 {
85  for (auto inp : input) {
86  o2::utils::Str::trim(inp);
87 
88  if (fs::is_directory(inp)) {
89  processDirectory(inp);
90  } else if (mSelRegex && !std::regex_match(inp, *mSelRegex.get())) { // provided selector does not match, treat as a txt file with list
91  std::ifstream listFile(inp);
92  if (!listFile.good()) {
93  LOGP(error, "file {} pretends to be a list of inputs but does not exist", inp);
94  continue;
95  }
96  std::string line;
97  std::vector<std::string> newInput;
98  while (getline(listFile, line)) {
99  o2::utils::Str::trim(line);
100  if (line[0] == '#') { // ignore commented file
101  continue;
102  }
103  newInput.push_back(line);
104  }
105  processInput(newInput);
106  } else { // should be local or remote data file
107  addInputFile(inp);
108  }
109  }
110 }

129 bool FileFetcher::addInputFile(const std::string& fname)
130 {
131  static bool alienErrorPrinted = false;
132  if (mRemRegex && std::regex_match(fname, *mRemRegex.get())) {
133  mInputFiles.emplace_back(FileRef{fname, mNoRemoteCopy ? fname : createCopyName(fname), true, false});
134  if (fname.find("alien:") == 0) {
135  if (!gGrid && !TGrid::Connect("alien://") && !alienErrorPrinted) {
136  LOG(error) << "File name starts with alien but connection to Grid failed";
137  alienErrorPrinted = true;
138  }
139  }
140  mNRemote++;
141  } else if (fs::exists(fname)) { // local file
142  mInputFiles.emplace_back(FileRef{fname, "", false, false});
143  } else {
144  LOGP(error, "file {} pretends to be local but does not exist", fname);
145  return false;
146  }
147  return true;
148 }

151 std::string FileFetcher::createCopyName(const std::string& fname) const
152 {
153  std::string cpnam{}, cpnamP = fname;
154  for (auto& c : cpnamP) {
155  if (!std::isalnum(c) && c != '.' && c != '-') {
156  c = '_';
157  }
158  }
159  while (1) {
160  cpnam = fmt::format("{}/{}_{}", mCopyDirName, o2::utils::Str::getRandomString(12), cpnamP);
161  if (!fs::exists(cpnam)) {
162  break;
163  }
164  }
165  return cpnam;
166 }

235 //____________________________________________________________
236 void FileFetcher::fetcher()
237 {
238  // data fetching/copying thread
239  size_t fileEntry = -1ul;
240 
241  if (!getNFiles()) {
242  mRunning = false;
243  return;
244  }
245 
246  // BOOST requires a locale set
247  try {
248  std::locale loc("");
249  } catch (const std::exception& e) {
250  setenv("LC_ALL", "C", 1);
251  try {
252  std::locale loc("");
253  LOG(info) << "Setting locale";
254  } catch (const std::exception& e) {
255  LOG(info) << "Setting locale failed: " << e.what();
256  return;
257  }
258  }
259 
260  while (mRunning) {
261  mNLoops = mNFilesProc / getNFiles();
262  if (mNLoops > mMaxLoops) {
263  LOGP(info, "Finished file fetching: {} of {} files fetched successfully in {} iterations", mNFilesProcOK, mNFilesProc, mMaxLoops);
264  mRunning = false;
265  break;
266  }
267  if (getQueueSize() >= mMaxInQueue) {
268  std::this_thread::sleep_for(5ms);
269  continue;
270  }
271  fileEntry = (fileEntry + 1) % getNFiles();
272  if (fileEntry == 0 && mNLoops > 0) {
273  LOG(info) << "Fetcher starts new iteration " << mNLoops;
274  }
275  mNFilesProc++;
276  auto& fileRef = mInputFiles[fileEntry];
277  if (fileRef.copied || !fileRef.remote || mNoRemoteCopy) {
278  mQueue.push(fileEntry);
279  mNFilesProcOK++;
280  } else { // need to copy
281  if (copyFile(fileEntry)) {
282  fileRef.copied = true;
283  mQueue.push(fileEntry);
284  mNFilesProcOK++;
285  }
286  }
287  }
288 }

303 bool FileFetcher::copyFile(size_t id)
304 {
305  // copy remote file to local setCopyDirName. Adaptation for Gvozden's code from SubTimeFrameFileSource::DataFetcherThread()
306  if (mCopyCmd.find("alien") != std::string::npos) {
307  if (!gGrid && !TGrid::Connect("alien://")) {
308  LOG(error) << "Copy command refers to alien but connection to Grid failed";
309  }
310  }
311  auto realCmd = std::regex_replace(std::regex_replace(mCopyCmd, std::regex("\\?src"), mInputFiles[id].getOrigName()), std::regex("\\?dst"), mInputFiles[id].getLocalName());
312  std::vector<std::string> copyParams{"-c", realCmd};
313  bp::child copyChild(bp::search_path("sh"), copyParams, bp::std_err > mCopyCmdLogFile, bp::std_out > mCopyCmdLogFile);
314  while (!copyChild.wait_for(5s)) {
315  LOGP(info, "FileFetcher: waiting for copy command. cmd={}", realCmd);
316  }
317  const auto sysRet = copyChild.exit_code();
318  if (sysRet != 0) {
319  LOGP(warning, "FileFetcher: non-zero exit code {} for cmd={}", sysRet, realCmd);
320  }
321  if (!fs::is_regular_file(mInputFiles[id].getLocalName()) || fs::is_empty(mInputFiles[id].getLocalName())) {
322  LOGP(error, "FileFetcher: failed for copy command {}", realCmd);
323  return false;
324  }
325  mCopied[mInputFiles[id].getLocalName()] = id + 1;
326  return true;
327 }
