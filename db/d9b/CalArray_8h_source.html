160 template <class T>
161 void CalArray<T>::initData()
162 {
163  const auto& mapper = Mapper::instance();
164 
165  switch (mPadSubset) {
166  case PadSubset::ROC: {
167  mData.resize(ROC(mPadSubsetNumber).rocType() == RocType::IROC ? mapper.getPadsInIROC() : mapper.getPadsInOROC());
168  if (mName.empty()) {
169  setName(boost::str(boost::format("ROC_%1$02d") % mPadSubsetNumber));
170  }
171  break;
172  }
173  case PadSubset::Partition: {
174  mData.resize(mapper.getPartitionInfo(mPadSubsetNumber % mapper.getNumberOfPartitions()).getNumberOfPads());
175  if (mName.empty()) {
176  setName(boost::str(boost::format("Partition_%1$03d") % mPadSubsetNumber));
177  }
178  break;
179  }
180  case PadSubset::Region: {
181  mData.resize(mapper.getPadRegionInfo(mPadSubsetNumber % mapper.getNumberOfPadRegions()).getNumberOfPads());
182  if (mName.empty()) {
183  setName(boost::str(boost::format("Region_%1$03d") % mPadSubsetNumber));
184  }
185  break;
186  }
187  }
188 }
