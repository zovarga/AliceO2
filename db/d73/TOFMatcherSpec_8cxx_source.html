74 void TOFMatcherSpec::init(InitContext& ic)
75 {
76  mTimer.Stop();
77  mTimer.Reset();
78  //-------- init geometry and field --------//
79  o2::base::GeometryManager::loadGeometry();
80  o2::base::Propagator::initFieldFromGRP();
81  std::unique_ptr<o2::parameters::GRPObject> grp{o2::parameters::GRPObject::loadFrom()};
82 
83  // this is a hack to provide Mat.LUT from the local file, in general will be provided by the framework from CCDB
84  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
85  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
86  if (o2::utils::Str::pathExists(matLUTFile)) {
87  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
88  o2::base::Propagator::Instance()->setMatLUT(lut);
89  LOG(debug) << "Loaded material LUT from " << matLUTFile;
90  } else {
91  LOG(debug) << "Material LUT " << matLUTFile << " file is absent, only TGeo can be used";
92  }
93  if (mStrict) {
94  mMatcher.setHighPurity();
95  }
96 }

98 void TOFMatcherSpec::run(ProcessingContext& pc)
99 {
100  mTimer.Start(false);
101 
102  RecoContainer recoData;
103  recoData.collectData(pc, *mDataRequest.get());
104 
105  auto creationTime = DataRefUtils::getHeader<DataProcessingHeader*>(pc.inputs().getFirstValid(true))->creation;
106 
107  LOG(debug) << "isTrackSourceLoaded: TPC -> " << recoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPC);
108  LOG(debug) << "isTrackSourceLoaded: ITSTPC -> " << recoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPC);
109  LOG(debug) << "isTrackSourceLoaded: TPCTRD -> " << recoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRD);
110  LOG(debug) << "isTrackSourceLoaded: ITSTPCTRD -> " << recoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRD);
111 
112  bool isTPCused = recoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPC);
113  bool isITSTPCused = recoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPC);
114  bool isTPCTRDused = recoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRD);
115  bool isITSTPCTRDused = recoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRD);
116  uint32_t ss = o2::globaltracking::getSubSpec(mStrict ? o2::globaltracking::MatchingType::Strict : o2::globaltracking::MatchingType::Standard);
117  mMatcher.setFIT(mUseFIT);
118 
119  mMatcher.setTS(creationTime);
120 
121  mMatcher.run(recoData);
122 
123  if (isTPCused) {
124  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "MTC_TPC", ss, Lifetime::Timeframe}, mMatcher.getMatchedTrackVector(o2::dataformats::MatchInfoTOFReco::TrackType::TPC));
125  if (mUseMC) {
126  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "MCMTC_TPC", ss, Lifetime::Timeframe}, mMatcher.getMatchedTOFLabelsVector(o2::dataformats::MatchInfoTOFReco::TrackType::TPC));
127  }
128 
129  auto nmatch = mMatcher.getMatchedTrackVector(o2::dataformats::MatchInfoTOFReco::TrackType::TPC).size();
130  if (mDoTPCRefit) {
131  LOG(debug) << "Refitting " << nmatch << " matched TPC tracks with TOF time info";
132  } else {
133  LOG(debug) << "Shifting Z for " << nmatch << " matched TPC tracks according to TOF time info";
134  }
135  auto& tracksTPCTOF = pc.outputs().make<std::vector<o2::dataformats::TrackTPCTOF>>(OutputRef{"tpctofTracks", ss}, nmatch);
136  mMatcher.makeConstrainedTPCTracks(tracksTPCTOF);
137  }
138 
139  if (isITSTPCused) {
140  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "MTC_ITSTPC", 0, Lifetime::Timeframe}, mMatcher.getMatchedTrackVector(o2::dataformats::MatchInfoTOFReco::TrackType::ITSTPC));
141  if (mUseMC) {
142  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "MCMTC_ITSTPC", 0, Lifetime::Timeframe}, mMatcher.getMatchedTOFLabelsVector(o2::dataformats::MatchInfoTOFReco::TrackType::ITSTPC));
143  }
144  }
145 
146  if (isTPCTRDused) {
147  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "MTC_TPCTRD", ss, Lifetime::Timeframe}, mMatcher.getMatchedTrackVector(o2::dataformats::MatchInfoTOFReco::TrackType::TPCTRD));
148  if (mUseMC) {
149  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "MCMTC_TPCTRD", ss, Lifetime::Timeframe}, mMatcher.getMatchedTOFLabelsVector(o2::dataformats::MatchInfoTOFReco::TrackType::TPCTRD));
150  }
151  }
152 
153  if (isITSTPCTRDused) {
154  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "MTC_ITSTPCTRD", 0, Lifetime::Timeframe}, mMatcher.getMatchedTrackVector(o2::dataformats::MatchInfoTOFReco::TrackType::ITSTPCTRD));
155  if (mUseMC) {
156  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "MCMTC_ITSTPCTRD", 0, Lifetime::Timeframe}, mMatcher.getMatchedTOFLabelsVector(o2::dataformats::MatchInfoTOFReco::TrackType::ITSTPCTRD));
157  }
158  }
159 
160  // TODO: TRD-matched tracks
161  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "CALIBDATA", 0, Lifetime::Timeframe}, mMatcher.getCalibVector());
162 
163  mTimer.Stop();
164 }

172 DataProcessorSpec getTOFMatcherSpec(GID::mask_t src, bool useMC, bool useFIT, bool tpcRefit, bool strict)
173 {
174  uint32_t ss = o2::globaltracking::getSubSpec(strict ? o2::globaltracking::MatchingType::Strict : o2::globaltracking::MatchingType::Standard);
175  auto dataRequest = std::make_shared<DataRequest>();
176  if (strict) {
177  dataRequest->setMatchingInputStrict();
178  }
179  dataRequest->requestTracks(src, useMC);
180  dataRequest->requestClusters(GID::getSourceMask(GID::TOF), useMC);
181  if (useFIT) {
182  dataRequest->requestClusters(GID::getSourceMask(GID::FT0), false);
183  }
184 
185  std::vector<OutputSpec> outputs;
186  if (GID::includesSource(GID::TPC, src)) {
187  outputs.emplace_back(o2::header::gDataOriginTOF, "MTC_TPC", ss, Lifetime::Timeframe);
188  outputs.emplace_back(OutputLabel{"tpctofTracks"}, o2::header::gDataOriginTOF, "TOFTRACKS_TPC", ss, Lifetime::Timeframe);
189  if (useMC) {
190  outputs.emplace_back(o2::header::gDataOriginTOF, "MCMTC_TPC", ss, Lifetime::Timeframe);
191  }
192  }
193  if (GID::includesSource(GID::ITSTPC, src)) {
194  outputs.emplace_back(o2::header::gDataOriginTOF, "MTC_ITSTPC", 0, Lifetime::Timeframe);
195  if (useMC) {
196  outputs.emplace_back(o2::header::gDataOriginTOF, "MCMTC_ITSTPC", 0, Lifetime::Timeframe);
197  }
198  }
199  if (GID::includesSource(GID::ITSTPCTRD, src)) {
200  outputs.emplace_back(o2::header::gDataOriginTOF, "MTC_ITSTPCTRD", 0, Lifetime::Timeframe);
201  if (useMC) {
202  outputs.emplace_back(o2::header::gDataOriginTOF, "MCMTC_ITSTPCTRD", 0, Lifetime::Timeframe);
203  }
204  }
205  if (GID::includesSource(GID::TPCTRD, src)) {
206  outputs.emplace_back(o2::header::gDataOriginTOF, "MTC_TPCTRD", ss, Lifetime::Timeframe);
207  if (useMC) {
208  outputs.emplace_back(o2::header::gDataOriginTOF, "MCMTC_TPCTRD", ss, Lifetime::Timeframe);
209  }
210  }
211  outputs.emplace_back(o2::header::gDataOriginTOF, "CALIBDATA", 0, Lifetime::Timeframe);
212 
213  return DataProcessorSpec{
214  "tof-matcher",
215  dataRequest->inputs,
216  outputs,
217  AlgorithmSpec{adaptFromTask<TOFMatcherSpec>(dataRequest, useMC, useFIT, tpcRefit, strict)},
218  Options{
219  {"material-lut-path", VariantType::String, "", {"Path of the material LUT file"}}}};
220 }
