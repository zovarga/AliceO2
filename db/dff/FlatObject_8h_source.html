120 template <typename T>
121 T* resizeArray(T*& ptr, int oldSize, int newSize, T* newPtr = nullptr)
122 {
123  // Resize array pointed by ptr. T must be a POD class.
124  // If the non-null newPtr is provided, use it instead of allocating a new one.
125  // In this case it is up to the user to ensure that it has at least newSize slots allocated.
126  // Return original array pointer, so that the user can manage previously allocate memory
127  if (oldSize < 0) {
128  oldSize = 0;
129  }
130  if (newSize > 0) {
131  if (!newPtr) {
132  newPtr = new T[newSize];
133  }
134  int mcp = std::min(newSize, oldSize);
135  std::memmove(newPtr, ptr, mcp * sizeof(T));
136  if (newSize > oldSize) {
137  std::memset(newPtr + mcp, 0, (newSize - oldSize) * sizeof(T));
138  }
139  }
140  T* oldPtr = ptr;
141  ptr = newPtr;
142  return oldPtr;
143 }

368 inline void FlatObject::cloneFromObject(const FlatObject& obj, char* newFlatBufferPtr)
369 {
373  // that we want to relocate the obj to external buffer)
374 
375  assert(obj.isConstructed());
376 
377  // providing *this with internal buffer as obj makes sens only if we want to conver it to object with PROVIDED external buffer
378  assert(!(!newFlatBufferPtr && obj.mFlatBufferPtr == mFlatBufferPtr && obj.isBufferInternal()));
379 
380  char* oldPtr = resizeArray(mFlatBufferPtr, mFlatBufferSize, obj.mFlatBufferSize, newFlatBufferPtr);
381 
382  if (isBufferInternal()) {
383  delete[] oldPtr; // delete old buffer if owned
384  }
385  mFlatBufferSize = obj.mFlatBufferSize;
386  mFlatBufferContainer = newFlatBufferPtr ? nullptr : mFlatBufferPtr; // external buffer is not provided, make object to own the buffer
387  std::memcpy(mFlatBufferPtr, obj.mFlatBufferPtr, obj.mFlatBufferSize);
388  mConstructionMask = (unsigned int)ConstructionState::Constructed;
389 }

405 template <class T>
406 inline std::string FlatObject::stressTest(T& obj)
407 {
411  std::string err;
412 
413  if (!obj.isConstructed()) {
414  return "tested object is not constructed!";
415  }
416 
417  T tst;
418  tst.cloneFromObject(obj, nullptr);
419  if (!tst.isConstructed() || !tst.isBufferInternal()) {
420  return "error at cloneFromObject()!";
421  }
422 
423  obj.destroy();
424 
425  char* buf0 = tst.releaseInternalBuffer();
426  char* buf1 = new char[tst.getFlatBufferSize()];
427  char* buf2 = new char[tst.getFlatBufferSize()];
428  std::memcpy(buf1, tst.getFlatBufferPtr(), tst.getFlatBufferSize());
429  tst.setActualBufferAddress(buf1);
430  delete[] buf0;
431 
432  tst.setFutureBufferAddress(buf2);
433  std::memcpy(buf2, buf1, tst.getFlatBufferSize());
434  delete[] buf1;
435 
436  if (tst.isBufferInternal()) {
437  return err = "error, buffer should be external!";
438  }
439 
440  tst.adoptInternalBuffer(buf2);
441  if (!tst.isBufferInternal()) {
442  return err = "error, buffer should be internal!";
443  }
444 
445  obj.cloneFromObject(tst, nullptr);
446  if (!obj.isBufferInternal()) {
447  return err = "error, buffer should be internal!";
448  }
449 
450  return err;
451 }

475 inline int FlatObject::writeToFile(T& obj, TFile& outf, const char* name)
476 {

482  return -1;
483  }
484 
485  bool isBufferExternal = !obj.isBufferInternal();
486  if (isBufferExternal) {
487  obj.adoptInternalBuffer(obj.mFlatBufferPtr);
488  }
489  outf.WriteObjectAny(&obj, T::Class(), name);
490  if (isBufferExternal) {
491  obj.clearInternalBufferPtr();
492  }
493  return 0;
494 }

496 template <class T, class TFile>
497 inline T* FlatObject::readFromFile(TFile& inpf, const char* name)
498 {
500 
501  if (inpf.IsZombie()) {
502  LOG(error) << "Failed to read from file " << inpf.GetName();
503  return nullptr;
504  }
505  T* pobj = reinterpret_cast<T*>(inpf.GetObjectChecked(name, T::Class()));
506  if (!pobj) {
507  LOG(error) << "Failed to load " << name << " from " << inpf.GetName();
508  return nullptr;
509  }
510  if (pobj->mFlatBufferSize > 0 && pobj->mFlatBufferContainer == nullptr) {
511  LOG(error) << "Failed to load " << name << " from " << inpf.GetName() << ": empty flat buffer container";
512  return nullptr;
513  }
514  pobj->setActualBufferAddress(pobj->mFlatBufferContainer);
515  return pobj;
516 }
