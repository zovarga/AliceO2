66 constexpr uint32_t crc32(char const* str, int length)
67 {
68  uint32_t crc = 0xFFFFFFFF;
69  for (auto j = 0; j <= length; ++j) {
70  crc = (crc >> 8) ^ crc_table[(crc ^ static_cast<unsigned int>(str[j])) & 0x000000FF];
71  }
72  return crc;
73 }

75 constexpr uint32_t compile_time_hash(char const* str)
76 {
77  return crc32(str, static_cast<int>(__builtin_strlen(str)) - 1) ^ 0xFFFFFFFF;
78 }
