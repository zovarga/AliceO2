38 void DEDigitizer::process(const Hit& hit, int evID, int srcID)
39 {
40  MCCompLabel label(hit.GetTrackID(), evID, srcID);
41  Point3D<float> pos(hit.GetX(), hit.GetY(), hit.GetZ());
42 
43  //convert energy to charge
44  auto charge = mResponse.etocharge(hit.GetEnergyLoss());
45 
46  auto time = mIR.differenceInBC(o2::raw::HBFUtils::Instance().orbitFirst);
47  // digit time will disappear anyway ? (same information in ROFRecord)
48 
49  //transformation from global to local
50  Point3D<float> lpos;
51  mTransformation.MasterToLocal(pos, lpos);
52 
53  auto anodpos = mResponse.getAnod(lpos.X());
54  auto fracplane = mResponse.chargeCorr();
55  auto chargebend = fracplane * charge;
56  auto chargenon = charge / fracplane;
57 
58  //borders of charge gen.
59  auto xMin = anodpos - mResponse.getQspreadX() * mResponse.getSigmaIntegration() * 0.5;
60  auto xMax = anodpos + mResponse.getQspreadX() * mResponse.getSigmaIntegration() * 0.5;
61  auto yMin = lpos.Y() - mResponse.getQspreadY() * mResponse.getSigmaIntegration() * 0.5;
62  auto yMax = lpos.Y() + mResponse.getQspreadY() * mResponse.getSigmaIntegration() * 0.5;
63 
64  //get segmentation for detector element
65  auto localX = anodpos;
66  auto localY = lpos.Y();
67 
68  //get area for signal induction from segmentation
69  //single pad as check
70  int padidbendcent = 0;
71  int padidnoncent = 0;
72  int ndigits = 0;
73 
74  bool padexists = mSegmentation.findPadPairByPosition(localX, localY, padidbendcent, padidnoncent);
75  if (!padexists) {
76  LOGP(warning, "Did not find _any_ pad for localX,Y={},{} for DeId {}", localX, localY, mDeId);
77  return;
78  }
79 
80  std::vector<int> padids;
81 
82  // get all pads within the defined bounding box ...
83  mSegmentation.forEachPadInArea(xMin, yMin, xMax, yMax, [&padids](int padid) {
84  padids.emplace_back(padid);
85  });
86 
87  // ... and loop over all those pads to compute the charge on each of them
88  for (auto padid : padids) {
89  auto dx = mSegmentation.padSizeX(padid) * 0.5;
90  auto dy = mSegmentation.padSizeY(padid) * 0.5;
91  auto xmin = (localX - mSegmentation.padPositionX(padid)) - dx;
92  auto xmax = xmin + 2 * dx;
93  auto ymin = (localY - mSegmentation.padPositionY(padid)) - dy;
94  auto ymax = ymin + 2 * dy;
95  auto q = mResponse.chargePadfraction(xmin, xmax, ymin, ymax);
96  if (mResponse.aboveThreshold(q)) {
97  if (mSegmentation.isBendingPad(padid)) {
98  q *= chargebend;
99  } else {
100  q *= chargenon;
101  }
102  if (q > 0) {
103  mCharges[padid] += q;
104  mLabels[padid].emplace_back(label);
105  }
106  }
107  }
108 }

110 void DEDigitizer::addNoise(float noiseProba)
111 {
112  std::random_device rd;
113  std::mt19937 mt(rd());
114 
115  float mean = noiseProba * mSegmentation.nofPads();
116  float sigma = mean / std::sqrt(mean);
117 
118  std::normal_distribution<float> gaus(mean, sigma);
119 
120  int nofNoisyPads = std::ceil(gaus(mt));
121 
122  std::uniform_int_distribution<int> ids(0, mSegmentation.nofPads() - 1);
123 
124  float chargeNoise = 1.2;
125  // FIXME: draw this also from some distribution (according to
126  // some parameters in DigitizerParam)
127  for (auto i = 0; i < nofNoisyPads; i++) {
128  auto padid = ids(mt);
129  mCharges[padid] += chargeNoise;
130  mLabels[padid].emplace_back(true);
131  }
132 }

134 void DEDigitizer::extractDigitsAndLabels(std::vector<Digit>& digits,
135  o2::dataformats::MCTruthContainer<o2::MCCompLabel>& labels)
136 {
137  int dataindex = labels.getIndexedSize();
138  for (auto padid = 0; padid < mCharges.size(); ++padid) {
139  auto q = mCharges[padid];
140  if (q <= 0) {
141  continue;
142  }
143  // FIXME: this is just to compare with previous MCHDigitizer
144  auto signal = (uint32_t)q * mResponse.getInverseChargeThreshold();
145  auto padc = signal * mResponse.getChargeThreshold();
146  auto adc = mResponse.response(TMath::Nint(padc));
147  if (adc > 0) {
148  auto time = mIR.differenceInBC(o2::raw::HBFUtils::Instance().orbitFirst);
149  digits.emplace_back(mDeId, padid, adc, time, 1);
150  for (auto element : mLabels[padid]) {
151  labels.addElement(dataindex, element);
152  }
153  ++dataindex;
154  }
155  }
156 }
