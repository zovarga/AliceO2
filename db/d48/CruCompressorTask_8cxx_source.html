50 uint64_t CruCompressorTask::buildEventOutput()
51 {
52  //mReader holds the vectors of tracklets and digits.
53  // tracklets are 64 bit
54  // digits are DigitMCMHeader and DigitMCMData
55 
56  uint64_t currentpos = 0;
57  uint64_t trailer = 0xeeeeeeeeeeeeeeeeLL;
58  //first we write a start rdh block
59  CompressedRawHeader* trackletheader = (CompressedRawHeader*)&mOutBuffer[0];
60  trackletheader->format = 1;
61  trackletheader->eventtime = 99;
62  currentpos = 1;
63  //write the
64  std::vector<o2::trd::TriggerRecord> ir;
65  std::vector<o2::trd::Tracklet64> tracklets;
66  std::vector<o2::trd::Digit> digits;
67  mReader.getParsedObjects(tracklets, digits, ir);
68  trackletheader->bc = ir[0].getBCData().bc;
69  trackletheader->orbit = ir[0].getBCData().orbit;
70  trackletheader->padding = 0xeeee;
71  trackletheader->size = mReader.sumTrackletsFound() * 8; // to get to bytes. TODO compare to getTrackletsFound
72  for (auto tracklet : mReader.getTracklets(ir[0].getBCData())) {
73  //convert tracklet to 64 bit and add to blob
74  mOutBuffer[currentpos++] = tracklet.getTrackletWord();
75  }
76  CompressedRawTrackletDigitSeperator* tracklettrailer = (CompressedRawTrackletDigitSeperator*)&mOutBuffer[currentpos];
77  tracklettrailer->word = trailer;
78  currentpos++;
79  CompressedRawHeader* digitheader = (CompressedRawHeader*)&mOutBuffer[currentpos];
80  currentpos++;
81  digitheader->format = 2;
82  digitheader->eventtime = 99;
83 
84  for (auto digit : mReader.getDigits(ir[0].getBCData())) {
85  uint64_t* word;
86  word = &mOutBuffer[currentpos];
87  DigitMCMHeader mcmheader;
88  mcmheader.eventcount = 1;
89  mcmheader.mcm = digit.getMCM();
90  mcmheader.rob = digit.getROB();
91  mcmheader.yearflag = 1;
92  mcmheader.eventcount = 1;
93  mcmheader.res = 0xc; // formst is reservedto be 1100 binary
94  //unpack the digit.
95  mcmheader.word = (*word) >> 32;
96  currentpos++;
97  }
98  CompressedRawTrackletDigitSeperator* digittrailer = (CompressedRawTrackletDigitSeperator*)&mOutBuffer[currentpos];
99  digittrailer->word = trailer;
100  currentpos++;
101  //as far as I can tell this is almost always going to be blank.
102  CompressedRawHeader* configheader = (CompressedRawHeader*)&mOutBuffer[currentpos];
103  currentpos++;
104  configheader->size = 2;
105  configheader->format = 3;
106  configheader->eventtime = 99;
107  CompressedRawTrackletDigitSeperator* configtrailer = (CompressedRawTrackletDigitSeperator*)&mOutBuffer[currentpos];
108  configtrailer->word = trailer;
109  //finally we write a stop rdh block
110 
111  return currentpos;
112 }

114 void CruCompressorTask::run(ProcessingContext& pc)
115 {
116  LOG(info) << "TRD Compression Task run method";
117 
118  auto device = pc.services().get<o2::framework::RawDeviceService>().device();
119  auto outputRoutes = pc.services().get<o2::framework::RawDeviceService>().spec().outputs;
120  auto fairMQChannel = outputRoutes.at(0).channel;
121  int inputcount = 0;
122  /* loop over inputs routes */
123  for (auto iit = pc.inputs().begin(), iend = pc.inputs().end(); iit != iend; ++iit) {
124  if (!iit.isValid()) {
125  continue;
126  }
127  //LOG(info) << "iit.mInputs " << iit.mInputs.
128  /* prepare output parts */
129  FairMQParts parts;
130 
131  /* loop over input parts */
132  for (auto const& ref : iit) {
133 
134  auto headerIn = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
135  auto dataProcessingHeaderIn = DataRefUtils::getHeader<o2::framework::DataProcessingHeader*>(ref);
136  auto payloadIn = ref.payload;
137  auto payloadInSize = headerIn->payloadSize;
138  std::cout << "payload In is : " << std::hex << payloadIn << std::endl;
139  std::cout << "payload In is : " << std::dec << payloadIn << std::endl;
140  std::cout << "payload In size is : " << std::dec << payloadInSize << std::endl;
141  mReader.setDataBuffer(payloadIn);
142  mReader.setDataBufferSize(payloadInSize);
143  mReader.setVerbose(mVerbose);
144  mReader.setDataVerbose(mDataVerbose);
145  mReader.setHeaderVerbose(mHeaderVerbose);
146  /* run */
147  mReader.run();
148 
149  auto payloadOutSize = buildEventOutput();
150 
151  auto payloadOutSizeBytes = payloadOutSize * 8; // payloadoutsize in bytes.
152  LOG(info) << "outgoing message has a data size of : " << payloadOutSize;
153  if (payloadOutSizeBytes > 32 * 1024) {
154  LOG(warn) << " buffer size for data is >32kB so will span rdh";
155  }
156  int numberofpiecestocutinto = payloadOutSizeBytes / (32 * 1024);
157  int segmentsize = 32 * 1024 - 0x40; // 0x40 is the size of the rdh header in bytes.
158  //the above will drop the decimal due to int.
159  numberofpiecestocutinto++;
160  for (int datasegment = 0; datasegment < numberofpiecestocutinto; ++datasegment) {
161  auto payloadMessage = device->NewMessage(payloadOutSize);
162  std::memcpy(payloadMessage->GetData(), (char*)mOutBuffer.data() + datasegment * segmentsize, payloadOutSize);
163  /* output */
164  auto headerOut = *headerIn;
165  auto dataProcessingHeaderOut = *dataProcessingHeaderIn;
166  headerOut.dataDescription = "CDATA";
167  headerOut.payloadSize = payloadOutSizeBytes;
168  // what to do about the packet count?
169  //headerOut.packetCounter;
170  o2::header::Stack headerStack{headerOut, dataProcessingHeaderOut};
171  auto headerMessage = device->NewMessage(headerStack.size());
172  std::memcpy(headerMessage->GetData(), headerStack.data(), headerStack.size());
173 
174  /* add parts */
175  parts.AddPart(std::move(headerMessage));
176  parts.AddPart(std::move(payloadMessage));
177  }
178  }
179 
180  /* send message */
181  device->Send(parts, fairMQChannel);
182  }
183 }
