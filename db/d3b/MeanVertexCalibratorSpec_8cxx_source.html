76 void MeanVertexCalibDevice::sendOutput(DataAllocator& output)
77 {
78 
79  // extract CCDB infos and calibration objects, convert it to TMemFile and send them to the output
80  // TODO in principle, this routine is generic, can be moved to Utils.h
81 
82  using clbUtils = o2::calibration::Utils;
83  const auto& payloadVec = mCalibrator->getMeanVertexObjectVector();
84  auto& infoVec = mCalibrator->getMeanVertexObjectInfoVector(); // use non-const version as we update it
85  assert(payloadVec.size() == infoVec.size());
86 
87  for (uint32_t i = 0; i < payloadVec.size(); i++) {
88  auto& w = infoVec[i];
89  auto image = o2::ccdb::CcdbApi::createObjectImage(&payloadVec[i], &w);
90  LOG(info) << "Sending object " << w.getPath() << "/" << w.getFileName() << " of size " << image->size()
91  << " bytes, valid for " << w.getStartValidityTimestamp() << " : " << w.getEndValidityTimestamp();
92 
93  output.snapshot(Output{clbUtils::gDataOriginCDBPayload, "MEANVERTEX", i}, *image.get()); // vector<char>
94  output.snapshot(Output{clbUtils::gDataOriginCDBWrapper, "MEANVERTEX", i}, w); // root-serialized
95  }
96  if (payloadVec.size()) {
97  mCalibrator->initOutput(); // reset the outputs once they are already sent
98  }
99 }
