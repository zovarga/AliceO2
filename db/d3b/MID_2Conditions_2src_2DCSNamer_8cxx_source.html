107 std::optional<ID> detElemId2DCS(int deId)
108 {
109  if (deId < 0 || deId > 71) {
110  return std::nullopt;
111  }
112  int chamberId = detparams::getChamber(deId);
113  int id = 1 + detparams::getRPCLine(deId);
114  Side side = detparams::isRightSide(deId) ? Side::Inside : Side::Outside;
115 
116  switch (chamberId) {
117  case 0:
118  chamberId = 11;
119  break;
120  case 1:
121  chamberId = 12;
122  break;
123  case 2:
124  chamberId = 21;
125  break;
126  case 3:
127  chamberId = 22;
128  break;
129  default:
130  throw std::runtime_error(fmt::format("invalid chamberId={} : expected between 0 and 3", chamberId));
131  }
132 
133  return std::optional<ID>{{id, side, chamberId}};
134 }

136 std::vector<std::string> aliases(std::vector<MeasurementType> types)
137 {
138 
139  std::vector<std::string> patterns;
140  for (auto deId = 0; deId < detparams::NDetectionElements; ++deId) {
141  auto id = detElemId2DCS(deId);
142  auto pattern = fmt::format("{}_{}_MT{}_RPC{}_HV.", detPrefix,
143  sideName(id->side), id->chamberId, id->number) +
144  "%s";
145  patterns.emplace_back(pattern);
146  }
147  std::vector<std::string> aliases;
148 
149  for (auto m : types) {
150  if (m == MeasurementType::HV_V || m == MeasurementType::HV_I) {
151  auto aliasPerType = measurement(patterns, m);
152  aliases.insert(aliases.begin(), aliasPerType.begin(), aliasPerType.end());
153  }
154  }
155  return aliases;
156 }
