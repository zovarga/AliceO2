41 void CTFCoder::compress(CompressedDigits& cd, const gsl::span<const BCData>& digitVec, const gsl::span<const ChannelData>& channelVec)
42 {
43  // convert digits/channel to their compressed version
44  cd.clear();
45  if (!digitVec.size()) {
46  return;
47  }
48  const auto& dig0 = digitVec[0];
49  cd.header.det = mDet;
50  cd.header.nTriggers = digitVec.size();
51  cd.header.firstOrbit = dig0.ir.orbit;
52  cd.header.firstBC = dig0.ir.bc;
53 
54  cd.trigger.resize(cd.header.nTriggers);
55  cd.bcInc.resize(cd.header.nTriggers);
56  cd.orbitInc.resize(cd.header.nTriggers);
57  cd.nChan.resize(cd.header.nTriggers);
58 
59  cd.idChan.resize(channelVec.size());
60  cd.time.resize(channelVec.size());
61  cd.charge.resize(channelVec.size());
62 
63  uint16_t prevBC = cd.header.firstBC;
64  uint32_t prevOrbit = cd.header.firstOrbit;
65  uint32_t ccount = 0;
66  for (uint32_t idig = 0; idig < cd.header.nTriggers; idig++) {
67  const auto& digit = digitVec[idig];
68  const auto chanels = digit.getBunchChannelData(channelVec); // we assume the channels are sorted
69 
70  // fill trigger info
71  cd.trigger[idig] = digit.getTriggers().triggerSignals;
72  if (prevOrbit == digit.ir.orbit) {
73  cd.bcInc[idig] = digit.ir.bc - prevBC;
74  cd.orbitInc[idig] = 0;
75  } else {
76  cd.bcInc[idig] = digit.ir.bc;
77  cd.orbitInc[idig] = digit.ir.orbit - prevOrbit;
78  }
79  prevBC = digit.ir.bc;
80  prevOrbit = digit.ir.orbit;
81  // fill channels info
82  cd.nChan[idig] = chanels.size();
83  if (!cd.nChan[idig]) {
84  LOG(debug) << "Digits with no channels";
85  continue;
86  }
87  uint8_t prevChan = 0;
88  for (uint8_t ic = 0; ic < cd.nChan[idig]; ic++) {
89  assert(prevChan <= chanels[ic].pmtNumber);
90  cd.idChan[ccount] = chanels[ic].pmtNumber - prevChan;
91  cd.time[ccount] = chanels[ic].time; // make sure it fits to short!!!
92  cd.charge[ccount] = chanels[ic].chargeAdc; // make sure we really need short!!!
93  prevChan = chanels[ic].pmtNumber;
94  ccount++;
95  }
96  }
97 }
