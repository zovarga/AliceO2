40 template <typename RawReaderType>
41 class FITDataReaderDPLSpec : public Task
42 {
43  public:
44  FITDataReaderDPLSpec(const RawReaderType& rawReader) : mRawReader(rawReader) {}
45  FITDataReaderDPLSpec() = delete;
46  ~FITDataReaderDPLSpec() override = default;
47  typedef RawReaderType RawReader_t;
48  RawReader_t mRawReader;
49  void init(InitContext& ic) final
50  {
51  auto ccdbUrl = ic.options().get<std::string>("ccdb-path");
52  auto lutPath = ic.options().get<std::string>("lut-path");
53  if (ccdbUrl != "") {
54  RawReader_t::LookupTable_t::setCCDBurl(ccdbUrl);
55  }
56  if (lutPath != "") {
57  RawReader_t::LookupTable_t::setLUTpath(lutPath);
58  }
59  RawReader_t::LookupTable_t::Instance().printFullMap();
60  auto nReserveVecDig = ic.options().get<int>("reserve-vec-dig");
61  auto nReserveVecChData = ic.options().get<int>("reserve-vec-chdata");
62  auto nReserveVecBuffer = ic.options().get<int>("reserve-vec-buffer");
63  auto nReserveMapDig = ic.options().get<int>("reserve-map-dig");
64  if (nReserveVecDig || nReserveVecChData) {
65  mRawReader.reserveVecDPL(nReserveVecDig, nReserveVecChData);
66  }
67  if (nReserveVecBuffer || nReserveMapDig) {
68  mRawReader.reserve(nReserveVecBuffer, nReserveMapDig);
69  }
70  }
71  void run(ProcessingContext& pc) final
72  {
73  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload this means that the "delayed message"
74  // mechanism created it in absence of real data from upstream. Processor should send empty output to not block the workflow
75  {
76  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
77  std::vector<InputSpec> dummy{InputSpec{"dummy", ConcreteDataMatcher{mRawReader.mDataOrigin, o2::header::gDataDescriptionRawData, 0xDEADBEEF}}};
78  for (const auto& ref : InputRecordWalker(pc.inputs(), dummy)) {
79  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
80  if (dh->payloadSize == 0) {
81  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
82  if (++contDeadBeef <= maxWarn) {
83  LOGP(warning, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
84  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize,
85  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
86  }
87  mRawReader.makeSnapshot(pc); // send empty output
88  return;
89  }
90  }
91  contDeadBeef = 0; // if good data, reset the counter
92  }
93  std::vector<InputSpec> filter{InputSpec{"filter", ConcreteDataTypeMatcher{mRawReader.mDataOrigin, o2::header::gDataDescriptionRawData}, Lifetime::Timeframe}};
94  DPLRawParser parser(pc.inputs(), filter);
95  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
96  //Proccessing each page
97  auto rdhPtr = it.get_if<o2::header::RAWDataHeader>();
98  gsl::span<const uint8_t> payload(it.data(), it.size());
99  mRawReader.process(payload, int(rdhPtr->linkID), int(rdhPtr->endPointID));
100  }
101  mRawReader.accumulateDigits();
102  mRawReader.makeSnapshot(pc);
103  mRawReader.clear();
104  }
105 };

107 template <typename RawReaderType>
108 framework::DataProcessorSpec getFITDataReaderDPLSpec(const RawReaderType& rawReader, bool askSTFDist)
109 {
110  std::vector<OutputSpec> outputSpec;
111  rawReader.configureOutputSpec(outputSpec);
112  std::vector<InputSpec> inputSpec{{"STF", ConcreteDataTypeMatcher{rawReader.mDataOrigin, "RAWDATA"}, Lifetime::Optional}};
113  if (askSTFDist) {
114  inputSpec.emplace_back("STFDist", "FLP", "DISTSUBTIMEFRAME", 0, Lifetime::Timeframe);
115  }
116  std::string dataProcName = rawReader.mDataOrigin.template as<std::string>();
117  std::for_each(dataProcName.begin(), dataProcName.end(), [](char& c) { c = ::tolower(c); });
118  dataProcName += "-datareader-dpl";
119  LOG(info) << dataProcName;
120  return DataProcessorSpec{
121  dataProcName,
122  inputSpec,
123  outputSpec,
124  adaptFromTask<FITDataReaderDPLSpec<RawReaderType>>(rawReader),
125  {o2::framework::ConfigParamSpec{"ccdb-path", VariantType::String, "", {"CCDB url which contains LookupTable"}},
126  o2::framework::ConfigParamSpec{"lut-path", VariantType::String, "", {"LookupTable path, e.g. FT0/LookupTable"}},
127  o2::framework::ConfigParamSpec{"reserve-vec-dig", VariantType::Int, 0, {"Reserve memory for Digit vector, to DPL channel"}},
128  o2::framework::ConfigParamSpec{"reserve-vec-chdata", VariantType::Int, 0, {"Reserve memory for ChannelData vector, to DPL channel"}},
129  o2::framework::ConfigParamSpec{"reserve-vec-buffer", VariantType::Int, 0, {"Reserve memory for DataBlock vector, buffer for each page"}},
130  o2::framework::ConfigParamSpec{"reserve-map-dig", VariantType::Int, 0, {"Reserve memory for Digit map, mapping in RawReader"}}}};
131 }
