116 GPUd() void GPUTPCCompressionTrackModel::Init(float x, float y, float z, float alpha, unsigned char qPt, const GPUParam& GPUrestrict() param)
117 {
118  // initialize track model
119  mX = x;
120  mAlpha = alpha;
121  mCosAlpha = CAMath::Cos(alpha);
122  mSinAlpha = CAMath::Sin(alpha);
123  mP[0] = y;
124  mP[1] = z;
125  mP[2] = 0.f;
126  mP[3] = 0.f;
127  mP[4] = (qPt - 127.f) * (20.f / 127.f);
128  resetCovariance();
129  mNDF = -5;
130  mBz = param.par.constBz;
131  float pti = CAMath::Abs(mP[4]);
132  if (pti < 1.e-4f) {
133  pti = 1.e-4f; // set 10.000 GeV momentum for straight track
134  }
135  mTrk.x = x;
136  mTrk.y = y;
137  mTrk.z = z;
138  mTrk.q = (mP[4] >= 0) ? 1.f : -1.f;
139  mTrk.pt = 1.f / pti;
140  mTrk.p = mTrk.pt;
141  mTrk.px = mTrk.pt;
142  mTrk.py = 0.f;
143  mTrk.pz = 0.f;
144  mTrk.qpt = mTrk.q * pti;
145  calculateMaterialCorrection();
146 }

148 GPUd() int GPUTPCCompressionTrackModel::Propagate(float x, float alpha)
149 {
150  // constrain sin(phi)
151  if (mP[2] > MaxSinPhi) {
152  mP[2] = MaxSinPhi;
153  } else if (mP[2] < -MaxSinPhi) {
154  mP[2] = -MaxSinPhi;
155  }
156  // propagate track parameters to specified x
157  if (CAMath::Abs(alpha - mAlpha) > 1.e-4) {
158  if (rotateToAlpha(alpha) != 0) {
159  return -2;
160  }
161  }
162  if (CAMath::Abs(x - mX) < 1.e-7f) {
163  mX = x;
164  return 0;
165  }
166 
167  // propagate mTrk to t0e
168  PhysicalTrackModel t0e(mTrk);
169  float dLp = 0;
170  if (CAMath::Abs(x - t0e.x) < 1.e-8f) {
171  return 0;
172  }
173  if (propagateToXBzLightNoUpdate(t0e, x, mBz, dLp)) {
174  return 1;
175  }
176  updatePhysicalTrackValues(t0e);
177  if (CAMath::Abs(t0e.sinphi) >= MaxSinPhi) {
178  return -3;
179  }
180  return followLinearization(t0e, mBz, dLp);
181 }

183 GPUd() int GPUTPCCompressionTrackModel::Filter(float y, float z, int iRow)
184 {
185  // apply kalman filter update with measurement y/z
186  float err2Y, err2Z;
187  getClusterRMS2(iRow, z, mTrk.sinphi, mTrk.dzds, err2Y, err2Z);
188  if (mNDF == -5) {
189  // first measurement: no need to filter, as the result is known in advance. so just set it
190  // ignore offline statistical errors for now (as is also done by default)
191  mP[0] = y;
192  mP[1] = z;
193  mC[0] = err2Y;
194  mC[2] = err2Z;
195  mNDF = -3;
196  return 0;
197  }
198 
199  // constrain sin(phi)
200  if (mP[2] > MaxSinPhi) {
201  mP[2] = MaxSinPhi;
202  } else if (mP[2] < -MaxSinPhi) {
203  mP[2] = -MaxSinPhi;
204  }
205 
206  const float d00 = mC[0], d01 = mC[1], d02 = mC[3], d03 = mC[6], d04 = mC[10];
207  const float d10 = mC[1], d11 = mC[2], d12 = mC[4], d13 = mC[7], d14 = mC[11];
208 
209  const float z0 = y - mP[0];
210  const float z1 = z - mP[1];
211  float w0, w1, w2;
212  if (mNDF <= 0) {
213  w0 = 1.f / (err2Y + d00);
214  w1 = 0;
215  w2 = 1.f / (err2Z + d11);
216  } else {
217  w0 = d11 + err2Z, w1 = d10, w2 = d00 + err2Y;
218  { // Invert symmetric matrix
219  float det = w0 * w2 - w1 * w1;
220  if (CAMath::Abs(det) < 1.e-10f) {
221  return -1;
222  }
223  det = 1.f / det;
224  w0 = w0 * det;
225  w1 = -w1 * det;
226  w2 = w2 * det;
227  }
228  }
229  mNDF += 2;
230 
231  if (mNDF <= 0) {
232  const float k00 = d00 * w0;
233  const float k20 = d02 * w0;
234  const float k40 = d04 * w0;
235  const float k11 = d11 * w2;
236  const float k31 = d13 * w2;
237  mP[0] += k00 * z0;
238  mP[1] += k11 * z1;
239  mP[2] += k20 * z0;
240  mP[3] += k31 * z1;
241  mP[4] += k40 * z0;
242 
243  mC[0] -= k00 * d00;
244  mC[2] -= k11 * d11;
245  mC[3] -= k20 * d00;
246  mC[5] -= k20 * d02;
247  mC[7] -= k31 * d11;
248  mC[9] -= k31 * d13;
249  mC[10] -= k00 * d04;
250  mC[12] -= k40 * d02;
251  mC[14] -= k40 * d04;
252  } else {
253  const float k00 = d00 * w0 + d01 * w1;
254  const float k01 = d00 * w1 + d10 * w2;
255  const float k10 = d01 * w0 + d11 * w1;
256  const float k11 = d01 * w1 + d11 * w2;
257  const float k20 = d02 * w0 + d12 * w1;
258  const float k21 = d02 * w1 + d12 * w2;
259  const float k30 = d03 * w0 + d13 * w1;
260  const float k31 = d03 * w1 + d13 * w2;
261  const float k40 = d04 * w0 + d14 * w1;
262  const float k41 = d04 * w1 + d14 * w2;
263 
264  mP[0] += k00 * z0 + k01 * z1;
265  mP[1] += k10 * z0 + k11 * z1;
266  mP[2] += k20 * z0 + k21 * z1;
267  mP[3] += k30 * z0 + k31 * z1;
268  mP[4] += k40 * z0 + k41 * z1;
269 
270  mC[0] -= k00 * d00 + k01 * d10;
271 
272  mC[2] -= k10 * d01 + k11 * d11;
273 
274  mC[3] -= k20 * d00 + k21 * d10;
275  mC[5] -= k20 * d02 + k21 * d12;
276 
277  mC[7] -= k30 * d01 + k31 * d11;
278  mC[9] -= k30 * d03 + k31 * d13;
279 
280  mC[10] -= k40 * d00 + k41 * d10;
281  mC[12] -= k40 * d02 + k41 * d12;
282  mC[14] -= k40 * d04 + k41 * d14;
283 
284  mC[1] -= k10 * d00 + k11 * d10;
285  mC[4] -= k20 * d01 + k21 * d11;
286  mC[6] -= k30 * d00 + k31 * d10;
287  mC[8] -= k30 * d02 + k31 * d12;
288  mC[11] -= k40 * d01 + k41 * d11;
289  mC[13] -= k40 * d03 + k41 * d13;
290  }
291  return 0;
292 }

294 GPUd() int GPUTPCCompressionTrackModel::Mirror()
295 {
296  float dy = -2.f * mTrk.q * mTrk.px / mBz;
297  float dS; // path in XY
298  {
299  float chord = dy; // chord to the extrapolated point == |dy|*sign(x direction)
300  float sa = -mTrk.cosphi; // sin( half of the rotation angle ) == (chord/2) / radius
301 
302  // dS = (Pt/b)*2*arcsin( sa )
303  // = (Pt/b)*2*sa*(1 + 1/6 sa^2 + 3/40 sa^4 + 5/112 sa^6 +... )
304  // = chord*(1 + 1/6 sa^2 + 3/40 sa^4 + 5/112 sa^6 +... )
305 
306  float sa2 = sa * sa;
307  const float k2 = 1.f / 6.f;
308  const float k4 = 3.f / 40.f;
309  // const float k6 = 5.f/112.f;
310  dS = chord + chord * sa2 * (k2 + k4 * sa2);
311  // dS = sqrtf(pt2)/b*2.*CAMath::ASin( sa );
312  }
313 
314  if (mTrk.sinphi < 0.f) {
315  dS = -dS;
316  }
317 
318  mTrk.y = mTrk.y + 2.f * dy; // TODO check why dy is added TWICE to the track position
319  mTrk.z = mTrk.z + 2.f * mTrk.dzds * dS;
320  changeDirection();
321 
322  // Energy Loss
323 
324  float dL = CAMath::Copysign(dS * mTrk.dlds, -1.f); // we are in flight direction
325 
326  float& mC40 = mC[10];
327  float& mC41 = mC[11];
328  float& mC42 = mC[12];
329  float& mC43 = mC[13];
330  float& mC44 = mC[14];
331 
332  float dLmask = 0.f;
333  bool maskMS = (CAMath::Abs(dL) < mMaterial.DLMax);
334  if (maskMS) {
335  dLmask = dL;
336  }
337  float dLabs = CAMath::Abs(dLmask);
338  float corr = 1.f - mMaterial.EP2 * dLmask;
339 
340  float corrInv = 1.f / corr;
341  mTrk.px *= corrInv;
342  mTrk.py *= corrInv;
343  mTrk.pz *= corrInv;
344  mTrk.pt *= corrInv;
345  mTrk.p *= corrInv;
346  mTrk.qpt *= corr;
347 
348  mP[4] *= corr;
349 
350  mC40 *= corr;
351  mC41 *= corr;
352  mC42 *= corr;
353  mC43 *= corr;
354  mC44 = mC44 * corr * corr + dLabs * mMaterial.sigmadE2;
355 
356  return 0;
357 }

359 GPUd() void GPUTPCCompressionTrackModel::updatePhysicalTrackValues(PhysicalTrackModel& trk)
360 {
361  float px = trk.px;
362  if (CAMath::Abs(px) < 1.e-4f) {
363  px = CAMath::Copysign(1.e-4f, px);
364  }
365 
366  trk.pt = sqrt(px * px + trk.py * trk.py);
367  float pti = 1.f / trk.pt;
368  trk.p = sqrt(px * px + trk.py * trk.py + trk.pz * trk.pz);
369  trk.sinphi = trk.py * pti;
370  trk.cosphi = px * pti;
371  trk.secphi = trk.pt / px;
372  trk.dzds = trk.pz * pti;
373  trk.dlds = trk.p * pti;
374  trk.qpt = trk.q * pti;
375 }

395 GPUd() int GPUTPCCompressionTrackModel::rotateToAlpha(float newAlpha)
396 {
397  //
398  // Rotate the track coordinate system in XY to the angle newAlpha
399  // return value is error code (0==no error)
400  //
401 
402  float newCosAlpha = CAMath::Cos(newAlpha);
403  float newSinAlpha = CAMath::Sin(newAlpha);
404 
405  float cc = newCosAlpha * mCosAlpha + newSinAlpha * mSinAlpha; // cos(newAlpha - mAlpha);
406  float ss = newSinAlpha * mCosAlpha - newCosAlpha * mSinAlpha; // sin(newAlpha - mAlpha);
407 
408  PhysicalTrackModel t0 = mTrk;
409 
410  float x0 = mTrk.x;
411  float y0 = mTrk.y;
412  float px0 = mTrk.px;
413  float py0 = mTrk.py;
414 
415  if (CAMath::Abs(mP[2]) >= MaxSinPhi || CAMath::Abs(px0) < (1 - MaxSinPhi)) {
416  return -1;
417  }
418 
419  // rotate t0 track
420  float px1 = px0 * cc + py0 * ss;
421  float py1 = -px0 * ss + py0 * cc;
422 
423  {
424  t0.x = x0 * cc + y0 * ss;
425  t0.y = -x0 * ss + y0 * cc;
426  t0.px = px1;
427  t0.py = py1;
428  updatePhysicalTrackValues(t0);
429  }
430 
431  if (CAMath::Abs(py1) > MaxSinPhi * mTrk.pt || CAMath::Abs(px1) < (1 - MaxSinPhi)) {
432  return -1;
433  }
434 
435  // calculate X of rotated track:
436  float trackX = x0 * cc + ss * mP[0];
437 
438  // transport t0 to trackX
439  float dLp = 0;
440  if (propagateToXBzLightNoUpdate(t0, trackX, mBz, dLp)) {
441  return -1;
442  }
443  updatePhysicalTrackValues(t0);
444 
445  if (CAMath::Abs(t0.sinphi) >= MaxSinPhi) {
446  return -1;
447  }
448 
449  // now t0 is rotated and propagated, all checks are passed
450 
451  // Rotate track using mTrk for linearisation. After rotation X is not fixed, but has a covariance
452 
453  // Y Z Sin DzDs q/p
454  // Jacobian J0 = { { j0, 0, 0, 0, 0 }, // Y
455  // { 0, 1, 0, 0, 0 }, // Z
456  // { 0, 0, j1, 0, 0 }, // SinPhi
457  // { 0, 0, 0, 1, 0 }, // DzDs
458  // { 0, 0, 0, 0, 1 }, // q/p
459  // { j2, 0, 0, 0, 0 } }// X (rotated )
460 
461  float j0 = cc;
462  float j1 = px1 / px0;
463  float j2 = ss;
464  // float dy = mT->Y() - y0;
465  // float ds = mT->SinPhi() - mTrk.SinPhi();
466 
467  mX = trackX; // == x0*cc + ss*mP[0] == t0.x + j0*dy;
468  mP[0] = -x0 * ss + cc * mP[0]; //== t0.y + j0*dy;
469  // mP[2] = py1/pt0 + j1*ds; // == t0.sinphi + j1*ds; // use py1, since t0.sinphi can have different sign
470  mP[2] = -CAMath::Sqrt(1.f - mP[2] * mP[2]) * ss + mP[2] * cc;
471 
472  // Rotate cov. matrix Cr = J0 x C x J0T. Cr has one more row+column for X:
473  float* c = mC;
474 
475  float c15 = c[0] * j0 * j2;
476  float c16 = c[1] * j2;
477  float c17 = c[3] * j1 * j2;
478  float c18 = c[6] * j2;
479  float c19 = c[10] * j2;
480  float c20 = c[0] * j2 * j2;
481 
482  c[0] *= j0 * j0;
483  c[3] *= j0;
484  c[10] *= j0;
485 
486  c[3] *= j1;
487  c[5] *= j1 * j1;
488  c[12] *= j1;
489 
490  if (setDirectionAlongX(t0)) { // change direction if Px < 0
491  mP[2] = -mP[2];
492  mP[3] = -mP[3];
493  mP[4] = -mP[4];
494  c[3] = -c[3]; // covariances with SinPhi
495  c[4] = -c[4];
496  c17 = -c17;
497  c[6] = -c[6]; // covariances with DzDs
498  c[7] = -c[7];
499  c18 = -c18;
500  c[10] = -c[10]; // covariances with QPt
501  c[11] = -c[11];
502  c19 = -c19;
503  }
504 
505  // Now fix the X coordinate: so to say, transport track T to fixed X = mX.
506  // only covariance changes. Use rotated and transported t0 for linearisation
507  float j3 = -t0.py / t0.px;
508  float j4 = -t0.pz / t0.px;
509  float j5 = t0.qpt * mBz;
510 
511  // Y Z Sin DzDs q/p X
512  // Jacobian J1 = { { 1, 0, 0, 0, 0, j3 }, // Y
513  // { 0, 1, 0, 0, 0, j4 }, // Z
514  // { 0, 0, 1, 0, 0, j5 }, // SinPhi
515  // { 0, 0, 0, 1, 0, 0 }, // DzDs
516  // { 0, 0, 0, 0, 1, 0 } }; // q/p
517 
518  float h15 = c15 + c20 * j3;
519  float h16 = c16 + c20 * j4;
520  float h17 = c17 + c20 * j5;
521 
522  c[0] += j3 * (c15 + h15);
523 
524  c[2] += j4 * (c16 + h16);
525 
526  c[3] += c17 * j3 + h15 * j5;
527  c[5] += j5 * (c17 + h17);
528 
529  c[7] += c18 * j4;
530  // c[ 9] = c[ 9];
531 
532  c[10] += c19 * j3;
533  c[12] += c19 * j5;
534  // c[14] = c[14];
535 
536  mAlpha = newAlpha;
537  mCosAlpha = newCosAlpha;
538  mSinAlpha = newSinAlpha;
539  mTrk = t0;
540 
541  return 0;
542 }

544 GPUd() int GPUTPCCompressionTrackModel::propagateToXBzLightNoUpdate(PhysicalTrackModel& t, float x, float Bz, float& dLp)
545 {
546  //
547  // transport the track to X=x in magnetic field B = ( 0, 0, Bz[kG*0.000299792458] )
548  // dLp is a return value == path length / track momentum [cm/(GeV/c)]
549  // the method returns error code (0 == no error)
550  //
551  // Additional values are not recalculated, UpdateValues() has to be called afterwards!!
552  //
553  float b = t.q * Bz;
554  float pt2 = t.px * t.px + t.py * t.py;
555  float dx = x - t.x;
556  float pye = t.py - dx * b; // extrapolated py
557  float pxe2 = pt2 - pye * pye;
558 
559  if (t.px < (1.f - MaxSinPhi) || pxe2 < (1.f - MaxSinPhi) * (1.f - MaxSinPhi)) {
560  return -1; // can not transport to x=x
561  }
562  float pxe = CAMath::Sqrt(pxe2); // extrapolated px
563  float pti = 1.f / CAMath::Sqrt(pt2);
564 
565  float ty = (t.py + pye) / (t.px + pxe);
566  float dy = dx * ty;
567  float dS; // path in XY
568  {
569  float chord = dx * CAMath::Sqrt(1.f + ty * ty); // chord to the extrapolated point == sqrt(dx^2+dy^2)*sign(dx)
570  float sa = 0.5f * chord * b * pti; // sin( half of the rotation angle ) == (chord/2) / radius
571 
572  // dS = (Pt/b)*2*arcsin( sa )
573  // = (Pt/b)*2*sa*(1 + 1/6 sa^2 + 3/40 sa^4 + 5/112 sa^6 +... )
574  // = chord*(1 + 1/6 sa^2 + 3/40 sa^4 + 5/112 sa^6 +... )
575 
576  float sa2 = sa * sa;
577  const float k2 = 1.f / 6.f;
578  const float k4 = 3.f / 40.f;
579  // const float k6 = 5.f/112.f;
580  dS = chord + chord * sa2 * (k2 + k4 * sa2);
581  // dS = sqrt(pt2)/b*2.*CAMath::ASin( sa );
582  }
583 
584  dLp = pti * dS; // path in XYZ / p == path in XY / pt
585 
586  float dz = t.pz * dLp;
587 
588  t.x = x;
589  t.y += dy;
590  t.z += dz;
591  t.px = pxe;
592  t.py = pye;
593  return 0;
594 }

614 GPUd() int GPUTPCCompressionTrackModel::followLinearization(const PhysicalTrackModel& t0e, float Bz, float dLp)
615 {

617 
618  // propagate track and cov matrix with derivatives for (0,0,Bz) field
619 
620  float dS = dLp * t0e.pt;
621  float dL = CAMath::Abs(dLp * t0e.p);
622 
623  dL = -dL; // we are always in flight direction
624 
625  float ey = mTrk.sinphi;
626  float ex = mTrk.cosphi;
627  float exi = mTrk.secphi;
628  float ey1 = t0e.sinphi;
629  float ex1 = t0e.cosphi;
630  float ex1i = t0e.secphi;

632  float k = -mTrk.qpt * Bz;
633  float dx = t0e.x - mTrk.x;
634  float kdx = k * dx;
635  float cc = ex + ex1;
636  float cci = 1.f / cc;
637 
638  float dxcci = dx * cci;
639  float hh = dxcci * ex1i * (1.f + ex * ex1 + ey * ey1);
640 
641  float j02 = exi * hh;
642  float j04 = -Bz * dxcci * hh;
643  float j13 = dS;
644  float j24 = -dx * Bz;

647 
648  float d0 = p[0] - mTrk.y;
649  float d1 = p[1] - mTrk.z;
650  float d2 = p[2] - mTrk.sinphi;
651  float d3 = p[3] - mTrk.dzds;
652  float d4 = p[4] - mTrk.qpt;
653 
654  float newSinPhi = ey1 + d2 + j24 * d4;
655  if (mNDF >= 15 && CAMath::Abs(newSinPhi) > MaxSinPhi) {
656  return -4;
657  }
658 
659  mTrk = t0e;
660  mX = t0e.x;
661  p[0] = t0e.y + d0 + j02 * d2 + j04 * d4;
662  p[1] = t0e.z + d1 + j13 * d3;
663  p[2] = newSinPhi;
664  p[3] = t0e.dzds + d3;
665  p[4] = t0e.qpt + d4;

670  float c21 = c[4];
671  float c22 = c[5];
672

679  float c41 = c[11];

681  float c43 = c[13];
682  float c44 = c[14];

685  float c20ph04c42 = c20 + j04 * c42;
686  float j02c22 = j02 * c22;
687  float j04c44 = j04 * c44;
688 
689  float n6 = c30 + j02 * c32 + j04 * c43;
690  float n7 = c31 + j13 * c33;
691  float n10 = c40 + j02 * c42 + j04c44;
692  float n11 = c41 + j13 * c43;
693  float n12 = c42 + j24 * c44;
694 
695  c[0] += j02 * j02c22 + j04 * j04c44 + 2.f * (j02 * c20ph04c42 + j04 * c40);
696  c[1] += j02 * c21 + j04 * c41 + j13 * n6;
697  c[2] += j13 * (c31 + n7);
698  c[3] = c20ph04c42 + j02c22 + j24 * n10;
699  c[4] = c21 + j13 * c32 + j24 * n11;
700  c[5] = c22 + j24 * (c42 + n12);
701  c[6] = n6;
702  c[7] = n7;
703  c[8] = c32 + c43 * j24;
704  c[10] = n10;
705  c[11] = n11;
706  c[12] = n12;
707  } else {
708  float c00 = c[0];
709  float c10 = c[1];
710  float c11 = c[2];
711 
712  float ss = ey + ey1;
713  float tg = ss * cci;
714  float xx = 1.f - 0.25f * kdx * kdx * (1.f + tg * tg);
715  if (xx < 1.e-8f) {
716  return -1;
717  }
718  xx = CAMath::Sqrt(xx);
719  float yy = CAMath::Sqrt(ss * ss + cc * cc);
720 
721  float j12 = dx * mTrk.dzds * tg * (2.f + tg * (ey * exi + ey1 * ex1i)) / (xx * yy);
722  float j14 = 0;
723  if (CAMath::Abs(mTrk.qpt) > 1.e-6f) {
724  j14 = (2.f * xx * ex1i * dx / yy - dS) * mTrk.dzds / mTrk.qpt;
725  } else {
726  j14 = -mTrk.dzds * Bz * dx * dx * exi * exi * exi * (0.5f * ey + (1.f / 3.f) * kdx * (1 + 2.f * ey * ey) * exi * exi);
727  }
728 
729  p[1] += j12 * d2 + j14 * d4;
730 
731  float h00 = c00 + c20 * j02 + c40 * j04;
732  // float h01 = c10 + c21*j02 + c41*j04;
733  float h02 = c20 + c22 * j02 + c42 * j04;
734  // float h03 = c30 + c32*j02 + c43*j04;
735  float h04 = c40 + c42 * j02 + c44 * j04;

738  float h11 = c11 + c21 * j12 + c31 * j13 + c41 * j14;
739  float h12 = c21 + c22 * j12 + c32 * j13 + c42 * j14;

741  float h14 = c41 + c42 * j12 + c43 * j13 + c44 * j14;

743  float h20 = c20 + c40 * j24;
744  float h21 = c21 + c41 * j24;
745  float h22 = c22 + c42 * j24;
746  float h23 = c32 + c43 * j24;
747  float h24 = c42 + c44 * j24;
748 
749  c[0] = h00 + h02 * j02 + h04 * j04;
750 
751  c[1] = h10 + h12 * j02 + h14 * j04;
752  c[2] = h11 + h12 * j12 + h13 * j13 + h14 * j14;
753

755  c[4] = h21 + h22 * j12 + h23 * j13 + h24 * j14;
756  c[5] = h22 + h24 * j24;

760  c[8] = c32 + c43 * j24;

762

764  c[11] = c41 + c42 * j12 + c43 * j13 + c44 * j14;
765  c[12] = c42 + c44 * j24;
766  // c[13] = c43;
767  // c[14] = c44;
768  }
769 
770  float& mC22 = c[5];
771  float& mC33 = c[9];
772  float& mC40 = c[10];
773  float& mC41 = c[11];
774  float& mC42 = c[12];
775  float& mC43 = c[13];
776  float& mC44 = c[14];

779  bool maskMS = (CAMath::Abs(dL) < mMaterial.DLMax);
780  if (maskMS) {
781  dLmask = dL;
782  }
783  float dLabs = CAMath::Abs(dLmask);
784 
785  // Energy Loss
786  {
787  // std::cout<<"APPLY ENERGY LOSS!!!"<<std::endl;
788  float corr = 1.f - mMaterial.EP2 * dLmask;
789  float corrInv = 1.f / corr;

792  mTrk.pz *= corrInv;
793  mTrk.pt *= corrInv;
794  mTrk.p *= corrInv;
795  mTrk.qpt *= corr;
796

798

800  mC41 *= corr;
801  mC42 *= corr;
802  mC43 *= corr;
803  mC44 = mC44 * corr * corr + dLabs * mMaterial.sigmadE2;
804  }
805  // Multiple Scattering
806 
807  {

809  mC33 += dLabs * mMaterial.k33;
810  mC43 += dLabs * mMaterial.k43;
811  mC44 += dLabs * mMaterial.k44;
812  }
813 
814  return 0;

817 GPUd() void GPUTPCCompressionTrackModel::calculateMaterialCorrection()
818 {
819  const float mass = 0.13957f;
820 
821  float qpt = mTrk.qpt;
822  if (CAMath::Abs(qpt) > 20) {
823  qpt = 20;
824  }
825 
826  float w2 = (1.f + mTrk.dzds * mTrk.dzds); //==(P/pt)2
827  float pti2 = qpt * qpt;
828  if (pti2 < 1.e-4f) {
829  pti2 = 1.e-4f;
830  }
831 
832  float mass2 = mass * mass;
833  float beta2 = w2 / (w2 + mass2 * pti2);
834 
835  float p2 = w2 / pti2; // impuls 2
836  float betheRho = approximateBetheBloch(p2 / mass2) * mMaterial.rho;
837  float E = CAMath::Sqrt(p2 + mass2);
838  float theta2 = (14.1f * 14.1f / 1.e6f) / (beta2 * p2) * mMaterial.radLenInv;
839 
840  mMaterial.EP2 = E / p2;
841 
842  // Approximate energy loss fluctuation (M.Ivanov)
843 
844  const float knst = 0.07f; // To be tuned.
845  mMaterial.sigmadE2 = knst * mMaterial.EP2 * qpt;
846  mMaterial.sigmadE2 = mMaterial.sigmadE2 * mMaterial.sigmadE2;
847 
848  mMaterial.k22 = theta2 * w2;
849  mMaterial.k33 = mMaterial.k22 * w2;
850  mMaterial.k43 = 0.f;
851  mMaterial.k44 = theta2 * mTrk.dzds * mTrk.dzds * pti2;
852 
853  float br = (betheRho > 1.e-8f) ? betheRho : 1.e-8f;
854  mMaterial.DLMax = 0.3f * E / br;
855  mMaterial.EP2 *= betheRho;
856  mMaterial.sigmadE2 = mMaterial.sigmadE2 * betheRho; // + mMaterial.fK44;
857 }

859 GPUd() float GPUTPCCompressionTrackModel::approximateBetheBloch(float beta2)
860 {
861  //------------------------------------------------------------------
862  // This is an approximation of the Bethe-Bloch formula with
863  // the density effect taken into account at beta*gamma > 3.5
864  // (the approximation is reasonable only for solid materials)
865  //------------------------------------------------------------------
866 
867  const float log0 = log(5940.f);
868  const float log1 = log(3.5f * 5940.f);
869 
870  bool bad = (beta2 >= .999f) || (beta2 < 1.e-8f);
871 
872  if (bad) {
873  return 0.f;
874  }
875 
876  float a = beta2 / (1.f - beta2);
877  float b = 0.5f * log(a);
878  float d = 0.153e-3f / beta2;
879  float c = b - beta2;
880 
881  float ret = d * (log0 + b + c);
882  float case1 = d * (log1 + c);
883 
884  if (a > 3.5f * 3.5f) {
885  ret = case1;
886  }
887 
888  return ret;
889 }
