37 void DigitizerSpec::initDigitizerTask(framework::InitContext& ic)
38 {
39 
40  auto simulatePileup = ic.options().get<int>("pileup");
41  if (simulatePileup) { // set readout time and dead time parameters
42  mReadoutTime = o2::phos::PHOSSimParams::Instance().mReadoutTimePU; //PHOS readout time in ns
43  mDeadTime = o2::phos::PHOSSimParams::Instance().mDeadTimePU; //PHOS dead time (should include readout => mReadoutTime< mDeadTime)
44  } else {
45  mReadoutTime = o2::phos::PHOSSimParams::Instance().mReadoutTime; //PHOS readout time in ns
46  mDeadTime = o2::phos::PHOSSimParams::Instance().mDeadTime; //PHOS dead time (should include readout => mReadoutTime< mDeadTime)
47  }
48 
49  // init digitizer
50  mDigitizer.init();
51  auto mctruth = ic.options().get<bool>("mctruth");
52  if (!mctruth) {
53  mDigitizer.processMC(false);
54  }
55 
56  if (mHits) {
57  delete mHits;
58  }
59  mHits = new std::vector<Hit>();
60 }

76 void DigitizerSpec::run(framework::ProcessingContext& pc)
77 {
78  // read collision context from input
79  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
80  context->initSimChains(o2::detectors::DetID::PHS, mSimChains);
81  auto& timesview = context->getEventRecords();
82  LOG(debug) << "GOT " << timesview.size() << " COLLISSION TIMES";
83 
84  // if there is nothing to do ... return
85  int n = timesview.size();
86  if (n == 0) {
87  return;
88  }
89 
90  TStopwatch timer;
91  timer.Start();
92 
93  if (mDigitizer.runStartTime() == 0) { //not set yet
94  long runStartTime = timesview[0].getTimeNS();
95  mDigitizer.setRunStartTime(runStartTime); //set timestamp to access CCDB if necessary
96  }
97 
98  LOG(info) << " CALLING PHOS DIGITIZATION ";
99  std::vector<TriggerRecord> triggers;
100 
101  int indexStart = mDigitsOut.size();
102  auto& eventParts = context->getEventParts();
103  //if this is last stream of hits and we can write directly to final vector of digits? Otherwize use temporary vectors
104  mDigitsFinal.clear();
105  mDigitsTmp.clear();
106  bool isLastStream = true;
107  double eventTime = timesview[0].getTimeNS() - o2::phos::PHOSSimParams::Instance().mDeadTime; //checked above that list not empty
108  int eventId;
109  // loop over all composite collisions given from context
110  // (aka loop over all the interaction records)
111  for (int collID = 0; collID < n; ++collID) {
112  double dt = timesview[collID].getTimeNS() - eventTime; //start new PHOS readout, continue current or dead time?
113  if (dt > mReadoutTime && dt < mDeadTime) { //dead time, skip event
114  continue;
115  }
116 
117  if (dt >= o2::phos::PHOSSimParams::Instance().mDeadTime) { // start new event
118  //new event
119  eventTime = timesview[collID].getTimeNS();
120  dt = 0.;
121  eventId = collID;
122  }
123 
124  //Check if next event has to be added to this read-out
125  if (collID < n - 1) {
126  isLastStream = (timesview[collID + 1].getTimeNS() - eventTime > mReadoutTime);
127  } else {
128  isLastStream = true;
129  }
130 
131  // for each collision, loop over the constituents event and source IDs
132  // (background signal merging is basically taking place here)
133  // merge new hist to current digit list
134  auto part = eventParts[collID].begin();
135  while (part != eventParts[collID].end()) {
136  // get the hits for this event and this source
137  int source = part->sourceID;
138  int entry = part->entryID;
139  retrieveHits("PHSHit", source, entry);
140  part++;
141  if (part == eventParts[collID].end() && isLastStream) { //last stream, copy digits directly to output vector
142  mDigitizer.processHits(mHits, mDigitsFinal, mDigitsOut, mLabels, entry, source, dt);
143  mDigitsFinal.clear();
144  //finalyze previous event and clean
145  // Add trigger record
146  triggers.emplace_back(timesview[eventId], indexStart, mDigitsOut.size() - indexStart);
147  indexStart = mDigitsOut.size();
148  } else { //Fill intermediate digitvector
149  mDigitsTmp.swap(mDigitsFinal);
150  mDigitizer.processHits(mHits, mDigitsTmp, mDigitsFinal, mLabels, entry, source, dt);
151  mDigitsTmp.clear();
152  }
153  }
154  }
155  LOG(debug) << "Have " << mLabels.getNElements() << " PHOS labels ";
156  // here we have all digits and we can send them to consumer (aka snapshot it onto output)
157  pc.outputs().snapshot(Output{"PHS", "DIGITS", 0, Lifetime::Timeframe}, mDigitsOut);
158  pc.outputs().snapshot(Output{"PHS", "DIGITTRIGREC", 0, Lifetime::Timeframe}, triggers);
159  if (pc.outputs().isAllowed({"PHS", "DIGITSMCTR", 0})) {
160  pc.outputs().snapshot(Output{"PHS", "DIGITSMCTR", 0, Lifetime::Timeframe}, mLabels);
161  }
162 
163  // PHOS is always a triggering detector
164  const o2::parameters::GRPObject::ROMode roMode = o2::parameters::GRPObject::TRIGGERING;
165  LOG(debug) << "PHOS: Sending ROMode= " << roMode << " to GRPUpdater";
166  pc.outputs().snapshot(Output{"PHS", "ROMode", 0, Lifetime::Timeframe}, roMode);
167 
168  timer.Stop();
169  LOG(info) << "Digitization took " << timer.CpuTime() << "s";
170 
171  // we should be only called once; tell DPL that this process is ready to exit
172  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
173 }

175 DataProcessorSpec getPHOSDigitizerSpec(int channel, bool mctruth)
176 {
177  std::vector<OutputSpec> outputs;
178  outputs.emplace_back("PHS", "DIGITS", 0, Lifetime::Timeframe);
179  outputs.emplace_back("PHS", "DIGITTRIGREC", 0, Lifetime::Timeframe);
180  if (mctruth) {
181  outputs.emplace_back("PHS", "DIGITSMCTR", 0, Lifetime::Timeframe);
182  }
183  outputs.emplace_back("PHS", "ROMode", 0, Lifetime::Timeframe);
184 
185  // create the full data processor spec using
186  // a name identifier
187  // input description
188  // algorithmic description (here a lambda getting called once to setup the actual processing function)
189  // options that can be used for this processor (here: input file names where to take the hits)
190  return DataProcessorSpec{
191  "PHOSDigitizer", Inputs{InputSpec{"collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe}},
192  outputs,
193  AlgorithmSpec{o2::framework::adaptFromTask<DigitizerSpec>()},
194  Options{{"pileup", VariantType::Int, 1, {"whether to run in continuous time mode"}},
195  {"mctruth", VariantType::Bool, true, {"whether to process MC info"}}}};
196 }
