36 void ZDCDataReaderDPLSpec::run(ProcessingContext& pc)
37 {
38  mRawReader.clear();
39 
40  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload this means that the "delayed message"
41  // mechanism created it in absence of real data from upstream. Processor should send empty output to not block the workflow
42  {
43  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
44  std::vector<InputSpec> dummy{InputSpec{"dummy", ConcreteDataMatcher{o2::header::gDataOriginZDC, o2::header::gDataDescriptionRawData, 0xDEADBEEF}}};
45  for (const auto& ref : InputRecordWalker(pc.inputs(), dummy)) {
46  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
47  if (dh->payloadSize == 0) {
48  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
49  if (++contDeadBeef <= maxWarn) {
50  LOGP(warning, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
51  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize,
52  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
53  }
54  mRawReader.makeSnapshot(pc); // send empty output
55  return;
56  }
57  }
58  contDeadBeef = 0; // if good data, reset the counter
59  }
60 
61  DPLRawParser parser(pc.inputs(), o2::framework::select("zdc:ZDC/RAWDATA"));
62 
63  //>> update Time-dependent CCDB stuff, at the moment set the moduleconfig only once
64  if (!mRawReader.getModuleConfig()) {
65  /*long timeStamp = 0; // TIMESTAMP SHOULD NOT BE 0
66  mgr.setTimestamp(timeStamp);*/
67  auto& mgr = o2::ccdb::BasicCCDBManager::instance();
68  auto moduleConfig = mgr.get<o2::zdc::ModuleConfig>(o2::zdc::CCDBPathConfigModule);
69  if (!moduleConfig) {
70  LOG(fatal) << "Cannot module configuratio for timestamp " << mgr.getTimestamp();
71  return;
72  } else {
73  LOG(info) << "Loaded module configuration for timestamp " << mgr.getTimestamp();
74  }
75  mRawReader.setModuleConfig(moduleConfig);
76  mRawReader.setTriggerMask();
77  mRawReader.setVerifyTrigger(mVerifyTrigger);
78  LOG(info) << "Check of trigger condition during conversion is " << (mVerifyTrigger ? "ON" : "OFF");
79  }
80  uint64_t count = 0;
81  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
82  //Proccessing each page
83  count++;
84  auto rdhPtr = it.get_if<o2::header::RAWDataHeader>();
85  gsl::span<const uint8_t> payload(it.data(), it.size());
86  mRawReader.processBinaryData(payload, rdhPtr->linkID);
87  }
88  LOG(info) << "Pages: " << count;
89  mRawReader.accumulateDigits();
90  mRawReader.makeSnapshot(pc);
91 }
