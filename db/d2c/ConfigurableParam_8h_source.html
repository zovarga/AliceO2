32 // Base class for a configurable parameter.
33 //
34 // A configurable parameter (ConfigurableParameter) is a simple class, defining
35 // a few (pod) properties/members which are registered
36 // in a global (boost) property tree / structure.
37 //
38 // The features that we provide here are:
39 // *) Automatic translation from C++ data description to INI/JSON/XML
40 // format via ROOT introspection and boost property trees and
41 // the possibility to readably save the configuration
42 // *) Serialization/Deserialization into ROOT binary blobs (for the purpose
43 // of writing/retrieving parameters from CCDB and to pass parameters along the processing chain)
44 // *) Automatic integration of sub-classes into a common configuration
45 // *) Be able to query properties from high level interfaces (just knowing
46 // *) Be able to set properties from high-level interfaces (and modifying the underlying
47 // C++ object)
48 // *) Automatic ability to modify parameters from the command-line
49 // *) Keeping track of the provenance of individual parameter values; The user is able
50 // to see whether is parameter is defaulted-code-value/coming-from-CCDB/coming-from-comandline
51 //
52 // Note that concrete parameter sub-classes **must** be implemented
53 // by inheriting from ConfigurableParamHelper and not from this class.
54 //
55 // ---------------------
56 // Example: To define a parameter class TPCGasParameters, one does the following:
57 //
58 // class TPCGasParamer : public ConfigurableParamHelper<TPCGasParameter>
59 // {
60 // public:
61 // double getGasDensity() const { return mGasDensity; }
62 // private: // define properties AND default values
63 // double mGasDensity = 1.23;
64 // int mGasMaterialID = 1;
65 //
66 // O2ParamDef(TPCGasParameter, TPCGas); // a macro implementing some magic
67 // }
68 //
69 //
70 // We can now query the parameters in various ways
71 // - All parameter classes are singletons and we can say: TPCGasParameter::Instance().getGasDensity();
72 // - We can query by key (using classname + parameter name) from the global registry:
73 // - ConfigurableParameter::getValueAs<double>("TPCGas", "mGasDensity");
74 //
75 // We can modify the parameters via the global registry together with an automatic syncing
76 // of the underlying C++ object:
77 // - ConfigurableParameter::setValue("TPCGas.mGasDensity", "0.5");
78 //
79 // - TPCGasParameter::Instance().getGasParameter() will now return 0.5;
80 //
81 // This feature allows to easily modify parameters at run-time via a textual representation
82 // (for example by giving strings on the command line)
83 //
84 // The collection of all parameter keys and values can be stored to a human/machine readable
85 // file
86 // - ConfigurableParameter::writeJSON("thisconfiguration.json")

88 struct EnumLegalValues {
89  std::vector<std::pair<std::string, int>> vvalues;
90 
91  bool isLegal(const std::string& value) const
92  {
93  for (auto& v : vvalues) {
94  if (v.first == value) {
95  return true;
96  }
97  }
98  return false;
99  }
100 
101  bool isLegal(int value) const
102  {
103  for (auto& v : vvalues) {
104  if (v.second == value) {
105  return true;
106  }
107  }
108  return false;
109  }
110 
111  std::string toString() const;
112  int getIntValue(const std::string& value) const;
113 };

198  template <typename T>
199  static void setValue(std::string const& mainkey, std::string const& subkey, T x)
200  {
201  if (!sIsFullyInitialized) {
202  initialize();
203  }
204  assert(sPtree);
205  try {
206  auto key = mainkey + "." + subkey;
207  if (sPtree->get_optional<std::string>(key).is_initialized()) {
208  sPtree->put(key, x);
209  auto changed = updateThroughStorageMap(mainkey, subkey, typeid(T), (void*)&x);
210  if (changed != EParamUpdateStatus::Failed) {
211  sValueProvenanceMap->find(key)->second = kRT; // set to runtime
212  }
213  }
214  } catch (std::exception const& e) {
215  std::cerr << "Error in setValue (T) " << e.what() << "\n";
216  }
217  }
218 
219  // specialized for std::string
220  // which means that the type will be converted internally
221  static void setValue(std::string const& key, std::string const& valuestring)
222  {
223  if (!sIsFullyInitialized) {
224  initialize();
225  }
226  assert(sPtree);
227  try {
228  if (sPtree->get_optional<std::string>(key).is_initialized()) {
229  sPtree->put(key, valuestring);
230  auto changed = updateThroughStorageMapWithConversion(key, valuestring);
231  if (changed != EParamUpdateStatus::Failed) {
232  sValueProvenanceMap->find(key)->second = kRT; // set to runtime
233  }
234  }
235  } catch (std::exception const& e) {
236  std::cerr << "Error in setValue (string) " << e.what() << "\n";
237  }
