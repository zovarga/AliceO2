42 void FT0CalibTimeSlewing::fillGraph(int channel, TH2F* histo)
43 {
44  LOG(info) << "FT0CalibTimeSlewing::fillGraph " << channel << " entries " << int(histo->GetEntries());
45  double shiftchannel = 0;
46  TH1D* hist_Proj = histo->ProjectionY();
47  TFitResultPtr res = hist_Proj->Fit("gaus", "SQ");
48  if ((Int_t)res == 0) {
49  shiftchannel = res->Parameter(1);
50  }
51  Double_t xgr[NUMBER_OF_HISTOGRAM_BINS_X] = {};
52  Double_t ygr[NUMBER_OF_HISTOGRAM_BINS_X] = {};
53  TH1D* proj = nullptr;
54  int nbins = 0;
55  for (int ibin = 1; ibin < NUMBER_OF_HISTOGRAM_BINS_X; ibin++) {
56  xgr[ibin] = histo->GetXaxis()->GetBinCenter(ibin);
57  proj = histo->ProjectionY(Form("proj_px%i", ibin), ibin, ibin + 1);
58  if (proj->GetEntries() < 500) {
59  ygr[ibin] = 0;
60  continue;
61  }
62  TFitResultPtr r = proj->Fit("gaus", "SQ");
63  if ((Int_t)r == 0) {
64  ygr[ibin] = r->Parameter(1) - shiftchannel;
65  }
66  nbins++;
67  LOG(info) << "channel " << channel << " bin " << ibin << " x " << xgr[ibin] << " y " << ygr[ibin] << " ent " << proj->GetEntries() << " sigma " << r->Parameter(2) << " shiftchannel " << shiftchannel;
68  }
69  TGraph* grTimeAmp = new TGraph(nbins + 5, xgr, ygr);
70  mTimeSlewing[channel] = *grTimeAmp;
71 }
