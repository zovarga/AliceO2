301 inline void hash_combine(std::size_t& seed) {}
302 
303 template <typename T, typename... Rest>
304 inline void hash_combine(std::size_t& seed, const T& v, Rest... rest)
305 {
306  std::hash<T> hasher;
307  seed ^= hasher(v) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
308  hash_combine(seed, rest...);
309 }
