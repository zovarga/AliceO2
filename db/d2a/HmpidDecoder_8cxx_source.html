185 int HmpidDecoder::checkType(uint32_t wp, int* p1, int* p2, int* p3, int* p4)
186 {
187  if ((wp & 0x0000ffff) == 0x000036A8 || (wp & 0x0000ffff) == 0x000032A8 || (wp & 0x0000ffff) == 0x000030A0 || (wp & 0x0800ffff) == 0x080010A0) {
188  *p2 = (wp & 0x03ff0000) >> 16; // Number of words of row
189  *p1 = wp & 0x0000ffff;
190  return (WTYPE_ROW);
191  }
192  if ((wp & 0xfff00000) >> 20 == 0xAB0) {
193  *p2 = (wp & 0x000fff00) >> 8; // Number of words of Segment
194  *p1 = (wp & 0xfff00000) >> 20;
195  *p3 = wp & 0x0000000F;
196  if (*p3 < 4 && *p3 > 0) {
197  return (WTYPE_EOS);
198  }
199  }
200  // #EX MASK Raul 0x3803FF80 # ex mask 0xF803FF80 - this is EoE marker 0586800B0
201  if ((wp & 0x0803FF80) == 0x08000080) {
202  *p1 = (wp & 0x07c00000) >> 22;
203  *p2 = (wp & 0x003C0000) >> 18;
204  *p3 = (wp & 0x0000007F);
205  if (*p1 < 25 && *p2 < 11) {
206  return (WTYPE_EOE);
207  }
208  }
209  if ((wp & 0x08000000) == 0) { // # this is a pad
210  // PAD:0000.0ccc.ccdd.ddnn.nnnn.vvvv.vvvv.vvvv :: c=col,d=dilo,n=chan,v=value
211  *p1 = (wp & 0x07c00000) >> 22;
212  *p2 = (wp & 0x003C0000) >> 18;
213  *p3 = (wp & 0x0003F000) >> 12;
214  *p4 = (wp & 0x00000FFF);
215  if (*p1 > 0 && *p1 < 25 && *p2 > 0 && *p2 < 11 && *p3 < 48) {
216  return (WTYPE_PAD);
217  }
218  } else {
219  return (WTYPE_NONE);
220  }
221  return (WTYPE_NONE);
222 }

486 void HmpidDecoder::decodePage(uint32_t** streamBuf)
487 {
488  int equipmentIndex;
489  try {
490  getHeaderFromStream(streamBuf);
491  } catch (int e) {
492  // The stream end !
493  LOG(debug) << "End main decoding loop !";
494  throw TH_BUFFEREMPTY;
495  }
496  try {
497  decodeHeader(*streamBuf, &equipmentIndex);
498  } catch (int e) {
499  LOG(error) << "Failed to decode the Header !";
500  throw TH_WRONGHEADER;
501  }
502 
503  HmpidEquipment* eq = evaluateHeaderContents(equipmentIndex);
504 
505  uint32_t wpprev = 0;
506  uint32_t wp = 0;
507  int newOne = true;
508  int p1, p2, p3, p4;
509  int error;
510  int type;
511  bool isIt;
512 
513  int payIndex = 0;
514  while (payIndex < mNumberWordToRead) { //start the payload loop word by word
515  if (newOne == true) {
516  wpprev = wp;
517  if (!getWordFromStream(&wp)) { // end the stream
518  break;
519  }
520  type = checkType(wp, &p1, &p2, &p3, &p4);
521  if (type == WTYPE_NONE) {
522  if (eq->mWillBePad == true) { // try to recover the first pad !
523  type = checkType((wp & 0xF7FFFFFF), &p1, &p2, &p3, &p4);
524  if (type == WTYPE_PAD && p3 == 0 && eq->mWordsPerDilogicCounter == 0) {
525  newOne = false; // # reprocess as pad
526  continue;
527  }
528  }
529  eq->setError(ERR_NOTKNOWN);
530  LOG(debug) << "Equip=" << mEquipment << sErrorDescription[ERR_NOTKNOWN] << " [" << wp << "]";
531  eq->mWordsPerRowCounter++;
532  eq->mWordsPerSegCounter++;
533  payIndex++;
534  continue;
535  }
536  }
537  if (mEquipment == 8) {
538  LOG(info) << "Event" << eq->mEventNumber << " >" << std::hex << wp << std::dec << "<" << type;
539  }
540  if (eq->mWillBeRowMarker == true) { // #shoud be a Row Marker
541  if (type == WTYPE_ROW) {
542  eq->mColumnCounter++;
543  eq->mWordsPerSegCounter++;
544  eq->mRowSize = p2;
545  switch (p2) {
546  case 0: // Empty column
547  eq->setError(ERR_ROWMARKEMPTY);
548  LOG(debug) << "Equip=" << mEquipment << sErrorDescription[ERR_ROWMARKEMPTY] << " col=" << (eq->mSegment) * 8 + eq->mColumnCounter << "[" << p1 << "]";
549  eq->mWillBeRowMarker = true;
550  break;
551  case 0x3FF: // Error in column
552  eq->setError(ERR_ROWMARKERROR);
553  LOG(debug) << "Equip=" << mEquipment << sErrorDescription[ERR_ROWMARKERROR] << " col=" << (eq->mSegment) * 8 + eq->mColumnCounter << "[" << p1 << "]";
554  eq->mWillBeRowMarker = true;
555  break;
556  case 0x3FE: // Masked column
557  LOG(info) << "Equip=" << mEquipment << "The column=" << (eq->mSegment) * 8 + eq->mColumnCounter << " is Masked !";
558  eq->mWillBeRowMarker = true;
559  break;
560  default:
561  eq->mWillBeRowMarker = false;
562  eq->mWillBePad = true;
563  break;
564  }
565  newOne = true;
566  } else {
567  if (wp == wpprev) {
568  eq->setError(ERR_DUPLICATEPAD);
569  LOG(debug) << "Equip=" << mEquipment << sErrorDescription[ERR_DUPLICATEPAD] << " col=" << (eq->mSegment) * 8 + eq->mColumnCounter << "[" << p1 << "]";
570  newOne = true;
571  } else if (type == WTYPE_EOE) { // # Could be a EoE
572  eq->mColumnCounter++;
573  eq->setError(ERR_ROWMARKWRONG);
574  eq->mWillBeRowMarker = false;
575  eq->mWillBePad = true;
576  newOne = true;
577  } else if (type == WTYPE_PAD) { //# Could be a PAD
578  eq->mColumnCounter++;
579  eq->setError(ERR_ROWMARKLOST);
580  LOG(debug) << "Equip=" << mEquipment << sErrorDescription[ERR_ROWMARKLOST] << " col=" << (eq->mSegment) * 8 + eq->mColumnCounter << "[" << p1 << "]";
581  eq->mWillBeRowMarker = false;
582  eq->mWillBePad = true;
583  newOne = true;
584  } else if (type == WTYPE_EOS) { // # Could be a EoS
585  eq->mWillBeRowMarker = false;
586  eq->mWillBeSegmentMarker = true;
587  newOne = false;
588  } else {
589  eq->mColumnCounter++;
590  eq->setError(ERR_ROWMARKLOST);
591  LOG(debug) << "Equip=" << mEquipment << sErrorDescription[ERR_ROWMARKLOST] << " col=" << (eq->mSegment) * 8 + eq->mColumnCounter << "[" << p1 << "]";
592  eq->mWillBeRowMarker = false;
593  eq->mWillBePad = true;
594  newOne = true;
595  }
596  }
597  } else if (eq->mWillBePad == true) { // # We expect a pad
598  //# PAD:0000.0ccc.ccdd.ddnn.nnnn.vvvv.vvvv.vvvv :: c=col,d=dilo,n=chan,v=value
599  // c = 1..24 d = 1..10 n = 0..47
600  if (type == WTYPE_PAD) {
601  newOne = true;
602  if (wp == wpprev) {
603  eq->setError(ERR_DUPLICATEPAD);
604  LOG(debug) << "Equip=" << mEquipment << sErrorDescription[ERR_DUPLICATEPAD] << " col=" << (eq->mSegment) * 8 + eq->mColumnCounter << "[" << p1 << "]";
605  } else if (p1 != (eq->mSegment * 8 + eq->mColumnCounter)) { // # Manage
606  // We try to recover the RowMarker misunderstanding
607  isIt = isRowMarker(wp, &error, &p2, &p1);
608  if (isIt == true && error == false) {
609  type = WTYPE_ROW;
610  newOne = false;
611  eq->mWillBeEoE = true;
612  eq->mWillBePad = false;
613  } else {
614  LOG(debug) << "Equip=" << mEquipment << " Mismatch in column"
615  << " col=" << (eq->mSegment) * 8 + eq->mColumnCounter << "[" << p1 << "]";
616  eq->mColumnCounter = p1 % 8;
617  }
618  } else {
619  setPad(eq, p1 - 1, p2 - 1, p3, p4);
620  if (mEquipment == 8) {
