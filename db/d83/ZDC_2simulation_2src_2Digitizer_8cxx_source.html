30 Digitizer::ModuleConfAux::ModuleConfAux(const Module& md) : id(md.id)
31 {
32  if (md.id < 0 || md.id >= NModules) {
33  LOG(fatal) << "Module id = " << md.id << " not in allowed range [0:" << NModules << ")";
34  }
35  // construct aux helper from full module description
36  for (int ic = Module::MaxChannels; ic--;) {
37  if (md.channelID[ic] >= IdDummy) {
38  if (md.readChannel[ic]) {
39  readChannels |= 0x1 << (NChPerModule * md.id + ic);
40  }
41  if (md.trigChannel[ic]) {
42  trigChannels |= 0x1 << (NChPerModule * md.id + ic);
43  }
44  }
45  }

49 // this will process hits and fill the digit vector with digits which are finalized
50 void Digitizer::process(const std::vector<o2::zdc::Hit>& hits,
51  std::vector<o2::zdc::BCData>& digitsBC,
52  std::vector<o2::zdc::ChannelData>& digitsCh,
53  o2::dataformats::MCTruthContainer<o2::zdc::MCLabel>& labels)
54 {
55  // loop over all hits and produce digits
56  LOG(debug) << "Processing IR = " << mIR << " | NHits = " << hits.size();
57 
58  flush(digitsBC, digitsCh, labels); // flush cached signal which cannot be affect by new event
59 
60  for (auto& hit : hits) {
61 
62  std::array<o2::InteractionRecord, NBC2Cache> cachedIR;
63  // hit of given IR can con
64  // for each hit find out sector + detector information
65  int detID = hit.GetDetectorID();
66  int secID = hit.getSector();
67  float nPhotons;
68  if (detID == ZEM) {
69  // ZEM calorimeters have only common PM
70  nPhotons = hit.getPMCLightYield();
71  } else {
72  nPhotons = (secID == Common) ? hit.getPMCLightYield() : hit.getPMQLightYield();
73  }
74  if (!nPhotons) {
75  continue;
76  }
77  if (nPhotons < 0 || nPhotons > 1e6) {
78  int chan = toChannel(detID, secID);
79  LOG(error) << "Anomalous number of photons " << nPhotons << " for channel " << chan << '(' << channelName(chan) << ')';
80  continue;
81  }
82 
83  double hTime = hit.GetTime() - getTOFCorrection(detID); // account for TOF to detector
84  hTime += mIR.getTimeNS();
85  //
86  o2::InteractionRecord irHit(hTime); // BC in which the hit appears (might be different from interaction BC for slow particles)
87 
88  // nominal time of the BC to which the hit will be attributed
89  double bcTime = o2::InteractionRecord::bc2ns(irHit.bc, irHit.orbit);
90 
91  int nCachedIR = 0;
92  for (int i = BCCacheMin; i < BCCacheMax + 1; i++) {
93  double tNS = hTime + o2::constants::lhc::LHCBunchSpacingNS * i;
94  cachedIR[nCachedIR].setFromNS(tNS);
95  if (tNS < 0 && cachedIR[nCachedIR] > irHit) {
96  continue; // don't go to negative BC/orbit (it will wrap)
97  }
98  getCreateBCCache(cachedIR[nCachedIR++]); // ensure existence of cached container
99  }
100  auto channel = toChannel(detID, secID);
101  phe2Sample(nPhotons, hit.getParentID(), hTime, cachedIR, nCachedIR, channel);
102  // if digit for this sector does not exist, create one otherwise add to it
103  }
104 }

106 //______________________________________________________________________________
107 void Digitizer::flush(std::vector<o2::zdc::BCData>& digitsBC,
108  std::vector<o2::zdc::ChannelData>& digitsCh,
109  o2::dataformats::MCTruthContainer<o2::zdc::MCLabel>& labels)
110 {
111  // do we have something to flush? We can do this only for cached BC data which is distanced from currently processed BC by NBCReadOut
112  int lastDoneBCid = -1, diff2Last = 0;
113  int nCached = mCache.size();
114  if (nCached < 1) {
115  return;
116  }
117  if (mIR.differenceInBC(mCache.back()) > -BCCacheMin) {
118  LOG(debug) << "Generating new pedestal BL fluct. for BC range " << mCache.front() << " : " << mCache.back();
119  generatePedestal();
120  } else {
121  return;
122  }
123  o2::InteractionRecord ir0(mCache.front());
124  int cacheSpan = 1 + mCache.back().differenceInBC(ir0);
125  LOG(debug) << "Cache spans " << cacheSpan << " with " << nCached << " BCs cached";
126 
127  mFastCache.clear();
128  mFastCache.resize(cacheSpan, nullptr);
129  mStoreChanMask.clear();
130  mStoreChanMask.resize(cacheSpan + mNBCAHead, 0);
131 
132  for (int ibc = nCached; ibc--;) { // digitize BCs which might not be affected by future events
133  auto& bc = mCache[ibc];
134  lastDoneBCid = ibc;
135  if (!bc.digitized) {
136  digitizeBC(bc);
137  }
138  int bcSlot = mCache[ibc].differenceInBC(ir0);
139  mFastCache[bcSlot] = &mCache[ibc]; // add to fast access cache
140  }
141  mDummyBC.clear();
142  digitizeBC(mDummyBC);
143 
144  // check trigger condition for digitized BCs
145  for (int ibc = 0; ibc < cacheSpan; ibc++) {
146  if (mFastCache[ibc] && !mFastCache[ibc]->triggerChecked) {
147  triggerBC(ibc);
148  }
149  } // all allowed BCs are checked for trigger
150 
151  // store triggered BC with requested few BCs ahead
152  for (int ibc = -mNBCAHead; ibc < cacheSpan; ibc++) {
153  auto bcr = mStoreChanMask[ibc + mNBCAHead];
154  if (!bcr) {
155  continue; // nothing to store for this BC
156  }
157  BCCache* bcPtr = nullptr;
158  if (ibc < 0 || !mFastCache[ibc]) {
159  mDummyBC = ir0 + ibc; // fix the IR of the dummy BC
160  bcPtr = &mDummyBC;
161  } else {
162  bcPtr = mFastCache[ibc];
163  }
164  storeBC(*bcPtr, bcr, digitsBC, digitsCh, labels);
165  } // all allowed BCs are checked for trigger
166 
167  // clean cache for BCs which are not needed anymore
168  LOG(debug) << "Cleaning cache";
169  mCache.erase(mCache.begin(), mCache.end());
170 }

173 void Digitizer::generatePedestal()
174 {
175  for (int idet : {ZNA, ZPA, ZNC, ZPC}) {
176  int chanSum = toChannel(idet, Sum);
177  mPedestalBLFluct[chanSum] = 0.;
178  int comm = toChannel(idet, Common);
179  mPedestalBLFluct[comm] = gRandom->Gaus(0, mSimCondition->channels[comm].pedestalFluct);
180  for (int ic : {Ch1, Ch2, Ch3, Ch4}) {
181  int chan = toChannel(idet, ic);
182  mPedestalBLFluct[chanSum] += mPedestalBLFluct[chan] = gRandom->Gaus(0, mSimCondition->channels[chan].pedestalFluct);
183  }
184  }
185  mPedestalBLFluct[IdZEM1] = gRandom->Gaus(0, mSimCondition->channels[IdZEM1].pedestalFluct);
186  mPedestalBLFluct[IdZEM2] = gRandom->Gaus(0, mSimCondition->channels[IdZEM2].pedestalFluct);
187 }

189 //______________________________________________________________________________
190 void Digitizer::digitizeBC(BCCache& bc)
191 {
192  auto& bcdata = bc.data;
193  auto& bcdigi = bc.digi;
194  // apply gain
195  for (int idet : {ZNA, ZPA, ZNC, ZPC}) {
196  for (int ic : {Common, Ch1, Ch2, Ch3, Ch4}) {
197  int chan = toChannel(idet, ic);
198  auto gain = mSimCondition->channels[chan].gain;
199  for (int ib = NTimeBinsPerBC; ib--;) {
200  bcdata[chan][ib] *= gain;
201  }
202  }
203  }
204  for (int ib = NTimeBinsPerBC; ib--;) {
205  bcdata[IdZEM1][ib] *= mSimCondition->channels[IdZEM1].gain;
206  bcdata[IdZEM2][ib] *= mSimCondition->channels[IdZEM2].gain;
207  }
208  // Prepare sum of towers before adding noise
209  for (int ib = NTimeBinsPerBC; ib--;) {
210  bcdata[IdZNASum][ib] = mSimCondition->channels[IdZNASum].gain *
211  (bcdata[IdZNA1][ib] + bcdata[IdZNA2][ib] + bcdata[IdZNA3][ib] + bcdata[IdZNA4][ib]);
212  bcdata[IdZPASum][ib] = mSimCondition->channels[IdZPASum].gain *
213  (bcdata[IdZPA1][ib] + bcdata[IdZPA2][ib] + bcdata[IdZPA3][ib] + bcdata[IdZPA4][ib]);
214  bcdata[IdZNCSum][ib] = mSimCondition->channels[IdZNCSum].gain *
215  (bcdata[IdZNC1][ib] + bcdata[IdZNC2][ib] + bcdata[IdZNC3][ib] + bcdata[IdZNC4][ib]);
216  bcdata[IdZPCSum][ib] = mSimCondition->channels[IdZPCSum].gain *
217  (bcdata[IdZPC1][ib] + bcdata[IdZPC2][ib] + bcdata[IdZPC3][ib] + bcdata[IdZPC4][ib]);
218  }
219  // Digitize the signals connected to each channel of the different modules
220  for (const auto& md : mModuleConfig->modules) {
221  if (md.id >= 0 && md.id < NModules) {
222  for (int ic = Module::MaxChannels; ic--;) {
223  int id = md.channelID[ic];
224  if (id >= 0 && id < NChannels) {
225  const auto& chanConf = mSimCondition->channels[id];
226  auto pedBaseLine = mSimCondition->channels[id].pedestal;
227  // Geographical position of signal in the setup
228  auto ipos = NChPerModule * md.id + ic;
229  for (int ib = NTimeBinsPerBC; ib--;) {
230  bcdigi[ipos][ib] = bcdata[id][ib] + gRandom->Gaus(pedBaseLine + mPedestalBLFluct[id], chanConf.pedestalNoise);
231  int adc = std::nearbyint(bcdigi[ipos][ib]);
232  bcdigi[ipos][ib] = adc < ADCMax ? (adc > ADCMin ? adc : ADCMin) : ADCMax;
233  }
234  LOG(debug) << "md " << md.id << " ch " << ic << " sig " << id << " " << ChannelNames[id]
235  << bcdigi[ipos][0] << " " << bcdigi[ipos][1] << " " << bcdigi[ipos][2] << " " << bcdigi[ipos][3] << " " << bcdigi[ipos][4] << " " << bcdigi[ipos][5] << " "
236  << bcdigi[ipos][6] << " " << bcdigi[ipos][7] << " " << bcdigi[ipos][8] << " " << bcdigi[ipos][9] << " " << bcdigi[ipos][10] << " " << bcdigi[ipos][11];
237  }
238  }
239  }
240  }
241  bc.digitized = true;
242 }

244 //______________________________________________________________________________
245 bool Digitizer::triggerBC(int ibc)
246 {
247  // check trigger for the cached BC in the position ibc
248  auto& bcCached = *mFastCache[ibc];
249 
250  LOG(debug) << "CHECK TRIGGER " << ibc << " IR=" << bcCached;
251 
252  // Check trigger condition regardless of run type, will apply later the trigger mask
253  for (const auto& md : mModuleConfig->modules) {
254  if (md.id >= 0 && md.id < NModules) {
255  for (int ic = Module::MaxChannels; ic--;) {
256  //int id=md.channelID[ic];
257  auto trigCh = md.trigChannelConf[ic];
258  int id = trigCh.id;
259  if (id >= 0 && id < NChannels) {
260  auto ipos = NChPerModule * md.id + ic;
261  int last1 = trigCh.last + 2;
262  bool okPrev = false;
263 #ifdef ZDC_DOUBLE_TRIGGER_CONDITION
264  // look for 2 consecutive bins (the 1st one spanning trigCh.first : trigCh.last range) so that
265  // signal[bin]-signal[bin+trigCh.shift] > trigCh.threshold
266  for (int ib = trigCh.first; ib < last1; ib++) { // ib may be negative, so we shift by offs and look in the ADC cache
267  int binF, bcFidx = ibc + binHelper(ib, binF);
268  int binL, bcLidx = ibc + binHelper(ib + trigCh.shift, binL);
269  const auto& bcF = (bcFidx < 0 || !mFastCache[bcFidx]) ? mDummyBC : *mFastCache[bcFidx];
270  const auto& bcL = (bcLidx < 0 || !mFastCache[bcLidx]) ? mDummyBC : *mFastCache[bcLidx];
271  bool ok = bcF.digi[ipos][binF] - bcL.digi[ipos][binL] > trigCh.threshold;
272  if (ok && okPrev) { // trigger ok!
273  bcCached.trigChanMask |= 0x1 << (NChPerModule * md.id + ic); // register trigger mask
274  LOG(debug) << bcF.digi[ipos][binF] << " - " << bcL.digi[ipos][binL] << " = " << bcF.digi[ipos][binF] - bcL.digi[ipos][binL] << " > " << trigCh.threshold;
275  LOG(debug) << " hit [" << md.id << "," << ic << "] " << int(id) << "(" << ChannelNames[id] << ") => " << bcCached.trigChanMask;
276  break;
277  }
278  okPrev = ok;
279  }
280 #else
281  bool okPPrev = false;
282  // look for 3 consecutive bins (the 1st one spanning trigCh.first : trigCh.last range) so that
283  // signal[bin]-signal[bin+trigCh.shift] > trigCh.threshold
284  for (int ib = trigCh.first - 1; ib < last1; ib++) { // ib may be negative, so we shift by offs and look in the ADC cache
285  int binF, bcFidx = ibc + binHelper(ib, binF);
286  int binL, bcLidx = ibc + binHelper(ib + trigCh.shift, binL);
287  const auto& bcF = (bcFidx < 0 || !mFastCache[bcFidx]) ? mDummyBC : *mFastCache[bcFidx];
288  const auto& bcL = (bcLidx < 0 || !mFastCache[bcLidx]) ? mDummyBC : *mFastCache[bcLidx];
289  bool ok = bcF.digi[ipos][binF] - bcL.digi[ipos][binL] > trigCh.threshold;
290  if (ok && okPrev && okPPrev) { // trigger ok!
291  bcCached.trigChanMask |= 0x1 << (NChPerModule * md.id + ic); // register trigger mask
292  LOG(debug) << bcF.digi[ipos][binF] << " - " << bcL.digi[ipos][binL] << " = " << bcF.digi[ipos][binF] - bcL.digi[ipos][binL] << " > " << trigCh.threshold;
293  LOG(debug) << " hit [" << md.id << "," << ic << "] " << int(id) << "(" << ChannelNames[id] << ") => " << bcCached.trigChanMask;
294  break;
295  }
296  okPPrev = okPrev;
297  okPrev = ok;
298  }
299 #endif
300  }
301  }
302  }
303  }
304 
305  // just check if this BC IR corresponds to external trigger
306  if (!mIRExternalTrigger.empty() && mIRExternalTrigger.front() == bcCached) {
307  bcCached.extTrig = ALICETriggerMask;
308  mIRExternalTrigger.pop_front(); // suppress accounted external trigger
309  }
310  // This works in autotrigger mode, partially for triggered mode
311  if (bcCached.trigChanMask & mTriggerableChanMask) { // there are triggered channels, flag modules/channels to read
312  for (int ibcr = ibc - mNBCAHead; ibcr <= ibc; ibcr++) {
313  auto& bcr = mStoreChanMask[ibcr + mNBCAHead];
314  for (const auto& mdh : mModConfAux) {
315  if (bcCached.trigChanMask & mdh.trigChannels) { // are there triggered channels in this module?
316  bcr |= mdh.readChannels; // flag channels to store
317  }
318  }
319  }
320  }
321  bcCached.triggerChecked = true;
322  return bcCached.trigChanMask;
323 }

326 void Digitizer::storeBC(const BCCache& bc, uint32_t chan2Store,
327  std::vector<o2::zdc::BCData>& digitsBC, std::vector<o2::zdc::ChannelData>& digitsCh,
328  o2::dataformats::MCTruthContainer<o2::zdc::MCLabel>& labels)
329 {
330  // store selected data of selected BC
331  if (!chan2Store) {
332  return;
333  }
334  LOG(debug) << "Storing ch: " << chanPattern(chan2Store) << " trigger: " << chanPattern(bc.trigChanMask) << " for BC " << bc;
335 
336  int first = digitsCh.size(), nSto = 0;
337  for (const auto& md : mModuleConfig->modules) {
338  if (md.id >= 0 && md.id < NModules) {
339  for (int ic = 0; ic < Module::MaxChannels; ic++) {
340  auto ipos = NChPerModule * md.id + ic;
341  if (chan2Store & (0x1 << ipos)) {
342  digitsCh.emplace_back(md.channelID[ic], bc.digi[ipos]);
343  nSto++;
344  }
345  }
346  }
347  }
348 
349  int nBC = digitsBC.size();
350  digitsBC.emplace_back(first, nSto, bc, chan2Store, bc.trigChanMask, bc.extTrig);
351  // TODO clarify if we want to store MC labels for all channels or only for stored ones
352  if (!mSkipMCLabels) {
353  for (const auto& lbl : bc.labels) {
354  if (chan2Store & (0x1 << lbl.getChannel())) {
355  labels.addElement(nBC, lbl);
356  }
357  }
358  }
359 }

362 void Digitizer::phe2Sample(int nphe, int parID, double timeHit, std::array<o2::InteractionRecord, NBC2Cache> const& cachedIR, int nCachedIR, int channel)
363 {
364  //function to simulate the waveform from no. of photoelectrons seen in a given sample
365  // for electrons at timeInSample wrt beginning of the sample
366 
367  double time0 = cachedIR[0].bc2ns(); // start time of the 1st cashed BC
368  const auto& chanConfig = mSimCondition->channels[channel];
369 
370  float timeDiff = time0 - timeHit;
371  int sample = (timeDiff - gRandom->Gaus(chanConfig.timePosition, chanConfig.timeJitter)) * ChannelSimCondition::ShapeBinWidthInv + chanConfig.ampMinID;
372  int ir = 0;
373  bool stop = false;
374 
375  do {
376  auto bcCache = getBCCache(cachedIR[ir]);
377  bool added = false;
378  for (int ib = 0; ib < NTimeBinsPerBC; ib++) {
379  if (sample >= chanConfig.shape.size()) {
380  stop = true;
381  break;
382  }
383  if (sample >= 0) {
384  auto signal = chanConfig.shape[sample] * nphe; // signal not accounting for the gain
385  (*bcCache).data[channel][ib] += signal;
386  added = true;
387  }
388  sample += o2::constants::lhc::LHCBunchSpacingNS / NTimeBinsPerBC * ChannelSimCondition::ShapeBinWidthInv; // sample increment for next bin
389  }
390  if (added) {
391  (*bcCache).labels.emplace_back(parID, mEventID, mSrcID, channel);
392  }
393  } while (++ir < nCachedIR && !stop);
394 }

397 o2::zdc::Digitizer::BCCache& Digitizer::getCreateBCCache(const o2::InteractionRecord& ir)
398 {
399  if (mCache.empty() || mCache.back() < ir) {
400  mCache.emplace_back();
401  auto& cb = mCache.back();
402  cb = ir;
403  return cb;
404  }
405  if (mCache.front() > ir) {
406  mCache.emplace_front();
407  auto& cb = mCache.front();
408  cb = ir;
409  return cb;
410  }
411 
412  for (auto cb = mCache.begin(); cb != mCache.end(); cb++) {
413  if ((*cb) == ir) {
414  return *cb;
415  }
416  if (ir < (*cb)) {
417  auto cbnew = mCache.emplace(cb); // insert new element before cb
418  (*cbnew) = ir;
419  return (*cbnew);
420  }
421  }
422  return mCache.front();
423 }

455 //______________________________________________________________________________
456 void Digitizer::refreshCCDB()
457 {
458  // fetch ccdb objects. TODO: decide if this stays here or goes to the Spec
459  auto& mgr = o2::ccdb::BasicCCDBManager::instance();
460  if (!mModuleConfig) { // load this only once
461  mModuleConfig = mgr.get<ModuleConfig>(CCDBPathConfigModule);
462  LOG(info) << "Loaded module configuration for timestamp " << mgr.getTimestamp();
463  // fetch trigger info
464  mTriggerConfig.clear();
465  mModConfAux.clear();
466  for (const auto& md : mModuleConfig->modules) {
467  if (md.id >= 0 && md.id < NModules) {
468  mModConfAux.emplace_back(md);
469  for (int ic = Module::MaxChannels; ic--;) {
470  // We consider all channels that can produce a hit
471  if (md.trigChannel[ic] || (md.trigChannelConf[ic].shift > 0 && md.trigChannelConf[ic].threshold > 0)) {
472  const auto& trgChanConf = md.trigChannelConf[ic];
473  if (trgChanConf.last + trgChanConf.shift + 1 >= NTimeBinsPerBC) {
474  LOG(error) << "Wrong trigger settings";
475  }
476  mTriggerConfig.emplace_back(trgChanConf);
477  // We insert in the trigger mask only the channels that are actually triggering
478  // Trigger mask is geographical, bit position is relative to the module and channel
479  // where signal is connected
480  if (md.trigChannel[ic]) {
481  LOG(info) << "Adding channel [" << md.id << "," << ic << "] " << int(trgChanConf.id) << '(' << channelName(trgChanConf.id) << ") as triggering one";
482  // TODO insert check if bit is already used. Should never happen
483  mTriggerableChanMask |= 0x1 << (NChPerModule * md.id + ic);
484  } else {
485  LOG(info) << "Adding channel [" << md.id << "," << ic << "] " << int(trgChanConf.id) << '(' << channelName(trgChanConf.id) << ") as discriminator";
486  }
487  if (trgChanConf.first < mTrigBinMin) {
488  mTrigBinMin = trgChanConf.first;
489  }
490  if (trgChanConf.last + trgChanConf.shift > mTrigBinMax) {
491  mTrigBinMax = trgChanConf.last + trgChanConf.shift;
492  }
493  }
494  if (md.feeID[ic] < 0 || md.feeID[ic] >= NLinks) {
495  LOG(fatal) << "FEEID " << md.feeID[ic] << " not in allowed range [0:" << NLinks << ")";
496  }
497  }
498  } else {
499  LOG(fatal) << "Module id: " << md.id << " is out of range";

589 void Digitizer::assignTriggerBits(uint32_t ibc, std::vector<BCData>& bcData)
590 {
591  // Triggers refer to the HW trigger conditions (32 possible channels)
592 
593  uint32_t nbcTot = bcData.size();
594  auto& currBC = bcData[ibc];
595 
596  for (int is = -1; is < 4; is++) {
597  int ibc_peek = ibc + is;
598  if (ibc_peek < 0) {
599  continue;
600  }
601  if (ibc_peek >= nbcTot) {
602  break;
603  }
604  const auto& otherBC = bcData[ibc_peek];
605  auto diffBC = otherBC.ir.differenceInBC(currBC.ir);
606  if (diffBC < -1) {
607  continue;
608  }
609  if (diffBC > 3) {
610  break;
611  }
612  if (otherBC.ext_triggers && diffBC >= 0) {
613  for (int im = 0; im < NModules; im++) {
614  currBC.moduleTriggers[im] |= 0x1 << diffBC;
615  }
616  }
617  if (otherBC.triggers) {
618  // Assign trigger bits in payload
619  for (int im = 0; im < NModules; im++) {
620  uint32_t tmask = (0xf << (im * NChPerModule)) & mTriggerMask;
621  if (otherBC.triggers & tmask) {
622  currBC.moduleTriggers[im] |= 0x1 << (5 + diffBC);
623  }
624  }
625  }
626  }
627  // Printout before cleanup
628  //currBC.print(mTriggerMask);
629 }

631 void Digitizer::Finalize(std::vector<BCData>& bcData, std::vector<o2::zdc::OrbitData>& pData)
632 {
633  // Compute scalers for digitized data (works only for triggering channels)
634  for (Int_t ipd = 0; ipd < pData.size(); ipd++) {
635  for (int id = 0; id < NChannels; id++) {
636  pData[ipd].scaler[id] = 0;
637  }
638  }
639  for (int im = 0; im < NModules; im++) {
640  for (int ic = 0; ic < NChPerModule; ic++) {
641  uint32_t mask = 0x1 << (im * NChPerModule + ic);
642  auto id = mModuleConfig->modules[im].channelID[ic];
643  for (uint32_t ibc = 0; ibc < bcData.size(); ibc++) {
644  auto& currBC = bcData[ibc];
645  if ((currBC.triggers & mask) && (mReadoutMask & mask)) {
646  for (Int_t ipd = 0; ipd < pData.size(); ipd++) {
647  if (pData[ipd].ir.orbit == currBC.ir.orbit) {
648  pData[ipd].scaler[id]++;
649  }
650  }
651  }
652  }
653  }
654  }
655  // Default is to mask trigger bits, we can leave them for debugging
656  if (mMaskTriggerBits) {
657  for (uint32_t ibc = 0; ibc < bcData.size(); ibc++) {
658  auto& currBC = bcData[ibc];
659  for (int im = 0; im < NModules; im++) {
660  // Cleanup hits if module has no trigger
661  uint32_t mmask = 0xf << im;
662  if ((currBC.triggers & mTriggerMask & mmask) == 0) {
663  currBC.triggers = currBC.triggers & (~mmask);
664  }
665  }
666  // Cleanup trigger bits for channels that are not readout
667  currBC.triggers &= mReadoutMask;
668  // Printout after cleanup
669  //currBC.print(mTriggerMask);
670  }
671  }
672 }

675 void Digitizer::findEmptyBunches(const std::bitset<o2::constants::lhc::LHCMaxBunches>& bunchPattern)
676 {
677  mNEmptyBCs = 0;
678  for (int ib = 0; ib < o2::constants::lhc::LHCMaxBunches; ib++) {
679  int mb = (ib + 31) % o2::constants::lhc::LHCMaxBunches; // beam gas from back of calorimeter
680  int m1 = ib ? ((ib - 1) % o2::constants::lhc::LHCMaxBunches) : (o2::constants::lhc::LHCMaxBunches - 1); // previous bunch
681  int cb = ib; // current bunch crossing
682  int p1 = (ib + 1) % o2::constants::lhc::LHCMaxBunches; // colliding + 1
683  int p2 = (ib + 2) % o2::constants::lhc::LHCMaxBunches; // colliding + 2
684  int p3 = (ib + 3) % o2::constants::lhc::LHCMaxBunches; // colliding + 3
685  if (!(bunchPattern[mb] || bunchPattern[m1] || bunchPattern[cb] || bunchPattern[p1] || bunchPattern[p2] || bunchPattern[p3])) {
686  mNEmptyBCs++;
687  }
688  }
689  LOG(info) << "There are " << mNEmptyBCs << " clean empty bunches";
690 }

693 void Digitizer::updatePedestalReference(OrbitData& pdata)
694 {
695  // Compute or update baseline reference
696  for (uint32_t id = 0; id < NChannels; id++) {
697  auto base_m = mSimCondition->channels[id].pedestal; // Average pedestal
698  auto base_s = mSimCondition->channels[id].pedestalFluct; // Baseline oscillations
699  auto base_n = mSimCondition->channels[id].pedestalNoise; // Electronic noise
700  float ped = gRandom->Gaus(12. * mNEmptyBCs * base_m, 12. * 2. * base_s * std::sqrt(0.5 * mNEmptyBCs)) // 2 for fluctuation every 2 BCs
701  + gRandom->Gaus(0, base_n * std::sqrt(12. * mNEmptyBCs));
702  int16_t peds = std::round(8. * ped / mNEmptyBCs / 12.);
703  if (peds < SHRT_MIN) {
704  peds = SHRT_MIN;
705  } else if (peds > SHRT_MAX) {
706  peds = SHRT_MAX;
707  }
708  pdata.data[id] = peds;
709  }
