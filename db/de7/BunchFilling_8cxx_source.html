156 BunchFilling::Pattern BunchFilling::createPattern(const std::string& p)
157 {
158  // create bit pattern from string according to convention of Run1/2:
159  // The string has the following syntax:
160  // "25L 25(2H2LH 3(23HL))"
161  // - H/h -> 1 L/l -> 0
162  // - spaces, new lines are white characters
163  std::vector<unsigned char> input;
164  for (unsigned char c : p) {
165  if (c == '(' || c == ')' || std::isdigit(c)) {
166  input.push_back(c);
167  } else if (c == 'h' || c == 'H') {
168  input.push_back('H');
169  } else if (c == 'l' || c == 'L') {
170  input.push_back('L');
171  } else if (!std::isspace(c)) {
172  throw std::runtime_error(fmt::format("forbidden character ({}) in input string ({})", c, p));
173  }
174  }
175  input.push_back(0); // flag the end
176  BunchFilling::Pattern patt;
177  int ibit = 0, level = 0;
178  const unsigned char* in = input.data();
179  if (!parsePattern(in, patt, ibit, level)) {
180  throw std::runtime_error(fmt::format("failed to extract BC pattern from input string ({})", p));
181  }
182  return patt;
183 }

186 std::string BunchFilling::buckets2PatternString(const std::vector<int>& buckets, int ibeam)
187 {
188  // create bunches pattern string from filled buckets, in the format parsed by o2::BunchFilling::createPattern
189  Pattern patt;
190  std::string pattS;
191  for (int b : buckets) {
192  patt[o2::constants::lhc::LHCBunch2P2BC(b / 10, o2::constants::lhc::BeamDirection(ibeam))] = true;
193  }
194  int nh = 0;
195  for (int i = 0; i < o2::constants::lhc::LHCMaxBunches; i++) {
196  if (patt[i]) {
197  if (nh) {
198  pattS += fmt::format("{}L", nh);
199  nh = 0; // reset holes
200  }
201  pattS += "H";
202  } else {
203  nh++;
204  }
205  }
206  return pattS;
207 }

210 bool BunchFilling::parsePattern(const unsigned char*& input, BunchFilling::Pattern& patt, int& ibit, int& level)
211 {
212  // this is analog of AliTriggerBCMask::Bcm2Bits
213  level++;
214  int repetion = 1;
215  auto setBunch = [&patt, &ibit](bool v) {
216  if (ibit < patt.size()) {
217  patt[ibit++] = v;
218  } else {
219  throw std::runtime_error(fmt::format("provided pattern overflow max bunch {}", patt.size()));
220  }
221  };
222  while (1) {
223  if (*input == 0) {
224  if (level > 1) {
225  LOG(error) << "Missing )";
226  return false;
227  }
228  break;
229  }
230  if (*input == 'H') {
231  for (int i = 0; i < repetion; i++) {
232  setBunch(true);
233  }
234  repetion = 1;
235  input++;
236  } else if (*input == 'L') {
237  for (int i = 0; i < repetion; i++) {
238  setBunch(false);
239  }
240  repetion = 1;
241  input++;
242  } else if (std::isdigit(*input)) {
243  repetion = int((*input++) - '0');
244  while (*input && std::isdigit(*input)) {
245  repetion = repetion * 10 + int(*(input++) - '0');
246  }
247  } else if (*input == '(') {
248  input++;
249  auto ibit1 = ibit;
250  if (!parsePattern(input, patt, ibit, level)) {
251  return false;
252  }
253  auto ibit2 = ibit;
254  for (int i = 0; i < repetion - 1; i++) {
255  for (int j = ibit1; j < ibit2; j++) {
256  setBunch(patt[j]);
257  }
258  }
259  repetion = 1;
260  } else if (*input == ')') {
261  input++;
262  if (level <= 1) {
263  LOG(error) << "Incorrectly placed )";
264  return false;
265  }
266  break;
267  } else {
268  LOG(error) << "Unexpected symbol " << *input;
269  return false;
270  }
271  }
272  level--;
273  return true;
274 }
