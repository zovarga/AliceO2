171  static constexpr int binHelper(int ib, int& binInSiftedBC)
172  {
173  if (ib < 0) {
174  binInSiftedBC = NTimeBinsPerBC - 1 + (1 + ib) % NTimeBinsPerBC;
175  return (ib - (NTimeBinsPerBC - 1)) / NTimeBinsPerBC;
176  } else {
177  binInSiftedBC = ib % NTimeBinsPerBC;
178  return ib / NTimeBinsPerBC;
179  }
