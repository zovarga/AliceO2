112 template <typename CHARGESUM>
113 void UserLogicElinkDecoder<CHARGESUM>::append(uint64_t data50, uint8_t error, bool incomplete)
114 {
115 #ifdef ULDEBUG
116  debugHeader() << (*this) << fmt::format(" --> append50 {:013x} error {} incomplete {} data10={:d} {:d} {:d} {:d} {:d}\n", data50, error, incomplete, static_cast<uint10_t>(data50 & 0x3FF), static_cast<uint10_t>((data50 & 0xFFC00) >> 10), static_cast<uint10_t>((data50 & 0x3FF00000) >> 20), static_cast<uint10_t>((data50 & 0xFFC0000000) >> 30), static_cast<uint10_t>((data50 & 0x3FF0000000000) >> 40));
117 #endif
118 
119  if (isSync(data50)) {
120 #ifdef ULDEBUG
121  debugHeader() << (*this) << fmt::format(" --> SYNC word found {:013x}\n", data50);
122 #endif
123  clear();
124  transition(State::WaitingHeader);
125  return;
126  }
127 
128  auto data = data50;
129 
130  int i;
131  for (i = 0; i < 5; i++) {
132  bool packetEnd = append10(static_cast<uint10_t>(data & 0x3FF));
133  data >>= 10;
134 #ifdef ULDEBUG
135  if (incomplete) {
136  debugHeader() << (*this) << fmt::format(" --> incomplete {} packetEnd @i={}\n", incomplete, packetEnd, i);
137  }
138 #endif
139  if (hasError()) {
140 #ifdef ULDEBUG
141  debugHeader() << (*this) << " reset due to hasError\n";
142 #endif
143  reset();
144  break;
145  }
146  if (incomplete && packetEnd) {
147 #ifdef ULDEBUG
148  debugHeader() << (*this) << " stop due to isIncomplete\n";
149 #endif
150  break;
151  }
152  }
153 
154  if (incomplete && (i == 5) && (mState != State::WaitingSync)) {
155 #ifdef ULDEBUG
156  debugHeader() << (*this) << " data packet end not found when isIncomplete --> resetting\n";
157 #endif
158  sendError(static_cast<int8_t>(mSampaHeader.chipAddress()), static_cast<uint32_t>(ErrorBadIncompleteWord));
159  reset();
160  }

176 template <typename CHARGESUM>
177 bool UserLogicElinkDecoder<CHARGESUM>::append10(uint10_t data10)
178 {
179  bool result = false;
180 #ifdef ULDEBUG
181  debugHeader() << (*this) << fmt::format(" --> data10 {:d}\n", data10);
182 #endif
183  switch (mState) {
184  case State::WaitingHeader:
185  setHeaderPart(data10);
186  if (isHeaderComplete()) {
187  completeHeader();
188  if (isSync(mSampaHeader.uint64())) {
189  reset();
190  } else if (mSampaHeader.packetType() == SampaPacketType::HeartBeat) {
191  if (mSampaHeader.isHeartbeat()) {
192  sendHBPacket();
193  transition(State::WaitingHeader);
194  result = true;
195  } else {
196  mErrorMessage = "badly formatted Heartbeat packet";
197  sendError(static_cast<int8_t>(mSampaHeader.chipAddress()), static_cast<uint32_t>(ErrorBadHeartBeatPacket));
198  reset();
199  }
200  } else {
201  if (mSampaHeader.nof10BitWords() > 2) {
202  transition(State::WaitingSize);
203  } else {
204  reset();
205  }
206  }
207  }
208  break;
209  case State::WaitingSize:
210  if (moreWordsToRead()) {
211  setClusterSize(data10);
212  if (hasError()) {
213  return false;
214  }
215  transition(State::WaitingTime);
216  } else {
217  mErrorMessage = "WaitingSize but no more words";
218  return false;
219  }
220  break;
221  case State::WaitingTime:
222  if (moreWordsToRead()) {
223  setClusterTime(data10);
224  transition(State::WaitingSample);
225  } else {
226  mErrorMessage = "WaitingTime but no more words";
227  return false;
228  }
229  break;
230  case State::WaitingSample:
231  if (moreSampleToRead()) {
232  setSample(data10);
233  }
234  if (!moreSampleToRead()) {
235  if (moreWordsToRead()) {
236  transition(State::WaitingSize);
237  } else {
238  transition(State::WaitingHeader);
239  result = true;
240  }
241  }
242  break;
243  default:
244  break;
245  };
246  return result;
247 }

281 template <typename CHARGESUM>
282 void UserLogicElinkDecoder<CHARGESUM>::completeHeader()
283 {
284  uint64_t header{0};
285  for (auto i = 0; i < mHeaderParts.size(); i++) {
286  header += (static_cast<uint64_t>(mHeaderParts[i]) << (10 * i));
287  }
288 
289  mSampaHeader = SampaHeader(header);
290  mNof10BitWords = mSampaHeader.nof10BitWords();
291 
292 #ifdef ULDEBUG
293  debugHeader() << (*this) << fmt::format(" --> completeHeader {:013X}\n", header);
294  debugHeader() << "\n";
295  std::stringstream s(o2::mch::raw::asString(mSampaHeader));
296  std::string part;
297  while (std::getline(s, part, '\n')) {
298  debugHeader() << (*this) << part << "\n";
299  }
300  debugHeader() << "\n";
301 #endif
302 
303  mHeaderParts.clear();
304 }

367 template <typename CHARGESUM>
368 void UserLogicElinkDecoder<CHARGESUM>::setClusterSize(uint10_t value)
369 {
370  oneLess10BitWord();
371  mClusterSize = value;
372  if (CHARGESUM()()) {
373  mSamplesToRead = 2;
374  } else {
375  mSamplesToRead = mClusterSize;
376  }
377  int checkSize = mSamplesToRead + 2 - mSampaHeader.nof10BitWords();
378  mErrorMessage = std::nullopt;
379  if (mClusterSize == 0) {
380  mErrorMessage = "cluster size is zero";
381  sendError(static_cast<int8_t>(mSampaHeader.chipAddress()), static_cast<uint32_t>(ErrorBadClusterSize));
382  }
383  if (checkSize > 0) {
384  mErrorMessage = "number of samples bigger than nof10BitWords";
385  sendError(static_cast<int8_t>(mSampaHeader.chipAddress()), static_cast<uint32_t>(ErrorBadClusterSize));
386  }
387 #ifdef ULDEBUG
388  debugHeader() << (*this) << " --> size=" << mClusterSize << " samples=" << mSamplesToRead << "\n";
