204 inline void CalibdEdxTrackTopologySpline::setSplinesFromFile(TFile& inpf)
205 {
206  FlatObject::startConstruction();
207 
208  int buffSize = 0;
209  int offsets1[FSplines];
210  int offsets2[FSplines];
211 
212  for (unsigned int ireg = 0; ireg < FSplines; ++ireg) {
213  std::string splinename = fmt::format("spline_qMax_region{}", ireg);
214  SplineType* splineTmpqMax = SplineType::readFromFile(inpf, splinename.data());
215  mCalibSplinesqMax[ireg] = *splineTmpqMax;
216  buffSize = alignSize(buffSize, mCalibSplinesqMax[ireg].getBufferAlignmentBytes());
217  offsets1[ireg] = buffSize;
218  buffSize += mCalibSplinesqMax[ireg].getFlatBufferSize();
219  delete splineTmpqMax;
220  }
221 
222  for (unsigned int ireg = 0; ireg < FSplines; ++ireg) {
223  std::string splinename = fmt::format("spline_qTot_region{}", ireg);
224  SplineType* splineTmpqTot = SplineType::readFromFile(inpf, splinename.data());
225  mCalibSplinesqTot[ireg] = *splineTmpqTot;
226  buffSize = alignSize(buffSize, mCalibSplinesqTot[ireg].getBufferAlignmentBytes());
227  offsets2[ireg] = buffSize;
228  buffSize += mCalibSplinesqTot[ireg].getFlatBufferSize();
229  delete splineTmpqTot;
230  }
231 
232  FlatObject::finishConstruction(buffSize);
233 
234  for (unsigned int i = 0; i < FSplines; i++) {
235  mCalibSplinesqMax[i].moveBufferTo(mFlatBufferPtr + offsets1[i]);
236  }
237  for (unsigned int i = 0; i < FSplines; i++) {
238  mCalibSplinesqTot[i].moveBufferTo(mFlatBufferPtr + offsets2[i]);
239  }
240 }
