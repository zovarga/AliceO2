165 void EventStorage::unpackData(std::vector<TriggerRecord>& triggers, std::vector<Tracklet64>& tracklets, std::vector<Digit>& digits)
166 {
167  int digitcount = 0;
168  int trackletcount = 0;
169  for (auto& event : mEventRecords) {
170  tracklets.insert(std::end(tracklets), std::begin(event.getTracklets()), std::end(event.getTracklets()));
171  digits.insert(std::end(digits), std::begin(event.getDigits()), std::end(event.getDigits()));
172  triggers.emplace_back(event.getBCData(), digitcount, event.getDigits().size(), trackletcount, event.getTracklets().size());
173  digitcount += event.getDigits().size();
174  trackletcount += event.getTracklets().size();
175  }

178 void EventStorage::sendData(o2::framework::ProcessingContext& pc, bool generatestats)
179 {
180  //at this point we know the total number of tracklets and digits and triggers.
181  auto dataReadStart = std::chrono::high_resolution_clock::now();
182  uint64_t trackletcount = 0;
183  uint64_t digitcount = 0;
184  uint64_t triggercount = 0;
185  sumTrackletsDigitsTriggers(trackletcount, digitcount, triggercount);
186  std::vector<Tracklet64> tracklets;
187  tracklets.reserve(trackletcount);
188  std::vector<Digit> digits;
189  digits.reserve(digitcount);
190  std::vector<TriggerRecord> triggers;
191  triggers.reserve(triggercount);
192  for (auto& event : mEventRecords) {
193  //sort tracklets
194  event.sortByHCID();
195  //TODO do this sort in parallel over the events
196  tracklets.insert(std::end(tracklets), std::begin(event.getTracklets()), std::end(event.getTracklets()));
197  digits.insert(std::end(digits), std::begin(event.getDigits()), std::end(event.getDigits()));
198  triggers.emplace_back(event.getBCData(), digitcount, event.getDigits().size(), trackletcount, event.getTracklets().size());
199  digitcount += event.getDigits().size();
200  trackletcount += event.getTracklets().size();
201  }
202  //TODO change to adopt instead of having this additional copy.
203  //
204  pc.outputs().snapshot(o2::framework::Output{o2::header::gDataOriginTRD, "DIGITS", 0, o2::framework::Lifetime::Timeframe}, digits);
205  pc.outputs().snapshot(o2::framework::Output{o2::header::gDataOriginTRD, "TRACKLETS", 0, o2::framework::Lifetime::Timeframe}, tracklets);
206  pc.outputs().snapshot(o2::framework::Output{o2::header::gDataOriginTRD, "TRKTRGRD", 0, o2::framework::Lifetime::Timeframe}, triggers);
207  if (generatestats) {
208  accumulateStats();
209  pc.outputs().snapshot(o2::framework::Output{o2::header::gDataOriginTRD, "RAWSTATS", 0, o2::framework::Lifetime::Timeframe}, mTFStats);
210  }
211  mEventRecords.clear();
212  std::chrono::duration<double, std::micro> dataReadTime = std::chrono::high_resolution_clock::now() - dataReadStart;
213  LOG(debug) << "Preparing for sending and sending data took " << std::chrono::duration_cast<std::chrono::milliseconds>(dataReadTime).count() << "ms";
214  if (mPackagingTime != nullptr) {
215  mPackagingTime->Fill((int)std::chrono::duration_cast<std::chrono::microseconds>(dataReadTime).count());
216  }
217 }

219 void EventStorage::accumulateStats()
220 {
221  int eventcount = mEventRecords.size();
222  int sumtracklets = 0;
223  int sumdigits = 0;
224  //std::chrono::duration sumdigittime=std::chrono::steady_clock::duration::zero();
225  //std::chrono::duration sumtracklettime=std::chrono::steady_clock::duration::zero();
226  int sumdigittime = 0;
227  int sumtracklettime = 0;
228  int sumtime = 0;
229  uint64_t sumwordsrejected = 0;
230  uint64_t sumwordsread = 0;
231  for (auto event : mEventRecords) {
232  sumtracklets += event.mEventStats.mTrackletsFound;
233  sumdigits += event.mEventStats.mDigitsFound;
234  sumtracklettime += event.mEventStats.mTimeTakenForTracklets;
235  sumdigittime += event.mEventStats.mTimeTakenForDigits;
236  sumwordsrejected += event.mEventStats.mWordsRejected;
237  sumwordsread += event.mEventStats.mWordsRead;
238  }
239  if (eventcount != 0) {
240  mTFStats.mTrackletsPerEvent = sumtracklets / eventcount;
241  mTFStats.mDigitsPerEvent = sumdigits / eventcount;
242  mTFStats.mTimeTakenForTracklets = sumtracklettime;
243  mTFStats.mTimeTakenForDigits = sumdigittime;
244  }
245 }
