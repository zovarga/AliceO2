187  // add element for a particular dataindex
188  // at the moment only strictly consecutive modes are supported
189  void addElement(uint32_t dataindex, TruthElement const& element)
190  {
191  if (dataindex < mHeaderArray.size()) {
192  // look if we have something for this dataindex already
193  // must currently be the last one
194  if (dataindex != (mHeaderArray.size() - 1)) {
195  throw std::runtime_error("MCTruthContainer: unsupported code path");
196  }
197  } else {
198  // assert(dataindex == mHeaderArray.size());
199 
200  // add empty holes
201  int holes = dataindex - mHeaderArray.size();
202  assert(holes >= 0);
203  for (int i = 0; i < holes; ++i) {
204  mHeaderArray.emplace_back(mTruthArray.size());
205  }
206  // add a new one
207  mHeaderArray.emplace_back(mTruthArray.size());
208  }
209  mTruthArray.emplace_back(element);
210  }

234  // Add element at last position or for a previous index
235  // (at random access position).
236  // This might be a slow process since data has to be moved internally
237  // so this function should be used with care.
238  void addElementRandomAccess(uint32_t dataindex, TruthElement const& element)
239  {
240  if (dataindex >= mHeaderArray.size()) {
241  // a new dataindex -> push element at back
242 
243  // we still forbid to leave holes
244  assert(dataindex == mHeaderArray.size());
245 
246  mHeaderArray.resize(dataindex + 1);
247  mHeaderArray[dataindex] = mTruthArray.size();
248  mTruthArray.emplace_back(element);
249  } else {
250  // if appending at end use fast function
251  if (dataindex == mHeaderArray.size() - 1) {
252  addElement(dataindex, element);
253  return;
254  }
255 
256  // existing dataindex
257  // have to:
258  // a) move data;
259  // b) insert new element;
260  // c) adjust indices of all headers right to this
261  auto currentindex = mHeaderArray[dataindex].index;
262  auto lastindex = currentindex + getSize(dataindex);
263  assert(currentindex >= 0);
264 
265  // resize truth array
266  mTruthArray.resize(mTruthArray.size() + 1);
267  // move data (have to do from right to left)
268  for (int i = mTruthArray.size() - 1; i > lastindex; --i) {
269  mTruthArray[i] = mTruthArray[i - 1];
270  }
271  // insert new element
272  mTruthArray[lastindex] = element;
273 
274  // fix headers
275  for (uint32_t i = dataindex + 1; i < mHeaderArray.size(); ++i) {
276  auto oldindex = mHeaderArray[i].index;
277  mHeaderArray[i].index = (oldindex != (uint32_t)-1) ? oldindex + 1 : oldindex;
278  }
279  }
280  }

289  // merge another container to the back of this one
290  void mergeAtBack(MCTruthContainer<TruthElement> const& other)
291  {
292  const auto oldtruthsize = mTruthArray.size();
293  const auto oldheadersize = mHeaderArray.size();
294 
295  // copy from other
296  std::copy(other.mHeaderArray.begin(), other.mHeaderArray.end(), std::back_inserter(mHeaderArray));
297  std::copy(other.mTruthArray.begin(), other.mTruthArray.end(), std::back_inserter(mTruthArray));
298 
299  // adjust information of newly attached part
300  for (uint32_t i = oldheadersize; i < mHeaderArray.size(); ++i) {
301  mHeaderArray[i].index += oldtruthsize;
302  }
303  }

305  // merge part of another container ("n" entries starting from "from") to the back of this one
306  void mergeAtBack(MCTruthContainer<TruthElement> const& other, size_t from, size_t n)
307  {
308  const auto oldtruthsize = mTruthArray.size();
309  const auto oldheadersize = mHeaderArray.size();
310  auto endIdx = from + n;
311  assert(endIdx <= other.mHeaderArray.size());
312  const auto* headBeg = &other.mHeaderArray[from];
313  const auto* headEnd = headBeg + n;
314  const auto* trtArrBeg = &other.mTruthArray[other.getMCTruthHeader(from).index];
315  const auto* trtArrEnd = (endIdx == other.mHeaderArray.size()) ? (&other.mTruthArray.back()) + 1 : &other.mTruthArray[other.getMCTruthHeader(endIdx).index];
316 
317  // copy from other
318  std::copy(headBeg, headEnd, std::back_inserter(mHeaderArray));
319  std::copy(trtArrBeg, trtArrEnd, std::back_inserter(mTruthArray));
320  long offset = long(oldtruthsize) - other.getMCTruthHeader(from).index;
321  // adjust information of newly attached part
322  for (uint32_t i = oldheadersize; i < mHeaderArray.size(); ++i) {
323  mHeaderArray[i].index += offset;
324  }
325  }

331  template <typename ContainerType>
332  size_t flatten_to(ContainerType& container) const
333  {
334  size_t bufferSize = sizeof(FlatHeader) + sizeof(MCTruthHeaderElement) * mHeaderArray.size() + sizeof(TruthElement) * mTruthArray.size();
335  container.resize((bufferSize / sizeof(typename ContainerType::value_type)) + ((bufferSize % sizeof(typename ContainerType::value_type)) > 0 ? 1 : 0));
336  char* target = reinterpret_cast<char*>(container.data());
337  auto& flatheader = *reinterpret_cast<FlatHeader*>(target);
338  target += sizeof(FlatHeader);
339  flatheader.version = 1;
340  flatheader.sizeofHeaderElement = sizeof(MCTruthHeaderElement);
341  flatheader.sizeofTruthElement = sizeof(TruthElement);
342  flatheader.reserved = 0;
343  flatheader.nofHeaderElements = mHeaderArray.size();
344  flatheader.nofTruthElements = mTruthArray.size();
345  size_t copySize = flatheader.sizeofHeaderElement * flatheader.nofHeaderElements;
346  memcpy(target, mHeaderArray.data(), copySize);
347  target += copySize;
348  copySize = flatheader.sizeofTruthElement * flatheader.nofTruthElements;
349  memcpy(target, mTruthArray.data(), copySize);
350  return bufferSize;
351  }

356  void restore_from(const char* buffer, size_t bufferSize)
357  {
358  if (buffer == nullptr || bufferSize < sizeof(FlatHeader)) {
359  return;
360  }
361  auto* source = buffer;
362  auto& flatheader = *reinterpret_cast<FlatHeader const*>(source);
363  source += sizeof(FlatHeader);
364  if (bufferSize < sizeof(FlatHeader) + flatheader.sizeofHeaderElement * flatheader.nofHeaderElements + flatheader.sizeofTruthElement * flatheader.nofTruthElements) {
365  throw std::runtime_error("inconsistent buffer size: too small");
366  return;
367  }
368  if (flatheader.sizeofHeaderElement != sizeof(MCTruthHeaderElement) || flatheader.sizeofTruthElement != sizeof(TruthElement)) {
369  // not yet handled
370  throw std::runtime_error("member element sizes don't match");
371  }
372  // TODO: with a spectator memory ressource the vectors can be built directly
373  // over the original buffer, there is the implementation for a memory ressource
374  // working on a FairMQ message, here we would need two memory resources over
375  // the two ranges of the input buffer
376  // for now doing a copy
377  mHeaderArray.resize(flatheader.nofHeaderElements);
378  mTruthArray.resize(flatheader.nofTruthElements);
379  size_t copySize = flatheader.sizeofHeaderElement * flatheader.nofHeaderElements;
380  memcpy(mHeaderArray.data(), source, copySize);
381  source += copySize;
382  copySize = flatheader.sizeofTruthElement * flatheader.nofTruthElements;
383  memcpy(mTruthArray.data(), source, copySize);
384  }
