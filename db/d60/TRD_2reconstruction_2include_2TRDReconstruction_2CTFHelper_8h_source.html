37  : mTrigRec(trgRec), mTrkData(trkData), mDigData(digData), mTrkStart(trkData.size()), mDigStart(digData.size())
38  {

40  for (const auto& trg : mTrigRec) {

45  mDigStart[trg.getFirstDigit()] = true;
46  }

50  CTFHeader createHeader()
51  {
52  CTFHeader h{o2::detectors::DetID::TRD, 0, 1, 0, // dummy timestamp, version 1.0

67  template <typename I, typename D, typename T, int M = 1>
68  class _Iter
69  {
70  public:
71  using difference_type = int64_t;
72  using value_type = T;
73  using pointer = const T*;
74  using reference = const T&;
75  using iterator_category = std::random_access_iterator_tag;
76 
77  _Iter(const gsl::span<const D>& data, bool end = false) : mData(data), mIndex(end ? M * data.size() : 0){};
78  _Iter() = default;
79 
80  const I& operator++()
81  {
82  ++mIndex;
83  return (I&)(*this);
84  }
85 
86  const I& operator--()
87  {
88  mIndex--;
89  return (I&)(*this);
90  }
91 
92  difference_type operator-(const I& other) const { return mIndex - other.mIndex; }
93 
94  difference_type operator-(size_t idx) const { return mIndex - idx; }
95 
96  const I& operator-(size_t idx)
97  {
98  mIndex -= idx;
99  return (I&)(*this);
100  }
101 
102  bool operator!=(const I& other) const { return mIndex != other.mIndex; }
103  bool operator==(const I& other) const { return mIndex == other.mIndex; }
104  bool operator>(const I& other) const { return mIndex > other.mIndex; }
105  bool operator<(const I& other) const { return mIndex < other.mIndex; }
106 
107  protected:
108  gsl::span<const D> mData{};
109  size_t mIndex = 0;
110  };

113  // BC difference wrt previous if in the same orbit, otherwise the abs.value.
114  // For the very 1st entry return 0 (diff wrt 1st BC in the CTF header)
115  class Iter_bcIncTrig : public _Iter<Iter_bcIncTrig, TriggerRecord, uint16_t>
116  {
117  public:
118  using _Iter<Iter_bcIncTrig, TriggerRecord, uint16_t>::_Iter;
119  value_type operator*() const
120  {
121  if (mIndex) {
122  if (mData[mIndex].getBCData().orbit == mData[mIndex - 1].getBCData().orbit) {
123  return mData[mIndex].getBCData().bc - mData[mIndex - 1].getBCData().bc;
124  } else {
125  return mData[mIndex].getBCData().bc;
126  }
127  }
128  return 0;
129  }
130  };
