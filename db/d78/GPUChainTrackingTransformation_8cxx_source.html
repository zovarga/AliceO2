35 int GPUChainTracking::ConvertNativeToClusterData()
36 {
37 #ifdef GPUCA_HAVE_O2HEADERS
38  mRec->PushNonPersistentMemory(qStr2Tag("TPCTRANS"));
39  const auto& threadContext = GetThreadContext();
40  bool doGPU = GetRecoStepsGPU() & RecoStep::TPCConversion;
41  GPUTPCConvert& convert = processors()->tpcConverter;
42  GPUTPCConvert& convertShadow = doGPU ? processorsShadow()->tpcConverter : convert;
43 
44  bool transferClusters = false;
45  if (doGPU) {
46  if (!(mRec->GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCClusterFinding)) {
47  mInputsHost->mNClusterNative = mInputsShadow->mNClusterNative = mIOPtrs.clustersNative->nClustersTotal;
48  AllocateRegisteredMemory(mInputsHost->mResourceClusterNativeBuffer);
49  processorsShadow()->ioPtrs.clustersNative = mInputsShadow->mPclusterNativeAccess;
50  WriteToConstantMemory(RecoStep::TPCConversion, (char*)&processors()->ioPtrs - (char*)processors(), &processorsShadow()->ioPtrs, sizeof(processorsShadow()->ioPtrs), 0);
51  *mInputsHost->mPclusterNativeAccess = *mIOPtrs.clustersNative;
52  mInputsHost->mPclusterNativeAccess->clustersLinear = mInputsShadow->mPclusterNativeBuffer;
53  mInputsHost->mPclusterNativeAccess->setOffsetPtrs();
54  GPUMemCpy(RecoStep::TPCConversion, mInputsShadow->mPclusterNativeBuffer, mIOPtrs.clustersNative->clustersLinear, sizeof(mIOPtrs.clustersNative->clustersLinear[0]) * mIOPtrs.clustersNative->nClustersTotal, 0, true);
55  TransferMemoryResourceLinkToGPU(RecoStep::TPCConversion, mInputsHost->mResourceClusterNativeAccess, 0);
56  transferClusters = true;
57  }
58  }
59  if (!param().par.earlyTpcTransform) {
60  if (GetProcessingSettings().debugLevel >= 3) {
61  GPUInfo("Early transform inactive, skipping TPC Early transformation kernel, transformed on the fly during slice data creation / refit");
62  }
63  if (transferClusters) {
64  SynchronizeStream(0); // TODO: Synchronize implicitly with next step
65  }
66  return 0;
67  }
68  SetupGPUProcessor(&convert, true);
69  for (unsigned int i = 0; i < NSLICES; i++) {
70  convert.mMemory->clusters[i] = convertShadow.mClusters + mIOPtrs.clustersNative->clusterOffset[i][0];
71  }
72 
73  WriteToConstantMemory(RecoStep::TPCConversion, (char*)&processors()->tpcConverter - (char*)processors(), &convertShadow, sizeof(convertShadow), 0);
74  TransferMemoryResourcesToGPU(RecoStep::TPCConversion, &convert, 0);
75  runKernel<GPUTPCConvertKernel>(GetGridBlk(NSLICES * GPUCA_ROW_COUNT, 0), krnlRunRangeNone, krnlEventNone);
76  TransferMemoryResourcesToHost(RecoStep::TPCConversion, &convert, 0);
77  SynchronizeStream(0);
78 
79  for (unsigned int i = 0; i < NSLICES; i++) {
80  mIOPtrs.nClusterData[i] = (i == NSLICES - 1 ? mIOPtrs.clustersNative->nClustersTotal : mIOPtrs.clustersNative->clusterOffset[i + 1][0]) - mIOPtrs.clustersNative->clusterOffset[i][0];
81  mIOPtrs.clusterData[i] = convert.mClusters + mIOPtrs.clustersNative->clusterOffset[i][0];
82  }
83  mRec->PopNonPersistentMemory(RecoStep::TPCConversion, qStr2Tag("TPCTRANS"));
84 #endif
85  return 0;
86 }

154 int GPUChainTracking::ForwardTPCDigits()
155 {
156 #ifdef GPUCA_HAVE_O2HEADERS
157  if (GetRecoStepsGPU() & RecoStep::TPCClusterFinding) {
158  throw std::runtime_error("Cannot forward TPC digits with Clusterizer on GPU");
159  }
160  std::vector<ClusterNative> tmp[NSLICES][GPUCA_ROW_COUNT];
161  unsigned int nTotal = 0;
162  const float zsThreshold = param().rec.tpc.zsThreshold;
163  for (int i = 0; i < NSLICES; i++) {
164  for (unsigned int j = 0; j < mIOPtrs.tpcPackedDigits->nTPCDigits[i]; j++) {
165  const auto& d = mIOPtrs.tpcPackedDigits->tpcDigits[i][j];
166  if (d.getChargeFloat() >= zsThreshold) {
167  ClusterNative c;
168  c.setTimeFlags(d.getTimeStamp(), 0);
169  c.setPad(d.getPad());
170  c.setSigmaTime(1);
171  c.setSigmaPad(1);
172  c.qTot = c.qMax = d.getChargeFloat();
173  tmp[i][d.getRow()].emplace_back(c);
174  nTotal++;
175  }
176  }
177  }
178  mIOMem.clustersNative.reset(new ClusterNative[nTotal]);
179  nTotal = 0;
180  mClusterNativeAccess->clustersLinear = mIOMem.clustersNative.get();
181  for (int i = 0; i < NSLICES; i++) {
182  for (int j = 0; j < GPUCA_ROW_COUNT; j++) {
183  mClusterNativeAccess->nClusters[i][j] = tmp[i][j].size();
184  memcpy(&mIOMem.clustersNative[nTotal], tmp[i][j].data(), tmp[i][j].size() * sizeof(*mClusterNativeAccess->clustersLinear));
185  nTotal += tmp[i][j].size();
186  }
187  }
188  mClusterNativeAccess->setOffsetPtrs();
189  mIOPtrs.tpcPackedDigits = nullptr;
190  mIOPtrs.clustersNative = mClusterNativeAccess.get();
191  GPUInfo("Forwarded %u TPC clusters", nTotal);
192  mRec->MemoryScalers()->nTPCHits = nTotal;
193 #endif
194  return 0;
195 }
