72 void HmpidCoder2::setRDHFields(int eq)
73 {
74  int st, en;
75  uint32_t wr = (mBusyTime << 9) | ((mHmpidErrorFlag & 0x01F) << 4) | (mHmpidFrwVersion & 0x0F);
76  st = (eq < 0 || eq >= Geo::MAXEQUIPMENTS) ? 0 : eq;
77  en = (eq < 0 || eq >= Geo::MAXEQUIPMENTS) ? Geo::MAXEQUIPMENTS : eq + 1;
78  for (int l = st; l < en; l++) {
79  o2::raw::RawFileWriter::LinkData& link = mWriter.getLinkWithSubSpec(mTheRFWLinks[l]);
80  RDHAny* RDHptr = link.getLastRDH();
81  if (RDHptr != nullptr) {
82  o2::raw::RDHUtils::setDetectorField(RDHptr, wr);
83  }
84  }
85  return;
86 }

107 void HmpidCoder2::fillTheOutputBuffer(uint32_t* padMap)
108 {
109  uint32_t rowMarker, segMarker, eoeMarker, padWord;
110  uint32_t rowSize;
111  uint32_t ptr = 0;
112  int pads[Geo::MAXEQUIPMENTS];
113  int padsCount;
114  int segSize;
115 
116  for (int i = 0; i < Geo::MAXEQUIPMENTS; i++) {
117  mEventSizePerEquipment[i] = 0;
118  }
119 
120  for (int eq = 0; eq < mNumberOfEquipments; eq++) {
121  int startPtr = ptr;
122  padsCount = 0;
123  for (int s = 1; s <= Geo::N_SEGMENTS; s++) {
124  segSize = 0;
125  for (int c = 1; c <= Geo::N_COLXSEGMENT; c++) {
126  // ---- Pre-calculate the size of each column
127  for (int j = 0; j < Geo::N_DILOGICS; j++) {
128  pads[j] = 0;
129  }
130  rowSize = 0;
131  for (int j = 0; j < Geo::N_DILOGICS; j++) {
132  for (int k = 0; k < Geo::N_CHANNELS; k++) {
133  int idx = getEquipmentPadIndex(eq, ((s - 1) * Geo::N_COLXSEGMENT + (c - 1)), j, k);
134  if (padMap[idx] > 0) {
135  pads[j]++;
136  rowSize++;
137  padsCount++;
138  }
139  }
140  }
141  rowSize += Geo::N_DILOGICS;
142  segSize += (rowSize + 1);
143  rowMarker = 0x000036A8 | ((rowSize << 16) & 0x03ff0000);
144 
145  // ---- fills the Payload Buffer
146  mPayloadBufferPtr[ptr++] = rowMarker;
147  int col = (s - 1) * Geo::N_COLXSEGMENT + c;
148  for (int d = 1; d <= Geo::N_DILOGICS; d++) {
149  for (int p = 0; p < Geo::N_CHANNELS; p++) {
150  int idx = getEquipmentPadIndex(eq, ((s - 1) * Geo::N_COLXSEGMENT + (c - 1)), (d - 1), p);
151  if (padMap[idx] > 0) {
152  padWord = ((col << 22) & 0x07c00000) | ((d << 18) & 0x003C0000) | ((p << 12) & 0x0003F000) | (padMap[idx] & 0x00000FFF);
153  mPayloadBufferPtr[ptr++] = padWord;
154  }
155  }
156  eoeMarker = 0x08000080 | ((col << 22) & 0x07c00000) | (d << 18 & 0x003C0000) | (pads[d - 1] & 0x0000007F);
157  mPayloadBufferPtr[ptr++] = eoeMarker;
158  }
159  }
160  segSize += 1;
161  segMarker = 0xAB000000 | ((segSize << 8) & 0x000fff00) | (s & 0x0000000F);
162  mPayloadBufferPtr[ptr++] = segMarker;
163  }
164  mPadsCoded += padsCount;
165  mEventPadsPerEquipment[eq] = padsCount;
166  mEventSizePerEquipment[eq] = ptr - startPtr;
167  }
168  return;
169 }

177 void HmpidCoder2::writePaginatedEvent(uint32_t orbit, uint16_t bc)
178 {
179  uint32_t* ptrStartEquipment = mPayloadBufferPtr;
180  for (int eq = 0; eq < mNumberOfEquipments; eq++) {
181  int EventSize = mEventSizePerEquipment[eq];
182  LOG(debug) << "writePaginatedEvent() Eq=" << eq << " Size:" << EventSize << " Pads:" << mEventPadsPerEquipment[eq] << " Orbit:" << orbit << " BC:" << bc;
183  if (mEventPadsPerEquipment[eq] > 0 || !mSkipEmptyEvents) { // Skips the Events with 0 Pads
184  mWriter.addData(ReadOut::FeeId(eq),
185  ReadOut::CruId(eq),
186  ReadOut::LnkId(eq),
187  0,
188  {bc, orbit},
189  gsl::span<char>(reinterpret_cast<char*>(ptrStartEquipment),
190  EventSize * sizeof(uint32_t)),
191  false,
192  (uint32_t)((mBusyTime << 9) | ((mHmpidErrorFlag & 0x01F) << 4) | (mHmpidFrwVersion & 0x0F)));
193  // We fill the fields !
194  // TODO: we can fill the detector field with Simulated Data
195  setDetectorSpecificFields(0.000001 * EventSize);
196  setRDHFields(eq);
197  }
198  ptrStartEquipment += EventSize;
199  }
200  return;
201 }

210 void HmpidCoder2::codeEventChunkDigits(std::vector<o2::hmpid::Digit>& digits, InteractionRecord ir)
211 {
212  int eq, col, dil, cha, mo, x, y, idx;
213  uint32_t orbit = ir.orbit;
214  uint16_t bc = ir.bc;
215 
216  int padsCount = 0;
217  LOG(debug) << "Manage chunk Orbit :" << orbit << " BC:" << bc << " Digits size:" << digits.size();
218  for (o2::hmpid::Digit d : digits) {
219  Digit::pad2Equipment(d.getPadID(), &eq, &col, &dil, &cha); // From Digit to Hardware coords
220  eq = ReadOut::FeeId(eq); // converts the Equipment Id in Cru/Link position ref
221  idx = getEquipmentPadIndex(eq, col, dil, cha); // finally to the unique padmap index
222  if (mPadMap[idx] != 0) { // We already have the pad set
223  LOG(warning) << "Duplicated DIGIT =" << d << " (" << eq << "," << col << "," << dil << "," << cha << ")" << idx;
224  } else {
225  mPadMap[idx] = d.getCharge();
226  padsCount++;
227  }
228  }
229  fillTheOutputBuffer(mPadMap); // Fill the Buffer for all Equipments per Event
230  writePaginatedEvent(orbit, bc);
231  std::memset(mPadMap, 0, sizeof(uint32_t) * Geo::N_HMPIDTOTALPADS); // Update for the new event
232  return;
233 }

241 void HmpidCoder2::openOutputStream(const std::string& outputFileName, const std::string& fileFor)
242 {
243  RAWDataHeader rdh; // by default, v6 is used currently.
244  for (int eq = 0; eq < mNumberOfEquipments; eq++) {
245  rdh.feeId = ReadOut::FeeId(eq);
246  rdh.cruID = ReadOut::CruId(eq);
247  rdh.linkID = ReadOut::LnkId(eq);
248  rdh.endPointID = 0;
249  std::string outfname;
250  if (fileFor == "link") {
251  outfname = fmt::format("{}_{}_feeid{}.raw", outputFileName, ReadOut::FlpHostName(eq), ReadOut::FeeId(eq));
252  } else if (fileFor == "flp") {
253  outfname = fmt::format("{}_{}.raw", outputFileName, ReadOut::FlpHostName(eq));
254  } else if (fileFor == "all") {
255  outfname = fmt::format("{}.raw", outputFileName);
256  } else if (fileFor == "cru") {
257  outfname = fmt::format("{}_{}.raw", outputFileName, ReadOut::FlpHostName(eq));
258  } else {
259  throw std::runtime_error(fmt::format("unknown raw file grouping option {}", fileFor));
260  }
261 
262  mWriter.registerLink(rdh, outfname); // register the link
263  LinkSubSpec_t ap = RDHUtils::getSubSpec(ReadOut::CruId(eq), ReadOut::LnkId(eq), 0, ReadOut::FeeId(eq));
264  mTheRFWLinks[eq] = ap; // Store the RawFileWriter Link ID
265  }
266  return;
267 }
