38  public:
39  void init(o2::framework::InitContext& ic) final
40  {
41  bool isTFsendingPolicy = ic.options().get<bool>("tf-sending-policy");
42  int maxEnt = ic.options().get<int>("max-number-hits-to-fill-tree");
43  int slotL = ic.options().get<int>("tf-per-slot");
44  bool absMaxEnt = ic.options().get<bool>("is-max-number-hits-to-fill-tree-absolute");
45  mCollector = std::make_unique<o2::ft0::FT0CalibCollector>(isTFsendingPolicy, maxEnt);
46  }
47 
48  void run(o2::framework::ProcessingContext& pc) final
49  {
50  auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime; // is this the timestamp of the current TF?
51  auto data = pc.inputs().get<gsl::span<o2::ft0::FT0CalibrationInfoObject>>("input");
52  mCollector->process(tfcounter, data);
53  sendOutput(pc.outputs());
54  }
55 
56  void endOfStream(o2::framework::EndOfStreamContext& ec) final
57  {
58  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
59  mCollector->checkSlotsToFinalize(INFINITE_TF);
60  // we force finalizing slot zero (unless everything was already finalized), no matter how many entries we had
61  if (mCollector->getNSlots() != 0) {
62  mCollector->finalizeSlot(mCollector->getSlot(0));
63  }
64  sendOutput(ec.outputs());
65  }
66 
67  private:
68  std::unique_ptr<o2::ft0::FT0CalibCollector> mCollector;
69  int mMaxNumOfHits = 0;
70 
71  //________________________________________________________________
72  void sendOutput(DataAllocator& output)
73  {
74  // in output we send the calibration tree
75  auto& collectedInfo = mCollector->getCollectedCalibInfo();
76  LOG(debug) << "In CollectorSpec sendOutput: size = " << collectedInfo.size();
77  if (collectedInfo.size()) {
78  auto entries = collectedInfo.size();
79  // this means that we are ready to send the output
80  auto entriesPerChannel = mCollector->getEntriesPerChannel();
81  output.snapshot(Output{o2::header::gDataOriginFT0, "COLLECTEDINFO", 0, Lifetime::Timeframe}, collectedInfo);
82  output.snapshot(Output{o2::header::gDataOriginFT0, "ENTRIESCH", 0, Lifetime::Timeframe}, entriesPerChannel);
83  mCollector->initOutput(); // reset the output for the next round
84  }
85  }
86 };
