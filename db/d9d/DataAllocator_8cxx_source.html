45 std::string const& DataAllocator::matchDataHeader(const Output& spec, size_t timeslice)
46 {
47  // FIXME: we should take timeframeId into account as well.
48  for (auto& output : mAllowedOutputRoutes) {
49  if (DataSpecUtils::match(output.matcher, spec.origin, spec.description, spec.subSpec) && ((timeslice % output.maxTimeslices) == output.timeslice)) {
50  return output.channel;
51  }
52  }
53  throw runtime_error_f(
54  "Worker is not authorised to create message with "
55  "origin(%s) description(%s) subSpec(%d)",
56  spec.origin.as<std::string>().c_str(),
57  spec.description.as<std::string>().c_str(),
58  spec.subSpec);
59 }

114 void DataAllocator::addPartToContext(FairMQMessagePtr&& payloadMessage, const Output& spec,
115  o2::header::SerializationMethod serializationMethod)
116 {
117  auto& timingInfo = mRegistry->get<TimingInfo>();
118  std::string const& channel = matchDataHeader(spec, timingInfo.timeslice);
119  auto headerMessage = headerMessageFromOutput(spec, channel, serializationMethod, 0);
120 
121  // FIXME: this is kind of ugly, we know that we can change the content of the
122  // header message because we have just created it, but the API declares it const
123  const DataHeader* cdh = o2::header::get<DataHeader*>(headerMessage->GetData());
124  DataHeader* dh = const_cast<DataHeader*>(cdh);
125  dh->payloadSize = payloadMessage->GetSize();
126  auto& context = mRegistry->get<MessageContext>();
127  // make_scoped creates the context object inside of a scope handler, since it goes out of
128  // scope immediately, the created object is scheduled and can be directly sent if the context
129  // is configured with the dispatcher callback
130  context.make_scoped<MessageContext::TrivialObject>(std::move(headerMessage), std::move(payloadMessage), channel);
131 }

145 void doWriteTable(std::shared_ptr<FairMQResizableBuffer> b, arrow::Table* table)
146 {
147  auto mock = std::make_shared<arrow::io::MockOutputStream>();
148  int64_t expectedSize = 0;
149  auto mockWriter = arrow::ipc::MakeStreamWriter(mock.get(), table->schema());
150  arrow::Status outStatus;
151  if (O2_BUILTIN_LIKELY(table->num_rows() != 0)) {
152  outStatus = mockWriter.ValueOrDie()->WriteTable(*table);
153  } else {
154  std::vector<std::shared_ptr<arrow::Array>> columns;
155  columns.resize(table->columns().size());
156  for (size_t ci = 0; ci < table->columns().size(); ci++) {
157  columns[ci] = table->column(ci)->chunk(0);
158  }
159  auto batch = arrow::RecordBatch::Make(table->schema(), 0, columns);
160  outStatus = mockWriter.ValueOrDie()->WriteRecordBatch(*batch);
161  }
162 
163  expectedSize = mock->Tell().ValueOrDie();
164  auto reserve = b->Reserve(expectedSize);
165  if (reserve.ok() == false) {
166  throw std::runtime_error("Unable to reserve memory for table");
167  }
168 
169  auto stream = std::make_shared<FairMQOutputStream>(b);
170  auto outBatch = arrow::ipc::MakeStreamWriter(stream.get(), table->schema());
171  if (outBatch.ok() == false) {
172  throw ::std::runtime_error("Unable to create batch writer");
173  }
174 
175  if (O2_BUILTIN_UNLIKELY(table->num_rows() == 0)) {
176  std::vector<std::shared_ptr<arrow::Array>> columns;
177  columns.resize(table->columns().size());
178  for (size_t ci = 0; ci < table->columns().size(); ci++) {
179  columns[ci] = table->column(ci)->chunk(0);
180  }
181  auto batch = arrow::RecordBatch::Make(table->schema(), 0, columns);
182  outStatus = outBatch.ValueOrDie()->WriteRecordBatch(*batch);
183  } else {
184  outStatus = outBatch.ValueOrDie()->WriteTable(*table);
185  }
186 
187  if (outStatus.ok() == false) {
188  throw std::runtime_error("Unable to Write table");
189  }
190 }

192 void DataAllocator::adopt(const Output& spec, TableBuilder* tb)
193 {
194  auto& timingInfo = mRegistry->get<TimingInfo>();
195  std::string const& channel = matchDataHeader(spec, timingInfo.timeslice);
196  auto header = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodArrow, 0);
197  auto& context = mRegistry->get<ArrowContext>();
198 
199  auto creator = [device = context.proxy().getDevice()](size_t s) -> std::unique_ptr<FairMQMessage> { return device->NewMessage(s); };
200  auto buffer = std::make_shared<FairMQResizableBuffer>(creator);
201 
205  std::shared_ptr<TableBuilder> p(tb);
206  auto finalizer = [payload = p](std::shared_ptr<FairMQResizableBuffer> b) -> void {
207  auto table = payload->finalize();
208  doWriteTable(b, table.get());
209  };
210 
211  context.addBuffer(std::move(header), buffer, std::move(finalizer), channel);
212 }

268 Output DataAllocator::getOutputByBind(OutputRef&& ref)
269 {
270  if (ref.label.empty()) {
271  throw runtime_error("Invalid (empty) OutputRef provided.");
272  }
273  for (auto ri = 0ul, re = mAllowedOutputRoutes.size(); ri != re; ++ri) {
274  if (mAllowedOutputRoutes[ri].matcher.binding.value == ref.label) {
275  auto spec = mAllowedOutputRoutes[ri].matcher;
276  auto dataType = DataSpecUtils::asConcreteDataTypeMatcher(spec);
277  return Output{dataType.origin, dataType.description, ref.subSpec, spec.lifetime, std::move(ref.headerStack)};
278  }
279  }
280  throw runtime_error_f("Unable to find OutputSpec with label %s", ref.label.c_str());
281  O2_BUILTIN_UNREACHABLE();
282 }
