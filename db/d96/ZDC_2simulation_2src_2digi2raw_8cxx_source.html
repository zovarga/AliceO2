104 void digi2raw(const std::string& inpName, const std::string& outDir, int verbosity, const std::string& fileFor, uint32_t rdhV, const std::string& ccdbHost, int superPageSizeInB)
105 {
106  //std::string ccdbHost = "http://ccdb-test.cern.ch:8080";
107  auto& mgr = o2::ccdb::BasicCCDBManager::instance();
108  mgr.setURL(ccdbHost);
109  /*long timeStamp = 0; // TIMESTAMP SHOULD NOT BE 0!
110  if (timeStamp == mgr.getTimestamp()) {
111  return;
112  }
113  mgr.setTimestamp(timeStamp);*/
114  auto moduleConfig = mgr.get<o2::zdc::ModuleConfig>(o2::zdc::CCDBPathConfigModule);
115  if (!moduleConfig) {
116  LOG(fatal) << "Cannot module configuratio for timestamp " << mgr.getTimestamp();
117  return;
118  }
119  LOG(info) << "Loaded module configuration for timestamp " << mgr.getTimestamp();
120 
121  auto simCondition = mgr.get<o2::zdc::SimCondition>(o2::zdc::CCDBPathConfigSim);
122  if (!simCondition) {
123  LOG(fatal) << "Cannot get simulation configuration for timestamp " << mgr.getTimestamp();
124  return;
125  }
126  LOG(info) << "Loaded simulation configuration for timestamp " << mgr.getTimestamp();
127  simCondition->print();
128 
129  const auto* ctx = o2::steer::DigitizationContext::loadFromFile("collisioncontext.root");
130  const auto& bcfill = ctx->getBunchFilling();
131  auto bf = ctx->getBunchFilling();
132  if (verbosity > 0) {
133  bf.print();
134  }
135  auto bp = bf.getPattern();
136 
137  TStopwatch swTot;
138  swTot.Start();
139 
140  o2::zdc::Digits2Raw d2r;
141  d2r.setFileFor(fileFor);
142  d2r.setVerbosity(verbosity);
143  auto& wr = d2r.getWriter();
144  std::string inputGRP = o2::base::NameConf::getGRPFileName();
145  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
146  wr.setContinuousReadout(grp->isDetContinuousReadOut(o2::detectors::DetID::ZDC)); // must be set explicitly
147  wr.setSuperPageSize(superPageSizeInB);
148  wr.useRDHVersion(rdhV);
149 
150  o2::raw::assertOutputDirectory(outDir);
151 
152  std::string outDirName(outDir);
153  if (outDirName.back() != '/') {
154  outDirName += '/';
155  }
156 
157  d2r.setModuleConfig(moduleConfig);
158  d2r.setSimCondition(simCondition);
159  d2r.emptyBunches(bp);
160  d2r.setVerbosity(verbosity);
161  d2r.processDigits(outDirName, inpName);
162  wr.writeConfFile(wr.getOrigin().str, "RAWDATA", o2::utils::Str::concat_string(outDirName, wr.getOrigin().str, "raw.cfg"));
163  //
164  swTot.Stop();
165  swTot.Print();
166 }
