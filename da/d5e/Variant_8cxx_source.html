75  oss << val.get<float>();
76  break;
77  case VariantType::Double:

87  printArray<int>(oss, val.get<int*>(), val.size());

91  break;
92  case VariantType::ArrayDouble:
93  printArray<double>(oss, val.get<double*>(), val.size());
94  break;
95  case VariantType::ArrayBool:
96  printArray<bool>(oss, val.get<bool*>(), val.size());
97  break;
98  case VariantType::ArrayString:
99  printArray<std::string>(oss, val.get<std::string*>(), val.size());
100  break;
101  case VariantType::Array2DInt:
102  printMatrix<int>(oss, val.get<Array2D<int>>());
103  break;
104  case VariantType::Array2DFloat:
105  printMatrix<float>(oss, val.get<Array2D<float>>());
106  break;
107  case VariantType::Array2DDouble:
108  printMatrix<double>(oss, val.get<Array2D<double>>());
109  break;
110  case VariantType::Empty:
111  break;
112  case VariantType::Dict:
113  oss << "{}";

129 Variant::Variant(const Variant& other) : mType(other.mType)
130 {
131  // In case this is an array we need to duplicate it to avoid
132  // double deletion.
133  switch (mType) {
134  case variant_trait_v<const char*>:
135  mSize = other.mSize;
136  variant_helper<storage_t, const char*>::set(&mStore, other.get<const char*>());
137  return;
138  case variant_trait_v<int*>:
139  mSize = other.mSize;
140  variant_helper<storage_t, int*>::set(&mStore, other.get<int*>(), mSize);
141  return;
142  case variant_trait_v<float*>:
143  mSize = other.mSize;
144  variant_helper<storage_t, float*>::set(&mStore, other.get<float*>(), mSize);
145  return;

191 Variant::~Variant()
192 {
193  // In case we allocated an array, we
194  // should delete it.
195  switch (mType) {
196  case variant_trait_v<const char*>:
197  case variant_trait_v<int*>:
198  case variant_trait_v<float*>:
199  case variant_trait_v<double*>:
200  case variant_trait_v<bool*>:
201  case variant_trait_v<std::string*>:
202  if (reinterpret_cast<void**>(&mStore) != nullptr) {
203  free(*reinterpret_cast<void**>(&mStore));
204  }
205  return;
206  default:
207  return;
208  }
