51 // just need to implement 2 special methods init + run (there is no need to inherit from anything)
52 class TOFDPLClustererTask
53 {
54  bool mUseMC = true;
55  bool mUseCCDB = false;
56  bool mIsCalib = false;
57  bool mIsCosmic = false;
58  int mTimeWin = 5000;

62  explicit TOFDPLClustererTask(bool useMC, bool useCCDB, bool doCalib, bool isCosmic, std::string ccdb_url) : mUseMC(useMC), mUseCCDB(useCCDB), mIsCalib(doCalib), mIsCosmic(isCosmic), mCCDBurl(ccdb_url) {}
63  void init(framework::InitContext& ic)
64  {
65  // nothing special to be set up
66  mTimer.Stop();
67  mTimer.Reset();
68 
69  mTimeWin = ic.options().get<int>("cluster-time-window");
70  LOG(debug) << "Is calibration from cluster on? " << mIsCalib;
71  LOG(debug) << "DeltaTime for clusterization = " << mTimeWin << " ps";
72  LOG(debug) << "Is cosmics? " << mIsCosmic;
73 
74  mClusterer.setCalibFromCluster(mIsCalib);
75  mClusterer.setDeltaTforClustering(mTimeWin);
76 
77  // initialize collision context
78  auto mcReader = std::make_unique<o2::steer::MCKinematicsReader>("collisioncontext.root");
79  auto context = mcReader->getDigitizationContext();
80  if (context) {
81  auto bcf = context->getBunchFilling();
82  std::bitset<3564> isInBC = bcf.getBCPattern();
83  for (unsigned int i = 0; i < isInBC.size(); i++) {
84  if (isInBC.test(i)) {
85  o2::tof::Utils::addInteractionBC(i, true);
86  }
87  }
88  }
89  }

91  void run(framework::ProcessingContext& pc)
92  {
93  mTimer.Start(false);
94  // get digit data
95  auto digits = pc.inputs().get<gsl::span<o2::tof::Digit>>("tofdigits");
96  auto row = pc.inputs().get<gsl::span<o2::tof::ReadoutWindowData>>("readoutwin");
97  auto dia = pc.inputs().get<o2::tof::Diagnostic*>("diafreq");
98  auto patterns = pc.inputs().get<pmr::vector<unsigned char>>("patterns");
99 
100  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true));
101  mClusterer.setFirstOrbit(dh->firstTForbit);
102 
103  auto labelvector = std::make_shared<std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>>();
104  if (mUseMC) {
105  auto digitlabels = pc.inputs().get<std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>*>("tofdigitlabels");
106  *labelvector.get() = std::move(*digitlabels);
107  mClusterer.setMCTruthContainer(&mClsLabels);
108  mClsLabels.clear();
109  }
110 
111  o2::dataformats::CalibLHCphaseTOF lhcPhaseObj;
112  auto channelCalibObj = std::make_unique<o2::dataformats::CalibTimeSlewingParamTOF>();
113 
114  /*
115  if (mUseCCDB) { // read calibration objects from ccdb
116  // check LHC phase
117  auto lhcPhase = pc.inputs().get<o2::dataformats::CalibLHCphaseTOF*>("tofccdbLHCphase");
118  auto channelCalib = pc.inputs().get<o2::dataformats::CalibTimeSlewingParamTOF*>("tofccdbChannelCalib");
119 
120  o2::dataformats::CalibLHCphaseTOF lhcPhaseObjTmp = std::move(*lhcPhase);
121  o2::dataformats::CalibTimeSlewingParamTOF channelCalibObjTmp = std::move(*channelCalib);
122 
123  // make a copy in global scope
124  lhcPhaseObj = lhcPhaseObjTmp;
125  *channelCalibObj = channelCalibObjTmp;
126  } else { // calibration objects set to zero
127 */
128  lhcPhaseObj.addLHCphase(0, 0);
129  lhcPhaseObj.addLHCphase(2000000000, 0);
130 
131  for (int ich = 0; ich < o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELS; ich++) {
132  channelCalibObj->addTimeSlewingInfo(ich, 0, 0);
133  int sector = ich / o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELXSECTOR;
134  int channelInSector = ich % o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELXSECTOR;
135  channelCalibObj->setFractionUnderPeak(sector, channelInSector, 1);
136  }
137  // }
138 
139  o2::tof::CalibTOFapi calibapi(long(0), &lhcPhaseObj, channelCalibObj.get());
140 
141  if (mUseCCDB) {
142  calibapi.setURL(mCCDBurl.c_str());
143  auto creationTime = DataRefUtils::getHeader<DataProcessingHeader*>(pc.inputs().getFirstValid(true))->creation;
144  calibapi.setTimeStamp(creationTime / 1000);
145  LOG(info) << "CCDB required from TOF clusterizer with timestamp " << creationTime / 1000 << " from URL " << mCCDBurl.c_str();
146 
147  LOG(info) << "read LHCphase";
148  calibapi.readLHCphase();
149  LOG(info) << "read time slewing";
150  calibapi.readTimeSlewingParam();
151  LOG(info) << "read daignostic";
152  calibapi.readDiagnosticFrequencies();
153  } else {
154  LOG(info) << "No CCDB requested by TOF";
155  }
156 
157  mClusterer.setCalibApi(&calibapi);
158 
159  mClusterer.clearDiagnostic();
160  mClusterer.addDiagnostic(*dia);
161 
162  // call actual clustering routine
163  mClustersArray.clear();
164  if (mIsCalib) {
165  mClusterer.getInfoFromCluster()->clear();
166  }
167 
168  if (mIsCosmic) {
169  mCosmicProcessor.clear();
170  }
171 
172  for (unsigned int i = 0; i < row.size(); i++) {
173  //fill trm pattern but process them in clusterize since they are for readout windows
174  calibapi.resetTRMErrors();
175 
176  // loop over crates
177  int kw = 0;
178  for (int crate = 0; crate < 72; crate++) {
179  int slot = -1;
180  int nwords = kw + row[i].getDiagnosticInCrate(crate);
181  int eword;
182  for (; kw < nwords; kw++) {
183  if (patterns[kw] > 28) { // new slot
184  if (slot > -1) { // fill previous
185  calibapi.processError(crate, slot, eword);
186  }
187  slot = patterns[kw] - 28;
188  eword = 0;
189 
190  if (slot < 3 || slot > 12) { // not a valid slot -> don't fill otherwise mapping fails (slot 1 crate 0 is a bad condition)
191  slot = -1;
192  // LOG(info) << "not a valid slot in diagnostic words: slot = " << slot << " for crate " << crate;
193  }
194  } else if (slot > -1) { // process error in this slot
195  eword += 1 << patterns[kw];
196  }
197  }
198  } // end crate loop
199 
200  //printf("# TOF readout window for clusterization = %d/%lu (N digits = %d)\n", i, row.size(), row[i].size());
201  auto digitsRO = row[i].getBunchChannelData(digits);
202  mReader.setDigitArray(&digitsRO);
203 
204  if (mIsCosmic) {
205  mCosmicProcessor.process(mReader, i != 0);
206  }
207 
208  if (mUseMC) {
209  mClusterer.process(mReader, mClustersArray, &(labelvector->at(i)));
210  } else {
211  mClusterer.process(mReader, mClustersArray, nullptr);
212  }
213  }
214  LOG(debug) << "TOF CLUSTERER : TRANSFORMED " << digits.size()
215  << " DIGITS TO " << mClustersArray.size() << " CLUSTERS";
216 
217  // send clusters
218  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "CLUSTERS", 0, Lifetime::Timeframe}, mClustersArray);
219  // send labels
220  if (mUseMC) {
221  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "CLUSTERSMCTR", 0, Lifetime::Timeframe}, mClsLabels);
222  }
223 
224  if (mIsCalib) {
225  std::vector<CalibInfoCluster>* clusterCalInfo = mClusterer.getInfoFromCluster();
226  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "INFOCALCLUS", 0, Lifetime::Timeframe}, *clusterCalInfo);
227  }
228 
229  if (mIsCosmic) {
230  std::vector<CosmicInfo>* cosmicInfo = mCosmicProcessor.getCosmicInfo();
231  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "INFOCOSMICS", 0, Lifetime::Timeframe}, *cosmicInfo);
232  std::vector<CalibInfoTrackCl>* cosmicTrack = mCosmicProcessor.getCosmicTrack();
233  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "INFOTRACKCOS", 0, Lifetime::Timeframe}, *cosmicTrack);
234  std::vector<int>* cosmicTrackSize = mCosmicProcessor.getCosmicTrackSize();
235  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "INFOTRACKSIZE", 0, Lifetime::Timeframe}, *cosmicTrackSize);
236  }
237 
238  mTimer.Stop();
239  }
240 
241  void endOfStream(EndOfStreamContext& ec)
242  {
243  LOGF(debug, "TOF Clusterer total timing: Cpu: %.3e Real: %.3e s in %d slots",
244  mTimer.CpuTime(), mTimer.RealTime(), mTimer.Counter() - 1);
245  }

249  Clusterer mClusterer;
250  CosmicProcessor mCosmicProcessor;
251  TStopwatch mTimer;
252 
253  std::vector<Cluster> mClustersArray;
254  MCLabelContainer mClsLabels;
255  std::vector<CalibInfoCluster> mClusterCalInfo;

258 o2::framework::DataProcessorSpec getTOFClusterizerSpec(bool useMC, bool useCCDB, bool doCalib, bool isCosmic, std::string ccdb_url)
259 {
260  std::vector<InputSpec> inputs;
261  inputs.emplace_back("tofdigits", o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe);
262  inputs.emplace_back("readoutwin", o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe);
263  inputs.emplace_back("diafreq", o2::header::gDataOriginTOF, "DIAFREQ", 0, Lifetime::Timeframe);
264  inputs.emplace_back("patterns", o2::header::gDataOriginTOF, "PATTERNS", 0, Lifetime::Timeframe);
265 
266  // if (useCCDB) {
267  // inputs.emplace_back("tofccdbLHCphase", o2::header::gDataOriginTOF, "LHCphase");
268  // inputs.emplace_back("tofccdbChannelCalib", o2::header::gDataOriginTOF, "ChannelCalib");
269  // }
270  if (useMC) {
271  inputs.emplace_back("tofdigitlabels", o2::header::gDataOriginTOF, "DIGITSMCTR", 0, Lifetime::Timeframe);
272  }
273 
274  std::vector<OutputSpec> outputs;
275  outputs.emplace_back(o2::header::gDataOriginTOF, "CLUSTERS", 0, Lifetime::Timeframe);
276  if (useMC) {
277  outputs.emplace_back(o2::header::gDataOriginTOF, "CLUSTERSMCTR", 0, Lifetime::Timeframe);
278  }
279 
280  if (doCalib) {
281  outputs.emplace_back(o2::header::gDataOriginTOF, "INFOCALCLUS", 0, Lifetime::Timeframe);
282  }
283  if (isCosmic) {
284  outputs.emplace_back(o2::header::gDataOriginTOF, "INFOCOSMICS", 0, Lifetime::Timeframe);
285  outputs.emplace_back(o2::header::gDataOriginTOF, "INFOTRACKCOS", 0, Lifetime::Timeframe);
286  outputs.emplace_back(o2::header::gDataOriginTOF, "INFOTRACKSIZE", 0, Lifetime::Timeframe);
287  }
288 
289  return DataProcessorSpec{
290  "TOFClusterer",
291  inputs,
292  outputs,
293  AlgorithmSpec{adaptFromTask<TOFDPLClustererTask>(useMC, useCCDB, doCalib, isCosmic, ccdb_url)},
294  Options{{"cluster-time-window", VariantType::Int, 5000, {"time window for clusterization in ps"}}}};
295 }
