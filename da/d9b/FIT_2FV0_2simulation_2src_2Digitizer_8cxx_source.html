38 void Digitizer::init()
39 {
40  LOG(info) << "init";
41  mNBins = FV0DigParam::Instance().waveformNbins; //Will be computed using detector set-up from CDB
42  mBinSize = FV0DigParam::Instance().waveformBinWidth; //Will be set-up from CDB
43  mNTimeBinsPerBC = std::lround(o2::constants::lhc::LHCBunchSpacingNS / mBinSize); // 1920 bins/BC
44 
45  for (Int_t detID = 0; detID < Constants::nFv0Channels; detID++) {
46  mPmtChargeVsTime[detID].resize(mNBins);
47  mLastBCCache.mPmtChargeVsTime[detID].resize(mNBins);
48  }
49 
51  TF1Convolution convolutionRingA1ToA4("expo", "landau", 5.e-09, 90.e-09, false);
52  TF1 convolutionRingA1ToA4Fn("convolutionFn", convolutionRingA1ToA4, 5.e-09, 90.e-09, convolutionRingA1ToA4.GetNpar());
53  convolutionRingA1ToA4Fn.SetParameters(FV0DigParam::Instance().constRingA1ToA4, FV0DigParam::Instance().slopeRingA1ToA4,
54  FV0DigParam::Instance().mpvRingA1ToA4, FV0DigParam::Instance().sigmaRingA1ToA4);
55 
57  TF1Convolution convolutionRing5("expo", "landau", 5.e-09, 90.e-09, false);
58  TF1 convolutionRing5Fn("convolutionFn", convolutionRing5, 5.e-09, 90.e-09, convolutionRing5.GetNpar());
59  convolutionRing5Fn.SetParameters(FV0DigParam::Instance().constRing5, FV0DigParam::Instance().slopeRing5,
60  FV0DigParam::Instance().mpvRing5, FV0DigParam::Instance().sigmaRing5);
62  mPmtResponseGlobalRingA1ToA4.resize(mNBins);
63  const float binSizeInNs = mBinSize * 1.e-09; // to convert ns into sec
64  double x = (binSizeInNs) / 2.0;
65  for (auto& y : mPmtResponseGlobalRingA1ToA4) {
66  y = FV0DigParam::Instance().normRingA1ToA4 // normalisation to have MIP adc at 16
67  * convolutionRingA1ToA4Fn.Eval(x + FV0DigParam::Instance().offsetRingA1ToA4); //offset to adjust mean position of waveform
68  x += binSizeInNs;
69  }
71  mPmtResponseGlobalRing5.resize(mNBins);
72  x = (binSizeInNs) / 2.0;
73  for (auto& y : mPmtResponseGlobalRing5) {
74  y = FV0DigParam::Instance().normRing5 // normalisation to have MIP adc at 16
75  * convolutionRing5Fn.Eval(x + FV0DigParam::Instance().offsetRing5); //offset to adjust mean position of waveform
76  x += binSizeInNs;
77  }
78  mLastBCCache.clear();
79  mCfdStartIndex.fill(0);
80  LOG(info) << "init -> finished";
81 }

83 void Digitizer::process(const std::vector<o2::fv0::Hit>& hits,
84  std::vector<o2::fv0::BCData>& digitsBC,
85  std::vector<o2::fv0::ChannelData>& digitsCh,
86  std::vector<o2::fv0::DetTrigInput>& digitsTrig,
87  o2::dataformats::MCTruthContainer<o2::fv0::MCLabel>& labels)
88 {
89  LOG(debug) << "Begin with " << hits.size() << " hits";
90  flush(digitsBC, digitsCh, digitsTrig, labels); // flush cached signal which cannot be affect by new event
91 
92  std::vector<int> hitIdx(hits.size());
93  std::iota(std::begin(hitIdx), std::end(hitIdx), 0);
94  std::sort(std::begin(hitIdx), std::end(hitIdx),
95  [&hits](int a, int b) { return hits[a].GetTrackID() < hits[b].GetTrackID(); });
96 
97  // use ordered hits
98  for (auto ids : hitIdx) {
99  const auto& hit = hits[ids];
100  Int_t detId = hit.GetDetectorID();
101  Double_t hitEdep = hit.GetHitValue() * 1e3; //convert to MeV
102  Float_t const hitTime = hit.GetTime() * 1e9;
103  // TODO: check how big is inaccuracy if more than 1 'below-threshold' particles hit the same detector cell
104  if (hitEdep < FV0DigParam::Instance().singleMipThreshold || hitTime > FV0DigParam::Instance().singleHitTimeThreshold) {
105  continue;
106  }
107  float distanceFromXc = 0;
108  if (Geometry::instance()->isRing5(detId)) {
109  distanceFromXc = getDistFromCellCenter(detId, hit.GetX(), hit.GetY());
110  }
111 
112  int iChannelPerCell = 0;
113  while (iChannelPerCell < 2) { // loop over 2 channels, into which signal from each cell in ring 5 is split
114  if (Geometry::instance()->isRing5(detId)) {
115  // The first channel number is located counter-clockwise from the cell center
116  // and remains identical to the detector number, the second one is clockwise and incremented by 8
117  if (iChannelPerCell == 1) {
118  detId += 8;
119  }
120  // Split signal magnitude to fractions depending on the distance of the hit from the cell center
121  hitEdep = (hit.GetHitValue() * 1e3) * getSignalFraction(distanceFromXc, iChannelPerCell == 0);
122  // LOG(info) << " detId: " << detId << "-" << iChannelPerCell << " hitEdep: " << hitEdep << " distanceFromXc: " << distanceFromXc;
123  ++iChannelPerCell;
124  } else {
125  iChannelPerCell = 2; // not a ring 5 cell -> don't repeat the loop
126  }
127  Double_t const nPhotons = hitEdep * DP::N_PHOTONS_PER_MEV;
128  float const nPhE = SimulateLightYield(detId, nPhotons);
129  float const mipFraction = float(nPhE / FV0DigParam::Instance().avgNumberPhElectronPerMip);
130  Float_t timeHit = hitTime;
131  timeHit += mIntRecord.getTimeNS();
132  o2::InteractionTimeRecord const irHit(timeHit);
133  std::array<o2::InteractionRecord, NBC2Cache> cachedIR;
134  int nCachedIR = 0;
135  for (int i = BCCacheMin; i < BCCacheMax + 1; i++) {
136  double const tNS = timeHit + o2::constants::lhc::LHCBunchSpacingNS * i;
137  cachedIR[nCachedIR].setFromNS(tNS);
138  if (tNS < 0 && cachedIR[nCachedIR] > irHit) {
139  continue; // don't go to negative BC/orbit (it will wrap)
140  }
141  setBCCache(cachedIR[nCachedIR++]); // ensure existence of cached container
142  } //BCCache loop
143  createPulse(mipFraction, hit.GetTrackID(), hitTime, cachedIR, nCachedIR, detId);
144 
145  } //while loop
146  } //hitloop
147 }

149 void Digitizer::createPulse(float mipFraction, int parID, const double hitTime,
150  std::array<o2::InteractionRecord, NBC2Cache> const& cachedIR, int nCachedIR, const int detId)
151 {
152 
153  std::array<bool, NBC2Cache> added;
154  added.fill(false);
155 
156  for (int ir = 0; ir < NBC2Cache; ir++) {
157  auto bcCache = getBCCache(cachedIR[ir]);
158  for (int ich = 0; ich < Constants::nFv0Channels; ich++) {
159  (*bcCache).mPmtChargeVsTime[ich].resize(mNTimeBinsPerBC);
160  }
161  }
162 
164  Int_t const NBinShift = std::lround((hitTime - FV0DigParam::Instance().globalTimeOfFlight) / FV0DigParam::Instance().waveformBinWidth);
165 
166  if (NBinShift >= 0 && NBinShift < FV0DigParam::Instance().waveformNbins) {
167  mPmtResponseTemp.resize(FV0DigParam::Instance().waveformNbins, 0.);
168  if (isRing5(detId)) {
169  std::memcpy(&mPmtResponseTemp[NBinShift], &mPmtResponseGlobalRing5[0],
170  sizeof(double) * (FV0DigParam::Instance().waveformNbins - NBinShift));
171  } else {
172  std::memcpy(&mPmtResponseTemp[NBinShift], &mPmtResponseGlobalRingA1ToA4[0],
173  sizeof(double) * (FV0DigParam::Instance().waveformNbins - NBinShift));
174  }
175  } else {
176  if (isRing5(detId)) {
177  mPmtResponseTemp = mPmtResponseGlobalRing5;
178  mPmtResponseTemp.erase(mPmtResponseTemp.begin(), mPmtResponseTemp.begin() + abs(NBinShift));
179  } else {
180  mPmtResponseTemp = mPmtResponseGlobalRingA1ToA4;
181  mPmtResponseTemp.erase(mPmtResponseTemp.begin(), mPmtResponseTemp.begin() + abs(NBinShift));
182  }
183 
184  mPmtResponseTemp.resize(FV0DigParam::Instance().waveformNbins);
185  }
186 
187  for (int ir = 0; ir < int(mPmtResponseTemp.size() / mNTimeBinsPerBC); ir++) {
188  auto bcCache = getBCCache(cachedIR[ir]);
189 
190  for (int iBin = 0; iBin < mNTimeBinsPerBC; iBin++) {
191  (*bcCache).mPmtChargeVsTime[detId][iBin] += (mPmtResponseTemp[ir * mNTimeBinsPerBC + iBin] * mipFraction);
192  }
193  added[ir] = true;
194  }
196  for (int ir = 0; ir < nCachedIR; ir++) {
197  if (added[ir]) {
198  auto bcCache = getBCCache(cachedIR[ir]);
199  (*bcCache).labels.emplace_back(parID, mEventId, mSrcId, detId);
200  }
201  }
202 }

204 void Digitizer::flush(std::vector<o2::fv0::BCData>& digitsBC,
205  std::vector<o2::fv0::ChannelData>& digitsCh,
206  std::vector<o2::fv0::DetTrigInput>& digitsTrig,
207  o2::dataformats::MCTruthContainer<o2::fv0::MCLabel>& labels)
208 {
209  while (!mCache.empty()) {
210  auto const& bc = mCache.front();
211  if (mIntRecord.differenceInBC(bc) > NBC2Cache) { // Build events that are separated by NBC2Cache BCs from current BC
212  storeBC(bc, digitsBC, digitsCh, digitsTrig, labels);
213  mCache.pop_front();
214  } else {
215  return;
216  }
217  }
218 }

220 void Digitizer::storeBC(const BCCache& bc,
221  std::vector<o2::fv0::BCData>& digitsBC,
222  std::vector<o2::fv0::ChannelData>& digitsCh,
223  std::vector<o2::fv0::DetTrigInput>& digitsTrig,
224  o2::dataformats::MCTruthContainer<o2::fv0::MCLabel>& labels)
225 
226 {
227  size_t const nBC = digitsBC.size(); // save before digitsBC is being modified
228  size_t const first = digitsCh.size(); // save before digitsCh is being modified
229  size_t nStored = 0;
230  double totalChargeAllRing = 0;
231  double nSignalInner = 0;
232  double nSignalOuter = 0;
233 
234  if (mLastBCCache.differenceInBC(bc) != 1) { // if the last buffered BC is not the one before the current BC
235  mLastBCCache.clear(); // clear the bufffer (mPmtChargeVsTime set to 0s)
236  mCfdStartIndex.fill(0); // reset all start indices to 0, i.e., to the beginning of the BC
237  }
238 
239  for (int iPmt = 0; iPmt < Constants::nFv0Channels; iPmt++) {
240  // run the CFD: this updates the start index for the next BC in case the CFD dead time ends in the next BC
241  double cfdWithOffset = SimulateTimeCfd(mCfdStartIndex[iPmt], mLastBCCache.mPmtChargeVsTime[iPmt], bc.mPmtChargeVsTime[iPmt]);
242  double cfdZero = cfdWithOffset - FV0DigParam::Instance().avgCfdTimeForMip;
243 
244  if (cfdZero < -FV0DigParam::Instance().cfdCheckWindow || cfdZero > FV0DigParam::Instance().cfdCheckWindow) {
245  continue;
246  }
247  float totalCharge = IntegrateCharge(bc.mPmtChargeVsTime[iPmt]);
248  if (totalCharge > (FV0DigParam::Instance().maxCountInAdc * (1. / DP::INV_CHARGE_PER_ADC)) && FV0DigParam::Instance().useMaxChInAdc) {
249  totalChargeAllRing += FV0DigParam::Instance().maxCountInAdc * (1. / DP::INV_CHARGE_PER_ADC);
250  } else {
251  totalChargeAllRing += totalCharge;
252  }
253  totalCharge *= DP::INV_CHARGE_PER_ADC; //convert coulomb to adc
254  if (totalCharge > FV0DigParam::Instance().maxCountInAdc && FV0DigParam::Instance().useMaxChInAdc) {
255  totalCharge = FV0DigParam::Instance().maxCountInAdc; //max adc channel for one PMT
256  }
257  cfdZero *= DP::INV_TIME_PER_TDCCHANNEL;
258 
259  digitsCh.emplace_back(iPmt, std::lround(cfdZero), std::lround(totalCharge));
260  ++nStored;
261  //---trigger---
262  if (iPmt < 24) {
263  nSignalInner++;
264  } else {
265  nSignalOuter++;
266  }
267  }
268  // save BC information for the CFD detector
269  mLastBCCache = bc;
270  if (nStored < 1) {
271  return;
272  }
273  totalChargeAllRing *= DP::INV_CHARGE_PER_ADC;
274  //LOG(info)<<"Total charge ADC " <<totalChargeAllRing ;
276  bool isMinBias, isMinBiasInner, isMinBiasOuter, isHighMult, isDummy;
277  isMinBias = nStored > 0;
278  isMinBiasInner = nSignalInner > 0; //ring 1,2 and 3
279  isMinBiasOuter = nSignalOuter > 0; //ring 4 and 5
280  isHighMult = totalChargeAllRing > FV0DigParam::Instance().adcChargeHighMultTh;
281  isDummy = false;
282 
283  Triggers triggers;
284  triggers.setTriggers(isMinBias, isMinBiasInner, isMinBiasOuter, isHighMult, isDummy, nStored, totalChargeAllRing);
285  digitsBC.emplace_back(first, nStored, bc, triggers);
286  digitsTrig.emplace_back(bc, isMinBias, isMinBiasInner, isMinBiasOuter, isHighMult, isDummy);
287  for (auto const& lbl : bc.labels) {
288  labels.addElement(nBC, lbl);
289  }
290 }

296 Int_t Digitizer::SimulateLightYield(Int_t pmt, Int_t nPhot) const
297 {
298  const Float_t epsilon = 0.0001f;
299  const Float_t p = FV0DigParam::Instance().lightYield * FV0DigParam::Instance().photoCathodeEfficiency;
300  if ((fabs(1.0f - p) < epsilon) || nPhot == 0) {
301  return nPhot;
302  }
303  const Int_t n = Int_t(nPhot < 100
304  ? gRandom->Binomial(nPhot, p)
305  : gRandom->Gaus((p * nPhot) + 0.5, TMath::Sqrt(p * (1. - p) * nPhot)));
306  return n;
307 }

309 Float_t Digitizer::IntegrateCharge(const ChannelBCDataF& pulse) const
310 {
311  int const chargeIntMin = FV0DigParam::Instance().isIntegrateFull ? 0 : (FV0DigParam::Instance().avgCfdTimeForMip - 6.0) / mBinSize; //Charge integration offset (cfd mean time - 6 ns)
312  int const chargeIntMax = FV0DigParam::Instance().isIntegrateFull ? mNTimeBinsPerBC : (FV0DigParam::Instance().avgCfdTimeForMip + 14.0) / mBinSize; //Charge integration offset (cfd mean time + 14 ns)
313  if (chargeIntMin < 0 || chargeIntMin > mNTimeBinsPerBC || chargeIntMax > mNTimeBinsPerBC) {
314  LOG(fatal) << "invalid indicess: chargeInMin=" << chargeIntMin << " chargeIntMax=" << chargeIntMax;
315  }
316  Float_t totalCharge = 0.0f;
317  for (int iTimeBin = chargeIntMin; iTimeBin < chargeIntMax; iTimeBin++) {
318  totalCharge += pulse[iTimeBin];
319  }
320  return totalCharge;
321 }

323 Float_t Digitizer::SimulateTimeCfd(int& startIndex, const ChannelBCDataF& pulseLast, const ChannelBCDataF& pulse) const
324 {
325  Float_t timeCfd = -1024.0f;
326 
327  if (pulse.empty()) {
328  startIndex = 0;
329  return timeCfd;
330  }
331 
332  Float_t const cfdThrInCoulomb = FV0DigParam::Instance().mCFD_trsh * 1e-3 / 50 * mBinSize * 1e-9; // convert mV into Coulomb assuming 50 Ohm
333 
334  Int_t const binShift = TMath::Nint(FV0DigParam::Instance().timeShiftCfd / mBinSize);
335  Float_t sigPrev = 5 * pulseLast[mNTimeBinsPerBC - binShift - 1] - pulseLast[mNTimeBinsPerBC - 1]; // CFD output from the last bin of the last BC
336  for (Int_t iTimeBin = 0; iTimeBin < mNTimeBinsPerBC; ++iTimeBin) {
337  Float_t const sigCurrent = 5.0f * (iTimeBin >= binShift ? pulse[iTimeBin - binShift] : pulseLast[mNTimeBinsPerBC - binShift + iTimeBin]) - pulse[iTimeBin];
338  if (iTimeBin >= startIndex && std::abs(pulse[iTimeBin]) > cfdThrInCoulomb) { // enable
339  if (sigPrev < 0.0f && sigCurrent >= 0.0f) { // test for zero-crossing
340  timeCfd = Float_t(iTimeBin) * mBinSize;
341  startIndex = iTimeBin + std::lround(FV0DigParam::Instance().mCFDdeadTime / mBinSize); // update startIndex (CFD dead time)
342  if (startIndex < mNTimeBinsPerBC) {
343  startIndex = 0; // dead-time ends in same BC: no impact on the following BC
344  } else {
345  startIndex -= mNTimeBinsPerBC;
346  }
347  if (startIndex > mNTimeBinsPerBC) {
348  LOG(fatal) << "CFD dead-time was set to > 25 ns";
349  }
350  break; // only detects the 1st zero-crossing in the BC
351  }
352  }
353  sigPrev = sigCurrent;
354  }
355  return timeCfd;
356 }

358 float Digitizer::getDistFromCellCenter(UInt_t cellId, double hitx, double hity)
359 {
360  Geometry* geo = Geometry::instance();
361 
362  // Parametrize the line (ax+by+c=0) that crosses the detector center and the cell's middle point
363  Point3Dsimple* pCell = &geo->getCellCenter(cellId);
364  float x0, y0, z0;
365  geo->getGlobalPosition(x0, y0, z0);
366  double a = -(y0 - pCell->y) / (x0 - pCell->x);
367  double b = 1;
368  double c = -(y0 - a * x0);
369  //Return the distance from hit to this line
370  return (a * hitx + b * hity + c) / TMath::Sqrt(a * a + b * b);
371 }

384 o2::fv0::Digitizer::BCCache& Digitizer::setBCCache(const o2::InteractionRecord& ir)
385 {
386  if (mCache.empty() || mCache.back() < ir) {
387  mCache.emplace_back();
388  auto& cb = mCache.back();
389  cb = ir;
390  return cb;
391  }
392  if (mCache.front() > ir) {
393  mCache.emplace_front();
394  auto& cb = mCache.front();
395  cb = ir;
396  return cb;
397  }
398  for (auto cb = mCache.begin(); cb != mCache.end(); cb++) {
399  if ((*cb) == ir) {
400  return *cb;
401  }
402  if (ir < (*cb)) {
403  auto cbnew = mCache.emplace(cb); // insert new element before cb
404  (*cbnew) = ir;
405  return (*cbnew);
406  }
407  }
408  return mCache.front();
409 }
