46  void init(o2::framework::InitContext& ic) final

58  mCalibrator->setUpdateAtTheEndOfRunOnly();

61  //_________________________________________________________________________________________________
62  void logStats(size_t dataSize)
63  {
64  static auto loggerStart = std::chrono::high_resolution_clock::now();
65  static auto loggerEnd = loggerStart;
66  static size_t nDigits = 0;
67  static size_t nTF = 0;
68 
69  if (mLoggingInterval == 0) {
70  return;
71  }
72 
73  nDigits += dataSize;
74  nTF += 1;
75 
76  loggerEnd = std::chrono::high_resolution_clock::now();
77  std::chrono::duration<double, std::milli> loggerElapsed = loggerEnd - loggerStart;
78  if (loggerElapsed.count() > 1000) {
79  LOG(info) << "received " << nDigits << " digits in " << nTF << " time frames";
80  nDigits = 0;
81  nTF = 0;
82  loggerStart = std::chrono::high_resolution_clock::now();
83  }
84  }

86  //_________________________________________________________________________________________________
87  void run(o2::framework::ProcessingContext& pc) final
88  {
89  const o2::framework::DataProcessingHeader* header = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("digits").header);
90  if (!header) {
91  return;
92  }
93  auto tfcounter = header->startTime; // is this the timestamp of the current TF?
94 
95  auto data = pc.inputs().get<gsl::span<o2::mch::calibration::PedestalDigit>>("digits");
96  mCalibrator->process(tfcounter, data);
97 
98  logStats(data.size());
99  }

101  //_________________________________________________________________________________________________
102  void endOfStream(o2::framework::EndOfStreamContext& ec) final
103  {
104  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;

108  sendOutput(ec.outputs());

116  //________________________________________________________________
117  void sendOutput(DataAllocator& output)
118  {
119  auto createBuffer = [&](auto& vec, size_t& size) {
120  size = vec.empty() ? 0 : sizeof(*(vec.begin())) * vec.size();
121  char* buf = nullptr;
122  if (size > 0) {
123  buf = (char*)malloc(size);
124  if (buf) {
125  char* p = buf;
126  size_t sizeofElement = sizeof(*(vec.begin()));
127  for (auto& element : vec) {
128  memcpy(p, &element, sizeofElement);
129  p += sizeofElement;
130  }
131  }
132  }
133  return buf;
134  };
135 
136  // extract CCDB infos and calibration objects, convert it to TMemFile and send them to the output
137  // TODO in principle, this routine is generic, can be moved to Utils.h
138  using clbUtils = o2::calibration::Utils;
139  const auto& payload = mCalibrator->getBadChannelsVector();
140  auto& info = mCalibrator->getBadChannelsInfo(); // use non-const version as we update it
141  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
142  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
143  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
144  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "MCH_BADCHAN", 0}, *image.get());
145  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "MCH_BADCHAN", 0}, info); // root-serialized
146 
147  size_t pedestalsSize;
148  char* pedestalsBuffer = createBuffer(mCalibrator->getPedestalsVector(), pedestalsSize);
149  auto freefct = [](void* data, void*) { free(data); };
150  output.adoptChunk(Output{"MCH", "PEDESTALS", 0}, pedestalsBuffer, pedestalsSize, freefct, nullptr);
151 
152  mCalibrator->initOutput(); // reset the outputs once they are already sent
153  }

162 DataProcessorSpec getMCHPedestalCalibSpec(const char* specName, const std::string inputSpec)
163 {
164  constexpr int64_t INFINITE_TF = 0xffffffffffffffff;
165  using device = o2::mch::calibration::PedestalCalibDevice;
166  using clbUtils = o2::calibration::Utils;
167 
168  std::vector<OutputSpec> outputs;
169  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "MCH_BADCHAN"}, Lifetime::Sporadic);
170  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "MCH_BADCHAN"}, Lifetime::Sporadic);
171  outputs.emplace_back(OutputSpec{"MCH", "PEDESTALS", 0, Lifetime::Sporadic});
172 
173  return DataProcessorSpec{
174  specName,
175  select(inputSpec.data()),
176  outputs,
177  AlgorithmSpec{adaptFromTask<device>()},
178  Options{
179  {"logging-interval", VariantType::Int, 0, {"time interval in seconds between logging messages (set to zero to disable)"}},
180  {"tf-per-slot", VariantType::Int64, INFINITE_TF - 10, {"number of TFs per calibration time slot"}},
181  {"max-delay", VariantType::Int, 1, {"number of slots in past to consider"}},
182  {"pedestal-threshold", VariantType::Float, 200.0f, {"maximum allowed pedestal value"}},
183  {"noise-threshold", VariantType::Float, 2.0f, {"maximum allowed noise value"}},
184  {"ccdb-path", VariantType::String, o2::base::NameConf::getCCDBServer(), {"Path to CCDB"}}}};
185 }
