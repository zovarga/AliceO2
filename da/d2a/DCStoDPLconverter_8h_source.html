53 o2f::InjectorFunction dcs2dpl(std::unordered_map<DPID, o2h::DataDescription>& dpid2group, uint64_t startTime, uint64_t step, bool verbose = false)
54 {
55 
56  auto timesliceId = std::make_shared<size_t>(startTime);
57  return [dpid2group, timesliceId, step, verbose](FairMQDevice& device, FairMQParts& parts, o2f::ChannelRetriever channelRetriever) {
58  static std::unordered_map<DPID, DPCOM> cache; // will keep only the latest measurement in the 1-second wide window for each DPID
59  static auto timer = std::chrono::system_clock::now();
60 
61  LOG(debug) << "In lambda function: ********* Size of unordered_map (--> number of defined groups) = " << dpid2group.size();
62  // We first iterate over the parts of the received message
63  for (size_t i = 0; i < parts.Size(); ++i) { // DCS sends only 1 part, but we should be able to receive more
64  auto nDPCOM = parts.At(i)->GetSize() / sizeof(DPCOM); // number of DPCOM in current part
65  for (size_t j = 0; j < nDPCOM; j++) {
66  const auto& src = *(reinterpret_cast<const DPCOM*>(parts.At(i)->GetData()) + j);
67  // do we want to check if this DP was requested ?
68  auto mapEl = dpid2group.find(src.id);
69  if (verbose) {
70  LOG(info) << "Received DP " << src.id << " (data = " << src.data << "), matched to output-> " << (mapEl == dpid2group.end() ? "none " : mapEl->second.as<std::string>());
71  }
72  if (mapEl != dpid2group.end()) {
73  auto& dst = cache[src.id] = src; // this is needed in case in the 1s window we get a new value for the same DP
74  }
75  }
76  }
77 
78  auto timerNow = std::chrono::system_clock::now();
79  std::chrono::duration<double, std::ratio<1>> duration = timerNow - timer;
80  if (duration.count() > 1) { //did we accumulate for 1 sec?
81  *timesliceId += step; // we increment only if we send something
82  std::unordered_map<o2h::DataDescription, pmr::vector<DPCOM>, std::hash<o2h::DataDescription>> outputs;
83  // in the cache we have the final values of the DPs that we should put in the output
84  // distribute DPs over the vectors for each requested output
85  for (auto& it : cache) {
86  auto mapEl = dpid2group.find(it.first);
87  if (mapEl != dpid2group.end()) {
88  outputs[mapEl->second].push_back(it.second);
89  }
90  }
91  std::uint64_t creation = std::chrono::time_point_cast<std::chrono::milliseconds>(timerNow).time_since_epoch().count();
92  // create and send output messages
93  for (auto& it : outputs) {
94  o2h::DataHeader hdr(it.first, "DCS", 0);
95  o2f::OutputSpec outsp{hdr.dataOrigin, hdr.dataDescription, hdr.subSpecification};
96  if (it.second.empty()) {
97  LOG(warning) << "No data for OutputSpec " << outsp;
98  continue;
99  }
100  auto channel = channelRetriever(outsp, *timesliceId);
101  if (channel.empty()) {
102  LOG(warning) << "No output channel found for OutputSpec " << outsp << ", discarding its data";
103  it.second.clear();
104  continue;
105  }
106 
107  hdr.tfCounter = *timesliceId; // this also
108  hdr.payloadSerializationMethod = o2h::gSerializationMethodNone;
109  hdr.splitPayloadParts = 1;
110  hdr.splitPayloadIndex = 1;
111  hdr.payloadSize = it.second.size() * sizeof(DPCOM);
112  hdr.firstTForbit = 0; // this should be irrelevant for DCS
113  o2h::Stack headerStack{hdr, o2::framework::DataProcessingHeader{*timesliceId, 1, creation}};
114  auto fmqFactory = device.GetChannel(channel).Transport();
115  auto hdMessage = fmqFactory->CreateMessage(headerStack.size(), fair::mq::Alignment{64});
116  auto plMessage = fmqFactory->CreateMessage(hdr.payloadSize, fair::mq::Alignment{64});
117  memcpy(hdMessage->GetData(), headerStack.data(), headerStack.size());
118  memcpy(plMessage->GetData(), it.second.data(), hdr.payloadSize);
119  if (verbose) {
120  LOGP(info, "Pushing {} DPs to {} for TimeSlice {} at {}", it.second.size(), o2f::DataSpecUtils::describe(outsp), *timesliceId, creation);
121  }
122  it.second.clear();
123  FairMQParts outParts;
124  outParts.AddPart(std::move(hdMessage));
125  outParts.AddPart(std::move(plMessage));
126  o2f::sendOnChannel(device, outParts, channel);
127  }
128 
129  timer = timerNow;
130  cache.clear();
131  }
132  };
133 }
