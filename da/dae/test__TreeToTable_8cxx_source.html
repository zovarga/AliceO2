29 BOOST_AUTO_TEST_CASE(TreeToTableConversion)
30 {
32  Int_t ndp = 17;
33 
34  TFile f1("tree2table.root", "RECREATE");
35  TTree t1("t1", "a simple Tree with simple variables");
36  Bool_t ok, ts[5] = {false};
37  Float_t px, py, pz;
38  Double_t random;
39  Int_t ev;
40  uint8_t b;
41  const Int_t nelem = 9;
42  Double_t ij[nelem] = {0};
43  float xyzw[96];
44  memset(xyzw, 1, 96 * 4);
45  TString leaflist = Form("ij[%i]/D", nelem);
46 
47  Int_t ncols = 10;
48  t1.Branch("ok", &ok, "ok/O");
49  t1.Branch("px", &px, "px/F");
50  t1.Branch("py", &py, "py/F");
51  t1.Branch("pz", &pz, "pz/F");
52  t1.Branch("random", &random, "random/D");
53  t1.Branch("ev", &ev, "ev/I");
54  t1.Branch("ij", ij, leaflist.Data());
55  t1.Branch("tests", ts, "tests[5]/O");
56  t1.Branch("xyzw", xyzw, "xyzw[96]/F");
57  t1.Branch("small", &b, "small/b");
58 
59  //fill the tree
60  int ntruein[2] = {0};
61  for (int i = 0; i < ndp; i++) {
62  ok = (i % 2) == 0;
63  if (ok) {
64  ntruein[0]++;
65  }
66  gRandom->Rannor(px, py);
67  pz = px * px + py * py;
68  random = gRandom->Rndm();
69  ev = i + 1;
70  b = i % 3;
71  for (Int_t jj = 0; jj < nelem; jj++) {
72  ij[jj] = i + 100 * jj;
73  }
74  for (Int_t jj = 0; jj < 5; jj++) {
75  ts[jj] = (((i + jj) % 2) == 0);
76  if (ts[jj]) {
77  ntruein[1]++;
78  }
79  }
80 
81  t1.Fill();
82  }
83  t1.Write();
84 
85  // Create an arrow table from this.
86  TreeToTable tr2ta;
87  tr2ta.addAllColumns(&t1);
88  tr2ta.fill(&t1);
89  auto table = tr2ta.finalize();
90  f1.Close();
91 
92  // test result
93  BOOST_REQUIRE_EQUAL(table->Validate().ok(), true);
94  BOOST_REQUIRE_EQUAL(table->num_rows(), ndp);
95  BOOST_REQUIRE_EQUAL(table->num_columns(), ncols);
96 
97  BOOST_REQUIRE_EQUAL(table->column(0)->type()->id(), arrow::Type::BOOL);
98  BOOST_REQUIRE_EQUAL(table->column(1)->type()->id(), arrow::Type::FLOAT);
99  BOOST_REQUIRE_EQUAL(table->column(2)->type()->id(), arrow::Type::FLOAT);
100  BOOST_REQUIRE_EQUAL(table->column(3)->type()->id(), arrow::Type::FLOAT);
101  BOOST_REQUIRE_EQUAL(table->column(4)->type()->id(), arrow::Type::DOUBLE);
102  BOOST_REQUIRE_EQUAL(table->column(5)->type()->id(), arrow::Type::INT32);
103  BOOST_REQUIRE_EQUAL(table->column(6)->type()->id(), arrow::Type::FIXED_SIZE_LIST);
104  BOOST_REQUIRE_EQUAL(table->column(7)->type()->id(), arrow::Type::FIXED_SIZE_LIST);
105  BOOST_REQUIRE_EQUAL(table->column(8)->type()->id(), arrow::Type::FIXED_SIZE_LIST);
106  BOOST_REQUIRE_EQUAL(table->column(9)->type()->id(), arrow::Type::UINT8);
107 
108  BOOST_REQUIRE(table->column(0)->type()->Equals(arrow::boolean()));
109  BOOST_REQUIRE(table->column(1)->type()->Equals(arrow::float32()));
110  BOOST_REQUIRE(table->column(2)->type()->Equals(arrow::float32()));
111  BOOST_REQUIRE(table->column(3)->type()->Equals(arrow::float32()));
112  BOOST_REQUIRE(table->column(4)->type()->Equals(arrow::float64()));
113  BOOST_REQUIRE(table->column(5)->type()->Equals(arrow::int32()));
114  BOOST_REQUIRE(table->column(6)->type()->Equals(arrow::fixed_size_list(arrow::float64(), nelem)));
115  BOOST_REQUIRE(table->column(7)->type()->Equals(arrow::fixed_size_list(arrow::boolean(), 5)));
116  BOOST_REQUIRE(table->column(8)->type()->Equals(arrow::fixed_size_list(arrow::float32(), 96)));
117  BOOST_REQUIRE(table->column(9)->type()->Equals(arrow::uint8()));
118 
119  // count number of rows with ok==true
120  int ntrueout = 0;
121  auto chunks = table->column(0);
122  BOOST_REQUIRE_NE(chunks.get(), nullptr);
123 
124  auto oks = std::dynamic_pointer_cast<arrow::BooleanArray>(chunks->chunk(0));
125  BOOST_REQUIRE_NE(oks.get(), nullptr);
126 
127  for (int ii = 0; ii < table->num_rows(); ii++) {
128  ntrueout += oks->Value(ii) ? 1 : 0;
129  }
130  BOOST_REQUIRE_EQUAL(ntruein[0], ntrueout);
131 
132  // count number of ts with ts==true
133  chunks = table->column(7);
134  BOOST_REQUIRE_NE(chunks.get(), nullptr);
135 
136  auto chunkToUse = std::static_pointer_cast<arrow::FixedSizeListArray>(chunks->chunk(0))->values();
137  BOOST_REQUIRE_NE(chunkToUse.get(), nullptr);
138 
139  auto tests = std::dynamic_pointer_cast<arrow::BooleanArray>(chunkToUse);
140  ntrueout = 0;
141  for (int ii = 0; ii < table->num_rows() * 5; ii++) {
142  ntrueout += tests->Value(ii) ? 1 : 0;
143  }
144  BOOST_REQUIRE_EQUAL(ntruein[1], ntrueout);
145 
146  // save table as tree
147  TFile* f2 = TFile::Open("table2tree.root", "RECREATE");
148  TableToTree ta2tr(table, f2, "mytree");
149  ta2tr.addAllBranches();
150 
151  auto t2 = ta2tr.process();
152  auto br = (TBranch*)t2->GetBranch("ok");
153  BOOST_REQUIRE_EQUAL(t2->GetEntries(), ndp);
154  BOOST_REQUIRE_EQUAL(br->GetEntries(), ndp);
155  br = (TBranch*)t2->GetBranch("tests");
156  BOOST_REQUIRE_EQUAL(br->GetEntries(), ndp);
157 
158  f2->Close();
159 }
