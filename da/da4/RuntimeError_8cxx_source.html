90 void demangled_backtrace_symbols(void** stackTrace, unsigned int stackDepth, int fd)
91 {
92  char** stackStrings;
93  stackStrings = backtrace_symbols(stackTrace, stackDepth);
94  for (size_t i = 1; i < stackDepth; i++) {
95 
96  size_t sz = 64000; // 64K ought to be enough for our templates...
97  char* function = static_cast<char*>(malloc(sz));
98  char *begin = nullptr, *end = nullptr;
99  // find the last space and address offset surrounding the mangled name
100 #if __APPLE__
101  for (char* j = stackStrings[i]; *j; ++j) {
102  if (*j == ' ' && *(j + 1) != '+') {
103  begin = j;
104  } else if (*j == ' ' && *(j + 1) == '+') {
105  end = j;
106  break;
107  }
108  }
109 #else
110  for (char* j = stackStrings[i]; *j; ++j) {
111  if (*j == '(') {
112  begin = j;
113  } else if (*j == '+') {
114  end = j;
115  break;
116  }
117  }
118 #endif
119  if (begin && end) {
120  *begin++ = '\0';
121  *end = '\0';
122  // found our mangled name, now in [begin, end)
123 
124  int status;
125  char* ret = abi::__cxa_demangle(begin, function, &sz, &status);
126  if (ret) {
127  // return value may be a realloc() of the input
128  function = ret;
129  } else {
130  // demangling failed, just pretend it's a C function with no args
131  std::strncpy(function, begin, sz);
132  std::strncat(function, "()", sz);
133  function[sz - 1] = '\0';
134  }
135  dprintf(fd, " %s: %s\n", stackStrings[i], function);
136  } else {
137  // didn't find the mangled name, just print the whole line
138  dprintf(fd, " %s\n", stackStrings[i]);
139  }
140  free(function);
141  }
142  free(stackStrings); // malloc()ed by backtrace_symbols
143  fsync(fd);
144 }
