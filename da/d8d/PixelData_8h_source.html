32 class PixelData
33 {
34 
35  public:
36  PixelData(const Digit* dig) : mRow(dig->getRow()), mCol(dig->getColumn()) {}
37  PixelData(uint16_t r = 0, uint16_t c = 0) : mRow(r), mCol(c) {}
38  uint16_t getRow() const { return mRow & RowMask; }
39  uint16_t getCol() const { return mCol; }
40  bool isMasked() const { return mRow & MaskBit; }
41  void setMask() { mRow |= MaskBit; }
42  void unsetMask() { mRow &= RowMask; }
43 
45  uint16_t getRowDirect() const { return mRow; }
46 
47  bool operator==(const PixelData& dt) const
48  {
50  return (getCol() == dt.getCol()) && (getRow() == dt.getRow());
51  }
52 
53  bool operator>(const PixelData& dt) const
54  {
56  if (getCol() == dt.getCol()) {
57  return getRow() > dt.getRow();
58  }
59  return getCol() > dt.getCol();
60  }
61 
62  bool operator<(const PixelData& dt) const
63  {
65  if (getCol() == dt.getCol()) {
66  return getRow() < dt.getRow();
67  }
68  return getCol() < dt.getCol();
69  }
70 
71  bool isNeighbour(const PixelData& dt, int maxDist) const
72  {
74  return (std::abs(static_cast<int>(getCol()) - static_cast<int>(dt.getCol())) <= maxDist &&
75  std::abs(static_cast<int>(getRow()) - static_cast<int>(dt.getRow())) <= maxDist);
76  }
77 
78  int compare(const PixelData& dt) const
79  {
81  return operator==(dt) ? 0 : (operator>(dt) ? 1 : -1);
82  }
83 
84  static constexpr uint32_t DummyROF = 0xffffffff;
85  static constexpr uint32_t DummyChipID = 0xffff;
86 
87  private:
88  void sanityCheck() const;
89  static constexpr int RowMask = 0x7FFF;
90  static constexpr int MaskBit = 0x8000;
91  uint16_t mRow = 0;
92  uint16_t mCol = 0;
93 
94  ClassDefNV(PixelData, 1);
95 };

167  void maskFiredInSample(const ChipPixelData& sample)
168  {
170  const auto& pixelsS = sample.getData();
171  uint32_t nC = mPixels.size();
172  if (!nC) {
173  return;
174  }
175  uint32_t nS = pixelsS.size();
176  if (!nS) {
177  return;
178  }
179  uint32_t itC = 0, itS = 0;
180  while (itC < nC && itS < nS) {
181  auto& pix0 = mPixels[itC];
182  const auto& pixC = pixelsS[itS];
183  if (pix0 == pixC) { // same
184  pix0.setMask();
185  if (mFirstUnmasked == itC++) { // mFirstUnmasked should flag 1st unmasked pixel entry
186  mFirstUnmasked = itC;
187  }
188  itS++;
189  } else if (pix0 < pixC) {
190  itC++;
191  } else {
192  itS++;
193  }
194  }
195  }

197  void maskFiredInSample(const ChipPixelData& sample, int maxDist)
198  {
200  const auto& pixelsS = sample.getData();
201  int nC = mPixels.size();
202  if (!nC) {
203  return;
204  }
205  int nS = pixelsS.size();
206  if (!nS) {
207  return;
208  }
209  for (int itC = 0, itS = 0; itC < nC; itC++) {
210  auto& pix0 = mPixels[itC];
211 
212  // seek to itS which is inferior than itC - maxDist
213  auto mincol = pix0.getCol() > maxDist ? pix0.getCol() - maxDist : 0;
214  auto minrow = pix0.getRowDirect() > maxDist ? pix0.getRowDirect() - maxDist : 0;
215  if (itS == nS) { // in case itS lool below reached the end
216  itS--;
217  }
218  while ((pixelsS[itS].getCol() > mincol || pixelsS[itS].getRow() > minrow) && itS > 0) {
219  itS--;
220  }
221  for (; itS < nS; itS++) {
222  const auto& pixC = pixelsS[itS];
223 
224  auto drow = static_cast<int>(pixC.getRow()) - static_cast<int>(pix0.getRowDirect());
225  auto dcol = static_cast<int>(pixC.getCol()) - static_cast<int>(pix0.getCol());
226 
227  if (dcol > maxDist || (dcol == maxDist && drow > maxDist)) {
228  break; // all higher itS will not match to this itC also
229  }
230  if (dcol < -maxDist || (drow > maxDist || drow < -maxDist)) {
231  continue;
232  } else {
233  pix0.setMask();
234  if (int(mFirstUnmasked) == itC) { // mFirstUnmasked should flag 1st unmasked pixel entry
235  mFirstUnmasked = itC + 1;
236  }
237  break;
238  }
239  }
240  }
241  }
