63 template <class T, int I, typename... Args>
64 int GPUReconstructionCPUBackend::runKernelBackend(krnlSetup& _xyz, const Args&... args)
65 {
66  auto& x = _xyz.x;
67  auto& y = _xyz.y;
68  if (x.device == krnlDeviceType::Device) {
69  throw std::runtime_error("Cannot run device kernel on host");
70  }
71  if (x.nThreads != 1) {
72  throw std::runtime_error("Cannot run device kernel on host with nThreads != 1");
73  }
74  unsigned int num = y.num == 0 || y.num == -1 ? 1 : y.num;
75  for (unsigned int k = 0; k < num; k++) {
76  int ompThreads = mProcessingSettings.ompKernels ? (mProcessingSettings.ompKernels == 2 ? ((mProcessingSettings.ompThreads + mNestedLoopOmpFactor - 1) / mNestedLoopOmpFactor) : mProcessingSettings.ompThreads) : 1;
77  if (ompThreads > 1) {
78  if (mProcessingSettings.debugLevel >= 5) {
79  printf("Running %d ompThreads\n", ompThreads);
80  }
81  GPUCA_OPENMP(parallel for num_threads(ompThreads))
82  for (unsigned int iB = 0; iB < x.nBlocks; iB++) {
83  typename T::GPUSharedMemory smem;
84  T::template Thread<I>(x.nBlocks, 1, iB, 0, smem, T::Processor(*mHostConstantMem)[y.start + k], args...);
85  }
86  } else {
87  for (unsigned int iB = 0; iB < x.nBlocks; iB++) {
88  typename T::GPUSharedMemory smem;
89  T::template Thread<I>(x.nBlocks, 1, iB, 0, smem, T::Processor(*mHostConstantMem)[y.start + k], args...);
90  }
91  }
92  }
93  return 0;
94 }

118 size_t GPUReconstructionCPU::TransferMemoryResourcesHelper(GPUProcessor* proc, int stream, bool all, bool toGPU)
119 {
120  int inc = toGPU ? GPUMemoryResource::MEMORY_INPUT_FLAG : GPUMemoryResource::MEMORY_OUTPUT_FLAG;
121  int exc = toGPU ? GPUMemoryResource::MEMORY_OUTPUT_FLAG : GPUMemoryResource::MEMORY_INPUT_FLAG;
122  size_t n = 0;
123  for (unsigned int i = 0; i < mMemoryResources.size(); i++) {
124  GPUMemoryResource& res = mMemoryResources[i];
125  if (res.mPtr == nullptr) {
126  continue;
127  }
128  if (proc && res.mProcessor != proc) {
129  continue;
130  }
131  if (!(res.mType & GPUMemoryResource::MEMORY_GPU) || (res.mType & GPUMemoryResource::MEMORY_CUSTOM_TRANSFER)) {
132  continue;
133  }
134  if (!mProcessingSettings.keepAllMemory && !all && (res.mType & exc) && !(res.mType & inc)) {
135  continue;
136  }
137  if (toGPU) {
138  n += TransferMemoryResourceToGPU(&mMemoryResources[i], stream);
139  } else {
140  n += TransferMemoryResourceToHost(&mMemoryResources[i], stream);
141  }
142  }
143  return n;
144 }

190 int GPUReconstructionCPU::RunChains()
191 {
192  mMemoryScalers->temporaryFactor = 1.;
193  mStatNEvents++;
194  mNEventsProcessed++;
195 
196  timerTotal.Start();
197  if (mProcessingSettings.doublePipeline) {
198  if (EnqueuePipeline()) {
199  return 1;
200  }
201  } else {
202  if (mThreadId != GetThread()) {
203  if (mProcessingSettings.debugLevel >= 2) {
204  GPUInfo("Thread changed, migrating context, Previous Thread: %d, New Thread: %d", mThreadId, GetThread());
205  }
206  mThreadId = GetThread();
207  }
208  if (mSlaves.size() || mMaster) {
209  WriteConstantParams(); // Reinitialize
210  }
211  for (unsigned int i = 0; i < mChains.size(); i++) {
212  int retVal = mChains[i]->RunChain();
213  if (retVal) {
214  return retVal;
215  }
216  }
217  }
218  timerTotal.Stop();
219 
220  mStatWallTime = (timerTotal.GetElapsedTime() * 1000000. / mStatNEvents);
221  if (GetProcessingSettings().debugLevel >= 1) {
222  double kernelTotal = 0;
223  std::vector<double> kernelStepTimes(GPUDataTypes::N_RECO_STEPS);
224 
225  for (unsigned int i = 0; i < mTimers.size(); i++) {
226  double time = 0;
227  if (mTimers[i] == nullptr) {
228  continue;
229  }
230  for (int j = 0; j < mTimers[i]->num; j++) {
231  HighResTimer& timer = mTimers[i]->timer[j];
232  time += timer.GetElapsedTime();
233  if (mProcessingSettings.resetTimers) {
234  timer.Reset();
235  }
236  }
237 
238  char type = mTimers[i]->type;
239  if (type == 0) {
240  kernelTotal += time;
241  int stepNum = getRecoStepNum(mTimers[i]->step);
242  kernelStepTimes[stepNum] += time;
243  }
244  type = type == 0 ? 'K' : 'C';
245  char bandwidth[256] = "";
246  if (mTimers[i]->memSize && mStatNEvents && time != 0.) {
247  snprintf(bandwidth, 256, " (%6.3f GB/s - %'14lu bytes)", mTimers[i]->memSize / time * 1e-9, (unsigned long)(mTimers[i]->memSize / mStatNEvents));
248  }
249  printf("Execution Time: Task (%c %8ux): %50s Time: %'10d us%s\n", type, mTimers[i]->count, mTimers[i]->name.c_str(), (int)(time * 1000000 / mStatNEvents), bandwidth);
250  if (mProcessingSettings.resetTimers) {
251  mTimers[i]->count = 0;
252  mTimers[i]->memSize = 0;
253  }
254  }
255  for (int i = 0; i < GPUDataTypes::N_RECO_STEPS; i++) {
256  if (kernelStepTimes[i] != 0. || mTimersRecoSteps[i].timerTotal.GetElapsedTime() != 0.) {
257  printf("Execution Time: Step : %11s %38s Time: %'10d us ( Total Time : %'14d us)\n", "Tasks", GPUDataTypes::RECO_STEP_NAMES[i], (int)(kernelStepTimes[i] * 1000000 / mStatNEvents), (int)(mTimersRecoSteps[i].timerTotal.GetElapsedTime() * 1000000 / mStatNEvents));
258  }
259  if (mTimersRecoSteps[i].bytesToGPU) {
260  printf("Execution Time: Step (D %8ux): %11s %38s Time: %'10d us (%6.3f GB/s - %'14lu bytes - %'14lu per call)\n", mTimersRecoSteps[i].countToGPU, "DMA to GPU", GPUDataTypes::RECO_STEP_NAMES[i], (int)(mTimersRecoSteps[i].timerToGPU.GetElapsedTime() * 1000000 / mStatNEvents),
261  mTimersRecoSteps[i].bytesToGPU / mTimersRecoSteps[i].timerToGPU.GetElapsedTime() * 1e-9, mTimersRecoSteps[i].bytesToGPU / mStatNEvents, mTimersRecoSteps[i].bytesToGPU / mTimersRecoSteps[i].countToGPU);
262  }
263  if (mTimersRecoSteps[i].bytesToHost) {
264  printf("Execution Time: Step (D %8ux): %11s %38s Time: %'10d us (%6.3f GB/s - %'14lu bytes - %'14lu per call)\n", mTimersRecoSteps[i].countToHost, "DMA to Host", GPUDataTypes::RECO_STEP_NAMES[i], (int)(mTimersRecoSteps[i].timerToHost.GetElapsedTime() * 1000000 / mStatNEvents),
265  mTimersRecoSteps[i].bytesToHost / mTimersRecoSteps[i].timerToHost.GetElapsedTime() * 1e-9, mTimersRecoSteps[i].bytesToHost / mStatNEvents, mTimersRecoSteps[i].bytesToHost / mTimersRecoSteps[i].countToHost);
266  }
267  if (mProcessingSettings.resetTimers) {
268  mTimersRecoSteps[i].bytesToGPU = mTimersRecoSteps[i].bytesToHost = 0;
269  mTimersRecoSteps[i].timerToGPU.Reset();
270  mTimersRecoSteps[i].timerToHost.Reset();
271  mTimersRecoSteps[i].timerTotal.Reset();
272  mTimersRecoSteps[i].countToGPU = 0;
273  mTimersRecoSteps[i].countToHost = 0;
274  }
275  }
276  for (int i = 0; i < GPUDataTypes::N_GENERAL_STEPS; i++) {
277  if (mTimersGeneralSteps[i].GetElapsedTime() != 0.) {
278  printf("Execution Time: General Step : %50s Time: %'10d us\n", GPUDataTypes::GENERAL_STEP_NAMES[i], (int)(mTimersGeneralSteps[i].GetElapsedTime() * 1000000 / mStatNEvents));
279  }
280  }
281  mStatKernelTime = kernelTotal * 1000000 / mStatNEvents;
282  printf("Execution Time: Total : %50s Time: %'10d us\n", "Total Kernel", (int)mStatKernelTime);
283  printf("Execution Time: Total : %50s Time: %'10d us\n", "Total Wall", (int)mStatWallTime);
284  } else if (GetProcessingSettings().debugLevel >= 0) {
285  GPUInfo("Total Wall Time: %d us", (int)mStatWallTime);
286  }
287  if (mProcessingSettings.resetTimers) {
288  mStatNEvents = 0;
289  timerTotal.Reset();
290  }
291 
292  return 0;
293 }

316 GPUReconstructionCPU::timerMeta* GPUReconstructionCPU::insertTimer(unsigned int id, std::string&& name, int J, int num, int type, RecoStep step)
317 {
318  while (timerFlag.test_and_set()) {
319  ;
320  }
321  if (mTimers.size() <= id) {
322  mTimers.resize(id + 1);
323  }
324  if (mTimers[id] == nullptr) {
325  if (J >= 0) {
326  name += std::to_string(J);
327  }
328  mTimers[id].reset(new timerMeta{std::unique_ptr<HighResTimer[]>{new HighResTimer[num]}, name, num, type, 1u, step, (size_t)0});
329  } else {
330  mTimers[id]->count++;
331  }
332  timerMeta* retVal = mTimers[id].get();
333  timerFlag.clear();
334  return retVal;
335 }
