77 template <typename source_T, typename dest_T, std::enable_if_t<(sizeof(dest_T) >= sizeof(source_T)), bool> = true>
78 inline size_t calculateNDestTElements(size_t sourceElems) noexcept
79 {
80  const size_t sizeOfSourceArray = sourceElems * sizeof(source_T);
81  return sizeOfSourceArray / sizeof(dest_T) + (sizeOfSourceArray % sizeof(dest_T) != 0);
82 };

84 template <typename source_T, typename dest_T, std::enable_if_t<(sizeof(dest_T) >= sizeof(source_T)), bool> = true>
85 inline size_t calculatePaddedSize(size_t nElems) noexcept
86 {
87  const size_t sizeOfSourceT = sizeof(source_T);
88  const size_t sizeOfDestT = sizeof(dest_T);
89 
90  // this is equivalent to (sizeOfSourceT / sizeOfDestT) * std::ceil(sizeOfSourceArray/ sizeOfDestT)
91  return (sizeOfDestT / sizeOfSourceT) * calculateNDestTElements<source_T, dest_T>(nElems);
92 };

164 template <typename W = uint32_t>
165 struct Block {
166 
167  Registry* registry = nullptr;
168  int nDict = 0; // dictionary length (if any)
169  int nData = 0; // length of data
170  int nLiterals = 0; // length of literals vector (if any)
171  int nStored = 0; // total length
172  W* payload = nullptr; //[nStored];
173 
174  inline const W* getDict() const { return nDict ? payload : nullptr; }
175  inline const W* getData() const { return nData ? (payload + nDict) : nullptr; }
176  inline const W* getDataPointer() const { return payload ? (payload + nDict) : nullptr; } // needed when nData is not set yet
177  inline const W* getLiterals() const { return nLiterals ? (payload + nDict + nData) : nullptr; }
178 
179  inline W* getCreatePayload() { return payload ? payload : (registry ? (payload = reinterpret_cast<W*>(registry->getFreeBlockStart())) : nullptr); }
180  inline W* getCreateDict() { return payload ? payload : getCreatePayload(); }
181  inline W* getCreateData() { return payload ? (payload + nDict) : getCreatePayload(); }
182  inline W* getCreateLiterals() { return payload ? payload + (nDict + nData) : getCreatePayload(); }
183 
184  inline auto getOffsDict() { return reinterpret_cast<std::uintptr_t>(getCreateDict()) - reinterpret_cast<std::uintptr_t>(registry->head); }
185  inline auto getOffsData() { return reinterpret_cast<std::uintptr_t>(getCreateData()) - reinterpret_cast<std::uintptr_t>(registry->head); }
186  inline auto getOffsLiterals() { return reinterpret_cast<std::uintptr_t>(getCreateLiterals()) - reinterpret_cast<std::uintptr_t>(registry->head); }
187 
188  inline void setNDict(int _ndict)
189  {
190  nDict = _ndict;
191  nStored += nDict;
192  }
193 
194  inline void setNData(int _ndata)
195  {
196  nData = _ndata;
197  nStored += nData;
198  }
199 
200  inline void setNLiterals(int _nliterals)
201  {
202  nLiterals = _nliterals;
203  nStored += nLiterals;
204  }
205 
206  inline int getNDict() const { return nDict; }
207  inline int getNData() const { return nData; }
208  inline int getNLiterals() const { return nLiterals; }
209  inline int getNStored() const { return nStored; }
210 
211  ~Block()
212  {
213  if (!registry) { // this is a standalone block owning its data
214  delete[] payload;
215  }
216  }
217 
219  void clear()
220  {
221  nDict = 0;
222  nData = 0;
223  nLiterals = 0;
224  nStored = 0;
225  payload = nullptr;
226  }
227 
229  static size_t estimateSize(int n)
230  {
231  return alignSize(n * sizeof(W));
232  }
233 
234  // store a dictionary in an empty block
235  void storeDict(int _ndict, const W* _dict)
236  {
237  if (getNStored() > 0) {
238  throw std::runtime_error("trying to write in occupied block");
239  }
240  size_t sz = estimateSize(_ndict);
241  assert(registry); // this method is valid only for flat version, which has a registry
242  assert(sz <= registry->getFreeSize());
243  assert((_ndict > 0) == (_dict != nullptr));
244  setNDict(_ndict);
245  if (nDict) {
246  memcpy(getCreateDict(), _dict, _ndict * sizeof(W));
247  realignBlock();
248  }
249  };
250 
251  // store a dictionary to a block which can either be empty or contain a dict.
252  void storeData(int _ndata, const W* _data)
253  {
254  if (getNStored() > getNDict()) {
255  throw std::runtime_error("trying to write in occupied block");
256  }
257 
258  size_t sz = estimateSize(_ndata);
259  assert(registry); // this method is valid only for flat version, which has a registry
260  assert(sz <= registry->getFreeSize());
261  assert((_ndata > 0) == (_data != nullptr));
262  setNData(_ndata);
263  if (nData) {
264  memcpy(getCreateData(), _data, _ndata * sizeof(W));
265  realignBlock();
266  }
267  }
268 
269  // store a dictionary to a block which can either be empty or contain a dict.
270  void storeLiterals(int _nliterals, const W* _literals)
271  {
272  if (getNStored() > getNDict() + getNData()) {
273  throw std::runtime_error("trying to write in occupied block");
274  }
275 
276  size_t sz = estimateSize(_nliterals);
277  assert(registry); // this method is valid only for flat version, which has a registry
278  assert(sz <= registry->getFreeSize());
279  // assert((_nliterals > 0) == (_literals != nullptr));
280  setNLiterals(_nliterals);
281  if (nLiterals) {
282  memcpy(getCreateLiterals(), _literals, _nliterals * sizeof(W));
283  realignBlock();
284  }
285  }
286 
287  // resize block and free up unused buffer space.
288  void realignBlock()
289  {
290  size_t sz = estimateSize(getNStored());
291  registry->offsFreeStart = (reinterpret_cast<char*>(payload) - registry->head) + sz;
292  }
293 
295  void store(int _ndict, int _ndata, int _nliterals, const W* _dict, const W* _data, const W* _literals)
296  {
297  size_t sz = estimateSize(_ndict + _ndata + _nliterals);
298  assert(registry); // this method is valid only for flat version, which has a registry
299  assert(sz <= registry->getFreeSize());
300  assert((_ndict > 0) == (_dict != nullptr));
301  assert((_ndata > 0) == (_data != nullptr));
302  // assert(_literals == _data + _nliterals);
303  setNDict(_ndict);
304  setNData(_ndata);
305  setNLiterals(_nliterals);
306  getCreatePayload(); // do this even for empty block!!!
307  if (getNStored()) {
308  payload = reinterpret_cast<W*>(registry->getFreeBlockStart());
309  if (getNDict()) {
310  memcpy(getCreateDict(), _dict, _ndict * sizeof(W));
311  }
312  if (getNData()) {
313  memcpy(getCreateData(), _data, _ndata * sizeof(W));
314  }
315  if (getNLiterals()) {
316  memcpy(getCreateLiterals(), _literals, _nliterals * sizeof(W));
317  }
318  }
319  realignBlock();
320  }
321 
323  void relocate(const char* oldHead, char* newHeadData, char* newHeadRegistry)
324  {
325  payload = relocatePointer(oldHead, newHeadData, payload);
326  registry = relocatePointer(oldHead, newHeadRegistry, registry);
327  }

511 template <typename H, int N, typename W>
512 template <typename VD>
513 void EncodedBlocks<H, N, W>::readFromTree(VD& vec, TTree& tree, const std::string& name, int ev)
514 {
515  auto tmp = create(vec);
516  if (!readTreeBranch(tree, o2::utils::Str::concat_string(name, "_wrapper."), *tmp, ev)) {
517  throw std::runtime_error(fmt::format("Failed to read CTF header for {}", name));
518  }
519  tmp = tmp->expand(vec, tmp->estimateSizeFromMetadata());
520  const auto& meta = tmp->getMetadata();
521  for (int i = 0; i < N; i++) {
522  Block<W> bl;
523  readTreeBranch(tree, o2::utils::Str::concat_string(name, "_block.", std::to_string(i), "."), bl, ev);
524  assert(meta[i].nDictWords == bl.getNDict());
525  assert(meta[i].nDataWords == bl.getNData());
526  assert(meta[i].nLiteralWords == bl.getNLiterals());
527  tmp->mBlocks[i].store(bl.getNDict(), bl.getNData(), bl.getNLiterals(), bl.getDict(), bl.getData(), bl.getLiterals());
528  }
529 }

533 template <typename H, int N, typename W>
534 size_t EncodedBlocks<H, N, W>::appendToTree(TTree& tree, const std::string& name) const
535 {
536  long s = 0;
537  s += fillTreeBranch(tree, o2::utils::Str::concat_string(name, "_wrapper."), const_cast<base&>(*this), WrappersCompressionLevel, WrappersSplitLevel);
538  for (int i = 0; i < N; i++) {

540  s += fillTreeBranch(tree, o2::utils::Str::concat_string(name, "_block.", std::to_string(i), "."), const_cast<Block<W>&>(mBlocks[i]), compression);
541  }
542  tree.SetEntries(tree.GetEntries() + 1);
543  return s;
544 }

548 template <typename H, int N, typename W>
549 template <typename D>
550 bool EncodedBlocks<H, N, W>::readTreeBranch(TTree& tree, const std::string& brname, D& dt, int ev)
551 {
552  auto* br = tree.GetBranch(brname.c_str());
553  if (!br) {
554  LOG(debug) << "Branch " << brname << " is absent";
555  return false;
556  }
557  auto* ptr = &dt;
558  br->SetAddress(&ptr);
559  br->GetEntry(ev);
560  br->ResetAddress();
561  return true;
562 }

570  auto* br = tree.GetBranch(brname.c_str());

574  }

635 {

765 template <typename H, int N, typename W>
766 template <typename D_IT, std::enable_if_t<detail::is_iterator_v<D_IT>, bool>>
767 void EncodedBlocks<H, N, W>::decode(D_IT dest, // iterator to destination
768  int slot, // slot of the block to decode
769  const void* decoderExt) const // optional externally provided decoder
770 {
771  // get references to the right data
772  const auto& block = mBlocks[slot];
773  const auto& md = mMetadata[slot];
774 
775  using dest_t = typename std::iterator_traits<D_IT>::value_type;
776 
777  // decode
778  if (block.getNStored()) {
779  if (md.opt == Metadata::OptStore::EENCODE) {
780  if (!decoderExt && !block.getNDict()) {
781  LOG(error) << "Dictionaty is not saved for slot " << slot << " and no external decoder is provided";
782  throw std::runtime_error("Dictionary is not saved and no external decoder provided");
783  }
784  const o2::rans::LiteralDecoder64<dest_t>* decoder = reinterpret_cast<const o2::rans::LiteralDecoder64<dest_t>*>(decoderExt);
785  std::unique_ptr<o2::rans::LiteralDecoder64<dest_t>> decoderLoc;
786  if (block.getNDict()) { // if dictionaty is saved, prefer it
787  decoderLoc = std::make_unique<o2::rans::LiteralDecoder64<dest_t>>(this->getFrequencyTable(slot));
788  decoder = decoderLoc.get();
789  } else { // verify that decoded corresponds to stored metadata
790  if (md.min != decoder->getMinSymbol()) {
791  LOG(error) << "Mismatch between min=" << md.min << " symbol in metadata and those in external decoder "
792  << decoder->getMinSymbol() << " for slot " << slot;
793  throw std::runtime_error("Mismatch between min symbol in metadata and the one in external decoder");
794  }
795  }
796  // load incompressible symbols if they existed
797  std::vector<dest_t> literals;
798  if (block.getNLiterals()) {
799  // note: here we have to use md.nLiterals (original number of literal words) rather than md.nLiteralWords == block.getNLiterals()
800  // (number of W-words in the EncodedBlock occupied by literals) as we cast literals stored in W-word array
801  // to D-word array
802  literals = std::vector<dest_t>{reinterpret_cast<const dest_t*>(block.getLiterals()), reinterpret_cast<const dest_t*>(block.getLiterals()) + md.nLiterals};
803  }
804  decoder->process(block.getData() + block.getNData(), dest, md.messageLength, literals);
805  } else { // data was stored as is
806  using destPtr_t = typename std::iterator_traits<D_IT>::pointer;
807  destPtr_t srcBegin = reinterpret_cast<destPtr_t>(block.payload);
808  destPtr_t srcEnd = srcBegin + md.messageLength * sizeof(dest_t);
809  std::copy(srcBegin, srcEnd, dest);
810  // std::memcpy(dest, block.payload, md.messageLength * sizeof(dest_t));
811  }
812  }
813 }

816 template <typename H, int N, typename W>
817 template <typename input_IT, typename buffer_T>
818 void EncodedBlocks<H, N, W>::encode(const input_IT srcBegin, // iterator begin of source message
819  const input_IT srcEnd, // iterator end of source message
820  int slot, // slot in encoded data to fill
821  uint8_t symbolTablePrecision, // encoding into
822  Metadata::OptStore opt, // option for data compression
823  buffer_T* buffer, // optional buffer (vector) providing memory for encoded blocks
824  const void* encoderExt, // optional external encoder
825  float memfc) // memory allocation margin factor
826 {
827 
828  using storageBuffer_t = W;
829  using input_t = typename std::iterator_traits<input_IT>::value_type;
830  using ransEncoder_t = typename rans::LiteralEncoder64<input_t>;
831  using ransState_t = typename ransEncoder_t::coder_t;
832  using ransStream_t = typename ransEncoder_t::stream_t;
833 
834  // assert at compile time that output types align so that padding is not necessary.
835  static_assert(std::is_same_v<storageBuffer_t, ransStream_t>);
836  static_assert(std::is_same_v<storageBuffer_t, typename rans::count_t>);
837 
838  // fill a new block
839  assert(slot == mRegistry.nFilledBlocks);
840  mRegistry.nFilledBlocks++;
841 
842  const size_t messageLength = std::distance(srcBegin, srcEnd);
843  // cover three cases:
844  // * empty source message: no entropy coding
845  // * source message to pass through without any entropy coding
846  // * source message where entropy coding should be applied
847 
848  // case 1: empty source message
849  if (messageLength == 0) {
850  mMetadata[slot] = Metadata{0, 0, sizeof(input_t), sizeof(ransState_t), sizeof(ransStream_t), symbolTablePrecision, Metadata::OptStore::NODATA, 0, 0, 0, 0, 0};
851  return;
852  }
853 
854  auto* thisBlock = &mBlocks[slot];
855  auto* thisMetadata = &mMetadata[slot];
856 
857  // resize underlying buffer of block if necessary and update all pointers.
858  auto expandStorage = [&](int additionalElements) {
859  auto* const blockHead = get(thisBlock->registry->head); // extract pointer from the block, as "this" might be invalid
860  const size_t additionalSize = blockHead->estimateBlockSize(additionalElements); // size in bytes!!!
861  if (additionalSize >= thisBlock->registry->getFreeSize()) {
862  LOG(debug) << "Slot " << slot << ": free size: " << thisBlock->registry->getFreeSize() << ", need " << additionalSize << " for " << additionalElements << " words";
863  if (buffer) {
864  blockHead->expand(*buffer, blockHead->size() + (additionalSize - blockHead->getFreeSize()));
865  thisMetadata = &(get(buffer->data())->mMetadata[slot]);
866  thisBlock = &(get(buffer->data())->mBlocks[slot]); // in case of resizing this and any this.xxx becomes invalid
867  } else {
868  throw std::runtime_error("no room for encoded block in provided container");
869  }
870  }
871  };
872 
873  // case 3: message where entropy coding should be applied
874  if (opt == Metadata::OptStore::EENCODE) {
875  // build symbol statistics
876  constexpr size_t SizeEstMarginAbs = 10 * 1024;
877  const float SizeEstMarginRel = 1.5 * memfc;
878 
879  const auto [inplaceEncoder, frequencyTable] = [&]() {
880  if (encoderExt) {
881  return std::make_tuple(ransEncoder_t{}, rans::FrequencyTable{});
882  } else {
883  rans::FrequencyTable frequencyTable = rans::makeFrequencyTableFromSamples(srcBegin, srcEnd);
884  RenormedFrequencyTable renormedFrequencyTable = rans::renorm(frequencyTable, symbolTablePrecision);
885  return std::make_tuple(ransEncoder_t{renormedFrequencyTable}, frequencyTable);
886  }
887  }();
888  ransEncoder_t const* const encoder = encoderExt ? reinterpret_cast<ransEncoder_t const* const>(encoderExt) : &inplaceEncoder;
889 
890  // estimate size of encode buffer
891  int dataSize = rans::calculateMaxBufferSize(messageLength, encoder->getAlphabetRangeBits(), sizeof(input_t)); // size in bytes
892  // preliminary expansion of storage based on dict size + estimated size of encode buffer
893  dataSize = SizeEstMarginAbs + int(SizeEstMarginRel * (dataSize / sizeof(storageBuffer_t))) + (sizeof(input_t) < sizeof(storageBuffer_t)); // size in words of output stream
894  expandStorage(frequencyTable.size() + dataSize);
895  // store dictionary first
896  if (!frequencyTable.empty()) {
897  thisBlock->storeDict(frequencyTable.size(), frequencyTable.data());
898  LOGP(debug, "StoreDict {} bytes, offs: {}:{}", frequencyTable.size() * sizeof(W), thisBlock->getOffsDict(), thisBlock->getOffsDict() + frequencyTable.size() * sizeof(W));
899  }
900  // vector of incompressible literal symbols
901  std::vector<input_t> literals;
902  // directly encode source message into block buffer.
903  storageBuffer_t* const blockBufferBegin = thisBlock->getCreateData();
904  const size_t maxBufferSize = thisBlock->registry->getFreeSize(); // note: "this" might be not valid after expandStorage call!!!
905  const auto encodedMessageEnd = encoder->process(srcBegin, srcEnd, blockBufferBegin, literals);
906  rans::utils::checkBounds(encodedMessageEnd, blockBufferBegin + maxBufferSize / sizeof(W));
907  dataSize = encodedMessageEnd - thisBlock->getDataPointer();
908  thisBlock->setNData(dataSize);
909  thisBlock->realignBlock();
910  LOGP(debug, "StoreData {} bytes, offs: {}:{}", dataSize * sizeof(W), thisBlock->getOffsData(), thisBlock->getOffsData() + dataSize * sizeof(W));
911  // update the size claimed by encode message directly inside the block
912 
913  // store incompressible symbols if any
914  const size_t nLiteralSymbols = literals.size();
915  const size_t nLiteralWords = [&]() {
916  if (!literals.empty()) {
917  const size_t nSymbols = literals.size();
918  // introduce padding in case literals don't align;
919  const size_t nLiteralSymbolsPadded = calculatePaddedSize<input_t, storageBuffer_t>(nSymbols);
920  literals.resize(nLiteralSymbolsPadded, {});
921 
922  const size_t nLiteralStorageElems = calculateNDestTElements<input_t, storageBuffer_t>(nSymbols);
923  expandStorage(nLiteralStorageElems);
924  thisBlock->storeLiterals(nLiteralStorageElems, reinterpret_cast<const storageBuffer_t*>(literals.data()));
925  LOGP(debug, "StoreLiterals {} bytes, offs: {}:{}", nLiteralStorageElems * sizeof(W), thisBlock->getOffsLiterals(), thisBlock->getOffsLiterals() + nLiteralStorageElems * sizeof(W));
926  return nLiteralStorageElems;
927  }
928  return size_t(0);
929  }();
930 
931  *thisMetadata = Metadata{messageLength,
932  nLiteralSymbols,
933  sizeof(input_t),
934  sizeof(ransState_t),
935  sizeof(ransStream_t),
936  static_cast<uint8_t>(encoder->getSymbolTablePrecision()),
937  opt,
938  encoder->getMinSymbol(),
939  encoder->getMaxSymbol(),
940  static_cast<int32_t>(frequencyTable.size()),
941  dataSize,
942  static_cast<int32_t>(nLiteralWords)};
943  } else { // store original data w/o EEncoding
944  // FIXME(milettri): we should be able to do without an intermediate vector;
945  // provided iterator is not necessarily pointer, need to use intermediate vector!!!
946 
947  // introduce padding in case literals don't align;
948  const size_t nSourceElemsPadded = calculatePaddedSize<input_t, storageBuffer_t>(messageLength);
949  std::vector<input_t> tmp(nSourceElemsPadded, {});
950  std::copy(srcBegin, srcEnd, std::begin(tmp));
951 
952  const size_t nBufferElems = calculateNDestTElements<input_t, storageBuffer_t>(messageLength);
953  expandStorage(nBufferElems);
954  thisBlock->storeData(thisMetadata->nDataWords, reinterpret_cast<const storageBuffer_t*>(tmp.data()));
955 
956  *thisMetadata = Metadata{messageLength, 0, sizeof(input_t), sizeof(ransState_t), sizeof(storageBuffer_t), symbolTablePrecision, opt, 0, 0, 0, static_cast<int>(nBufferElems), 0};
957  }
958 }

961 template <typename H, int N, typename W>
962 std::vector<char> EncodedBlocks<H, N, W>::createDictionaryBlocks(const std::vector<o2::rans::FrequencyTable>& vfreq, const std::vector<Metadata>& vmd)
963 {
964  if (vfreq.size() != N) {
965  throw std::runtime_error(fmt::format("mismatch between the size of frequencies vector {} and number of blocks {}", vfreq.size(), N));
966  }
967  size_t sz = alignSize(sizeof(EncodedBlocks<H, N, W>));
968  for (int ib = 0; ib < N; ib++) {
969  sz += Block<W>::estimateSize(vfreq[ib].size());
970  }
971  std::vector<char> vdict(sz); // memory space for dictionary
972  auto dictBlocks = create(vdict.data(), sz);
973  for (int ib = 0; ib < N; ib++) {
974  if (vfreq[ib].size()) {
975  LOG(info) << "adding dictionary of " << vfreq[ib].size() << " words for block " << ib << ", min/max= " << vfreq[ib].getMinSymbol() << "/" << vfreq[ib].getMaxSymbol();
976  dictBlocks->mBlocks[ib].storeDict(vfreq[ib].size(), vfreq[ib].data());
977  dictBlocks = get(vdict.data()); // !!! rellocation might have invalidated dictBlocks pointer
978  dictBlocks->mMetadata[ib] = vmd[ib];
979  dictBlocks->mMetadata[ib].opt = Metadata::OptStore::ROOTCompression; // we will compress the dictionary with root!
980  dictBlocks->mBlocks[ib].realignBlock();
981  } else {
982  dictBlocks->mMetadata[ib].opt = Metadata::OptStore::NONE;
983  }
984  dictBlocks->mRegistry.nFilledBlocks++;
985  }
986  return std::move(vdict);
987 }
