79 void TOFMatchChecker::checkMatching(GID gid)
80 {
81  if (!gid.includesDet(DetID::TOF)) {
82  return;
83  }
84  const o2::dataformats::MatchInfoTOF& match = mRecoData.getTOFMatch(gid);
85 
86  int trksource = 5;
87  if (gid.getSource() == GID::TPCTOF) {
88  trksource = 0;
89  } else if (gid.getSource() == GID::ITSTPCTOF) {
90  trksource = 1;
91  } else if (gid.getSource() == GID::TPCTRDTOF) {
92  trksource = 2;
93  } else if (gid.getSource() == GID::ITSTPCTRDTOF) {
94  trksource = 3;
95  }
96 
97  const char* sources[5] = {"TPC", "ITS-TPC", "TPC-TRD", "ITS-TPC-TRD", "NONE"};
98 
99  int tofcl = match.getIdxTOFCl();
100  int trIndex = match.getTrackIndex();
101  float chi2 = match.getChi2();
102  float ttof = mTOFClustersArrayInp[tofcl].getTime();
103  float x = mTOFClustersArrayInp[tofcl].getX();
104  float y = mTOFClustersArrayInp[tofcl].getY();
105  float z = mTOFClustersArrayInp[tofcl].getZ();
106  int sector = mTOFClustersArrayInp[tofcl].getSector();
107  LOG(info) << "trkSource=" << sources[trksource] << " -- cl=" << tofcl << " - trk=" << trIndex << " - chi2 =" << chi2 << " - time=" << ttof << " - coordinates (to be rotated, sector=" << sector << ") = (" << x << "," << y << "," << z << ")";
108 
109  // digit coordinates
110  int mainCh = mTOFClustersArrayInp[tofcl].getMainContributingChannel();
111  int addCh;
112  int det[5];
113  float pos[3];
114  o2::tof::Geo::getVolumeIndices(mainCh, det);
115  o2::tof::Geo::getPos(det, pos);
116  LOG(debug) << "Cluster mult = " << mTOFClustersArrayInp[tofcl].getNumOfContributingChannels() << " - main channel (" << mainCh << ") -> pos(" << pos[0] << "," << pos[1] << "," << pos[2] << ")";
117 
118  // top->+48, bottom->-48, left->-1, right->+1
119  if (mTOFClustersArrayInp[tofcl].getNumOfContributingChannels() > 1) {
120  if (mTOFClustersArrayInp[tofcl].isAdditionalChannelSet(o2::tof::Cluster::kUpRight)) {
121  addCh = mainCh + 48 + 1;
122  o2::tof::Geo::getVolumeIndices(addCh, det);
123  o2::tof::Geo::getPos(det, pos);
124  LOG(debug) << "top right (" << addCh << ") -> pos(" << pos[0] << "," << pos[1] << "," << pos[2] << ")";
125  }
126  if (mTOFClustersArrayInp[tofcl].isAdditionalChannelSet(o2::tof::Cluster::kUp)) {
127  addCh = mainCh + 48;
128  o2::tof::Geo::getVolumeIndices(addCh, det);
129  o2::tof::Geo::getPos(det, pos);
130  LOG(debug) << "top (" << addCh << ") -> pos(" << pos[0] << "," << pos[1] << "," << pos[2] << ")";
131  }
132  if (mTOFClustersArrayInp[tofcl].isAdditionalChannelSet(o2::tof::Cluster::kUpLeft)) {
133  addCh = mainCh + 48 - 1;
134  o2::tof::Geo::getVolumeIndices(addCh, det);
135  o2::tof::Geo::getPos(det, pos);
136  LOG(debug) << "top left (" << addCh << ") -> pos(" << pos[0] << "," << pos[1] << "," << pos[2] << ")";
137  }
138  if (mTOFClustersArrayInp[tofcl].isAdditionalChannelSet(o2::tof::Cluster::kRight)) {
139  addCh = mainCh + 1;
140  o2::tof::Geo::getVolumeIndices(addCh, det);
141  o2::tof::Geo::getPos(det, pos);
142  LOG(debug) << "right (" << addCh << ") -> pos(" << pos[0] << "," << pos[1] << "," << pos[2] << ")";
143  }
144  if (mTOFClustersArrayInp[tofcl].isAdditionalChannelSet(o2::tof::Cluster::kLeft)) {
145  addCh = mainCh - 1;
146  o2::tof::Geo::getVolumeIndices(addCh, det);
147  o2::tof::Geo::getPos(det, pos);
148  LOG(debug) << "left (" << addCh << ") -> pos(" << pos[0] << "," << pos[1] << "," << pos[2] << ")";
149  }
150  if (mTOFClustersArrayInp[tofcl].isAdditionalChannelSet(o2::tof::Cluster::kDownRight)) {
151  addCh = mainCh - 48 + 1;
152  o2::tof::Geo::getVolumeIndices(addCh, det);
153  o2::tof::Geo::getPos(det, pos);
154  LOG(debug) << "down right (" << addCh << ") -> pos(" << pos[0] << "," << pos[1] << "," << pos[2] << ")";
155  }
156  if (mTOFClustersArrayInp[tofcl].isAdditionalChannelSet(o2::tof::Cluster::kDown)) {
157  addCh = mainCh - 48;
158  o2::tof::Geo::getVolumeIndices(addCh, det);
159  o2::tof::Geo::getPos(det, pos);
160  LOG(debug) << "down (" << addCh << ") -> pos(" << pos[0] << "," << pos[1] << "," << pos[2] << ")";
161  }
162  if (mTOFClustersArrayInp[tofcl].isAdditionalChannelSet(o2::tof::Cluster::kDownLeft)) {
163  addCh = mainCh - 48 - 1;
164  o2::tof::Geo::getVolumeIndices(addCh, det);
165  o2::tof::Geo::getPos(det, pos);
166  LOG(debug) << "down left (" << addCh << ") -> pos(" << pos[0] << "," << pos[1] << "," << pos[2] << ")";
167  }
168  }
169 
170  LOG(debug) << "";
171 }

183 void TOFMatchChecker::run(ProcessingContext& pc)
184 {
185  mTimer.Start(false);
186 
187  mRecoData.collectData(pc, *mDataRequest.get());
188 
189  mIsTPC = (mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTOF) && mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTOF));
190  mIsITSTPC = (mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTOF) && mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTOF));
191  mIsITSTPCTRD = (mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRD) && mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRDTOF));
192  mIsTPCTRD = (mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRD) && mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRDTOF));
193 
194  mTOFClustersArrayInp = mRecoData.getTOFClusters();
195 
196  LOG(debug) << "isTrackSourceLoaded: TPC -> " << mIsTPC << " (t=" << mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTOF) << ",m=" << mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTOF) << ")";
197  LOG(debug) << "isTrackSourceLoaded: ITSTPC -> " << mIsITSTPC << " (t=" << mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTOF) << ",m=" << mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTOF) << ")";
198  LOG(debug) << "isTrackSourceLoaded: TPCTRD -> " << mIsTPCTRD << " (t=" << mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRDTOF) << ",m=" << mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRDTOF) << ")";
199  LOG(debug) << "isTrackSourceLoaded: ITSTPCTRD -> " << mIsITSTPCTRD << " (t=" << mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRDTOF) << ",m=" << mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRDTOF) << ")";
200  LOG(debug) << "TOF cluster size = " << mTOFClustersArrayInp.size();
201 
202  if (!mTOFClustersArrayInp.size()) {
203  return;
204  }
205 
206  auto creator = [this](auto& trk, GID gid, float time0, float terr) {
207  this->checkMatching(gid);
208  return true;
209  };
210  mRecoData.createTracksVariadic(creator);
211 
212  mTimer.Stop();
213 }
