59 void EntropyEncoderSpec::run(ProcessingContext& pc)
60 {
61  auto cput = mTimer.CpuTime();
62  mTimer.Start(false);
63  auto rofs = pc.inputs().get<gsl::span<o2::mch::ROFRecord>>("rofs");
64  auto digits = pc.inputs().get<gsl::span<o2::mch::Digit>>("digits");
65 
66  auto& buffer = pc.outputs().make<std::vector<o2::ctf::BufferType>>(Output{"MCH", "CTFDATA", 0, Lifetime::Timeframe});
67  mCTFCoder.encode(buffer, rofs, digits);
68  auto eeb = CTF::get(buffer.data()); // cast to container pointer
69  eeb->compactify(); // eliminate unnecessary padding
70  buffer.resize(eeb->size()); // shrink buffer to strictly necessary size
71  // eeb->print();
72  mTimer.Stop();
73  LOG(info) << fmt::format("Created encoded data ({} digits and {} rofs) of size {} ({:5.1f} MB) for MCH in {:5.1f} s ",
74  digits.size(), rofs.size(), eeb->size(), eeb->size() / 1024.0 / 1024, mTimer.CpuTime() - cput);
75 }

84 {
85  std::vector<InputSpec> inputs;

87  inputs.emplace_back("digits", "MCH", "DIGITS", 0, Lifetime::Timeframe);
88 
89  return DataProcessorSpec{

91  inputs,

93  AlgorithmSpec{adaptFromTask<EntropyEncoderSpec>()},
