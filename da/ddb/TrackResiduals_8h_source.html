656 //_____________________________________________________
657 inline float TrackResiduals::getDXI(int ix) const
658 {
659  if (mUniformBins[VoxX]) {
660  return mDXI;
661  } else {
662  if (ix < param::NRowsPerROC[0]) {
663  // we are in the IROC
664  return 1.f / param::RowDX[0];
665  } else if (ix > param::NRowsAccumulated[param::NROCTypes - 2]) {
666  // we are in the last OROC
667  return 1.f / param::RowDX[param::NROCTypes - 1];
668  } else if (ix < param::NRowsAccumulated[1]) {
669  // OROC1
670  return 1.f / param::RowDX[1];
671  } else {
672  // OROC2
673  return 1.f / param::RowDX[2];
674  }
675  }
676 }

688  return (0.5f + ip) * mDY2X[ix] - mMaxY2X[ix];

737 //_____________________________________________________
738 inline int TrackResiduals::getXBinExact(float x) const
739 {
740  if (mUniformBins[VoxX]) {
741  int ix = (x - param::MinX) * mDXI;
742  return (ix < 0 || ix >= mNXBins) ? -1 : ix;
743  }
744  return getRowID(x);
745 }

751  return bx > -1 ? (bx < mNXBins ? bx : mNXBins - 1) : 0;
752 }
753 
754 //_____________________________________________________
755 inline int TrackResiduals::getY2XBinExact(float y2x, int ix) const
756 {
757  if (y2x < -mMaxY2X[ix]) {
758  return -1;
759  }
760  if (y2x > mMaxY2X[ix]) {
761  return mNY2XBins;
762  }
763  if (mUniformBins[VoxF]) {
764  return static_cast<int>((y2x + mMaxY2X[ix]) * getDY2XI(ix));
765  }
766  y2x /= mMaxY2X[ix];
767  for (int iBin = 0; iBin < mNY2XBins; ++iBin) {
768  if (y2x < mY2XBinsCenter[iBin] + mY2XBinsDH[iBin]) {
769  return iBin;
770  }
771  }
772  return mNY2XBins;
773 }

775 //_____________________________________________________
776 inline int TrackResiduals::getY2XBin(float y2x, int ix) const
777 {
778  int bp = getY2XBinExact(y2x, ix);
779  return bp > -1 ? (bp < mNY2XBins ? bp : mNY2XBins - 1) : 0;
780 }

782 //_____________________________________________________
783 inline int TrackResiduals::getZ2XBinExact(float z2x) const
784 {
785  if (mUniformBins[VoxZ]) {
786  float bz = z2x * getDZ2XI();
787  if (bz >= mNZ2XBins) {
788  return -1;
789  }
790  if (bz < 0) {
791  // accounting for clusters which were moved to the wrong side
792  // TODO: how can this happen?
793  bz = 0;
794  }
795  return static_cast<int>(bz);
796  }
797  for (int iBin = 0; iBin < mNZ2XBins; ++iBin) {
798  if (z2x < mZ2XBinsCenter[iBin] + mZ2XBinsDH[iBin]) {
