156  currentsaveddatacount += rdhpayload;
157  totaldataread += offsetToNext;
158  // move to next rdh
159  rdh = reinterpret_cast<const o2::header::RDHAny*>(reinterpret_cast<const char*>(rdh) + offsetToNext);
160  if ((char*)(rdh) < (char*)&mHBFPayload[0] + mDataBufferSize) {
161  //if (reinterpret_cast<const o2::header::RDHAny*>(rdh) < (char*)&mHBFPayload[0] + mDataBufferSize) {
162  // we can still copy into this buffer.
163  } else {

165  LOG(warn) << "next rdh exceeds the bounds of the cru payload buffer";

167  if (mVerbose) {
168  LOG(info) << "rdh position is out of bounds of the buffer";
169  o2::raw::RDHUtils::printRDH(rdh);
170  }
171  }
172  return false; //-1;
173  }
174  }
175  //increment the data pointer by the size of the stop rdh.
176  mDataPointer = reinterpret_cast<const uint32_t*>(reinterpret_cast<const char*>(rdh) + o2::raw::RDHUtils::getOffsetToNext(rdh));
177  // at this point the entire HBF data payload is sitting in mHBFPayload and the total data count is mTotalHBFPayLoad
178  int counthalfcru = 0;
179  mHBFoffset32 = 0;
180 
181  while ((mHBFoffset32 < ((mTotalHBFPayLoad) / 4))) { // the blank event of eeeeee at the end
182  if (mHeaderVerbose) {
183  LOG(info) << "Looping over cruheaders in HBF, loop count " << counthalfcru << " current offset is" << mHBFoffset32 << " total payload is " << mTotalHBFPayLoad / 4 << " raw :" << mTotalHBFPayLoad;
184  }
185  int halfcruprocess = processHalfCRU(mHBFoffset32);
186  if (mVerbose) {
187  switch (halfcruprocess) {
188  case -1:
189  LOG(info) << "ignored rdh event ";

191  case 0:
192  LOG(error) << "figure out what now";
193  break;
194  case 1:
195  LOG(info) << "all good parsing half cru";
196  break;
197  default:
198  return true;
199  }
200  }
201  //take care of the case where there is an "empty" rdh containing all 0xeeeeeeee as payload.
202  if (mTotalHBFPayLoad / 4 - mHBFoffset32 == 8 && mHBFPayload[mHBFoffset32 + 7] == o2::trd::constants::CRUPADDING32) {
203  mHBFoffset32 += 8;
204  }

212  }

268 int CruRawReader::parseDigitHCHeader()
269 {
270  //mHBFoffset is the current offset into the current buffer,
271  //
272  uint32_t dhcheader = mHBFPayload[mHBFoffset32++];
273  std::array<uint32_t, 4> headers{0};
274  if (mByteSwap) {
275  // byte swap if needed.
276  o2::trd::HelperMethods::swapByteOrder(dhcheader);
277  }
278  mDigitHCHeader.word = dhcheader;
279  if (mDigitHCHeader.major == 0 && mDigitHCHeader.minor == 0 && mDigitHCHeader.numberHCW == 0) {
280  //hack this data into something resembling usable.
281  mDigitHCHeader.major = mHalfChamberMajor;
282  mDigitHCHeader.minor = 42; // to keep me entertained
283  mDigitHCHeader.numberHCW = mHalfChamberWords;
284  if (mHalfChamberWords == 0 || mHalfChamberMajor == 0) {
285  //LOG(warn) << "we have a messed up halfchamber header and you have only set the halfchamber command line option to zero, hex dump of data and revisit what it should be.";
286  // already in histograms
287  }
288  }
289 
290  int additionalHeaderWords = mDigitHCHeader.numberHCW;
291  if (additionalHeaderWords >= 3) {
292  increment2dHist(TRDParsingDigitHeaderCountGT3, mFEEID.supermodule * 2 + mHalfChamberSide[0], mStack[0], mLayer[0]);
293  //TODO graph this and stats it
294  if (mMaxErrsPrinted > 0) {
295  LOG(alarm) << "Error parsing DigitHCHeader, too many additional words count=" << additionalHeaderWords;
296  printDigitHCHeader(mDigitHCHeader, &headers[0]);
297  checkNoErr();
298  }
299  return -1;
300  }
301  for (int headerwordcount = 0; headerwordcount < additionalHeaderWords; ++headerwordcount) {
302  headers[headerwordcount] = mHBFPayload[mHBFoffset32++];
303  if (mByteSwap) {
304  // byte swap if needed.
305  o2::trd::HelperMethods::swapByteOrder(headers[headerwordcount]);
306  }
307  switch (getDigitHCHeaderWordType(headers[headerwordcount])) {
308  case 1: // header header1;
309  mDigitHCHeader1.word = headers[headerwordcount];
310  if (mDigitHCHeader1.res != 0x1) {
311  //LOG(alarm) << "Digit HC Header 1 reserved : " << std::hex << mDigitHCHeader1.res << " raw: 0x" << mDigitHCHeader1.word;
312  increment2dHist(TRDParsingDigitHeaderWrong1, mFEEID.supermodule * 2 + mHalfChamberSide[0], mStack[0], mLayer[0]);
313  }
314  break;
315  case 2: // header header2;
316  mDigitHCHeader2.word = headers[headerwordcount];
317  if (mDigitHCHeader2.res != 0b110001) {
318  // LOG(alarm) << "Digit HC Header 2 reserved : " << std::hex << mDigitHCHeader2.res << " raw: 0x" << mDigitHCHeader2.word;
319  increment2dHist(TRDParsingDigitHeaderWrong2, mFEEID.supermodule * 2 + mHalfChamberSide[0], mStack[0], mLayer[0]);
320  }
321  break;
322  case 3: // header header3;
323  mDigitHCHeader3.word = headers[headerwordcount];
324  if (mDigitHCHeader3.res != 0b110101) {
325  // LOG(alarm) << "Digit HC Header 3 reserved : " << std::hex << mDigitHCHeader3.res << " raw: 0x" << mDigitHCHeader3.word;
326  increment2dHist(TRDParsingDigitHeaderWrong3, mFEEID.supermodule * 2 + mHalfChamberSide[0], mStack[0], mLayer[0]);
327  }
328  break;
329  default:
330  //LOG(alarm) << "Error parsing DigitHCHeader at word:" << headerwordcount << " looking at 0x:" << std::hex << mHBFPayload[mHBFoffset32 - 1];
331  increment2dHist(TRDParsingDigitHeaderWrong4, mFEEID.supermodule * 2 + mHalfChamberSide[0], mStack[0], mLayer[0]);
332  }
333  }
334  if (mHeaderVerbose) {
335  printDigitHCHeader(mDigitHCHeader, &headers[0]);
336  }
337  return 1;
