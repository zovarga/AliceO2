21 // Composed Label to encode MC event and the source (file) + weight
22 
23 class MCEventLabel
24 {
25  private:
26  static constexpr uint32_t Dummy = 0xffffffff;
27  uint32_t mLabel = Dummy;
28 
29  public:
30  static constexpr int nbitsEvID = 16; // number of bits reserved for MC event ID
31  static constexpr int nbitsSrcID = 7; // number of bits reserved for MC source ID
32  static constexpr int nbitsCorrW = sizeof(mLabel) * 8 - nbitsEvID - nbitsSrcID - 1;
33  static constexpr uint32_t NotSet = (0x1 << (nbitsEvID + nbitsSrcID)) - 1;
34 
35  // Mask to extract MC event ID
36  static constexpr uint32_t MaskEvID = (0x1 << nbitsEvID) - 1;
37  // Mask to extract MC source ID
38  static constexpr uint32_t MaskSrcID = (0x1 << nbitsSrcID) - 1;
39  // Mask to extract MC source and event ID only
40  static constexpr uint32_t MaskSrcEvID = (MaskSrcID << nbitsEvID) | MaskEvID;
41  // Mask to extract MC correct contribution weight
42  static constexpr uint32_t MaskCorrW = (0x1 << nbitsCorrW) - 1;
43  static constexpr float WeightNorm = 1. / float(MaskCorrW);
44 
45  MCEventLabel(int evID, int srcID, float corrw = 1.0) { set(evID, srcID, corrw); }
46  MCEventLabel() = default;
47  ~MCEventLabel() = default;
48 
49  // check if label was assigned
50  bool isSet() const { return (mLabel & NotSet) != NotSet; }
51  // check if label was not assigned
52  bool isEmpty() const { return (mLabel & NotSet) == NotSet; }
53 
54  // conversion operator
55  operator uint32_t() const { return mLabel; }
56  // allow to retrieve bare label
57  uint32_t getRawValue() const { return mLabel; }
58 
59  // get only combined identifier, discarding weight info
60  uint32_t getIDOnly() const { return mLabel & MaskSrcEvID; }
61 
62  // compare
63  bool compare(const MCEventLabel& other, bool strict = false) const
64  {
65  return strict ? (getRawValue() == other.getRawValue()) : (getIDOnly() == other.getIDOnly());
66  }
67 
68  // comparison operator, compares only label, not eventual weight or correctness info
69  bool operator==(const MCEventLabel& other) const { return compare(other); }
70 
71  // invalidate
72  void unset() { mLabel = NotSet; }
73 
75  void set(int evID, int srcID, float corrW)
76  {
77  uint32_t iw = static_cast<uint32_t>(std::round(corrW * MaskCorrW));
78  assert(iw <= MaskCorrW);
79  mLabel = (iw << (nbitsEvID + nbitsSrcID)) | ((MaskSrcID & static_cast<uint32_t>(srcID)) << nbitsEvID) | (MaskEvID & static_cast<uint32_t>(evID));
80  }
81  void setCorrWeight(float corrW)
82  {
83  uint32_t iw = static_cast<uint32_t>(std::round(corrW * MaskCorrW));
84  assert(iw <= MaskCorrW);
85  mLabel = (mLabel & ((MaskSrcID << nbitsEvID) | MaskEvID)) | (iw << (nbitsEvID + nbitsSrcID));
86  }
87 
88  int getEventID() const
89  {
90  auto res = mLabel & MaskEvID;
91  return res <= MaxEventID() ? res : -1;
92  }
93 
94  int getSourceID() const
95  {
96  auto res = (mLabel >> nbitsEvID) & MaskSrcID;
97  return res <= MaxSourceID() ? res : -1;
98  }
99 
100  float getCorrWeight() const { return ((mLabel >> (nbitsEvID + nbitsSrcID)) & MaskCorrW) * WeightNorm; }
101 
102  void get(int& evID, int& srcID, float& corrW)
103  {
105  evID = getEventID();
106  srcID = getSourceID();
107  corrW = getCorrWeight();
108  }
109 
110  void print() const;
111 
112  static constexpr uint32_t MaxSourceID() { return MaskSrcID - 1; }
113  static constexpr uint32_t MaxEventID() { return MaskEvID - 1; }
114  static constexpr float WeightPrecision() { return WeightNorm; }
115  ClassDefNV(MCEventLabel, 1);
116 };
