84 void AlignableSensorTRD::dPosTraDParCalib(const AlignmentPoint* pnt, double* deriv, int calibID, const AlignableVolume* parent) const
85 {
86  // calculate point position X,Y,Z derivatives wrt calibration parameter calibID of given parent
87  // parent=0 means top detector object calibration
88  //
89  deriv[0] = deriv[1] = deriv[2] = 0;
90  //
91  if (!parent) { // TRD detector global calibration
92  //
93  switch (calibID) {
94  case AlignableDetectorTRD::kCalibNRCCorrDzDtgl: { // correction for Non-Crossing tracklets Z,Y shift: Z -> Z + calib*tgl, Y -> Y + calib*tgl*tilt*sign(tilt);
95  double sgYZ = pnt->getYZErrTracking()[1]; // makes sense only for nonRC tracklets
96  if (Abs(sgYZ) > 0.01) {
97  const double kTilt = 2. * TMath::DegToRad();
98  deriv[2] = pnt->getTrParamWSA()[AlignmentPoint::kParTgl];
99  deriv[1] = deriv[2] * Sign(kTilt, sgYZ);
100  }
101  break;
102  }
103  //
104  case AlignableDetectorTRD::kCalibDVT: { // correction for bias in VdriftT
105  // error in VdriftT equivalent to shift in X at which Y measurement is evaluated
106  // Y -> Y + dVdriftT * tg_phi, where tg_phi is the slope of the track in YX plane
107  double snp = pnt->getTrParamWSA(AlignmentPoint::kParSnp), slpY = snp / Sqrt((1 - snp) * (1 + snp));
108  deriv[1] = slpY;
109  break;
110  }
111 
112  default:
113  break;
114  };
115  }
116  //
117 }

120 AlignmentPoint* AlignableSensorTRD::TrackPoint2AlgPoint(int pntId, const AliTrackPointArray* trpArr, const AliESDtrack* tr)
121 {
122  // convert the pntId-th point to AlignmentPoint
123  //
124  AlignableDetectorTRD* det = (AlignableDetectorTRD*)getDetector();
125  AlignmentPoint* pnt = det->getPointFromPool();
126  pnt->setSensor(this);
127  //
128  double tra[3], locId[3], loc[3],
129  glo[3] = {trpArr->GetX()[pntId], trpArr->GetY()[pntId], trpArr->GetZ()[pntId]};
130  const TGeoHMatrix& matL2Grec = getMatrixL2GReco(); // local to global matrix used for reconstruction
131  const TGeoHMatrix& matT2L = getMatrixT2L(); // matrix for tracking to local frame translation
132  //
133  // undo reco-time alignment
134  matL2Grec.MasterToLocal(glo, locId); // go to local frame using reco-time matrix, here we recover ideal measurement
135  //
136  getMatrixClAlg().LocalToMaster(locId, loc); // apply alignment
137  //
138  matT2L.MasterToLocal(loc, tra); // go to tracking frame
139  //
140  /*
141  double gloT[3];
142  TGeoHMatrix t2g;
143  getMatrixT2G(t2g); t2g.LocalToMaster(tra,gloT);
144  printf("\n%5d %s\n",getVolID(), getSymName());
145  printf("GloOR: %+.4e %+.4e %+.4e\n",glo[0],glo[1],glo[2]);
146  printf("LocID: %+.4e %+.4e %+.4e\n",locId[0],locId[1],locId[2]);
147  printf("LocML: %+.4e %+.4e %+.4e\n",loc[0],loc[1],loc[2]);
148  printf("Tra : %+.4e %+.4e %+.4e\n",tra[0],tra[1],tra[2]);
149  printf("GloTR: %+.4e %+.4e %+.4e\n",gloT[0],gloT[1],gloT[2]);
150  */
151  //
152  if (!det->getUseErrorParam()) {
153  // convert error
154  TGeoHMatrix hcov;
155  double hcovel[9];
156  const float* pntcov = trpArr->GetCov() + pntId * 6; // 6 elements per error matrix
157  hcovel[0] = double(pntcov[0]);
158  hcovel[1] = double(pntcov[1]);
159  hcovel[2] = double(pntcov[2]);
160  hcovel[3] = double(pntcov[1]);
161  hcovel[4] = double(pntcov[3]);
162  hcovel[5] = double(pntcov[4]);
163  hcovel[6] = double(pntcov[2]);
164  hcovel[7] = double(pntcov[4]);
165  hcovel[8] = double(pntcov[5]);
166  hcov.SetRotation(hcovel);
167  hcov.Multiply(&matL2Grec);
168  const TGeoHMatrix& l2gi = matL2Grec;
169  hcov.MultiplyLeft(&l2gi); // errors in local frame
170  hcov.Multiply(&matT2L);
171  const TGeoHMatrix& t2li = matT2L.Inverse();
172  hcov.MultiplyLeft(&t2li); // errors in tracking frame
173  //
174  double* hcovscl = hcov.GetRotationMatrix();
175  const double* sysE = getAddError(); // additional syst error
176  pnt->setYZErrTracking(hcovscl[4] + sysE[0] * sysE[0], hcovscl[5], hcovscl[8] + sysE[1] * sysE[1]);
177  } else { // errors will be calculated just before using the point in the fit, using track info
178  pnt->setYZErrTracking(0, 0, 0);
179  pnt->setNeedUpdateFromTrack();
180  }
181  pnt->setXYZTracking(tra[0], tra[1], tra[2]);
182  pnt->setAlphaSens(getAlpTracking());
183  pnt->setXSens(getXTracking());
184  pnt->setDetID(det->getDetID());
185  pnt->setSID(getSID());
186  //
187  pnt->setContainsMeasurement();
188  //
189  // Apply calibrations
190  // Correction for NonRC points to account for most probable Z for non-crossing
191  {
192  const double kTilt = 2. * TMath::DegToRad();
193  // is it pad crrossing?
194  double* errYZ = (double*)pnt->getYZErrTracking();
195  double sgYZ = errYZ[1];
196  if (TMath::Abs(sgYZ) < 0.01) { // crossing
197  // increase errors since the error
198  const double* extraErrRC = det->GetExtraErrRC();
199  errYZ[0] += extraErrRC[0] * extraErrRC[0];
200  errYZ[2] += extraErrRC[1] * extraErrRC[1];
201  } else { // account for probability to not cross the row
202  double* pYZ = (double*)pnt->getYZTracking();
203  double corrZ = det->GetNonRCCorrDzDtglWithCal() * tr->GetTgl();
204  pYZ[1] += corrZ;
205  pYZ[0] += corrZ * Sign(kTilt, sgYZ); // Y and Z are correlated
206  }
207  }
208  //
209  // Correction for DVT, equivalent to shift in X at which Y is evaluated: dY = tg_phi * dvt
210  {
211  double dvt = det->GetCorrDVTWithCal();
212  if (Abs(dvt) > AlmostZeroD) {
213  AliExternalTrackParam trc = *tr;
214  if (!trc.RotateParamOnly(getAlpTracking()))
215  return 0;
216  double snp = trc.GetSnpAt(pnt->getXPoint(), AliTrackerBase::GetBz());
217  if (Abs(snp) > AlmostOneD)
218  return 0;
219  double slpY = snp / Sqrt((1 - snp) * (1 + snp));
220  double* pYZ = (double*)pnt->getYZTracking();
221  pYZ[0] += dvt * slpY;
222  }
223  }
224  //
225  pnt->init();
226  //
227  return pnt;
228  //
229 }
