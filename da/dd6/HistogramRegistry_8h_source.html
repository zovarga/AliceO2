193 template <typename T, typename... Ts>
194 void HistFiller::fillHistAny(std::shared_ptr<T> hist, const Ts&... positionAndWeight)
195 {
196  constexpr int nArgs = sizeof...(Ts);
197 
198  constexpr bool validTH3 = (std::is_same_v<TH3, T> && (nArgs == 3 || nArgs == 4));
199  constexpr bool validTH2 = (std::is_same_v<TH2, T> && (nArgs == 2 || nArgs == 3));
200  constexpr bool validTH1 = (std::is_same_v<TH1, T> && (nArgs == 1 || nArgs == 2));
201  constexpr bool validTProfile3D = (std::is_same_v<TProfile3D, T> && (nArgs == 4 || nArgs == 5));
202  constexpr bool validTProfile2D = (std::is_same_v<TProfile2D, T> && (nArgs == 3 || nArgs == 4));
203  constexpr bool validTProfile = (std::is_same_v<TProfile, T> && (nArgs == 2 || nArgs == 3));
204 
205  constexpr bool validSimpleFill = validTH1 || validTH2 || validTH3 || validTProfile || validTProfile2D || validTProfile3D;
206  // unfortunately we dont know at compile the dimension of THn(Sparse)
207  constexpr bool validComplexFill = std::is_base_of_v<THnBase, T>;
208  constexpr bool validComplexFillStep = std::is_base_of_v<StepTHn, T>;
209 
210  if constexpr (validSimpleFill) {
211  hist->Fill(static_cast<double>(positionAndWeight)...);
212  } else if constexpr (validComplexFillStep) {
213  hist->Fill(positionAndWeight...); // first argument in pack is iStep, dimension check is done in StepTHn itself
214  } else if constexpr (validComplexFill) {
215  double tempArray[] = {static_cast<double>(positionAndWeight)...};
216  double weight{1.};
217  constexpr int nArgsMinusOne = nArgs - 1;
218  if (hist->GetNdimensions() == nArgsMinusOne) {
219  weight = tempArray[nArgsMinusOne];
220  } else if (hist->GetNdimensions() != nArgs) {
221  LOGF(fatal, "The number of arguments in fill function called for histogram %s is incompatible with histogram dimensions.", hist->GetName());
222  }
223  hist->Fill(tempArray, weight);
224  } else {
225  LOGF(fatal, "The number of arguments in fill function called for histogram %s is incompatible with histogram dimensions.", hist->GetName());
226  }
227 }

229 template <typename... Cs, typename R, typename T>
230 void HistFiller::fillHistAny(std::shared_ptr<R> hist, const T& table, const o2::framework::expressions::Filter& filter)
231 {
232  if constexpr (std::is_base_of_v<StepTHn, T>) {
233  LOGF(fatal, "Table filling is not (yet?) supported for StepTHn.");
234  return;
235  }
236  auto s = o2::framework::expressions::createSelection(table.asArrowTable(), filter);
237  auto filtered = o2::soa::Filtered<T>{{table.asArrowTable()}, s};
238  for (auto& t : filtered) {
239  fillHistAny(hist, (*(static_cast<Cs>(t).getIterator()))...);
240  }
241 }

243 template <typename T>
244 double HistFiller::getSize(std::shared_ptr<T> hist, double fillFraction)
245 {
246  double size{0.};
247  if constexpr (std::is_base_of_v<TH1, T>) {
248  size = hist->GetNcells() * (HistFiller::getBaseElementSize(hist.get()) + ((hist->GetSumw2()->fN) ? sizeof(double) : 0.));
249  } else if constexpr (std::is_base_of_v<THn, T>) {
250  size = hist->GetNbins() * (HistFiller::getBaseElementSize(hist.get()) + ((hist->GetSumw2() != -1.) ? sizeof(double) : 0.));
251  } else if constexpr (std::is_base_of_v<THnSparse, T>) {
252  // THnSparse has massive overhead and should only be used when histogram is large and a very small fraction of bins is filled
253  double nBinsTotal = 1.;
254  int compCoordSize = 0; // size required to store a compact coordinate representation
255  for (int d = 0; d < hist->GetNdimensions(); ++d) {
256  int nBins = hist->GetAxis(d)->GetNbins() + 2;
257  nBinsTotal *= nBins;
258 
259  // number of bits needed to store compact coordinates
260  int b = 1;
261  while (nBins /= 2) {
262  ++b;
263  }
264  compCoordSize += b;
265  }
266  compCoordSize = (compCoordSize + 7) / 8; // turn bits into bytes
267 
268  // THnSparse stores the data in an array of chunks (THnSparseArrayChunk), each containing a fixed number of bins (e.g. 1024 * 16)
269  double nBinsFilled = fillFraction * nBinsTotal;
270  int nCunks = ceil(nBinsFilled / hist->GetChunkSize());
271  int chunkOverhead = sizeof(THnSparseArrayChunk);
272 
273  // each chunk holds array of compact bin-coordinates and an array of bin content (+ one of bin error if requested)
274  double binSize = compCoordSize + HistFiller::getBaseElementSize(hist.get()) + ((hist->GetSumw2() != -1.) ? sizeof(double) : 0.);
275  size = nCunks * (chunkOverhead + hist->GetChunkSize() * binSize);
276  // since THnSparse must keep track of all the stored bins, it stores a map that
277  // relates the compact bin coordinates (or a hash thereof) to a linear index
278  // this index determines in which chunk and therein at which position to find / store bin coordinate and content
279  size += nBinsFilled * 3 * sizeof(Long64_t); // hash, key, value; not sure why 3 are needed here...
280  }
281  return size / 1048576.;
282 }

303 template <typename B, typename T>
304 int HistFiller::getBaseElementSize(T* ptr)
305 {
306  if constexpr (std::is_base_of_v<THn, T>) {
307  if (dynamic_cast<THnT<B>*>(ptr)) {
308  return sizeof(B);
309  }
310  } else if constexpr (std::is_base_of_v<THnSparse, T>) {
311  if (dynamic_cast<THnSparseT<B>*>(ptr)) {
312  TDataMember* dm = B::Class()->GetDataMember("fArray");
313  return dm ? dm->GetDataType()->Size() : 0;
314  }
315  } else if constexpr (std::is_base_of_v<TH1, T>) {
316  if (auto arrayPtr = dynamic_cast<B*>(ptr)) {
317  return sizeof(arrayPtr->At(0));
318  }
319  }
320  return 0;
321 }

375 template <typename T>
376 HistPtr HistogramRegistry::insertClone(const HistName& histName, const std::shared_ptr<T> originalHist)
377 {
378  validateHistName(histName.str, histName.hash);
379  for (auto i = 0u; i < MAX_REGISTRY_SIZE; ++i) {
380  TObject* rawPtr = nullptr;
381  std::visit([&](const auto& sharedPtr) { rawPtr = sharedPtr.get(); }, mRegistryValue[imask(histName.idx + i)]);
382  if (!rawPtr) {
383  registerName(histName.str);
384  mRegistryKey[imask(histName.idx + i)] = histName.hash;
385  mRegistryValue[imask(histName.idx + i)] = std::shared_ptr<T>(static_cast<T*>(originalHist->Clone(histName.str)));
386  lookup += i;
387  return mRegistryValue[imask(histName.idx + i)];
388  }
389  }
390  LOGF(fatal, R"(Internal array of HistogramRegistry "%s" is full.)", mName);
391  return HistPtr();
392 }

394 template <typename T>
395 uint32_t HistogramRegistry::getHistIndex(const T& histName)
396 {
397  if (O2_BUILTIN_LIKELY(histName.hash == mRegistryKey[histName.idx])) {
398  return histName.idx;
399  }
400  for (auto i = 1u; i < MAX_REGISTRY_SIZE; ++i) {
401  if (histName.hash == mRegistryKey[imask(histName.idx + i)]) {
402  return imask(histName.idx + i);
403  }
404  }
405  throw runtime_error_f(R"(Could not find histogram "%s" in HistogramRegistry "%s"!)", histName.str, mName.data());
406 }
