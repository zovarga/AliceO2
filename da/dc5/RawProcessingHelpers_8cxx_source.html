25 bool raw_processing_helpers::processZSdata(const char* data, size_t size, rdh_utils::FEEIDType feeId, uint32_t orbit, uint32_t referenceOrbit, uint32_t syncOffsetReference, ADCCallback fillADC, bool useTimeBin)
26 {
27  const auto& mapper = Mapper::instance();
28 
29  const auto link = rdh_utils::getLink(feeId);
30  const auto cruID = rdh_utils::getCRU(feeId);
31  const auto endPoint = rdh_utils::getEndPoint(feeId);
32  const CRU cru(cruID);
33  const int fecLinkOffsetCRU = (mapper.getPartitionInfo(cru.partition()).getNumberOfFECs() + 1) / 2;
34  const int fecInPartition = link + endPoint * fecLinkOffsetCRU;
35 
36  // temporarily store the sync offset until it is available in the ZS header
37  // WARNING: This only works until the TB counter wrapped afterwards the alignment might change
38  const int globalLinkID = int(link) + int(endPoint * 12);
39  const int tpcGlobalLinkID = cruID * 24 + globalLinkID;
40  static std::array<uint32_t, 360 * 24> syncOffsetLinks;
41 
42  const uint32_t maxBunches = (uint32_t)o2::constants::lhc::LHCMaxBunches;
43  int globalBCOffset = int(orbit - referenceOrbit) * o2::constants::lhc::LHCMaxBunches;
44  static int triggerBCOffset = 0;
45 
46  bool hasData{false};
47 
48  zerosupp_link_based::ContainerZS* zsdata = (zerosupp_link_based::ContainerZS*)data;
49  const zerosupp_link_based::ContainerZS* const zsdataEnd = (zerosupp_link_based::ContainerZS*)(data + size);
50 
51  while (zsdata < zsdataEnd) {
52  const auto& header = zsdata->cont.header;
53 
54  // align to header word if needed
55  if (!header.hasCorrectMagicWord()) {
56  zsdata = (zerosupp_link_based::ContainerZS*)((const char*)zsdata + sizeof(zerosupp_link_based::Header));
57  if (!header.isFillWord()) {
58  LOGP(error, "Bad LinkZS magic word (0x{:08x}), for feeId 0x{:05x} (CRU: {:3}, link: {:2}, EP {}) , skipping data block", header.magicWord, feeId, rdh_utils::getCRU(feeId), rdh_utils::getLink(feeId), rdh_utils::getEndPoint(feeId));
59  LOGP(error, "Full 128b word is: 0x{:016x}{:016x}", header.word1, header.word0);
60  }
61  continue;
62  }
63 
64  // set trigger offset and skip trigger info
65  if (header.isTriggerInfo()) {
66  // for the moment only skip the trigger info
67  const auto triggerInfo = (zerosupp_link_based::TriggerContainer*)zsdata;
68  const auto triggerOrbit = triggerInfo->triggerInfo.getOrbit();
69  const auto triggerBC = triggerInfo->triggerInfo.bunchCrossing;
70  triggerBCOffset = (int(triggerOrbit) - int(referenceOrbit)) * maxBunches + triggerBC;
71  LOGP(debug, "orbit: {}, triggerOrbit: {}, triggerBC: {}, triggerBCOffset: {}", orbit, triggerOrbit, triggerBC, triggerBCOffset);
72  zsdata = (zerosupp_link_based::ContainerZS*)((const char*)zsdata + sizeof(zerosupp_link_based::Header) * (1 + header.numWordsPayload));
73  continue;
74  }
75 
76  const auto channelBits = zsdata->getChannelBits();
77  const uint32_t expectedWords = std::ceil(channelBits.count() * 0.1f);
78  const uint32_t numberOfWords = zsdata->getDataWords();
79  assert(expectedWords == numberOfWords);
80 
81  const uint32_t bunchCrossingHeader = zsdata->getBunchCrossing() + syncOffsetReference;
82  uint32_t syncOffset = header.syncOffsetBC;
83 
84  if (useTimeBin) {
85  const uint32_t timebinHeader = (header.syncOffsetCRUCycles << 8) | header.syncOffsetBC;
86  if (syncOffsetLinks[tpcGlobalLinkID] == 0) {
87  syncOffsetLinks[tpcGlobalLinkID] = (bunchCrossingHeader + maxBunches - (timebinHeader * 8) % maxBunches) % maxBunches % 16;
88  }
89  syncOffset = syncOffsetLinks[tpcGlobalLinkID];
90  }
91 
92  const int bcOffset = (int(globalBCOffset) + int(bunchCrossingHeader) - int(syncOffset)) - triggerBCOffset;
93  const int timebin = bcOffset / constants::LHCBCPERTIMEBIN;
94  if (bcOffset < 0) {
95  LOGP(debug, "skipping time bin with negative BC offset (globalBCoffset (({} - {}) * {} = {}) + bunchCrossingHeader ({}) - syncOffset({}) - triggerBCOffset({})) = {}", orbit, referenceOrbit, o2::constants::lhc::LHCMaxBunches, globalBCOffset, bunchCrossingHeader, syncOffset, triggerBCOffset, bcOffset);
96 
97  // go to next time bin
98  zsdata = zsdata->next();
99  continue;
100  }
101 
102  std::size_t processedChannels = 0;
103  for (std::size_t ichannel = 0; ichannel < channelBits.size(); ++ichannel) {
104  if (!channelBits[ichannel]) {
105  continue;
106  }
107 
108  // adc value
109  const auto adcValue = zsdata->getADCValueFloat(processedChannels);
110 
111  // mapping to row, pad sector
112  int sampaOnFEC{}, channelOnSAMPA{};
113  Mapper::getSampaAndChannelOnFEC(cruID, ichannel, sampaOnFEC, channelOnSAMPA);
114  const auto padSecPos = mapper.padSecPos(cru, fecInPartition, sampaOnFEC, channelOnSAMPA);
115  const auto& padPos = padSecPos.getPadPos();
116 
117  // add digit using callback
118  fillADC(int(cruID), int(padPos.getRow()), int(padPos.getPad()), timebin, adcValue);
119 
120  ++processedChannels;
121  hasData = true;
122  }
123 
124  // go to next time bin
125  zsdata = zsdata->next();
126  }
127 
128  return hasData;
129 }
