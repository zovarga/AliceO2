47 void TrapSimulator::init(TrapConfig* trapconfig, int det, int robPos, int mcmPos)
48 {
49  //
50  // Initialize the class with new MCM position information
51  //
52  mDetector = det;
53  mRobPos = robPos;
54  mMcmPos = mcmPos;
55 
56  uint64_t row = mFeeParam->getPadRowFromMCM(mRobPos, mMcmPos); // need uint64_t type to assemble mTrkltWordEmpty below
57  uint64_t column = mMcmPos % NMCMROBINCOL;
58  // prepare a part of the MCM header, what still is missing are the 3 x 8 bits from the charges
59  // < 1 | padrow (4 bits) | column (2 bits) | 00..0 (8 bit) | 00..0 (8 bit) | 00..0 (8 bit) | 1 >
60  mMcmHeaderEmpty = (1 << 31) | (row << 27) | (column << 25) | 1;
61  // prepare the part of the Tracklet64 which is common to all tracklets of this MCM
62  uint64_t hcid = 2 * mDetector + (mRobPos % 2);
63  mTrkltWordEmpty = (1UL << Tracklet64::formatbs) | (hcid << Tracklet64::hcidbs) | (row << Tracklet64::padrowbs) | (column << Tracklet64::colbs);
64 
65  if (!mInitialized) {
66  mTrapConfig = trapconfig;
67  mNTimeBin = mTrapConfig->getTrapReg(TrapConfig::kC13CPUA, mDetector, mRobPos, mMcmPos);
68  mZSMap.resize(NADCMCM);
69  mADCR.resize(mNTimeBin * NADCMCM);
70  mADCF.resize(mNTimeBin * NADCMCM);
71  }
72 
73  mInitialized = true;
74  reset();
75 }

77 void TrapSimulator::reset()
78 {
79  // Resets the data values and internal filter registers
80  // by re-initialising them
81 
82  if (!checkInitialized()) {
83  return;
84  }
85 
86  //clear the adc data
87  std::fill(mADCR.begin(), mADCR.end(), 0);
88  std::fill(mADCF.begin(), mADCF.end(), 0);
89  std::fill(mADCDigitIndices.begin(), mADCDigitIndices.end(), -1);
90 
91  for (auto filterreg : mInternalFilterRegisters) {
92  filterreg.ClearReg();
93  }
94 
95  // Default unread, low active bit mask
96  std::fill(mZSMap.begin(), mZSMap.end(), 0);
97  std::fill(mMCMT.begin(), mMCMT.end(), 0);
98 
99  filterPedestalInit();
100  //filterGainInit(); // we do not use the gain filter anyway, so disable it completely
101  filterTailInit();
102 
103  for (auto& fitreg : mFitReg) {
104  fitreg.ClearReg();
105  }
106  mADCFilled = 0;
107 
108  mTrackletArray64.clear();
109  mTrackletDigitCount.clear();
110  mTrackletDigitIndices.clear();
111 
112  mDataIsSet = false;
113 }

426 void TrapSimulator::noiseTest(int nsamples, int mean, int sigma, int inputGain, int inputTail)
427 {
428  // This function can be used to test the filters.
429  // It feeds nsamples of ADC values with a gaussian distribution specified by mean and sigma.
430  // The filter chain implemented here consists of:
431  // Pedestal -> Gain -> Tail
432  // With inputGain and inputTail the input to the gain and tail filter, respectively,
433  // can be chosen where
434  // 0: noise input
435  // 1: pedestal output
436  // 2: gain output
437  // The input has to be chosen from a stage before.
438  // The filter behaviour is controlled by the TRAP parameters from TrapConfig in the
439  // same way as in normal simulation.
440  // The functions produces four histograms with the values at the different stages.
441 
442  if (!checkInitialized()) {
443  return;
444  }
445 
446  std::string nameInputGain;
447  std::string nameInputTail;
448 
449  switch (inputGain) {
450  case 0:
451  nameInputGain = "Noise";
452  break;
453 
454  case 1:
455  nameInputGain = "Pedestal";
456  break;
457 
458  default:
459  LOG(error) << "Undefined input to tail cancellation filter";
460  return;
461  }
462 
463  switch (inputTail) {
464  case 0:
465  nameInputTail = "Noise";
466  break;
467 
468  case 1:
469  nameInputTail = "Pedestal";
470  break;
471 
472  case 2:
473  nameInputTail = "Gain";
474  break;
475 
476  default:
477  LOG(error) << "Undefined input to tail cancellation filter";
478  return;
479  }
480 
481  TH1F* h = new TH1F("noise", "Gaussian Noise;sample;ADC count",
482  nsamples, 0, nsamples);
483  TH1F* hfp = new TH1F("ped", "Noise #rightarrow Pedestal filter;sample;ADC count", nsamples, 0, nsamples);
484  TH1F* hfg = new TH1F("gain",
485  (nameInputGain + "#rightarrow Gain;sample;ADC count").c_str(),
486  nsamples, 0, nsamples);
487  TH1F* hft = new TH1F("tail",
488  (nameInputTail + "#rightarrow Tail;sample;ADC count").c_str(),
489  nsamples, 0, nsamples);
490  h->SetStats(false);
491  hfp->SetStats(false);
492  hfg->SetStats(false);
493  hft->SetStats(false);
494 
495  for (int i = 0; i < nsamples; i++) {
496  int value; // ADC count with noise (10 bit)
497  int valuep; // pedestal filter output (12 bit)
498  int valueg; // gain filter output (12 bit)
499  int valuet; // tail filter value (12 bit)
500  value = (int)gRandom->Gaus(mean, sigma); // generate noise with gaussian distribution
501  h->SetBinContent(i, value);
502 
503  valuep = filterPedestalNextSample(1, 0, ((int)value) << 2);
504 
505  if (inputGain == 0) {
506  valueg = filterGainNextSample(1, ((int)value) << 2);
507  } else {
508  valueg = filterGainNextSample(1, valuep);
509  }
510 
511  if (inputTail == 0) {
512  valuet = filterTailNextSample(1, ((int)value) << 2);
513  } else if (inputTail == 1) {
514  valuet = filterTailNextSample(1, valuep);
515  } else {
516  valuet = filterTailNextSample(1, valueg);
517  }
518 
519  hfp->SetBinContent(i, valuep >> 2);
520  hfg->SetBinContent(i, valueg >> 2);
521  hft->SetBinContent(i, valuet >> 2);
522  }
523 
524  TCanvas* c = new TCanvas;
525  c->Divide(2, 2);
526  c->cd(1);
527  h->Draw();
528  c->cd(2);
529  hfp->Draw();
530  c->cd(3);
531  hfg->Draw();
532  c->cd(4);
533  hft->Draw();
534 }

570 {
571  // Plots the data stored in a 2-dim. timebin vs. ADC channel plot.
572  // The choice selects what data is plotted and is enumated in the header.
573  // PLOTRAW - plot raw data (default)
574  // PLOTFILTERED - plot filtered data (meaningless if R is specified)
575  // In addition to the ADC values:
576  // PLOTHITS - plot hits
577  // PLOTTRACKLETS - plot tracklets
578  if (!checkInitialized()) {
579  return;
580  }
581  TFile* rootfile = new TFile("trdtrackletplots.root", "UPDATE");

858  // The output depends on the internal registers and, thus, the
859  // history of the filter.
860  LOG(debug) << "BEGIN: " << __FILE__ << ":" << __func__ << ":" << __LINE__;
861 
862  unsigned short fpnp = mTrapConfig->getTrapReg(TrapConfig::kFPNP, mDetector, mRobPos, mMcmPos); // 0..511 -> 0..127.75, pedestal at the output

867  unsigned short inpAdd;

872  if (timebin == 0) // the accumulator is disabled in the drift time

875  mInternalFilterRegisters[adc].mPedAcc = (mInternalFilterRegisters[adc].mPedAcc + correction) & 0x7FFFFFFF; // 31 bits

879  // LOG(info) << "Pedestal filter bypassed";

884 
885  if (inpAdd <= accumulatorShifted) {
886  return 0;
887  } else {
888  inpAdd = inpAdd - accumulatorShifted;

891  } else {

893  }

995 void TrapSimulator::filterTailInit(int baseline)
996 {
997  // Initializes the tail filter assuming that the input has
998  // been at the baseline value (configured by FTFP) for a
999  // sufficiently long time.
1000 
1001  // exponents and weight calculated from configuration
1002  unsigned short alphaLong = 0x3ff & mTrapConfig->getTrapReg(TrapConfig::kFTAL, mDetector, mRobPos, mMcmPos); // the weight of the long component
1003  unsigned short lambdaLong = (1 << 10) | (1 << 9) | (mTrapConfig->getTrapReg(TrapConfig::kFTLL, mDetector, mRobPos, mMcmPos) & 0x1FF); // the multiplier
1004  unsigned short lambdaShort = (0 << 10) | (1 << 9) | (mTrapConfig->getTrapReg(TrapConfig::kFTLS, mDetector, mRobPos, mMcmPos) & 0x1FF); // the multiplier
1005 
1006  float lambdaL = lambdaLong * 1.0 / (1 << 11);
1007  float lambdaS = lambdaShort * 1.0 / (1 << 11);
1008  float alphaL = alphaLong * 1.0 / (1 << 11);
1009  float qup, qdn;
1010  qup = (1 - lambdaL) * (1 - lambdaS);
1011  qdn = 1 - lambdaS * alphaL - lambdaL * (1 - alphaL);
1012  float kdc = qup / qdn;
1013 
1014  float ql, qs;
1015 
1016  if (baseline < 0) {
1017  baseline = mTrapConfig->getTrapReg(TrapConfig::kFPNP, mDetector, mRobPos, mMcmPos);
1018  }
1019 
1020  ql = lambdaL * (1 - lambdaS) * alphaL;
1021  qs = lambdaS * (1 - lambdaL) * (1 - alphaL);
1022 
1023  for (int adc = 0; adc < NADCMCM; adc++) {
1024  int value = baseline & 0xFFF;
1025  int corr = (value * mTrapConfig->getTrapReg(TrapConfig::TrapReg_t(TrapConfig::kFGF0 + adc), mDetector, mRobPos, mMcmPos)) >> 11;
1026  corr = corr > 0xfff ? 0xfff : corr;
1027  corr = addUintClipping(corr, mTrapConfig->getTrapReg(TrapConfig::TrapReg_t(TrapConfig::kFGA0 + adc), mDetector, mRobPos, mMcmPos), 12);
1028 
1029  float kt = kdc * baseline;

1037 unsigned short TrapSimulator::filterTailNextSample(int adc, unsigned short value)
1038 {
1039  // Returns the output of the tail filter for the given input value.
1040  // The output depends on the internal registers and, thus, the
1041  // history of the filter.
1042 
1043  // exponents and weight calculated from configuration
1044  unsigned short alphaLong = 0x3ff & mTrapConfig->getTrapReg(TrapConfig::kFTAL, mDetector, mRobPos, mMcmPos); // the weight of the long component
1045  unsigned short lambdaLong = (1 << 10) | (1 << 9) | (mTrapConfig->getTrapReg(TrapConfig::kFTLL, mDetector, mRobPos, mMcmPos) & 0x1FF); // the multiplier of the long component
1046  unsigned short lambdaShort = (0 << 10) | (1 << 9) | (mTrapConfig->getTrapReg(TrapConfig::kFTLS, mDetector, mRobPos, mMcmPos) & 0x1FF); // the multiplier of the short component
1047 
1048  // intermediate signals
1049  unsigned int aDiff;
1050  unsigned int alInpv;
1051  unsigned short aQ;
1052  unsigned int tmp;
1053 
1054  unsigned short inpVolt = value & 0xFFF; // 12 bits
1055 
1056  // add the present generator outputs
1057  aQ = addUintClipping(mInternalFilterRegisters[adc].mTailAmplLong, mInternalFilterRegisters[adc].mTailAmplShort, 12);
1058 
1059  // calculate the difference between the input and the generated signal
1060  if (inpVolt > aQ) {
1061  aDiff = inpVolt - aQ;
1062  } else {
1063  aDiff = 0;
1064  }
1065 
1066  // the inputs to the two generators, weighted
1067  alInpv = (aDiff * alphaLong) >> 11;
1068 
1069  // the new values of the registers, used next time
1070  // long component
1071  tmp = addUintClipping(mInternalFilterRegisters[adc].mTailAmplLong, alInpv, 12);
1072  tmp = (tmp * lambdaLong) >> 11;
1073  mInternalFilterRegisters[adc].mTailAmplLong = tmp & 0xFFF;
1074  // short component
1075  tmp = addUintClipping(mInternalFilterRegisters[adc].mTailAmplShort, aDiff - alInpv, 12);
1076  tmp = (tmp * lambdaShort) >> 11;
1077  mInternalFilterRegisters[adc].mTailAmplShort = tmp & 0xFFF;
1078 
1079  // the output of the filter
1080  if (mTrapConfig->getTrapReg(TrapConfig::kFTBY, mDetector, mRobPos, mMcmPos) == 0) { // bypass mode, active low
1081  return value;
1082  } else {
1083  return aDiff;
1084  }
