51  {"disable-mc", o2::framework::VariantType::Bool, false, {"disable sending of MC information, TBI"}},

54  {"use-ccdb", o2::framework::VariantType::Bool, false, {"enable access to ccdb tof calibration objects"}},
55  {"input-desc", o2::framework::VariantType::String, "CRAWDATA", {"Input specs description string"}},
56  {"disable-root-input", o2::framework::VariantType::Bool, false, {"disable root-files input readers"}},
57  {"disable-root-output", o2::framework::VariantType::Bool, false, {"disable root-files output writers"}},
58  {"conet-mode", o2::framework::VariantType::Bool, false, {"enable conet mode"}},
59  {"configKeyValues", o2::framework::VariantType::String, "", {"Semicolon separated key=value strings ..."}},
60  {"disable-row-writing", o2::framework::VariantType::Bool, false, {"disable ROW in Digit writing"}},
61  {"write-decoding-errors", o2::framework::VariantType::Bool, false, {"trace errors in digits output when decoding"}},
62  {"ignore-dist-stf", VariantType::Bool, false, {"do not subscribe to FLP/DISTSUBTIMEFRAME/0 message (no lost TF recovery)"}},
63  {"calib-cluster", VariantType::Bool, false, {"to enable calib info production from clusters"}},
64  {"cosmics", VariantType::Bool, false, {"to enable cosmics utils"}}};
65  o2::raw::HBFUtilsInitializer::addConfigOption(options);
66  std::swap(workflowOptions, options);

87 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
88 {
89  WorkflowSpec specs;
90 
91  // the lane configuration defines the subspecification ids to be distributed among the lanes.
92  auto nLanes = cfgc.options().get<int>("tof-lanes");
93  auto inputType = cfgc.options().get<std::string>("input-type");
94  auto outputType = cfgc.options().get<std::string>("output-type");
95 
96  bool writecluster = 0;
97  bool writedigit = 0;
98  bool writeraw = 0;
99  bool writectf = 0;
100  bool writeerr = 0;
101 
102  if (outputType.rfind("clusters") < outputType.size()) {
103  writecluster = 1;
104  }
105  if (outputType.rfind("digits") < outputType.size()) {
106  writedigit = 1;
107  }
108  if (outputType.rfind("raw") < outputType.size()) {
109  writeraw = 1;
110  }
111  if (outputType.rfind("ctf") < outputType.size()) {
112  writectf = 1;
113  }
114 
115  bool dgtinput = 0;
116  bool clusterinput = 0;
117  bool rawinput = 0;
118  if (inputType == "digits") {
119  dgtinput = 1;
120  } else if (inputType == "clusters") {
121  clusterinput = 1;
122  } else if (inputType == "raw") {
123  rawinput = 1;
124  writeerr = cfgc.options().get<bool>("write-decoding-errors");
125  }
126 
127  if (!cfgc.helpOnCommandLine()) {
128  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
129  }
130 
131  auto useMC = !cfgc.options().get<bool>("disable-mc");
132  auto useCCDB = cfgc.options().get<bool>("use-ccdb");
133  bool disableRootInput = cfgc.options().get<bool>("disable-root-input") || rawinput;
134  bool disableRootOutput = cfgc.options().get<bool>("disable-root-output");
135  bool conetmode = cfgc.options().get<bool>("conet-mode");
136  bool disableROWwriting = cfgc.options().get<bool>("disable-row-writing");
137  auto isCalibFromCluster = cfgc.options().get<bool>("calib-cluster");
138  auto isCosmics = cfgc.options().get<bool>("cosmics");
139  auto ignoreDistStf = cfgc.options().get<bool>("ignore-dist-stf");
140  auto ccdb_url = o2::base::NameConf::getCCDBServer();
141 
142  LOG(debug) << "TOF RECO WORKFLOW configuration";
143  LOG(debug) << "TOF input = " << cfgc.options().get<std::string>("input-type");
144  LOG(debug) << "TOF output = " << cfgc.options().get<std::string>("output-type");
145  LOG(debug) << "TOF sectors = " << cfgc.options().get<std::string>("tof-sectors");
146  LOG(debug) << "TOF disable-mc = " << cfgc.options().get<std::string>("disable-mc");
147  LOG(debug) << "TOF lanes = " << cfgc.options().get<std::string>("tof-lanes");
148  LOG(debug) << "TOF use-ccdb = " << cfgc.options().get<std::string>("use-ccdb");
149  if (useCCDB) {
150  LOG(debug) << "CCDB url = " << ccdb_url;
151  }
152  LOG(debug) << "TOF disable-root-input = " << disableRootInput;
153  LOG(debug) << "TOF disable-root-output = " << disableRootOutput;
154  LOG(debug) << "TOF conet-mode = " << conetmode;
155  LOG(debug) << "TOF ignore Dist Stf = " << ignoreDistStf;
156  LOG(debug) << "TOF disable-row-writing = " << disableROWwriting;
157  LOG(debug) << "TOF write-decoding-errors = " << writeerr;
158 
159  if (clusterinput && !disableRootInput) {
160  LOG(debug) << "Insert TOF Cluster Reader";
161  specs.emplace_back(o2::tof::getClusterReaderSpec(useMC));
162  } else if (dgtinput) {
163  // TOF clusterizer
164  if (!disableRootInput) {
165  LOG(debug) << "Insert TOF Digit reader from file";
166  specs.emplace_back(o2::tof::getDigitReaderSpec(useMC));
167  }
168  if (writeraw) {
169  LOG(debug) << "Insert TOF Raw writer";
170  specs.emplace_back(o2::tof::getTOFRawWriterSpec());
171  }
172  } else if (rawinput) {
173  LOG(debug) << "Insert TOF Compressed Raw Decoder";
174  auto inputDesc = cfgc.options().get<std::string>("input-desc");
175  specs.emplace_back(o2::tof::getCompressedDecodingSpec(inputDesc, conetmode, !ignoreDistStf));
176  useMC = 0;
177 
178  if (writedigit && !disableRootOutput) {
179  // add TOF digit writer without mc labels
180  LOG(debug) << "Insert TOF Digit Writer";
181  specs.emplace_back(o2::tof::getTOFDigitWriterSpec(0, writeerr));
182  }
183  }
184 
185  if (!clusterinput && writecluster) {
186  LOG(debug) << "Insert TOF Clusterizer";
187  specs.emplace_back(o2::tof::getTOFClusterizerSpec(useMC, useCCDB, isCalibFromCluster, isCosmics, ccdb_url.c_str()));
188  if (writecluster && !disableRootOutput) {
189  LOG(debug) << "Insert TOF Cluster Writer";
190  specs.emplace_back(o2::tof::getTOFClusterWriterSpec(useMC));
191  }
192  }
193 
194  if (writectf) {
195  LOG(debug) << "Insert TOF CTF encoder";
196  specs.emplace_back(o2::tof::getEntropyEncoderSpec());
197  }
198 
199  LOG(debug) << "Number of active devices = " << specs.size();
200 
201  // configure dpl timer to inject correct firstTFOrbit: start from the 1st orbit of TF containing 1st sampled orbit
202  o2::raw::HBFUtilsInitializer hbfIni(cfgc, specs);
203 
204  return std::move(specs);
205 }
