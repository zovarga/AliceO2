110 uint32_t getQFromRaw(const o2::trd::TrackletMCMHeader* header, const o2::trd::TrackletMCMData* data, int pidindex, int trackletindex)
111 {
112  uint32_t pid = 0;
113  uint32_t qa, qb;
114  //PID VERSION 1
115  //frist part of pid is in the TrackletMCMHeader
116  switch (trackletindex) {
117  case 0:
118  qa = header->pid0;
119  break;
120  case 1:
121  qa = header->pid1;
122  break;
123  case 2:
124  qa = header->pid2;
125  break;
126  default:
127  LOG(warn) << " unknown trackletindex of " << trackletindex << " to getQFromRaw : " << pidindex;
128  break;
129  }
130  //qa is 6bits of Q2 and 2 bits of Q1
131  //second part of pid is in the TrackletMCMData
132  qb = data->pid;
133  //qb is 7 bits Q0 and 5 bits of Q1
134  switch (pidindex) {
135  case 0: //Q0
136  pid = (qb >> 5) & 0x7f; // 7 bits at the top of all of Q0
137  break;
138  case 1: //Q1
139  pid = ((qa & 0x3) << 5) | (qb >> 5); // 2 bits of qb and 5 bits of qb for Q1 .. 7 bits
140  break;
141  case 2: //Q2
142  pid = (qa >> 2) & 0x2f; // 6 bits shifted down by bits 2 to 8 ... 6 bits
143  break;
144  default:
145  LOG(warn) << " unknown pid index of : " << pidindex;
146  break;
147  }
148  //PID VERSION 2
149  /*
150  switch(pidindex) {
151  case 0 : pid=qa&0xffc>>2;break;
152  case 1 : pid=((qa&0x3)<<5)|(qb>>6);break;
153  case 2 : pid=qb&0x3f;break;
154  default : LOG(warn) << " unknown pid index of : " << pidindex;
155  }
156  */
157  return pid;
158 }
