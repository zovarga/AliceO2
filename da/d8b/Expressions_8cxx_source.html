30 size_t Filter::designateSubtrees(Node* node, size_t index)
31 {
32  std::stack<NodeRecord> path;
33  auto local_index = index;
34  path.emplace(node, 0);
35 
36  while (path.empty() == false) {
37  auto& top = path.top();
38  top.node_ptr->index = local_index;
39  path.pop();
40  if (top.node_ptr->condition != nullptr) {
41  // start new subtrees
42  index = designateSubtrees(top.node_ptr->left.get(), local_index + 1);
43  index = designateSubtrees(top.node_ptr->condition.get(), index + 1);
44  index = designateSubtrees(top.node_ptr->right.get(), index + 1);
45  } else {
46  // continue current subtree
47  if (top.node_ptr->left != nullptr) {
48  path.emplace(top.node_ptr->left.get(), 0);
49  }
50  if (top.node_ptr->right != nullptr) {
51  path.emplace(top.node_ptr->right.get(), 0);
52  }
53  }
54  }
55 
56  return index;
57 }

157 void updatePlaceholders(Filter& filter, InitContext& context)
158 {
159  std::stack<NodeRecord> path;
160 
161  // insert the top node into stack
162  path.emplace(filter.node.get(), 0);
163 
164  auto updateNode = [&](Node* node) {
165  if (node->self.index() == 3) {
166  std::get_if<3>(&node->self)->reset(context);
167  }
168  };
169 
170  // while the stack is not empty
171  while (path.empty() == false) {
172  auto& top = path.top();
173  updateNode(top.node_ptr);
174 
175  auto* leftp = top.node_ptr->left.get();
176  auto* rightp = top.node_ptr->right.get();
177  auto* condp = top.node_ptr->condition.get();
178  path.pop();
179 
180  if (leftp != nullptr) {
181  path.emplace(leftp, 0);
182  }
183  if (rightp != nullptr) {
184  path.emplace(rightp, 0);
185  }
186  if (condp != nullptr) {
187  path.emplace(condp, 0);
188  }
189  }
190 }

192 Operations createOperations(Filter const& expression)
193 {
194  Operations OperationSpecs;
195  std::stack<NodeRecord> path;
196  auto isLeaf = [](Node const* const node) {
197  return ((node->left == nullptr) && (node->right == nullptr));
198  };
199 
200  auto processLeaf = [](Node const* const node) {
201  return std::visit(
202  overloaded{
203  [lh = LiteralNodeHelper{}](LiteralNode const& node) { return lh(node); },
204  [bh = BindingNodeHelper{}](BindingNode const& node) { return bh(node); },
205  [ph = PlaceholderNodeHelper{}](PlaceholderNode const& node) { return ph(node); },
206  [](auto&&) { return DatumSpec{}; }},
207  node->self);
208  };
209 
210  size_t index = 0;
211  // insert the top node into stack
212  path.emplace(expression.node.get(), index++);
213 
214  // while the stack is not empty
215  while (path.empty() == false) {
216  auto& top = path.top();
217 
218  // create operation spec, pop the node and add its children
219  auto operationSpec =
220  std::visit(
221  overloaded{
222  [&](OpNode node) { return ColumnOperationSpec{node.op, top.node_ptr->index}; },
223  [&](ConditionalNode) { return ColumnOperationSpec{BasicOp::Conditional, top.node_ptr->index}; },
224  [](auto&&) { return ColumnOperationSpec{}; }},
225  top.node_ptr->self);
226 
227  operationSpec.result = DatumSpec{top.index, operationSpec.type};
228  path.pop();
229 
230  auto* left = top.node_ptr->left.get();
231  bool leftLeaf = isLeaf(left);
232  size_t li = 0;
233  if (leftLeaf) {
234  operationSpec.left = processLeaf(left);
235  } else {
236  li = index;
237  operationSpec.left = DatumSpec{index++, atype::NA};
238  }
239 
240  decltype(left) right = nullptr;
241  if (top.node_ptr->right != nullptr) {
242  right = top.node_ptr->right.get();
243  }
244  bool rightLeaf = true;
245  if (right != nullptr) {
246  rightLeaf = isLeaf(right);
247  }
248  size_t ri = 0;
249  auto isUnary = false;
250  if (top.node_ptr->right == nullptr) {
251  operationSpec.right = DatumSpec{};
252  isUnary = true;
253  } else {
254  if (rightLeaf) {
255  operationSpec.right = processLeaf(right);
256  } else {
257  ri = index;
258  operationSpec.right = DatumSpec{index++, atype::NA};
259  }
260  }
261 
262  decltype(left) condition = nullptr;
263  if (top.node_ptr->condition != nullptr) {
264  condition = top.node_ptr->condition.get();
265  }
266  bool condleaf = condition != nullptr ? isLeaf(condition) : true;
267  size_t ci = 0;
268  if (condition != nullptr) {
269  if (condleaf) {
270  operationSpec.condition = processLeaf(condition);
271  } else {
272  ci = index;
273  operationSpec.condition = DatumSpec{index++, atype::BOOL};
274  }
275  } else {
276  operationSpec.condition = DatumSpec{};
277  }
278 
279  OperationSpecs.push_back(std::move(operationSpec));
280  if (!leftLeaf) {
281  path.emplace(left, li);
282  }
283  if (!isUnary && !rightLeaf) {
284  path.emplace(right, ri);
285  }
286  if (!condleaf) {
287  path.emplace(condition, ci);
288  }
289  }
290  // at this stage the operations vector is created, but the field types are
291  // only set for the logical operations and leaf nodes
292  std::vector<atype::type> resultTypes;
293  resultTypes.resize(OperationSpecs.size());
294 
295  auto inferResultType = [&resultTypes](DatumSpec& left, DatumSpec& right) {
296  // if the left datum is monostate (error)
297  if (left.datum.index() == 0) {
298  throw runtime_error("Malformed operation spec: empty left datum");
299  }
300 
301  // check if the datums are references
302  if (left.datum.index() == 1) {
303  left.type = resultTypes[std::get<size_t>(left.datum)];
304  }
305 
306  if (right.datum.index() == 1) {
307  right.type = resultTypes[std::get<size_t>(right.datum)];
308  }
309 
310  auto t1 = left.type;
311  auto t2 = right.type;
312  // if the right datum is monostate (unary op)
313  if (right.datum.index() == 0) {
314  if (t1 == atype::DOUBLE) {
315  return atype::DOUBLE;
316  }
317  return atype::FLOAT;
318  }
319 
320  if (t1 == t2) {
321  return t1;
322  }
323 
324  auto isIntType = [](auto t) {
325  return (t == atype::UINT8) || (t == atype::INT8) || (t == atype::UINT16) || (t == atype::INT16) || (t == atype::UINT32) || (t == atype::INT32) || (t == atype::UINT64) || (t == atype::INT64);
326  };
327 
328  if (isIntType(t1)) {
329  if (t2 == atype::FLOAT) {
330  return atype::FLOAT;
331  }
332  if (t2 == atype::DOUBLE) {
333  return atype::DOUBLE;
334  }
335  if (isIntType(t2)) {
336  if (t1 > t2) {
337  return t1;
338  }
339  return t2;
340  }
341  }
342  if (t1 == atype::FLOAT) {
343  if (isIntType(t2)) {
344  return atype::FLOAT;
345  }
346  if (t2 == atype::DOUBLE) {
347  return atype::DOUBLE;
348  }
349  }
350  if (t1 == atype::DOUBLE) {
351  return atype::DOUBLE;
352  }
353  throw runtime_error_f("Invalid combination of argument types %d and %d", t1, t2);
354  };
355 
356  for (auto it = OperationSpecs.rbegin(); it != OperationSpecs.rend(); ++it) {
357  auto type = inferResultType(it->left, it->right);
358  if (it->type == atype::NA) {
359  it->type = type;
360  }
361 
362  it->result.type = it->type;
363  resultTypes[std::get<size_t>(it->result.datum)] = it->type;
364  }
365 
366  return OperationSpecs;
367 }

424 gandiva::Selection createSelection(std::shared_ptr<arrow::Table> const& table, std::shared_ptr<gandiva::Filter> const& gfilter)
425 {
426  gandiva::Selection selection;
427  auto s = gandiva::SelectionVector::MakeInt64(table->num_rows(),
428  arrow::default_memory_pool(),
429  &selection);
430  if (!s.ok()) {
431  throw runtime_error_f("Cannot allocate selection vector %s", s.ToString().c_str());
432  }
433  if (table->num_rows() == 0) {
434  return selection;
435  }
436  arrow::TableBatchReader reader(*table);
437  std::shared_ptr<arrow::RecordBatch> batch;
438  while (true) {
439  s = reader.ReadNext(&batch);
440  if (!s.ok()) {
441  throw runtime_error_f("Cannot read batches from table %s", s.ToString().c_str());
442  }
443  if (batch == nullptr) {
444  break;
445  }
446  s = gfilter->Evaluate(*batch, selection);
447  if (!s.ok()) {
448  throw runtime_error_f("Cannot apply filter %s", s.ToString().c_str());
449  }
450  }
451 
452  return selection;
453 }

482 gandiva::NodePtr createExpressionTree(Operations const& opSpecs,
483  gandiva::SchemaPtr const& Schema)
484 {
485  std::vector<gandiva::NodePtr> opNodes;
486  opNodes.resize(opSpecs.size());
487  std::fill(opNodes.begin(), opNodes.end(), nullptr);
488  std::unordered_map<std::string, gandiva::NodePtr> fieldNodes;
489  std::unordered_map<size_t, gandiva::NodePtr> subtrees;
490 
491  auto datumNode = [Schema, &opNodes, &fieldNodes](DatumSpec const& spec) {
492  if (spec.datum.index() == 0) {
493  return gandiva::NodePtr(nullptr);
494  }
495  if (spec.datum.index() == 1) {
496  return opNodes[std::get<size_t>(spec.datum)];
497  }
498 
499  if (spec.datum.index() == 2) {
500  auto content = std::get<LiteralNode::var_t>(spec.datum);
501  switch (content.index()) {
502  case 0: //int
503  return gandiva::TreeExprBuilder::MakeLiteral(static_cast<int32_t>(std::get<int>(content)));
504  case 1: //bool
505  return gandiva::TreeExprBuilder::MakeLiteral(std::get<bool>(content));
506  case 2: //float
507  return gandiva::TreeExprBuilder::MakeLiteral(std::get<float>(content));
508  case 3: //double
509  return gandiva::TreeExprBuilder::MakeLiteral(std::get<double>(content));
510  case 4: //uint8
511  return gandiva::TreeExprBuilder::MakeLiteral(std::get<uint8_t>(content));
512  case 5: //int64
513  return gandiva::TreeExprBuilder::MakeLiteral(std::get<int64_t>(content));
514  case 6: //int16
515  return gandiva::TreeExprBuilder::MakeLiteral(std::get<int16_t>(content));
516  case 7: //uint16
517  return gandiva::TreeExprBuilder::MakeLiteral(std::get<uint16_t>(content));
518  case 8: //int8
519  return gandiva::TreeExprBuilder::MakeLiteral(std::get<int8_t>(content));
520  case 9: //uint32
521  return gandiva::TreeExprBuilder::MakeLiteral(std::get<uint32_t>(content));
522  case 10: //uint64
523  return gandiva::TreeExprBuilder::MakeLiteral(std::get<uint64_t>(content));
524  default:
525  throw runtime_error("Malformed LiteralNode");
526  }
527  }
528 
529  if (spec.datum.index() == 3) {
530  auto name = std::get<std::string>(spec.datum);
531  auto lookup = fieldNodes.find(name);
532  if (lookup != fieldNodes.end()) {
533  return lookup->second;
534  }
535  auto field = Schema->GetFieldByName(name);
536  if (field == nullptr) {
537  throw runtime_error_f("Cannot find field \"%s\"", name.c_str());
538  }
539  auto node = gandiva::TreeExprBuilder::MakeField(field);
540  fieldNodes.insert({name, node});
541  return node;
542  }
543  throw runtime_error("Malformed DatumSpec");
544  };
545 
546  gandiva::NodePtr tree = nullptr;
547  for (auto it = opSpecs.rbegin(); it != opSpecs.rend(); ++it) {
548  auto leftNode = datumNode(it->left);
549  auto rightNode = datumNode(it->right);
550  auto condNode = datumNode(it->condition);
551 
552  auto insertUpcastNode = [&](gandiva::NodePtr node, atype::type t) {
553  if (t != it->type) {
554  auto upcast = gandiva::TreeExprBuilder::MakeFunction(upcastTo(it->type), {node}, concreteArrowType(it->type));
555  node = upcast;
556  }
557  return node;
558  };
559 
560  auto insertEqualizeUpcastNode = [&](gandiva::NodePtr& node1, gandiva::NodePtr& node2, atype::type t1, atype::type t2) {
561  if (t2 > t1) {
562  auto upcast = gandiva::TreeExprBuilder::MakeFunction(upcastTo(t2), {node1}, concreteArrowType(t2));
563  node1 = upcast;
564  } else if (t1 > t2) {
565  auto upcast = gandiva::TreeExprBuilder::MakeFunction(upcastTo(t1), {node2}, concreteArrowType(t1));
566  node2 = upcast;
567  }
568  };
569 
570  gandiva::NodePtr temp_node;
571 
572  switch (it->op) {
573  case BasicOp::LogicalOr:
574  temp_node = gandiva::TreeExprBuilder::MakeOr({leftNode, rightNode});
575  break;
576  case BasicOp::LogicalAnd:
577  temp_node = gandiva::TreeExprBuilder::MakeAnd({leftNode, rightNode});
578  break;
579  case BasicOp::Conditional:
580  temp_node = gandiva::TreeExprBuilder::MakeIf(condNode, leftNode, rightNode, concreteArrowType(it->type));
581  break;
582  default:
583  if (it->op < BasicOp::Sqrt) {
584  if (it->type != atype::BOOL) {
585  leftNode = insertUpcastNode(leftNode, it->left.type);
586  rightNode = insertUpcastNode(rightNode, it->right.type);
587  } else if (it->op == BasicOp::Equal || it->op == BasicOp::NotEqual) {
588  insertEqualizeUpcastNode(leftNode, rightNode, it->left.type, it->right.type);
589  }
590  temp_node = gandiva::TreeExprBuilder::MakeFunction(basicOperationsMap[it->op], {leftNode, rightNode}, concreteArrowType(it->type));
591  } else {
592  leftNode = insertUpcastNode(leftNode, it->left.type);
593  temp_node = gandiva::TreeExprBuilder::MakeFunction(basicOperationsMap[it->op], {leftNode}, concreteArrowType(it->type));
594  }
595  break;
596  }
597  if (it->index == 0) {
598  tree = temp_node;
599  } else {
600  auto subtree = subtrees.find(it->index);
601  if (subtree == subtrees.end()) {
602  subtrees.insert({it->index, temp_node});
603  } else {
604  subtree->second = temp_node;
605  }
606  }
607  opNodes[std::get<size_t>(it->result.datum)] = temp_node;
608  }
609 
610  return tree;
611 }

613 bool isTableCompatible(std::set<size_t> const& hashes, Operations const& specs)
614 {
615  std::set<size_t> opHashes;
616  for (auto& spec : specs) {
617  if (spec.left.datum.index() == 3) {
618  opHashes.insert(spec.left.hash);
619  }
620  if (spec.right.datum.index() == 3) {
621  opHashes.insert(spec.right.hash);
622  }
623  }
624 
625  return std::includes(hashes.begin(), hashes.end(),
626  opHashes.begin(), opHashes.end());
627 }

629 bool isSchemaCompatible(gandiva::SchemaPtr const& Schema, Operations const& opSpecs)
630 {
631  std::set<std::string> opFieldNames;
632  for (auto& spec : opSpecs) {
633  if (spec.left.datum.index() == 3) {
634  opFieldNames.insert(std::get<std::string>(spec.left.datum));
635  }
636  if (spec.right.datum.index() == 3) {
637  opFieldNames.insert(std::get<std::string>(spec.right.datum));
638  }
639  }
640 
641  std::set<std::string> schemaFieldNames;
642  for (auto& field : Schema->fields()) {
643  schemaFieldNames.insert(field->name());
644  }
645 
646  return std::includes(schemaFieldNames.begin(), schemaFieldNames.end(),
647  opFieldNames.begin(), opFieldNames.end());
648 }

650 void updateExpressionInfos(expressions::Filter const& filter, std::vector<ExpressionInfo>& eInfos)
651 {
652  if (eInfos.empty()) {
653  throw runtime_error("Empty expression info vector.");
654  }
655  Operations ops = createOperations(filter);
656  for (auto& info : eInfos) {
657  if (isTableCompatible(info.hashes, ops)) {
658  auto tree = createExpressionTree(ops, info.schema);
660  if (info.tree != nullptr) {
661  info.tree = gandiva::TreeExprBuilder::MakeAnd({info.tree, tree});
662  } else {
663  info.tree = tree;
664  }
665  }
666  }
