40  o2::InteractionRecord ir(0, 0);
41  std::array<float, NTimeBinsPerBC> chanVals;
42 
43  // BCData and ChannelData
44  for (int irof = 0; irof < 1000; irof++) {
45  ir += 1 + gRandom->Integer(100); // randomly increaing BC
46 
47  uint32_t channPatt = 0, triggers = 0;
48  int8_t ich = -1;
49  int firstChEntry = chandata.size();
50  while ((ich += 1 + gRandom->Poisson(2.)) < NDigiChannels) {
51  channPatt |= 0x1 << ich;
52  for (int i = 0; i < NTimeBinsPerBC; i++) {
53  chanVals[i] = gRandom->Integer(0xffff);
54  }
55  if (gRandom->Rndm() > 0.4) {
56  triggers |= 0x1 << ich;
57  }
58  chandata.emplace_back(ich, chanVals);
59  }
60  auto& bcd = bcdata.emplace_back(firstChEntry, chandata.size() - firstChEntry, ir, channPatt, triggers, gRandom->Integer(0xff));
61  for (int im = 0; im < NModules; im++) {
62  bcd.moduleTriggers[im] = gRandom->Rndm() > 0.7 ? gRandom->Integer((0x1 << 10) - 1) : 0;
63  }
64  }
65 
66  // OrbitData
67  const auto &irFirst = bcdata.front().ir, irLast = bcdata.back().ir;
68  o2::InteractionRecord irPed(o2::constants::lhc::LHCMaxBunches - 1, irFirst.orbit);
69  int norbits = irLast.orbit - irFirst.orbit + 1;
70  pedsdata.resize(norbits);
71  for (int i = 0; i < norbits; i++) {
72  pedsdata[i].ir = irPed;
73  for (int ic = 0; ic < NChannels; ic++) {
74  pedsdata[i].data[ic] = gRandom->Integer(0xffff);
75  pedsdata[i].scaler[ic] = (i > 0 ? pedsdata[i].scaler[ic - 1] : 0) + gRandom->Integer(20);
76  }
77  irPed.orbit++;
78  }
79 
80  sw.Start();

82  {
83  CTFCoder coder;
84  coder.encode(vec, bcdata, chandata, pedsdata); // compress
85  }
