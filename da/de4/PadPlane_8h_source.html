94  GPUd() double getTiltOffset(int row, double rowOffset) const
95  {
96  if (row == 0 || row == mNrows - 1) {
97  return mTiltingTan * (rowOffset - 0.5 * mLengthOPad);
98  } else {
99  return mTiltingTan * (rowOffset - 0.5 * mLengthIPad);
100  }
101  };
102  GPUd() double getPadRowOffset(int row, double z) const
103  {
104  if ((row < 0) || (row >= mNrows)) {
105  return -1.0;
106  } else {
107  return mPadRow[row] + mPadRowSMOffset - z;
108  }
109  };
110  GPUd() double getPadRowOffsetROC(int row, double z) const
111  {
112  if ((row < 0) || (row >= mNrows)) {
113  return -1.0;
114  } else {
115  return mPadRow[row] - z;
116  }
117  };
118 
119  GPUd() double getPadColOffset(int col, double rphi) const
120  {
121  if ((col < 0) || (col >= mNcols)) {
122  return -1.0;
123  } else {
124  return rphi - mPadCol[col];
125  }
126  };
