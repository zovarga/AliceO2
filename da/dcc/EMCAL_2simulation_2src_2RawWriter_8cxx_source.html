24 void RawWriter::init()
25 {
26  mRawWriter = std::make_unique<o2::raw::RawFileWriter>(o2::header::gDataOriginEMC, false);
27  mRawWriter->setCarryOverCallBack(this);
28 
29  // initialize mappers
30  if (!mMappingHandler) {
31  mMappingHandler = std::make_unique<o2::emcal::MappingHandler>();
32  }
33 
34  for (auto iddl = 0; iddl < 40; iddl++) {
35  // For EMCAL set
36  // - FEE ID = DDL ID
37  // - C-RORC and link increasing with DDL ID
38  // @TODO replace with link assignment on production FLPs,
39  // eventually storing in CCDB
40 
41  // initialize containers for SRU
42  SRUDigitContainer srucont;
43  srucont.mSRUid = iddl;
44  mSRUdata.push_back(srucont);
45 
46  // Skip empty links with these ddl IDs,
47  // ddl ID 21 and 39 are empty links, while 23 and 36 are connected to LEDmon only
48  if (iddl == 21 || iddl == 22 || iddl == 36 || iddl == 39) {
49  continue;
50  }
51 
52  auto [crorc, link] = mGeometry->getLinkAssignment(iddl);
53  auto flpID = (iddl <= 23) ? 146 : 147;
54  std::string rawfilename = mOutputLocation;
55  switch (mFileFor) {
56  case FileFor_t::kFullDet:
57  rawfilename += "/emcal.raw";
58  break;
59  case FileFor_t::kSubDet:
60  rawfilename += fmt::format("/EMC_alio2-cr1-flp{:d}.raw", flpID);
61  break;
62  case FileFor_t::kCRORC:
63  rawfilename += fmt::format("/EMC_alio2-cr1-flp{:d}_cru{:d}.raw", flpID, crorc);
64  break;
65  case FileFor_t::kLink:
66  // Pileup simulation based on DigitsWriteoutBuffer (EMCAL-681) - AliceO2 â€“ H. Hassan
67  rawfilename += fmt::format("/EMC_alio2-cr1-flp{:d}_cru{:d}_lnk{:d}.raw", flpID, crorc, link);
68  break;
69  }
70  mRawWriter->registerLink(iddl, crorc, link, 0, rawfilename.data());
71  }

82 bool RawWriter::processTrigger(const o2::emcal::TriggerRecord& trg)
83 {
84  for (auto& srucont : mSRUdata) {
85  srucont.mChannels.clear();
86  }
87  std::vector<o2::emcal::Digit*>* bunchDigits;
88  int lasttower = -1;
89  for (auto& dig : gsl::span(mDigits.data() + trg.getFirstEntry(), trg.getNumberOfObjects())) {
90  auto tower = dig.getTower();
91  if (tower != lasttower) {
92  lasttower = tower;
93  if (tower > 20000) {
94  std::cout << "Wrong cell ID " << tower << std::endl;
95  }
96  auto onlineindices = mGeometry->getOnlineID(tower);
97  int sruID = std::get<0>(onlineindices);
98  auto towerdata = mSRUdata[sruID].mChannels.find(tower);
99  if (towerdata == mSRUdata[sruID].mChannels.end()) {
100  mSRUdata[sruID].mChannels[tower] = {std::get<1>(onlineindices), std::get<2>(onlineindices), std::vector<o2::emcal::Digit*>(mNADCSamples)};
101  bunchDigits = &(mSRUdata[sruID].mChannels[tower].mDigits);
102  memset(bunchDigits->data(), 0, sizeof(o2::emcal::Digit*) * mNADCSamples);
103  } else {
104  bunchDigits = &(towerdata->second.mDigits);
105  }
106  }
107 
108  // Get time sample of the digit:
109  // Digitizer stores the time sample in ns, needs to be converted to time sample dividing
110  // by the length of the time sample
111  auto timesample = int(dig.getTimeStamp() / emcal::constants::EMCAL_TIMESAMPLE);
112  if (timesample >= mNADCSamples) {
113  LOG(error) << "Digit time sample " << timesample << " outside range [0," << mNADCSamples << "]";
114  continue;
115  }
116  (*bunchDigits)[timesample] = &dig;
117  }
118 
119  // Create and fill DMA pages for each channel
120  LOG(debug) << "encode data";
121  for (auto srucont : mSRUdata) {
122 
123  std::vector<char> payload; // this must be initialized per SRU, becuase pages are per SRU, therefore the payload was not reset.
124 
125  if (srucont.mSRUid == 21 || srucont.mSRUid == 22 || srucont.mSRUid == 36 || srucont.mSRUid == 39) {
126  continue;
127  }
128 
129  for (const auto& [tower, channel] : srucont.mChannels) {
130  // Find out hardware address of the channel
131  auto hwaddress = mMappingHandler->getMappingForDDL(srucont.mSRUid).getHardwareAddress(channel.mRow, channel.mCol, ChannelType_t::HIGH_GAIN); // @TODO distinguish between high- and low-gain cells
132 
133  std::vector<int> rawbunches;
134  int nbunches = 0;
135  for (auto& bunch : findBunches(channel.mDigits)) {
136  if (!bunch.mADCs.size()) {
137  LOG(error) << "Found bunch with without ADC entries - skipping ...";
138  continue;
139  }
140  rawbunches.push_back(bunch.mADCs.size() + 2); // add 2 words for header information
141  rawbunches.push_back(bunch.mStarttime);
142  for (auto adc : bunch.mADCs) {
143  rawbunches.push_back(adc);
144  }
145  nbunches++;
146  }
147  if (!rawbunches.size()) {
148  LOG(debug) << "No bunch selected";
149  continue;
150  }
151  LOG(debug) << "Selected " << nbunches << " bunches";
152 
153  auto encodedbunches = encodeBunchData(rawbunches);
154  auto chanhead = createChannelHeader(hwaddress, rawbunches.size(), false);
155  char* chanheadwords = reinterpret_cast<char*>(&chanhead);
156  uint32_t* testheader = reinterpret_cast<uint32_t*>(chanheadwords);
157  if ((*testheader >> 30) & 1) {
158  // header pattern found, check that the payload size is properly reflecting the number of words
159  uint32_t payloadsizeRead = ((*testheader >> 16) & 0x3FF);
160  uint32_t nwordsRead = (payloadsizeRead + 2) / 3;
161  if (encodedbunches.size() != nwordsRead) {
162  LOG(error) << "Mismatch in number of 32-bit words, encoded " << encodedbunches.size() << ", recalculated " << nwordsRead << std::endl;
163  LOG(error) << "Payload size: " << payloadsizeRead << ", number of words: " << rawbunches.size() << ", encodeed words " << encodedbunches.size() << ", calculated words " << nwordsRead << std::endl;
164  } else {
165  LOG(debug) << "Matching number of payload 32-bit words, encoded " << encodedbunches.size() << ", decoded " << nwordsRead;
166  }
167  } else {
168  LOG(error) << "Header without header bit detected ..." << std::endl;
169  }
170  for (int iword = 0; iword < sizeof(ChannelHeader) / sizeof(char); iword++) {
171  payload.emplace_back(chanheadwords[iword]);
172  }
173  char* channelwords = reinterpret_cast<char*>(encodedbunches.data());
174  for (auto iword = 0; iword < encodedbunches.size() * sizeof(int) / sizeof(char); iword++) {
175  payload.emplace_back(channelwords[iword]);
176  }
177  }
178 
179  if (!payload.size()) {
180  // [EMCAL-699] No payload found in SRU
181  // Still the link is not completely ignored but a trailer with 0-payloadsize is added
182  LOG(debug) << "Payload buffer has size 0 - only write empty trailer" << std::endl;
183  }
184  LOG(debug) << "Payload buffer has size " << payload.size();
185 
186  // Create RCU trailer
187  auto trailerwords = createRCUTrailer(payload.size() / 4, 100., trg.getBCData().toLong(), srucont.mSRUid);
188  for (auto word : trailerwords) {
189  payload.emplace_back(word);
190  }
191 
192  // register output data
193  auto ddlid = srucont.mSRUid;
194  auto [crorc, link] = mGeometry->getLinkAssignment(ddlid);
195  LOG(debug1) << "Adding payload with size " << payload.size() << " (" << payload.size() / 4 << " ALTRO words)";
196  mRawWriter->addData(ddlid, crorc, link, 0, trg.getBCData(), payload, false, trg.getTriggerBits());
197  }
198  LOG(debug) << "Done";
199  return true;
200 }

202 std::vector<AltroBunch> RawWriter::findBunches(const std::vector<o2::emcal::Digit*>& channelDigits)
203 {
204  std::vector<AltroBunch> result;
205  AltroBunch currentBunch;
206  bool bunchStarted = false;
207  // Digits in ALTRO bunch in time-reversed order
208  int itime;
209  for (itime = channelDigits.size() - 1; itime >= 0; itime--) {
210  auto dig = channelDigits[itime];
211  if (!dig) {
212  if (bunchStarted) {
213  // we have a bunch which is started and needs to be closed
214  // check if the ALTRO bunch has a minimum amount of ADCs
215  if (currentBunch.mADCs.size() >= mMinADCBunch) {
216  // Bunch selected, set start time and push to bunches
217  currentBunch.mStarttime = itime + 1;
218  result.push_back(currentBunch);
219  currentBunch = AltroBunch();
220  bunchStarted = false;
221  }
222  }
223  continue;
224  }
225  int adc = dig->getAmplitudeADC();
226  if (adc < mPedestal) {
227  // ADC value below threshold
228  // in case we have an open bunch it needs to be stopped bunch
229  // Set the start time to the time sample of previous (passing) digit
230  if (bunchStarted) {
231  // check if the ALTRO bunch has a minimum amount of ADCs
232  if (currentBunch.mADCs.size() >= mMinADCBunch) {
233  // Bunch selected, set start time and push to bunches
234  currentBunch.mStarttime = itime + 1;
235  result.push_back(currentBunch);
236  currentBunch = AltroBunch();
237  bunchStarted = false;
238  }
239  }
240  }
241  // Valid ADC value, if the bunch is closed we start a new bunch
242  if (!bunchStarted) {
243  bunchStarted = true;
244  }
245  currentBunch.mADCs.emplace_back(adc);
246  }
247  // if we have a last bunch set time start time to the time bin of teh previous digit
248  if (bunchStarted) {
249  if (currentBunch.mADCs.size() >= mMinADCBunch) {
250  currentBunch.mStarttime = itime + 1;
251  result.push_back(currentBunch);
252  }
253  }
254  return result;
255 }

257 std::vector<int> RawWriter::encodeBunchData(const std::vector<int>& data)
258 {
259  std::vector<int> encoded;
260  CaloBunchWord currentword;
261  currentword.mDataWord = 0;
262  int wordnumber = 0;
263  for (auto adc : data) {
264  if (adc > 0x3FF) {
265  LOG(error) << "Exceeding max ADC count for 10 bit ALTRO word: " << adc << " (max: 1023)" << std::endl;
266  }
267  switch (wordnumber) {
268  case 0:
269  currentword.mWord0 = adc;
270  break;
271  case 1:
272  currentword.mWord1 = adc;
273  break;
274  case 2:
275  currentword.mWord2 = adc;
276  break;
277  };
278  wordnumber++;
279  if (wordnumber == 3) {
280  // start new word;
281  encoded.push_back(currentword.mDataWord);
282  currentword.mDataWord = 0;
283  wordnumber = 0;
284  }
285  }
286  if (wordnumber) {
287  encoded.push_back(currentword.mDataWord);
288  }
289  return encoded;
290 }

302 std::vector<char> RawWriter::createRCUTrailer(int payloadsize, double timesample, uint64_t triggertime, int feeID)
303 {
304  RCUTrailer trailer;
305  trailer.setPayloadSize(payloadsize);
306  trailer.setTimeSamplePhaseNS(triggertime, timesample);
307 
308  // You can find details about these settings here https://alice.its.cern.ch/jira/browse/EMCAL-650
309  trailer.setRCUID(feeID);
310  trailer.setFirmwareVersion(2);
311  trailer.setActiveFECsA(0x0);
312  trailer.setActiveFECsB(0x1);
313  trailer.setBaselineCorrection(0);
314  trailer.setPolarity(false);
315  trailer.setNumberOfPresamples(0);
316  trailer.setNumberOfPostsamples(0);
317  trailer.setSecondBaselineCorrection(false);
318  trailer.setGlitchFilter(0);
319  trailer.setNumberOfNonZeroSuppressedPostsamples(1);
320  trailer.setNumberOfNonZeroSuppressedPresamples(1);
321  trailer.setNumberOfPretriggerSamples(0);
322  trailer.setNumberOfSamplesPerChannel(15);
323  // For MC we don't simulate pedestals. In order to prevent pedestal subtraction
324  // in the raw fitter we set the zero suppression to true in the RCU trailer
325  trailer.setZeroSuppression(true);
326  trailer.setSparseReadout(true);
327  trailer.setNumberOfAltroBuffers(RCUTrailer::BufferMode_t::NBUFFERS4);
328 
329  auto trailerwords = trailer.encode();
330  std::vector<char> encoded(trailerwords.size() * sizeof(uint32_t));
331  memcpy(encoded.data(), trailerwords.data(), trailerwords.size() * sizeof(uint32_t));
332  return encoded;
333 }
