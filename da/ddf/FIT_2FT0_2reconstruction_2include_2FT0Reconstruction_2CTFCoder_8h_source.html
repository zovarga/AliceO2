70 template <typename VEC>
71 void CTFCoder::encode(VEC& buff, const gsl::span<const Digit>& digitVec, const gsl::span<const ChannelData>& channelVec)
72 {
73  using MD = o2::ctf::Metadata::OptStore;
74  // what to do which each field: see o2::ctd::Metadata explanation
75  constexpr MD optField[CTF::getNBlocks()] = {
76  MD::EENCODE, // BLC_trigger
77  MD::EENCODE, // BLC_bcInc
78  MD::EENCODE, // BLC_orbitInc
79  MD::EENCODE, // BLC_nChan
80  MD::EENCODE, // BLC_status
81  MD::EENCODE, // BLC_idChan
82  MD::EENCODE, // BLC_qtcChain
83  MD::EENCODE, // BLC_cfdTime
84  MD::EENCODE // BLC_qtcAmpl
85  };
86  CompressedDigits cd;
87  compress(cd, digitVec, channelVec);
88 
89  // book output size with some margin
90  auto szIni = estimateCompressedSize(cd);
91  buff.resize(szIni);
92 
93  auto ec = CTF::create(buff);
94  using ECB = CTF::base;
95 
96  ec->setHeader(cd.header);
97  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
98  ec->getANSHeader().majorVersion = 0;
99  ec->getANSHeader().minorVersion = 1;
100  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
101 #define ENCODEFT0(part, slot, bits) CTF::get(buff.data())->encode(part, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
102  // clang-format off
103  ENCODEFT0(cd.trigger, CTF::BLC_trigger, 0);
104  ENCODEFT0(cd.bcInc, CTF::BLC_bcInc, 0);
105  ENCODEFT0(cd.orbitInc, CTF::BLC_orbitInc, 0);
106  ENCODEFT0(cd.nChan, CTF::BLC_nChan, 0);
107  ENCODEFT0(cd.eventStatus, CTF::BLC_status, 0);
108  ENCODEFT0(cd.idChan , CTF::BLC_idChan, 0);
109  ENCODEFT0(cd.qtcChain, CTF::BLC_qtcChain, 0);
110  ENCODEFT0(cd.cfdTime, CTF::BLC_cfdTime, 0);
111  ENCODEFT0(cd.qtcAmpl, CTF::BLC_qtcAmpl, 0);
112  // clang-format on
113  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
114 }

141 template <typename VDIG, typename VCHAN>
142 void CTFCoder::decompress(const CompressedDigits& cd, VDIG& digitVec, VCHAN& channelVec)
143 {
144  digitVec.clear();
145  channelVec.clear();
146  digitVec.reserve(cd.header.nTriggers);
147  channelVec.reserve(cd.idChan.size());
148 
149  uint32_t firstEntry = 0, clCount = 0, chipCount = 0;
150  o2::InteractionRecord ir(cd.header.firstBC, cd.header.firstOrbit);
151 
152  for (uint32_t idig = 0; idig < cd.header.nTriggers; idig++) {
153  // restore ROFRecord
154  if (cd.orbitInc[idig]) { // non-0 increment => new orbit
155  ir.bc = cd.bcInc[idig]; // bcInc has absolute meaning
156  ir.orbit += cd.orbitInc[idig];
157  } else {
158  ir.bc += cd.bcInc[idig];
159  }
160  Triggers trig;
161  trig.triggersignals = cd.trigger[idig];
162  const auto& params = DigitizationParameters::Instance();
163  int triggerGate = params.mTime_trg_gate;
164  firstEntry = channelVec.size();
165  uint8_t chID = 0;
166  int amplA = 0, amplC = 0, timeA = 0, timeC = 0;
167 
168  for (uint8_t ic = 0; ic < cd.nChan[idig]; ic++) {
169  auto icc = channelVec.size();
170  const auto& chan = channelVec.emplace_back((chID += cd.idChan[icc]), cd.cfdTime[icc], cd.qtcAmpl[icc], cd.qtcChain[icc]);
171  if (std::abs(chan.CFDTime) < triggerGate) {
172  if (chan.ChId < 4 * uint8_t(Geometry::NCellsA)) { // A side
173  amplA += chan.QTCAmpl;
174  timeA += chan.CFDTime;
175  trig.nChanA++;
176 
177  } else {
178  amplC += chan.QTCAmpl;
179  timeC += chan.CFDTime;
180  trig.nChanC++;
181  }
182  }
183  }
184  if (trig.nChanA) {
185  trig.timeA = timeA / trig.nChanA;
186  trig.amplA = amplA / 8;
187  }
188  if (trig.nChanC) {
189  trig.timeC = timeC / trig.nChanC;
190  trig.amplC = amplC / 8;
191  }
192  auto& d = digitVec.emplace_back(firstEntry, cd.nChan[idig], ir, trig, idig);
193  d.setEventStatus(cd.eventStatus[idig]);
194  }
195 }
