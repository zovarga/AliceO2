101  inline float signalForm(float x) const
102  { // table lookup for the signal shape
103  if (x <= 0.0f) {
104  return 0.0f;
105  }
106  float const y = x / DigitizationParameters::Instance().mBunchWidth * DP::SIGNAL_TABLE_SIZE;
107  int const index = std::floor(y);
108  if (index + 1 >= DP::SIGNAL_TABLE_SIZE) {
109  return mSignalTable.back();
110  }
111  float const rem = y - index;
112  return mSignalTable[index] + rem * (mSignalTable[index + 1] - mSignalTable[index]);
113  }
114 
115  template <typename VcType>
116  inline VcType signalFormVc(VcType x) const
117  { // table lookup for the signal shape (SIMD version)
118  // implemented as template function, so that we don't need to include <Vc/Vc> here
119  auto const y = x / DigitizationParameters::Instance().mBunchWidth * DP::SIGNAL_TABLE_SIZE;
120  typename VcType::IndexType const index = floor(y);
121  auto const rem = y - index;
122  VcType val(0);
123  for (size_t i = 0; i < VcType::size(); ++i) {
124  if (y[i] < 0.0f) {
125  continue;
126  }
127  if (index[i] + 1 < DP::SIGNAL_TABLE_SIZE) {
128  val[i] = mSignalTable[index[i]] + rem[i] * (mSignalTable[index[i] + 1] - mSignalTable[index[i]]);
129  } else {
130  val[i] = mSignalTable.back();
131  }
132  }
133  return val;
134  }
