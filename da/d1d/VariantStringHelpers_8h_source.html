20 template <typename T>
21 T lexical_cast(std::string const& input)
22 {
23  if constexpr (std::is_same_v<T, int>) {
24  return std::stoi(input, nullptr);
25  } else if constexpr (std::is_same_v<T, int64_t>) {
26  return std::stol(input, nullptr);
27  } else if constexpr (std::is_same_v<T, float>) {
28  return std::stof(input, nullptr);
29  } else if constexpr (std::is_same_v<T, double>) {
30  return std::stod(input, nullptr);
31  } else if constexpr (std::is_same_v<T, bool>) {
32  return static_cast<bool>(std::stoi(input, nullptr));
33  }
34 }

37 std::vector<T> stringToVector(std::string const& input)
38 {

41  assert(input[0] == variant_array_symbol<T>::symbol);
42  std::regex nmatch(R"((?:(?!=,)|(?!=\[))[+-]?\d+\.?\d*(?:[eE][+-]?\d+)?(?=,|\]))");
43  auto end = std::sregex_iterator();

66 template <typename T>
67 Array2D<T> stringToArray2D(std::string const& input)
68 {
69  std::vector<T> cache;
70  assert(input[0] == variant_array_symbol<T>::symbol);
71  std::regex mrows(R"(\[[^\[\]]+\])");
72  std::regex marray(R"((?:(?!=,)|(?!=\[))[+-]?\d+\.?\d*(?:[eE][+-]?\d+)?(?=,|\]))");
73  auto end = std::sregex_iterator();
74  auto rows = std::sregex_iterator(input.begin(), input.end(), mrows);
75  uint32_t nrows = 0;
76  uint32_t ncols = 0;
77  bool first = true;
78  for (auto& row = rows; row != end; ++row) {
79  auto str = row->str();
80  auto values = std::sregex_iterator(str.begin(), str.end(), marray);
81  if (first) {
82  ncols = 0;
83  }
84  for (auto& v = values; v != end; ++v) {
85  cache.push_back(lexical_cast<T>(v->str()));
86  if (first) {
87  ++ncols;
88  }
89  }
90  if (first) {
91  first = false;
92  }
93  ++nrows;
94  }
95  return Array2D<T>{cache, nrows, ncols};
