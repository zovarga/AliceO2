32 int PVertexer::runVertexing(const gsl::span<o2d::GlobalTrackID> gids, const gsl::span<o2::InteractionRecord> bcData,
33  std::vector<PVertex>& vertices, std::vector<o2d::VtxTrackIndex>& vertexTrackIDs, std::vector<V2TRef>& v2tRefs,
34  gsl::span<const o2::MCCompLabel> lblTracks, std::vector<o2::MCEventLabel>& lblVtx)
35 {
36  dbscan_clusterize();
37  std::vector<PVertex> verticesLoc;
38  std::vector<uint32_t> trackIDs;
39  std::vector<V2TRef> v2tRefsLoc;
40  std::vector<float> validationTimes;
41  std::vector<o2::MCEventLabel> lblVtxLoc;
42  for (auto tc : mTimeZClusters) {
43  VertexingInput inp;
44  inp.idRange = gsl::span<int>(tc.trackIDs);
45  inp.scaleSigma2 = mPVParams->iniScale2;
46  inp.timeEst = tc.timeEst;
47 #ifdef _PV_DEBUG_TREE_
48  doDBScanDump(inp, lblTracks);
49 #endif
50  findVertices(inp, verticesLoc, trackIDs, v2tRefsLoc);
51  }
52 
53  // sort in time
54  std::vector<int> vtTimeSortID(verticesLoc.size());
55  std::iota(vtTimeSortID.begin(), vtTimeSortID.end(), 0);
56  std::sort(vtTimeSortID.begin(), vtTimeSortID.end(), [&verticesLoc](int i, int j) {
57  return verticesLoc[i].getTimeStamp().getTimeStamp() < verticesLoc[j].getTimeStamp().getTimeStamp();
58  });
59 
60 #ifdef _PV_DEBUG_TREE_
61  if (lblTracks.size()) { // at this stage labels are needed just for the debug output
62  createMCLabels(lblTracks, trackIDs, v2tRefsLoc, lblVtxLoc);
63  }
64 #endif
65 
66  if (mPVParams->applyDebrisReduction) {
67  reduceDebris(verticesLoc, vtTimeSortID, lblVtxLoc);
68  }
69  if (mPVParams->applyReattachment) {
70  reAttach(verticesLoc, vtTimeSortID, trackIDs, v2tRefsLoc);
71  }
72 
73  if (lblTracks.size()) { // at this stage labels are needed just for the debug output
74  createMCLabels(lblTracks, trackIDs, v2tRefsLoc, lblVtxLoc);
75  }
76 
77 #ifdef _PV_DEBUG_TREE_
78  doVtxDump(verticesLoc, trackIDs, v2tRefsLoc, lblTracks);
79 #endif
80 
81  vertices.clear();
82  v2tRefs.clear();
83  vertexTrackIDs.clear();
84  vertices.reserve(verticesLoc.size());
85  v2tRefs.reserve(v2tRefsLoc.size());
86  vertexTrackIDs.reserve(trackIDs.size());
87 
88  int trCopied = 0, count = 0, vtimeID = 0;
89  for (auto i : vtTimeSortID) {
90  if (i < 0) {
91  continue; // vertex was suppressed
92  }
93  auto& vtx = verticesLoc[i];
94 
95  bool irSet = setCompatibleIR(vtx);
96  if (!irSet) {
97  continue;
98  }
99  // do we need to validate by Int. records ?
100  if (mValidateWithIR) {
101  auto bestMatch = getBestIR(vtx, bcData, vtimeID);
102  if (bestMatch.first >= 0) {
103  vtx.setFlags(PVertex::TimeValidated);
104  if (bestMatch.second == 1) {
105  vtx.setIR(bcData[bestMatch.first]);
106  }
107  LOG(debug) << "Validated with t0 " << bestMatch.first << " with " << bestMatch.second << " candidates";
108  } else if (vtx.getNContributors() >= mPVParams->minNContributorsForIRcut) {
109  LOG(debug) << "Discarding " << vtx;
110  continue; // reject
111  }
112  }
113  vertices.push_back(vtx);
114  if (lblVtxLoc.size()) {
115  lblVtx.push_back(lblVtxLoc[i]);
116  }
117  int it = v2tRefsLoc[i].getFirstEntry(), itEnd = it + v2tRefsLoc[i].getEntries(), dest0 = vertexTrackIDs.size();
118  for (; it < itEnd; it++) {
119  const auto& trc = mTracksPool[trackIDs[it]];
120  auto& gid = vertexTrackIDs.emplace_back(trc.gid); // assign global track ID
121  gid.setPVContributor();
122  }
123  v2tRefs.emplace_back(dest0, v2tRefsLoc[i].getEntries());
124  LOG(debug) << "#" << count++ << " " << vertices.back() << " | " << v2tRefs.back().getEntries() << " indices from " << v2tRefs.back().getFirstEntry(); // RS REM
125  }
126 
127  return vertices.size();
128 }

131 int PVertexer::findVertices(const VertexingInput& input, std::vector<PVertex>& vertices, std::vector<uint32_t>& trackIDs, std::vector<V2TRef>& v2tRefs)
132 {
133  // find vertices using tracks with indices (sorted in time) from idRange from "tracks" pool. The pool may containt arbitrary number of tracks,
134  // only those which are in the idRange and have canUse()==true, will be used.
135  // Results are placed in vertices and v2tRefs vectors
136 
137  int nfound = 0, ntr = 0;
138  auto seedHistoTZ = buildHistoTZ(input); // histo for seeding peak finding
139 
140 #ifdef _PV_DEBUG_TREE_
141  static int dbsCount = -1;
142  dbsCount++;
143  int trialCount = 0; // TODO REM
144  auto hh = seedHistoTZ.createTH2F(o2::utils::Str::concat_string("htz", std::to_string(dbsCount)));
145  hh->SetDirectory(nullptr);
146  mDebugDumpFile->cd();
147  hh->Write();
148 #endif
149 
150  int nTrials = 0;
151  while (nfound < mPVParams->maxVerticesPerCluster && nTrials < mPVParams->maxTrialsPerCluster) {
152  int peakBin = seedHistoTZ.findPeakBin();
153  if (!seedHistoTZ.isValidBin(peakBin)) {
154  break;
155  }
156  int peakBinT = seedHistoTZ.getXBin(peakBin), peakBinZ = seedHistoTZ.getYBin(peakBin);
157  float tv = seedHistoTZ.getBinXCenter(peakBinT);
158  float zv = seedHistoTZ.getBinYCenter(peakBinZ);
159  LOG(debug) << "Seeding with T=" << tv << " Z=" << zv << " bin " << peakBin << " on trial " << nTrials << " for vertex " << nfound;
160 
161  PVertex vtx;
162  vtx.setXYZ(mMeanVertex.getX(), mMeanVertex.getY(), zv);
163  vtx.setTimeStamp({tv, 0.f});
164  if (findVertex(input, vtx)) {
165  finalizeVertex(input, vtx, vertices, v2tRefs, trackIDs, &seedHistoTZ);
166  nfound++;
167  nTrials = 0;
168  } else { // suppress failed seeding bin and its proximities
169  seedHistoTZ.setBinContent(peakBin, -1);
170  }
171  nTrials++;
172 
173 #ifdef _PV_DEBUG_TREE_
174  auto hh1 = seedHistoTZ.createTH2F(o2::utils::Str::concat_string("htz", std::to_string(dbsCount), "_", std::to_string(trialCount++)));
175  hh1->SetDirectory(nullptr);
176  mDebugDumpFile->cd();
177  hh1->Write();
178 #endif
179  }
180  return nfound;
181 }

184 bool PVertexer::findVertex(const VertexingInput& input, PVertex& vtx)
185 {
186  // fit vertex taking provided vertex as a seed
187  // tracks pool may contain arbitrary number of tracks, only those which are in
188  // the idRange (indices of tracks sorted in time) will be used.
189 
190  int ntr = input.idRange.size(); // RSREM
191 
192  VertexSeed vtxSeed(vtx);
193  vtxSeed.setScale(input.scaleSigma2, mTukey2I);
194  vtxSeed.scaleSigma2Prev = input.scaleSigma2;
195  // vtxSeed.setTimeStamp( timeEstimate(input) );
196  LOG(debug) << "Start time guess: " << vtxSeed.getTimeStamp();
197  vtx.setChi2(1.e30);
198  //
199  FitStatus result = FitStatus::IterateFurther;
200  bool found = false;
201  while (result == FitStatus::IterateFurther) {
202  vtxSeed.resetForNewIteration();
203  vtxSeed.nIterations++;
204  LOG(debug) << "iter " << vtxSeed.nIterations << " with scale=" << vtxSeed.scaleSigma2 << " prevScale=" << vtxSeed.scaleSigma2Prev
205  << " ntr=" << ntr << " Zv=" << vtxSeed.getZ() << " Tv=" << vtxSeed.getTimeStamp().getTimeStamp();
206  result = fitIteration(input, vtxSeed);
207 
208  if (result == FitStatus::OK) {
209  result = evalIterations(vtxSeed, vtx);
210  } else if (result == FitStatus::NotEnoughTracks) {
211  if (vtxSeed.nIterations <= mPVParams->maxIterations && upscaleSigma(vtxSeed)) {
212  LOG(debug) << "Upscaling scale to " << vtxSeed.scaleSigma2;
213  result = FitStatus::IterateFurther;
214  continue; // redo with stronger rescaling
215  } else {
216  break;
217  }
218  } else if (result == FitStatus::PoolEmpty || result == FitStatus::Failure) {
219  break;
220  } else {
221  LOG(fatal) << "Unknown fit status " << int(result);
222  }
223  }
224  LOG(debug) << "Stopped with scale=" << vtxSeed.scaleSigma2 << " prevScale=" << vtxSeed.scaleSigma2Prev << " result = " << int(result) << " ntr=" << ntr;
225 
226  if (result != FitStatus::OK) {
227  vtx.setChi2(vtxSeed.maxScaleSigma2Tested);
228  return false;
229  } else {
230  return true;
231  }
232 }

235 PVertexer::FitStatus PVertexer::fitIteration(const VertexingInput& input, VertexSeed& vtxSeed)
236 {
237  int nTested = 0;
238  for (int i : input.idRange) {
239  if (mTracksPool[i].canUse()) {
240  accountTrack(mTracksPool[i], vtxSeed);
241  // printf("#%d z:%f t:%f te:%f w:%f wh:%f\n", nTested, mTracksPool[i].z, mTracksPool[i].timeEst.getTimeStamp(),
242  // mTracksPool[i].timeEst.getTimeStampError(), mTracksPool[i].wgh, mTracksPool[i].wghHisto);
243  nTested++;
244  }
245  }
246 
247  vtxSeed.maxScaleSigma2Tested = vtxSeed.scaleSigma2;
248  if (vtxSeed.getNContributors() < mPVParams->minTracksPerVtx) {
249  return nTested < mPVParams->minTracksPerVtx ? FitStatus::PoolEmpty : FitStatus::NotEnoughTracks;
250  }
251  if (mPVParams->useMeanVertexConstraint) {
252  applyConstraint(vtxSeed);
253  }
254  if (!solveVertex(vtxSeed)) {
255  return FitStatus::Failure;
256  }
257  return FitStatus::OK;
258 }

261 void PVertexer::accountTrack(TrackVF& trc, VertexSeed& vtxSeed) const
262 {
263  // deltas defined as track - vertex
264  bool useTime = vtxSeed.getTimeStamp().getTimeStampError() >= 0.f;
265  auto chi2T = trc.evalChi2ToVertex(vtxSeed, useTime && mPVParams->useTimeInChi2);
266  float wghT = (1.f - chi2T * vtxSeed.scaleSig2ITuk2I); // weighted distance to vertex
267  if (wghT < kAlmost0F) {
268  trc.wgh = 0.f;
269  return;
270  }
271  wghT *= wghT;
272  float syyI(trc.sig2YI), szzI(trc.sig2ZI), syzI(trc.sigYZI);
273 
274  auto timeErrorFromTB = [&trc]() {
275  // decide if the time error is from the time bracket rather than gaussian error
276  return trc.gid.getSource() == GTrackID::ITS;
277  };
278 
279  //
280  vtxSeed.wghSum += wghT;
281  vtxSeed.wghChi2 += wghT * chi2T;
282  //
283  syyI *= wghT;
284  syzI *= wghT;
285  szzI *= wghT;
286  trc.wgh = wghT;
287  //
288  // aux variables
289  double tmpSP = trc.sinAlp * trc.tgP, tmpCP = trc.cosAlp * trc.tgP,
290  tmpSC = trc.sinAlp + tmpCP, tmpCS = -trc.cosAlp + tmpSP,
291  tmpCL = trc.cosAlp * trc.tgL, tmpSL = trc.sinAlp * trc.tgL,
292  tmpYXP = trc.y - trc.tgP * trc.x, tmpZXL = trc.z - trc.tgL * trc.x,
293  tmpCLzz = tmpCL * szzI, tmpSLzz = tmpSL * szzI, tmpSCyz = tmpSC * syzI,
294  tmpCSyz = tmpCS * syzI, tmpCSyy = tmpCS * syyI, tmpSCyy = tmpSC * syyI,
295  tmpSLyz = tmpSL * syzI, tmpCLyz = tmpCL * syzI;
296  //
297  // symmetric matrix equation
298  vtxSeed.cxx += tmpCL * (tmpCLzz + tmpSCyz + tmpSCyz) + tmpSC * tmpSCyy; // dchi^2/dx/dx
299  vtxSeed.cxy += tmpCL * (tmpSLzz + tmpCSyz) + tmpSL * tmpSCyz + tmpSC * tmpCSyy; // dchi^2/dx/dy
300  vtxSeed.cxz += -trc.sinAlp * syzI - tmpCLzz - tmpCP * syzI; // dchi^2/dx/dz
301  vtxSeed.cx0 += -(tmpCLyz + tmpSCyy) * tmpYXP - (tmpCLzz + tmpSCyz) * tmpZXL; // RHS
302  //
303  vtxSeed.cyy += tmpSL * (tmpSLzz + tmpCSyz + tmpCSyz) + tmpCS * tmpCSyy; // dchi^2/dy/dy
304  vtxSeed.cyz += -(tmpCSyz + tmpSLzz); // dchi^2/dy/dz
305  vtxSeed.cy0 += -tmpYXP * (tmpCSyy + tmpSLyz) - tmpZXL * (tmpCSyz + tmpSLzz); // RHS
306  //
307  vtxSeed.czz += szzI; // dchi^2/dz/dz
308  vtxSeed.cz0 += tmpZXL * szzI + tmpYXP * syzI; // RHS
309  //
310  if (useTime) {
311  float trErr2I = wghT / (trc.timeEst.getTimeStampError() * trc.timeEst.getTimeStampError());
312  if (timeErrorFromTB()) {
313  vtxSeed.tMeanAccTB += trc.timeEst.getTimeStamp() * trErr2I;
314  vtxSeed.tMeanAccErrTB += trErr2I;
315  vtxSeed.nContributorsTB++;
316  vtxSeed.wghSumTB += wghT;
317  } else {
318  vtxSeed.tMeanAcc += trc.timeEst.getTimeStamp() * trErr2I;
319  vtxSeed.tMeanAccErr += trErr2I;
320  }
321  }
322  vtxSeed.addContributor();
323 }

326 bool PVertexer::solveVertex(VertexSeed& vtxSeed) const
327 {
328  ROOT::Math::SMatrix<double, 3, 3, ROOT::Math::MatRepSym<double, 3>> mat;
329  mat(0, 0) = vtxSeed.cxx;
330  mat(0, 1) = vtxSeed.cxy;
331  mat(0, 2) = vtxSeed.cxz;
332  mat(1, 1) = vtxSeed.cyy;
333  mat(1, 2) = vtxSeed.cyz;
334  mat(2, 2) = vtxSeed.czz;
335  if (!mat.InvertFast()) {
336  LOG(error) << "Failed to invert matrix" << mat;
337  return false;
338  }
339  ROOT::Math::SVector<double, 3> rhs(vtxSeed.cx0, vtxSeed.cy0, vtxSeed.cz0);
340  auto sol = mat * rhs;
341  vtxSeed.setXYZ(sol(0), sol(1), sol(2));
342  vtxSeed.setCov(mat(0, 0), mat(1, 0), mat(1, 1), mat(2, 0), mat(2, 1), mat(2, 2));
343  if (vtxSeed.tMeanAccErr + vtxSeed.tMeanAccErrTB > 0.) {
344  // since the time error from the ITS measurements does not improve with statistics, we downscale it with number of such tracks
345  auto t = vtxSeed.tMeanAcc;
346  auto e2i = vtxSeed.tMeanAccErr;
347  if (vtxSeed.wghSumTB > 0.) {
348  t += vtxSeed.tMeanAccTB / vtxSeed.wghSumTB;
349  e2i += vtxSeed.tMeanAccErrTB / vtxSeed.wghSumTB;
350  }
351  auto err2 = 1. / e2i;
352  vtxSeed.setTimeStamp({float(t * err2), float(std::sqrt(err2))});
353  }
354 
355  vtxSeed.setChi2((vtxSeed.getNContributors() - vtxSeed.wghSum) / vtxSeed.scaleSig2ITuk2I); // calculate chi^2
356  auto newScale = vtxSeed.wghChi2 / vtxSeed.wghSum;
357  LOG(debug) << "Solve: wghChi2=" << vtxSeed.wghChi2 << " wghSum=" << vtxSeed.wghSum << " -> scale= " << newScale << " old scale " << vtxSeed.scaleSigma2 << " prevScale: " << vtxSeed.scaleSigma2Prev;
358  vtxSeed.setScale(newScale < mPVParams->minScale2 ? mPVParams->minScale2 : newScale, mTukey2I);
359  return true;
360 }

363 PVertexer::FitStatus PVertexer::evalIterations(VertexSeed& vtxSeed, PVertex& vtx) const
364 {
365  // decide if new iteration should be done, prepare next one if needed
366  // if scaleSigma2 reached its lower limit stop
367  PVertexer::FitStatus result = PVertexer::FitStatus::IterateFurther;
368 
369  if (vtxSeed.nIterations > mPVParams->maxIterations) {
370  result = PVertexer::FitStatus::Failure;
371  } else if (vtxSeed.scaleSigma2Prev <= mPVParams->minScale2 + kAlmost0F) {
372  result = PVertexer::FitStatus::OK;
373  LOG(debug) << "stop on simga :" << vtxSeed.scaleSigma2 << " prev: " << vtxSeed.scaleSigma2Prev;
374  }
375  if (fair::Logger::Logging(fair::Severity::debug)) {
376  auto dchi = (vtx.getChi2() - vtxSeed.getChi2()) / vtxSeed.getChi2();
377  auto dx = vtxSeed.getX() - vtx.getX(), dy = vtxSeed.getY() - vtx.getY(), dz = vtxSeed.getZ() - vtx.getZ();
378  auto dst = std::sqrt(dx * dx + dy * dy + dz * dz);
379 
380  LOG(debug) << "dChi:" << vtx.getChi2() << "->" << vtxSeed.getChi2() << " :-> " << dchi;
381  LOG(debug) << "dx: " << dx << " dy: " << dy << " dz: " << dz << " -> " << dst;
382  }
383 
384  vtx = reinterpret_cast<const PVertex&>(vtxSeed);
385 
386  if (result == PVertexer::FitStatus::OK) {
387  auto chi2Mean = vtxSeed.getChi2() / vtxSeed.getNContributors();
388  if (chi2Mean > mPVParams->maxChi2Mean) {
389  result = PVertexer::FitStatus::Failure;
390  LOG(debug) << "Rejecting at iteration " << vtxSeed.nIterations << " and ScalePrev " << vtxSeed.scaleSigma2Prev << " with meanChi2 = " << chi2Mean;
391  } else {
392  return result;
393  }
394  }
395 
396  if (vtxSeed.scaleSigma2 > vtxSeed.scaleSigma2Prev) {
397  if (++vtxSeed.nScaleIncrease > mPVParams->maxNScaleIncreased) {
398  result = PVertexer::FitStatus::Failure;
399  LOG(debug) << "Rejecting at iteration " << vtxSeed.nIterations << " with NScaleIncreased " << vtxSeed.nScaleIncrease;
400  }
401  } else if (vtxSeed.scaleSigma2 > mPVParams->slowConvergenceFactor * vtxSeed.scaleSigma2Prev) {
402  if (++vtxSeed.nScaleSlowConvergence > mPVParams->maxNScaleSlowConvergence) {
403  if (vtxSeed.scaleSigma2 < mPVParams->acceptableScale2) {
404  vtxSeed.setScale(mPVParams->minScale2, mTukey2I);
405  LOG(debug) << "Forcing scale2 to " << mPVParams->minScale2;
406  result = PVertexer::FitStatus::IterateFurther;
407  } else {
408  result = PVertexer::FitStatus::Failure;
409  LOG(debug) << "Rejecting at iteration " << vtxSeed.nIterations << " with NScaleSlowConvergence " << vtxSeed.nScaleSlowConvergence;
410  }
411  }
412  } else {
413  vtxSeed.nScaleSlowConvergence = 0;
414  }
415 
416  return result;
417 }

420 void PVertexer::reAttach(std::vector<PVertex>& vertices, std::vector<int>& timeSort, std::vector<uint32_t>& trackIDs, std::vector<V2TRef>& v2tRefs)
421 {
422  float tRange = 0.5 * std::max(mITSROFrameLengthMUS, mPVParams->dbscanDeltaT) + mPVParams->timeMarginReattach; // consider only vertices in this proximity to tracks
423  std::vector<std::pair<int, TimeEst>> vtvec; // valid vertex times and indices
424  int nvtOrig = vertices.size();
425  vtvec.reserve(nvtOrig);
426  mTimeZClusters.resize(nvtOrig);
427  for (int ivt = 0; ivt < nvtOrig; ivt++) {
428  mTimeZClusters[ivt].trackIDs.clear();
429  mTimeZClusters[ivt].trackIDs.reserve(int(vertices[ivt].getNContributors() * 1.2));
430  }
431  for (auto ivt : timeSort) {
432  if (ivt >= 0) {
433  vtvec.emplace_back(ivt, vertices[ivt].getTimeStamp());
434  }
435  }
436  int ntr = mTracksPool.size(), nvt = vtvec.size();
437  int vtStart = 0;
438  for (int itr = 0; itr < ntr; itr++) {
439  auto& trc = mTracksPool[itr];
440  trc.vtxID = TrackVF::kNoVtx;
441  trc.wgh = kAlmost0F;
442  for (int ivt = vtStart; ivt < nvt; ivt++) {
443  auto dt = vtvec[ivt].second.getTimeStamp() - trc.timeEst.getTimeStamp();
444  if (dt < -tRange) { // all following tracks will have higher time than the vertex vtStart, move it
445  vtStart = ivt + 1;
446  continue;
447  }
448  if (dt > tRange) { // all following vertices will be have higher time than this track, stop checking
449  break;
450  }
451  bool useTime = trc.gid.getSource() != GTrackID::ITS && mPVParams->useTimeInChi2; // TODO Check if it is ok to not account time error for ITS tracks
452  float wgh = 1.f - mTukey2I * trc.evalChi2ToVertex(vertices[vtvec[ivt].first], useTime);
453  if (wgh > trc.wgh) {
454  trc.wgh = wgh;
455  trc.vtxID = vtvec[ivt].first;
456  }
457  }
458  if (trc.vtxID != TrackVF::kNoVtx) {
459  mTimeZClusters[trc.vtxID].trackIDs.push_back(itr);
460  trc.vtxID = TrackVF::kNoVtx; // to enable for using in the fit
461  trc.bin = -1;
462  }
463  }
464  // refit vertices with reattached tracks
465  v2tRefs.clear();
466  trackIDs.clear();
467  std::vector<PVertex> verticesUpd;
468  for (int ivt = 0; ivt < nvtOrig; ivt++) {
469  auto& clusZT = mTimeZClusters[ivt];
470  auto& vtx = vertices[ivt];
471  if (clusZT.trackIDs.size() < mPVParams->minTracksPerVtx) {
472  continue;
473  }
474  VertexingInput inp;
475  inp.idRange = gsl::span<int>(clusZT.trackIDs);
476  inp.scaleSigma2 = 1.;
477  inp.timeEst = vtx.getTimeStamp();
478  if (!findVertex(inp, vtx)) {

493 //___________________________________________________________________
494 void PVertexer::reduceDebris(std::vector<PVertex>& vertices, std::vector<int>& timeSort, const std::vector<o2::MCEventLabel>& lblVtx)
495 {
496  // eliminate low multiplicity vertices in the close proximity of high mult ones, assuming that these are their debries
497  // The timeSort vector indicates the time ordering of the vertices
498  int nv = vertices.size();
499  std::vector<int> multSort(nv); // sort time indices in multiplicity
500  std::iota(multSort.begin(), multSort.end(), 0);
501  std::sort(multSort.begin(), multSort.end(), [&timeSort, vertices](int i, int j) {
502  return vertices[timeSort[i]].getNContributors() > vertices[timeSort[j]].getNContributors();
503  });
504 
505  // suppress vertex pointed by j if needed, if time difference between i and j is too large, return true to stop checking
506  // pairs starting with i.
507  auto checkPair = [&vertices, &timeSort, &lblVtx, this](int i, int j) {
508  auto &vtI = vertices[timeSort[i]], &vtJ = vertices[timeSort[j]];
509  auto tDiff = std::abs(vtI.getTimeStamp().getTimeStamp() - vtJ.getTimeStamp().getTimeStamp());
510  if (tDiff > this->mPVParams->maxTDiffDebris) {
511  return true; // don't continue checking other neighbours in time
512  }
513  if (vtI.getNContributors() < vtJ.getNContributors()) {
514  return false; // comparison goes from higher to lower mult vtx
515  }
516  bool rej = true;
517  float zDiff = std::abs(vtI.getZ() - vtJ.getZ());
518  if (zDiff > this->mPVParams->maxZDiffDebris) { // cannot be reduced as too far in Z
519 #ifndef _PV_DEBUG_TREE_
520  return false;
521 #endif
522  rej = false;
523  }
524  float multRat = float(vtJ.getNContributors()) / float(vtI.getNContributors());
525  if (multRat > this->mPVParams->maxMultRatDebris) {
526 #ifndef _PV_DEBUG_TREE_
527  return false;
528 #endif
529  rej = false;
530  }
531  float tiE = vtI.getTimeStamp().getTimeStampError(), tjE = vtJ.getTimeStamp().getTimeStampError();
532  float chi2z = zDiff * zDiff / (vtI.getSigmaZ2() + vtJ.getSigmaZ2() + this->mPVParams->addZSigma2Debris);
533  float chi2t = tDiff * tDiff / (tiE * tiE + tjE * tjE + this->mPVParams->addTimeSigma2Debris);
534  if (chi2z + chi2t > this->mPVParams->maxChi2TZDebris) {
535 #ifndef _PV_DEBUG_TREE_
536  return false;
537 #endif
538  rej = false;
539  }
540  // all veto cuts passed, declare as fake!
541 #ifdef _PV_DEBUG_TREE_
542  o2::MCEventLabel dummyLbl;
543  this->mDebugDumpPVComp.emplace_back(PVtxCompDump{vtI, vtJ, chi2z, chi2t, rej});
544  if (!lblVtx.empty()) {
545  this->mDebugDumpPVCompLbl0.push_back(lblVtx[timeSort[i]]);
546  this->mDebugDumpPVCompLbl1.push_back(lblVtx[timeSort[j]]);
547  }
548 #endif
549  if (rej) {
550  timeSort[j] = -1;
551  vtJ.setNContributors(0);
552  }
553  return false;
554  };
555 
556  for (int im = 0; im < nv; im++) { // loop from highest multiplicity to lowest one
557  int it = multSort[im];
558  if (timeSort[it] < 0) { // if <0, the vertex was already discarded
559  continue;
560  }
561 
562  int itL = it; // look for vertices with smaller time
563  while (itL) {
564  if (timeSort[--itL] >= 0) { // if <0, the vertex was already discarded
565  if (checkPair(it, itL)) { // if too far in time, don't compare further
566  break;
567  }
568  }
569  } // itL loop
570  int itH = it; // look for vertices with higher time
571  while (++itH < nv) {
572  if (timeSort[itH] >= 0) { // if <0, the vertex was already discarded
573  if (checkPair(it, itH)) { // if too far in time, don't compare further
574  break;
575  }
576  }
577  } // itH loop
578  }
579 #ifdef _PV_DEBUG_TREE_
580  mDebugVtxCompTree->Fill();
581  mDebugDumpPVComp.clear();
582  mDebugDumpPVCompLbl0.clear();
583  mDebugDumpPVCompLbl1.clear();
584 #endif
585 }

762 //___________________________________________________________________
763 bool PVertexer::setCompatibleIR(PVertex& vtx)
764 {
765  // assign compatible IRs accounting for the bunch filling scheme
766  const auto& vtxT = vtx.getTimeStamp();
767  o2::InteractionRecord irMin(mStartIR), irMax(mStartIR);
768  auto rangeT = std::max(mPVParams->minTError, mPVParams->nSigmaTimeCut * std::min(mPVParams->maxTError, vtxT.getTimeStampError())) + mPVParams->timeMarginVertexTime;
769  float t = vtxT.getTimeStamp() + mPVParams->timeBiasMS;
770  if (t > rangeT) {
771  irMin += o2::InteractionRecord(1.e3 * (t - rangeT));
772  }
773  if (t < -rangeT) {
774  return false; // discard vertex at negative time
775  }
776  irMax += o2::InteractionRecord(1.e3 * (t + rangeT)); // RS TODO: make sure that irMax does not exceed TF edge
777  irMax++; // to account for rounding
778  // restrict using bunch filling
779  int bc = mClosestBunchAbove[irMin.bc];
780  LOG(debug) << "irMin.bc = " << irMin.bc << " bcAbove = " << bc;
781  if (bc < irMin.bc) {
782  irMin.orbit++;
783  }
784  irMin.bc = bc;
785  bc = mClosestBunchBelow[irMax.bc];
786  LOG(debug) << "irMax.bc = " << irMax.bc << " bcBelow = " << bc;
787  if (bc > irMax.bc) {
788  if (irMax.orbit == 0) {
789  return false;
790  }
791  irMax.orbit--;
792  }
793  irMax.bc = bc;
794  vtx.setIRMin(irMin);
795  vtx.setIRMax(irMax);
796  if (irMin > irMax) {
797  LOG(debug) << "Reject VTX " << vtx.asString() << " trange = " << rangeT;
798  }
799  return irMax >= irMin;
800 }

802 //___________________________________________________________________
803 int PVertexer::dbscan_RangeQuery(int id, std::vector<int>& cand, std::vector<int>& status)
804 {
805  // find neighbours for dbscan cluster core point candidate
806  // Since we use asymmetric distance definition, is it bit more complex than simple search within chi2 proximity
807  int nFound = 0;
808  const auto& tI = mTracksPool[id];
809  int ntr = mTracksPool.size();
810 
811  auto procPnt = [this, &tI, &status, &cand, &nFound, id](int idN) {
812  const auto& tL = this->mTracksPool[idN];
813  if (std::abs(tI.timeEst.getTimeStamp() - tL.timeEst.getTimeStamp()) > this->mPVParams->dbscanDeltaT) {
814  return -1;
815  }
816  auto statN = status[idN], stat = status[id];
817  if (statN >= 0 && (stat < 0 || (stat >= 0 && statN != stat))) { // do not consider as a neighbour if already added to other cluster
818  return 0;
819  }
820  auto dist2 = tL.getDist2(tI);
821  if (dist2 < this->mPVParams->dbscanMaxDist2) {
822  nFound++;
823  if (statN < 0 && statN > DBS_INCHECK) { // no point in adding for check already assigned point, or which is already in the list (i.e. < INCHECK)
824  cand.push_back(idN);
825  status[idN] += DBS_INCHECK; // flag that the track is in the candidates list (i.e. DBS_UDEF-10 = -12 or DPB_NOISE-10 = -11).
826  }
827  }
828  return 1;
829  };
830  int idL = id;
831  while (--idL >= 0) { // index in time decreasing direction
832  if (procPnt(idL) < 0) {
833  break;
834  }
835  }
836  int idU = id;
837  while (++idU < ntr) { // index in time increasing direction
838  if (procPnt(idU) < 0) {
839  break;
840  }
841  }
842  return nFound;
843 }

845 //_____________________________________________________
846 void PVertexer::dbscan_clusterize()
847 {
848  mTimeZClusters.clear();
849  int ntr = mTracksPool.size();
850  std::vector<int> status(ntr, DBS_UNDEF);
851  TStopwatch timer;
852  int clID = -1;
853 
854  std::vector<int> nbVec;
855  for (int it = 0; it < ntr; it++) {
856  if (status[it] != DBS_UNDEF) {
857  continue;
858  }
859  nbVec.clear();
860  auto nnb0 = dbscan_RangeQuery(it, nbVec, status);
861  int minNeighbours = mPVParams->minTracksPerVtx - 1;
862  if (nnb0 < minNeighbours) {
863  status[it] = DBS_NOISE; // noise
864  continue;
865  }
866  if (nnb0 > minNeighbours) {
867  minNeighbours = std::max(minNeighbours, int(nnb0 * mPVParams->dbscanAdaptCoef));
868  }
869  status[it] = ++clID;
870  auto& clusVec = mTimeZClusters.emplace_back().trackIDs; // new cluster
871  clusVec.push_back(it);
872 
873  for (int j = 0; j < nnb0; j++) {
874  int jt = nbVec[j];
875  auto statjt = status[jt];
876  if (statjt >= 0) {
877  LOG(error) << "assigned track " << jt << " with status " << statjt << " head is " << it << " clID= " << clID;
878  continue;
879  }
880  status[jt] = clID;
881  clusVec.push_back(jt);
882  if (statjt == DBS_NOISE + DBS_INCHECK) { // was border point, no check for being core point is needed
883  continue;
884  }
885  int ncurr = nbVec.size();
886  if (clusVec.size() > minNeighbours) {
887  minNeighbours = std::max(minNeighbours, int(clusVec.size() * mPVParams->dbscanAdaptCoef));
888  }
889  auto nnb1 = dbscan_RangeQuery(jt, nbVec, status);
890  if (nnb1 < minNeighbours) {
891  for (unsigned k = ncurr; k < nbVec.size(); k++) {
892  if (status[nbVec[k]] < DBS_INCHECK) {
893  status[nbVec[k]] -= DBS_INCHECK; // remove from checks
894  }
895  }
896  nbVec.resize(ncurr); // not a core point, reset the seeds pool to the state before RangeQuery
897  } else {
898  nnb0 = nbVec.size(); // core point, its neighbours need to be checked
899  }
900  }
901  }
902 
903  for (auto& clus : mTimeZClusters) {
904  if (clus.trackIDs.size() < mPVParams->minTracksPerVtx) {
905  clus.trackIDs.clear();
906  continue;
907  }
908  float tMean = 0;
909  for (const auto tid : clus.trackIDs) {
910  tMean += mTracksPool[tid].timeEst.getTimeStamp();
911  }
912  clus.timeEst.setTimeStamp(tMean / clus.trackIDs.size());
913  }
914  timer.Stop();
915  LOG(info) << "Found " << mTimeZClusters.size() << " seeding clusters from DBSCAN in " << timer.CpuTime() << " CPU s";
916 }
