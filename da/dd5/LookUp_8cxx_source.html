41 int LookUp::groupFinder(int nRow, int nCol)
42 {
43  int row_index = nRow / TopologyDictionary::RowClassSpan;
44  if (nRow % TopologyDictionary::RowClassSpan == 0) {
45  row_index--;
46  }
47  int col_index = nCol / TopologyDictionary::ColClassSpan;
48  if (nCol % TopologyDictionary::RowClassSpan == 0) {
49  col_index--;
50  }
51  int grNum = -1;
52  if (row_index > TopologyDictionary::MaxNumberOfRowClasses || col_index > TopologyDictionary::MaxNumberOfColClasses) {
53  grNum = TopologyDictionary::NumberOfRareGroups - 1;
54  } else {
55  grNum = row_index * TopologyDictionary::MaxNumberOfColClasses + col_index;
56  }
57  return grNum;
58 }

60 int LookUp::findGroupID(int nRow, int nCol, const unsigned char patt[ClusterPattern::MaxPatternBytes]) const
61 {
62  int nBits = nRow * nCol;
63  // Small topology
64  if (nBits < 9) {
65  int ID = mDictionary.mSmallTopologiesLUT[(nCol - 1) * 255 + (int)patt[0]];
66  if (ID >= 0) {
67  return ID;
68  } else { //small rare topology (inside groups)
69  int index = groupFinder(nRow, nCol);
70  auto res = mDictionary.mGroupMap.find(index);
71  return res == mDictionary.mGroupMap.end() ? -1 : res->second;
72  }
73  }
74  // Big topology
75  unsigned long hash = ClusterTopology::getCompleteHash(nRow, nCol, patt);
76  auto ret = mDictionary.mCommonMap.find(hash);
77  if (ret != mDictionary.mCommonMap.end()) {
78  return ret->second;
79  } else { // Big rare topology (inside groups)
80  int index = groupFinder(nRow, nCol);
81  auto res = mDictionary.mGroupMap.find(index);
82  return res == mDictionary.mGroupMap.end() ? -1 : res->second;
83  }
84 }
