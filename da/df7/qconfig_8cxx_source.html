244  template <typename... Args>

246  {
247  auto settings = qConfigGetSettings(args...);
248  const bool boolType = optionType != 1 && std::is_same<T, bool>::value;
249  const char* arguments = settings.doSet ? " (" : (settings.doDefault || optionType == 1 || boolType) ? " [arg] (" : optionType == 2 ? " [...] (" : " arg (";
250  char argBuffer[4] = {0};
251  unsigned int argBufferPos = 0;
252  if (optnameshort && preoptshort) {
253  argBuffer[argBufferPos++] = '-';
254  }
255  if (optnameshort) {
256  argBuffer[argBufferPos++] = preoptshort == 0 ? '-' : preoptshort;
257  argBuffer[argBufferPos++] = optnameshort;
258  }
259  std::cout << "\t" << name << ": " << argBuffer << (optnameshort == 0 ? "" : arguments) << "--";
260  if (optname[0] == 0) {
261  std::cout << preopt << name;
262  } else {
263  std::cout << optname;
264  }
265  std::cout << (optnameshort == 0 ? arguments : ", ") << "type: " << type;
266  if (optionType == 0) {
267  std::cout << ", default: " << def;
268  }
269  if (optionType == 1) {
270  std::cout << ", sets " << name << " to " << def;
271  }
272  if (settings.checkMin) {
273  std::cout << ", minimum: " << settings.min;
274  }
275  if (settings.checkMax) {
276  std::cout << ", maximum: " << settings.max;
277  }
278  std::cout << ")\n\t\t" << help << ".\n";
279  if (settings.doSet) {
280  std::cout << "\t\tSets " << name << " to " << settings.set << ".\n";
281  } else if (settings.doDefault) {
282  std::cout << "\t\tIf no argument is supplied, " << name << " is set to " << settings.set << ".\n";
283  } else if (boolType) {
284  std::cout << "\t\tIf no argument is supplied, " << name << " is set to true.\n";
285  }
286  if (optionType == 2) {
287  std::cout << "\t\tCan be set multiple times, accepts multiple arguments.\n";
288  }
289  std::cout << "\n";
