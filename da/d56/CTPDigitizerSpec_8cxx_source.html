48  //const bool withQED = context->isQEDProvided();
49  //auto& timesview = context->getEventRecords(withQED);
50  // read ctp inputs from input
51  auto ft0inputs = pc.inputs().get<gsl::span<o2::ft0::DetTrigInput>>("ft0");
52  auto fv0inputs = pc.inputs().get<gsl::span<o2::fv0::DetTrigInput>>("fv0");
53 
54  std::vector<o2::ctp::CTPInputDigit> finputs;
55  TStopwatch timer;
56  timer.Start();
57  LOG(info) << "CALLING CTP DIGITIZATION";
58  // Input order: T0, V0, ... but O need also poisition of inputs DETInputs
59  for (const auto& inp : ft0inputs) {
60  finputs.emplace_back(CTPInputDigit{inp.mIntRecord, inp.mInputs, o2::detectors::DetID::FT0});
61  }
62  for (const auto& inp : fv0inputs) {
63  finputs.emplace_back(CTPInputDigit{inp.mIntRecord, inp.mInputs, o2::detectors::DetID::FV0});
64  }
65  gsl::span<CTPInputDigit> ginputs(finputs);
66  auto digits = mDigitizer.process(ginputs);
67  // send out to next stage
68  LOG(info) << "CTP DIGITS being sent.";
69  pc.outputs().snapshot(Output{"CTP", "DIGITS", 0, Lifetime::Timeframe}, digits);
70  LOG(info) << "CTP PRESENT being sent.";
71  pc.outputs().snapshot(Output{"CTP", "ROMode", 0, Lifetime::Timeframe}, mROMode);
72  timer.Stop();
73  LOG(info) << "CTP Digitization took " << timer.CpuTime() << "s";
74  }
75
