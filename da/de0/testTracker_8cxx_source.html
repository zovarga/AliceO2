64 TrackClusters getTrackClusters(const Track& track)
65 {
66  TrackClusters trCl;
67  trCl.track = track;
68  trCl.nFiredChambers = 0;
69  Mapping::MpStripIndex stripIndex;
70  MpArea area;
71  Cluster cl;
72  for (int ich = 0; ich < 4; ++ich) {
73  auto hits = helper.hitFinder.getLocalPositions(track, ich);
74  bool isFired = false;
75  for (auto& hit : hits) {
76  int deId = hit.deId;
77  float xPos = hit.xCoor;
78  float yPos = hit.yCoor;
79  stripIndex = helper.mapping.stripByPosition(xPos, yPos, 0, deId, false);
80  if (!stripIndex.isValid()) {
81  continue;
82  }
83  cl.deId = deId;
84  area = helper.mapping.stripByLocation(stripIndex.strip, 0, stripIndex.line, stripIndex.column, deId);
85  cl.yCoor = area.getCenterY();
86  cl.yErr = area.getHalfSizeY() / std::sqrt(3.);
87  stripIndex = helper.mapping.stripByPosition(xPos, yPos, 1, deId, false);
88  area = helper.mapping.stripByLocation(stripIndex.strip, 1, stripIndex.line, stripIndex.column, deId);
89  cl.xCoor = area.getCenterX();
90  cl.xErr = area.getHalfSizeX() / std::sqrt(3.);
91  trCl.clusters.push_back(cl);
92  isFired = true;
93  } // loop on fired pos
94  if (isFired) {
95  ++(trCl.nFiredChambers);
96  }
97  }
98  return trCl;
99 }

121 BOOST_DATA_TEST_CASE(TestMultipleTracks, boost::unit_test::data::xrange(1, 9), nTracksPerEvent)
122 {
123  float chi2Cut = helper.tracker.getSigmaCut() * helper.tracker.getSigmaCut();
124  int nTotFakes = 0, nTotReconstructible = 0;
125  for (int ievt = 0; ievt < 1000; ++ievt) {
126  std::vector<TrackClusters> trackClusters = getTrackClusters(nTracksPerEvent);
127  std::vector<Cluster> clusters;
128 
129  // Fill string for debugging
130  std::stringstream ss;
131  ss << "\n";
132  int itr = -1;
133  for (auto& trCl : trackClusters) {
134  ++itr;
135  ss << "Track " << itr << ": " << trCl.track << "\n";
136  for (auto& cl : trCl.clusters) {
137  clusters.push_back(cl);
138  ss << " deId " << (int)cl.deId << " pos: (" << cl.xCoor << ", " << cl.yCoor << ")";
139  }
140  ss << "\n";
141  }
142 
143  // Run tracker algorithm
144  helper.tracker.process(clusters);
145 
146  // Further strings for debugging
147  ss << " Reconstructed tracks:\n";
148  for (size_t ireco = 0; ireco < helper.tracker.getTracks().size(); ++ireco) {
149  ss << " " << helper.tracker.getTracks()[ireco] << "\n";
150  }
151 
152  // Check that all reconstructible tracks are reconstructed
153  size_t nReconstructible = 0;
154  itr = -1;
155  for (auto& trCl : trackClusters) {
156  ++itr;
157  bool isReco = false;
158  for (size_t ireco = 0; ireco < helper.tracker.getTracks().size(); ++ireco) {
159  if (helper.tracker.getTracks()[ireco].isCompatible(trCl.track, chi2Cut)) {
160  isReco = true;
161  break;
162  }
163  }
164  bool isReconstructible = trCl.isReconstructible();
165  // If the number of tracks is small, we can check that:
166  // 1) all reconstructible tracks are reconstructed
167  // 2) all non-reconstructible tracks are not
168  // Case 2), however, is not always valid when we have many tracks,
169  // since we can combine clusters from different tracks to build fakes
170  bool testTrack = (isReconstructible || nTracksPerEvent < 4);
171  if (testTrack) {
172  BOOST_TEST(isReco == isReconstructible, ss.str() << " track " << itr << " reco " << isReco
173  << " != reconstructible " << isReconstructible);
174  }
175  if (isReconstructible) {
176  ++nReconstructible;
177  }
178  } // loop on input tracks
179  nTotReconstructible += nReconstructible;
180  int nFakes = helper.tracker.getTracks().size() - nReconstructible;
181  if (nFakes > 0) {
182  ++nTotFakes;
183  }
184  } // loop on events
185  // To show the following message, run the test with: --log_level=message
186  BOOST_TEST_MESSAGE("Fraction of fake tracks: " << (double)nTotFakes / (double)nTotReconstructible);
187 }
