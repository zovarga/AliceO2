41 int RawReaderZDC::processWord(const uint32_t* word)
42 {
43  if (word == nullptr) {
44  LOG(error) << "NULL pointer";
45  return 1;
46  }
47  if ((word[0] & 0x3) == Id_w0) {
48  for (int32_t iw = 0; iw < NWPerGBTW; iw++) {
49  mCh.w[0][iw] = word[iw];
50  }
51  } else if ((word[0] & 0x3) == Id_w1) {
52  if (mCh.f.fixed_0 == Id_w0) {
53  for (int32_t iw = 0; iw < NWPerGBTW; iw++) {
54  mCh.w[1][iw] = word[iw];
55  }
56  } else {
57  LOG(error) << "Wrong word sequence";
58  mCh.f.fixed_0 = Id_wn;
59  mCh.f.fixed_1 = Id_wn;
60  mCh.f.fixed_2 = Id_wn;
61  }
62  } else if ((word[0] & 0x3) == Id_w2) {
63  if (mCh.f.fixed_0 == Id_w0 && mCh.f.fixed_1 == Id_w1) {
64  for (int32_t iw = 0; iw < NWPerGBTW; iw++) {
65  mCh.w[2][iw] = word[iw];
66  }
67  process(mCh);
68  } else {
69  LOG(error) << "Wrong word sequence";
70  }
71  mCh.f.fixed_0 = Id_wn;
72  mCh.f.fixed_1 = Id_wn;
73  mCh.f.fixed_2 = Id_wn;
74  } else {
75  // Word not present in payload
76  LOG(fatal) << "Event format error";
77  return 1;
78  }
79  return 0;
80 }

96 int RawReaderZDC::getDigits(std::vector<BCData>& digitsBC, std::vector<ChannelData>& digitsCh, std::vector<OrbitData>& orbitData)
97 {
98  if (mModuleConfig == nullptr) {
99  LOG(fatal) << "Missing ModuleConfig";
100  return 0;
101  }
102  int bcCounter = mMapData.size();
103  LOG(info) << "Processing #bc " << bcCounter;
104  for (auto& [ir, ev] : mMapData) {
105  // TODO: Error check
106  // Pedestal data
107  if (ir.bc == 3563) {
108  auto& pdata = orbitData.emplace_back();
109  pdata.ir = ir;
110  for (int32_t im = 0; im < NModules; im++) {
111  for (int32_t ic = 0; ic < NChPerModule; ic++) {
112  if (ev.data[im][ic].f.fixed_0 == Id_w0 && ev.data[im][ic].f.fixed_1 == Id_w1 && ev.data[im][ic].f.fixed_2 == Id_w2) {
113  // Identify connected channel
114  auto id = mModuleConfig->modules[im].channelID[ic];
115  int offset = ev.data[im][ic].f.offset - 32768;
116  pdata.data[id] = offset;
117  pdata.scaler[id] = ev.data[im][ic].f.hits;
118  } else if (ev.data[im][ic].f.fixed_0 == 0 && ev.data[im][ic].f.fixed_1 == 0 && ev.data[im][ic].f.fixed_2 == 0) {
119  // Empty channel
120  } else {
121  LOG(error) << "Data format error";
122  }
123  }
124  }
125  }
126  // BC data
127  auto& bcdata = digitsBC.emplace_back();
128  bcdata.ir = ir;
129  // An inconsistent event has as at least one inconsistent module
130  bool inconsistent_event = false;
131  bool filled_event = false;
132  bcdata.ref.setFirstEntry(digitsCh.size());
133  uint32_t ncd = 0;
134  // Channel data
135  for (int32_t im = 0; im < NModules; im++) {
136  ModuleTriggerMapData mt;
137  mt.w = 0;
138  bool filled_module = false;
139  bool inconsistent_module = false;
140  for (int32_t ic = 0; ic < NChPerModule; ic++) {
141  // Check if payload is present for channel
142  if (ev.data[im][ic].f.fixed_0 == Id_w0 && ev.data[im][ic].f.fixed_1 == Id_w1 && ev.data[im][ic].f.fixed_2 == Id_w2) {
143  bcdata.channels |= 0x1 << (NChPerModule * im + ic); // Flag channel as present
144  auto& ch = ev.data[im][ic];
145  uint16_t us[12];
146  us[0] = ch.f.s00;
147  us[1] = ch.f.s01;
148  us[2] = ch.f.s02;
149  us[3] = ch.f.s03;
150  us[4] = ch.f.s04;
151  us[5] = ch.f.s05;
152  us[6] = ch.f.s06;
153  us[7] = ch.f.s07;
154  us[8] = ch.f.s08;
155  us[9] = ch.f.s09;
156  us[10] = ch.f.s10;
157  us[11] = ch.f.s11;
158  // Identify connected channel
159  auto& chd = digitsCh.emplace_back();
160  auto id = mModuleConfig->modules[im].channelID[ic];
161  chd.id = id;
162  for (int32_t is = 0; is < NTimeBinsPerBC; is++) {
163  if (us[is] > ADCMax) {
164  chd.data[is] = us[is] - ADCRange;
165  } else {
166  chd.data[is] = us[is];
167  }
168  }
169  // Trigger bits
170  if (ch.f.Hit) {
171  bcdata.triggers |= (0x1 << (im * NChPerModule + ic));
172  }
173  // TODO: Alice trigger bits
174  // TODO: consistency checks
175  if (filled_event == false) {
176  mt.f.Alice_0 = ch.f.Alice_0;
177  mt.f.Alice_1 = ch.f.Alice_1;
178  mt.f.Alice_2 = ch.f.Alice_2;
179  mt.f.Alice_3 = ch.f.Alice_3;
180  filled_event = true;
181  } else if (mt.f.Alice_0 != ch.f.Alice_0 || mt.f.Alice_1 != ch.f.Alice_1 || mt.f.Alice_2 != ch.f.Alice_2 || mt.f.Alice_3 != ch.f.Alice_3) {
182  inconsistent_event = true;
183  }
184  if (filled_module == false) {
185  mt.f.Auto_m = ch.f.Auto_m;
186  mt.f.Auto_0 = ch.f.Auto_0;
187  mt.f.Auto_1 = ch.f.Auto_1;
188  mt.f.Auto_2 = ch.f.Auto_2;
189  mt.f.Auto_3 = ch.f.Auto_3;
190  filled_module = true;
191  } else if (mt.f.Auto_m != ch.f.Auto_m || mt.f.Auto_0 != ch.f.Auto_0 || mt.f.Auto_1 != ch.f.Auto_1 || mt.f.Auto_2 != ch.f.Auto_2 || mt.f.Auto_3 != ch.f.Auto_3) {
192  inconsistent_module = true;
193  }
194  // Verify trigger condition (if requested)
195  if (mVerifyTrigger) {
196  if ((mt.f.Alice_0 || mt.f.Alice_1) || (mt.f.Alice_2 && (mt.f.Auto_0 || mt.f.Auto_m)) || (mt.f.Alice_3 && mt.f.Auto_0) || (mt.f.Auto_0 || mt.f.Auto_1)) {
197  ncd++;
198  } else {
199  digitsCh.pop_back();
200  }
201  } else {
202  ncd++;
203  }
204  } else if (ev.data[im][ic].f.fixed_0 == 0 && ev.data[im][ic].f.fixed_1 == 0 && ev.data[im][ic].f.fixed_2 == 0) {
205  // Empty channel
206  } else {
207  LOG(error) << "Data format error";
208  }
209  }
210  bcdata.moduleTriggers[im] = mt.w;
211  if (inconsistent_module == true) {
212  inconsistent_event = true;
213  }
214  }
215  if (ncd == 0) {
216  // Remove empty orbits (keep pedestal information)
217  digitsBC.pop_back();
218  } else {
219  bcdata.ref.setEntries(ncd);
220  if (mDumpData) {
221  bcdata.print(mTriggerMask);
222  auto first_entry = bcdata.ref.getFirstEntry();
223  for (Int_t icd = 0; icd < ncd; icd++) {
224  digitsCh[icd + first_entry].print();
225  }
226  }
227  }
228  if (inconsistent_event) {
229  LOG(error) << "Inconsistent event";
230  for (int32_t im = 0; im < NModules; im++) {
231  for (int32_t ic = 0; ic < NChPerModule; ic++) {
232  if (ev.data[im][ic].f.fixed_0 == Id_w0 && ev.data[im][ic].f.fixed_1 == Id_w1 && ev.data[im][ic].f.fixed_2 == Id_w2) {
233  for (int32_t iw = 0; iw < NWPerBc; iw++) {
234  o2::zdc::Digits2Raw::print_gbt_word((const uint32_t*)&ev.data[im][ic].w[iw][0]);
235  }
236  }
237  }
238  }
239  }
240  }
241 
242  mMapData.clear();
243  return bcCounter;
244 }

247 void RawReaderZDC::setTriggerMask()
248 {
249  mTriggerMask = 0;
250  std::string printTriggerMask{};
251 
252  for (int im = 0; im < NModules; im++) {
253  if (im > 0) {
254  printTriggerMask += " ";
255  }
256  printTriggerMask += std::to_string(im);
257  printTriggerMask += "[";
258  for (int ic = 0; ic < NChPerModule; ic++) {
259  if (mModuleConfig->modules[im].trigChannel[ic]) {
260  uint32_t tmask = 0x1 << (im * NChPerModule + ic);
261  mTriggerMask = mTriggerMask | tmask;
262  printTriggerMask += "T";
263  } else {
264  printTriggerMask += " ";
265  }
266  }
267  printTriggerMask += "]";
268  uint32_t mytmask = mTriggerMask >> (im * NChPerModule);
269  LOGF(info, "Trigger mask for module %d 0123 %c%c%c%c", im,
270  mytmask & 0x1 ? 'T' : 'N', mytmask & 0x2 ? 'T' : 'N', mytmask & 0x4 ? 'T' : 'N', mytmask & 0x8 ? 'T' : 'N');
271  }
272  LOGF(info, "trigger_mask=0x%08x %s", mTriggerMask, printTriggerMask.c_str());
273 }
