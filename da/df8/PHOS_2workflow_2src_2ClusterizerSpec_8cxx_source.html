54 void ClusterizerSpec::run(framework::ProcessingContext& ctx)
55 {
56 
57  if (mUseDigits) {
58  LOG(debug) << "PHOSClusterizer - run on digits called";
59 
60  auto dataref = ctx.inputs().get("digits");
61  auto const* phosheader = o2::framework::DataRefUtils::getHeader<o2::phos::PHOSBlockHeader*>(dataref);
62  if (!phosheader->mHasPayload) {
63  mOutputClusters.clear();
64  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CLUSTERS", 0, o2::framework::Lifetime::Timeframe}, mOutputClusters);
65  if (mFullCluOutput) {
66  mOutputCluElements.clear();
67  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CLUELEMENTS", 0, o2::framework::Lifetime::Timeframe}, mOutputCluElements);
68  }
69  mOutputClusterTrigRecs.clear();
70  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CLUSTERTRIGREC", 0, o2::framework::Lifetime::Timeframe}, mOutputClusterTrigRecs);
71  if (mPropagateMC) {
72  mOutputTruthCont.clear();
73  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CLUSTERTRUEMC", 0, o2::framework::Lifetime::Timeframe}, mOutputTruthCont);
74  }
75  return;
76  }
77  // auto digits = ctx.inputs().get<gsl::span<o2::phos::Digit>>("digits");
78  auto digits = ctx.inputs().get<std::vector<o2::phos::Digit>>("digits");
79  auto digitsTR = ctx.inputs().get<std::vector<o2::phos::TriggerRecord>>("digitTriggerRecords");
80  LOG(debug) << "[PHOSClusterizer - run] Received " << digitsTR.size() << " TR, running clusterizer ...";
81  // const o2::dataformats::MCTruthContainer<MCLabel>* truthcont=nullptr;
82  if (mPropagateMC) {
83  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::phos::MCLabel>> truthcont(ctx.inputs().get<o2::dataformats::MCTruthContainer<o2::phos::MCLabel>*>("digitsmctr"));
84  mClusterizer.process(digits, digitsTR, truthcont.get(), mOutputClusters, mOutputCluElements, mOutputClusterTrigRecs, mOutputTruthCont); // Find clusters on digits (pass by ref)
85  } else {
86  mClusterizer.process(digits, digitsTR, nullptr, mOutputClusters, mOutputCluElements, mOutputClusterTrigRecs, mOutputTruthCont); // Find clusters on digits (pass by ref)
87  }
88  } else {
89 
90  LOG(debug) << "PHOSClusterizer - run run on cells called";
91 
92  auto cells = ctx.inputs().get<std::vector<o2::phos::Cell>>("cells");
93  // auto cells = ctx.inputs().get<gsl::span<o2::phos::Cell>>("cells");
94  LOG(debug) << "[PHOSClusterizer - run] Received " << cells.size() << " cells, running clusterizer ...";
95  // auto cellsTR = ctx.inputs().get<gsl::span<o2::phos::TriggerRecord>>("cellTriggerRecords");
96  auto cellsTR = ctx.inputs().get<std::vector<o2::phos::TriggerRecord>>("cellTriggerRecords");
97  if (mPropagateMC) {
98  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::phos::MCLabel>> truthcont(ctx.inputs().get<o2::dataformats::MCTruthContainer<o2::phos::MCLabel>*>("cellsmctr"));
99  // truthmap = ctx.inputs().get<gsl::span<uint>>("cellssmcmap");
100  mClusterizer.processCells(cells, cellsTR, truthcont.get(), mOutputClusters, mOutputCluElements, mOutputClusterTrigRecs, mOutputTruthCont); // Find clusters on digits (pass by ref)
101  } else {
102  mClusterizer.processCells(cells, cellsTR, nullptr, mOutputClusters, mOutputCluElements, mOutputClusterTrigRecs, mOutputTruthCont); // Find clusters on digits (pass by ref)
103  }
104  }
105 
106  if (mPropagateMC) {
107  LOG(debug) << "[PHOSClusterizer - run] Writing " << mOutputClusters.size() << " clusters, " << mOutputClusterTrigRecs.size() << "TR and " << mOutputTruthCont.getIndexedSize() << " Labels";
108  } else {
109  LOG(debug) << "[PHOSClusterizer - run] Writing " << mOutputClusters.size() << " clusters and " << mOutputClusterTrigRecs.size() << " TR";
110  }
111  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CLUSTERS", 0, o2::framework::Lifetime::Timeframe}, mOutputClusters);
112  if (mFullCluOutput) {
113  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CLUELEMENTS", 0, o2::framework::Lifetime::Timeframe}, mOutputCluElements);
114  }
115  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CLUSTERTRIGREC", 0, o2::framework::Lifetime::Timeframe}, mOutputClusterTrigRecs);
116  if (mPropagateMC) {
117  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CLUSTERTRUEMC", 0, o2::framework::Lifetime::Timeframe}, mOutputTruthCont);
118  }
119 }

121 o2::framework::DataProcessorSpec o2::phos::reco_workflow::getClusterizerSpec(bool propagateMC, bool fullClu)
122 {
123  std::vector<o2::framework::InputSpec> inputs;
124  std::vector<o2::framework::OutputSpec> outputs;
125  inputs.emplace_back("digits", o2::header::gDataOriginPHS, "DIGITS", 0, o2::framework::Lifetime::Timeframe);
126  inputs.emplace_back("digitTriggerRecords", o2::header::gDataOriginPHS, "DIGITTRIGREC", 0, o2::framework::Lifetime::Timeframe);
127  if (propagateMC) {
128  inputs.emplace_back("digitsmctr", "PHS", "DIGITSMCTR", 0, o2::framework::Lifetime::Timeframe);
129  }
130  outputs.emplace_back("PHS", "CLUSTERS", 0, o2::framework::Lifetime::Timeframe);
131  if (fullClu) {
132  outputs.emplace_back("PHS", "CLUELEMENTS", 0, o2::framework::Lifetime::Timeframe);
133  }
134  outputs.emplace_back("PHS", "CLUSTERTRIGREC", 0, o2::framework::Lifetime::Timeframe);
135  if (propagateMC) {
136  outputs.emplace_back("PHS", "CLUSTERTRUEMC", 0, o2::framework::Lifetime::Timeframe);
137  }
138 
139  return o2::framework::DataProcessorSpec{"PHOSClusterizerSpec",
140  inputs,
141  outputs,
142  o2::framework::adaptFromTask<o2::phos::reco_workflow::ClusterizerSpec>(propagateMC, true, fullClu)};
143 }
