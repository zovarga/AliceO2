120  boost::property_tree::ptree& pt,
121  boost::program_options::variables_map const& vmap,
122  boost::property_tree::ptree& provenance)

128  continue;
129  }
130  try {

132  case VariantType::Int:
133  pt.put(key, vmap[key].as<int>());
134  break;
135  case VariantType::UInt8:
136  pt.put(key, vmap[key].as<uint8_t>());
137  break;
138  case VariantType::UInt16:

152  break;
153  case VariantType::Double:
154  pt.put(key, vmap[key].as<double>());

170  case VariantType::ArrayDouble:
171  pt.put_child(key, vectorToBranch<double>(stringToVector<double>(vmap[key].as<std::string>())));
172  break;
173  case VariantType::ArrayBool:
174  // pt.put_child(key, vectorToBranch<bool>(stringToVector<bool>(vmap[key].as<std::string>())));
175  break;
176  case VariantType::ArrayString:

191  case VariantType::Unknown:

207 template <typename T>
208 auto replaceLabels(LabeledArray<T>& input, LabeledArray<T>&& spec)
209 {
210  if (!input.getLabelsCols().empty() || !input.getLabelsRows().empty()) {
211  return false;
212  }
213  if (spec.getLabelsCols().empty() == false) {
214  input.replaceLabelsCols(spec.getLabelsCols());
215  }
216  if (spec.getLabelsRows().empty() == false) {
217  input.replaceLabelsRows(spec.getLabelsRows());
218  }
219  return true;
220 }

222 void PropertyTreeHelpers::populate(std::vector<ConfigParamSpec> const& schema,
223  boost::property_tree::ptree& pt,
224  boost::property_tree::ptree const& in,
225  boost::property_tree::ptree& provenance,
226  std::string const& provenanceLabel)
227 {
228  for (auto const& spec : schema) {
229  // strip short version to get the correct key
230  std::string key = spec.name.substr(0, spec.name.find(','));
231  auto it = in.get_child_optional(key);
232  if (!it) {
233  it = in.get_child_optional(boost::property_tree::path(key, '/'));
234  }
235  if (!it) {
236  continue;
237  }
238  try {
239  switch (spec.type) {
240  case VariantType::Int:
241  pt.put(key, (*it).get_value<int>());
242  break;
243  case VariantType::UInt8:
244  pt.put(key, (*it).get_value<uint8_t>());
245  break;
246  case VariantType::UInt16:
247  pt.put(key, (*it).get_value<uint16_t>());
248  break;
249  case VariantType::UInt32:
250  pt.put(key, (*it).get_value<uint32_t>());
251  break;
252  case VariantType::UInt64:
253  pt.put(key, (*it).get_value<uint64_t>());
254  break;
255  case VariantType::Int64:
256  pt.put(key, (*it).get_value<int64_t>());
257  break;
258  case VariantType::Float:
259  pt.put(key, (*it).get_value<float>());
260  break;
261  case VariantType::Double:
262  pt.put(key, (*it).get_value<double>());
263  break;
264  case VariantType::String:
265  pt.put(key, (*it).get_value<std::string>());
266  break;
267  case VariantType::Bool:
268  pt.put(key, (*it).get_value<bool>());
269  break;
270  case VariantType::Dict:
271  case VariantType::ArrayInt:
272  case VariantType::ArrayFloat:
273  case VariantType::ArrayDouble:
274  case VariantType::ArrayBool:
275  case VariantType::ArrayString:
276  case VariantType::Array2DInt:
277  case VariantType::Array2DFloat:
278  case VariantType::Array2DDouble:
279  pt.put_child(key, *it);
280  break;
281  case VariantType::LabeledArrayInt: {
282  auto v = labeledArrayFromBranch<int>(it.value());
283  if (!replaceLabels(v, spec.defaultValue.get<LabeledArray<int>>())) {
284  pt.put_child(key, *it);
285  } else {
286  pt.put_child(key, labeledArrayToBranch(std::move(v)));
287  }
288  }; break;
289  case VariantType::LabeledArrayFloat: {
290  auto v = labeledArrayFromBranch<float>(it.value());
291  if (!replaceLabels(v, spec.defaultValue.get<LabeledArray<float>>())) {
292  pt.put_child(key, *it);
293  } else {
294  pt.put_child(key, labeledArrayToBranch(std::move(v)));
295  }
296  }; break;
297  case VariantType::LabeledArrayDouble: {
298  auto v = labeledArrayFromBranch<double>(it.value());
299  if (!replaceLabels(v, spec.defaultValue.get<LabeledArray<double>>())) {
300  pt.put_child(key, *it);
301  } else {
302  pt.put_child(key, labeledArrayToBranch(std::move(v)));
303  }
304  }; break;
305  case VariantType::Unknown:
306  case VariantType::Empty:
307  default:
308  throw std::runtime_error("Unknown variant type");
309  }
310  provenance.put(key, provenanceLabel);
311  } catch (std::runtime_error& re) {
312  throw;
313  } catch (std::exception& e) {
314  throw std::invalid_argument(std::string("missing option: ") + key + " (" + e.what() + ")");
315  } catch (...) {
316  throw std::invalid_argument(std::string("missing option: ") + key);
317  }
318  }
319 }

323 void traverseRecursive(const boost::property_tree::ptree& parent,
324  const boost::property_tree::ptree::path_type& childPath,
325  const boost::property_tree::ptree& child,
326  PropertyTreeHelpers::WalkerFunction& method)
327 {
328  using boost::property_tree::ptree;
329 
330  method(parent, childPath, child);
331  for (ptree::const_iterator it = child.begin(); it != child.end(); ++it) {
332  ptree::path_type curPath = childPath / ptree::path_type(it->first);
333  traverseRecursive(parent, curPath, it->second, method);
334  }
335 }
