30 void ClusterizerSpec::run(framework::ProcessingContext& ctx)
31 {
32  LOG(info) << "Starting ClusterizerSpec::run() ";
33  LOG(debug) << "CPVClusterizer - run on digits called";
34 
35  auto digits = ctx.inputs().get<std::vector<Digit>>("digits");
36 
37  if (!digits.size()) { // nothing to process
38  LOG(info) << "ClusterizerSpec::run() : no digits; moving on";
39  //ctx.services().get<o2::framework::ControlService>().readyToQuit(framework::QuitRequest::Me);
40  mOutputClusters.clear();
41  ctx.outputs().snapshot(o2::framework::Output{"CPV", "CLUSTERS", 0, o2::framework::Lifetime::Timeframe}, mOutputClusters);
42  mOutputClusterTrigRecs.clear();
43  ctx.outputs().snapshot(o2::framework::Output{"CPV", "CLUSTERTRIGRECS", 0, o2::framework::Lifetime::Timeframe}, mOutputClusterTrigRecs);
44  if (mPropagateMC) {
45  mOutputTruthCont.clear();
46  ctx.outputs().snapshot(o2::framework::Output{"CPV", "CLUSTERTRUEMC", 0, o2::framework::Lifetime::Timeframe}, mOutputTruthCont);
47  }
48  return;
49  }
50  auto digitsTR = ctx.inputs().get<std::vector<o2::cpv::TriggerRecord>>("digitTriggerRecords");
51 
52  //const o2::dataformats::MCTruthContainer<MCCompLabel>* truthcont = nullptr;
53  // DO NOT TRY TO USE const pointer for MCTruthContainer, it is somehow spoiling whole array
54  if (mPropagateMC) {
55  auto truthcont = ctx.inputs().get<o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("digitsmctr");
56  mClusterizer.process(digits, digitsTR, truthcont.get(), &mOutputClusters, &mOutputClusterTrigRecs, &mOutputTruthCont); // Find clusters with MC Truth
57  } else {
58  mClusterizer.process(digits, digitsTR, nullptr, &mOutputClusters, &mOutputClusterTrigRecs, &mOutputTruthCont); // Find clusters without MC Truth
59  }
60 
61  LOG(debug) << "CPVClusterizer::run() : Received " << digitsTR.size() << " TR, calling clusterizer ...";
62 
63  ctx.outputs().snapshot(o2::framework::Output{"CPV", "CLUSTERS", 0, o2::framework::Lifetime::Timeframe}, mOutputClusters);
64  ctx.outputs().snapshot(o2::framework::Output{"CPV", "CLUSTERTRIGRECS", 0, o2::framework::Lifetime::Timeframe}, mOutputClusterTrigRecs);
65  if (mPropagateMC) {
66  ctx.outputs().snapshot(o2::framework::Output{"CPV", "CLUSTERTRUEMC", 0, o2::framework::Lifetime::Timeframe}, mOutputTruthCont);
67  }
68  LOG(info) << "Finished, wrote " << mOutputClusters.size() << " clusters, " << mOutputClusterTrigRecs.size() << "TR and " << mOutputTruthCont.getIndexedSize() << " Labels";
69 }

84 
85  return o2::framework::DataProcessorSpec{"CPVClusterizerSpec",
