42 void Utils::addInteractionBC(int bc, bool fromCollisonCotext)
43 {
44  if (fromCollisonCotext) { // align to TOF
45  if (bc + Geo::LATENCY_ADJ_LHC_IN_BC < 0) {
46  mFillScheme.push_back(bc + Geo::LATENCY_ADJ_LHC_IN_BC + Geo::BC_IN_ORBIT);
47  } else if (bc + Geo::LATENCY_ADJ_LHC_IN_BC >= Geo::BC_IN_ORBIT) {
48  mFillScheme.push_back(bc + Geo::LATENCY_ADJ_LHC_IN_BC - Geo::BC_IN_ORBIT);
49  } else {
50  mFillScheme.push_back(bc + Geo::LATENCY_ADJ_LHC_IN_BC);
51  }
52  } else {
53  mFillScheme.push_back(bc);
54  }
55 }

74 void Utils::computeLHCphase()
75 {
76  static std::vector<o2::dataformats::CalibInfoTOF> tracks;
77  tracks.clear();
78  for (int i = 0; i < NTRACKS_REQUESTED; i++) {
79  tracks.push_back(mCalibTracks[i]);
80  }
81 
82  auto evtime = evTimeMaker<std::vector<o2::dataformats::CalibInfoTOF>, o2::dataformats::CalibInfoTOF, filterCalib<o2::dataformats::CalibInfoTOF>>(tracks, 6.0f, true);
83 
84  if (evtime.eventTimeError < 100) { // udpate LHCphase
85  mPhases[mIsample] = evtime.eventTime;
86  mIsample = (mIsample + 1) % 100;
87  if (mNsample < 100) {
88  mNsample++;
89  }
90  }
91 
92  mLHCPhase = 0;
93  for (int i = 0; i < mNsample; i++) {
94  mLHCPhase += mPhases[i];
95  }
96  mLHCPhase /= mNsample;
97 }

112 double Utils::subtractInteractionBC(double time, bool subLatency)
113 {
114  static const int deltalat = o2::tof::Geo::BC_IN_ORBIT - o2::tof::Geo::LATENCYWINDOW_IN_BC;
115  int bc = int(time * o2::tof::Geo::BC_TIME_INPS_INV + 0.2);
116 
117  if (subLatency) {
118  if (bc >= o2::tof::Geo::LATENCYWINDOW_IN_BC) {
119  bc -= o2::tof::Geo::LATENCYWINDOW_IN_BC;
120  time -= o2::tof::Geo::LATENCYWINDOW_IN_BC * o2::tof::Geo::BC_TIME_INPS;
121  } else {
122  bc += deltalat;
123  time += deltalat * o2::tof::Geo::BC_TIME_INPS;
124  }
125  }
126 
127  int bcOrbit = bc % o2::constants::lhc::LHCMaxBunches;
128 
129  int dbc = o2::constants::lhc::LHCMaxBunches, bcc = bc;
130  for (int k = 0; k < getNinteractionBC(); k++) { // get bc from fill scheme closest
131  if (abs(bcOrbit - getInteractionBC(k)) < dbc) {
132  bcc = bc - bcOrbit + getInteractionBC(k);
133  dbc = abs(bcOrbit - getInteractionBC(k));
134  }
135  if (abs(bcOrbit - getInteractionBC(k) + o2::constants::lhc::LHCMaxBunches) < dbc) { // in case k is close to the right border (last BC of the orbit)
136  bcc = bc - bcOrbit + getInteractionBC(k) - o2::constants::lhc::LHCMaxBunches;
137  dbc = abs(bcOrbit - getInteractionBC(k) + o2::constants::lhc::LHCMaxBunches);
138  }
139  if (abs(bcOrbit - getInteractionBC(k) - o2::constants::lhc::LHCMaxBunches) < dbc) { // in case k is close to the left border (BC=0)
140  bcc = bc - bcOrbit + getInteractionBC(k) + o2::constants::lhc::LHCMaxBunches;
141  dbc = abs(bcOrbit - getInteractionBC(k) - o2::constants::lhc::LHCMaxBunches);
142  }
143  }
144  time -= o2::tof::Geo::BC_TIME_INPS * bcc;
145 
146  return time;
147 }

186 void Utils::addBC(float toftime, bool subLatency)
187 {
188  if (!mIsInit) {
189  init();
190  mIsInit = true;
191  }
192 
193  if (mNautodet > MAX_NUM_EVENT_AUTODETECT) {
194  if (!hasFillScheme()) { // detect fill scheme
195  int thres = mMaxBC / 2;
196  for (int i = 0; i < o2::constants::lhc::LHCMaxBunches; i++) {
197  if (mBCmult[i] > thres) { // good bunch
198  addInteractionBC(i);
199  }
200  }
201  }
202  return;
203  }
204 
205  // just fill
206  static const int deltalat = o2::tof::Geo::BC_IN_ORBIT - o2::tof::Geo::LATENCYWINDOW_IN_BC;
207  int bc = int(toftime * o2::tof::Geo::BC_TIME_INPS_INV + 0.2) % o2::constants::lhc::LHCMaxBunches;
208 
209  if (subLatency) {
210  if (bc >= o2::tof::Geo::LATENCYWINDOW_IN_BC) {
211  bc -= o2::tof::Geo::LATENCYWINDOW_IN_BC;
212  toftime -= o2::tof::Geo::LATENCYWINDOW_IN_BC * o2::tof::Geo::BC_TIME_INPS;
213  } else {
214  bc += deltalat;
215  toftime += deltalat * o2::tof::Geo::BC_TIME_INPS;
216  }
217  }
218 
219  mBCmult[bc]++;
220 
221  if (mBCmult[bc] > mMaxBC) {
222  mMaxBC = mBCmult[bc];
223  }
224  mNautodet++;
225 }
