168 GPUdi() bool GlobalTrackID::includesDet(DetID id, GlobalTrackID::mask_t srcm)
169 {
170  for (int i = 0; i < NSources; i++) {
171  if (includesSource(i, srcm) && (getSourceDetectorsMask(i) & id.getMask()).any()) {
172  return true;
173  }
174  }
175  return false;
176 }
