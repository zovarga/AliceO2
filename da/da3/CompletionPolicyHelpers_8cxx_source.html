28 CompletionPolicy CompletionPolicyHelpers::defineByNameOrigin(std::string const& name, std::string const& origin, CompletionPolicy::CompletionOp op)
29 {
30  auto matcher = [name](DeviceSpec const& device) -> bool {
31  return std::regex_match(device.name.begin(), device.name.end(), std::regex(name));
32  };
33 
34  auto originReceived = std::make_shared<std::vector<uint64_t>>();
35 
36  auto callback = [originReceived, origin, op](InputSpan const& inputRefs) -> CompletionPolicy::CompletionOp {
37  // update list of the start times of inputs with origin @origin
38  for (auto& ref : inputRefs) {
39  if (ref.header != nullptr) {
40  auto header = CompletionPolicyHelpers::getHeader<o2::header::DataHeader>(ref);
41  if (header->dataOrigin.str == origin) {
42  auto startTime = DataRefUtils::getHeader<DataProcessingHeader*>(ref)->startTime;
43  auto it = std::find(originReceived->begin(), originReceived->end(), startTime);
44  if (it == originReceived->end()) {
45  originReceived->emplace_back(startTime);
46  }
47  }
48  }
49  }
50 
51  // find out if all inputs which are not of origin @origin have a corresponding entry in originReceived
52  // if one is missing then we have to wait
53  for (auto& ref : inputRefs) {
54  if (ref.header != nullptr) {
55  auto header = CompletionPolicyHelpers::getHeader<o2::header::DataHeader>(ref);
56  if (header->dataOrigin.str != origin) {
57  auto startTime = DataRefUtils::getHeader<DataProcessingHeader*>(ref)->startTime;
58  auto it = std::find(originReceived->begin(), originReceived->end(), startTime);
59  if (it == originReceived->end()) {
60  LOGP(info, "Have to wait until message of origin {} with startTime {} has arrived.", origin, startTime);
61  return CompletionPolicy::CompletionOp::Wait;
62  }
63  }
64  }
65  }
66  return op;
67  };
68  return CompletionPolicy{"wait-origin", matcher, callback};
69 
70  O2_BUILTIN_UNREACHABLE();
71 }

73 CompletionPolicy CompletionPolicyHelpers::defineByName(std::string const& name, CompletionPolicy::CompletionOp op)
74 {
75  auto matcher = [name](DeviceSpec const& device) -> bool {
76  return std::regex_match(device.name.begin(), device.name.end(), std::regex(name));
77  };
78  auto callback = [op](InputSpan const&) -> CompletionPolicy::CompletionOp {
79  return op;
80  };
81  switch (op) {
82  case CompletionPolicy::CompletionOp::Consume:
83  return CompletionPolicy{"always-consume", matcher, callback};
84  break;
85  case CompletionPolicy::CompletionOp::ConsumeExisting:
86  return CompletionPolicy{"consume-existing", matcher, callback};
87  break;
88  case CompletionPolicy::CompletionOp::Process:
89  return CompletionPolicy{"always-process", matcher, callback};
90  break;
91  case CompletionPolicy::CompletionOp::Wait:
92  return CompletionPolicy{"always-wait", matcher, callback};
93  break;
94  case CompletionPolicy::CompletionOp::Discard:
95  return CompletionPolicy{"always-discard", matcher, callback};
96  break;
97  }
98  O2_BUILTIN_UNREACHABLE();
99 }

114 CompletionPolicy CompletionPolicyHelpers::consumeExistingWhenAny(const char* name, CompletionPolicy::Matcher matcher)
115 {
116  return CompletionPolicy{
117  name,
118  matcher,
119  [](InputSpan const& inputs, std::vector<InputSpec> const& specs) -> CompletionPolicy::CompletionOp {
120  size_t present = 0;
121  size_t current = 0;
122  size_t withPayload = 0;
123  size_t sporadic = 0;
124  size_t maxSporadic = 0;
125  size_t i = 0;
126  for (auto& input : inputs) {
127  auto& spec = specs[i++];
128  if (spec.lifetime == Lifetime::Sporadic) {
129  maxSporadic++;
130  }
131  if (input.header != nullptr) {
132  present++;
133  if (spec.lifetime == Lifetime::Sporadic) {
134  sporadic++;
135  }
136  }
137  if (input.payload != nullptr) {
138  withPayload++;
139  }
140  current++;
141  }
142  // * In case we have all inputs but the sporadic ones: Consume, since we do not know if the sporadic ones
143  // will ever come.
144  // * In case we have only sporadic inputs: Consume, since we do not know if we already Consumed
145  // the non sporadic ones above.
146  // * In case we do not have payloads: Wait
147  // * In all other cases we consume what is there, but we wait for the non sporadic ones to be complete
148  // (i.e. we wait for present + maxSporadic).
149  if (present - sporadic + maxSporadic == inputs.size()) {
150  return CompletionPolicy::CompletionOp::Consume;
151  } else if (present - sporadic == 0) {
152  return CompletionPolicy::CompletionOp::Consume;
153  } else if (withPayload == 0) {
154  return CompletionPolicy::CompletionOp::Wait;
155  }
156  return CompletionPolicy::CompletionOp::ConsumeExisting;
157  }
158 
159  };
160 }

165  for (auto& input : inputs) {

175 CompletionPolicy CompletionPolicyHelpers::processWhenAny(const char* name, CompletionPolicy::Matcher matcher)
176 {
177  auto callback = [](InputSpan const& inputs) -> CompletionPolicy::CompletionOp {
178  size_t present = 0;
179  for (auto& input : inputs) {
180  if (input.header != nullptr) {
181  present++;
182  }
183  }
184  if (present == inputs.size()) {
185  return CompletionPolicy::CompletionOp::Consume;
186  } else if (present == 0) {
187  return CompletionPolicy::CompletionOp::Wait;
188  }
189  return CompletionPolicy::CompletionOp::Process;
190  };
191  return CompletionPolicy{name, matcher, callback};
192 }
