108 inline void SegmentationSuperAlpide::localToDetectorUnchecked(float xRow, float zCol, int& iRow, int& iCol)
109 {
110  // convert to row/col w/o over/underflow check
111  xRow = 0.5 * (ActiveMatrixSizeRows - PassiveEdgeTop + PassiveEdgeReadOut) - xRow; // coordinate wrt top edge of Active matrix
112  zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt left edge of Active matrix
113  iRow = int(xRow / PitchRow);
114  iCol = int(zCol / PitchCol);
115  if (xRow < 0) {
116  iRow -= 1;
117  }
118  if (zCol < 0) {
119  iCol -= 1;
120  }
121 }

123 inline bool SegmentationSuperAlpide::localToDetector(float xRow, float zCol, int& iRow, int& iCol)
124 {
125  // convert to row/col
126  xRow = 0.5 * (ActiveMatrixSizeRows - PassiveEdgeTop + PassiveEdgeReadOut) - xRow; // coordinate wrt left edge of Active matrix
127  zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt bottom edge of Active matrix
128  if (xRow < 0 || xRow >= ActiveMatrixSizeRows || zCol < 0 || zCol >= ActiveMatrixSizeCols) {
129  iRow = iCol = -1;
130  return false;
131  }
132  iRow = int(xRow / PitchRow);
133  iCol = int(zCol / PitchCol);
134  return true;
135 }
