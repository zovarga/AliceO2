35 void ServiceRegistry::registerService(hash_type typeHash, void* service, ServiceKind kind, uint64_t threadId, const char* name) const
36 {
37  hash_type id = typeHash & MAX_SERVICES_MASK;
38  hash_type threadHashId = (typeHash ^ threadId) & MAX_SERVICES_MASK;
39  // If kind is not stream, there is only one copy of our service.
40  // So we look if it is already registered and reused it if it is.
41  // If not, we register it as thread id 0 and as the passed one.
42  if (kind != ServiceKind::Stream && threadId != 0) {
43  void* oldService = this->get(typeHash, 0, kind);
44  if (oldService == nullptr) {
45  registerService(typeHash, service, kind, 0);
46  } else {
47  service = oldService;
48  }
49  }
50  for (uint8_t i = 0; i < MAX_DISTANCE; ++i) {
51  // If the service slot was not taken, take it atomically
52  bool expected = false;
53  if (mServicesBooked[i + threadHashId].compare_exchange_strong(expected, true,
54  std::memory_order_seq_cst)) {
55  mServicesValue[i + threadHashId] = service;
56  mServicesMeta[i + threadHashId] = ServiceMeta{kind, threadId};
57  mServicesKey[i + threadHashId] = typeHash;
58  std::atomic_thread_fence(std::memory_order_release);
59  return;
60  }
61  }
62  throw std::runtime_error(std::string("Unable to find a spot in the registry for service ") +
63  std::to_string(typeHash) +
64  ". Make sure you use const / non-const correctly.");
65 }

67 void ServiceRegistry::declareService(ServiceSpec const& spec, DeviceState& state, fair::mq::ProgOptions& options)
68 {
69  mSpecs.push_back(spec);
70  // Services which are not stream must have a single instance created upfront.
71  if (spec.kind != ServiceKind::Stream) {
72  ServiceHandle handle = spec.init(*this, state, options);
73  this->registerService(handle.hash, handle.instance, handle.kind, 0, handle.name.c_str());
74  this->bindService(spec, handle.instance);
75  }
76 }

78 void ServiceRegistry::bindService(ServiceSpec const& spec, void* service)
79 {
80  static TracyLockableN(std::mutex, bindMutex, "bind mutex");
81  std::scoped_lock<LockableBase(std::mutex)> lock(bindMutex);
82  if (spec.preProcessing) {
83  mPreProcessingHandles.push_back(ServiceProcessingHandle{spec.preProcessing, service});
84  }
85  if (spec.postProcessing) {
86  mPostProcessingHandles.push_back(ServiceProcessingHandle{spec.postProcessing, service});
87  }
88  if (spec.preDangling) {
89  mPreDanglingHandles.push_back(ServiceDanglingHandle{spec.preDangling, service});
90  }
91  if (spec.postDangling) {
92  mPostDanglingHandles.push_back(ServiceDanglingHandle{spec.postDangling, service});
93  }
94  if (spec.preEOS) {
95  mPreEOSHandles.push_back(ServiceEOSHandle{spec.preEOS, service});
96  }
97  if (spec.postEOS) {
98  mPostEOSHandles.push_back(ServiceEOSHandle{spec.postEOS, service});
99  }
100  if (spec.postDispatching) {
101  mPostDispatchingHandles.push_back(ServiceDispatchingHandle{spec.postDispatching, service});
102  }
103  if (spec.start) {
104  mPreStartHandles.push_back(ServiceStartHandle{spec.start, service});
105  }
106  if (spec.stop) {
107  mPostStopHandles.push_back(ServiceStopHandle{spec.stop, service});
108  }
109  if (spec.exit) {
110  mPreExitHandles.push_back(ServiceExitHandle{spec.exit, service});
111  }
