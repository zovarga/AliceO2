70  _Iter() = default;

105  // BC difference wrt previous if in the same orbit, otherwise the abs.value.
106  // For the very 1st entry return 0 (diff wrt 1st BC in the CTF header)
107  class Iter_bcIncTrig : public _Iter<Iter_bcIncTrig, Trigger, uint16_t>
108  {
109  public:
110  using _Iter<Iter_bcIncTrig, Trigger, uint16_t>::_Iter;
111  value_type operator*() const
112  {
113  if (mIndex) {
114  if (mData[mIndex].getOrbit() == mData[mIndex - 1].getOrbit()) {
115  return mData[mIndex].getBc() - mData[mIndex - 1].getBc();
116  } else {
117  return mData[mIndex].getBc();
118  }
119  }
120  return 0;
121  }
122  };
