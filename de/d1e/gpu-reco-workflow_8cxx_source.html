105 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
106 {
107  WorkflowSpec specs;
108  std::vector<int> tpcSectors(o2::tpc::Sector::MAXSECTOR);
109  std::iota(tpcSectors.begin(), tpcSectors.end(), 0);
110 
111  auto inputType = cfgc.options().get<std::string>("input-type");
112  bool doMC = !cfgc.options().get<bool>("disable-mc");
113 
114  o2::conf::ConfigurableParam::updateFromFile(cfgc.options().get<std::string>("configFile"));
115  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
116  o2::conf::ConfigurableParam::writeINI("o2gpurecoworkflow_configuration.ini");
117 
118  std::vector<ioType> outputTypes, inputTypes;
119  try {
120  outputTypes = o2::RangeTokenizer::tokenize<ioType>(cfgc.options().get<std::string>("output-type"), [](std::string const& token) { return OutputMap.at(token); });
121  inputTypes = o2::RangeTokenizer::tokenize<ioType>(cfgc.options().get<std::string>("input-type"), [](std::string const& token) { return InputMap.at(token); });
122  } catch (std::out_of_range&) {
123  throw std::invalid_argument("invalid input / output type");
124  }
125 
126  auto isEnabled = [](auto& list, ioType type) {
127  return std::find(list.begin(), list.end(), type) != list.end();
128  };
129 
130  o2::gpu::gpuworkflow::Config cfg;
131  cfg.decompressTPC = isEnabled(inputTypes, ioType::CompClustCTF);
132  cfg.decompressTPCFromROOT = isEnabled(inputTypes, ioType::CompClustROOT);
133  cfg.zsDecoder = isEnabled(inputTypes, ioType::ZSRaw);
134  cfg.zsOnTheFly = isEnabled(inputTypes, ioType::ZSRawOTF);
135  cfg.caClusterer = cfg.zsDecoder || cfg.zsOnTheFly || isEnabled(inputTypes, ioType::Digits);
136  cfg.outputTracks = isEnabled(outputTypes, ioType::Tracks);
137  cfg.outputCompClusters = isEnabled(outputTypes, ioType::CompClustROOT);
138  cfg.outputCompClustersFlat = isEnabled(outputTypes, ioType::CompClustCTF);
139  cfg.outputCAClusters = isEnabled(outputTypes, ioType::Clusters);
140  cfg.outputQA = isEnabled(outputTypes, ioType::QA);
141  cfg.outputSharedClusterMap = (cfg.outputCAClusters || cfg.caClusterer || isEnabled(inputTypes, ioType::Clusters)) && cfg.outputTracks && !isEnabled(outputTypes, ioType::NoSharedMap);
142  cfg.processMC = doMC;
143  cfg.sendClustersPerSector = isEnabled(outputTypes, ioType::SendClustersPerSector);
144  cfg.askDISTSTF = !cfgc.options().get<bool>("ignore-dist-stf");
145  cfg.readTRDtracklets = isEnabled(inputTypes, ioType::TRDTracklets);
146  cfg.runTRDTracking = isEnabled(outputTypes, ioType::TRDTracks);
147  specs.emplace_back(o2::gpu::getGPURecoWorkflowSpec(&gPolicyData, cfg, tpcSectors, gTpcSectorMask, "gpu-reconstruction"));
148 
149  if (!cfgc.options().get<bool>("ignore-dist-stf")) {
150  GlobalTrackID::mask_t srcTrk = GlobalTrackID::getSourcesMask("none");
151  GlobalTrackID::mask_t srcCl = GlobalTrackID::getSourcesMask("TPC");
152  o2::globaltracking::InputHelper::addInputSpecs(cfgc, specs, srcCl, srcTrk, srcTrk, doMC);
153  }
154 
155  // configure dpl timer to inject correct firstTFOrbit: start from the 1st orbit of TF containing 1st sampled orbit
156  o2::raw::HBFUtilsInitializer hbfIni(cfgc, specs);
157 
158  return std::move(specs);
159 }
