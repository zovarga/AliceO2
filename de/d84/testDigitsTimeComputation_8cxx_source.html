75 static std::vector<RawDigit> processDigits(const std::vector<RawDigit>& digits, uint32_t tfOrbit, uint32_t tfBunchCrossing)
76 {
77  SampaChannelHandler channelHandler;
78  RdhHandler rdhHandler;
79 
80  bool ds2manu = false;

85  bool useDummyElecMap = false;
86  DataDecoder decoder{channelHandler, rdhHandler, sampaBcOffset, "", "", ds2manu, mDebug, useDummyElecMap};
87 
88  decoder.setFirstOrbitInTF(tfOrbit);
89  decoder.setDigits(digits);
90 
91  for (auto& digit : digits) {
92  auto& info = digit.info;
93  auto chipId = DataDecoder::getChipId(info.solar, info.ds, info.chip);
94  decoder.updateTimeFrameStartRecord(chipId, tfOrbit, tfBunchCrossing);
95  }
96  decoder.computeDigitsTime();
97 
98  auto& digitsOut = decoder.getDigits();
99  return digitsOut;
100 }
