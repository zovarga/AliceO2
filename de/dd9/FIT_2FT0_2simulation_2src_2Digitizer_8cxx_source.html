33 // signal shape function
34 template <typename Float>
35 Float signalForm_i(Float x)
36 {
37  using namespace std;
38  Float const a = -0.45458;
39  Float const b = -0.83344945;
40  return x > Float(0) ? -(exp(b * x) - exp(a * x)) / Float(7.8446501) : Float(0);
41 };
42 
43 // integrated signal shape function
44 inline float signalForm_integral(float x)
45 {
46  using namespace std;
47  double const a = -0.45458;
48  double const b = -0.83344945;
49  if (x < 0) {
50  x = 0;
51  }
52  return -(exp(b * x) / b - exp(a * x) / a) / 7.8446501;
53 };
54 
55 // SIMD version of the integrated signal shape function
56 inline Vc::float_v signalForm_integralVc(Vc::float_v x)
57 {
58  auto const mask = (x >= 0.0f);
59  Vc::float_v arg(0);
60  arg.assign(x, mask); // branchless if
61  Vc::float_v const a(-0.45458f);
62  Vc::float_v const b(-0.83344945f);
63  Vc::float_v result = -(Vc::exp(b * arg) / b - Vc::exp(a * arg) / a) / 7.8446501f;
64  return result;
65 };

68 Digitizer::CFDOutput Digitizer::get_time(const std::vector<float>& times, float deadTime)
69 {
70  assert(std::is_sorted(std::begin(times), std::end(times)));
71 
72  // get a new batch of random values
73  for (float& n : mNoiseSamples) {
74  n = mRndGaus.getNextValue();
75  }
76 
77  // this lambda function evaluates the signal shape at a give time
78  auto value_at = [&](float time) {
79  float val = 0;
80  // (1) sum over individual hits
81  Vc::float_v acc(0);
82  Vc::float_v tableVal(0);
83  const float* tp = times.data();
84  size_t m = times.size() / Vc::float_v::size();
85  for (size_t i = 0; i < m; ++i) {
86  tableVal.load(tp);
87  tp += Vc::float_v::size();
88  Vc::prefetchForOneRead(tp);
89  acc += signalFormVc(time - tableVal);
90  }
91  val += acc.sum();
92  // non-SIMD tail
93  for (size_t i = Vc::float_v::size() * m; i < times.size(); ++i, ++tp) {
94  val += signalForm(time - (*tp));
95  }
96  // (2) add noise
97  // find the right indices into the sinc table
98  int timeIndex = std::lround(time / DigitizationParameters::Instance().mNoisePeriod * mSincTable.size());
99  int timeOffset = timeIndex / mSincTable.size();
100  timeIndex %= mSincTable.size();
101  if (timeOffset >= mNumNoiseSamples) { // this happens when time >= 25 ns
102  timeOffset = mNumNoiseSamples - 1;
103  LOG(debug) << "timeOffset >= mNumNoiseSamples";
104  }
105  if (timeOffset <= -mNumNoiseSamples) { // this happens when time <= -25 ns
106  timeOffset = -mNumNoiseSamples + 1;
107  LOG(debug) << "timeOffset <= -mNumNoiseSamples";
108  }
109  Vc::float_v noiseVal(0);
110  const float* np = mNoiseSamples.data();
111  tp = mSincTable[timeIndex].data() + mNumNoiseSamples - timeOffset;
112  acc = 0.0f;
113  m = mNumNoiseSamples / Vc::float_v::Size;
114  for (size_t i = 0; i < m; ++i) {
115  tableVal.load(tp);
116  tp += Vc::float_v::Size;
117  Vc::prefetchForOneRead(tp);
118  noiseVal.load(np);
119  np += Vc::float_v::Size;
120  Vc::prefetchForOneRead(np);
121  acc += noiseVal * tableVal;
122  }
123  val += acc.sum(); // horizontal sum
124  // non-SIMD tail
125  for (size_t i = Vc::float_v::Size * m; i < mNumNoiseSamples; ++i, ++tp, ++np) {
126  val += (*np) * (*tp);
127  }
128  return val;
129  };
130  auto const min_time = std::max(deadTime, *std::min_element(std::begin(times),
131  std::end(times)));
132  CFDOutput result{std::nullopt, -0.5f * DigitizationParameters::Instance().mBunchWidth};
133  bool is_positive = true;
134 
135  // reset the chache
136  std::fill_n(std::begin(mSignalCache), std::size(mSignalCache), -1.0f);
137  const auto& params = DigitizationParameters::Instance();
138  // we need double precision for time in order to match previous behaviour
139  for (double time = min_time; time < 0.5 * params.mBunchWidth; time += DP::SIGNAL_CACHE_DT) {
140  float const val = value_at(time);
141  int const index = std::lround((time + 0.5 * params.mBunchWidth) / DP::SIGNAL_CACHE_DT);
142  if (index >= 0 && index < mSignalCache.size()) { // save the value for later use
143  mSignalCache[index] = val;
144  }
145  // look up the time-shifted signal value from the past
146  float val_prev = 0.0f;
147  int const index_prev = std::lround((time - params.mCFDShiftPos + 0.5f * params.mBunchWidth) / DP::SIGNAL_CACHE_DT);
148  val_prev = ((index_prev < 0 || index_prev >= mSignalCache.size() || mSignalCache[index_prev] < 0.0f)
149  ? value_at(time - params.mCFDShiftPos) // was not computed before
150  : mSignalCache[index_prev]); // is available in the cache
151  float const cfd_val = 5.0f * val_prev - val;
152  if (std::abs(val) > params.mCFD_trsh && !is_positive && cfd_val > 0.0f) {
153  if (!result.particle) {
154  result.particle = time;
155  }
156  result.deadTime = time + params.mCFDdeadTime;
157  time += params.mCFDdeadTime - DP::SIGNAL_CACHE_DT;
158  is_positive = true;
159  } else {
160  is_positive = cfd_val > 0.0f;
161  }
162  }
163  return result;
164 }

166 double Digitizer::measure_amplitude(const std::vector<float>& times) const
167 {
168  float const from = DigitizationParameters::Instance().mAmpRecordLow;
169  float const to = from + DigitizationParameters::Instance().mAmpRecordUp;
170  // SIMD version has a negligible effect on the total wall time
171  Vc::float_v acc(0);
172  Vc::float_v tv(0);
173  const float* tp = times.data();
174  size_t const m = times.size() / Vc::float_v::Size;
175  for (size_t i = 0; i < m; ++i) {
176  tv.load(tp);
177  tp += Vc::float_v::Size;
178  Vc::prefetchForOneRead(tp);
179  acc += signalForm_integralVc(to - tv) - signalForm_integralVc(from - tv);
180  }
181  float result = acc.sum(); // horizontal sum
182  // non-SIMD tail
183  for (size_t i = Vc::float_v::Size * m; i < times.size(); ++i, ++tp) {
184  result += signalForm_integral(to - (*tp)) - signalForm_integral(from - (*tp));
185  }
186  return result;
187 }

189 void Digitizer::process(const std::vector<o2::ft0::HitType>* hits,
190  std::vector<o2::ft0::Digit>& digitsBC,
191  std::vector<o2::ft0::ChannelData>& digitsCh,
192  std::vector<o2::ft0::DetTrigInput>& digitsTrig,
193  o2::dataformats::MCTruthContainer<o2::ft0::MCLabel>& label)
194 {
195  ;
196  //Calculating signal time, amplitude in mean_time +- time_gate --------------
197  LOG(debug) << " process firstBCinDeque " << firstBCinDeque << " mIntRecord " << mIntRecord;
198  if (firstBCinDeque != mIntRecord) {
199  flush(digitsBC, digitsCh, digitsTrig, label);
200  }
201 
202  Int_t parent = -10;
203  for (auto const& hit : *hits) {
204  if (hit.GetEnergyLoss() > 0) {
205  continue;
206  }
207  const auto& params = DigitizationParameters::Instance();
208  Int_t hit_ch = hit.GetDetectorID();
209  Bool_t is_A_side = (hit_ch < 4 * mGeometry.NCellsA);
210  Float_t time_compensate = is_A_side ? params.mA_side_cable_cmps : params.mC_side_cable_cmps;
211  Double_t hit_time = hit.GetTime() - time_compensate;
212  if (hit_time > 150) {
213  continue; //not collect very slow particles
214  }
215  auto relBC = o2::InteractionRecord{hit_time};
216  if (mCache.size() <= relBC.bc) {
217  mCache.resize(relBC.bc + 1);
218  }
219  mCache[relBC.bc].hits.emplace_back(BCCache::particle{hit_ch, hit_time - relBC.bc2ns()});
220  //charge particles in MCLabel
221  Int_t parentID = hit.GetTrackID();
222  if (parentID != parent) {
223  mCache[relBC.bc].labels.emplace(parentID, mEventID, mSrcID, hit_ch);
224  parent = parentID;
225  }
226  }
227 }

229 void Digitizer::storeBC(BCCache& bc,
230  std::vector<o2::ft0::Digit>& digitsBC,
231  std::vector<o2::ft0::ChannelData>& digitsCh,
232  std::vector<o2::ft0::DetTrigInput>& digitsTrig,
233  o2::dataformats::MCTruthContainer<o2::ft0::MCLabel>& labels)
234 {
235  if (bc.hits.empty()) {
236  return;
237  }
238  int n_hit_A = 0, n_hit_C = 0, mean_time_A = 0, mean_time_C = 0;
239  int summ_ampl_A = 0, summ_ampl_C = 0;
240  int vertex_time;
241  const auto& params = DigitizationParameters::Instance();
242  int first = digitsCh.size(), nStored = 0;
243  auto& particles = bc.hits;
244  std::sort(std::begin(particles), std::end(particles));
245  auto channel_end = particles.begin();
246  std::vector<float> channel_times;
247  for (Int_t ipmt = 0; ipmt < params.mMCPs; ++ipmt) {
248  auto channel_begin = channel_end;
249  channel_end = std::find_if(channel_begin, particles.end(),
250  [ipmt](BCCache::particle const& p) { return p.hit_ch != ipmt; });
251  if (channel_end - channel_begin < params.mAmp_trsh) {
252  continue;
253  }
254  channel_times.resize(channel_end - channel_begin);
255  std::transform(channel_begin, channel_end, channel_times.begin(), [](BCCache::particle const& p) { return p.hit_time; });
256  int chain = (std::rand() % 2) ? 1 : 0;
257  auto cfd = get_time(channel_times, mDeadTimes[ipmt].intrec.bc2ns() -
258  firstBCinDeque.bc2ns() +
259  mDeadTimes[ipmt].deadTime);
260  mDeadTimes[ipmt].intrec = firstBCinDeque;
261  mDeadTimes[ipmt].deadTime = cfd.deadTime;
262 
263  if (!cfd.particle) {
264  continue;
265  }
266  int smeared_time = 1000. * (*cfd.particle - params.mCfdShift) * params.mChannelWidthInverse;
267  bool is_time_in_signal_gate = (smeared_time > -params.mTime_trg_gate && smeared_time < params.mTime_trg_gate);
268  float charge = measure_amplitude(channel_times) * params.mCharge2amp;
269  float amp = is_time_in_signal_gate ? params.mMV_2_Nchannels * charge : 0;
270  if (amp > 4095) {
271  amp = 4095;
272  }
273  LOG(debug) << mEventID << " bc " << firstBCinDeque.bc << " orbit " << firstBCinDeque.orbit << ", ipmt " << ipmt << ", smeared_time " << smeared_time << " nStored " << nStored;
274  digitsCh.emplace_back(ipmt, smeared_time, int(amp), chain);
275  nStored++;
276 
277  // fill triggers
278 
279  Bool_t is_A_side = (ipmt < 4 * mGeometry.NCellsA);
280  if (!is_time_in_signal_gate) {
281  continue;
282  }
283 
284  if (is_A_side) {
285  n_hit_A++;
286  summ_ampl_A += amp;
287  mean_time_A += smeared_time;
288  } else {
289  n_hit_C++;
290  summ_ampl_C += amp;
291  mean_time_C += smeared_time;
292  }
293  }
294  Bool_t is_A, is_C, isVertex, is_Central, is_SemiCentral = 0;
295  is_A = n_hit_A > 0;
296  is_C = n_hit_C > 0;
297  is_Central = summ_ampl_A + summ_ampl_C >= params.mtrg_central_trh;
298  is_SemiCentral = summ_ampl_A + summ_ampl_C >= params.mtrg_semicentral_trh;
299  uint32_t amplA = is_A ? summ_ampl_A * 0.125 : 0; // sum amplitude A side / 8 (hardware)
300  uint32_t amplC = is_C ? summ_ampl_C * 0.125 : 0; // sum amplitude C side / 8 (hardware)
301  int timeA = is_A ? mean_time_A / n_hit_A : 0; // average time A side
302  int timeC = is_C ? mean_time_C / n_hit_C : 0; // average time C side
303  vertex_time = (timeC - timeA) * 0.5;
304  isVertex = is_A && is_C && (vertex_time > -params.mTime_trg_gate && vertex_time < params.mTime_trg_gate);
305  LOG(debug) << " A " << is_A << " timeA " << timeA << " mean_time_A " << mean_time_A << " n_hit_A " << n_hit_A << " C " << is_C << " timeC " << timeC << " mean_time_C " << mean_time_C << " n_hit_C " << n_hit_C << " vertex_time " << vertex_time;
306  Triggers triggers;
307  if (nStored > 0) {
308  triggers.setTriggers(is_A, is_C, isVertex, is_Central, is_SemiCentral, int8_t(n_hit_A), int8_t(n_hit_C),
309  amplA, amplC, timeA, timeC);
310  digitsBC.emplace_back(first, nStored, firstBCinDeque, triggers, mEventID - 1);
311  digitsTrig.emplace_back(firstBCinDeque, is_A, is_C, isVertex, is_Central, is_SemiCentral);
312  size_t const nBC = digitsBC.size();
313  for (auto const& lbl : bc.labels) {
314  labels.addElement(nBC - 1, lbl);
315  }
316  }
317  // Debug output -------------------------------------------------------------
318 
319  LOG(info) << "Event ID: " << mEventID << ", bc " << firstBCinDeque.bc << ", N hit " << bc.hits.size();
320  LOG(info) << "N hit A: " << int(triggers.nChanA) << " N hit C: " << int(triggers.nChanC) << " summ ampl A: " << int(triggers.amplA)
321  << " summ ampl C: " << int(triggers.amplC) << " mean time A: " << triggers.timeA
322  << " mean time C: " << triggers.timeC << " nStored " << nStored;
323 
324  LOG(info) << "IS A " << triggers.getOrA() << " IsC " << triggers.getOrC() << " vertex " << triggers.getVertex() << " is Central " << triggers.getCen() << " is SemiCentral " << triggers.getSCen();
325 }

359 void Digitizer::initParameters()
360 {
361  auto const sinc = [](double x) { x *= TMath::Pi(); return (std::abs(x) < 1e-12) ? 1.0 : std::sin(x) / x; };
362 
363  // number of noise samples in one BC
364  const auto& params = DigitizationParameters::Instance();
365  mNumNoiseSamples = std::ceil(params.mBunchWidth / params.mNoisePeriod);
366  mNoiseSamples.resize(mNumNoiseSamples);
367 
368  // set up tables with sinc function values (times noiseVar)
369  for (size_t i = 0, n = mSincTable.size(); i < n; ++i) {
370  float const time = i / float(n) * params.mNoisePeriod; // [0 .. 1/params.mNoisePeriod)
371  LOG(debug) << "initParameters " << i << "/" << n << " " << time;
372  // we make a table of sinc values between -num_noise_samples and 2*num_noise_samples
373  mSincTable[i].resize(3 * mNumNoiseSamples);
374  for (int j = -mNumNoiseSamples; j < 2 * mNumNoiseSamples; ++j) {
375  mSincTable[i][mNumNoiseSamples + j] = params.mNoiseVar * sinc((time + 0.5f * params.mBunchWidth) / params.mNoisePeriod - j);
376  }
377  }
378  // set up the lookup table for the signal form
379  for (size_t i = 0; i < DP::SIGNAL_TABLE_SIZE; ++i) {
380  float const x = float(i) / float(DP::SIGNAL_TABLE_SIZE) * params.mBunchWidth;
381  mSignalTable[i] = signalForm_i(x);
382  }
383 
384  // cache for signal time series used by the CFD -BC/2 .. +3BC/2
385  mSignalCache.resize(std::lround(params.mBunchWidth / DP::SIGNAL_CACHE_DT));
386 }
