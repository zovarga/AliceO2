89 BOOST_AUTO_TEST_CASE(AverageGroupSector_test)
90 {
91  std::srand(std::time(nullptr));
92 
93  const static auto& paramIDCGroup = ParameterIDCGroup::Instance();
94  BOOST_CHECK(paramIDCGroup.method == AveragingMethod::FAST);
95 
96  // create some random IDCs for one TPC side
97  const auto idcsPerSide = getIDCsSide();
98 
99  std::array<std::vector<float>, Mapper::NREGIONS> idcsRegion;
100  for (int i = 0; i < Mapper::NREGIONS; ++i) {
101  idcsRegion[i] = getIDCs(i);
102  }
103 
104  // get the total sum of the reference IDCs (consider rounding of floats)
105  unsigned long refSumIDCs = 0;
106  for (const auto val : idcsPerSide) {
107  refSumIDCs += static_cast<unsigned long>(val + 0.1f);
108  }
109  refSumIDCs *= 2;
110 
111  // converting the IDCs to IDCDelta object which is used forthe averating and grouping
112  IDCDelta<float> idcs;
113  idcs.getIDCDelta(Side::A) = idcsPerSide;
114  idcs.getIDCDelta(Side::C) = idcsPerSide;
115 
116  for (int iter = 0; iter < NITER; ++iter) {
117  // type=0: do nothing special for grouping at sector edges, typ1=1: do group pads at sector edges in pad direction, typ1=2: group pads at sector edges in pad+row direction
118  for (int type = 0; type < 3; ++type) {
119  const unsigned char grPadTmp = genRand(MAXGRPAD) + 1;
120  const unsigned char grRowTmp = genRand(MAXGRROW) + 1;
121  const unsigned char grRowThrTmp = genRand(MAXGRTHR);
122  const unsigned char grPadThrTmp = genRand(MAXGRTHR);
123  const unsigned int groupPadsSectorEdges = type == 0 ? 0 : 10 * genRandSecEdge() + type - 1;
124  std::array<unsigned char, Mapper::NREGIONS> grPad{};
125  std::fill(grPad.begin(), grPad.end(), grPadTmp);
126  std::array<unsigned char, Mapper::NREGIONS> grRow{};
127  std::fill(grRow.begin(), grRow.end(), grRowTmp);
128  std::array<unsigned char, Mapper::NREGIONS> grRowThr{};
129  std::fill(grRowThr.begin(), grRowThr.end(), grRowThrTmp);
130  std::array<unsigned char, Mapper::NREGIONS> grPadThr{};
131  std::fill(grPadThr.begin(), grPadThr.end(), grPadThrTmp);
132 
133  // perform the grouping
134  IDCAverageGroup<IDCAverageGroupTPC> idcaverage(grPad, grRow, grRowThr, grPadThr, groupPadsSectorEdges);
135  idcaverage.setIDCs(idcs);
136  idcaverage.processIDCs();
137 
138  std::array<IDCAverageGroup<IDCAverageGroupCRU>, Mapper::NREGIONS> idcaverageCRU{
139  IDCAverageGroup<IDCAverageGroupCRU>(grPadTmp, grRowTmp, grRowThrTmp, grPadThrTmp, groupPadsSectorEdges, 0),
140  IDCAverageGroup<IDCAverageGroupCRU>(grPadTmp, grRowTmp, grRowThrTmp, grPadThrTmp, groupPadsSectorEdges, 1),
141  IDCAverageGroup<IDCAverageGroupCRU>(grPadTmp, grRowTmp, grRowThrTmp, grPadThrTmp, groupPadsSectorEdges, 2),
142  IDCAverageGroup<IDCAverageGroupCRU>(grPadTmp, grRowTmp, grRowThrTmp, grPadThrTmp, groupPadsSectorEdges, 3),
143  IDCAverageGroup<IDCAverageGroupCRU>(grPadTmp, grRowTmp, grRowThrTmp, grPadThrTmp, groupPadsSectorEdges, 4),
144  IDCAverageGroup<IDCAverageGroupCRU>(grPadTmp, grRowTmp, grRowThrTmp, grPadThrTmp, groupPadsSectorEdges, 5),
145  IDCAverageGroup<IDCAverageGroupCRU>(grPadTmp, grRowTmp, grRowThrTmp, grPadThrTmp, groupPadsSectorEdges, 6),
146  IDCAverageGroup<IDCAverageGroupCRU>(grPadTmp, grRowTmp, grRowThrTmp, grPadThrTmp, groupPadsSectorEdges, 7),
147  IDCAverageGroup<IDCAverageGroupCRU>(grPadTmp, grRowTmp, grRowThrTmp, grPadThrTmp, groupPadsSectorEdges, 8),
148  IDCAverageGroup<IDCAverageGroupCRU>(grPadTmp, grRowTmp, grRowThrTmp, grPadThrTmp, groupPadsSectorEdges, 9)};
149 
150  for (unsigned int region = 0; region < Mapper::NREGIONS; ++region) {
151  idcaverageCRU[region].setIDCs(idcsRegion[region]);
152  idcaverageCRU[region].processIDCs();
153  }
154 
155  // get the total sum of the grouped IDCs (consider rounding of floats)
156  unsigned long sumGroupedIDCs = 0;
157  unsigned long sumGroupedIDCsCRU = 0;
158  std::vector<unsigned long> sumGroupedIDCsSector(NINTEGRATIONINTERVALS * Mapper::NSECTORS);
159  for (unsigned int i = 0; i < NINTEGRATIONINTERVALS; ++i) {
160  for (unsigned int sector = 0; sector < Mapper::NSECTORS; ++sector) {
161  const int index = sector + i * Mapper::NSECTORS;
162  for (unsigned int region = 0; region < Mapper::NREGIONS; ++region) {
163  for (unsigned int irow = 0; irow < Mapper::ROWSPERREGION[region]; ++irow) {
164  for (unsigned int ipad = 0; ipad < Mapper::PADSPERROW[region][irow]; ++ipad) {
165  const auto valSector = static_cast<unsigned long>(idcaverage.getUngroupedIDCDeltaVal(sector, region, irow, ipad, i) + 0.1f);
166  const auto valCRU = static_cast<unsigned long>(idcaverageCRU[region].getUngroupedIDCValLocal(irow, ipad, i) + 0.1f);
167  sumGroupedIDCsSector[index] += valSector;
168  sumGroupedIDCsCRU += valCRU;
169  BOOST_CHECK(valSector == valCRU);
170  }
171  }
172  }
173  sumGroupedIDCs += sumGroupedIDCsSector[i];
174  }
175  }
176 
177  // comparing the sum from the reference IDCs to the sum of the grouped IDCs
178  BOOST_CHECK(refSumIDCs == sumGroupedIDCs);
179  BOOST_CHECK(refSumIDCs == sumGroupedIDCsCRU);
180 
181  // comparing the reference sum of IDCs per sector with the grouped IDCs per sector
182  for (const auto idcsSec : sumGroupedIDCsSector) {
183  BOOST_CHECK(refSumIDCs / sumGroupedIDCsSector.size() == idcsSec);
184  }
185  }
186  }
