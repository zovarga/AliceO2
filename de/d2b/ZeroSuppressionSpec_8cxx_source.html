47  void run(o2::framework::ProcessingContext& pc)
48  {
49  const auto patterns = pc.inputs().get<gsl::span<ColumnData>>("mid_data_mc");
50 
51  const auto inROFRecords = pc.inputs().get<gsl::span<ROFRecord>>("mid_data_mc_rof");
52 
53  const auto inMCContainer = mUseMC ? pc.inputs().get<const o2::dataformats::MCTruthContainer<MCLabel>*>("mid_data_mc_labels") : nullptr;
54 
55  o2::dataformats::MCTruthContainer<MCLabel> outMCContainer;
56 
57  std::vector<ROFRecord> zsROFs, tmpROFs(1);
58  std::vector<ColumnData> zsData;
59  for (auto& rof : inROFRecords) {
60  mConverter.process(patterns.subspan(rof.firstEntry, rof.nEntries));
61  if (!mConverter.getData().empty()) {
62  std::vector<ROBoard> decodedData = mConverter.getData();
63  mResponse.applyZeroSuppression(decodedData);
64  tmpROFs.front().interactionRecord = rof.interactionRecord;
65  tmpROFs.front().eventType = rof.eventType;
66  tmpROFs.front().firstEntry = 0;
67  tmpROFs.front().nEntries = decodedData.size();
68  mAggregator.process(decodedData, tmpROFs);
69  auto& tmpOut = mAggregator.getData();
70  zsROFs.emplace_back(rof.interactionRecord, rof.eventType, zsData.size(), tmpOut.size());
71  zsData.insert(zsData.end(), tmpOut.begin(), tmpOut.end());
72 
73  if (mUseMC) {
74  for (auto outColIt = zsData.begin() + zsROFs.back().firstEntry, outEnd = zsData.begin() + zsROFs.back().firstEntry + zsROFs.back().nEntries; outColIt != outEnd; ++outColIt) {
75  for (auto inColIt = patterns.begin() + rof.firstEntry, inEnd = patterns.begin() + rof.firstEntry + rof.nEntries; inColIt != inEnd; ++inColIt) {
76  if (inColIt->deId == outColIt->deId && inColIt->columnId == outColIt->columnId) {
77  auto inIdx = std::distance(patterns.begin(), inColIt);
78  auto outIdx = std::distance(zsData.begin(), outColIt);
79  outMCContainer.addElements(outIdx, inMCContainer->getLabels(inIdx));
80  break;
81  }
82  }
83  }
84  }
85  }
86  }
87 
88  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "DATA", 0, of::Lifetime::Timeframe}, zsData);
89  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "DATAROF", 0, of::Lifetime::Timeframe}, zsROFs);
90  if (mUseMC) {
91  pc.outputs().snapshot(of::Output{header::gDataOriginMID, "DATALABELS", 0, of::Lifetime::Timeframe}, outMCContainer);
92  }
93  }
94 
95  private:
96  ColumnDataToLocalBoard mConverter{};
97  DecodedDataAggregator mAggregator{};
98  ROBoardResponse mResponse{};
99  bool mUseMC{true};
100 };
