62 void DigitRecoSpec::run(ProcessingContext& pc)
63 {
64  if (!mInitialized) {
65  mInitialized = true;
66  // Initialization from CCDB
67  auto& mgr = o2::ccdb::BasicCCDBManager::instance();
68  mgr.setURL(mccdbHost);
69  /*long timeStamp = 0; // TIMESTAMP SHOULD NOT BE 0
70  if (timeStamp == mgr.getTimestamp()) {
71  return;
72  }
73  mgr.setTimestamp(timeStamp);*/
74 
75  std::string loadedConfFiles = "Loaded ZDC configuration files for timestamp " + std::to_string(mgr.getTimestamp()) + ":";
76  auto* moduleConfig = mgr.get<o2::zdc::ModuleConfig>(o2::zdc::CCDBPathConfigModule);
77  if (!moduleConfig) {
78  LOG(fatal) << "Missing ModuleConfig ZDC configuration object";
79  return;
80  } else {
81  loadedConfFiles += " ModuleConfig";
82  }
83  if (mVerbosity > DbgZero) {
84  LOG(info) << "Loaded ZDC module configuration for timestamp " << mgr.getTimestamp();
85  moduleConfig->print();
86  }
87 
88  // Configuration parameters for ZDC reconstruction
89  auto* recoConfigZDC = mgr.get<o2::zdc::RecoConfigZDC>(o2::zdc::CCDBPathRecoConfigZDC);
90  if (!recoConfigZDC) {
91  LOG(info) << loadedConfFiles;
92  LOG(fatal) << "Missing RecoConfigZDC object";
93  return;
94  } else {
95  loadedConfFiles += " RecoConfigZDC";
96  }
97  if (mVerbosity > DbgZero) {
98  LOG(info) << "Loaded RecoConfigZDC for timestamp " << mgr.getTimestamp();
99  recoConfigZDC->print();
100  }
101 
102  // TDC centering
103  auto* tdcParam = mgr.get<o2::zdc::ZDCTDCParam>(o2::zdc::CCDBPathTDCCalib);
104  if (!tdcParam) {
105  LOG(info) << loadedConfFiles;
106  LOG(fatal) << "Missing ZDCTDCParam calibration object";
107  return;
108  } else {
109  loadedConfFiles += " ZDCTDCParam";
110  }
111  if (mVerbosity > DbgZero) {
112  LOG(info) << "Loaded TDC centering ZDCTDCParam for timestamp " << mgr.getTimestamp();
113  tdcParam->print();
114  }
115 
116  // Energy calibration
117  auto* energyParam = mgr.get<o2::zdc::ZDCEnergyParam>(o2::zdc::CCDBPathEnergyCalib);
118  if (!energyParam) {
119  LOG(warning) << "Missing ZDCEnergyParam calibration object - using default";
120  } else {
121  loadedConfFiles += " ZDCEnergyParam";
122  if (mVerbosity > DbgZero) {
123  LOG(info) << "Loaded Energy calibration ZDCEnergyParam for timestamp " << mgr.getTimestamp();
124  energyParam->print();
125  }
126  }
127 
128  // Tower calibration
129  auto* towerParam = mgr.get<o2::zdc::ZDCTowerParam>(o2::zdc::CCDBPathTowerCalib);
130  if (!towerParam) {
131  LOG(warning) << "Missing ZDCTowerParam calibration object - using default";
132  } else {
133  loadedConfFiles += " ZDCTowerParam";
134  if (mVerbosity > DbgZero) {
135  LOG(info) << "Loaded Tower calibration ZDCTowerParam for timestamp " << mgr.getTimestamp();
136  towerParam->print();
137  }
138  }
139 
140  LOG(info) << loadedConfFiles;
141 
142  mDR.setModuleConfig(moduleConfig);
143  mDR.setRecoConfigZDC(recoConfigZDC);
144  mDR.setTDCParam(tdcParam);
145  mDR.setEnergyParam(energyParam);
146  mDR.setTowerParam(towerParam);
147 
148  if (mDebugOut) {
149  mDR.setDebugOutput();
150  }
151 
152  mDR.setVerbosity(mVerbosity);
153 
154  mDR.init();
155  }
156  auto cput = mTimer.CpuTime();
157  mTimer.Start(false);
158  auto bcdata = pc.inputs().get<gsl::span<o2::zdc::BCData>>("trig");
159  auto chans = pc.inputs().get<gsl::span<o2::zdc::ChannelData>>("chan");
160  auto peds = pc.inputs().get<gsl::span<o2::zdc::OrbitData>>("peds");
161 
162  mDR.process(peds, bcdata, chans);
163  const std::vector<o2::zdc::RecEventAux>& recAux = mDR.getReco();
164 
165  RecEvent recEvent;
166  int32_t nte = 0, ntt = 0;
167  for (auto reca : recAux) {
168  int32_t ne = reca.ezdc.size();
169  int32_t nt = 0;
170  for (int32_t it = 0; it < o2::zdc::NTDCChannels; it++) {
171  for (int32_t ih = 0; ih < reca.ntdc[it]; ih++) {
172  if (nt == 0) {
173  recEvent.addBC(reca.ir, reca.channels, reca.triggers);
174  }
175  nt++;
176  recEvent.addTDC(it, reca.TDCVal[it][ih], reca.TDCAmp[it][ih]);
177  }
178  }
179  if (ne > 0 && nt == 0) {
180  recEvent.addBC(reca.ir);
181  }
182  if (ne > 0) {
183  std::map<uint8_t, float>::iterator it;
184  for (it = reca.ezdc.begin(); it != reca.ezdc.end(); it++) {
185  recEvent.addEnergy(it->first, it->second);
186  }
187  }
188  nte += ne;
189  ntt += nt;
190  if (mVerbosity > 0 && (nt > 0 || ne > 0)) {
191  printf("Orbit %9u bc %4u ntdc %2d ne %2d\n", reca.ir.orbit, reca.ir.bc, nt, ne);
192  }
193  // Event information
194  recEvent.addInfo(reca.tdcPedQC, MsgTDCPedQC);
195  recEvent.addInfo(reca.tdcPedMissing, MsgTDCPedMissing);
196  recEvent.addInfo(reca.adcPedOr, MsgADCPedOr);
197  recEvent.addInfo(reca.adcPedQC, MsgADCPedQC);
198  recEvent.addInfo(reca.adcPedMissing, MsgADCPedMissing);
199  }
200  LOG(info) << "Reconstructed " << ntt << " signal TDCs and " << nte << " ZDC energies in " << recEvent.mRecBC.size() << "/" << recAux.size() << " b.c.";
201  // TODO: rate information for all channels
202  // TODO: summary of reconstruction to be collected by DQM?
203  pc.outputs().snapshot(Output{"ZDC", "BCREC", 0, Lifetime::Timeframe}, recEvent.mRecBC);
204  pc.outputs().snapshot(Output{"ZDC", "ENERGY", 0, Lifetime::Timeframe}, recEvent.mEnergy);
205  pc.outputs().snapshot(Output{"ZDC", "TDCDATA", 0, Lifetime::Timeframe}, recEvent.mTDCData);
206  pc.outputs().snapshot(Output{"ZDC", "INFO", 0, Lifetime::Timeframe}, recEvent.mInfo);
207  mTimer.Stop();
208 }

217 framework::DataProcessorSpec getDigitRecoSpec(const int verbosity = 0, const bool enableDebugOut = false)
218 {
219  std::vector<InputSpec> inputs;
220  inputs.emplace_back("trig", "ZDC", "DIGITSBC", 0, Lifetime::Timeframe);
221  inputs.emplace_back("chan", "ZDC", "DIGITSCH", 0, Lifetime::Timeframe);
222  inputs.emplace_back("peds", "ZDC", "DIGITSPD", 0, Lifetime::Timeframe);
223 
224  std::vector<OutputSpec> outputs;
225  outputs.emplace_back("ZDC", "BCREC", 0, Lifetime::Timeframe);
226  outputs.emplace_back("ZDC", "ENERGY", 0, Lifetime::Timeframe);
227  outputs.emplace_back("ZDC", "TDCDATA", 0, Lifetime::Timeframe);
228  outputs.emplace_back("ZDC", "INFO", 0, Lifetime::Timeframe);
229 
230  return DataProcessorSpec{
231  "zdc-digi-reco",
232  inputs,
233  outputs,
234  AlgorithmSpec{adaptFromTask<DigitRecoSpec>(verbosity, enableDebugOut)},
235  o2::framework::Options{{"ccdb-url", o2::framework::VariantType::String, o2::base::NameConf::getCCDBServer(), {"CCDB Url"}}}};
236 }
