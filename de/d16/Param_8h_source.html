146  static double lAbsWin(double eV) { return (eV < 8.2) * (818.8638 - 301.0436 * eV + 36.89642 * eV * eV - 1.507555 * eV * eV * eV) + (eV >= 8.2) * 0.0001; } // fit from DiMauro data 28.10.03

174  void lors2Mars(Int_t c, double x, double y, double* m, Int_t pl = kPc) const
175  {

177  switch (pl) {

190  }

204  void mars2LorsVec(Int_t c, double* m, double& th, double& ph) const
205  {

208  float pt = TMath::Sqrt(l[0] * l[0] + l[1] * l[1]);
