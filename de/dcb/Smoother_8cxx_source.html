21 constexpr std::array<double, 3> getInverseSymm2D(const std::array<double, 3>& mat)
22 {
23  const double det = mat[0] * mat[2] - mat[1] * mat[1];
24  return std::array<double, 3>{mat[2] / det, -mat[1] / det, mat[0] / det};
25 }

28 template <unsigned int D>
29 Smoother<D>::Smoother(TrackITSExt& track, int smoothingLayer, const ROframe& event, float bZ, o2::base::PropagatorF::MatCorrType corr) : mLayerToSmooth{smoothingLayer},
30  mBz(bZ),
31  mCorr(corr)
32 {
33 #if defined(CA_DEBUG) || defined(CA_STANDALONE_DEBUGGER)
34  mDebugger = new StandaloneDebugger("dbg_ITSSmootherCPU.root");
35 #endif
36 
37  auto propInstance = o2::base::Propagator::Instance();
38  const TrackingFrameInfo& originalTf = event.getTrackingFrameInfoOnLayer(mLayerToSmooth).at(track.getClusterIndex(mLayerToSmooth));
39 
40  mOutwardsTrack = track; // This track will be propagated outwards inside the smoother! (as last step of fitting did inward propagation)
41  mInwardsTrack = {track.getParamOut(), // This track will be propagated inwards inside the smoother!
42  static_cast<short>(mOutwardsTrack.getNumberOfClusters()), -999, static_cast<std::uint32_t>(event.getROFrameId()),
43  mOutwardsTrack.getParamOut(), mOutwardsTrack.getClusterIndexes()};
44 
45  mOutwardsTrack.resetCovariance();
46  mOutwardsTrack.setChi2(0);
47  mInwardsTrack.resetCovariance();
48  mInwardsTrack.setChi2(0);
49 
50  bool statusOutw{false};
51  bool statusInw{false};
52 
54  // Outward propagation
55  for (size_t iLayer{0}; iLayer < mLayerToSmooth; ++iLayer) {
56  if (mOutwardsTrack.getClusterIndex(iLayer) == constants::its::UnusedIndex) { // Shorter tracks
57  continue;
58  }
59  const TrackingFrameInfo& tF = event.getTrackingFrameInfoOnLayer(iLayer).at(mOutwardsTrack.getClusterIndex(iLayer));
60  statusOutw = mOutwardsTrack.rotate(tF.alphaTrackingFrame);
61  statusOutw &= propInstance->propagateToX(mOutwardsTrack,
62  tF.xTrackingFrame,
63  mBz,
64  o2::base::PropagatorImpl<float>::MAX_SIN_PHI,
65  o2::base::PropagatorImpl<float>::MAX_STEP,
66  mCorr);
67  mOutwardsTrack.setChi2(mOutwardsTrack.getChi2() + mOutwardsTrack.getPredictedChi2(tF.positionTrackingFrame, tF.covarianceTrackingFrame));
68  statusOutw &= mOutwardsTrack.o2::track::TrackParCov::update(tF.positionTrackingFrame, tF.covarianceTrackingFrame);
69  // LOG(info) << "Outwards loop on inwards track, layer: " << iLayer << " x: " << mOutwardsTrack.getX();
70  }
71 
72  // Prediction on the previously outwards-propagated track is done on a copy, as the process seems to be not reversible
73  auto outwardsClone = mOutwardsTrack;
74  statusOutw = outwardsClone.rotate(originalTf.alphaTrackingFrame);
75  statusOutw &= propInstance->propagateToX(outwardsClone,
76  originalTf.xTrackingFrame,
77  mBz,
78  o2::base::PropagatorImpl<float>::MAX_SIN_PHI,
79  o2::base::PropagatorImpl<float>::MAX_STEP,
80  mCorr);
82  // Inward propagation
83  for (size_t iLayer{D - 1}; iLayer > mLayerToSmooth; --iLayer) {
84  if (mInwardsTrack.getClusterIndex(iLayer) == constants::its::UnusedIndex) { // Shorter tracks
85  continue;
86  }
87  const TrackingFrameInfo& tF = event.getTrackingFrameInfoOnLayer(iLayer).at(mInwardsTrack.getClusterIndex(iLayer));
88  statusInw = mInwardsTrack.rotate(tF.alphaTrackingFrame);
89  statusInw &= propInstance->propagateToX(mInwardsTrack,
90  tF.xTrackingFrame,
91  mBz,
92  o2::base::PropagatorImpl<float>::MAX_SIN_PHI,
93  o2::base::PropagatorImpl<float>::MAX_STEP,
94  mCorr);
95  mInwardsTrack.setChi2(mInwardsTrack.getChi2() + mInwardsTrack.getPredictedChi2(tF.positionTrackingFrame, tF.covarianceTrackingFrame));
96  statusInw &= mInwardsTrack.o2::track::TrackParCov::update(tF.positionTrackingFrame, tF.covarianceTrackingFrame);
97  // LOG(info) << "Inwards loop on outwards track, layer: " << iLayer << " x: " << mInwardsTrack.getX();
98  }
99 
100  // Prediction on the previously inwards-propagated track is done on a copy, as the process seems to be not revesible
101  auto inwardsClone = mInwardsTrack;
102  statusInw = inwardsClone.rotate(originalTf.alphaTrackingFrame);
103  statusInw &= propInstance->propagateToX(inwardsClone,
104  originalTf.xTrackingFrame,
105  mBz,
106  o2::base::PropagatorImpl<float>::MAX_SIN_PHI,
107  o2::base::PropagatorImpl<float>::MAX_STEP,
108  mCorr);
109  // Compute weighted local chi2
110  mInitStatus = statusInw && statusOutw;
111  if (mInitStatus) {
112  mBestChi2 = computeSmoothedPredictedChi2(inwardsClone, outwardsClone, originalTf.positionTrackingFrame, originalTf.covarianceTrackingFrame);
113  mLastChi2 = mBestChi2;
114  LOG(info) << "Smoothed chi2 on original cluster: " << mBestChi2;
115  }
116 }

129 template <unsigned int D>
130 float Smoother<D>::computeSmoothedPredictedChi2(const o2::track::TrackParCov& firstTrack, // outwards track: from innermost cluster to outermost
131  const o2::track::TrackParCov& secondTrack, // inwards track: from outermost cluster to innermost
132  const std::array<float, 2>& cls,
133  const std::array<float, 3>& clCov)
134 {
135  // Tracks need to be already propagated, compute only chi2
136  // Symmetric covariances assumed
137 
138  if (firstTrack.getX() != secondTrack.getX()) {
139  LOG(fatal) << "Tracks need to be propagated to the same point! secondTrack.X=" << secondTrack.getX() << " firstTrack.X=" << firstTrack.getX();
140  }
141 
142  std::array<double, 2> pp1 = {static_cast<double>(firstTrack.getY()), static_cast<double>(firstTrack.getZ())}; // P1: predicted Y,Z points
143  std::array<double, 2> pp2 = {static_cast<double>(secondTrack.getY()), static_cast<double>(secondTrack.getZ())}; // P2: predicted Y,Z points
144 
145  std::array<double, 3> c1 = {static_cast<double>(firstTrack.getSigmaY2()),
146  static_cast<double>(firstTrack.getSigmaZY()),
147  static_cast<double>(firstTrack.getSigmaZ2())}; // Cov. track 1
148 
149  std::array<double, 3> c2 = {static_cast<double>(secondTrack.getSigmaY2()),
150  static_cast<double>(secondTrack.getSigmaZY()),
151  static_cast<double>(secondTrack.getSigmaZ2())}; // Cov. track 2
152 
153  std::array<double, 3> w1 = getInverseSymm2D(c1); // weight matrices
154  std::array<double, 3> w2 = getInverseSymm2D(c2);
155 
156  std::array<double, 3> w1w2 = {w1[0] + w2[0], w1[1] + w2[1], w1[2] + w2[2]}; // (W1 + W2)
157  std::array<double, 3> C = getInverseSymm2D(w1w2); // C = (W1+W2)^-1
158 
159  std::array<double, 2> w1pp1 = {w1[0] * pp1[0] + w1[1] * pp1[1], w1[1] * pp1[0] + w1[2] * pp1[1]}; // W1 * P1
160  std::array<double, 2> w2pp2 = {w2[0] * pp2[0] + w2[1] * pp2[1], w2[1] * pp2[0] + w2[2] * pp2[1]}; // W2 * P2
161 
162  double Y = C[0] * (w1pp1[0] + w2pp2[0]) + C[1] * (w1pp1[1] + w2pp2[1]); // Pp: weighted normalized combination of the predictions:
163  double Z = C[1] * (w1pp1[0] + w2pp2[0]) + C[2] * (w1pp1[1] + w2pp2[1]); // Pp = [(W1 * P1) + (W2 * P2)] / (W1 + W2)
164 
165  std::array<double, 2> delta = {Y - cls[0], Z - cls[1]}; // Δ = Pp - X, X: space point of cluster (Y,Z)
166  std::array<double, 3> CCp = {C[0] + static_cast<double>(clCov[0]), C[1] + static_cast<double>(clCov[1]), C[2] + static_cast<double>(clCov[2])}; // Transformation of cluster covmat: CCp = C + Cov
167  std::array<double, 3> Wp = getInverseSymm2D(CCp); // Get weight matrix: Wp = CCp^-1
168 
169  float chi2 = static_cast<float>(delta[0] * (Wp[0] * delta[0] + Wp[1] * delta[1]) + delta[1] * (Wp[1] * delta[0] + Wp[2] * delta[1])); // chi2 = tΔ * (Wp * Δ)
170 
171  // #ifdef CA_DEBUG
172  LOG(info) << "Cluster_y: " << cls[0] << " Cluster_z: " << cls[1];
173  LOG(info) << "\t\t- Covariance cluster: Y2: " << clCov[0] << " YZ: " << clCov[1] << " Z2: " << clCov[2];
174  LOG(info) << "\t\t- Propagated t1_y: " << pp1[0] << " t1_z: " << pp1[1];
175  LOG(info) << "\t\t- Propagated t2_y: " << pp2[0] << " t2_z: " << pp2[1];
176  LOG(info) << "\t\t- Covariance t1: sY2: " << c1[0] << " sYZ: " << c1[1] << " sZ2: " << c1[2];
177  LOG(info) << "\t\t- Covariance t2: sY2: " << c2[0] << " sYZ: " << c2[1] << " sZ2: " << c2[2];
178  LOG(info) << "Smoother prediction Y: " << Y << " Z: " << Z;
179  LOG(info) << "\t\t- Delta_y: " << delta[0] << " Delta_z: " << delta[1];
180  LOG(info) << "\t\t- Covariance Pr: Y2: " << C[0] << " YZ: " << C[1] << " Z2: " << C[2];
181  LOG(info) << "\t\t- predicted chi2 t1: " << firstTrack.getPredictedChi2(cls, clCov);
182  LOG(info) << "\t\t- predicted chi2 t2: " << secondTrack.getPredictedChi2(cls, clCov);
183  // #endif
184  return chi2;
185 }

188 bool Smoother<D>::testCluster(const int clusterId, const ROframe& event)
189 {
190  if (!mInitStatus) {
191  return false;
192  }
193  auto propInstance = o2::base::Propagator::Instance();
194  const TrackingFrameInfo& testTf = event.getTrackingFrameInfoOnLayer(mLayerToSmooth).at(clusterId);
195 
196  bool statusOutw{false};
197  bool statusInw{false};
198 
199  // Prediction on the previously outwards-propagated track is done on a copy, as the process seems to be not reversible
200  auto outwardsClone = mOutwardsTrack;
201  statusOutw = outwardsClone.rotate(testTf.alphaTrackingFrame);
202  statusOutw &= propInstance->propagateToX(outwardsClone,
203  testTf.xTrackingFrame,
204  mBz,
205  o2::base::PropagatorImpl<float>::MAX_SIN_PHI,
206  o2::base::PropagatorImpl<float>::MAX_STEP,
207  mCorr);
208 
209  // Prediction on the previously inwards-propagated track is done on a copy, as the process seems to be not reversible
210  auto inwardsClone = mInwardsTrack;
211  statusInw = inwardsClone.rotate(testTf.alphaTrackingFrame);
212  statusInw &= propInstance->propagateToX(inwardsClone,
213  testTf.xTrackingFrame,
214  mBz,
215  o2::base::PropagatorImpl<float>::MAX_SIN_PHI,
216  o2::base::PropagatorImpl<float>::MAX_STEP,
217  mCorr);
218  bool testStatus = statusOutw && statusInw;
219  if (!(statusOutw && statusInw)) {
220  LOG(warning) << "Failed propagation in smoother!";
221  return false;
222  }
223 
224  // Compute weighted local chi2
225  mLastChi2 = computeSmoothedPredictedChi2(inwardsClone, outwardsClone, testTf.positionTrackingFrame, testTf.covarianceTrackingFrame);
226  LOG(info) << "Smoothed chi2 on tested cluster: " << mLastChi2;
227 
228  return true;
229 }
