37 template <typename state_T, typename stream_T>
38 class Encoder
39 {
40  __extension__ using uint128_t = unsigned __int128;
41 
42  // the Coder works either with a 64Bit state and 32 bit streaming or
43  //a 32 Bit state and 8 Bit streaming We need to make sure it gets initialized with
44  //the right template arguments at compile time.
45  static_assert((sizeof(state_T) == sizeof(uint32_t) && sizeof(stream_T) == sizeof(uint8_t)) ||
46  (sizeof(state_T) == sizeof(uint64_t) && sizeof(stream_T) == sizeof(uint32_t)),
47  "Coder can either be 32Bit with 8 Bit stream type or 64 Bit Type with 32 Bit stream type");
48 
49  public:
50  explicit Encoder(size_t symbolTablePrecission) noexcept;
51 
52  template <typename stream_IT>
53  stream_IT flush(stream_IT outputIter);
54 
55  // Encodes a given symbol.
56  template <typename stream_IT>
57  stream_IT putSymbol(stream_IT outputIter, const EncoderSymbol<state_T>& symbol);
58 
59  private:
60  state_T mState{LOWER_BOUND};
61  size_t mSymbolTablePrecission{};
62 
63  // Renormalize the encoder.
64  template <typename stream_IT>
65  std::tuple<state_T, stream_IT> renorm(state_T state, stream_IT outputIter, uint32_t frequency);
66 
67  // L ('l' in the paper) is the lower bound of our normalization interval.
68  // Between this and our byte-aligned emission, we use 31 (not 32!) bits.
69  // This is done intentionally because exact reciprocals for 31-bit uints
70  // fit in 32-bit uints: this permits some optimizations during encoding.
71  inline static constexpr state_T LOWER_BOUND = needs64Bit<state_T>() ? (1u << 31) : (1u << 23); // lower bound of our normalization interval
72 
73  inline static constexpr state_T STREAM_BITS = sizeof(stream_T) * 8; // lower bound of our normalization interval
74 };

80 template <typename stream_IT>
81 stream_IT Encoder<state_T, stream_T>::flush(stream_IT outputIter)
82 {
83  stream_IT streamPosition = outputIter;
84  if constexpr (needs64Bit<state_T>()) {
85  ++streamPosition;
86  *streamPosition = static_cast<stream_T>(mState >> 32);
87  ++streamPosition;
88  *streamPosition = static_cast<stream_T>(mState >> 0);

103 template <typename state_T, typename stream_T>
104 template <typename stream_IT>
105 stream_IT Encoder<state_T, stream_T>::putSymbol(stream_IT outputIter, const EncoderSymbol<state_T>& symbol)
106 {
107 
108  assert(symbol.getFrequency() != 0); // can't encode symbol with freq=0
109 
110  // renormalize
111  const auto [newState, streamPosition] = renorm(mState, outputIter, symbol.getFrequency());
112 
113  // x = C(s,x)
114  state_T quotient = 0;
115 
116  if constexpr (needs64Bit<state_T>()) {
117  // This code needs support for 64-bit long multiplies with 128-bit result
118  // (or more precisely, the top 64 bits of a 128-bit result).
119  quotient = static_cast<state_T>((static_cast<uint128_t>(newState) * symbol.getReciprocalFrequency()) >> 64);
120  } else {
121  quotient = static_cast<state_T>((static_cast<uint64_t>(newState) * symbol.getReciprocalFrequency()) >> 32);
122  }
123  quotient = quotient >> symbol.getReciprocalShift();
124 
125  mState = newState + symbol.getBias() + quotient * symbol.getFrequencyComplement();
126  return streamPosition;
127 };

129 template <typename state_T, typename stream_T>
130 template <typename stream_IT>
131 inline std::tuple<state_T, stream_IT> Encoder<state_T, stream_T>::renorm(state_T state, stream_IT outputIter, uint32_t frequency)
132 {
133  state_T maxState = ((LOWER_BOUND >> mSymbolTablePrecission) << STREAM_BITS) * frequency; // this turns into a shift.
134  if (state >= maxState) {
135  if constexpr (needs64Bit<state_T>()) {
136  ++outputIter;
137  *outputIter = static_cast<stream_T>(state);
138  state >>= STREAM_BITS;
139  assert(state < maxState);
140  } else {
141  do {
142  ++outputIter;
143  //stream out 8 Bits
144  *outputIter = static_cast<stream_T>(state & 0xff);
145  state >>= STREAM_BITS;
146  } while (state >= maxState);
147  }
148  }
149  return std::make_tuple(state, outputIter);
150 };
