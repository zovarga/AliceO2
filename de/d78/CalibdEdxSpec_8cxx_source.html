35 class CalibdEdxDevice : public Task
36 {
37  public:
38  void init(framework::InitContext& ic) final
39  {
40  const int minEntriesSector = ic.options().get<int>("min-entries-sector");
41  const int minEntries1D = ic.options().get<int>("min-entries-1d");
42  const int minEntries2D = ic.options().get<int>("min-entries-2d");
43 
44  const int dEdxBins = ic.options().get<int>("dedxbins");
45  const int zBins = ic.options().get<int>("zbins");
46  const int angularBins = ic.options().get<int>("angularbins");
47  const float mindEdx = ic.options().get<float>("min-dedx");
48  const float maxdEdx = ic.options().get<float>("max-dedx");
49 
50  mDumpToFile = ic.options().get<bool>("file-dump");
51  float field = ic.options().get<float>("field");
52 
53  if (field <= -10.f) {
54  const auto inputGRP = o2::base::NameConf::getGRPFileName();
55  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
56  if (grp != nullptr) {
57  field = 5.00668f * grp->getL3Current() / 30000.;
58  LOGP(info, "Using GRP file to set the magnetic field to {} kG", field);
59  }
60  }
61 
62  mCalib = std::make_unique<CalibdEdx>(mindEdx, maxdEdx, dEdxBins, zBins, angularBins);
63  mCalib->setApplyCuts(false);
64  mCalib->setFitCuts({minEntriesSector, minEntries1D, minEntries2D});
65  mCalib->setField(field);
66  }
67 
68  void run(ProcessingContext& pc) final
69  {
70  const auto tfcounter = o2::header::get<DataProcessingHeader*>(pc.inputs().get("tracks").header)->startTime;
71  const auto tracks = pc.inputs().get<gsl::span<TrackTPC>>("tracks");
72 
73  LOGP(info, "Processing TF {} with {} tracks", tfcounter, tracks.size());
74 
75  mCalib->fill(tracks);
76  // sendOutput(pc.outputs());
77  }
78 
79  void endOfStream(EndOfStreamContext& eos) final
80  {
81  LOGP(info, "Finalizing calibration");
82  mCalib->finalize();
83  mCalib->print();
84  sendOutput(eos.outputs());
85 
86  if (mDumpToFile) {
87  mCalib->getCalib().writeToFile("calibdEdx.root");
88  }
89  }
90 
91  private:
92  void sendOutput(DataAllocator& output)
93  {
94  using clbUtils = o2::calibration::Utils;
95  const auto& corr = mCalib->getCalib();
96 
97  o2::ccdb::CcdbObjectInfo info;
98 
99  const auto now = std::chrono::system_clock::now();
100  const long timeStart = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()).count();
101  const long timeEnd = 99999999999999;
102 
103  info.setPath("TPC/Calib/dEdx");
104  info.setStartValidityTimestamp(timeStart);
105  info.setEndValidityTimestamp(timeEnd);
106 
107  auto image = o2::ccdb::CcdbApi::createObjectImage(&corr, &info);
108 
109  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", info.getPath(), info.getFileName(), image->size(), info.getStartValidityTimestamp(), info.getEndValidityTimestamp());
110  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "TPC_CalibdEdx", 0}, *image.get());
111  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "TPC_CalibdEdx", 0}, info);
112  }
113 
114  bool mDumpToFile{};
115  std::unique_ptr<CalibdEdx> mCalib;
116 };

139  {"min-dedx", VariantType::Float, 5.0f, {"minimum value for the dEdx histograms"}},
140  {"max-dedx", VariantType::Float, 100.0f, {"maximum value for the dEdx histograms"}},
