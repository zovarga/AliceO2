52 void PHOSTurnonCalibDevice::endOfStream(o2::framework::EndOfStreamContext& ec)
53 {
54  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
55  mCalibrator->checkSlotsToFinalize(INFINITE_TF);
56  mCalibrator->endOfStream();
57  mTriggerMap.reset(new TriggerMap(mCalibrator->getCalibration()));
58  if (checkFitResult()) {
59  //Calculate and send final object to CCDB
60  auto flName = o2::ccdb::CcdbApi::generateFileName("TriggerMap");
61  std::map<std::string, std::string> md;
62  o2::ccdb::CcdbObjectInfo info("PHS/Calib/TriggerMap", "TriggerMap", flName, md, mRunStartTime, 99999999999999);
63  info.setMetaData(md);
64  auto image = o2::ccdb::CcdbApi::createObjectImage(mTriggerMap.get(), &info);
65 
66  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName()
67  << " of size " << image->size()
68  << " bytes, valid for " << info.getStartValidityTimestamp()
69  << " : " << info.getEndValidityTimestamp();
70 
71  header::DataHeader::SubSpecificationType subSpec{(header::DataHeader::SubSpecificationType)0};
72  ec.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "PHOS_Tunron", subSpec}, *image.get());
73  ec.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "PHOS_Tunron", subSpec}, info);
74  } else {
75  LOG(error) << "Incorrect fit results";
76  }
