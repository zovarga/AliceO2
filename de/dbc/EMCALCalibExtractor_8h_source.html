67  template <typename... axes>
68  o2::emcal::BadChannelMap calibrateBadChannels(boost::histogram::histogram<axes...>& hist)
69  {
70  // calculate the mean and sigma of the mEsumHisto
71  auto fitValues = o2::utils::fitBoostHistoWithGaus<double>(hist);
72  double mean = fitValues.at(1);
73  double sigma = fitValues.at(2);
74 
75  // calculate the "good cell window from the mean"
76  double maxVal = mean + mSigma * sigma;
77  double minVal = mean - mSigma * sigma;
78  o2::emcal::BadChannelMap mOutputBCM;
79  // now loop through the cells and determine the mask for a given cell
80  for (int cellID = 0; cellID < 17664; cellID++) {
81  double E = hist.at(cellID);
82  // if in the good window, mark the cell as good
83  // for now we won't do warm cells - the definition of this is unclear.
84  if (E == 0) {
85  mOutputBCM.addBadChannel(cellID, o2::emcal::BadChannelMap::MaskType_t::DEAD_CELL);
86  } else if (E < maxVal || E > minVal) {
87  mOutputBCM.addBadChannel(cellID, o2::emcal::BadChannelMap::MaskType_t::GOOD_CELL);
88  } else {
89  mOutputBCM.addBadChannel(cellID, o2::emcal::BadChannelMap::MaskType_t::BAD_CELL);
90  }
91  }
92 
93  return mOutputBCM;
94  }

100  template <typename... axes>
101  o2::emcal::TimeCalibrationParams calibrateTime(boost::histogram::histogram<axes...>& hist, double minTime = 0, double maxTime = 1000)
102  {
103 
104  auto histReduced = boost::histogram::algorithm::reduce(hist, boost::histogram::algorithm::shrink(minTime, maxTime), boost::histogram::algorithm::shrink(0, NCELLS));
105 
106  o2::emcal::TimeCalibrationParams TCP;
107 
108  double mean = 0;
109 
110 #if (defined(WITH_OPENMP) && !defined(__CLING__))
111  if (mNThreads < 1) {
112  mNThreads = std::min(omp_get_max_threads(), NCELLS);
113  }
114  LOG(info) << "Number of threads that will be used = " << mNThreads;
115 #pragma omp parallel for num_threads(mNThreads)
116 #else
117  LOG(info) << "OPEN MP will not be used for the time calibration";
118  mNThreads = 1;
119 #endif
120 
121  for (unsigned int i = 0; i < NCELLS; ++i) {
122 
123  // project boost histogram to 1d just for 1 cell
124  auto boostHist1d = o2::utils::ProjectBoostHistoXFast(histReduced, i, i + 1);
125  // ToDo: maybe cut histo to max +- 25 ns
126 
127  try {
128  auto fitValues = o2::utils::fitBoostHistoWithGaus<double>(boostHist1d);
129  mean = fitValues.at(1);
130  // add mean to time calib params
131  TCP.addTimeCalibParam(i, mean, 0);
132  } catch (o2::utils::FitGausError_t) {
133  TCP.addTimeCalibParam(i, mean, 0); // take calib value of last cell; or 400 ns shift default value
134  }
135  }
136  return TCP;
137  }
