102 {

137 bool Tracker::processSide(bool isRight, bool isInward)
138 {
140  int firstCh = (isInward) ? 3 : 0;
141  int secondCh = (isInward) ? 2 : 1;
142  int rpcOffset1 = detparams::getDEId(isRight, firstCh, 0);
143  int rpcOffset2 = detparams::getDEId(isRight, secondCh, 0);
144 
145  // loop on RPCs in first plane
146  Track track;
147  for (int irpc = 0; irpc < 9; ++irpc) {
148  int deId1 = rpcOffset1 + irpc;
149  for (auto clIdx1 : mClusterIndexes[deId1]) {
150  // loop on clusters of the RPC in the first plane
151  auto& cl1 = mClusters[clIdx1];
152  int firstRpc = getFirstNeighbourRPC(irpc);
153  int lastRpc = getLastNeighbourRPC(irpc);
154  for (int irpc2 = firstRpc; irpc2 <= lastRpc; ++irpc2) {
155  // loop on (neighbour) RPCs in second plane
156  int deId2 = rpcOffset2 + irpc2;
157  for (auto clIdx2 : mClusterIndexes[deId2]) {
158  // loop on clusters of the RPC in the second plane
159  auto& cl2 = mClusters[clIdx2];
160 
161  if (!makeTrackSeed(track, cl1, cl2)) {
162  continue;
163  }
164 
165  track.setClusterMatchedUnchecked(firstCh, clIdx1);
166  track.setClusterMatchedUnchecked(secondCh, clIdx2);
167  track.setClusterMatchedUnchecked(3 - firstCh, -1);
168  track.setClusterMatchedUnchecked(3 - secondCh, -1);
169  std::invoke(mFollowTrack, this, track, isRight, isInward);
170  } // loop on clusters in second plane
171  } // loop on RPCs in second plane
172  } // loop on clusters in first plane
173  } // loop on RPCs in first plane
174  return true;
175 }

178 bool Tracker::makeTrackSeed(Track& track, const Cluster& cl1, const Cluster& cl2) const
179 {
181 
182  // First check if the delta_x between the two clusters is not too large
183  double dZ = cl2.zCoor - cl1.zCoor;
184  double dZ2 = dZ * dZ;
185  double nonBendingSlope = (cl2.xCoor - cl1.xCoor) / dZ;
186  double nonBendingImpactParam = std::abs(cl2.xCoor - cl2.zCoor * nonBendingSlope);
187  double nonBendingImpactParamErr = std::sqrt(
188  (cl1.zCoor * cl1.zCoor * cl2.getEX2() + cl2.zCoor * cl2.zCoor * cl1.getEX2()) / dZ2);
189  if ((nonBendingImpactParam - mSigmaCut * nonBendingImpactParamErr) > mImpactParamCut) {
190  return false;
191  }
192 
193  // Then start making the track (from 2 points)
194  track.setPosition(cl2.xCoor, cl2.yCoor, cl2.zCoor);
195  track.setDirection(nonBendingSlope, (cl2.yCoor - cl1.yCoor) / dZ, 1.);
196  track.setCovarianceParameters(cl2.getEX2(), // x-x
197  cl2.getEY2(), // y-y
198  (cl1.getEX2() + cl2.getEX2()) / dZ2, // slopeX-slopeX
199  (cl1.getEY2() + cl2.getEY2()) / dZ2, // slopeY-slopeY
200  cl2.getEX2() / dZ, // x-slopeX
201  cl2.getEY2() / dZ); // y-slopeY
202  track.setChi2(0.);
203  track.setNDF(0);
204 
205  return true;
206 }

210 {
220 
221  std::unordered_set<int> excludedClusters;
222

224 
225  // look for clusters in both chambers or in first chamber only
226  findAllClusters(track, isRight, 1, 0, 8, 0, excludedClusters, true);
227 
228  // look for clusters in the second chamber only
229  findAllClusters(track, isRight, 0, 0, 8, -1, excludedClusters, true);
230 
231  } else {
232 
233  // exclude clusters from tracks already found by inward tracking (with 4/4 clusters)
234  excludeUsedClusters(track, 1, 0, excludedClusters);
235 
236  // look for clusters in first chamber only
237  findAllClusters(track, isRight, 2, 0, 8, -1, excludedClusters, true);
238 
239  // look for clusters in the second chamber only
240  findAllClusters(track, isRight, 3, 0, 8, -1, excludedClusters, true);
241  }
242

247 bool Tracker::findAllClusters(const Track& track, bool isRight, int chamber, int firstRPC, int lastRPC, int nextChamber,
248  std::unordered_set<int>& excludedClusters, bool excludeClusters)
249 {
253 
254  int rpcOffset = detparams::getDEId(isRight, chamber, 0);
255  bool clusterFound = false;
256  Track newTrack;
257 
258  for (int irpc = firstRPC; irpc <= lastRPC; ++irpc) {
259  int deId = rpcOffset + irpc;
260  for (auto clIdx : mClusterIndexes[deId]) {
261 
262  // skip excluded clusters
263  if (excludeClusters && excludedClusters.count(clIdx) > 0) {
264  continue;
265  }
266 
267  // try to attach this cluster
268  if (!tryOneCluster(track, chamber, clIdx, newTrack)) {
269  continue;
270  }
271  clusterFound = true;
272 
273  // add this cluster to the list to be excluded later on
274  if (!excludeClusters) {
275  excludedClusters.emplace(clIdx);
276  }
277 
278  // store this track extrapolated to MT11 unless compatible clusters are found in the next chamber (if any)
279  if (nextChamber < 0 || !findAllClusters(newTrack, isRight, nextChamber, getFirstNeighbourRPC(irpc),
280  getLastNeighbourRPC(irpc), -1, excludedClusters, false)) {
281  newTrack.propagateToZ(SMT11Z);
282  mTracks.emplace_back(newTrack);
283  }
284  }
285  }
286 
287  return clusterFound;
288 }

291 bool Tracker::followTrackKeepBest(const Track& track, bool isRight, bool isInward)
292 {
295 
296  // This algorithm is works generally well,
297  // but tests show that it can fail reconstructing the generated track
298  // in case of two tracks close in space.
299  // The main issue is that we do not have charge distribution information, and the strips are large,
300  // so it can happen that the track obtained by combining the hits of the first track in MT1
301  // and of the second track in MT2 can sometimes have a better chi2 of the real tracks.
302  // Since this algorithm only keeps the best track, it might chose the fake instead of the real track.
303  // The algorithm is however fast, so it should be preferred in case of low tracks multiplicity
304 
305  std::array<int, 2> chamberOrder;
306  chamberOrder[0] = isInward ? 1 : 2;
307  chamberOrder[1] = isInward ? 0 : 3;
308 
309  Track bestTrack;
310 
311  // loop on next two chambers
312  for (int ich = 0; ich < 2; ++ich) {
313  findNextCluster(track, isRight, isInward, chamberOrder[ich], 0, 8, bestTrack);
314  if (bestTrack.getNDF() == 4) {
315  // We already have a track with 4 clusters: no need to search for a track with only one cluster
316  // in the next chamber
317  break;
318  }
319  }
320 
321  if (bestTrack.getNDF() > 0) {
322  // Extrapolate to MT11
323  bestTrack.propagateToZ(SMT11Z);
324 
325  // Add the track if it is not compatible or better than the ones we already have
326  if (tryAddTrack(bestTrack)) {
327  return true;
328  }
329  }
330 
331  return false;
332 }

335 bool Tracker::findNextCluster(const Track& track, bool isRight, bool isInward, int chamber, int firstRPC, int lastRPC,
336  Track& bestTrack) const
337 {
339  int nextChamber = (isInward) ? chamber - 1 : chamber + 1;
340  int rpcOffset = detparams::getDEId(isRight, chamber, 0);
341  Track newTrack;
342  for (int irpc = firstRPC; irpc <= lastRPC; ++irpc) {
343  int deId = rpcOffset + irpc;
344  for (auto clIdx : mClusterIndexes[deId]) {
345  if (!tryOneCluster(track, chamber, clIdx, newTrack)) {
346  continue;
347  }
348  if (nextChamber >= 0 && nextChamber <= 3) {
349  // We found a cluster in the first chamber of the station
350  // We search for a cluster in the last chamber, this time limiting to the RPC above and below this one
351  findNextCluster(newTrack, isRight, isInward, nextChamber,
352  getFirstNeighbourRPC(irpc), getLastNeighbourRPC(irpc), bestTrack);
353  }
354  if (newTrack.getNDF() > bestTrack.getNDF() ||
355  (newTrack.getNDF() == bestTrack.getNDF() && newTrack.getChi2() < bestTrack.getChi2())) {
356  // Prefer tracks with a larger number of attached clusters, even if the chi2 is worse
357  bestTrack = newTrack;
358  }
359  } // loop on clusters
360  } // loop on RPC
361 
362  return (bestTrack.getNDF() > 0);
363 }

366 bool Tracker::tryOneCluster(const Track& track, int chamber, int clIdx, Track& newTrack) const
367 {
370 
371  newTrack = track;
372  auto& cl = mClusters[clIdx];
373  newTrack.propagateToZ(cl.zCoor);
374 
375  double diff[2] = {cl.xCoor - newTrack.getPositionX(), cl.yCoor - newTrack.getPositionY()};
376  double err2[2] = {newTrack.getCovarianceParameter(Track::CovarianceParamIndex::VarX) + cl.getEX2(),
377  newTrack.getCovarianceParameter(Track::CovarianceParamIndex::VarY) + cl.getEY2()};
378  double chi2 = diff[0] * diff[0] / err2[0] + diff[1] * diff[1] / err2[1];
379  if (chi2 > mMaxChi2) {
380  return false;
381  }
382 
383  runKalmanFilter(newTrack, cl);
384  newTrack.setChi2(track.getChi2() + chi2);
385  newTrack.setNDF(track.getNDF() + 2);
386  newTrack.setClusterMatchedUnchecked(chamber, clIdx);
387 
388  return true;
389 }

391 //__________________________________________________________________________
392 void Tracker::runKalmanFilter(Track& track, const Cluster& cluster) const
393 {
396 
397  double pos[2] = {track.getPositionX(), track.getPositionY()};
398  double dir[2] = {track.getDirectionX(), track.getDirectionY()};
399  const std::array<float, 6>& covParams = track.getCovarianceParameters();
400  double clusPos[2] = {cluster.xCoor, cluster.yCoor};
401  double clusterSigma[2] = {cluster.getEX2(), cluster.getEY2()};
402 
403  std::array<float, 6> newCovParams;
404  double newPos[2], newDir[2];
405  for (int idx = 0; idx < 2; ++idx) {
406  int slopeIdx = idx + 2;
407  int covIdx = idx + 4;
408 
409  double den = clusterSigma[idx] + covParams[idx];
410  assert(den != 0.);
411  // if ( den == 0. ) return 2.*mMaxChi2;
412 
413  // Compute the new covariance
414  // cov -> (W+U)^-1
415  // where W = (old_cov)^-1
416  // and U = inverse of uncertainties of cluster
417  // s_x^2 -> s_x^2 * cl_s_x^2 / (s_x^2 + cl_s_x^2)
418  newCovParams[idx] = covParams[idx] * clusterSigma[idx] / den;
419  // cov(x,slopeX) -> cov(x,slopeX) * cl_s_x^2 / (s_x^2 + cl_s_x^2)
420  newCovParams[covIdx] = covParams[covIdx] * clusterSigma[idx] / den;
421  // s_slopeX^2 -> s_slopeX^2 - cov(x,slopeX)^2 / (s_x^2 + cl_s_x^2)
422  newCovParams[slopeIdx] = covParams[slopeIdx] - covParams[covIdx] * covParams[covIdx] / den;
423 
424  double diff = clusPos[idx] - pos[idx];
425 
426  // New parameters: p' = ((W+U)^-1)U(m-p) + p
427  // x -> x + ( cl_x - x ) * s_x^2 / (s_x^2 + cl_s_x^2)
428  newPos[idx] = pos[idx] + diff * covParams[idx] / den;
429  // slopeX -> slopeX + ( cl_x - x ) * cov(x,slopeX) / (s_x^2 + cl_s_x^2)
430  newDir[idx] = dir[idx] + diff * covParams[covIdx] / den;
431  }
432 
433  // Save the new parameters
434  track.setPosition(newPos[0], newPos[1], cluster.zCoor);
435  track.setDirection(newDir[0], newDir[1], 1.);
436 
437  // Save the new parameters covariance matrix
438  track.setCovarianceParameters(newCovParams);
439 }

442 bool Tracker::tryAddTrack(const Track& track)
443 {
449 
450  // We divide the chi2 by two since we want to consider only the uncertainty
451  // on one of the two tracks. We further reduce to 0.4 since we want to account
452  // for the case where one of the two reconstructed tracks has a much better precision
453  // of the other
454  float chi2Cut = 0.4 * mSigmaCut * mSigmaCut;
455  for (auto checkTrack = mTracks.begin() + mTrackOffset; checkTrack != mTracks.end(); ++checkTrack) {
456  int nCommonClusters = 0;
457  for (int ich = 0; ich < 4; ++ich) {
458  if (track.getClusterMatchedUnchecked(ich) == checkTrack->getClusterMatchedUnchecked(ich)) {
459  ++nCommonClusters;
460  }
461  }
462  if (nCommonClusters == 4) {
463  return false;
464  }
465  if (nCommonClusters == 3 && track.isCompatible(*checkTrack, chi2Cut)) {
466  // The new track is compatible with an existing one
467  if (track.getNDF() > checkTrack->getNDF() ||
468  (track.getNDF() == checkTrack->getNDF() && track.getChi2() < checkTrack->getChi2())) {
469  // The new track has more cluster or is more precise than the old one: replace it!
470  *checkTrack = track;
471  }
472  return false;
473  }
474  }
475 
476  // The new track is not compatible with the previous ones: keep it
477  mTracks.emplace_back(track);
478  return true;
479 }
