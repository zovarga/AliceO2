53 int MFTDCSProcessor::process(const gsl::span<const DPCOM> dps)
54 {
55 
56  // first we check which DPs are missing - if some are, it means that
57  // the delta map was sent
58  if (mVerbose) {
59  LOG(info) << "\n\n\nProcessing new TF\n-----------------";
60  }
61  if (!mStartTFset) {
62  mStartTF = mTF;
63  mStartTFset = true;
64  }
65 
66  std::unordered_map<DPID, DPVAL> mapin;
67  for (auto& it : dps) {
68  mapin[it.id] = it.data;
69  }
70  for (auto& it : mPids) {
71  const auto& el = mapin.find(it.first);
72  if (el == mapin.end()) {
73  LOG(debug) << "DP " << it.first << " not found in map";
74  } else {
75  LOG(debug) << "DP " << it.first << " found in map";
76  }
77  }
78 
79  // now we process all DPs, one by one
80  for (const auto& it : dps) {
81  // we process only the DPs defined in the configuration
82  const auto& el = mPids.find(it.id);
83  if (el == mPids.end()) {
84  LOG(info) << "DP " << it.id << " not found in MFTDCSProcessor, we will not process it";
85  continue;
86  }
87  /*
88  //it.id = DataPointIdentifier
89  //id.data = DataPointValue
90  const DPCOM new_it(new_id,it.data);
91  processDP(new_it);
92  */
93  processDP(it);
94 
95  mPids[it.id] = true;
96  }
97 
98  //updateCurrentAnalogCCDB();
99  updateDPsCCDB();
100 
101  return 0;
102 }

106 int MFTDCSProcessor::processDP(const DPCOM& dpcom)
107 {
108 
109  // processing single DP
110 
111  auto& dpid = dpcom.id;
112  const auto& type = dpid.get_type();
113  auto& val = dpcom.data;
114  if (mVerbose) {
115  if (type == RAW_DOUBLE) {
116  LOG(info);
117  LOG(info) << "Processing DP = " << dpcom << ", with value = " << o2::dcs::getValue<double>(dpcom);
118  } else if (type == RAW_INT) {
119  LOG(info);
120  LOG(info) << "Processing DP = " << dpcom << ", with value = " << o2::dcs::getValue<int32_t>(dpcom);
121  }
122  }
123 
124  auto flags = val.get_flags();
125 
126  // now I need to access the correct element
127  if (type == RAW_DOUBLE) {
128  // for these DPs, we will store the first, last, mid value, plus the value where the maximum variation occurred
129  auto& dvect = mDpsdoublesmap[dpid];
130  LOG(info) << "mDpsdoublesmap[dpid].size() = " << dvect.size();
131  auto etime = val.get_epoch_time();
132  if (dvect.size() == 0 || etime != dvect.back().get_epoch_time()) { // we check
133  // that we did not get the
134  // same timestamp as the
135  // latest one
136  dvect.push_back(val);
137  }
138  }
139 
140  return 0;
141 }

145 void MFTDCSProcessor::updateDPsCCDB()
146 {
147 
148  // here we create the object to then be sent to CCDB
149  LOG(info) << "Finalizing";
150  union Converter {
151  uint64_t raw_data;
152  double double_value;
153  } converter0, converter1;
154 
155  for (const auto& it : mPids) {
156  const auto& type = it.first.get_type();
157  if (type == o2::dcs::RAW_DOUBLE) {
158  auto& mftdcs = mMFTDCS[it.first];
159  if (it.second == true) { // we processed the DP at least 1x
160  auto& dpvect = mDpsdoublesmap[it.first];
161  mftdcs.firstValue.first = dpvect[0].get_epoch_time();
162  converter0.raw_data = dpvect[0].payload_pt1;
163  mftdcs.firstValue.second = converter0.double_value;
164  mftdcs.lastValue.first = dpvect.back().get_epoch_time();
165  converter0.raw_data = dpvect.back().payload_pt1;
166  mftdcs.lastValue.second = converter0.double_value;
167  // now I will look for the max change
168  if (dpvect.size() > 1) {
169  auto deltatime = dpvect.back().get_epoch_time() - dpvect[0].get_epoch_time();
170  if (deltatime < 60000) {
171  // if we did not cover at least 1 minute,
172  // max variation is defined as the difference between first and last value
173  converter0.raw_data = dpvect[0].payload_pt1;
174  converter1.raw_data = dpvect.back().payload_pt1;
175  double delta = std::abs(converter0.double_value - converter1.double_value);
176  mftdcs.maxChange.first = deltatime; // is it ok to do like this, as in Run 2?
177  mftdcs.maxChange.second = delta;
178  } else {
179  for (auto i = 0; i < dpvect.size() - 1; ++i) {
180  for (auto j = i + 1; j < dpvect.size(); ++j) {
181  auto deltatime = dpvect[j].get_epoch_time() - dpvect[i].get_epoch_time();
182  if (deltatime >= 60000) { // we check every min; epoch_time in ms
183  converter0.raw_data = dpvect[i].payload_pt1;
184  converter1.raw_data = dpvect[j].payload_pt1;
185  double delta = std::abs(converter0.double_value - converter1.double_value);
186  if (delta > mftdcs.maxChange.second) {
187  mftdcs.maxChange.first = deltatime; // is it ok to do like this, as in Run 2?
188  mftdcs.maxChange.second = delta;
189  }
190  }
191  }
192  }
193  }
194  // mid point
195  auto midIdx = dpvect.size() / 2 - 1;
196  mftdcs.midValue.first = dpvect[midIdx].get_epoch_time();
197  converter0.raw_data = dpvect[midIdx].payload_pt1;
198  mftdcs.midValue.second = converter0.double_value;
199  } else {
200  mftdcs.maxChange.first = dpvect[0].get_epoch_time();
201  converter0.raw_data = dpvect[0].payload_pt1;
202  mftdcs.maxChange.second = converter0.double_value;
203  mftdcs.midValue.first = dpvect[0].get_epoch_time();
204  converter0.raw_data = dpvect[0].payload_pt1;
205  mftdcs.midValue.second = converter0.double_value;
206  }
207  }
208  if (mVerbose) {
209  LOG(info) << "PID = " << it.first.get_alias();
210  mftdcs.print();
211  }
212  }
213  }
214 
215  std::map<std::string, std::string> md;
216  md["responsible"] = "Satoshi Yano";
217  prepareCCDBobjectInfo(mMFTDCS, mccdbDPsInfo, "MFT/Condition/DCSDPs", mTF, md);
218 
219  return;
220 }
