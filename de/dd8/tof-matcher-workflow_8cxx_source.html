62 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
63 {
64  WorkflowSpec specs;
65 
66  // Update the (declared) parameters if changed from the command line
67  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
68  // write the configuration used for the workflow
69  o2::conf::ConfigurableParam::writeINI("o2match-tof-workflow_configuration.ini");
70 
71  auto useMC = !configcontext.options().get<bool>("disable-mc");
72  auto disableRootIn = configcontext.options().get<bool>("disable-root-input");
73  auto disableRootOut = configcontext.options().get<bool>("disable-root-output");
74  auto useFIT = configcontext.options().get<bool>("use-fit");
75  auto useCCDB = configcontext.options().get<bool>("use-ccdb");
76  auto strict = configcontext.options().get<bool>("strict-matching");
77  auto diagnostic = configcontext.options().get<bool>("enable-dia");
78 
79  bool writematching = 0;
80  bool writecalib = 0;
81  auto outputType = configcontext.options().get<std::string>("output-type");
82  if (outputType.rfind("matching-info") < outputType.size()) {
83  writematching = 1;
84  }
85  if (outputType.rfind("calib-info") < outputType.size()) {
86  writecalib = 1;
87  }
88 
89  if (!writecalib) {
90  useFIT = false;
91  }
92 
93  LOG(debug) << "TOF MATCHER WORKFLOW configuration";
94  LOG(debug) << "TOF track inputs = " << configcontext.options().get<std::string>("track-sources");
95  LOG(debug) << "TOF output = " << outputType;
96  LOG(debug) << "TOF disable-mc = " << configcontext.options().get<std::string>("disable-mc");
97  LOG(debug) << "TOF use-ccdb = " << useCCDB;
98  LOG(debug) << "TOF use-fit = " << useFIT;
99  LOG(debug) << "TOF disable-root-input = " << disableRootIn;
100  LOG(debug) << "TOF disable-root-output = " << disableRootOut;
101  LOG(debug) << "TOF matching in strict mode = " << strict;
102 
103  //GID::mask_t alowedSources = GID::getSourcesMask("TPC,ITS-TPC");
104  GID::mask_t alowedSources = GID::getSourcesMask("TPC,ITS-TPC,TPC-TRD,ITS-TPC-TRD");
105 
106  GID::mask_t src = alowedSources & GID::getSourcesMask(configcontext.options().get<std::string>("track-sources"));
107  if (strict && (src & ~GID::getSourcesMask("TPC,TPC-TRD")).any()) {
108  LOGP(warning, "In strict matching mode only TPC and TPC-TRD sources allowed, {} asked, redefining", GID::getSourcesNames(src));
109  src &= GID::getSourcesMask("TPC,TPC-TRD");
110  }
111  GID::mask_t mcmaskcl;
112  GID::mask_t nonemask = GID::getSourcesMask(GID::NONE);
113  GID::mask_t clustermask = GID::getSourcesMask("TOF");
114  if (useFIT) {
115  clustermask |= GID::getSourceMask(GID::FT0);
116  }
117 
118  if (useMC) {
119  mcmaskcl |= GID::getSourceMask(GID::TOF);
120  }
121  // pass strict flag to fetch eventual TPC-TRD input with correct subspec
122  o2::globaltracking::InputHelper::addInputSpecs(configcontext, specs, clustermask, nonemask, src, useMC, mcmaskcl, GID::getSourcesMask(GID::ALL), strict);
123 
124  specs.emplace_back(o2::globaltracking::getTOFMatcherSpec(src, useMC, useFIT, false, strict)); // doTPCrefit not yet supported (need to load TPC clusters?)
125 
126  if (!disableRootOut) {
127  if (writematching) {
128  if (GID::includesSource(GID::TPC, src)) { // matching to TPC was requested
129  specs.emplace_back(o2::tof::getTOFMatchedWriterSpec(useMC, "o2match_tof_tpc.root", true, (int)o2::dataformats::MatchInfoTOFReco::TrackType::TPC, strict));
130  }
131  if (GID::includesSource(GID::ITSTPC, src)) { // matching to ITS-TPC was requested, there is not strict mode in this case
132  specs.emplace_back(o2::tof::getTOFMatchedWriterSpec(useMC, "o2match_tof_itstpc.root", false, (int)o2::dataformats::MatchInfoTOFReco::TrackType::ITSTPC, false));
133  }
134  if (GID::includesSource(GID::TPCTRD, src)) { // matching to TPC-TRD was requested, there is not strict mode in this case
135  specs.emplace_back(o2::tof::getTOFMatchedWriterSpec(useMC, "o2match_tof_tpctrd.root", false, (int)o2::dataformats::MatchInfoTOFReco::TrackType::TPCTRD, strict));
136  }
137  if (GID::includesSource(GID::ITSTPCTRD, src)) { // matching to ITS-TPC-TRD was requested, there is not strict mode in this case
138  specs.emplace_back(o2::tof::getTOFMatchedWriterSpec(useMC, "o2match_tof_itstpctrd.root", false, (int)o2::dataformats::MatchInfoTOFReco::TrackType::ITSTPCTRD, false));
139  }
140  }
141  if (writecalib) {
142  specs.emplace_back(o2::tof::getTOFCalibWriterSpec("o2calib_tof.root", 0, diagnostic));
143  }
144  }
145 
146  // configure dpl timer to inject correct firstTFOrbit: start from the 1st orbit of TF containing 1st sampled orbit
147  o2::raw::HBFUtilsInitializer hbfIni(configcontext, specs);
148 
149  return std::move(specs);
150 }
