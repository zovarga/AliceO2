28 void Clusterer::process(int nThreads, PixelReader& reader, CompClusCont* compClus,
29  PatternCont* patterns, ROFRecCont* vecROFRec, MCTruth* labelsCl)
30 {
31 #ifdef _PERFORM_TIMING_
32  mTimer.Start(kFALSE);
33 #endif
34  if (nThreads < 1) {
35  nThreads = 1;
36  }
37  auto autoDecode = reader.getDecodeNextAuto();
38  do {
39  if (autoDecode) {
40  reader.setDecodeNextAuto(false); // internally do not autodecode
41  if (!reader.decodeNextTrigger()) {
42  break; // on the fly decoding was requested, but there were no data left
43  }
44  }
45  if (reader.getInteractionRecord().isDummy()) {
46  continue; // No IR info was found
47  }
48  // pre-fetch all non-empty chips of current ROF
49  ChipPixelData* curChipData = nullptr;
50  mFiredChipsPtr.clear();
51  size_t nPix = 0;
52  while ((curChipData = reader.getNextChipData(mChips))) {
53  mFiredChipsPtr.push_back(curChipData);
54  nPix += curChipData->getData().size();
55  }
56 
57  auto& rof = vecROFRec->emplace_back(reader.getInteractionRecord(), 0, compClus->size(), 0); // create new ROF
58 
59  uint16_t nFired = mFiredChipsPtr.size();
60  if (!nFired) {
61  if (autoDecode) {
62  continue;
63  }
64  break; // just 1 ROF was asked to be processed
65  }
66  if (nFired < nThreads) {
67  nThreads = nFired;
68  }
69 #ifdef WITH_OPENMP
70  omp_set_num_threads(nThreads);
71 #else
72  nThreads = 1;
73 #endif
74  uint16_t chipStep = nThreads > 1 ? (nThreads == 2 ? 20 : (nThreads < 5 ? 5 : 1)) : nFired;
75  int dynGrp = std::min(4, std::max(1, nThreads / 2));
76  if (nThreads > mThreads.size()) {
77  int oldSz = mThreads.size();
78  mThreads.resize(nThreads);
79  for (int i = oldSz; i < nThreads; i++) {
80  mThreads[i] = std::make_unique<ClustererThread>(this);
81  }
82  }
83 #ifdef WITH_OPENMP
84 #pragma omp parallel for schedule(dynamic, dynGrp)
85  //>> start of MT region
86  for (uint16_t ic = 0; ic < nFired; ic += chipStep) {
87  auto ith = omp_get_thread_num();
88  if (nThreads > 1) {
89  mThreads[ith]->process(ic, std::min(chipStep, uint16_t(nFired - ic)),
90  &mThreads[ith]->compClusters,
91  patterns ? &mThreads[ith]->patterns : nullptr,
92  labelsCl ? reader.getDigitsMCTruth() : nullptr,
93  labelsCl ? &mThreads[ith]->labels : nullptr, rof);
94  } else { // put directly to the destination
95  mThreads[0]->process(0, nFired, compClus, patterns, labelsCl ? reader.getDigitsMCTruth() : nullptr, labelsCl, rof);
96  }
97  }
98  //<< end of MT region
99 #else
100  mThreads[0]->process(0, nFired, compClus, patterns, labelsCl ? reader.getDigitsMCTruth() : nullptr, labelsCl, rof);
101 #endif
102  // copy data of all threads but the 1st one to final destination
103  if (nThreads > 1) {
104 #ifdef _PERFORM_TIMING_
105  mTimerMerge.Start(false);
106 #endif
107  size_t nClTot = 0, nPattTot = 0;
108  int chid = 0, thrStatIdx[nThreads];
109  for (int ith = 0; ith < nThreads; ith++) {
110  thrStatIdx[ith] = 0;
111  nClTot += mThreads[ith]->compClusters.size();
112  nPattTot += mThreads[ith]->patterns.size();
113  }
114  compClus->reserve(nClTot);
115  if (patterns) {
116  patterns->reserve(nPattTot);
117  }
118  while (chid < nFired) {
119  for (int ith = 0; ith < nThreads; ith++) {
120  if (thrStatIdx[ith] >= mThreads[ith]->stats.size()) {
121  continue;
122  }
123  const auto& stat = mThreads[ith]->stats[thrStatIdx[ith]];
124  if (stat.firstChip == chid) {
125  thrStatIdx[ith]++;
126  chid += stat.nChips; // next chip to look
127  const auto clbeg = mThreads[ith]->compClusters.begin() + stat.firstClus;
128  auto szold = compClus->size();
129  compClus->insert(compClus->end(), clbeg, clbeg + stat.nClus);
130  if (patterns) {
131  const auto ptbeg = mThreads[ith]->patterns.begin() + stat.firstPatt;
132  patterns->insert(patterns->end(), ptbeg, ptbeg + stat.nPatt);
133  }
134  if (labelsCl) {
135  labelsCl->mergeAtBack(mThreads[ith]->labels, stat.firstClus, stat.nClus);
136  }
137  }
138  }
139  }
140  for (int ith = 0; ith < nThreads; ith++) {
141  mThreads[ith]->patterns.clear();
142  mThreads[ith]->compClusters.clear();
143  mThreads[ith]->labels.clear();
144  mThreads[ith]->stats.clear();
145  }
146 #ifdef _PERFORM_TIMING_
147  mTimerMerge.Stop();
148 #endif
149  } else {
150  mThreads[0]->stats.clear();
151  }
152  rof.setNEntries(compClus->size() - rof.getFirstEntry()); // update
153  } while (autoDecode);
154  reader.setDecodeNextAuto(autoDecode); // restore setting
155 #ifdef _PERFORM_TIMING_
156  mTimer.Stop();
157 #endif
158 }

161 void Clusterer::ClustererThread::process(uint16_t chip, uint16_t nChips, CompClusCont* compClusPtr, PatternCont* patternsPtr,
162  const ConstMCTruth* labelsDigPtr, MCTruth* labelsClPtr, const ROFRecord& rofPtr)
163 {
164  if (stats.empty() || stats.back().firstChip + stats.back().nChips < chip) { // there is a jump, register new block
165  stats.emplace_back(ThreadStat{chip, 0, uint32_t(compClusPtr->size()), patternsPtr ? uint32_t(patternsPtr->size()) : 0, 0, 0});
166  }
167  for (int ic = 0; ic < nChips; ic++) {
168  auto* curChipData = parent->mFiredChipsPtr[chip + ic];
169  auto chipID = curChipData->getChipID();
170  if (parent->mMaxBCSeparationToMask > 0) { // mask pixels fired from the previous ROF
171  const auto& chipInPrevROF = parent->mChipsOld[chipID];
172  if (std::abs(rofPtr.getBCData().differenceInBC(chipInPrevROF.getInteractionRecord())) < parent->mMaxBCSeparationToMask) {
173  parent->mMaxRowColDiffToMask ? curChipData->maskFiredInSample(parent->mChipsOld[chipID], parent->mMaxRowColDiffToMask) : curChipData->maskFiredInSample(parent->mChipsOld[chipID]);
174  }
175  }
176  auto nclus0 = compClusPtr->size();
177  auto validPixID = curChipData->getFirstUnmasked();
178  auto npix = curChipData->getData().size();
179  if (validPixID < npix) { // chip data may have all of its pixels masked!
180  auto valp = validPixID++;
181  if (validPixID == npix) { // special case of a single pixel fired on the chip
182  finishChipSingleHitFast(valp, curChipData, compClusPtr, patternsPtr, labelsDigPtr, labelsClPtr);
183  } else {
184  initChip(curChipData, valp);
185  for (; validPixID < npix; validPixID++) {
186  if (!curChipData->getData()[validPixID].isMasked()) {
187  updateChip(curChipData, validPixID);
188  }
189  }
190  finishChip(curChipData, compClusPtr, patternsPtr, labelsDigPtr, labelsClPtr);
191  }
192  }
193  if (parent->mMaxBCSeparationToMask > 0) { // current chip data will be used in the next ROF to mask overflow pixels
194  parent->mChipsOld[chipID].swap(*curChipData);
195  }
196  }
197  auto& currStat = stats.back();
198  currStat.nChips += nChips;
199  currStat.nClus = compClusPtr->size() - currStat.firstClus;
200  currStat.nPatt = patternsPtr ? (patternsPtr->size() - currStat.firstPatt) : 0;
201 }

204 void Clusterer::ClustererThread::finishChip(ChipPixelData* curChipData, CompClusCont* compClusPtr,
205  PatternCont* patternsPtr, const ConstMCTruth* labelsDigPtr, MCTruth* labelsClusPtr)
206 {
207  auto clustersCount = compClusPtr->size();
208  const auto& pixData = curChipData->getData();
209  for (int i1 = 0; i1 < preClusterHeads.size(); ++i1) {
210  auto ci = preClusterIndices[i1];
211  if (ci < 0) {
212  continue;
213  }
214  BBox bbox(curChipData->getChipID());
215  int nlab = 0;
216  int next = preClusterHeads[i1];
217  pixArrBuff.clear();
218  while (next >= 0) {
219  const auto& pixEntry = pixels[next];
220  const auto pix = pixData[pixEntry.second];
221  pixArrBuff.push_back(pix); // needed for cluster topology
222  bbox.adjust(pix.getRowDirect(), pix.getCol());
223  if (labelsClusPtr) { // the MCtruth for this pixel is at curChipData->startID+pixEntry.second
224  fetchMCLabels(pixEntry.second + curChipData->getStartID(), labelsDigPtr, nlab);
225  }
226  next = pixEntry.first;
227  }
228  preClusterIndices[i1] = -1;
229  for (int i2 = i1 + 1; i2 < preClusterHeads.size(); ++i2) {
230  if (preClusterIndices[i2] != ci) {
231  continue;
232  }
233  next = preClusterHeads[i2];
234  while (next >= 0) {
235  const auto& pixEntry = pixels[next];
236  const auto pix = pixData[pixEntry.second]; // PixelData
237  pixArrBuff.push_back(pix); // needed for cluster topology
238  bbox.adjust(pix.getRowDirect(), pix.getCol());
239  if (labelsClusPtr) { // the MCtruth for this pixel is at curChipData->startID+pixEntry.second
240  fetchMCLabels(pixEntry.second + curChipData->getStartID(), labelsDigPtr, nlab);
241  }
242  next = pixEntry.first;
243  }
244  preClusterIndices[i2] = -1;
245  }
246  if (bbox.isAcceptableSize()) {
247  parent->streamCluster(pixArrBuff, &labelsBuff, bbox, parent->mPattIdConverter, compClusPtr, patternsPtr, labelsClusPtr, nlab);
248  } else {
249  auto warnLeft = MaxHugeClusWarn - parent->mNHugeClus;
250  if (warnLeft > 0) {
251  LOGP(alarm, "Splitting a huge cluster: chipID {}, rows {}:{} cols {}:{}{}", bbox.chipID, bbox.rowMin, bbox.rowMax, bbox.colMin, bbox.colMax,
252  warnLeft == 1 ? " (Further warnings will be muted)" : "");
253 #ifdef WITH_OPENMP
254 #pragma omp critical
255 #endif
256  {
257  parent->mNHugeClus++;
258  }
259  }
260  BBox bboxT(bbox); // truncated box
261  std::vector<PixelData> pixbuf;
262  do {
263  bboxT.rowMin = bbox.rowMin;
264  bboxT.colMax = std::min(bbox.colMax, uint16_t(bboxT.colMin + o2::itsmft::ClusterPattern::MaxColSpan - 1));
265  do { // Select a subset of pixels fitting the reduced bounding box
266  bboxT.rowMax = std::min(bbox.rowMax, uint16_t(bboxT.rowMin + o2::itsmft::ClusterPattern::MaxRowSpan - 1));
267  for (const auto& pix : pixArrBuff) {
268  if (bboxT.isInside(pix.getRowDirect(), pix.getCol())) {
269  pixbuf.push_back(pix);
270  }
271  }
272  if (!pixbuf.empty()) { // Stream a piece of cluster only if the reduced bounding box is not empty
273  parent->streamCluster(pixbuf, &labelsBuff, bboxT, parent->mPattIdConverter, compClusPtr, patternsPtr, labelsClusPtr, nlab, true);
274  pixbuf.clear();
275  }
276  bboxT.rowMin = bboxT.rowMax + 1;
277  } while (bboxT.rowMin < bbox.rowMax);
278  bboxT.colMin = bboxT.colMax + 1;
279  } while (bboxT.colMin < bbox.colMax);
280  }
281  }
282 }

285 void Clusterer::ClustererThread::finishChipSingleHitFast(uint32_t hit, ChipPixelData* curChipData, CompClusCont* compClusPtr,
286  PatternCont* patternsPtr, const ConstMCTruth* labelsDigPtr, MCTruth* labelsClusPtr)
287 {
288  auto clustersCount = compClusPtr->size();
289  auto pix = curChipData->getData()[hit];
290  uint16_t row = pix.getRowDirect(), col = pix.getCol();
291 
292  if (labelsClusPtr) { // MC labels were requested
293  int nlab = 0;
294  fetchMCLabels(curChipData->getStartID() + hit, labelsDigPtr, nlab);
295  auto cnt = compClusPtr->size();
296  for (int i = nlab; i--;) {
297  labelsClusPtr->addElement(cnt, labelsBuff[i]);
298  }
299  }
300 
301  // add to compact clusters, which must be always filled
302  unsigned char patt[ClusterPattern::MaxPatternBytes]{0x1 << (7 - (0 % 8))}; // unrolled 1 hit version of full loop in finishChip
303  uint16_t pattID = (parent->mPattIdConverter.size() == 0) ? CompCluster::InvalidPatternID : parent->mPattIdConverter.findGroupID(1, 1, patt);
304  if ((pattID == CompCluster::InvalidPatternID || parent->mPattIdConverter.isGroup(pattID)) && patternsPtr) {
305  patternsPtr->emplace_back(1); // rowspan
306  patternsPtr->emplace_back(1); // colspan
307  patternsPtr->insert(patternsPtr->end(), std::begin(patt), std::begin(patt) + 1);
308  }
309  compClusPtr->emplace_back(row, col, pattID, curChipData->getChipID());
310 }

324 void Clusterer::ClustererThread::initChip(const ChipPixelData* curChipData, uint32_t first)
325 {
326  // init chip with the 1st unmasked pixel (entry "from" in the mChipData)
327  prev = column1 + 1;
328  curr = column2 + 1;
329  resetColumn(curr);
330 
331  pixels.clear();
332  preClusterHeads.clear();
333  preClusterIndices.clear();
334  auto pix = curChipData->getData()[first];
335  currCol = pix.getCol();
336  curr[pix.getRowDirect()] = 0; // can use getRowDirect since the pixel is not masked
337  // start the first pre-cluster
338  preClusterHeads.push_back(0);
339  preClusterIndices.push_back(0);
340  pixels.emplace_back(-1, first); // id of current pixel
341  noLeftCol = true; // flag that there is no column on the left to check yet

345 void Clusterer::ClustererThread::updateChip(const ChipPixelData* curChipData, uint32_t ip)
346 {
347  const auto pix = curChipData->getData()[ip];
348  uint16_t row = pix.getRowDirect(); // can use getRowDirect since the pixel is not masked
349  if (currCol != pix.getCol()) { // switch the buffers
350  swapColumnBuffers();
351  resetColumn(curr);
352  noLeftCol = false;
353  if (pix.getCol() > currCol + 1) {
354  // no connection with previous column, this pixel cannot belong to any of the
355  // existing preclusters, create a new precluster and flag to check only the row above for next pixels of this column
356  currCol = pix.getCol();
357  addNewPrecluster(ip, row);
358  noLeftCol = true;
359  return;
360  }
361  currCol = pix.getCol();
362  }
363 
364  Bool_t orphan = true;
365 
366  if (noLeftCol) { // check only the row above
367  if (curr[row - 1] >= 0) {
368  expandPreCluster(ip, row, curr[row - 1]); // attach to the precluster of the previous row
369  return;
370  }
371  } else {
372 #ifdef _ALLOW_DIAGONAL_ALPIDE_CLUSTERS_
373  int neighbours[]{curr[row - 1], prev[row], prev[row + 1], prev[row - 1]};
374 #else
375  int neighbours[]{curr[row - 1], prev[row]};
376 #endif
377  for (auto pci : neighbours) {
378  if (pci < 0) {
379  continue;
380  }
381  if (orphan) {
382  expandPreCluster(ip, row, pci); // attach to the adjascent precluster
383  orphan = false;
384  continue;
385  }
386  // reassign precluster index to smallest one
387  if (preClusterIndices[pci] < preClusterIndices[curr[row]]) {
388  preClusterIndices[curr[row]] = preClusterIndices[pci];
389  } else {
390  preClusterIndices[pci] = preClusterIndices[curr[row]];
391  }
392  }
393  }
394  if (orphan) {
395  addNewPrecluster(ip, row); // start new precluster
396  }
397 }

400 void Clusterer::ClustererThread::fetchMCLabels(int digID, const ConstMCTruth* labelsDig, int& nfilled)
401 {
402  // transfer MC labels to cluster
403  if (nfilled >= MaxLabels) {
404  return;
405  }

407  for (int i = lbls.size(); i--;) {
408  int ic = nfilled;
409  for (; ic--;) { // check if the label is already present
410  if (labelsBuff[ic] == lbls[i]) {
411  return; // label is found, do nothing
412  }
413  }
414  labelsBuff[nfilled++] = lbls[i];
415  if (nfilled >= MaxLabels) {
416  break;
417  }
418  }
