37 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
38 {
39  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
40 
41  auto trigRecFilterActive = configcontext.options().get<bool>("filter-trigrec");
42  // MC labels are passed through for the global tracking downstream
43  // in case ROOT output is requested the tracklet labels are duplicated
44  bool useMC = !configcontext.options().get<bool>("disable-mc");
45  int timestamp = configcontext.options().get<int>("timestamp");
46 
47  WorkflowSpec spec;
48 
49  if (!configcontext.options().get<bool>("disable-root-input")) {
50  // cannot use InputHelper here, since we have to create the calibrated tracklets first
51  spec.emplace_back(o2::trd::getTRDTrackletReaderSpec(useMC, false));
52  }
53 
54  if (trigRecFilterActive) {
55  o2::globaltracking::InputHelper::addInputSpecsIRFramesITS(configcontext, spec);
56  }
57 
58  spec.emplace_back(o2::trd::getTRDTrackletTransformerSpec(trigRecFilterActive, timestamp));
59 
60  if (!configcontext.options().get<bool>("disable-root-output")) {
61  spec.emplace_back(o2::trd::getTRDCalibratedTrackletWriterSpec(useMC));
62  }
63 
64  return spec;
65 }
