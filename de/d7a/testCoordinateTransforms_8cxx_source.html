44 BOOST_AUTO_TEST_CASE(LocaltoRCTest)
45 {
46  auto mGeo = o2::trd::Geometry::instance();
47  mGeo->createPadPlaneArray();
48  mGeo->createClusterMatrixArray();
49 
50  int hcid = 776;
51  // This C1 chamber has 16 pad rows with I pad length = 90mm and O pad length = 75mm
52  int detector = hcid / 2;
53  int stack = mGeo->getStack(detector);
54  int layer = mGeo->getLayer(detector);
55 
56  auto padPlane = mGeo->getPadPlane(layer, stack);
57  double lengthIPad = padPlane->getLengthIPad();
58  double lengthOPad = padPlane->getLengthOPad();
59 
60  double padIWidth = padPlane->getWidthIPad();
61  double padOWidth = padPlane->getWidthOPad();
62 
63  double tiltingAngle = padPlane->getTiltingAngle();
64 
65  // Test padrows
66  auto p1 = padPlane->getPadRow(0);
67  // Center of the chamber. This should return the lower edge of padrow 8.
68  // Since we are using float values, the lower edge of padrow 8 correspond with float value 8.0.
69  testRCPoint(p1, 8.);
70 
71  auto p2 = padPlane->getPadRow(lengthIPad / 2.);
72  // With an I pad length of 9 cm, 9. / 2. should put us half way into the preceeding pad (pad 7) since padrow number
73  // decreses in positive z direction.
74  testRCPoint(p2, 7.5);
75 
76  auto p3 = padPlane->getPadRow(-lengthIPad / 2.);
77  // Same as above but in the other direction.
78  testRCPoint(p3, 8.5);
79 
80  auto p4 = padPlane->getPadRow(lengthIPad * 4.2);
81  // Arbitrary distance in z.
82  testRCPoint(p4, 8 - 4.2);
83 
84  auto p5 = padPlane->getPadRow(lengthIPad * 7 + lengthOPad);
85  // Lower border case. Take center and add 7 pads * 9 cm + 1 pad * 7.5 cm.
86  testRCPoint(p5, 0);
87 
88  auto p6 = padPlane->getPadRow(-lengthIPad * 7 - lengthOPad);
89  // Upper border case. Take center and subtract 7 pads * 9 cm - 1 pad * 7.5 cm.
90  testRCPoint(p6, 16);
91 
92  // Test pads with pad tilting
93  double p13 = padPlane->getPad(0, 0.01);
94  // Center of chamber plus epsilon (in z). Note the discontinuity in pad vs. z at z=0. This puts us at a point on the lower
95  // (in z direction) end of padrow 7. Since we have a pad tilt of -2 deg (pads tilted clockwise for particle coming
96  // from interaction vertex). After a lot thought and doodles on Miro (https://miro.com/app/board/o9J_lKgybMc=/) we
97  // find that we expect a small negative offset which would place us in the upper half of pad 71.
98  // To calculate that offset, we multiply the tangent of the tilting angle by the distance that our point is away
99  // from the center of its local padrow. Some unit conversions are also neccessary...
100  testRCPoint(p13, 72 + TMath::Tan(TMath::DegToRad() * tiltingAngle) * (0.5 * lengthIPad - 0.01) / padIWidth);
101 
102  double p14 = padPlane->getPad(0, -lengthIPad / 2.);
103  // Move from center of chamber to to center of padrow 8 = 8.5.
104  // This should now place us as edge of pad 72 = 72.0 since no offset is applied from pad tilt.
105  testRCPoint(p14, 72);
106 
107  double p15 = padPlane->getPad(0, lengthIPad / 2.);
108  // Should be the same in the other direction at padrow = 7.5
109  testRCPoint(p15, 72);
110 
111  double p16 = padPlane->getPad(padIWidth * 42, lengthIPad / 2.);
112  // Adding an arbitrary number of pads should just increase position by same number
113  testRCPoint(p16, 72 + 42);
114 
115  double p17 = padPlane->getPad(padIWidth * 42, -lengthIPad * 4 - 2.3);
116  // Moving to arbitrary point in both y and z. In y, we are 42 pads from the center which would be 72 + 42 = 114
117  // if we were in the center of the padrow and no pad tilting is considered. However, we are not in the center of
118  // a padrow, but rather 2.3 cm into padrow 8 + 4 = 12. This puts us below the center which is at 4.5 cm
119  // into the padrow and therefore, we expect a small postiive offset since the pad tilting angle is -2 deg.
120  testRCPoint(p17, 72 + 42 + TMath::Tan(TMath::DegToRad() * tiltingAngle) * (2.3 - 0.5 * lengthIPad) / padIWidth);
121 
122  double p18 = padPlane->getPad(padIWidth * 71 + padOWidth, lengthIPad / 2.);
123  // Border case right on the upper edge of the padrow in y direction and at the center of the padrow in z
124  testRCPoint(p18, 144);
125 
126  double p19 = padPlane->getPad(-padIWidth * 71 - padOWidth, lengthIPad / 2.);
127  // Border case right on the lower edge of the padrow in y direction and at the center of the padrow in z
128  testRCPoint(p19, 0);
129 
130  double p20 = padPlane->getPad(0, lengthIPad * 7 + 1.5);
131  // Ensure that shorter length of outer padrows is considered correctly
132  testRCPoint(p20, 72 + TMath::Tan(TMath::DegToRad() * tiltingAngle) * (0.5 * lengthOPad - 1.5) / padIWidth);
133 }
