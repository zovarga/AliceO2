139  //________________________________________________________________
140  void run(o2::framework::ProcessingContext& pc) final
141  {
142  auto tfid = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime;
143  auto dps = pc.inputs().get<gsl::span<DPCOM>>("input");
144 
145  mProcessor->setTF(tfid);
146  mProcessor->process(dps);
147 
148  auto timeNow = HighResClock::now();
149  Duration elapsedTime = timeNow - mTimer; // in seconds
150 
151  LOG(info) << "mDPsUpdateInterval " << mDPsUpdateInterval << "[sec.]";
152 
153  if (elapsedTime.count() >= mDPsUpdateInterval) {
154  sendDPsoutput(pc.outputs());
155  mTimer = timeNow;
156  }
157  }
158 
159  //________________________________________________________________
160  void endOfStream(o2::framework::EndOfStreamContext& ec) final
161  {
162  sendDPsoutput(ec.outputs());
163  }

165  private:
166  std::unique_ptr<MFTDCSProcessor> mProcessor;

168  int64_t mDPsUpdateInterval;
169 
170  long mStart;
171  long mEnd;
172 
173  //________________________________________________________________
174  void sendDPsoutput(DataAllocator& output)
175  {
176  // extract CCDB infos and calibration object for DPs
177  mProcessor->updateDPsCCDB();
178  const auto& payload = mProcessor->getMFTDPsInfo();
179  auto& info = mProcessor->getccdbDPsInfo();
180 
181  long tstart = mStart;
182  long tend = mEnd;
183 
184  if (tstart == -1) {
185  tstart = o2::ccdb::getCurrentTimestamp();
186  }
187 
188  if (tend == -1) {
189  constexpr long SECONDSPERYEAR = 365 * 24 * 60 * 60;
190  tend = o2::ccdb::getFutureTimestamp(SECONDSPERYEAR);
191  }
192 
193  info.setStartValidityTimestamp(tstart);
194  info.setEndValidityTimestamp(tend);
195 
196  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
197  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
198  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
199  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "MFT_DCSDPs", 0}, *image.get());
200  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "MFT_DCSDPs", 0}, info);
201  mProcessor->clearDPsinfo();
202  }
