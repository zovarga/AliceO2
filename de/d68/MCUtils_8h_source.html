73 inline bool isStable(int pdg)
74 {
75  //
76  // Decide whether particle (pdg) is stable
77  //
78 
79  // All ions/nucleons are considered as stable
80  // Nuclear code is 10LZZZAAAI
81  if (pdg > 1000000000) {
82  return true;
83  }
84 
85  const int kNstable = 18;
86  int pdgStable[kNstable] = {
87  kGamma, // Photon
88  kElectron, // Electron
89  kMuonPlus, // Muon
90  kPiPlus, // Pion
91  kKPlus, // Kaon
92  kK0Short, // K0s
93  kK0Long, // K0l
94  kProton, // Proton
95  kNeutron, // Neutron
96  kLambda0, // Lambda_0
97  kSigmaMinus, // Sigma Minus
98  kSigmaPlus, // Sigma Plus
99  3312, // Xsi Minus
100  3322, // Xsi
101  3334, // Omega
102  kNuE, // Electron Neutrino
103  kNuMu, // Muon Neutrino
104  kNuTau // Tau Neutrino
105  };
106 
107  // this is linear search ---> a hash map or binary search should be more appropriate??
108  for (int i = 0; i < kNstable; ++i) {
109  if (pdg == std::abs(pdgStable[i])) {
110  return true;
111  }
112  }
113  return false;
114 }
