56 void TrackerDPL::init(InitContext& ic)
57 {
58  mTimer.Stop();
59  mTimer.Reset();
60 
61  auto* chainITS = mRecChain->AddChain<o2::gpu::GPUChainITS>();
62  mVertexer = std::make_unique<Vertexer>(chainITS->GetITSVertexerTraits());
63  mTracker = std::make_unique<Tracker>(new TrackerTraitsCPU);
64 
65  auto filename = ic.options().get<std::string>("grp-file");
66  const auto grp = parameters::GRPObject::loadFrom(filename);
67  if (grp) {
68  mGRP.reset(grp);
69  base::Propagator::initFieldFromGRP(grp);
70  auto field = static_cast<field::MagneticField*>(TGeoGlobalMagField::Instance()->GetField());
71 
72  base::GeometryManager::loadGeometry();
73  GeometryTGeo* geom = GeometryTGeo::Instance();
74  geom->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L, o2::math_utils::TransformType::T2GRot,
75  o2::math_utils::TransformType::T2G));
76 
77  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
78  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
79  if (o2::utils::Str::pathExists(matLUTFile)) {
80  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
81  o2::base::Propagator::Instance()->setMatLUT(lut);
82  mTracker->setCorrType(o2::base::PropagatorImpl<float>::MatCorrType::USEMatCorrLUT);
83  LOG(info) << "Loaded material LUT from " << matLUTFile;
84  } else {
85  LOG(info) << "Material LUT " << matLUTFile << " file is absent, only heuristic material correction can be used";
86  }
87 
88  std::vector<TrackingParameters> trackParams;
89  std::vector<MemoryParameters> memParams;
90 
91  mRunVertexer = true;
92  if (mMode == "async") {
93 
94  trackParams.resize(2);
95  trackParams[1].TrackletMinPt = 0.2f;
96  trackParams[1].CellDeltaTanLambdaSigma *= 2.;
97  trackParams[1].MinTrackLength = 4;
98  memParams.resize(2);
99  LOG(info) << "Initializing tracker in async. phase reconstruction with " << trackParams.size() << " passes";
100 
101  } else if (mMode == "sync_misaligned") {
102 
103  trackParams.resize(1);
104  trackParams[0].PhiBins = 32;
105  trackParams[0].ZBins = 64;
106  trackParams[0].CellDeltaTanLambdaSigma *= 10;
107  trackParams[0].LayerMisalignment[0] = 3.e-2;
108  trackParams[0].LayerMisalignment[1] = 3.e-2;
109  trackParams[0].LayerMisalignment[2] = 3.e-2;
110  trackParams[0].LayerMisalignment[3] = 1.e-1;
111  trackParams[0].LayerMisalignment[4] = 1.e-1;
112  trackParams[0].LayerMisalignment[5] = 1.e-1;
113  trackParams[0].LayerMisalignment[6] = 1.e-1;
114  trackParams[0].FitIterationMaxChi2[0] = 100.;
115  trackParams[0].FitIterationMaxChi2[1] = 50.;
116  trackParams[0].MinTrackLength = 4;
117  memParams.resize(1);
118  LOG(info) << "Initializing tracker in misaligned sync. phase reconstruction with " << trackParams.size() << " passes";
119 
120  } else if (mMode == "sync") {
121  memParams.resize(1);
122  trackParams.resize(1);
123  LOG(info) << "Initializing tracker in sync. phase reconstruction with " << trackParams.size() << " passes";
124  } else if (mMode == "cosmics") {
125  mRunVertexer = false;
126  trackParams.resize(1);
127  memParams.resize(1);
128  trackParams[0].MinTrackLength = 4;
129  trackParams[0].CellDeltaTanLambdaSigma *= 10;
130  trackParams[0].PhiBins = 4;
131  trackParams[0].ZBins = 16;
132  trackParams[0].PVres = 1.e5f;
133  trackParams[0].LayerMisalignment[0] = 3.e-2;
134  trackParams[0].LayerMisalignment[1] = 3.e-2;
135  trackParams[0].LayerMisalignment[2] = 3.e-2;
136  trackParams[0].LayerMisalignment[3] = 1.e-1;
137  trackParams[0].LayerMisalignment[4] = 1.e-1;
138  trackParams[0].LayerMisalignment[5] = 1.e-1;
139  trackParams[0].LayerMisalignment[6] = 1.e-1;
140  trackParams[0].FitIterationMaxChi2[0] = 100.;
141  trackParams[0].FitIterationMaxChi2[1] = 50.;
142  LOG(info) << "Initializing tracker in reconstruction for cosmics with " << trackParams.size() << " passes";
143 
144  } else {
145  throw std::runtime_error(fmt::format("Unsupported ITS tracking mode {:s} ", mMode));
146  }
147  mTracker->setParameters(memParams, trackParams);
148 
149  mVertexer->getGlobalConfiguration();
150  mTracker->getGlobalConfiguration();
151  LOG(info) << Form("Using %s for material budget approximation", (mTracker->isMatLUT() ? "lookup table" : "TGeometry"));
152 
153  double origD[3] = {0., 0., 0.};
154  mBz = field->getBz(origD);
155  } else {
156  throw std::runtime_error(o2::utils::Str::concat_string("Cannot retrieve GRP from the ", filename));
157  }
158 
159  std::string dictPath = o2::itsmft::ClustererParam<o2::detectors::DetID::ITS>::Instance().dictFilePath;
160  std::string dictFile = o2::base::DetectorNameConf::getAlpideClusterDictionaryFileName(o2::detectors::DetID::ITS, dictPath);
161  if (o2::utils::Str::pathExists(dictFile)) {
162  mDict.readFromFile(dictFile);
163  LOG(info) << "Tracker running with a provided dictionary: " << dictFile;
164  } else {
165  LOG(info) << "Dictionary " << dictFile << " is absent, Tracker expects cluster patterns";
166  }
167 }

169 void TrackerDPL::run(ProcessingContext& pc)
170 {
171  mTimer.Start(false);
172  auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
173  gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
174 
175  // code further down does assignment to the rofs and the altered object is used for output
176  // we therefore need a copy of the vector rather than an object created directly on the input data,
177  // the output vector however is created directly inside the message memory thus avoiding copy by
178  // snapshot
179  auto rofsinput = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
180  auto& rofs = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"ITS", "ITSTrackROF", 0, Lifetime::Timeframe}, rofsinput.begin(), rofsinput.end());
181 
182  auto& irFrames = pc.outputs().make<std::vector<o2::dataformats::IRFrame>>(Output{"ITS", "IRFRAMES", 0, Lifetime::Timeframe});
183 
184  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance(); // RS: this should come from CCDB
185  int nBCPerTF = alpParams.roFrameLengthInBC;
186 
187  LOG(info) << "ITSTracker pulled " << compClusters.size() << " clusters, " << rofs.size() << " RO frames";
188 
189  const dataformats::MCTruthContainer<MCCompLabel>* labels = nullptr;
190  gsl::span<itsmft::MC2ROFRecord const> mc2rofs;
191  if (mIsMC) {
192  labels = pc.inputs().get<const dataformats::MCTruthContainer<MCCompLabel>*>("labels").release();
193  // get the array as read-only span, a snapshot is send forward
194  mc2rofs = pc.inputs().get<gsl::span<itsmft::MC2ROFRecord>>("MC2ROframes");
195  LOG(info) << labels->getIndexedSize() << " MC label objects , in " << mc2rofs.size() << " MC events";
196  }
197 
198  std::vector<o2::its::TrackITSExt> tracks;
199  auto& allClusIdx = pc.outputs().make<std::vector<int>>(Output{"ITS", "TRACKCLSID", 0, Lifetime::Timeframe});
200  std::vector<o2::MCCompLabel> trackLabels;
201  auto& allTracks = pc.outputs().make<std::vector<o2::its::TrackITS>>(Output{"ITS", "TRACKS", 0, Lifetime::Timeframe});
202  std::vector<o2::MCCompLabel> allTrackLabels;
203 
204  auto& vertROFvec = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"ITS", "VERTICESROF", 0, Lifetime::Timeframe});
205  auto& vertices = pc.outputs().make<std::vector<Vertex>>(Output{"ITS", "VERTICES", 0, Lifetime::Timeframe});
206 
207  std::uint32_t roFrame = 0;
208  ROframe event(0, 7);
209 
210  bool continuous = mGRP->isDetContinuousReadOut("ITS");
211  LOG(info) << "ITSTracker RO: continuous=" << continuous;
212 
213  const auto& multEstConf = FastMultEstConfig::Instance(); // parameters for mult estimation and cuts
214  FastMultEst multEst; // mult estimator
215 
216  TimeFrame mTimeFrame;
217  mTracker->adoptTimeFrame(mTimeFrame);
218  mTracker->setBz(mBz);
219 
220  gsl::span<const unsigned char>::iterator pattIt = patterns.begin();
221 
222  gsl::span<itsmft::ROFRecord> rofspan(rofs);
223  mTimeFrame.loadROFrameData(rofspan, compClusters, pattIt, mDict, labels);
224  pattIt = patterns.begin();
225  std::vector<int> savedROF;
226  auto logger = [&](std::string s) { LOG(info) << s; };
227  float vertexerElapsedTime{0.f};
228  int nclUsed = 0;
229 
230  std::vector<bool> processingMask;
231  int cutClusterMult{0}, cutVertexMult{0}, cutTotalMult{0};
232  for (auto& rof : rofspan) {
233  nclUsed += ioutils::loadROFrameData(rof, event, compClusters, pattIt, mDict, labels);
234  // prepare in advance output ROFRecords, even if this ROF to be rejected
235  int first = allTracks.size();
236 
237  // for vertices output
238  auto& vtxROF = vertROFvec.emplace_back(rof); // register entry and number of vertices in the
239  vtxROF.setFirstEntry(vertices.size()); // dedicated ROFRecord
240  vtxROF.setNEntries(0);
241 
242  bool multCut = (multEstConf.cutMultClusLow <= 0 && multEstConf.cutMultClusHigh <= 0); // cut was requested
243  if (!multCut) {
244  float mult = multEst.process(rof.getROFData(compClusters));
245  multCut = mult >= multEstConf.cutMultClusLow && mult <= multEstConf.cutMultClusHigh;
246  LOG(debug) << fmt::format("ROF {} rejected by the cluster multiplicity selection [{},{}]", processingMask.size(), multEstConf.cutMultClusLow, multEstConf.cutMultClusHigh);
247  cutClusterMult += !multCut;
248  }
249 
250  std::vector<Vertex> vtxVecLoc;
251  if (multCut) {
252  if (mRunVertexer) {
253  vertexerElapsedTime += mVertexer->clustersToVertices(event, false, logger);
254  auto allVerts = mVertexer->exportVertices();
255  multCut = allVerts.size() == 0;
256  for (const auto& vtx : allVerts) {
257  if (vtx.getNContributors() < multEstConf.cutMultVtxLow || (multEstConf.cutMultVtxHigh > 0 && vtx.getNContributors() > multEstConf.cutMultVtxHigh)) {
258  continue; // skip vertex of unwanted multiplicity
259  }
260  multCut = true; // At least one passes the selection
261  vtxVecLoc.push_back(vtx);
262  }
263  } else {
264  vtxVecLoc.emplace_back(Vertex());
265  vtxVecLoc.back().setNContributors(1);
266  }
267 
268  if (!multCut) {
269  LOG(debug) << fmt::format("ROF {} rejected by the vertex multiplicity selection [{},{}]", processingMask.size(), multEstConf.cutMultVtxLow, multEstConf.cutMultVtxHigh);
270  cutVertexMult++;
271  }
272  }
273  cutTotalMult += !multCut;
274  processingMask.push_back(multCut);
275  mTimeFrame.addPrimaryVertices(vtxVecLoc);
276 
277  vtxROF.setNEntries(vtxVecLoc.size());
278  for (const auto& vtx : vtxVecLoc) {
279  vertices.push_back(vtx);
280  }
281  savedROF.push_back(roFrame);
282  roFrame++;
283  }
284 
285  LOG(info) << fmt::format(" - In total, multiplicity selection rejected {}/{} ROFs", cutTotalMult, rofspan.size());
286  LOG(info) << fmt::format("\t - Cluster multiplicity selection rejected {}/{} ROFs", cutClusterMult, rofspan.size());
287  LOG(info) << fmt::format("\t - Vertex multiplicity selection rejected {}/{} ROFs", cutVertexMult, rofspan.size());
288  LOG(info) << fmt::format(" - Vertex seeding total elapsed time: {} ms for {} clusters in {} ROFs", vertexerElapsedTime, nclUsed, rofspan.size());
289  LOG(info) << fmt::format(" - Beam position computed for the TF: {}, {}", mTimeFrame.getBeamX(), mTimeFrame.getBeamY());
290 
291  mTimeFrame.setMultiplicityCutMask(processingMask);
292  mTracker->clustersToTracks(logger);
293  if (mTimeFrame.hasBogusClusters()) {
294  LOG(warning) << fmt::format(" - The processed timeframe had {} clusters with wild z coordinates, check the dictionaries", mTimeFrame.hasBogusClusters());
295  }
296 
297  for (unsigned int iROF{0}; iROF < rofs.size(); ++iROF) {
298 
299  auto& rof{rofs[iROF]};
300  tracks = mTimeFrame.getTracks(iROF);
301  trackLabels = mTimeFrame.getTracksLabel(iROF);
302  auto number{tracks.size()};
303  auto first{allTracks.size()};
304  int offset = -rof.getFirstEntry(); // cluster entry!!!
305  rof.setFirstEntry(first);
306  rof.setNEntries(number);
307 
308  if (tracks.size()) {
309  irFrames.emplace_back(rof.getBCData(), rof.getBCData() + nBCPerTF - 1);
310  }
311  std::copy(trackLabels.begin(), trackLabels.end(), std::back_inserter(allTrackLabels));
312  // Some conversions that needs to be moved in the tracker internals
313  for (unsigned int iTrk{0}; iTrk < tracks.size(); ++iTrk) {
314  auto& trc{tracks[iTrk]};
315  trc.setFirstClusterEntry(allClusIdx.size()); // before adding tracks, create final cluster indices
316  int ncl = trc.getNumberOfClusters(), nclf = 0;
317  for (int ic = TrackITSExt::MaxClusters; ic--;) { // track internally keeps in->out cluster indices, but we want to store the references as out->in!!!
318  auto clid = trc.getClusterIndex(ic);
319  if (clid >= 0) {
320  allClusIdx.push_back(clid);
321  nclf++;
322  }
323  }
324  assert(ncl == nclf);
325  allTracks.emplace_back(trc);
326  }
327  }
328 
329  LOG(info) << "ITSTracker pushed " << allTracks.size() << " tracks";
330  if (mIsMC) {
331  LOG(info) << "ITSTracker pushed " << allTrackLabels.size() << " track labels";
332 
333  pc.outputs().snapshot(Output{"ITS", "TRACKSMCTR", 0, Lifetime::Timeframe}, allTrackLabels);
334  pc.outputs().snapshot(Output{"ITS", "ITSTrackMC2ROF", 0, Lifetime::Timeframe}, mc2rofs);
335  }
336  mTimer.Stop();
337 }

345 DataProcessorSpec getTrackerSpec(bool useMC, const std::string& trModeS, o2::gpu::GPUDataTypes::DeviceType dType)
346 {
347  std::vector<InputSpec> inputs;
348  inputs.emplace_back("compClusters", "ITS", "COMPCLUSTERS", 0, Lifetime::Timeframe);
349  inputs.emplace_back("patterns", "ITS", "PATTERNS", 0, Lifetime::Timeframe);
350  inputs.emplace_back("ROframes", "ITS", "CLUSTERSROF", 0, Lifetime::Timeframe);
351 
352  std::vector<OutputSpec> outputs;
353  outputs.emplace_back("ITS", "TRACKS", 0, Lifetime::Timeframe);
354  outputs.emplace_back("ITS", "TRACKCLSID", 0, Lifetime::Timeframe);
355  outputs.emplace_back("ITS", "ITSTrackROF", 0, Lifetime::Timeframe);
356  outputs.emplace_back("ITS", "VERTICES", 0, Lifetime::Timeframe);
357  outputs.emplace_back("ITS", "VERTICESROF", 0, Lifetime::Timeframe);
358  outputs.emplace_back("ITS", "IRFRAMES", 0, Lifetime::Timeframe);
359 
360  if (useMC) {
361  inputs.emplace_back("labels", "ITS", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
362  inputs.emplace_back("MC2ROframes", "ITS", "CLUSTERSMC2ROF", 0, Lifetime::Timeframe);
363  outputs.emplace_back("ITS", "TRACKSMCTR", 0, Lifetime::Timeframe);
364  outputs.emplace_back("ITS", "ITSTrackMC2ROF", 0, Lifetime::Timeframe);
365  outputs.emplace_back("ITS", "VERTICES", 0, Lifetime::Timeframe);
366  }
367 
368  return DataProcessorSpec{
369  "its-tracker",
370  inputs,
371  outputs,
372  AlgorithmSpec{adaptFromTask<TrackerDPL>(useMC, trModeS, dType)},
373  Options{
374  {"grp-file", VariantType::String, "o2sim_grp.root", {"Name of the grp file"}},
375  {"material-lut-path", VariantType::String, "", {"Path of the material LUT file"}}}};
376 }
