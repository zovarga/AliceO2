92  this->mGID = tree["gid"].GetString();

94  this->mGID = "track";
95  }
96  this->mPolyX.reserve(count.GetInt());
97  this->mPolyY.reserve(count.GetInt());
98  this->mPolyZ.reserve(count.GetInt());
99  for (auto& v : jsonPolyX.GetArray()) {
100  mPolyX.push_back(v.GetDouble());
101  }
102  for (auto& v : jsonPolyY.GetArray()) {
103  mPolyY.push_back(v.GetDouble());
104  }
105  for (auto& v : jsonPolyZ.GetArray()) {
106  mPolyZ.push_back(v.GetDouble());
107  }
108  if (tree.HasMember("mClusters")) {
109  rapidjson::Value& jsonClusters = tree["mClusters"];
110  auto jsonArray = jsonClusters.GetArray();
111  this->mClusters.reserve(jsonArray.Size());
112  for (auto& v : jsonClusters.GetArray()) {
113  mClusters.emplace_back(v);
114  }
115  }

118 rapidjson::Value VisualisationTrack::jsonTree(rapidjson::Document::AllocatorType& allocator)
119 {
120  rapidjson::Value tree(rapidjson::kObjectType);
121  rapidjson::Value jsonPolyX(rapidjson::kArrayType);
122  rapidjson::Value jsonPolyY(rapidjson::kArrayType);
123  rapidjson::Value jsonPolyZ(rapidjson::kArrayType);
124  rapidjson::Value jsonStartCoordinates(rapidjson::kArrayType);
125 
126  tree.AddMember("count", rapidjson::Value().SetInt(this->getPointCount()), allocator);
127  tree.AddMember("source", rapidjson::Value().SetInt(this->mSource), allocator);
128  rapidjson::Value gid;
129  gid.SetString(this->mGID.c_str(), this->mGID.size(), allocator);
130  tree.AddMember("gid", gid, allocator);
131  tree.AddMember("time", rapidjson::Value().SetFloat(isnan(this->mTime) ? 0 : this->mTime), allocator);
132  tree.AddMember("charge", rapidjson::Value().SetInt(this->mCharge), allocator);
133  tree.AddMember("theta", rapidjson::Value().SetFloat(isnan(this->mTheta) ? 0 : this->mTheta), allocator);
134  tree.AddMember("phi", rapidjson::Value().SetFloat(isnan(this->mPhi) ? 0 : this->mPhi), allocator);
135  tree.AddMember("eta", rapidjson::Value().SetFloat(isnan(this->mEta) ? 0 : this->mEta), allocator);
136  tree.AddMember("PID", rapidjson::Value().SetInt(this->mPID), allocator);
137 
138  jsonStartCoordinates.PushBack((float)mStartCoordinates[0], allocator);
139  jsonStartCoordinates.PushBack((float)mStartCoordinates[1], allocator);
140  jsonStartCoordinates.PushBack((float)mStartCoordinates[2], allocator);
141  tree.AddMember("jsonStartingXYZ", jsonStartCoordinates, allocator);
142 
143  for (size_t i = 0; i < this->getPointCount(); i++) {
144  jsonPolyX.PushBack((float)mPolyX[i], allocator);
145  jsonPolyY.PushBack((float)mPolyY[i], allocator);
146  jsonPolyZ.PushBack((float)mPolyZ[i], allocator);
147  }
148  tree.AddMember("mPolyX", jsonPolyX, allocator);
149  tree.AddMember("mPolyY", jsonPolyY, allocator);
150  tree.AddMember("mPolyZ", jsonPolyZ, allocator);
151 
152  rapidjson::Value jsonClusters(rapidjson::kArrayType);
153  for (size_t i = 0; i < this->mClusters.size(); i++) {
154  jsonClusters.PushBack(this->mClusters[i].jsonTree(allocator), allocator);
155  }
156  tree.AddMember("mClusters", jsonClusters, allocator);
157 
158  return tree;
159 }
