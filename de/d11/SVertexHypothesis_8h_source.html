47  float calcMass2(float p2Pos, float p2Neg, float p2V0) const
48  {
49  // calculate v0 mass from squared momentum of its prongs and total momentum
50  float ePos = std::sqrt(p2Pos + getMass2PosProng()), eNeg = std::sqrt(p2Neg + getMass2NegProng()), eV0 = ePos + eNeg;
51  return eV0 * eV0 - p2V0;
52  }
53 
54  float calcMass(float p2Pos, float p2Neg, float p2V0) const { return std::sqrt(calcMass2(p2Pos, p2Neg, p2V0)); }
55 
56  bool check(float p2Pos, float p2Neg, float p2V0, float ptV0) const
57  { // check if given mass and pt is matching to hypothesis
58  return check(calcMass(p2Pos, p2Neg, p2V0), ptV0);
59  }
60 
61  bool check(float mass, float pt) const
62  { // check if given mass and pt is matching to hypothesis
63  return std::abs(mass - getMassV0Hyp()) < getMargin(pt);
64  }
65 
66  float getSigma(float pt) const { return mPars[SigmaM] * (1.f + mPars[CPt] * pt); }
67  float getMargin(float pt) const { return mPars[NSigmaM] * getSigma(pt) + mPars[MarginM]; }
