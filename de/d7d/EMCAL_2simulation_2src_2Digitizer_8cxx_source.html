35 void Digitizer::init()
36 {
37  mSimParam = &(o2::emcal::SimParam::Instance());
38  mLiveTime = mSimParam->getLiveTime();
39  mBusyTime = mSimParam->getBusyTime();
40  mRandomGenerator = new TRandom3(std::chrono::high_resolution_clock::now().time_since_epoch().count());
41 
42  float tau = mSimParam->getTimeResponseTau();
43  float N = mSimParam->getTimeResponsePower();
44  float delay = std::fmod(mSimParam->getSignalDelay() / constants::EMCAL_TIMESAMPLE, 1);
45  mDelay = ((int)(std::floor(mSimParam->getSignalDelay() / constants::EMCAL_TIMESAMPLE)));
46 
47  mSmearEnergy = mSimParam->doSmearEnergy();
48  mSimulateTimeResponse = mSimParam->doSimulateTimeResponse();
49  mRemoveDigitsBelowThreshold = mSimParam->doRemoveDigitsBelowThreshold();
50  mSimulateNoiseDigits = mSimParam->doSimulateNoiseDigits();
51 
52  mTimeBinOffset.clear();
53  mAmplitudeInTimeBins.clear();
54 
55  // for each phase create a template distribution
56  TF1 RawResponse("RawResponse", rawResponseFunction, 0, 256, 5);
57  RawResponse.SetParameters(1., 0., tau, N, 0.);
58 
59  for (int i = 0; i < 4; i++) {
60  int offset = 0;
61  mTimeBinOffset.push_back(offset);
62 
63  std::vector<double> sf;
64  RawResponse.SetParameter(1, 0.25 * i);
65 
66  for (int j = 0; j < constants::EMCAL_MAXTIMEBINS; j++) {
67  sf.push_back(RawResponse.Eval(j - mTimeWindowStart));
68  }
69  mAmplitudeInTimeBins.push_back(sf);
70  }
71 }

74 double Digitizer::rawResponseFunction(double* x, double* par)
75 {
76  double signal = 0.;
77  double tau = par[2];
78  double n = par[3];
79  double ped = par[4];
80  double xx = (x[0] - par[1] + tau) / tau;
81 
82  // par[0] amp, par[1] peak time
83 
84  if (xx <= 0) {
85  signal = ped;
86  } else {
87  signal = ped + par[0] * std::pow(xx, n) * std::exp(n * (1 - xx));
88  }
89 
90  return signal;
91 }

111 void Digitizer::process(const std::vector<LabeledDigit>& labeledSDigits)
112 {
113 
114  //mDigits.reserve();
115 
116  for (auto labeleddigit : labeledSDigits) {
117 
118  sampleSDigit(labeleddigit.getDigit());
119 
120  for (auto digit : mTempDigitVector) {
121  Int_t id = digit.getTower();
122 
123  auto labels = labeleddigit.getLabels();
124  LabeledDigit d(digit, labels[0]);
125  for (auto label : labels) {
126  if (digit.getAmplitude() == 0) {
127  label.setAmplitudeFraction(0);

129  if (label == labels.front()) {
130  continue;
131  }
132  d.addLabel(label);
133  }
134  mDigits[id].push_back(d);
135  }
136  }
137 
138  mEmpty = false;

142 void Digitizer::sampleSDigit(const Digit& sDigit)
143 {
144  mTempDigitVector.clear();
145  Int_t tower = sDigit.getTower();
146  Double_t energy = sDigit.getAmplitude();
147 
148  if (mSmearEnergy) {
149  energy = smearEnergy(energy);
150  }
151 
152  if (energy < __DBL_EPSILON__) {
153  return;
154  }
155 
156  if (mSimulateTimeResponse) {
157  for (int j = 0; j < mAmplitudeInTimeBins.at(mPhase).size(); j++) {
158  double val = energy * (mAmplitudeInTimeBins.at(mPhase).at(j));
159 
160  double digitTime = (mEventTimeOffset + j - mTimeBinOffset.at(mPhase) + mDelay - mTimeWindowStart) * constants::EMCAL_TIMESAMPLE;
161  if (preTriggerCollision() && digitTime >= (mLiveTime + mBusyTime)) {
162  digitTime = digitTime - (mLiveTime + mBusyTime);
163  }
164 
165  Digit digit(tower, val, digitTime);
166  mTempDigitVector.push_back(digit);
167  }
168  } else {
169  Digit digit(tower, energy, mEventTime + mDelay * constants::EMCAL_TIMESAMPLE);
170  mTempDigitVector.push_back(digit);
171  }
172 }

175 double Digitizer::smearEnergy(double energy)
176 {
177  Double_t fluct = (energy * mSimParam->getMeanPhotonElectron()) / mSimParam->getGainFluctuations();
178  energy *= mRandomGenerator->Poisson(fluct) / fluct;
179  return energy;
180 }

183 void Digitizer::setEventTime(double t)
184 {
185  // assign event time, it should be in a strictly increasing order
186  // convert to ns
187  t *= mCoeffToNanoSecond;
188 
189  if (t < mEventTime) {
190  LOG(fatal) << "New event time (" << t << ") is < previous event time (" << mEventTime << ")";
191  }
192 
193  if ((t - mTriggerTime) >= (mLiveTime + mBusyTime)) {
194  mTriggerTime = t;
195  }
196 
197  mEventTime = t - mTriggerTime;
198 
199  mPhase = ((int)(std::fmod(mEventTime, 100) / 25));
200 
201  mEventTimeOffset = ((int)((mEventTime - std::fmod(mEventTime, 100) + 0.1) / 100));
202 
203  if (preTriggerCollision()) {
204  mEventTimeOffset = ((int)((mEventTime - std::fmod(mEventTime, 100) + 0.1) / 100)) - (mLiveTime + mBusyTime - mPreTriggerTime) / 100 + mAfterTriggerTime / 100;
205  }
206 
207  if (mPhase == 4) {
208  mPhase = 0;
209  mEventTimeOffset++;
210  }
211 }

214 void Digitizer::addNoiseDigits(LabeledDigit& d1)
215 {
216  double amplitude = d1.getAmplitude();
217  double sigma = mSimParam->getPinNoise();
218  if (amplitude > constants::EMCAL_HGLGTRANSITION * constants::EMCAL_ADCENERGY) {
219  sigma = mSimParam->getPinNoiseLG();
220  }
221 
222  double noise = std::abs(mRandomGenerator->Gaus(0, sigma));
223  MCLabel label(true, 1.0);
224  LabeledDigit d(d1.getTower(), noise, d1.getTimeStamp(), label);
225  d1 += d;
226 }

229 void Digitizer::fillOutputContainer(std::vector<Digit>& digits, o2::dataformats::MCTruthContainer<o2::emcal::MCLabel>& labelsout)
230 {
231  std::list<LabeledDigit> l;
232 
233  for (auto [tower, digitsList] : mDigits) {
234 
235  if (digitsList.size() == 0) {
236  continue;
237  }
238  digitsList.sort();
239 
240  for (auto& ld : digitsList) {
241 
242  // Loop over all digits in the time sample and sum the digits that falls in one time bin
243  for (auto ld1 = digitsList.begin(); ld1 != digitsList.end(); ++ld1) {
244 
245  if (ld == *ld1) {
246  continue;
247  }
248 
249  std::vector<decltype(digitsList.begin())> toDelete;
250 
251  if (ld.canAdd(*ld1)) {
252  ld += *ld1;
253  toDelete.push_back(ld1);
254  }
255  for (auto del : toDelete) {
256  digitsList.erase(del);
257  }
258  }
259 
260  if (mSimulateNoiseDigits) {
261  addNoiseDigits(ld);
262  }
263 
264  if (mRemoveDigitsBelowThreshold && (ld.getAmplitude() < (mSimParam->getDigitThreshold() * constants::EMCAL_ADCENERGY))) {
265  continue;
266  }
267  if (ld.getAmplitude() < 0) {
268  continue;
269  }
270  if ((ld.getTimeStamp() >= mSimParam->getLiveTime()) || (ld.getTimeStamp() < 0)) {
271  continue;
272  }
273 
274  l.push_back(ld);
275  }
276  }
277  l.sort();
278 
279  for (auto d : l) {
280  Digit digit = d.getDigit();
281  std::vector<MCLabel> labels = d.getLabels();
282  digits.push_back(digit);
283 
284  Int_t LabelIndex = labelsout.getIndexedSize();
285  for (auto label : labels) {
286  labelsout.addElementRandomAccess(LabelIndex, label);
287  }
288  }
289 
290  mDigits.clear();
291  mEmpty = true;
292 }
