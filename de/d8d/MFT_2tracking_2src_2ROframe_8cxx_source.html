42 template <typename T>
43 void ROframe<T>::sortClusters()
44 {
45  Int_t nClsInLayer, binPrevIndex, clsMinIndex, clsMaxIndex, jClsLayer;
46  // sort the clusters in R-Phi
47  for (Int_t iLayer = 0; iLayer < constants::mft::LayersNumber; ++iLayer) {
48  if (mClusters[iLayer].size() == 0) {
49  continue;
50  }
51  // sort clusters in layer according to the bin index
52  sort(mClusters[iLayer].begin(), mClusters[iLayer].end(),
53  [](Cluster& c1, Cluster& c2) { return c1.indexTableBin < c2.indexTableBin; });
54  // find the cluster local index range in each bin
55  // index = element position in the vector
56  nClsInLayer = mClusters[iLayer].size();
57  binPrevIndex = mClusters[iLayer].at(0).indexTableBin;
58  clsMinIndex = 0;
59  for (jClsLayer = 1; jClsLayer < nClsInLayer; ++jClsLayer) {
60  if (mClusters[iLayer].at(jClsLayer).indexTableBin == binPrevIndex) {
61  continue;
62  }
63 
64  clsMaxIndex = jClsLayer - 1;
65 
66  mClusterBinIndexRange[iLayer][binPrevIndex] = std::pair<Int_t, Int_t>(clsMinIndex, clsMaxIndex);
67 
68  binPrevIndex = mClusters[iLayer].at(jClsLayer).indexTableBin;
69  clsMinIndex = jClsLayer;
70  } // clusters
71 
72  // last cluster
73  clsMaxIndex = jClsLayer - 1;
74 
75  mClusterBinIndexRange[iLayer][binPrevIndex] = std::pair<Int_t, Int_t>(clsMinIndex, clsMaxIndex);
76  } // layers
77 }
