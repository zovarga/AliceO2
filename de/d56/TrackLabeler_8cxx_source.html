38 std::vector<MCCompLabel> TrackLabeler::findLabels(const Track& track, const o2::dataformats::MCTruthContainer<MCClusterLabel>& inMCContainer) const
39 {
41  std::vector<MCCompLabel> allLabels;
42  std::vector<int> counts;
43  for (int ich = 0; ich < 4; ++ich) {
44  auto icl = track.getClusterMatched(ich);
45  if (icl < 0) {
46  continue;
47  }
48 
49  // First check if the BP and NBP were ever fired by a track
50  // If they are, it means that the chamber was efficient
51  bool bothFired = areBothSidesFired(inMCContainer.getLabels(icl));
52 
53  for (auto& label : inMCContainer.getLabels(icl)) {
54  // This track fired only one cathode
55  bool oneFired = (!label.isFiredBP() || !label.isFiredNBP());
56  if (bothFired && oneFired) {
57  // This condition means that the cluster was made from
58  // a track in the BP and a different track in the NBP.
59  // This happens when we have two tracks in the same column.
60  // This means that the cluster is a fake one, so we discard it
61  continue;
62  }
63  bool isNew = true;
64  for (size_t idx = 0; idx < allLabels.size(); ++idx) {
65  if (allLabels[idx] == label) {
66  ++counts[idx];
67  isNew = false;
68  break;
69  }
70  }
71  if (isNew) {
72  allLabels.emplace_back(label);
73  counts.emplace_back(1);
74  }
75  }
76  }
77 
78  std::vector<MCCompLabel> labels;
79 
80  for (size_t idx = 0; idx < allLabels.size(); ++idx) {
81  if (counts[idx] >= 3) {
82  labels.emplace_back(allLabels[idx]);
83  }
84  }
85 
86  return labels;
87 }
