125 template <typename F>
126 constexpr F meanPhiSmall(F phi0, F phi1)
127 {
128  // return mean phi, assume phis in 0:2pi
129  F phi;
130  if (!okForPhiMin(phi0, phi1)) {
131  phi = phi0;
132  phi0 = phi1;
133  phi1 = phi;
134  }
135  if (phi0 > phi1) {
136  phi = (phi1 - (constants::math::TwoPI - phi0)) / 2; // wrap
137  } else {
138  phi = (phi0 + phi1) / 2;
139  }
140  bringTo02Pi(phi);
141  return phi;
142 };

182 inline constexpr int findKeyIndex(int key, const int* arr, int n) noexcept
183 {
184  // finds index of key in the array
185  int imn = 0;
186  int imx = n - 1;
187  while (imx >= imn) {
188  const int mid = (imx + imn) >> 1;
189  if (arr[mid] == key) {
190  return mid;
191  }
192 
193  if (arr[mid] < key) {
194  imn = mid + 1;
195  } else {
196  imx = mid - 1;
197  }
198  }
199  return -1;
200 }
