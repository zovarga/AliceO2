237 template <typename S, typename T>
238 struct variant_helper {
239  static void set(S* store, T value)
240  {
241  new (reinterpret_cast<T*>(store)) T{};
242  *(reinterpret_cast<T*>(store)) = value;
243  }
244  static void set(S* store, T values, size_t size)
245  {
246  *reinterpret_cast<T*>(store) = reinterpret_cast<T>(std::memcpy(std::malloc(size * sizeof(std::remove_pointer_t<T>)), reinterpret_cast<void*>(values), size * sizeof(std::remove_pointer_t<T>)));
247  }
248 
249  static T get(const S* store) { return *(reinterpret_cast<const T*>(store)); }
250 };

274 class Variant
275 {
276  using storage_t = std::aligned_union<8, int, int64_t, uint8_t, uint16_t, uint32_t, uint64_t,
277  const char*, float, double, bool,
278  int*, float*, double*, bool*,
279  Array2D<int>, Array2D<float>, Array2D<double>,
280  LabeledArray<int>, LabeledArray<float>, LabeledArray<double>>::type;
281 
282  public:
283  Variant(VariantType type = VariantType::Unknown) : mType{type}, mSize{1} {}
284 
285  template <typename T>
286  Variant(T value) : mType{variant_trait_v<T>}, mSize{1}
287  {
288  variant_helper<storage_t, decltype(value)>::set(&mStore, value);
289  }
290 
291  template <typename T>
292  Variant(T values, size_t size) : mType{variant_trait_v<T>}, mSize{size}
293  {
294  variant_helper<storage_t, T>::set(&mStore, values, mSize);
295  }
296 
297  template <typename T>
298  Variant(std::vector<T>& values) : mType{variant_trait_v<T*>}, mSize{values.size()}
299  {
300  variant_helper<storage_t, T*>::set(&mStore, values.data(), mSize);
301  }
302 
303  template <typename T>
304  Variant(std::initializer_list<T>) : mType{VariantType::Unknown}, mSize{1}
305  {
306  static_assert(sizeof(T) == 0,
307  "brace-enclosed initializer list forbidden for Variant"
308  "\n did you accidentally put braces around the default value?");
309  }

314  Variant& operator=(const Variant& other);

317  template <typename T>
318  T get() const
319  {
320  if (mType != variant_trait_v<T>) {
321  throw runtime_error("Mismatch between types");
322  }
323  return variant_helper<storage_t, T>::get(&mStore);
324  }
325 
326  template <typename T>
327  void set(T value)
328  {
329  return variant_helper<storage_t, T>::set(&mStore, value);
330  }
331 
332  template <typename T>
333  void set(T value, size_t size)
334  {
335  mSize = size;
336  return variant_helper<storage_t, T>::set(&mStore, value, mSize);
337  }
338 
339  template <typename T>
340  void set(std::vector<T>& values)
341  {
342  return variant_helper<storage_t, T*>::set(&mStore, values.data(), values.size());
343  }
344 
345  VariantType type() const { return mType; }
346  size_t size() const { return mSize; }
347  std::string asString() const;
348 
349  private:

351  storage_t mStore;
