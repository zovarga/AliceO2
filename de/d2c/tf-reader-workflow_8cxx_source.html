51 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)

60  rinp.detListNonRawOnly = configcontext.options().get<std::string>("non-raw-only-det");
61  rinp.rawChannelConfig = configcontext.options().get<std::string>("raw-channel-config");
62  rinp.delay_us = uint64_t(1e6 * configcontext.options().get<float>("delay")); // delay in microseconds
63  rinp.verbosity = configcontext.options().get<int>("tf-reader-verbosity");
64  rinp.maxTFCache = std::max(1, configcontext.options().get<int>("max-cached-tf"));
65  rinp.maxFileCache = std::max(1, configcontext.options().get<int>("max-cached-files"));
66  rinp.copyCmd = configcontext.options().get<std::string>("copy-cmd");
67  rinp.tffileRegex = configcontext.options().get<std::string>("tf-file-regex");
68  rinp.remoteRegex = configcontext.options().get<std::string>("remote-regex");
69  rinp.sendDummyForMissing = !configcontext.options().get<bool>("disable-dummy-output");
70  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
71 
72  WorkflowSpec specs;
73  specs.emplace_back(o2::rawdd::getTFReaderSpec(rinp));
