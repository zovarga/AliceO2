385 bool GeometryMisAligner::matrixToAngles(const double* rot, double& psi, double& theta, double& phi)
386 {
391 
392  if (std::abs(rot[0]) < 1e-7 || std::abs(rot[8]) < 1e-7) {
393  LOG(error) << "Failed to extract roll-pitch-yall angles!";
394  return false;
395  }
396  psi = std::atan2(-rot[5], rot[8]);
397  theta = std::asin(rot[2]);
398  phi = std::atan2(-rot[1], rot[0]);
399  return true;
400 }
