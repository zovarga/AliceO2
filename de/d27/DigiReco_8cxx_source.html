22 void DigiReco::init()
23 {
24  LOG(info) << "Initialization of ZDC reconstruction";
25  // Load configuration parameters
26  auto& sopt = ZDCSimParam::Instance();
27  mIsContinuous = sopt.continuous;
28  mNBCAHead = mIsContinuous ? sopt.nBCAheadCont : sopt.nBCAheadTrig;
29 
30  if (!mModuleConfig) {
31  LOG(fatal) << "Missing ModuleConfig configuration object";
32  return;
33  }
34 
35  // Prepare tapered sinc function
36  // tsc/TSN =3.75 (~ 4) and TSL*TSN*sqrt(2)/tsc >> 1 (n. of sigma)
37  const O2_ZDC_DIGIRECO_FLT tsc = 750;
38  int n = TSL * TSN;
39  for (int tsi = 0; tsi <= n; tsi++) {
40  O2_ZDC_DIGIRECO_FLT arg1 = TMath::Pi() * O2_ZDC_DIGIRECO_FLT(tsi) / O2_ZDC_DIGIRECO_FLT(TSN);
41  O2_ZDC_DIGIRECO_FLT fs = 1;
42  if (arg1 != 0) {
43  fs = TMath::Sin(arg1) / arg1;
44  }
45  O2_ZDC_DIGIRECO_FLT arg2 = O2_ZDC_DIGIRECO_FLT(tsi) / tsc;
46  O2_ZDC_DIGIRECO_FLT fg = TMath::Exp(-arg2 * arg2);
47  mTS[n + tsi] = fs * fg;
48  mTS[n - tsi] = mTS[n + tsi]; // Function is even
49  }
50 
51  if (mTreeDbg) {
52  // Open debug file
53  LOG(info) << "ZDC DigiReco: opening debug output";
54  mDbg = std::make_unique<TFile>("ZDCRecoDbg.root", "recreate");
55  mTDbg = std::make_unique<TTree>("zdcr", "ZDCReco");
56  mTDbg->Branch("zdcr", "RecEventAux", &mRec);
57  }
58 
59  // Update reconstruction parameters
60  //auto& ropt=RecoParamZDC::Instance();
61  o2::zdc::RecoParamZDC& ropt = const_cast<o2::zdc::RecoParamZDC&>(RecoParamZDC::Instance());
62 
63  // Fill maps to decode the pattern of channels with hit
64  for (int itdc = 0; itdc < o2::zdc::NTDCChannels; itdc++) {
65  // If the reconstruction parameters were not manually set
66  if (ropt.tmod[itdc] < 0 || ropt.tch[itdc] < 0) {
67  int isig = TDCSignal[itdc];
68  for (int im = 0; im < NModules; im++) {
69  for (uint32_t ic = 0; ic < NChPerModule; ic++) {
70  if (mModuleConfig->modules[im].channelID[ic] == isig && mModuleConfig->modules[im].readChannel[ic]) {
71  //ropt.updateFromString(TString::Format("RecoParamZDC.tmod[%d]=%d;",itdc,im));
72  //ropt.updateFromString(TString::Format("RecoParamZDC.tch[%d]=%d;",itdc,ic));
73  ropt.tmod[itdc] = im;
74  ropt.tch[itdc] = ic;
75  // Fill mask to identify TDC channels
76  mTDCMask[itdc] = (0x1 << (4 * im + ic));
77  goto next_itdc;
78  }
79  }
80  }
81  }
82  next_itdc:;
83  if (mVerbosity > DbgZero) {
84  LOG(info) << "TDC " << itdc << "(" << ChannelNames[TDCSignal[itdc]] << ")"
85  << " mod " << ropt.tmod[itdc] << " ch " << ropt.tch[itdc];
86  }
87  }
88 
89  // TDC calibration
90  for (int itdc = 0; itdc < o2::zdc::NTDCChannels; itdc++) {
91  float fval = ropt.tdc_shift[itdc];
92  // If the reconstruction parameters were not manually set
93  if (fval < 0) {
94  // Check if calibration object is present
95  if (!mTDCParam) {
96  LOG(fatal) << "TDC " << itdc << " missing configuration object and no manual override";
97  } else {
98  fval = mTDCParam->getShift(itdc) / FTDCVal;
99  }
100  }
101  auto val = std::nearbyint(fval);
102  if (val < kMinShort) {
103  LOG(fatal) << "Shift for TDC " << itdc << " " << val << " is out of range";
104  }
105  if (val > kMaxShort) {
106  LOG(fatal) << "Shift for TDC " << itdc << " " << val << " is out of range";
107  }
108  tdc_shift[itdc] = val;
109  if (mVerbosity > DbgZero) {
110  LOG(info) << itdc << " " << ChannelNames[TDCSignal[itdc]] << " shift= " << tdc_shift[itdc] << " i.s. = " << val * FTDCVal << " ns";
111  }
112  }
113 
114  // TDC search zone
115  for (int itdc = 0; itdc < o2::zdc::NTDCChannels; itdc++) {
116  // If the reconstruction parameters were not manually set
117  if (ropt.tdc_search[itdc] <= 0) {
118  if (!mRecoConfigZDC) {
119  LOG(fatal) << "Search zone for TDC " << itdc << " missing configuration object and no manual override";
120  } else {
121  ropt.tdc_search[itdc] = mRecoConfigZDC->tdc_search[itdc];
122  }
123  }
124  if (mVerbosity > DbgZero) {
125  LOG(info) << itdc << " " << ChannelNames[TDCSignal[itdc]] << " search= " << ropt.tdc_search[itdc] << " i.s. = " << ropt.tdc_search[itdc] * FTDCVal << " ns";
126  }
127  }
128 
129  // Energy calibration
130  for (int il = 0; il < ChEnergyCalib.size(); il++) {
131  if (ropt.energy_calib[ChEnergyCalib[il]] > 0) {
132  LOG(info) << "Energy Calibration from command line " << ChannelNames[ChEnergyCalib[il]] << " = " << ropt.energy_calib[ChEnergyCalib[il]];
133  } else if (mEnergyParam && mEnergyParam->energy_calib[ChEnergyCalib[il]] > 0) {
134  ropt.energy_calib[ChEnergyCalib[il]] = mEnergyParam->energy_calib[ChEnergyCalib[il]];
135  if (mVerbosity > DbgZero) {
136  LOG(info) << "Energy Calibration from CCDB " << ChannelNames[ChEnergyCalib[il]] << " = " << ropt.energy_calib[ChEnergyCalib[il]];
137  }
138  } else {
139  if (ChEnergyCalib[il] == CaloCommonPM[ChEnergyCalib[il]]) {
140  // Is a common PM or a ZEM
141  ropt.energy_calib[ChEnergyCalib[il]] = 1;
142  LOG(warning) << "Default Energy Calibration " << ChannelNames[ChEnergyCalib[il]] << " = " << ropt.energy_calib[ChEnergyCalib[il]];
143  } else {
144  // Is one of the analog sums -> same calibration as common PM
145  // N.B. the calibration for common has already been set in the loop
146  ropt.energy_calib[ChEnergyCalib[il]] = ropt.energy_calib[CaloCommonPM[il]];
147  if (mVerbosity > DbgZero) {
148  LOG(info) << "SUM Energy Calibration " << ChannelNames[ChEnergyCalib[il]] << " = " << ropt.energy_calib[ChEnergyCalib[il]];
149  }
150  }
151  }
152  }
153 
154  // Tower calibration
155  for (int il = 0; il < ChTowerCalib.size(); il++) {
156  if (ropt.tower_calib[ChTowerCalib[il]] > 0) {
157  LOG(info) << "Tower Calibration from command line " << ChannelNames[ChTowerCalib[il]] << " = " << ropt.tower_calib[ChTowerCalib[il]];
158  } else if (mTowerParam && mTowerParam->tower_calib[ChTowerCalib[il]] > 0) {
159  ropt.tower_calib[ChTowerCalib[il]] = mTowerParam->tower_calib[ChTowerCalib[il]];
160  if (mVerbosity > DbgZero) {
161  LOG(info) << "Tower Calibration from CCDB " << ChannelNames[ChTowerCalib[il]] << " = " << ropt.tower_calib[ChTowerCalib[il]];
162  }
163  } else {
164  ropt.tower_calib[ChTowerCalib[il]] = 1;
165  LOG(warning) << "Default Tower Calibration " << ChannelNames[ChTowerCalib[il]] << " = " << ropt.tower_calib[ChTowerCalib[il]];
166  }
167  }
168 
169  // Tower energy calibration
170  for (int il = 0; il < ChTowerCalib.size(); il++) {
171  if (ropt.energy_calib[ChTowerCalib[il]] > 0) {
172  LOG(info) << "Tower Energy Calibration from command line " << ChannelNames[ChTowerCalib[il]] << " = " << ropt.energy_calib[ChTowerCalib[il]];
173  } else {
174  ropt.energy_calib[ChTowerCalib[il]] = ropt.tower_calib[ChTowerCalib[il]] * ropt.energy_calib[CaloCommonPM[ChTowerCalib[il]]];
175  if (mVerbosity > DbgZero) {
176  LOG(info) << "Tower Energy Calibration " << ChannelNames[ChTowerCalib[il]] << " = " << ropt.energy_calib[ChTowerCalib[il]];
177  }
178  }
179  }
180 
181  // Fill maps channel maps for integration
182  for (int ich = 0; ich < NChannels; ich++) {
183  // If the reconstruction parameters were not manually set
184  if (ropt.amod[ich] < 0 || ropt.ach[ich] < 0) {
185  for (int im = 0; im < NModules; im++) {
186  for (uint32_t ic = 0; ic < NChPerModule; ic++) {
187  if (mModuleConfig->modules[im].channelID[ic] == ich && mModuleConfig->modules[im].readChannel[ic]) {
188  ropt.amod[ich] = im;
189  ropt.ach[ich] = ic;
190  goto next_ich;
191  }
192  }
193  }
194  }
195  next_ich:;
196  if (mVerbosity > DbgZero) {
197  LOG(info) << "ADC " << ich << "(" << ChannelNames[ich] << ") mod " << ropt.amod[ich] << " ch " << ropt.ach[ich];
198  }
199  }
200 
201  // Integration ranges
202  for (int ich = 0; ich < NChannels; ich++) {
203  // If the reconstruction parameters were not manually set
204  if (ropt.beg_int[ich] == DummyIntRange || ropt.end_int[ich] == DummyIntRange) {
205  if (!mRecoConfigZDC) {
206  LOG(fatal) << "Integration for signal " << ich << " missing configuration object and no manual override";
207  } else {
208  ropt.beg_int[ich] = mRecoConfigZDC->beg_int[ich];
209  ropt.end_int[ich] = mRecoConfigZDC->end_int[ich];
210  }
211  }
212  if (ropt.beg_ped_int[ich] == DummyIntRange || ropt.end_ped_int[ich] == DummyIntRange) {
213  if (!mRecoConfigZDC) {
214  LOG(error) << "Integration for pedestal " << ich << " missing configuration object and no manual override";
215  } else {
216  ropt.beg_ped_int[ich] = mRecoConfigZDC->beg_ped_int[ich];
217  ropt.end_ped_int[ich] = mRecoConfigZDC->end_ped_int[ich];
218  }
219  }
220  if (mVerbosity > DbgZero) {
221  LOG(info) << ChannelNames[ich] << " integration: signal=[" << ropt.beg_int[ich] << ":" << ropt.end_int[ich] << "] pedestal=[" << ropt.beg_ped_int[ich] << ":" << ropt.end_ped_int[ich] << "]";
222  }
223  }
224 } // init

226 int DigiReco::process(const gsl::span<const o2::zdc::OrbitData>& orbitdata, const gsl::span<const o2::zdc::BCData>& bcdata, const gsl::span<const o2::zdc::ChannelData>& chdata)
227 {
228  // We assume that vectors contain data from a full time frame
229  mOrbitData = orbitdata;
230  mBCData = bcdata;
231  mChData = chdata;
232 
233  // Initialization of lookup structure for pedestals
234  mOrbit.clear();
235  int norb = mOrbitData.size();
236  if (mVerbosity >= DbgFull) {
237  LOG(info) << "Dump of pedestal data lookup table";
238  }
239  for (int iorb = 0; iorb < norb; iorb++) {
240  mOrbit[mOrbitData[iorb].ir.orbit] = iorb;
241  if (mVerbosity >= DbgFull) {
242  LOG(info) << "mOrbitData[" << mOrbitData[iorb].ir.orbit << "] = " << iorb;
243  }
244  }
245  mNBC = mBCData.size();
246  mReco.clear();
247  mReco.resize(mNBC);
248  // Initialization of reco structure
249  for (int ibc = 0; ibc < mNBC; ibc++) {
250  auto& bcr = mReco[ibc];
251 #ifdef O2_ZDC_TDC_C_ARRAY
252  for (int itdc = 0; itdc < NTDCChannels; itdc++) {
253  for (int i = 0; i < MaxTDCValues; i++) {
254  bcr.tdcVal[itdc][i] = kMinShort;
255  bcr.tdcAmp[itdc][i] = kMinShort;
256  }
257  }
258 #endif
259  auto& bcd = mBCData[ibc];
260  bcr.ir = bcd.ir;
261  int chEnt = bcd.ref.getFirstEntry();
262  for (int ic = 0; ic < bcd.ref.getEntries(); ic++) {
263  auto& chd = mChData[chEnt];
264  if (chd.id > IdDummy && chd.id < NChannels) {
265  bcr.ref[chd.id] = chEnt;
266  }
267  chEnt++;
268  }
269  }
270 
271  // Probably this is not necessary
272  // for(int itdc=0; itdc<NTDCChannels; itdc++){
273  // mReco.pattern[itdc]=0;
274  // for(int itb=0; itb<NTimeBinsPerBC; itb++){
275  // mReco.fired[itdc][itb]=0;
276  // }
277  // for(int isb=0; isb<mNSB; isb++){
278  // mReco.inter[itdc][isb]=0;
279  // }
280  // }
281 
282  // Assign interaction record and event information
283  for (int ibc = 0; ibc < mNBC; ibc++) {
284  mReco[ibc].ir = mBCData[ibc].ir;
285  mReco[ibc].channels = mBCData[ibc].channels;
286  mReco[ibc].triggers = mBCData[ibc].triggers;
287  }
288 
289  // Find consecutive bunch crossings and perform signal interpolation
290  // in the identified ranges (in the reconstruction method we take into
291  // account for signals that do not span the entire reange)
292  int seq_beg = 0;
293  int seq_end = 0;
294  LOG(info) << "Processing ZDC reconstruction for " << mNBC << " bunch crossings";
295  for (int ibc = 0; ibc < mNBC; ibc++) {
296  auto& ir = mBCData[seq_end].ir;
297  auto bcd = mBCData[ibc].ir.differenceInBC(ir);
298  if (bcd < 0) {
299  LOG(fatal) << "Orbit number is not increasing " << mBCData[seq_end].ir.orbit << "." << mBCData[seq_end].ir.bc << " followed by " << mBCData[ibc].ir.orbit << "." << mBCData[ibc].ir.bc;
300  return __LINE__;
301  } else if (bcd > 1) {
302  // Detected a gap
303  reconstruct(seq_beg, seq_end);
304  seq_beg = ibc;
305  seq_end = ibc;
306  } else if (ibc == (mNBC - 1)) {
307  // Last bunch
308  seq_end = ibc;
309  reconstruct(seq_beg, seq_end);
310  seq_beg = mNBC;
311  seq_end = mNBC;
312  } else {
313  // Look for another bunch
314  seq_end = ibc;
315  }
316  }
317 
318  return 0;
319 } // process

321 int DigiReco::reconstruct(int ibeg, int iend)
322 {
323  // Process consecutive BCs
324  if (ibeg == iend) {
325  if (mReco[ibeg].ir.bc == (o2::constants::lhc::LHCMaxBunches - 1)) {
326  mNLastLonely++;
327  } else {
328  mNLonely++;
329  LOG(info) << "Lonely bunch " << mReco[ibeg].ir.orbit << "." << mReco[ibeg].ir.bc;
330  }
331  return 0;
332  }
333 
334  if (mVerbosity >= DbgFull) {
335  LOG(info) << __func__ << "(" << ibeg << "," << iend << "): " << mReco[ibeg].ir.orbit << "." << mReco[ibeg].ir.bc << " - " << mReco[iend].ir.orbit << "." << mReco[iend].ir.bc;
336  }
337 
338  // Get reconstruction parameters
339  auto& ropt = RecoParamZDC::Instance();
340 
341  // Apply differential discrimination with triple condition
342  for (int itdc = 0; itdc < NTDCChannels; itdc++) {
343  // Check if channel has valid data for consecutive bunches in current bunch range
344  // N.B. there are events recorded from ibeg-iend but we are not sure if it is the
345  // case for every TDC channel
346  int istart = -1, istop = -1;
347  // Loop allows for gaps in the data sequence for each TDC channel
348  for (int ibun = ibeg; ibun <= iend; ibun++) {
349  if (mBCData[ibun].channels & mTDCMask[itdc]) { // TDC channel has data for this event
350  if (istart < 0) {
351  istart = ibun;
352  }
353  istop = ibun;
354  } else { // No data from channel
355  // A gap is detected
356  if (istart >= 0 && (istop - istart) > 0) {
357  // Need data for at least two consecutive bunch crossings
358  processTrigger(itdc, istart, istop);
359  }
360  istart = -1;
361  istop = -1;
362  }
363  }
364  // Check if there are consecutive bunch crossings at the end of group
365  if (istart >= 0 && (istop - istart) > 0) {
366  processTrigger(itdc, istart, istop);
367  }
368  }
369 
370  // Loop on bunches after trigger evaluation
371  // Reconstruct integrated charges and fill output tree
372  // TODO: compare average pedestal with estimation from current event
373  // TODO: failover in case of discrepancy
374  for (int ibun = ibeg; ibun <= iend; ibun++) {
375  updateOffsets(ibun);
376  auto& rec = mReco[ibun];
377  for (int itdc = 0; itdc < NTDCChannels; itdc++) {
378 #ifdef O2_ZDC_DEBUG
379  if (rec.fired[itdc] != 0x0) {
380  printf("%d %u.%u TDC %d [%s] %04hx -> ", ibun, rec.ir.orbit, rec.ir.bc, itdc, ChannelNames[TDCSignal[itdc]].data(), rec.fired[itdc]);
381  for (int isam = 0; isam < NTimeBinsPerBC; isam++) {
382  printf("%d", rec.fired[itdc] & mMask[isam] ? 1 : 0);
383  }
384  printf("\n");
385  }
386 #endif
387  rec.pattern[itdc] = 0;
388  for (int32_t i = 0; i < rec.ntdc[itdc]; i++) {
389 #ifdef O2_ZDC_DEBUG
390  LOG(info) << "tdc " << i << " [" << ChannelNames[TDCSignal[itdc]] << "] " << rec.TDCAmp[itdc][i] << " @ " << rec.TDCVal[itdc][i];
391 #endif
392  // There is a TDC value in the search zone around main-main position
393  if (std::abs(rec.TDCVal[itdc][i]) < ropt.tdc_search[itdc]) {
394  rec.pattern[itdc] = 1;
395  }
396 #ifdef O2_ZDC_DEBUG
397  else {
398  LOG(info) << rec.TDCVal[itdc][i] << " " << ropt.tdc_search[itdc];
399  }
400 #endif
401  }
402  }
403 #ifdef O2_ZDC_DEBUG
404  printf("%d %u.%-4u TDC PATTERN: ", ibun, mReco[ibun].ir.orbit, mReco[ibun].ir.bc);
405  for (int itdc = 0; itdc < NTDCChannels; itdc++) {
406  printf("%d", rec.pattern[itdc]);
407  }
408  printf("\n");
409 #endif
410  // Check if coincidence of common PM and sum of towers is satisfied
411  bool fired[NChannels] = {0};
412  // Side A
413  if ((rec.pattern[TDCZNAC] || ropt.bitset[TDCZNAC]) && (rec.pattern[TDCZNAS] || ropt.bitset[TDCZNAS])) {
414  for (int ich = IdZNAC; ich <= IdZNASum; ich++) {
415  fired[ich] = true;
416  }
417  }
418  if ((rec.pattern[TDCZPAC] || ropt.bitset[TDCZPAC]) && (rec.pattern[TDCZPAS] || ropt.bitset[TDCZPAS])) {
419  for (int ich = IdZPAC; ich <= IdZPASum; ich++) {
420  fired[ich] = true;
421  }
422  }
423  // ZEM1 and ZEM2 are not in coincidence
424  fired[IdZEM1] = rec.pattern[TDCZEM1];
425  fired[IdZEM2] = rec.pattern[TDCZEM2];
426  // Side C
427  if ((rec.pattern[TDCZNCC] || ropt.bitset[TDCZNCC]) && (rec.pattern[TDCZNCS] || ropt.bitset[TDCZNCS])) {
428  for (int ich = IdZNCC; ich <= IdZNCSum; ich++) {
429  fired[ich] = true;
430  }
431  }
432  if ((rec.pattern[TDCZPCC] || ropt.bitset[TDCZPCC]) && (rec.pattern[TDCZPCS] || ropt.bitset[TDCZPCS])) {
433  for (int ich = IdZPCC; ich <= IdZPCSum; ich++) {
434  fired[ich] = true;
435  }
436  }
437  // TODO: option to use a less restrictive definition of "fired" using for example
438  // just the autotrigger bits instead of using TDCs
439  if (mVerbosity >= DbgFull) {
440  printf("%d FIRED ", ibun);
441  printf("ZNA:%d%d%d%d%d%d ZPA:%d%d%d%d%d%d ZEM:%d%d ZNC:%d%d%d%d%d%d ZPC:%d%d%d%d%d%d\n",
442  fired[IdZNAC], fired[IdZNA1], fired[IdZNA2], fired[IdZNA3], fired[IdZNA4], fired[IdZNASum],
443  fired[IdZPAC], fired[IdZPA1], fired[IdZPA2], fired[IdZPA3], fired[IdZPA4], fired[IdZPASum],
444  fired[IdZEM1], fired[IdZEM2],
445  fired[IdZNCC], fired[IdZNC1], fired[IdZNC2], fired[IdZNC3], fired[IdZNC4], fired[IdZNCSum],
446  fired[IdZPCC], fired[IdZPC1], fired[IdZPC2], fired[IdZPC3], fired[IdZPC4], fired[IdZPCSum]);
447  }
448  // Get Orbit pedestals
449  updateOffsets(ibun);
450  for (int ich = 0; ich < NChannels; ich++) {
451  // Check if the corresponding TDC is fired
452  if (fired[ich]) {
453  // Check if channel data are present in payload
454  auto ref = mReco[ibun].ref[ich];
455  if (ref < ZDCRefInitVal) {
456  // Energy reconstruction Assignment of TDCs
457  // Compute event by event pedestal
458  bool hasEvPed = false;
459  float evPed = 0;
460  if (ibun > ibeg) {
461  auto ref_m = mReco[ibun - 1].ref[ich];
462  if (ropt.beg_ped_int[ich] >= 0 || ref_m < ZDCRefInitVal) {
463  for (int is = ropt.beg_ped_int[ich]; is <= ropt.end_ped_int[ich]; is++) {
464  if (is < 0) {
465  // Sample is in previous BC
466  evPed += float(mChData[ref_m].data[is + NTimeBinsPerBC]);
467  } else {
468  // Sample is in current BC
469  evPed += float(mChData[ref].data[is]);
470  }
471  }
472  evPed /= float(ropt.end_ped_int[ich] - ropt.beg_ped_int[ich] + 1);
473  hasEvPed = true;
474  }
475  }
476  float myPed = std::numeric_limits<float>::infinity();
477  // TODO: compare event pedestal with orbit pedestal to detect pile-up
478  // from previous bunch. If this is the case we use orbit pedestal
479  // instead of event pedestal
480  if (hasEvPed) {
481  myPed = evPed;
482  rec.adcPedEv[ich] = true;
483  } else if (mSource[ich] == PedOr) {
484  myPed = mOffset[ich];
485  rec.adcPedOr[ich] = true;
486  } else if (mSource[ich] == PedQC) {

561 void DigiReco::processTrigger(int itdc, int ibeg, int iend)
562 {
563 #ifdef O2_ZDC_DEBUG
564  LOG(info) << __func__ << "(itdc=" << itdc << "[" << ChannelNames[TDCSignal[itdc]] << "] ," << ibeg << "," << iend << "): " << mReco[ibeg].ir.orbit << "." << mReco[ibeg].ir.bc << " - " << mReco[iend].ir.orbit << "." << mReco[iend].ir.bc;
565 #endif
566  // Get reconstruction parameters
567  auto& ropt = RecoParamZDC::Instance();
568 
569  int nbun = iend - ibeg + 1;
570  int maxs2 = NTimeBinsPerBC * nbun - 1;
571  int shift = ropt.tsh[itdc];
572  int thr = ropt.tth[itdc];
573 
574  int is1 = 0, is2 = 1;
575  uint8_t isfired = 0;
576 #ifdef O2_ZDC_DEBUG
577  int16_t m[3] = {0};
578  int16_t s[3] = {0};
579 #endif
580  int it1 = 0, it2 = 0, ib1 = -1, ib2 = -1;
581  for (;;) {
582  // Shift data
583  isfired = isfired << 1;
584 #ifdef O2_ZDC_DEBUG
585  for (int i = 2; i > 0; i--) {
586  m[i] = m[i - 1];
587  s[i] = s[i - 1];
588  }
589 #endif
590  // Bunches and samples that are used in the difference
591  int b1 = ibeg + is1 / NTimeBinsPerBC;
592  int b2 = ibeg + is2 / NTimeBinsPerBC;
593  int s1 = is1 % NTimeBinsPerBC;
594  int s2 = is2 % NTimeBinsPerBC;
595  auto ref_m = mReco[b1].ref[TDCSignal[itdc]]; // reference to minuend
596  auto ref_s = mReco[b2].ref[TDCSignal[itdc]]; // reference to subtrahend
597  // Check data consistency before computing difference
598  if (ref_m == ZDCRefInitVal || ref_s == ZDCRefInitVal) {
599  LOG(fatal) << "Missing information for bunch crossing";
600  return;
601  }
602  // TODO: More checks that bunch crossings are indeed consecutive
603  int diff = mChData[ref_m].data[s1] - mChData[ref_s].data[s2];
604  // Triple trigger condition
605 #ifdef O2_ZDC_DEBUG
606  m[0] = mChData[ref_m].data[s1];
607  s[0] = mChData[ref_s].data[s2];
608 #endif
609  if (diff > thr) {
610  isfired = isfired | 0x1;
611  if ((isfired & 0x7) == 0x7) {
612  // Fired bit is assigned to the second sample, i.e. to the one that can identify the
613  // signal peak position
614  mReco[b2].fired[itdc] |= mMask[s2];
615 #ifdef O2_ZDC_DEBUG
616  LOG(info) << itdc << " " << ChannelNames[TDCSignal[itdc]] << " Fired @ " << mReco[b2].ir.orbit << "." << mReco[b2].ir.bc << ".s" << s2
617  << " (" << m[2] << " - (" << s[2] << ")) = " << (m[2] - s[2]) << " > " << thr
618  << " && (" << m[1] << " - (" << s[1] << ")) = " << (m[1] - s[1]) << " > " << thr
619  << " && (s" << s1 << ":" << m[0] << " - (s" << s2 << ":" << s[0] << ")) = " << diff << " > " << thr;
620 #endif
621  }
622  }
623  if (is2 >= shift) {
624  is1++;
625  }
626  if (is2 < maxs2) {
627  is2++;
628  }
629  if (is1 == maxs2) {
630  break;
631  }
632  }
633  interpolate(itdc, ibeg, iend);
634 } // processTrigger

636 O2_ZDC_DIGIRECO_FLT DigiReco::getPoint(int itdc, int ibeg, int iend, int i)
637 {
638  constexpr int nsbun = TSN * NTimeBinsPerBC; // Total number of interpolated points per bunch crossing
639  if (i >= mNtot || i < 0) {
640  LOG(fatal) << "Error addressing TDC itdc=" << itdc << " i=" << i << " mNtot=" << mNtot;
641  return std::numeric_limits<float>::infinity();
642  }
643  // Constant extrapolation at the beginning and at the end of the array
644  if (i < TSNH) {
645  // Return value of first sample
646  return mFirstSample;
647  } else if (i >= mIlast) {
648  // Return value of last sample
649  return mLastSample;
650  } else {
651  // Identification of the point to be assigned
652  int ibun = ibeg + i / nsbun;
653  // Interpolation between acquired points (N.B. from 0 to mNint)
654  i = i - TSNH;
655  int im = i % TSN;
656  if (im == 0) {
657  // This is an acquired point
658  int ip = (i / TSN) % NTimeBinsPerBC;
659  int ib = ibeg + (i / TSN) / NTimeBinsPerBC;
660  if (ib != ibun) {
661  LOG(fatal) << "ib=" << ib << " ibun=" << ibun;
662  return std::numeric_limits<float>::infinity();
663  }
664  return mChData[mReco[ibun].ref[TDCSignal[itdc]]].data[ip];
665  } else {
666  // Do the actual interpolation
667  O2_ZDC_DIGIRECO_FLT y = 0;
668  int ip = i / TSN;
669  O2_ZDC_DIGIRECO_FLT sum = 0;
670  for (int is = TSN - im, ii = ip - TSL + 1; is < NTS; is += TSN, ii++) {
671  // Default is first point in the array
672  O2_ZDC_DIGIRECO_FLT yy = mFirstSample;
673  if (ii > 0) {
674  if (ii < mNsam) {
675  int ip = ii % NTimeBinsPerBC;
676  int ib = ibeg + ii / NTimeBinsPerBC;
677  yy = mChData[mReco[ib].ref[TDCSignal[itdc]]].data[ip];
678  } else {
679  // Last acquired point
680  yy = mLastSample;
681  }
682  }
683  sum += mTS[is];
684  y += yy * mTS[is];
685  }
686  y = y / sum;
687  return y;
688  }
689  }

752 void DigiReco::interpolate(int itdc, int ibeg, int iend)
753 {
754 #ifdef O2_ZDC_DEBUG
755  LOG(info) << __func__ << "(itdc=" << itdc << "[" << ChannelNames[TDCSignal[itdc]] << "] ," << ibeg << "," << iend << "): " << mReco[ibeg].ir.orbit << "." << mReco[ibeg].ir.bc << " - " << mReco[iend].ir.orbit << "." << mReco[iend].ir.bc;
756 #endif
757  // TODO: get data from preceding time frame
758  constexpr int MaxTimeBin = NTimeBinsPerBC - 1; //< number of samples per BC
759  constexpr int nsbun = TSN * NTimeBinsPerBC; // Total number of interpolated points per bunch crossing
760  // Set data members for interpolation of the current TDC
761  mNbun = iend - ibeg + 1; // Number of adjacent bunches
762  mNsam = mNbun * NTimeBinsPerBC; // Number of acquired samples
763  mNtot = mNsam * TSN; // Total number of points in the interpolated arrays
764  mNint = (mNbun * NTimeBinsPerBC - 1) * TSN; // Total points in the interpolation region (-1)
765  mIlast = mNtot - TSNH; // Index of last acquired sample
766 
767  constexpr int nsp = 5; // Number of points to be searched
768 
769  // At this level there should be no need to check if the TDC channel is connected
770  // since a fatal should have been raised already
771  for (int ibun = ibeg; ibun <= iend; ibun++) {
772  auto ref = mReco[ibun].ref[TDCSignal[itdc]];
773  if (ref == ZDCRefInitVal) {
774  LOG(fatal) << "Missing information for bunch crossing";
775  }
776  }
777 
778  // Get reconstruction parameters
779  auto& ropt = RecoParamZDC::Instance();
780 
781  int imod = ropt.tmod[itdc]; // Module corresponding to TDC channel
782  int ich = ropt.tch[itdc]; // Hardware channel corresponding to TDC channel
783 
784  auto ref_beg = mReco[ibeg].ref[TDCSignal[itdc]];
785  auto ref_end = mReco[iend].ref[TDCSignal[itdc]];
786 
787  mFirstSample = mChData[ref_beg].data[0];
788  mLastSample = mChData[ref_end].data[NTimeBinsPerBC - 1];
789 
790  // Full interpolation
791 #ifdef O2_ZDC_INTERP_DEBUG
792  for (int i = 0; i < mNtot; i++) {
793  setPoint(itdc, ibeg, iend, i);
794  }
795 #endif
796 
797  // Looking for a local maximum in a search zone
798  float amp = std::numeric_limits<float>::infinity(); // Amplitude to be stored
799  int isam_amp = 0; // Sample at maximum amplitude (relative to beginning of group)
800  int ip_old = -1, ip_cur = -1, ib_cur = -1; // Current and old points
801  bool is_searchable = false; // Flag for point in the search zone for maximum amplitude
802  bool was_searchable = false; // Flag for point in the search zone for maximum amplitude
803  int ib[nsp] = {-1, -1, -1, -1, -1};
804  int ip[nsp] = {-1, -1, -1, -1, -1};
805  // N.B. Points at the extremes are constant therefore no local maximum
806  // can occur in these two regions
807  for (int i = 0; i < mNint; i++) {
808  int isam = i + TSNH;
809  // Check if trigger is fired for this point
810  // For the moment we don't take into account possible extensions of the search zone
811  // ip_cur can span several bunches and is used just to identify transitions
812  ip_cur = isam / TSN;
813  // Speed up computation
814  if (ip_cur != ip_old) {
815  ip_old = ip_cur;
816  for (int j = 0; j < 5; j++) {
817  ib[j] = -1;
818  ip[j] = -1;
819  }
820  // There are three possible triple conditions that involve current point (middle is current point)
821  ip[2] = ip_cur % NTimeBinsPerBC;
822  ib[2] = ibeg + ip_cur / NTimeBinsPerBC;
823  ib_cur = ib[2];
824  if (ip[2] > 0) {
825  ip[1] = ip[2] - 1;
826  ib[1] = ib[2];
827  } else if (ip[2] == 0) {
828  if (ib[2] > ibeg) {
829  ib[1] = ib[2] - 1;
830  ip[1] = MaxTimeBin;
831  }
832  }
833  if (ip[1] > 0) {
834  ip[0] = ip[1] - 1;
835  ib[0] = ib[1];
836  } else if (ip[1] == 0) {
837  if (ib[1] > ibeg) {
838  ib[0] = ib[1] - 1;
839  ip[0] = MaxTimeBin;
840  }
841  }
842  if (ip[2] < MaxTimeBin) {
843  ip[3] = ip[2] + 1;
844  ib[3] = ib[2];
845  } else if (ip[2] == MaxTimeBin) {
846  if (ib[2] < iend) {
847  ib[3] = ib[2] + 1;
848  ip[3] = 0;
849  }
850  }
851  if (ip[3] < MaxTimeBin) {
852  ip[4] = ip[3] + 1;
853  ib[4] = ib[3];
854  } else if (ip[3] == MaxTimeBin) {
855  if (ib[3] < iend) {
856  ib[4] = ib[3] + 1;
857  ip[4] = 0;
858  }
859  }
860  // meet the threshold condition
861  was_searchable = is_searchable;
862  // Search conditions with list of allowed patterns
863  // No need to double check ib[?] and ip[?] because either we assign both or none
864  uint16_t triggered = 0x0000;
865  for (int j = 0; j < 5; j++) {
866  if (ib[j] >= 0 && (mReco[ib[j]].fired[itdc] & mMask[ip[j]]) > 0) {
867  triggered |= (0x1 << j);
868  }
869  }
870  // Reject conditions:
871  // 00000
872  // 10001
873  // One among 10000 and 00001
874  // Accept conditions:
875  constexpr uint16_t accept[14] = {
876  // 0x01, // 00001 extend search zone before maximum
877  0x02, // 00010
878  0x04, // 00100
879  0x08, // 01000
880  0x10, // 10000 extend after
881  0x03, // 00011
882  0x06, // 00110
883  0x0c, // 01100
884  0x18, // 11000
885  0x07, // 00111
886  0x0e, // 01110
887  0x1c, // 11100
888  0x0f, // 01111
889  0x1e, // 11110
890  0x1f // 11111
891  };
892  // All other are not correct (->reject)
893  is_searchable = 0;
894  if (triggered != 0) {
895  for (int j = 0; j < 14; j++) {
896  if (triggered == accept[j]) {
897  is_searchable = 1;
898  break;
899  }
900  }
901  }
902  }
903  // We do not restrict search zone around expected main-main collision
904  // because we would like to be able to identify pile-up from collisions
905  // with satellites (buggy)
906 
907  // If we exit from searching zone
908  if (was_searchable && !is_searchable) {
909  if (amp <= ADCMax) {
910  // Store identified peak
911  int ibun = ibeg + isam_amp / nsbun;
912  updateOffsets(ibun);
913  if (mSource[ich] != PedND) {
914  amp = mOffset[ich] - amp;
915  } else {
916  LOGF(error, "%u.%-4d Missing pedestal for TDC %d %s ", mBCData[ibun].ir.orbit, mBCData[ibun].ir.bc, itdc, ChannelNames[TDCSignal[itdc]]);
917  amp = std::numeric_limits<float>::infinity();
918  }
919  int tdc = isam_amp % nsbun;
920  assignTDC(ibun, ibeg, iend, itdc, tdc, amp);
921  }
922  amp = std::numeric_limits<float>::infinity();
923  isam_amp = 0;
924  was_searchable = 0;
925  }
926  if (is_searchable) {
927  int mysam = isam % nsbun;
928 #ifndef O2_ZDC_INTERP_DEBUG
929  // Perform interpolation for the searched point
930  //setPoint(itdc, ibeg, iend, isam);
931  float myval = getPoint(itdc, ibeg, iend, isam);
932 #else
933  float myval = mReco[ib_cur].inter[itdc][mysam];
934 #endif
935  if (myval < amp) {
936  amp = myval;
937  isam_amp = isam;
938  }
939  }
940  }
941  // Trigger flag still present at the of the scan
942  if (is_searchable) {
943  // Add last identified peak
944  if (amp <= ADCMax) {
945  // Store identified peak
946  int ibun = ibeg + isam_amp / nsbun;
947  updateOffsets(ibun);
948  if (mSource[ich] != PedND) {
949  amp = mOffset[ich] - amp;
950  } else {
951  LOGF(error, "%u.%-4d Missing pedestal for TDC %d %s ", mBCData[ibun].ir.orbit, mBCData[ibun].ir.bc, itdc, ChannelNames[TDCSignal[itdc]]);
952  amp = std::numeric_limits<float>::infinity();
953  }
954  int tdc = isam_amp % nsbun;
955  assignTDC(ibun, ibeg, iend, itdc, tdc, amp);
956  }
957  }
958 } // interpolate

960 void DigiReco::assignTDC(int ibun, int ibeg, int iend, int itdc, int tdc, float amp)
961 {
962  constexpr int nsbun = TSN * NTimeBinsPerBC; // Total number of interpolated points per bunch crossing
963  constexpr int tdc_max = nsbun / 2;
964  constexpr int tdc_min = -tdc_max;
965 
966  // Apply tdc shift correction
967  int tdc_cor = tdc - tdc_shift[itdc];
968  // Correct bunch assignment
969  if (tdc_cor < tdc_min && ibun >= ibeg) {
970  // Assign to preceding bunch
971  ibun = ibun - 1;
972  tdc_cor = tdc_cor + nsbun;
973  } else if (tdc_cor >= tdc_max && ibun < iend) {
974  // Assign to following bunch
975  ibun = ibun + 1;
976  tdc_cor = tdc_cor - nsbun;
977  }
978  if (tdc_cor < kMinShort) {
979  LOG(error) << "TDC " << itdc << " " << tdc_cor << " is out of range";
980  tdc_cor = kMinShort;
981  }
982  if (tdc_cor > kMaxShort) {
983  LOG(error) << "TDC " << itdc << " " << tdc_cor << " is out of range";
984  tdc_cor = kMaxShort;
985  }
986 
987  auto& rec = mReco[ibun];
988 
989  // Assign to correct bunch
990  auto myamp = std::nearbyint(amp / FTDCAmp);
991  rec.TDCVal[itdc].push_back(tdc_cor);
992  rec.TDCAmp[itdc].push_back(myamp);
993  int& ihit = mReco[ibun].ntdc[itdc];
994 #ifdef O2_ZDC_TDC_C_ARRAY
995  if (ihit < MaxTDCValues) {
996  rec.tdcVal[itdc][ihit] = tdc_cor;
997  rec.tdcAmp[itdc][ihit] = myamp;
998  } else {
999  LOG(error) << rec.ir.orbit << "." << rec.ir.bc << " "
1000  << "ibun=" << ibun << " itdc=" << itdc << " tdc=" << tdc << " tdc_cor=" << tdc_cor * FTDCVal << " amp=" << amp * FTDCAmp << " OVERFLOW";
1001  }
1002 #endif
1003  int ich = TDCSignal[itdc];
1004  // Assign info about pedestal subtration
1005  if (mSource[ich] == PedOr) {
1006  rec.tdcPedOr[ich] = true;
1007  } else if (mSource[ich] == PedQC) {
1008  rec.tdcPedQC[ich] = true;
1009  } else if (mSource[ich] == PedEv) {
1010  // In present implementation this never happens
1011  rec.tdcPedEv[ich] = true;
1012  } else {
1013  rec.tdcPedMissing[ich] = true;
1014  }
1015 #ifdef O2_ZDC_DEBUG
1016  LOG(info) << mReco[ibun].ir.orbit << "." << mReco[ibun].ir.bc
1017  << " ibun=" << ibun << " itdc=" << itdc << " tdc=" << tdc << " tdc_cor=" << tdc_cor * FTDCVal << " amp=" << amp << " -> " << myamp
1018  << " pedSrc = " << mSource[ich];
1019 #endif
1020  ihit++;
1021 } // assignTDC
