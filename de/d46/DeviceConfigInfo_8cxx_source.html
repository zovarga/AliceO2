21 // Parses a config entry in the form
22 //
23 // [CONFIG] <key>=<vaue> <timestamp> <provenance>
24 bool DeviceConfigHelper::parseConfig(std::string_view s, ParsedConfigMatch& match)
25 {
26  const char* begin = s.begin();
27  const char* end = s.end();
28  if (s.size() > 17 && (strncmp("[CONFIG];", begin + 17, 9) != 0)) {
29  return false;
30  }
31  if (s.size() < 17 + 9) {
32  return false;
33  }
34  match.beginKey = s.data() + 9 + 17;
35  match.endKey = (char const*)memchr(match.beginKey, '=', s.size() - 9);
36  if (match.endKey == nullptr) {
37  return false;
38  }
39  match.beginValue = match.endKey + 1;
40  match.endValue = (char const*)memchr(match.beginValue, ';', end - match.beginValue);
41  if (match.endValue == nullptr) {
42  return false;
43  }
44 
45  char* next = nullptr;
46  match.timestamp = strtoll(match.endValue + 1, &next, 10);
47 
48  if (!next || *next != ';') {
49  return false;
50  }
51 
52  match.beginProvenance = next + 1;
53  match.endProvenance = end;
54 
55  return true;
56 }

58 bool DeviceConfigHelper::processConfig(ParsedConfigMatch& match,
59  DeviceInfo& info)
60 {
61  if (match.beginKey == nullptr || match.endKey == nullptr ||
62  match.beginValue == nullptr || match.endValue == nullptr ||
63  match.beginProvenance == nullptr || match.endProvenance == nullptr) {
64  return false;
65  }
66  auto keyString = std::string(match.beginKey, match.endKey - match.beginKey);
67  auto valueString = std::string(match.beginValue, match.endValue - match.beginValue);
68  auto provenanceString = std::string(match.beginProvenance, match.endProvenance - match.beginProvenance);
69  boost::property_tree::ptree branch;
70  std::stringstream ss{valueString};
71  try {
72  boost::property_tree::json_parser::read_json(ss, branch);
73  info.currentConfig.put_child(keyString, branch);
74  } catch (boost::exception&) {
75  // in case it is not a tree but a single value
76  info.currentConfig.put(keyString, valueString);
77  }
78 
79  info.currentProvenance.put(keyString, provenanceString);
80  return true;
81 }
