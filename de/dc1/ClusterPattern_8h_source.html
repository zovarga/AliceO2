146 template <typename Processor>
147 void ClusterPattern::process(Processor procRowCol)
148 {
149  auto cspan = getColumnSpan(), rspan = getRowSpan();
150  uint32_t nBits = cspan * rspan;
151  uint32_t nBytes = (nBits >> 3) + (nBits % 8 != 0);
152  const auto bmap = mBitmap.data() + 2;
153  uint16_t ic = 0, ir = 0;
154  for (unsigned int i = 0; i < nBytes; i++) {
155  int s = 128; // 0b10000000
156  while (s > 0) {
157  if ((bmap[i] & s) != 0) {
158  procRowCol(ir, ic);
159  }
160  ic++;
161  s >>= 1;
162  if (uint32_t(ir + 1) * ic == nBits) {
163  break;
164  }
165  if (ic == cspan) {
166  ic = 0;
167  ir++;
168  }
169  }
170  if (uint32_t(ir + 1) * ic == nBits) {
171  break;
172  }
173  }
174 }
