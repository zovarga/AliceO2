52 DataSamplingPolicy DataSamplingPolicy::fromConfiguration(const ptree& config)
53 {
54  auto name = config.get<std::string>("id");
55  DataSamplingPolicy policy(name);
56 
57  size_t outputId = 0;
58  std::vector<InputSpec> inputSpecs = DataDescriptorQueryBuilder::parse(config.get<std::string>("query").c_str());
59  std::vector<OutputSpec> outputSpecs;
60  // Optionally user can specify the outputs,
61  if (auto outputsQuery = config.get<std::string>("outputs", ""); !outputsQuery.empty()) {
62  std::vector<InputSpec> outputsAsInputSpecs = DataDescriptorQueryBuilder::parse(outputsQuery.c_str());
63  if (outputsAsInputSpecs.size() != inputSpecs.size()) {
64  throw std::runtime_error(
65  "The number of outputs should match the number of inputs (queries),"
66  " which is not the case for the policy '" +
67  name + "'(" +
68  std::to_string(inputSpecs.size()) + " inputs vs. " + std::to_string(outputsAsInputSpecs.size()) + " outputs).");
69  }
70  for (const auto& outputAsInputSpec : outputsAsInputSpecs) {
71  outputSpecs.emplace_back(DataSpecUtils::asOutputSpec(outputAsInputSpec));
72  outputSpecs.back().lifetime = Lifetime::QA;
73  }
74  } else { // otherwise default format will be used
75  for (const auto& inputSpec : inputSpecs) {
76  if (DataSpecUtils::getOptionalSubSpec(inputSpec).has_value()) {
77  outputSpecs.emplace_back(OutputSpec{
78  {inputSpec.binding},
79  createPolicyDataOrigin(),
80  createPolicyDataDescription(name, outputId++),
81  DataSpecUtils::getOptionalSubSpec(inputSpec).value(),
82  Lifetime::QA});
83  } else {
84  outputSpecs.emplace_back(OutputSpec{
85  {inputSpec.binding},
86  {createPolicyDataOrigin(), createPolicyDataDescription(name, outputId++)},
87  Lifetime::QA});
88  }
89  }
90  }
91  assert(inputSpecs.size() == outputSpecs.size());
92  for (size_t i = 0; i < inputSpecs.size(); i++) {
93  policy.registerPath(inputSpecs[i], outputSpecs[i]);
94  }
95 
96  for (const auto& conditionConfig : config.get_child("samplingConditions")) {
97  auto condition = DataSamplingConditionFactory::create(conditionConfig.second.get<std::string>("condition"));
98  condition->configure(conditionConfig.second);
99  policy.registerCondition(std::move(condition));
100  }
101 
102  policy.setFairMQOutputChannel(config.get_optional<std::string>("fairMQOutput").value_or(""));
103 
104  return policy;
105 }

113 bool DataSamplingPolicy::decide(const o2::framework::DataRef& dataRef)
114 {
115  bool decision = std::all_of(mConditions.begin(), mConditions.end(),
116  [dataRef](std::unique_ptr<DataSamplingCondition>& condition) {
117  return condition->decide(dataRef);
118  });
119 
120  mTotalAcceptedMessages += decision;
121  mTotalEvaluatedMessages++;
122 
123  return decision;
124 }

174 header::DataDescription DataSamplingPolicy::createPolicyDataDescription(std::string policyName, size_t id)
175 {
176  if (id > 99) {
177  throw std::runtime_error("Maximum 100 inputs in DataSamplingPolicy are supported. Call the developers if you really need more.");
178  }
179 
180  if (policyName.size() > 14) {
181  LOG(warning) << "DataSamplingPolicy name '" << policyName << "' is longer than 14 characters, we have to trim it. "
182  << "Use a shorter policy name to avoid potential output name conflicts.";
183  policyName.resize(14);
184  }
185 
186  header::DataDescription outputDescription;
187  outputDescription.runtimeInit((policyName + std::to_string(id)).c_str());
188  return outputDescription;
189 }
