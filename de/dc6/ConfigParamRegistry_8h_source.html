65  bool isDefault(const char* key) const
66  {
67  return mStore->store().count(key) > 0 && mStore->provenance(key) != "default";
68  }

70  template <typename T>
71  T get(const char* key) const
72  {
73  assert(mStore.get());
74  try {
75  if constexpr (isSimpleType<T>()) {
76  return mStore->store().get<T>(key);
77  } else if constexpr (std::is_same_v<T, std::string>) {
78  return mStore->store().get<std::string>(key);
79  } else if constexpr (std::is_same_v<T, std::string_view>) {
80  return std::string_view{mStore->store().get<std::string>(key)};
81  } else if constexpr (is_base_of_template<std::vector, T>::value) {
82  return vectorFromBranch<typename T::value_type>(mStore->store().get_child(key));
83  } else if constexpr (is_base_of_template<o2::framework::Array2D, T>::value) {
84  return array2DFromBranch<typename T::element_t>(mStore->store().get_child(key));
85  } else if constexpr (is_base_of_template<o2::framework::LabeledArray, T>::value) {
86  return labeledArrayFromBranch<typename T::element_t>(mStore->store().get_child(key));
87  } else if constexpr (std::is_same_v<T, boost::property_tree::ptree>) {
88  return mStore->store().get_child(key);
89  } else if constexpr (std::is_constructible_v<T, boost::property_tree::ptree>) {
90  return T{mStore->store().get_child(key)};
91  } else if constexpr (std::is_constructible_v<T, boost::property_tree::ptree> == false) {
92  static_assert(std::is_constructible_v<T, boost::property_tree::ptree> == false,
93  "Not a basic type and no constructor from ptree provided");
94  }
95  } catch (std::exception& e) {
96  throw std::invalid_argument(std::string("missing option: ") + key + " (" + e.what() + ")");
97  } catch (...) {
98  throw std::invalid_argument(std::string("error parsing option: ") + key);
99  }
100  throw std::invalid_argument(std::string("bad type for option: ") + key);
101  }
102 
103  private:
104  std::unique_ptr<ConfigParamStore> mStore;
105 };
