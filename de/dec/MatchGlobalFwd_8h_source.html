165  template <typename T>
166  bool propagateToNextClusterWithMCS(T& track, double z, int& startingLayerID, const int& newLayerID)
167  {
168  // Propagate track to the next cluster z position, adding angular MCS effects at the center of
169  // each disk crossed by the track. This method is valid only for track propagation between
170  // clusters at MFT layers positions. The startingLayerID is updated.
171 
172  if (startingLayerID == newLayerID) { // Same layer, nothing to do.
173  LOG(debug) << " => Propagate to next cluster with MCS : startingLayerID = " << startingLayerID << " = > "
174  << " newLayerID = " << newLayerID << " (NLayers = " << std::abs(newLayerID - startingLayerID)
175  << ") ; track.getZ() = " << track.getZ() << " => "
176  << "destination cluster z = " << z << " ; => Same layer: no MCS effects.";
177 
178  if (z != track.getZ()) {
179  track.propagateToZ(z, mBz);
180  }
181  return true;
182  }
183 
184  using o2::mft::constants::LayerZPosition;
185  auto startingZ = track.getZ();
186 
187  // https://stackoverflow.com/questions/1903954/is-there-a-standard-sign-function-signum-sgn-in-c-c
188  auto signum = [](auto a) {
189  return (0 < a) - (a < 0);
190  };
191  int direction = signum(newLayerID - startingLayerID); // takes values +1, 0, -1
192  auto currentLayer = startingLayerID;
193 
194  LOG(debug) << " => Propagate to next cluster with MCS : startingLayerID = " << startingLayerID << " = > "
195  << " newLayerID = " << newLayerID << " (NLayers = " << std::abs(newLayerID - startingLayerID)
196  << ") ; track.getZ() = " << track.getZ() << " => "
197  << "destination cluster z = " << z << " ; ";
198 
199  // Number of disks crossed by this track segment
200  while (currentLayer != newLayerID) {
201  auto nextlayer = currentLayer + direction;
202  auto nextZ = LayerZPosition[nextlayer];
203 
204  int NDisksMS;
205  if (nextZ - track.getZ() > 0) {
206  NDisksMS = (currentLayer % 2 == 0) ? (currentLayer - nextlayer) / 2 : (currentLayer - nextlayer + 1) / 2;
207  } else {
208  NDisksMS = (currentLayer % 2 == 0) ? (nextlayer - currentLayer + 1) / 2 : (nextlayer - currentLayer) / 2;
209  }
210 
211  LOG(debug) << "currentLayer = " << currentLayer << " ; "
212  << "nextlayer = " << nextlayer << " ; "
213  << "track.getZ() = " << track.getZ() << " ; "
214  << "nextZ = " << nextZ << " ; "
215  << "NDisksMS = " << NDisksMS;
216 
217  if ((NDisksMS * mMFTDiskThicknessInX0) != 0) {
218  track.addMCSEffect(NDisksMS * mMFTDiskThicknessInX0);
219  LOG(debug) << "Track covariances after MCS effects:";
220  LOG(debug) << track.getCovariances() << std::endl;
221  }
222 
223  LOG(debug) << " BeforeExtrap: X = " << track.getX() << " Y = " << track.getY() << " Z = " << track.getZ() << " Tgl = " << track.getTanl() << " Phi = " << track.getPhi() << " pz = " << track.getPz() << " q/pt = " << track.getInvQPt() << std::endl;
224 
225  track.propagateToZ(nextZ, mBz);
226 
227  currentLayer = nextlayer;
228  }
229  if (z != track.getZ()) {
230  track.propagateToZ(z, mBz);
231  }
232  startingLayerID = newLayerID;
233  return true;
234  }
