102  // the decodeTF() function processes the messages generated by the (sub)TimeFrame builder
103  void decodeTF(framework::ProcessingContext& pc)
104  {
105  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true));
106  mFirstTForbit = dh->firstTForbit;
107 
108  mDecoder->setFirstOrbitInTF(mFirstTForbit);
109 
110  if (mDebug) {
111  LOG(info) << "[DataDecoderSpec::run] first TF orbit is " << mFirstTForbit;
112  }
113 
114  // get the input buffer
115  auto& inputs = pc.inputs();
116  DPLRawParser parser(inputs, o2::framework::select(mInputSpec.c_str()));
117  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
118  auto const* raw = it.raw();
119  if (!raw) {
120  continue;
121  }
122  size_t payloadSize = it.size();
123 
124  gsl::span<const std::byte> buffer(reinterpret_cast<const std::byte*>(raw), sizeof(RDH) + payloadSize);
125  mDecoder->decodeBuffer(buffer);
126  }
127  }

130  // the decodeReadout() function processes the messages generated by o2-mch-cru-page-reader-workflow
131  void decodeReadout(const o2::framework::DataRef& input)
132  {
133  static int nFrame = 1;
134  // get the input buffer
135  if (input.spec->binding != "readout") {
136  return;
137  }
138 
139  auto const* raw = input.payload;
140  // size of payload
141  size_t payloadSize = DataRefUtils::getPayloadSize(input);
142 
143  if (mDebug) {
144  std::cout << nFrame << " payloadSize=" << payloadSize << std::endl;
145  }
146  nFrame += 1;
147  if (payloadSize == 0) {
148  return;
149  }
150 
151  const RDH* rdh = reinterpret_cast<const RDH*>(raw);
152  mFirstTForbit = o2::raw::RDHUtils::getHeartBeatOrbit(rdh);
153  mDecoder->setFirstOrbitInTF(mFirstTForbit);
154 
155  gsl::span<const std::byte> buffer(reinterpret_cast<const std::byte*>(raw), payloadSize);
156  mDecoder->decodeBuffer(buffer);
157  }

169  bool isDroppedTF(framework::ProcessingContext& pc)
170  {
175  constexpr auto origin = header::gDataOriginMCH;
176  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
177  o2::framework::InputSpec dummy{"dummy",
178  framework::ConcreteDataMatcher{origin,
179  header::gDataDescriptionRawData,
180  0xDEADBEEF}};
181  for (const auto& ref : o2::framework::InputRecordWalker(pc.inputs(), {dummy})) {
182  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
183  if (dh->payloadSize == 0) {
184  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
185  if (++contDeadBeef <= maxWarn) {
186  LOGP(warning, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
187  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize,
188  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
189  }
190  return true;
191  }
192  }
193  contDeadBeef = 0; // if good data, reset the counter
194  return false;
195  }

197  //_________________________________________________________________________________________________
198  void run(framework::ProcessingContext& pc)
199  {
200  if (isDroppedTF(pc)) {
201  sendEmptyOutput(pc.outputs());
202  return;
203  }
204 
205  static int32_t deltaMax = 0;
206 
207  auto createBuffer = [&](auto& vec, size_t& size) {
208  size = vec.empty() ? 0 : sizeof(*(vec.begin())) * vec.size();
209  char* buf = nullptr;
210  if (size > 0) {
211  buf = (char*)malloc(size);
212  if (buf) {
213  char* p = buf;
214  size_t sizeofElement = sizeof(*(vec.begin()));
215  for (auto& element : vec) {
216  memcpy(p, &element, sizeofElement);
217  p += sizeofElement;
218  }
219  }
220  }
221  return buf;
222  };
223 
224  auto tStart = std::chrono::high_resolution_clock::now();
225  mDecoder->reset();
226  for (auto&& input : pc.inputs()) {
227  if (input.spec->binding == "readout") {
228  decodeReadout(input);
229  }
230  if (input.spec->binding == "TF") {
231  decodeTF(pc);
232  }
233  }
234  mDecoder->computeDigitsTime();
235  auto tEnd = std::chrono::high_resolution_clock::now();
236  mTimeDecoding += tEnd - tStart;
237 
238  auto& digits = mDecoder->getDigits();
239  auto& orbits = mDecoder->getOrbits();
240 
241 #ifdef MCH_RAW_DATADECODER_DEBUG_DIGIT_TIME
242  constexpr int BCINORBIT = o2::constants::lhc::LHCMaxBunches;
243  int32_t bcMax = mNHBPerTF * 3564 - 1;
244  for (auto d : digits) {
245  if (d.getTime() < 0 || d.getTime() > bcMax) {
246  int32_t delta = d.getTime() - bcMax;
247  if (delta > deltaMax) {
248  deltaMax = delta;
249  std::cout << "Max digit time exceeded: TF ORBIT " << mDecoder->getFirstOrbitInTF().value() << " DE# " << d.getDetID() << " PadId " << d.getPadID() << " ADC " << d.getADC()
250  << " time " << d.getTime() << " (" << bcMax << ", delta=" << delta << ")" << std::endl;
251  }
252  }
253  }
254 #endif
255 
256  tStart = std::chrono::high_resolution_clock::now();
257  ROFFinder rofFinder(digits, mFirstTForbit);
258  rofFinder.process(mDummyROFs);
259  tEnd = std::chrono::high_resolution_clock::now();
260  mTimeROFFinder += tEnd - tStart;
261 
262  if (mDebug) {
263  rofFinder.dumpOutputDigits();
264  rofFinder.dumpOutputROFs();
265  }
266 
267  if (mCheckROFs) {
268  rofFinder.isRofTimeMonotonic();
269  rofFinder.isDigitsTimeAligned();
270  }
271 
272  // send the output buffer via DPL
273  size_t digitsSize, rofsSize, orbitsSize;
274  char* digitsBuffer = rofFinder.saveDigitsToBuffer(digitsSize);
275  char* rofsBuffer = rofFinder.saveROFRsToBuffer(rofsSize);
276  char* orbitsBuffer = createBuffer(orbits, orbitsSize);
277 
278  if (mDebug) {
279  std::cout << "digitsSize " << digitsSize << " rofsSize " << rofsSize << " orbitsSize " << orbitsSize << std::endl;
280  }
281 
282  // create the output message
283  auto freefct = [](void* data, void*) { free(data); };
284  pc.outputs().adoptChunk(Output{header::gDataOriginMCH, "DIGITS", 0}, digitsBuffer, digitsSize, freefct, nullptr);
285  pc.outputs().adoptChunk(Output{header::gDataOriginMCH, "DIGITROFS", 0}, rofsBuffer, rofsSize, freefct, nullptr);
286  pc.outputs().adoptChunk(Output{header::gDataOriginMCH, "ORBITS", 0}, orbitsBuffer, orbitsSize, freefct, nullptr);
287 
288  mTFcount += 1;
289  if (mErrorLogFrequency) {
290  if (mTFcount == 1 || mTFcount % mErrorLogFrequency == 0) {
291  mDecoder->logErrorMap(mTFcount);
292  }
293  }
294  }

312 o2::framework::DataProcessorSpec getDecodingSpec(const char* specName, std::string inputSpec,
313  bool askSTFDist)
314 {
315  auto inputs = o2::framework::select(inputSpec.c_str());
316  for (auto& inp : inputs) {
317  // mark input as optional in order not to block the workflow
318  // if our raw data happen to be missing in some TFs
319  inp.lifetime = Lifetime::Optional;
320  }
321  if (askSTFDist) {
322  // request the input FLP/DISTSUBTIMEFRAME/0 that is _guaranteed_
323  // to be present, even if none of our raw data is present.
324  inputs.emplace_back("stfDist", "FLP", "DISTSUBTIMEFRAME", 0, o2::framework::Lifetime::Timeframe);
325  }
326  o2::mch::raw::DataDecoderTask task(inputSpec);
327  return DataProcessorSpec{
328  specName,
329  inputs,
330  Outputs{OutputSpec{header::gDataOriginMCH, "DIGITS", 0, Lifetime::Timeframe},
331  OutputSpec{header::gDataOriginMCH, "DIGITROFS", 0, Lifetime::Timeframe},
332  OutputSpec{header::gDataOriginMCH, "ORBITS", 0, Lifetime::Timeframe}},
333  AlgorithmSpec{adaptFromTask<DataDecoderTask>(std::move(task))},
334  Options{{"mch-debug", VariantType::Bool, false, {"enable verbose output"}},
335  {"cru-map", VariantType::String, "", {"custom CRU mapping"}},
336  {"fec-map", VariantType::String, "", {"custom FEC mapping"}},
337  {"dummy-elecmap", VariantType::Bool, false, {"use dummy electronic mapping (for debug, temporary)"}},
338  {"ds2manu", VariantType::Bool, false, {"convert channel numbering from Run3 to Run1-2 order"}},
339  {"check-rofs", VariantType::Bool, false, {"perform consistency checks on the output ROFs"}},
340  {"dummy-rofs", VariantType::Bool, false, {"disable the ROFs finding algorithm"}},
341  {"error-log-frequency", VariantType::Int, 6000, {"log the error map at this frequency (in TF unit) (first TF is always logged, unless frequency is zero)"}}}};
342 }
