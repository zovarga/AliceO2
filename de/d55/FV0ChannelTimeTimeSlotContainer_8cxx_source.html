66 int16_t FV0ChannelTimeTimeSlotContainer::getMeanGaussianFitValue(std::size_t channelID) const
67 {
68 
69  static constexpr size_t MEAN_VALUE_INDEX_IN_OUTPUT_VECTOR = 1;
70 
71  if (0 == mEntriesPerChannel[channelID]) {
72  return 0;
73  }
74  LOG(debug) << " for channel " << int(channelID) << " entries " << mEntriesPerChannel[channelID];
75 
76  std::vector<double> channelHistogramData(NUMBER_OF_HISTOGRAM_BINS, 0);
77 
78  std::vector<double> outputGaussianFitValues;
79  double binWidth = (HISTOGRAM_RANGE - (-HISTOGRAM_RANGE)) / NUMBER_OF_HISTOGRAM_BINS;
80  double minGausFitRange = 0;
81  double maxGausFitRange = 0;
82  double MaxValOfHistogram = 0.0;
83 
84  for (int iBin = 0; iBin < NUMBER_OF_HISTOGRAM_BINS; ++iBin) {
85  channelHistogramData[iBin] = mHistogram.at(iBin, channelID);
86  }
87 
88  int maxElementIndex = std::max_element(channelHistogramData.begin(), channelHistogramData.end()) - channelHistogramData.begin();
89  int maxElement = *std::max_element(channelHistogramData.begin(), channelHistogramData.end());
90 
91  // calculating the min & max range values to fit gaussian
92  minGausFitRange = (-HISTOGRAM_RANGE + (maxElementIndex - sGausFitBins) * binWidth + binWidth / 2.0);
93  maxGausFitRange = (-HISTOGRAM_RANGE + (maxElementIndex + sGausFitBins) * binWidth + binWidth / 2.0);
94 
95  double returnCode = math_utils::fitGaus<double>(NUMBER_OF_HISTOGRAM_BINS, channelHistogramData.data(),
96  minGausFitRange, maxGausFitRange, outputGaussianFitValues);
97 
98  MaxValOfHistogram = (-HISTOGRAM_RANGE + maxElementIndex * binWidth + binWidth / 2.0);
99 
100  if (returnCode < 0) {
101  LOG(error) << "Gaussian fit error!";
102  return static_cast<int16_t>(std::round(MaxValOfHistogram));
103  //return 0;
104  }
105 
106  return static_cast<int16_t>(std::round(outputGaussianFitValues[MEAN_VALUE_INDEX_IN_OUTPUT_VECTOR]));
107 }
