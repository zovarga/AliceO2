114 {

177 template <typename T, typename ChunkingPolicy = Chunked>
178 class ColumnIterator : ChunkingPolicy
179 {
180  static constexpr char SCALE_FACTOR = std::is_same_v<std::decay_t<T>, bool> ? 3 : 0;
181 
182  public:
187  ColumnIterator(arrow::ChunkedArray const* column)
188  : mColumn{column},
189  mCurrent{nullptr},
190  mCurrentPos{nullptr},
191  mLast{nullptr},
192  mFirstIndex{0},
193  mCurrentChunk{0},
194  mOffset{0}
195  {
196  auto array = getCurrentArray();
197  mCurrent = reinterpret_cast<unwrap_t<T> const*>(array->values()->data()) + (mOffset >> SCALE_FACTOR);
198  mLast = mCurrent + array->length();
199  }
200 
201  ColumnIterator() = default;
202  ColumnIterator(ColumnIterator<T, ChunkingPolicy> const&) = default;
203  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy> const&) = default;
204 
205  ColumnIterator(ColumnIterator<T, ChunkingPolicy>&&) = default;
206  ColumnIterator<T, ChunkingPolicy>& operator=(ColumnIterator<T, ChunkingPolicy>&&) = default;
207 
209  void nextChunk() const
210  {
211  auto previousArray = getCurrentArray();
212  mFirstIndex += previousArray->length();
213 
214  mCurrentChunk++;
215  auto array = getCurrentArray();
216  mCurrent = reinterpret_cast<unwrap_t<T> const*>(array->values()->data()) + (mOffset >> SCALE_FACTOR) - (mFirstIndex >> SCALE_FACTOR);
217  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
218  }
219 
220  void prevChunk() const
221  {
222  auto previousArray = getCurrentArray();
223  mFirstIndex -= previousArray->length();
224 
225  mCurrentChunk--;
226  auto array = getCurrentArray();
227  mCurrent = reinterpret_cast<unwrap_t<T> const*>(array->values()->data()) + (mOffset >> SCALE_FACTOR) - (mFirstIndex >> SCALE_FACTOR);
228  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
229  }
230 
231  void moveToChunk(int chunk)
232  {
233  if (mCurrentChunk < chunk) {
234  while (mCurrentChunk != chunk) {
235  nextChunk();
236  }
237  } else {
238  while (mCurrentChunk != chunk) {
239  prevChunk();
240  }
241  }
242  }
243 
245  void moveToEnd()
246  {
247  mCurrentChunk = mColumn->num_chunks() - 1;
248  auto array = getCurrentArray();
249  mFirstIndex = mColumn->length() - array->length();
250  mCurrent = reinterpret_cast<unwrap_t<T> const*>(array->values()->data()) + (mOffset >> SCALE_FACTOR) - (mFirstIndex >> SCALE_FACTOR);
251  mLast = mCurrent + array->length() + (mFirstIndex >> SCALE_FACTOR);
252  }
253 
254  decltype(auto) operator*() const
255  {
256  if constexpr (ChunkingPolicy::chunked) {
257  if constexpr (std::is_same_v<arrow_array_for_t<T>, arrow::ListArray>) {
258  auto list = std::static_pointer_cast<arrow::ListArray>(mColumn->chunk(mCurrentChunk));
259  if (O2_BUILTIN_UNLIKELY(*mCurrentPos - mFirstIndex >= list->length())) {
260  nextChunk();
261  }
262  } else {
263  if (O2_BUILTIN_UNLIKELY(((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast))) {
264  nextChunk();
265  }
266  }
267  }
268  if constexpr (std::is_same_v<bool, std::decay_t<T>>) {
269  // FIXME: check if shifting the masked bit to the first position is better than != 0
270  return (*(mCurrent + (*mCurrentPos >> SCALE_FACTOR)) & (1 << ((*mCurrentPos + mOffset) & 0x7))) != 0;
271  } else if constexpr (std::is_same_v<arrow_array_for_t<T>, arrow::ListArray>) {
272  auto list = std::static_pointer_cast<arrow::ListArray>(mColumn->chunk(mCurrentChunk));
273  auto offset = list->value_offset(*mCurrentPos - mFirstIndex);
274  auto length = list->value_length(*mCurrentPos - mFirstIndex);
275  return gsl::span{mCurrent + mFirstIndex + offset, mCurrent + mFirstIndex + (offset + length)};
276  } else {
277  return *(mCurrent + (*mCurrentPos >> SCALE_FACTOR));
278  }
279  }
280 
281  // Move to the chunk which containts element pos
282  ColumnIterator<T>& moveToPos()
283  {
284  // If we get outside range of the current chunk, go to the next.
285  if constexpr (ChunkingPolicy::chunked) {
286  while (O2_BUILTIN_UNLIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) >= mLast)) {
287  nextChunk();
288  }
289  }
290  return *this;
291  }
292 
293  // Move to the chunk which containts element pos
294  ColumnIterator<T>& checkNextChunk()
295  {
296  if constexpr (ChunkingPolicy::chunked) {
297  if (O2_BUILTIN_LIKELY((mCurrent + (*mCurrentPos >> SCALE_FACTOR)) <= mLast)) {
298  return *this;
299  }
300  nextChunk();
301  }
302  return *this;
303  }
304 
305  mutable unwrap_t<T> const* mCurrent;
306  int64_t const* mCurrentPos;
307  mutable unwrap_t<T> const* mLast;
308  arrow::ChunkedArray const* mColumn;
309  mutable int mFirstIndex;
310  mutable int mCurrentChunk;
311  mutable int mOffset;
312 
313  private:
315  auto getCurrentArray() const
316  {
317  std::shared_ptr<arrow::Array> chunkToUse = mColumn->chunk(mCurrentChunk);
318  mOffset = chunkToUse->offset();
319  if constexpr (std::is_same_v<arrow_array_for_t<T>, arrow::FixedSizeListArray>) {
320  chunkToUse = std::dynamic_pointer_cast<arrow::FixedSizeListArray>(chunkToUse)->values();
321  return std::static_pointer_cast<arrow_array_for_t<value_for_t<T>>>(chunkToUse);
322  } else if constexpr (std::is_same_v<arrow_array_for_t<T>, arrow::ListArray>) {
323  chunkToUse = std::dynamic_pointer_cast<arrow::ListArray>(chunkToUse)->values();
324  mOffset = chunkToUse->offset();
325  return std::static_pointer_cast<arrow_array_for_t<value_for_t<T>>>(chunkToUse);
326  } else {
327  return std::static_pointer_cast<arrow_array_for_t<T>>(chunkToUse);
328  }
329  }
330 };
