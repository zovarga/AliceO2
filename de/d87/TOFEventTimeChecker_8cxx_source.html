165 void TOFEventTimeChecker::processEvent(std::vector<MyTrack>& tracks)
166 {
167  int nBC = int(tracks[0].tofSignalDouble() * o2::tof::Geo::BC_TIME_INPS_INV);
168  for (auto& track : tracks) {
169  if (!o2::tof::Utils::hasFillScheme()) {
170  track.mSignal = float(track.mSignalDouble - double(o2::tof::Geo::BC_TIME_INPS) * nBC);
171  track.mTrktime = (track.mTrktime - double(o2::tof::Geo::BC_TIME_INPS) * nBC);
172  } else {
173  double localTime = track.mSignalDouble;
174 
175  // get into orbit
176  int bcStarOrbit = int(localTime * o2::tof::Geo::BC_TIME_INPS_INV);
177  bcStarOrbit = (bcStarOrbit / o2::constants::lhc::LHCMaxBunches) * o2::constants::lhc::LHCMaxBunches; // truncation
178  localTime -= bcStarOrbit * o2::tof::Geo::BC_TIME_INPS;
179  track.mSignal = o2::tof::Utils::subtractInteractionBC(localTime);
180  localTime = track.mTrktime;
181  bcStarOrbit = int(localTime * o2::tof::Geo::BC_TIME_INPS_INV);
182  bcStarOrbit = (bcStarOrbit / o2::constants::lhc::LHCMaxBunches) * o2::constants::lhc::LHCMaxBunches; // truncation
183  localTime -= bcStarOrbit * o2::tof::Geo::BC_TIME_INPS;
184  track.mTrktime = o2::tof::Utils::subtractInteractionBC(localTime);
185  }
186  }
187 
188  auto evtime = o2::tof::evTimeMaker<std::vector<MyTrack>, MyTrack, MyFilter>(tracks);
189 
190  if (evtime.eventTime - o2::tof::Utils::mLHCPhase < -2000 || evtime.eventTime - o2::tof::Utils::mLHCPhase > 2000) {
191  return;
192  }
193  //
194  const float cinv = 33.35641;
195 
196  int nt = 0;
197  for (auto& track : tracks) {
198  mT0 = evtime.eventTime;
199  mT0Res = evtime.eventTimeError;
200 
201  float sumw = 1. / (mT0Res * mT0Res);
202  mT0 *= sumw;
203  mT0 -= evtime.weights[nt] * evtime.tracktime[nt];
204  sumw -= evtime.weights[nt];
205  mT0 /= sumw;
206  mT0Res = sqrt(1. / sumw);
207 
208  nt++;
209 
210  mCh = track.mCh;
211  mP = track.mP;
212  mPt = track.mPt;
213  mEta = track.mEta;
214  mSource = track.mSource;
215  mPhi = track.mPhi;
216  mChi2 = track.mChi2;
217  mL = track.mLength;
218  mTof = track.tofSignal();
219  mExpDe = track.tofExpSignalDe();
220  mExpPi = track.tofExpSignalPi();
221  mExpKa = track.tofExpSignalKa();
222  mExpPr = track.tofExpSignalPr();
223  mIsProb = track.mIsProb;
224  mTrktime = track.mTrktime;
225  mTrktimeRes = track.mTrktimeRes;
226  mDx = track.mDx;
227  mDz = track.mDz;
228 
229 #ifdef TDEBUG
230  mTree->Fill();
231 #endif
232 
233  // remove unphysical tracks
234  if (mTof - mT0 - mExpPi < -5000) {
235  continue;
236  }
237 
238  //Beta
239  float beta = mL / (mTof - mT0) * cinv;
240 
241  float betaexpPi = mL / mExpPi * cinv;
242  float betaexpKa = mL / mExpKa * cinv;
243  float betaexpPr = mL / mExpPr * cinv;
244 
245  //Mass
246  float mass = mP / beta * TMath::Sqrt(TMath::Abs(1 - beta * beta));
247  float massexpPi = mP / betaexpPi * TMath::Sqrt(TMath::Abs(1 - betaexpPi * betaexpPi));
248  float massexpKa = mP / betaexpKa * TMath::Sqrt(TMath::Abs(1 - betaexpKa * betaexpKa));
249  float massexpPr = mP / betaexpPr * TMath::Sqrt(TMath::Abs(1 - betaexpPr * betaexpPr));
250 
251  if (massexpPi < 0.13) { // remove wrong track lengths
252  continue;
253  }
254 
255  //Fill histos
256  mHTimePi->Fill(mTof - mT0 - mExpPi);
257  mHTimeKa->Fill(mTof - mT0 - mExpKa);
258  mHTimePr->Fill(mTof - mT0 - mExpPr);
259  mHMass->Fill(mass);
260  mHMassExpPi->Fill(massexpPi);
261  mHMassExpKa->Fill(massexpKa);
262  mHMassExpPr->Fill(massexpPr);
263  mHBetavsP->Fill(mP, beta);
264  mPBetavsPExpPi->Fill(mP, betaexpPi);
265  mPBetavsPExpKa->Fill(mP, betaexpKa);
266  mPBetavsPExpPr->Fill(mP, betaexpPr);
267  mHTimePivsP->Fill(mP, mTof - mT0 - mExpPi);
268  mHTimeKvsP->Fill(mP, mTof - mT0 - mExpKa);
269  mHTimePrvsP->Fill(mP, mTof - mT0 - mExpPr);
270  mHMassvsP->Fill(mP, mass);
271  mPMassvsPExpPi->Fill(mP, massexpPi);
272  mPMassvsPExpKa->Fill(mP, massexpKa);
273  mPMassvsPExpPr->Fill(mP, massexpPr);
274  if (mP > 0.7 && mP < 1.1) {
275  mHTimevsResEvtimePi->Fill(mT0Res, mTof - mT0 - mExpPi);
276  }
277  mHEventTimevsResEvtime->Fill(mT0Res, mT0);
278  }
279 }

281 void TOFEventTimeChecker::fillMatching(GID gid, float time0, float time0res)
282 {
283  MyTrack trk;
284  trk.mHasTOF = true;
285 
286  if (!gid.includesDet(DetID::TOF)) {
287  trk.mHasTOF = false;
288  }
289 
290  trk.mTrktime = time0 * 1E6;
291  trk.mTrktimeRes = time0res * 1E6;
292 
293  trk.mDx = 0;
294  trk.mDz = 0;
295 
296  int trksource = 5;
297  if (gid.getSource() == GID::TPCTOF) {
298  const o2::dataformats::MatchInfoTOF& match = mRecoData.getTOFMatch(gid);
299  const auto& array = mRecoData.getTPCTracks();
300  GID gTrackId = match.getTrackRef();
301  const auto& srctrk = array[gTrackId.getIndex()];
302  trk.mPt = srctrk.getPt();
303  trk.mP = srctrk.getP();
304  trk.mEta = srctrk.getEta();
305  trk.mPhi = srctrk.getPhi();
306  trk.mDx = match.getDXatTOF();
307  trk.mDz = match.getDZatTOF();
308  trksource = 0;
309  } else if (gid.getSource() == GID::TPC) {
310  const o2::dataformats::MatchInfoTOF& match = mRecoData.getTOFMatch(gid);
311  const auto& array = mRecoData.getTPCTracks();
312  GID gTrackId = gid;
313  const auto& srctrk = array[gTrackId.getIndex()];
314  trk.mPt = srctrk.getPt();
315  trk.mP = srctrk.getP();
316  trk.mEta = srctrk.getEta();
317  trk.mPhi = srctrk.getPhi();
318  trksource = 0;
319  } else if (gid.getSource() == GID::ITSTPCTOF) {
320  const o2::dataformats::MatchInfoTOF& match = mRecoData.getTOFMatch(gid);
321  const auto& array = mRecoData.getTPCITSTracks();
322  GID gTrackId = match.getTrackRef();
323  const auto& srctrk = array[gTrackId.getIndex()];
324  trk.mPt = srctrk.getPt();
325  trk.mP = srctrk.getP();
326  trk.mEta = srctrk.getEta();
327  trk.mPhi = srctrk.getPhi();
328  trksource = 1;
329  trk.mTrktime = srctrk.getTimeMUS().getTimeStamp() * 1E6;
330  trk.mTrktimeRes = srctrk.getTimeMUS().getTimeStampError() * 1E6;
331  trk.mDx = match.getDXatTOF();
332  trk.mDz = match.getDZatTOF();
333  } else if (gid.getSource() == GID::ITSTPC) {
334  const auto& array = mRecoData.getTPCITSTracks();
335  GID gTrackId = gid; //match.getTrackRef();
336  const auto& srctrk = array[gTrackId.getIndex()];
337  trk.mPt = srctrk.getPt();
338  trk.mP = srctrk.getP();
339  trk.mEta = srctrk.getEta();
340  trk.mPhi = srctrk.getPhi();
341  trksource = 1;
342  trk.mTrktime = srctrk.getTimeMUS().getTimeStamp() * 1E6;
343  trk.mTrktimeRes = srctrk.getTimeMUS().getTimeStampError() * 1E6;
344  } else if (gid.getSource() == GID::TPCTRDTOF) {
345  const o2::dataformats::MatchInfoTOF& match = mRecoData.getTOFMatch(gid);
346  const auto& array = mRecoData.getTPCTRDTracks<o2::trd::TrackTRD>();
347  GID gTrackId = match.getTrackRef();
348  const auto& srctrk = array[gTrackId.getIndex()];
349  trk.mPt = srctrk.getPt();
350  trk.mP = srctrk.getP();
351  trk.mEta = srctrk.getEta();
352  trk.mPhi = srctrk.getPhi();
353  trksource = 2;
354  trk.mDx = match.getDXatTOF();
355  trk.mDz = match.getDZatTOF();
356  } else if (gid.getSource() == GID::ITSTPCTRDTOF) {
357  const o2::dataformats::MatchInfoTOF& match = mRecoData.getTOFMatch(gid);
358  const auto& array = mRecoData.getITSTPCTRDTracks<o2::trd::TrackTRD>();
359  GID gTrackId = match.getTrackRef();
360  const auto& srctrk = array[gTrackId.getIndex()];
361  trk.mPt = srctrk.getPt();
362  trk.mP = srctrk.getP();
363  trk.mEta = srctrk.getEta();
364  trk.mPhi = srctrk.getPhi();
365  trksource = 3;
366  trk.mDx = match.getDXatTOF();
367  trk.mDz = match.getDZatTOF();
368  }
369 
370  trk.mSource = trksource;
371 
372  const char* sources[5] = {"TPC", "ITS-TPC", "TPC-TRD", "ITS-TPC-TRD", "NONE"};
373 
374  if (trk.mHasTOF) {
375  const o2::dataformats::MatchInfoTOF& match = mRecoData.getTOFMatch(gid);
376  const o2::track::TrackLTIntegral& info = match.getLTIntegralOut();
377 
378  if (info.getL() < 370) {
379  // trk.mHasTOF = false;
380  }
381 
382  trk.mExpDe = info.getTOF(5); // el
383  trk.expTimes[0] = info.getTOF(2); // pi
384  trk.expTimes[1] = info.getTOF(3); // ka
385  trk.expTimes[2] = info.getTOF(4); // pr
386  trk.expSigma[0] = 120; // dummy resolution (to be updated)
387  trk.expSigma[1] = 120; // dummy resolution (to be updated)
388  trk.expSigma[2] = 120; // dummy resolution (to be updated)
389 
390  trk.mChi2 = match.getChi2();
391 
392  int tofcl = match.getIdxTOFCl();
393  // trk.mSignal = mTOFClustersArrayInp[tofcl].getTime();
394  double tofsignal = match.getSignal();
395 
396  trk.mSignalDouble = tofsignal;
397 
398  //trk.mSignal = match.getSignal();
399  trk.mTOFChi2 = match.getChi2();
400  trk.mLength = info.getL();
401  // trk.mHypo = 0;
402  trk.mCh = mTOFClustersArrayInp[tofcl].getMainContributingChannel();
403 
404  if (mSlewing) { // let's calibrate
405  trk.mIsProb = mSlewing->isProblematic(trk.mCh);
406  if (mSlewing->isProblematic(trk.mCh)) {
407  // LOG(debug) << "skip channel " << trk.mCh << " since problematic";
408  // return;
409  }
410  float tot = mTOFClustersArrayInp[tofcl].getTot();
411  trk.mSignalDouble -= mSlewing->evalTimeSlewing(trk.mCh, tot);
412  LOG(debug) << "calibration -> " << mSlewing->evalTimeSlewing(trk.mCh, tot);
413  }
414  }
415  if (!trk.mHasTOF) {
416  trk.mExpDe = 0;
417  trk.expTimes[0] = 0;
418  trk.expTimes[1] = 0;
419  trk.expTimes[2] = 0;
420  trk.expSigma[0] = 0; // dummy resolution (to be updated)
421  trk.expSigma[1] = 0; // dummy resolution (to be updated)
422  trk.expSigma[2] = 0; // dummy resolution (to be updated)
423  trk.mSignalDouble = time0 * 1E6;
424  trk.mTOFChi2 = 9999;
425  trk.mLength = 0;
426  trk.mCh = -1;
427  trk.mChi2 = 0;
428  trk.mIsProb = 0;
429  }
430 
431  mMyTracks.push_back(trk);
432 }

509 void TOFEventTimeChecker::run(ProcessingContext& pc)
510 {
511  mTimer.Start(false);
512 
513  mMyTracks.clear();
514  mRecoData.collectData(pc, *mDataRequest.get());
515 
516  mIsTPC = (mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTOF) && mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTOF));
517  mIsITSTPC = (mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTOF) && mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTOF));
518  mIsITSTPCTRD = (mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRD) && mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRDTOF));
519  mIsTPCTRD = (mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRD) && mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRDTOF));
520 
521  mTOFClustersArrayInp = mRecoData.getTOFClusters();
522 
523  LOG(debug) << "isTrackSourceLoaded: TPC -> " << mIsTPC << " (t=" << mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTOF) << ",m=" << mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTOF) << ")";
524  LOG(debug) << "isTrackSourceLoaded: ITSTPC -> " << mIsITSTPC << " (t=" << mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTOF) << ",m=" << mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTOF) << ")";
525  LOG(debug) << "isTrackSourceLoaded: TPCTRD -> " << mIsTPCTRD << " (t=" << mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRDTOF) << ",m=" << mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::TPCTRDTOF) << ")";
526  LOG(debug) << "isTrackSourceLoaded: ITSTPCTRD -> " << mIsITSTPCTRD << " (t=" << mRecoData.isTrackSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRDTOF) << ",m=" << mRecoData.isMatchSourceLoaded(o2::dataformats::GlobalTrackID::Source::ITSTPCTRDTOF) << ")";
527  LOG(debug) << "TOF cluster size = " << mTOFClustersArrayInp.size();
528 
529  if (!mTOFClustersArrayInp.size()) {
530  return;
531  }
532 
533  auto creator = [this](auto& trk, GID gid, float time0, float terr) {
534  this->fillMatching(gid, time0, terr);
535  return true;
536  };
537  mRecoData.createTracksVariadic(creator);
538 
539  // sorting matching in time
540  std::sort(mMyTracks.begin(), mMyTracks.end(),
541  [](MyTrack a, MyTrack b) { return a.tofSignalDouble() < b.tofSignalDouble(); });
542 
543  for (auto& element : mMyTracks) { // loop print
544  LOG(debug) << "Time cluster = " << element.tofSignal() << " ps - pt = " << element.pt();
545  }
546 
547  std::vector<MyTrack> tracks;
548  for (int i = 0; i < mMyTracks.size(); i++) { // loop looking for interaction candidates
549  tracks.clear();
550  int ntrk = 1;
551  double time = mMyTracks[i].tofSignalDouble();
552  tracks.emplace_back(mMyTracks[i]);
553  for (; i < mMyTracks.size(); i++) {
554  double timeCurrent = mMyTracks[i].tofSignalDouble();
555  if (timeCurrent - time > 100E3) {
556  i--;
557  break;
558  }
559  tracks.emplace_back(mMyTracks[i]);
560  ntrk++;
561  }
562  if (ntrk > 0) { // good candidate with time
563  processEvent(tracks);
564  }
565  }
566 
567  mTimer.Stop();
568 }
