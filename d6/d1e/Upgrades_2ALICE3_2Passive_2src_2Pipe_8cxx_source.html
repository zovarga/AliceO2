65 void Alice3Pipe::ConstructGeometry()
66 {
67  createMaterials();
68  //
69  // Class describing the beam Alice3Pipe geometry
70  //
71  float z, zsh, z0;
72  //
73  // Rotation Matrices
74  //
75  const float kDegRad = TMath::Pi() / 180.;
76  // Rotation by 180 deg
77  TGeoRotation* rot180 = new TGeoRotation("rot180", 90., 180., 90., 90., 180., 0.);
78  TGeoRotation* rotyz = new TGeoRotation("rotyz", 90., 180., 0., 180., 90., 90.);
79  TGeoRotation* rotxz = new TGeoRotation("rotxz", 0., 0., 90., 90., 90., 180.);
80  //
81 
82  //
83  // Media
84  auto& matmgr = o2::base::MaterialManager::Instance();
85 
86  const TGeoMedium* kMedBe = matmgr.getTGeoMedium("ALICE3PIPE_BE");
87  const TGeoMedium* kMedVac = matmgr.getTGeoMedium("ALICE3PIPE_VACUUM");
88  const TGeoMedium* kMedVacNF = matmgr.getTGeoMedium("ALICE3PIPE_VACUUM_NF");
89  const TGeoMedium* kMedVacHC = matmgr.getTGeoMedium("ALICE3PIPE_VACUUM_HC");
90  const TGeoMedium* kMedVacNFHC = matmgr.getTGeoMedium("ALICE3PIPE_VACUUM_NFHC");
91 
92  // Top volume
93  TGeoVolume* top = gGeoManager->GetVolume("cave");
94  TGeoVolume* barrel = gGeoManager->GetVolume("barrel");
95  if (!barrel) {
96  LOG(fatal) << "Could not find the top volume";
97  }
98 
99  //---------------- Outermost Be pipe around the IP ----------
100  // Outer pipe has to be filled with vacuum. There we have also TRK layers, which we don't want to depend on the pipe volume.
101  // Eventually, we will depend on some information passed from the outside.
102  // For A3PIP-only simulations, we don't want TRK's shade.
103  // Strategy used here is to use a composite shape where shapes of TRK layers are subtracted to the vacuum volume
104  TGeoTube* outerBeTube = new TGeoTube("OUT_PIPEsh", mBeOuterPipeRmin, mBeOuterPipeRmin + mBeOuterPipeThick, mOuterIpHLength);
105  TGeoVolume* outerBeTubeVolume = new TGeoVolume("OUT_PIPE", outerBeTube, kMedBe);
106  outerBeTubeVolume->SetLineColor(kBlue);
107 
108  TGeoTube* outerBerylliumTubeVacuumBase = new TGeoTube("OUT_PIPEVACUUM_BASEsh", mBeInnerPipeRmin + mBeInnerPipeThick, mBeOuterPipeRmin, mOuterIpHLength); // Vacuum filling for outer pipe
109  TGeoCompositeShape* outerBerylliumTubeVacuumComposite; // Composite volume to subctract to vacuum
110  TGeoVolume* outerBerylliumTubeVacuumVolume; // Final volume to be used
111 
112  TString compositeFormula{"OUT_PIPEVACUUM_BASEsh"}; // If pipe is alone we won't subctract anything
113  TString subtractorsFormula;
114 
115  if (!mIsTRKActivated) {
116  std::vector<TGeoTube*> trkLayerShapes;
117  std::vector<std::array<float, 3>> layersQuotas = {std::array<float, 3>{0.5f, 50.f, 50.e-4}, // TODO: Set layers dynamically. {radius, zLen, thickness}
118  std::array<float, 3>{1.2f, 50.f, 50.e-4},
119  std::array<float, 3>{2.5f, 50.f, 50.e-4}};
120 
121  subtractorsFormula = "TRKLAYER_0sh"; // First volume to be subctracted (at least one has to be provided)
122  for (auto iLayer{0}; iLayer < layersQuotas.size(); ++iLayer) { // Create TRK layers shapes
123  auto& layerData = layersQuotas[iLayer];
124  trkLayerShapes.emplace_back(new TGeoTube(Form("TRKLAYER_%dsh", iLayer), layerData[0], layerData[0] + layerData[2], layerData[1] / 2));
125  if (iLayer > 0) {
126  subtractorsFormula += Form("+TRKLAYER_%dsh", iLayer);
127  }
128  }
129  LOG(debug) << "Subtractors formula before: " << subtractorsFormula;
130  subtractorsFormula = Form("-(%s)", subtractorsFormula.Data());
131  LOG(debug) << "Subtractors formula after: " << subtractorsFormula;
132 
133  outerBerylliumTubeVacuumComposite = new TGeoCompositeShape("OUT_PIPEVACUUMsh", (compositeFormula + subtractorsFormula).Data());
134  outerBerylliumTubeVacuumVolume = new TGeoVolume("OUT_PIPEVACUUM", outerBerylliumTubeVacuumComposite, kMedVac);
135  } else {
136  outerBerylliumTubeVacuumVolume = new TGeoVolume("OUT_PIPEVACUUM", outerBerylliumTubeVacuumBase, kMedVac);
137  }
138 
139  outerBerylliumTubeVacuumVolume->SetVisibility(1);
140  outerBerylliumTubeVacuumVolume->SetTransparency(50);
141  outerBerylliumTubeVacuumVolume->SetLineColor(kGreen);
142 
143  // outerBeTubeVolume->AddNode(outerBerylliumTubeVacuumVolume, 1, gGeoIdentity);
144  barrel->AddNode(outerBerylliumTubeVacuumVolume, 1, new TGeoTranslation(0, 30.f, 0));
145 
146  barrel->AddNode(outerBeTubeVolume, 1, new TGeoTranslation(0, 30.f, 0)); // Add to surrounding geometry
147 
148  //---------------- Innermost Be pipe around the IP ----------
149  TGeoTube* innerBeTube =
150  new TGeoTube("INN_PIPEsh", mBeInnerPipeRmin, mBeInnerPipeRmin + mBeInnerPipeThick, mInnerIpHLength);
151  TGeoVolume* innerBeTubeVolume = new TGeoVolume("INN_PIPE", innerBeTube, kMedBe);
152  innerBeTubeVolume->SetLineColor(kRed);
153 
154  TGeoTube* berylliumTubeVacuum =
155  new TGeoTube("INN_PIPEVACUUMsh", 0., mBeInnerPipeRmin, mInnerIpHLength);
156  TGeoVolume* innerBerylliumTubeVacuumVolume = new TGeoVolume("INN_PIPEVACUUM", berylliumTubeVacuum, kMedVac);
157  innerBerylliumTubeVacuumVolume->SetVisibility(1);
158  innerBerylliumTubeVacuumVolume->SetLineColor(kGreen);
159 
160  barrel->AddNode(innerBeTubeVolume, 1, new TGeoTranslation(0, 30.f, 0));
161  barrel->AddNode(innerBerylliumTubeVacuumVolume, 1, new TGeoTranslation(0, 30.f, 0));
162 }

164 void Alice3Pipe::createMaterials()
165 {
166  //
167  // Define materials for beam Alice3Pipe
168  //
169  Int_t isxfld = 2.;
170  float sxmgmx = 10.;
171  o2::base::Detector::initFieldTrackingParams(isxfld, sxmgmx);
172 
173  //
174  // Air
175  //
176  float aAir[4] = {12.0107, 14.0067, 15.9994, 39.948};
177  float zAir[4] = {6., 7., 8., 18.};
178  float wAir[4] = {0.000124, 0.755267, 0.231781, 0.012827};
179  float dAir = 1.20479E-3;
180  float dAir1 = 1.20479E-11;
181 
182  // ****************
183  // Defines tracking media parameters.
184  //
185  float epsil = .1; // Tracking precision,
186  float stemax = -0.01; // Maximum displacement for multiple scat
187  float tmaxfd = -20.; // Maximum angle due to field deflection
188  float deemax = -.3; // Maximum fractional energy loss, DLS
189  float stmin = -.8;
190  // ***************
191 
192  auto& matmgr = o2::base::MaterialManager::Instance();
193 
194  // Beryllium
195  matmgr.Material("ALICE3PIPE", 5, "BERILLIUM$", 9.01, 4., 1.848, 35.3, 36.7);
196  matmgr.Medium("ALICE3PIPE", 5, "BE", 5, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
197 
198  // Vacuum
199  matmgr.Mixture("ALICE3PIPE", 16, "VACUUM$ ", aAir, zAir, dAir1, 4, wAir);
200  matmgr.Mixture("ALICE3PIPE", 36, "VACUUM$_NF", aAir, zAir, dAir1, 4, wAir);
201  matmgr.Mixture("ALICE3PIPE", 56, "VACUUM$_HC ", aAir, zAir, dAir1, 4, wAir);
202  matmgr.Mixture("ALICE3PIPE", 76, "VACUUM$_NFHC", aAir, zAir, dAir1, 4, wAir);
203 
204  matmgr.Medium("ALICE3PIPE", 16, "VACUUM", 16, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
205  matmgr.Medium("ALICE3PIPE", 36, "VACUUM_NF", 36, 0, 0, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
206  matmgr.Medium("ALICE3PIPE", 56, "VACUUM_HC", 56, 0, isxfld, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
207  matmgr.Medium("ALICE3PIPE", 76, "VACUUM_NFHC", 76, 0, 0, sxmgmx, tmaxfd, stemax, deemax, epsil, stmin);
208 }
