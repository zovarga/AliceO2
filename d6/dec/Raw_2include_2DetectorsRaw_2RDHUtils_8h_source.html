56  template <typename H>
57  static constexpr int getVersion()
58  {
59 #ifndef GPUCA_GPUCODE_DEVICE
60  RDHAny::sanityCheckStrict<H>();
61  if (std::is_same<H, RDHv6>::value) {
62  return 6;
63  }
64  if (std::is_same<H, RDHv5>::value) {
65  return 5;
66  }
67  if (std::is_same<H, RDHv4>::value) {
68  return 4;
69  }
70 #else
71  return -1; // dummy value as this method will be used on the CPU only
72 #endif // GPUCA_GPUCODE_DEVICE
73  }
