29 std::vector<ColumnData> makeMasks(const ChannelScalers& scalers, unsigned long nEvents, double threshold, const std::vector<ColumnData>& refMasks)
30 {
32  uint32_t nThresholdEvents = static_cast<uint32_t>(threshold * nEvents);
33  ChannelMasksHandler maskHandler;
34  for (const auto scaler : scalers.getScalers()) {
35  if (scaler.second >= nThresholdEvents) {
36  maskHandler.switchOffChannel(scalers.getDeId(scaler.first), scalers.getColumnId(scaler.first), scalers.getLineId(scaler.first), scalers.getStrip(scaler.first), scalers.getCathode(scaler.first));
37  }
38  }
39  std::vector<ColumnData> masks(maskHandler.getMasks());
40 
41  if (!refMasks.empty()) {
42  ChannelMasksHandler defaultMaskHandler;
43  defaultMaskHandler.setFromChannelMasks(refMasks);
44 
45  for (auto& mask : masks) {
46  defaultMaskHandler.applyMask(mask);
47  }
48  }
49 
50  return masks;
51 }

53 std::vector<ColumnData> makeDefaultMasks()
54 {
56  Mapping mapping;
57  ChannelMasksHandler masksHandler;
58  uint16_t fullPattern = 0xFFFF;
59  for (int ide = 0; ide < detparams::NDetectionElements; ++ide) {
60  for (int icol = mapping.getFirstColumn(ide); icol < 7; ++icol) {
61  ColumnData mask;
62  mask.deId = static_cast<uint8_t>(ide);
63  mask.columnId = static_cast<uint8_t>(icol);
64  // int nFullPatterns = 0;
65  for (int iline = mapping.getFirstBoardBP(icol, ide), lastLine = mapping.getLastBoardBP(icol, ide); iline <= lastLine; ++iline) {
66  mask.setBendPattern(fullPattern, iline);
67  // ++nFullPatterns;
68  }
69  for (int istrip = 0; istrip < mapping.getNStripsNBP(icol, ide); ++istrip) {
70  mask.addStrip(istrip, 1, 0);
71  }
72  masksHandler.setFromChannelMask(mask);
73  }
74  }
75 
76  return masksHandler.getMasks();
77 }

79 std::vector<ROBoard> getActiveBoards(const FEEIdConfig& feeIdConfig, const CrateMasks& masks)
80 {
82  std::vector<ROBoard> activeBoards;
83  auto gbtUniqueIds = feeIdConfig.getConfiguredGBTUniqueIDs();
84  ROBoard board;
85  board.statusWord = raw::sCARDTYPE;
86  board.patternsBP.fill(0xFFFF);
87  board.patternsNBP.fill(0xFFFF);
88  board.firedChambers = 0xF;
89  for (auto& gbtUniqueId : gbtUniqueIds) {
90  auto mask = masks.getMask(gbtUniqueId);
91  auto crateId = crateparams::getCrateIdFromGBTUniqueId(gbtUniqueId);
92  for (int iloc = 0; iloc < 8; ++iloc) {
93  if ((mask >> iloc) & 0x1) {
94  board.boardId = raw::makeUniqueLocID(crateId, crateparams::getLocIdInCrate(gbtUniqueId, iloc));
95  activeBoards.emplace_back(board);
96  }
97  }
98  }
99  return activeBoards;
100 }

102 std::vector<ColumnData> makeDefaultMasksFromCrateConfig(const FEEIdConfig& feeIdConfig, const CrateMasks& crateMasks)
103 {
105  auto activeBoards = getActiveBoards(feeIdConfig, crateMasks);
106  std::vector<ROFRecord> rofs;
107  rofs.push_back({InteractionRecord(), EventType::Standard, 0, activeBoards.size()});
108  DecodedDataAggregator aggregator;
109  aggregator.process(activeBoards, rofs);
110  std::vector<ColumnData> maskedBoards(aggregator.getData());
111  ChannelMasksHandler masksHandler;
112  masksHandler.setFromChannelMasks(makeDefaultMasks());
113  for (auto col : maskedBoards) {
114  masksHandler.applyMask(col);
115  }
116 
117  return maskedBoards;
118 }
