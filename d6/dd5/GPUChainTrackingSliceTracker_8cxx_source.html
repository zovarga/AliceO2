25 int GPUChainTracking::GlobalTracking(unsigned int iSlice, int threadId, bool synchronizeOutput)
26 {
27  if (GetProcessingSettings().debugLevel >= 5) {
28  GPUInfo("GPU Tracker running Global Tracking for slice %u on thread %d\n", iSlice, threadId);
29  }
30 
31  GPUReconstruction::krnlDeviceType deviceType = GetProcessingSettings().fullMergerOnGPU ? GPUReconstruction::krnlDeviceType::Auto : GPUReconstruction::krnlDeviceType::CPU;
32  runKernel<GPUTPCGlobalTracking>(GetGridBlk(256, iSlice % mRec->NStreams(), deviceType), {iSlice});
33  if (GetProcessingSettings().fullMergerOnGPU) {
34  TransferMemoryResourceLinkToHost(RecoStep::TPCSliceTracking, processors()->tpcTrackers[iSlice].MemoryResCommon(), iSlice % mRec->NStreams());
35  }
36  if (synchronizeOutput) {
37  SynchronizeStream(iSlice % mRec->NStreams());
38  }
39 
40  if (GetProcessingSettings().debugLevel >= 5) {
41  GPUInfo("GPU Tracker finished Global Tracking for slice %u on thread %d\n", iSlice, threadId);
42  }
43  return (0);
44 }

46 int GPUChainTracking::RunTPCTrackingSlices()
47 {
48  if (mRec->GPUStuck()) {
49  GPUWarning("This GPU is stuck, processing of tracking for this event is skipped!");
50  return (1);
51  }
52 
53  const auto& threadContext = GetThreadContext();
54 
55  int retVal = RunTPCTrackingSlices_internal();
56  if (retVal) {
57  SynchronizeGPU();
58  }
59  if (retVal >= 2) {
60  ResetHelperThreads(retVal >= 3);
61  }
62  return (retVal != 0);
63 }

65 int GPUChainTracking::RunTPCTrackingSlices_internal()
66 {
67  if (GetProcessingSettings().debugLevel >= 2) {
68  GPUInfo("Running TPC Slice Tracker");
69  }
70  bool doGPU = GetRecoStepsGPU() & RecoStep::TPCSliceTracking;
71  bool doSliceDataOnGPU = processors()->tpcTrackers[0].SliceDataOnGPU();
72  if (!param().par.earlyTpcTransform) {
73  for (unsigned int i = 0; i < NSLICES; i++) {
74  processors()->tpcTrackers[i].Data().SetClusterData(nullptr, mIOPtrs.clustersNative->nClustersSector[i], mIOPtrs.clustersNative->clusterOffset[i][0]);
75  if (doGPU) {
76  processorsShadow()->tpcTrackers[i].Data().SetClusterData(nullptr, mIOPtrs.clustersNative->nClustersSector[i], mIOPtrs.clustersNative->clusterOffset[i][0]); // TODO: not needed I think, anyway copied in SetupGPUProcessor
77  }
78  }
79  mRec->MemoryScalers()->nTPCHits = mIOPtrs.clustersNative->nClustersTotal;
80  } else {
81  int offset = 0;
82  for (unsigned int i = 0; i < NSLICES; i++) {
83  processors()->tpcTrackers[i].Data().SetClusterData(mIOPtrs.clusterData[i], mIOPtrs.nClusterData[i], offset);
84 #ifdef GPUCA_HAVE_O2HEADERS
85  if (doGPU && GetRecoSteps().isSet(RecoStep::TPCConversion)) {
86  processorsShadow()->tpcTrackers[i].Data().SetClusterData(processorsShadow()->tpcConverter.mClusters + processors()->tpcTrackers[i].Data().ClusterIdOffset(), processors()->tpcTrackers[i].NHitsTotal(), processors()->tpcTrackers[i].Data().ClusterIdOffset());
87  }
88 #endif
89  offset += mIOPtrs.nClusterData[i];
90  }
91  mRec->MemoryScalers()->nTPCHits = offset;
92  }
93  GPUInfo("Event has %u TPC Clusters, %d TRD Tracklets", (unsigned int)mRec->MemoryScalers()->nTPCHits, mIOPtrs.nTRDTracklets);
94 
95  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
96  processors()->tpcTrackers[iSlice].SetMaxData(mIOPtrs); // First iteration to set data sizes
97  }
98  mRec->ComputeReuseMax(nullptr); // Resolve maximums for shared buffers
99  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
100  SetupGPUProcessor(&processors()->tpcTrackers[iSlice], false); // Prepare custom allocation for 1st stack level
101  mRec->AllocateRegisteredMemory(processors()->tpcTrackers[iSlice].MemoryResSliceScratch());
102  mRec->AllocateRegisteredMemory(processors()->tpcTrackers[iSlice].MemoryResSliceInput());
103  }
104  mRec->PushNonPersistentMemory(qStr2Tag("TPCSLTRK"));
105  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
106  SetupGPUProcessor(&processors()->tpcTrackers[iSlice], true); // Now we allocate
107  mRec->ResetRegisteredMemoryPointers(&processors()->tpcTrackers[iSlice]); // TODO: The above call breaks the GPU ptrs to already allocated memory. This fixes them. Should actually be cleaned up at the source.
108  processors()->tpcTrackers[iSlice].SetupCommonMemory();
109  }
110 
111  bool streamInit[GPUCA_MAX_STREAMS] = {false};
112  if (doGPU) {
113  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
114  processorsShadow()->tpcTrackers[iSlice].GPUParametersConst()->gpumem = (char*)mRec->DeviceMemoryBase();
115  // Initialize Startup Constants
116  processors()->tpcTrackers[iSlice].GPUParameters()->nextStartHit = (((getKernelProperties<GPUTPCTrackletConstructor, GPUTPCTrackletConstructor::allSlices>().minBlocks * BlockCount()) + NSLICES - 1 - iSlice) / NSLICES) * getKernelProperties<GPUTPCTrackletConstructor, GPUTPCTrackletConstructor::allSlices>().nThreads;
117  processorsShadow()->tpcTrackers[iSlice].SetGPUTextureBase(mRec->DeviceMemoryBase());
118  }
119 
120  if (!doSliceDataOnGPU) {
121  RunHelperThreads(&GPUChainTracking::HelperReadEvent, this, NSLICES);
122  }
123  if (PrepareTextures()) {
124  return (2);
125  }
126 
127  // Copy Tracker Object to GPU Memory
128  if (GetProcessingSettings().debugLevel >= 3) {
129  GPUInfo("Copying Tracker objects to GPU");
130  }
131  if (PrepareProfile()) {
132  return 2;
133  }
134 
135  WriteToConstantMemory(RecoStep::TPCSliceTracking, (char*)processors()->tpcTrackers - (char*)processors(), processorsShadow()->tpcTrackers, sizeof(GPUTPCTracker) * NSLICES, mRec->NStreams() - 1, &mEvents->init);
136 
137  for (int i = 0; i < mRec->NStreams() - 1; i++) {
138  streamInit[i] = false;
139  }
140  streamInit[mRec->NStreams() - 1] = true;
141  }
142  if (GPUDebug("Initialization (1)", 0)) {
143  return (2);
144  }
145 
146  int streamMap[NSLICES];
147 
148  bool error = false;
149  GPUCA_OPENMP(parallel for if(!doGPU && GetProcessingSettings().ompKernels != 1) num_threads(mRec->SetAndGetNestedLoopOmpFactor(!doGPU, NSLICES)))
150  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
151  GPUTPCTracker& trk = processors()->tpcTrackers[iSlice];
152  GPUTPCTracker& trkShadow = doGPU ? processorsShadow()->tpcTrackers[iSlice] : trk;
153  int useStream = (iSlice % mRec->NStreams());
154 
155  if (GetProcessingSettings().debugLevel >= 3) {
156  GPUInfo("Creating Slice Data (Slice %d)", iSlice);
157  }
158  if (doSliceDataOnGPU) {
159  TransferMemoryResourcesToGPU(RecoStep::TPCSliceTracking, &trk, useStream);
160  runKernel<GPUTPCCreateSliceData>(GetGridBlk(GPUCA_ROW_COUNT, useStream), {iSlice}, {nullptr, streamInit[useStream] ? nullptr : &mEvents->init});
161  streamInit[useStream] = true;
162  } else if (!doGPU || iSlice % (GetProcessingSettings().nDeviceHelperThreads + 1) == 0) {
163  if (ReadEvent(iSlice, 0)) {
164  GPUError("Error reading event");
165  error = 1;
166  continue;
167  }
168  } else {
169  if (GetProcessingSettings().debugLevel >= 3) {
170  GPUInfo("Waiting for helper thread %d", iSlice % (GetProcessingSettings().nDeviceHelperThreads + 1) - 1);
171  }
172  while (HelperDone(iSlice % (GetProcessingSettings().nDeviceHelperThreads + 1) - 1) < (int)iSlice) {
173  ;
174  }
175  if (HelperError(iSlice % (GetProcessingSettings().nDeviceHelperThreads + 1) - 1)) {
176  error = 1;
177  continue;
178  }
179  }
180  if (!doGPU && trk.CheckEmptySlice() && GetProcessingSettings().debugLevel == 0) {
181  continue;
182  }
183 
184  if (GetProcessingSettings().debugLevel >= 6) {
185  *mDebugFile << "\n\nReconstruction: Slice " << iSlice << "/" << NSLICES << std::endl;
186  if (GetProcessingSettings().debugMask & 1) {
187  if (doSliceDataOnGPU) {
188  TransferMemoryResourcesToHost(RecoStep::TPCSliceTracking, &trk, -1, true);
189  }
190  trk.DumpSliceData(*mDebugFile);
191  }
192  }
193 
194  // Initialize temporary memory where needed
195  if (GetProcessingSettings().debugLevel >= 3) {
196  GPUInfo("Copying Slice Data to GPU and initializing temporary memory");
197  }
198  if (GetProcessingSettings().keepDisplayMemory && !doSliceDataOnGPU) {
199  memset((void*)trk.Data().HitWeights(), 0, trkShadow.Data().NumberOfHitsPlusAlign() * sizeof(*trkShadow.Data().HitWeights()));
200  } else {
201  runKernel<GPUMemClean16>(GetGridAutoStep(useStream, RecoStep::TPCSliceTracking), krnlRunRangeNone, {}, trkShadow.Data().HitWeights(), trkShadow.Data().NumberOfHitsPlusAlign() * sizeof(*trkShadow.Data().HitWeights()));
202  }
203 
204  // Copy Data to GPU Global Memory
205  if (!doSliceDataOnGPU) {
206  TransferMemoryResourcesToGPU(RecoStep::TPCSliceTracking, &trk, useStream);
207  }
208  if (GPUDebug("Initialization (3)", useStream)) {
209  throw std::runtime_error("memcpy failure");
210  }
211 
212  runKernel<GPUTPCNeighboursFinder>(GetGridBlk(GPUCA_ROW_COUNT, useStream), {iSlice}, {nullptr, streamInit[useStream] ? nullptr : &mEvents->init});
213  streamInit[useStream] = true;
214 
215  if (GetProcessingSettings().keepDisplayMemory) {
216  TransferMemoryResourcesToHost(RecoStep::TPCSliceTracking, &trk, -1, true);
217  memcpy(trk.LinkTmpMemory(), mRec->Res(trk.MemoryResLinks()).Ptr(), mRec->Res(trk.MemoryResLinks()).Size());
218  if (GetProcessingSettings().debugMask & 2) {
219  trk.DumpLinks(*mDebugFile, 0);
220  }
221  }
222 
223  runKernel<GPUTPCNeighboursCleaner>(GetGridBlk(GPUCA_ROW_COUNT - 2, useStream), {iSlice});
224  DoDebugAndDump(RecoStep::TPCSliceTracking, 4, trk, &GPUTPCTracker::DumpLinks, *mDebugFile, 1);
225 
226  runKernel<GPUTPCStartHitsFinder>(GetGridBlk(GPUCA_ROW_COUNT - 6, useStream), {iSlice});
227 #ifdef GPUCA_SORT_STARTHITS_GPU
228  if (doGPU) {
229  runKernel<GPUTPCStartHitsSorter>(GetGridAuto(useStream), {iSlice});
230  }
231 #endif
232  DoDebugAndDump(RecoStep::TPCSliceTracking, 32, trk, &GPUTPCTracker::DumpStartHits, *mDebugFile);
233 
234  if (GetProcessingSettings().memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) {
235  trk.UpdateMaxData();
236  AllocateRegisteredMemory(trk.MemoryResTracklets());
237  AllocateRegisteredMemory(trk.MemoryResOutput());
238  }
239 
240  if (!(doGPU || GetProcessingSettings().debugLevel >= 1) || GetProcessingSettings().trackletConstructorInPipeline) {
241  runKernel<GPUTPCTrackletConstructor>(GetGridAuto(useStream), {iSlice});
242  DoDebugAndDump(RecoStep::TPCSliceTracking, 128, trk, &GPUTPCTracker::DumpTrackletHits, *mDebugFile);
243  if (GetProcessingSettings().debugMask & 256 && !GetProcessingSettings().comparableDebutOutput) {
244  trk.DumpHitWeights(*mDebugFile);
245  }
246  }
247 
248  if (!(doGPU || GetProcessingSettings().debugLevel >= 1) || GetProcessingSettings().trackletSelectorInPipeline) {
249  runKernel<GPUTPCTrackletSelector>(GetGridAuto(useStream), {iSlice});
250  runKernel<GPUTPCGlobalTrackingCopyNumbers>({1, -ThreadCount(), useStream}, {iSlice}, {}, 1);
251  TransferMemoryResourceLinkToHost(RecoStep::TPCSliceTracking, trk.MemoryResCommon(), useStream, &mEvents->slice[iSlice]);
252  streamMap[iSlice] = useStream;
253  if (GetProcessingSettings().debugLevel >= 3) {
254  GPUInfo("Slice %u, Number of tracks: %d", iSlice, *trk.NTracks());
255  }
256  DoDebugAndDump(RecoStep::TPCSliceTracking, 512, trk, &GPUTPCTracker::DumpTrackHits, *mDebugFile);
257  }
258  }
259  mRec->SetNestedLoopOmpFactor(1);
260  if (error) {
261  return (3);
262  }
263 
264  if (doGPU || GetProcessingSettings().debugLevel >= 1) {
265  ReleaseEvent(&mEvents->init);
266  if (!doSliceDataOnGPU) {
267  WaitForHelperThreads();
268  }
269 
270  if (!GetProcessingSettings().trackletSelectorInPipeline) {
271  if (GetProcessingSettings().trackletConstructorInPipeline) {
272  SynchronizeGPU();
273  } else {
274  for (int i = 0; i < mRec->NStreams(); i++) {
275  RecordMarker(&mEvents->stream[i], i);
276  }
277  runKernel<GPUTPCTrackletConstructor, 1>(GetGridAuto(0), krnlRunRangeNone, {&mEvents->single, mEvents->stream, mRec->NStreams()});
278  for (int i = 0; i < mRec->NStreams(); i++) {
279  ReleaseEvent(&mEvents->stream[i]);
280  }
281  SynchronizeEventAndRelease(&mEvents->single);
282  }
283 
284  if (GetProcessingSettings().debugLevel >= 4) {
285  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
286  DoDebugAndDump(RecoStep::TPCSliceTracking, 128, processors()->tpcTrackers[iSlice], &GPUTPCTracker::DumpTrackletHits, *mDebugFile);
287  }
288  }
289 
290  int runSlices = 0;
291  int useStream = 0;
292  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice += runSlices) {
293  if (runSlices < GetProcessingSettings().trackletSelectorSlices) {
294  runSlices++;
295  }
296  runSlices = CAMath::Min<int>(runSlices, NSLICES - iSlice);
297  if (getKernelProperties<GPUTPCTrackletSelector>().minBlocks * BlockCount() < (unsigned int)runSlices) {
298  runSlices = getKernelProperties<GPUTPCTrackletSelector>().minBlocks * BlockCount();
299  }
300 
301  if (GetProcessingSettings().debugLevel >= 3) {
302  GPUInfo("Running TPC Tracklet selector (Stream %d, Slice %d to %d)", useStream, iSlice, iSlice + runSlices);
303  }
304  runKernel<GPUTPCTrackletSelector>(GetGridAuto(useStream), {iSlice, runSlices});
305  runKernel<GPUTPCGlobalTrackingCopyNumbers>({1, -ThreadCount(), useStream}, {iSlice}, {}, runSlices);
306  for (unsigned int k = iSlice; k < iSlice + runSlices; k++) {
307  TransferMemoryResourceLinkToHost(RecoStep::TPCSliceTracking, processors()->tpcTrackers[k].MemoryResCommon(), useStream, &mEvents->slice[k]);
308  streamMap[k] = useStream;
309  }
310  useStream++;
311  if (useStream >= mRec->NStreams()) {
312  useStream = 0;
313  }
314  }
315  }
316 
317  mSliceSelectorReady = 0;
318 
319  std::array<bool, NSLICES> transferRunning;
320  transferRunning.fill(true);
321  if ((GetRecoStepsOutputs() & GPUDataTypes::InOutType::TPCSectorTracks) || (doGPU && !(GetRecoStepsGPU() & RecoStep::TPCMerging))) {
322  if (param().rec.tpc.globalTracking) {
323  mWriteOutputDone.fill(0);
324  }
325  RunHelperThreads(&GPUChainTracking::HelperOutput, this, NSLICES);
326 
327  unsigned int tmpSlice = 0;
328  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
329  if (GetProcessingSettings().debugLevel >= 3) {
330  GPUInfo("Transfering Tracks from GPU to Host");
331  }
332 
333  if (tmpSlice == iSlice) {
334  SynchronizeEvents(&mEvents->slice[iSlice]);
335  }
336  while (tmpSlice < NSLICES && (tmpSlice == iSlice || IsEventDone(&mEvents->slice[tmpSlice]))) {
337  ReleaseEvent(&mEvents->slice[tmpSlice]);
338  if (*processors()->tpcTrackers[tmpSlice].NTracks() > 0) {
339  TransferMemoryResourceLinkToHost(RecoStep::TPCSliceTracking, processors()->tpcTrackers[tmpSlice].MemoryResOutput(), streamMap[tmpSlice], &mEvents->slice[tmpSlice]);
340  } else {
341  transferRunning[tmpSlice] = false;
342  }
343  tmpSlice++;
344  }
345 
346  if (GetProcessingSettings().keepAllMemory) {
347  TransferMemoryResourcesToHost(RecoStep::TPCSliceTracking, &processors()->tpcTrackers[iSlice], -1, true);
348  if (!GetProcessingSettings().trackletConstructorInPipeline) {
349  if (GetProcessingSettings().debugMask & 256 && !GetProcessingSettings().comparableDebutOutput) {
350  processors()->tpcTrackers[iSlice].DumpHitWeights(*mDebugFile);
351  }
352  }
353  if (!GetProcessingSettings().trackletSelectorInPipeline) {
354  if (GetProcessingSettings().debugMask & 512) {
355  processors()->tpcTrackers[iSlice].DumpTrackHits(*mDebugFile);
356  }
357  }
358  }
359 
360  if (transferRunning[iSlice]) {
361  SynchronizeEvents(&mEvents->slice[iSlice]);
362  }
363  if (GetProcessingSettings().debugLevel >= 3) {
364  GPUInfo("Tracks Transfered: %d / %d", *processors()->tpcTrackers[iSlice].NTracks(), *processors()->tpcTrackers[iSlice].NTrackHits());
365  }
366 
367  if (GetProcessingSettings().debugLevel >= 3) {
368  GPUInfo("Data ready for slice %d, helper thread %d", iSlice, iSlice % (GetProcessingSettings().nDeviceHelperThreads + 1));
369  }
370  mSliceSelectorReady = iSlice;
371 
372  if (param().rec.tpc.globalTracking) {
373  for (unsigned int tmpSlice2a = 0; tmpSlice2a <= iSlice; tmpSlice2a += GetProcessingSettings().nDeviceHelperThreads + 1) {
374  unsigned int tmpSlice2 = GPUTPCGlobalTracking::GlobalTrackingSliceOrder(tmpSlice2a);
375  unsigned int sliceLeft, sliceRight;
376  GPUTPCGlobalTracking::GlobalTrackingSliceLeftRight(tmpSlice2, sliceLeft, sliceRight);
377 
378  if (tmpSlice2 <= iSlice && sliceLeft <= iSlice && sliceRight <= iSlice && mWriteOutputDone[tmpSlice2] == 0) {
379  GlobalTracking(tmpSlice2, 0);
380  WriteOutput(tmpSlice2, 0);
381  mWriteOutputDone[tmpSlice2] = 1;
382  }
383  }
384  } else {
385  if (iSlice % (GetProcessingSettings().nDeviceHelperThreads + 1) == 0) {
386  WriteOutput(iSlice, 0);
387  }
388  }
389  }
390  WaitForHelperThreads();
391  }
392  if (!(GetRecoStepsOutputs() & GPUDataTypes::InOutType::TPCSectorTracks) && param().rec.tpc.globalTracking) {
393  std::vector<bool> blocking(NSLICES * mRec->NStreams());
394  for (int i = 0; i < NSLICES; i++) {
395  for (int j = 0; j < mRec->NStreams(); j++) {
396  blocking[i * mRec->NStreams() + j] = i % mRec->NStreams() == j;
397  }
398  }
399  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
400  unsigned int tmpSlice = GPUTPCGlobalTracking::GlobalTrackingSliceOrder(iSlice);
401  if (!((GetRecoStepsOutputs() & GPUDataTypes::InOutType::TPCSectorTracks) || (doGPU && !(GetRecoStepsGPU() & RecoStep::TPCMerging)))) {
402  unsigned int sliceLeft, sliceRight;
403  GPUTPCGlobalTracking::GlobalTrackingSliceLeftRight(tmpSlice, sliceLeft, sliceRight);
404  if (!blocking[tmpSlice * mRec->NStreams() + sliceLeft % mRec->NStreams()]) {
405  StreamWaitForEvents(tmpSlice % mRec->NStreams(), &mEvents->slice[sliceLeft]);
406  blocking[tmpSlice * mRec->NStreams() + sliceLeft % mRec->NStreams()] = true;
407  }
408  if (!blocking[tmpSlice * mRec->NStreams() + sliceRight % mRec->NStreams()]) {
409  StreamWaitForEvents(tmpSlice % mRec->NStreams(), &mEvents->slice[sliceRight]);
410  blocking[tmpSlice * mRec->NStreams() + sliceRight % mRec->NStreams()] = true;
411  }
412  }
413  GlobalTracking(tmpSlice, 0, !GetProcessingSettings().fullMergerOnGPU);
414  }
415  }
416  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
417  if (transferRunning[iSlice]) {
418  ReleaseEvent(&mEvents->slice[iSlice]);
419  }
420  }
421  } else {
422  mSliceSelectorReady = NSLICES;
423  GPUCA_OPENMP(parallel for if(!doGPU && GetProcessingSettings().ompKernels != 1) num_threads(mRec->SetAndGetNestedLoopOmpFactor(!doGPU, NSLICES)))
424  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
425  if (param().rec.tpc.globalTracking) {
426  GlobalTracking(iSlice, 0);
427  }
428  if (GetRecoStepsOutputs() & GPUDataTypes::InOutType::TPCSectorTracks) {
429  WriteOutput(iSlice, 0);
430  }
431  }
432  mRec->SetNestedLoopOmpFactor(1);
433  }
434 
435  if (param().rec.tpc.globalTracking && GetProcessingSettings().debugLevel >= 3) {
436  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
437  GPUInfo("Slice %d - Tracks: Local %d Global %d - Hits: Local %d Global %d", iSlice,
438  processors()->tpcTrackers[iSlice].CommonMemory()->nLocalTracks, processors()->tpcTrackers[iSlice].CommonMemory()->nTracks, processors()->tpcTrackers[iSlice].CommonMemory()->nLocalTrackHits, processors()->tpcTrackers[iSlice].CommonMemory()->nTrackHits);
439  }
440  }
441 
442  if (GetProcessingSettings().debugMask & 1024 && !GetProcessingSettings().comparableDebutOutput) {
443  for (unsigned int i = 0; i < NSLICES; i++) {
444  processors()->tpcTrackers[i].DumpOutput(*mDebugFile);
445  }
446  }
447 
448  if (DoProfile()) {
449  return (1);
450  }
451  for (unsigned int i = 0; i < NSLICES; i++) {
452  mIOPtrs.nSliceTracks[i] = *processors()->tpcTrackers[i].NTracks();
453  mIOPtrs.sliceTracks[i] = processors()->tpcTrackers[i].Tracks();
454  mIOPtrs.nSliceClusters[i] = *processors()->tpcTrackers[i].NTrackHits();
455  mIOPtrs.sliceClusters[i] = processors()->tpcTrackers[i].TrackHits();
456  if (GetProcessingSettings().keepDisplayMemory && !GetProcessingSettings().keepAllMemory) {
457  TransferMemoryResourcesToHost(RecoStep::TPCSliceTracking, &processors()->tpcTrackers[i], -1, true);
458  }
459  }
460  if (GetProcessingSettings().debugLevel >= 2) {
461  GPUInfo("TPC Slice Tracker finished");
462  }
463  mRec->PopNonPersistentMemory(RecoStep::TPCSliceTracking, qStr2Tag("TPCSLTRK"));
464  return 0;
465 }

479 void GPUChainTracking::WriteOutput(int iSlice, int threadId)
480 {
481  if (GetProcessingSettings().debugLevel >= 5) {
482  GPUInfo("Running WriteOutput for slice %d on thread %d\n", iSlice, threadId);
483  }
484  if (GetProcessingSettings().nDeviceHelperThreads) {
485  while (mLockAtomic.test_and_set(std::memory_order_acquire)) {
486  ;
487  }
488  }
489  processors()->tpcTrackers[iSlice].WriteOutputPrepare();
490  if (GetProcessingSettings().nDeviceHelperThreads) {
491  mLockAtomic.clear();
492  }
493  processors()->tpcTrackers[iSlice].WriteOutput();
494  if (GetProcessingSettings().debugLevel >= 5) {
495  GPUInfo("Finished WriteOutput for slice %d on thread %d\n", iSlice, threadId);
