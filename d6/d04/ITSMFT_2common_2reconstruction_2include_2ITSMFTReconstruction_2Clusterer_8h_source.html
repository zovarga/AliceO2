79  struct BBox {
80  uint16_t chipID = 0xffff;
81  uint16_t rowMin = 0xffff;
82  uint16_t colMin = 0xffff;
83  uint16_t rowMax = 0;
84  uint16_t colMax = 0;
85  BBox(uint16_t c) : chipID(c) {}
86  bool isInside(uint16_t row, uint16_t col) const { return row >= rowMin && row <= rowMax && col >= colMin && col <= colMax; }
87  auto rowSpan() const { return rowMax - rowMin + 1; }
88  auto colSpan() const { return colMax - colMin + 1; }
89  bool isAcceptableSize() const { return colMax - colMin < o2::itsmft::ClusterPattern::MaxColSpan && rowMax - rowMin < o2::itsmft::ClusterPattern::MaxRowSpan; }
90  void clear()
91  {
92  rowMin = colMin = 0xffff;
93  rowMax = colMax = 0;
94  }
95  void adjust(uint16_t row, uint16_t col)
96  {
97  if (row < rowMin) {
98  rowMin = row;
99  }
100  if (row > rowMax) {
101  rowMax = row;
102  }
103  if (col < colMin) {
104  colMin = col;
105  }
106  if (col > colMax) {
107  colMax = col;
108  }
109  }
110  };

252 template <typename VCLUS, typename VPAT>
253 void Clusterer::streamCluster(const std::vector<PixelData>& pixbuf, const std::array<Label, MaxLabels>* lblBuff, const Clusterer::BBox& bbox, const LookUp& pattIdConverter,
254  VCLUS* compClusPtr, VPAT* patternsPtr, MCTruth* labelsClusPtr, int nlab, bool isHuge)
255 {
256  if (labelsClusPtr && lblBuff) { // MC labels were requested
257  auto cnt = compClusPtr->size();
258  for (int i = nlab; i--;) {
259  labelsClusPtr->addElement(cnt, (*lblBuff)[i]);
260  }
261  }
262  auto colSpanW = bbox.colSpan();
263  auto rowSpanW = bbox.rowSpan();
264  // add to compact clusters, which must be always filled
265  std::array<unsigned char, ClusterPattern::MaxPatternBytes> patt{};
266  for (const auto& pix : pixbuf) {
267  uint32_t ir = pix.getRowDirect() - bbox.rowMin, ic = pix.getCol() - bbox.colMin;
268  int nbits = ir * colSpanW + ic;
269  patt[nbits >> 3] |= (0x1 << (7 - (nbits % 8)));
270  }
271  uint16_t pattID = (isHuge || pattIdConverter.size() == 0) ? CompCluster::InvalidPatternID : pattIdConverter.findGroupID(rowSpanW, colSpanW, patt.data());
272  uint16_t row = bbox.rowMin, col = bbox.colMin;
273  if (pattID == CompCluster::InvalidPatternID || pattIdConverter.isGroup(pattID)) {
274  if (pattID != CompCluster::InvalidPatternID) {
275  //For groupped topologies, the reference pixel is the COG pixel
276  float xCOG = 0., zCOG = 0.;
277  ClusterPattern::getCOG(rowSpanW, colSpanW, patt.data(), xCOG, zCOG);
278  row += round(xCOG);
279  col += round(zCOG);
280  }
281  if (patternsPtr) {
282  patternsPtr->emplace_back((unsigned char)rowSpanW);
283  patternsPtr->emplace_back((unsigned char)colSpanW);
284  int nBytes = rowSpanW * colSpanW / 8;
285  if (((rowSpanW * colSpanW) % 8) != 0) {
286  nBytes++;
287  }
288  patternsPtr->insert(patternsPtr->end(), std::begin(patt), std::begin(patt) + nBytes);
289  }
290  }
291  compClusPtr->emplace_back(row, col, pattID, bbox.chipID);
292 }
