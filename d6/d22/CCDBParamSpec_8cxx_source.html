43 std::vector<ConfigParamSpec> ccdbParamSpec(std::string const& path, bool runDependent, std::vector<CCDBMetadata> metadata)
44 {
45  // Add here CCDB objecs which should be considered run dependent
46  static std::vector<std::string> runDependentObjects = {"GLO/GRP"};
47  if (std::any_of(runDependentObjects.begin(), runDependentObjects.end(), [&path](std::string const& s) { return path == s; })) {
48  runDependent = true;
49  }
50  std::vector<ConfigParamSpec> result{ccdbPathSpec(path), ccdbRunDependent(runDependent)};
51  for (auto& [key, value] : metadata) {
52  result.push_back(ccdbMetadataSpec(key, value));
53  }
54  return result;
55 }
