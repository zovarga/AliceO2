90 void AODProducerWorkflowDPL::collectBCs(const o2::globaltracking::RecoContainer& data,
91  const std::vector<o2::InteractionTimeRecord>& mcRecords,
92  std::map<uint64_t, int>& bcsMap)
93 {
94  const auto& primVertices = data.getPrimaryVertices();
95  const auto& fddRecPoints = data.getFDDRecPoints();
96  const auto& ft0RecPoints = data.getFT0RecPoints();
97  const auto& fv0RecPoints = data.getFV0RecPoints();
98  const auto& caloEMCCellsTRGR = data.getEMCALTriggers();
99  const auto& ctpDigits = data.getCTPDigits();
100  const auto& zdcBCRecData = data.getZDCBCRecData();
101 
102  bcsMap[mStartIR.toLong()] = 1; // store the start of TF
103 
104  // collecting non-empty BCs and enumerating them
105  for (auto& rec : mcRecords) {
106  uint64_t globalBC = rec.toLong();
107  bcsMap[globalBC] = 1;
108  }
109 
110  for (auto& fddRecPoint : fddRecPoints) {
111  uint64_t globalBC = fddRecPoint.getInteractionRecord().toLong();
112  bcsMap[globalBC] = 1;
113  }
114 
115  for (auto& ft0RecPoint : ft0RecPoints) {
116  uint64_t globalBC = ft0RecPoint.getInteractionRecord().toLong();
117  bcsMap[globalBC] = 1;
118  }
119 
120  for (auto& fv0RecPoint : fv0RecPoints) {
121  uint64_t globalBC = fv0RecPoint.getInteractionRecord().toLong();
122  bcsMap[globalBC] = 1;
123  }
124 
125  for (auto& zdcRecData : zdcBCRecData) {
126  uint64_t globalBC = zdcRecData.ir.toLong();
127  bcsMap[globalBC] = 1;
128  }
129 
130  for (auto& vertex : primVertices) {
131  auto& timeStamp = vertex.getTimeStamp();
132  double tsTimeStamp = timeStamp.getTimeStamp() * 1E3; // mus to ns
133  uint64_t globalBC = relativeTime_to_GlobalBC(tsTimeStamp);
134  bcsMap[globalBC] = 1;
135  }
136 
137  for (auto& emcaltrg : caloEMCCellsTRGR) {
138  uint64_t globalBC = emcaltrg.getBCData().toLong();
139  bcsMap[globalBC] = 1;
140  }
141 
142  for (auto& ctpDigit : ctpDigits) {
143  uint64_t globalBC = ctpDigit.intRecord.toLong();
144  bcsMap[globalBC] = 1;
145  }
146 
147  int bcID = 0;
148  for (auto& item : bcsMap) {
149  item.second = bcID;
150  bcID++;
151  }
152 }

154 uint64_t AODProducerWorkflowDPL::getTFNumber(const o2::InteractionRecord& tfStartIR, int runNumber)
155 {
156  auto& mgr = o2::ccdb::BasicCCDBManager::instance();
157  o2::ccdb::CcdbApi ccdb_api;
158  const std::string rct_path = "RCT/RunInformation/";
159  const std::string start_orbit_path = "Trigger/StartOrbit";
160 
161  mgr.setURL(o2::base::NameConf::getCCDBServer());
162  ccdb_api.init(o2::base::NameConf::getCCDBServer());
163 
164  std::map<int, int>* mapStartOrbit = mgr.get<std::map<int, int>>(start_orbit_path);
165  int64_t ts = 0;
166  std::map<std::string, std::string> metadata;
167  std::map<std::string, std::string> headers;
168  const std::string run_path = Form("%s/%i", rct_path.data(), runNumber);
169  headers = ccdb_api.retrieveHeaders(run_path, metadata, -1);
170  ts = atol(headers["SOR"].c_str());
171 
172  // ccdb returns timestamp in mus
173  // mus to ms
174  ts = ts / 1000;
175 
176  uint32_t initialOrbit = mapStartOrbit->at(runNumber);
177  uint16_t firstRecBC = tfStartIR.bc;
178  uint32_t firstRecOrbit = tfStartIR.orbit;
179  const o2::InteractionRecord firstRec(firstRecBC, firstRecOrbit);
180  ts += firstRec.bc2ns() / 1000000;
181 
182  return ts;
183 };

247 template <typename mftTracksCursorType, typename AmbigMFTTracksCursorType>
248 void AODProducerWorkflowDPL::addToMFTTracksTable(mftTracksCursorType& mftTracksCursor, AmbigMFTTracksCursorType& ambigMFTTracksCursor,
249  GIndex trackID, const o2::globaltracking::RecoContainer& data, int collisionID,
250  std::uint64_t collisionBC, const std::map<uint64_t, int>& bcsMap)
251 {
252  // mft tracks
253  int bcSlice[2] = {-1, -1};
254  const auto& track = data.getMFTTrack(trackID);
255  const auto& rof = data.getMFTTracksROFRecords()[mMFTROFs[trackID.getIndex()]];
256  float trackTime = rof.getBCData().differenceInBC(mStartIR) * o2::constants::lhc::LHCBunchSpacingNS + mMFTROFrameHalfLengthNS;
257  float trackTimeRes = mMFTROFrameHalfLengthNS;
258  bool needBCSlice = collisionID < 0 || trackID.isAmbiguous();
259  std::uint64_t bcOfTimeRef;
260  if (needBCSlice) {
261  double error = mTimeMarginTrackTime + trackTimeRes;
262  bcOfTimeRef = fillBCSlice(bcSlice, trackTime - error, trackTime + error, bcsMap);
263  } else {
264  bcOfTimeRef = collisionBC - mStartIR.toLong(); // by default (unambiguous) track time is wrt collision BC
265  }
266  trackTime -= bcOfTimeRef * o2::constants::lhc::LHCBunchSpacingNS;
267 
268  mftTracksCursor(0,
269  collisionID,
270  track.getX(),
271  track.getY(),
272  truncateFloatFraction(track.getZ(), mTrackX), // for the forward tracks Z has the same role as X in barrel
273  truncateFloatFraction(track.getPhi(), mTrackAlpha),
274  truncateFloatFraction(track.getTanl(), mTrackTgl),
275  truncateFloatFraction(track.getInvQPt(), mTrack1Pt),
276  track.getNumberOfPoints(),
277  truncateFloatFraction(track.getTrackChi2(), mTrackChi2),
278  truncateFloatFraction(trackTime, mTrackTime),
279  truncateFloatFraction(trackTimeRes, mTrackTimeError));
280  if (needBCSlice) {
281  ambigMFTTracksCursor(0, mTableTrMFTID, bcSlice);
282  }
283  mGIDToTableMFTID.emplace(trackID, mTableTrMFTID);
284  mTableTrMFTID++;
285 }

287 template <typename TracksCursorType, typename TracksCovCursorType, typename TracksExtraCursorType, typename AmbigTracksCursorType,
288  typename MFTTracksCursorType, typename AmbigMFTTracksCursorType,
289  typename FwdTracksCursorType, typename FwdTracksCovCursorType, typename AmbigFwdTracksCursorType>
290 void AODProducerWorkflowDPL::fillTrackTablesPerCollision(int collisionID,
291  std::uint64_t collisionBC,
292  const o2::dataformats::VtxTrackRef& trackRef,
293  const gsl::span<const GIndex>& GIndices,
294  const o2::globaltracking::RecoContainer& data,
295  TracksCursorType& tracksCursor,
296  TracksCovCursorType& tracksCovCursor,
297  TracksExtraCursorType& tracksExtraCursor,
298  AmbigTracksCursorType& ambigTracksCursor,
299  MFTTracksCursorType& mftTracksCursor,
300  AmbigMFTTracksCursorType& ambigMFTTracksCursor,
301  FwdTracksCursorType& fwdTracksCursor,
302  FwdTracksCovCursorType& fwdTracksCovCursor,
303  AmbigFwdTracksCursorType& ambigFwdTracksCursor,
304  const std::map<uint64_t, int>& bcsMap)
305 {
306  for (int src = GIndex::NSources; src--;) {
307  int start = trackRef.getFirstEntryOfSource(src);
308  int end = start + trackRef.getEntriesOfSource(src);
309  for (int ti = start; ti < end; ti++) {
310  auto& trackIndex = GIndices[ti];
311  if (GIndex::includesSource(src, mInputSources)) {
312  if (src == GIndex::Source::MFT) { // MFT tracks are treated separately since they are stored in a different table
313  if (trackIndex.isAmbiguous() && mGIDToTableMFTID.find(trackIndex) != mGIDToTableMFTID.end()) { // was it already stored ?
314  continue;
315  }
316  addToMFTTracksTable(mftTracksCursor, ambigMFTTracksCursor, trackIndex, data, collisionID, collisionBC, bcsMap);
317  } else if (src == GIndex::Source::MCH || src == GIndex::Source::MFTMCH) { // FwdTracks tracks are treated separately since they are stored in a different table
318  if (trackIndex.isAmbiguous() && mGIDToTableFwdID.find(trackIndex) != mGIDToTableFwdID.end()) { // was it already stored ?
319  continue;
320  }
321  addToFwdTracksTable(fwdTracksCursor, fwdTracksCovCursor, ambigFwdTracksCursor, trackIndex, data, collisionID, collisionBC, bcsMap);
322  } else {
323  // barrel track: normal tracks table
324  if (trackIndex.isAmbiguous() && mGIDToTableID.find(trackIndex) != mGIDToTableID.end()) { // was it already stored ?
325  continue;
326  }
327  auto extraInfoHolder = processBarrelTrack(collisionID, collisionBC, trackIndex, data, bcsMap);
328  if (extraInfoHolder.trackTimeRes < 0.f) { // failed or rejected?
329  LOG(warning) << "Barrel track " << trackIndex << " has no time set, rejection is not expected : time=" << extraInfoHolder.trackTime
330  << " timeErr=" << extraInfoHolder.trackTimeRes << " BCSlice: " << extraInfoHolder.bcSlice[0] << ":" << extraInfoHolder.bcSlice[1];
331  continue;
332  }
333  addToTracksTable(tracksCursor, tracksCovCursor, data.getTrackParam(trackIndex), collisionID);
334  addToTracksExtraTable(tracksExtraCursor, extraInfoHolder);
335  // collecting table indices of barrel tracks for V0s table
336  if (extraInfoHolder.bcSlice[0] >= 0) {
337  ambigTracksCursor(0, mTableTrID, extraInfoHolder.bcSlice);
338  }
339  mGIDToTableID.emplace(trackIndex, mTableTrID);
340  mTableTrID++;
341  }
342  }
343  }
344  }
345 }

347 template <typename FwdTracksCursorType, typename FwdTracksCovCursorType, typename AmbigFwdTracksCursorType>
348 void AODProducerWorkflowDPL::addToFwdTracksTable(FwdTracksCursorType& fwdTracksCursor, FwdTracksCovCursorType& fwdTracksCovCursor,
349  AmbigFwdTracksCursorType& ambigFwdTracksCursor, GIndex trackID,
350  const o2::globaltracking::RecoContainer& data, int collisionID, std::uint64_t collisionBC,
351  const std::map<uint64_t, int>& bcsMap)
352 {
353 
354  // table columns must be floats, not double
355  uint8_t trackTypeId;
356  float x;
357  float y;
358  float z;
359  float rabs;
360  float phi;
361  float tanl;
362  float invqpt;
363  float chi2;
364  float pdca;
365  int nClusters;
366  float chi2matchmchmid = -1.0;
367  float chi2matchmchmft = -1.0;
368  float matchscoremchmft = -1.0;
369  int matchmfttrackid = -1;
370  int matchmchtrackid = -1;
371  uint16_t mchBitMap = 0;
372  uint8_t midBitMap = 0;
373  uint32_t midBoards = 0;
374  float trackTime = -999.f;
375  float trackTimeRes = -999.f;
376 
377  float sigX = 0;
378  float sigY = 0;
379  float sigPhi = 0;
380  float sigTgl = 0;
381  float sig1Pt = 0;
382 
383  int8_t rhoXY = 0;
384  int8_t rhoPhiX = 0;
385  int8_t rhoPhiY = 0;
386  int8_t rhoTglX = 0;
387  int8_t rhoTglY = 0;
388  int8_t rhoTglPhi = 0;
389  int8_t rho1PtX = 0;
390  int8_t rho1PtY = 0;
391  int8_t rho1PtPhi = 0;
392  int8_t rho1PtTgl = 0;
393  int bcSlice[2] = {-1, -1};
394  bool errGaussian = true; // decide if the errors are gaussian or just an interval
395 
396  if (trackID.getSource() == GIndex::MCH) { // This is a MCH track
397  trackTypeId = o2::aod::fwdtrack::MCHStandaloneTrack;
398  // mch standalone tracks extrapolated to vertex
399  // compute 3 sets of tracks parameters :
400  // - at vertex
401  // - at DCA
402  // - at the end of the absorber
403  // extrapolate to vertex
404  float vx = 0, vy = 0, vz = 0;
405  if (collisionID >= 0) {
406  const auto& v = data.getPrimaryVertex(collisionID);
407  vx = v.getX();
408  vy = v.getY();
409  vz = v.getZ();
410  }
411  const auto track = data.getMCHTrack(trackID);
412  o2::mch::TrackParam trackParamAtVertex(track.getZ(), track.getParameters());
413  double errVtx{0.0}; // FIXME: get errors associated with vertex if available
414  double errVty{0.0};
415  if (!o2::mch::TrackExtrap::extrapToVertex(trackParamAtVertex, vx, vy, vz, errVtx, errVty)) {
416  return;
417  }
418 
419  // extrapolate to DCA
420  o2::mch::TrackParam trackParamAtDCA(track.getZ(), track.getParameters());
421  if (!o2::mch::TrackExtrap::extrapToVertexWithoutBranson(trackParamAtDCA, vz)) {
422  return;
423  }
424 
425  // extrapolate to the end of the absorber
426  o2::mch::TrackParam trackParamAtRAbs(track.getZ(), track.getParameters());
427  if (!o2::mch::TrackExtrap::extrapToZ(trackParamAtRAbs, -505.)) { // FIXME: replace hardcoded 505
428  return;
429  }
430 
431  double dcaX = trackParamAtDCA.getNonBendingCoor() - vx;
432  double dcaY = trackParamAtDCA.getBendingCoor() - vy;
433  double dca = std::sqrt(dcaX * dcaX + dcaY * dcaY);
434 
435  double xAbs = trackParamAtRAbs.getNonBendingCoor();
436  double yAbs = trackParamAtRAbs.getBendingCoor();
437 
438  double px = trackParamAtVertex.px();
439  double py = trackParamAtVertex.py();
440  double pz = trackParamAtVertex.pz();
441 
442  double pt = std::sqrt(px * px + py * py);
443  double dphi = std::atan2(py, px);
444  double dtanl = pz / pt;
445  double dinvqpt = 1.0 / (trackParamAtVertex.getCharge() * pt);
446  double dpdca = trackParamAtVertex.p() * dca;
447  double dchi2 = track.getChi2OverNDF();
448 
449  x = trackParamAtVertex.getNonBendingCoor();
450  y = trackParamAtVertex.getBendingCoor();
451  z = trackParamAtVertex.getZ();
452  rabs = std::sqrt(xAbs * xAbs + yAbs * yAbs);
453  phi = dphi;
454  tanl = dtanl;
455  invqpt = dinvqpt;
456  chi2 = dchi2;
457  pdca = dpdca;
458  nClusters = track.getNClusters();
459 
460  sigX = TMath::Sqrt(trackParamAtVertex.getCovariances()(0, 0));
461  sigY = TMath::Sqrt(trackParamAtVertex.getCovariances()(1, 1));
462  sigPhi = TMath::Sqrt(trackParamAtVertex.getCovariances()(2, 2));
463  sigTgl = TMath::Sqrt(trackParamAtVertex.getCovariances()(3, 3));
464  sig1Pt = TMath::Sqrt(trackParamAtVertex.getCovariances()(4, 4));
465  rhoXY = (Char_t)(128. * trackParamAtVertex.getCovariances()(0, 1) / (sigX * sigY));
466  rhoPhiX = (Char_t)(128. * trackParamAtVertex.getCovariances()(0, 2) / (sigPhi * sigX));
467  rhoPhiY = (Char_t)(128. * trackParamAtVertex.getCovariances()(1, 2) / (sigPhi * sigY));
468  rhoTglX = (Char_t)(128. * trackParamAtVertex.getCovariances()(0, 3) / (sigTgl * sigX));
469  rhoTglY = (Char_t)(128. * trackParamAtVertex.getCovariances()(1, 3) / (sigTgl * sigY));
470  rhoTglPhi = (Char_t)(128. * trackParamAtVertex.getCovariances()(2, 3) / (sigTgl * sigPhi));
471  rho1PtX = (Char_t)(128. * trackParamAtVertex.getCovariances()(0, 4) / (sig1Pt * sigX));
472  rho1PtY = (Char_t)(128. * trackParamAtVertex.getCovariances()(1, 4) / (sig1Pt * sigY));
473  rho1PtPhi = (Char_t)(128. * trackParamAtVertex.getCovariances()(2, 4) / (sig1Pt * sigPhi));
474  rho1PtTgl = (Char_t)(128. * trackParamAtVertex.getCovariances()(3, 4) / (sig1Pt * sigTgl));
475 
476  const auto& rof = data.getMCHTracksROFRecords()[mMCHROFs[trackID.getIndex()]];

1881  // see VertexTrackMatcher::extractTracks creator method, i.e. central time estimated +- uncertainty defined as:
1882  // 1) for tracks having a gaussian time error: PVertexerParams.nSigmaTimeTrack * trackSigma + PVertexerParams.timeMarginTrackTime
1883  // 2) for tracks having time uncertainty in a fixed time interval (TPC,ITS,MFT..): half of the interval + PVertexerParams.timeMarginTrackTime
1884  // The track time in the TrackExtraInfo is stored in ns wrt the collision BC for unambigous tracks and wrt bcSlice[0] for ambiguous ones,
1885  // with convention for errors: trackSigma in case (1) and half of the time interval for case (2) above.
1886 
1887  // find indices of widest slice of global BCs in the map compatible with provided BC range. bcsMap is guaranteed to be non-empty
1888  uint64_t bcMin = relativeTime_to_GlobalBC(tmin), bcMax = relativeTime_to_GlobalBC(tmax);
1889  auto lower = bcsMap.lower_bound(bcMin), upper = bcsMap.upper_bound(bcMax);
1890  if (lower == bcsMap.end()) {
1891  --lower;
1892  }
1893  if (upper != lower) {
1894  --upper;
1895  }
1896  slice[0] = std::distance(bcsMap.begin(), lower);
1897  slice[1] = std::distance(bcsMap.begin(), upper);
1898  auto bcOfTimeRef = lower->first - this->mStartIR.toLong();
1899  LOG(debug) << "BC slice t:" << tmin << " " << slice[0] << "(" << lower->first << "/" << lower->second << ")"
1900  << " t: " << tmax << " " << slice[1] << "(" << upper->first << "/" << upper->second << ")"
1901  << " bcref: " << bcOfTimeRef;
