88 Bool_t Detector::ProcessHits(FairVolume* v)
89 {
90  // This method is called from the MC stepping
91 
92  // Track only charged particles and photons
93  bool isPhotonTrack = false;
94  Int_t particlePdg = fMC->TrackPid();
95  if (particlePdg == 22) { // If particle is standard PDG photon
96  isPhotonTrack = true;
97  }
98  if (!(isPhotonTrack || fMC->TrackCharge())) {
99  return kFALSE;
100  }
101 
102  // Check track status to define when hit is started and when it is stopped
103  bool startHit = false, stopHit = false;
104  if ((fMC->IsTrackEntering()) || (fMC->IsTrackInside() && !mTrackData.mHitStarted)) {
105  startHit = true;
106  } else if ((fMC->IsTrackExiting() || fMC->IsTrackOut() || fMC->IsTrackStop())) {
107  stopHit = true;
108  }
109 
110  // Increment energy loss at all steps except entrance
111  if (!startHit) {
112  mTrackData.mEnergyLoss += fMC->Edep();
113  }
114 
115  // Track is entering or created in the volume
116  // Start registering new hit, defined as the combination of all the steps from a given particle
117  if (startHit) {
118  mTrackData.mHitStarted = true;
119  mTrackData.mEnergyLoss = 0.;
120  fMC->TrackMomentum(mTrackData.mMomentumStart);
121  fMC->TrackPosition(mTrackData.mPositionStart);
122  }
123 
124  // Track is exiting or stopped within the volume, finalize recording of this hit and save it
125  if (stopHit) {
126  TLorentzVector positionStop;
127  fMC->TrackPosition(positionStop);
128  Int_t trackID = fMC->GetStack()->GetCurrentTrackNumber();
129 
130  // Get unique ID of the detector cell (sensitive volume)
131  Int_t cellId = mGeometry->getCurrentCellId(fMC);
132 
133  math_utils::Point3D<float> posStart(mTrackData.mPositionStart.X(), mTrackData.mPositionStart.Y(), mTrackData.mPositionStart.Z());
134  math_utils::Point3D<float> posStop(positionStop.X(), positionStop.Y(), positionStop.Z());
135  math_utils::Vector3D<float> momStart(mTrackData.mMomentumStart.Px(), mTrackData.mMomentumStart.Py(), mTrackData.mMomentumStart.Pz());
136  addHit(trackID, cellId, posStart, posStop, momStart,
137  mTrackData.mMomentumStart.E(), positionStop.T(),
138  mTrackData.mEnergyLoss, particlePdg);
139  } else {
140  return kFALSE; // do noting more
141  }
142 
143  return kTRUE;
144 }

184  // TODO: Verify which of the following 2 lines is correct

186  // Float_t wScint[nScint] = { 0.08528, 0.91472 }; // based on chemical composition of base: polyvinyltoluene
187  const Float_t dScint = 1.023;
188 
189  // PMMA plastic mixture: (C5O2H8)n, same for plastic fiber support and for the fiber core
190  // Fiber cladding is different, but it comprises only 3% of the fiber volume, so it is not included
191  const Int_t nPlast = 3;
192  Float_t aPlast[nPlast] = {1.00784, 12.0107, 15.999};
193  Float_t zPlast[nPlast] = {1, 6, 8};

200 
201  // Densities of fiber-equivalent material, for five density regions in front of the PMTs
202  const Int_t nFiberPMTs = 5;
203  Float_t dFiberPMTs[nFiberPMTs] = {0.109313, 0.217216, 0.364493, 1.373307, 1.406480};

205  // Aluminum

210  // Titanium grade 5 (https://en.wikipedia.org/wiki/Titanium_alloy) without iron and oxygen

220  Float_t zPMT[nPMT] = {29, 30, 14, 8, 6, 1, 7, 26, 24, 5, 51, 55, 4, 13};
221  Float_t wPMT[nPMT] = {0.07, 0.02, 0.14, 0.21, 0.11, 0.02, 0.02, 0.04, 0.01, 0.01, 0.00, 0.00, 0.01, 0.34};
222  const Float_t dPMT = Geometry::getPmtDensity();
223 
224  Int_t matId = 0; // tmp material id number

226 
227  Float_t tmaxfd = -10.0; // max deflection angle due to magnetic field in one step
228  Float_t stemax = 0.1; // max step allowed [cm]

243  o2::base::Detector::Medium(Air, "Air$", matId, unsens, fieldType, maxField,
244  tmaxfd, stemax, deemax, epsil, stmin);
245 
246  o2::base::Detector::Mixture(++matId, "Scintillator$", aScint, zScint, dScint, nScint, wScint);
247  o2::base::Detector::Medium(Scintillator, "Scintillator$", matId, unsens, fieldType, maxField,

251  o2::base::Detector::Medium(Plastic, "Plastic$", matId, unsens, fieldType, maxField,
252  tmaxfd, stemax, deemax, epsil, stmin);
253 
254  for (int i = 0; i < nFiberRings; i++) {
255  o2::base::Detector::Mixture(++matId, Form("FiberRing%i$", i + 1), aPlast, zPlast, dFiberRings[i], nPlast, wPlast);

266  o2::base::Detector::Material(++matId, "Aluminium$", aAlu, zAlu, dAlu, 8.9, 999);
