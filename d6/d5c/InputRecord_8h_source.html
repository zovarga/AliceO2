201  template <typename T = DataRef, typename R>
202  decltype(auto) get(R binding, int part = 0) const
203  {
204  DataRef ref{nullptr, nullptr};
205  using decayed = std::decay_t<R>;
206  // Get the actual dataref
207  if constexpr (std::is_same_v<decayed, char const*> ||
208  std::is_same_v<decayed, char*> ||
209  std::is_same_v<decayed, std::string>) {
210  try {
211  int pos = -1;
212  if constexpr (std::is_same_v<decayed, std::string>) {
213  pos = getPos(binding.c_str());
214  } else {
215  pos = getPos(binding);
216  }
217  if (pos < 0) {
218  throw std::invalid_argument("no matching route found for " + std::string(binding));
219  }
220  ref = this->getByPos(pos, part);
221  } catch (const std::exception& e) {
222  if constexpr (std::is_same_v<decayed, std::string>) {
223  throw runtime_error_f("Unknown argument requested %s - %s", binding.c_str(), e.what());
224  } else {
225  throw runtime_error_f("Unknown argument requested %s - %s", binding, e.what());
226  }
227  }
228  } else if constexpr (std::is_same_v<decayed, DataRef>) {
229  ref = binding;
230  } else {
231  static_assert(always_static_assert_v<R>, "Unknown binding type");
232  }
233  if constexpr (std::is_same_v<std::decay_t<T>, DataRef>) {
234  return ref;
235  } else if constexpr (std::is_same<T, std::string>::value) {
236  // substitution for std::string
237  // If we ask for a string, we need to duplicate it because we do not want
238  // the buffer to be deleted when it goes out of scope. The string is built
239  // from the data and its lengh, null-termination is not necessary.
240  // return std::string object
241  return std::string(ref.payload, DataRefUtils::getPayloadSize(ref));
242 
243  // implementation (c)
244  } else if constexpr (std::is_same<T, char const*>::value) {
245  // substitution for const char*
246  // If we ask for a char const *, we simply point to the payload. Notice this
247  // is meant for C-style strings which are expected to be null terminated.
248  // If you want to actually get hold of the buffer, use gsl::span<char> as that will
249  // give you the size as well.
250  // return pointer to payload content
251  return reinterpret_cast<char const*>(ref.payload);
252 
253  // implementation (d)
254  } else if constexpr (std::is_same<T, TableConsumer>::value) {
255  // substitution for TableConsumer
256  // For the moment this is dummy, as it requires proper support to
257  // create the RDataSource from the arrow buffer.
258  auto data = reinterpret_cast<uint8_t const*>(ref.payload);
259  return std::make_unique<TableConsumer>(data, DataRefUtils::getPayloadSize(ref));
260 
261  // implementation (e)
262  } else if constexpr (framework::is_boost_serializable<T>::value || is_specialization_v<T, BoostSerialized>) {
263  // substitution for boost-serialized entities
264  // We have to deserialize the ostringstream.
265  // FIXME: check that the string is null terminated.
266  // @return deserialized copy of payload
267  auto str = std::string(ref.payload, DataRefUtils::getPayloadSize(ref));
268  assert(DataRefUtils::getPayloadSize(ref) == sizeof(T));
269  if constexpr (is_specialization_v<T, BoostSerialized>) {
270  return o2::utils::BoostDeserialize<typename T::wrapped_type>(str);
271  } else {
272  return o2::utils::BoostDeserialize<T>(str);
273  }
274 
275  // implementation (f)
276  } else if constexpr (is_span<T>::value) {
277  // substitution for span of messageable objects
278  // FIXME: there will be std::span in C++20
279  static_assert(is_messageable<typename T::value_type>::value, "span can only be created for messageable types");
280  auto header = DataRefUtils::getHeader<header::DataHeader*>(ref);
281  assert(header);
282  if (sizeof(typename T::value_type) > 1 && header->payloadSerializationMethod != o2::header::gSerializationMethodNone) {
283  throw runtime_error("Inconsistent serialization method for extracting span");
284  }
285  using ValueT = typename T::value_type;
286  auto payloadSize = DataRefUtils::getPayloadSize(ref);
287  if (payloadSize % sizeof(ValueT)) {
288  throw runtime_error(("Inconsistent type and payload size at " + std::string(ref.spec->binding) + "(" + DataSpecUtils::describe(*ref.spec) + ")" +
289  ": type size " + std::to_string(sizeof(ValueT)) +
290  " payload size " + std::to_string(payloadSize))
291  .c_str());
292  }
293  return gsl::span<ValueT const>(reinterpret_cast<ValueT const*>(ref.payload), payloadSize / sizeof(ValueT));
294 
295  // implementation (g)
296  } else if constexpr (is_container<T>::value) {
297  // currently implemented only for vectors
298  if constexpr (is_specialization_v<std::remove_const_t<T>, std::vector>) {
299  auto header = DataRefUtils::getHeader<header::DataHeader*>(ref);
300  auto payloadSize = DataRefUtils::getPayloadSize(ref);
301  auto method = header->payloadSerializationMethod;
302  if (method == o2::header::gSerializationMethodNone) {
303  // TODO: construct a vector spectator
304  // this is a quick solution now which makes a copy of the plain vector data
305  auto* start = reinterpret_cast<typename T::value_type const*>(ref.payload);
306  auto* end = start + payloadSize / sizeof(typename T::value_type);
307  T result(start, end);
308  return result;
309  } else if (method == o2::header::gSerializationMethodROOT) {
315  using NonConstT = typename std::remove_const<T>::type;
316  if constexpr (is_specialization_v<T, ROOTSerialized> == true || has_root_dictionary<T>::value == true) {
317  // we expect the unique_ptr to hold an object, exception should have been thrown
318  // otherwise
319  auto object = DataRefUtils::as<NonConstT>(ref);
320  // need to swap the content of the deserialized container to a local variable to force return
321  // value optimization
322  T container;
323  std::swap(const_cast<NonConstT&>(container), *object);
324  return container;
325  } else {
326  throw runtime_error("No supported conversion function for ROOT serialized message");
327  }
328  } else {
329  throw runtime_error("Attempt to extract object from message with unsupported serialization type");
330  }
331  } else {
332  static_assert(always_static_assert_v<T>, "unsupported code path");
333  }
334 
335  // implementation (h)
336  } else if constexpr (is_messageable<T>::value) {
337  // extract a messageable type by reference
338  // Cast content of payload bound by @a binding to known type.
339  // we need to check the serialization type, the cast makes only sense for
340  // unserialized objects
341  using DataHeader = o2::header::DataHeader;
342 
343  auto header = DataRefUtils::getHeader<header::DataHeader*>(ref);
344  auto method = header->payloadSerializationMethod;
345  if (method != o2::header::gSerializationMethodNone) {
346  // FIXME: we could in principle support serialized content here as well if we
347  // store all extracted objects internally and provide cleanup
348  throw runtime_error("Can not extract a plain object from serialized message");
349  }
350  return *reinterpret_cast<T const*>(ref.payload);
351 
352  // implementation (i)
353  } else if constexpr (std::is_pointer<T>::value &&
354  (is_messageable<typename std::remove_pointer<T>::type>::value || has_root_dictionary<typename std::remove_pointer<T>::type>::value)) {
355  // extract a messageable type or object with ROOT dictionary by pointer
356  // return unique_ptr to message content with custom deleter
357  using DataHeader = o2::header::DataHeader;
358  using ValueT = typename std::remove_pointer<T>::type;
359 
360  auto header = DataRefUtils::getHeader<header::DataHeader*>(ref);
361  auto payloadSize = DataRefUtils::getPayloadSize(ref);
362  auto method = header->payloadSerializationMethod;
363  if (method == o2::header::gSerializationMethodNone) {
364  if constexpr (is_messageable<ValueT>::value) {
365  auto const* ptr = reinterpret_cast<ValueT const*>(ref.payload);
366  // return type with non-owning Deleter instance
367  std::unique_ptr<ValueT const, Deleter<ValueT const>> result(ptr, Deleter<ValueT const>(false));
368  return result;
369  } else if constexpr (is_specialization_v<ValueT, std::vector> && has_messageable_value_type<ValueT>::value) {
370  // TODO: construct a vector spectator
371  // this is a quick solution now which makes a copy of the plain vector data
372  auto* start = reinterpret_cast<typename ValueT::value_type const*>(ref.payload);
373  auto* end = start + payloadSize / sizeof(typename ValueT::value_type);
374  auto container = std::make_unique<ValueT>(start, end);
375  std::unique_ptr<ValueT const, Deleter<ValueT const>> result(container.release(), Deleter<ValueT const>(true));
376  return result;
377  }
378  throw runtime_error("unsupported code path");
379  } else if (method == o2::header::gSerializationMethodROOT) {
380  // This supports the common case of retrieving a root object and getting pointer.
381  // Notice that this will return a copy of the actual contents of the buffer, because
382  // the buffer is actually serialised, for this reason we return a unique_ptr<T>.
383  // FIXME: does it make more sense to keep ownership of all the deserialised
384  // objects in a single place so that we can avoid duplicate deserializations?
385  // explicitely specify serialization method to ROOT-serialized because type T
386  // is messageable and a different method would be deduced in DataRefUtils
387  // return type with owning Deleter instance, forwarding to default_deleter
388  std::unique_ptr<ValueT const, Deleter<ValueT const>> result(DataRefUtils::as<ROOTSerialized<ValueT>>(ref).release());
389  return result;
390  } else if (method == o2::header::gSerializationMethodCCDB) {
391  // This is to support deserialising objects from CCDB. Contrary to what happens for
392  // other objects, those objects are most likely long lived, so we
393  // keep around an instance of the associated object and deserialise it only when
394  // it's updated.
395  // FIXME: add ability to apply callbacks to deserialised objects.
396  auto id = ObjectCache::Id::fromRef(ref);
397  ConcreteDataMatcher matcher{header->dataOrigin, header->dataDescription, header->subSpecification};
398  // If the matcher does not have an entry in the cache, deserialise it
399  // and cache the deserialised object at the given id.
400  auto path = fmt::format("{}", DataSpecUtils::describe(matcher));
401  LOGP(info, "{}", path);
402  auto& cache = mRegistry.get<ObjectCache>();
403  auto& callbacks = mRegistry.get<CallbackService>();
404  auto cacheEntry = cache.matcherToId.find(path);
405  if (cacheEntry == cache.matcherToId.end()) {
406  cache.matcherToId.insert(std::make_pair(path, id));
407  std::unique_ptr<ValueT const, Deleter<ValueT const>> result(DataRefUtils::as<CCDBSerialized<ValueT>>(ref).release(), false);
408  void* obj = (void*)result.get();
409  callbacks(CallbackService::Id::CCDBDeserialised, (ConcreteDataMatcher&)matcher, (void*)obj);
410  cache.idToObject[id] = obj;
411  LOGP(info, "Caching in {} ptr to {} ({})", id.value, path, obj);
412  return result;
413  }
414  auto& oldId = cacheEntry->second;
415  // The id in the cache is the same, let's simply return it.
416  if (oldId.value == id.value) {
417  std::unique_ptr<ValueT const, Deleter<ValueT const>> result((ValueT const*)cache.idToObject[id], false);
418  LOGP(info, "Returning cached entry {} for {} ({})", id.value, path, (void*)result.get());
419  return result;
420  }
421  // The id in the cache is different. Let's destroy the old cached entry
422  // and create a new one.
423  delete reinterpret_cast<ValueT*>(cache.idToObject[oldId]);
424  std::unique_ptr<ValueT const, Deleter<ValueT const>> result(DataRefUtils::as<CCDBSerialized<ValueT>>(ref).release(), false);
425  void* obj = (void*)result.get();
426  callbacks(CallbackService::Id::CCDBDeserialised, (ConcreteDataMatcher&)matcher, (void*)obj);
427  cache.idToObject[id] = obj;
428  LOGP(info, "Replacing cached entry {} with {} for {} ({})", oldId.value, id.value, path, obj);
429  oldId.value = id.value;
430  return result;
431  } else {
432  throw runtime_error("Attempt to extract object from message with unsupported serialization type");
433  }
434  } else if constexpr (has_root_dictionary<T>::value) {
435  // retrieving ROOT objects follows the pointer approach, i.e. T* has to be specified
436  // as template parameter and a unique_ptr will be returned, std vectors of ROOT serializable
437  // objects can be retrieved by move, this is handled above in the "container" code branch
438  static_assert(always_static_assert_v<T>, "ROOT objects need to be retrieved by pointer");
439  } else {
440  // non-messageable objects for which serialization method can not be derived by type,
441  // the operation depends on the transmitted serialization method
442  auto header = DataRefUtils::getHeader<header::DataHeader*>(ref);
443  auto method = header->payloadSerializationMethod;
444  if (method == o2::header::gSerializationMethodNone) {
445  // this code path is only selected if the type is non-messageable
446  throw runtime_error(
447  "Type mismatch: attempt to extract a non-messagable object "
448  "from message with unserialized data");
449  } else if (method == o2::header::gSerializationMethodROOT) {
450  // explicitely specify serialization method to ROOT-serialized because type T
451  // is messageable and a different method would be deduced in DataRefUtils
452  // return type with owning Deleter instance, forwarding to default_deleter
453  std::unique_ptr<T const, Deleter<T const>> result(DataRefUtils::as<ROOTSerialized<T>>(ref).release());
454  return result;
455  } else {
456  throw runtime_error("Attempt to extract object from message with unsupported serialization type");
457  }
458  }

506  Iterator(ParentType const* parent, size_t position = 0, size_t size = 0)
507  : mParent(parent), mPosition(position), mSize(size > position ? size : position), mElement{nullptr, nullptr, nullptr}
508  {
509  if (mPosition < mSize) {
510  if (mParent->isValid(mPosition)) {
511  mElement = mParent->getByPos(mPosition);
512  } else {
513  ++(*this);
514  }
515  }
516  }
517 
518  ~Iterator() = default;
519 
520  // prefix increment
521  SelfType& operator++()
522  {
523  while (mPosition < mSize && ++mPosition < mSize) {
524  if (!mParent->isValid(mPosition)) {
525  continue;
526  }
527  mElement = mParent->getByPos(mPosition);
528  break;
529  }
530  if (mPosition >= mSize) {
531  // reset the element to the default value of the type
532  mElement = ElementType{};
533  }
534  return *this;
535  }
536  // postfix increment
537  SelfType operator++(int /*unused*/)
538  {
539  SelfType copy(*this);
540  operator++();

542  }

544  reference operator*() const
545  {
546  return mElement;
547  }

550  {
