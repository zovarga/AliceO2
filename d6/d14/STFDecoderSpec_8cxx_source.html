54 template <class Mapping>
55 void STFDecoder<Mapping>::init(InitContext& ic)
56 {
57  try {
58  mDecoder = std::make_unique<RawPixelDecoder<Mapping>>();
59  auto v0 = o2::utils::Str::tokenize(mInputSpec, ':');
60  auto v1 = o2::utils::Str::tokenize(v0[1], '/');
61  header::DataOrigin dataOrig;
62  header::DataDescription dataDesc;
63  dataOrig.runtimeInit(v1[0].c_str());
64  dataDesc.runtimeInit(v1[1].c_str());
65  mDecoder->setUserDataOrigin(dataOrig);
66  mDecoder->setUserDataDescription(dataDesc);
67  mDecoder->init();
68  } catch (const std::exception& e) {
69  LOG(error) << "exception was thrown in decoder creation: " << e.what();
70  throw;
71  } catch (...) {
72  LOG(error) << "non-std::exception was thrown in decoder creation";
73  throw;
74  }
75 
76  auto detID = Mapping::getDetID();
77  if (detID == o2::detectors::DetID::ITS) {
78  mDictName = o2::itsmft::ClustererParam<o2::detectors::DetID::ITS>::Instance().dictFilePath;
79  mNoiseName = o2::itsmft::ClustererParam<o2::detectors::DetID::ITS>::Instance().noiseFilePath;
80  } else {
81  mDictName = o2::itsmft::ClustererParam<o2::detectors::DetID::MFT>::Instance().dictFilePath;
82  mNoiseName = o2::itsmft::ClustererParam<o2::detectors::DetID::MFT>::Instance().noiseFilePath;
83  }
84  mNoiseName = o2::base::DetectorNameConf::getNoiseFileName(detID, mNoiseName, "root");
85  mDictName = o2::base::DetectorNameConf::getAlpideClusterDictionaryFileName(detID, mDictName);
86 
87  try {
88  mNThreads = std::max(1, ic.options().get<int>("nthreads"));
89  mDecoder->setNThreads(mNThreads);
90  mDecoder->setFormat(ic.options().get<bool>("old-format") ? GBTLink::OldFormat : GBTLink::NewFormat);
91  mUnmutExtraLanes = ic.options().get<bool>("unmute-extra-lanes");
92  mVerbosity = ic.options().get<int>("decoder-verbosity");
93  mDumpOnError = ic.options().get<int>("raw-data-dumps");
94  if (mDumpOnError < 0 || mDumpOnError >= int(GBTLink::RawDataDumps::DUMP_NTYPES)) {
95  throw std::runtime_error(fmt::format("unknown raw data dump level {} requested", mDumpOnError));
96  }
97  mDecoder->setFillCalibData(mDoCalibData);
98  if (o2::utils::Str::pathExists(mNoiseName)) {
99  TFile* f = TFile::Open(mNoiseName.data(), "old");
100  auto pnoise = (NoiseMap*)f->Get("ccdb_object");
101  AlpideCoder::setNoisyPixels(pnoise);
102  LOG(info) << mSelfName << " loading noise map file: " << mNoiseName;
103  } else {
104  LOG(info) << mSelfName << " Noise file " << mNoiseName << " is absent, " << Mapping::getName() << " running without noise suppression";
105  }
106  } catch (const std::exception& e) {
107  LOG(error) << "exception was thrown in decoder configuration: " << e.what();
108  throw;
109  } catch (...) {
110  LOG(error) << "non-std::exception was thrown in decoder configuration";
111  throw;
112  }
113 
114  if (mDoClusters) {
115  try {
116  mClusterer = std::make_unique<Clusterer>();
117  mClusterer->setNChips(Mapping::getNChips());
118  const auto grp = o2::parameters::GRPObject::loadFrom();
119  if (grp) {
120  mClusterer->setContinuousReadOut(grp->isDetContinuousReadOut(detID));
121  } else {
122  throw std::runtime_error("failed to retrieve GRP");
123  }
124 
125  // settings for the fired pixel overflow masking
126  const auto& alpParams = DPLAlpideParam<Mapping::getDetID()>::Instance();
127  const auto& clParams = ClustererParam<Mapping::getDetID()>::Instance();
128  auto nbc = clParams.maxBCDiffToMaskBias;
129  nbc += mClusterer->isContinuousReadOut() ? alpParams.roFrameLengthInBC : (alpParams.roFrameLengthTrig / o2::constants::lhc::LHCBunchSpacingNS);
130  mClusterer->setMaxBCSeparationToMask(nbc);
131  mClusterer->setMaxRowColDiffToMask(clParams.maxRowColDiffToMask);
132 
133  if (o2::utils::Str::pathExists(mDictName)) {
134  mClusterer->loadDictionary(mDictName);
135  LOG(info) << mSelfName << " clusterer running with a provided dictionary: " << mDictName;
136  } else {
137  LOG(info) << mSelfName << " Dictionary " << mDictName << " is absent, " << Mapping::getName() << " clusterer expects cluster patterns";
138  }
139  mClusterer->print();
140  } catch (const std::exception& e) {
141  LOG(error) << "exception was thrown in clustrizer configuration: " << e.what();
142  throw;
143  } catch (...) {
144  LOG(error) << "non-std::exception was thrown in clusterizer configuration";
145  throw;
146  }
147  }

151 template <class Mapping>
152 void STFDecoder<Mapping>::run(ProcessingContext& pc)
153 {
154  static bool firstCall = true;
155  if (firstCall) {
156  firstCall = false;
157  mDecoder->setInstanceID(pc.services().get<const o2::framework::DeviceSpec>().inputTimesliceId);
158  mDecoder->setNInstances(pc.services().get<const o2::framework::DeviceSpec>().maxInputTimeslices);
159  mDecoder->setVerbosity(mDecoder->getInstanceID() == 0 ? mVerbosity : (mUnmutExtraLanes ? mVerbosity : -1));
160  mAllowReporting &= (mDecoder->getInstanceID() == 0) || mUnmutExtraLanes;
161  }
162  int nSlots = pc.inputs().getNofParts(0);
163  double timeCPU0 = mTimer.CpuTime(), timeReal0 = mTimer.RealTime();
164  mTimer.Start(false);
165  mDecoder->startNewTF(pc.inputs());
166  auto orig = Mapping::getOrigin();
167  std::vector<o2::itsmft::CompClusterExt> clusCompVec;
168  std::vector<o2::itsmft::ROFRecord> clusROFVec;
169  std::vector<unsigned char> clusPattVec;
170 
171  std::vector<Digit> digVec;
172  std::vector<GBTCalibData> calVec;
173  std::vector<ROFRecord> digROFVec;
174 
175  if (mDoDigits) {
176  digVec.reserve(mEstNDig);
177  digROFVec.reserve(mEstNROF);
178  }
179  if (mDoClusters) {
180  clusCompVec.reserve(mEstNClus);
181  clusROFVec.reserve(mEstNROF);
182  clusPattVec.reserve(mEstNClusPatt);
183  }
184  if (mDoCalibData) {
185  calVec.reserve(mEstNCalib);
186  }
187 
188  mDecoder->setDecodeNextAuto(false);
189  while (mDecoder->decodeNextTrigger()) {
190  if (mDoDigits) { // call before clusterization, since the latter will hide the digits
191  mDecoder->fillDecodedDigits(digVec, digROFVec); // lot of copying involved
192  if (mDoCalibData) {
193  mDecoder->fillCalibData(calVec);
194  }
195  }
196  if (mDoClusters) { // !!! THREADS !!!
197  mClusterer->process(mNThreads, *mDecoder.get(), &clusCompVec, mDoPatterns ? &clusPattVec : nullptr, &clusROFVec);
198  }
199  }
200 
201  if (mDoDigits) {
202  pc.outputs().snapshot(Output{orig, "DIGITS", 0, Lifetime::Timeframe}, digVec);
203  pc.outputs().snapshot(Output{orig, "DIGITSROF", 0, Lifetime::Timeframe}, digROFVec);
204  mEstNDig = std::max(mEstNDig, size_t(digVec.size() * 1.2));
205  mEstNROF = std::max(mEstNROF, size_t(digROFVec.size() * 1.2));
206  if (mDoCalibData) {
207  pc.outputs().snapshot(Output{orig, "GBTCALIB", 0, Lifetime::Timeframe}, calVec);
208  mEstNCalib = std::max(mEstNCalib, size_t(calVec.size() * 1.2));
209  }
210  }
211 
212  if (mDoClusters) { // we are not obliged to create vectors which are not requested, but other devices might not know the options of this one
213  pc.outputs().snapshot(Output{orig, "COMPCLUSTERS", 0, Lifetime::Timeframe}, clusCompVec);
214  pc.outputs().snapshot(Output{orig, "PATTERNS", 0, Lifetime::Timeframe}, clusPattVec);
215  pc.outputs().snapshot(Output{orig, "CLUSTERSROF", 0, Lifetime::Timeframe}, clusROFVec);
216  mEstNClus = std::max(mEstNClus, size_t(clusCompVec.size() * 1.2));
217  mEstNClusPatt = std::max(mEstNClusPatt, size_t(clusPattVec.size() * 1.2));
218  mEstNROF = std::max(mEstNROF, size_t(clusROFVec.size() * 1.2));
219  }
220 
221  if (mDumpOnError != int(GBTLink::RawDataDumps::DUMP_NONE)) {
222  mDecoder->produceRawDataDumps(mDumpOnError, DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true)));
223  }
224 
225  if (mDoClusters) {
226  LOG(debug) << mSelfName << " Built " << clusCompVec.size() << " clusters in " << clusROFVec.size() << " ROFs";
227  }
228  if (mDoDigits) {
229  LOG(debug) << mSelfName << " Decoded " << digVec.size() << " Digits in " << digROFVec.size() << " ROFs";
230  }
231  mTimer.Stop();
232  auto tfID = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true))->tfCounter;
233  LOG(debug) << mSelfName << " Total time for TF " << tfID << '(' << mTFCounter << ") : CPU: " << mTimer.CpuTime() - timeCPU0 << " Real: " << mTimer.RealTime() - timeReal0;
234  mTFCounter++;
235 }

239 {
240  if (mFinalizeDone) {
241  return;
242  }
243  mFinalizeDone = true;
244  LOGF(info, "%s statistics:", mSelfName);
245  LOGF(info, "%s Total STF decoding%s timing (w/o disk IO): Cpu: %.3e Real: %.3e s in %d slots", mSelfName,
246  mDoClusters ? "/clustering" : "", mTimer.CpuTime(), mTimer.RealTime(), mTimer.Counter() - 1);
247  if (mDecoder && mAllowReporting) {
248  mDecoder->printReport();
249  }
250  if (mClusterer) {
251  mClusterer->print();
252  }

255 DataProcessorSpec getSTFDecoderSpec(const STFDecoderInp& inp)
256 {
257  std::vector<OutputSpec> outputs;
258  if (inp.doDigits) {
259  outputs.emplace_back(inp.origin, "DIGITS", 0, Lifetime::Timeframe);
260  outputs.emplace_back(inp.origin, "DIGITSROF", 0, Lifetime::Timeframe);
261  if (inp.doCalib) {
262  outputs.emplace_back(inp.origin, "GBTCALIB", 0, Lifetime::Timeframe);
263  }
264  }
265  if (inp.doClusters) {
266  outputs.emplace_back(inp.origin, "COMPCLUSTERS", 0, Lifetime::Timeframe);
267  outputs.emplace_back(inp.origin, "CLUSTERSROF", 0, Lifetime::Timeframe);
268  // in principle, we don't need to open this input if we don't need to send real data,
269  // but other devices expecting it do not know about options of this device: problem?
270  // if (doClusters && doPatterns)
271  outputs.emplace_back(inp.origin, "PATTERNS", 0, Lifetime::Timeframe);
272  }
273 
274  auto inputs = o2::framework::select(inp.inputSpec.c_str());
275  if (inp.askSTFDist) {
276  for (auto& ins : inputs) { // mark input as optional in order not to block the workflow if our raw data happen to be missing in some TFs
277  ins.lifetime = Lifetime::Optional;
278  }
279  // request the input FLP/DISTSUBTIMEFRAME/0 that is _guaranteed_ to be present, even if none of our raw data is present.
280  inputs.emplace_back("stfDist", "FLP", "DISTSUBTIMEFRAME", 0, o2::framework::Lifetime::Timeframe);
281  }
282 
283  return DataProcessorSpec{
284  inp.deviceName,
285  inputs,
286  outputs,
287  inp.origin == o2::header::gDataOriginITS ? AlgorithmSpec{adaptFromTask<STFDecoder<ChipMappingITS>>(inp)} : AlgorithmSpec{adaptFromTask<STFDecoder<ChipMappingMFT>>(inp)},
288  Options{
289  {"nthreads", VariantType::Int, 1, {"Number of decoding/clustering threads"}},
290  {"old-format", VariantType::Bool, false, {"Use old format (1 trigger per CRU page)"}},
291  {"decoder-verbosity", VariantType::Int, 0, {"Verbosity level (-1: silent, 0: errors, 1: headers, 2: data) of 1st lane"}},
292  {"raw-data-dumps", VariantType::Int, int(GBTLink::RawDataDumps::DUMP_HBF), {"Raw data dumps on error (0: none, 1: HBF for link, 2: whole TF for all links"}},
293  {"unmute-extra-lanes", VariantType::Bool, false, {"allow extra lanes to be as verbose as 1st one"}}}};
294 }
