28 template <int I>
29 int GPUChainTracking::RunTRDTracking()
30 {
31  if (!processors()->trdTrackerGPU.IsInitialized()) {
32  return 1;
33  }
34 
35  auto& Tracker = processors()->getTRDTracker<I>();
36  Tracker.Reset();
37  if (mIOPtrs.nTRDTracklets == 0) {
38  return 0;
39  }
40  Tracker.SetGenerateSpacePoints(mIOPtrs.trdSpacePoints == nullptr);
41 
42  mRec->PushNonPersistentMemory(qStr2Tag("TRDTRACK"));
43  SetupGPUProcessor(&Tracker, true);
44 
45  if constexpr (I == GPUTRDTrackerKernels::gpuVersion) {
46  for (unsigned int i = 0; i < mIOPtrs.nMergedTracks; i++) {
47  const GPUTPCGMMergedTrack& trk = mIOPtrs.mergedTracks[i];
48  if (!Tracker.PreCheckTrackTRDCandidate(trk)) {
49  continue;
50  }
51  const GPUTRDTrackGPU& trktrd = param().rec.tpc.nWaysOuter ? (GPUTRDTrackGPU)trk.OuterParam() : (GPUTRDTrackGPU)trk;
52  if (!Tracker.CheckTrackTRDCandidate(trktrd)) {
53  continue;
54  }
55 
56  if (Tracker.LoadTrack(trktrd, i, false)) {
57  return 1;
58  }
59  }
60  } else {
61 #ifdef GPUCA_HAVE_O2HEADERS
62  for (unsigned int i = 0; i < mIOPtrs.nOutputTracksTPCO2; i++) {
63  const auto& trk = mIOPtrs.outputTracksTPCO2[i];
64  if (!Tracker.PreCheckTrackTRDCandidate(trk)) {
65  continue;
66  }
67  const GPUTRDTrack& trktrd = (GPUTRDTrack)trk;
68  if (!Tracker.CheckTrackTRDCandidate(trktrd)) {
69  continue;
70  }
71 
72  if (Tracker.LoadTrack(trktrd, i, false)) {
73  return 1;
74  }
75  }
76 #endif
77  }
78 
79  DoTRDGPUTracking<I>();
80 
81  mIOPtrs.nTRDTracks = Tracker.NTracks();
82  if constexpr (I == GPUTRDTrackerKernels::gpuVersion) {
83  mIOPtrs.trdTracks = Tracker.Tracks();
84  mIOPtrs.trdTracksO2 = nullptr;
85  } else {
86 #ifdef GPUCA_HAVE_O2HEADERS
87  mIOPtrs.trdTracks = nullptr;
88  mIOPtrs.trdTracksO2 = Tracker.Tracks();
89 #endif
90  }
91  mRec->PopNonPersistentMemory(RecoStep::TRDTracking, qStr2Tag("TRDTRACK"));
92 
93  return 0;
94 }

96 template <int I, class T>
97 int GPUChainTracking::DoTRDGPUTracking(T* externalInstance)
98 {
99 #ifdef GPUCA_HAVE_O2HEADERS
100  bool doGPU = GetRecoStepsGPU() & RecoStep::TRDTracking;
101  auto* Tracker = &processors()->getTRDTracker<I>();
102  auto* TrackerShadow = doGPU ? &processorsShadow()->getTRDTracker<I>() : Tracker;
103  if (externalInstance) {
104  if constexpr (std::is_same_v<decltype(Tracker), decltype(externalInstance)>) {
105  Tracker = externalInstance;
106  } else {
107  throw std::runtime_error("Must not provide external instance that does not match template type");
108  }
109  }
110  Tracker->PrepareTracking(this);
111 
112  int useStream = 0;
113 
114  const auto& threadContext = GetThreadContext();
115  SetupGPUProcessor(Tracker, false);
116  if (doGPU) {
117  TrackerShadow->OverrideGPUGeometry(reinterpret_cast<GPUTRDGeometry*>(mFlatObjectsDevice.mCalibObjects.trdGeometry));
118  mInputsHost->mNTRDTracklets = mInputsShadow->mNTRDTracklets = processorsShadow()->ioPtrs.nTRDTracklets = mIOPtrs.nTRDTracklets;
119  mInputsHost->mNTRDTriggerRecords = mInputsShadow->mNTRDTriggerRecords = processorsShadow()->ioPtrs.nTRDTriggerRecords = mIOPtrs.nTRDTriggerRecords;
120  mInputsHost->mDoSpacepoints = mInputsShadow->mDoSpacepoints = !Tracker->GenerateSpacepoints();
121  AllocateRegisteredMemory(mInputsHost->mResourceTRD);
122  processorsShadow()->ioPtrs.trdTracklets = mInputsShadow->mTRDTracklets;
123  processorsShadow()->ioPtrs.trdSpacePoints = Tracker->GenerateSpacepoints() ? Tracker->SpacePoints() : mInputsShadow->mTRDSpacePoints;
124  if constexpr (std::is_same_v<decltype(processorsShadow()->ioPtrs.trdTracks), decltype(TrackerShadow->Tracks())>) {
125  processorsShadow()->ioPtrs.trdTracks = TrackerShadow->Tracks();
126  } else {
127  processorsShadow()->ioPtrs.trdTracks = nullptr;
128  }
129  processorsShadow()->ioPtrs.nTRDTracks = mIOPtrs.nTRDTracks;
130  processorsShadow()->ioPtrs.trdTriggerTimes = mInputsShadow->mTRDTriggerTimes;
131  processorsShadow()->ioPtrs.trdTrackletIdxFirst = mInputsShadow->mTRDTrackletIdxFirst;
132  GPUMemCpy(RecoStep::TRDTracking, mInputsShadow->mTRDTracklets, mIOPtrs.trdTracklets, sizeof(*mIOPtrs.trdTracklets) * mIOPtrs.nTRDTracklets, useStream, true);
133  if (!Tracker->GenerateSpacepoints()) {
134  GPUMemCpy(RecoStep::TRDTracking, mInputsShadow->mTRDSpacePoints, mIOPtrs.trdSpacePoints, sizeof(*mIOPtrs.trdSpacePoints) * mIOPtrs.nTRDTracklets, useStream, true);
135  }
136  GPUMemCpy(RecoStep::TRDTracking, mInputsShadow->mTRDTriggerTimes, mIOPtrs.trdTriggerTimes, sizeof(*mIOPtrs.trdTriggerTimes) * mIOPtrs.nTRDTriggerRecords, useStream, true);
137  GPUMemCpy(RecoStep::TRDTracking, mInputsShadow->mTRDTrackletIdxFirst, mIOPtrs.trdTrackletIdxFirst, sizeof(*mIOPtrs.trdTrackletIdxFirst) * mIOPtrs.nTRDTriggerRecords, useStream, true);
138  if (mIOPtrs.trdTrigRecMask) {
139  processorsShadow()->ioPtrs.trdTrigRecMask = mInputsShadow->mTRDTrigRecMask;
140  GPUMemCpy(RecoStep::TRDTracking, mInputsShadow->mTRDTrigRecMask, mIOPtrs.trdTrigRecMask, sizeof(*mIOPtrs.trdTrigRecMask) * mIOPtrs.nTRDTriggerRecords, useStream, true);
141  } else {
142  processorsShadow()->ioPtrs.trdTrigRecMask = nullptr;
143  }
144  WriteToConstantMemory(RecoStep::TRDTracking, (char*)&processors()->ioPtrs - (char*)processors(), &processorsShadow()->ioPtrs, sizeof(processorsShadow()->ioPtrs), useStream);
145  WriteToConstantMemory(RecoStep::TRDTracking, (char*)&processors()->trdTrackerGPU - (char*)processors(), TrackerShadow, sizeof(*TrackerShadow), useStream);
146  }
147 
148  TransferMemoryResourcesToGPU(RecoStep::TRDTracking, Tracker, useStream);
149  runKernel<GPUTRDTrackerKernels, I>(GetGridAuto(useStream), krnlRunRangeNone, krnlEventNone, externalInstance ? Tracker : nullptr);
150  TransferMemoryResourcesToHost(RecoStep::TRDTracking, Tracker, useStream);
151  SynchronizeStream(useStream);
152 
153  if (GetProcessingSettings().debugLevel >= 2) {
154  GPUInfo("GPU TRD tracker Finished");
155  }
156 #endif
157  return (0);
158 }
