58  explicit LookUpTable(bool fillLinearly)
59  : mTopoVector(Nmodules * NChPerMod, {0, 0}),
60  mInvTopo(mTopoVector.size(), 0)
61  {
62  if (fillLinearly) {
63  LOG(info) << "Mapping of global channel and (PM, PM channel) pair";
64  for (int link = 0; link < Nmodules; ++link) {
65  for (int ch = 0; ch < NChPerMod; ++ch) {
66  mTopoVector[link * NChPerMod + ch] = o2::fdd::Topo{link, ch};
67  }
68  }
69  } else {
70  // TODO: If needed: implement more realistic splitting: 1 ring -> 1 PM instead of linear
71  LOG(warning) << "Don't use it - not implemented yet.";
72  }
73 
74  // Fill inverted LUT - matters only if LUT is not linear
75  for (size_t channel = 0; channel < mTopoVector.size(); ++channel) {
76  mInvTopo[getIdx(mTopoVector[channel].modLink, mTopoVector[channel].modCh)] = channel;
77  }
78  }
79 
80  int getChannel(int link, int mcp, int ep = 0) const { return mInvTopo[getIdx(link, mcp)]; }
81  int getLink(int channel) const { return mTopoVector[channel].modLink; }
82  int getModChannel(int channel) const { return mTopoVector[channel].modCh; }
83  int getTcmLink() const { return Nmodules; }
84  bool isTCM(int link, int ep) const { return link == 2 && ep == 0; }
85  Topo getTopoPM(int globalChannelID) const { return mTopoVector[globalChannelID]; }
86  Topo getTopoTCM() const { return Topo{getTcmLink(), 0}; }
87  std::size_t getNchannels() const { return mTopoVector.size(); } //get number of global PM channels
88  void printFullMap() const
89  {
90  LOG(info) << "o2::fdd::LookUpTable::printFullMap(): mTopoVector: [globalCh link modCh]";
91  for (size_t channel = 0; channel < mTopoVector.size(); ++channel) {
92  LOG(info) << " " << channel << " " << mTopoVector[channel].modLink << " " << mTopoVector[channel].modCh;
93  }
94  LOG(info) << "o2::fdd::LookUpTable::printFullMap(): mInvTopo: [idx globalCh link modCh]";
95  for (size_t idx = 0; idx < mInvTopo.size(); ++idx) {
96  LOG(info) << " " << idx << " " << mInvTopo[idx] << " " << getLinkFromIdx(mInvTopo[idx]) << " " << getModChannelFromIdx(mInvTopo[idx]);
97  }
98  }
99 
100  static o2::fdd::LookUpTable linear()
101  {
102  return o2::fdd::LookUpTable{1};
103  }
104 
105  private:
106  std::vector<Topo> mTopoVector; // iterator of each vector element gives the global channel number
107  std::vector<int> mInvTopo; // each element is an iterator of mTopoVector
108 
109  static int getIdx(int link, int modCh)
110  {
111  assert(modCh < NChPerMod);
112  return link * NChPerMod + modCh;
113  }
114  static int getLinkFromIdx(int idx) { return idx / NChPerMod; }
115  static int getModChannelFromIdx(int idx) { return idx % NChPerMod; }
116 
117  ClassDefNV(LookUpTable, 1);
118 };

152  std::map<Topo_t, RDHtype> mapResult;
153  const uint16_t cruID = 0; //constant
154  const uint32_t endPointID = 0; //constant
155  uint64_t feeID = 0; //increments
156  //PM
157  for (int iCh = 0; iCh < Instance().getNchannels(); iCh++) {
158  auto pairInserted = mapResult.insert({makeGlobalTopo(Instance().getTopoPM(iCh)), RDHtype{}});
159  if (pairInserted.second) {
160  auto& rdhObj = pairInserted.first->second;
161  const auto& topoObj = pairInserted.first->first;
162  if constexpr (std::is_same<RDHhelper, void>::value) {
163  rdhObj.linkID = topoObj.modLink;
164  rdhObj.endPointID = endPointID;
165  rdhObj.feeId = feeID;
166  rdhObj.cruID = cruID;
167  } else //Using RDHUtils
168  {
169  RDHhelper::setLinkID(&rdhObj, topoObj.modLink);
170  RDHhelper::setEndPointID(&rdhObj, endPointID);
171  RDHhelper::setFEEID(&rdhObj, feeID);
172  RDHhelper::setCRUID(&rdhObj, cruID);
173  }
174  feeID++;
175  }
176  }
177  //TCM
178  {
179  auto pairInserted = mapResult.insert({makeGlobalTopo(Instance().getTopoTCM()), RDHtype{}});
180  if (pairInserted.second) {
181  auto& rdhObj = pairInserted.first->second;
182  const auto& topoObj = pairInserted.first->first;
