62 void PHOSRunbyrunSlot::fill(const gsl::span<const Cluster>& clusters, const gsl::span<const TriggerRecord>& trs)
63 {
64  if (!mBadMap) {
65  if (mUseCCDB) {
66  LOG(info) << "Retrieving BadMap from CCDB";
67  o2::ccdb::CcdbApi ccdb;
68  ccdb.init(mCCDBPath); // or http://localhost:8080 for a local installation
69  std::map<std::string, std::string> metadata;
70  mBadMap.reset(ccdb.retrieveFromTFileAny<BadChannelsMap>("PHS/Calib/BadChannels", metadata, mRunStartTime));
71 
72  if (!mBadMap) { //was not read from CCDB, but expected
73  LOG(fatal) << "Can not read BadMap from CCDB, you may use --not-use-ccdb option to create default bad map";
74  }
75  } else {
76  LOG(info) << "Do not use CCDB, create default BadMap";
77  mBadMap.reset(new BadChannelsMap());
78  }
79  }
80 
81  for (auto& tr : trs) {
82 
83  int firstCluInEvent = tr.getFirstEntry();
84  int lastCluInEvent = firstCluInEvent + tr.getNumberOfObjects();
85 
86  //TODO!!! Get MFT0 vertex
87  TVector3 vertex = {0., 0., 0.};
88  mBuffer->startNewEvent(); // mark stored clusters to be used for Mixing
89  for (int i = firstCluInEvent; i < lastCluInEvent; i++) {
90  const Cluster& clu = clusters[i];
91 
92  if (!checkCluster(clu)) {
93  continue;
94  }
95  // prepare TLorentsVector
96  float posX, posZ;
97  clu.getLocalPosition(posX, posZ);
98  TVector3 vec3;
99  Geometry::GetInstance("Run3")->local2Global(clu.module(), posX, posZ, vec3);
100  vec3 -= vertex;
101  float e = clu.getEnergy();
102  vec3 *= 1. / vec3.Mag();
103  TLorentzVector v(vec3.X() * e, vec3.Y() * e, vec3.Z() * e, e);
104  for (short ip = mBuffer->size(); ip--;) {
105  const TLorentzVector& vp = mBuffer->getEntry(ip);
106  TLorentzVector sum = v + vp;
107  if (sum.Pt() > mPtCut) {
108  if (mBuffer->isCurrentEvent(ip)) { //same (real) event
109  mReMi[2 * clu.module()](sum.M()); // put all high-pt pairs to bin 4-6 GeV
110  } else { //Mixed
111  mReMi[2 * clu.module() + 1](sum.M()); // put all high-pt pairs to bin 4-6 GeV
112  }
113  }
114  }
115  mBuffer->addEntry(v);
116  }
117  }

123 bool PHOSRunbyrunSlot::checkCluster(const Cluster& clu)
124 {
125  if (clu.getEnergy() > 1.e-4) {
126  return false;
127  }
128  //First check BadMap
129  float posX, posZ;
130  clu.getLocalPosition(posX, posZ);
131  short absId;
132  Geometry::relPosToAbsId(clu.module(), posX, posZ, absId);
133  if (!mBadMap->isChannelGood(absId)) {
134  return false;
135  }
136  return (clu.getEnergy() > 0.3 && clu.getMultiplicity() > 1);
137 }

222 void PHOSRunbyrunCalibrator::endOfStream()
223 {
224  //Merge collected in different slots histograms
225  TF1 fRatio("ratio", this, &PHOSRunbyrunCalibrator::CBRatio, 0, 1, 6, "PHOSRunbyrunCalibrator", "CBRatio");
226  TF1 fBg("background", this, &PHOSRunbyrunCalibrator::bg, 0, 1, 6, "PHOSRunbyrunCalibrator", "bg");
227  TF1 fSignal("signal", this, &PHOSRunbyrunCalibrator::CBSignal, 0, 1, 6, "PHOSRunbyrunCalibrator", "CBSignal");
228  //fit inv mass distributions
229  TFile fout("mgg.root", "recreate");
230  for (int mod = 0; mod < 4; mod++) {
231  mReMi[2 * mod]->Sumw2();
232  mReMi[2 * mod + 1]->Sumw2();
233  TH1D* tmp = (TH1D*)mReMi[2 * mod]->Clone("Ratio");
234  tmp->Divide(mReMi[2 * mod + 1]);
235  fRatio.SetParameters(1., 0.134, 0.005, 0.01, 0., 0.);
236  tmp->Fit(&fRatio, "q", "", 0.07, 0.22);
237  fBg.SetParameters(fRatio.GetParameter(3), fRatio.GetParameter(4), fRatio.GetParameter(5));
238  mReMi[2 * mod + 1]->Multiply(&fBg);
239  // mReMi[2*mod]->Add(mReMi[2*mod+1],-1.) ;
240  fSignal.SetParameters(0.3 * mReMi[2 * mod]->Integral(65, 67, ""), 0.135, 0.005);
241  mReMi[2 * mod]->Fit(&fSignal, "q", "", 0.07, 0.22);
242  mRunByRun[2 * mod] = fSignal.GetParameter(1);
243  mRunByRun[2 * mod + 1] = fSignal.GetParError(1);
244  tmp->Write();
245  mReMi[2 * mod]->Write();
246  mReMi[2 * mod + 1]->Write();
247  delete tmp;
248  //Clear before next period?
249  mReMi[2 * mod]->Reset();
250  mReMi[2 * mod + 1]->Reset();
251  }
252  fout.Close();
253 }
254 double PHOSRunbyrunCalibrator::CBRatio(double* x, double* par)
255 {
256  double m = par[1];
257  double s = par[2];
258  const double n = 4.1983;
259  const double a = 1.5;
260  const double A = TMath::Power((n / TMath::Abs(a)), n) * TMath::Exp(-a * a / 2);
261  const double B = n / TMath::Abs(a) - TMath::Abs(a);
262  double dx = (x[0] - m) / s;
263  if (dx > -a) {
264  return par[0] * exp(-dx * dx / 2.) +
265  par[3] + par[4] * x[0] + par[5] * x[0] * x[0];
266  } else {
267  return par[0] * A * TMath::Power((B - dx), -n) +
268  par[3] + par[4] * x[0] + par[5] * x[0] * x[0];
269  }
270 }
271 double PHOSRunbyrunCalibrator::CBSignal(double* x, double* par)
272 {
273  double m = par[1];
274  double s = par[2];
275  const double n = 4.1983;
276  const double a = 1.5;
277  const double A = TMath::Power((n / TMath::Abs(a)), n) * TMath::Exp(-a * a / 2);
278  const double B = n / TMath::Abs(a) - TMath::Abs(a);
279  double dx = (x[0] - m) / s;
280  if (dx > -a) {
281  return par[0] * exp(-dx * dx / 2.) + par[3];
282  } else {
283  return par[0] * A * TMath::Power((B - dx), -n) + par[3];
284  }
285 }
