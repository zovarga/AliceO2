29 void CalibLaserTracks::fill(const gsl::span<const TrackTPC> tracks)
30 {
31  // ===| clean up TF data |===
32  mZmatchPairsTFA.clear();
33  mZmatchPairsTFC.clear();
34  mCalibDataTF.reset();
35 
36  if (!tracks.size()) {
37  return;
38  }
39 
40  // ===| associate tracks with ideal laser track positions |===
41  for (const auto& track : tracks) {
42  processTrack(track);
43  }
44 
45  // ===| set TF start and end times |===
46  if (mCalibData.firstTime == 0 && mCalibData.lastTime == 0) {
47  mCalibData.firstTime = mTFstart;
48  }
49 
50  auto tfEnd = mTFend;
51  if (tfEnd == 0) {
52  tfEnd = mTFstart;
53  }
54  mCalibData.lastTime = tfEnd;
55 
56  mCalibDataTF.firstTime = mTFstart;
57  mCalibDataTF.lastTime = tfEnd;
58 
59  // ===| TF counters |===
60  ++mCalibData.processedTFs;
61  ++mCalibDataTF.processedTFs;
62 
63  // ===| finalize TF processing |===
64  endTF();
65 }

68 void CalibLaserTracks::processTrack(const TrackTPC& track)
69 {
70  if (track.hasBothSidesClusters()) {
71  return;
72  }
73 
74  // use outer parameters which are closest to the laser mirrors
75  auto parOutAtLtr = track.getOuterParam();
76 
77  // track should have been alreay propagated close to the laser mirrors
78  if (parOutAtLtr.getX() < 220) {
79  return;
80  }
81 
82  // recalculate z position based on trigger or CE position if needed
83  float zTrack = parOutAtLtr.getZ();
84 
85  // TODO: calculation has to be improved
86  if (mTriggerPos < 0) {
87  // use CE for time 0
88  const float zOffset = (track.getTime0() + mTriggerPos) * mZbinWidth * mDriftV + 250;
89  //printf("time0: %.2f, trigger pos: %d, zTrack: %.2f, zOffset: %.2f\n", track.getTime0(), mTriggerPos, zTrack, zOffset);
90  zTrack += zOffset;
91  parOutAtLtr.setZ(zTrack);
92  } else if (mTriggerPos > 0) {
93  }
94 
95  if (std::abs(zTrack) > 300) {
96  return;
97  }
98 
99  // try association with ideal laser track and rotate parameters
100  const int side = track.hasCSideClusters();
101  const int laserTrackID = findLaserTrackID(parOutAtLtr, side);
102 
103  if (laserTrackID < 0 || laserTrackID >= LaserTrack::NumberOfTracks) {
104  return;
105  }
106 
107  auto ltr = mLaserTracks.getTrack(laserTrackID);
108  parOutAtLtr.rotateParam(ltr.getAlpha());
109  parOutAtLtr.propagateParamTo(ltr.getX(), mBz);
110 
111  if (ltr.getSide() == 0) {
112  mZmatchPairsA.emplace_back(TimePair{ltr.getZ(), parOutAtLtr.getZ(), mTFstart});
113  mZmatchPairsTFA.emplace_back(TimePair{ltr.getZ(), parOutAtLtr.getZ(), mTFstart});
114  } else {
115  mZmatchPairsC.emplace_back(TimePair{ltr.getZ(), parOutAtLtr.getZ(), mTFstart});
116  mZmatchPairsTFC.emplace_back(TimePair{ltr.getZ(), parOutAtLtr.getZ(), mTFstart});
117  }
118 
119  mCalibData.matchedLtrIDs.emplace_back(laserTrackID);
120  mCalibDataTF.matchedLtrIDs.emplace_back(laserTrackID);
121 
122  // ===| debug output |========================================================
123  if (mWriteDebugTree) {
124  if (!mDebugStream) {
125  mDebugStream = std::make_unique<o2::utils::TreeStreamRedirector>(mDebugOutputName.data(), "recreate");
126  }
127 
128  auto writeTrack = track;
129  *mDebugStream << "ltrMatch"
130  << "tfStart=" << mTFstart
131  << "tfEnd=" << mTFend
132  << "ltr=" << ltr // matched ideal laser track
133  << "trOutLtr=" << parOutAtLtr // track rotated and propagated to ideal track position
134  << "TPCTracks=" << writeTrack // original TPC track
135  << "\n";
136  }
137 }

140 int CalibLaserTracks::findLaserTrackID(TrackPar outerParam, int side)
141 {
142  // ===| rotate outer param to closes laser rod |===
143  const auto phisec = getPhiNearbyLaserRod(outerParam, side);
144  if (!outerParam.rotateParam(phisec)) {
145  return -1;
146  }
147 
148  if (side < 0) {
149  side = outerParam.getZ() < 0;
150  }
151 
152  // ===| laser rod |===
153  const int rod = std::nearbyint((phisec - LaserTrack::FirstRodPhi[side]) / LaserTrack::RodDistancePhi);
154 
155  // ===| laser bundle |===
156  float mindist = 1000;
157 
158  const auto outerParamZ = std::abs(outerParam.getZ());
159 
160  int bundle = -1;
161  for (size_t i = 0; i < LaserTrack::CoarseBundleZPos.size(); ++i) {
162  const float dist = std::abs(outerParamZ - LaserTrack::CoarseBundleZPos[i]);
163  if (dist < mindist) {
164  mindist = dist;
165  bundle = i;
166  }
167  }
168 
169  if (bundle < 0) {
170  return -1;
171  }
172 
173  // ===| laser beam |===
174  const auto outerParamsInBundle = mLaserTracks.getTracksInBundle(side, rod, bundle);
175  mindist = 1000;
176 
177  int beam = -1;
178  for (int i = 0; i < outerParamsInBundle.size(); ++i) {
179  const auto louterParam = outerParamsInBundle[i];
180  if (i == 0) {
181  outerParam.propagateParamTo(louterParam.getX(), mBz);
182  }
183  const float dist = std::abs(outerParam.getSnp() - louterParam.getSnp());
184  if (dist < mindist) {
185  mindist = dist;
186  beam = i;
187  }
188  }
189 
190  if (mindist > 0.01) {
191  return -1;
192  }
193 
194  // ===| track ID from side, rod, bundle and beam |===
195  const int trackID = LaserTrack::NumberOfTracks / 2 * side +
196  LaserTrack::BundlesPerRod * LaserTrack::TracksPerBundle * rod +
197  LaserTrack::TracksPerBundle * bundle +
198  beam;
199 
200  return trackID;
201 }

204 float CalibLaserTracks::getPhiNearbyLaserRod(const TrackPar& param, int side)
205 {
206  const auto xyzGlo = param.getXYZGlo();
207  const auto phiTrack = o2::math_utils::to02PiGen(std::atan2(xyzGlo.Y(), xyzGlo.X()) - LaserTrack::FirstRodPhi[side % 2]);
208  const auto phiRod = o2::math_utils::to02PiGen(std::nearbyint(phiTrack / LaserTrack::RodDistancePhi) * LaserTrack::RodDistancePhi + LaserTrack::FirstRodPhi[side % 2]);
209 
210  return phiRod;
211 }

214 bool CalibLaserTracks::hasNearbyLaserRod(const TrackPar& param, int side)
215 {
216  const auto xyzGlo = param.getXYZGlo();
217  const auto phiTrack = o2::math_utils::to02PiGen(std::atan2(xyzGlo.Y(), xyzGlo.X()) - LaserTrack::FirstRodPhi[side % 2]);
218  const auto phiRod = o2::math_utils::to02PiGen(std::nearbyint(phiTrack / LaserTrack::RodDistancePhi) * LaserTrack::RodDistancePhi);
219 
220  return std::abs(phiRod - phiTrack) < LaserTrack::SectorSpanRad / 4.;
221 }

233 void CalibLaserTracks::merge(const CalibLaserTracks* other)
234 {
235  if (!other) {
236  return;
237  }
238  mCalibData.processedTFs += other->mCalibData.processedTFs;
239 
240  const auto sizeAthis = mZmatchPairsA.size();
241  const auto sizeCthis = mZmatchPairsC.size();
242  const auto sizeAother = other->mZmatchPairsA.size();
243  const auto sizeCother = other->mZmatchPairsC.size();
244 
245  mZmatchPairsA.insert(mZmatchPairsA.end(), other->mZmatchPairsA.begin(), other->mZmatchPairsA.end());
246  mZmatchPairsC.insert(mZmatchPairsC.end(), other->mZmatchPairsC.begin(), other->mZmatchPairsC.end());
247 
248  auto& ltrIDs = mCalibData.matchedLtrIDs;
249  auto& ltrIDsOther = other->mCalibData.matchedLtrIDs;
250  ltrIDs.insert(ltrIDs.end(), ltrIDsOther.begin(), ltrIDsOther.end());
251 
252  mCalibData.firstTime = std::min(mCalibData.firstTime, other->mCalibData.firstTime);
253  mCalibData.lastTime = std::max(mCalibData.lastTime, other->mCalibData.lastTime);
254 
255  sort(mZmatchPairsA);
256  sort(mZmatchPairsC);
257 
258  LOGP(info, "Merged CalibLaserTracks with mached pairs {} / {} + {} / {} = {} / {} (this +_other A- / C-Side)", sizeAthis, sizeCthis, sizeAother, sizeCother, mZmatchPairsA.size(), mZmatchPairsC.size());
259 }

318 TimePair CalibLaserTracks::fit(const std::vector<TimePair>& trackMatches) const
319 {
320  if (!trackMatches.size()) {
321  return TimePair();
322  }
323 
324  static TLinearFitter fit(2, "pol1");
325  fit.StoreData(false);
326  fit.ClearPoints();
327 
328  uint64_t meanTime = 0;
329  for (const auto& point : trackMatches) {
330  double x = point.x1;
331  double y = point.x2;
332  fit.AddPoint(&x, y);
333 
334  meanTime += point.time;
335  }
336 
337  meanTime /= uint64_t(trackMatches.size());
338 
339  const float robustFraction = 0.9;
340  const int minPoints = 6;
341 
342  if (trackMatches.size() < size_t(minPoints / robustFraction)) {
343  return TimePair({0, 0, meanTime});
344  }
345 
346  //fit.EvalRobust(robustFraction);
347  fit.Eval();
348 
349  TimePair retVal;
350  retVal.x1 = float(fit.GetParameter(0));
351  retVal.x2 = float(fit.GetParameter(1));
352  retVal.time = meanTime;
353 
354  return retVal;
355 }
