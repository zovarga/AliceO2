196 //__________________________________________________________
197 void GloFwdAssessment::runBasicQC(o2::framework::ProcessingContext& ctx)
198 {
199 
200  // get tracks
201  mMFTTracks = ctx.inputs().get<gsl::span<o2::mft::TrackMFT>>("mfttracks");
202  mMCHTracks = ctx.inputs().get<gsl::span<o2::mch::TrackMCH>>("mchtracks");
203  mGlobalFwdTracks = ctx.inputs().get<gsl::span<o2::dataformats::GlobalFwdTrack>>("fwdtracks");
204 
205  if (mUseMC) {
206  // get labels
207  mMFTTrackLabels = ctx.inputs().get<gsl::span<MCCompLabel>>("mfttrklabels");
208  mMCHTrackLabels = ctx.inputs().get<gsl::span<MCCompLabel>>("mchtrklabels");
209  mFwdTrackLabels = ctx.inputs().get<gsl::span<MCCompLabel>>("fwdtrklabels");
210  }
211 
212  for (auto& oneTrack : mGlobalFwdTracks) {
213  if (mMIDFilterEnabled and (oneTrack.getMIDMatchingChi2() < 0)) { // MID filter
214  continue;
215  }
216  const auto nClusters = mMFTTracks[oneTrack.getMFTTrackID()].getNumberOfPoints();
217  mTrackNumberOfClusters->Fill(nClusters);
218  mTrackInvQPt->Fill(oneTrack.getInvQPt());
219  mTrackChi2->Fill(oneTrack.getTrackChi2());
220  mTrackCharge->Fill(oneTrack.getCharge());
221  mTrackPhi->Fill(oneTrack.getPhi());
222  mTrackEta->Fill(oneTrack.getEta());
223  mTrackTanl->Fill(oneTrack.getTanl());
224 
225  for (auto minNClusters : sMinNClustersList) {
226  if (nClusters >= minNClusters) {
227  mTrackEtaNCls[minNClusters - sMinNClustersList[0]]->Fill(oneTrack.getEta());
228  mTrackPhiNCls[minNClusters - sMinNClustersList[0]]->Fill(oneTrack.getPhi());
229  mTrackXYNCls[minNClusters - sMinNClustersList[0]]->Fill(oneTrack.getX(), oneTrack.getY());
230  mTrackEtaPhiNCls[minNClusters - sMinNClustersList[0]]->Fill(oneTrack.getEta(), oneTrack.getPhi());
231  }
232  }
233  }
234 }

238 {

247  } // events

251 //__________________________________________________________
252 void GloFwdAssessment::processPairables()
253 {
254  int trackID = 0, evnID = 0, srcID = 0;
255  bool fake = false;
256  std::unordered_map<o2::MCCompLabel, std::array<bool, 2>> mcPairables;
257 
258  // Loop MCH Tracks
259  auto nMCHTracks = mMCHTracks.size();
260  for (int iTrk = 0; iTrk < nMCHTracks; ++iTrk) {
261  auto mchLabel = mMCHTrackLabels[iTrk];
262 
263  mchLabel.get(trackID, evnID, srcID, fake);
264  if (fake) {
265  continue;
266  }
267  mcPairables[mchLabel][0] = true;
268  }
269 
270  // Loop MFT Tracks
271  auto nMFTTracks = mMFTTracks.size();
272  for (int iTrk = 0; iTrk < nMFTTracks; ++iTrk) {
273  auto mftLabel = mMFTTrackLabels[iTrk];
274 
275  mftLabel.get(trackID, evnID, srcID, fake);
276  if (fake) {
277  continue;
278  }
279  auto t = mcPairables.find(mftLabel);
280  if (t != mcPairables.end()) {
281  t->second[1] = true;
282  }
283  }
284 
285  // Identify and process pairables
286  for (auto& testPair : mcPairables) {
287  auto& boolPair = testPair.second;
288  if (boolPair[0] and boolPair[1]) {
289  mPairables[testPair.first] = true;
290  auto const* mcParticle = mcReader.getTrack(testPair.first);
291  srcID = testPair.first.getSourceID();
292  evnID = testPair.first.getEventID();
293  auto evH = mcReader.getMCEventHeader(srcID, evnID);
294  addMCParticletoHistos(mcParticle, kPairable, evH);
295  }
296  }
297 }

299 //__________________________________________________________
300 void GloFwdAssessment::processRecoAndTrueTracks()
301 {
302 
303  auto trkId = 0;
304  for (auto fwdTrack : mGlobalFwdTracks) {
305  if (mMIDFilterEnabled and (fwdTrack.getMIDMatchingChi2() < 0)) { // MID filter
306  trkId++;
307  continue;
308  }
309  auto pt_Rec = fwdTrack.getPt();
310  auto invQPt_Rec = fwdTrack.getInvQPt();
311  auto px_mch = mMCHTracks[fwdTrack.getMCHTrackID()].getPx();
312  auto py_mch = mMCHTracks[fwdTrack.getMCHTrackID()].getPy();
313  auto invQPt_MCH = mMCHTracks[fwdTrack.getMCHTrackID()].getSign() / sqrt(px_mch * px_mch + py_mch * py_mch);
314  auto eta_Rec = std::abs(fwdTrack.getEta());
315  auto phi_Rec = fwdTrack.getPhi();
316  auto nMFTClusters = mMFTTracks[fwdTrack.getMFTTrackID()].getNumberOfPoints();
317  auto Chi2_Rec = fwdTrack.getTrackChi2();
318  int Q_Rec = fwdTrack.getCharge();
319  auto matchChi2 = fwdTrack.getMFTMCHMatchingChi2();
320  auto matchMatchScore = fwdTrack.getMFTMCHMatchingScore();
321 
322  auto TrackType = kReco;
323  mHistPtVsEta[TrackType]->Fill(eta_Rec, pt_Rec);
324  mHistPhiVsEta[TrackType]->Fill(eta_Rec, phi_Rec);
325  mHistPhiVsPt[TrackType]->Fill(pt_Rec, phi_Rec);
326 
327  mTH3Histos[kTH3GMTrackPtEtaChi2]->Fill(pt_Rec, eta_Rec, matchChi2);
328  mTH3Histos[kTH3GMTrackPtEtaMatchScore]->Fill(pt_Rec, eta_Rec, matchMatchScore);
329  if (fwdTrack.isCloseMatch()) {
330  mTH3Histos[kTH3GMCloseMatchPtEtaChi2]->Fill(pt_Rec, eta_Rec, matchChi2);
331  mTH3Histos[kTH3GMCloseMatchPtEtaMatchScore]->Fill(pt_Rec, eta_Rec, matchMatchScore);
332  }
333 
334  auto trackLabel = mFwdTrackLabels[trkId];
335  if (trackLabel.isCorrect()) {
336  TrackType = kRecoTrue;
337  auto evH = mcReader.getMCEventHeader(trackLabel.getSourceID(), trackLabel.getEventID());
338  auto zVtx = evH.GetZ();
339 
340  auto const* mcParticle = mcReader.getTrack(trackLabel);
341  auto etaGen = std::abs(mcParticle->GetEta());
342  auto phiGen = TMath::ATan2(mcParticle->Py(), mcParticle->Px());
343  auto ptGen = mcParticle->GetPt();
344  auto vxGen = mcParticle->GetStartVertexCoordinatesX();
345  auto vyGen = mcParticle->GetStartVertexCoordinatesY();
346  auto vzGen = mcParticle->GetStartVertexCoordinatesZ();
347  auto tanlGen = mcParticle->Pz() / mcParticle->GetPt();
348 
349  auto pdgcode_MC = mcParticle->GetPdgCode();
350  int Q_Gen;
351  if (TDatabasePDG::Instance()->GetParticle(pdgcode_MC)) {
352  Q_Gen = TDatabasePDG::Instance()->GetParticle(pdgcode_MC)->Charge() / 3;
353  } else {
354  continue;
355  }
356  auto invQPtGen = 1.0 * Q_Gen / ptGen;
357  fwdTrack.propagateToZ(vzGen, mBz);
358 
359  // Residuals at vertex
360  auto x_res = fwdTrack.getX() - vxGen;
361  auto y_res = fwdTrack.getY() - vyGen;
362  auto eta_res = fwdTrack.getEta() - etaGen;
363  auto phi_res = fwdTrack.getPhi() - phiGen;
364  auto tanl_res = fwdTrack.getTanl() - tanlGen;
365  auto invQPt_res = invQPt_Rec - invQPtGen;
366  mHistPtVsEta[TrackType]->Fill(eta_Rec, pt_Rec);
367  mHistPhiVsEta[TrackType]->Fill(eta_Rec, phi_Rec);
368  mHistPhiVsPt[TrackType]->Fill(pt_Rec, phi_Rec);
369  mHistZvtxVsEta[TrackType]->Fill(eta_Rec, zVtx);
370 
371  mHistPhiRecVsPhiGen->Fill(phiGen, phi_Rec);
372  mHistEtaRecVsEtaGen->Fill(etaGen, eta_Rec);
373 
375  auto d_Charge = Q_Rec - Q_Gen;
376  mChargeMatchEff->Fill(!d_Charge, ptGen);
377 
378  mTH3Histos[kTH3GMTrackDeltaXVertexPtEta]->Fill(ptGen, etaGen, 1e4 * x_res);
379  mTH3Histos[kTH3GMTrackDeltaYVertexPtEta]->Fill(ptGen, etaGen, 1e4 * y_res);
380  mTH3Histos[kTH3GMTrackDeltaXDeltaYEta]->Fill(etaGen, 1e4 * x_res, 1e4 * y_res);
381  mTH3Histos[kTH3GMTrackDeltaXDeltaYPt]->Fill(ptGen, 1e4 * x_res, 1e4 * y_res);
382  mTH3Histos[kTH3GMTrackXPullPtEta]->Fill(ptGen, etaGen, x_res / sqrt(fwdTrack.getCovariances()(0, 0)));
383  mTH3Histos[kTH3GMTrackYPullPtEta]->Fill(ptGen, etaGen, y_res / sqrt(fwdTrack.getCovariances()(1, 1)));
384  mTH3Histos[kTH3GMTrackPhiPullPtEta]->Fill(ptGen, etaGen, phi_res / sqrt(fwdTrack.getCovariances()(2, 2)));
385  mTH3Histos[kTH3GMTrackTanlPullPtEta]->Fill(ptGen, etaGen, tanl_res / sqrt(fwdTrack.getCovariances()(3, 3)));
386  mTH3Histos[kTH3GMTrackInvQPtPullPtEta]->Fill(ptGen, etaGen, invQPt_res / sqrt(fwdTrack.getCovariances()(4, 4)));
387  mTH3Histos[kTH3GMTrackInvQPtResolutionPtEta]->Fill(ptGen, etaGen, (invQPt_Rec - invQPtGen) / invQPtGen);
388  mTH3Histos[kTH3GMTrackInvQPtResMCHPtEta]->Fill(ptGen, etaGen, (invQPt_MCH - invQPtGen) / invQPtGen);
389  mTH3Histos[kTH3GMTrackReducedChi2PtEta]->Fill(ptGen, etaGen, Chi2_Rec / (2 * nMFTClusters - 5));
390  mTH3Histos[kTH3GMTruePtEtaChi2]->Fill(pt_Rec, eta_Rec, matchChi2);
391  mTH3Histos[kTH3GMTruePtEtaMatchScore]->Fill(pt_Rec, eta_Rec, matchMatchScore);
392  }
393  trkId++;
394  }
395 }

397 //__________________________________________________________
398 void GloFwdAssessment::addMCParticletoHistos(const MCTrack* mcTr, const int TrackType, const o2::dataformats::MCEventHeader& evH)
399 {
400  auto zVtx = evH.GetZ();
401 
402  auto pt = mcTr->GetPt();
403  auto eta = -1 * mcTr->GetEta();
404  auto phi = mcTr->GetPhi();
405  o2::math_utils::bringToPMPiGend(phi);
406  auto z = mcTr->GetStartVertexCoordinatesZ();
407  auto R = sqrt(pow(mcTr->GetStartVertexCoordinatesX(), 2) + pow(mcTr->GetStartVertexCoordinatesY(), 2));
408 
409  mHistPtVsEta[TrackType]->Fill(eta, pt);
410  mHistPhiVsEta[TrackType]->Fill(eta, phi);
411  mHistPhiVsPt[TrackType]->Fill(pt, phi);
412  mHistZvtxVsEta[TrackType]->Fill(eta, zVtx);
413  if (TrackType == kGen || TrackType == kPairable) {
414  mHistRVsZ[TrackType]->Fill(z, R);
415  }
416  if (TrackType == kPairable) {
417  mTH3Histos[kTH3GMPairablePtEtaZ]->Fill(pt, eta, zVtx);
418  }
419 }

422 void GloFwdAssessment::getHistos(TObjArray& objar)
423 {

425  objar.Add(mTrackNumberOfClusters.get());
426  objar.Add(mTrackInvQPt.get());
427  objar.Add(mTrackChi2.get());
428  objar.Add(mTrackCharge.get());

430  objar.Add(mTrackEta.get());
