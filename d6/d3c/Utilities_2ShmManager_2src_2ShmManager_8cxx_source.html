66  void AddSegments(const vector<string>& _segments, bool zero)
67  {
68  for (const auto& s : _segments) {
69  vector<string> conf;
70  boost::algorithm::split(conf, s, boost::algorithm::is_any_of(","));
71  if (conf.size() != 3) {
72  LOG(error) << "incorrect format for --segments. Expecting pairs of <id>,<size><numaid>.";
73  fair::mq::shmem::Monitor::Cleanup(fair::mq::shmem::ShmId{shmId});
74  throw runtime_error("incorrect format for --segments. Expecting pairs of <id>,<size>,<numaid>.");
75  }
76  uint16_t id = stoi(conf.at(0));
77  uint64_t size = stoull(conf.at(1));
78  segmentCfgs.emplace_back(fair::mq::shmem::SegmentConfig{id, size, "rbtree_best_fit"});
79 
80 #if !defined(__MACH__) && !defined(__APPLE__)
81  int numaid = stoi(conf.at(2));
82  if (numaid == -2) {
83  LOG(info) << "Setting memory allocation to process default";
84  syscall(SYS_set_mempolicy, MPOL_DEFAULT, nullptr, 0);
85  } else if (numaid == -1) {
86  LOG(info) << "Setting memory allocation to NUMA interleaving";
87  unsigned long nodemask = 0xffffff;
88  syscall(SYS_set_mempolicy, MPOL_INTERLEAVE, &nodemask, sizeof(nodemask) * 8);
89  } else {
90  LOG(info) << "Setting memory allocation to NUMA id " << numaid;
91  unsigned long nodemask = 1 << numaid;
92  syscall(SYS_set_mempolicy, MPOL_BIND, &nodemask, sizeof(nodemask) * 8);
93  }
94 #endif
95 
96  auto ret = segments.emplace(id, fair::mq::shmem::Segment(shmId, id, size, fair::mq::shmem::rbTreeBestFit));
97  fair::mq::shmem::Segment& segment = ret.first->second;
98  LOG(info) << "Created segment " << id << " of size " << segment.GetSize() << ", starting at " << segment.GetData() << ". Locking...";
99  segment.Lock();
100  LOG(info) << "Done.";
101  if (zero) {
102  LOG(info) << "Zeroing...";
103  segment.Zero();
104  LOG(info) << "Done.";
105  }
106  }
107  }

109  void AddRegions(const vector<string>& _regions, bool zero)
110  {
111  for (const auto& r : _regions) {
112  vector<string> conf;
113  boost::algorithm::split(conf, r, boost::algorithm::is_any_of(","));
114  if (conf.size() != 3) {
115  LOG(error) << "incorrect format for --regions. Expecting pairs of <id>,<size>,<numaid>.";
116  fair::mq::shmem::Monitor::Cleanup(fair::mq::shmem::ShmId{shmId});
117  throw runtime_error("incorrect format for --regions. Expecting pairs of <id>,<size>,<numaid>.");
118  }
119  uint16_t id = stoi(conf.at(0));
120  uint64_t size = stoull(conf.at(1));
121  fair::mq::RegionConfig cfg;
122  cfg.id = id;
123  cfg.size = size;
124  regionCfgs.push_back(cfg);
125 
126 #if !defined(__MACH__) && !defined(__APPLE__)
127  int numaid = stoi(conf.at(2));
128  if (numaid == -2) {
129  LOG(info) << "Setting memory allocation to process default";
130  syscall(SYS_set_mempolicy, MPOL_DEFAULT, nullptr, 0);
131  } else if (numaid == -1) {
132  LOG(info) << "Setting memory allocation to NUMA interleaving";
133  unsigned long nodemask = 0xffffff;
134  syscall(SYS_set_mempolicy, MPOL_INTERLEAVE, &nodemask, sizeof(nodemask) * 8);
135  } else {
136  LOG(info) << "Setting memory allocation to NUMA id " << numaid;
137  unsigned long nodemask = 1 << numaid;
138  syscall(SYS_set_mempolicy, MPOL_BIND, &nodemask, sizeof(nodemask) * 8);
139  }
140 #endif
141 
142  auto ret = regions.emplace(id, make_unique<fair::mq::shmem::UnmanagedRegion>(shmId, id, size));
143  fair::mq::shmem::UnmanagedRegion& region = *(ret.first->second);
144  LOG(info) << "Created unamanged region " << id << " of size " << region.GetSize() << ", starting at " << region.GetData() << ". Locking...";
145  region.Lock();
146  LOG(info) << "Done.";
147  if (zero) {
148  LOG(info) << "Zeroing...";
149  region.Zero();
150  LOG(info) << "Done.";
151  }
152  }
153  }
