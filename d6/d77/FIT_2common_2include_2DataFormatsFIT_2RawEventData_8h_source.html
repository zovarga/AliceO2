57 struct EventData {
58  static constexpr size_t PayloadSize = 5;
59  static constexpr size_t PayloadPerGBTword = 10;
60  static constexpr size_t MinNelements = 0;
61  static constexpr size_t MaxNelements = 12;
62  //
63  static constexpr int BitFlagPos = 25; // position of first bit flag(numberADC)
64 
65  union {
66  uint64_t word = {0};
67  struct {
68  int64_t time : 12;
69  int64_t charge : 13;
70  uint64_t numberADC : 1, //25 bit
71  isDoubleEvent : 1,
72  isTimeInfoNOTvalid : 1,
73  isCFDinADCgate : 1,
74  isTimeInfoLate : 1,
75  isAmpHigh : 1,
76  isEventInTVDC : 1,
77  isTimeInfoLost : 1,
78  reservedField : 3,
79  channelID : 4;
80  };
81  };
82  void generateFlags()
83  {
84  numberADC = std::rand() % 2;
85  isDoubleEvent = 0;
86  isTimeInfoNOTvalid = 0;
87  isCFDinADCgate = 1;
88  isTimeInfoLate = 0;
89  isAmpHigh = 0;
90  isEventInTVDC = 1;
91  isTimeInfoLost = 0;
92  }
93  uint8_t getFlagWord() const
94  {
95  return uint8_t(word >> BitFlagPos);
96  }
97  void print() const;
