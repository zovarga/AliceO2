795 template <unsigned int D, unsigned int N>
796 template <class T>
797 GPUdi() void Inverter<D, N>::InvertBunchKaufman(MatRepSymGPU<T, D>& rhs, int& ifail)
798 {
799  typedef T value_type;
800  int i, j, k, s;
801  int pivrow;
802  const int nrow = MatRepSymGPU<T, D>::kRows;
803 
804  SVectorGPU<T, MatRepSymGPU<T, D>::kRows> xvec;
805  SVectorGPU<int, MatRepSymGPU<T, D>::kRows> pivv;
806 
807  typedef int* pivIter;
808  typedef T* mIter;
809 
810  mIter x = xvec.begin();
811  // x[i] is used as helper storage, needs to have at least size nrow.
812  pivIter piv = pivv.begin();
813  // piv[i] is used to store details of exchanges
814 
815  value_type temp1, temp2;
816  mIter ip, mjj, iq;
817  value_type lambda, sigma;
818  const value_type alpha = .6404; // = (1+sqrt(17))/8
819  // LM (04/2009) remove this useless check (it is not in LAPACK) which fails inversion of
820  // a matrix with values < epsilon in the diagonal
821  //
822  // const double epsilon = 32*std::numeric_limits<T>::epsilon();
823  // whenever a sum of two doubles is below or equal to epsilon
824  // it is set to zero.
825  // this constant could be set to zero but then the algorithm
826  // doesn't neccessarily detect that a matrix is singular
827 
828  for (i = 0; i < nrow; i++) {
829  piv[i] = i + 1;
830  }
831 
832  ifail = 0;
833 
834  // compute the factorization P*A*P^T = L * D * L^T
835  // L is unit lower triangular, D is direct sum of 1x1 and 2x2 matrices
836  // L and D^-1 are stored in A = *this, P is stored in piv[]
837 
838  for (j = 1; j < nrow; j += s) // main loop over columns
839  {
840  mjj = rhs.Array() + j * (j - 1) / 2 + j - 1;
841  lambda = 0; // compute lambda = max of A(j+1:n,j)
842  pivrow = j + 1;
843  ip = rhs.Array() + (j + 1) * j / 2 + j - 1;
844  for (i = j + 1; i <= nrow; ip += i++) {
845  if (o2::gpu::GPUCommonMath::Abs(*ip) > lambda) {
846  lambda = o2::gpu::GPUCommonMath::Abs(*ip);
847  pivrow = i;
848  }
849  }
850 
851  if (lambda == 0) {
852  if (*mjj == 0) {
853  ifail = 1;
854  return;
855  }
856  s = 1;
857  *mjj = 1.0f / *mjj;
858  } else {
859  if (o2::gpu::GPUCommonMath::Abs(*mjj) >= lambda * alpha) {
860  s = 1;
861  pivrow = j;
862  } else {
863  sigma = 0; // compute sigma = max A(pivrow, j:pivrow-1)
864  ip = rhs.Array() + pivrow * (pivrow - 1) / 2 + j - 1;
865  for (k = j; k < pivrow; k++) {
866  if (o2::gpu::GPUCommonMath::Abs(*ip) > sigma) {
867  sigma = o2::gpu::GPUCommonMath::Abs(*ip);
868  }
869  ip++;
870  }
871  // sigma cannot be zero because it is at least lambda which is not zero
872  if (o2::gpu::GPUCommonMath::Abs(*mjj) >= alpha * lambda * (lambda / sigma)) {
873  s = 1;
874  pivrow = j;
875  } else if (o2::gpu::GPUCommonMath::Abs(*(rhs.Array() + pivrow * (pivrow - 1) / 2 + pivrow - 1)) >= alpha * sigma) {
876  s = 1;
877  } else {
878  s = 2;
879  }
880  }
881  if (pivrow == j) // no permutation neccessary
882  {
883  piv[j - 1] = pivrow;
884  if (*mjj == 0) {
885  ifail = 1;
886  return;
887  }
888  temp2 = *mjj = 1.0f / *mjj; // invert D(j,j)
889 
890  // update A(j+1:n, j+1,n)
891  for (i = j + 1; i <= nrow; i++) {
892  temp1 = *(rhs.Array() + i * (i - 1) / 2 + j - 1) * temp2;
893  ip = rhs.Array() + i * (i - 1) / 2 + j;
894  for (k = j + 1; k <= i; k++) {
895  *ip -= static_cast<T>(temp1 * *(rhs.Array() + k * (k - 1) / 2 + j - 1));
896  // if (o2::gpu::GPUCommonMath::Abs(*ip) <= epsilon)
897  // *ip=0;
898  ip++;
899  }
900  }
901  // update L
902  ip = rhs.Array() + (j + 1) * j / 2 + j - 1;
903  for (i = j + 1; i <= nrow; ip += i++) {
904  *ip *= static_cast<T>(temp2);
905  }
906  } else if (s == 1) // 1x1 pivot
907  {
908  piv[j - 1] = pivrow;
909 
910  // interchange rows and columns j and pivrow in
911  // submatrix (j:n,j:n)
912  ip = rhs.Array() + pivrow * (pivrow - 1) / 2 + j;
913  for (i = j + 1; i < pivrow; i++, ip++) {
914  temp1 = *(rhs.Array() + i * (i - 1) / 2 + j - 1);
915  *(rhs.Array() + i * (i - 1) / 2 + j - 1) = *ip;
916  *ip = static_cast<T>(temp1);
917  }
918  temp1 = *mjj;
919  *mjj = *(rhs.Array() + pivrow * (pivrow - 1) / 2 + pivrow - 1);
920  *(rhs.Array() + pivrow * (pivrow - 1) / 2 + pivrow - 1) = static_cast<T>(temp1);
921  ip = rhs.Array() + (pivrow + 1) * pivrow / 2 + j - 1;
922  iq = ip + pivrow - j;
923  for (i = pivrow + 1; i <= nrow; ip += i, iq += i++) {
924  temp1 = *iq;
925  *iq = *ip;
926  *ip = static_cast<T>(temp1);
927  }
928 
929  if (*mjj == 0) {
930  ifail = 1;
931  return;
932  }
933  temp2 = *mjj = 1.0f / *mjj; // invert D(j,j)
934 
935  // update A(j+1:n, j+1:n)
936  for (i = j + 1; i <= nrow; i++) {
937  temp1 = *(rhs.Array() + i * (i - 1) / 2 + j - 1) * temp2;
938  ip = rhs.Array() + i * (i - 1) / 2 + j;
939  for (k = j + 1; k <= i; k++) {
940  *ip -= static_cast<T>(temp1 * *(rhs.Array() + k * (k - 1) / 2 + j - 1));
941  // if (o2::gpu::GPUCommonMath::Abs(*ip) <= epsilon)
942  // *ip=0;
943  ip++;
944  }
945  }
946  // update L
947  ip = rhs.Array() + (j + 1) * j / 2 + j - 1;
948  for (i = j + 1; i <= nrow; ip += i++) {
949  *ip *= static_cast<T>(temp2);
950  }
951  } else // s=2, ie use a 2x2 pivot
952  {
953  piv[j - 1] = -pivrow;
954  piv[j] = 0; // that means this is the second row of a 2x2 pivot
955 
956  if (j + 1 != pivrow) {
957  // interchange rows and columns j+1 and pivrow in
958  // submatrix (j:n,j:n)
959  ip = rhs.Array() + pivrow * (pivrow - 1) / 2 + j + 1;
960  for (i = j + 2; i < pivrow; i++, ip++) {
961  temp1 = *(rhs.Array() + i * (i - 1) / 2 + j);
962  *(rhs.Array() + i * (i - 1) / 2 + j) = *ip;
963  *ip = static_cast<T>(temp1);
964  }
965  temp1 = *(mjj + j + 1);
966  *(mjj + j + 1) =
967  *(rhs.Array() + pivrow * (pivrow - 1) / 2 + pivrow - 1);
968  *(rhs.Array() + pivrow * (pivrow - 1) / 2 + pivrow - 1) = static_cast<T>(temp1);
969  temp1 = *(mjj + j);
970  *(mjj + j) = *(rhs.Array() + pivrow * (pivrow - 1) / 2 + j - 1);
971  *(rhs.Array() + pivrow * (pivrow - 1) / 2 + j - 1) = static_cast<T>(temp1);
972  ip = rhs.Array() + (pivrow + 1) * pivrow / 2 + j;
973  iq = ip + pivrow - (j + 1);
974  for (i = pivrow + 1; i <= nrow; ip += i, iq += i++) {
975  temp1 = *iq;
976  *iq = *ip;
977  *ip = static_cast<T>(temp1);
978  }
979  }
980  // invert D(j:j+1,j:j+1)
981  temp2 = *mjj * *(mjj + j + 1) - *(mjj + j) * *(mjj + j);
982  if (temp2 == 0) {
983  printf("SymMatrix::bunch_invert: error in pivot choice");
984  }

986  // this quotient is guaranteed to exist by the choice
987  // of the pivot

989  *mjj = static_cast<T>(*(mjj + j + 1) * temp2);
990  *(mjj + j + 1) = static_cast<T>(temp1 * temp2);
991  *(mjj + j) = static_cast<T>(-*(mjj + j) * temp2);
992 
993  if (j < nrow - 1) // otherwise do nothing
994  {
995  // update A(j+2:n, j+2:n)
996  for (i = j + 2; i <= nrow; i++) {
997  ip = rhs.Array() + i * (i - 1) / 2 + j - 1;
998  temp1 = *ip * *mjj + *(ip + 1) * *(mjj + j);
999  // if (o2::gpu::GPUCommonMath::Abs(temp1 ) <= epsilon)
1000  // temp1 = 0;
1001  temp2 = *ip * *(mjj + j) + *(ip + 1) * *(mjj + j + 1);
1002  // if (o2::gpu::GPUCommonMath::Abs(temp2 ) <= epsilon)
1003  // temp2 = 0;
1004  for (k = j + 2; k <= i; k++) {
1005  ip = rhs.Array() + i * (i - 1) / 2 + k - 1;
1006  iq = rhs.Array() + k * (k - 1) / 2 + j - 1;
1007  *ip -= static_cast<T>(temp1 * *iq + temp2 * *(iq + 1));
1008  // if (o2::gpu::GPUCommonMath::Abs(*ip) <= epsilon)
1009  // *ip = 0;
1010  }
1011  }
1012  // update L
1013  for (i = j + 2; i <= nrow; i++) {
1014  ip = rhs.Array() + i * (i - 1) / 2 + j - 1;
1015  temp1 = *ip * *mjj + *(ip + 1) * *(mjj + j);
1016  // if (o2::gpu::GPUCommonMath::Abs(temp1) <= epsilon)
1017  // temp1 = 0;
1018  *(ip + 1) = *ip * *(mjj + j) + *(ip + 1) * *(mjj + j + 1);
1019  // if (o2::gpu::GPUCommonMath::Abs(*(ip+1)) <= epsilon)
1020  // *(ip+1) = 0;
1021  *ip = static_cast<T>(temp1);
1022  }
1023  }
1024  }
1025  }
1026  } // end of main loop over columns
1027 
1028  if (j == nrow) // the the last pivot is 1x1
1029  {
1030  mjj = rhs.Array() + j * (j - 1) / 2 + j - 1;
1031  if (*mjj == 0) {
1032  ifail = 1;
1033  return;
1034  } else {
1035  *mjj = 1.0f / *mjj;
1036  }
1037  } // end of last pivot code
1038 
1039  // computing the inverse from the factorization
1040 
1041  for (j = nrow; j >= 1; j -= s) // loop over columns
1042  {
1043  mjj = rhs.Array() + j * (j - 1) / 2 + j - 1;
1044  if (piv[j - 1] > 0) // 1x1 pivot, compute column j of inverse
1045  {
1046  s = 1;
1047  if (j < nrow) {
1048  ip = rhs.Array() + (j + 1) * j / 2 + j - 1;
1049  for (i = 0; i < nrow - j; ip += 1 + j + i++) {
1050  x[i] = *ip;
1051  }
1052  for (i = j + 1; i <= nrow; i++) {
1053  temp2 = 0;

1056  temp2 += *ip++ * x[k];
1057  }

1059  temp2 += *ip * x[k];

1177  value_type s11, s12;

1181  // double epsilon = 8*std::numeric_limits<T>::epsilon();
1182  // could be set to zero (like it was before)
1183  // but then the algorithm often doesn't detect
1184  // that a matrix is singular

1189  jfail = jrange;
1190  int nxch = 0;

1192  mIter mj = rhs.Array();
1193  mIter mjj = mj;
1194  for (unsigned int j = 1; j <= n; j++) {
1195  unsigned int k = j;
1196  p = (o2::gpu::GPUCommonMath::Abs(*mjj));
1197  if (j != n) {
1198  mIter mij = mj + n + j - 1;
1199  for (unsigned int i = j + 1; i <= n; i++) {
1200  q = (o2::gpu::GPUCommonMath::Abs(*(mij)));
1201  if (q > p) {
1202  k = i;
1203  p = q;
1204  }
1205  mij += n;
1206  }

1208  if (p <= epsilon) {
1209  det = 0;
1210  ifail = imposs;
1211  jfail = jrange;
1212  return ifail;
1213  }

1215  // must not change. So I change it twice.
1216  }

1218  mIter mkl = rhs.Array() + (k - 1) * n;

1220  tf = *mjl;
1221  *(mjl++) = *mkl;
1222  *(mkl++) = static_cast<T>(tf);
1223  }

1225  ir[nxch] = (((j) << 12) + (k));
1226  } else {

1228  det = 0.0;
1229  ifail = imposs;
1230  jfail = jrange;
1231  return ifail;
1232  }
1233  }
