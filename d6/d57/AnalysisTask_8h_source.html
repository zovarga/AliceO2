162  template <typename T>
163  static auto extractTableFromRecord(InputRecord& record)
164  {
165  if constexpr (soa::is_type_with_metadata_v<aod::MetadataTrait<T>>) {
166  auto table = record.get<TableConsumer>(aod::MetadataTrait<T>::metadata::tableLabel())->asArrowTable();

170  return table;

187  template <typename T, typename... Os>
188  static auto extractFilteredFromRecord(InputRecord& record, ExpressionInfo& info, pack<Os...> const&)
189  {
190  auto table = o2::soa::ArrowHelpers::joinTables(std::vector<std::shared_ptr<arrow::Table>>{extractTableFromRecord<Os>(record)...});
191  if (info.tree != nullptr && info.filter == nullptr) {
192  info.filter = framework::expressions::createFilter(table->schema(), framework::expressions::makeCondition(info.tree));
193  }
194  if (info.tree != nullptr && info.filter != nullptr && info.resetSelection == true) {
195  info.selection = framework::expressions::createSelection(table, info.filter);
196  info.resetSelection = false;
197  }
198  if constexpr (!framework::is_base_of_template<soa::SmallGroups, std::decay_t<T>>::value) {
199  if (info.selection == nullptr) {
200  throw runtime_error_f("Null selection for %d (arg %d), missing Filter declaration?", info.processHash, info.argumentIndex);
201  }
202  }
203  if constexpr (soa::is_soa_iterator_t<T>::value) {
204  return typename T::parent_t({table}, info.selection);
205  } else {
206  return T({table}, info.selection);
207  }
208  }

210  template <typename T, int AI>
211  static auto extractSomethingFromRecord(InputRecord& record, std::vector<ExpressionInfo>& infos, size_t phash)
212  {
213  using decayed = std::decay_t<T>;
214 
215  if constexpr (soa::is_soa_filtered_t<decayed>::value) {
216  return extractFilteredFromRecord<decayed>(record, *std::find_if(infos.begin(), infos.end(), [&phash](ExpressionInfo const& i) { return (i.processHash == phash && i.argumentIndex == AI); }), soa::make_originals_from_type<decayed>());
217  } else if constexpr (soa::is_soa_iterator_t<decayed>::value) {
218  if constexpr (std::is_same_v<typename decayed::policy_t, soa::FilteredIndexPolicy>) {
219  return extractFilteredFromRecord<decayed>(record, *std::find_if(infos.begin(), infos.end(), [&phash](ExpressionInfo const& i) { return (i.processHash == phash && i.argumentIndex == AI); }), soa::make_originals_from_type<decayed>());
220  } else {
221  return extractFromRecord<decayed>(record, soa::make_originals_from_type<decayed>());
222  }
223  } else {
224  return extractFromRecord<decayed>(record, soa::make_originals_from_type<decayed>());
225  }
226  O2_BUILTIN_UNREACHABLE();
227  }

251  template <typename Task, typename R, typename C, typename Grouping, typename... Associated>
252  static void invokeProcess(Task& task, InputRecord& inputs, R (C::*processingFunction)(Grouping, Associated...), std::vector<ExpressionInfo>& infos)
253  {
254  using G = std::decay_t<Grouping>;
255  auto groupingTable = AnalysisDataProcessorBuilder::bindGroupingTable(inputs, processingFunction, infos);
256 
257  // set filtered tables for partitions with grouping
258  homogeneous_apply_refs([&groupingTable](auto& x) {
259  PartitionManager<std::decay_t<decltype(x)>>::setPartition(x, groupingTable);
260  PartitionManager<std::decay_t<decltype(x)>>::bindInternalIndices(x, &groupingTable);
261  return true;
262  },
263  task);
264 
265  if constexpr (sizeof...(Associated) == 0) {
266  // single argument to process
267  homogeneous_apply_refs([&groupingTable](auto& x) {
268  PartitionManager<std::decay_t<decltype(x)>>::bindExternalIndices(x, &groupingTable);
269  GroupedCombinationManager<std::decay_t<decltype(x)>>::setGroupedCombination(x, groupingTable);
270  return true;
271  },
272  task);
273  if constexpr (soa::is_soa_iterator_t<G>::value) {
274  for (auto& element : groupingTable) {
275  std::invoke(processingFunction, task, *element);
276  }
277  } else {
278  static_assert(soa::is_soa_table_like_t<G>::value,
279  "Single argument of process() should be a table-like or an iterator");
280  std::invoke(processingFunction, task, groupingTable);
281  }
282  } else {
283  // multiple arguments to process
284  static_assert(((soa::is_soa_iterator_t<std::decay_t<Associated>>::value == false) && ...),
285  "Associated arguments of process() should not be iterators");
286  auto associatedTables = AnalysisDataProcessorBuilder::bindAssociatedTables(inputs, processingFunction, infos);
287  //pre-bind self indices
288  std::apply(
289  [&](auto&... t) {
290  (homogeneous_apply_refs(
291  [&](auto& p) {
292  PartitionManager<std::decay_t<decltype(p)>>::bindInternalIndices(p, &t);
293  return true;
294  },
295  task),
296  ...);
297  },
298  associatedTables);
299 
300  auto binder = [&](auto&& x) {
301  x.bindExternalIndices(&groupingTable, &std::get<std::decay_t<Associated>>(associatedTables)...);
302  homogeneous_apply_refs([&x](auto& t) {
303  PartitionManager<std::decay_t<decltype(t)>>::setPartition(t, x);
304  PartitionManager<std::decay_t<decltype(t)>>::bindExternalIndices(t, &x);
305  return true;
306  },
307  task);
308  };
309  groupingTable.bindExternalIndices(&std::get<std::decay_t<Associated>>(associatedTables)...);
310 
311  // always pre-bind full tables to support index hierarchy
312  std::apply(
313  [&](auto&&... x) {
314  (binder(x), ...);
315  },
316  associatedTables);
317 
318  // GroupedCombinations bound separately, as they should be set once for all associated tables
319  auto hashes = std::get<0>(associatedTables);
320  auto realAssociated = tuple_tail(associatedTables);
321  homogeneous_apply_refs([&groupingTable, &hashes, &realAssociated](auto& t) {
322  GroupedCombinationManager<std::decay_t<decltype(t)>>::setGroupedCombination(t, hashes, groupingTable, realAssociated);
323  return true;
324  },
325  task);
326 
327  if constexpr (soa::is_soa_iterator_t<std::decay_t<G>>::value) {
328  // grouping case
329  auto slicer = GroupSlicer(groupingTable, associatedTables);
330  for (auto& slice : slicer) {
331  auto associatedSlices = slice.associatedTables();
332 
333  std::apply(
334  [&](auto&&... x) {
335  (binder(x), ...);
336  },
337  associatedSlices);
338 
339  // bind partitions and grouping table
340  homogeneous_apply_refs([&groupingTable](auto& x) {
341  PartitionManager<std::decay_t<decltype(x)>>::bindExternalIndices(x, &groupingTable);
342  return true;
343  },
344  task);
345 
346  invokeProcessWithArgsGeneric(task, processingFunction, slice.groupingElement(), associatedSlices);
347  }
348  } else {
349  // non-grouping case
350 
351  // bind partitions and grouping table
352  homogeneous_apply_refs([&groupingTable](auto& x) {
353  PartitionManager<std::decay_t<decltype(x)>>::bindExternalIndices(x, &groupingTable);
354  return true;
355  },
356  task);
357 
358  invokeProcessWithArgsGeneric(task, processingFunction, groupingTable, associatedTables);
359  }
360  }
361  }

512 template <typename T, typename... Args>
513 DataProcessorSpec adaptAnalysisTask(ConfigContext const& ctx, Args&&... args)
514 {
515  TH1::AddDirectory(false);
516 
517  auto [name_str, task] = getTaskNameSetProcesses<T>(args...);
518 
519  auto suffix = ctx.options().get<std::string>("workflow-suffix");
520  if (!suffix.empty()) {
521  name_str += suffix;
522  }
523  const char* name = name_str.c_str();
524 
525  auto hash = compile_time_hash(name);
526 
527  std::vector<OutputSpec> outputs;
528  std::vector<InputSpec> inputs;
529  std::vector<ConfigParamSpec> options;
530  std::vector<ExpressionInfo> expressionInfos;
531 
533  homogeneous_apply_refs([&options, &hash](auto& x) { return OptionManager<std::decay_t<decltype(x)>>::appendOption(options, x); }, *task.get());
535  homogeneous_apply_refs([&inputs](auto& x) { return ConditionManager<std::decay_t<decltype(x)>>::appendCondition(inputs, x); }, *task.get());
536 
538  if constexpr (has_process_v<T>) {
539  AnalysisDataProcessorBuilder::inputsFromArgs(&T::process, "default", true, inputs, expressionInfos);
540  }
541  homogeneous_apply_refs(
542  [name = name_str, &expressionInfos, &inputs](auto& x) {
543  using D = std::decay_t<decltype(x)>;
544  if constexpr (is_base_of_template<ProcessConfigurable, D>::value) {
545  // this pushes (argumentIndex,processHash,schemaPtr,nullptr) into expressionInfos for arguments that are Filtered/filtered_iterators
546  AnalysisDataProcessorBuilder::inputsFromArgs(x.process, (name + "/" + x.name).c_str(), x.value, inputs, expressionInfos);
547  return true;
548  }
549  return false;
550  },
551  *task.get());
552 
553  // avoid self-forwarding if process methods subscribe to same tables
554  std::sort(inputs.begin(), inputs.end(), [](InputSpec const& a, InputSpec const& b) { return a.binding < b.binding; });
555  auto last = std::unique(inputs.begin(), inputs.end(), [](InputSpec const& a, InputSpec const& b) { return a.binding == b.binding; });
556  inputs.erase(last, inputs.end());
557 
558  // request base tables for spawnable extended tables
559  // this checks for duplications
560  homogeneous_apply_refs([&inputs](auto& x) {
561  return SpawnManager<std::decay_t<decltype(x)>>::requestInputs(inputs, x);
562  },
563  *task.get());
564 
565  //request base tables for indices to be built
566  homogeneous_apply_refs([&inputs](auto& x) {
567  return IndexManager<std::decay_t<decltype(x)>>::requestInputs(inputs, x);
568  },
569  *task.get());
570 
571  // no static way to check if the task defines any processing, we can only make sure it subscribes to at least something
572  if (inputs.empty() == true) {

574  }

579  homogeneous_apply_refs([&requiredServices](auto& x) { return ServiceManager<std::decay_t<decltype(x)>>::add(requiredServices, x); }, *task.get());

582  homogeneous_apply_refs([&ic](auto&& x) { return OptionManager<std::decay_t<decltype(x)>>::prepare(ic, x); }, *task.get());
583  homogeneous_apply_refs([&ic](auto&& x) { return ServiceManager<std::decay_t<decltype(x)>>::prepare(ic, x); }, *task.get());

589  };

594  [&ic](auto& x) -> bool { return FilterManager<std::decay_t<decltype(x)>>::updatePlaceholders(x, ic); },
595  *task.get());
597  homogeneous_apply_refs(
598  [&ic](auto& x) -> bool { PartitionManager<std::decay_t<decltype(x)>>::updatePlaceholders(x, ic); return true; },
599  *task.get());
601  homogeneous_apply_refs([&expressionInfos](auto& x) {
602  return FilterManager<std::decay_t<decltype(x)>>::createExpressionTrees(x, expressionInfos);
603  },
604  *task.get());
605 
606  if constexpr (has_init_v<T>) {
607  task->init(ic);
608  }
609 
610  return [task, expressionInfos](ProcessingContext& pc) mutable {
611  // load the ccdb object from their cache
612  homogeneous_apply_refs([&pc](auto&& x) { return ConditionManager<std::decay_t<decltype(x)>>::newDataframe(pc.inputs(), x); }, *task.get());
613  // reset partitions once per dataframe
614  homogeneous_apply_refs([](auto&& x) { return PartitionManager<std::decay_t<decltype(x)>>::newDataframe(x); }, *task.get());
615  // reset selections for the next dataframe
616  for (auto& info : expressionInfos) {
617  info.resetSelection = true;
618  }
619  homogeneous_apply_refs([&pc](auto&& x) { return OutputManager<std::decay_t<decltype(x)>>::prepare(pc, x); }, *task.get());
620  if constexpr (has_run_v<T>) {
621  task->run(pc);
622  }
623  if constexpr (has_process_v<T>) {
624  AnalysisDataProcessorBuilder::invokeProcess(*(task.get()), pc.inputs(), &T::process, expressionInfos);
625  }
626  homogeneous_apply_refs(
627  [&pc, &expressionInfos, &task](auto& x) mutable {
628  if constexpr (is_base_of_template<ProcessConfigurable, std::decay_t<decltype(x)>>::value) {
629  if (x.value == true) {
630  AnalysisDataProcessorBuilder::invokeProcess(*task.get(), pc.inputs(), x.process, expressionInfos);
631  return true;
632  }
633  }
634  return false;
635  },
636  *task.get());
637 
638  homogeneous_apply_refs([&pc](auto&& x) { return OutputManager<std::decay_t<decltype(x)>>::finalize(pc, x); }, *task.get());
639  };
640  }};
