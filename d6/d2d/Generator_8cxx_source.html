59 Bool_t
60  Generator::ReadEvent(FairPrimaryGenerator* primGen)
61 {
65  while (true) {
66  mReadEventCounter++;
67 
69  mParticles.clear();
70 
72  if (!generateEvent()) {
73  return kFALSE;
74  }
75 
77  if (!importParticles()) {
78  return kFALSE;
79  }
80 
82  if (triggerEvent()) {
83  mTriggerOkHook(mParticles, mReadEventCounter);
84  break;
85  } else {
86  mTriggerFalseHook(mParticles, mReadEventCounter);
87  }
88  }
89 
91  if (!addTracks(primGen)) {
92  return kFALSE;
93  }
94 
96  auto header = primGen->GetEvent();
97  auto o2header = dynamic_cast<o2::dataformats::MCEventHeader*>(header);
98  if (!header) {
99  LOG(fatal) << "MC event header is not a 'o2::dataformats::MCEventHeader' object";
100  return kFALSE;
101  }
102  updateHeader(o2header);
103 
105  return kTRUE;
106 }

159 Bool_t
160  Generator::triggerEvent()
161 {
165  if (mTriggers.size() == 0 && mDeepTriggers.size() == 0) {
166  return kTRUE;
167  }
168 
170  Bool_t triggered;
171  if (mTriggerMode == kTriggerOFF) {
172  return kTRUE;
173  } else if (mTriggerMode == kTriggerOR) {
174  triggered = kFALSE;
175  } else if (mTriggerMode == kTriggerAND) {
176  triggered = kTRUE;
177  } else {
178  return kTRUE;
179  }
180 
182  for (const auto& trigger : mTriggers) {
183  auto retval = trigger(mParticles);
184  if (mTriggerMode == kTriggerOR) {
185  triggered |= retval;
186  }
187  if (mTriggerMode == kTriggerAND) {
188  triggered &= retval;
189  }
190  }
191 
193  for (const auto& trigger : mDeepTriggers) {
194  auto retval = trigger(mInterface, mInterfaceName);
195  if (mTriggerMode == kTriggerOR) {
196  triggered |= retval;
197  }
198  if (mTriggerMode == kTriggerAND) {
199  triggered &= retval;
200  }
201  }
202 
204  return triggered;
205 }
