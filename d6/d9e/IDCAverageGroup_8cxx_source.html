36 template <>
37 void o2::tpc::IDCAverageGroup<o2::tpc::IDCAverageGroupCRU>::init()
38 {
39  unsigned int maxValues = 0;
40  for (unsigned int i = 0; i < Mapper::NREGIONS; ++i) {
41  const unsigned int maxGroup = (this->mIDCsGrouped.getGroupRows() + this->mIDCsGrouped.getGroupLastRowsThreshold()) * (this->mIDCsGrouped.getGroupPads() + this->mIDCsGrouped.getGroupLastPadsThreshold() + Mapper::ADDITIONALPADSPERROW[i].back());
42  if (maxGroup > maxValues) {
43  maxValues = maxGroup;
44  }
45  }
46 
47  for (auto& rob : this->mRobustAverage) {
48  rob.reserve(maxValues);
49  }
50 
51  // init weights
52  const float sigmaEdge = 1.f;
53  this->mWeightsPad.reserve(mOverlapPads);
54  for (int i = 0; i < mOverlapPads; ++i) {
55  const float groupPadsHalf = this->mIDCsGrouped.getGroupPads() / 2.f;
56  const float sigmaPad = groupPadsHalf / sigmaEdge; // assume 3-sigma at the edge of the last pad
57  this->mWeightsPad.emplace_back(normal_dist(groupPadsHalf + i, sigmaPad));
58  }
59 
60  this->mWeightsRow.reserve(mOverlapRows);
61  for (int i = 0; i < mOverlapRows; ++i) {
62  const float groupRowsHalf = this->mIDCsGrouped.getGroupRows() / 2.f;
63  const float sigmaRow = groupRowsHalf / sigmaEdge; // assume 3-sigma at the edge of the last pad
64  this->mWeightsRow.emplace_back(normal_dist(groupRowsHalf + i, sigmaRow));
65  }
66 }

68 template <>
69 void o2::tpc::IDCAverageGroup<o2::tpc::IDCAverageGroupTPC>::init()
70 {
71  unsigned int maxValues = 0;
72  for (unsigned int i = 0; i < Mapper::NREGIONS; ++i) {
73  const unsigned int maxGroup = (this->mIDCGroupHelperSector.getGroupingParameter().getGroupRows(i) + this->mIDCGroupHelperSector.getGroupingParameter().getGroupLastRowsThreshold(i)) * (this->mIDCGroupHelperSector.getGroupingParameter().getGroupPads(i) + this->mIDCGroupHelperSector.getGroupingParameter().getGroupLastPadsThreshold(i) + Mapper::ADDITIONALPADSPERROW[i].back());
74  if (maxGroup > maxValues) {
75  maxValues = maxGroup;
76  }
77  }
78 
79  for (auto& rob : this->mRobustAverage) {
80  rob.reserve(maxValues);
81  }
82 
83  // init weights
84  for (unsigned int region = 0; region < Mapper::NREGIONS; ++region) {
85  const float sigmaEdge = 1.f;
86  this->mWeightsPad[region].reserve(mOverlapPads);
87  for (unsigned int i = 0; i < mOverlapPads; ++i) {
88  const float groupPadsHalf = this->mIDCGroupHelperSector.getGroupingParameter().getGroupPads(i) / 2.f;
89  const float sigmaPad = groupPadsHalf / sigmaEdge; // assume 3-sigma at the edge of the last pad
90  this->mWeightsPad[i].emplace_back(normal_dist(groupPadsHalf + i, sigmaPad));
91  }
92 
93  this->mWeightsRow[region].reserve(mOverlapRows);
94  for (unsigned int i = 0; i < mOverlapRows; ++i) {
95  const float groupRowsHalf = this->mIDCGroupHelperSector.getGroupingParameter().getGroupRows(i) / 2.f;
96  const float sigmaRow = groupRowsHalf / sigmaEdge; // assume 3-sigma at the edge of the last pad
97  this->mWeightsRow[i].emplace_back(normal_dist(groupRowsHalf + i, sigmaRow));
98  }
99  }

121 template <class Type>
122 float o2::tpc::IDCAverageGroup<Type>::normal_dist(const float x, const float sigma)
123 {
124  const float fac = x / sigma;
125  return std::exp(-fac * fac / 2);

144  for (int thread = 0; thread < sNThreads; ++thread) {
145  idcStruct[thread].setThreadNum(thread);
146  }
147 
148 #pragma omp parallel for num_threads(sNThreads)
149  for (unsigned int iCRU = 0; iCRU < CRU::MaxCRU; ++iCRU) {
150  const unsigned int threadNum = omp_get_thread_num();
151  const CRU cru(iCRU);

154  idcStruct[threadNum].setIntegrationInterval(integrationInterval);
155  loopOverGroups(idcStruct[threadNum]);
156  }

273 template <class Type>
274 template <class LoopType>
275 void o2::tpc::IDCAverageGroup<Type>::loopOverGroups(IDCAverageGroupHelper<LoopType>& idcStruct)
276 {
277  const unsigned int region = idcStruct.getRegion();
278  const int groupRows = idcStruct.getGroupRows();
279  const int groupPads = idcStruct.getGroupPads();
280  const int lastRow = idcStruct.getLastRow();
281  const int groupPadsSectorEdges = idcStruct.getTotalGroupPadsSectorEdges();
282  unsigned int rowGrouped = 0;
283 
284  if (groupPadsSectorEdges) {
285  const auto groupingType = idcStruct.getEdgePadGroupingType();
286  const bool groupRowsEdge = groupingType == EdgePadGroupingMethod::ROWS;
287  const int groupedPads = idcStruct.getGroupedPadsSectorEdges();
288  const int endrow = groupRowsEdge ? lastRow + 1 : Mapper::ROWSPERREGION[region];
289  const int stepRow = groupRowsEdge ? groupRows : 1;
290  for (int ulrow = 0; ulrow < endrow; ulrow += stepRow) {
291  const bool bNotLastrow = ulrow != lastRow;
292 
293  if constexpr (std::is_same_v<LoopType, IDCAverageGroupDraw>) {
294  idcStruct.mCol = ulrow / stepRow;
295  }
296 
297  for (int iYLocalSide = 0; iYLocalSide < 2; ++iYLocalSide) {
298  int pad = 0;
299  for (int padGroup = 0; padGroup < groupedPads; ++padGroup) {
300  const int nPadsPerGroup = idcStruct.getPadsInGroupSectorEdges(padGroup);
301  for (int padInGroup = 0; padInGroup < nPadsPerGroup; ++padInGroup) {
302  const int endRow = (groupRowsEdge && (ulrow + stepRow >= Mapper::ROWSPERREGION[region] || !bNotLastrow)) ? (Mapper::ROWSPERREGION[region] - ulrow) : stepRow; // last row in this group
303  for (int iRowMerge = 0; iRowMerge < endRow; ++iRowMerge) {
304  const int irow = ulrow + iRowMerge;
305  const int ungroupedPad = !iYLocalSide ? pad : Mapper::PADSPERROW[region][irow] - pad - 1;
306  const int padInRegion = Mapper::OFFSETCRULOCAL[region][irow] + ungroupedPad;
307 
308  if constexpr (std::is_same_v<LoopType, IDCAverageGroupCRU> || std::is_same_v<LoopType, IDCAverageGroupTPC>) {
309  const auto flag = mPadStatus->getCalArray(idcStruct.getCRU()).getValue(padInRegion);
310  if (flag == PadFlags::flagDeadPad) {
311  continue;
312  }
313  idcStruct.addValue(padInRegion, 1);
314  } else {
315  const GlobalPadNumber padNum = o2::tpc::Mapper::getGlobalPadNumber(irow, ungroupedPad, region);
316  drawLatex(idcStruct, padNum, padInRegion, true);
317  }
318  }
319  ++pad;
320  }
321  if constexpr (std::is_same_v<LoopType, IDCAverageGroupCRU> || std::is_same_v<LoopType, IDCAverageGroupTPC>) {
322  const auto groupedIDCVal = idcStruct.getGroupedIDC();
323  const int ungroupedPad = !iYLocalSide ? pad - 1 : Mapper::PADSPERROW[region][ulrow] - pad;
324  idcStruct.setSectorEdgeIDC(ulrow, ungroupedPad);
325  idcStruct.clearRobustAverage();
326  } else {
327  ++idcStruct.mGroupCounter;
328  ++idcStruct.mCol;
329  }
330  }
331  }
332  }
333  }
334 
335  // loop over ungrouped row
336  for (int iRow = 0; iRow <= lastRow; iRow += groupRows) {
337  const bool bNotLastrow = iRow != lastRow;
338 
339  // the sectors is divide in to two parts around ylocal=0 to get the same simmetric grouping around ylocal=0
340  for (int iYLocalSide = 0; iYLocalSide < 2; ++iYLocalSide) {
341  if constexpr (std::is_same_v<LoopType, IDCAverageGroupDraw>) {
342  idcStruct.mCol = region + iRow / groupRows + iYLocalSide;
343  }
344  unsigned int padGrouped = iYLocalSide ? idcStruct.getPadsPerRow(rowGrouped) / 2 : idcStruct.getPadsPerRow(rowGrouped) / 2 - 1; // grouped pad in pad direction
345  const int nPadsStart = Mapper::PADSPERROW[region][iRow] / 2; // first ungrouped pad in pad direction
346  const int nPadsEnd = idcStruct.getLastPad(iRow) + nPadsStart; // last grouped pad in pad direction
347 
348  // loop over ungrouped pads
349  for (int iPad = nPadsStart; iPad <= nPadsEnd; iPad += groupPads) {
350  if constexpr (std::is_same_v<LoopType, IDCAverageGroupCRU> || std::is_same_v<LoopType, IDCAverageGroupTPC>) {
351  idcStruct.clearRobustAverage();
352  }
353 
354  const int startRow = ((iRow - mOverlapRows) < 0) ? 0 : -mOverlapRows; // first row in this group
355  const int endRow = ((iRow + groupRows + mOverlapRows) >= Mapper::ROWSPERREGION[region] || !bNotLastrow) ? (Mapper::ROWSPERREGION[region] - iRow) : (mOverlapRows + groupRows); // last row in this group
356  for (int iRowMerge = startRow; iRowMerge < endRow; ++iRowMerge) {
357  const bool bOverlapRowRight = iRowMerge >= groupRows;
358  const unsigned int ungroupedRow = iRow + iRowMerge;
359  const int offsPad = static_cast<int>(Mapper::ADDITIONALPADSPERROW[region][ungroupedRow]) - static_cast<int>(Mapper::ADDITIONALPADSPERROW[region][iRow]); // offset due to additional pads in pad direction in the current row compared to the first row in the group
360 
361  const bool lastPad = iPad == nPadsEnd;
362  const int padEnd = lastPad ? (static_cast<int>(Mapper::PADSPERROW[region][ungroupedRow]) - iPad - groupPadsSectorEdges) : (groupPads + offsPad + mOverlapPads); // last ungrouped pad in pad direction
363  const int padStart = offsPad - mOverlapPads; // first ungrouped pad in pad direction
364 
365  for (int ipadMerge = padStart; ipadMerge < padEnd; ++ipadMerge) {
366  const unsigned int ungroupedPad = iYLocalSide ? (iPad + ipadMerge) : Mapper::PADSPERROW[region][ungroupedRow] - (iPad + ipadMerge) - 1;
367  const unsigned int padInRegion = Mapper::OFFSETCRULOCAL[region][ungroupedRow] + ungroupedPad;
368 
369  // averaging and grouping
370  if constexpr (std::is_same_v<LoopType, IDCAverageGroupCRU> || std::is_same_v<LoopType, IDCAverageGroupTPC>) {
371  // check status flag
372  const auto flag = mPadStatus->getCalArray(idcStruct.getCRU()).getValue(padInRegion);
373 
374  // TODO add more cases...
375  if (flag == PadFlags::flagDeadPad) {
376  // dead pad. just skip this
377  continue;
378  }
379 
380  float weight = 1;
381  // set weight for outer pads which are not in the main group
382  if (mOverlapRows && mOverlapPads) {
383  if (iRowMerge < 0) {
384  // everything on the left border
385  const int relPosRow = std::abs(iRowMerge);
386  if (ipadMerge < offsPad) {
387  const int relPosPad = std::abs(ipadMerge - offsPad);
388  weight = idcStruct.getWeight(relPosRow, relPosPad);
389  } else if (!lastPad && ipadMerge >= (groupPads + offsPad)) {
390  const int relPosPad = std::abs(1 + ipadMerge - (groupPads + offsPad));
391  weight = idcStruct.getWeight(relPosRow, relPosPad);
392  } else {
393  weight = idcStruct.getWeightRow(relPosRow);
394  }
395  } else if (bNotLastrow && bOverlapRowRight) {
396  const int relPosRow = std::abs(1 + iRowMerge - (groupRows));
397  if (ipadMerge < offsPad) {
398  const int relPosPad = std::abs(ipadMerge - offsPad);
399  weight = idcStruct.getWeight(relPosRow, relPosPad);
400  } else if (!lastPad && ipadMerge >= (groupPads + offsPad)) {
401  const int relPosPad = std::abs(1 + ipadMerge - (groupPads + offsPad));
402  weight = idcStruct.getWeight(relPosRow, relPosPad);
403  } else {
404  weight = idcStruct.getWeightRow(relPosRow);
405  }
406  } else if (ipadMerge < offsPad) {
407  // bottom
408  const int relPadPos = std::abs(ipadMerge - offsPad);
409  weight = idcStruct.getWeightPad(relPadPos);
410  } else if (!lastPad && ipadMerge >= (groupPads + offsPad)) {
411  const int relPadPos = std::abs(1 + ipadMerge - (groupPads + offsPad));
412  weight = idcStruct.getWeightPad(relPadPos);
413  } else {
414  }
415  }
416  idcStruct.addValue(padInRegion, weight);
417  } else {
418  // drawing the grouping
419  const GlobalPadNumber padNum = o2::tpc::Mapper::getGlobalPadNumber(ungroupedRow, ungroupedPad, region);
420  const bool fillNotPoly = iRowMerge < 0 || (bNotLastrow && bOverlapRowRight) || (ipadMerge < offsPad) || (!lastPad && ipadMerge >= (groupPads + offsPad));
421  drawLatex(idcStruct, padNum, padInRegion, !fillNotPoly, idcStruct.mColors.size());
422  }
423  }
424  }
425  if constexpr (std::is_same_v<LoopType, IDCAverageGroupCRU> || std::is_same_v<LoopType, IDCAverageGroupTPC>) {
426  idcStruct.setGroupedIDC(rowGrouped, padGrouped);
427  } else {
428  ++idcStruct.mGroupCounter;
429  ++idcStruct.mCol;
430  }
431  iYLocalSide ? ++padGrouped : --padGrouped;
432  }
433  }
434  ++rowGrouped;
435  }
436 }

536 template <class Type>
537 void o2::tpc::IDCAverageGroup<Type>::createDebugTree(const IDCAverageGroupHelper<Type>& idcStruct, o2::utils::TreeStreamRedirector& pcstream)
538 {
539  const Mapper& mapper = Mapper::instance();
540  unsigned int cru = idcStruct.getCRU();
541  const CRU cruTmp(cru);
542  unsigned int sector = cruTmp.sector();
543  unsigned int region = idcStruct.getRegion();
544  unsigned int integrationInterval = idcStruct.getIntegrationInterval();
545 
546  const unsigned long padsPerCRU = Mapper::PADSPERREGION[region];
547  std::vector<unsigned int> vRow(padsPerCRU);
548  std::vector<unsigned int> vPad(padsPerCRU);
549  std::vector<float> vXPos(padsPerCRU);
550  std::vector<float> vYPos(padsPerCRU);
551  std::vector<float> vGlobalXPos(padsPerCRU);
552  std::vector<float> vGlobalYPos(padsPerCRU);
553  std::vector<float> idcsPerIntegrationInterval(padsPerCRU); // idcs for one time bin
554  std::vector<float> groupedidcsPerIntegrationInterval(padsPerCRU); // idcs for one time bin
555  std::vector<float> invPadArea(padsPerCRU);
556 
557  for (unsigned int iPad = 0; iPad < padsPerCRU; ++iPad) {
558  const GlobalPadNumber globalNum = Mapper::GLOBALPADOFFSET[region] + iPad;
559  const auto& padPosLocal = mapper.padPos(globalNum);
560  vRow[iPad] = padPosLocal.getRow();
561  vPad[iPad] = padPosLocal.getPad();
562  vXPos[iPad] = mapper.getPadCentre(padPosLocal).X();
563  vYPos[iPad] = mapper.getPadCentre(padPosLocal).Y();
564  invPadArea[iPad] = Mapper::INVPADAREA[region];
565  const GlobalPosition2D globalPos = mapper.LocalToGlobal(LocalPosition2D(vXPos[iPad], vYPos[iPad]), cruTmp.sector());
566  vGlobalXPos[iPad] = globalPos.X();
567  vGlobalYPos[iPad] = globalPos.Y();
568  idcsPerIntegrationInterval[iPad] = idcStruct.getUngroupedIDCVal(iPad);
569  groupedidcsPerIntegrationInterval[iPad] = idcStruct.getGroupedIDCValGlobal(vRow[iPad], vPad[iPad]);
570  }
571 
572  pcstream << "tree"
573  << "cru=" << cru
574  << "sector=" << sector
575  << "region=" << region
576  << "integrationInterval=" << integrationInterval
577  << "IDCUngrouped.=" << idcsPerIntegrationInterval
578  << "IDCGrouped.=" << groupedidcsPerIntegrationInterval
579  << "invPadArea.=" << invPadArea
580  << "pad.=" << vPad
581  << "row.=" << vRow
582  << "lx.=" << vXPos
583  << "ly.=" << vYPos
584  << "gx.=" << vGlobalXPos
585  << "gy.=" << vGlobalYPos
586  << "\n";
587 }
