38 template <typename state_T, typename stream_T>
39 class Decoder
40 {
41  // the Coder works either with a 64Bit state and 32 bit streaming or
42  //a 32 Bit state and 8 Bit streaming We need to make sure it gets initialized with
43  //the right template arguments at compile time.
44  static_assert((sizeof(state_T) == sizeof(uint32_t) && sizeof(stream_T) == sizeof(uint8_t)) ||
45  (sizeof(state_T) == sizeof(uint64_t) && sizeof(stream_T) == sizeof(uint32_t)),
46  "Coder can either be 32Bit with 8 Bit stream type or 64 Bit Type with 32 Bit stream type");

52  // Unlike the encoder, the decoder works forwards as you'd expect.
53  template <typename stream_IT>

57  uint32_t get();

60  template <typename stream_IT>
61  stream_IT advanceSymbol(stream_IT inputIter, const DecoderSymbol& sym);
62 
63  private:
64  state_T mState{};
65  size_t mSymbolTablePrecission{};
66 
67  // Renormalize.
68  template <typename stream_IT>
69  std::tuple<state_T, stream_IT> renorm(state_T x, stream_IT iter);
70 
71  // L ('l' in the paper) is the lower bound of our normalization interval.
72  // Between this and our byte-aligned emission, we use 31 (not 32!) bits.
73  // This is done intentionally because exact reciprocals for 31-bit uints
74  // fit in 32-bit uints: this permits some optimizations during encoding.
75  inline static constexpr state_T LOWER_BOUND = needs64Bit<state_T>() ? (1u << 31) : (1u << 23); // lower bound of our normalization interval
76 
77  inline static constexpr state_T STREAM_BITS = sizeof(stream_T) * 8; // lower bound of our normalization interval

83 template <typename state_T, typename stream_T>
84 template <typename stream_IT>
85 stream_IT Decoder<state_T, stream_T>::init(stream_IT inputIter)
86 {
87 
88  state_T newState = 0;
89  stream_IT streamPosition = inputIter;
90 
91  if constexpr (needs64Bit<state_T>()) {
92  newState = static_cast<state_T>(*streamPosition) << 0;
93  --streamPosition;
94  newState |= static_cast<state_T>(*streamPosition) << 32;
95  --streamPosition;
96  assert(std::distance(streamPosition, inputIter) == 2);
97  } else {
98  newState = static_cast<state_T>(*streamPosition) << 0;
99  --streamPosition;
100  newState |= static_cast<state_T>(*streamPosition) << 8;
101  --streamPosition;
102  newState |= static_cast<state_T>(*streamPosition) << 16;
103  --streamPosition;
104  newState |= static_cast<state_T>(*streamPosition) << 24;
105  --streamPosition;
106  assert(std::distance(streamPosition, inputIter) == 4);
107  }
108 
109  mState = newState;
110  return streamPosition;

119 template <typename state_T, typename stream_T>
120 template <typename stream_IT>
121 stream_IT Decoder<state_T, stream_T>::advanceSymbol(stream_IT inputIter, const DecoderSymbol& symbol)
122 {
123  static_assert(std::is_same<typename std::iterator_traits<stream_IT>::value_type, stream_T>::value);
124 
125  state_T mask = (pow2(mSymbolTablePrecission)) - 1;
126 
127  // s, x = D(x)
128  state_T newState = mState;
129  newState = symbol.getFrequency() * (newState >> mSymbolTablePrecission) + (newState & mask) - symbol.getCumulative();
130 
131  // renormalize
132  const auto [renormedState, newStreamPosition] = this->renorm(newState, inputIter);
133  mState = renormedState;
134  return newStreamPosition;
135 };

137 template <typename state_T, typename stream_T>
138 template <typename stream_IT>
139 inline std::tuple<state_T, stream_IT> Decoder<state_T, stream_T>::renorm(state_T state, stream_IT inputIter)
140 {
141  static_assert(std::is_same<typename std::iterator_traits<stream_IT>::value_type, stream_T>::value);
142 
143  stream_IT streamPosition = inputIter;
144 
145  // renormalize
146  if (state < LOWER_BOUND) {
147  if constexpr (needs64Bit<state_T>()) {
148  state = (state << STREAM_BITS) | *streamPosition;
149  --streamPosition;
150  assert(state >= LOWER_BOUND);
151  } else {
152 
153  do {
154  state = (state << STREAM_BITS) | *streamPosition;
155  --streamPosition;
156  } while (state < LOWER_BOUND);
157  }
158  }
159  return std::make_tuple(state, streamPosition);
160 }
