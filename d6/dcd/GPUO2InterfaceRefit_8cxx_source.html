25 void GPUO2InterfaceRefit::fillSharedClustersMap(const ClusterNativeAccess* cl, const gsl::span<const TrackTPC> trks, const TPCClRefElem* trackRef, unsigned char* shmap)
26 {
27  if (!cl || !shmap) {
28  throw std::runtime_error("Must provide clusters access and preallocated recepient for shared map");
29  }
30  memset(shmap, 0, sizeof(char) * cl->nClustersTotal);
31  for (unsigned int i = 0; i < trks.size(); i++) {
32  for (unsigned int j = 0; j < trks[i].getNClusterReferences(); j++) {
33  size_t idx = &trks[i].getCluster(trackRef, j, *cl) - cl->clustersLinear;
34  shmap[idx] = shmap[idx] ? 2 : 1;
35  }
36  }
37  for (unsigned int i = 0; i < cl->nClustersTotal; i++) {
38  shmap[i] = (shmap[i] > 1 ? GPUTPCGMMergedTrackHit::flagShared : 0) | cl->clustersLinear[i].getFlags();
39  }
40 }

43 {
44  if (cl->nClustersTotal) {
45  if (sharedmap == nullptr && trks == nullptr) {
46  throw std::runtime_error("Must provide either shared cluster map or vector of tpc tracks to build the map");
47  }
48  if (sharedmap == nullptr) {
49  mSharedMap.resize(cl->nClustersTotal);
50  sharedmap = mSharedMap.data();
51  fillSharedClustersMap(cl, *trks, trackRef, mSharedMap.data());
52  }

62 }
