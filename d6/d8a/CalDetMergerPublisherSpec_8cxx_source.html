62  void run(o2::framework::ProcessingContext& pc) final
63  {
64  int nSlots = pc.inputs().getNofParts(0);
65  assert(pc.inputs().getNofParts(1) == nSlots);
66 
67  mRunNumber = processing_helpers::getRunNumber(pc);
68 
69  for (int isl = 0; isl < nSlots; isl++) {
70  const auto type = pc.inputs().get<int>("clbInfo", isl);
71  const auto pld = pc.inputs().get<gsl::span<char>>("clbPayload", isl); // this is actually an image of TMemFile
72  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().get("clbInfo", isl));
73  const auto subSpec = dh->subSpecification;
74  const int lane = subSpec >> 4;
75  const int calibType = subSpec & 0xf;
76 
77  // const auto& path = wrp->getPath();
78  TMemFile f("file", (char*)&pld[0], pld.size(), "READ");
79  if (!f.IsZombie()) {
80  auto calDetMap = f.Get<CalDetMap>("data");
81  if (calDetMap) {
82  if (mMergedCalDetsMap.size() == 0) {
83  mCalDetMapType = CDBType(type);
84  for (auto& [key, obj] : *calDetMap) {
85  mMergedCalDetsMap[key] = obj;
86  }
87  } else {
88  if (int(mCalDetMapType) != type) {
89  LOGP(fatal, "received CalDetMap of different type for merging, previous: {}, present{}", CDBTypeMap.at(mCalDetMapType), CDBTypeMap.at(CDBType(type)));
90  }
91  for (auto& [key, obj] : *calDetMap) {
92  mMergedCalDetsMap[key] += obj;
93  }
94  }
95  }
96 
97  auto calDet = f.Get<o2::tpc::CalDet<float>>("data");
98  if (calDet) {
99  if (mMergedCalDets.find(type) == mMergedCalDets.end()) {
100  mMergedCalDets[type] = *calDet;
101  } else {
102  mMergedCalDets[type] += *calDet;
103  }
104  }
105  }
106  f.Close();
107 
108  LOGP(info, "getting slot {}, subspec {:#8x}, lane {}, type {}", isl, subSpec, lane, calibType);
109  // if (mReceivedLanes.test(lane)) {
110  // LOGP(warning, "lane {} received multiple times", lane);
111  // }
112  mReceivedLanes.set(lane);
113  }
114 
115  if (mReceivedLanes.count() == mLanesToExpect) {
116  LOGP(info, "data of all lanes received");
117  if (mPublishAfterComplete) {
118  LOGP(info, "publishing after all data was received");
119  dumpCalibData();
120  sendOutput(pc.outputs());
121 
122  // reset calibration objects
123  mMergedCalDetsMap.clear();
124  for (auto& [type, object] : mMergedCalDets) {
125  object = 0;
126  }
127  mCalibDumped = false;
128  }
129  mReceivedLanes.reset();
130  }
131  }
132 
133  void endOfStream(o2::framework::EndOfStreamContext& ec) final
134  {
135  LOGP(info, "endOfStream");
136 
137  if (mReceivedLanes.count() == mLanesToExpect) {
138  dumpCalibData();
139  sendOutput(ec.outputs());
140  } else {
141  LOGP(info, "Received lanes {} does not match expected lanes {}, object already sent", mReceivedLanes.count(), mLanesToExpect);
142  }
143  ec.services().get<ControlService>().readyToQuit(QuitRequest::Me);
144  }

161  //____________________________________________________________________________
162  void sendOutput(DataAllocator& output)
163  {
164  // CDBStorage::MetaData_t md;
165 
166  // perhaps should be changed to time of the run
167  const auto now = std::chrono::system_clock::now();
168  const long timeStart = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()).count();
169  const long timeEnd = 99999999999999;
170 
171  std::map<std::string, std::string> md;
172 
173  if (mMergedCalDetsMap.size() > 0) {
174  o2::ccdb::CcdbObjectInfo w;
175  auto image = o2::ccdb::CcdbApi::createObjectImage(&mMergedCalDetsMap, &w);
176 
177  w.setPath(CDBTypeMap.at(mCalDetMapType));
178  w.setStartValidityTimestamp(timeStart);
179  w.setEndValidityTimestamp(timeEnd);
180 
181  md = w.getMetaData();
182  md[o2::base::NameConf::CCDBRunTag.data()] = std::to_string(mRunNumber);
183  w.setMetaData(md);
184 
185  LOGP(info, "Sending object {}/{} of size {} bytes, valid for {} : {}", w.getPath(), w.getFileName(), image->size(), w.getStartValidityTimestamp(), w.getEndValidityTimestamp());
186 
187  o2::header::DataHeader::SubSpecificationType subSpec{(o2::header::DataHeader::SubSpecificationType)mCalDetMapType};
188  output.snapshot(Output{clbUtils::gDataOriginCDBPayload, "TPC_CALIB", subSpec}, *image.get());
189  output.snapshot(Output{clbUtils::gDataOriginCDBWrapper, "TPC_CALIB", subSpec}, w);
190  }
191 
192  for (auto& [type, object] : mMergedCalDets) {
193  o2::ccdb::CcdbObjectInfo w;
194  auto image = o2::ccdb::CcdbApi::createObjectImage(&object, &w);
195 
196  w.setPath(CDBTypeMap.at(CDBType(type)));
197  w.setStartValidityTimestamp(timeStart);
198  w.setEndValidityTimestamp(timeEnd);
199 
200  md = w.getMetaData();
201  md[o2::base::NameConf::CCDBRunTag.data()] = std::to_string(mRunNumber);
202  w.setMetaData(md);
203 
204  LOG(info) << "Sending object " << w.getPath() << "/" << w.getFileName() << " of size " << image->size()
205  << " bytes, valid for " << w.getStartValidityTimestamp() << " : " << w.getEndValidityTimestamp();
206 
207  o2::header::DataHeader::SubSpecificationType subSpec{(o2::header::DataHeader::SubSpecificationType)type};
208  output.snapshot(Output{clbUtils::gDataOriginCDBPayload, "TPC_CALIB", subSpec}, *image.get());
209  output.snapshot(Output{clbUtils::gDataOriginCDBWrapper, "TPC_CALIB", subSpec}, w);
210  }
211  }

214  void dumpCalibData()
215  {
216  if (mDirectFileDump && !mCalibDumped) {
217  LOGP(info, "Dumping output to file");
218  std::string fileName = "merged_CalDet.root";
219  if (mMergedCalDetsMap.size()) {
220  const auto& cdbType = CDBTypeMap.at(mCalDetMapType);
221  const auto name = cdbType.substr(cdbType.rfind("/") + 1);
222  fileName = fmt::format("merged_{}.root", name);
223  }
224  TFile f(fileName.data(), "recreate");
225  for (auto& [key, object] : mMergedCalDetsMap) {
226  f.WriteObject(&object, object.getName().data());
227  }
228  for (auto& [type, object] : mMergedCalDets) {
229  f.WriteObject(&object, object.getName().data());
230  }
231  mCalibDumped = true;
232  }
233  }
