23 // matrix-like wrapper for C-array
24 // has no range checks
25 template <typename T>
26 struct Array2D {
27  using element_t = T;
28 
29  Array2D()
30  : data{nullptr},
31  rows{0},
32  cols{0}
33  {
34  }
35 
36  Array2D(T const* data_, uint32_t r, uint32_t c)
37  : rows{r}, cols{c}
38  {
39  data = new T[rows * cols];
40  for (auto i = 0U; i < rows; ++i) {
41  for (auto j = 0U; j < cols; ++j) {
42  data[i * cols + j] = data_[i * cols + j];
43  }
44  }
45  }
46  Array2D(std::vector<T> data_, uint32_t r, uint32_t c)
47  : rows{r}, cols{c}
48  {
49  data = new T[rows * cols];
50  for (auto i = 0U; i < rows; ++i) {
51  for (auto j = 0U; j < cols; ++j) {
52  data[i * cols + j] = data_[i * cols + j];
53  }
54  }
55  }
56 
57  Array2D(Array2D<T> const& other)
58  : rows{other.rows},
59  cols{other.cols}
60  {
61  data = new T[rows * cols];
62  for (auto i = 0U; i < rows; ++i) {
63  for (auto j = 0U; j < cols; ++j) {
64  data[i * cols + j] = other.data[i * cols + j];
65  }
66  }
67  }
68 
69  Array2D(Array2D<T>&& other)
70  : rows{other.rows},
71  cols{other.cols}
72  {
73  data = other.data;
74  other.data = nullptr;
75  other.rows = 0;
76  other.cols = 0;
77  }
78 
79  Array2D& operator=(Array2D<T> const& other)
80  {
81  this->rows = other.rows;
82  this->cols = other.cols;
83  data = new T[rows * cols];
84  for (auto i = 0U; i < rows; ++i) {
85  for (auto j = 0U; j < cols; ++j) {
86  data[i * cols + j] = *(other.data + (i * cols + j));
87  }
88  }
89  return *this;
90  }
91 
92  Array2D& operator=(Array2D<T>&& other)
93  {
94  this->rows = other.rows;
95  this->cols = other.cols;
96  data = other.data;
97  other.data = nullptr;
98  other.rows = 0;
99  other.cols = 0;
100  return *this;
101  }
102 
103  ~Array2D()
104  {
105  if (data != nullptr) {
106  delete[] data;
107  }
108  }
109 
110  T operator()(uint32_t y, uint32_t x) const
111  {
112  return data[y * cols + x];
113  }
114  T* operator[](uint32_t y) const
115  {
116  return data + y * cols;
117  }
118 
119  T* data;
120  uint32_t rows;
121  uint32_t cols;
122 };
