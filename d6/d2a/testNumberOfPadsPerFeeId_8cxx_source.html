25 int nofPadsPerFeeId(uint16_t feeid)
26 {
27  int npads{0};
28  auto feeLink2Solar = createFeeLink2SolarMapper<ElectronicMapperGenerated>();
29  auto solars = getSolarUIDsPerFeeId<ElectronicMapperGenerated>(feeid);
30  for (auto solar : solars) {
31  auto dualSampas = getDualSampas<ElectronicMapperGenerated>(solar);
32  for (auto ds : dualSampas) {
33  const o2::mch::mapping::Segmentation& seg = o2::mch::mapping::segmentation(ds.deId());
34  for (auto channel = 0; channel < 64; channel++) {
35  auto paduid = seg.findPadByFEE(ds.dsId(), channel);
36  if (seg.isValid(paduid)) {
37  npads++;
38  }
39  }
40  }
41  }
42  return npads;
43 }
