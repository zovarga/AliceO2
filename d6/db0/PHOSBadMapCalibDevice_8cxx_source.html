99 void PHOSCalibCollector::scanClusters(o2::framework::ProcessingContext& pc)
100 {
101 
102  // auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime; // is this the timestamp of the current TF?
103 
104  auto clusters = pc.inputs().get<gsl::span<o2::phos::Cluster>>("clusters");
105  auto cluTR = pc.inputs().get<gsl::span<o2::phos::TriggerRecord>>("cluTR");
106  LOG(info) << "Processing TF with " << clusters.size() << " clusters and " << cluTR.size() << " TriggerRecords";
107 
108  for (auto& tr : cluTR) {
109 
110  //Mark new event
111  //First goes new event marker + BC (16 bit), next word orbit (32 bit)
112  EventHeader h = {0};
113  h.mMarker = 16383;
114  h.mBC = tr.getBCData().bc;
115  mDigits.push_back(h.mDataWord);
116  mDigits.push_back(tr.getBCData().orbit);
117 
118  int iclu = 0;
119  int firstCluInEvent = tr.getFirstEntry();
120  int lastCluInEvent = firstCluInEvent + tr.getNumberOfObjects();
121 
122  mBuffer->startNewEvent(); // mark stored clusters to be used for Mixing
123  for (int i = firstCluInEvent; i < lastCluInEvent; i++) {
124  const Cluster& clu = clusters[i];
125 
126  fillTimeMassHisto(clu);
127  bool isGood = checkCluster(clu);
128 
129  auto cluList = clu.getElementList();
130  for (auto ce = cluList->begin(); ce != cluList->end(); ce++) {
131  short absId = ce->absId;
132  //Fill cells from cluster for next iterations
133  short adcCounts = ce->energy / mCalibParams->getGain(absId);
134  // Need to chale LG gain too to fit dynamic range
135  if (!ce->isHG) {
136  adcCounts /= mCalibParams->getHGLGRatio(absId);
137  }
138  CalibDigit d = {0};
139  d.mAddress = absId;
140  d.mAdcAmp = adcCounts;
141  d.mHgLg = ce->isHG;
142  d.mBadChannel = isGood;
143  d.mCluster = (i - firstCluInEvent) % kMaxCluInEvent;
144  mDigits.push_back(d.mDataWord);
145  if (i - firstCluInEvent > kMaxCluInEvent) {
146  //Normally this is not critical as indexes are used "locally", i.e. are compared to previous/next
147  LOG(info) << "Too many clusters per event:" << i - firstCluInEvent << ", apply more strict selection; clusters with same indexes will appear";
148  }
149  }
150  }
151  }

154 void PHOSCalibCollector::readDigits()
155 {
156  // open files from the list
157  // for each file read digits, calibrate them and form clusters
158  // fill inv mass and time (for check) histograms
159  std::ifstream ifs(mdigitsfilelist, std::ifstream::in);
160  if (!ifs.is_open()) {
161  LOG(error) << "can not open file " << mdigitsfilelist;
162  return;
163  }
164 
165  std::string filename;
166  while (ifs >> filename) {
167  TFile f(filename.data(), "READ");
168  if (!f.IsOpen()) {
169  LOG(error) << "can not read file " << filename;
170  continue;
171  }
172  std::vector<uint32_t>* digits;
173  f.GetObject("Digits", digits);
174 
175  std::vector<uint32_t>::const_iterator digIt = digits->cbegin();
176  std::vector<uint32_t>::const_iterator digEnd = digits->cend();
177  Cluster clu;
178  bool isNextNewEvent;
179  mBuffer->startNewEvent(); // mark stored clusters to be used for Mixing
180  while (nextCluster(digIt, digEnd, clu, isNextNewEvent)) {
181 
182  fillTimeMassHisto(clu);
183  if (isNextNewEvent) {
184  mBuffer->startNewEvent(); // mark stored clusters to be used for Mixing
185  }
186  }
187  delete digits;
188  f.Close();
189  }
190 }

192 bool PHOSCalibCollector::nextCluster(std::vector<uint32_t>::const_iterator digitIt, std::vector<uint32_t>::const_iterator digitEnd,
193  Cluster& clu, bool& isNextNewEvent)
194 {
195  //Scan digits belonging to cluster
196  // return true if cluster read
197  isNextNewEvent = false;
198  clu.reset();
199  int cluIndex = -1;
200  while (digitIt != digitEnd) {
201  CalibDigit d = {*digitIt};
202  if (d.mAddress == 16383) { //impossible address, marker of new event
203  if (clu.getMultiplicity() > 0) { //already read cluster: calculate its parameters; do not go into next event
204  isNextNewEvent = true;
205  break;
206  }
207  //If just started cluster, read new event. Two first words event header
208  EventHeader h = {*digitIt};
209  mEvBC = h.mBC; //current event BC
210  digitIt++;
211  mEvOrbit = *digitIt; //current event orbit
212  digitIt++;
213  continue;
214  }
215  if (cluIndex == -1) { //start new cluster
216  cluIndex = d.mCluster;
217  } else {
218  if (cluIndex != d.mCluster) { //digit belongs to text cluster: all digits from current read
219  break;
220  }
221  }
222  short absId = d.mAddress;
223  float energy = d.mAdcAmp * mCalibParams->getGain(absId);
224  if (!d.mHgLg) {
225  energy *= mCalibParams->getHGLGRatio(absId);
226  }
227  clu.addDigit(absId, energy, 0., 0, 1.);
228  digitIt++;
229  }
230  if (digitIt == digitEnd && clu.getMultiplicity() == 0) {
231  return false;
232  }
233  //Evaluate clu parameters
234  clu.purify();
235  clu.evalAll();
236  return true;
237 }
