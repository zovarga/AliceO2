39 const int gTreeSize = 10; // elements in the test tree
40 DataProcessorSpec getSourceSpec()
41 {
42  auto initFct = [](InitContext& ic) {
43  // create a test tree in a temporary file
44  std::string fileName = gSystem->TempDirectory();
45  fileName += "/test_RootTreeReader.root";
46 
47  {
48  std::unique_ptr<TFile> testFile(TFile::Open(fileName.c_str(), "RECREATE"));
49  std::unique_ptr<TTree> testTree = std::make_unique<TTree>("testtree", "testtree");
50 
51  std::vector<o2::test::TriviallyCopyable> msgblarray;
52  std::vector<o2::test::Polymorphic> valarray;
53  auto* branch1 = testTree->Branch("msgblarray", &msgblarray);
54  auto* branch2 = testTree->Branch("dataarray", &valarray);
55 
56  for (int entry = 0; entry < gTreeSize; entry++) {
57  msgblarray.clear();
58  valarray.clear();
59  for (int idx = 0; idx < entry + 1; ++idx) {
60  msgblarray.emplace_back((entry * 10) + idx, 0, 0);
61  valarray.emplace_back((entry * 10) + idx);
62  }
63  testTree->Fill();
64  }
65  testTree->Write();
66  testTree->SetDirectory(nullptr);
67  testFile->Close();
68  }

70  constexpr auto persistency = Lifetime::Transient;
71  auto reader = std::make_shared<RootTreeReader>("testtree", // tree name
72  fileName.c_str(), // input file name
73  RootTreeReader::BranchDefinition<std::vector<o2::test::TriviallyCopyable>>{Output{"TST", "ARRAYOFMSGBL", 0, persistency}, "msgblarray"},
74  Output{"TST", "ARRAYOFDATA", 0, persistency},
75  "dataarray",
76  RootTreeReader::PublishingMode::Single);

78  auto processingFct = [reader](ProcessingContext& pc) {
79  if (reader->getCount() >= gTreeSize) {
80  return;
81  }
82  if (reader->getCount() == 0) {
83  // add two additional headers on the stack in the first entry
84  o2::header::NameHeader<16> auxHeader("extended_info");
85  o2::header::DataHeader dummyheader;
86  (++(*reader))(pc, auxHeader, dummyheader);
87  } else {
88  // test signature without headers for the rest of the entries
89  (++(*reader))(pc);
90  }
91  if ((reader->getCount()) >= gTreeSize) {
92  pc.services().get<ControlService>().endOfStream();
93  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
94  }
95  };
96 
97  return processingFct;
98  };

100  return DataProcessorSpec{"source", // name of the processor
101  {},
102  {OutputSpec{"TST", "ARRAYOFDATA"},
103  OutputSpec{"TST", "ARRAYOFMSGBL"}},
104  AlgorithmSpec(initFct)};

107 DataProcessorSpec getSinkSpec()
108 {
109  auto processingFct = [](ProcessingContext& pc) {
110  static int counter = 0;
111  using DataHeader = o2::header::DataHeader;
112  for (auto& input : pc.inputs()) {
113  auto dh = DataRefUtils::getHeader<const DataHeader*>(input);
114  LOG(info) << dh->dataOrigin.str << " " << dh->dataDescription.str << " " << dh->payloadSize;
115  }
116  auto data = pc.inputs().get<std::vector<o2::test::Polymorphic>>("input1");
117  if (counter == 0) {
118  // the first entry comes together with additional headers on the stack, test those ...
119  auto auxHeader = DataRefUtils::getHeader<o2::header::NameHeader<16>*>(pc.inputs().get("input1"));
120  ASSERT_ERROR(auxHeader != nullptr);
121  if (auxHeader != nullptr) {
122  ASSERT_ERROR(std::string("extended_info") == auxHeader->getName());
123  o2::header::hexDump("", auxHeader, auxHeader->headerSize);
124  auto dummyheader = auxHeader->next();
125  ASSERT_ERROR(dummyheader != nullptr && dummyheader->size() == sizeof(o2::header::DataHeader));
126  }
127  }
128 
129  LOG(info) << "count: " << counter << " data elements:" << data.size();
130  ASSERT_ERROR(counter + 1 == data.size());
131 
132  // retrieving the unserialized message as vector
133  auto msgblvec = pc.inputs().get<std::vector<o2::test::TriviallyCopyable>*>("input2");
134  ASSERT_ERROR(counter + 1 == msgblvec->size());
135 
136  // retrieving the unserialized message as span
137  auto msgblspan = pc.inputs().get<gsl::span<o2::test::TriviallyCopyable>>("input2");
138  ASSERT_ERROR(counter + 1 == msgblspan.size());
139 
140  for (unsigned int idx = 0; idx < data.size(); idx++) {
141  LOG(info) << data[idx].get();
142  auto expected = 10 * counter + idx;
143  ASSERT_ERROR(data[idx].get() == expected);
144  ASSERT_ERROR(((*msgblvec)[idx] == o2::test::TriviallyCopyable{expected, 0, 0}));
145  ASSERT_ERROR((msgblspan[idx] == o2::test::TriviallyCopyable{expected, 0, 0}));
146  }
147 
148  ++counter;
149  };
150 
151  return DataProcessorSpec{"sink", // name of the processor
152  {InputSpec{"input1", "TST", "ARRAYOFDATA"},
153  InputSpec{"input2", "TST", "ARRAYOFMSGBL"}},
154  Outputs{},
155  AlgorithmSpec(processingFct)};
156 }
