62 void TRDTrackBasedCalibDevice::run(ProcessingContext& pc)
63 {
64  if (!mDataHeaderSet) {
65  mOutput = std::make_unique<Output>(o2::header::gDataOriginTRD, "ANGRESHISTS", 0, Lifetime::Timeframe);
66  mDataHeaderSet = true;
67  }
68  RecoContainer recoData;
69  recoData.collectData(pc, *mDataRequest.get());
70  mCalibrator.setInput(recoData);
71  mCalibrator.calculateAngResHistos();
72  ++mNumberOfProcessedTFs;
73  if (mNumberOfProcessedTFs % 200 == 0) {
74  pc.outputs().snapshot(*mOutput, mCalibrator.getAngResHistos());
75  mDataHeaderSet = false;
76  mNumberOfProcessedTFs = 0;
77  mCalibrator.reset();
78  }
79 }

90 DataProcessorSpec getTRDTrackBasedCalibSpec(o2::dataformats::GlobalTrackID::mask_t src)
91 {
92  std::vector<OutputSpec> outputs;
93  auto dataRequest = std::make_shared<DataRequest>();
94 
95  GTrackID::mask_t srcTrk;
96  if (GTrackID::includesSource(GTrackID::Source::ITSTPC, src)) {
97  LOGF(info, "Found ITS-TPC tracks as input, loading ITS-TPC-TRD");
98  srcTrk |= GTrackID::getSourcesMask("ITS-TPC-TRD");
99  }
100  if (GTrackID::includesSource(GTrackID::Source::TPC, src)) {
101  LOGF(info, "Found TPC tracks as input, loading TPC-TRD");
102  srcTrk |= GTrackID::getSourcesMask("TPC-TRD");
103  }
104  GTrackID::mask_t srcClu = GTrackID::getSourcesMask("TRD"); // we don't need all clusters, only TRD tracklets
105  dataRequest->requestTracks(srcTrk, false);
106  dataRequest->requestClusters(srcClu, false);
107 
108  outputs.emplace_back(o2::header::gDataOriginTRD, "ANGRESHISTS", 0, Lifetime::Timeframe);
109 
110  return DataProcessorSpec{
111  "trd-trackbased-calib",
112  dataRequest->inputs,
113  outputs,
114  AlgorithmSpec{adaptFromTask<TRDTrackBasedCalibDevice>(dataRequest)},
115  Options{}};
116 }
