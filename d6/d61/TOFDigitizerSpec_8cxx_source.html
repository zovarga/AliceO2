46  void initDigitizerTask(framework::InitContext& ic) override
47  {
48  LOG(info) << "Initializing TOF digitization";
49 
50  mSimChains = std::move(std::make_unique<std::vector<TChain*>>());
51 
52  // the instance of the actual digitizer
53  mDigitizer = std::move(std::make_unique<o2::tof::Digitizer>());
54  // containers for digits and labels
55  mDigits = std::move(std::make_unique<std::vector<o2::tof::Digit>>());
56  mLabels = std::move(std::make_unique<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>());
57 
58  // init digitizer
59  mDigitizer->init();
60  const bool isContinuous = ic.options().get<int>("pileup");
61  LOG(info) << "CONTINUOUS " << isContinuous;
62  mDigitizer->setContinuous(isContinuous);
63  mDigitizer->setMCTruthContainer(mLabels.get());
64  LOG(info) << "TOF initialization done";
65  }

67  void run(framework::ProcessingContext& pc)
68  {
69  static bool finished = false;
70  if (finished) {
71  return;
72  }
73  // RS: at the moment using hardcoded flag for continuos readout
74  static o2::parameters::GRPObject::ROMode roMode = o2::parameters::GRPObject::CONTINUOUS;
75 
76  // read collision context from input
77  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
78  auto& timesview = context->getEventRecords();
79  LOG(debug) << "GOT " << timesview.size() << " COLLISSION TIMES";
80 
81  context->initSimChains(o2::detectors::DetID::TOF, *mSimChains.get());
82 
83  // if there is nothing to do ... return
84  if (timesview.size() == 0) {
85  return;
86  }
87 
88  TStopwatch timer;
89  timer.Start();
90 
91  LOG(info) << " CALLING TOF DIGITIZATION ";
92  o2::dataformats::CalibLHCphaseTOF lhcPhaseObj;
93  o2::dataformats::CalibTimeSlewingParamTOF channelCalibObj;
94 
95  /*
96  if (mUseCCDB) { // read calibration objects from ccdb
97  // check LHC phase
98  auto lhcPhase = pc.inputs().get<o2::dataformats::CalibLHCphaseTOF*>("tofccdbLHCphase");
99  auto channelCalib = pc.inputs().get<o2::dataformats::CalibTimeSlewingParamTOF*>("tofccdbChannelCalib");
100 
101  o2::dataformats::CalibLHCphaseTOF lhcPhaseObjTmp = std::move(*lhcPhase);
102  o2::dataformats::CalibTimeSlewingParamTOF channelCalibObjTmp = std::move(*channelCalib);
103 
104  // make a copy in global scope
105  lhcPhaseObj = lhcPhaseObjTmp;
106  channelCalibObj = channelCalibObjTmp;
107  } else { // calibration objects set to zero
108  lhcPhaseObj.addLHCphase(0, 0);
109  lhcPhaseObj.addLHCphase(2000000000, 0);
110 
111  for (int ich = 0; ich < o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELS; ich++) {
112  channelCalibObj.addTimeSlewingInfo(ich, 0, 0);
113  int sector = ich / o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELXSECTOR;
114  int channelInSector = ich % o2::dataformats::CalibTimeSlewingParamTOF::NCHANNELXSECTOR;
115  channelCalibObj.setFractionUnderPeak(sector, channelInSector, 1);
116  }
117  }
118 */
119 
120  o2::tof::CalibTOFapi calibapi(long(0), &lhcPhaseObj, &channelCalibObj);
121  mDigitizer->setCalibApi(&calibapi);
122 
123  if (mUseCCDB) {
124  calibapi.setURL(mCCDBurl.c_str());
125  long ts = mTimestamp;
126  calibapi.setTimeStamp(ts);
127  calibapi.readTimeSlewingParam();
128  calibapi.readDiagnosticFrequencies();
129  calibapi.readActiveMap();
130  }
131 
132  static std::vector<o2::tof::HitType> hits;
133 
134  auto& eventParts = context->getEventParts();
135  // loop over all composite collisions given from context
136  // (aka loop over all the interaction records)
137  for (int collID = 0; collID < timesview.size(); ++collID) {
138  mDigitizer->setEventTime(timesview[collID]);
139 
140  // for each collision, loop over the constituents event and source IDs
141  // (background signal merging is basically taking place here)
142  for (auto& part : eventParts[collID]) {
143  mDigitizer->setEventID(part.entryID);
144  mDigitizer->setSrcID(part.sourceID);
145 
146  // get the hits for this event and this source
147  hits.clear();
148  context->retrieveHits(*mSimChains.get(), "TOFHit", part.sourceID, part.entryID, &hits);
149 
150  // LOG(info) << "For collision " << collID << " eventID " << part.entryID << " found " << hits.size() << " hits ";
151 
152  // call actual digitization procedure
153  mLabels->clear();
154  mDigits->clear();
155  mDigitizer->process(&hits, mDigits.get());
156  }
157  }
158  if (mDigitizer->isContinuous()) {
159  mDigits->clear();
160  mLabels->clear();
161  mDigitizer->flushOutputContainer(*mDigits.get());
162  }
163 
164  std::vector<Digit>* digitsVector = mDigitizer->getDigitPerTimeFrame();
165  std::vector<ReadoutWindowData>* readoutwindow = mDigitizer->getReadoutWindowData();
166  std::vector<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>* mcLabVecOfVec = mDigitizer->getMCTruthPerTimeFrame();
167 
168  LOG(info) << "Post " << digitsVector->size() << " digits in " << readoutwindow->size() << " RO windows";
169 
170  // here we have all digits and we can send them to consumer (aka snapshot it onto output)
171  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe}, *digitsVector);
172  if (pc.outputs().isAllowed({o2::header::gDataOriginTOF, "DIGITSMCTR", 0})) {
173  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "DIGITSMCTR", 0, Lifetime::Timeframe}, *mcLabVecOfVec);
174  }
175  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe}, *readoutwindow);
176 
177  // send empty pattern from digitizer (it may change in future)
178  std::vector<uint8_t>& patterns = mDigitizer->getPatterns();
179  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "PATTERNS", 0, Lifetime::Timeframe}, patterns);
180 
181  DigitHeader& digitH = mDigitizer->getDigitHeader();
182  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "DIGITHEADER", 0, Lifetime::Timeframe}, digitH);
183 
184  LOG(info) << "TOF: Sending ROMode= " << roMode << " to GRPUpdater";
185  pc.outputs().snapshot(Output{o2::header::gDataOriginTOF, "ROMode", 0, Lifetime::Timeframe}, roMode);
186 
187  timer.Stop();
188  LOG(info) << "Digitization took " << timer.CpuTime() << "s";
189 
190  // we should be only called once; tell DPL that this process is ready to exit
191  pc.services().get<ControlService>().endOfStream();
192 
193  finished = true;
194  }
195

200  std::unique_ptr<o2::dataformats::MCTruthContainer<o2::MCCompLabel>> mLabels;
201  bool mUseCCDB = false;
202  std::string mCCDBurl;
203  int mTimestamp = 0;
204 };

206 DataProcessorSpec getTOFDigitizerSpec(int channel, bool useCCDB, bool mctruth, std::string ccdb_url, int timestamp)
207 {
208  // create the full data processor spec using
209  // a name identifier
210  // input description
211  // algorithmic description (here a lambda getting called once to setup the actual processing function)
212  // options that can be used for this processor (here: input file names where to take the hits)
213  std::vector<InputSpec> inputs;
214  inputs.emplace_back("collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
215  // if (useCCDB) {
216  // inputs.emplace_back("tofccdbLHCphase", o2::header::gDataOriginTOF, "LHCphase");
217  // inputs.emplace_back("tofccdbChannelCalib", o2::header::gDataOriginTOF, "ChannelCalib");
218  // }
219  std::vector<OutputSpec> outputs;
220  outputs.emplace_back(o2::header::gDataOriginTOF, "DIGITHEADER", 0, Lifetime::Timeframe);
221  outputs.emplace_back(o2::header::gDataOriginTOF, "DIGITS", 0, Lifetime::Timeframe);
222  outputs.emplace_back(o2::header::gDataOriginTOF, "READOUTWINDOW", 0, Lifetime::Timeframe);
223  outputs.emplace_back(o2::header::gDataOriginTOF, "PATTERNS", 0, Lifetime::Timeframe);
224  if (mctruth) {
225  outputs.emplace_back(o2::header::gDataOriginTOF, "DIGITSMCTR", 0, Lifetime::Timeframe);
226  }
227  outputs.emplace_back(o2::header::gDataOriginTOF, "ROMode", 0, Lifetime::Timeframe);
228  return DataProcessorSpec{
229  "TOFDigitizer",
230  inputs,
231  outputs,
232  AlgorithmSpec{adaptFromTask<TOFDPLDigitizerTask>(useCCDB, ccdb_url, timestamp)},
233  Options{{"pileup", VariantType::Int, 1, {"whether to run in continuous time mode"}}}
234  // I can't use VariantType::Bool as it seems to have a problem
235  };
236 }
