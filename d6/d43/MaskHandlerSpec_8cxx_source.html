57  void printSummary()
58  {
59  std::string name = "calib";
60  o2::mid::ColumnDataToLocalBoard colToBoard;
61  o2::mid::ROBoardConfigHandler roBoardCfgHandler;
62 
63  for (auto& masks : mMasksHandlers) {
64  auto maskVec = masks.getMasks();
65  colToBoard.process(maskVec);
66  auto roMasks = colToBoard.getData();
67  if (maskVec.empty()) {
68  LOG(info) << "No problematic digit found in " << name << " events";
69  } else {
70  LOG(info) << "Problematic digits found in " << name << " events. Corresponding masks:";
71  for (auto& mask : maskVec) {
72  LOG(info) << mask;
73  }
74  std::cout << "\nCorresponding boards masks:" << std::endl;
75  for (auto& board : roMasks) {
76  std::cout << board << std::endl;
77  }
78  }
79 
80  if (!mOutFilename.empty()) {
81  auto idx = mOutFilename.find_last_of("/");
82  auto insertIdx = (idx == std::string::npos) ? 0 : idx + 1;
83  std::string outFilename = mOutFilename;
84  outFilename.insert(insertIdx, "_");
85  outFilename.insert(insertIdx, name.c_str());
86  auto fullMasks = masks.getMasksFull(o2::mid::makeDefaultMasks());
87  colToBoard.process(fullMasks);
88  roBoardCfgHandler.updateMasks(colToBoard.getData());
89  roBoardCfgHandler.write(outFilename.c_str());
90  }
91 
92  name = "FET";
93  }

96  void
97  run(o2::framework::ProcessingContext& pc)
98  {
99  auto tStart = std::chrono::high_resolution_clock::now();
100 
101  std::array<gsl::span<const ColumnData>, 2> masks;
102 
103  masks[0] = pc.inputs().get<gsl::span<ColumnData>>("mid_noise_mask");
104  masks[1] = pc.inputs().get<gsl::span<ColumnData>>("mid_dead_mask");
105 
106  auto tAlgoStart = std::chrono::high_resolution_clock::now();
107 
108  bool isChanged = false;
109  for (int itype = 0; itype < 2; ++itype) {
110  ChannelMasksHandler chMasks;
111  for (auto& mask : masks[itype]) {
112  chMasks.setFromChannelMask(mask);
113  }
114  if (!(chMasks == mMasksHandlers[itype])) {
115  mMasksHandlers[itype] = chMasks;
116  isChanged = true;
117  }
118  }
119 
120  if (isChanged) {
121  printSummary();
122  }
123 
124  mTimerMaskHandler += std::chrono::high_resolution_clock::now() - tAlgoStart;
125 
126  mTimer += std::chrono::high_resolution_clock::now() - tStart;
127  }
128 
129  private:
130  std::array<ChannelMasksHandler, 2> mMasksHandlers{};
131  std::chrono::duration<double> mTimer{0};
132  std::chrono::duration<double> mTimerMaskHandler{0};
133  std::string mOutFilename{};
134 };
