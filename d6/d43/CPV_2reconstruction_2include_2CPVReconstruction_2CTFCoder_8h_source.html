58 template <typename VEC>
59 void CTFCoder::encode(VEC& buff, const gsl::span<const TriggerRecord>& trigData, const gsl::span<const Cluster>& cluData)
60 {
61  using MD = o2::ctf::Metadata::OptStore;
62  // what to do which each field: see o2::ctd::Metadata explanation
63  constexpr MD optField[CTF::getNBlocks()] = {
64  MD::EENCODE, // BLC_bcIncTrig
65  MD::EENCODE, // BLC_orbitIncTrig
66  MD::EENCODE, // BLC_entriesTrig
67  MD::EENCODE, // BLC_posX
68  MD::EENCODE, // BLC_posZ
69  MD::EENCODE, // BLC_energy
70  MD::EENCODE // BLC_status
71  };
72 
73  CTFHelper helper(trigData, cluData);
74 
75  // book output size with some margin
76  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
77  buff.resize(szIni);
78 
79  auto ec = CTF::create(buff);
80  using ECB = CTF::base;
81 
82  ec->setHeader(helper.createHeader());
83  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
84  ec->getANSHeader().majorVersion = 0;
85  ec->getANSHeader().minorVersion = 1;
86  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
87 #define ENCODECPV(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
88  // clang-format off
89  ENCODECPV(helper.begin_bcIncTrig(), helper.end_bcIncTrig(), CTF::BLC_bcIncTrig, 0);
90  ENCODECPV(helper.begin_orbitIncTrig(), helper.end_orbitIncTrig(), CTF::BLC_orbitIncTrig, 0);
91  ENCODECPV(helper.begin_entriesTrig(), helper.end_entriesTrig(), CTF::BLC_entriesTrig, 0);
92 
93  ENCODECPV(helper.begin_posX(), helper.end_posX(), CTF::BLC_posX, 0);
94  ENCODECPV(helper.begin_posZ(), helper.end_posZ(), CTF::BLC_posZ, 0);
95  ENCODECPV(helper.begin_energy(), helper.end_energy(), CTF::BLC_energy, 0);
96  ENCODECPV(helper.begin_status(), helper.end_status(), CTF::BLC_status, 0);
97  // clang-format on
98  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
99 }

102 template <typename VTRG, typename VCLUSTER>
103 void CTFCoder::decode(const CTF::base& ec, VTRG& trigVec, VCLUSTER& cluVec)
104 {
105  auto header = ec.getHeader();
106  checkDictVersion(static_cast<const o2::ctf::CTFDictHeader&>(header));
107  ec.print(getPrefix(), mVerbosity);
108  std::vector<uint16_t> bcInc, entries, posX, posZ;
109  std::vector<uint32_t> orbitInc;
110  std::vector<uint8_t> energy, status;
111 
112 #define DECODECPV(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
113  // clang-format off
114  DECODECPV(bcInc, CTF::BLC_bcIncTrig);
115  DECODECPV(orbitInc, CTF::BLC_orbitIncTrig);
116  DECODECPV(entries, CTF::BLC_entriesTrig);
117  DECODECPV(posX, CTF::BLC_posX);
118  DECODECPV(posZ, CTF::BLC_posZ);
119  DECODECPV(energy, CTF::BLC_energy);
120  DECODECPV(status, CTF::BLC_status);
121  // clang-format on
122  //
123  trigVec.clear();
124  cluVec.clear();
125  trigVec.reserve(header.nTriggers);
126  status.reserve(header.nClusters);
127 
128  uint32_t firstEntry = 0, cluCount = 0;
129  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
130 
131  Cluster clu;
132  for (uint32_t itrig = 0; itrig < header.nTriggers; itrig++) {
133  // restore TrigRecord
134  if (orbitInc[itrig]) { // non-0 increment => new orbit
135  ir.bc = bcInc[itrig]; // bcInc has absolute meaning
136  ir.orbit += orbitInc[itrig];
137  } else {
138  ir.bc += bcInc[itrig];
139  }
140 
141  firstEntry = cluVec.size();
142  for (uint16_t ic = 0; ic < entries[itrig]; ic++) {
143  clu.setPacked(posX[cluCount], posZ[cluCount], energy[cluCount], status[cluCount]);
144  cluVec.emplace_back(clu);
145  cluCount++;
146  }
147  trigVec.emplace_back(ir, firstEntry, entries[itrig]);
148  }
149  assert(cluCount == header.nClusters);
150 }
