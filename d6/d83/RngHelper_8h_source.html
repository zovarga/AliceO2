44  // static function to get a true random number from /dev/urandom
45  template <typename T>
46  static T readURandom()
47  {
48  int randomDataHandle = open("/dev/urandom", O_RDONLY);
49  if (randomDataHandle < 0) {
50  // something went wrong
51  } else {
52  T seed;
53  auto result = read(randomDataHandle, &seed, sizeof(T));
54  if (result < 0) {
55  // something went wrong
56  }
57  close(randomDataHandle);
58  return seed;
59  }
60  return T(0);
61  }
62 };
