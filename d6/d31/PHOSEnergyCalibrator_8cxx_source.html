50 void PHOSEnergySlot::fill(const gsl::span<const Cluster>& clusters, const gsl::span<const CluElement>& cluelements, const gsl::span<const TriggerRecord>& cluTR)
51 {
52  //Scan current list of clusters
53  //Fill time, non-linearity and mgg histograms
54  //Fill list of re-calibraiable digits
55  for (auto& tr : cluTR) {
56 
57  //Mark new event
58  //First goes new event marker + BC (16 bit), next word orbit (32 bit)
59  EventHeader h = {0};
60  h.mMarker = 16383;
61  h.mBC = tr.getBCData().bc;
62  mDigits.push_back(h.mDataWord);
63  mDigits.push_back(tr.getBCData().orbit);
64 
65  int iclu = 0;
66  int firstCluInEvent = tr.getFirstEntry();
67  int lastCluInEvent = firstCluInEvent + tr.getNumberOfObjects();
68 
69  mBuffer->startNewEvent(); // mark stored clusters to be used for Mixing
70  for (int i = firstCluInEvent; i < lastCluInEvent; i++) {
71  const Cluster& clu = clusters[i];
72  if (clu.getEnergy() < 1.e-4) { //There was problem in unfolding and cluster parameters not calculated
73  continue;
74  }
75  fillTimeMassHisto(clu, cluelements);
76 
77  uint32_t firstCE = clu.getFirstCluEl();
78  uint32_t lastCE = clu.getLastCluEl();
79  for (int idig = firstCE; idig < lastCE; idig++) {
80  const CluElement& ce = cluelements[idig];
81  short absId = ce.absId;
82  //Fill cells from cluster for next iterations
83  short adcCounts = ce.energy / mCalibParams->getGain(absId);
84  // Need to chale LG gain too to fit dynamic range
85  if (!ce.isHG) {
86  adcCounts /= mCalibParams->getHGLGRatio(absId);
87  }
88  CalibDigit d = {0};
89  d.mAddress = absId;
90  d.mAdcAmp = adcCounts;
91  d.mHgLg = ce.isHG;
92  d.mCluster = (i - firstCluInEvent) % kMaxCluInEvent;
93  mDigits.push_back(d.mDataWord);
94  if (i - firstCluInEvent > kMaxCluInEvent) {
95  //Normally this is not critical as indexes are used "locally", i.e. are compared to previous/next
96  LOG(info) << "Too many clusters per event:" << i - firstCluInEvent << ", apply more strict selection; clusters with same indexes will appear";
97  }
98  }
99  }
100  }
101 }

108 void PHOSEnergySlot::fillTimeMassHisto(const Cluster& clu, const gsl::span<const CluElement>& cluelements)
109 {
110  // Fill time distributions only for cells in cluster
111  uint32_t firstCE = clu.getFirstCluEl();
112  uint32_t lastCE = clu.getLastCluEl();
113 
114  for (int idig = firstCE; idig < lastCE; idig++) {
115  const CluElement& ce = cluelements[idig];
116  short absId = ce.absId;
117  if (ce.isHG) {
118  if (ce.energy > mEminHGTime) {
119  mHistos.fill(ETCalibHistos::kTimeHGPerCell, absId, ce.time);
120  }
121  mHistos.fill(ETCalibHistos::kTimeHGSlewing, ce.time, ce.energy);
122  } else {
123  if (ce.energy > mEminLGTime) {
124  mHistos.fill(ETCalibHistos::kTimeLGPerCell, absId, ce.time);
125  }
126  mHistos.fill(ETCalibHistos::kTimeLGSlewing, ce.time, ce.energy);
127  }
128  }
129 
130  //Real and Mixed inv mass distributions
131  // prepare TLorentsVector
132  float posX, posZ;
133  clu.getLocalPosition(posX, posZ);
134  TVector3 vec3;
135  mGeom->local2Global(clu.module(), posX, posZ, vec3);
136  vec3 -= mVertex;
137  float e = clu.getEnergy();
138  short absId;
139  mGeom->relPosToAbsId(clu.module(), posX, posZ, absId);
140 
141  vec3 *= 1. / vec3.Mag();
142  TLorentzVector v(vec3.X() * e, vec3.Y() * e, vec3.Z() * e, e);
143  // Fill calibration histograms for all cells, even bad, but partners in inv, mass should be good
144  bool isGood = checkCluster(clu);
145  for (short ip = mBuffer->size(); ip--;) {
146  const TLorentzVector& vp = mBuffer->getEntry(ip);
147  TLorentzVector sum = v + vp;
148  if (mBuffer->isCurrentEvent(ip)) { //same (real) event
149  if (isGood) {
150  mHistos.fill(ETCalibHistos::kReInvMassNonlin, e, sum.M());
151  }
152  if (sum.Pt() > mPtMin) {
153  mHistos.fill(ETCalibHistos::kReInvMassPerCell, absId, sum.M());
154  }
155  } else { //Mixed
156  if (isGood) {
157  mHistos.fill(ETCalibHistos::kMiInvMassNonlin, e, sum.M());
158  }
159  if (sum.Pt() > mPtMin) {
160  mHistos.fill(ETCalibHistos::kMiInvMassPerCell, absId, sum.M());
161  }
162  }
163  }
164 
165  //Add to list ot partners only if cluster is good
166  if (isGood) {
167  mBuffer->addEntry(v);
168  }
169 }

171 bool PHOSEnergySlot::checkCluster(const Cluster& clu)
172 {
173  //First check BadMap
174  float posX, posZ;
175  clu.getLocalPosition(posX, posZ);
176  short absId;
177  Geometry::relPosToAbsId(clu.module(), posX, posZ, absId);
178  if (!mBadMap->isChannelGood(absId)) {
179  return false;
180  }
181 
182  return (clu.getEnergy() > 0.3 && clu.getMultiplicity() > 1);
183 }

195 void PHOSEnergyCalibrator::finalizeSlot(Slot& slot)
196 {
197 
198  // Extract results for the single slot
199  es* c = slot.getContainer();
200  LOG(info) << "Finalize slot " << slot.getTFStart() << " <= TF <= " << slot.getTFEnd();
201  //Add histos
202  mHistos->merge(c->getCollectedHistos());
203  //Add collected Digits
204  auto tmpD = c->getCollectedDigits();
205  //Add to list or write to file directly?
206  if (!mFout) { //not open yet?
207  LOG(info) << "Writing CalibDigits to file " << mdigitsfilename.data();
208  mFout.reset(TFile::Open(mdigitsfilename.data(), "recreate"));
209  }
210  int nbites = mFout->WriteObjectAny(&tmpD, "std::vector<uint32_t>", Form("Digits%d", mChank++));
211  LOG(info) << "Writing " << tmpD.size() << " CalibDigits, wrote " << nbites << "bytes";
212  c->clear();
213 }
