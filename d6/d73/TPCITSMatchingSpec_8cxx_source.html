80 void TPCITSMatchingDPL::init(InitContext& ic)
81 {
82  mTimer.Stop();
83  mTimer.Reset();
84  //-------- init geometry and field --------//
85  o2::base::GeometryManager::loadGeometry();
86  o2::base::Propagator::initFieldFromGRP();
87  std::unique_ptr<o2::parameters::GRPObject> grp{o2::parameters::GRPObject::loadFrom()};
88  mMatching.setSkipTPCOnly(mSkipTPCOnly);
89  mMatching.setITSTriggered(!grp->isDetContinuousReadOut(o2::detectors::DetID::ITS));
90  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
91  if (mMatching.isITSTriggered()) {
92  mMatching.setITSROFrameLengthMUS(alpParams.roFrameLengthTrig / 1.e3); // ITS ROFrame duration in \mus
93  } else {
94  mMatching.setITSROFrameLengthInBC(alpParams.roFrameLengthInBC); // ITS ROFrame duration in \mus
95  }
96  mMatching.setMCTruthOn(mUseMC);
97  mMatching.setUseFT0(mUseFT0);
98  mMatching.setVDriftCalib(mCalibMode);
99  mMatching.setNThreads(std::max(1, ic.options().get<int>("nthreads")));
100  mMatching.setUseBCFilling(!ic.options().get<bool>("ignore-bc-check"));
101  //
102  std::string dictPath = o2::itsmft::ClustererParam<o2::detectors::DetID::ITS>::Instance().dictFilePath;
103  std::string dictFile = o2::base::DetectorNameConf::getAlpideClusterDictionaryFileName(o2::detectors::DetID::ITS, dictPath);
104  if (o2::utils::Str::pathExists(dictFile)) {
105  mITSDict.readFromFile(dictFile);
106  LOG(info) << "Matching is running with a provided ITS dictionary: " << dictFile;
107  } else {
108  LOG(info) << "Dictionary " << dictFile << " is absent, Matching expects ITS cluster patterns";
109  }
110  mMatching.setITSDictionary(&mITSDict);
111 
112  // this is a hack to provide Mat.LUT from the local file, in general will be provided by the framework from CCDB
113  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
114  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
115  if (o2::utils::Str::pathExists(matLUTFile)) {
116  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
117  o2::base::Propagator::Instance()->setMatLUT(lut);
118  LOG(info) << "Loaded material LUT from " << matLUTFile;
119  } else {
120  LOG(info) << "Material LUT " << matLUTFile << " file is absent, only TGeo can be used";
121  }
122 
123  int dbgFlags = ic.options().get<int>("debug-tree-flags");
124  mMatching.setDebugFlag(dbgFlags);
125 
126  // set bunch filling. Eventually, this should come from CCDB
127  if (mMatching.getUseBCFilling()) {
128  const auto* digctx = o2::steer::DigitizationContext::loadFromFile();
129  const auto& bcfill = digctx->getBunchFilling();
130  mMatching.setBunchFilling(bcfill);
131  }
132  mMatching.init();
133  //
134 }

136 void TPCITSMatchingDPL::run(ProcessingContext& pc)
137 {
138  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true));
139  LOG(info) << " startOrbit: " << dh->firstTForbit;
140  mTimer.Start(false);
141  RecoContainer recoData;
142  recoData.collectData(pc, *mDataRequest.get());
143 
144  mMatching.run(recoData);
145 
146  pc.outputs().snapshot(Output{"GLO", "TPCITS", 0, Lifetime::Timeframe}, mMatching.getMatchedTracks());
147  pc.outputs().snapshot(Output{"GLO", "TPCITSAB_REFS", 0, Lifetime::Timeframe}, mMatching.getABTrackletRefs());
148  pc.outputs().snapshot(Output{"GLO", "TPCITSAB_CLID", 0, Lifetime::Timeframe}, mMatching.getABTrackletClusterIDs());
149  if (mUseMC) {
150  pc.outputs().snapshot(Output{"GLO", "TPCITS_MC", 0, Lifetime::Timeframe}, mMatching.getMatchLabels());
151  pc.outputs().snapshot(Output{"GLO", "TPCITSAB_MC", 0, Lifetime::Timeframe}, mMatching.getABTrackletLabels());
152  }
153 
154  if (mCalibMode) {
155  pc.outputs().snapshot(Output{"GLO", "TPCITS_VDTGL", 0, Lifetime::Timeframe}, mMatching.getTglITSTPC());
156  }
157  mTimer.Stop();
158 }

167 DataProcessorSpec getTPCITSMatchingSpec(GTrackID::mask_t src, bool useFT0, bool calib, bool skipTPCOnly, bool useMC)
168 {
169  std::vector<OutputSpec> outputs;
170  auto dataRequest = std::make_shared<DataRequest>();
171  if ((src & GTrackID::getSourcesMask("TPC-TRD,TPC-TOF,TPC-TRD-TOF")).any()) { // preliminary stage of extended workflow ?
172  dataRequest->setMatchingInputStrict();
173  }
174 
175  dataRequest->requestTracks(src, useMC);
176  dataRequest->requestTPCClusters(false);
177  dataRequest->requestITSClusters(useMC); // Only ITS clusters labels are needed for the afterburner
178  if (useFT0) {
179  dataRequest->requestFT0RecPoints(false);
180  }
181  outputs.emplace_back("GLO", "TPCITS", 0, Lifetime::Timeframe);
182  outputs.emplace_back("GLO", "TPCITSAB_REFS", 0, Lifetime::Timeframe); // AftetBurner ITS tracklet references (referred by GlobalTrackID::ITSAB) on cluster indices
183  outputs.emplace_back("GLO", "TPCITSAB_CLID", 0, Lifetime::Timeframe); // cluster indices of ITS tracklets attached by the AfterBurner
184 
185  if (calib) {
186  outputs.emplace_back("GLO", "TPCITS_VDTGL", 0, Lifetime::Timeframe);
187  }
188 
189  if (useMC) {
190  outputs.emplace_back("GLO", "TPCITS_MC", 0, Lifetime::Timeframe);
191  outputs.emplace_back("GLO", "TPCITSAB_MC", 0, Lifetime::Timeframe); // AfterBurner ITS tracklet MC
192  }
193 
194  return DataProcessorSpec{
195  "itstpc-track-matcher",
196  dataRequest->inputs,
197  outputs,
198  AlgorithmSpec{adaptFromTask<TPCITSMatchingDPL>(dataRequest, useFT0, calib, skipTPCOnly, useMC)},
199  Options{
200  {"nthreads", VariantType::Int, 1, {"Number of afterburner threads"}},
201  {"material-lut-path", VariantType::String, "", {"Path of the material LUT file"}},
202  {"ignore-bc-check", VariantType::Bool, false, {"Do not check match candidate against BC filling"}},
203  {"debug-tree-flags", VariantType::Int, 0, {"DebugFlagTypes bit-pattern for debug tree"}}}};
204 }
