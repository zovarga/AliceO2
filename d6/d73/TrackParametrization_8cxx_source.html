46 template <typename value_T>
47 GPUd() TrackParametrization<value_T>::TrackParametrization(const dim3_t& xyz, const dim3_t& pxpypz, int charge, bool sectorAlpha, const PID pid)
48  : mX{0.f}, mAlpha{0.f}, mP{0.f}
49 {
50  // construct track param from kinematics
51 
52  // Alpha of the frame is defined as:
53  // sectorAlpha == false : -> angle of pt direction
54  // sectorAlpha == true : -> angle of the sector from X,Y coordinate for r>1
55  // angle of pt direction for r==0
56  //
57  //
58  constexpr value_t kSafe = 1e-5;
59  value_t radPos2 = xyz[0] * xyz[0] + xyz[1] * xyz[1];
60  value_t alp = 0;
61  if (sectorAlpha || radPos2 < 1) {
62  alp = math_utils::detail::atan2<value_T>(pxpypz[1], pxpypz[0]);
63  } else {
64  alp = math_utils::detail::atan2<value_T>(xyz[1], xyz[0]);
65  }
66  if (sectorAlpha) {
67  alp = math_utils::detail::angle2Alpha<value_t>(alp);
68  }
69  //
70  value_t sn, cs;
71  math_utils::detail::sincos(alp, sn, cs);
72  // protection against cosp<0
73  if (cs * pxpypz[0] + sn * pxpypz[1] < 0) {
74  LOG(debug) << "alpha from phiPos() will invalidate this track parameters, overriding to alpha from phi()";
75  alp = math_utils::detail::atan2<value_T>(pxpypz[1], pxpypz[0]);
76  if (sectorAlpha) {
77  alp = math_utils::detail::angle2Alpha<value_t>(alp);
78  }
79  math_utils::detail::sincos(alp, sn, cs);
80  }
81 
82  // protection: avoid alpha being too close to 0 or +-pi/2
83  if (math_utils::detail::abs<value_T>(sn) < 2 * kSafe) {
84  if (alp > 0) {
85  alp += alp < constants::math::PIHalf ? 2 * kSafe : -2 * kSafe;
86  } else {
87  alp += alp > -constants::math::PIHalf ? -2 * kSafe : 2 * kSafe;
88  }
89  math_utils::detail::sincos(alp, sn, cs);
90  } else if (math_utils::detail::abs<value_T>(cs) < 2 * kSafe) {
91  if (alp > 0) {
92  alp += alp > constants::math::PIHalf ? 2 * kSafe : -2 * kSafe;
93  } else {
94  alp += alp > -constants::math::PIHalf ? 2 * kSafe : -2 * kSafe;
95  }
96  math_utils::detail::sincos(alp, sn, cs);
97  }
98  // get the vertex of origin and the momentum
99  dim3_t ver{xyz[0], xyz[1], xyz[2]};
100  dim3_t mom{pxpypz[0], pxpypz[1], pxpypz[2]};
101  //
102  // Rotate to the local coordinate system
103  math_utils::detail::rotateZ<value_t>(ver, -alp);
104  math_utils::detail::rotateZ<value_t>(mom, -alp);
105  //
106  value_t ptI = 1.f / sqrt(mom[0] * mom[0] + mom[1] * mom[1]);
107  mX = ver[0];
108  mAlpha = alp;
109  mP[kY] = ver[1];
110  mP[kZ] = ver[2];
111  mP[kSnp] = mom[1] * ptI;
112  mP[kTgl] = mom[2] * ptI;
113  mAbsCharge = math_utils::detail::abs<value_T>(charge);
114  mP[kQ2Pt] = charge ? ptI * charge : ptI;
115  mPID = pid;
116  //
117  if (math_utils::detail::abs<value_T>(1 - getSnp()) < kSafe) {
118  mP[kSnp] = 1.f - kSafe; // Protection
119  } else if (math_utils::detail::abs<value_T>(-1 - getSnp()) < kSafe) {
120  mP[kSnp] = -1.f + kSafe; // Protection
121  }
122  //
123 }

126 template <typename value_T>
127 GPUd() bool TrackParametrization<value_T>::getPxPyPzGlo(dim3_t& pxyz) const
128 {
129  // track momentum
130  if (math_utils::detail::abs<value_T>(getQ2Pt()) < constants::math::Almost0 || math_utils::detail::abs<value_T>(getSnp()) > constants::math::Almost1) {
131  return false;
132  }
133  value_t cs, sn, pt = getPt();
134  value_t r = math_utils::detail::sqrt<value_T>((1.f - getSnp()) * (1.f + getSnp()));
135  math_utils::detail::sincos(getAlpha(), sn, cs);
136  pxyz[0] = pt * (r * cs - getSnp() * sn);
137  pxyz[1] = pt * (getSnp() * cs + r * sn);
138  pxyz[2] = pt * getTgl();
139  return true;
140 }

143 template <typename value_T>
144 GPUd() bool TrackParametrization<value_T>::getPosDirGlo(gpu::gpustd::array<value_t, 9>& posdirp) const
145 {
146  // fill vector with lab x,y,z,px/p,py/p,pz/p,p,sinAlpha,cosAlpha
147  value_t ptI = math_utils::detail::abs<value_T>(getQ2Pt());
148  value_t snp = getSnp();
149  if (ptI < constants::math::Almost0 || math_utils::detail::abs<value_T>(snp) > constants::math::Almost1) {
150  return false;
151  }
152  value_t &sn = posdirp[7], &cs = posdirp[8];
153  value_t csp = math_utils::detail::sqrt<value_T>((1.f - snp) * (1.f + snp));
154  value_t cstht = math_utils::detail::sqrt<value_T>(1.f + getTgl() * getTgl());
155  value_t csthti = 1.f / cstht;
156  math_utils::detail::sincos(getAlpha(), sn, cs);
157  posdirp[0] = getX() * cs - getY() * sn;
158  posdirp[1] = getX() * sn + getY() * cs;
159  posdirp[2] = getZ();
160  posdirp[3] = (csp * cs - snp * sn) * csthti; // px/p
161  posdirp[4] = (snp * cs + csp * sn) * csthti; // py/p
162  posdirp[5] = getTgl() * csthti; // pz/p
163  posdirp[6] = cstht / ptI; // p
164  return true;
165 }

168 template <typename value_T>
169 GPUd() bool TrackParametrization<value_T>::rotateParam(value_t alpha)
170 {
171  // rotate to alpha frame
172  if (math_utils::detail::abs<value_T>(getSnp()) > constants::math::Almost1) {
173  LOGP(debug, "Precondition is not satisfied: |sin(phi)|>1 ! {:f}", getSnp());
174  return false;
175  }
176  //
177  math_utils::detail::bringToPMPi<value_t>(alpha);
178  //
179  value_t ca = 0, sa = 0;
180  math_utils::detail::sincos(alpha - getAlpha(), sa, ca);
181  value_t snp = getSnp(), csp = math_utils::detail::sqrt<value_T>((1.f - snp) * (1.f + snp)); // Improve precision
182  // RS: check if rotation does no invalidate track model (cos(local_phi)>=0, i.e. particle
183  // direction in local frame is along the X axis
184  if ((csp * ca + snp * sa) < 0) {
185  //LOGF(warning,"Rotation failed: local cos(phi) would become {:.2f}", csp * ca + snp * sa);
186  return false;
187  }
188  //
189  value_t tmp = snp * ca - csp * sa;
190  if (math_utils::detail::abs<value_T>(tmp) > constants::math::Almost1) {
191  LOGP(debug, "Rotation failed: new snp {:.2f}", tmp);
192  return false;
193  }
194  value_t xold = getX(), yold = getY();
195  mAlpha = alpha;
196  mX = xold * ca + yold * sa;
197  mP[kY] = -xold * sa + yold * ca;
198  mP[kSnp] = tmp;
199  return true;
200 }

203 template <typename value_T>
204 GPUd() bool TrackParametrization<value_T>::propagateParamTo(value_t xk, const dim3_t& b)
205 {
206  //----------------------------------------------------------------
207  // Extrapolate this track params (w/o cov matrix) to the plane X=xk in the field b[].
208  //
209  // X [cm] is in the "tracking coordinate system" of this track.
210  // b[]={Bx,By,Bz} [kG] is in the Global coordidate system.
211  //----------------------------------------------------------------
212 
213  value_t dx = xk - getX();
214  if (math_utils::detail::abs<value_T>(dx) < constants::math::Almost0) {
215  return true;
216  }
217  // Do not propagate tracks outside the ALICE detector
218  if (math_utils::detail::abs<value_T>(dx) > 1e5 || math_utils::detail::abs<value_T>(getY()) > 1e5 || math_utils::detail::abs<value_T>(getZ()) > 1e5) {
219  LOGP(warning, "Anomalous track, target X:{:f}", xk);
220  return false;
221  }
222  value_t crv = getCurvature(b[2]);
223  if (crv == 0.) {
224  return propagateParamTo(xk, 0.); // for the straight-line propagation use 1D field method
225  }
226 
227  value_t x2r = crv * dx;
228  value_t f1 = getSnp(), f2 = f1 + x2r;
229  if (math_utils::detail::abs<value_T>(f1) > constants::math::Almost1 || math_utils::detail::abs<value_T>(f2) > constants::math::Almost1) {
230  return false;
231  }
232  value_t r1 = math_utils::detail::sqrt<value_T>((1.f - f1) * (1.f + f1));
233  if (math_utils::detail::abs<value_T>(r1) < constants::math::Almost0) {
234  return false;
235  }
236  value_t r2 = math_utils::detail::sqrt<value_T>((1.f - f2) * (1.f + f2));
237  if (math_utils::detail::abs<value_T>(r2) < constants::math::Almost0) {
238  return false;
239  }
240  value_t dy2dx = (f1 + f2) / (r1 + r2);
241  value_t step = (math_utils::detail::abs<value_T>(x2r) < 0.05f) ? dx * math_utils::detail::abs<value_T>(r2 + f2 * dy2dx) // chord
242  : 2.f * CAMath::ASin(0.5f * dx * math_utils::detail::sqrt<value_T>(1.f + dy2dx * dy2dx) * crv) / crv; // arc
243  step *= math_utils::detail::sqrt<value_T>(1.f + getTgl() * getTgl());
244  //
245  // get the track x,y,z,px/p,py/p,pz/p,p,sinAlpha,cosAlpha in the Global System
246  gpu::gpustd::array<value_t, 9> vecLab{0.f};
247  if (!getPosDirGlo(vecLab)) {
248  return false;
249  }
250 
251  // rotate to the system where Bx=By=0.
252  value_t bxy2 = b[0] * b[0] + b[1] * b[1];
253  value_t bt = math_utils::detail::sqrt<value_T>(bxy2);
254  value_t cosphi = 1.f, sinphi = 0.f;
255  if (bt > constants::math::Almost0) {
256  cosphi = b[0] / bt;
257  sinphi = b[1] / bt;
258  }
259  value_t bb = math_utils::detail::sqrt<value_T>(bxy2 + b[2] * b[2]);
260  value_t costet = 1.f, sintet = 0.f;
261  if (bb > constants::math::Almost0) {
262  costet = b[2] / bb;
263  sintet = bt / bb;
264  }
265  gpu::gpustd::array<value_t, 7> vect{costet * cosphi * vecLab[0] + costet * sinphi * vecLab[1] - sintet * vecLab[2],
266  -sinphi * vecLab[0] + cosphi * vecLab[1],
267  sintet * cosphi * vecLab[0] + sintet * sinphi * vecLab[1] + costet * vecLab[2],
268  costet * cosphi * vecLab[3] + costet * sinphi * vecLab[4] - sintet * vecLab[5],
269  -sinphi * vecLab[3] + cosphi * vecLab[4],
270  sintet * cosphi * vecLab[3] + sintet * sinphi * vecLab[4] + costet * vecLab[5],
271  vecLab[6]};
272 
273  // Do the helix step
274  value_t q = getCharge();
275  g3helx3(q * bb, step, vect);
276 
277  // rotate back to the Global System
278  vecLab[0] = cosphi * costet * vect[0] - sinphi * vect[1] + cosphi * sintet * vect[2];
279  vecLab[1] = sinphi * costet * vect[0] + cosphi * vect[1] + sinphi * sintet * vect[2];
280  vecLab[2] = -sintet * vect[0] + costet * vect[2];
281 
282  vecLab[3] = cosphi * costet * vect[3] - sinphi * vect[4] + cosphi * sintet * vect[5];
283  vecLab[4] = sinphi * costet * vect[3] + cosphi * vect[4] + sinphi * sintet * vect[5];
284  vecLab[5] = -sintet * vect[3] + costet * vect[5];
285 
286  // rotate back to the Tracking System
287  value_t sinalp = -vecLab[7], cosalp = vecLab[8];
288  value_t t = cosalp * vecLab[0] - sinalp * vecLab[1];
289  vecLab[1] = sinalp * vecLab[0] + cosalp * vecLab[1];
290  vecLab[0] = t;
291  t = cosalp * vecLab[3] - sinalp * vecLab[4];
292  vecLab[4] = sinalp * vecLab[3] + cosalp * vecLab[4];
293  vecLab[3] = t;
294 
295  // Do the final correcting step to the target plane (linear approximation)
296  value_t x = vecLab[0], y = vecLab[1], z = vecLab[2];
297  if (math_utils::detail::abs<value_T>(dx) > constants::math::Almost0) {
298  if (math_utils::detail::abs<value_T>(vecLab[3]) < constants::math::Almost0) {
299  return false;
300  }
301  dx = xk - vecLab[0];
302  x += dx;
303  y += vecLab[4] / vecLab[3] * dx;
304  z += vecLab[5] / vecLab[3] * dx;
305  }
306 
307  // Calculate the track parameters
308  t = 1.f / math_utils::detail::sqrt<value_T>(vecLab[3] * vecLab[3] + vecLab[4] * vecLab[4]);
309  mX = xk;
310  mP[kY] = y;
311  mP[kZ] = z;
312  mP[kSnp] = vecLab[4] * t;
313  mP[kTgl] = vecLab[5] * t;
314  mP[kQ2Pt] = q * t / vecLab[6];
315 
316  return true;
317 }

320 template <typename value_T>
321 GPUd() bool TrackParametrization<value_T>::propagateParamTo(value_t xk, value_t b)
322 {
323  //----------------------------------------------------------------
324  // propagate this track to the plane X=xk (cm) in the field "b" (kG)
325  // Only parameters are propagated, not the matrix. To be used for small
326  // distances only (<mm, i.e. misalignment)
327  //----------------------------------------------------------------
328  value_t dx = xk - getX();
329  if (math_utils::detail::abs<value_T>(dx) < constants::math::Almost0) {
330  return true;
331  }
332  value_t crv = (math_utils::detail::abs<value_T>(b) < constants::math::Almost0) ? 0.f : getCurvature(b);
333  value_t x2r = crv * dx;
334  value_t f1 = getSnp(), f2 = f1 + x2r;
335  if ((math_utils::detail::abs<value_T>(f1) > constants::math::Almost1) || (math_utils::detail::abs<value_T>(f2) > constants::math::Almost1)) {
336  return false;
337  }
338  value_t r1 = math_utils::detail::sqrt<value_T>((1.f - f1) * (1.f + f1));
339  if (math_utils::detail::abs<value_T>(r1) < constants::math::Almost0) {
340  return false;
341  }
342  value_t r2 = math_utils::detail::sqrt<value_T>((1.f - f2) * (1.f + f2));
343  if (math_utils::detail::abs<value_T>(r2) < constants::math::Almost0) {
344  return false;
345  }
346  mX = xk;
347  double dy2dx = (f1 + f2) / (r1 + r2);
348  mP[kY] += dx * dy2dx;
349  mP[kSnp] += x2r;
350  if (math_utils::detail::abs<value_T>(x2r) < 0.05f) {
351  mP[kZ] += dx * (r2 + f2 * dy2dx) * getTgl();
352  } else {
353  // for small dx/R the linear apporximation of the arc by the segment is OK,
354  // but at large dx/R the error is very large and leads to incorrect Z propagation
355  // angle traversed delta = 2*asin(dist_start_end / R / 2), hence the arc is: R*deltaPhi
356  // The dist_start_end is obtained from sqrt(dx^2+dy^2) = x/(r1+r2)*sqrt(2+f1*f2+r1*r2)
357  // double chord = dx*TMath::Sqrt(1+dy2dx*dy2dx); // distance from old position to new one
358  // double rot = 2*TMath::ASin(0.5*chord*crv); // angular difference seen from the circle center
359  // track1 += rot/crv*track3;
360  //
361  value_t rot = CAMath::ASin(r1 * f2 - r2 * f1); // more economic version from Yura.
362  if (f1 * f1 + f2 * f2 > 1.f && f1 * f2 < 0.f) { // special cases of large rotations or large abs angles
363  if (f2 > 0.f) {
364  rot = constants::math::PI - rot; //
365  } else {
366  rot = -constants::math::PI - rot;
367  }
368  }
369  mP[kZ] += getTgl() / crv * rot;
370  }
371  return true;
372 }

375 template <typename value_T>
376 GPUd() bool TrackParametrization<value_T>::propagateParamToDCA(const math_utils::Point3D<value_t>& vtx, value_t b, dim2_t* dca, value_t maxD)
377 {
378  // propagate track to DCA to the vertex
379  value_t sn, cs, alp = getAlpha();
380  math_utils::detail::sincos(alp, sn, cs);
381  value_t x = getX(), y = getY(), snp = getSnp(), csp = math_utils::detail::sqrt<value_T>((1.f - snp) * (1.f + snp));
382  value_t xv = vtx.X() * cs + vtx.Y() * sn, yv = -vtx.X() * sn + vtx.Y() * cs, zv = vtx.Z();
383  x -= xv;
384  y -= yv;
385  //Estimate the impact parameter neglecting the track curvature
386  value_t d = math_utils::detail::abs<value_T>(x * snp - y * csp);
387  if (d > maxD) {
388  return false;
389  }
390  value_t crv = getCurvature(b);
391  value_t tgfv = -(crv * x - snp) / (crv * y + csp);
392  sn = tgfv / math_utils::detail::sqrt<value_T>(1.f + tgfv * tgfv);
393  cs = math_utils::detail::sqrt<value_T>((1.f - sn) * (1.f + sn));
394  cs = (math_utils::detail::abs<value_T>(tgfv) > constants::math::Almost0) ? sn / tgfv : constants::math::Almost1;
395 
396  x = xv * cs + yv * sn;
397  yv = -xv * sn + yv * cs;
398  xv = x;
399 
400  auto tmpT(*this); // operate on the copy to recover after the failure
401  alp += math_utils::detail::asin<value_T>(sn);
402  if (!tmpT.rotateParam(alp) || !tmpT.propagateParamTo(xv, b)) {
403 #ifndef GPUCA_ALIGPUCODE
404  LOG(debug) << "failed to propagate to alpha=" << alp << " X=" << xv << " for vertex "
405  << vtx.X() << ' ' << vtx.Y() << ' ' << vtx.Z() << " | Track is: " << tmpT.asString();
406 #else
407  LOG(debug) << "failed to propagate to alpha=" << alp << " X=" << xv << " for vertex " << vtx.X() << ' ' << vtx.Y() << ' ' << vtx.Z();
408 #endif
409  return false;
410  }
411  *this = tmpT;
412  if (dca) {
413  (*dca)[0] = getY() - yv;
414  (*dca)[1] = getZ() - zv;
415  }
416  return true;
417 }

420 template <typename value_T>
421 GPUd() bool TrackParametrization<value_T>::getYZAt(value_t xk, value_t b, value_t& y, value_t& z) const
422 {
423  //----------------------------------------------------------------
424  // estimate Y,Z in tracking frame at given X
425  //----------------------------------------------------------------
426  value_t dx = xk - getX();
427  y = mP[kY];
428  z = mP[kZ];
429  if (math_utils::detail::abs<value_T>(dx) < constants::math::Almost0) {
430  return true;
431  }
432  value_t crv = getCurvature(b);
433  value_t x2r = crv * dx;
434  value_t f1 = getSnp(), f2 = f1 + x2r;
435  if ((math_utils::detail::abs<value_T>(f1) > constants::math::Almost1) || (math_utils::detail::abs<value_T>(f2) > constants::math::Almost1)) {
436  return false;
437  }
438  value_t r1 = math_utils::detail::sqrt<value_T>((1.f - f1) * (1.f + f1));
439  if (math_utils::detail::abs<value_T>(r1) < constants::math::Almost0) {
440  return false;
441  }
442  value_t r2 = math_utils::detail::sqrt<value_T>((1.f - f2) * (1.f + f2));
443  if (math_utils::detail::abs<value_T>(r2) < constants::math::Almost0) {
444  return false;
445  }
446  double dy2dx = (f1 + f2) / (r1 + r2);
447  y += dx * dy2dx;
448  if (math_utils::detail::abs<value_T>(x2r) < 0.05f) {
449  z += dx * (r2 + f2 * dy2dx) * getTgl();
450  } else {
451  // for small dx/R the linear apporximation of the arc by the segment is OK,
452  // but at large dx/R the error is very large and leads to incorrect Z propagation
453  // angle traversed delta = 2*asin(dist_start_end / R / 2), hence the arc is: R*deltaPhi
454  // The dist_start_end is obtained from sqrt(dx^2+dy^2) = x/(r1+r2)*sqrt(2+f1*f2+r1*r2)
455  // double chord = dx*TMath::Sqrt(1+dy2dx*dy2dx); // distance from old position to new one
456  // double rot = 2*TMath::ASin(0.5*chord*crv); // angular difference seen from the circle center
457  // track1 += rot/crv*track3;
458  //
459  value_t rot = CAMath::ASin(r1 * f2 - r2 * f1); // more economic version from Yura.
460  if (f1 * f1 + f2 * f2 > 1.f && f1 * f2 < 0.f) { // special cases of large rotations or large abs angles

527 template <typename value_T>
528 GPUd() bool TrackParametrization<value_T>::getXatLabR(value_t r, value_t& x, value_t bz, track::DirType dir) const
529 {
530  // Get local X of the track position estimated at the radius lab radius r.
531  // The track curvature is accounted exactly
532  //
533  // The flag "dir" can be used to remove the ambiguity of which intersection to take (out of 2 possible)
534  // DirAuto (==0) - take the intersection closest to the current track position
535  // DirOutward (==1) - go along the track (increasing mX)
536  // DirInward (==-1) - go backward (decreasing mX)
537  //
538  const auto fy = mP[0], sn = mP[2];
539  const value_t kEps = 1.e-6;
540  //
541  auto crv = getCurvature(bz);
542  if (math_utils::detail::abs<value_T>(crv) > constants::math::Almost0) { // helix
543  // get center of the track circle
544  math_utils::CircleXY<value_t> circle;
545  getCircleParamsLoc(bz, circle);
546  value_t r0 = math_utils::detail::sqrt<value_T>(circle.getCenterD2());
547  if (r0 <= constants::math::Almost0) {
548  return false; // the track is concentric to circle
549  }
550  value_t tR2r0 = 1.f, g = 0.f, tmp = 0.f;
551  if (math_utils::detail::abs<value_T>(circle.rC - r0) > kEps) {
552  tR2r0 = circle.rC / r0;
553  g = 0.5f * (r * r / (r0 * circle.rC) - tR2r0 - 1.f / tR2r0);
554  tmp = 1.f + g * tR2r0;
555  } else {
556  tR2r0 = 1.0;
557  g = 0.5f * r * r / (r0 * circle.rC) - 1.f;
558  tmp = 0.5f * r * r / (r0 * r0);
559  }
560  value_t det = (1.f - g) * (1.f + g);
561  if (det < 0.f) {
562  return false; // does not reach raduis r
563  }
564  det = math_utils::detail::sqrt<value_T>(det);
565  //
566  // the intersection happens in 2 points: {circle.xC+tR*C,circle.yC+tR*S}
567  // with C=f*c0+-|s0|*det and S=f*s0-+c0 sign(s0)*det
568  // where s0 and c0 make direction for the circle center (=circle.xC/r0 and circle.yC/r0)
569  //
570  x = circle.xC * tmp;
571  value_t y = circle.yC * tmp;
572  if (math_utils::detail::abs<value_T>(circle.yC) > constants::math::Almost0) { // when circle.yC==0 the x,y is unique
573  value_t dfx = tR2r0 * math_utils::detail::abs<value_T>(circle.yC) * det;
574  value_t dfy = tR2r0 * circle.xC * (circle.yC > 0.f ? det : -det);
575  if (dir == DirAuto) { // chose the one which corresponds to smallest step
576  value_t delta = (x - mX) * dfx - (y - fy) * dfy; // the choice of + in C will lead to smaller step if delta<0
577  x += delta < 0.f ? dfx : -dfx;
578  } else if (dir == DirOutward) { // along track direction: x must be > mX
579  x -= dfx; // try the smallest step (dfx is positive)
580  value_t dfeps = mX - x; // handle special case of very small step
581  if (dfeps < -kEps) {
582  return true;
583  }
584  if (math_utils::detail::abs<value_T>(dfeps) < kEps && math_utils::detail::abs<value_T>(mX * mX + fy * fy - r * r) < kEps) { // are we already in right r?
585  return mX;
586  }
587  x += dfx + dfx;
588  value_t dxm = x - mX;
589  if (dxm > 0.f) {
590  return true;
591  } else if (dxm < -kEps) {
592  return false;
593  }
594  x = mX; // don't move
595  } else { // backward: x must be < mX
596  x += dfx; // try the smallest step (dfx is positive)
597  value_t dfeps = x - mX; // handle special case of very small step
598  if (dfeps < -kEps) {
599  return true;
600  }
601  if (math_utils::detail::abs<value_T>(dfeps) < kEps && math_utils::detail::abs<value_T>(mX * mX + fy * fy - r * r) < kEps) { // are we already in right r?
602  return mX;
603  }
604  x -= dfx + dfx;
605  value_t dxm = x - mX;
606  if (dxm < 0.f) {
607  return true;
608  }
609  if (dxm > kEps) {
610  return false;
611  }
612  x = mX; // don't move
613  }
614  } else { // special case: track touching the circle just in 1 point
615  if ((dir == DirOutward && x < mX) || (dir == DirInward && x > mX)) {
616  return false;
617  }
618  }
619  } else { // this is a straight track
620  if (math_utils::detail::abs<value_T>(sn) >= constants::math::Almost1) { // || to Y axis
621  value_t det = (r - mX) * (r + mX);
622  if (det < 0.f) {
623  return false; // does not reach raduis r
624  }
625  x = mX;
626  if (dir == DirAuto) {
627  return true;
628  }
629  det = math_utils::detail::sqrt<value_T>(det);
630  if (dir == DirOutward) { // along the track direction
631  if (sn > 0.f) {
632  if (fy > det) {
633  return false; // track is along Y axis and above the circle
634  }
635  } else {
636  if (fy < -det) {
637  return false; // track is against Y axis amd belo the circle
638  }
639  }
640  } else if (dir == DirInward) { // against track direction
641  if (sn > 0.f) {
642  if (fy < -det) {
643  return false; // track is along Y axis
644  }
645  } else if (fy > det) {
646  return false; // track is against Y axis
647  }
648  }
649  } else if (math_utils::detail::abs<value_T>(sn) <= constants::math::Almost0) { // || to X axis
650  value_t det = (r - fy) * (r + fy);
651  if (det < 0.f) {
652  return false; // does not reach raduis r
653  }
654  det = math_utils::detail::sqrt<value_T>(det);
655  if (dir == DirAuto) {
656  x = mX > 0.f ? det : -det; // choose the solution requiring the smalest step
657  return true;
658  } else if (dir == DirOutward) { // along the track direction
659  if (mX > det) {
660  return false; // current point is in on the right from the circle
661  } else {
662  x = (mX < -det) ? -det : det; // on the left : within the circle
663  }
664  } else { // against the track direction
665  if (mX < -det) {
666  return false;
667  } else {
668  x = mX > det ? det : -det;
669  }
670  }
671  } else { // general case of straight line
672  value_t cs = math_utils::detail::sqrt<value_T>((1.f - sn) * (1.f + sn));
673  value_t xsyc = mX * sn - fy * cs;
674  value_t det = (r - xsyc) * (r + xsyc);
675  if (det < 0.f) {
676  return false; // does not reach raduis r
677  }
678  det = math_utils::detail::sqrt<value_T>(det);
679  value_t xcys = mX * cs + fy * sn;
680  value_t t = -xcys;
681  if (dir == DirAuto) {
682  t += t > 0.f ? -det : det; // chose the solution requiring the smalest step
683  } else if (dir > 0) { // go in increasing mX direction. ( t+-det > 0)
684  if (t >= -det) {
685  t += -det; // take minimal step giving t>0
686  } else {
687  return false; // both solutions have negative t
688  }
689  } else { // go in increasing mX direction. (t+-det < 0)
690  if (t < det) {
691  t -= det; // take minimal step giving t<0
692  } else {
693  return false; // both solutions have positive t
694  }
695  }
696  x = mX + cs * t;
697  }
698  }
699  //
700  return true;
701 }

704 template <typename value_T>
705 GPUd() bool TrackParametrization<value_T>::correctForELoss(value_t xrho, bool anglecorr, value_t dedx)
706 {
707  //------------------------------------------------------------------
708  // This function corrects the track parameters for the energy loss in crossed material.
709  // "xrho" - is the product length*density (g/cm^2).
710  // It should be passed as negative when propagating tracks
711  // from the intreaction point to the outside of the central barrel.
712  // "dedx" - mean enery loss (GeV/(g/cm^2), if <=kCalcdEdxAuto : calculate on the fly
713  // "anglecorr" - switch for the angular correction
714  //------------------------------------------------------------------
715  constexpr value_t kMaxELossFrac = 0.3f; // max allowed fractional eloss
716  constexpr value_t kMinP = 0.01f; // kill below this momentum
717 
718  // Apply angle correction, if requested
719  if (anglecorr) {
720  value_t csp2 = (1.f - getSnp()) * (1.f + getSnp()); // cos(phi)^2
721  value_t cst2I = (1.f + getTgl() * getTgl()); // 1/cos(lambda)^2
722  value_t angle = math_utils::detail::sqrt<value_T>(cst2I / (csp2));
723  xrho *= angle;
724  }
725  value_t p = getP();
726  value_t p2 = p * p;
727  value_t e2 = p2 + getPID().getMass2();
728  value_t beta2 = p2 / e2;
729 
730  // Calculating the energy loss corrections************************
731  if ((xrho != 0.f) && (beta2 < 1.f)) {
732  if (dedx < kCalcdEdxAuto + constants::math::Almost1) { // request to calculate dedx on the fly
733  dedx = BetheBlochSolid(p / getPID().getMass());
734  if (mAbsCharge != 1) {
735  dedx *= mAbsCharge * mAbsCharge;
736  }
737  }
738 
739  value_t dE = dedx * xrho;
740  value_t e = math_utils::detail::sqrt<value_T>(e2);
741  if (math_utils::detail::abs<value_T>(dE) > kMaxELossFrac * e) {
742  return false; // 30% energy loss is too much!
743  }
744  value_t eupd = e + dE;
745  value_t pupd2 = eupd * eupd - getPID().getMass2();
746  if (pupd2 < kMinP * kMinP) {
747  return false;
748  }
749  setQ2Pt(getQ2Pt() * p / math_utils::detail::sqrt<value_T>(pupd2));
750  }
751 
752  return true;
753 }
