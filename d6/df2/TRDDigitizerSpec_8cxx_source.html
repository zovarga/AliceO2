55  void run(framework::ProcessingContext& pc)
56  {
57  static bool finished = false;
58  if (finished) {
59  return;
60  }
61  LOG(info) << "Doing TRD digitization";
62 
63  bool mctruth = pc.outputs().isAllowed({"TRD", "LABELS", 0});
64 
65  Calibrations simcal;
66  // the timestamp can be extracted from the DPL header (it is set in SimReader)
67  const auto ref = pc.inputs().getFirstValid(true);
68  auto creationTime = DataRefUtils::getHeader<DataProcessingHeader*>(ref)->creation;
69  simcal.getCCDBObjects(creationTime);
70  mDigitizer.setCalibrations(&simcal);
71 
72  // read collision context from input
73  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
74  context->initSimChains(o2::detectors::DetID::TRD, mSimChains);
75  auto& irecords = context->getEventRecords();
76 
77  for (auto& record : irecords) {
78  LOG(debug) << "TRD TIME RECEIVED " << record.getTimeNS();
79  }
80 
81  auto& eventParts = context->getEventParts();
82  std::vector<o2::trd::Digit> digitsAccum; // accumulator for digits
83  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labelsAccum;
84  std::vector<TriggerRecord> triggers;
85 
86  std::vector<o2::trd::Digit> digits; // digits which get filled
87  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labels; // labels which get filled
88 
89  o2::InteractionTimeRecord currentTime; // the current time
90  o2::InteractionTimeRecord triggerTime; // the time at which the TRD start reading out a signal
91  bool firstEvent = true; // Flag for the first event processed
92 
93  TStopwatch timer;
94  timer.Start();
95  // loop over all composite collisions given from context
96  // (aka loop over all the interaction records)
97  for (int collID = 0; collID < irecords.size(); ++collID) {
98  currentTime = irecords[collID];
99  // Trigger logic implemented here
100  bool isNewTrigger = true; // flag newly accepted readout trigger
101  if (firstEvent) {
102  triggerTime = currentTime;
103  firstEvent = false;
104  } else {
105  double dT = currentTime.getTimeNS() - triggerTime.getTimeNS();
106  if (dT < o2::trd::constants::BUSY_TIME) {
107  // BUSY_TIME = READOUT_TIME + DEAD_TIME, if less than that, pile up the signals and update the last time
108  isNewTrigger = false;
109  mDigitizer.pileup();
110  } else {
111  // A new signal can be received, and the detector read it out:
112  // flush previous stored digits, labels and keep a trigger record
113  // then update the trigger time to the new one
114  mDigitizer.flush(digits, labels);
115  assert(digits.size() == labels.getIndexedSize());
116  // Add trigger record, and send digits to the accumulator
117  triggers.emplace_back(triggerTime, digitsAccum.size(), digits.size());
118  std::copy(digits.begin(), digits.end(), std::back_inserter(digitsAccum));
119  if (mctruth) {
120  labelsAccum.mergeAtBack(labels);
121  }
122  triggerTime = currentTime;
123  digits.clear();
124  labels.clear();
125  }
126  }
127 
128  mDigitizer.setEventTime(triggerTime.getTimeNS());
129  if (isNewTrigger) {
130  mDigitizer.setTriggerTime(triggerTime.getTimeNS());
131  }
132 
133  // for each collision, loop over the constituents event and source IDs
134  // (background signal merging is basically taking place here)
135  for (auto& part : eventParts[collID]) {
136  mDigitizer.setEventID(part.entryID);
137  mDigitizer.setSrcID(part.sourceID);
138  // get the hits for this event and this source and process them
139  std::vector<o2::trd::Hit> hits;
140  context->retrieveHits(mSimChains, "TRDHit", part.sourceID, part.entryID, &hits);
141  mDigitizer.process(hits);
142  }
143  }
144 
145  // Force flush of the digits that remain in the digitizer cache
146  mDigitizer.flush(digits, labels);
147  assert(digits.size() == labels.getIndexedSize());
148  triggers.emplace_back(triggerTime, digitsAccum.size(), digits.size());
149  std::copy(digits.begin(), digits.end(), std::back_inserter(digitsAccum));
150  if (mctruth) {
151  labelsAccum.mergeAtBack(labels);
152  }
153  LOGF(info, "List of TRD chambers with at least one drift velocity out of range: %s", mDigitizer.dumpFlaggedChambers());
154  timer.Stop();
155  LOGF(info, "TRD digitization timing: Cpu: %.3e Real: %.3e s", timer.CpuTime(), timer.RealTime());
156 
157  LOG(info) << "TRD: Sending " << digitsAccum.size() << " digits";
158  pc.outputs().snapshot(Output{"TRD", "DIGITS", 0, Lifetime::Timeframe}, digitsAccum);
159  if (mctruth) {
160  LOG(info) << "TRD: Sending " << labelsAccum.getNElements() << " labels";
161  // we are flattening the labels and write to managed shared memory container for further communication
162  auto& sharedlabels = pc.outputs().make<o2::dataformats::ConstMCTruthContainer<o2::MCCompLabel>>(Output{"TRD", "LABELS", 0, Lifetime::Timeframe});
163  labelsAccum.flatten_to(sharedlabels);
164  }
165  LOG(info) << "TRD: Sending ROMode= " << mROMode << " to GRPUpdater";
166  pc.outputs().snapshot(Output{"TRD", "ROMode", 0, Lifetime::Timeframe}, mROMode);
167  LOG(info) << "TRD: Sending trigger records";
168  pc.outputs().snapshot(Output{"TRD", "TRGRDIG", 0, Lifetime::Timeframe}, triggers);
169  // we should be only called once; tell DPL that this process is ready to exit
170  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
171  finished = true;
172  }
173 
174  private:
175  Digitizer mDigitizer;
176  std::vector<TChain*> mSimChains;
177  // RS: at the moment using hardcoded flag for continuos readout
178  o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::PRESENT; // readout mode
179 }; // namespace trd

181 o2::framework::DataProcessorSpec getTRDDigitizerSpec(int channel, bool mctruth)
182 {
183  // create the full data processor spec using
184  // a name identifier
185  // input description
186  // algorithmic description (here a lambda getting called once to setup the actual processing function)
187  // options that can be used for this processor (here: input file names where to take the hits)
188  std::vector<OutputSpec> outputs;
189  outputs.emplace_back("TRD", "DIGITS", 0, Lifetime::Timeframe);
190  outputs.emplace_back("TRD", "TRGRDIG", 0, Lifetime::Timeframe);
191  if (mctruth) {

194  outputs.emplace_back("TRD", "ROMode", 0, Lifetime::Timeframe);
195 
196  return DataProcessorSpec{

198  Inputs{InputSpec{"collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe}},
199 
200  outputs,
201 
202  AlgorithmSpec{adaptFromTask<TRDDPLDigitizerTask>()},
