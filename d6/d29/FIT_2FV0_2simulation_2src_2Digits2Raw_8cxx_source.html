23 void Digits2Raw::readDigits(const std::string& outDir, const std::string& fileDigitsName)
24 {
25  LOG(info) << "==============FV0: Digits2Raw::convertDigits" << std::endl;
26  LookUpTable lut(true);
27 
28  std::string outd = outDir;
29  if (outd.back() != '/') {
30  outd += '/';
31  }
32 
33  // Register PM links linearly
34  for (int iPmLink = 0; iPmLink < Constants::nPms; ++iPmLink) {
35  uint64_t feeId = uint64_t(iPmLink);
36  uint32_t linkId = uint32_t(iPmLink);
37  std::string outFileLink = mOutputPerLink ? (outd + "fv0_link" + std::to_string(iPmLink) + ".raw") : (outd + "fv0.raw");
38  LOG(info) << " Register PM link: " << iPmLink << " to file: " << outFileLink;
39  mWriter.registerLink(feeId, sCruId, linkId, sEndPointId, outFileLink);
40  }
41 
42  // Register TCM link separately
43  std::string outFileLink = mOutputPerLink ? (outd + "fv0_link" + std::to_string(sTcmLink) + ".raw") : (outd + "fv0.raw");
44  LOG(info) << " Register TCM link: " << outFileLink;
45  mWriter.registerLink(uint64_t(sTcmLink), sCruId, sTcmLink, sEndPointId, outFileLink);
46 
47  TFile* fdig = TFile::Open(fileDigitsName.data());
48  assert(fdig != nullptr);
49  LOG(info) << "Open digits file: " << fileDigitsName.data();
50 
51  TTree* digTree = (TTree*)fdig->Get("o2sim");
52  std::vector<o2::fv0::BCData> digitsBC, *fv0BCDataPtr = &digitsBC;
53  std::vector<o2::fv0::ChannelData> digitsCh, *fv0ChDataPtr = &digitsCh;
54  digTree->SetBranchAddress("FV0DigitBC", &fv0BCDataPtr);
55  digTree->SetBranchAddress("FV0DigitCh", &fv0ChDataPtr);
56 
57  for (int ient = 0; ient < digTree->GetEntries(); ient++) {
58  digTree->GetEntry(ient);
59  int nbc = digitsBC.size();
60  for (int ibc = 0; ibc < nbc; ibc++) {
61  auto& bcd = digitsBC[ibc];
62  auto channels = bcd.getBunchChannelData(digitsCh);
63  if (!channels.empty()) {
64  LOG(debug) << "o2::fv0::Digits2Raw::readDigits(): Start to convertDigits() at ibc = " << ibc << " " << bcd.ir
65  << " iCh0:" << bcd.ref.getFirstEntry() << " nentries:" << bcd.ref.getEntries();
66  convertDigits(bcd, channels, lut);
67  }
68  }
69  }
70 }

72 void Digits2Raw::convertDigits(o2::fv0::BCData bcdigits, gsl::span<const ChannelData> pmchannels,
73  const o2::fv0::LookUpTable& lut)
74 {
75  const o2::InteractionRecord intRecord = bcdigits.getIntRecord();
76  int prevPmLink = -1;
77  int iChannelPerLink = 0;
78  int nch = pmchannels.size();
79  std::stringstream ss;
80  ss << " Number of channels: " << nch << " (Ch, PMT, Q, T)\n";
81  for (int ich = 0; ich < nch; ich++) {
82  if (pmchannels[ich].chargeAdc != 0) {
83  ss << " " << std::setw(2) << ich
84  << std::setw(3) << pmchannels[ich].pmtNumber
85  << std::setw(5) << pmchannels[ich].chargeAdc
86  << std::setw(7) << std::setprecision(3) << pmchannels[ich].time << "\n";
87  }
88  }
89  LOG(debug) << ss.str().substr(0, ss.str().size() - 1);
90 
91  for (int ich = 0; ich < nch; ich++) {
92  int nLinkPm = lut.getLink(pmchannels[ich].pmtNumber);
93  if (nLinkPm != prevPmLink) {
94  if (prevPmLink >= 0) {
95  fillSecondHalfWordAndAddData(iChannelPerLink, prevPmLink, intRecord);
96  }
97  makeGBTHeader(mRawEventData.mEventHeader, nLinkPm, intRecord);
98  iChannelPerLink = 0;
99  prevPmLink = nLinkPm;
100  }
101  if (pmchannels[ich].chargeAdc != 0) {
102  LOG(debug) << " Store data for channel: " << ich << " PmLink = " << nLinkPm << " ";
103  auto& newData = mRawEventData.mEventData[iChannelPerLink];
104  newData.charge = pmchannels[ich].chargeAdc;
105  newData.time = pmchannels[ich].time;
106  newData.generateFlags();
107  newData.channelID = lut.getPmChannel(pmchannels[ich].pmtNumber);
108  iChannelPerLink++;
109  }
110  if (ich == nch - 1) {
111  fillSecondHalfWordAndAddData(iChannelPerLink, prevPmLink, intRecord);
112  }
113  }
114 
115  // TCM
116  makeGBTHeader(mRawEventData.mEventHeader, sTcmLink, intRecord);
117  mRawEventData.mEventHeader.nGBTWords = 1;
118  auto& tcmdata = mRawEventData.mTCMdata;
119  /*
120  tcmdata.orA = 1;
121  tcmdata.orC = 0;
122  tcmdata.sCen = 0;
123  tcmdata.cen = 0;
124  tcmdata.vertex = 1;
125  */
126  tcmdata.nChanA = 0;
127  tcmdata.nChanC = 0;
128  tcmdata.amplA = 0;
129 
130  tcmdata.amplC = 0;
131  tcmdata.timeA = 0;
132  tcmdata.timeC = 0;
133  //Temporary for tests
134  tcmdata.orA = bool(bcdigits.mTriggers.triggerSignals & (1 << 0));
135  tcmdata.orC = bool(bcdigits.mTriggers.triggerSignals & (1 << 1));
136  tcmdata.sCen = bool(bcdigits.mTriggers.triggerSignals & (1 << 2));
137  tcmdata.cen = bool(bcdigits.mTriggers.triggerSignals & (1 << 3));
138  tcmdata.vertex = bool(bcdigits.mTriggers.triggerSignals & (1 << 4));
139  tcmdata.laser = bool(bcdigits.mTriggers.triggerSignals & (1 << 5));
140  tcmdata.nChanA = bcdigits.mTriggers.nChanA;
141  tcmdata.amplA = bcdigits.mTriggers.amplA;
142  //
143  auto data = mRawEventData.to_vector(kTRUE); //for tcm module
144  uint32_t linkId = uint32_t(sTcmLink);
145  uint64_t feeId = uint64_t(sTcmLink);
146  mWriter.addData(feeId, sCruId, linkId, sEndPointId, intRecord, data);
147 
148  // fill mEventData[iChannelPerLink] with 0s to flag that this is a dummy data
149  uint nGBTWords = uint((iChannelPerLink + 1) / 2);
150  if ((iChannelPerLink % 2) == 1) {
151  mRawEventData.mEventData[iChannelPerLink] = {};
152  }
153  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
154  // LOG(debug) << " last link: " << prevPmLink;
155 }

169 void Digits2Raw::fillSecondHalfWordAndAddData(int iChannelPerLink, int prevPmLink, const o2::InteractionRecord& ir)
170 {
171  uint nGBTWords = uint((iChannelPerLink + 1) / 2);
172  if ((iChannelPerLink % 2) == 1) {
173  mRawEventData.mEventData[iChannelPerLink] = {};
174  LOG(debug) << " Fill up empty second half-word.";
175  }
176  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
177  auto data = mRawEventData.to_vector(false);
178  uint32_t linkId = uint32_t(prevPmLink);
179  uint64_t feeId = uint64_t(prevPmLink);
180  mWriter.addData(feeId, sCruId, linkId, sEndPointId, ir, data);
181  LOG(debug) << " Switch prevPmLink: " << prevPmLink << ". Save data with nGBTWords="
182  << nGBTWords << " in header. Last channel: " << iChannelPerLink;
183 }
