24  LOG(info) << "Setup global cuts and processes";
25 
26  // ------>>>> IMPORTANT!!!!
27  // For a correct comparison between GEANE and MC (pull distributions)
28  // or for a simulation without the generation of secondary particles:
29  // 1. set LOSS = 2, DRAY = 0, BREM = 1
30  // 2. set the following cut values: CUTGAM, CUTELE, CUTNEU, CUTHAD, CUTMUO = 1 MeV or less
31  // BCUTE, BCUTM, DCUTE, DCUTM, PPCUTM = 10 TeV
32  // (For an explanation of the chosen values, please refer to the GEANT User's Guide
33  // or to message #5362 in the PandaRoot Forum >> Monte Carlo Engines >> g3Config.C thread)
34  //
35  // The default settings refer to a complete simulation which generates and follows also the secondary particles.
36 
37  // \note All following settings could also be set in Cave since it is always loaded.
38  // Use MaterialManager to set processes and cuts
39  auto& mgr = o2::base::MaterialManager::Instance();
40  auto& params = o2::GlobalProcessCutSimParam::Instance();
41 
42  LOG(info) << "Set default settings for processes and cuts.";
43  mgr.DefaultProcesses({{EProc::kPAIR, params.PAIR},
44  {EProc::kCOMP, params.COMP},
45  {EProc::kPHOT, params.PHOT},
46  {EProc::kPFIS, params.PFIS},
47  {EProc::kDRAY, params.DRAY},
48  {EProc::kANNI, params.ANNI},
49  {EProc::kBREM, params.BREM},
50  {EProc::kHADR, params.HADR},
51  {EProc::kMUNU, params.MUNU},
52  {EProc::kDCAY, params.DCAY},
53  {EProc::kLOSS, params.LOSS},
54  {EProc::kMULS, params.MULS},
55  {EProc::kCKOV, params.CKOV}});
57  mgr.DefaultCuts({{ECut::kCUTGAM, params.CUTGAM},
58  {ECut::kCUTELE, params.CUTELE},
59  {ECut::kCUTNEU, params.CUTNEU},
60  {ECut::kCUTHAD, params.CUTHAD},
61  {ECut::kCUTMUO, params.CUTMUO},
62  {ECut::kBCUTE, params.BCUTE},
63  {ECut::kBCUTM, params.BCUTM},
64  {ECut::kDCUTE, params.DCUTE},
65  {ECut::kDCUTM, params.DCUTM},
66  {ECut::kPPCUTM, params.PPCUTM},
67  {ECut::kTOFMAX, params.TOFMAX}});
69  const char* settingProc = mgr.specialProcessesEnabled() ? "enabled" : "disabled";
70  const char* settingCut = mgr.specialCutsEnabled() ? "enabled" : "disabled";
