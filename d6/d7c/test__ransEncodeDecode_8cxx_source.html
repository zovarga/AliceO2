70 template <

72  template <typename, typename, typename> class decoder_T,
73  typename coder_T, class dictString_T, class testString_T>
74 struct EncodeDecodeBase {
75  public:
76  using params_t = Params<coder_T>;
77 
78  EncodeDecodeBase()
79  {
80  dictString_T source;
81  std::string& s = source.data;
82  o2::rans::RenormedFrequencyTable frequencyTable = o2::rans::renorm(o2::rans::makeFrequencyTableFromSamples(std::begin(s), std::end(s)), params_t::symbolTablePrecision);
83 
84  encoder = decltype(encoder)(frequencyTable);
85  decoder = decltype(decoder)(frequencyTable);
86 
87  const auto [min, max] = [&s]() {
88  const auto [minIter, maxIter] = std::minmax_element(s.begin(), s.end());
89  const char min = minIter == s.end() ? 0 : *minIter;
90  const char max = maxIter == s.end() ? 0 : *maxIter + 1;
91  return std::make_tuple(min, max);
92  }();
93 
94  const size_t alphabetRangeBits = o2::rans::internal::numBitsForNSymbols(max - min + 1 + 1);
95 
96  BOOST_CHECK_EQUAL(encoder.getSymbolTablePrecision(), params_t::symbolTablePrecision);
97  BOOST_CHECK_EQUAL(encoder.getAlphabetRangeBits(), alphabetRangeBits);
98  BOOST_CHECK_EQUAL(encoder.getMinSymbol(), min);
99  BOOST_CHECK_EQUAL(encoder.getMaxSymbol(), max);
100 
101  BOOST_CHECK_EQUAL(decoder.getSymbolTablePrecision(), params_t::symbolTablePrecision);
102  BOOST_CHECK_EQUAL(decoder.getAlphabetRangeBits(), alphabetRangeBits);
103  BOOST_CHECK_EQUAL(decoder.getMinSymbol(), min);
104  BOOST_CHECK_EQUAL(decoder.getMaxSymbol(), max);
105  }
