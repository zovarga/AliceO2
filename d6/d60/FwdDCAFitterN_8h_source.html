33 struct FwdTrackCovI {
34  float sxx, syy, sxy, szz;
35 
36  FwdTrackCovI(const o2::track::TrackParCovFwd& trc, float zerrFactor = 1.) { set(trc, zerrFactor); }
37  FwdTrackCovI() = default;
38  void set(const o2::track::TrackParCovFwd& trc, float zerrFactor = 1)
39  {
40  float cxx = trc.getSigma2X(), cyy = trc.getSigma2Y(), cxy = trc.getSigmaXY(), czz = cyy * zerrFactor;
41  float detXY = cxx * cyy - cxy * cxy;
42  if (detXY > 0.) {
43  auto detXYI = 1. / detXY;
44  sxx = cyy * detXYI;
45  syy = cxx * detXYI;
46  sxy = -cxy * detXYI;
47  szz = 1. / czz;
48  } else {
49  throw std::runtime_error("invalid track covariance");
50  }
51  }
52 };

56 struct FwdTrackDeriv {
57  float dxdz, dydz, d2xdz2, d2ydz2;
58  FwdTrackDeriv() = default;
59  FwdTrackDeriv(const o2::track::TrackParFwd& trc, float bz) { set(trc, bz); }
60  void set(const o2::track::TrackParFwd& trc, float bz)
61  {
62  float snp = trc.getSnp(), csp = std::sqrt((1. - snp) * (1. + snp)), cspI = 1. / csp, crv2c = trc.getCurvature(bz), tgl = trc.getTanl(), tglI = 1. / tgl;
63  if (crv2c == 0.) {
64  crv2c = (trc.getCharge()) * 0.3 * bz * (-1e-3);
65  }
66 
67  dxdz = csp * tglI;
68  dydz = snp * tglI;
69  d2xdz2 = crv2c * snp * tglI * tglI;
70  d2ydz2 = -crv2c * csp * tglI * tglI;
71  }
72 };

298 template <int N, typename... Args>
299 template <class... Tr>
300 int FwdDCAFitterN<N, Args...>::process(const Tr&... args)
301 {
302 
303  static_assert(sizeof...(args) == N, "incorrect number of input tracks");
304  assign(0, args...);
305  clear();
306 
307  for (int i = 0; i < N; i++) {
308  mTrAux[i].set(*mOrigTrPtr[i], mBz);
309  }
310 
311  if (!mCrossings.set(mTrAux[0], *mOrigTrPtr[0], mTrAux[1], *mOrigTrPtr[1])) { // even for N>2 it should be enough to test just 1 loop
312  return 0; // no crossing
313  }
314 
315  if (mCrossings.nDCA == MAXHYP) { // if there are 2 candidates
316  auto dst2 = (mCrossings.xDCA[0] - mCrossings.xDCA[1]) * (mCrossings.xDCA[0] - mCrossings.xDCA[1]) +
317  (mCrossings.yDCA[0] - mCrossings.yDCA[1]) * (mCrossings.yDCA[0] - mCrossings.yDCA[1]);
318 
319  if (dst2 < mMaxDist2ToMergeSeeds) {
320  mCrossings.nDCA = 1;
321  mCrossings.xDCA[0] = 0.5 * (mCrossings.xDCA[0] + mCrossings.xDCA[1]);
322  mCrossings.yDCA[0] = 0.5 * (mCrossings.yDCA[0] + mCrossings.yDCA[1]);
323  }
324  }
325 
326  // check all crossings
327  for (int ic = 0; ic < mCrossings.nDCA; ic++) {
328  // check if radius is acceptable
329  if (mCrossings.xDCA[ic] * mCrossings.xDCA[ic] + mCrossings.yDCA[ic] * mCrossings.yDCA[ic] > mMaxR2) {
330  continue;
331  }
332 
333  mCrossIDCur = ic;
334  mCrossIDAlt = (mCrossings.nDCA == 2 && mAllowAltPreference) ? 1 - ic : -1; // works for max 2 crossings
335  mNIters[mCurHyp] = 0;
336  mTrPropDone[mCurHyp] = false;
337  mChi2[mCurHyp] = -1.;
338 
339  findZatXY_mid(mCurHyp);
340 
341  if (mUseAbsDCA ? minimizeChi2NoErr() : minimizeChi2()) {
342  mOrder[mCurHyp] = mCurHyp;
343  if (mPropagateToPCA && !FwdpropagateTracksToVertex(mCurHyp)) {
344  continue;
345  }
346  mCurHyp++;
347  }
348  }
349 
350  for (int i = mCurHyp; i--;) { // order in quality
351  for (int j = i; j--;) {
352  if (mChi2[mOrder[i]] < mChi2[mOrder[j]]) {
353  std::swap(mOrder[i], mOrder[j]);
354  }
355  }
356  }
357 
358  return mCurHyp;
359 }

362 template <int N, typename... Args>
363 bool FwdDCAFitterN<N, Args...>::FwdcalcPCACoefs()
364 {
365  //< calculate Ti matrices for global vertex decomposition to V = sum_{0<i<N} Ti pi, see EQ.T in the ref
366  if (!FwdcalcInverseWeight()) {
367  return false;
368  }
369  for (int i = N; i--;) { // build Mi*Ei matrix, with Mi = I
370  const auto& tcov = mTrcEInv[mCurHyp][i];
371  MatStd3D miei;
372 
373  miei[0][0] = tcov.sxx;
374  miei[0][1] = tcov.sxy;
375  miei[1][0] = tcov.sxy;
376  miei[1][1] = tcov.syy;
377  miei[2][2] = tcov.szz;
378 
379  mTrCFVT[mCurHyp][i] = mWeightInv * miei;
380  }
381  return true;
382 }

385 template <int N, typename... Args>
386 bool FwdDCAFitterN<N, Args...>::FwdcalcInverseWeight()
387 {
388  //< calculate [sum_{0<j<N} M_j*E_j*M_j^T]^-1 used for Ti matrices, see EQ.T, with M_i = I
389  auto* arrmat = mWeightInv.Array();
390  memset(arrmat, 0, sizeof(mWeightInv));
391  enum { XX,
392  XY,
393  YY,
394  XZ,
395  YZ,
396  ZZ };
397  for (int i = N; i--;) {
398  const auto& tcov = mTrcEInv[mCurHyp][i];
399  arrmat[XX] += tcov.sxx;
400  arrmat[XY] += tcov.sxy;
401  arrmat[XZ] += 0;
402  arrmat[YY] += tcov.syy;
403  arrmat[YZ] += 0;
404  arrmat[ZZ] += tcov.szz;
405  }
406 
407  // invert 3x3 symmetrix matrix
408  return mWeightInv.Invert();
409 }

412 template <int N, typename... Args>
413 void FwdDCAFitterN<N, Args...>::FwdcalcResidDerivatives()
414 {
415  //< calculate matrix of derivatives for weighted chi2: residual i vs parameter Z of track j
416  MatStd3D matMT;
417  for (int i = N; i--;) { // residual being differentiated
418  // const auto& taux = mTrAux[i];
419  for (int j = N; j--;) { // track over which we differentiate
420  const auto& matT = mTrCFVT[mCurHyp][j]; // coefficient matrix for track J
421  const auto& trDz = mTrDer[mCurHyp][j]; // track point derivs over track Z param
422  auto& dr1 = mDResidDz[i][j];
423  auto& dr2 = mD2ResidDz2[i][j];
424  // calculate M_i^transverse * T_j , M_i^transverse=I -> MT=T
425  matMT[0][0] = matT[0][0];
426  matMT[0][1] = matT[0][1];
427  matMT[0][2] = matT[0][2];
428  matMT[1][0] = matT[1][0];
429  matMT[1][1] = matT[1][1];
430  matMT[1][2] = matT[1][2];
431  matMT[2][0] = matT[2][0];
432  matMT[2][1] = matT[2][1];
433  matMT[2][2] = matT[2][2];
434 
435  // calculate DResid_i/Dz_j = (delta_ij - M_i^tr * T_j) * DTrack_k/Dz_k
436  dr1[0] = -(matMT[0][0] * trDz.dxdz + matMT[0][1] * trDz.dydz + matMT[0][2]);
437  dr1[1] = -(matMT[1][0] * trDz.dxdz + matMT[1][1] * trDz.dydz + matMT[1][2]);
438  dr1[2] = -(matMT[2][0] * trDz.dxdz + matMT[2][1] * trDz.dydz + matMT[2][2]);
439 
440  // calculate D2Resid_I/(Dz_J Dz_K) = (delta_ijk - M_i^tr * T_j * delta_jk) * D2Track_k/dz_k^2
441  dr2[0] = -(matMT[0][1] * trDz.d2ydz2 + matMT[0][0] * trDz.d2xdz2);
442  dr2[1] = -(matMT[1][1] * trDz.d2ydz2 + matMT[1][0] * trDz.d2xdz2);
443  dr2[2] = -(matMT[2][1] * trDz.d2ydz2 + matMT[2][0] * trDz.d2xdz2);
444 
445  if (i == j) {
446  dr1[0] += trDz.dxdz;
447  dr1[1] += trDz.dydz;
448  dr1[2] += 1.;
449 
450  dr2[0] += trDz.d2xdz2;
451  dr2[1] += trDz.d2ydz2;
452  }
453  } // track over which we differentiate
454  } // residual being differentiated
455 }

458 template <int N, typename... Args>
459 void FwdDCAFitterN<N, Args...>::FwdcalcResidDerivativesNoErr()
460 {
461  //< calculate matrix of derivatives for absolute distance chi2: residual i vs parameter Z of track j
462  constexpr double NInv1 = 1. - NInv; // profit from Rii = I/Ninv
463  for (int i = N; i--;) { // residual being differentiated
464  const auto& trDzi = mTrDer[mCurHyp][i]; // track point derivs over track Z param
465  auto& dr1ii = mDResidDz[i][i];
466  auto& dr2ii = mD2ResidDz2[i][i];
467 
468  dr1ii[0] = NInv1 * trDzi.dxdz;
469  dr1ii[1] = NInv1 * trDzi.dydz;
470  dr1ii[2] = NInv1;
471 
472  dr2ii[0] = NInv1 * trDzi.d2xdz2;
473  dr2ii[1] = NInv1 * trDzi.d2ydz2;
474  dr2ii[2] = 0;
475 
476  for (int j = i; j--;) { // track over which we differentiate
477  auto& dr1ij = mDResidDz[i][j];
478  auto& dr1ji = mDResidDz[j][i];
479  const auto& trDzj = mTrDer[mCurHyp][j]; // track point derivs over track Z param
480 
481  // calculate DResid_i/Dz_j = (delta_ij - R_ij) * DTrack_j/Dz_j for j<i
482  dr1ij[0] = -trDzj.dxdz * NInv;
483  dr1ij[1] = -trDzj.dydz * NInv;
484  dr1ij[2] = -1 * NInv;
485 
486  // calculate DResid_j/Dz_i = (delta_ij - R_ji) * DTrack_i/Dz_i for j<i
487  dr1ji[0] = -trDzi.dxdz * NInv;
488  dr1ji[1] = -trDzi.dydz * NInv;
489  dr1ji[2] = -1 * NInv;
490 
491  auto& dr2ij = mD2ResidDz2[i][j];
492  auto& dr2ji = mD2ResidDz2[j][i];
493 
494  // calculate D2Resid_I/(Dz_J Dz_K) = (delta_ij - Rij) * D2Track_j/dz_j^2 * delta_jk for j<i
495  dr2ij[0] = -trDzj.d2xdz2 * NInv;
496  dr2ij[1] = -trDzj.d2ydz2 * NInv;
497  dr2ij[2] = 0;
498 
499  // calculate D2Resid_j/(Dz_i Dz_k) = (delta_ij - Rji) * D2Track_i/dz_i^2 * delta_ik for j<i
500  dr2ji[0] = -trDzi.d2xdz2 * NInv;
501  dr2ji[1] = -trDzi.d2ydz2 * NInv;

509 template <int N, typename... Args>
510 void FwdDCAFitterN<N, Args...>::FwdcalcChi2Derivatives()
511 {
512  //< calculate 1st and 2nd derivatives of wighted DCA (chi2) over track parameters Z
513  std::array<std::array<Vec3D, N>, N> covIDrDz; // tempory vectors of covI_j * dres_j/dz_i
514 
515  // chi2 1st derivative
516  for (int i = N; i--;) {
517  auto& dchi1 = mDChi2Dz[i]; // DChi2/Dz_i = sum_j { res_j * covI_j * Dres_j/Dz_i }
518  dchi1 = 0;
519  for (int j = N; j--;) {
520  const auto& res = mTrRes[mCurHyp][j]; // vector of residuals of track j
521  const auto& covI = mTrcEInv[mCurHyp][j]; // inverse cov matrix of track j
522  const auto& dr1 = mDResidDz[j][i]; // vector of j-th residuals 1st derivative over Z param of track i
523  auto& cidr = covIDrDz[i][j]; // vector covI_j * dres_j/dz_i, save for 2nd derivative calculation
524  cidr[0] = covI.sxx * dr1[0] + covI.sxy * dr1[1];
525  cidr[1] = covI.sxy * dr1[0] + covI.syy * dr1[1];
526  cidr[2] = covI.szz * dr1[2];
527 
528  dchi1 += ROOT::Math::Dot(res, cidr);
529  }
530  }
531 
532  // chi2 2nd derivative
533  for (int i = N; i--;) {
534  for (int j = i + 1; j--;) { // symmetric matrix
535  auto& dchi2 = mD2Chi2Dz2[i][j]; // D2Chi2/Dz_i/Dz_j = sum_k { Dres_k/Dz_j * covI_k * Dres_k/Dz_i + res_k * covI_k * D2res_k/Dz_i/Dz_j }
536  dchi2 = 0;
537  for (int k = N; k--;) {
538  const auto& dr1j = mDResidDz[k][j]; // vector of k-th residuals 1st derivative over Z param of track j
539  const auto& cidrkj = covIDrDz[i][k]; // vector covI_k * dres_k/dz_i
540  dchi2 += ROOT::Math::Dot(dr1j, cidrkj);
541  if (k == j) {
542  const auto& res = mTrRes[mCurHyp][k]; // vector of residuals of track k
543  const auto& covI = mTrcEInv[mCurHyp][k]; // inverse cov matrix of track k
544  const auto& dr2ij = mD2ResidDz2[k][j]; // vector of k-th residuals 2nd derivative over Z params of track j
545  dchi2 += res[0] * (covI.sxx * dr2ij[0] + covI.sxy * dr2ij[1]) + res[1] * (covI.sxy * dr2ij[0] + covI.syy * dr2ij[1]) + res[2] * covI.szz * dr2ij[2];
546  }
547  }
548  }
549  }
550 }

1084  double sumX = 0.;
1085  double sumY = 0.;
1086

1088  mCandTr[mCurHyp][i] = *mOrigTrPtr[i];

1098

1100  sumY = sumY + y[i];
1101  }
1102

1104  mPCA[mCurHyp][1] = sumY / N;

1107  return false;
1108  }
1109

1111  return false;
1112  }

1114  FwdcalcTrackResiduals(); // current track residuals
1115  float chi2Upd, chi2 = FwdcalcChi2();

1117  calcTrackDerivatives(); // current track derivatives (1st and 2nd)
1118  FwdcalcResidDerivatives(); // current residals derivatives (1st and 2nd)
1119  FwdcalcChi2Derivatives(); // current chi2 derivatives (1st and 2nd) to proceed for dz calculation
1120 
1121  // do Newton-Rapson iteration with corrections = - dchi2/d{x0..xN} * [ d^2chi2/d{x0..xN}^2 ]^-1
1122  if (!mD2Chi2Dz2.Invert()) {
1123  return false;
1124  }
1125

1142  chi2 = chi2Upd;

1144  }
1145 
1146  chi2 = chi2Upd;
1147  } while (++mNIters[mCurHyp] < mMaxIter);
1148
