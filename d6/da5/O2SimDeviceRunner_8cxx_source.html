40 void sigaction_handler(int signal, siginfo_t* signal_info, void*)
41 {
42  auto pid = getpid();
43  LOG(info) << pid << " caught signal " << signal << " from source " << signal_info->si_pid;
44  auto groupid = getpgrp();
45  if (pid == gMasterProcess) {
46  killpg(pid, signal); // master kills whole process group
47  } else {
48  if (signal_info->si_pid != gDriverProcess) {
49  // forward to master if coming internally
50  kill(groupid, signal);
51  }
52  }
53  if (signal_info->si_pid == gDriverProcess) {
54  _exit(0); // external requests are not treated as error
55  }
56  if (signal == SIGTERM) {
57  // normal termination is not error
58  _exit(0);
59  }
60  // we treat internal signal interruption as an error
61  // because only ordinary termination is good in the context of the distributed system
62  _exit(128 + signal);
63 }

170 void pinToCPU(unsigned int cpuid)
171 {
172  auto affinity = getenv("ALICE_CPUAFFINITY");
173  if (affinity) {
174  // MacOS does not support this API so we add a protection
175 #ifndef __APPLE__
176 
177  pthread_t thread;
178 
179  thread = pthread_self();
180 
181  cpu_set_t cpuset;
182  CPU_ZERO(&cpuset);
183  CPU_SET(cpuid, &cpuset);
184 
185  auto s = pthread_setaffinity_np(thread, sizeof(cpu_set_t), &cpuset);
186  if (s != 0) {
187  LOG(warning) << "FAILED TO SET PTHREAD AFFINITY";
188  }
189 
190  /* Check the actual affinity mask assigned to the thread */
191  s = pthread_getaffinity_np(thread, sizeof(cpu_set_t), &cpuset);
192  if (s != 0) {
193  LOG(warning) << "FAILED TO GET PTHREAD AFFINITY";
194  }
195 
196  for (int j = 0; j < CPU_SETSIZE; j++) {
197  if (CPU_ISSET(j, &cpuset)) {
198  LOG(info) << "ENABLED CPU " << j;
199  }
200  }
201 #else
202  LOG(warn) << "CPU AFFINITY NOT IMPLEMENTED ON APPLE";
203 #endif
204  }
205 }

207 bool waitForControlInput()
208 {
209  auto factory = FairMQTransportFactory::CreateTransportFactory("zeromq");
210  auto channel = FairMQChannel{"o2sim-control", "sub", factory};
211  auto controlsocketname = getenv("ALICE_O2SIMCONTROL");
212  LOG(debug) << "AWAITING CONTROL ON SOCKETNAME " << controlsocketname;
213  channel.Connect(std::string(controlsocketname));
214  channel.Validate();
215  std::unique_ptr<FairMQMessage> reply(channel.NewMessage());
216 
217  LOG(debug) << "WAITING FOR INPUT";
218  if (channel.Receive(reply) > 0) {
219  auto data = reply->GetData();
220  auto size = reply->GetSize();
221 
222  std::string command(reinterpret_cast<char const*>(data), size);
223  LOG(info) << "message: " << command;
224 
225  o2::conf::SimReconfigData reconfig;
226  o2::conf::parseSimReconfigFromString(command, reconfig);
227  if (reconfig.stop) {
228  LOG(info) << "Stop asked, shutting down";
229  return false;
230  }
231  LOG(info) << "Processing " << reconfig.nEvents << " new events";
232  } else {
233  LOG(info) << "NOTHING RECEIVED";
234  }
235  return true;
236 }

238 int main(int argc, char* argv[])
239 {
240  struct sigaction act;
241  memset(&act, 0, sizeof act);
242  sigemptyset(&act.sa_mask);
243  act.sa_sigaction = &sigaction_handler;
244  act.sa_flags = SA_SIGINFO; // <--- enable sigaction
245 
246  std::vector<int> handledsignals = {SIGTERM, SIGINT, SIGQUIT, SIGSEGV, SIGBUS, SIGFPE}; // <--- may need to be completed
247  // remember that SIGKILL can't be handled
248  for (auto s : handledsignals) {
249  if (sigaction(s, &act, nullptr)) {
250  LOG(error) << "Could not install signal handler for " << s;
251  exit(EXIT_FAILURE);
252  }
253  }
254 
255  // extract the path to FairMQ config
256  bpo::options_description desc{"Options"};
257  // clang-format off
258  desc.add_options()
259  ("control","control type")
260  ("id","ID")
261  ("config-key","config key")
262  ("mq-config",bpo::value<std::string>(),"path to FairMQ config")
263  ("severity","log severity");
264  // clang-format on
265  bpo::variables_map vm;
266  bpo::store(parse_command_line(argc, argv, desc), vm);
267  bpo::notify(vm);
268 
269  std::string FMQconfig;
270  if (vm.count("mq-config")) {
271  FMQconfig = vm["mq-config"].as<std::string>();
272  }
273 
274  auto internalfork = getenv("ALICE_SIMFORKINTERNAL");
275  if (internalfork) {
276  int driverPID = getppid();
277  auto pubchannel = o2::simpubsub::createPUBChannel(o2::simpubsub::getPublishAddress("o2sim-worker-notifications", driverPID));
278 
279  if (FMQconfig.empty()) {
280  throw std::runtime_error("This should never be called without FairMQ config.");
281  }
282  // read the JSON config
283  FILE* fp = fopen(FMQconfig.c_str(), "r");
284  constexpr unsigned short usmax = std::numeric_limits<unsigned short>::max() - 1;
285  char readBuffer[usmax];
286  rapidjson::FileReadStream is(fp, readBuffer, sizeof(readBuffer));
287  rapidjson::Document d;
288  d.ParseStream(is);
289  fclose(fp);
290 
291  // retrieve correct server and merger URLs
292  std::string serveraddress;
293  std::string mergeraddress;
294  std::string serverstatus_address;
295  std::string s;
296 
297  auto& options = d["fairMQOptions"];
298  assert(options.IsObject());
299  for (auto option = options.MemberBegin(); option != options.MemberEnd(); ++option) {
300  s = option->name.GetString();
301  if (s == "devices") {
302  assert(option->value.IsArray());
303  auto devices = option->value.GetArray();
304  for (auto& device : devices) {
305  s = device["id"].GetString();
306  if (s == "primary-server") {
307  auto channels = device["channels"].GetArray();
308  auto sockets = (channels[0])["sockets"].GetArray();
309  auto address = (sockets[0])["address"].GetString();
310  serveraddress = address;
311  sockets = (channels[1])["sockets"].GetArray();
312  address = (sockets[0])["address"].GetString();
313  serverstatus_address = address;
314  }
315  if (s == "hitmerger") {
316  auto channels = device["channels"].GetArray();
317  for (auto& channel : channels) {
318  s = channel["name"].GetString();
319  if (s == "simdata") {
320  auto sockets = channel["sockets"].GetArray();
321  auto address = (sockets[0])["address"].GetString();
322  mergeraddress = address;
323  }
324  }
325  }
326  }
327  }
328  }
329 
330  LOG(info) << "Parsed primary server address " << serveraddress;
331  LOG(info) << "Parsed primary server status address " << serverstatus_address;
332  LOG(info) << "Parsed merger address " << mergeraddress;
333  if (serveraddress.empty() || mergeraddress.empty()) {
334  throw std::runtime_error("Could not determine server or merger URLs.");
335  }
336  // This is a solution based on initializing the simulation once
337  // and then fork the process to share the simulation memory across
338  // many processes. Here we are not using FairMQDevices and just setup
339  // some channels manually and do our own runloop.
340 
341  // we init the simulation first
342  std::unique_ptr<FairRunSim> simrun;
343  // TODO: take the addresses from somewhere else
344  if (!initializeSim("zeromq", serverstatus_address, simrun)) {
345  LOG(error) << "Could not initialize simulation";
346  return 1;
347  }
348 
349  // should be factored out?
350  unsigned int nworkers = std::max(1u, std::thread::hardware_concurrency() / 2);
351  auto f = getenv("ALICE_NSIMWORKERS");
352  if (f) {
353  nworkers = static_cast<unsigned int>(std::stoi(f));
354  }
355  LOG(info) << "Running with " << nworkers << " sim workers ";
356 
357  gMasterProcess = getpid();
358  gDriverProcess = getppid();
359  // then we fork and create a device in each fork
360  for (auto i = 0u; i < nworkers; ++i) {
361  // we use the current process as one of the workers as it has nothing else to do
362  auto pid = (i == nworkers - 1) ? 0 : fork();
363  if (pid == 0) {
364  // Each worker can publish its progress/state on a ZMQ channel.
365  // We actually use a push/pull mechanism to collect all messages in the
366  // master worker which can then publish using PUB/SUB.
367  // auto factory = FairMQTransportFactory::CreateTransportFactory("zeromq");
368  auto collectAndPubThreadFunction = [driverPID, &pubchannel]() {
369  auto collectorchannel = o2::simpubsub::createPUBChannel(o2::simpubsub::getPublishAddress("o2sim-workerinternal", driverPID), "pull");
370  std::unique_ptr<FairMQMessage> msg(collectorchannel.NewMessage());
371 
372  while (true) {
373  if (collectorchannel.Receive(msg) > 0) {
374  auto data = msg->GetData();
375  auto size = msg->GetSize();
376  std::string text(reinterpret_cast<char const*>(data), size);
377  // LOG(info) << "Collector message: " << text;
378  o2::simpubsub::publishMessage(pubchannel, text);
379  }
380  }
381  };
382  if (i == nworkers - 1) { // <---- extremely important to take non-forked version since ZMQ sockets do not behave well on fork
383  std::vector<std::thread> threads;
384  threads.push_back(std::thread(collectAndPubThreadFunction));
385  threads.back().detach();
386  }
387 
388  // everyone else is getting a push socket for notifications
389  auto pushchannel = o2::simpubsub::createPUBChannel(o2::simpubsub::getPublishAddress("o2sim-workerinternal", driverPID), "push");
390 
391  // we will try to pin each worker to a particular CPU
392  // this can be made configurable via environment variables??
393  pinToCPU(i);
394 
395  auto kernelSetup = initSim("zeromq", serveraddress, serverstatus_address, mergeraddress, i);
396 
397  std::stringstream worker;
398  worker << "WORKER" << i;
399  o2::simpubsub::publishMessage(pushchannel, o2::simpubsub::simStatusString(worker.str(), "STATUS", "SETUP COMPLETED"));
400 
401  auto& conf = o2::conf::SimConfig::Instance();
402 
403  bool more = true;
404  while (more) {
405  runSim(kernelSetup);
406 
407  if (conf.asService()) {
408  LOG(info) << "IN SERVICE MODE WAITING";
409  o2::simpubsub::publishMessage(pushchannel, o2::simpubsub::simStatusString(worker.str(), "STATUS", "AWAITING INPUT"));
410  more = waitForControlInput();
411  usleep(100); // --> why?
412  } else {
413  o2::simpubsub::publishMessage(pushchannel, o2::simpubsub::simStatusString(worker.str(), "STATUS", "TERMINATING"));
414 
415  LOG(info) << "FINISHING";
416  more = false;
417  }
418  }
419  sleep(10); // ---> give some time for message to be delivered to merger (destructing too early might affect the ZQM buffers)
420  // The process will in any case be terminated by the main o2-sim driver.
421 
422  // destruct setup (using _exit due to problems in ROOT shutdown (segmentation violations)
423  // Clearly at some moment, a more robust solution would be appreciated
424  _exit(0);
425  } else {
426  gChildProcesses.push_back(pid);
427  }
428  }
429  int status;
430  wait(&status); /* only the parent waits */
431  _exit(0);
432  } else {
433  // This the solution where we setup an ordinary FairMQDevice
434  // (each if which will setup its own simulation). Parallelism
435  // is achieved outside by instantiating multiple device processes.
436  _exit(initAndRunDevice(argc, argv));
437  }
