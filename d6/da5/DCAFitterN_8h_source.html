31 struct TrackCovI {
32  float sxx, syy, syz, szz;
33 
34  TrackCovI(const o2::track::TrackParCov& trc, float xerrFactor = 1.) { set(trc, xerrFactor); }
35 
36  TrackCovI() = default;
37 
38  void set(const o2::track::TrackParCov& trc, float xerrFactor = 1)
39  {
40  // we assign Y error to X for DCA calculation
41  // (otherwise for quazi-collinear tracks the X will not be constrained)
42  float cyy = trc.getSigmaY2(), czz = trc.getSigmaZ2(), cyz = trc.getSigmaZY(), cxx = cyy * xerrFactor;
43  float detYZ = cyy * czz - cyz * cyz;
44  if (detYZ > 0.) {
45  auto detYZI = 1. / detYZ;
46  sxx = 1. / cxx;
47  syy = czz * detYZI;
48  syz = -cyz * detYZI;
49  szz = cyy * detYZI;
50  } else {
51  throw std::runtime_error("invalid track covariance");
52  }
53  }
54 };

58 struct TrackDeriv {
59  float dydx, dzdx, d2ydx2, d2zdx2;
60  TrackDeriv() = default;
61  TrackDeriv(const o2::track::TrackPar& trc, float bz) { set(trc, bz); }
62  void set(const o2::track::TrackPar& trc, float bz)
63  {
64  float snp = trc.getSnp(), csp = std::sqrt((1. - snp) * (1. + snp)), cspI = 1. / csp, crv2c = trc.getCurvature(bz) * cspI;
65  dydx = snp * cspI; // = snp/csp
66  dzdx = trc.getTgl() * cspI; // = tgl/csp
67  d2ydx2 = crv2c * cspI * cspI; // = crv/csp^3
68  d2zdx2 = crv2c * dzdx * dydx; // = crv*tgl*snp/csp^3
69  }

328 template <int N, typename... Args>
329 template <class... Tr>
330 int DCAFitterN<N, Args...>::process(const Tr&... args)
331 {
332  // This is a main entry point: fit PCA of N tracks
333  static_assert(sizeof...(args) == N, "incorrect number of input tracks");
334  assign(0, args...);
335  clear();
336  for (int i = 0; i < N; i++) {
337  mTrAux[i].set(*mOrigTrPtr[i], mBz);
338  }
339  if (!mCrossings.set(mTrAux[0], *mOrigTrPtr[0], mTrAux[1], *mOrigTrPtr[1], mMaxDXYIni)) { // even for N>2 it should be enough to test just 1 loop
340  return 0; // no crossing
341  }
342  if (mUseAbsDCA) {
343  calcRMatrices(); // needed for fast residuals derivatives calculation in case of abs. distance minimization
344  }
345  if (mCrossings.nDCA == MAXHYP) { // if there are 2 candidates and they are too close, chose their mean as a starting point
346  auto dst2 = (mCrossings.xDCA[0] - mCrossings.xDCA[1]) * (mCrossings.xDCA[0] - mCrossings.xDCA[1]) +
347  (mCrossings.yDCA[0] - mCrossings.yDCA[1]) * (mCrossings.yDCA[0] - mCrossings.yDCA[1]);
348  if (dst2 < mMaxDist2ToMergeSeeds) {
349  mCrossings.nDCA = 1;
350  mCrossings.xDCA[0] = 0.5 * (mCrossings.xDCA[0] + mCrossings.xDCA[1]);
351  mCrossings.yDCA[0] = 0.5 * (mCrossings.yDCA[0] + mCrossings.yDCA[1]);
352  }
353  }
354  // check all crossings
355  for (int ic = 0; ic < mCrossings.nDCA; ic++) {
356  // check if radius is acceptable
357  if (mCrossings.xDCA[ic] * mCrossings.xDCA[ic] + mCrossings.yDCA[ic] * mCrossings.yDCA[ic] > mMaxR2) {
358  continue;
359  }
360  mCrossIDCur = ic;
361  mCrossIDAlt = (mCrossings.nDCA == 2 && mAllowAltPreference) ? 1 - ic : -1; // works for max 2 crossings
362  mNIters[mCurHyp] = 0;
363  mTrPropDone[mCurHyp] = false;
364  mChi2[mCurHyp] = -1.;
365  mPCA[mCurHyp][0] = mCrossings.xDCA[ic];
366  mPCA[mCurHyp][1] = mCrossings.yDCA[ic];
367 
368  if (mUseAbsDCA ? minimizeChi2NoErr() : minimizeChi2()) {
369  mOrder[mCurHyp] = mCurHyp;
370  if (mPropagateToPCA && !propagateTracksToVertex(mCurHyp)) {
371  continue; // discard candidate if failed to propagate to it
372  }
373  mCurHyp++;
374  }
375  }
376 
377  for (int i = mCurHyp; i--;) { // order in quality
378  for (int j = i; j--;) {
379  if (mChi2[mOrder[i]] < mChi2[mOrder[j]]) {
380  std::swap(mOrder[i], mOrder[j]);
381  }
382  }
383  }
384  return mCurHyp;
385 }

388 template <int N, typename... Args>
389 bool DCAFitterN<N, Args...>::calcPCACoefs()
390 {
391  //< calculate Ti matrices for global vertex decomposition to V = sum_{0<i<N} Ti pi, see EQ.T in the ref
392  if (!calcInverseWeight()) {
393  return false;
394  }
395  for (int i = N; i--;) { // build Mi*Ei matrix
396  const auto& taux = mTrAux[i];
397  const auto& tcov = mTrcEInv[mCurHyp][i];
398  MatStd3D miei;
399  miei[0][0] = taux.c * tcov.sxx;
400  miei[0][1] = -taux.s * tcov.syy;
401  miei[0][2] = -taux.s * tcov.syz;
402  miei[1][0] = taux.s * tcov.sxx;
403  miei[1][1] = taux.c * tcov.syy;
404  miei[1][2] = taux.c * tcov.syz;
405  //miei[2][0] = 0;
406  miei[2][1] = tcov.syz;
407  miei[2][2] = tcov.szz;
408  mTrCFVT[mCurHyp][i] = mWeightInv * miei;
409  }
410  return true;
411 }

414 template <int N, typename... Args>
415 bool DCAFitterN<N, Args...>::calcInverseWeight()
416 {
417  //< calculate [sum_{0<j<N} M_j*E_j*M_j^T]^-1 used for Ti matrices, see EQ.T
418  auto* arrmat = mWeightInv.Array();
419  memset(arrmat, 0, sizeof(mWeightInv));
420  enum { XX,
421  XY,
422  YY,
423  XZ,
424  YZ,
425  ZZ };
426  for (int i = N; i--;) {
427  const auto& taux = mTrAux[i];
428  const auto& tcov = mTrcEInv[mCurHyp][i];
429  arrmat[XX] += taux.cc * tcov.sxx + taux.ss * tcov.syy;
430  arrmat[XY] += taux.cs * (tcov.sxx - tcov.syy);
431  arrmat[XZ] += -taux.s * tcov.syz;
432  arrmat[YY] += taux.cc * tcov.syy + taux.ss * tcov.sxx;
433  arrmat[YZ] += taux.c * tcov.syz;
434  arrmat[ZZ] += tcov.szz;
435  }
436  // invert 3x3 symmetrix matrix
437  return mWeightInv.Invert();
438 }

441 template <int N, typename... Args>
442 void DCAFitterN<N, Args...>::calcResidDerivatives()
443 {
444  //< calculate matrix of derivatives for weighted chi2: residual i vs parameter X of track j
445  MatStd3D matMT;
446  for (int i = N; i--;) { // residual being differentiated
447  const auto& taux = mTrAux[i];
448  for (int j = N; j--;) { // track over which we differentiate
449  const auto& matT = mTrCFVT[mCurHyp][j]; // coefficient matrix for track J
450  const auto& trDx = mTrDer[mCurHyp][j]; // track point derivs over track X param
451  auto& dr1 = mDResidDx[i][j];
452  auto& dr2 = mD2ResidDx2[i][j];
453  // calculate M_i^tr * T_j
454  matMT[0][0] = taux.c * matT[0][0] + taux.s * matT[1][0];
455  matMT[0][1] = taux.c * matT[0][1] + taux.s * matT[1][1];
456  matMT[0][2] = taux.c * matT[0][2] + taux.s * matT[1][2];
457  matMT[1][0] = -taux.s * matT[0][0] + taux.c * matT[1][0];
458  matMT[1][1] = -taux.s * matT[0][1] + taux.c * matT[1][1];
459  matMT[1][2] = -taux.s * matT[0][2] + taux.c * matT[1][2];
460  matMT[2][0] = matT[2][0];
461  matMT[2][1] = matT[2][1];
462  matMT[2][2] = matT[2][2];
463 
464  // calculate DResid_i/Dx_j = (delta_ij - M_i^tr * T_j) * DTrack_k/Dx_k
465  dr1[0] = -(matMT[0][0] + matMT[0][1] * trDx.dydx + matMT[0][2] * trDx.dzdx);
466  dr1[1] = -(matMT[1][0] + matMT[1][1] * trDx.dydx + matMT[1][2] * trDx.dzdx);
467  dr1[2] = -(matMT[2][0] + matMT[2][1] * trDx.dydx + matMT[2][2] * trDx.dzdx);
468 
469  // calculate D2Resid_I/(Dx_J Dx_K) = (delta_ijk - M_i^tr * T_j * delta_jk) * D2Track_k/dx_k^2
470  dr2[0] = -(matMT[0][1] * trDx.d2ydx2 + matMT[0][2] * trDx.d2zdx2);
471  dr2[1] = -(matMT[1][1] * trDx.d2ydx2 + matMT[1][2] * trDx.d2zdx2);
472  dr2[2] = -(matMT[2][1] * trDx.d2ydx2 + matMT[2][2] * trDx.d2zdx2);
473 
474  if (i == j) {
475  dr1[0] += 1.;
476  dr1[1] += trDx.dydx;
477  dr1[2] += trDx.dzdx;
478 
479  dr2[1] += trDx.d2ydx2;
480  dr2[2] += trDx.d2zdx2;
481  }
482  } // track over which we differentiate
483  } // residual being differentiated
484 }

486 //__________________________________________________________________________
487 template <int N, typename... Args>
488 void DCAFitterN<N, Args...>::calcResidDerivativesNoErr()
489 {
490  //< calculate matrix of derivatives for absolute distance chi2: residual i vs parameter X of track j
491  constexpr double NInv1 = 1. - NInv; // profit from Rii = I/Ninv
492  for (int i = N; i--;) { // residual being differentiated
493  const auto& trDxi = mTrDer[mCurHyp][i]; // track point derivs over track X param
494  auto& dr1ii = mDResidDx[i][i];
495  auto& dr2ii = mD2ResidDx2[i][i];
496  dr1ii[0] = NInv1;
497  dr1ii[1] = NInv1 * trDxi.dydx;
498  dr1ii[2] = NInv1 * trDxi.dzdx;
499 
500  dr2ii[0] = 0;
501  dr2ii[1] = NInv1 * trDxi.d2ydx2;
502  dr2ii[2] = NInv1 * trDxi.d2zdx2;
503 
504  for (int j = i; j--;) { // track over which we differentiate
505  auto& dr1ij = mDResidDx[i][j];
506  auto& dr1ji = mDResidDx[j][i];
507  const auto& trDxj = mTrDer[mCurHyp][j]; // track point derivs over track X param
508  auto cij = mCosDif[i][j], sij = mSinDif[i][j]; // M_i^T*M_j / N matrices non-trivial elements = {ci*cj+si*sj , si*cj-ci*sj }, see 5 in ref.
509 
510  // calculate DResid_i/Dx_j = (delta_ij - R_ij) * DTrack_j/Dx_j for j<i
511  dr1ij[0] = -(cij + sij * trDxj.dydx);
512  dr1ij[1] = -(-sij + cij * trDxj.dydx);
513  dr1ij[2] = -trDxj.dzdx * NInv;
514 
515  // calculate DResid_j/Dx_i = (delta_ij - R_ji) * DTrack_i/Dx_i for j<i
516  dr1ji[0] = -(cij - sij * trDxi.dydx);
517  dr1ji[1] = -(sij + cij * trDxi.dydx);
518  dr1ji[2] = -trDxi.dzdx * NInv;
519 
520  auto& dr2ij = mD2ResidDx2[i][j];
521  auto& dr2ji = mD2ResidDx2[j][i];
522  // calculate D2Resid_I/(Dx_J Dx_K) = (delta_ij - Rij) * D2Track_j/dx_j^2 * delta_jk for j<i
523  dr2ij[0] = -sij * trDxj.d2ydx2;
524  dr2ij[1] = -cij * trDxj.d2ydx2;
525  dr2ij[2] = -trDxj.d2zdx2 * NInv;
526 
527  // calculate D2Resid_j/(Dx_i Dx_k) = (delta_ij - Rji) * D2Track_i/dx_i^2 * delta_ik for j<i
528  dr2ji[0] = sij * trDxi.d2ydx2;
529  dr2ji[1] = -cij * trDxi.d2ydx2;
530  dr2ji[2] = -trDxi.d2zdx2 * NInv;
531 
532  } // track over which we differentiate
533  } // residual being differentiated
534 }

538 void DCAFitterN<N, Args...>::calcRMatrices()
539 {
540  //< calculate Rij = 1/N M_i^T * M_j matrices (rotation from j-th track to i-th track frame)

547  }

552 template <int N, typename... Args>
553 void DCAFitterN<N, Args...>::calcChi2Derivatives()
554 {
555  //< calculate 1st and 2nd derivatives of wighted DCA (chi2) over track parameters X, see EQ.Chi2 in the ref
556  std::array<std::array<Vec3D, N>, N> covIDrDx; // tempory vectors of covI_j * dres_j/dx_i
557 
558  // chi2 1st derivative
559  for (int i = N; i--;) {
560  auto& dchi1 = mDChi2Dx[i]; // DChi2/Dx_i = sum_j { res_j * covI_j * Dres_j/Dx_i }
561  dchi1 = 0;
562  for (int j = N; j--;) {
563  const auto& res = mTrRes[mCurHyp][j]; // vector of residuals of track j
564  const auto& covI = mTrcEInv[mCurHyp][j]; // inverse cov matrix of track j
565  const auto& dr1 = mDResidDx[j][i]; // vector of j-th residuals 1st derivative over X param of track i
566  auto& cidr = covIDrDx[i][j]; // vector covI_j * dres_j/dx_i, save for 2nd derivative calculation
567  cidr[0] = covI.sxx * dr1[0];
568  cidr[1] = covI.syy * dr1[1] + covI.syz * dr1[2];
569  cidr[2] = covI.syz * dr1[1] + covI.szz * dr1[2];
570  // calculate res_i * covI_j * dres_j/dx_i
571  dchi1 += ROOT::Math::Dot(res, cidr);
572  }
573  }
574  // chi2 2nd derivative
575  for (int i = N; i--;) {
576  for (int j = i + 1; j--;) { // symmetric matrix
577  auto& dchi2 = mD2Chi2Dx2[i][j]; // D2Chi2/Dx_i/Dx_j = sum_k { Dres_k/Dx_j * covI_k * Dres_k/Dx_i + res_k * covI_k * D2res_k/Dx_i/Dx_j }
578  dchi2 = 0;
579  for (int k = N; k--;) {
580  const auto& dr1j = mDResidDx[k][j]; // vector of k-th residuals 1st derivative over X param of track j
581  const auto& cidrkj = covIDrDx[i][k]; // vector covI_k * dres_k/dx_i
582  dchi2 += ROOT::Math::Dot(dr1j, cidrkj);
583  if (k == j) {
584  const auto& res = mTrRes[mCurHyp][k]; // vector of residuals of track k
585  const auto& covI = mTrcEInv[mCurHyp][k]; // inverse cov matrix of track k
586  const auto& dr2ij = mD2ResidDx2[k][j]; // vector of k-th residuals 2nd derivative over X params of track j
587  dchi2 += res[0] * covI.sxx * dr2ij[0] + res[1] * (covI.syy * dr2ij[1] + covI.syz * dr2ij[2]) + res[2] * (covI.syz * dr2ij[1] + covI.szz * dr2ij[2]);
588  }
589  }
590  }
591  }
592 }

595 template <int N, typename... Args>
596 void DCAFitterN<N, Args...>::calcChi2DerivativesNoErr()
597 {
598  //< calculate 1st and 2nd derivatives of abs DCA (chi2) over track parameters X, see (6) in the ref
599  for (int i = N; i--;) {
600  auto& dchi1 = mDChi2Dx[i]; // DChi2/Dx_i = sum_j { res_j * Dres_j/Dx_i }
601  dchi1 = 0; // chi2 1st derivative
602  for (int j = N; j--;) {
603  const auto& res = mTrRes[mCurHyp][j]; // vector of residuals of track j
604  const auto& dr1 = mDResidDx[j][i]; // vector of j-th residuals 1st derivative over X param of track i
605  dchi1 += ROOT::Math::Dot(res, dr1);
606  if (i >= j) { // symmetrix matrix
607  // chi2 2nd derivative
608  auto& dchi2 = mD2Chi2Dx2[i][j]; // D2Chi2/Dx_i/Dx_j = sum_k { Dres_k/Dx_j * covI_k * Dres_k/Dx_i + res_k * covI_k * D2res_k/Dx_i/Dx_j }
609  dchi2 = ROOT::Math::Dot(mTrRes[mCurHyp][i], mD2ResidDx2[i][j]);
610  for (int k = N; k--;) {
611  dchi2 += ROOT::Math::Dot(mDResidDx[k][i], mDResidDx[k][j]);
612  }
613  }
614  }
615  }
616 }

797 template <int N, typename... Args>
798 bool DCAFitterN<N, Args...>::minimizeChi2()
799 {
800  // find best chi2 (weighted DCA) of N tracks in the vicinity of the seed PCA
801  for (int i = N; i--;) {
802  mCandTr[mCurHyp][i] = *mOrigTrPtr[i];
803  auto x = mTrAux[i].c * mPCA[mCurHyp][0] + mTrAux[i].s * mPCA[mCurHyp][1]; // X of PCA in the track frame
804  if (x < mMinXSeed || !propagateToX(mCandTr[mCurHyp][i], x)) {
805  return false;
806  }
807  setTrackPos(mTrPos[mCurHyp][i], mCandTr[mCurHyp][i]); // prepare positions
808  mTrcEInv[mCurHyp][i].set(mCandTr[mCurHyp][i], XerrFactor); // prepare inverse cov.matrices at starting point
809  }
810 
811  if (mMaxDZIni > 0 && !roughDZCut()) { // apply rough cut on tracks Z difference
812  return false;
813  }
814 
815  if (!calcPCACoefs()) { // prepare tracks contribution matrices to the global PCA
816  return false;
817  }
818  calcPCA(); // current PCA
819  calcTrackResiduals(); // current track residuals
820  float chi2Upd, chi2 = calcChi2();
821  do {
822  calcTrackDerivatives(); // current track derivatives (1st and 2nd)
823  calcResidDerivatives(); // current residals derivatives (1st and 2nd)
824  calcChi2Derivatives(); // current chi2 derivatives (1st and 2nd)
825 
826  // do Newton-Rapson iteration with corrections = - dchi2/d{x0..xN} * [ d^2chi2/d{x0..xN}^2 ]^-1
827  if (!mD2Chi2Dx2.Invert()) {
828  LOG(error) << "InversionFailed";
829  return false;
830  }
831  VecND dx = mD2Chi2Dx2 * mDChi2Dx;
832  if (!correctTracks(dx)) {
833  return false;
834  }
835  calcPCA(); // updated PCA
836  if (mCrossIDAlt >= 0 && closerToAlternative()) {
837  mAllowAltPreference = false;
838  return false;
839  }
840  calcTrackResiduals(); // updated residuals
841  chi2Upd = calcChi2(); // updated chi2
842  if (getAbsMax(dx) < mMinParamChange || chi2Upd > chi2 * mMinRelChi2Change) {
843  chi2 = chi2Upd;
844  break; // converged
845  }
846  chi2 = chi2Upd;
847  } while (++mNIters[mCurHyp] < mMaxIter);
848  //
849  mChi2[mCurHyp] = chi2 * NInv;
850  return mChi2[mCurHyp] < mMaxChi2;
851 }
