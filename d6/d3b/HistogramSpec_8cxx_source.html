18 void AxisSpec::makeLogaritmic()
19 {
20  if (binEdges.size() > 2) {
21  LOG(fatal) << "Cannot make a variabled bin width axis logaritmic";
22  }
23 
24  const double min = binEdges[0];
25  if (min <= 0.) {
26  LOG(fatal) << "Cannot have the first bin limit of the log. axis below 0: " << min;
27  }
28  const double max = binEdges[1];
29  binEdges.clear();
30  const double logmin = std::log10(min);
31  const double logmax = std::log10(max);
32  const int nbins = nBins.value();
33  const double logdelta = (logmax - logmin) / (static_cast<double>(nbins));
34  const double log10 = std::log10(10.);
35  LOG(debug) << "Making a logaritmic binning from " << min << " to " << max << " with " << nbins << " bins";
36  for (int i = 0; i < nbins + 1; i++) {
37  const auto nextEdge = std::pow(10, logmin + i * logdelta);
38  LOG(debug) << i << "/" << nbins - 1 << ": " << nextEdge;
39  binEdges.push_back(nextEdge);
40  }
41  nBins = std::nullopt;
42 }

44 // main function for creating arbirtary histograms
45 template <typename T>
46 std::unique_ptr<T> HistFactory::createHist(const HistogramSpec& histSpec)
47 {
48  constexpr std::size_t MAX_DIM{10};
49  const std::size_t nAxes{histSpec.config.axes.size()};
50  if (nAxes == 0 || nAxes > MAX_DIM) {
51  LOGF(fatal, "The histogram specification contains no (or too many) axes.");
52  return nullptr;
53  }
54 
55  int nBins[MAX_DIM]{0};
56  double lowerBounds[MAX_DIM]{0.};
57  double upperBounds[MAX_DIM]{0.};
58 
59  // first figure out number of bins and dimensions
60  for (std::size_t i = 0; i < nAxes; i++) {
61  nBins[i] = (histSpec.config.axes[i].nBins) ? *histSpec.config.axes[i].nBins : histSpec.config.axes[i].binEdges.size() - 1;
62  lowerBounds[i] = histSpec.config.axes[i].binEdges.front();
63  upperBounds[i] = histSpec.config.axes[i].binEdges.back();
64  }
65 
66  // create histogram
67  std::unique_ptr<T> hist{generateHist<T>(histSpec.name, histSpec.title, nAxes, nBins, lowerBounds, upperBounds, histSpec.config.nSteps)};
68  if (!hist) {
69  LOGF(fatal, "The number of dimensions specified for histogram %s does not match the type.", histSpec.name);
70  return nullptr;
71  }
72 
73  // set axis properties
74  for (std::size_t i = 0; i < nAxes; i++) {
75  TAxis* axis{getAxis(i, hist.get())};
76  if (axis) {
77  if (histSpec.config.axes[i].title) {
78  axis->SetTitle((*histSpec.config.axes[i].title).data());
79  }
80 
81  // this helps to have axes not only called 0,1,2... in ndim histos
82  if constexpr (std::is_base_of_v<THnBase, T>) {
83  if (histSpec.config.axes[i].name) {
84  axis->SetName((std::string(axis->GetName()) + "-" + *histSpec.config.axes[i].name).data());
85  }
86  }
87 
88  // move the bin edges in case a variable binning was requested
89  if (!histSpec.config.axes[i].nBins) {
90  if (!std::is_sorted(std::begin(histSpec.config.axes[i].binEdges), std::end(histSpec.config.axes[i].binEdges))) {
91  LOGF(fatal, "The bin edges in histogram %s are not in increasing order!", histSpec.name);
92  return nullptr;
93  }
94  axis->Set(nBins[i], histSpec.config.axes[i].binEdges.data());
95  }
96  }
97  }
98  if (histSpec.callSumw2) {
99  hist->Sumw2();
100  }
101  return hist;
102 }

125 // helper function to get the axis via index for any type of root histogram
126 template <typename T>
127 TAxis* HistFactory::getAxis(const int i, T* hist)
128 {
129  if constexpr (std::is_base_of_v<THnBase, T> || std::is_base_of_v<StepTHn, T>) {
130  return hist->GetAxis(i);
131  } else {
132  if (i == 0) {
133  return hist->GetXaxis();
134  } else if (i == 1) {
135  return hist->GetYaxis();
136  } else if (i == 2) {
137  return hist->GetZaxis();
138  } else {
139  return nullptr;
140  }
141  }

144 // helper function to generate the actual histograms
145 template <typename T>
146 T* HistFactory::generateHist(const std::string& name, const std::string& title, const std::size_t nDim,
147  const int nBins[], const double lowerBounds[], const double upperBounds[], const int nSteps)
148 {
149  if constexpr (std::is_base_of_v<StepTHn, T>) {
150  return new T(name.data(), title.data(), nSteps, nDim, nBins, lowerBounds, upperBounds);
151  } else if constexpr (std::is_base_of_v<THnBase, T>) {
152  return new T(name.data(), title.data(), nDim, nBins, lowerBounds, upperBounds);
153  } else if constexpr (std::is_base_of_v<TH3, T>) {
154  return (nDim != 3) ? nullptr
155  : new T(name.data(), title.data(), nBins[0], lowerBounds[0],
156  upperBounds[0], nBins[1], lowerBounds[1], upperBounds[1],
157  nBins[2], lowerBounds[2], upperBounds[2]);
158  } else if constexpr (std::is_base_of_v<TH2, T>) {
159  return (nDim != 2) ? nullptr
160  : new T(name.data(), title.data(), nBins[0], lowerBounds[0],
161  upperBounds[0], nBins[1], lowerBounds[1], upperBounds[1]);
162  } else if constexpr (std::is_base_of_v<TH1, T>) {
163  return (nDim != 1)
164  ? nullptr
165  : new T(name.data(), title.data(), nBins[0], lowerBounds[0], upperBounds[0]);
166  }
167  return nullptr;
168 }

172 std::optional<HistPtr> HistFactory::castToVariant(std::shared_ptr<TObject> obj)
173 {
174  if (obj->InheritsFrom(T::Class())) {

177  return std::nullopt;

180 template <typename T, typename Next, typename... Rest>
181 std::optional<HistPtr> HistFactory::castToVariant(std::shared_ptr<TObject> obj)
182 {
183  if (auto hist = castToVariant<T>(obj)) {
184  return hist;
185  }
186  return castToVariant<Next, Rest...>(obj);

189 std::optional<HistPtr> HistFactory::castToVariant(std::shared_ptr<TObject> obj)
190 {

193  return castToVariant<THn, THnSparse, TProfile3D, TH3, TProfile2D, TH2, TProfile, TH1, StepTHn>(obj);

195  return std::nullopt;
