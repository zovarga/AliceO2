58 void CookedTrackerDPL::init(InitContext& ic)
59 {
60  mTimer.Stop();
61  mTimer.Reset();
62  auto nthreads = ic.options().get<int>("nthreads");
63  mTracker.setNumberOfThreads(nthreads);
64  auto filename = ic.options().get<std::string>("grp-file");
65  const auto grp = o2::parameters::GRPObject::loadFrom(filename);
66  if (grp) {
67  mVertexerPtr->getGlobalConfiguration();
68 
69  mGRP.reset(grp);
70  o2::base::Propagator::initFieldFromGRP(grp);
71  auto field = static_cast<o2::field::MagneticField*>(TGeoGlobalMagField::Instance()->GetField());
72 
73  o2::base::GeometryManager::loadGeometry();
74  o2::its::GeometryTGeo* geom = o2::its::GeometryTGeo::Instance();
75  geom->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L, o2::math_utils::TransformType::T2GRot,
76  o2::math_utils::TransformType::T2G));
77  mTracker.setGeometry(geom);
78 
79  mTracker.setConfigParams();
80  LOG(info) << "Tracking mode " << mMode;
81  if (mMode == "cosmics") {
82  LOG(info) << "Setting cosmics parameters...";
83  mTracker.setParametersCosmics();
84  mRunVertexer = false;
85  }
86 
87  double origD[3] = {0., 0., 0.};
88  mTracker.setBz(field->getBz(origD));
89 
90  bool continuous = mGRP->isDetContinuousReadOut("ITS");
91  LOG(info) << "ITSCookedTracker RO: continuous=" << continuous;
92  mTracker.setContinuousMode(continuous);
93  } else {
94  throw std::runtime_error(o2::utils::Str::concat_string("Cannot retrieve GRP from the ", filename));
95  }
96 
97  std::string dictPath = o2::itsmft::ClustererParam<o2::detectors::DetID::ITS>::Instance().dictFilePath;
98  std::string dictFile = o2::base::DetectorNameConf::getAlpideClusterDictionaryFileName(o2::detectors::DetID::ITS, dictPath);
99  if (o2::utils::Str::pathExists(dictFile)) {
100  mDict.readFromFile(dictFile);
101  LOG(info) << "Tracker running with a provided dictionary: " << dictFile;
102  } else {
103  LOG(info) << "Dictionary " << dictFile << " is absent, Tracker expects cluster patterns";
104  }
105 }

107 void CookedTrackerDPL::run(ProcessingContext& pc)
108 {
109  mTimer.Start(false);
110  auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
111  gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
112 
113  // code further down does assignment to the rofs and the altered object is used for output
114  // we therefore need a copy of the vector rather than an object created directly on the input data,
115  // the output vector however is created directly inside the message memory thus avoiding copy by
116  // snapshot
117  auto rofsinput = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
118  auto& rofs = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"ITS", "ITSTrackROF", 0, Lifetime::Timeframe}, rofsinput.begin(), rofsinput.end());
119 
120  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>> labels;
121  gsl::span<itsmft::MC2ROFRecord const> mc2rofs;
122  if (mUseMC) {
123  labels = pc.inputs().get<const o2::dataformats::MCTruthContainer<o2::MCCompLabel>*>("labels");
124  // get the array as read-onlt span, a snapshot is send forward
125  mc2rofs = pc.inputs().get<gsl::span<itsmft::MC2ROFRecord>>("MC2ROframes");
126  }
127  const auto& multEstConf = FastMultEstConfig::Instance(); // parameters for mult estimation and cuts
128  FastMultEst multEst; // mult estimator
129 
130  LOG(info) << "ITSCookedTracker pulled " << compClusters.size() << " clusters, in " << rofs.size() << " RO frames";
131 
132  std::vector<o2::MCCompLabel> trackLabels;
133  if (mUseMC) {
134  mTracker.setMCTruthContainers(labels.get(), &trackLabels);
135  }
136 
137  o2::its::ROframe event(0, 7);
138 
139  auto& vertROFvec = pc.outputs().make<std::vector<o2::itsmft::ROFRecord>>(Output{"ITS", "VERTICESROF", 0, Lifetime::Timeframe});
140  auto& vertices = pc.outputs().make<std::vector<Vertex>>(Output{"ITS", "VERTICES", 0, Lifetime::Timeframe});
141  auto& tracks = pc.outputs().make<std::vector<o2::its::TrackITS>>(Output{"ITS", "TRACKS", 0, Lifetime::Timeframe});
142  auto& clusIdx = pc.outputs().make<std::vector<int>>(Output{"ITS", "TRACKCLSID", 0, Lifetime::Timeframe});
143  auto& irFrames = pc.outputs().make<std::vector<o2::dataformats::IRFrame>>(Output{"ITS", "IRFRAMES", 0, Lifetime::Timeframe});
144 
145  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance(); // RS: this should come from CCDB
146  int nBCPerTF = mTracker.getContinuousMode() ? alpParams.roFrameLengthInBC : alpParams.roFrameLengthTrig;
147 
148  gsl::span<const unsigned char>::iterator pattIt = patterns.begin();
149  for (auto& rof : rofs) {
150  auto& vtxROF = vertROFvec.emplace_back(rof); // register entry and number of vertices in the
151  vtxROF.setFirstEntry(vertices.size());
152  vtxROF.setNEntries(0);
153 
154  auto it = pattIt;
155 
156  // fast cluster mult. cut if asked (e.g. sync. mode)
157  if (rof.getNEntries() && (multEstConf.cutMultClusLow > 0 || multEstConf.cutMultClusHigh > 0)) { // cut was requested
158  auto mult = multEst.process(rof.getROFData(compClusters));
159  if (mult < multEstConf.cutMultClusLow || (multEstConf.cutMultClusHigh > 0 && mult > multEstConf.cutMultClusHigh)) {
160  LOG(info) << "Estimated cluster mult. " << mult << " is outside of requested range "
161  << multEstConf.cutMultClusLow << " : " << multEstConf.cutMultClusHigh << " | ROF " << rof.getBCData();
162  rof.setFirstEntry(tracks.size());
163  rof.setNEntries(0);
164  continue;
165  }
166  }
167 
168  if (mRunVertexer) {
169  o2::its::ioutils::loadROFrameData(rof, event, compClusters, pattIt, mDict, labels.get());
170  mVertexerPtr->clustersToVertices(event, false, [&](std::string s) { LOG(info) << s; });
171  }
172  auto vtxVecLoc = mVertexerPtr->exportVertices();
173 
174  if (multEstConf.cutMultVtxLow > 0 || multEstConf.cutMultVtxHigh > 0) { // cut was requested
175  std::vector<o2::dataformats::Vertex<o2::dataformats::TimeStamp<int>>> vtxVecSel;
176  vtxVecSel.swap(vtxVecLoc);
177  for (const auto& vtx : vtxVecSel) {
178  if (vtx.getNContributors() < multEstConf.cutMultVtxLow || (multEstConf.cutMultVtxHigh > 0 && vtx.getNContributors() > multEstConf.cutMultVtxHigh)) {
179  LOG(info) << "Found vertex mult. " << vtx.getNContributors() << " is outside of requested range "
180  << multEstConf.cutMultVtxLow << " : " << multEstConf.cutMultVtxHigh << " | ROF " << rof.getBCData();
181  continue; // skip vertex of unwanted multiplicity
182  }
183  vtxVecLoc.push_back(vtx);
184  }
185  }
186  if (vtxVecLoc.empty()) {
187  if (multEstConf.cutMultVtxLow < 1) { // do blind search only if there is no cut on the low mult vertices
188  vtxVecLoc.emplace_back();
189  } else {
190  rof.setFirstEntry(tracks.size());
191  rof.setNEntries(0);
192  continue;
193  }
194  } else { // save vetrices
195  vtxROF.setNEntries(vtxVecLoc.size());
196  for (const auto& vtx : vtxVecLoc) {
197  vertices.push_back(vtx);
198  }
199  }
200  mTracker.setVertices(vtxVecLoc);
201  mTracker.process(compClusters, it, mDict, tracks, clusIdx, rof);
202  if (tracks.size()) {
203  irFrames.emplace_back(rof.getBCData(), rof.getBCData() + nBCPerTF - 1);
204  }
205  }
206 
207  LOG(info) << "ITSCookedTracker pushed " << tracks.size() << " tracks";
208 
209  if (mUseMC) {
210  pc.outputs().snapshot(Output{"ITS", "TRACKSMCTR", 0, Lifetime::Timeframe}, trackLabels);
211  pc.outputs().snapshot(Output{"ITS", "ITSTrackMC2ROF", 0, Lifetime::Timeframe}, mc2rofs);
212  }
213  mTimer.Stop();
214 }

222 DataProcessorSpec getCookedTrackerSpec(bool useMC, const std::string& trMode)
223 {
224  std::vector<InputSpec> inputs;
225  inputs.emplace_back("compClusters", "ITS", "COMPCLUSTERS", 0, Lifetime::Timeframe);
226  inputs.emplace_back("patterns", "ITS", "PATTERNS", 0, Lifetime::Timeframe);
227  inputs.emplace_back("ROframes", "ITS", "CLUSTERSROF", 0, Lifetime::Timeframe);
228 
229  std::vector<OutputSpec> outputs;
230  outputs.emplace_back("ITS", "TRACKS", 0, Lifetime::Timeframe);
231  outputs.emplace_back("ITS", "TRACKCLSID", 0, Lifetime::Timeframe);
232  outputs.emplace_back("ITS", "ITSTrackROF", 0, Lifetime::Timeframe);
233  outputs.emplace_back("ITS", "VERTICES", 0, Lifetime::Timeframe);
234  outputs.emplace_back("ITS", "VERTICESROF", 0, Lifetime::Timeframe);
235  outputs.emplace_back("ITS", "IRFRAMES", 0, Lifetime::Timeframe);
236 
237  if (useMC) {
238  inputs.emplace_back("labels", "ITS", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
239  inputs.emplace_back("MC2ROframes", "ITS", "CLUSTERSMC2ROF", 0, Lifetime::Timeframe);
240  outputs.emplace_back("ITS", "TRACKSMCTR", 0, Lifetime::Timeframe);
241  outputs.emplace_back("ITS", "ITSTrackMC2ROF", 0, Lifetime::Timeframe);
242  }
243 
244  return DataProcessorSpec{
245  "its-cooked-tracker",
246  inputs,
247  outputs,
248  AlgorithmSpec{adaptFromTask<CookedTrackerDPL>(useMC, trMode)},
249  Options{
250  {"grp-file", VariantType::String, "o2sim_grp.root", {"Name of the grp file"}},
251  {"nthreads", VariantType::Int, 1, {"Number of threads"}}}};
252 }
