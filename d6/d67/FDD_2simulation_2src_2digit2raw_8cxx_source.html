100 void digit2raw(const std::string& inpName, const std::string& outDir, int verbosity, const std::string& fileFor, uint32_t rdhV, bool noEmptyHBF, const std::string& flpName, const std::string& ccdbUrl, const std::string& lutPath, int superPageSizeInB)
101 {
102  TStopwatch swTot;
103  swTot.Start();
104  o2::fdd::RawWriterFDD m2r;
105  m2r.setFileFor(fileFor);
106  m2r.setFlpName(flpName);
107  m2r.setVerbosity(verbosity);
108  if (ccdbUrl != "") {
109  m2r.setCCDBurl(ccdbUrl);
110  }
111  if (lutPath != "") {
112  m2r.setLUTpath(lutPath);
113  }
114  auto& wr = m2r.getWriter();
115  std::string inputGRP = o2::base::NameConf::getGRPFileName();
116  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
117  wr.setContinuousReadout(grp->isDetContinuousReadOut(o2::detectors::DetID::FDD)); // must be set explicitly
118  wr.setSuperPageSize(superPageSizeInB);
119  wr.useRDHVersion(rdhV);
120  wr.setDontFillEmptyHBF(noEmptyHBF);
121 
122  o2::raw::assertOutputDirectory(outDir);
123 
124  std::string outDirName(outDir);
125  if (outDirName.back() != '/') {
126  outDirName += '/';
127  }
128 
129  m2r.convertDigitsToRaw(outDirName, inpName);
130  wr.writeConfFile(wr.getOrigin().str, "RAWDATA", o2::utils::Str::concat_string(outDirName, wr.getOrigin().str, "raw.cfg"));
131  //
132  swTot.Stop();
133  swTot.Print();
134 }
