70 void CompressedAnalysisTask::run(ProcessingContext& pc)
71 {
72 
74  if (mStatus) {
75  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
76  return;
77  }
78 
80  for (auto iit = pc.inputs().begin(), iend = pc.inputs().end(); iit != iend; ++iit) {
81  if (!iit.isValid()) {
82  continue;
83  }
84 
86  for (auto const& ref : iit) {
87 
88  const auto* headerIn = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
89  auto payloadIn = ref.payload;
90  auto payloadInSize = headerIn->payloadSize;
91 
92  mAnalysis->setDecoderBuffer(payloadIn);
93  mAnalysis->setDecoderBufferSize(payloadInSize);
94  mAnalysis->run();
95  }
96  }
