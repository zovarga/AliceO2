51  void init(InitContext& ic)
52  {
53  auto filename = utils::Str::concat_string(utils::Str::rectifyDirectory(ic.options().get<std::string>("input-dir")),
54  ic.options().get<std::string>("mid-digit-infile"));
55  if (mUseMC) {
56  mReader = std::make_unique<RootTreeReader>("o2sim", filename.c_str(), -1,
57  RootTreeReader::PublishingMode::Single,
58  RootTreeReader::BranchDefinition<std::vector<ColumnDataMC>>{
59  Output{header::gDataOriginMID, mDescriptions[0], 0, Lifetime::Timeframe}, "MIDDigit"},
60  RootTreeReader::BranchDefinition<std::vector<ROFRecord>>{
61  Output{header::gDataOriginMID, mDescriptions[1], 0, Lifetime::Timeframe}, "MIDROFRecords"},
62  RootTreeReader::BranchDefinition<dataformats::MCTruthContainer<MCLabel>>{
63  Output{header::gDataOriginMID, mDescriptions[2], 0, Lifetime::Timeframe}, "MIDDigitMCLabels"},
64  &mPublishDigits);
65  } else {
66  mReader = std::make_unique<RootTreeReader>("o2sim", filename.c_str(), -1,
67  RootTreeReader::PublishingMode::Single,
68  RootTreeReader::BranchDefinition<std::vector<ColumnDataMC>>{
69  Output{header::gDataOriginMID, mDescriptions[0], 0, Lifetime::Timeframe}, "MIDDigit"},
70  RootTreeReader::BranchDefinition<std::vector<ROFRecord>>{
71  Output{header::gDataOriginMID, mDescriptions[1], 0, Lifetime::Timeframe}, "MIDROFRecords"},
72  &mPublishDigits);
73  }
74  }

89  RootTreeReader::SpecialPublishHook mPublishDigits{
90  [](std::string_view name, ProcessingContext& pc, Output const& output, char* data) -> bool {
91  if (name == "MIDDigit") {
92  auto inputDigits = reinterpret_cast<std::vector<ColumnDataMC>*>(data);
93  std::vector<ColumnData> digits{};
94  digits.insert(digits.end(), inputDigits->begin(), inputDigits->end());
95  pc.outputs().snapshot(output, digits);
96  LOG(debug) << "MIDDigitsReader pushed " << digits.size() << " digits";
97  return true;
98  }
99  return false;
100  }};
101 };

103 DataProcessorSpec getDigitReaderSpec(bool useMC, const char* baseDescription)
104 {
105  std::vector<OutputSpec> outputs;
106  std::vector<header::DataDescription> descriptions;
107  std::stringstream ss;
108  ss << "A:" << header::gDataOriginMID.as<std::string>() << "/" << baseDescription << "/0";
109  ss << ";B:" << header::gDataOriginMID.as<std::string>() << "/" << baseDescription << "ROF/0";
110  if (useMC) {
111  ss << ";C:" << header::gDataOriginMID.as<std::string>() << "/" << baseDescription << "LABELS/0";
112  }
113  auto matchers = select(ss.str().c_str());
114  for (auto& matcher : matchers) {
115  outputs.emplace_back(DataSpecUtils::asOutputSpec(matcher));
116  descriptions.emplace_back(DataSpecUtils::asConcreteDataDescription(matcher));
117  }
118 
119  return DataProcessorSpec{
120  "MIDDigitsReader",
121  Inputs{},
122  outputs,
123  AlgorithmSpec{adaptFromTask<DigitsReaderDeviceDPL>(useMC, descriptions)},
124  Options{{"mid-digit-infile", VariantType::String, "middigits.root", {"Name of the input file"}},
125  {"input-dir", VariantType::String, "none", {"Input directory"}}}};
126 }
