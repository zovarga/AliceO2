96 std::tuple<int, int> CaloRawFitter::findPeakRegion(const gsl::span<double> adcValues, short indexMaxADC, int threshold) const
97 {
98 
99  int first(0), last(0);
100  int tmpfirst = indexMaxADC;
101  int tmplast = indexMaxADC;
102  double prevFirst = adcValues[indexMaxADC];
103  double prevLast = adcValues[indexMaxADC];
104  bool firstJump = false;
105  bool lastJump = false;
106 
107  while ((tmpfirst >= 0) && (adcValues[tmpfirst] >= threshold) && (!firstJump)) {
108  // jump check:
109  if (tmpfirst != indexMaxADC) { // neighbor to maxindex can share peak with maxindex
110  if (adcValues[tmpfirst] >= prevFirst) {
111  firstJump = true;
112  }
113  }
114  prevFirst = adcValues[tmpfirst];
115  tmpfirst--;
116  }
117 
118  while ((tmplast < adcValues.size()) && (adcValues[tmplast] >= threshold) && (!lastJump)) {
119  // jump check:
120  if (tmplast != indexMaxADC) { // neighbor to maxindex can share peak with maxindex
121  if (adcValues[tmplast] >= prevLast) {
122  lastJump = true;
123  }
124  }
125  prevLast = adcValues[tmplast];
126  tmplast++;
127  }
128 
129  // we keep one pre- or post- sample if we can (as in online)
130  // check though if we ended up on a 'jump', or out of bounds: if so, back up
131  if (firstJump || tmpfirst < 0) {
132  tmpfirst++;
133  }
134  if (lastJump || tmplast >= adcValues.size()) {
135  tmplast--;
136  }
137 
138  first = tmpfirst;
139  last = tmplast;
140 
141  return std::make_tuple(first, last);
142 }

144 std::tuple<float, std::array<double, constants::EMCAL_MAXTIMEBINS>> CaloRawFitter::reverseAndSubtractPed(const Bunch& bunch) const
145 {
146  std::array<double, constants::EMCAL_MAXTIMEBINS> outarray;
147  int length = bunch.getBunchLength();
148  const gsl::span<const uint16_t> sig(bunch.getADC());
149 
150  double ped = mIsZerosupressed ? 0. : evaluatePedestal(sig, length);
151 
152  for (int i = 0; i < length; i++) {
153  outarray[i] = sig[length - i - 1] - ped;
154  }
155 
156  return std::make_tuple(ped, outarray);
157 }

159 double CaloRawFitter::evaluatePedestal(const gsl::span<const uint16_t> data, std::optional<int> length) const
160 {
161  if (!mNsamplePed) {
162  throw RawFitterError_t::SAMPLE_UNINITIALIZED;
163  }
164  if (data.size() < mNsamplePed) {
165  return 0.;
166  }
167  return static_cast<double>(std::accumulate(data.begin(), data.begin() + mNsamplePed, 0)) / mNsamplePed;
168 }

207 std::tuple<short, short, short> CaloRawFitter::selectMaximumBunch(const gsl::span<const Bunch>& bunchvector)
208 {
209  short bunchindex = -1;
210  short indexMaxInBunch(0), maxADCallBunches(-1);
211 
212  for (unsigned int i = 0; i < bunchvector.size(); i++) {
213  auto [maxADC, maxIndex] = getMaxAmplitudeBunch(bunchvector[i]); // CRAP PTH, bug fix, trouble if more than one bunches
214  if (isInTimeRange(maxIndex, mMaxTimeIndex, mMinTimeIndex)) {
215  if (maxADC > maxADCallBunches) {
216  bunchindex = i;
217  indexMaxInBunch = maxIndex;
218  maxADCallBunches = maxADC;
219  }
220  }
221  }
222 
223  if (bunchindex >= 0) {
224  // reject bunch if the max. ADC value is at the edges of a bunch
225  if (isMaxADCBunchEdge(bunchvector[bunchindex])) {
226  throw RawFitterError_t::BUNCH_NOT_OK;
227  }
228  }
229 
230  return std::make_tuple(bunchindex, indexMaxInBunch, maxADCallBunches);
231 }

242 double CaloRawFitter::calculateChi2(double amp, double time,
243  int first, int last,
244  double adcErr, double tau) const
245 {
246  if (first == last || first < 0) { // signal consists of single sample, chi2 estimate (0) not too well defined..
247  // or, first is negative, the indices are not valid
248  throw RawFitterError_t::CHI2_ERROR;
249  }
250 
251  int nsamples = last - first + 1;
252 
253  double chi2 = 0;
254 
255  for (int i = 0; i < nsamples; i++) {
256  int x = first + i; // timebin
257  double xx = (x - time + tau) / tau; // help variable
258  double f = 0.0;
259 
260  if (xx > 0) {
261  f = amp * xx * xx * TMath::Exp(2 * (1 - xx));
262  }
263 
264  double dy = mReversed[x] - f;
265  chi2 += dy * dy;
266  }
267 
268  if (adcErr > 0.0) { // weight chi2
269  chi2 /= (adcErr * adcErr);
270  }
271 
272  return chi2;
273 }
274 std::tuple<int, int, float, short, short, float, int, int> CaloRawFitter::preFitEvaluateSamples(const gsl::span<const Bunch> bunchvector, int adcThreshold)
275 {
276 
277  int nsamples(0), first(0), last(0), indexMaxADCRReveresed(0);
278  double peakADC(0.), pedestal(0.);
279 
280  // Reset buffer for reversed bunch, no matter whether the bunch could be selected or not
281  mReversed.fill(0);
282 
283  // select the bunch with the highest amplitude unless any time constraints is set
284  auto [bunchindex, indexMaxADC, adcMAX] = selectMaximumBunch(bunchvector);
285 
286  // something valid was found, and non-zero amplitude
287  if (bunchindex >= 0) {
288  if (adcMAX >= adcThreshold) {
289  // use more convenient numbering and possibly subtract pedestal
290 
291  int bunchlength = bunchvector[bunchindex].getBunchLength();
292  const std::vector<uint16_t>& sig = bunchvector[bunchindex].getADC();
293 
294  if (!mIsZerosupressed) {
295  pedestal = evaluatePedestal(sig, bunchlength);
296  }
297 
298  int testindexReverse = -1;
299  for (int i = 0; i < bunchlength; i++) {
300  mReversed[i] = sig[bunchlength - i - 1] - pedestal;
301  if (mReversed[i] > peakADC) {
302  peakADC = mReversed[i];
303  testindexReverse = i;
304  }
305  }
306 
307  if (peakADC >= adcThreshold) // possibly significant signal
308  {
309  // select array around max to possibly be used in fit
310  indexMaxADCRReveresed = indexMaxADC - bunchvector[bunchindex].getStartTime();
311  std::tie(first, last) = findPeakRegion(gsl::span<double>(&mReversed[0], bunchvector[bunchindex].getBunchLength()), indexMaxADCRReveresed, adcThreshold);
312 
313  // sanity check: maximum should not be in first or last bin
314  // if we should do a fit
315  if (first != indexMaxADCRReveresed && last != indexMaxADCRReveresed) {
316  // calculate how many samples we have
317  nsamples = last - first + 1;
318  }
319  }
320  } else {
321  throw RawFitterError_t::LOW_SIGNAL;
322  }
323  } else {
324  throw RawFitterError_t::BUNCH_NOT_OK;
325  }
326 
327  return std::make_tuple(nsamples, bunchindex, peakADC, adcMAX, indexMaxADCRReveresed, pedestal, first, last);
328 }
