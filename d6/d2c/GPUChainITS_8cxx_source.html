80 int GPUChainITS::RunITSTrackFit(std::vector<o2::its::Road>& roads, std::vector<const o2::its::Cluster*>& clusters, std::vector<const o2::its::Cell*>& cells, const std::vector<std::vector<o2::its::TrackingFrameInfo>>& tf, std::vector<o2::its::TrackITSExt>& tracks)

83  bool doGPU = GetRecoStepsGPU() & RecoStep::ITSTracking;
84  GPUITSFitter& Fitter = processors()->itsFitter;
85  GPUITSFitter& FitterShadow = doGPU ? processorsShadow()->itsFitter : Fitter;
86 
87  Fitter.clearMemory();
88  Fitter.SetNumberOfRoads(roads.size());
89  for (unsigned int i = 0; i < clusters.size(); i++) {
90  Fitter.SetNumberTF(i, tf[i].size());

92  Fitter.SetMaxData(processors()->ioPtrs);
93  std::copy(clusters.begin(), clusters.end(), Fitter.clusters());

96  std::copy(roads.begin(), roads.end(), Fitter.roads());

99  }
100 
101  WriteToConstantMemory(RecoStep::ITSTracking, (char*)&processors()->itsFitter - (char*)processors(), &FitterShadow, sizeof(FitterShadow), 0);
102  TransferMemoryResourcesToGPU(RecoStep::ITSTracking, &Fitter, 0);
103  runKernel<GPUITSFitterKernel>(GetGridBlk(BlockCount(), 0), krnlRunRangeNone, krnlEventNone);
104  TransferMemoryResourcesToHost(RecoStep::ITSTracking, &Fitter, 0);
105 
106  SynchronizeGPU();
107 
108  for (unsigned int i = 0; i < Fitter.NumberOfTracks(); i++) {
109  auto& trkin = Fitter.tracks()[i];
110 
111  tracks.emplace_back(TrackITSExt{{trkin.X(),

115  (short int)((trkin.NDF() + 5) / 2),
116  trkin.Chi2(),
117  {trkin.mOuterParam.X,
118  trkin.mOuterParam.alpha,
119  {trkin.mOuterParam.P[0], trkin.mOuterParam.P[1], trkin.mOuterParam.P[2], trkin.mOuterParam.P[3], trkin.mOuterParam.P[4]},
120  {trkin.mOuterParam.C[0], trkin.mOuterParam.C[1], trkin.mOuterParam.C[2], trkin.mOuterParam.C[3], trkin.mOuterParam.C[4], trkin.mOuterParam.C[5], trkin.mOuterParam.C[6], trkin.mOuterParam.C[7], trkin.mOuterParam.C[8], trkin.mOuterParam.C[9],
121  trkin.mOuterParam.C[10], trkin.mOuterParam.C[11], trkin.mOuterParam.C[12], trkin.mOuterParam.C[13], trkin.mOuterParam.C[14]}},

123  }
