107 template <typename Input, typename Container>
108 template <typename DATA>
109 bool TimeSlotCalibration<Input, Container>::process(TFType tf, const DATA& data)
110 {
111 
112  // process current TF
113 
114  int maxDelay = mMaxSlotsDelay * mSlotLength;
115  if (!mUpdateAtTheEndOfRunOnly) { // if you update at the end of run only, then you accept everything
116  if (tf < mLastClosedTF || (!mSlots.empty() && getLastSlot().getTFStart() > tf + maxDelay)) { // ignore TF; note that if you have only 1 timeslot
117  // which is INFINITE_TF wide, then maxDelay
118  // does not matter: you won't accept TFs from the past,
119  // so the first condition will be used
120  LOG(info) << "Ignoring TF " << tf << ", mLastClosedTF = " << mLastClosedTF;
121  return false;
122  }
123  }
124 
125  auto& slotTF = getSlotForTF(tf);
126  slotTF.getContainer()->fill(data);
127  if (tf > mMaxSeenTF) {
128  mMaxSeenTF = tf; // keep track of the most recent TF processed
129  }
130  if (!mUpdateAtTheEndOfRunOnly) { // if you update at the end of run only, you don't check at every TF which slots can be closed
131  // check if some slots are done
132  checkSlotsToFinalize(tf, maxDelay);
133  }
134 
135  return true;
136 }

170 template <typename Input, typename Container>
171 void TimeSlotCalibration<Input, Container>::checkSlotsToFinalize(TFType tf, int maxDelay)
172 {
173  // Check which slots can be finalized, provided the newly arrived TF is tf
174 
175  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
176  constexpr int64_t INFINITE_TF_int64 = std::numeric_limits<long>::max() - 1; // this is used to define the end
177  // of the slot in case it is "std::numeric_limits<long>::max()"
178  // long (so we need to subtract 1)
179 
180  // if we have one slot only which is INFINITE_TF_int64 long, and we are not at the end of run (tf != INFINITE_TF),
181  // we need to check if we got enough statistics, and if so, redefine the slot
182  if (mSlots.size() == 1 && mSlots[0].getTFEnd() == INFINITE_TF_int64) {
183  uint64_t checkInterval = mCheckIntervalInfiniteSlot + mLastCheckedTFInfiniteSlot;
184  if (mWasCheckedInfiniteSlot) {
185  checkInterval = mCheckDeltaIntervalInfiniteSlot + mLastCheckedTFInfiniteSlot;
186  }
187  if (tf >= checkInterval || tf == INFINITE_TF) {
188  LOG(debug) << "mMaxSeenTF = " << mMaxSeenTF << ", mLastCheckedTFInfiniteSlot = " << mLastCheckedTFInfiniteSlot << ", checkInterval = " << checkInterval << ", mSlots[0].getTFStart() = " << mSlots[0].getTFStart();
189  if (tf == INFINITE_TF) {
190  LOG(info) << "End of run reached, trying to calibrate what we have, if we have enough statistics";
191  } else {
192  LOG(info) << "Calibrating as soon as we have enough statistics:";
193  LOG(info) << "Update interval passed (" << checkInterval << "), checking slot for " << mSlots[0].getTFStart() << " <= TF <= " << mSlots[0].getTFEnd();
194  }
195  mLastCheckedTFInfiniteSlot = tf;
196  if (hasEnoughData(mSlots[0])) {
197  mWasCheckedInfiniteSlot = false;
198  mSlots[0].setTFStart(mLastClosedTF);
199  mSlots[0].setTFEnd(mMaxSeenTF);
200  LOG(info) << "Finalizing slot for " << mSlots[0].getTFStart() << " <= TF <= " << mSlots[0].getTFEnd();
201  finalizeSlot(mSlots[0]); // will be removed after finalization
202  mLastClosedTF = mSlots[0].getTFEnd() + 1; // will not accept any TF below this
203  mSlots.erase(mSlots.begin());
204  // creating a new slot if we are not at the end of run
205  if (tf != INFINITE_TF) {
206  LOG(info) << "Creating new slot for " << mLastClosedTF << " <= TF <= " << INFINITE_TF_int64;
207  emplaceNewSlot(true, mLastClosedTF, INFINITE_TF_int64);
208  }
209  } else {
210  LOG(info) << "Not enough data to calibrate";
211  mWasCheckedInfiniteSlot = true;
212  }
213  } else {
214  LOG(debug) << "Not trying to calibrate: either not at EoS, or update interval not passed";
215  }
216  } else {
217  // check if some slots are done
218  for (auto slot = mSlots.begin(); slot != mSlots.end();) {
219  //if (maxDelay == 0 || (slot->getTFEnd() + maxDelay) < tf) {
220  if ((slot->getTFEnd() + maxDelay) < tf) {
221  if (hasEnoughData(*slot)) {
222  LOG(debug) << "Finalizing slot for " << slot->getTFStart() << " <= TF <= " << slot->getTFEnd();
223  finalizeSlot(*slot); // will be removed after finalization
224  } else if ((slot + 1) != mSlots.end()) {
225  LOG(info) << "Merging underpopulated slot " << slot->getTFStart() << " <= TF <= " << slot->getTFEnd()
226  << " to slot " << (slot + 1)->getTFStart() << " <= TF <= " << (slot + 1)->getTFEnd();
227  (slot + 1)->mergeToPrevious(*slot);
228  } else {
229  LOG(info) << "Discard underpopulated slot " << slot->getTFStart() << " <= TF <= " << slot->getTFEnd();
230  break; // slot has no enough stat. and there is no other slot to merge it to
231  }
232  mLastClosedTF = slot->getTFEnd() + 1; // will not accept any TF below this
233  LOG(info) << "closing slot " << slot->getTFStart() << " <= TF <= " << slot->getTFEnd();
234  slot = mSlots.erase(slot);
235  } else {
236  break; // all following slots will be even closer to the new TF
237  }
238  }
239  }
240 }

243 template <typename Input, typename Container>
244 void TimeSlotCalibration<Input, Container>::finalizeOldestSlot()
245 {
246  // Enforce finalization and removal of the oldest slot
247  if (mSlots.empty()) {
248  LOG(warning) << "There are no slots defined";
249  return;
250  }
251  finalizeSlot(mSlots.front());
252  mLastClosedTF = mSlots.front().getTFEnd() + 1; // do not accept any TF below this
253  mSlots.erase(mSlots.begin());
254 }

257 template <typename Input, typename Container>
258 inline TFType TimeSlotCalibration<Input, Container>::tf2SlotMin(TFType tf) const
259 {
260 
261  // returns the min TF of the slot to which "tf" belongs
262 
263  if (tf < mFirstTF) {
264  throw std::runtime_error("invalide TF");
265  }
266  if (mUpdateAtTheEndOfRunOnly) {
267  return mFirstTF;
268  }
269  return TFType((tf - mFirstTF) / mSlotLength) * mSlotLength + mFirstTF;
270 }

273 template <typename Input, typename Container>
274 TimeSlot<Container>& TimeSlotCalibration<Input, Container>::getSlotForTF(TFType tf)
275 {
276 
277  LOG(debug) << "Getting slot for TF " << tf;
278 
279  if (mUpdateAtTheEndOfRunOnly) {
280  if (!mSlots.empty() && mSlots.back().getTFEnd() < tf) {
281  mSlots.back().setTFEnd(tf);
282  } else if (mSlots.empty()) {
283  emplaceNewSlot(true, mFirstTF, tf);
284  }
285  return mSlots.back();
286  }
287 
288  if (!mSlots.empty() && mSlots.front().getTFStart() > tf) { // we need to add a slot to the beginning
289  auto tfmn = tf2SlotMin(mSlots.front().getTFStart() - 1); // min TF of the slot corresponding to a TF smaller than the first seen
290  auto tftgt = tf2SlotMin(tf); // min TF of the slot to which the TF "tf" would belong
291  while (tfmn >= tftgt) {
292  LOG(info) << "Adding new slot for " << tfmn << " <= TF <= " << tfmn + mSlotLength - 1;
293  emplaceNewSlot(true, tfmn, tfmn + mSlotLength - 1);
294  if (!tfmn) {
295  break;
296  }
297  tfmn = tf2SlotMin(mSlots.front().getTFStart() - 1);
298  }
299  return mSlots[0];
300  }
301  for (auto it = mSlots.begin(); it != mSlots.end(); it++) {
302  auto rel = (*it).relateToTF(tf);
303  if (rel == 0) {
304  return (*it);
305  }
306  }
307  // need to add in the end
308  auto tfmn = mSlots.empty() ? tf2SlotMin(tf) : tf2SlotMin(mSlots.back().getTFEnd() + 1);
309  do {
310  LOG(info) << "Adding new slot for " << tfmn << " <= TF <= " << tfmn + mSlotLength - 1;
311  emplaceNewSlot(false, tfmn, tfmn + mSlotLength - 1);
312  tfmn = tf2SlotMin(mSlots.back().getTFEnd() + 1);
313  } while (tf > mSlots.back().getTFEnd());
314 
315  return mSlots.back();
316 }
