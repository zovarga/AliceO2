26 std::vector<double> helpers::makeLogBinning(const int nbins, const double min, const double max)
27 {
28  assert(min > 0);
29  assert(min < max);
30 
31  std::vector<double> binLim(nbins + 1);
32 
33  const double expMax = std::log(max / min);
34  const double binWidth = expMax / nbins;
35 
36  binLim[0] = min;
37  binLim[nbins] = max;
38 
39  for (Int_t i = 1; i < nbins; ++i) {
40  binLim[i] = min * std::exp(i * binWidth);
41  }
42 
43  return binLim;
44 }

82
