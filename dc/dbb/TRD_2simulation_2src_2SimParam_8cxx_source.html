54 double SimParam::timeResponse(double time) const
55 {
56  //
57  // Applies the preamp shaper time response
58  // (We assume a signal rise time of 0.2us = fTRFlo/2.
59  //
60 
61  double rt = (time - .5 * mTRFlo) * mInvTRFwid;
62  int iBin = (int)rt;
63  double dt = rt - iBin;
64  if ((iBin >= 0) && (iBin + 1 < mTRFbin)) {
65  return mTRFsmp[iBin] + (mTRFsmp[iBin + 1] - mTRFsmp[iBin]) * dt;
66  } else {
67  return 0.0;
68  }
69 }

87 void SimParam::init()
88 {
89  //
90  // initializes the parameter class for a given gas mixture
91  //
92 
93  if (isXenon()) {
94  // The range and the binwidth for the sampled TRF
95  mTRFbin = 200;
96  // Start 0.2 mus before the signal
97  mTRFlo = -0.4;
98  // End the maximum drift time after the signal
99  mTRFhi = 3.58;
100  // Standard gas gain
101  mGasGain = 4000.0;
102  } else if (isArgon()) {
103  // The range and the binwidth for the sampled TRF
104  mTRFbin = 50;
105  // Start 0.2 mus before the signal
106  mTRFlo = 0.02;
107  // End the maximum drift time after the signal
108  mTRFhi = 1.98;
109  // Higher gas gain
110  mGasGain = 8000.0;
111  } else {
112  LOG(fatal) << "Not a valid gas mixture!\n";
113  }
114  mInvTRFwid = ((float)mTRFbin) / (mTRFhi - mTRFlo); // Inverse of the bin width of the integrated TRF
115 
116  // Create the sampled TRF
117  sampleTRF();
118 }
