22 BOOST_AUTO_TEST_CASE(TestTMessageSerializer)
23 {
24  const char* testname = "testname";
25  const char* testtitle = "testtitle";
26  using namespace o2::framework;
27 
28  TObjArray array;
29  array.SetOwner();
30  array.Add(new TNamed(testname, testtitle));
31 
32  FairTMessage msg;
33  TMessageSerializer::serialize(msg, &array);
34 
35  auto buf = as_span(msg);
36  BOOST_CHECK_EQUAL(buf.size(), msg.BufferSize());
37  BOOST_CHECK_EQUAL(static_cast<void*>(buf.data()), static_cast<void*>(msg.Buffer()));
38  // test deserialization with TObject as target class (default)
39  auto out = TMessageSerializer::deserialize(buf);
40 
41  TObjArray* outarr = dynamic_cast<TObjArray*>(out.get());
42  BOOST_CHECK_EQUAL(out.get(), outarr);
43  TNamed* named = dynamic_cast<TNamed*>(outarr->At(0));
44  BOOST_CHECK_EQUAL(static_cast<void*>(named), static_cast<void*>(outarr->At(0)));
45  BOOST_CHECK_EQUAL(named->GetName(), testname);
46  BOOST_CHECK_EQUAL(named->GetTitle(), testtitle);
47 
48  // test deserialization with a wrong target class and check the exception
49  BOOST_CHECK_EXCEPTION(TMessageSerializer::deserialize<TNamed>(buf),
50  RuntimeErrorRef,
51  [](RuntimeErrorRef const& ref) {
52  auto& e = error_from_ref(ref);
53  std::string expected("can not convert serialized class TObjArray into target class TNamed");
54  return expected == e.what;
55  });
56 }
