45 // we need to add workflow options before including Framework/runDataProcessing
46 void customize(std::vector<o2::framework::ConfigParamSpec>& workflowOptions)
47 {
48  // option allowing to set parameters
49  std::vector<o2::framework::ConfigParamSpec> options;
50  options.push_back(ConfigParamSpec{"ctf-input", VariantType::String, "none", {"comma-separated list CTF input files"}});
51  options.push_back(ConfigParamSpec{"onlyDet", VariantType::String, std::string{DetID::ALL}, {"comma-separated list of detectors to accept. Overrides skipDet"}});
52  options.push_back(ConfigParamSpec{"skipDet", VariantType::String, std::string{DetID::NONE}, {"comma-separate list of detectors to skip"}});
53  options.push_back(ConfigParamSpec{"max-tf", VariantType::Int, -1, {"max CTFs to process (<= 0 : infinite)"}});
54  options.push_back(ConfigParamSpec{"loop", VariantType::Int, 0, {"loop N times (infinite for N<0)"}});
55  options.push_back(ConfigParamSpec{"delay", VariantType::Float, 0.f, {"delay in seconds between consecutive TFs sending"}});
56  options.push_back(ConfigParamSpec{"copy-cmd", VariantType::String, "alien_cp ?src file://?dst", {"copy command for remote files or no-copy to avoid copying"}}); // Use "XrdSecPROTOCOL=sss,unix xrdcp -N root://eosaliceo2.cern.ch/?src ?dst" for direct EOS access
57  options.push_back(ConfigParamSpec{"ctf-file-regex", VariantType::String, ".*o2_ctf_run.+\\.root$", {"regex string to identify CTF files"}});
58  options.push_back(ConfigParamSpec{"remote-regex", VariantType::String, "^(alien://|)/alice/data/.+", {"regex string to identify remote files"}}); // Use "^/eos/aliceo2/.+" for direct EOS access
59  options.push_back(ConfigParamSpec{"max-cached-files", VariantType::Int, 3, {"max CTF files queued (copied for remote source)"}});
60  options.push_back(ConfigParamSpec{"allow-missing-detectors", VariantType::Bool, false, {"send empty message if detector is missing in the CTF (otherwise throw)"}});
61  options.push_back(ConfigParamSpec{"ctf-reader-verbosity", VariantType::Int, 0, {"verbosity level (0: summary per detector, 1: summary per block"}});
62  options.push_back(ConfigParamSpec{"configKeyValues", VariantType::String, "", {"Semicolon separated key=value strings"}});
63  //
64  options.push_back(ConfigParamSpec{"its-digits", VariantType::Bool, false, {"convert ITS clusters to digits"}});
65  options.push_back(ConfigParamSpec{"mft-digits", VariantType::Bool, false, {"convert MFT clusters to digits"}});
66 
67  std::swap(workflowOptions, options);
68 }

73 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
74 {
75  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
76  o2::ctf::CTFReaderInp ctfInput;
77 
78  WorkflowSpec specs;
79  std::string allowedDetectors = "ITS,TPC,TRD,TOF,PHS,CPV,EMC,HMP,MFT,MCH,MID,ZDC,FT0,FV0,FDD,CTP"; // FIXME: explicit list to avoid problem with upgrade detectors
80  auto mskOnly = DetID::getMask(configcontext.options().get<std::string>("onlyDet"));
81  auto mskSkip = DetID::getMask(configcontext.options().get<std::string>("skipDet"));
82  if (mskOnly.any()) {
83  ctfInput.detMask &= mskOnly;
84  } else {
85  ctfInput.detMask ^= mskSkip;
86  }
87  ctfInput.detMask &= DetID::getMask(allowedDetectors);
88  ctfInput.inpdata = configcontext.options().get<std::string>("ctf-input");
89  if (ctfInput.inpdata.empty() || ctfInput.inpdata == "none") {
90  if (!configcontext.helpOnCommandLine()) {
91  throw std::runtime_error("--ctf-input <file,...> is not provided");
92  }
93  ctfInput.inpdata = "";
94  }
95 
96  ctfInput.maxLoops = configcontext.options().get<int>("loop");
97  if (ctfInput.maxLoops < 0) {
98  ctfInput.maxLoops = 0x7fffffff;
99  }
100  ctfInput.delay_us = int32_t(1e6 * configcontext.options().get<float>("delay")); // delay in microseconds
101  if (ctfInput.delay_us < 0) {
102  ctfInput.delay_us = 0;
103  }
104  int n = configcontext.options().get<int>("max-tf");
105  ctfInput.maxTFs = n > 0 ? n : 0x7fffffff;
106 
107  ctfInput.maxFileCache = std::max(1, configcontext.options().get<int>("max-cached-files"));
108 
109  ctfInput.copyCmd = configcontext.options().get<std::string>("copy-cmd");
110  ctfInput.tffileRegex = configcontext.options().get<std::string>("ctf-file-regex");
111  ctfInput.remoteRegex = configcontext.options().get<std::string>("remote-regex");
112  ctfInput.allowMissingDetectors = configcontext.options().get<bool>("allow-missing-detectors");
113 
114  specs.push_back(o2::ctf::getCTFReaderSpec(ctfInput));
115  int verbosity = configcontext.options().get<int>("ctf-reader-verbosity");
116 
117  // add decodors for all allowed detectors.
118  if (ctfInput.detMask[DetID::ITS]) {
119  specs.push_back(o2::itsmft::getEntropyDecoderSpec(DetID::getDataOrigin(DetID::ITS), verbosity, configcontext.options().get<bool>("its-digits")));
120  }
121  if (ctfInput.detMask[DetID::MFT]) {
122  specs.push_back(o2::itsmft::getEntropyDecoderSpec(DetID::getDataOrigin(DetID::MFT), verbosity, configcontext.options().get<bool>("mft-digits")));
123  }
124  if (ctfInput.detMask[DetID::TPC]) {
125  specs.push_back(o2::tpc::getEntropyDecoderSpec(verbosity));
126  }
127  if (ctfInput.detMask[DetID::TRD]) {
128  specs.push_back(o2::trd::getEntropyDecoderSpec(verbosity));
129  }
130  if (ctfInput.detMask[DetID::TOF]) {
131  specs.push_back(o2::tof::getEntropyDecoderSpec(verbosity));
132  }
133  if (ctfInput.detMask[DetID::FT0]) {
134  specs.push_back(o2::ft0::getEntropyDecoderSpec(verbosity));
135  }
136  if (ctfInput.detMask[DetID::FV0]) {
137  specs.push_back(o2::fv0::getEntropyDecoderSpec(verbosity));
138  }
139  if (ctfInput.detMask[DetID::FDD]) {
140  specs.push_back(o2::fdd::getEntropyDecoderSpec(verbosity));
141  }
142  if (ctfInput.detMask[DetID::MID]) {
143  specs.push_back(o2::mid::getEntropyDecoderSpec(verbosity));
144  }
145  if (ctfInput.detMask[DetID::MCH]) {
146  specs.push_back(o2::mch::getEntropyDecoderSpec(verbosity));
147  }
148  if (ctfInput.detMask[DetID::EMC]) {
149  specs.push_back(o2::emcal::getEntropyDecoderSpec(verbosity));
150  }
151  if (ctfInput.detMask[DetID::PHS]) {
152  specs.push_back(o2::phos::getEntropyDecoderSpec(verbosity));
153  }
154  if (ctfInput.detMask[DetID::CPV]) {
155  specs.push_back(o2::cpv::getEntropyDecoderSpec(verbosity));
156  }
157  if (ctfInput.detMask[DetID::ZDC]) {
158  specs.push_back(o2::zdc::getEntropyDecoderSpec(verbosity));
159  }
160  if (ctfInput.detMask[DetID::HMP]) {
161  specs.push_back(o2::hmpid::getEntropyDecoderSpec(verbosity));
162  }
163  if (ctfInput.detMask[DetID::CTP]) {
164  specs.push_back(o2::ctp::getEntropyDecoderSpec(verbosity));
165  }
166 
167  return std::move(specs);
168 }
