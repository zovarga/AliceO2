91 GPUdi() void GPUdEdx::checkSubThresh(int roc)
92 {
93  if (roc != mLastROC) {
94  if (mNSubThresh && mCount + mNSubThresh <= MAX_NCL) {
95  for (int i = 0; i < mNSubThresh; i++) {
96  mChargeTot[mCount] = (GPUCA_DEDX_STORAGE_TYPE)(mSubThreshMinTot * scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::factor + scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::round);
97  mChargeMax[mCount++] = (GPUCA_DEDX_STORAGE_TYPE)(mSubThreshMinMax * scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::factor + scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::round);
98  }
99  mNClsROC[mLastROC] += mNSubThresh;
100  mNClsROCSubThresh[mLastROC] += mNSubThresh;
101  }
102  mNSubThresh = 0;
103  mSubThreshMinTot = 1e10f;
104  mSubThreshMinMax = 1e10f;
105  }
106 
107  mLastROC = roc;
108 }

110 GPUdnii() void GPUdEdx::fillCluster(float qtot, float qmax, int padRow, unsigned char slice, float trackSnp, float trackTgl, const GPUParam& GPUrestrict() param, const GPUCalibObjectsConst& calib, float z, float pad, float relTime)
111 {
112  if (mCount >= MAX_NCL) {
113  return;
114  }
115 
116  // container containing all the dE/dx corrections
117  auto calibContainer = calib.dEdxCalibContainer;
118 
119  const int roc = param.tpcGeometry.GetROC(padRow);
120  checkSubThresh(roc);
121  float snp2 = trackSnp * trackSnp;
122  if (snp2 > GPUCA_MAX_SIN_PHI_LOW) {
123  snp2 = GPUCA_MAX_SIN_PHI_LOW;
124  }
125 
126  // setting maximum for snp for which the calibration object was created
127  const float maxSnp = calibContainer->getMaxSinPhiTopologyCorrection();
128  float snp = CAMath::Abs(trackSnp);
129  if (snp > maxSnp) {
130  snp = maxSnp;
131  }
132 
133  // tanTheta local dip angle: z angle - dz/dx (cm/cm)
134  const float sec2 = 1.f / (1.f - snp2);
135  const float tgl2 = trackTgl * trackTgl;
136  float tanTheta = CAMath::Sqrt(tgl2 * sec2);
137  const float maxTanTheta = calibContainer->getMaxTanThetaTopologyCorrection();
138  if (tanTheta > maxTanTheta) {
139  tanTheta = maxTanTheta;
140  }
141 
142  // getting the topology correction
143  const float absRelPad = CAMath::Abs(pad - int(pad + 0.5f));
144  const int region = param.tpcGeometry.GetRegion(padRow);
145  z = CAMath::Abs(z);
146  const float threshold = calibContainer->getZeroSupressionThreshold(slice, padRow, pad); // TODO: Use the mean zero supresion threshold of all pads in the cluster?
147  const float qMaxTopologyCorr = calibContainer->getTopologyCorrection(region, o2::tpc::ChargeType::Max, tanTheta, snp, z, absRelPad, relTime, threshold);
148  const float qTotTopologyCorr = calibContainer->getTopologyCorrection(region, o2::tpc::ChargeType::Tot, tanTheta, snp, z, absRelPad, relTime, threshold);
149 
150  qmax /= qMaxTopologyCorr;
151  qtot /= qTotTopologyCorr;
152 
153  tpc::StackID stack{
154  slice,
155  static_cast<tpc::GEMstack>(roc)};
156 
157  // getting the residual dE/dx correction
158  const float qMaxResidualCorr = calibContainer->getResidualCorrection(stack, tpc::ChargeType::Max, z, trackTgl);
159  const float qTotResidualCorr = calibContainer->getResidualCorrection(stack, tpc::ChargeType::Tot, z, trackTgl);
160  qmax /= qMaxResidualCorr;
161  qtot /= qTotResidualCorr;
162 
163  mChargeTot[mCount] = (GPUCA_DEDX_STORAGE_TYPE)(qtot * scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::factor + scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::round);
164  mChargeMax[mCount++] = (GPUCA_DEDX_STORAGE_TYPE)(qmax * scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::factor + scalingFactor<GPUCA_DEDX_STORAGE_TYPE>::round);
165  mNClsROC[roc]++;
166  if (qtot < mSubThreshMinTot) {
167  mSubThreshMinTot = qtot;
168  }
169  if (qmax < mSubThreshMinMax) {
170  mSubThreshMinMax = qmax;
171  }
172 }
