117 void PedestalsCalculationTask::recordPedInFiles()
118 {
119  double Average;
120  double Variance;
121  double Samples;
122  double SumOfCharge;
123  double SumOfSquares;
124 
125  uint32_t Buffer;
126  uint32_t Pedestal;
127  uint32_t Threshold;
128  char padsFileName[1024];
129 
130  for (int e = 0; e < Geo::MAXEQUIPMENTS; e++) {
131  if (mDeco->getAverageEventSize(e) == 0) {
132  continue;
133  }
134  sprintf(padsFileName, "%s_%d.dat", mPedestalsBasePath.c_str(), e);
135  FILE* fpads = fopen(padsFileName, "w");
136  if (fpads == nullptr) {
137  mExTimer.logMes("error creating the file = " + std::string(padsFileName));
138  LOG(error) << "error creating the file = " << padsFileName;
139  return;
140  }
141  for (int c = 0; c < Geo::N_COLUMNS; c++) {
142  for (int d = 0; d < Geo::N_DILOGICS; d++) {
143  for (int h = 0; h < Geo::N_CHANNELS; h++) {
144  Samples = (double)mDeco->getChannelSamples(e, c, d, h);
145  SumOfCharge = mDeco->getChannelSum(e, c, d, h);
146  SumOfSquares = mDeco->getChannelSquare(e, c, d, h);
147 
148  if (Samples > 0) {
149  Average = SumOfCharge / Samples;
150  Variance = sqrt(abs((Samples * SumOfSquares) - (SumOfCharge * SumOfCharge))) / Samples;
151  } else {
152  Average = 0;
153  Variance = 0;
154  }
155  Pedestal = (uint32_t)Average;
156  Threshold = (uint32_t)(Variance * mSigmaCut + Average);
157  Buffer = ((Threshold & 0x001FF) << 9) | (Pedestal & 0x001FF);
158  fprintf(fpads, "%05X\n", Buffer);
159  }
160  for (int h = 48; h < 64; h++) {
161  fprintf(fpads, "%05X\n", 0);
162  }
163  }
164  }
165  mExTimer.logMes("End write the equipment = " + std::to_string(e));
166  fprintf(fpads, "%05X\n", 0xA0A0A);
167  fclose(fpads);
168  }
169  mExTimer.logMes("End Writing the pedestals ! Digits decoded = " + std::to_string(mTotalDigits) + " Frames received = " + std::to_string(mTotalFrames));
170  return;
171 }

173 void PedestalsCalculationTask::recordPedInDcsCcdb()
174 {
175  // create the root structure
176  LOG(info) << "Store Pedestals in DCS CCDB ";
177 
178  float xb, yb, ch, Samples;
179  double SumOfCharge, SumOfSquares, Average, Variance;
180  uint32_t Pedestal, Threshold, PedThr;
181 
182  o2::dcs::DCSconfigObject_t pedestalsConfig;
183 
184 // Setup dimensions for Equipment granularity with 48 channels/dilogic
185 #define PEDTHFORMAT "%05X,"
186 #define COLUMNTAIL false
187 #define PEDTHBYTES 6
188  int bufferDim = PEDTHBYTES * Geo::N_CHANNELS * Geo::N_DILOGICS * Geo::N_COLUMNS + 10;
189  char* outBuffer = new char[bufferDim];
190  char* inserPtr;
191 
192  for (int e = 0; e < Geo::MAXEQUIPMENTS; e++) {
193  if (mDeco->getAverageEventSize(e) == 0) { // skip the empty equipment
194  continue;
195  }
196  inserPtr = outBuffer;
197  // algoritm based on equipment granularity
198  for (int c = 0; c < Geo::N_COLUMNS; c++) {
199  for (int d = 0; d < Geo::N_DILOGICS; d++) {
200  for (int h = 0; h < Geo::N_CHANNELS; h++) {
201  Samples = (double)mDeco->getChannelSamples(e, c, d, h);
202  SumOfCharge = mDeco->getChannelSum(e, c, d, h);
203  SumOfSquares = mDeco->getChannelSquare(e, c, d, h);
204  if (Samples > 0) {
205  Average = SumOfCharge / Samples;
206  Variance = sqrt(abs((Samples * SumOfSquares) - (SumOfCharge * SumOfCharge))) / Samples;
207  } else {
208  Average = 0;
209  Variance = 0;
210  }
211  Pedestal = (uint32_t)Average;
212  Threshold = (uint32_t)(Variance * mSigmaCut + Average);
213  PedThr = ((Threshold & 0x001FF) << 9) | (Pedestal & 0x001FF);
214  sprintf(inserPtr, PEDTHFORMAT, PedThr);
215  inserPtr += PEDTHBYTES;
216  }
217  if (COLUMNTAIL) {
218  for (int h = 48; h < 64; h++) {
219  sprintf(inserPtr, PEDTHFORMAT, 0);
220  inserPtr += PEDTHBYTES;
221  }
222  }
223  }
224  }
225  mExTimer.logMes("End write the equipment = " + std::to_string(e));
226  sprintf(inserPtr, "%05X\n", 0xA0A0A); // The closure value
227  inserPtr += 6;
228  *inserPtr = '\0'; // close the string rap.
229  o2::dcs::addConfigItem(pedestalsConfig, "Equipment" + std::to_string(e), (const char*)outBuffer);
230  }
231 
232  struct timeval tp;
233  gettimeofday(&tp, nullptr);
234  uint64_t ms = tp.tv_sec * 1000 + tp.tv_usec / 1000;
235  uint64_t minTimeStamp = ms;
236  uint64_t maxTimeStamp = ms + 1;
237 
238  char filename[1024];
239  sprintf(filename, "%s/%s/PedThre.root", mPedestalsCCDBBasePath.c_str(), mPedestalTag.c_str());
240  mExTimer.logMes("File name = >" + std::string(filename) + "< (" + mPedestalsCCDBBasePath + "," + mPedestalTag);
241  TFile outputFile(filename, "recreate");
242  outputFile.WriteObjectAny(&pedestalsConfig, "std::vector<char>", "DCSConfig");
243  outputFile.Close();
244 
245  mDbMetadata.emplace("Tag", mPedestalTag.c_str());
246  mDBapi.storeAsTFileAny(&pedestalsConfig, filename, mDbMetadata, minTimeStamp, maxTimeStamp);
247 
248  mExTimer.logMes("End Writing the pedestals ! Digits decoded = " + std::to_string(mTotalDigits) + " Frames received = " + std::to_string(mTotalFrames));
249 
250  return;
251 }

253 void PedestalsCalculationTask::recordPedInCcdb()
254 {
255  // create the root structure
256  LOG(info) << "Store Pedestals in ccdb ";
257 
258  float xb, yb, ch;
259  double Samples, SumOfCharge, SumOfSquares, Average, Variance;
260 
261  TObjArray aSigmas(Geo::N_MODULES);
262  TObjArray aPedestals(Geo::N_MODULES);
263 
264  for (int i = 0; i < Geo::N_MODULES; i++) {
265  aSigmas.AddAt(new TMatrixF(Geo::N_XROWS, Geo::N_YCOLS), i);
266  aPedestals.AddAt(new TMatrixF(Geo::N_XROWS, Geo::N_YCOLS), i);
267  }
268 
269  for (int m = 0; m < o2::hmpid::Geo::N_MODULES; m++) {
270  if (mDeco->getAverageEventSize(m * 2) == 0 && mDeco->getAverageEventSize(m * 2 + 1) == 0) {
271  continue; // If no events skip the chamber
272  }
273  TMatrixF* pS = (TMatrixF*)aSigmas.At(m);
274  TMatrixF* pP = (TMatrixF*)aPedestals.At(m);
275 
276  for (int x = 0; x < o2::hmpid::Geo::N_XROWS; x++) {
277  for (int y = 0; y < o2::hmpid::Geo::N_YCOLS; y++) {
278 
279  Samples = (double)mDeco->getPadSamples(m, x, y);
280  SumOfCharge = mDeco->getPadSum(m, x, y);
281  SumOfSquares = mDeco->getPadSquares(m, x, y);
282  if (Samples > 0) {
283  (*pP)(x, y) = SumOfCharge / Samples;
284  (*pS)(x, y) = sqrt(abs((Samples * SumOfSquares) - (SumOfCharge * SumOfCharge))) / Samples;
285  } else {
286  (*pP)(x, y) = 0;
287  (*pS)(x, y) = 0;
288  }
289  }
290  }
291  }
292  struct timeval tp;
293  gettimeofday(&tp, nullptr);
294  uint64_t ms = tp.tv_sec * 1000 + tp.tv_usec / 1000;
295  uint64_t minTimeStamp = ms;
296  uint64_t maxTimeStamp = ms + 1;
297 
298  for (int i = 0; i < Geo::N_MODULES; i++) {
299  if (mDeco->getAverageEventSize(i * 2) == 0 && mDeco->getAverageEventSize(i * 2 + 1) == 0) {
300  continue; // If no events skip the chamber
301  }
302  TString filename = TString::Format("%s/%s/Mean_%d", mPedestalsCCDBBasePath.c_str(), mPedestalTag.c_str(), i);
303  mDbMetadata.emplace("Tag", mPedestalTag.c_str());
304  mDBapi.storeAsTFileAny(aPedestals.At(i), filename.Data(), mDbMetadata, minTimeStamp, maxTimeStamp);
305  }
306  for (int i = 0; i < Geo::N_MODULES; i++) {
307  if (mDeco->getAverageEventSize(i * 2) == 0 && mDeco->getAverageEventSize(i * 2 + 1) == 0) {
308  continue; // If no events skip the chamber
309  }
310  TString filename = TString::Format("%s/%s/Sigma_%d", mPedestalsCCDBBasePath.c_str(), mPedestalTag.c_str(), i);
311  mDbMetadata.emplace("Tag", mPedestalTag.c_str());
312  mDBapi.storeAsTFileAny(aSigmas.At(i), filename.Data(), mDbMetadata, minTimeStamp, maxTimeStamp);
313  }
314  return;
315 }

318 // the decodeTF() function processes the the messages generated by the (sub)TimeFrame builder
319 void PedestalsCalculationTask::decodeTF(framework::ProcessingContext& pc)
320 {
321  LOG(debug) << "*********** In decodeTF **************";
322  // get the input buffer
323  auto& inputs = pc.inputs();
324  DPLRawParser parser(inputs, o2::framework::select("TF:HMP/RAWDATA"));
325  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
326  uint32_t* theBuffer = (uint32_t*)it.raw();
327  mDeco->setUpStream(theBuffer, it.size() + it.offset());
328  try {
329  if (mFastAlgorithm) {
330  mDeco->decodePageFast(&theBuffer);
331  } else {
332  mDeco->decodePage(&theBuffer);
333  }
334  } catch (int e) {
335  // The stream end !
336  LOG(debug) << "End Page decoding !";
337  }
338  mTotalFrames++;
339  mTotalDigits += mDeco->mDigits.size();
340  }
341  return;
342 }
