63  // this is the virtual interface for context objects
64  class ContextObject
65  {
66  public:
67  ContextObject() = delete;
68  ContextObject(FairMQMessagePtr&& headerMsg, FairMQMessagePtr&& payloadMsg, const std::string& bindingChannel)
69  : mParts{}, mChannel{bindingChannel}
70  {
71  mParts.AddPart(std::move(headerMsg));
72  mParts.AddPart(std::move(payloadMsg));
73  }
74  ContextObject(FairMQMessagePtr&& headerMsg, const std::string& bindingChannel)
75  : mParts{}, mChannel{bindingChannel}
76  {
77  mParts.AddPart(std::move(headerMsg));
78  }
79  virtual ~ContextObject() = default;
80 
84  virtual FairMQParts finalize()
85  {
86  FairMQParts parts = std::move(mParts);
87  assert(parts.Size() == 2);
88  auto* header = o2::header::get<o2::header::DataHeader*>(parts.At(0)->GetData());
89  if (header == nullptr) {
90  throw std::logic_error("No valid header message found");
91  } else {
92  // o2::header::get returns const pointer, but here we can change the message
93  const_cast<o2::header::DataHeader*>(header)->payloadSize = parts.At(1)->GetSize();
94  }
95  // return value optimization returns by move
96  return parts;
97  }
98 
100  const std::string& channel() const
101  {
102  return mChannel;
103  }
104 
105  bool empty() const
106  {
107  return mParts.Size() == 0;
108  }
109 
110  o2::header::DataHeader const* header()
111  {
112  // we would expect this function to be const but the FairMQParts API does not allow this
113  if (empty() || mParts.At(0) == nullptr) {
114  return nullptr;
115  }
116  return o2::header::get<o2::header::DataHeader*>(mParts.At(0)->GetData());
117  }
118 
119  o2::framework::DataProcessingHeader const* dataProcessingHeader()
120  {
121  if (empty() || mParts.At(0) == nullptr) {
122  return nullptr;
123  }
124  return o2::header::get<o2::framework::DataProcessingHeader*>(mParts.At(0)->GetData());
125  }
126 
127  o2::header::Stack const* headerStack()
128  {
129  // we would expect this function to be const but the FairMQParts API does not allow this
130  if (empty() || mParts.At(0) == nullptr) {
131  return nullptr;
132  }
133  return o2::header::get<o2::header::DataHeader*>(mParts.At(0)->GetData()) ? reinterpret_cast<o2::header::Stack*>(mParts.At(0)->GetData()) : nullptr;
134  }
135 
136  protected:
137  FairMQParts mParts;
138  std::string const& mChannel;
139  };

241  ContainerRefObject() = delete;
243  template <typename ContextType, typename... Args>
244  ContainerRefObject(ContextType* context, FairMQMessagePtr&& headerMsg, const std::string& bindingChannel, int index, Args&&... args)
245  : ContextObject(std::forward<FairMQMessagePtr>(headerMsg), context->getChannelRef(bindingChannel)),
246  // the transport factory
247  mFactory{context->proxy().getTransport(bindingChannel, index)},
248  // the memory resource takes ownership of the message
249  mResource{mFactory ? AlignedMemoryResource(mFactory->GetMemoryResource()) : AlignedMemoryResource(nullptr)},
250  // create the vector with apropriate underlying memory resource for the message
251  mData{std::forward<Args>(args)..., pmr::polymorphic_allocator<value_type>(&mResource)}
252  {
253  // FIXME: drop this repeated check and make sure at initial setup of devices that everything is fine
254  // introduce error policy
255  if (mFactory == nullptr) {
256  throw runtime_error_f("failed to get transport factory for channel %s", bindingChannel.c_str());
257  }
258  if (mResource.isValid() == false) {
259  throw runtime_error_f("no memory resource for channel %s", bindingChannel.c_str());
260  }
261  }
262  ~ContainerRefObject() override = default;
263

267  {
268  assert(mParts.Size() == 1);
269  auto payloadMsg = o2::pmr::getMessage(std::move(mData));

272  }

481  void schedule(Messages::value_type&& message)
482  {
483  auto const* header = message->header();
484  if (header == nullptr) {
485  throw std::logic_error("No valid header message found");
486  }
487  mScheduledMessages.emplace_back(std::move(message));
488  if (mDispatchControl.dispatch != nullptr) {
489  // send all scheduled messages if there is no trigger callback or its result is true
490  if (mDispatchControl.trigger == nullptr || mDispatchControl.trigger(*header)) {
491  std::unordered_map<std::string const*, FairMQParts> outputs;
492  for (auto& message : mScheduledMessages) {
493  FairMQParts parts = message->finalize();
494  assert(message->empty());
495  assert(parts.Size() == 2);
496  for (auto& part : parts) {
497  outputs[&(message->channel())].AddPart(std::move(part));
498  }
499  }
500  for (auto& [channel, parts] : outputs) {
501  mDispatchControl.dispatch(std::move(parts), *channel, DefaultChannelIndex);
502  }
503  mScheduledMessages.clear();
504  }
505  }
506  }
507
