39 static void BM_TreeToTable(benchmark::State& state)
40 {
41 
42  // initialize a random generator
43  std::default_random_engine e1(1234567891);
44  std::uniform_real_distribution<double> rd(0, 1);
45  std::normal_distribution<float> rf(5., 2.);
46  std::discrete_distribution<ULong64_t> rl({10, 20, 30, 30, 5, 5});
47  std::discrete_distribution<int> ri({10, 20, 30, 30, 5, 5});
48 
49  // create a table and fill the columns with random numbers
50  TableBuilder builder;
51  auto rowWriter =
52  builder.persist<double, float, ULong64_t, int>({"a", "b", "c", "d"});
53  for (auto i = 0; i < state.range(0); ++i) {
54  rowWriter(0, rd(e1), rf(e1), rl(e1), ri(e1));
55  }
56  auto table = builder.finalize();
57 
58  // now convert the table to a tree
59  TFile fout("tree2table.root", "RECREATE");
60  TableToTree ta2tr(table, &fout, "tree2table");
61  ta2tr.addAllBranches();
62  ta2tr.process();
63  fout.Close();
64 
65  // read tree and convert to table again
66  TFile* f = nullptr;
67  TreeToTable* tr2ta = nullptr;
68  for (auto _ : state) {
69 
70  // Open file and create tree
71  f = new TFile("tree2table.root", "READ");
72  auto tr = (TTree*)f->Get("tree2table");
73 
74  // benchmark TreeToTable
75  if (tr) {
76  tr2ta = new TreeToTable;
77  tr2ta->addAllColumns(tr);
78  tr2ta->fill(tr);
79  auto ta = tr2ta->finalize();
80  } else {
81  LOG(info) << "tree is empty!";
82  }
83 
84  // clean up
85  delete tr2ta;
86 
87  f->Close();
88  delete f;
89  }
90 
91  state.SetBytesProcessed(state.iterations() * state.range(0) * 24);
92 }
