56  c = -1;
57  }
58  for (size_t iCluster{0}; iCluster < numClusters; ++iCluster) {
59  int extIndex = track.getClusterIndex(iCluster);
60  if (extIndex == -1) {
61  continue;
62  }
63  o2::MCCompLabel mcLabel = *(event.getClusterLabels(iCluster, extIndex).begin());
64  bool found = false;
65 
66  for (size_t iOcc{0}; iOcc < occurrences.size(); ++iOcc) {
67  std::pair<o2::MCCompLabel, int>& occurrence = occurrences[iOcc];
68  if (mcLabel == occurrence.first) {
69  ++occurrence.second;
70  found = true;
71  break;
72  }
73  }
74  if (!found) {
75  occurrences.emplace_back(mcLabel, 1);
76  }
77  }
78 
79  if (occurrences.size() > 1) {
80  isFake = true;
81  }
82  std::sort(std::begin(occurrences), std::end(occurrences), [](auto e1, auto e2) {
83  return e1.second > e2.second;
84  });
85  mainLabel = occurrences[0].first;
86 
87  for (size_t iOcc{1}; iOcc < occurrences.size(); ++iOcc) {
88  if (occurrences[iOcc].second == occurrences[0].second) {
89  isAmbiguousId = true;
90  break;
91  }
92  }
93 
94  for (size_t iCluster{0}; iCluster < numClusters; ++iCluster) {
95  int extIndex = track.getClusterIndex(iCluster);
96  if (extIndex == -1) {
97  continue;
98  }
99  o2::MCCompLabel lbl = *(event.getClusterLabels(iCluster, extIndex).begin());
100  if (lbl == mainLabel && occurrences[0].second > 1 && !lbl.isNoise()) { // if we have MaxClusters fake clusters -> occurrences[0].second = 1
101  clusStatuses[iCluster] = 1;
102  } else {
103  clusStatuses[iCluster] = 0;
104  ++nFakeClusters;
105  }
106  }
107  if (storeClusters) {
108  for (auto iCluster{0}; iCluster < numClusters; ++iCluster) {
109  const int index = track.getClusterIndex(iCluster);
110  if (index != constants::its::UnusedIndex) {
111  clusters[iCluster] = pvc->getClusters()[iCluster][index];
