87 void Detector::ConstructGeometry()

89  LOG(debug) << "Creating FT0 geometry\n";

91 
92  TGeoVolumeAssembly* stlinA = new TGeoVolumeAssembly("FT0A"); // A side mother
93  TGeoVolumeAssembly* stlinC = new TGeoVolumeAssembly("FT0C"); // C side mother

95  Geometry geometry;
96  Float_t zdetA = geometry.ZdetA;
97  Float_t zdetC = geometry.ZdetC;
98  int nCellsA = geometry.NCellsA;
99  int nCellsC = geometry.NCellsC;
100 
101  for (int ipos = 0; ipos < nCellsA; ipos++) {
102  mPosModuleAx[ipos] = geometry.centerMCP(ipos).X();
103  mPosModuleAy[ipos] = geometry.centerMCP(ipos).Y();
104  }
105 
106  // FIT interior
107  TVirtualMC::GetMC()->Gsvolu("0MOD", "BOX", getMediumID(kAir), mInStart, 3);
108  TGeoVolume* ins = gGeoManager->GetVolume("0MOD");
109  //
110  TGeoTranslation* tr[nCellsA + nCellsC];
111  TString nameTr;
112  // A side Translations
113  for (Int_t itr = 0; itr < Geometry::NCellsA; itr++) {
114  nameTr = Form("0TR%i", itr + 1);
115  float z = -mStartA[2] + mInStart[2];
116  tr[itr] = new TGeoTranslation(nameTr.Data(), mPosModuleAx[itr], mPosModuleAy[itr], z);
117  tr[itr]->RegisterYourself();
118  stlinA->AddNode(ins, itr, tr[itr]);
119  LOG(debug) << " A geom " << itr << " " << mPosModuleAx[itr] << " " << mPosModuleAy[itr];
120  }
121  SetCablesA(stlinA);
122  //Add FT0-A support Structure to the geometry
123  stlinA->AddNode(constructFrameGeometry(), 1, new TGeoTranslation(0, 0, -mStartA[2] + mInStart[2]));
124 
125  // C Side
126  TGeoRotation* rot[nCellsC];
127  TString nameRot;
128  TGeoCombiTrans* com[nCellsC];
129  TGeoCombiTrans* comCable[nCellsC];
130  TString nameCom;
131 
132  for (Int_t itr = Geometry::NCellsA; itr < Geometry::NCellsA + nCellsC; itr++) {
133  nameTr = Form("0TR%i", itr + 1);
134  nameRot = Form("0Rot%i", itr + 1);

136  float ac1 = geometry.tiltMCP(ic).X();
137  float bc1 = geometry.tiltMCP(ic).Y();
138  float gc1 = geometry.tiltMCP(ic).Z();

140  LOG(debug) << " rot geom " << ic << " " << ac1 << " " << bc1 << " " << gc1;
141  rot[ic]->RegisterYourself();
142  mPosModuleCx[ic] = geometry.centerMCP(ic + nCellsA).X();
143  mPosModuleCy[ic] = geometry.centerMCP(ic + nCellsA).Y();
144  mPosModuleCz[ic] = geometry.centerMCP(ic + nCellsA).Z() - 80; // !!! fix later
145  com[ic] = new TGeoCombiTrans(mPosModuleCx[ic], mPosModuleCy[ic], mPosModuleCz[ic], rot[ic]);
146  TGeoHMatrix hm = *com[ic];
147  TGeoHMatrix* ph = new TGeoHMatrix(hm);
148  stlinC->AddNode(ins, itr, ph);
149  //cables
150  TGeoVolume* cables = SetCablesSize(itr);
151  LOG(debug) << " C " << mPosModuleCx[ic] << " " << mPosModuleCy[ic];
152  // cables->Print();
153  comCable[ic] = new TGeoCombiTrans(mPosModuleCx[ic], mPosModuleCy[ic], mPosModuleCz[ic] + mInStart[2] + 0.2, rot[ic]);
154  TGeoHMatrix hmCable = *comCable[ic];
155  TGeoHMatrix* phCable = new TGeoHMatrix(hmCable);
156  stlinC->AddNode(cables, itr, comCable[ic]);
157  }
158 
159  TGeoVolume* alice = gGeoManager->GetVolume("barrel");
160  alice->AddNode(stlinA, 1, new TGeoTranslation(0, 30., zdetA));
161  TGeoRotation* rotC = new TGeoRotation("rotC", 90., 0., 90., 90., 180., 0.);
162  alice->AddNode(stlinC, 1, new TGeoCombiTrans(0., 30., -zdetC, rotC));
163 
164  // MCP + 4 x wrapped radiator + 4xphotocathod + MCP + Al top in front of radiators
165  SetOneMCP(ins);
166  //SetCablesC(stlinC);

178 void Detector::SetOneMCP(TGeoVolume* ins)
179 {
180 
181  Double_t x, y, z;
182 
183  Float_t ptop[3] = {1.324, 1.324, 1.}; // Cherenkov radiator
184  Float_t ptopref[3] = {1.3241, 1.3241, 1.}; // Cherenkov radiator wrapped with reflector
185  Double_t prfv[3] = {0.0002, 1.323, 1.}; // Vertical refracting layer bettwen radiators and between radiator and not optical Air
186  Double_t prfh[3] = {1.323, 0.0002, 1.}; // Horizontal refracting layer bettwen radiators and ...
187  Float_t pmcp[3] = {2.949, 2.949, 0.66}; // MCP
188  Float_t pmcpinner[3] = {2.749, 2.749, 0.1};
189  Float_t pmcpbase[3] = {2.949, 2.949, 0.675};
190  Float_t pmcpside[3] = {0.15, 2.949, 0.65};
191  Float_t pmcptopglass[3] = {2.949, 2.949, 0.1}; // MCP top glass optical
192  Float_t preg[3] = {1.324, 1.324, 0.005}; // Photcathode
193  // Entry window (glass)
194  TVirtualMC::GetMC()->Gsvolu("0TOP", "BOX", getMediumID(kOpGlass), ptop, 3); // Glass radiator
195  TGeoVolume* top = gGeoManager->GetVolume("0TOP");
196  TVirtualMC::GetMC()->Gsvolu("0TRE", "BOX", getMediumID(kAir), ptopref, 3); // Air: wrapped radiator
197  TGeoVolume* topref = gGeoManager->GetVolume("0TRE");
198  TVirtualMC::GetMC()->Gsvolu("0RFV", "BOX", getMediumID(kOptAl), prfv, 3); // Optical Air vertical
199  TGeoVolume* rfv = gGeoManager->GetVolume("0RFV");
200  TVirtualMC::GetMC()->Gsvolu("0RFH", "BOX", getMediumID(kOptAl), prfh, 3); // Optical Air horizontal
201  TGeoVolume* rfh = gGeoManager->GetVolume("0RFH");
202 
203  TVirtualMC::GetMC()->Gsvolu("0REG", "BOX", getMediumID(kOpGlassCathode), preg, 3);
204  TGeoVolume* cat = gGeoManager->GetVolume("0REG");
205 
206  //wrapped radiator + reflecting layers
207 
208  Int_t ntops = 0, nrfvs = 0, nrfhs = 0;
209  x = y = z = 0;
210  topref->AddNode(top, 1, new TGeoTranslation(0, 0, 0));
211  float xinv = -ptop[0] - prfv[0];
212  topref->AddNode(rfv, 1, new TGeoTranslation(xinv, 0, 0));
213  xinv = ptop[0] + prfv[0];
214  topref->AddNode(rfv, 2, new TGeoTranslation(xinv, 0, 0));
215  float yinv = -ptop[1] - prfh[1];
216  topref->AddNode(rfh, 1, new TGeoTranslation(0, yinv, 0));
217  yinv = ptop[1] + prfh[1];
218  topref->AddNode(rfh, 2, new TGeoTranslation(0, yinv, 0));
219 
220  //container for radiator, cathode
221  for (Int_t ix = 0; ix < 2; ix++) {
222  float xin = -mInStart[0] + 0.3 + (ix + 0.5) * 2 * ptopref[0];
223  for (Int_t iy = 0; iy < 2; iy++) {
224  float yin = -mInStart[1] + 0.3 + (iy + 0.5) * 2 * ptopref[1];
225  ntops++;
226  z = -mInStart[2] + ptopref[2];
227  ins->AddNode(topref, ntops, new TGeoTranslation(xin, yin, z));
228  LOG(debug) << " n " << ntops << " x " << xin << " y " << yin << " z radiator " << z;
229  z += ptopref[2] + 2. * pmcptopglass[2] + preg[2];
230  ins->AddNode(cat, ntops, new TGeoTranslation(xin, yin, z));
231  LOG(debug) << " n " << ntops << " x " << xin << " y " << yin << " z cathod " << z;
232  }
233  }
234  // MCP
235  TVirtualMC::GetMC()->Gsvolu("0MTO", "BOX", getMediumID(kOpGlass), pmcptopglass, 3); //Op Glass
236  TGeoVolume* mcptop = gGeoManager->GetVolume("0MTO");
237  z = -mInStart[2] + 2 * ptopref[2] + pmcptopglass[2];
238  ins->AddNode(mcptop, 1, new TGeoTranslation(0, 0, z));
239 
240  TVirtualMC::GetMC()->Gsvolu("0MCP", "BOX", getMediumID(kAir), pmcp, 3); //glass
241  TGeoVolume* mcp = gGeoManager->GetVolume("0MCP");
242  z = -mInStart[2] + 2 * ptopref[2] + 2 * pmcptopglass[2] + 2 * preg[2] + pmcp[2];
243  ins->AddNode(mcp, 1, new TGeoTranslation(0, 0, z));
244 
245  TVirtualMC::GetMC()->Gsvolu("0MSI", "BOX", getMediumID(kMCPwalls), pmcpside, 3); //glass
246  TGeoVolume* mcpside = gGeoManager->GetVolume("0MSI");
247  x = -pmcp[0] + pmcpside[0];
248  y = -pmcp[1] + pmcpside[1];
249  mcp->AddNode(mcpside, 1, new TGeoTranslation(x, y, 0));
250  x = pmcp[0] - pmcpside[0];
251  y = pmcp[1] - pmcpside[1];
252  mcp->AddNode(mcpside, 2, new TGeoTranslation(x, y, 0));
253  x = -pmcp[1] + pmcpside[1];
254  y = -pmcp[0] + pmcpside[0];
255  mcp->AddNode(mcpside, 3, new TGeoCombiTrans(x, y, 0, new TGeoRotation("R2", 90, 0, 0)));
256  x = pmcp[1] - pmcpside[1];
257  y = pmcp[0] - pmcpside[0];
258  mcp->AddNode(mcpside, 4, new TGeoCombiTrans(x, y, 0, new TGeoRotation("R2", 90, 0, 0)));
259 
260  TVirtualMC::GetMC()->Gsvolu("0MBA", "BOX", getMediumID(kCeramic), pmcpbase, 3); //glass
261  TGeoVolume* mcpbase = gGeoManager->GetVolume("0MBA");
262  z = -mInStart[2] + 2 * ptopref[2] + pmcptopglass[2] + 2 * pmcp[2] + pmcpbase[2];
263  ins->AddNode(mcpbase, 1, new TGeoTranslation(0, 0, z));
264 }

384 // Class wrapper for construction of FT0-A support structure
385 // The frame is constructed by defining two aluminum boxes that are placed in an L-shape,
386 // with material sequentially removed to re-create the CAD drawings,
387 // including sockets defined by the parameters of the sensitive elements that they are placed into
388 // Two L-shaped elements form the full support structure, with one reflected about the axes of symmetry
389 // First written by Joe Crowley and revised by Jason Pruitt from Cal Poly in 2019-2021
390 TGeoVolume* Detector::constructFrameGeometry()
391 {
392  // define the media
393  TGeoMedium* Vacuum = gGeoManager->GetMedium("FT0_Vacuum$");
394  TGeoMedium* Al = gGeoManager->GetMedium("FT0_Aluminium$");
395 
396  // make a volume assembly for the frame
397  TGeoVolumeAssembly* FT0_Frame = new TGeoVolumeAssembly("FT0_Frame");
398 
399  // define translations for the quartz radiator and PMT sockets
400  defineTransformations();
401 
402  // frame1 and frame2 are rectangles that approximate the outline of one L
403  // shape of the frame
404  TGeoBBox* frame1 = new TGeoBBox("frame1", sFrame1X / 2, sFrame1Y / 2, sFrameZ / 2);
405  TGeoBBox* frame2 = new TGeoBBox("frame2", sFrame2X / 2, sFrame2Y / 2, sFrameZ / 2);
406 
407  // the following elements are subtracted from frame1 and frame2 to better
408  // approximate the CAD shape
409  TGeoBBox* rect1 = new TGeoBBox("rect1", sRect1X / 2, sRect1Y / 2, sFrameZ / 2);
410  TGeoBBox* rect2 = new TGeoBBox("rect2", sRect2X / 2, sRect2Y / 2 + sEps, sFrameZ / 2 - sMountZ / 2);
411  TGeoBBox* rect3 = new TGeoBBox("rect3", sRect3X / 2, sRect3Y / 2, sFrameZ / 2);
412  TGeoBBox* rect4 = new TGeoBBox("rect4", sRect4X / 2, sRect4Y / 2, sFrameZ / 2);
413  TGeoBBox* rect5 = new TGeoBBox("rect5", sRect5X / 2 + sEps, sRect5Y / 2 + sEps, sFrameZ / 2 + sEps);
414  TGeoBBox* rect6 = new TGeoBBox("rect6", sRect6X / 2 + sEps, sRect6Y / 2 + sEps, sFrameZ / 2 + sEps);
415  TGeoBBox* rect7 = new TGeoBBox("rect7", sRect7X / 2 + sEps, sRect7Y / 2 + sEps, sFrameZ / 2 - sMountZ / 2 + sEps);
416  TGeoBBox* rect8 = new TGeoBBox("rect8", sRect8X / 2 + sEps, sRect8Y / 2 + sEps, sFrameZ / 2 + sEps);
417 
418  // Define a value to overcut the coincidence between the closure of the tube
419  // and the edge of the rectangle to eliminate artifacts
420  Double_t flopsErr = .00001;
421  // PMT and quartz radiator shapes provide the dimensions of the sockets to be subtracted
422  // from the frame that will make room for a sensitive element to fit
423  TGeoBBox* quartzRadiator = new TGeoBBox("quartzRadiator", sQuartzRadiatorSide / 2, sQuartzRadiatorSide / 2, sQuartzRadiatorZ / 2);
424  TGeoBBox* pmtBox = new TGeoBBox("pmtBox", sPmtSide / 2 + sEps, sPmtSide / 2 + sEps, sPmtZ / 2 + sEps);
425 
426  // these two shapes create a subtraction so that the corners of the holes that
427  // seat the sens elements are rounded
428  TGeoBBox* pmtCornerRect = new TGeoBBox("pmtCornerRect", sCornerRadius / 2 - flopsErr, sCornerRadius / 2 - flopsErr, sPmtZ / 2);
429  TGeoTube* pmtCornerTube = new TGeoTube("pmtCornerTube", 0, sCornerRadius, sPmtZ / 2 + sEps);
