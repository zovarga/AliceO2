92 static void patchPage(gsl::span<const std::byte> rdhBuffer, bool verbose)
93 {
94  static int sNrdhs = 0;
95  auto& rdhAny = *reinterpret_cast<RDH*>(const_cast<std::byte*>(&(rdhBuffer[0])));
96  sNrdhs++;
97 
98  auto existingFeeId = o2::raw::RDHUtils::getFEEID(rdhAny);
99  if (existingFeeId == 0) {
100  // early versions of raw data did not set the feeId
101  // which we need to select the right decoder
102  auto cruId = o2::raw::RDHUtils::getCRUID(rdhAny) & 0xFF;
103  auto flags = o2::raw::RDHUtils::getCRUID(rdhAny) & 0xFF00;
104  auto endpoint = o2::raw::RDHUtils::getEndPointID(rdhAny);
105  uint32_t feeId = cruId * 2 + endpoint + flags;
106  o2::raw::RDHUtils::setFEEID(rdhAny, feeId);
107  }
108 
109  if (verbose) {
110  std::cout << "RDH number " << sNrdhs << "--\n";
111  o2::raw::RDHUtils::printRDH(rdhAny);
112  }
113 };

172  void decodePage(gsl::span<const std::byte> page)
173  {
174  static int Nrdhs = 0;
175  size_t ndigits{0};
176 
177  uint32_t orbit;
178 
179  auto tStart = std::chrono::high_resolution_clock::now();
180 
181  auto channelHandler = [&](DsElecId dsElecId, uint8_t channel, o2::mch::raw::SampaCluster sc) {
182  auto solarId = dsElecId.solarId();
183  auto dsId = dsElecId.elinkId();
184  if (mDebug) {
185  std::cout << "New digit: SOLAR " << (int)solarId << " DS " << (int)dsId << " CH " << (int)channel << std::endl;
186  }
187 
188  mDigits.emplace_back(o2::mch::calibration::PedestalDigit(solarId, dsId, channel, sc.bunchCrossing, 0, sc.samples));
189  ++ndigits;
190  };
191 
192  auto errorHandler = [&](DsElecId dsElecId, int8_t chip, uint32_t error) {
193  auto solarId = dsElecId.solarId();
194  auto dsId = dsElecId.elinkId();
195 
196  mErrors.emplace_back(o2::mch::DecoderError(solarId, dsId, chip, error));
197  };
198 
199  patchPage(page, mDebug);
200 
201  if (mDebug) {
202  auto& rdhAny = *reinterpret_cast<RDH*>(const_cast<std::byte*>(&(page[0])));
203  Nrdhs += 1;
204  std::cout << Nrdhs << "--\n";
205  o2::raw::RDHUtils::printRDH(rdhAny);
206  }
207 
208  if (!mDecoder) {
209  DecodedDataHandlers handlers;
210  handlers.sampaChannelHandler = channelHandler;
211  handlers.sampaErrorHandler = errorHandler;
212  mDecoder = mFee2Solar ? o2::mch::raw::createPageDecoder(page, handlers, mFee2Solar)
213  : o2::mch::raw::createPageDecoder(page, handlers);
214  }
215  try {
216  mDecoder(page);
217  } catch (std::exception& e) {
218  std::cout << e.what() << '\n';
219  }
220  }

222  //_________________________________________________________________________________________________
223  void decodeBuffer(gsl::span<const std::byte> buf)
224  {
225  if (mDebug) {
226  std::cout << "\n\n============================\nStart of new buffer\n";
227  }
228  size_t bufSize = buf.size();
229  size_t pageStart = 0;
230  while (bufSize > pageStart) {
231  RDH* rdh = reinterpret_cast<RDH*>(const_cast<std::byte*>(&(buf[pageStart])));
232  auto rdhVersion = o2::raw::RDHUtils::getVersion(rdh);
233  auto rdhHeaderSize = o2::raw::RDHUtils::getHeaderSize(rdh);
234  if (rdhHeaderSize != 64) {
235  break;
236  }
237  auto pageSize = o2::raw::RDHUtils::getOffsetToNext(rdh);
238 
239  gsl::span<const std::byte> page(reinterpret_cast<const std::byte*>(rdh), pageSize);
240  decodePage(page);
241 
242  pageStart += pageSize;
243  }
244  }

247  // the decodeTF() function processes the messages generated by the (sub)TimeFrame builder
248  void decodeTF(framework::ProcessingContext& pc)
249  {
250  // get the input buffer
251  auto& inputs = pc.inputs();
252  std::vector<InputSpec> filter = {{"check", ConcreteDataTypeMatcher{o2::header::gDataOriginMCH, "RAWDATA"}, Lifetime::Timeframe}};
253  DPLRawParser parser(inputs, filter);
254 
255  auto tStart = std::chrono::high_resolution_clock::now();
256  size_t totPayloadSize = 0;
257  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
258  auto const* raw = it.raw();
259  if (!raw) {
260  continue;
261  }
262  size_t payloadSize = it.size();
263  totPayloadSize += payloadSize;
264 
265  gsl::span<const std::byte> buffer(reinterpret_cast<const std::byte*>(raw), sizeof(RDH) + payloadSize);
266  decodeBuffer(buffer);
267  }
268  auto tEnd = std::chrono::high_resolution_clock::now();
269 
270  if (totPayloadSize > 0) {
271  std::chrono::duration<double, std::milli> elapsed = tEnd - tStart;
272  mTimeDecoder += elapsed;
273  if (!mTimeDecoderMin || (elapsed < mTimeDecoderMin)) {
274  mTimeDecoderMin = elapsed;
275  }
276  if (!mTimeDecoderMax || (elapsed > mTimeDecoderMax)) {
277  mTimeDecoderMax = elapsed;
278  }
279  mTFcount += 1;
280  }
281  }

297  void logStats()
298  {
299  static auto loggerStart = std::chrono::high_resolution_clock::now();
300  static auto loggerEnd = loggerStart;
301  static uint64_t nDigits = 0;
302  static uint64_t nTF = 0;
303 
304  if (mLoggingInterval == 0) {
305  return;
306  }
307 
308  nDigits += mDigits.size();
309  nTF += 1;
310 
311  loggerEnd = std::chrono::high_resolution_clock::now();
312  std::chrono::duration<double, std::milli> loggerElapsed = loggerEnd - loggerStart;
313  if (loggerElapsed.count() > mLoggingInterval) {
314  LOG(info) << "Processed " << nDigits << " digits in " << nTF << " time frames";
315  nDigits = 0;
316  nTF = 0;
317  loggerStart = std::chrono::high_resolution_clock::now();
318  }

321  //_________________________________________________________________________________________________
322  void run(framework::ProcessingContext& pc)
323  {
324  auto createBuffer = [&](auto& vec, size_t& size) {
325  size = vec.empty() ? 0 : sizeof(*(vec.begin())) * vec.size();
326  char* buf = nullptr;
327  if (size > 0) {
328  buf = (char*)malloc(size);
329  if (buf) {
330  char* p = buf;
331  size_t sizeofElement = sizeof(*(vec.begin()));
332  for (auto& element : vec) {
333  memcpy(p, &element, sizeofElement);
334  p += sizeofElement;
335  }
336  }
337  }
338  return buf;
339  };
340 
341  reset();
342  for (auto&& input : pc.inputs()) {
343  if (input.spec->binding == "TF") {
344  decodeTF(pc);
345  }
346  if (input.spec->binding == "readout") {
347  decodeReadout(input);
348  }
349  }
350 
351  size_t digitsSize;
352  char* digitsBuffer = createBuffer(mDigits, digitsSize);
353 
354  size_t errorsSize;
355  char* errorsBuffer = createBuffer(mErrors, errorsSize);
356 
357  // create the output message
358  auto freefct = [](void* data, void*) { free(data); };
359  pc.outputs().adoptChunk(Output{header::gDataOriginMCH, "PDIGITS", 0}, digitsBuffer, digitsSize, freefct, nullptr);
360  pc.outputs().adoptChunk(Output{header::gDataOriginMCH, "ERRORS", 0}, errorsBuffer, errorsSize, freefct, nullptr);
361 
362  logStats();
363  }
