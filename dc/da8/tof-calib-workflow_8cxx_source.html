36 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
37 {
38  WorkflowSpec specs;
39  auto useCCDB = configcontext.options().get<bool>("use-ccdb");
40  auto doLHCcalib = configcontext.options().get<bool>("do-lhc-phase");
41  auto doChannelOffsetCalib = configcontext.options().get<bool>("do-channel-offset");
42  auto attachChannelOffsetToLHCphase = configcontext.options().get<bool>("attach-channel-offset-to-lhcphase");
43  auto isCosmics = configcontext.options().get<bool>("cosmics");
44  auto perstrip = configcontext.options().get<bool>("perstrip");
45  auto safe = configcontext.options().get<bool>("safe-mode");
46 
47  if (isCosmics) {
48  LOG(info) << "Cosmics set!!!! No LHC phase, Yes channel offset";
49  doChannelOffsetCalib = true;
50  doLHCcalib = false;
51  }
52 
53  if (!doLHCcalib && attachChannelOffsetToLHCphase) {
54  LOG(info) << "Over-writing attachChannelOffsetToLHCphase because we are not doing the LHCphase calibration";
55  attachChannelOffsetToLHCphase = false;
56  }
57 
58  LOG(info) << "TOF Calibration workflow: options";
59  LOG(info) << "doLHCcalib = " << doLHCcalib;
60  LOG(info) << "doChannelOffsetCalib = " << doChannelOffsetCalib;
61  LOG(info) << "useCCDB = " << useCCDB;
62  LOG(info) << "attachChannelOffsetToLHCphase = " << attachChannelOffsetToLHCphase;
63  if (doLHCcalib) {
64  specs.emplace_back(getLHCClockCalibDeviceSpec());
65  }
66  if (doChannelOffsetCalib) {
67  if (!isCosmics) {
68  specs.emplace_back(getTOFChannelCalibDeviceSpec<o2::dataformats::CalibInfoTOF>(useCCDB, attachChannelOffsetToLHCphase, isCosmics, perstrip, safe));
69  } else {
70  specs.emplace_back(getTOFChannelCalibDeviceSpec<o2::tof::CalibInfoCluster>(useCCDB, attachChannelOffsetToLHCphase, isCosmics));
71  }
72  }
73  return specs;
74 }
