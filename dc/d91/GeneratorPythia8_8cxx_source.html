87  if (!hooks) {
88  LOG(fatal) << "Failed to init \'Pythia8\': problem with user hooks configuration ";
89  return false;
90  }
91  setUserHooks(hooks);
92  }
93 
94 #if PYTHIA_VERSION_INTEGER < 8300
95 
103  mPythia.readString("HadronLevel:Decay off");
104 #endif
105 
107  if (!mPythia.init()) {
108  LOG(fatal) << "Failed to init \'Pythia8\': init returned with error";
109  return false;
110  }
111

163 Bool_t
164  GeneratorPythia8::importParticles(Pythia8::Event& event)
165 {
168  /* loop over particles */
169  // auto weight = mPythia.info.weight(); // TBD: use weights
170  auto nParticles = event.size();
171  for (Int_t iparticle = 1; iparticle < nParticles; iparticle++) { // first particle is system
172  auto particle = event[iparticle];
173  auto pdg = particle.id();
174  auto st = particle.statusHepMC();
175  auto px = particle.px();
176  auto py = particle.py();
177  auto pz = particle.pz();
178  auto et = particle.e();
179  auto vx = particle.xProd();
180  auto vy = particle.yProd();
181  auto vz = particle.zProd();
182  auto vt = particle.tProd();
183  auto m1 = particle.mother1() - 1;
184  auto m2 = particle.mother2() - 1;
185  auto d1 = particle.daughter1() - 1;
186  auto d2 = particle.daughter2() - 1;
187  mParticles.push_back(TParticle(pdg, st, m1, m2, d1, d2, px, py, pz, et, vx, vy, vz, vt));
188  }
189 
191  return kTRUE;
192 }

238 void GeneratorPythia8::selectFromAncestor(int ancestor, Pythia8::Event& inputEvent, Pythia8::Event& outputEvent)
239 {
240 
245  // recursive selection via lambda function
246  std::set<int> selected;
247  std::function<void(int)> select;
248  select = [&](int i) {
249  selected.insert(i);
250  auto dl = inputEvent[i].daughterList();
251  for (auto j : dl) {
252  select(j);
253  }
254  };
255  select(ancestor);
256 
257  // map selected particle index to output index
258  std::map<int, int> indexMap;
259  int index = outputEvent.size();
260  for (auto i : selected) {
261  indexMap[i] = index++;
262  }
263 
264  // adjust mother/daughter indices and append to output event
265  for (auto i : selected) {
266  auto p = mPythia.event[i];
267  auto m1 = indexMap[p.mother1()];
268  auto m2 = indexMap[p.mother2()];
269  auto d1 = indexMap[p.daughter1()];
270  auto d2 = indexMap[p.daughter2()];
271  p.mothers(m1, m2);
272  p.daughters(d1, d2);
273 
274  outputEvent.append(p);
275  }
276 }

280 void GeneratorPythia8::getNcoll(const Pythia8::Info& info, int& nColl)
281 {
282 
285 #if PYTHIA_VERSION_INTEGER < 8300
286  auto hiinfo = info.hiinfo;
287 #else
288  auto hiinfo = info.hiInfo;
289 #endif
290 
291  nColl = 0;
292 
293  if (!hiinfo) {
294  return;
295  }
296 
297  // loop over sub-collisions
298  auto scptr = hiinfo->subCollisionsPtr();
299  for (auto sc : *scptr) {
300 
301  // wounded nucleon flag in projectile/target
302  auto pW = sc.proj->status() == Pythia8::Nucleon::ABS; // according to C.Bierlich this should be == Nucleon::ABS
303  auto tW = sc.targ->status() == Pythia8::Nucleon::ABS;
304 
305  // increase number of collisions if both are wounded
306  if (pW && tW) {
307  nColl++;
308  }
309  }
310 }

326 void GeneratorPythia8::getNpart(const Pythia8::Info& info, int& nProtonProj, int& nNeutronProj, int& nProtonTarg, int& nNeutronTarg)
327 {
328 
331 #if PYTHIA_VERSION_INTEGER < 8300
332  auto hiinfo = info.hiinfo;
333 #else
334  auto hiinfo = info.hiInfo;
335 #endif
336 
337  nProtonProj = nNeutronProj = nProtonTarg = nNeutronTarg = 0;
338  if (!hiinfo) {
339  return;
340  }
341 
342  // keep track of wounded nucleons
343  std::vector<Pythia8::Nucleon*> projW;
344  std::vector<Pythia8::Nucleon*> targW;
345 
346  // loop over sub-collisions
347  auto scptr = hiinfo->subCollisionsPtr();
348  for (auto sc : *scptr) {
349 
350  // wounded nucleon flag in projectile/target
351  auto pW = sc.proj->status() == Pythia8::Nucleon::ABS || sc.proj->status() == Pythia8::Nucleon::DIFF; // according to C.Bierlich this should be == Nucleon::ABS || Nucleon::DIFF
352  auto tW = sc.targ->status() == Pythia8::Nucleon::ABS || sc.targ->status() == Pythia8::Nucleon::DIFF;
353 
354  // increase number of wounded projectile nucleons if not yet in the wounded vector
355  if (pW && std::find(projW.begin(), projW.end(), sc.proj) == projW.end()) {
356  projW.push_back(sc.proj);
357  if (sc.proj->id() == 2212) {
358  nProtonProj++;
359  } else if (sc.proj->id() == 2112) {
360  nNeutronProj++;
361  }
362  }
363 
364  // increase number of wounded target nucleons if not yet in the wounded vector
365  if (tW && std::find(targW.begin(), targW.end(), sc.targ) == targW.end()) {
366  targW.push_back(sc.targ);
367  if (sc.targ->id() == 2212) {
368  nProtonTarg++;
369  } else if (sc.targ->id() == 2112) {
370  nNeutronTarg++;
371  }
372  }
373  }
374 }

378 void GeneratorPythia8::getNremn(const Pythia8::Event& event, int& nProtonProj, int& nNeutronProj, int& nProtonTarg, int& nNeutronTarg)
379 {
380 
383  // reset
384  nProtonProj = nNeutronProj = nProtonTarg = nNeutronTarg = 0;
385  auto nNucRem = 0;
386 
387  // particle loop
388  auto nparticles = event.size();
389  for (int ipa = 0; ipa < nparticles; ++ipa) {
390  const auto particle = event[ipa];
391  auto pdg = particle.id();
392 
393  // nuclear remnants have pdg code = ±10LZZZAAA9
394  if (pdg < 1000000000) {
395  continue; // must be nucleus
396  }
397  if (pdg % 10 != 9) {
398  continue; // first digit must be 9
399  }
400  nNucRem++;
401 
402  // extract A, Z and L from pdg code
403  pdg /= 10;
404  auto A = pdg % 1000;
405  pdg /= 1000;
406  auto Z = pdg % 1000;
407  pdg /= 1000;
408  auto L = pdg % 10;
409 
410  if (particle.pz() > 0.) {
411  nProtonProj = Z;
412  nNeutronProj = A - Z;
413  }
414  if (particle.pz() < 0.) {
415  nProtonTarg = Z;
416  nNeutronTarg = A - Z;
417  }
418 
419  } // end of particle loop
420 
421  if (nNucRem > 2) {
422  LOG(warning) << " GeneratorPythia8: found more than two nuclear remnants (weird)";
423  }
424 }
