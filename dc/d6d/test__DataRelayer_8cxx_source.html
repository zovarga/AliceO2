36 // A simple test where an input is provided
37 // and the subsequent InputRecord is immediately requested.
38 BOOST_AUTO_TEST_CASE(TestNoWait)
39 {
40  Monitoring metrics;
41  InputSpec spec{"clusters", "TPC", "CLUSTERS"};
42 
43  std::vector<InputRoute> inputs = {
44  InputRoute{spec, 0, "Fake", 0}};
45 
46  std::vector<ForwardRoute> forwards;
47  TimesliceIndex index{1};
48 
49  auto policy = CompletionPolicyHelpers::consumeWhenAny();
50  DataRelayer relayer(policy, inputs, metrics, index);
51  relayer.setPipelineLength(4);
52 
53  // Let's create a dummy O2 Message with two headers in the stack:
54  // - DataHeader matching the one provided in the input
55  DataHeader dh;
56  dh.dataDescription = "CLUSTERS";
57  dh.dataOrigin = "TPC";
58  dh.subSpecification = 0;
59  dh.splitPayloadIndex = 0;
60  dh.splitPayloadParts = 1;
61 
62  DataProcessingHeader dph{0, 1};
63  auto transport = FairMQTransportFactory::CreateTransportFactory("zeromq");
64  std::array<FairMQMessagePtr, 2> messages;
65  auto channelAlloc = o2::pmr::getTransportAllocator(transport.get());
66  messages[0] = o2::pmr::getMessage(Stack{channelAlloc, dh, dph});
67  messages[1] = transport->CreateMessage(1000);
68  FairMQMessagePtr& header = messages[0];
69  FairMQMessagePtr& payload = messages[1];
70  relayer.relay(header->GetData(), messages.data(), messages.size());
71  std::vector<RecordAction> ready;
72  relayer.getReadyToProcess(ready);
73  BOOST_REQUIRE_EQUAL(ready.size(), 1);
74  BOOST_CHECK_EQUAL(ready[0].slot.index, 0);
75  BOOST_CHECK_EQUAL(ready[0].op, CompletionPolicy::CompletionOp::Consume);
76  BOOST_CHECK_EQUAL(header.get(), nullptr);
77  BOOST_CHECK_EQUAL(payload.get(), nullptr);
78  auto result = relayer.consumeAllInputsForTimeslice(ready[0].slot);
79  // one MessageSet with one PartRef with header and payload
80  BOOST_REQUIRE_EQUAL(result.size(), 1);
81  BOOST_REQUIRE_EQUAL(result.at(0).size(), 1);
82 }

301 // This tests a simple cache pruning, where a single input is shifted out of
302 // the cache.
303 BOOST_AUTO_TEST_CASE(TestCache)
304 {
305  Monitoring metrics;
306  InputSpec spec{"clusters", "TPC", "CLUSTERS"};
307 
308  std::vector<InputRoute> inputs = {
309  InputRoute{spec, 0, "Fake", 0}};
310  std::vector<ForwardRoute> forwards;
311 
312  auto policy = CompletionPolicyHelpers::consumeWhenAll();
313  TimesliceIndex index{1};
314  DataRelayer relayer(policy, inputs, metrics, index);
315  // Only two messages to fill the cache.
316  relayer.setPipelineLength(2);
317 
318  // Let's create a dummy O2 Message with two headers in the stack:
319  // - DataHeader matching the one provided in the input
320  DataHeader dh;
321  dh.dataDescription = "CLUSTERS";
322  dh.dataOrigin = "TPC";
323  dh.subSpecification = 0;
324  dh.splitPayloadIndex = 0;
325  dh.splitPayloadParts = 1;
326 
327  DataProcessingHeader dph{0, 1};
328  auto transport = FairMQTransportFactory::CreateTransportFactory("zeromq");
329  auto channelAlloc = o2::pmr::getTransportAllocator(transport.get());
330  auto createMessage = [&transport, &channelAlloc, &relayer, &dh](auto const& h) {
331  std::array<FairMQMessagePtr, 2> messages;
332  messages[0] = o2::pmr::getMessage(Stack{channelAlloc, dh, h});
333  messages[1] = transport->CreateMessage(1000);
334  FairMQMessagePtr& header = messages[0];
335  FairMQMessagePtr& payload = messages[1];
336  auto res = relayer.relay(header->GetData(), messages.data(), messages.size());
337  BOOST_REQUIRE(res != DataRelayer::RelayChoice::WillRelay || header.get() == nullptr);
338  BOOST_REQUIRE(res != DataRelayer::RelayChoice::WillRelay || payload.get() == nullptr);
339  BOOST_REQUIRE(res != DataRelayer::RelayChoice::Backpressured || header.get() != nullptr);
340  BOOST_REQUIRE(res != DataRelayer::RelayChoice::Backpressured || payload.get() != nullptr);
341  };
342 
343  // This fills the cache, and then empties it.
344  createMessage(DataProcessingHeader{0, 1});
345  createMessage(DataProcessingHeader{1, 1});
346  std::vector<RecordAction> ready;
347  relayer.getReadyToProcess(ready);
348  BOOST_REQUIRE_EQUAL(ready.size(), 2);
349  BOOST_CHECK_EQUAL(ready[0].slot.index, 1);
350  BOOST_CHECK_EQUAL(ready[1].slot.index, 0);
351  BOOST_CHECK_EQUAL(ready[0].op, CompletionPolicy::CompletionOp::Consume);
352  BOOST_CHECK_EQUAL(ready[1].op, CompletionPolicy::CompletionOp::Consume);
353  for (size_t i = 0; i < ready.size(); ++i) {
354  auto result = relayer.consumeAllInputsForTimeslice(ready[i].slot);
355  }
356 
357  // This fills the cache and makes 2 obsolete.
358  createMessage(DataProcessingHeader{2, 1});
359  createMessage(DataProcessingHeader{3, 1});
360  createMessage(DataProcessingHeader{4, 1});
361  ready.clear();
362  relayer.getReadyToProcess(ready);
363  BOOST_REQUIRE_EQUAL(ready.size(), 2);
364 
365  auto result1 = relayer.consumeAllInputsForTimeslice(ready[0].slot);
366  auto result2 = relayer.consumeAllInputsForTimeslice(ready[1].slot);
367  // One for the header, one for the payload
368  BOOST_REQUIRE_EQUAL(result1.size(), 1);
369  BOOST_REQUIRE_EQUAL(result2.size(), 1);
370 }
