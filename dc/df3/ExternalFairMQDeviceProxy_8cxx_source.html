61 void sendOnChannel(FairMQDevice& device, FairMQParts& messages, std::string const& channel)
62 {
63  // Note: DPL is only setting up one instance of a channel while FairMQ allows to have an
64  // array of channels, the index is 0 in the call
65  constexpr auto index = 0;
66  LOG(debug) << "sending " << messages.Size() << " messages on " << channel;
67  // TODO: we can make this configurable
68  const int maxTimeout = 10000;
69  int timeout = 0;
70  // try dispatch with increasing timeout in order to also drop a warning if the dispatching
71  // has been tried multiple times within max timeout
72  // since we do not want any messages to be dropped at this stage, we stay in the loop until
73  // the downstream congestion is resolved
74  // TODO: we might want to treat this error condition some levels higher up, but for
75  // the moment its an appropriate solution. The important thing is not to drop
76  // messages and to be informed about the congestion.
77  while (device.Send(messages, channel, index, timeout) < 0) {
78  if (timeout == 0) {
79  timeout = 1;
80  } else if (timeout < maxTimeout) {
81  timeout *= 10;
82  } else {
83  LOG(error) << "failed to dispatch messages on channel " << channel << ", downstream queue might be full\n"
84  << "or unconnected. No data is dropped, keep on trying, but this will hold the reading from\n"
85  << "the input and expose back-pressure upstream. RESOLVE DOWNSTREAM CONGESTION to continue";
86  if (timeout == maxTimeout) {
87  // we add 1ms to disable the warning below
88  timeout += 1;
89  }
90  }
91  if (device.NewStatePending()) {
92  LOG(error) << "device state change is requested, dropping " << messages.Size() << " pending message(s)\n"
93  << "on channel " << channel << "\n"
94  << "ATTENTION: DATA IS LOST! Could not dispatch data to downstream consumer(s), check if\n"
95  << "consumers have been terminated too early";
96  // make sure we disable the warning below
97  timeout = maxTimeout + 1;
98  break;
99  }
100  }
101  // FIXME: we need a better logic for avoiding message spam
102  if (timeout > 1 && timeout <= maxTimeout) {
103  LOG(warning) << "dispatching on channel " << channel << " was delayed by " << timeout << " ms";
104  }
105  // TODO: feeling this is a bit awkward, but the interface of FairMQParts does not provide a
106  // method to clear the content.
107  // Maybe the FairMQ API can be improved at some point. Actually the ownership of all messages should be passed
108  // on to the transport and the messages should be empty after sending and the parts content can be cleared.
109  //assert(std::accumulate(messages.begin(), messages.end(), true, [](bool a, auto const& msg) {return a && (msg.get() == nullptr);}));
110  messages.fParts.clear();
111 }

125 void sendOnChannel(FairMQDevice& device, o2::header::Stack&& headerStack, FairMQMessagePtr&& payloadMessage, OutputSpec const& spec, ChannelRetriever& channelRetriever)
126 {
127  const auto* dph = o2::header::get<DataProcessingHeader*>(headerStack.data());
128  if (!dph) {
129  LOG(error) << "Header Stack does not follow the O2 data model, DataProcessingHeader missing";
130  return;
131  }
132  auto channelName = channelRetriever(spec, dph->startTime);
133  constexpr auto index = 0;
134  if (channelName.empty()) {
135  LOG(warning) << "can not find matching channel for " << DataSpecUtils::describe(spec);
136  return;
137  }
138  for (auto& channelInfo : device.fChannels) {
139  if (channelInfo.first != channelName) {
140  continue;
141  }
142  assert(channelInfo.second.size() == 1);
143  // allocate the header message using the underlying transport of the channel
144  auto channelAlloc = o2::pmr::getTransportAllocator(channelInfo.second[index].Transport());
145  FairMQMessagePtr headerMessage = o2::pmr::getMessage(std::move(headerStack), channelAlloc);
146 
147  FairMQParts out;
148  out.AddPart(std::move(headerMessage));
149  out.AddPart(std::move(payloadMessage));
150  sendOnChannel(device, out, channelName);
151  return;
152  }
153  LOG(error) << "internal mismatch, can not find channel " << channelName << " in the list of channel infos of the device";
154 }

188 {
189  bool throwOnUnmatchedInputs = config.throwOnUnmatchedInputs;
190  // structure to hold information on the unmatched data and print a warning at cleanup
191  class DroppedDataSpecs
192  {
193  public:
194  DroppedDataSpecs() = default;
195  ~DroppedDataSpecs()
196  {
197  warning();
198  }

200  bool find(std::string const& desc) const
201  {
202  return descriptions.find(desc) != std::string::npos;
203  }
204

210  void warning() const
211  {
212  if (not descriptions.empty()) {
213  LOG(warning) << "Some input data could not be matched by filter rules to output specs\n"
214  << "Active rules: " << descriptions << "\n"
215  << "DROPPING OF THESE MESSAGES HAS BEEN ENABLED BY CONFIGURATION";
216  }
217  }
218 
219  private:
220  std::string descriptions;

223  return [filterSpecs = std::move(filterSpecs), throwOnUnmatchedInputs, droppedDataSpecs = std::make_shared<DroppedDataSpecs>()](FairMQDevice& device, FairMQParts& parts, ChannelRetriever channelRetriever) {
224  std::unordered_map<std::string, FairMQParts> outputs;
225  std::vector<std::string> unmatchedDescriptions;
226  static int64_t dplCounter = -1;
227  dplCounter++;
228  for (size_t msgidx = 0; msgidx < parts.Size(); msgidx += 2) {
229  const auto dh = o2::header::get<DataHeader*>(parts.At(msgidx)->GetData());
230  if (!dh) {
231  LOG(error) << "data on input " << msgidx << " does not follow the O2 data model, DataHeader missing";
232  continue;
233  }
234  auto dph = o2::header::get<DataProcessingHeader*>(parts.At(msgidx)->GetData());
235  if (!dph) {
236  LOG(error) << "data on input " << msgidx << " does not follow the O2 data model, DataProcessingHeader missing";
237  continue;
238  }
239  const_cast<DataProcessingHeader*>(dph)->startTime = dplCounter;
240  LOG(debug) << msgidx << ": " << DataSpecUtils::describe(OutputSpec{dh->dataOrigin, dh->dataDescription, dh->subSpecification}) << " part " << dh->splitPayloadIndex << " of " << dh->splitPayloadParts << " payload " << parts.At(msgidx + 1)->GetSize();
241 
242  OutputSpec query{dh->dataOrigin, dh->dataDescription, dh->subSpecification};
243  LOG(debug) << "processing " << DataSpecUtils::describe(OutputSpec{dh->dataOrigin, dh->dataDescription, dh->subSpecification}) << " time slice " << dph->startTime << " part " << dh->splitPayloadIndex << " of " << dh->splitPayloadParts;
244  size_t finalBlockIndex = 0;
245  std::string channelName = "";
246 
247  for (auto const& spec : filterSpecs) {
248  // filter on the specified OutputSpecs, the default value is a ConcreteDataTypeMatcher with origin and description 'any'
249  if (DataSpecUtils::match(spec, OutputSpec{{header::gDataOriginAny, header::gDataDescriptionAny}}) ||
250  DataSpecUtils::match(spec, query)) {
251  channelName = channelRetriever(query, dph->startTime);
252  if (channelName.empty()) {
253  LOG(warning) << "can not find matching channel, not able to adopt " << DataSpecUtils::describe(query);
254  }
255  break;
256  }
257  }
258  if (!channelName.empty()) {
259  if (dh->splitPayloadParts > 0 && dh->splitPayloadParts == dh->splitPayloadIndex) {
260  // this is indicating a sequence of payloads following the header
261  // FIXME: we will probably also set the DataHeader version
262  finalBlockIndex = msgidx + dh->splitPayloadParts + 1;
263  } else {
264  // We can consider the next splitPayloadParts as one block of messages pairs
265  // because we are guaranteed they are all the same.
266  // If splitPayloadParts = 0, we assume that means there is only one (header, payload)
267  // pair.
268  finalBlockIndex = msgidx + (dh->splitPayloadParts > 0 ? dh->splitPayloadParts : 1) * 2;
269  }
270  assert(finalBlockIndex >= msgidx + 2);
271  if (finalBlockIndex > parts.Size()) {
272  // TODO error handling
273  //LOGP(error, "DataHeader::splitPayloadParts invalid");
274  continue;
275  }
276 
277  // the checks for consistency of split payload parts are of informative nature
278  // forwarding happens independently
279  //if (dh->splitPayloadParts > 1 && dh->splitPayloadParts != std::numeric_limits<decltype(dh->splitPayloadParts)>::max()) {
280  // if (lastSplitPartIndex == -1 && dh->splitPayloadIndex != 0) {
281  // LOG(warning) << "wrong split part index, expecting the first of " << dh->splitPayloadParts << " part(s)";
282  // } else if (dh->splitPayloadIndex != lastSplitPartIndex + 1) {
283  // LOG(warning) << "unordered split parts, expecting part " << lastSplitPartIndex + 1 << ", got " << dh->splitPayloadIndex
284  // << " of " << dh->splitPayloadParts;
285  // } else if (channelNameForSplitParts.empty() == false && channelName != channelNameForSplitParts) {
286  // LOG(error) << "inconsistent channel for split part " << dh->splitPayloadIndex
287  // << ", matching " << channelName << ", expecting " << channelNameForSplitParts;
288  // }
289  //}
290  LOGP(debug, "associating {} part(s) at index {} to channel {} ({})", finalBlockIndex - msgidx, msgidx, channelName, outputs[channelName].Size());
291  for (; msgidx < finalBlockIndex; ++msgidx) {
292  outputs[channelName].AddPart(std::move(parts.At(msgidx)));
293  }
294  msgidx -= 2;
295  }
296  if (finalBlockIndex == 0 && !DataSpecUtils::match(query, "DPL", "EOS", 0)) {
297  unmatchedDescriptions.emplace_back(DataSpecUtils::describe(query));
298  }
299  } // end of loop over parts
300 
301  for (auto& [channelName, channelParts] : outputs) {
302  if (channelParts.Size() == 0) {
303  continue;
304  }
305  sendOnChannel(device, channelParts, channelName);
306  }
307  if (not unmatchedDescriptions.empty()) {
308  if (throwOnUnmatchedInputs) {
309  std::string descriptions;
310  for (auto const& desc : unmatchedDescriptions) {
311  descriptions += "\n " + desc;
312  }
313  throw std::runtime_error("No matching filter rule for input data " + descriptions +
314  "\n Add appropriate matcher(s) to dataspec definition or allow to drop unmatched data");
315  } else {
316  bool changed = false;
317  for (auto const& desc : unmatchedDescriptions) {
318  if (not droppedDataSpecs->find(desc)) {
319  // a new description
320  droppedDataSpecs->add(desc);
321  changed = true;
322  }
323  }
324  if (changed) {
325  droppedDataSpecs->warning();
326  }
327  }
328  }
329  };
330 }

332 InjectorFunction incrementalConverter(OutputSpec const& spec, uint64_t startTime, uint64_t step)
333 {
334  auto timesliceId = std::make_shared<size_t>(startTime);
335 
336  return [timesliceId, spec, step](FairMQDevice& device, FairMQParts& parts, ChannelRetriever channelRetriever) {
337  // We iterate on all the parts and we send them two by two,
338  // adding the appropriate O2 header.
339  for (size_t i = 0; i < parts.Size(); ++i) {
340  DataHeader dh;
341 
342  // FIXME: this only supports fully specified output specs...
343  ConcreteDataMatcher matcher = DataSpecUtils::asConcreteDataMatcher(spec);
344  dh.dataOrigin = matcher.origin;
345  dh.dataDescription = matcher.description;
346  dh.subSpecification = matcher.subSpec;
347  dh.payloadSize = parts.At(i)->GetSize();
348 
349  DataProcessingHeader dph{*timesliceId, 0};
350  *timesliceId += step;
351  //we have to move the incoming data
352  o2::header::Stack headerStack{dh, dph};
353 
354  sendOnChannel(device, std::move(headerStack), std::move(parts.At(i)), spec, channelRetriever);
355  }
356  };
357 }

359 DataProcessorSpec specifyExternalFairMQDeviceProxy(char const* name,
360  std::vector<OutputSpec> const& outputs,
361  char const* defaultChannelConfig,
362  std::function<void(FairMQDevice&,
363  FairMQParts&,
364  ChannelRetriever)>
365  converter,
366  uint64_t minSHM)
367 {
368  DataProcessorSpec spec;
369  spec.name = strdup(name);
370  spec.inputs = {};
371  spec.outputs = outputs;
372  // The Init method will register a new "Out of band" channel and
373  // attach an OnData to it which is responsible for converting incoming
374  // messages into DPL messages.
375  spec.algorithm = AlgorithmSpec{[converter, channel = spec.name, minSHM](InitContext& ctx) {
376  auto device = ctx.services().get<RawDeviceService>().device();
377  // make a copy of the output routes and pass to the lambda by move
378  auto outputRoutes = ctx.services().get<RawDeviceService>().spec().outputs;
379  auto outputChannels = ctx.services().get<RawDeviceService>().spec().outputChannels;
380  assert(device);
381 
382  // check that the name used for registering the OnData callback corresponds
383  // to the configured output channel, unfortunately we can not automatically
384  // deduce this from list of channels without knowing the name, because there
385  // will be multiple channels. At least we throw a more informative exception.
386  // FairMQDevice calls the custom init before the channels have been configured
387  // so we do the check before starting in a dedicated callback
388  auto channelConfigurationChecker = [channel, device]() {
389  if (device->fChannels.count(channel) == 0) {
390  throw std::runtime_error("the required out-of-band channel '" + channel + "' has not been configured, please check the name in the channel configuration");
391  }
392  };
393  ctx.services().get<CallbackService>().set(CallbackService::Id::Start, channelConfigurationChecker);
394  // Converter should pump messages
395 
396  auto dataHandler = [device, converter, outputRoutes = std::move(outputRoutes), control = &ctx.services().get<ControlService>(), outputChannels = std::move(outputChannels)](FairMQParts& inputs, int) {
397  // pass a copy of the outputRoutes
398  auto channelRetriever = [&outputRoutes](OutputSpec const& query, DataProcessingHeader::StartTime timeslice) -> std::string {
399  for (auto& route : outputRoutes) {
400  LOG(debug) << "matching: " << DataSpecUtils::describe(query) << " to route " << DataSpecUtils::describe(route.matcher);
401  if (DataSpecUtils::match(route.matcher, query) && ((timeslice % route.maxTimeslices) == route.timeslice)) {
402  return route.channel;
403  }
404  }
405  return std::string("");
406  };
407 
408  auto checkEos = [&inputs]() -> bool {
409  std::string channelNameForSplitParts;
410  for (size_t msgidx = 0; msgidx < inputs.Size() / 2; ++msgidx) {
411  auto const sih = o2::header::get<SourceInfoHeader*>(inputs.At(msgidx * 2)->GetData());
412  if (sih != nullptr && sih->state == InputChannelState::Completed) {
413  return true;
414  }
415  }
416  return false;
417  };
418  // we buffer the condition since the converter will forward messages by move
419  bool doEos = checkEos();
420  converter(*device, inputs, channelRetriever);
421 
422  if (doEos) {
423  for (auto const& channel : outputChannels) {
424  DataProcessingHelpers::sendEndOfStream(*device, channel);
425  }
426  control->readyToQuit(QuitRequest::Me);
427  }
428  };
429 
430  auto runHandler = [dataHandler, channel, minSHM](ProcessingContext& ctx) {
431  static RateLimiter limiter;
432  auto device = ctx.services().get<RawDeviceService>().device();
433  limiter.check(ctx, std::stoi(device->fConfig->GetValue<std::string>("timeframes-rate-limit")), minSHM);
434 
435  FairMQParts parts;
436  device->Receive(parts, channel, 0);
437  dataHandler(parts, 0);
438  };
439 
440  return runHandler;
441  }};
442  const char* d = strdup(((std::string(defaultChannelConfig).find("name=") == std::string::npos ? (std::string("name=") + name + ",") : "") + std::string(defaultChannelConfig)).c_str());
443  spec.options = {
444  ConfigParamSpec{"channel-config", VariantType::String, d, {"Out-of-band channel config"}}};
445  return spec;
446 }

455 DataProcessorSpec specifyFairMQDeviceOutputProxy(char const* name,
456  Inputs const& inputSpecs,
457  const char* defaultChannelConfig)
458 {
459  DataProcessorSpec spec;
460  spec.name = name;
461  spec.inputs = inputSpecs;
462  spec.outputs = {};
463  spec.algorithm = adaptStateful([inputSpecs](CallbackService& callbacks, RawDeviceService& rds, DeviceSpec const& deviceSpec, ConfigParamRegistry const& options) {
464  // we can retrieve the channel name from the channel configuration string
465  // FIXME: even if a --channel-config option is specified on the command line, always the default string
466  // is retrieved from the config registry. The channel name thus needs to be configured in the default
467  // string AND must match the name in an optional channel config.
468  std::string channelConfig = options.get<std::string>("channel-config");
469  std::regex r{R"(name=([^,]*))"};
470  std::vector<std::string> values{std::sregex_token_iterator{std::begin(channelConfig), std::end(channelConfig), r, 1},
471  std::sregex_token_iterator{}};
472  if (values.size() != 1 || values[0].empty()) {
473  throw std::runtime_error("failed to extract channel name from channel configuration parameter '" + channelConfig + "'");
474  }
475  std::string outputChannelName = values[0];
476 
477  auto* device = rds.device();
478  // check that the input spec bindings have corresponding output channels
479  // FairMQDevice calls the custom init before the channels have been configured
480  // so we do the check before starting in a dedicated callback
481  auto channelConfigurationChecker = [inputSpecs = std::move(inputSpecs), device, outputChannelName]() {
482  LOG(info) << "checking channel configuration";
483  if (device->fChannels.count(outputChannelName) == 0) {
484  throw std::runtime_error("no corresponding output channel found for input '" + outputChannelName + "'");
485  }
486  };
487  callbacks.set(CallbackService::Id::Start, channelConfigurationChecker);
488  auto lastDataProcessingHeader = std::make_shared<DataProcessingHeader>(0, 0);
489 
490  if (deviceSpec.forwards.size() > 0) {
491  // check that no internal forwards are existing, i.e. that proxy is at the end of the workflow
492  // in principle we can be less strict here if we check only for the defined input specs that there
493  // are no internal forwards
494  throw std::runtime_error("can not add forward targets outside DPL if internal forwards are existing, the proxy must be at the end of the workflow");
495  }
496  for (auto const& inputSpec : inputSpecs) {
497  // this is a prototype, in principle we want to have all spec objects const
498  // and so only the const object can be retrieved from service registry
499  ForwardRoute route{0, 1, inputSpec, outputChannelName};
500  const_cast<DeviceSpec&>(deviceSpec).forwards.emplace_back(route);
501  }
502 
503  auto forwardEos = [device, lastDataProcessingHeader, outputChannelName](EndOfStreamContext&) {
504  // DPL implements an internal end of stream signal, which is propagated through
505  // all downstream channels if a source is dry, make it available to other external
506  // devices via a message of type {DPL/EOS/0}
507  for (auto& channelInfo : device->fChannels) {
508  auto& channelName = channelInfo.first;
509  if (channelName != outputChannelName) {
510  continue;
511  }
512  DataHeader dh;
513  dh.dataOrigin = "DPL";
514  dh.dataDescription = "EOS";
515  dh.subSpecification = 0;
516  dh.payloadSize = 0;
517  dh.payloadSerializationMethod = o2::header::gSerializationMethodNone;
518  dh.tfCounter = 0;
519  dh.firstTForbit = 0;
520  SourceInfoHeader sih;
521  sih.state = InputChannelState::Completed;
522  // allocate the header message using the underlying transport of the channel
523  auto channelAlloc = o2::pmr::getTransportAllocator(channelInfo.second[0].Transport());
524  auto headerMessage = o2::pmr::getMessage(o2::header::Stack{channelAlloc, dh, *lastDataProcessingHeader, sih});
525  FairMQParts out;
526  out.AddPart(std::move(headerMessage));
527  // add empty payload message
528  out.AddPart(std::move(device->NewMessageFor(channelName, 0, 0)));
529  sendOnChannel(*device, out, channelName);
530  }
531  };
532  callbacks.set(CallbackService::Id::EndOfStream, forwardEos);
533 
534  return adaptStateless([lastDataProcessingHeader](RawDeviceService& rds, InputRecord& inputs) {
535  for (size_t ii = 0; ii != inputs.size(); ++ii) {
536  for (size_t pi = 0; pi < inputs.getNofParts(ii); ++pi) {
537  auto part = inputs.getByPos(ii, pi);
538  const auto* dph = o2::header::get<DataProcessingHeader*>(part.header);
539  if (dph) {
540  // FIXME: should we implement an assignment operator for DataProcessingHeader?
541  lastDataProcessingHeader->startTime = dph->startTime;
542  lastDataProcessingHeader->duration = dph->duration;

557 DataProcessorSpec specifyFairMQDeviceMultiOutputProxy(char const* name,
558  Inputs const& inputSpecs,
559  const char* defaultChannelConfig,
560  ChannelSelector channelSelector)
561 {
562  // FIXME: this looks like a code duplication with the function above, check if the
563  // two can be combined
564  DataProcessorSpec spec;
565  spec.name = name;
566  spec.inputs = inputSpecs;
567  spec.outputs = {};
568  spec.algorithm = adaptStateful([inputSpecs, channelSelector](CallbackService& callbacks, RawDeviceService& rds, const DeviceSpec& deviceSpec) {
569  auto device = rds.device();
570  // check that the input spec bindings have corresponding output channels
571  // FairMQDevice calls the custom init before the channels have been configured
572  // so we do the check before starting in a dedicated callback
573  // also we set forwards for all input specs and keep a list of all channels so we can send EOS on them
574  auto channelNames = std::make_shared<std::vector<std::string>>();
575  auto channelConfigurationInitializer = [inputSpecs = std::move(inputSpecs), device, channelSelector, &deviceSpec, channelNames]() {
576  if (deviceSpec.forwards.size() > 0) {
577  // check that no internal forwards are existing, i.e. that proxy is at the end of the workflow
578  // in principle we can be less strict here if we check only for the defined input specs that there
579  // are no internal forwards
580  throw std::runtime_error("can not add forward targets outside DPL if internal forwards are existing, the proxy must be at the end of the workflow");
581  }
582  for (auto const& spec : inputSpecs) {
583  auto channel = channelSelector(spec, device->fChannels);
584  if (device->fChannels.count(channel) == 0) {
585  throw std::runtime_error("no corresponding output channel found for input '" + channel + "'");
586  }
587  ForwardRoute route{0, 1, spec, channel};
588  // this we will try to fix on the framework level, there will be an API to
589  // set external routes. Basically, this has to be added while setting up the
590  // workflow. After that, the actual spec provided by the service is supposed
591  // to be const by design
592  const_cast<DeviceSpec&>(deviceSpec).forwards.emplace_back(route);
593 
594  channelNames->emplace_back(std::move(channel));
595  }
596  };
597  callbacks.set(CallbackService::Id::Start, channelConfigurationInitializer);
598 
599  auto lastDataProcessingHeader = std::make_shared<DataProcessingHeader>(0, 0);
600  auto forwardEos = [device, lastDataProcessingHeader, channelNames](EndOfStreamContext&) {
601  // DPL implements an internal end of stream signal, which is propagated through
602  // all downstream channels if a source is dry, make it available to other external
603  // devices via a message of type {DPL/EOS/0}
604  for (auto& channelInfo : device->fChannels) {
605  auto& channelName = channelInfo.first;
606  auto checkChannel = [channelNames = std::move(*channelNames)](std::string const& name) -> bool {
607  for (auto const& n : channelNames) {
608  if (n == name) {
609  return true;
610  }
611  }
612  return false;
613  };
614  if (!checkChannel(channelName)) {
615  continue;
616  }
617  DataHeader dh;
618  dh.dataOrigin = "DPL";
619  dh.dataDescription = "EOS";
620  dh.subSpecification = 0;
621  dh.payloadSize = 0;
622  dh.payloadSerializationMethod = o2::header::gSerializationMethodNone;
623  dh.tfCounter = 0;
624  dh.firstTForbit = 0;
625  SourceInfoHeader sih;
626  sih.state = InputChannelState::Completed;
627  // allocate the header message using the underlying transport of the channel
628  auto channelAlloc = o2::pmr::getTransportAllocator(channelInfo.second[0].Transport());
629  auto headerMessage = o2::pmr::getMessage(o2::header::Stack{channelAlloc, dh, *lastDataProcessingHeader, sih});
630  FairMQParts out;
631  out.AddPart(std::move(headerMessage));
632  // add empty payload message
633  out.AddPart(std::move(device->NewMessageFor(channelName, 0, 0)));
634  sendOnChannel(*device, out, channelName);
635  }
636  };
637  callbacks.set(CallbackService::Id::EndOfStream, forwardEos);
638 
639  return adaptStateless([channelSelector, lastDataProcessingHeader](RawDeviceService& rds, InputRecord& inputs) {
640  // there is nothing to do if the forwarding is handled on the framework level
641  // as forward routes but we need to keep a copy of the last DataProcessingHeader
642  // for sending the EOS
643  for (size_t ii = 0; ii != inputs.size(); ++ii) {
644  for (size_t pi = 0; pi < inputs.getNofParts(ii); ++pi) {
645  auto part = inputs.getByPos(ii, pi);
646  const auto* dph = o2::header::get<DataProcessingHeader*>(part.header);
647  if (dph) {
648  // FIXME: should we implement an assignment operator for DataProcessingHeader?
649  lastDataProcessingHeader->startTime = dph->startTime;
650  lastDataProcessingHeader->duration = dph->duration;
651  lastDataProcessingHeader->creation = dph->creation;
652  }
653  }
654  }
655  });
656  });
657  const char* d = strdup(((std::string(defaultChannelConfig).find("name=") == std::string::npos ? (std::string("name=") + name + ",") : "") + std::string(defaultChannelConfig)).c_str());
658  spec.options = {
659  ConfigParamSpec{"channel-config", VariantType::String, d, {"Out-of-band channel config"}},
660  };
661 
662  return spec;
663 }
