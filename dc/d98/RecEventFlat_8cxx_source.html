27 int RecEventFlat::next()
28 {
29  if (mEntry >= mNEntries) {
30  return 0;
31  }
32 
33  tdcPedEv.fill(false);
34  tdcPedOr.fill(false);
35  tdcPedQC.fill(false);
36  tdcPedMissing.fill(false);
37  adcPedEv.fill(false);
38  adcPedOr.fill(false);
39  adcPedQC.fill(false);
40  adcPedMissing.fill(false);
41  ezdc.clear();
42  for (int itdc = 0; itdc < NTDCChannels; itdc++) {
43  TDCVal[itdc].clear();
44  TDCAmp[itdc].clear();
45  }
46 
47  // Get References
48  mCurB = mRecBC->at(mEntry);
49  mCurB.getRef(mFirstE, mNE, mFirstT, mNT, mFirstI, mNI);
50  mStopE = mFirstE + mNE;
51  mStopT = mFirstT + mNT;
52  mStopI = mFirstI + mNI;
53 
54  ir = mCurB.ir;
55  channels = mCurB.channels;
56  triggers = mCurB.triggers;
57 
58  // Decode event info
59  int infoState = 0;
60  uint16_t code = 0;
61  uint32_t map = 0;
62  for (int i = mFirstI; i < mStopI; i++) {
63  uint16_t info = mInfo->at(i);
64  if (infoState == 0) {
65  if (info & 0x8000) {
66  LOGF(error, "Inconsistent info stream at word %d: 0x%4u", i, info);
67  break;
68  }
69  code = info & 0x03ff;
70  uint8_t ch = (info >> 10) & 0x1f;
71  if (ch == 0x1f) {
72  infoState = 1;
73  } else if (ch < NChannels) {
74  decodeInfo(ch, code);
75  } else {
76  LOGF(error, "Info about non existing channel: %u", ch);
77  }
78  } else if (infoState == 1) {
79  if (info & 0x8000) {
80  map = info & 0x7fff;
81  } else {
82  LOGF(error, "Inconsistent info stream at word %d: 0x%4u", i, info);
83  break;
84  }
85  infoState = 2;
86  } else if (infoState == 2) {
87  if (info & 0x8000) {
88  uint32_t maph = info & 0x7fff;
89  map = (maph << 15) | map;
90  decodeMapInfo(map, code);
91  } else {
92  LOGF(error, "Inconsistent info stream at word %d: 0x%4u", i, info);
93  break;
94  }
95  infoState = 0;
96  }
97  }
98 
99  // Decode energy
100  for (int i = mFirstE; i < mStopE; i++) {
101  auto myenergy = mEnergy->at(i);
102  auto ch = myenergy.ch();
103  ezdc[ch] = myenergy.energy();
104  // Assign implicit event info
105  if (adcPedOr[ch] == false && adcPedQC[ch] == false && adcPedMissing[ch] == false) {
106  adcPedEv[ch] = true;
107  }
108  }
109  // Decode TDCs
110  for (int i = mFirstT; i < mStopT; i++) {
111  auto mytdc = mTDCData->at(i);
112  auto ch = mytdc.ch();
113  if (ch < NTDCChannels) {
114  TDCVal[ch].push_back(mytdc.val);
115  TDCAmp[ch].push_back(mytdc.amp);
116  // Assign implicit event info
117  if (tdcPedQC[ch] == false && tdcPedMissing[ch] == false) {
118  tdcPedOr[ch] = true;
119  }
120  }
121  }
122 
123  mEntry++;
124  return mEntry;
125 }

128 {

139 void RecEventFlat::decodeInfo(uint8_t ch, uint16_t code)
140 {
141  if (mVerbosity != DbgZero) {
142  printf("%9u.%04u Info: ch=%2d (%s) code=%-4u (%s)\n", ir.orbit, ir.bc, ch, ch < NChannels ? ChannelNames[ch].data() : "N.D.",
143  code, code < MsgEnd ? MsgText[code].data() : "undefined");
144  }
145  if (code == MsgTDCPedQC) {
146  tdcPedQC[ch] = true;
147  }
148  if (code == MsgTDCPedMissing) {
149  tdcPedMissing[ch] = true;
150  }
151  if (code == MsgADCPedOr) {
152  adcPedOr[ch] = true;
153  }
154  if (code == MsgADCPedQC) {
155  adcPedQC[ch] = true;
156  }
157  if (code == MsgADCPedMissing) {
158  adcPedMissing[ch] = true;
159  }
160 }

162 void RecEventFlat::print() const
163 {
164  printf("%9u.%04u ", ir.orbit, ir.bc);
165  printf("nE %2d pos %d nT %2d pos %d nI %2d pos %d\n", mNE, mFirstE, mNT, mFirstT, mNI, mFirstI);
166  printf("%9u.%04u ", ir.orbit, ir.bc);
167  printf("Read:");
168  for (int ic = 0; ic < NDigiChannels; ic++) {
169  if (ic % NChPerModule == 0) {
170  if (ic == 0) {
171  printf(" %d[", ic / NChPerModule);
172  } else {
173  printf("] %d[", ic / NChPerModule);
174  }
175  }
176  if (channels & (0x1 << ic)) {
177  printf("R");
178  } else {
179  printf(" ");
180  }
181  }
182  printf("]\n");
183  printf("%9u.%04u ", ir.orbit, ir.bc);
184  printf("Hits:");
185  for (int ic = 0; ic < NDigiChannels; ic++) {
186  if (ic % NChPerModule == 0) {
187  if (ic == 0) {
188  printf(" %d[", ic / NChPerModule);
189  } else {
190  printf("] %d[", ic / NChPerModule);
191  }
192  }
193  bool is_hit = triggers & (0x1 << ic);
194  bool is_trig = mTriggerMask & (0x1 << ic);
195  if (is_trig) {
196  if (is_hit) {
197  printf("T");
198  } else {
199  printf(".");
200  }
201  } else {
202  if (is_hit) {
203  printf("H");
204  } else {
205  printf(" ");
206  }
207  }
208  }
209  printf("]\n");
210 }
