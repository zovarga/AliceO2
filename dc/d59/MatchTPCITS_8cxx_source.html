71 void MatchTPCITS::run(const o2::globaltracking::RecoContainer& inp)
72 {
74  if (!mInitDone) {
75  LOG(fatal) << "init() was not done yet";
76  }
77  mRecoCont = &inp;
78  mStartIR = inp.startIR;
79  updateTimeDependentParams();
80 
81  mTimer[SWTot].Start(false);
82 
83  clear();
84  while (1) {
85  if (!prepareITSData() || !prepareTPCData() || !prepareFITData()) {
86  break;
87  }
88 
89  mTimer[SWDoMatching].Start(false);
90  for (int sec = o2::constants::math::NSectors; sec--;) {
91  doMatching(sec);
92  }
93  mTimer[SWDoMatching].Stop();
94  if (0) { // enabling this creates very verbose output
95  mTimer[SWTot].Stop();
96  printCandidatesTPC();
97  printCandidatesITS();
98  mTimer[SWTot].Start(false);
99  }
100 
101  selectBestMatches();
102 
103  refitWinners();
104 
105  if (mUseFT0 && Params::Instance().runAfterBurner) {
106  runAfterBurner();
107  }
108 
109 #ifdef _ALLOW_DEBUG_TREES_
110  if (mDBGOut && isDebugFlag(WinnerMatchesTree)) {
111  dumpWinnerMatches();
112  }
113 #endif
114  break;
115  }
116  mTimer[SWTot].Stop();
117 
118  if (mParams->verbosity > 0) {
119  for (int i = 0; i < NStopWatches; i++) {
120  LOGF(info, "Timing for %15s: Cpu: %.3e Real: %.3e s in %d slots of TF#%d", TimerName[i], mTimer[i].CpuTime(), mTimer[i].RealTime(), mTimer[i].Counter() - 1, mTFCount);
121  }
122  }
123  mTFCount++;
124 }

220 void MatchTPCITS::updateTimeDependentParams()
221 {
223  auto& gasParam = o2::tpc::ParameterGas::Instance();
224  auto& elParam = o2::tpc::ParameterElectronics::Instance();
225  auto& detParam = o2::tpc::ParameterDetector::Instance();
226  mTPCTBinMUS = elParam.ZbinWidth;
227  mTPCTBinNS = mTPCTBinMUS * 1e3;
228  mTPCVDrift0 = gasParam.DriftV;
229  mTPCZMax = detParam.TPClength;
230  mTPCTBinMUSInv = 1. / mTPCTBinMUS;
231  assert(mITSROFrameLengthMUS > 0.0f);
232  mTPCBin2Z = mTPCTBinMUS * mTPCVDrift0;
233  mZ2TPCBin = 1. / mTPCBin2Z;
234  mTPCVDrift0Inv = 1. / mTPCVDrift0;
235  mNTPCBinsFullDrift = mTPCZMax * mZ2TPCBin;
236  mTPCTimeEdgeTSafeMargin = z2TPCBin(mParams->safeMarginTPCTimeEdge);
237  mTPCExtConstrainedNSigmaInv = 1.f / mParams->tpcExtConstrainedNSigma;
238  mBz = o2::base::Propagator::Instance()->getNominalBz();
239  mFieldON = std::abs(mBz) > 0.01;
240 
241  mMinTPCTrackPtInv = (mFieldON && mParams->minTPCTrackR > 0) ? 1. / std::abs(mParams->minTPCTrackR * mBz * o2::constants::math::B2C) : 999.;
242  mMinITSTrackPtInv = (mFieldON && mParams->minITSTrackR > 0) ? 1. / std::abs(mParams->minITSTrackR * mBz * o2::constants::math::B2C) : 999.;
243 
244  o2::math_utils::Point3D<float> p0(90., 1., 1), p1(90., 100., 100.);
245  auto matbd = o2::base::Propagator::Instance()->getMatBudget(mParams->matCorr, p0, p1);
246  mTPCmeanX0Inv = matbd.meanX2X0 / matbd.length;
247 }

250 void MatchTPCITS::selectBestMatches()
251 {
253  mTimer[SWSelectBest].Start(false);
254  int nValidated = 0, iter = 0, nValidatedTotal = 0;
255 
256  do {
257  nValidated = 0;
258  int ntpc = mTPCWork.size(), nremaining = 0;
259  for (int it = 0; it < ntpc; it++) {
260  auto& tTPC = mTPCWork[it];
261  if (isDisabledTPC(tTPC) || isValidatedTPC(tTPC)) {
262  continue;
263  }
264  nremaining++;
265  if (validateTPCMatch(it)) {
266  nValidated++;
267  continue;
268  }
269  }
270  if (mParams->verbosity > 0) {
271  LOGP(info, "iter {}: Validated {} of {} remaining matches", iter, nValidated, nremaining);
272  }
273  iter++;
274  nValidatedTotal += nValidated;
275  } while (nValidated);
276 
277  mTimer[SWSelectBest].Stop();
278  LOGP(info, "Validated {} matches for {} TPC tracks in {} iterations", nValidatedTotal, mTPCWork.size(), iter);
279 }

281 //______________________________________________
282 bool MatchTPCITS::validateTPCMatch(int iTPC)
283 {
284  const auto& tTPC = mTPCWork[iTPC];
285  auto& rcTPC = mMatchRecordsTPC[tTPC.matchID]; // best TPC->ITS match
286  /* // should never happen
287  if (rcTPC.nextRecID == Validated) {
288  LOG(warning) << "TPC->ITS was already validated";
289  return false; // RS do we need this
290  }
291  */
292  // check if it is consistent with corresponding ITS->TPC match
293  auto& tITS = mITSWork[rcTPC.partnerID]; // partner ITS track
294  auto& rcITS = mMatchRecordsITS[tITS.matchID]; // best ITS->TPC match record
295  if (rcITS.nextRecID == Validated) {
296  return false;
297  }
298  if (rcITS.partnerID == iTPC) { // is best matching TPC track for this ITS track actually iTPC?
299  // unlink winner TPC track from all ITS candidates except winning one
300  int nextTPC = rcTPC.nextRecID;
301  while (nextTPC > MinusOne) {
302  auto& rcTPCrem = mMatchRecordsTPC[nextTPC];
303  removeTPCfromITS(iTPC, rcTPCrem.partnerID); // remove references on mtID from ITS match=rcTPCrem.partnerID
304  nextTPC = rcTPCrem.nextRecID;
305  }
306  rcTPC.nextRecID = Validated;
307  int itsWinID = rcTPC.partnerID;
308 
309  // unlink winner ITS match from all TPC matches using it
310  int nextITS = rcITS.nextRecID;
311  while (nextITS > MinusOne) {
312  auto& rcITSrem = mMatchRecordsITS[nextITS];
313  removeITSfromTPC(itsWinID, rcITSrem.partnerID); // remove references on itsWinID from TPC match=rcITSrem.partnerID
314  nextITS = rcITSrem.nextRecID;
315  }
316  rcITS.nextRecID = Validated;
317  return true;
318  }
319  return false;
320 }

348 void MatchTPCITS::addTPCSeed(const o2::track::TrackParCov& _tr, float t0, float terr, GTrackID srcGID, int tpcID)
349 {
350  // account single TPC seed, can be from standalone TPC track or constrained track from match to TRD and/or TOF
351  const float SQRT12DInv = 2. / sqrt(12.);
352  if (_tr.getX() > o2::constants::geom::XTPCInnerRef + 0.1 || std::abs(_tr.getQ2Pt()) > mMinTPCTrackPtInv) {
353  return;
354  }
355  const auto& tpcOrig = mTPCTracksArray[tpcID];
356  // discard tracks w/o certain number of total or innermost pads (last cluster is innermost one)
357  if (tpcOrig.getNClusterReferences() < mParams->minTPCClusters) {
358  return;
359  }
360  uint8_t clSect = 0, clRow = 0;
361  uint32_t clIdx = 0;
362  tpcOrig.getClusterReference(mTPCTrackClusIdx, tpcOrig.getNClusterReferences() - 1, clSect, clRow, clIdx);
363  if (clRow > mParams->askMinTPCRow) {
364  return;
365  }
366  // create working copy of track param
367  bool extConstrained = srcGID.getSource() != GTrackID::TPC;
368  if (extConstrained) {
369  terr *= mParams->tpcExtConstrainedNSigma;
370  } else {
371  terr += tpcTimeBin2MUS(tpcOrig.hasBothSidesClusters() ? mParams->safeMarginTPCITSTimeBin : mTPCTimeEdgeTSafeMargin);
372  }
373  auto& trc = mTPCWork.emplace_back(
374  TrackLocTPC{_tr, {t0 - terr, t0 + terr}, extConstrained ? t0 : tpcTimeBin2MUS(tpcOrig.getTime0()),
375  // for A/C constrained tracks the terr is half-interval, for externally constrained tracks it is sigma*Nsigma
376  terr * (extConstrained ? mTPCExtConstrainedNSigmaInv : SQRT12DInv),
377  tpcID,
378  srcGID,
379  MinusOne,
380  (extConstrained || tpcOrig.hasBothSidesClusters()) ? TrackLocTPC::Constrained : (tpcOrig.hasASideClustersOnly() ? TrackLocTPC::ASide : TrackLocTPC::CSide)});
381  // propagate to matching Xref
382  if (!propagateToRefX(trc)) {
383  mTPCWork.pop_back(); // discard track whose propagation to XMatchingRef failed
384  return;
385  }
386  if (mMCTruthON) {
387  mTPCLblWork.emplace_back(mTPCTrkLabels[tpcID]);
388  }
389  // cache work track index
390  mTPCSectIndexCache[o2::math_utils::angle2Sector(trc.getAlpha())].push_back(mTPCWork.size() - 1);
391 }

393 //______________________________________________
394 bool MatchTPCITS::prepareTPCData()
395 {
397  mTimer[SWPrepTPC].Start(false);
398  const auto& inp = *mRecoCont;
399 
400  mTPCTracksArray = inp.getTPCTracks();
401  mTPCTrackClusIdx = inp.getTPCTracksClusterRefs();
402  mTPCClusterIdxStruct = &inp.inputsTPCclusters->clusterIndex;
403  mTPCRefitterShMap = inp.clusterShMapTPC;
404  if (mMCTruthON) {
405  mTPCTrkLabels = inp.getTPCTracksMCLabels();
406  }
407 
408  int ntr = mTPCTracksArray.size();
409  mMatchRecordsTPC.reserve(mParams->maxMatchCandidates * ntr); // number of records might be actually more than N tracks!
410  mTPCWork.reserve(ntr);
411  if (mMCTruthON) {
412  mTPCLblWork.reserve(ntr);
413  }
414  for (int sec = o2::constants::math::NSectors; sec--;) {
415  mTPCSectIndexCache[sec].reserve(100 + 1.2 * ntr / o2::constants::math::NSectors);
416  }
417 
418  auto creator = [this](auto& trk, GTrackID gid, float time0, float terr) {
419  if constexpr (isITSTrack<decltype(trk)>()) {
420  // do nothing, ITS tracks will be processed in a direct loop over ROFs
421  }
422  if constexpr (!std::is_base_of_v<o2::track::TrackParCov, std::decay_t<decltype(trk)>>) {
423  return true;
424  } else if (std::abs(trk.getQ2Pt()) > mMinTPCTrackPtInv) {
425  return true;
426  }
427  if constexpr (isTPCTrack<decltype(trk)>()) {
428  // unconstrained TPC track, with t0 = TrackTPC.getTime0+0.5*(DeltaFwd-DeltaBwd) and terr = 0.5*(DeltaFwd+DeltaBwd) in TimeBins
429  if (!this->mSkipTPCOnly) {
430  this->addTPCSeed(trk, this->tpcTimeBin2MUS(time0), this->tpcTimeBin2MUS(terr), gid, gid.getIndex());
431  }
432  }
433  if constexpr (isTPCTOFTrack<decltype(trk)>()) {
434  // TPC track constrained by TOF time, time and its error in \mus
435  this->addTPCSeed(trk, time0, terr, gid, this->mRecoCont->getTPCContributorGID(gid));
436  }
437  if constexpr (isTRDTrack<decltype(trk)>()) {
438  // TPC track constrained by TRD trigger time, time and its error in \mus
439  LOG(error) << "Not ready yet for TPC-TRD tracks";
440  }
441  if constexpr (isTPCTRDTOFTrack<decltype(trk)>()) {
442  // TPC track constrained by TRD and TOF time, time and its error in \mus
443  LOG(error) << "Not ready yet for TPC-TRD-TOF tracks";
444  }
445  return true;
446  };
447  mRecoCont->createTracksVariadic(creator);
448 
449  float maxTime = 0;
450  int nITSROFs = mITSROFTimes.size();
451  // sort tracks in each sector according to their timeMax
452  for (int sec = o2::constants::math::NSectors; sec--;) {
453  auto& indexCache = mTPCSectIndexCache[sec];
454  if (mParams->verbosity > 0) {
455  LOG(info) << "Sorting sector" << sec << " | " << indexCache.size() << " TPC tracks";
456  }
457  if (!indexCache.size()) {
458  continue;
459  }
460  std::sort(indexCache.begin(), indexCache.end(), [this](int a, int b) {
461  auto& trcA = mTPCWork[a];
462  auto& trcB = mTPCWork[b];
463  return (trcA.tBracket.getMax() - trcB.tBracket.getMax()) < 0.;
464  });
465 
466  // build array of 1st entries with tmax corresponding to each ITS ROF (or trigger),
467  // TPC tracks below this entry cannot match to ITS tracks of this and higher ROFs
468 
469  float tmax = mTPCWork[indexCache.back()].tBracket.getMax();
470  if (maxTime < tmax) {
471  maxTime = tmax;
472  }
473  int nbins = 1 + (mITSTriggered ? time2ITSROFrameTrig(tmax, 0) : time2ITSROFrameCont(tmax));
474  auto& timeStart = mTPCTimeStart[sec];
475  timeStart.resize(nbins, -1);
476  int itsROF = 0;
477 
478  timeStart[0] = 0;
479  for (int itr = 0; itr < (int)indexCache.size(); itr++) {
480  auto& trc = mTPCWork[indexCache[itr]];
481  while (itsROF < nITSROFs && !(trc.tBracket < mITSROFTimes[itsROF])) { // 1st ITS frame afte max allowed time for this TPC track
482  itsROF++;
483  }
484  int itsROFMatch = itsROF;
485  if (itsROFMatch && timeStart[--itsROFMatch] == -1) { // register ITSrof preceding the one which exceeds the TPC track tmax
486  timeStart[itsROFMatch] = itr;
487  }
488  }
489  for (int i = 1; i < nbins; i++) {
490  if (timeStart[i] == -1) { // fill gaps with preceding indices
491  timeStart[i] = timeStart[i - 1];
492  }
493  }
494  } // loop over tracks of single sector
495 
496  // FIXME We probably don't need this
497  /*
498  // create mapping from TPC time to ITS ROFs
499  if (mITSROFTimes.back() < maxTime) {
500  maxTime = mITSROFTimes.back().getMax();
501  }
502  int nb = int(maxTime) + 1;
503  mITSROFofTPCBin.resize(nb, -1);
504  int itsROF = 0;
505  for (int ib = 0; ib < nb; ib++) {
506  while (itsROF < nITSROFs && ib < mITSROFTimes[itsROF].getMin()) {
507  itsROF++;
508  }
509  mITSROFofTPCBin[ib] = itsROF;
510  }
511 */
512  mTPCRefitter = std::make_unique<o2::gpu::GPUO2InterfaceRefit>(mTPCClusterIdxStruct, mTPCTransform.get(), mBz, mTPCTrackClusIdx.data(), mTPCRefitterShMap.data(), nullptr, o2::base::Propagator::Instance());
513 
514  mTimer[SWPrepTPC].Stop();
515  return mTPCWork.size() > 0;
516 }

519 bool MatchTPCITS::prepareITSData()
520 {
521  static size_t errCount = 0;
522  constexpr size_t MaxErrors2Report = 10;
523  // Do preparatory work for matching
524  mTimer[SWPrepITS].Start(false);
525  const auto& inp = *mRecoCont;
526 
527  // ITS clusters
528  mITSClusterROFRec = inp.getITSClustersROFRecords();
529  const auto clusITS = inp.getITSClusters();
530  if (mITSClusterROFRec.empty() || clusITS.empty()) {
531  LOG(info) << "No ITS clusters";
532  return false;
533  }
534  const auto patterns = inp.getITSClustersPatterns();
535  auto pattIt = patterns.begin();
536  mITSClustersArray.reserve(clusITS.size());
537  o2::its::ioutils::convertCompactClusters(clusITS, pattIt, mITSClustersArray, *mITSDict);
538  if (mMCTruthON) {
539  mITSClsLabels = inp.mcITSClusters.get();
540  }
541 
542  // ITS tracks
543  mITSTracksArray = inp.getITSTracks();
544  mITSTrackClusIdx = inp.getITSTracksClusterRefs();
545  mITSTrackROFRec = inp.getITSTracksROFRecords();
546  if (mMCTruthON) {
547  mITSTrkLabels = inp.getITSTracksMCLabels();
548  }
549  int nROFs = mITSTrackROFRec.size();
550  mITSWork.reserve(mITSTracksArray.size());
551 
552  // total N ITS clusters in TF
553  const auto& lastClROF = mITSClusterROFRec[nROFs - 1];
554  int nITSClus = lastClROF.getFirstEntry() + lastClROF.getNEntries();
555  mABClusterLinkIndex.resize(nITSClus, MinusOne);
556  for (int sec = o2::constants::math::NSectors; sec--;) {
557  mITSTimeStart[sec].resize(nROFs, -1); // start of ITS work tracks in every sector
558  }
559 
560  for (int irof = 0; irof < nROFs; irof++) {
561  const auto& rofRec = mITSTrackROFRec[irof];
562  int nBC = rofRec.getBCData().differenceInBC(mStartIR);
563  if (nBC < 0) {
564  if (++errCount < MaxErrors2Report) {
565  LOG(error) << "ITS ROF start " << rofRec.getBCData() << " precedes TF 1st orbit " << mStartIR;
566  }
567  continue;
568  }
569  float tMin = nBC * o2::constants::lhc::LHCBunchSpacingMUS;
570  float tMax = (nBC + mITSROFrameLengthInBC) * o2::constants::lhc::LHCBunchSpacingMUS;
571  if (!mITSTriggered) {
572  auto irofCont = nBC / mITSROFrameLengthInBC;
573  if (mITSTrackROFContMapping.size() <= irofCont) { // there might be gaps in the non-empty rofs, this will map continuous ROFs index to non empty ones
574  mITSTrackROFContMapping.resize((1 + irofCont / 128) * 128, 0);
575  }
576  mITSTrackROFContMapping[irofCont] = irof;
577  }
578 
579  mITSROFTimes.emplace_back(tMin, tMax); // ITS ROF min/max time
580 
581  for (int sec = o2::constants::math::NSectors; sec--;) { // start of sector's tracks for this ROF
582  mITSTimeStart[sec][irof] = mITSSectIndexCache[sec].size(); // The sorting does not affect this
583  }
584 
585  int trlim = rofRec.getFirstEntry() + rofRec.getNEntries();
586  for (int it = rofRec.getFirstEntry(); it < trlim; it++) {
587  const auto& trcOrig = mITSTracksArray[it];
588  if (mParams->runAfterBurner) {
589  flagUsedITSClusters(trcOrig);
590  }
591  if (trcOrig.getParamOut().getX() < 1.) {
592  continue; // backward refit failed
593  }
594  if (std::abs(trcOrig.getQ2Pt()) > mMinITSTrackPtInv) {
595  continue;
596  }
597  int nWorkTracks = mITSWork.size();
598  // working copy of outer track param
599  auto& trc = mITSWork.emplace_back(TrackLocITS{trcOrig.getParamOut(), {tMin, tMax}, it, irof, MinusOne});
600  if (!trc.rotate(o2::math_utils::angle2Alpha(trc.getPhiPos()))) {
601  mITSWork.pop_back(); // discard failed track
602  continue;
603  }
604  // make sure the track is at the ref. radius
605  if (!propagateToRefX(trc)) {
606  mITSWork.pop_back(); // discard failed track
607  continue; // add to cache only those ITS tracks which reached ref.X and have reasonable snp
608  }
609  if (mMCTruthON) {
610  mITSLblWork.emplace_back(mITSTrkLabels[it]);
611  }
612  // cache work track index
613  int sector = o2::math_utils::angle2Sector(trc.getAlpha());
614  mITSSectIndexCache[sector].push_back(nWorkTracks);
615 
616  // If the ITS track is very close to the sector edge, it may match also to a TPC track in the neighb. sector.
617  // For a track with Yr and Phir at Xr the distance^2 between the poisition of this track in the neighb. sector
618  // when propagated to Xr (in this neighbouring sector) and the edge will be (neglecting the curvature)
619  // [(Xr*tg(10)-Yr)/(tgPhir+tg70)]^2 / cos(70)^2 // for the next sector
620  // [(Xr*tg(10)+Yr)/(tgPhir-tg70)]^2 / cos(70)^2 // for the prev sector
621  // Distances to the sector edges in neighbourings sectors (at Xref in theit proper frames)
622  float tgp = trc.getSnp();
623  tgp /= std::sqrt((1.f - tgp) * (1.f + tgp)); // tan of track direction XY
624 
625  // sector up
626  float dy2Up = (YMaxAtXMatchingRef - trc.getY()) / (tgp + Tan70);
627  if ((dy2Up * dy2Up * Cos70I2) < mSectEdgeMargin2) { // need to check this track for matching in sector up
628  addLastTrackCloneForNeighbourSector(sector < (o2::constants::math::NSectors - 1) ? sector + 1 : 0);
629  }
630  // sector down
631  float dy2Dn = (YMaxAtXMatchingRef + trc.getY()) / (tgp - Tan70);
632  if ((dy2Dn * dy2Dn * Cos70I2) < mSectEdgeMargin2) { // need to check this track for matching in sector down
633  addLastTrackCloneForNeighbourSector(sector > 1 ? sector - 1 : o2::constants::math::NSectors - 1);
634  }
635  }
636  }
637 
638  if (!mITSTriggered) { // fill the gaps;
639  int nr = mITSTrackROFContMapping.size();
640  for (int i = 1; i < nr; i++) {
641  if (mITSTrackROFContMapping[i] < mITSTrackROFContMapping[i - 1]) {
642  mITSTrackROFContMapping[i] = mITSTrackROFContMapping[i - 1];
643  }
644  }
645  }
646 
647  // sort tracks in each sector according to their min time, then tgl
648  // RSTODO: sorting in tgl will be dangerous once the tracks with different time uncertaincies will be added
649  for (int sec = o2::constants::math::NSectors; sec--;) {
650  auto& indexCache = mITSSectIndexCache[sec];
651  if (mParams->verbosity > 0) {
652  LOG(info) << "Sorting sector" << sec << " | " << indexCache.size() << " ITS tracks";
653  }
654  if (!indexCache.size()) {
655  continue;
656  }
657  std::sort(indexCache.begin(), indexCache.end(), [this](int a, int b) {
658  auto& trackA = mITSWork[a];
659  auto& trackB = mITSWork[b];
660  if (trackA.tBracket.getMin() < trackB.tBracket.getMin()) {
661  return true;
662  } else if (trackA.tBracket.getMin() > trackB.tBracket.getMin()) {
663  return false;
664  }
665  return trackA.getTgl() < trackB.getTgl();
666  });
667  } // loop over tracks of single sector
668  mMatchRecordsITS.reserve(mITSWork.size() * mParams->maxMatchCandidates);
669  mTimer[SWPrepITS].Stop();
670 
671  return nITSClus > 0;
672 }

686 void MatchTPCITS::doMatching(int sec)
687 {
689  auto& cacheITS = mITSSectIndexCache[sec]; // array of cached ITS track indices for this sector
690  auto& cacheTPC = mTPCSectIndexCache[sec]; // array of cached ITS track indices for this sector
691  auto& timeStartTPC = mTPCTimeStart[sec]; // array of 1st TPC track with timeMax in ITS ROFrame
692  auto& timeStartITS = mITSTimeStart[sec];
693  int nTracksTPC = cacheTPC.size(), nTracksITS = cacheITS.size();
694  if (!nTracksTPC || !nTracksITS) {
695  if (mParams->verbosity > 0) {
696  LOG(info) << "Matchng sector " << sec << " : N tracks TPC:" << nTracksTPC << " ITS:" << nTracksITS << " in sector " << sec;
697  }
698  return;
699  }
700 
702  float maxTDriftSafe = tpcTimeBin2MUS(mNTPCBinsFullDrift + mParams->safeMarginTPCITSTimeBin + mTPCTimeEdgeTSafeMargin);
703  float vdErrT = tpcTimeBin2MUS(mZ2TPCBin * mParams->maxVDriftUncertainty);
704 
705  // get min ROFrame of ITS tracks currently in cache
706  auto minROFITS = mITSWork[cacheITS.front()].roFrame;
707 
708  if (minROFITS >= int(timeStartTPC.size())) {
709  LOG(info) << "ITS min ROFrame " << minROFITS << " exceeds all cached TPC track ROF eqiuvalent " << cacheTPC.size() - 1;
710  return;
711  }
712 
713  int nCheckTPCControl = 0, nCheckITSControl = 0, nMatchesControl = 0; // temporary
714  int idxMinTPC = timeStartTPC[minROFITS]; // index of 1st cached TPC track within cached ITS ROFrames
715  auto t2nbs = tpcTimeBin2MUS(mZ2TPCBin * mParams->tpcTimeICMatchingNSigma);
716  bool checkInteractionCandidates = mUseFT0 && mParams->validateMatchByFIT != MatchTPCITSParams::Disable;
717 
718  int itsROBin = 0;
719  for (int itpc = idxMinTPC; itpc < nTracksTPC; itpc++) {
720  auto& trefTPC = mTPCWork[cacheTPC[itpc]];
721  // estimate ITS 1st ROframe bin this track may match to: TPC track are sorted according to their
722  // timeMax, hence the timeMax - MaxmNTPCBinsFullDrift are non-decreasing
723  auto tmn = trefTPC.tBracket.getMax() - maxTDriftSafe;
724  itsROBin = mITSTriggered ? time2ITSROFrameTrig(tmn, itsROBin) : time2ITSROFrameCont(tmn);
725 
726  if (itsROBin >= int(timeStartITS.size())) { // time of TPC track exceeds the max time of ITS in the cache
727  break;
728  }
729  int iits0 = timeStartITS[itsROBin];
730  nCheckTPCControl++;
731  for (auto iits = iits0; iits < nTracksITS; iits++) {
732  auto& trefITS = mITSWork[cacheITS[iits]];
733  // compare if the ITS and TPC tracks may overlap in time
734  LOG(debug) << "TPC bracket: " << trefTPC.tBracket.asString() << " ITS bracket: " << trefITS.tBracket.asString() << " TPCtgl: " << trefTPC.getTgl() << " ITStgl: " << trefITS.getTgl();
735  if (trefTPC.tBracket < trefITS.tBracket) { // since TPC tracks are sorted in timeMax and ITS tracks are sorted in timeMin all following ITS tracks also will not match
736  break;
737  }
738  if (trefTPC.tBracket > trefITS.tBracket) { // its bracket precedes TPC bracket
739  continue;
740  }
741 
742  // is corrected TPC track time compatible with ITS ROF expressed
743  auto deltaT = (trefITS.getZ() - trefTPC.getZ()) * mTPCVDrift0Inv; // drift time difference corresponding to Z differences
744  auto timeCorr = trefTPC.getCorrectedTime(deltaT); // TPC time required to match to Z of ITS track
745  auto timeCorrErr = std::sqrt(trefITS.getSigmaZ2() + trefTPC.getSigmaZ2()) * t2nbs + mParams->safeMarginTimeCorrErr; // nsigma*error
746  if (mVDriftCalibOn) {
747  timeCorrErr += vdErrT * (250. - abs(trefITS.getZ())); // account for the extra error from TPC VDrift uncertainty
748  }
749  o2::math_utils::Bracketf_t trange(timeCorr - timeCorrErr, timeCorr + timeCorrErr);
750  LOG(debug) << "TPC range: " << trange.asString() << " ITS bracket: " << trefITS.tBracket.asString() << " DZ: " << (trefITS.getZ() - trefTPC.getZ()) << " DT: " << timeCorr;
751  if (trefITS.tBracket.isOutside(trange)) {
752  continue;
753  }
754  if (timeCorr < 0) { // RS TODO: similar check will be needed to other TF edge
755  if (timeCorr + mParams->tfEdgeTimeToleranceMUS < 0) {
756  //continue;
757  }
758  }
759 
760  nCheckITSControl++;
761  float chi2 = -1;
762  int rejFlag = compareTPCITSTracks(trefITS, trefTPC, chi2);
763 
764 #ifdef _ALLOW_DEBUG_TREES_
765  if (mDBGOut && ((rejFlag == Accept && isDebugFlag(MatchTreeAccOnly)) || isDebugFlag(MatchTreeAll))) {
766  fillTPCITSmatchTree(cacheITS[iits], cacheTPC[itpc], rejFlag, chi2, timeCorr);
767  }
768 #endif
769  /*
770  // RS: this might be dangerous for ITS tracks with different time coverages.
771  if (rejFlag == RejectOnTgl) {
772  // ITS tracks in each ROFrame are ordered in Tgl, hence if this check failed on Tgl check
773  // (i.e. tgl_its>tgl_tpc+tolerance), then all other ITS tracks in this ROFrame will also have tgl too large.
774  // Jump on the 1st ITS track of the next ROFrame
775  int rof = trefITS.roFrame;
776  bool stop = false;
777  do {
778  if (++rof >= int(timeStartITS.size())) {
779  stop = true;
780  break; // no more ITS ROFrames in cache
781  }
782  iits = timeStartITS[rof] - 1; // next track to be checked -1
783  } while (iits <= timeStartITS[trefITS.roFrame]); // skip empty bins
784  if (stop) {
785  break;
786  }
787  continue;
788  }
789  */
790  if (rejFlag != Accept) {
791  continue;
792  }
793  int matchedIC = MinusOne;
794  if (!isCosmics()) {
795  // validate by bunch filling scheme
796  if (mUseBCFilling) {
797  auto irBracket = tBracket2IRBracket(trange);
798  if (irBracket.isInvalid()) {
799  continue;
800  }
801  }
802  if (checkInteractionCandidates) {
803  // check if corrected TPC track time is compatible with any of interaction times
804  auto interactionRefs = mITSROFIntCandEntries[trefITS.roFrame]; // reference on interaction candidates compatible with this track
805  int nic = interactionRefs.getEntries();
806  if (nic) {
807  int idIC = interactionRefs.getFirstEntry(), maxIC = idIC + nic;
808  for (; idIC < maxIC; idIC++) {
809  auto cmp = mInteractions[idIC].tBracket.isOutside(trange);
810  if (cmp == o2::math_utils::Bracketf_t::Above) { // trange is above this interaction candidate, the following ones may match
811  continue;
812  }
813  if (cmp == o2::math_utils::Bracketf_t::Inside) {
814  matchedIC = idIC;
815  }
816  break; // we loop till 1st matching IC or the one above the trange (since IC are ordered, all others will be above too)
817  }
818  }
819  }
820  if (mParams->validateMatchByFIT == MatchTPCITSParams::Require && matchedIC == MinusOne) {
821  continue;
822  }
823  }
824  registerMatchRecordTPC(cacheITS[iits], cacheTPC[itpc], chi2, matchedIC); // register matching candidate
825  nMatchesControl++;
826  }
827  }
828  if (mParams->verbosity > 0) {
829  LOG(info) << "Match sector " << sec << " N tracks TPC:" << nTracksTPC << " ITS:" << nTracksITS
830  << " N TPC tracks checked: " << nCheckTPCControl << " (starting from " << idxMinTPC
831  << "), checks: " << nCheckITSControl << ", matches:" << nMatchesControl;
832  }
833 }

856 //______________________________________________
857 bool MatchTPCITS::registerMatchRecordTPC(int iITS, int iTPC, float chi2, int candIC)
858 {
861  auto& tTPC = mTPCWork[iTPC]; // get MatchRecord structure of this TPC track, create if none
862  if (tTPC.matchID < 0) { // no matches yet, just add new record
863  registerMatchRecordITS(iITS, iTPC, chi2, candIC); // register TPC track in the ITS records
864  tTPC.matchID = mMatchRecordsTPC.size(); // new record will be added in the end
865  mMatchRecordsTPC.emplace_back(iITS, chi2, MinusOne, candIC); // create new record with empty reference on next match
866  return true;
867  }
868 
869  int count = 0, nextID = tTPC.matchID, topID = MinusOne;
870  do {
871  auto& nextMatchRec = mMatchRecordsTPC[nextID];
872  count++;
873  if (!nextMatchRec.isBetter(chi2, candIC)) { // need to insert new record before nextMatchRec?
874  if (count < mParams->maxMatchCandidates) {
875  break; // will insert in front of nextID
876  } else { // max number of candidates reached, will overwrite the last one
877  suppressMatchRecordITS(nextMatchRec.partnerID, iTPC); // flag as disabled the overriden ITS match
878  registerMatchRecordITS(iITS, iTPC, chi2, candIC); // register TPC track entry in the ITS records
879  // reuse the record of suppressed ITS match to store better one
880  nextMatchRec.chi2 = chi2;
881  nextMatchRec.matchedIC = candIC;
882  nextMatchRec.partnerID = iITS;
883  return true;
884  }
885  }
886  topID = nextID; // check next match record
887  nextID = nextMatchRec.nextRecID;
888  } while (nextID > MinusOne);
889 
890  // if count == mParams->maxMatchCandidates, the max number of candidates was already reached, and the
891  // new candidated was either discarded (if its chi2 is worst one) or has overwritten worst
892  // existing candidate. Otherwise, we need to add new entry
893  if (count < mParams->maxMatchCandidates) {
894  if (topID < 0) { // the new match is top candidate
895  topID = tTPC.matchID = mMatchRecordsTPC.size(); // register new record as top one
896  } else { // there are better candidates
897  topID = mMatchRecordsTPC[topID].nextRecID = mMatchRecordsTPC.size(); // register to his parent
898  }
899  // nextID==-1 will mean that the while loop run over all candidates->the new one is the worst (goes to the end)
900  registerMatchRecordITS(iITS, iTPC, chi2, candIC); // register TPC track in the ITS records
901  mMatchRecordsTPC.emplace_back(iITS, chi2, nextID, candIC); // create new record with empty reference on next match
902  // make sure that after addition the number of candidates don't exceed allowed number
903  count++;
904  while (nextID > MinusOne) {
905  if (count > mParams->maxMatchCandidates) {
906  suppressMatchRecordITS(mMatchRecordsTPC[nextID].partnerID, iTPC);
907  // exclude nextID record, w/o changing topID (which becomes the last record)
908  nextID = mMatchRecordsTPC[topID].nextRecID = mMatchRecordsTPC[nextID].nextRecID;
909  continue;
910  }
911  count++;
912  topID = nextID;
913  nextID = mMatchRecordsTPC[nextID].nextRecID;
914  }
915  return true;
916  } else {
917  return false; // unless nextID was assigned OverrideExisting, new candidate was discarded
918  }
919 }

922 void MatchTPCITS::registerMatchRecordITS(int iITS, int iTPC, float chi2, int candIC)
923 {
925  auto& tITS = mITSWork[iITS];
926  int idnew = mMatchRecordsITS.size();
927  auto& newRecord = mMatchRecordsITS.emplace_back(iTPC, chi2, MinusOne, candIC); // associate iTPC with this record

929  tITS.matchID = idnew;
930  return;
931  }

941  } else {
942  mMatchRecordsITS[topID].nextRecID = idnew; // new record will follow existing better one
943  }
944  return;
945  }
946  topID = nextRecord;
947  nextRecord = mMatchRecordsITS[nextRecord].nextRecID;
948  } while (nextRecord > MinusOne);
949 
950  // if we reached here, the new record should be added in the end
951  mMatchRecordsITS[topID].nextRecID = idnew; // register new link
952 }

954 //______________________________________________
955 int MatchTPCITS::compareTPCITSTracks(const TrackLocITS& tITS, const TrackLocTPC& tTPC, float& chi2) const
956 {
958  chi2 = -1.f;
959  int rejFlag = Accept;
960  float diff; // make rough check differences and their nsigmas
961 
962  // start with check on Tgl, since rjection on it will allow to profit from sorting
963  diff = tITS.getParam(o2::track::kTgl) - tTPC.getParam(o2::track::kTgl);
964  if ((rejFlag = roughCheckDif(diff, mParams->crudeAbsDiffCut[o2::track::kTgl], RejectOnTgl))) {
965  return rejFlag;
966  }
967  auto err2Tgl = tITS.getDiagError2(o2::track::kTgl) + tTPC.getDiagError2(o2::track::kTgl);
968  if (mVDriftCalibOn) {
969  auto addErr = tITS.getParam(o2::track::kTgl) * mParams->maxVDriftUncertainty;
970  err2Tgl += addErr * addErr; // account for VDrift uncertainty
971  }
972  diff *= diff / err2Tgl;
973  if ((rejFlag = roughCheckDif(diff, mParams->crudeNSigma2Cut[o2::track::kTgl], RejectOnTgl + NSigmaShift))) {
974  return rejFlag;
975  }
976  diff = tITS.getParam(o2::track::kY) - tTPC.getParam(o2::track::kY);
977  if ((rejFlag = roughCheckDif(diff, mParams->crudeAbsDiffCut[o2::track::kY], RejectOnY))) {
978  return rejFlag;
979  }
980  diff *= diff / (tITS.getDiagError2(o2::track::kY) + tTPC.getDiagError2(o2::track::kY));
981  if ((rejFlag = roughCheckDif(diff, mParams->crudeNSigma2Cut[o2::track::kY], RejectOnY + NSigmaShift))) {
982  return rejFlag;
983  }
984 
985  if (tTPC.constraint == TrackLocTPC::Constrained) { // in continuous only constrained tracks can be compared in Z
986  diff = tITS.getParam(o2::track::kZ) - tTPC.getParam(o2::track::kZ);
987  if ((rejFlag = roughCheckDif(diff, mParams->crudeAbsDiffCut[o2::track::kZ], RejectOnZ))) {
988  return rejFlag;
989  }
990  diff *= diff / (tITS.getDiagError2(o2::track::kZ) + tTPC.getDiagError2(o2::track::kZ));
991  if ((rejFlag = roughCheckDif(diff, mParams->crudeNSigma2Cut[o2::track::kZ], RejectOnZ + NSigmaShift))) {
992  return rejFlag;
993  }
994  }
995 
996  diff = tITS.getParam(o2::track::kSnp) - tTPC.getParam(o2::track::kSnp);
997  if ((rejFlag = roughCheckDif(diff, mParams->crudeAbsDiffCut[o2::track::kSnp], RejectOnSnp))) {
998  return rejFlag;
999  }
1000  diff *= diff / (tITS.getDiagError2(o2::track::kSnp) + tTPC.getDiagError2(o2::track::kSnp));
1001  if ((rejFlag = roughCheckDif(diff, mParams->crudeNSigma2Cut[o2::track::kSnp], RejectOnSnp + NSigmaShift))) {
1002  return rejFlag;
1003  }
1004 
1005  diff = tITS.getParam(o2::track::kQ2Pt) - tTPC.getParam(o2::track::kQ2Pt);
1006  if ((rejFlag = roughCheckDif(diff, mParams->crudeAbsDiffCut[o2::track::kQ2Pt], RejectOnQ2Pt))) {
1007  return rejFlag;
1008  }
1009  diff *= diff / (tITS.getDiagError2(o2::track::kQ2Pt) + tTPC.getDiagError2(o2::track::kQ2Pt));
1010  if ((rejFlag = roughCheckDif(diff, mParams->crudeNSigma2Cut[o2::track::kQ2Pt], RejectOnQ2Pt + NSigmaShift))) {
1011  return rejFlag;
1012  }
1013  // calculate mutual chi2 excluding Z in continuos mode
1014  chi2 = getPredictedChi2NoZ(tITS, tTPC);
1015  if (chi2 > mParams->cutMatchingChi2) {
1016  return RejectOnChi2;
1017  }
1018 
1019  return Accept;
1020 }

1024 {

1793  !propagator->propagateToX(seedC, xTgt, true, MaxSnp, 2., mUseMatCorrFlag)) { // Bz-propagation only in ITS
1794  return -1;
1795  }

1797  float errZ = std::sqrt(seedC.getSigmaZ2());

1814  float phi = std::atan2(yCurr, xCurr); // RS: TODO : can we use fast atan2 here?
