60 template <typename value_T>
61 class Rotation2D
62 {
63  //
64  // class to perform rotation of 3D (around Z) and 2D points
65 
66  public:
67  using value_t = value_T;
68 
69  Rotation2D() = default;
70  Rotation2D(value_t cs, value_t sn) : mCos(cs), mSin(sn) {}
71  Rotation2D(value_t phiZ) : mCos(cos(phiZ)), mSin(sin(phiZ)) {}
72  ~Rotation2D() = default;
73  Rotation2D(const Rotation2D& src) = default;
74  Rotation2D(Rotation2D&& src) = default;
75  Rotation2D& operator=(const Rotation2D& src) = default;
76  Rotation2D& operator=(Rotation2D&& src) = default;
77 
78  void set(value_t phiZ)
79  {
80  mCos = cos(phiZ);
81  mSin = sin(phiZ);
82  }
83 
84  void set(value_t cs, value_t sn)
85  {
86  mCos = cs;
87  mSin = sn;
88  }
89 
90  void getComponents(value_t& cs, value_t& sn) const
91  {
92  cs = mCos;
93  sn = mSin;
94  }
95 
96  template <typename T>
97  Point3D<T> operator()(const Point3D<T>& v) const
98  { // local->master
99  return Point3D<T>(v.X() * mCos - v.Y() * mSin, v.X() * mSin + v.Y() * mCos, v.Z());
100  }
101 
102  template <typename T>
103  Point3D<T> operator^(const Point3D<T>& v) const
104  { // master->local
105  return Point3D<T>(v.X() * mCos + v.Y() * mSin, -v.X() * mSin + v.Y() * mCos, v.Z());
106  }
107 
108  template <typename T>
109  Vector3D<T> operator()(const Vector3D<T>& v) const
110  { // local->master
111  return Vector3D<T>(v.X() * mCos - v.Y() * mSin, v.X() * mSin + v.Y() * mCos, v.Z());
112  }
113 
114  template <typename T>
115  Vector3D<T> operator^(const Vector3D<T>& v) const
116  { // master->local
117  return Vector3D<T>(v.X() * mCos + v.Y() * mSin, -v.X() * mSin + v.Y() * mCos, v.Z());
118  }
119 
120  template <typename T>
121  Point2D<T> operator()(const Point2D<T>& v) const
122  { // local->master
123  return Point2D<T>(v.X() * mCos - v.Y() * mSin, v.X() * mSin + v.Y() * mCos);
124  }
125 
126  template <typename T>
127  Point2D<T> operator^(const Point2D<T>& v) const
128  { // master->local
129  return Point2D<T>(v.X() * mCos + v.Y() * mSin, -v.X() * mSin + v.Y() * mCos);
130  }
131 
132  template <typename T>
133  Vector2D<T> operator()(const Vector2D<T>& v) const
134  { // local->master
135  return Vector2D<T>(v.X() * mCos - v.Y() * mSin, v.X() * mSin + v.Y() * mCos);
136  }
137 
138  template <typename T>
139  Vector2D<T> operator^(const Vector2D<T>& v) const
140  { // master->local
141  return Vector2D<T>(v.X() * mCos + v.Y() * mSin, -v.X() * mSin + v.Y() * mCos);
142  }
143 
144  private:
145  value_t mCos = 1;
146  value_t mSin = 0;

156 class Transform3D : public ROOT::Math::Transform3D
157 {
158  //
159  // Class to perform geom.transformations (rotation and displacements only) in
160  // double precision over the cartesian points and vectors (float or double).
161  // Adds to the base ROOT::Math::Transform3D<double> class a convertor from
162  // TGeoMatrix.
163  // To be used instead of TGeoHMatrix for all transformations of hits,
164  // clusters etc.
165  //
166 
167  public:
168  Transform3D() = default;
169  Transform3D(const TGeoMatrix& m);
170  ~Transform3D() = default;
171 
172  // inherit assignment operators of the base class
173  using ROOT::Math::Transform3D::operator=;
174 
175  // to avoid conflict between the base Transform3D(const ForeignMatrix & m) and
176  // Transform3D(const TGeoMatrix &m) constructors we cannot inherit base c-tors,
177  // therefore we redefine them here
178  Transform3D(const ROOT::Math::Rotation3D& r, const Vector& v) : ROOT::Math::Transform3D(r, v) {}
179  Transform3D(const ROOT::Math::Rotation3D& r, const ROOT::Math::Translation3D& t) : ROOT::Math::Transform3D(r, t) {}
180  template <class IT>
181  Transform3D(IT begin, IT end) : ROOT::Math::Transform3D(begin, end)
182  {
183  }
184 
185  // conversion operator to TGeoHMatrix
186  operator TGeoHMatrix&() const
187  {
188  static TGeoHMatrix tmp;
189  double rot[9], tra[3];
190  GetComponents(rot[0], rot[1], rot[2], tra[0], rot[3], rot[4], rot[5], tra[1], rot[6], rot[7], rot[8], tra[2]);
191  tmp.SetRotation(rot);
192  tmp.SetTranslation(tra);
193  return tmp;
194  }
195 
196  void set(const TGeoMatrix& m); // set parameters from TGeoMatrix
197 
198  using ROOT::Math::Transform3D::operator();
199  // the local->master transformation for points and vectors can be
200  // done in operator form (inherited from base Transform3D) as
201  // Point3D pnt;
202  // Transform3D trans;
203  // auto pntTr0 = trans(pnt); // 1st version
204  // auto pntTr1 = trans*pnt; // 2nd version
205  //
206  // For the inverse transform we define our own operator^
207 
208  template <typename T>
209  Point3D<T> operator^(const Point3D<T>& p) const
210  { // master->local
211  return ApplyInverse(p);
212  }
213 
214  template <typename T>
215  Vector3D<T> operator^(const Vector3D<T>& v) const
216  { // local->master
217  return ApplyInverse(v);
218  }
219 
220  // TGeoHMatrix-like aliases
221  template <typename T>
222  void LocalToMaster(const Point3D<T>& loc, Point3D<T>& mst) const

225  }
