28 void Digits2Raw::processDigits(const std::string& outDir, const std::string& fileDigitsName)
29 {
30  auto& sopt = ZDCSimParam::Instance();
31  mIsContinuous = sopt.continuous;
32 
33  if (!mModuleConfig) {
34  LOG(fatal) << "Missing ModuleConfig configuration object";
35  return;
36  }
37 
38  if (!mSimCondition) {
39  LOG(fatal) << "Missing SimCondition configuration object";
40  return;
41  }
42 
43  if (mNEmpty < 0) {
44  LOG(fatal) << "Bunch crossing map is not initialized";
45  return;
46  }
47 
48  if (mNEmpty == 0) {
49  LOG(warning) << "Bunch crossing map has zero clean empty bunches";
50  }
51 
52  setTriggerMask();
53 
54  std::string outd = outDir;
55  if (outd.back() != '/') {
56  outd += '/';
57  }
58 
59  mLinkID = uint32_t(0);
60  mCruID = uint16_t(0);
61  mFLPID = uint16_t(0);
62  mEndPointID = uint32_t(0);
63  // TODO: assign FeeID from configuration object
64  for (int ilink = 0; ilink < NLinks; ilink++) {
65  uint64_t FeeID = uint64_t(ilink);
66  std::string outFileLink = o2::utils::Str::concat_string(outDir, "/", "ZDC");
67  if (mFileFor != "all") {
68  outFileLink += fmt::format("_{}", mFLP);
69  if (mFileFor != "flp") {
70  outFileLink += fmt::format("_cru{}_{}", mCruID, mEndPointID);
71  if (mFileFor != "cru") {
72  outFileLink += fmt::format("_lnk{}_feeid{}", ilink, FeeID);
73  if (mFileFor != "link") {
74  LOG(fatal) << "Not supported output file splitting: " << mFileFor;
75  throw std::runtime_error("invalid option provided for file grouping");
76  }
77  }
78  }
79  }
80  outFileLink += ".raw";
81  mWriter.registerLink(FeeID, mCruID, mLinkID, mEndPointID, outFileLink);
82  }
83 
84  std::unique_ptr<TFile> digiFile(TFile::Open(fileDigitsName.c_str()));
85  if (!digiFile || digiFile->IsZombie()) {
86  LOG(fatal) << "Failed to open input digits file " << fileDigitsName;
87  return;
88  }
89 
90  TTree* digiTree = (TTree*)digiFile->Get("o2sim");
91  if (!digiTree) {
92  LOG(fatal) << "Failed to get digits tree";
93  return;
94  }
95 
96  if (digiTree->GetBranch("ZDCDigitBC")) {
97  digiTree->SetBranchAddress("ZDCDigitBC", &mzdcBCDataPtr);
98  } else {
99  LOG(fatal) << "Branch ZDCDigitBC is missing";
100  return;
101  }
102 
103  if (digiTree->GetBranch("ZDCDigitCh")) {
104  digiTree->SetBranchAddress("ZDCDigitCh", &mzdcChDataPtr);
105  } else {
106  LOG(fatal) << "Branch ZDCDigitCh is missing";
107  return;
108  }
109 
110  if (digiTree->GetBranch("ZDCDigitOrbit")) {
111  digiTree->SetBranchAddress("ZDCDigitOrbit", &mzdcPedDataPtr);
112  } else {
113  LOG(fatal) << "Branch ZDCDigitOrbit is missing";
114  return;
115  }
116 
117  if (digiTree->GetBranchStatus("ZDCDigitLabels")) {
118  digiTree->SetBranchStatus("ZDCDigitLabel*", 0);
119  }
120 
121  for (int ient = 0; ient < digiTree->GetEntries(); ient++) {
122  digiTree->GetEntry(ient);
123  mNbc = mzdcBCData.size();
124  LOG(info) << "Entry " << ient << " : " << mNbc << " BCs stored";
125  for (int ibc = 0; ibc < mNbc; ibc++) {
126  mBCD = mzdcBCData[ibc];
127  convertDigits(ibc);
128  writeDigits();
129  // Detect last event or orbit change and insert last bunch
130  if (ibc == (mNbc - 1)) {
131  // For last event we need to close last orbit (if it is needed)
132  if (mzdcBCData[ibc].ir.bc != 3563) {
133  insertLastBunch(ibc, mzdcBCData[ibc].ir.orbit);
134  writeDigits();
135  }
136  } else {
137  auto this_orbit = mzdcBCData[ibc].ir.orbit;
138  auto next_orbit = mzdcBCData[ibc + 1].ir.orbit;
139  // If current bunch is last bunch in the orbit we don't insert it again
140  if (mzdcBCData[ibc].ir.bc == 3563) {
141  this_orbit = this_orbit + 1;
142  }
143  // We may need to insert more than one orbit
144  for (auto orbit = this_orbit; orbit < next_orbit; orbit++) {
145  insertLastBunch(ibc, orbit);
146  writeDigits();
147  }
148  }
149  }
150  }
151  digiFile->Close();
152 }

155 void Digits2Raw::setTriggerMask()
156 {
157  mTriggerMask = 0;
158  mPrintTriggerMask = "";
159  for (int32_t im = 0; im < NModules; im++) {
160  if (im > 0) {
161  mPrintTriggerMask += " ";
162  }
163  mPrintTriggerMask += std::to_string(im);
164  mPrintTriggerMask += "[";
165  for (uint32_t ic = 0; ic < NChPerModule; ic++) {
166  if (mModuleConfig->modules[im].trigChannel[ic]) {
167  uint32_t tmask = 0x1 << (im * NChPerModule + ic);
168  mTriggerMask = mTriggerMask | tmask;
169  mPrintTriggerMask += "T";
170  } else {
171  mPrintTriggerMask += " ";
172  }
173  }
174  mPrintTriggerMask += "]";
175  uint32_t mytmask = mTriggerMask >> (im * NChPerModule);
176  printf("Trigger mask for module %d 0123 %s%s%s%s\n", im,
177  mytmask & 0x1 ? "T" : "N",
178  mytmask & 0x2 ? "T" : "N",
179  mytmask & 0x4 ? "T" : "N",
180  mytmask & 0x8 ? "T" : "N");
181  }
182  printf("trigger_mask=0x%08x %s\n", mTriggerMask, mPrintTriggerMask.data());
183 }

199 //______________________________________________________________________________
200 inline void Digits2Raw::updatePedestalReference(int bc)
201 {
202  // Compute or update baseline reference
203  // In the last BC we copy what is stored in the digits
204  if (bc == 3563) {
205  int io = 0;
206  for (; io < mzdcPedData.size(); io++) {
207  uint32_t orbit = mBCD.ir.orbit;
208  if (orbit == mzdcPedData[io].ir.orbit) {
209  break;
210  }
211  }
212  if (io == mzdcPedData.size()) {
213  LOG(fatal) << "Cannot find orbit";
214  }
215 
216  for (int32_t im = 0; im < NModules; im++) {
217  for (int32_t ic = 0; ic < NChPerModule; ic++) {
218  // Identify connected channel
219  auto id = mModuleConfig->modules[im].channelID[ic];
220  double myped = mzdcPedData[io].data[id] + 32768.;
221  if (myped < 0) {
222  myped = 0;
223  }
224  if (myped > 65535) {
225  myped = 65535;
226  }
227  mPed[im][ic] = myped;
228  }
229  }
230  } else if (mEmpty[bc] > 0 && mEmpty[bc] != mLastNEmpty) {
231  // For the preceding bunch crossing we make-up the fields in a random walk
232  // fashion like in the hardware. The result however cannot be coherent with
233  // what is stored in the last bunch
234  for (int32_t im = 0; im < NModules; im++) {
235  for (int32_t ic = 0; ic < NChPerModule; ic++) {
236  // Identify connected channel
237  auto id = mModuleConfig->modules[im].channelID[ic];
238  auto base_m = mSimCondition->channels[id].pedestal; // Average pedestal
239  auto base_s = mSimCondition->channels[id].pedestalFluct; // Baseline oscillations
240  auto base_n = mSimCondition->channels[id].pedestalNoise; // Electronic noise
241  double deltan = mEmpty[bc] - mLastNEmpty;
242  // We assume to have a fluctuation every two bunch crossings
243  // Will need to tune this parameter
244  double k = 2.;
245  mSumPed[im][ic] += gRandom->Gaus(12. * deltan * base_m, 12. * k * base_s * TMath::Sqrt(deltan / k));
246  // Adding in quadrature the RMS of pedestal electronic noise
247  mSumPed[im][ic] += gRandom->Gaus(0, base_n * TMath::Sqrt(12. * deltan));
248  double myped = TMath::Nint(8. * mSumPed[im][ic] / double(mEmpty[bc]) / 12. + 32768);
249  if (myped < 0) {
250  myped = 0;
251  }
252  if (myped > 65535) {
253  myped = 65535;
254  }
255  mPed[im][ic] = myped;
256  }
257  }
258  mLastNEmpty = mEmpty[bc];
259  }
260 }

263 inline void Digits2Raw::resetOutputStructure(uint16_t bc, uint32_t orbit, bool is_dummy)
264 {
265  // Increment scalers and reset output structure
266  for (uint32_t im = 0; im < NModules; im++) {
267  for (uint32_t ic = 0; ic < NChPerModule; ic++) {
268  // Fixed words
269  mZDC.data[im][ic].w[0][0] = Id_w0;
270  mZDC.data[im][ic].w[0][1] = 0;
271  mZDC.data[im][ic].w[0][2] = 0;
272  mZDC.data[im][ic].w[0][3] = 0;
273  mZDC.data[im][ic].w[1][0] = Id_w1;
274  mZDC.data[im][ic].w[1][1] = 0;
275  mZDC.data[im][ic].w[1][2] = 0;
276  mZDC.data[im][ic].w[1][3] = 0;
277  mZDC.data[im][ic].w[2][0] = Id_w2;
278  mZDC.data[im][ic].w[2][1] = 0;
279  mZDC.data[im][ic].w[2][2] = 0;
280  mZDC.data[im][ic].w[2][3] = 0;
281  // Module and channel numbers
282  mZDC.data[im][ic].f.board = im;
283  mZDC.data[im][ic].f.ch = ic;
284  // Orbit and bunch crossing
285  mZDC.data[im][ic].f.orbit = orbit;
286  mZDC.data[im][ic].f.bc = bc;
287  // If channel is hit in current bunch crossing
288  if (!is_dummy) {
289  if (mBCD.triggers & (0x1 << (im * NChPerModule + ic))) {
290  mScalers[im][ic]++; // increment scalers
291  mZDC.data[im][ic].f.Hit = 1; // flag bunch crossing
292  }
293  }
294  mZDC.data[im][ic].f.hits = mScalers[im][ic];
295  mZDC.data[im][ic].f.offset = mPed[im][ic];
296  }
297  }
298 }

301 inline void Digits2Raw::assignTriggerBits(int ibc, uint16_t bc, uint32_t orbit, bool is_dummy)
302 {
303  // Triggers refer to the HW trigger conditions (32 possible channels)
304  // Autotrigger, current bunch crossing
305  ModuleTriggerMapData triggers;
306  // Autotrigger and ALICE trigger bits are zero for a dummy bunch crossing
307  if (!is_dummy) {
308  for (uint32_t im = 0; im < NModules; im++) {
309  triggers.w = mzdcBCData[ibc].moduleTriggers[im];
310  for (uint32_t ic = 0; ic < NChPerModule; ic++) {
311  mZDC.data[im][ic].f.Alice_0 = triggers.f.Alice_0;
312  mZDC.data[im][ic].f.Alice_1 = triggers.f.Alice_1;
313  mZDC.data[im][ic].f.Alice_2 = triggers.f.Alice_2;
314  mZDC.data[im][ic].f.Alice_3 = triggers.f.Alice_3;
315  mZDC.data[im][ic].f.Auto_m = triggers.f.Auto_m;
316  mZDC.data[im][ic].f.Auto_0 = triggers.f.Auto_0;
317  mZDC.data[im][ic].f.Auto_1 = triggers.f.Auto_1;
318  mZDC.data[im][ic].f.Auto_2 = triggers.f.Auto_2;
319  mZDC.data[im][ic].f.Auto_3 = triggers.f.Auto_3;
320  }
321  }
322  }

326 void Digits2Raw::insertLastBunch(int ibc, uint32_t orbit)
327 {
328 
329  // Orbit and bunch crossing identifiers
330  uint16_t bc = 3563;
331 
332  // Reset scalers at orbit change
333  if (orbit != mLastOrbit) {
334  resetSums(orbit);
335  }
336 
337  updatePedestalReference(bc);
338 
339  // Dummy bunch -> Do not increment scalers but reset output structure
340  resetOutputStructure(bc, orbit, true);
341 
342  // Compute autotrigger bits and assign ALICE trigger bits
343  assignTriggerBits(ibc, bc, orbit, true);
344 
345  // Insert payload for all channels
346  for (int32_t im = 0; im < NModules; im++) {
347  for (uint32_t ic = 0; ic < NChPerModule; ic++) {
348  if (mModuleConfig->modules[im].readChannel[ic]) {
349  auto id = mModuleConfig->modules[im].channelID[ic];
350  auto base_m = mSimCondition->channels[id].pedestal; // Average pedestal
351  auto base_s = mSimCondition->channels[id].pedestalFluct; // Baseline oscillations
352  auto base_n = mSimCondition->channels[id].pedestalNoise; // Electronic noise
353  double base = gRandom->Gaus(base_m, base_s);
354  double val = base + gRandom->Gaus(0, base_n);
355  mZDC.data[im][ic].f.s00 = val < ADCMax ? (val > ADCMin ? val : ADCMin) : ADCMax;
356  val = base + gRandom->Gaus(0, base_n);
357  mZDC.data[im][ic].f.s01 = val < ADCMax ? (val > ADCMin ? val : ADCMin) : ADCMax;
358  val = base + gRandom->Gaus(0, base_n);
359  mZDC.data[im][ic].f.s02 = val < ADCMax ? (val > ADCMin ? val : ADCMin) : ADCMax;
360  val = base + gRandom->Gaus(0, base_n);
361  mZDC.data[im][ic].f.s03 = val < ADCMax ? (val > ADCMin ? val : ADCMin) : ADCMax;
362  val = base + gRandom->Gaus(0, base_n);
363  mZDC.data[im][ic].f.s04 = val < ADCMax ? (val > ADCMin ? val : ADCMin) : ADCMax;
364  val = base + gRandom->Gaus(0, base_n);
365  mZDC.data[im][ic].f.s05 = val < ADCMax ? (val > ADCMin ? val : ADCMin) : ADCMax;
366  val = base + gRandom->Gaus(0, base_n);
367  mZDC.data[im][ic].f.s06 = val < ADCMax ? (val > ADCMin ? val : ADCMin) : ADCMax;
368  val = base + gRandom->Gaus(0, base_n);
369  mZDC.data[im][ic].f.s07 = val < ADCMax ? (val > ADCMin ? val : ADCMin) : ADCMax;
370  val = base + gRandom->Gaus(0, base_n);
371  mZDC.data[im][ic].f.s08 = val < ADCMax ? (val > ADCMin ? val : ADCMin) : ADCMax;
372  val = base + gRandom->Gaus(0, base_n);
373  mZDC.data[im][ic].f.s09 = val < ADCMax ? (val > ADCMin ? val : ADCMin) : ADCMax;
374  val = base + gRandom->Gaus(0, base_n);
375  mZDC.data[im][ic].f.s10 = val < ADCMax ? (val > ADCMin ? val : ADCMin) : ADCMax;
376  val = base + gRandom->Gaus(0, base_n);
377  mZDC.data[im][ic].f.s11 = val < ADCMax ? (val > ADCMin ? val : ADCMin) : ADCMax;
378  }
379  }
380  }
381 } // insertLastBunch

383 //______________________________________________________________________________
384 void Digits2Raw::convertDigits(int ibc)
385 {
386 
387  // Orbit and bunch crossing identifiers
388  uint16_t bc = mBCD.ir.bc;
389  uint32_t orbit = mBCD.ir.orbit;
390 
391  // Reset scalers at orbit change
392  if (orbit != mLastOrbit) {
393  resetSums(orbit);
394  }
395 
396  updatePedestalReference(bc);
397 
398  // Not a dummy bunch -> Reset output structure and eventually flag hits and increment scalers
399  resetOutputStructure(bc, orbit, false);
400 
401  // Compute autotrigger bits and assign ALICE trigger bits
402  assignTriggerBits(ibc, bc, orbit, false);
403 
404  if (mVerbosity > 0) {
405  mBCD.print(mTriggerMask);
406  }
407 
408  int chEnt = mBCD.ref.getFirstEntry();
409  for (int ic = 0; ic < mBCD.ref.getEntries(); ic++) {
410  const auto& chd = mzdcChData[chEnt++];
411  if (mVerbosity > 0) {
412  chd.print();
413  }
414  uint16_t bc = mBCD.ir.bc;
415  uint32_t orbit = mBCD.ir.orbit;
416  // Look for channel ID in digits and store channel (just one copy in output)
417  // This is a limitation of software but we are not supposed to acquire the
418  // same signal twice anyway
419  for (int32_t im = 0; im < NModules; im++) {
420  for (uint32_t ic = 0; ic < NChPerModule; ic++) {
421  if (mModuleConfig->modules[im].channelID[ic] == chd.id &&
422  mModuleConfig->modules[im].readChannel[ic]) {
423  int32_t is = 0;
424  mZDC.data[im][ic].f.s00 = chd.data[is++];
425  mZDC.data[im][ic].f.s01 = chd.data[is++];
426  mZDC.data[im][ic].f.s02 = chd.data[is++];
427  mZDC.data[im][ic].f.s03 = chd.data[is++];
428  mZDC.data[im][ic].f.s04 = chd.data[is++];
429  mZDC.data[im][ic].f.s05 = chd.data[is++];
430  mZDC.data[im][ic].f.s06 = chd.data[is++];
431  mZDC.data[im][ic].f.s07 = chd.data[is++];
432  mZDC.data[im][ic].f.s08 = chd.data[is++];
433  mZDC.data[im][ic].f.s09 = chd.data[is++];
434  mZDC.data[im][ic].f.s10 = chd.data[is++];
435  mZDC.data[im][ic].f.s11 = chd.data[is++];
436  break;
437  }
438  }
439  }
440  }

443 //______________________________________________________________________________
444 void Digits2Raw::writeDigits()
445 {
446  constexpr static int data_size = sizeof(uint32_t) * NWPerGBTW;
447  constexpr static gsl::span<char> empty;
448  // Local interaction record (true and empty bunches)
449  o2::InteractionRecord ir(mZDC.data[0][0].f.bc, mZDC.data[0][0].f.orbit);
450  for (uint32_t im = 0; im < o2::zdc::NModules; im++) {
451  // Check if module has been filled with data
452  // N.B. All channels are initialized if module is supposed to be readout
453  // Trigger bits are the same for all the channels connected to a module
454  bool TM = mZDC.data[im][0].f.Auto_m;
455  bool T0 = mZDC.data[im][0].f.Auto_0;
456  bool T1 = mZDC.data[im][0].f.Auto_1;
457  bool T2 = mZDC.data[im][0].f.Auto_2;
458  bool T3 = mZDC.data[im][0].f.Auto_3;
459  bool A0 = mZDC.data[im][0].f.Alice_0;
460  bool A1 = mZDC.data[im][0].f.Alice_1;
461  bool A2 = mZDC.data[im][0].f.Alice_2;
462  bool A3 = mZDC.data[im][0].f.Alice_3;
463  bool tcond_continuous = T0 || T1;
464  bool tcond_triggered = A0 || A1 || (A2 && (T0 || TM)) || (A3 && T0);
465  bool tcond_last = mZDC.data[im][0].f.bc == 3563;
466  // Condition to write GBT data
467  bool addedChData[NChPerModule] = {false, false, false, false};
468  if (tcond_triggered || (mIsContinuous && tcond_continuous) || (mZDC.data[im][0].f.bc == 3563)) {
469  for (uint32_t ic = 0; ic < o2::zdc::NChPerModule; ic++) {
470  uint64_t FeeID = 2 * im + ic / 2;
471  if (mModuleConfig->modules[im].readChannel[ic]) {
472  for (int32_t iw = 0; iw < o2::zdc::NWPerBc; iw++) {
473  gsl::span<char> payload{reinterpret_cast<char*>(&mZDC.data[im][ic].w[iw][0]), data_size};
474  mWriter.addData(FeeID, mCruID, mLinkID, mEndPointID, ir, payload);
475  }
476  addedChData[ic] = true;
477  }
478  }
479  }
480  // All links are registered, we add explicitly zero payload
481  if (addedChData[0] == false && addedChData[1] == false) {
482  uint64_t FeeID = 2 * im;
483  mWriter.addData(FeeID, mCruID, mLinkID, mEndPointID, ir, empty);
484  }
485  if (addedChData[2] == false && addedChData[3] == false) {
486  uint64_t FeeID = 2 * im + 1;
487  mWriter.addData(FeeID, mCruID, mLinkID, mEndPointID, ir, empty);
488  }
489  if (mVerbosity > 1) {
490  if (tcond_continuous) {
491  printf("M%d Cont. T0=%d || T1=%d\n", im, T0, T1);
492  }
493  if (tcond_triggered) {
494  printf("M%d Trig. %s A0=%d || A1=%d || (A2=%d && (T0=%d || TM=%d))=%d || (A3=%d && T0=%d )=%d\n", im, mIsContinuous ? "CM" : "TM", A0, A1, A2, T0, TM, A2 && (T0 || TM), A3, T0, A3 && T0);
495  }
496  if (mZDC.data[im][0].f.bc == 3563) {
497  printf("M%d is last BC\n", im);
498  }
499  if (tcond_triggered || (mIsContinuous && tcond_continuous) || (mZDC.data[im][0].f.bc == 3563)) {
500  for (uint32_t ic = 0; ic < o2::zdc::NChPerModule; ic++) {
501  if (mModuleConfig->modules[im].readChannel[ic]) {
502  for (int32_t iw = 0; iw < o2::zdc::NWPerBc; iw++) {
503  print_gbt_word(&mZDC.data[im][ic].w[iw][0], mModuleConfig);
504  }
505  }
506  }
507  } else {

604 void Digits2Raw::emptyBunches(std::bitset<3564>& bunchPattern)
605 {
606  const int LHCMaxBunches = o2::constants::lhc::LHCMaxBunches;
607  mNEmpty = 0;
608  for (int32_t ib = 0; ib < LHCMaxBunches; ib++) {
609  int32_t mb = (ib + 31) % LHCMaxBunches; // beam gas from back of calorimeter (31 bc earlier than a colliding bunch)
610  int32_t m1 = (ib + 1) % LHCMaxBunches; // next bunch is colliding (position -1)
611  int32_t p1 = (ib - 1 + LHCMaxBunches) % LHCMaxBunches; // current bc is 1 bc after colliding (position +1)
612  int32_t p2 = (ib - 2 + LHCMaxBunches) % LHCMaxBunches; // current bc is 2 bc after colliding
613  int32_t p3 = (ib - 3 + LHCMaxBunches) % LHCMaxBunches; // current bc is 3 bc after colliding
614  if (bunchPattern[mb] || bunchPattern[m1] || bunchPattern[ib] || bunchPattern[p1] || bunchPattern[p2] || bunchPattern[p3]) {
615  mEmpty[ib] = mNEmpty;
616  } else {
617  mNEmpty++;
618  mEmpty[ib] = mNEmpty;
619  }
620  }
621  LOG(info) << "There are " << mNEmpty << " clean empty bunches";
622 }
