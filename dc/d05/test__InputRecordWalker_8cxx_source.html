35 // simple helper struct to keep the InputRecord and ownership of messages
36 struct DataSet {
37  // not nice with the double vector but for quick unit test ok
38  using MessageSet = std::vector<std::unique_ptr<std::vector<char>>>;
39  using TaggedSet = std::pair<o2::header::DataOrigin, MessageSet>;
40  using Messages = std::vector<TaggedSet>;
41  using CheckType = std::vector<std::string>;
42  DataSet(std::vector<InputRoute>&& s, Messages&& m, CheckType&& v, ServiceRegistry& registry)
43  : schema{std::move(s)}, messages{std::move(m)}, span{[this](size_t i, size_t part) {
44  BOOST_REQUIRE(i < this->messages.size());
45  BOOST_REQUIRE(part < this->messages[i].second.size() / 2);
46  auto header = static_cast<char const*>(this->messages[i].second.at(2 * part)->data());
47  auto payload = static_cast<char const*>(this->messages[i].second.at(2 * part + 1)->data());
48  return DataRef{nullptr, header, payload};
49  },
50  [this](size_t i) { return i < this->messages.size() ? messages[i].second.size() / 2 : 0; }, this->messages.size()},
51  record{schema, span, registry},
52  values{std::move(v)}
53  {
54  BOOST_REQUIRE(messages.size() == schema.size());
55  }
56 
57  std::vector<InputRoute> schema;
58  Messages messages;
59  InputSpan span;
60  InputRecord record;
61  CheckType values;
62 };

64 DataSet createData()
65 {
66  static ServiceRegistry registry;
67  // Create the routes we want for the InputRecord
68  std::vector<InputSpec> inputspecs = {
69  InputSpec{"tpc", "TPC", "SOMEDATA", 0, Lifetime::Timeframe},
70  InputSpec{"its", ConcreteDataTypeMatcher{"ITS", "SOMEDATA"}, Lifetime::Timeframe},
71  InputSpec{"tof", "TOF", "SOMEDATA", 1, Lifetime::Timeframe}};
72 
73  size_t i = 0;
74  auto createRoute = [&i](const char* source, InputSpec& spec) {
75  return InputRoute{
76  spec,
77  i++,
78  source};
79  };
80 
81  std::vector<InputRoute> schema = {
82  createRoute("tpc_source", inputspecs[0]),
83  createRoute("its_source", inputspecs[1]),
84  createRoute("tof_source", inputspecs[2])};
85 
86  decltype(DataSet::values) checkValues;
87  DataSet::Messages messages;
88 
89  auto createMessage = [&messages, &checkValues](DataHeader dh) {
90  checkValues.emplace_back(fmt::format("{}_{}_{}", dh.dataOrigin, dh.dataDescription, dh.subSpecification));
91  std::string const& data = checkValues.back();
92  dh.payloadSize = data.size();
93  DataProcessingHeader dph{0, 1};
94  Stack stack{dh, dph};
95  auto it = messages.begin(), end = messages.end();
96  for (; it != end; ++it) {
97  if (it->first == dh.dataOrigin) {
98  break;
99  }
100  }
101  if (it == end) {
102  messages.resize(messages.size() + 1);
103  it = messages.end() - 1;
104  it->first = dh.dataOrigin;
105  }
106  auto& routemessages = it->second;
107  routemessages.emplace_back(std::make_unique<std::vector<char>>(stack.size()));
108  memcpy(routemessages.back()->data(), stack.data(), routemessages.back()->size());
109  routemessages.emplace_back(std::make_unique<std::vector<char>>(dh.payloadSize));
110  memcpy(routemessages.back()->data(), data.data(), routemessages.back()->size());
111  };
112 
113  // we create message for the 3 input routes, the messages have different page size
114  // and the second messages has 3 parts, each with the same page size
115  // the test value is written as payload after the RDH and all values are cached for
116  // later checking when parsing the data set
117  DataHeader dh1;
118  dh1.dataDescription = "SOMEDATA";
119  dh1.dataOrigin = "TPC";
120  dh1.subSpecification = 0;
121  dh1.payloadSerializationMethod = o2::header::gSerializationMethodNone;
122  DataHeader dh2;
123  dh2.dataDescription = "SOMEDATA";
124  dh2.dataOrigin = "ITS";
125  dh2.subSpecification = 0;
126  dh2.payloadSerializationMethod = o2::header::gSerializationMethodNone;
127  DataHeader dh3;
128  dh3.dataDescription = "SOMEDATA";
129  dh3.dataOrigin = "ITS";
130  dh3.subSpecification = 1;
131  dh3.payloadSerializationMethod = o2::header::gSerializationMethodNone;
132  DataHeader dh4;
133  dh4.dataDescription = "SOMEDATA";
134  dh4.dataOrigin = "TOF";
135  dh4.subSpecification = 255;
136  dh4.payloadSerializationMethod = o2::header::gSerializationMethodNone;
137  createMessage(dh1);
138  createMessage(dh2);
139  createMessage(dh3);
140  createMessage(dh4);
141 
142  return {std::move(schema), std::move(messages), std::move(checkValues), registry};
143 }
