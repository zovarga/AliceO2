129  static int getAPECode(uint8_t c, bool& ft)
130  {
131  if (c < 0xf2 || c > 0xfa) {
132  ft = false;
133  return -1;
134  }
135  ft = c >= 0xf4;
136  return APE_STRIP_START + c - 0xf2;
137  }
