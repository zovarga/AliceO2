161 template <typename T, typename... Targs>
162 constexpr T String2(char c, Targs... Fargs)
163 {
164  // number of arguments has either to match the size of the type, or the last element is treated
165  // as '0' if missing
166  // TODO: this is a bit of a question, do we allow the number of characters to
167  // be smaller than the size of the type and just pad with 0 like in the case
168  // of a char array argument?
169  static_assert(sizeof...(Targs) == sizeof(T) + 1 ||
170  sizeof...(Targs) == sizeof(T),
171  "number of arguments does not match the uint type width");
172  return internal::String2__<T>(c, Fargs...);
173 }

179 template <typename T, std::size_t N, std::size_t pos = 0, bool suppressAssert = false>
180 constexpr T String2(const char (&str)[N])
181 {
182  static_assert(std::is_integral<T>::value, "Non integral types not compatible with String2<type>");
183  static_assert(N >= pos, "Position is larger than the length of string");
184  static_assert(suppressAssert || N - pos <= sizeof(T) + 1,
185  "String parameter is longer than the size of the data type");
186  // clang-format off
187  return ((T)str[0 + pos] |
188  (str[0 + pos] && sizeof(T) >= 2 ? ((T)str[1 + pos] << (sizeof(T) >= 2 ? 8 : 0) |
189  (str[1 + pos] && sizeof(T) >= 4 ? ((T)str[2 + pos] << (sizeof(T) >= 4 ? 16 : 0) |
190  (str[2 + pos] && sizeof(T) >= 4 ? ((T)str[3 + pos] << (sizeof(T) >= 4 ? 24 : 0) |
191  (str[3 + pos] && sizeof(T) >= 8 ? ((T)str[4 + pos] << (sizeof(T) >= 8 ? 32 : 0) |
192  (str[4 + pos] && sizeof(T) >= 8 ? ((T)str[5 + pos] << (sizeof(T) >= 8 ? 40 : 0) |
193  (str[5 + pos] && sizeof(T) >= 8 ? ((T)str[6 + pos] << (sizeof(T) >= 8 ? 48 : 0) |
194  (str[6 + pos] && sizeof(T) >= 8 ? ((T)str[7 + pos] << (sizeof(T) >= 8 ? 56 : 0))
195  : 0)) : 0)) : 0)) : 0)) : 0)) : 0)) : 0));
196  // clang-format on
197 }

254  void runtimeInit(const char* string, short length = -1)
255  {
256  char* target = str;
257  char* targetEnd = target;
258  if (length >= 0 && length < (int)N) {
259  targetEnd += length;
260  } else {
261  targetEnd += N;
262  }
263  const char* source = string;
264  for (; source != nullptr && target < targetEnd && *source != 0; ++target, ++source) {
265  *target = *source;
266  }
267  targetEnd = str + N;
268  for (; target < targetEnd; ++target) {
269  *target = 0;
270  }
271  // require the string to be not longer than the descriptor size
272  if (source != nullptr && (*source == 0 || (length >= 0 && length <= (int)N))) {
273  } else {
274  throw std::invalid_argument("argument must not be longer than descriptor size");
275  }
276  }

443 template <typename HeaderType, typename std::enable_if_t<std::is_pointer<HeaderType>::value, int> = 0>
444 auto get(const std::byte* buffer, size_t /*len*/ = 0)
445 {
446  using HeaderConstPtrType = const typename std::remove_pointer<HeaderType>::type*;
447  using HeaderValueType = typename std::remove_pointer<HeaderType>::type;
448 
449  const BaseHeader* current = BaseHeader::get(buffer);
450  if (!current) {
451  return HeaderConstPtrType{nullptr};
452  }
453  if (current->description == HeaderValueType::sHeaderType) {
454  // FIXME: We need to think how to handle multiple versions.
455  // For the moment we require the version to exactly match the expected version, which
456  // is part of the header definition. The check function 'sanityCheck' will throw
457  // otherwise, we keep the code related to the exception outside the header file.
458  // Note: Can not check on size because the O2 data model requires variable size headers
459  // to be supported.
460  if (current->sanityCheck(HeaderValueType::sVersion)) {
461  return reinterpret_cast<HeaderConstPtrType>(current);
462  }
463  }
464  auto* prev = current;
465  while ((current = current->next())) {
466  prev = current;
467  if (current->description == HeaderValueType::sHeaderType) {
468  if (current->sanityCheck(HeaderValueType::sVersion)) {
469  return reinterpret_cast<HeaderConstPtrType>(current);
470  }
471  }
472  }
473  // if a next header was expected but not found, the header stack is inconsistent
474  // either the flag is wrongly set, or the size of the current header is wrong so
475  // that the BaseHeader information is not found at the expected offset.
476  if (prev->flagsNextHeader) {
477  prev->throwInconsistentStackError();
478  }
479  return HeaderConstPtrType{nullptr};
480 }

489 // utilities for converting strings to ints - tries to mimick the stadard API, but the result is
490 // strongly typed - can be used for fixed width and short unsigned integer types unlike the standard
491 // utilities.
492 template <typename T>
493 T strtoui(const char* str, char** str_end, int base) noexcept
494 {
495  static_assert(std::numeric_limits<T>::is_integer && !std::numeric_limits<T>::is_signed,
496  "Target must be an unsigned integer type");
497  // standard lib gives us 2 specializations of strtouX, so let's use them both where appropriate
498  if constexpr (sizeof(T) <= sizeof(uint32_t)) {
499  unsigned long res = strtoul(str, str_end, base);
500  if (res > std::numeric_limits<T>::max()) {
501  errno = ERANGE;
502  return std::numeric_limits<T>::max();
503  } else {
504  return static_cast<T>(res);
505  };
506  } else {
507  unsigned long long res = strtoull(str, str_end, base);
508  if (res > std::numeric_limits<T>::max()) {
509  errno = ERANGE;
510  return std::numeric_limits<T>::max();
511  } else {
512  return static_cast<T>(res);
513  };
514  }
515 };
