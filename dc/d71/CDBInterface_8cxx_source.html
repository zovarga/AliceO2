208 void CDBInterface::createDefaultPedestals()
209 {
210  // ===| create random pedestals |=============================================
211  mPedestals = std::make_unique<CalPad>("Pedestals");
212 
213  // distribution based on test beam data
214  const float meanPedestal = 72.5;
215  const float sigmaPedestal = 9.0;
216 
217  // set a minimum and maximum for the value
218  const float minPedestal = meanPedestal - 4 * sigmaPedestal;
219  const float maxPedestal = meanPedestal + 4 * sigmaPedestal;
220 
221  for (auto& calArray : mPedestals->getData()) {
222  for (auto& val : calArray.getData()) {
223  float random = gRandom->Gaus(meanPedestal, sigmaPedestal);
224  if (random < minPedestal) {
225  random = minPedestal;
226  }
227  if (random > maxPedestal) {
228  random = maxPedestal;
229  }
230  val = random;
231  }
232  }
233 }

292 bool CDBStorage::checkMetaData(MetaData_t metaData) const
293 {
294  if (!metaData.size()) {
295  LOGP(error, "no meta data set");
296  }
297 
298  const std::array<std::string_view, 3> requirements{
299  "Required",
300  "Recommended",
301  "Optional"};
302 
303  const std::array<std::vector<std::string_view>, 3> tests{{
304  {"Responsible", "Reason", "Intervention"}, // errors
305  {"JIRA"}, // warnings
306  {"Comment"} // infos
307  }};
308 
309  std::array<int, 3> counts{};
310 
311  for (size_t i = 0; i < requirements.size(); ++i) {
312  for (const auto& test : tests[i]) {
313  if (!metaData[test.data()].size()) {
314  const auto message = fmt::format("{} field {} not set in meta data", requirements[i], test);
315  if (i == 0) {
316  LOG(error) << message;
317  } else if (i == 1) {
318  LOG(warning) << message;
319  } else {
320  LOG(info) << message;
321  }
322  ++counts[i];
323  } else {
324  LOGP(debug, "{} field '{}' set to '{}'", requirements[i], test, metaData[test.data()]);
325  }
326  }
327  }
328 
329  return counts[0] == 0;
330 }
