36 class CalibLaserTracksDevice : public o2::framework::Task
37 {
38  public:
39  void init(o2::framework::InitContext& ic) final
40  {
41  mCalib.setWriteDebugTree(ic.options().get<bool>("write-debug"));
42  mMinNumberTFs = ic.options().get<int>("min-tfs");
43  mOnlyPublishOnEOS = ic.options().get<bool>("only-publish-on-eos");
44 
45  auto finishFunction = [this]() {
46  if (!mPublished) {
47  const auto nTFs = mCalib.getCalibData().processedTFs;
48  const auto nMatchA = mCalib.getMatchedPairsA();
49  const auto nMatchC = mCalib.getMatchedPairsC();
50  LOGP(error, "Calibration data was not published, laser track calibration might have enough statistics: {} ({}) matched tracks in {} TFs on the A (C) < {} min TFs * {} min matches per side per TF ", nMatchA, nMatchC, nTFs, mMinNumberTFs, CalibLaserTracks::MinTrackPerSidePerTF);
51  }
52  };
53  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, finishFunction);
54  }
55 
56  void run(o2::framework::ProcessingContext& pc) final
57  {
58  const auto dph = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header);
59  const auto startTime = dph->startTime;
60  const auto endTime = dph->startTime + dph->duration;
61  mRunNumber = processing_helpers::getRunNumber(pc);
62 
63  auto data = pc.inputs().get<gsl::span<TrackTPC>>("input");
64  mCalib.setTFtimes(startTime, endTime);
65  mCalib.fill(data);
66 
67  if (!mOnlyPublishOnEOS && mCalib.hasEnoughData(mMinNumberTFs) && !mPublished) {
68  sendOutput(pc.outputs());
69  }
70  }
71 
72  void endOfStream(o2::framework::EndOfStreamContext& ec) final
73  {
74  LOGP(info, "CalibLaserTracksDevice::endOfStream: Finalizing calibration");
75  if (!mCalib.hasEnoughData(mMinNumberTFs)) {
76  const auto nTFs = mCalib.getCalibData().processedTFs;
77  const auto nMatchA = mCalib.getMatchedPairsA();
78  const auto nMatchC = mCalib.getMatchedPairsC();
79  LOGP(warning, "laser track calibration does not have enough statistics: {} ({}) matched tracks in {} TFs on the A (C) < {} min TFs * {} min matches per side per TF ", nMatchA, nMatchC, nTFs, mMinNumberTFs, CalibLaserTracks::MinTrackPerSidePerTF);
80  }
81  sendOutput(ec.outputs());
82  }
83

97  std::map<std::string, std::string> md;

104 
105  md = w.getMetaData();

107  w.setMetaData(md);
108 
109  const auto now = std::chrono::system_clock::now();
110  const long timeStart = std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch()).count();
111  //const auto timeStart = ltrCalib.firstTime; //TODO: use once it is a correct time not TFid
112  const long timeEnd = 99999999999999;
113 
114  w.setPath("TPC/Calib/LaserTracks");
115  w.setStartValidityTimestamp(timeStart);
116  w.setEndValidityTimestamp(timeEnd);
117 
118  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", w.getPath(), w.getFileName(), image->size(), w.getStartValidityTimestamp(), w.getEndValidityTimestamp());
119  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "TPC_CalibLtr", 0}, *image.get());
120  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "TPC_CalibLtr", 0}, w);
121 
122  mPublished = true;
123  }

126 DataProcessorSpec getCalibLaserTracks(const std::string inputSpec)
127 {
128  using device = o2::tpc::CalibLaserTracksDevice;
129 
130  std::vector<OutputSpec> outputs;
131  outputs.emplace_back(ConcreteDataTypeMatcher{"TPC", "LtrCalibData"}, Lifetime::Sporadic);
132  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "TPC_CalibLtr"}, Lifetime::Sporadic);
133  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "TPC_CalibLtr"}, Lifetime::Sporadic);
134 
135  return DataProcessorSpec{
136  "tpc-calib-laser-tracks",
137  select(inputSpec.data()),
138  outputs,
139  AlgorithmSpec{adaptFromTask<device>()},
140  Options{
141  {"write-debug", VariantType::Bool, false, {"write a debug output tree."}},
142  {"min-tfs", VariantType::Int, 100, {"minimum number of TFs with enough laser tracks to finalize the calibration."}},
143  {"only-publish-on-eos", VariantType::Bool, false, {"only publish the calibration on eos, not during running"}},
144  }};
145 }
