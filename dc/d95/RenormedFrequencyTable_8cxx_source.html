24 RenormedFrequencyTable renorm(FrequencyTable frequencyTable, size_t newPrecision)
25 {
26 
27  using namespace internal;
28  LOG(trace) << "start rescaling frequency table";
29  RANSTimer t;
30  t.start();
31 
32  if (frequencyTable.empty()) {
33  LOG(warning) << "rescaling Frequency Table for empty message";
34  }
35 
36  if (newPrecision == 0) {
37  newPrecision = computeRenormingPrecision(frequencyTable);
38  }
39 
40  count_t nSamples = frequencyTable.getNumSamples();
41  count_t nIncompressible = frequencyTable.getIncompressibleSymbolFrequency();
42  count_t nUsedAlphabetSymbols = frequencyTable.getNUsedAlphabetSymbols();
43  const symbol_t offset = frequencyTable.getMinSymbol();
44 
45  // add an incompressible symbol if not present
46  if (!frequencyTable.hasIncompressibleSymbols()) {
47  nIncompressible = 1;
48  nSamples += nIncompressible;
49  nUsedAlphabetSymbols += 1;
50  }
51 
52  histogram_t frequencies = std::move(frequencyTable).release();
53  frequencies.push_back(nIncompressible);
54 
55  histogram_t cumulativeFrequencies(frequencies.size() + 1);
56  cumulativeFrequencies[0] = 0;
57  std::inclusive_scan(frequencies.begin(), frequencies.end(), ++cumulativeFrequencies.begin());
58 
59  auto getFrequency = [&cumulativeFrequencies](count_t i) { return cumulativeFrequencies[i + 1] - cumulativeFrequencies[i]; };
60 
61  const auto sortIdx = [&]() {
62  std::vector<size_t> indices;
63  indices.reserve(nUsedAlphabetSymbols);
64 
65  // we will sort only those memorize only those entries which can be used
66  for (size_t i = 0; i < frequencies.size(); i++) {
67  if (frequencies[i] != 0) {
68  indices.push_back(i);
69  }
70  }
71  std::sort(indices.begin(), indices.end(), [&](count_t i, count_t j) { return getFrequency(i) < getFrequency(j); });
72 
73  return indices;
74  }();
75 
76  // resample distribution based on cumulative frequencies
77  const count_t newCumulatedFrequency = pow2(newPrecision);
78  assert(newCumulatedFrequency >= nUsedAlphabetSymbols);
79  size_t needsShift = 0;
80  for (size_t i = 0; i < sortIdx.size(); i++) {
81  if (static_cast<count_t>(getFrequency(sortIdx[i])) * (newCumulatedFrequency - needsShift) / nSamples >= 1) {
82  break;
83  }
84  needsShift++;
85  }
86 
87  size_t shift = 0;
88  auto beforeUpdate = cumulativeFrequencies[0];
89  for (size_t i = 0; i < frequencies.size(); i++) {
90  if (frequencies[i] && static_cast<uint64_t>(cumulativeFrequencies[i + 1] - beforeUpdate) * (newCumulatedFrequency - needsShift) / nSamples < 1) {
91  shift++;
92  }
93  beforeUpdate = cumulativeFrequencies[i + 1];
94  cumulativeFrequencies[i + 1] = (static_cast<uint64_t>(newCumulatedFrequency - needsShift) * cumulativeFrequencies[i + 1]) / nSamples + shift;
95  }
96  assert(shift == needsShift);
97 
98  // verify
99 #if !defined(NDEBUG)
100  assert(cumulativeFrequencies.front() == 0 &&
101  cumulativeFrequencies.back() == newCumulatedFrequency);
102  for (size_t i = 0; i < frequencies.size(); i++) {
103  if (frequencies[i] == 0) {
104  assert(cumulativeFrequencies[i + 1] == cumulativeFrequencies[i]);
105  } else {
106  assert(cumulativeFrequencies[i + 1] > cumulativeFrequencies[i]);
107  }
108  }
109 #endif
110 
111  // calculate updated frequencies
112  for (size_t i = 0; i < frequencies.size(); i++) {
113  frequencies[i] = getFrequency(i);
114  }
115 
116  assert(frequencies.size() >= 1);
117  FrequencyTable rescaledFrequencies{frequencies.begin(), --frequencies.end(), offset, frequencies.back()};
118 
119  t.stop();
120  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
121 
122  LOG(trace) << "done rescaling frequency table";
123  return RenormedFrequencyTable{std::move(rescaledFrequencies), newPrecision};
124 };

126 RenormedFrequencyTable renormCutoffIncompressible(FrequencyTable frequencyTable, uint8_t newPrecision, uint8_t lowProbabilityCutoffBits)
127 {
128  using namespace internal;
129  LOG(trace) << "start rescaling frequency table";
130  RANSTimer t;
131  t.start();
132 
133  if (frequencyTable.empty()) {
134  LOG(warning) << "rescaling Frequency Table for empty message";
135  }
136 
137  if (newPrecision == 0) {
138  newPrecision = computeRenormingPrecision(frequencyTable);
139  }
140 
141  const count_t nSamplesRescaled = 1 << newPrecision;
142  const double_t probabilityCutOffThreshold = 1 / static_cast<double_t>(1 << (newPrecision + lowProbabilityCutoffBits));
143 
144  // scaling
145  double_t incompressibleSymbolProbability = static_cast<double_t>(frequencyTable.getIncompressibleSymbolFrequency()) / nSamplesRescaled;
146  count_t nSamplesRescaledUncorrected = 0;
147  std::vector<size_t> correctableIndices;
148  correctableIndices.reserve(frequencyTable.getNUsedAlphabetSymbols());
149 
150  auto scaleFrequency = [nSamplesRescaled](double_t symbolProbability) { return symbolProbability * nSamplesRescaled; };
151  auto roundDownFrequency = [](double_t i) { return static_cast<count_t>(i); };
152  auto roundUpFrequency = [roundDownFrequency](count_t i) { return roundDownFrequency(i) + 1; };
153  auto roundFrequency = [roundDownFrequency, roundUpFrequency](double_t rescaledFrequency) {
154  if (rescaledFrequency * rescaledFrequency <= (roundDownFrequency(rescaledFrequency) * roundUpFrequency(rescaledFrequency))) {
155  return roundDownFrequency(rescaledFrequency);
156  } else {
157  return roundUpFrequency(rescaledFrequency);
158  }
159  };
160 
161  histogram_t rescaledFrequencies(frequencyTable.size());
162 
163  for (size_t i = 0; i < frequencyTable.size(); ++i) {
164  const count_t frequency = frequencyTable.at(i);
165  const double_t symbolProbability = static_cast<double_t>(frequency) / static_cast<double_t>(frequencyTable.getNumSamples());
166  if (symbolProbability < probabilityCutOffThreshold) {
167  incompressibleSymbolProbability += symbolProbability;
168  rescaledFrequencies[i] = 0;
169  } else {
170  const double_t scaledFrequencyD = scaleFrequency(symbolProbability);
171  count_t rescaledFrequency = roundFrequency(scaledFrequencyD);
172  assert(rescaledFrequency > 0);
173  rescaledFrequencies[i] = rescaledFrequency;
174  nSamplesRescaledUncorrected += rescaledFrequency;
175  if (rescaledFrequency > 1) {
176  correctableIndices.push_back(i);
177  }
178  }
179  }
180 
181  // treat incompressible symbol
182  const count_t incompressibleSymbolFrequency = std::max(static_cast<count_t>(1), static_cast<count_t>(incompressibleSymbolProbability * nSamplesRescaled));
183  nSamplesRescaledUncorrected += incompressibleSymbolFrequency;
184 
185  // correction
186  std::stable_sort(correctableIndices.begin(), correctableIndices.end(), [&rescaledFrequencies](const count_t& a, const count_t& b) { return rescaledFrequencies[a] < rescaledFrequencies[b]; });
187 
188  int32_t nCorrections = nSamplesRescaled - nSamplesRescaledUncorrected;
189  const double_t rescalingFactor = static_cast<double_t>(nSamplesRescaled) / static_cast<double_t>(nSamplesRescaledUncorrected);
190 
191  for (auto index : correctableIndices) {
192  if (std::abs(nCorrections) > 0) {
193  const count_t uncorrectedFrequency = rescaledFrequencies[index];
194  int32_t correction = uncorrectedFrequency - roundFrequency(uncorrectedFrequency * rescalingFactor);
195 
196  if (nCorrections < 0) {
197  // overshoot - correct downwards by subtracting correction in [1,|nCorrections|]
198  correction = std::max(1, std::min(correction, std::abs(nCorrections)));
199  } else {
200  // correct upwards by subtracting correction in [-1, -nCorrections]
201  correction = std::min(-1, std::max(correction, -nCorrections));
202  }
203 
204  // the corrected frequency must be at least 1 though
205  const count_t correctedFrequency = std::max(1u, uncorrectedFrequency - correction);
206  nCorrections += uncorrectedFrequency - correctedFrequency;
207  rescaledFrequencies[index] = correctedFrequency;
208  } else {
209  break;
210  }
211  }
212 
213  if (std::abs(nCorrections) > 0) {
214  throw std::runtime_error(fmt::format("rANS rescaling incomplete: {} corrections Remaining", nCorrections));
215  }
216 
217  FrequencyTable newFrequencyTable{rescaledFrequencies.begin(), rescaledFrequencies.end(), frequencyTable.getMinSymbol(), incompressibleSymbolFrequency};
218 
219  t.stop();
220  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
221 
222  LOG(trace) << "done rescaling frequency table";
223  return RenormedFrequencyTable{std::move(newFrequencyTable), newPrecision};
224 }
