121 int ClusterPattern::getCOG(int rowSpan, int colSpan, const unsigned char patt[MaxPatternBytes], float& xCOG, float& zCOG)
122 {
123  int tempxCOG = 0, tempzCOG = 0, tempFiredPixels = 0, ic = 0, ir = 0;
124  int nBits = rowSpan * colSpan;
125  int nBytes = nBits / 8;
126  if (nBits % 8 != 0) {
127  nBytes++;
128  }
129  for (unsigned int i = 0; i < nBytes; i++) {
130  unsigned char tempChar = patt[i];
131  int s = 128; // 0b10000000
132  while (s > 0) {
133  if ((tempChar & s) != 0) {
134  tempFiredPixels++;
135  tempxCOG += ir;
136  tempzCOG += ic;
137  }
138  ic++;
139  s /= 2;
140  if ((ir + 1) * ic == nBits) {
141  break;
142  }
143  if (ic == colSpan) {
144  ic = 0;
145  ir++;
146  }
147  }
148  if ((ir + 1) * ic == nBits) {
149  break;
150  }
151  }
152  xCOG = float(tempxCOG) / tempFiredPixels;
153  zCOG = float(tempzCOG) / tempFiredPixels;
154 
155  return tempFiredPixels;
