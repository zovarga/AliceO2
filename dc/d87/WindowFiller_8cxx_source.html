31 // How data acquisition works in real data
32 /*
33  |<----------- 1 orbit ------------->|
34  ------|-----------|-----------|-----------|------
35  ^ ^ ^ ^ when triggers happen
36  |<--- latency ---|
37  |<- matching1->|
38  |<- matching2->|
39  |<- matching3->|
40  |<>| = overlap between two consecutive matching
41 
42 Norbit = number of orbits elapsed
43 Nbunch = bunch in the current orbit (0:3563)
44 Ntdc = number of tdc counts within the matching window --> for 1/3 orbit (0:3649535)
45 
46 raw time = trigger time (Norbit and Nbunch) - latency window + TDC(Ntdc)
47  */
48 
49 // What we implemented here (so far)
50 /*
51  |<----------- 1 orbit ------------->|
52  ------|-----------|-----------|-----------|------
53  |<- matching1->|
54  |<- matching2->|
55  |<- matching3->|
56  |<>| = overlap between two consecutive matching windows
57 
58 - NO OVERLAP between two consecutive windows at the moment (to be implemented)
59 - NO LATENCY WINDOW (we manage it during raw encoding/decoding) then digits already corrected
60 
61 NBC = Number of bunch since timeframe beginning = Norbit*3564 + Nbunch
62 Ntdc = here within the current BC -> (0:1023)
63 
64 digit time = NBC*1024 + Ntdc
65  */

320 void WindowFiller::checkIfReuseFutureDigits()
321 {
322  if (!mFutureDigits.size()) {
323  return;
324  }
325 
326  // check if digits stored very far in future match the new readout windows currently available
327  if (mFutureToBeSorted) {
328  // sort digit in descending BC order: kept last as first
329  std::sort(mFutureDigits.begin(), mFutureDigits.end(),
330  [](o2::tof::Digit a, o2::tof::Digit b) { return a.getBC() > b.getBC(); });
331  mFutureToBeSorted = false;
332  }
333 
334  int idigit = mFutureDigits.size() - 1;
335 
336  int bclimit = 999999; // if bc is larger than this value stop the search in the next loop since bc are ordered in descending order
337 
338  for (std::vector<Digit>::reverse_iterator digit = mFutureDigits.rbegin(); digit != mFutureDigits.rend(); ++digit) {
339 
340  if (digit->getBC() > bclimit) {
341  break;
342  }
343 
344  double timestamp = digit->getBC() * Geo::BC_TIME + digit->getTDC() * Geo::TDCBIN * 1E-3; // in ns
345  int isnext = Int_t(timestamp * Geo::READOUTWINDOW_INV) - (mReadoutWindowCurrent + 1); // to be replaced with uncalibrated time
346 
347  if (isnext < 0) { // we jump too ahead in future, digit will be not stored
348  LOG(debug) << "Digit lost because we jump too ahead in future. Current RO window=" << isnext << "\n";
349 
350  // remove digit from array in the future
351  int labelremoved = digit->getLabel();
352  mFutureDigits.erase(mFutureDigits.begin() + idigit);
353 
354  idigit--;
355 
356  continue;
357  }
358 
359  if (isnext < MAXWINDOWS - 1) { // move from digit buffer array to the proper window
360  std::vector<Strip>* strips = mStripsCurrent;
361 
362  if (isnext) {
363  strips = mStripsNext[isnext - 1];
364  }
365 
366  fillDigitsInStrip(strips, digit->getChannel(), digit->getTDC(), digit->getTOT(), digit->getBC(), digit->getChannel() / Geo::NPADS);
367 
368  // int labelremoved = digit->getLabel();
369  mFutureDigits.erase(mFutureDigits.begin() + idigit);
370 
371  } else {
372  bclimit = digit->getBC();
373  }
374  idigit--; // go back to the next position in the reverse iterator
375  } // close future digit loop
376 }

378 void WindowFiller::checkIfReuseFutureDigitsRO() // the same but using readout info information from raw
379 {
380  if (!mFutureDigits.size()) {
381  return;
382  }
383 
384  // check if digits stored very far in future match the new readout windows currently available
385  if (mFutureToBeSorted) {
386  // sort digit in descending BC order: kept last as first
387  std::sort(mFutureDigits.begin(), mFutureDigits.end(),
388  [](o2::tof::Digit a, o2::tof::Digit b) {
389  if (a.getTriggerOrbit() != b.getTriggerOrbit()) {
390  return a.getTriggerOrbit() > b.getTriggerOrbit();
391  }
392  if (a.getTriggerBunch() != b.getTriggerBunch()) {
393  return a.getTriggerBunch() > b.getTriggerBunch();
394  }
395  return a.getBC() > b.getBC();
396  });
397  mFutureToBeSorted = false;
398  }
399 
400  int idigit = mFutureDigits.size() - 1;
401 
402  int rolimit = 999999; // if bc is larger than this value stop the search in the next loop since bc are ordered in descending order
403 
404  for (std::vector<Digit>::reverse_iterator digit = mFutureDigits.rbegin(); digit != mFutureDigits.rend(); ++digit) {
405 
406  int row = (digit->getTriggerOrbit() - mFirstIR.orbit) * Geo::BC_IN_ORBIT + (digit->getTriggerBunch() - mFirstIR.bc) + 100; // N bunch id of the trigger from timeframe start + 100 bunches
407 
408  row *= Geo::BC_IN_WINDOW_INV;
409 
410  if (row > rolimit) {
411  break;
412  }
413 
414  int isnext = row - mReadoutWindowCurrent;
415 
416  if (isnext < 0) { // we jump too ahead in future, digit will be not stored
417  LOG(debug) << "Digit lost because we jump too ahead in future. Current RO window=" << isnext << "\n";
418 
419  // remove digit from array in the future
420  int labelremoved = digit->getLabel();
421  mFutureDigits.erase(mFutureDigits.begin() + idigit);
422 
423  idigit--;
424 
425  continue;
426  }
427 
428  if (isnext < MAXWINDOWS - 1) { // move from digit buffer array to the proper window
429  std::vector<Strip>* strips = mStripsCurrent;
430 
431  if (isnext) {
432  strips = mStripsNext[isnext - 1];
433  }
434 
435  if (mMaskNoiseRate < 0 || mChannelCounts[digit->getChannel()] < mMaskNoiseRate) {
436  fillDigitsInStrip(strips, digit->getChannel(), digit->getTDC(), digit->getTOT(), digit->getBC(), digit->getChannel() / Geo::NPADS);
437  }
438 
439  // int labelremoved = digit->getLabel();
440  mFutureDigits.erase(mFutureDigits.begin() + idigit);
441  } else {
442  rolimit = row;
443  }
444  idigit--; // go back to the next position in the reverse iterator
445  } // close future digit loop

448 void WindowFiller::fillDiagnosticFrequency()
449 {
450  bool isTOFempty = true;
451  // fill diagnostic frequency
452  for (int j = 0; j < mReadoutWindowData.size(); j++) {
453  mDiagnosticFrequency.fillROW();
454  for (int ic = 0; ic < 72; ic++) {
455  int dia = mReadoutWindowData[j].getDiagnosticInCrate(ic);
456  int slot = 0;
457  if (mReadoutWindowData[j].isEmptyCrate(ic)) {
458  mDiagnosticFrequency.fillEmptyCrate(ic);
459  if (dia) {
460  int fd = mReadoutWindowData[j].firstDia();
461  int lastdia = fd + dia;
462 
463  ULong64_t key;
464  for (int dd = fd; dd < lastdia; dd++) {
465  if (mPatterns[dd] >= 28) {
466  slot = mPatterns[dd] - 28;
467  key = mDiagnosticFrequency.getTRMKey(ic, slot);
468  continue;
469  }
470 
471  key += (1 << mPatterns[dd]);
472 
473  if (dd + 1 == lastdia || mPatterns[dd + 1] >= 28) {
474  mDiagnosticFrequency.fill(key);
475  }
476  }
477  }
478  } else {
479  isTOFempty = false;
480  }
481  }
482  }
483 
484  if (isTOFempty) {
485  mDiagnosticFrequency.fillEmptyTOF();
486  }
487 
488  // fill also noise diagnostic if the counts within the TF is larger than a threashold (default >=11, -> 1 kHZ)
489  int masknoise = mMaskNoiseRate;
490  if (masknoise < 0) {
491  masknoise = -masknoise;
492  }
493 
494  for (int i = 0; i < Geo::NCHANNELS; i++) {
495  if (mChannelCounts[i] >= masknoise) {
496  int additionalMask = 0;
497 
498  if (mChannelCounts[i] >= masknoise * 10) {
499  additionalMask += (1 << 19); // > 10 kHZ (if masknoise = 1 kHz)
500  if (mChannelCounts[i] >= masknoise * 100) {
501  additionalMask += (1 << 20); // > 100 kHZ (if masknoise = 1 kHz)
502  }
503  }
504 
505  //Fill noisy in diagnostic
506  mDiagnosticFrequency.fillNoisy(i + additionalMask, mReadoutWindowData.size());
507  }
508  }
