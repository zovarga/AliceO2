56  std::vector<char> to_vector(bool tcm)
57  {
58  constexpr int CRUWordSize = 16;
59  const char padding[CRUWordSize] = {0};
60 
61  std::vector<char> result(size() * CRUWordSize);
62  char* out = result.data();
63  if (!tcm) {
64  std::memcpy(out, &mEventHeader, sPayloadSize);
65  out += sPayloadSize;
66  LOG(debug) << "write header words " << (int)mEventHeader.nGBTWords << " orbit " << int(mEventHeader.orbit) << " bc " << int(mEventHeader.bc) << " out " << result.size();
67  if (mIsPadded) {
68  out += CRUWordSize - sPayloadSize;
69  }
70  for (int i = 0; i < mEventHeader.nGBTWords; ++i) {
71  std::memcpy(out, &mEventData[2 * i], sPayloadSizeFirstWord);
72  LOG(debug) << " 1st word " << mEventData[2 * i].channelID << " charge " << mEventData[2 * i].charge << " time " << mEventData[2 * i].time << " out " << result.size();
73  out += sPayloadSizeFirstWord;
74  std::memcpy(out, &mEventData[2 * i + 1], sPayloadSizeSecondWord);
75  out += sPayloadSizeSecondWord;
76  LOG(debug) << " 2nd word " << mEventData[2 * i + 1].channelID << " charge " << mEventData[2 * i + 1].charge << " time " << mEventData[2 * i + 1].time << " out " << result.size();
77  if (mIsPadded) {
78  out += CRUWordSize - sPayloadSizeSecondWord - sPayloadSizeFirstWord;
79  }
80  }
81  } else {
82  // TCM data
83  std::memcpy(out, &mEventHeader, sPayloadSize);
84  out += sPayloadSize;
85  LOG(debug) << "write TCM header words " << (int)mEventHeader.nGBTWords << " orbit " << int(mEventHeader.orbit) << " bc " << int(mEventHeader.bc) << " out " << result.size();
86  std::memcpy(out, &mTCMdata, sizeof(TCMdata));
87  out += sizeof(TCMdata);
88  LOG(debug) << "write TCM words " << sizeof(mTCMdata) << " orbit " << int(mEventHeader.orbit) << " bc " << int(mEventHeader.bc) << " out " << result.size() << " sum time A " << mTCMdata.timeA;
89  }
90 
91  return result;
92  }

108 class DataPageWriter
109 {
110  std::vector<char> mBuffer;
111  int mNpacketsInBuffer = 0;
112  std::vector<std::vector<char>> mPages;
113  std::vector<int> mNpackets;
114  static constexpr int MAX_Page_size = 8192;
115 
116  public:
117  o2::header::RAWDataHeader mRDH;
118  void flush(std::ostream& str)
119  {
120  writePage();
121  mRDH.stop = 0;
122  for (int page = 0; page < int(mPages.size()); ++page) {
123  mRDH.memorySize = mPages[page].size() + mRDH.headerSize;
124  mRDH.offsetToNext = mRDH.memorySize;
125  mRDH.packetCounter = mNpackets[page];
126  str.write(reinterpret_cast<const char*>(&mRDH), sizeof(mRDH));
127  str.write(mPages[page].data(), mPages[page].size());
128  mRDH.pageCnt++;
129  LOG(info) << " header " << mRDH.linkID << " end " << mRDH.endPointID;
130  }
131  if (!mPages.empty()) {
132  mRDH.memorySize = mRDH.headerSize;
133  mRDH.offsetToNext = mRDH.memorySize;
134  mRDH.stop = 1;
135  mRDH.pageCnt++;
136  str.write(reinterpret_cast<const char*>(&mRDH), sizeof(mRDH));
137  mPages.clear();
138  mNpackets.clear();
139  }
140  }
141 
142  void writePage()
143  {
144  if (mBuffer.size() == 0) {
145  return;
146  }
147  mPages.emplace_back(std::move(mBuffer));
148  LOG(debug) << " writePage " << mBuffer.size();
149  mNpackets.push_back(mNpacketsInBuffer);
150  mNpacketsInBuffer = 0;
151  mBuffer.clear();
152  }
153 
154  void write(std::vector<char> const& new_data)
155  {
156  if (mBuffer.size() + new_data.size() + mRDH.headerSize > MAX_Page_size) {
157  LOG(debug) << " write rest " << mBuffer.size() << " " << new_data.size() << " " << mRDH.headerSize;
158  writePage();
159  }
160  LOG(debug) << " write vector " << new_data.size() << " buffer " << mBuffer.size() << " RDH " << mRDH.headerSize << " new data " << new_data.data();
161  mBuffer.insert(mBuffer.end(), new_data.begin(), new_data.end());
162  mNpacketsInBuffer++;
163  LOG(debug) << " write vector end mBuffer.size " << mBuffer.size() << " mNpacketsInBuffer " << mNpacketsInBuffer << " newdtata " << new_data.size();
164  }
