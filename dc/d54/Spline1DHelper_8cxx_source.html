50 template <typename DataT>
51 void Spline1DHelper<DataT>::approximateFunctionClassic(Spline1DContainer<DataT>& spline,
52  double xMin, double xMax, std::function<void(double x, double f[/*spline.getFdimensions()*/])> F)
53 {
56 
57  int Ndim = spline.getYdimensions();
58  const int nKnots = spline.getNumberOfKnots();
59 
60  TMatrixD A(nKnots, nKnots);
61 
62  A.Zero();
63 
64  /*
65  const Spline1D::Knot& knot0 = spline.getKnot(i);
66  double x = (u - knot0.u) * knot0.Li; // scaled u
67  double cS1 = (6 - 12*x)*knot0.Li*knot0.Li;
68  double cZ0 = (6*x-4)*knot0.Li;
69  double cZ1 = (6*x-2)*knot0.Li;
70  // f''(u) = cS1*(f1-f0) + cZ0*z0 + cZ1*z1;
71  */
72 
73  // second derivative at knot0 is 0
74  {
75  const typename Spline1D<DataT>::Knot& knot0 = spline.getKnot(0);
76  double cZ0 = (-4) * knot0.Li;
77  double cZ1 = (-2) * knot0.Li;
78  // f''(u) = cS1*(f1-f0) + cZ0*z0 + cZ1*z1;
79  A(0, 0) = cZ0;
80  A(0, 1) = cZ1;
81  }
82 
83  // second derivative at knot nKnots-1 is 0
84  {
85  const typename Spline1D<DataT>::Knot& knot0 = spline.getKnot(nKnots - 2);
86  double cZ0 = (6 - 4) * knot0.Li;
87  double cZ1 = (6 - 2) * knot0.Li;
88  // f''(u) = cS1*(f1-f0) + cZ0*z0 + cZ1*z1;
89  A(nKnots - 1, nKnots - 2) = cZ0;
90  A(nKnots - 1, nKnots - 1) = cZ1;
91  }
92 
93  // second derivative at other knots is same from the left and from the right
94  for (int i = 1; i < nKnots - 1; i++) {
95  const typename Spline1D<DataT>::Knot& knot0 = spline.getKnot(i - 1);
96  double cZ0 = (6 - 4) * knot0.Li;
97  double cZ1_0 = (6 - 2) * knot0.Li;
98  // f''(u) = cS1*(f1-f0) + cZ0*z0 + cZ1*z1;
99 
100  const typename Spline1D<DataT>::Knot& knot1 = spline.getKnot(i);
101  double cZ1_1 = (-4) * knot1.Li;
102  double cZ2 = (-2) * knot1.Li;
103  // f''(u) = cS2*(f2-f1) + cZ1_1*z1 + cZ2*z2;
104  A(i, i - 1) = cZ0;
105  A(i, i) = cZ1_0 - cZ1_1;
106  A(i, i + 1) = -cZ2;
107  }
108 
109  A.Invert();
110 
111  spline.setXrange(xMin, xMax);
112  DataT* parameters = spline.getParameters();
113 
114  TVectorD b(nKnots);
115  b.Zero();
116 
117  double uToXscale = (((double)xMax) - xMin) / spline.getUmax();
118 
119  for (int i = 0; i < nKnots; ++i) {
120  const typename Spline1D<DataT>::Knot& knot = spline.getKnot(i);
121  double u = knot.u;
122  double f[Ndim];
123  F(xMin + u * uToXscale, f);
124  for (int dim = 0; dim < Ndim; dim++) {
125  parameters[(2 * i) * Ndim + dim] = f[dim];
126  }
127  }
128 
129  for (int dim = 0; dim < Ndim; dim++) {
130 
131  // second derivative at knot0 is 0
132  {
133  double f0 = parameters[(2 * 0) * Ndim + dim];
134  double f1 = parameters[(2 * 1) * Ndim + dim];
135  const typename Spline1D<DataT>::Knot& knot0 = spline.getKnot(0);
136  double cS1 = (6) * knot0.Li * knot0.Li;
137  // f''(u) = cS1*(f1-f0) + cZ0*z0 + cZ1*z1;
138  b(0) = -cS1 * (f1 - f0);
139  }
140 
141  // second derivative at knot nKnots-1 is 0
142  {
143  double f0 = parameters[2 * (nKnots - 2) * Ndim + dim];
144  double f1 = parameters[2 * (nKnots - 1) * Ndim + dim];
145  const typename Spline1D<DataT>::Knot& knot0 = spline.getKnot(nKnots - 2);
146  double cS1 = (6 - 12) * knot0.Li * knot0.Li;
147  // f''(u) = cS1*(f1-f0) + cZ0*z0 + cZ1*z1;
148  b(nKnots - 1) = -cS1 * (f1 - f0);
149  }
150 
151  // second derivative at other knots is same from the left and from the right
152  for (int i = 1; i < nKnots - 1; i++) {
153  double f0 = parameters[2 * (i - 1) * Ndim + dim];
154  double f1 = parameters[2 * (i)*Ndim + dim];
155  double f2 = parameters[2 * (i + 1) * Ndim + dim];
156  const typename Spline1D<DataT>::Knot& knot0 = spline.getKnot(i - 1);
157  double cS1 = (6 - 12) * knot0.Li * knot0.Li;
158  // f''(u) = cS1*(f1-f0) + cZ0*z0 + cZ1*z1;
159 
160  const typename Spline1D<DataT>::Knot& knot1 = spline.getKnot(i);
161  double cS2 = (6) * knot1.Li * knot1.Li;
162  // f''(u) = cS2*(f2-f1) + cZ1_1*z1 + cZ2*z2;
163  b(i) = -cS1 * (f1 - f0) + cS2 * (f2 - f1);
164  }
165 
166  TVectorD c = A * b;
167  for (int i = 0; i < nKnots; i++) {
168  parameters[(2 * i + 1) * Ndim + dim] = c[i];
169  }
170  }
171 }

223 template <typename DataT>
224 int Spline1DHelper<DataT>::setSpline(
225  const Spline1DContainer<DataT>& spline, int nFdimensions, int nAuxiliaryDataPoints)
226 {
227  // Prepare creation of a best-fit spline
228  //
229  // Data points will be set at all integer U (that includes all knots),
230  // plus at nAuxiliaryDataPoints points between the integers.
231  //
232  // nAuxiliaryDataPoints must be >= 2
233  //
234  // nAuxiliaryDataPoints==1 is also possible, but there must be at least
235  // one integer U without a knot, in order to get 2*nKnots data points in total.
236  //
237  // The return value is an error index, 0 means no error
238 
239  int ret = 0;
240 
241  mFdimensions = nFdimensions;
242  int nPoints = 0;
243  if (!spline.isConstructed()) {
244  ret = storeError(-1, "Spline1DHelper<DataT>::setSpline: input spline is not constructed");
245  mSpline.recreate(0, 2);
246  nAuxiliaryDataPoints = 2;
247  nPoints = 4;
248  } else {
249  std::vector<int> knots;
250  for (int i = 0; i < spline.getNumberOfKnots(); i++) {
251  knots.push_back(spline.getKnot(i).getU());
252  }
253  mSpline.recreate(0, spline.getNumberOfKnots(), knots.data());
254 
255  nPoints = 1 + mSpline.getUmax() + mSpline.getUmax() * nAuxiliaryDataPoints;
256  if (nPoints < 2 * mSpline.getNumberOfKnots()) {
257  nAuxiliaryDataPoints = 2;
258  nPoints = 1 + mSpline.getUmax() + mSpline.getUmax() * nAuxiliaryDataPoints;
259  ret = storeError(-3, "Spline1DHelper::setSpline: too few nAuxiliaryDataPoints, increase to 2");
260  }
261  }
262 
263  const int nPar = 2 * mSpline.getNumberOfKnots(); // n parameters for 1D
264 
265  mDataPoints.resize(nPoints);
266  double scalePoints2Knots = ((double)mSpline.getUmax()) / (nPoints - 1.);
267 
268  for (int i = 0; i < nPoints; ++i) {
269  DataPoint& p = mDataPoints[i];
270  double u = i * scalePoints2Knots;
271  int iKnot = mSpline.getLeftKnotIndexForU(u);
272  const typename Spline1D<double>::Knot& knot0 = mSpline.getKnot(iKnot);
273  const typename Spline1D<double>::Knot& knot1 = mSpline.getKnot(iKnot + 1);
274  double l = knot1.u - knot0.u;
275  double s = (u - knot0.u) * knot0.Li; // scaled u
276  double s2 = s * s;
277  double sm1 = s - 1.;
278 
279  p.iKnot = iKnot;
280  p.isKnot = 0;
281  p.u = u;
282  p.cS1 = s2 * (3. - 2. * s);
283  p.cS0 = 1. - p.cS1;
284  p.cZ0 = s * sm1 * sm1 * l;
285  p.cZ1 = s2 * sm1 * l;
286  }
287 
288  const int nKnots = mSpline.getNumberOfKnots();
289 
290  mKnotDataPoints.resize(nKnots);
291 
292  for (int i = 0; i < nKnots; ++i) {
293  const typename Spline1D<double>::Knot& knot = mSpline.getKnot(i);
294  int iu = (int)(knot.u + 0.1f);
295  mKnotDataPoints[i] = iu * (1 + nAuxiliaryDataPoints);
296  mDataPoints[mKnotDataPoints[i]].isKnot = 1;
297  }
298 
299  TMatrixDSym A(nPar);
300  A.Zero();
301 
302  for (int i = 0; i < nPoints; ++i) {
303  DataPoint& p = mDataPoints[i];
304  int j = p.iKnot * 2;
305  A(j + 0, j + 0) += p.cS0 * p.cS0;
306  A(j + 1, j + 0) += p.cS0 * p.cZ0;
307  A(j + 2, j + 0) += p.cS0 * p.cS1;
308  A(j + 3, j + 0) += p.cS0 * p.cZ1;
309 
310  A(j + 1, j + 1) += p.cZ0 * p.cZ0;
311  A(j + 2, j + 1) += p.cZ0 * p.cS1;
312  A(j + 3, j + 1) += p.cZ0 * p.cZ1;
313 
314  A(j + 2, j + 2) += p.cS1 * p.cS1;
315  A(j + 3, j + 2) += p.cS1 * p.cZ1;
316 
317  A(j + 3, j + 3) += p.cZ1 * p.cZ1;
318  }
319 
320  // copy symmetric matrix elements
321 
322  for (int i = 0; i < nPar; i++) {
323  for (int j = i + 1; j < nPar; j++) {
324  A(i, j) = A(j, i);
325  }
326  }
327 
328  TMatrixDSym Z(nKnots);
329  mLSMmatrixSvalues.resize(nKnots * nKnots);
330  for (int i = 0, k = 0; i < nKnots; i++) {
331  for (int j = 0; j < nKnots; j++, k++) {
332  mLSMmatrixSvalues[k] = A(i * 2 + 1, j * 2);
333  Z(i, j) = A(i * 2 + 1, j * 2 + 1);
334  }
335  }
336 
337  {
338  TDecompBK bk(A, 0);
339  bool ok = bk.Invert(A);
340 
341  if (!ok) {
342  ret = storeError(-4, "Spline1DHelper::setSpline: internal error - can not invert the matrix");
343  A.Zero();
344  }
345  mLSMmatrixFull.resize(nPar * nPar);
346  for (int i = 0, k = 0; i < nPar; i++) {
347  for (int j = 0; j < nPar; j++, k++) {
348  mLSMmatrixFull[k] = A(i, j);
349  }
350  }
351  }
352 
353  {
354  TDecompBK bk(Z, 0);
355  if (!bk.Invert(Z)) {
356  ret = storeError(-5, "Spline1DHelper::setSpline: internal error - can not invert the matrix");
357  Z.Zero();
358  }
359  mLSMmatrixSderivatives.resize(nKnots * nKnots);
360  for (int i = 0, k = 0; i < nKnots; i++) {
361  for (int j = 0; j < nKnots; j++, k++) {
362  mLSMmatrixSderivatives[k] = Z(i, j);
363  }
364  }
365  }
366 
367  return ret;
368 }

370 template <typename DataT>
371 void Spline1DHelper<DataT>::approximateFunction(
372  DataT* Sparameters,
373  const double DataPointF[/*getNumberOfDataPoints() x nFdim*/]) const
374 {
376 
377  const int nPar = 2 * mSpline.getNumberOfKnots();
378  double b[nPar];
379  for (int idim = 0; idim < mFdimensions; idim++) {
380  for (int i = 0; i < nPar; i++) {
381  b[i] = 0.;
382  }
383  for (int i = 0; i < getNumberOfDataPoints(); ++i) {
384  const DataPoint& p = mDataPoints[i];
385  double* bb = &(b[p.iKnot * 2]);
386  double f = (double)DataPointF[i * mFdimensions + idim];
387  bb[0] += f * p.cS0;
388  bb[1] += f * p.cZ0;
389  bb[2] += f * p.cS1;
390  bb[3] += f * p.cZ1;
391  }
392 
393  const double* row = mLSMmatrixFull.data();
394 
395  for (int i = 0; i < nPar; i++, row += nPar) {
396  double s = 0.;
397  for (int j = 0; j < nPar; j++) {
398  s += row[j] * b[j];
399  }
400  Sparameters[i * mFdimensions + idim] = (float)s;
401  }
402  }
403 }

429 template <typename DataT>
430 void Spline1DHelper<DataT>::approximateDerivatives(
431  DataT* Sparameters, const double DataPointF[/*getNumberOfDataPoints() x nFdim*/]) const
432 {
436 
437  const int nKnots = mSpline.getNumberOfKnots();
438  const int Ndim = mFdimensions;
439  double b[nKnots * mFdimensions];
440  for (int i = 0; i < nKnots * Ndim; i++) {
441  b[i] = 0.;
442  }
443 
444  for (int i = 0; i < getNumberOfDataPoints(); ++i) {
445  const DataPoint& p = mDataPoints[i];
446  if (p.isKnot) {
447  continue;
448  }
449  for (int d = 0; d < Ndim; d++) {
450  double f = (double)DataPointF[i * Ndim + d];
451  b[(p.iKnot + 0) * Ndim + d] += f * p.cZ0;
452  b[(p.iKnot + 1) * Ndim + d] += f * p.cZ1;
453  }
454  }
455 
456  const double* row = mLSMmatrixSvalues.data();
457  for (int i = 0; i < nKnots; ++i, row += nKnots) {
458  double s[Ndim];
459  for (int d = 0; d < Ndim; d++) {
460  s[d] = 0.;
461  }
462  for (int j = 0; j < nKnots; ++j) {
463  for (int d = 0; d < Ndim; d++) {
464  s[d] += row[j] * Sparameters[2 * j * Ndim + d];
465  }
466  }
467  for (int d = 0; d < Ndim; d++) {
468  b[i * Ndim + d] -= s[d];
469  }
470  }
471 
472  row = mLSMmatrixSderivatives.data();
473  for (int i = 0; i < nKnots; ++i, row += nKnots) {
474  double s[Ndim];
475  for (int d = 0; d < Ndim; d++) {
476  s[d] = 0.;
477  }
478  for (int j = 0; j < nKnots; ++j) {
479  for (int d = 0; d < Ndim; d++) {
480  s[d] += row[j] * b[j * Ndim + d];
481  }
482  }
483  for (int d = 0; d < Ndim; d++) {
484  Sparameters[(2 * i + 1) * Ndim + d] = (float)s[d];
485  }
486  }

489 #ifndef GPUCA_ALIROOT_LIB
490 template <typename DataT>
491 int Spline1DHelper<DataT>::test(const bool draw, const bool drawDataPoints)
492 {
493  using namespace std;
494 
495  // input function F
496 
497  const int Ndim = 5;
498  const int Fdegree = 4;
499  double Fcoeff[Ndim][2 * (Fdegree + 1)];
500 
501  auto F = [&](double x, double f[]) -> void {
502  double cosx[Fdegree + 1], sinx[Fdegree + 1];
503  double xi = 0;
504  for (int i = 0; i <= Fdegree; i++, xi += x) {
505  GPUCommonMath::SinCosd(xi, sinx[i], cosx[i]);
506  }
507  for (int dim = 0; dim < Ndim; dim++) {
508  f[dim] = 0; // Fcoeff[0]/2;
509  for (int i = 1; i <= Fdegree; i++) {
510  f[dim] += Fcoeff[dim][2 * i] * cosx[i] + Fcoeff[dim][2 * i + 1] * sinx[i];
511  }
512  }
513  };
514 
515  TCanvas* canv = nullptr;
516  TNtuple* nt = nullptr;
517  TNtuple* knots = nullptr;
518 
519  auto ask = [&]() -> bool {
520  if (!canv) {
521  return 0;
522  }
523  canv->Update();
524  cout << "type 'q ' to exit" << endl;
525  std::string str;
526  std::getline(std::cin, str);
527  return (str != "q" && str != ".q");
528  };
529 
530  std::cout << "Test 1D interpolation with the compact spline" << std::endl;
531 
532  int nTries = 100;
533 
534  if (draw) {
535  canv = new TCanvas("cQA", "Spline1D QA", 1000, 600);
536  nTries = 10000;
537  }
538 
539  double statDf1 = 0;
540  double statDf2 = 0;
541  double statDf1D = 0;
542  double statN = 0;
543 
544  int seed = 1;
545 
546  for (int itry = 0; itry < nTries; itry++) {
547 
548  // init random F
549  for (int dim = 0; dim < Ndim; dim++) {
550  gRandom->SetSeed(seed++);
551  for (int i = 0; i < 2 * (Fdegree + 1); i++) {
552  Fcoeff[dim][i] = gRandom->Uniform(-1, 1);
553  }
554  }
555 
556  // spline
557 
558  int nKnots = 4;
559  const int uMax = nKnots * 3;
560 
561  Spline1D<DataT, Ndim> spline1;
562  int knotsU[nKnots];
563 
564  do { // set knots randomly
565  knotsU[0] = 0;
566  double du = 1. * uMax / (nKnots - 1);
567  for (int i = 1; i < nKnots; i++) {
568  knotsU[i] = (int)(i * du); // + gRandom->Uniform(-du / 3, du / 3);
569  }
570  knotsU[nKnots - 1] = uMax;
571  spline1.recreate(nKnots, knotsU);
572  if (nKnots != spline1.getNumberOfKnots()) {
573  cout << "warning: n knots changed during the initialisation " << nKnots
574  << " -> " << spline1.getNumberOfKnots() << std::endl;
575  continue;
576  }
577  } while (0);
578 
579  std::string err = FlatObject::stressTest(spline1);
580  if (!err.empty()) {
581  cout << "error at FlatObject functionality: " << err << endl;
