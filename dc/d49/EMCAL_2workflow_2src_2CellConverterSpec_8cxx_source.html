26 void CellConverterSpec::init(framework::InitContext& ctx)
27 {
28  LOG(debug) << "[EMCALCellConverter - init] Initialize converter " << (mPropagateMC ? "with" : "without") << " MC truth container";
29 
30  if (!mGeometry) {
31  mGeometry = o2::emcal::Geometry::GetInstanceFromRunNumber(223409);
32  }
33  if (!mGeometry) {
34  LOG(error) << "Failure accessing geometry";
35  }
36 
37  auto fitmethod = ctx.options().get<std::string>("fitmethod");
38  if (fitmethod == "standard") {
39  LOG(info) << "Using standard raw fitter";
40  mRawFitter = std::unique_ptr<o2::emcal::CaloRawFitter>(new o2::emcal::CaloRawFitterStandard);
41  } else if (fitmethod == "gamma2") {
42  LOG(info) << "Using gamma2 raw fitter";
43  mRawFitter = std::unique_ptr<o2::emcal::CaloRawFitter>(new o2::emcal::CaloRawFitterGamma2);
44  }
45  mRawFitter->setAmpCut(0.);
46  mRawFitter->setL1Phase(0.);
47 }

49 void CellConverterSpec::run(framework::ProcessingContext& ctx)
50 {
51  LOG(debug) << "[EMCALCellConverter - run] called";
52  const double CONVADCGEV = 0.016; // Conversion from ADC counts to energy: E = 16 MeV / ADC
53 
54  mOutputCells.clear();
55  mOutputTriggers.clear();
56  auto digitsAll = ctx.inputs().get<gsl::span<o2::emcal::Digit>>("digits");
57  auto triggers = ctx.inputs().get<gsl::span<o2::emcal::TriggerRecord>>("triggers");
58  LOG(debug) << "[EMCALCellConverter - run] Received " << digitsAll.size() << " digits from " << triggers.size() << " trigger ...";
59  int currentstart = mOutputCells.size(), ncellsTrigger = 0;
60  for (const auto& trg : triggers) {
61  if (!trg.getNumberOfObjects()) {
62  mOutputTriggers.emplace_back(trg.getBCData(), trg.getTriggerBits(), currentstart, ncellsTrigger);
63  continue;
64  }
65  gsl::span<const o2::emcal::Digit> digits(digitsAll.data() + trg.getFirstEntry(), trg.getNumberOfObjects());
66 
67  for (const auto& srucont : digitsToBunches(digits)) {
68 
69  if (srucont.mSRUid == 21 || srucont.mSRUid == 22 || srucont.mSRUid == 36 || srucont.mSRUid == 39) {
70  continue;
71  }
72 
73  for (const auto& [tower, channelData] : srucont.mChannelsData) {
74 
75  // define the conatiner for the fit results, and perform the raw fitting using the stadnard raw fitter
76  CaloFitResults fitResults;
77  try {
78  fitResults = mRawFitter->evaluate(channelData.mChannelsBunches);
79 
80  if (fitResults.getAmp() < 0) {
81  fitResults.setAmp(0.);
82  }
83  if (fitResults.getTime() < 0) {
84  fitResults.setTime(0.);
85  }
86  } catch (CaloRawFitter::RawFitterError_t& fiterror) {
87  LOG(error) << "Failure in raw fitting: " << CaloRawFitter::createErrorMessage(fiterror);
88  }
89 
90  mOutputCells.emplace_back(tower, fitResults.getAmp() * CONVADCGEV, fitResults.getTime(), channelData.mChanType);
91  ncellsTrigger++;
92  }
93  }
94  mOutputTriggers.emplace_back(trg.getBCData(), trg.getTriggerBits(), currentstart, ncellsTrigger);
95  currentstart = mOutputCells.size();
96  ncellsTrigger = 0;
97  }
98  LOG(debug) << "[EMCALCellConverter - run] Writing " << mOutputCells.size() << " cells ...";
99  ctx.outputs().snapshot(o2::framework::Output{"EMC", "CELLS", 0, o2::framework::Lifetime::Timeframe}, mOutputCells);
100  ctx.outputs().snapshot(o2::framework::Output{"EMC", "CELLSTRGR", 0, o2::framework::Lifetime::Timeframe}, mOutputTriggers);
101  if (mPropagateMC) {
102  // copy mc truth container without modification
103  // as indexing doesn't change
104  auto truthcont = ctx.inputs().get<o2::dataformats::MCTruthContainer<o2::emcal::MCLabel>*>("digitsmctr");
105  ctx.outputs().snapshot(o2::framework::Output{"EMC", "CELLSMCTR", 0, o2::framework::Lifetime::Timeframe}, *truthcont);
106  }
107 }

109 std::vector<o2::emcal::SRUBunchContainer> CellConverterSpec::digitsToBunches(gsl::span<const o2::emcal::Digit> digits)
110 {
111 
112  std::vector<o2::emcal::SRUBunchContainer> sruBunchContainer;
113  std::vector<o2::emcal::DigitContainerPerSRU> sruDigitContainer;
114 
115  for (auto iddl = 0; iddl < 40; iddl++) {
116  o2::emcal::SRUBunchContainer srucontBunch;
117  srucontBunch.mSRUid = iddl;
118  sruBunchContainer.push_back(srucontBunch);
119 
120  o2::emcal::DigitContainerPerSRU srucontDigits;
121  srucontDigits.mSRUid = iddl;
122  sruDigitContainer.push_back(srucontDigits);
123  }
124 
125  std::vector<const o2::emcal::Digit*>* bunchDigits;
126  int lasttower = -1;
127  for (auto& dig : digits) {
128  auto tower = dig.getTower();
129  if (tower != lasttower) {
130  lasttower = tower;
131  if (tower > 20000) {
132  std::cout << "Wrong cell ID " << tower << std::endl;
133  }
134 
135  auto onlineindices = mGeometry->getOnlineID(tower);
136  int sruID = std::get<0>(onlineindices);
137 
138  auto towerdata = sruDigitContainer[sruID].mChannelsDigits.find(tower);
139  if (towerdata == sruDigitContainer[sruID].mChannelsDigits.end()) {
140  sruDigitContainer[sruID].mChannelsDigits[tower] = {dig.getType(), std::vector<const o2::emcal::Digit*>(o2::emcal::constants::EMCAL_MAXTIMEBINS)};
141  bunchDigits = &(sruDigitContainer[sruID].mChannelsDigits[tower].mChannelDigits);
142  memset(bunchDigits->data(), 0, sizeof(o2::emcal::Digit*) * o2::emcal::constants::EMCAL_MAXTIMEBINS);
143  } else {
144  bunchDigits = &(towerdata->second.mChannelDigits);
145  }
146  }
147 
148  // Get time sample of the digit:
149  // Digitizer stores the time sample in ns, needs to be converted to time sample dividing
150  // by the length of the time sample
151  auto timesample = int(dig.getTimeStamp() / emcal::constants::EMCAL_TIMESAMPLE);
152  if (timesample >= o2::emcal::constants::EMCAL_MAXTIMEBINS) {
153  LOG(error) << "Digit time sample " << timesample << " outside range [0," << o2::emcal::constants::EMCAL_MAXTIMEBINS << "]";
154  continue;
155  }
156  (*bunchDigits)[timesample] = &dig;
157  }
158 
159  for (auto srucont : sruDigitContainer) {
160 
161  if (srucont.mSRUid == 21 || srucont.mSRUid == 22 || srucont.mSRUid == 36 || srucont.mSRUid == 39) {
162  continue;
163  }
164 
165  for (const auto& [tower, channelDigits] : srucont.mChannelsDigits) {
166 
167  std::vector<o2::emcal::Bunch> rawbunches;
168  for (auto& bunch : findBunches(channelDigits.mChannelDigits)) {
169  rawbunches.emplace_back(bunch.mADCs.size(), bunch.mStarttime);
170  for (auto adc : bunch.mADCs) {
171  rawbunches.back().addADC(adc);
172  }
173  }
174 
175  sruBunchContainer[srucont.mSRUid].mChannelsData[tower] = {channelDigits.mChanType, rawbunches};
176  }
177  }
178 
179  return sruBunchContainer;
180 }

182 std::vector<o2::emcal::AltroBunch> CellConverterSpec::findBunches(const std::vector<const o2::emcal::Digit*>& channelDigits)
183 {
184  std::vector<AltroBunch> result;
185  AltroBunch currentBunch;
186  bool bunchStarted = false;
187  // Digits in ALTRO bunch in time-reversed order
188  int itime;
189  for (itime = channelDigits.size() - 1; itime >= 0; itime--) {
190  auto dig = channelDigits[itime];
191  if (!dig) {
192  if (bunchStarted) {
193  // we have a bunch which is started and needs to be closed
194  // check if the ALTRO bunch has a minimum amount of 3 ADCs
195  if (currentBunch.mADCs.size() >= 3) {
196  // Bunch selected, set start time and push to bunches
197  currentBunch.mStarttime = itime + 1;
198  result.push_back(currentBunch);
199  currentBunch = AltroBunch();
200  bunchStarted = false;
201  }
202  }
203  continue;
204  }
205  int adc = dig->getAmplitudeADC();
206  if (adc < 3) {
207  // ADC value below threshold
208  // in case we have an open bunch it needs to be stopped bunch
209  // Set the start time to the time sample of previous (passing) digit
210  if (bunchStarted) {
211  // check if the ALTRO bunch has a minimum amount of ADCs
212  if (currentBunch.mADCs.size() >= 3) {
213  // Bunch selected, set start time and push to bunches
214  currentBunch.mStarttime = itime + 1;
215  result.push_back(currentBunch);
216  currentBunch = AltroBunch();
217  bunchStarted = false;
218  }
219  }
220  }
221  // Valid ADC value, if the bunch is closed we start a new bunch
222  if (!bunchStarted) {
223  bunchStarted = true;
224  }
225  currentBunch.mADCs.emplace_back(adc);
226  }
227  // if we have a last bunch set time start time to the time bin of teh previous digit
228  if (bunchStarted) {
229  if (currentBunch.mADCs.size() >= 3) {
230  currentBunch.mStarttime = itime + 1;
231  result.push_back(currentBunch);
232  }
233  }
234  return result;
235 }

237 o2::framework::DataProcessorSpec o2::emcal::reco_workflow::getCellConverterSpec(bool propagateMC)
238 {
239  std::vector<o2::framework::InputSpec> inputs;
240  std::vector<o2::framework::OutputSpec> outputs;
241  inputs.emplace_back("digits", o2::header::gDataOriginEMC, "DIGITS", 0, o2::framework::Lifetime::Timeframe);

248  }
249  return o2::framework::DataProcessorSpec{"EMCALCellConverterSpec",
250  inputs,
251  outputs,
252  o2::framework::adaptFromTask<o2::emcal::reco_workflow::CellConverterSpec>(propagateMC),
253  o2::framework::Options{
