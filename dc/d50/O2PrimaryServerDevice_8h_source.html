72  {

77  // init magnetic field as it might be needed by the generator
78  if (TGeoGlobalMagField::Instance()->GetField() == nullptr) {

84  // look if we find a cached instances of Pythia8 or external generators in order to avoid
85  // (long) initialization times.
86  // This is evidently a bit weak, as generators might need reconfiguration (to be treated later).
87  // For now, we'd like to allow for fast switches between say a pythia8 instance and reading from kinematics
88  // to continue an already started simulation.
89  //
90  // Not using cached instances for external kinematics since these might change input filenames etc.
91  // and are in any case quickly setup.
92  mPrimGen = nullptr;
93  if (conf.getGenerator().compare("extkin") != 0 || conf.getGenerator().compare("extkinO2") != 0) {
94  auto iter = mPrimGeneratorCache.find(conf.getGenerator());
95  if (iter != mPrimGeneratorCache.end()) {
96  mPrimGen = iter->second;
97  LOG(info) << "Found cached generator for " << conf.getGenerator();
98  }
99  }
100 
101  if (mPrimGen == nullptr) {
102  mPrimGen = new o2::eventgen::PrimaryGenerator;
103  o2::eventgen::GeneratorFactory::setPrimaryGenerator(conf, mPrimGen);
104 
105  auto embedinto_filename = conf.getEmbedIntoFileName();
106  if (!embedinto_filename.empty()) {
107  mPrimGen->embedInto(embedinto_filename);
108  }
109 
110  mPrimGen->Init();
111 
112  mPrimGeneratorCache[conf.getGenerator()] = mPrimGen;
113  }
114  mPrimGen->SetEvent(&mEventHeader);
115 
116  LOG(info) << "Generator initialization took " << timer.CpuTime() << "s";
117  if (mMaxEvents > 0) {
118  generateEvent(); // generate a first event
119  }
120  }

146  // launches a thread that listens for status requests from outside asynchronously
147  void launchInfoThread()
148  {
149  static std::vector<std::thread> threads;
150  LOG(info) << "LAUNCHING STATUS THREAD";
151  auto lambda = [this]() {
152  while (mState != O2PrimaryServerState::Stopped) {
153  auto& channel = fChannels.at("o2sim-primserv-info").at(0);
154  if (!channel.IsValid()) {
155  LOG(error) << "channel primserv-info not valid";
156  }
157  std::unique_ptr<FairMQMessage> request(channel.NewSimpleMessage(-1));
158  int timeout = 100; // 100ms --> so as not to block and allow for proper termination of this thread
159  if (channel.Receive(request, timeout) > 0) {
160  LOG(info) << "INFO REQUEST RECEIVED";
161  if (*(int*)(request->GetData()) == (int)O2PrimaryServerInfoRequest::Status) {
162  LOG(info) << "Received status request";
163  // request needs to be a simple enum of type O2PrimaryServerInfoRequest
164  std::unique_ptr<FairMQMessage> reply(channel.NewSimpleMessage((int)mState.load()));
165  if (channel.Send(reply) > 0) {
166  LOG(info) << "Send status successful";
167  }
168  } else if (*(int*)request->GetData() == (int)O2PrimaryServerInfoRequest::Config) {
169  HandleConfigRequest(channel);
170  } else {
171  LOG(fatal) << "UNKNOWN REQUEST";
172  std::unique_ptr<FairMQMessage> reply(channel.NewSimpleMessage(404));
173  channel.Send(reply);
174  }
175  }
176  }
177  mInfoThreadStopped = true;
178  };
179  threads.push_back(std::thread(lambda));
180  threads.back().detach();
181  }

324  bool ConditionalRun() override
325  {
326  // we might come here in IDLE mode
327  if (mState == O2PrimaryServerState::Idle) {
328  if (mWaitingControlInput.load() == 0) {
329  if (mControlThread.joinable()) {
330  mControlThread.join();
331  }
332  mControlThread = std::thread(&O2PrimaryServerDevice::waitForControlInput, this);
333  }
334  }
335 
336  auto& channel = fChannels.at("primary-get").at(0);
337  PrimaryChunkRequest requestpayload;
338  std::unique_ptr<FairMQMessage> request(channel.NewSimpleMessage(requestpayload));
339  auto bytes = channel.Receive(request);
340  if (bytes < 0) {
341  LOG(error) << "Some error/interrupt occurred on socket during receive";
342  if (NewStatePending()) { // new state is typically pending if (term) signal was received
343  WaitForNextState();
344  // ask ourselves for termination of this loop
345  stateTransition(O2PrimaryServerState::Stopped, "CONDRUN");
346  }
347  return false;
348  }
349 
350  TStopwatch timer;
351  timer.Start();
352  auto& r = *((PrimaryChunkRequest*)(request->GetData()));
353  LOG(info) << "PARTICLE REQUEST IN STATE " << PrimStateToString[(int)mState.load()] << " from " << r.workerid << ":" << r.requestid;
354 
355  auto prestate = mState.load();
356  auto more = HandleRequest(request, 0, channel);
357  if (!more) {
358  if (mAsService) {
359  if (prestate == O2PrimaryServerState::ReadyToServe || prestate == O2PrimaryServerState::WaitingEvent) {
360  stateTransition(O2PrimaryServerState::Idle, "CONDRUN");
361  }
362  } else {
363  stateTransition(O2PrimaryServerState::Stopped, "CONDRUN");
364  }
365  }
366  timer.Stop();
367  auto time = timer.CpuTime();
368  LOG(info) << "COND-RUN TOOK " << time << " s";
369  return mState != O2PrimaryServerState::Stopped;
370  }

381  bool HandleRequest(FairMQMessagePtr& request, int /*index*/, FairMQChannel& channel)
382  {
383  // LOG(debug) << "GOT A REQUEST WITH SIZE " << request->GetSize();
384  // std::string requeststring(static_cast<char*>(request->GetData()), request->GetSize());
385  // LOG(info) << "NORMAL REQUEST STRING " << requeststring;
386  bool workavailable = true;
387  if (mEventCounter >= mMaxEvents && mNeedNewEvent) {
388  workavailable = false;
389  }
390  if (!(mState == O2PrimaryServerState::ReadyToServe || mState == O2PrimaryServerState::WaitingEvent)) {
391  // send a zero answer
392  workavailable = false;
393  }
394 
395  PrimaryChunkAnswer header{mState, workavailable};
396  FairMQParts reply;
397  std::unique_ptr<FairMQMessage> headermsg(channel.NewSimpleMessage(header));
398  reply.AddPart(std::move(headermsg));
399 
400  LOG(info) << "Received request for work " << mEventCounter << " " << mMaxEvents << " " << mNeedNewEvent << " available " << workavailable;
401  if (mNeedNewEvent) {
402  // we need a newly generated event now
403  if (mGeneratorThread.joinable()) {
404  try {
405  mGeneratorThread.join();
406  } catch (std::exception const& e) {
407  LOG(warn) << "Exception during thread join ..ignoring";
408  }
409  }
410  mNeedNewEvent = false;
411  mPartCounter = 0;
412  mEventCounter++;
413  }
414 
415  auto& prims = mStack->getPrimaries();
416  auto numberofparts = (int)std::ceil(prims.size() / (1. * mChunkGranularity));
417  // number of parts should be at least 1 (even if empty)
418  numberofparts = std::max(1, numberofparts);
419 
420  LOG(info) << "Have " << prims.size() << " " << numberofparts;
421 
422  o2::data::PrimaryChunk m;
423  o2::data::SubEventInfo i;
424  i.eventID = workavailable ? mEventCounter : -1;
425  i.maxEvents = mMaxEvents;
426  i.part = mPartCounter + 1;
427  i.nparts = numberofparts;
428  i.seed = mEventCounter + mInitialSeed;
429  i.index = m.mParticles.size();
430  i.mMCEventHeader = mEventHeader;
431  m.mSubEventInfo = i;
432 
433  if (workavailable) {
434  int endindex = prims.size() - mPartCounter * mChunkGranularity;
435  int startindex = prims.size() - (mPartCounter + 1) * mChunkGranularity;
436  LOG(info) << "indices " << startindex << " " << endindex;
437 
438  if (startindex < 0) {
439  startindex = 0;
440  }
441  if (endindex < 0) {
442  endindex = 0;
443  }
444 
445  for (int index = startindex; index < endindex; ++index) {
446  m.mParticles.emplace_back(prims[index]);
447  }
448 
449  LOG(info) << "Sending " << m.mParticles.size() << " particles";
450  LOG(info) << "treating ev " << mEventCounter << " part " << i.part << " out of " << i.nparts;
451 
452  // feedback to driver if new event started
453  if (mPipeToDriver != -1 && i.part == 1 && workavailable) {
454  if (write(mPipeToDriver, &mEventCounter, sizeof(mEventCounter))) {
455  }
456  }
457 
458  mPartCounter++;
459  if (mPartCounter == numberofparts) {
460  mNeedNewEvent = true;
461  // start generation of a new event
462  mGeneratorThread = std::thread(&O2PrimaryServerDevice::generateEvent, this);
463  }
464 
465  TMessage* tmsg = new TMessage(kMESS_OBJECT);
466  tmsg->WriteObjectAny((void*)&m, TClass::GetClass("o2::data::PrimaryChunk"));
467 
468  auto free_tmessage = [](void* data, void* hint) { delete static_cast<TMessage*>(hint); };
469 
470  std::unique_ptr<FairMQMessage> message(channel.NewMessage(tmsg->Buffer(), tmsg->BufferSize(), free_tmessage, tmsg));
471 
472  reply.AddPart(std::move(message));
473  }
474 
475  // send answer
476  TStopwatch timer;
477  timer.Start();
478  auto code = Send(reply, "primary-get", 0, 5000); // we introduce timeout in order not to block other requests
479  timer.Stop();
480  auto time = timer.CpuTime();
481  if (code > 0) {
482  LOG(info) << "Reply send in " << time << "s";
483  return workavailable;
484  } else {
485  LOG(warn) << "Sending process had problems. Return code : " << code << " time " << time << "s";
486  }
487  return false; // -> error should not get here
488  }

496  void waitForControlInput()
497  {
498  mWaitingControlInput.store(1);
499  stateTransition(O2PrimaryServerState::Idle, "CONTROL");
500 
501  o2::simpubsub::publishMessage(fChannels["primary-notifications"].at(0), o2::simpubsub::simStatusString("PRIMSERVER", "STATUS", "AWAITING INPUT"));
502  // this means we are idling
503 
504  auto factory = FairMQTransportFactory::CreateTransportFactory("zeromq");
505  auto channel = FairMQChannel{"o2sim-control", "sub", factory};
506  auto controlsocketname = getenv("ALICE_O2SIMCONTROL");
507  channel.Connect(std::string(controlsocketname));
508  channel.Validate();
509  std::unique_ptr<FairMQMessage> reply(channel.NewMessage());
510 
511  bool ok = false;
512 
513  LOG(info) << "WAITING FOR CONTROL INPUT";
514  if (channel.Receive(reply) > 0) {
515  stateTransition(O2PrimaryServerState::Initializing, "CONTROL");
516  auto data = reply->GetData();
517  auto size = reply->GetSize();
518 
519  std::string command(reinterpret_cast<char const*>(data), size);
520  LOG(info) << "message: " << command;
521 
522  o2::conf::SimReconfigData reconfig;
523  o2::conf::parseSimReconfigFromString(command, reconfig);
524  LOG(info) << "Processing " << reconfig.nEvents << " new events";
525  try {
526  LOG(info) << "REINIT START";
527  ok = ReInit(reconfig);
528  LOG(info) << "REINIT DONE";
529  } catch (std::exception e) {
530  LOG(info) << "Exception during reinit";
531  }
532  } else {
533  LOG(info) << "NOTHING RECEIVED";
534  }
535  if (ok) {
536  stateTransition(O2PrimaryServerState::ReadyToServe, "CONTROL");
537  } else {
538  stateTransition(O2PrimaryServerState::Stopped, "CONTROL");
539  }
540  mWaitingControlInput.store(0);
541  }
