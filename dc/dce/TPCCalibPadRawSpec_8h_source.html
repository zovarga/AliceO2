74  mRawReader.setADCDataCallback([this](const PadROCPos& padROCPos, const CRU& cru, const gsl::span<const uint32_t> data) -> int {
75  const int timeBins = mCalibration.update(padROCPos, cru, data);

83  return true;

87  mUseOldSubspec = ic.options().get<bool>("use-old-subspec");

89  mDirectFileDump = ic.options().get<bool>("direct-file-dump");

95  void run(o2::framework::ProcessingContext& pc) final
96  {
97  // in case the maximum number of events was reached don't do further processing
98  if (mReadyToQuit) {
99  return;
100  }
101 
102  if (mUseDigits) {
103  std::array<std::vector<Digit>, Sector::MAXSECTOR> digits;
104  copyDigits(pc.inputs(), digits);
105  mCalibration.setDigits(&digits);
106  mCalibration.processEvent();
107  } else {
108  auto& reader = mRawReader.getReaders()[0];
109  calib_processing_helper::processRawData(pc.inputs(), reader, mUseOldSubspec, mSectors);
110  mCalibration.endEvent();
111  mCalibration.endReader();
112  }
113 
114  mCalibration.incrementNEvents();
115  const auto nTFs = mCalibration.getNumberOfProcessedEvents();
116  LOGP(info, "Number of processed TFs: {} ({})", nTFs, mMaxEvents);
117 
118  if ((mPublishAfter && (nTFs % mPublishAfter) == 0)) {
119  LOGP(info, "Publishing after {} TFs", nTFs);
120  mCalibration.analyse();
121  dumpCalibData();
122  sendOutput(pc.outputs());
123  }
124 
125  if (mMaxEvents && (nTFs >= mMaxEvents) && !mCalibDumped) {
126  LOGP(info, "Maximm number of TFs reached ({}), no more processing will be done", mMaxEvents);
127  mReadyToQuit = true;
128  mCalibration.analyse();
129  dumpCalibData();
130  if (mForceQuit) {
131  pc.services().get<ControlService>().endOfStream();
132  pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
133  } else {
134  // pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
135  }
136  }
137  }
138 
139  void endOfStream(o2::framework::EndOfStreamContext& ec) final

142  mCalibration.analyse();
143  dumpCalibData();
144  sendOutput(ec.outputs());
145  ec.services().get<ControlService>().readyToQuit(QuitRequest::Me);
146  }

172  output.snapshot(Output{gDataOriginTPC, "CLBPART", subSpec}, *image.get());

174  }

189  void copyDigits(InputRecord& inputs, std::array<std::vector<Digit>, Sector::MAXSECTOR>& digits)
190  {
191  std::vector<InputSpec> filter = {
192  {"check", ConcreteDataTypeMatcher{"TPC", "DIGITS"}, Lifetime::Timeframe},
193  };
194  for (auto const& inputRef : InputRecordWalker(inputs)) {
195  auto const* sectorHeader = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(inputRef);
196  if (sectorHeader == nullptr) {
197  LOG(error) << "sector header missing on header stack for input on " << inputRef.spec->binding;
198  continue;
199  }
200  const int sector = sectorHeader->sector();
201  digits[sector] = inputs.get<std::vector<o2::tpc::Digit>>(inputRef);
202  }
203  }

206 template <typename... Args>
207 AlgorithmSpec getRawDevice(CDBType calibType, Args... args)
208 {
209  if (calibType == CDBType::CalPedestalNoise) {
210  return adaptFromTask<TPCCalibPedestalDevice<CalibPedestal>>(calibType, args...);
211  } else if (calibType == CDBType::CalPulser) {
212  return adaptFromTask<TPCCalibPedestalDevice<CalibPulser>>(calibType, args...);
213  } else if (calibType == CDBType::CalCE) {
214  return adaptFromTask<TPCCalibPedestalDevice<CalibPulser>>(calibType, args...);
215  } else {
216  return AlgorithmSpec{};
217  }

220 DataProcessorSpec getTPCCalibPadRawSpec(const std::string inputSpec, uint32_t ilane = 0, std::vector<int> sectors = {}, uint32_t publishAfterTFs = 0, CDBType rawType = CDBType::CalPedestalNoise)
221 {
222  const bool useDigitsAsInput = inputSpec.find("DIGITS") != std::string::npos;
223 
224  std::vector<o2::framework::OutputSpec> outputs;
225  outputs.emplace_back(ConcreteDataTypeMatcher{gDataOriginTPC, "CLBPART"});
226  outputs.emplace_back(ConcreteDataTypeMatcher{gDataOriginTPC, "CLBPARTINFO"});
227 
228  AlgorithmSpec spec;
229 
230  const auto id = fmt::format("calib-tpc-raw-{:02}", ilane);
231  return DataProcessorSpec{
232  id.data(),
233  select(inputSpec.data()),
234  outputs,
235  AlgorithmSpec{getRawDevice(rawType, ilane, sectors, publishAfterTFs, useDigitsAsInput)},
236  Options{
237  {"max-events", VariantType::Int, 0, {"maximum number of events to process"}},
238  {"use-old-subspec", VariantType::Bool, false, {"use old subsecifiation definition"}},
239  {"force-quit", VariantType::Bool, false, {"force quit after max-events have been reached"}},
240  {"direct-file-dump", VariantType::Bool, false, {"directly dump calibration to file"}},
241  } // end Options
242  }; // end DataProcessorSpec
