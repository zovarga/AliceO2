54 WorkflowSpec defineDataProcessing(ConfigContext const& config)
55 {
56 
57  bool verbose = config.options().get<bool>("verbose");
58  bool testMode = config.options().get<bool>("test-mode");
59  std::string detectorList = config.options().get<std::string>("detector-list");
60  o2::conf::ConfigurableParam::updateFromString(config.options().get<std::string>("configKeyValues"));
61  std::string url = config.options().get<std::string>("ccdb-url");
62 
63  std::unordered_map<DPID, o2h::DataDescription> dpid2DataDesc;
64 
65  if (testMode) {
66  DPID dpidtmp;
67  DPID::FILL(dpidtmp, "ADAPOS_LG/TEST_000100", DeliveryType::RAW_STRING);
68  dpid2DataDesc[dpidtmp] = "COMMON"; // i.e. this will go to {DCS/COMMON/0} OutputSpec
69  DPID::FILL(dpidtmp, "ADAPOS_LG/TEST_000110", DeliveryType::RAW_STRING);
70  dpid2DataDesc[dpidtmp] = "COMMON";
71  DPID::FILL(dpidtmp, "ADAPOS_LG/TEST_000200", DeliveryType::RAW_STRING);
72  dpid2DataDesc[dpidtmp] = "COMMON1";
73  DPID::FILL(dpidtmp, "ADAPOS_LG/TEST_000240", DeliveryType::RAW_INT);
74  dpid2DataDesc[dpidtmp] = "COMMON1";
75  }
76 
77  else {
78  auto& mgr = CcdbManager::instance();
79  mgr.setURL(url); // http://ccdb-test.cern.ch:8080 or http://localhost:8080 for a local installation
80  long ts = std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count();
81  std::regex re("[\\s,-]+");
82  std::sregex_token_iterator it(detectorList.begin(), detectorList.end(), re, -1);
83  std::sregex_token_iterator reg_end;
84  for (; it != reg_end; ++it) {
85  LOG(info) << "DCS DPs configured for detector " << it->str();
86  std::unordered_map<DPID, std::string>* dpid2Det = mgr.getForTimeStamp<std::unordered_map<DPID, std::string>>(it->str() + "/Config/DCSDPconfig", ts);
87  for (auto& el : *dpid2Det) {
88  o2::header::DataDescription tmpd;
89  tmpd.runtimeInit(el.second.c_str(), el.second.size());
90  dpid2DataDesc[el.first] = tmpd;
91  }
92  }
93  }
94 
95  // RS: here we should complete the attribution of different DPs to different outputs
96  // ...
97 
98  // now collect all required outputs to define OutputSpecs for specifyExternalFairMQDeviceProxy
99  std::unordered_map<o2h::DataDescription, int, std::hash<o2h::DataDescription>> outMap;
100  for (auto itdp : dpid2DataDesc) {
101  outMap[itdp.second]++;
102  }
103 
104  Outputs dcsOutputs;
105  for (auto itout : outMap) {
106  dcsOutputs.emplace_back("DCS", itout.first, 0, Lifetime::Timeframe);
107  }
108 
109  DataProcessorSpec dcsProxy = specifyExternalFairMQDeviceProxy(
110  "dcs-proxy",
111  std::move(dcsOutputs),
112  "type=pull,method=connect,address=tcp://aldcsadaposactor:60000,rateLogging=1,transport=zeromq",
113  dcs2dpl(dpid2DataDesc, 0, 1, verbose));
114 
115  WorkflowSpec workflow;
116  workflow.emplace_back(dcsProxy);
117  return workflow;
118 }
