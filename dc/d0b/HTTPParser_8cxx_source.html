27 void memmask(char* dst, char const* src, size_t size, uint32_t mask)
28 {
29  if (mask) {
30  char* m = (char*)&mask;
31  for (size_t len = 0; len < size; ++len) {
32  *dst++ = *src++ ^ m[len % 4];
33  }
34  } else {
35  memcpy(dst, src, size);
36  }
37 }

48 void encode_websocket_frames(std::vector<uv_buf_t>& outputs, char const* src, size_t size, WebSocketOpCode opcode, uint32_t mask)
49 {
50  void* finalHeader;
51  size_t headerSize;
52  char* buffer = nullptr;
53  char* startPayload = nullptr;
54  std::vector<uv_buf_t> result;
55 
56  if (size < 126) {
57  headerSize = sizeof(WebSocketFrameTiny);
58  buffer = (char*)malloc(headerSize + size);
59  WebSocketFrameTiny* header = (WebSocketFrameTiny*)buffer;
60  memset(buffer, 0, headerSize);
61  header->len = size;
62  } else if (size < 1 << 16) {
63  headerSize = sizeof(WebSocketFrameShort);
64  buffer = (char*)malloc(headerSize + size);
65  WebSocketFrameShort* header = (WebSocketFrameShort*)buffer;
66  memset(buffer, 0, headerSize);
67  header->len = 126;
68  header->len16 = htons(size);
69  } else {
70  headerSize = sizeof(WebSocketFrameHuge);
71  buffer = (char*)malloc(headerSize + size);
72  WebSocketFrameHuge* header = (WebSocketFrameHuge*)buffer;
73  memset(buffer, 0, headerSize);
74  header->len = 127;
75  header->len64 = htonll(size);
76  }
77  size_t fullHeaderSize = (mask ? 4 : 0) + headerSize;
78  startPayload = buffer + fullHeaderSize;
79  WebSocketFrameTiny* header = (WebSocketFrameTiny*)buffer;
80  header->fin = 1;
81  header->opcode = (unsigned char)opcode; // binary or text for now
82  // Mask is right before payload.
83  if (mask) {
84  *((uint32_t*)(startPayload - 4)) = mask;
85  }
86  header->mask = mask ? 1 : 0;
87  memmask(startPayload, src, size, mask);
88  outputs.push_back(uv_buf_init(buffer, size + fullHeaderSize));
89 }

91 void decode_websocket(char* start, size_t size, WebSocketHandler& handler)
92 {
93  // Handle the case in whiche the header is cut
94  if (handler.pendingHeaderSize) {
95  assert(handler.pendingHeader);
96  size_t pendingFullSize = handler.pendingHeaderSize + size;
97  char* pendingFull = new char[handler.pendingHeaderSize + size];
98  memcpy(pendingFull, handler.pendingHeader, handler.pendingHeaderSize);
99  memcpy(pendingFull + handler.pendingHeaderSize, start, size);
100  // We do not need the intermediate buffer anymore.
101  handler.pendingHeaderSize = 0;
102  delete[] handler.pendingHeader;
103  handler.pendingHeader = nullptr;
104  decode_websocket(pendingFull, pendingFullSize, handler);
105  delete[] pendingFull;
106  return;
107  }
108 
109  // Handle the case the previous message was cut in half
110  // by the I/O stack.
111  char* cur = start + handler.remainingSize;
112  if (handler.remainingSize) {
113  assert(handler.pendingBuffer);
114  auto newChunkSize = std::min(handler.remainingSize, size);
115  memcpy(handler.pendingBuffer + handler.pendingSize, start, newChunkSize);
116  handler.pendingSize += newChunkSize;
117  handler.remainingSize -= newChunkSize;
118  if (handler.remainingSize == 0) {
119  // One recursion should be enough.
120  decode_websocket(handler.pendingBuffer, handler.pendingSize, handler);
121  delete[] handler.pendingBuffer;
122  handler.pendingBuffer = nullptr;
123  }
124  }
125  handler.beginChunk();
126  // The + 2 is there because we need at least 2 bytes.
127  while (cur - start < size) {
128  WebSocketFrameTiny* header = (WebSocketFrameTiny*)cur;
129  size_t payloadSize = 0;
130  size_t headerSize = 0;
131  if ((cur + 2 - start >= size) ||
132  ((cur + 2 + 2 - start >= size) && header->len >= 126) ||
133  ((cur + 2 + 8 - start >= size) && header->len == 127)) {
134  // We do not have enough bytes for a tiny header. We copy in the pending header
135  handler.pendingHeaderSize = size - (cur - start);
136  handler.pendingHeader = new char[handler.pendingHeaderSize];
137  memcpy(handler.pendingHeader, cur, handler.pendingHeaderSize);
138  break;
139  }
140 
141  if (header->len < 126) {
142  payloadSize = header->len;
143  headerSize = 2 + (header->mask ? 4 : 0);
144  } else if (header->len == 126) {
145  WebSocketFrameShort* headerSmall = (WebSocketFrameShort*)cur;
146  payloadSize = ntohs(headerSmall->len16);
147  headerSize = 2 + 2 + (header->mask ? 4 : 0);
148  } else if (header->len == 127) {
149  WebSocketFrameHuge* headerSmall = (WebSocketFrameHuge*)cur;
150  payloadSize = ntohs(headerSmall->len64);
151  headerSize = 2 + 8 + (header->mask ? 4 : 0);
152  }
153  size_t availableSize = size - (cur - start);
154  if (availableSize < payloadSize + headerSize) {
155  handler.remainingSize = payloadSize + headerSize - availableSize;
156  handler.pendingSize = availableSize;
157  handler.pendingBuffer = new char[payloadSize + headerSize];
158  memcpy(handler.pendingBuffer, cur, availableSize);
159  break;
160  }
161  if (header->mask) {
162  int32_t mask = *(int32_t*)(cur + headerSize - 4);
163  memunmask(cur + headerSize, payloadSize, mask);
164  }
165  handler.frame(cur + headerSize, payloadSize);
166  cur += headerSize + payloadSize;
167  }
168  handler.endChunk();
169 }

209 void parse_http_request(char* start, size_t size, HTTPParser* parser)
210 {
211  enum HTTPState state = HTTPState::IN_START;
212  // Too short, let's try again...
213  if (size < 2) {
214  parser->remaining += std::string_view(start, size);
215  }
216  char* cur = start;
217  char* next = cur;
218  std::string_view lastToken(cur, 0);
219  std::string_view lastKey;
220  std::string_view lastValue;
221  std::string lastError;
222  if (parser->states.empty()) {
223  parser->states.push_back(HTTPState::IN_START);
224  }
225  char const* delimiters = nullptr;
226  char const* skippable = nullptr;
227  char const* separator = nullptr;
228  char const* spaces = "\t \v";
229  char const* colon = ":";
230  char const* newline = "\r\n";
231  bool done = false;
232 
233  while (!done) {
234  HTTPState state = parser->states.back();
235  parser->states.pop_back();
236  switch (state) {
237  case HTTPState::IN_START:
238  parser->states.push_back(HTTPState::BEGIN_METHOD);
239  break;
240  case HTTPState::IN_START_REPLY:
241  parser->states.push_back(HTTPState::BEGIN_REPLY_VERSION);
242  break;
243  case HTTPState::BEGIN_REPLY_VERSION:
244  parser->states.push_back(HTTPState::END_REPLY_VERSION);
245  parser->states.push_back(HTTPState::IN_CAPTURE_DELIMITERS);
246  delimiters = spaces;
247  break;
248  case HTTPState::END_REPLY_VERSION:
249  parser->replyVersion(lastToken);
250  parser->states.push_back(HTTPState::BEGIN_REPLY_CODE);
251  parser->states.push_back(HTTPState::IN_SKIP_CHARS);
252  skippable = spaces;
253  break;
254  case HTTPState::BEGIN_REPLY_CODE:
255  parser->states.push_back(HTTPState::END_REPLY_CODE);
256  parser->states.push_back(HTTPState::IN_CAPTURE_DELIMITERS);
257  delimiters = spaces;
258  break;
259  case HTTPState::END_REPLY_CODE:
260  parser->replyCode(lastToken);
261  parser->states.push_back(HTTPState::BEGIN_REPLY_MESSAGE);
262  parser->states.push_back(HTTPState::IN_SKIP_CHARS);
263  skippable = spaces;
264  break;
265  case HTTPState::BEGIN_REPLY_MESSAGE:
266  parser->states.push_back(HTTPState::END_REPLY_MESSAGE);
267  parser->states.push_back(HTTPState::IN_CAPTURE_SEPARATOR);
268  separator = newline;
269  break;
270  case HTTPState::END_REPLY_MESSAGE:
271  parser->replyMessage(lastToken);
272  parser->states.push_back(HTTPState::BEGIN_HEADERS);
273  break;
274  case HTTPState::BEGIN_METHOD:
275  parser->states.push_back(HTTPState::END_METHOD);
276  parser->states.push_back(HTTPState::IN_CAPTURE_DELIMITERS);
277  delimiters = spaces;
278  break;
279  case HTTPState::END_METHOD:
280  parser->method(lastToken);
281  parser->states.push_back(HTTPState::BEGIN_TARGET);
282  parser->states.push_back(HTTPState::IN_SKIP_CHARS);
283  skippable = spaces;
284  break;
285  case HTTPState::BEGIN_TARGET:
286  parser->states.push_back(HTTPState::END_TARGET);
287  parser->states.push_back(HTTPState::IN_CAPTURE_DELIMITERS);
288  delimiters = spaces;
289  break;
290  case HTTPState::END_TARGET:
291  parser->target(lastToken);
292  parser->states.push_back(HTTPState::BEGIN_VERSION);
293  parser->states.push_back(HTTPState::IN_SKIP_CHARS);
294  skippable = spaces;
295  break;
296  case HTTPState::BEGIN_VERSION:
297  parser->states.push_back(HTTPState::END_VERSION);
298  parser->states.push_back(HTTPState::IN_CAPTURE_SEPARATOR);
299  separator = newline;
300  break;
301  case HTTPState::END_VERSION:
302  parser->version(lastToken);
303  parser->states.push_back(HTTPState::BEGIN_HEADERS);
304  break;
305  case HTTPState::BEGIN_HEADERS:
306  parser->states.push_back(HTTPState::BEGIN_HEADER);
307  break;
308  case HTTPState::BEGIN_HEADER:
309  parser->states.push_back(HTTPState::BEGIN_HEADER_KEY);
310  break;
311  case HTTPState::BEGIN_HEADER_KEY:
312  parser->states.push_back(HTTPState::END_HEADER_KEY);
313  parser->states.push_back(HTTPState::IN_CAPTURE_SEPARATOR);
314  separator = colon;
315  break;
316  case HTTPState::END_HEADER_KEY:
317  lastKey = lastToken;
318  parser->states.push_back(HTTPState::BEGIN_HEADER_VALUE);
319  parser->states.push_back(HTTPState::IN_SKIP_CHARS);
320  skippable = spaces;
321  break;
322  case HTTPState::BEGIN_HEADER_VALUE:
323  parser->states.push_back(HTTPState::END_HEADER_VALUE);
324  parser->states.push_back(HTTPState::IN_CAPTURE_SEPARATOR);
325  separator = newline;
326  break;
327  case HTTPState::END_HEADER_VALUE:
328  lastValue = lastToken;
329  parser->states.push_back(HTTPState::END_HEADER);
330  break;
331  case HTTPState::END_HEADER:
332  if (strncmp("\r\n", next, 2) == 0) {
333  parser->header(lastKey, lastValue);
334  parser->states.push_back(HTTPState::END_HEADERS);
335  next += 2;
336  cur = next;
337  } else {
338  parser->header(lastKey, lastValue);
339  parser->states.push_back(HTTPState::BEGIN_HEADER);
340  cur = next;
341  }
342  break;
343  case HTTPState::END_HEADERS:
344  parser->endHeaders();
345  parser->states.push_back(HTTPState::BEGIN_BODY);
346  break;
347  case HTTPState::BEGIN_BODY: {
348  size_t bodySize = size - (cur - start);
349  parser->body(cur, bodySize);
350  next = cur + bodySize;
351  cur = next;
352  parser->states.push_back(HTTPState::BEGIN_BODY);
353  parser->states.push_back(HTTPState::IN_DONE);
354  } break;
355  case HTTPState::IN_SKIP_CHARS:
356  while (true) {
357  if (next - start == size) {
358  parser->remaining += std::string_view(cur, next - cur);
359  }
360  if (strchr(skippable, *next)) {
361  next++;
362  continue;
363  }
364  cur = next;
365  break;
366  }
367  break;
368  case HTTPState::IN_SEPARATOR:
369  if (memcmp(separator, cur, strlen(separator)) != 0) {
370  parser->states.push_back(HTTPState::IN_ERROR);
371  break;
372  }
373  next += strlen(separator);
374  cur = next;
375  break;
376  case HTTPState::IN_CAPTURE_DELIMITERS:
377  while (true) {
378  if (next - start == size) {
379  parser->remaining += std::string_view(cur, next - cur);
380  }
381  if (strchr(delimiters, *next) == nullptr) {
382  next++;
383  continue;
384  }
385  lastToken = std::string_view(cur, next - cur);
386  cur = next;
387  break;
388  }
389  break;
390  case HTTPState::IN_CAPTURE_SEPARATOR:
391  while (true) {
392  if (next + strlen(separator) - start == size) {
393  parser->remaining += std::string_view(cur, next - cur);
394  }
395  if (memcmp(separator, next, strlen(separator)) != 0) {
396  next++;
397  continue;
398  }
399  lastToken = std::string_view(cur, next - cur);
400  next += strlen(separator);
401  cur = next;
402  break;
403  }
404  break;
405  case HTTPState::IN_DONE:
406  // The only case in which there can be a pending state when IN_DONE, is if
407  // we plan to resume processing.
408  if (parser->states.size() == 1 && parser->states.back() == HTTPState::BEGIN_BODY) {
409  done = true;
410  } else if (parser->states.empty()) {
411  done = true;
412  } else {
413  parser->states.push_back(HTTPState::IN_ERROR);
414  }
415  break;
416  case HTTPState::IN_ERROR:
417  parser->error = lastError;
418  parser->states.clear();
419  done = true;
420  break;
421  default:
422  parser->states.push_back(HTTPState::IN_ERROR);
423  break;
424  }
425  }
426 }

428 std::pair<std::string, unsigned short> parse_websocket_url(char const* url)
429 {
430  std::string s = url;
431  if (s == "ws://") {
432  s = "ws://127.0.0.1:8080";
433  }
434  const std::regex urlMatcher("^ws://([0-9-_.]+)[:]([0-9]+)$");
435  std::smatch parts;
436  if (!std::regex_match(s, parts, urlMatcher)) {
437  throw runtime_error_f(
438  "Unable to parse driver client url: %s.\n"
439  "Format should be ws://[<driver ip>:<port>] e.g. ws://127.0.0.1:8080 or just ws://");
440  }
441  std::string ip = std::string{parts[1]};
442  auto portS = std::string(parts[2]);
443  unsigned short port = std::stoul(portS);
444  return {ip, port};
445 }
