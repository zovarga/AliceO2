35 class ReverseSymbolLookupTable
36 {
37  public:
38  using iterator_t = const symbol_t*;
39 
40  // TODO(milettri): fix once ROOT cling respects the standard http://wg21.link/p1286r2
41  ReverseSymbolLookupTable() noexcept {}; // NOLINT
42 
43  explicit ReverseSymbolLookupTable(const RenormedFrequencyTable& frequencyTable)
44  {
45  LOG(trace) << "start building reverse symbol lookup table";
46 
47  mLut.resize(frequencyTable.getNumSamples());
48  // go over all normal symbols
49  count_t cumulativeFrequency = 0;
50  for (size_t index = 0; index < frequencyTable.size(); ++index) {
51  const symbol_t symbol = frequencyTable.getMinSymbol() + index;
52  const count_t symbolFrequency = frequencyTable.at(index);
53  for (count_t cumulative = cumulativeFrequency; cumulative < cumulativeFrequency + symbolFrequency; ++cumulative) {
54  mLut[cumulative] = symbol;
55  }
56  cumulativeFrequency += symbolFrequency;
57  }
58 
59  // incompressible Symbol
60  const symbol_t symbol = frequencyTable.empty() ? 0 : frequencyTable.getMaxSymbol() + 1;
61  const count_t symbolFrequency = frequencyTable.getIncompressibleSymbolFrequency();
62  for (count_t cumulative = cumulativeFrequency; cumulative < cumulativeFrequency + symbolFrequency; ++cumulative) {
63  mLut[cumulative] = symbol;
64  }
65 
66 // advanced diagnostics for debug builds
67 #if !defined(NDEBUG)
68  LOG(debug2) << "reverseSymbolLookupTableProperties: {"
69  << "elements: " << mLut.size() << ", "
70  << "sizeB: " << mLut.size() * sizeof(typename std::decay_t<decltype(mLut)>::value_type) << "}";
71 #endif
72 
73  if (frequencyTable.empty()) {
74  LOG(warning) << "SymbolStatistics of empty message passed to " << __func__;
75  }
76 
77  LOG(trace) << "done building reverse symbol lookup table";
78  };
79 
80  inline size_t size() const noexcept { return mLut.size(); };
81  inline symbol_t operator[](count_t cumul) const noexcept
82  {
83  LOG_IF(fatal, cumul >= size()) << fmt::format("[iLUT]: {} exceeds max {}", cumul, size());
84  assert(cumul < size());
85  return mLut[cumul];
86  };
87  inline iterator_t begin() const noexcept { return mLut.data(); };
88  inline iterator_t end() const noexcept { return mLut.data() + size(); };
89 
90  private:
91  std::vector<symbol_t> mLut{};
92 };
