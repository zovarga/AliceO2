24 void* DataRefUtils::decodeCCDB(DataRef const& ref, std::type_info const& tinfo)
25 {
26  void* result = nullptr;
27  Int_t previousErrorLevel = gErrorIgnoreLevel;
28  gErrorIgnoreLevel = kFatal;
29  auto* dh = o2::header::get<o2::header::DataHeader*>(ref.header);
30  TMemFile memFile("name", const_cast<char*>(ref.payload), dh->payloadSize, "READ");
31  gErrorIgnoreLevel = previousErrorLevel;
32  if (memFile.IsZombie()) {
33  return nullptr;
34  }
35  TClass* tcl = TClass::GetClass(tinfo);
36  result = ccdb::CcdbApi::extractFromTFile(memFile, tcl);
37  if (!result) {
38  throw runtime_error_f("Couldn't retrieve object corresponding to %s from TFile", tcl->GetName());
39  }
40  memFile.Close();
41  return result;
42 }
