23 void GPUChainTracking::RunTPCTrackingMerger_MergeBorderTracks(char withinSlice, char mergeMode, GPUReconstruction::krnlDeviceType deviceType)
24 {
25  unsigned int n = withinSlice == -1 ? NSLICES / 2 : NSLICES;
26  bool doGPUall = GetRecoStepsGPU() & RecoStep::TPCMerging && GetProcessingSettings().fullMergerOnGPU;
27  if (GetProcessingSettings().alternateBorderSort && (!mRec->IsGPU() || doGPUall)) {
28  GPUTPCGMMerger& Merger = processors()->tpcMerger;
29  GPUTPCGMMerger& MergerShadow = doGPUall ? processorsShadow()->tpcMerger : Merger;
30  TransferMemoryResourceLinkToHost(RecoStep::TPCMerging, Merger.MemoryResMemory(), 0, &mEvents->init);
31  RecordMarker(&mEvents->single, 0);
32  for (unsigned int i = 0; i < n; i++) {
33  int stream = i % mRec->NStreams();
34  runKernel<GPUTPCGMMergerMergeBorders, 0>(GetGridAuto(stream, deviceType), krnlRunRangeNone, {nullptr, stream && i < (unsigned int)mRec->NStreams() ? &mEvents->single : nullptr}, i, withinSlice, mergeMode);
35  }
36  ReleaseEvent(&mEvents->single);
37  SynchronizeEventAndRelease(&mEvents->init);
38  for (unsigned int i = 0; i < n; i++) {
39  int stream = i % mRec->NStreams();
40  int n1, n2;
41  GPUTPCGMBorderTrack *b1, *b2;
42  int jSlice;
43  Merger.MergeBorderTracksSetup(n1, n2, b1, b2, jSlice, i, withinSlice, mergeMode);
44  gputpcgmmergertypes::GPUTPCGMBorderRange* range1 = MergerShadow.BorderRange(i);
45  gputpcgmmergertypes::GPUTPCGMBorderRange* range2 = MergerShadow.BorderRange(jSlice) + *processors()->tpcTrackers[jSlice].NTracks();
46  runKernel<GPUTPCGMMergerMergeBorders, 3>({1, -WarpSize(), stream, deviceType}, krnlRunRangeNone, krnlEventNone, range1, n1, 0);
47  runKernel<GPUTPCGMMergerMergeBorders, 3>({1, -WarpSize(), stream, deviceType}, krnlRunRangeNone, krnlEventNone, range2, n2, 1);
48  deviceEvent** e = nullptr;
49  int ne = 0;
50  if (i == n - 1) { // Synchronize all execution on stream 0 with the last kernel
51  ne = std::min<int>(n, mRec->NStreams());
52  for (int j = 1; j < ne; j++) {
53  RecordMarker(&mEvents->slice[j], j);
54  }
55  e = &mEvents->slice[1];
56  ne--;
57  stream = 0;
58  }
59  runKernel<GPUTPCGMMergerMergeBorders, 2>(GetGridAuto(stream, deviceType), krnlRunRangeNone, {nullptr, e, ne}, i, withinSlice, mergeMode);
60  }
61  } else {
62  for (unsigned int i = 0; i < n; i++) {
63  runKernel<GPUTPCGMMergerMergeBorders, 0>(GetGridAuto(0, deviceType), krnlRunRangeNone, krnlEventNone, i, withinSlice, mergeMode);
64  }
65  runKernel<GPUTPCGMMergerMergeBorders, 1>({2 * n, -WarpSize(), 0, deviceType}, krnlRunRangeNone, krnlEventNone, 0, withinSlice, mergeMode);
66  for (unsigned int i = 0; i < n; i++) {
67  runKernel<GPUTPCGMMergerMergeBorders, 2>(GetGridAuto(0, deviceType), krnlRunRangeNone, krnlEventNone, i, withinSlice, mergeMode);
68  }
69  }
70  mRec->ReturnVolatileDeviceMemory();
71 }

82 int GPUChainTracking::RunTPCTrackingMerger(bool synchronizeOutput)
83 {
84  if (GetProcessingSettings().debugLevel >= 6 && GetProcessingSettings().comparableDebutOutput && param().rec.tpc.mergerReadFromTrackerDirectly) {
85  for (unsigned int i = 0; i < NSLICES; i++) {
86  GPUTPCTracker& trk = processors()->tpcTrackers[i];
87  TransferMemoryResourcesToHost(RecoStep::NoRecoStep, &trk);
88  auto sorter = [](GPUTPCTrack& trk1, GPUTPCTrack& trk2) {
89  if (trk1.NHits() == trk2.NHits()) {
90  return trk1.Param().Y() > trk2.Param().Y();
91  }
92  return trk1.NHits() > trk2.NHits();
93  };
94  std::sort(trk.Tracks(), trk.Tracks() + trk.CommonMemory()->nLocalTracks, sorter);
95  std::sort(trk.Tracks() + trk.CommonMemory()->nLocalTracks, trk.Tracks() + *trk.NTracks(), sorter);
96  TransferMemoryResourcesToGPU(RecoStep::NoRecoStep, &trk, 0);
97  }
98  }
99  mRec->PushNonPersistentMemory(qStr2Tag("TPCMERGE"));
100  bool doGPU = GetRecoStepsGPU() & RecoStep::TPCMerging;
101  bool doGPUall = doGPU && GetProcessingSettings().fullMergerOnGPU;
102  GPUReconstruction::krnlDeviceType deviceType = doGPUall ? GPUReconstruction::krnlDeviceType::Auto : GPUReconstruction::krnlDeviceType::CPU;
103  unsigned int numBlocks = (!mRec->IsGPU() || doGPUall) ? BlockCount() : 1;
104  GPUTPCGMMerger& Merger = processors()->tpcMerger;
105  GPUTPCGMMerger& MergerShadow = doGPU ? processorsShadow()->tpcMerger : Merger;

116  AllocateRegisteredMemory(Merger.MemoryResOutputState(), mSubOutputControls[GPUTrackingOutputs::getIndex(&GPUTrackingOutputs::sharedClusterMap)]);
