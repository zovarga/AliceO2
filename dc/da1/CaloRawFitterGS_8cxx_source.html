28 void CaloRawFitterGS::init()
29 {
30  mSpikeThreshold = 5;
31  // prepare fitting arrays, once per lifetime
32  double k = o2::phos::PHOSSimParams::Instance().mSampleDecayTime;
33  ma0[0] = 0.;
34  ma1[0] = 0.;
35  ma2[0] = 0.;
36  ma3[0] = 0.;
37  ma4[0] = 0.;
38  for (int i = 1; i < NMAXSAMPLES; i++) {
39  double xi = k * i;
40  mexp[i] = exp(-xi);
41  double s = mexp[i] * mexp[i];
42  ma0[i] = ma0[i - 1] + s;
43  ma1[i] = ma1[i - 1] + s * xi;
44  ma2[i] = ma2[i - 1] + s * xi * xi;
45  ma3[i] = ma3[i - 1] + s * xi * xi * xi;
46  ma4[i] = ma4[i - 1] + s * xi * xi * xi * xi;
47  }
48 }

53  // Pedestal analysis mode
54  if (mPedestalRun) {
55  int nPed = signal.size();
56  mAmp = 0.;
57  mTime = 0.;
58  for (auto a : signal) {
59  mAmp += a;
60  mTime += a * a;
61  }
62  if (nPed > 0) {
63  mAmp /= nPed;
64  mTime = mTime / nPed - mAmp * mAmp;
65  if (mTime > 0.) {
66  mTime = sqrt(mTime);
67  }
68  }
69  mOverflow = false;
70  mStatus = kOK;
71  return kOK;
72  }
73 
74  mStatus = kNotEvaluated;
75  // Extract amplitude and time
76  mStatus = evalFit(signal);
77  return mStatus;

80 CaloRawFitterGS::FitStatus CaloRawFitterGS::evalFit(gsl::span<short unsigned int> signal)
81 {
82  // Calculate signal parameters (energy, time, quality) from array of samples
83  // Fit with semi-gaus function with free parameters time and amplitude
84  // Signal overflows if there are at least 3 samples of the same amplitude above 900
85 
86  // Calculate signal parameters (energy, time, quality) from array of samples
87  // Energy is a maximum sample minus pedestal 9
88  // Time is the first time bin
89  // Signal overflows is there are at least 3 samples of the same amplitude above 900
90 
91  int nSamples = signal.size();
92  if (nSamples == 0) {
93  mAmp = 0;
94  mTime = 0.;
95  mChi2 = 0.;
96  return kEmptyBunch;
97  }
98  if (nSamples == 1) {
99  mAmp = signal[0];
100  mTime = 0.;
101  mChi2 = 1.;
102  return kOK;
103  }
104 
105  mOverflow = false;
106 
107  // if pedestal should be subtracted first evaluate it
108  float pedMean = 0;
109  int nPed = 0;
110  if (mPedSubtract) {
111  nSamples = -mPreSamples;
112  // remember inverse time order
113  for (auto it = signal.rbegin(); (nPed < mPreSamples) && it != signal.rend(); ++it) {
114  nPed++;
115  pedMean += *it;
116  }
117  if (nPed > 0) {
118  pedMean /= nPed;
119  }
120  }
121 
122  float maxSample = 0.; // maximal sample value
123  int nMax = 0; // number of consequitive maximal samples
124  bool spike = false; // spike in previoud signal bin?
125  short ap = -1, app = -1; // remember previous values to evaluate spikes
126  double b0 = 0., b1 = 0., b2 = 0., y2 = 0.; // fit coeficients
127  double sa0 = 0., sa1 = 0., sa2 = 0., sa3 = 0., sa4 = 0.; // corrections in case of overflow
128 
129  int firstS = nSamples;
130  int j = TMath::Min(nSamples, NMAXSAMPLES - 1);
131  for (int i = 1; i <= j; i++) {
132  short a = signal[firstS - i] - pedMean; // remember inverse order of samples
133  float xi = i * mDecTime;
134  if (a > maxSample) {
135  maxSample = a;
136  nMax = 1;
137  } else {
138  if (a == maxSample) {
139  nMax++;
140  }
141  }
142  // check if there was a spike in previous step?
143  if (app > 0 && ap > 0) {
144  spike = (ap - a > mSpikeThreshold) && (ap - app > mSpikeThreshold);
145  }
146  if (spike) { // Try to recover: subtract last point contribution and replace by average of "app" and "a"
147  float atmp = 0.5 * (app + a);
148  float xiprev = xi - mDecTime;
149  float st = (atmp - ap) * mexp[i - 1];
150  b0 += st;
151  b1 += st * xiprev;
152  b2 += st * xiprev * xiprev;
153  y2 += atmp * atmp - ap * ap;
154  ap = a;
155  } else {
156  app = ap;
157  ap = a;
158  }
159  // Check if in saturation
160  if (maxSample > 900 && nMax >= 3) {
161  // Remove overflow points from the fit
162  if (!mOverflow) { // first time in this sample: remove two previous points
163  sa0 = ma0[i] - ma0[i - 2]; // can not appear at i<2
164  sa1 = ma1[i] - ma1[i - 2];
165  sa2 = ma2[i] - ma2[i - 2];
166  sa3 = ma3[i] - ma3[i - 2];
167  sa4 = ma4[i] - ma4[i - 2];
168  float st = ap * mexp[i - 1];
169  float xiprev = xi - mDecTime;
170  b0 -= st;
171  b1 -= st * xiprev;
172  b2 -= st * xiprev * xiprev;
173  y2 -= ap * ap;
174  st = app * mexp[i - 2];
175  xiprev -= mDecTime;
176  b0 -= st;
177  b1 -= st * xiprev;
178  b2 -= st * xiprev * xiprev;
179  y2 -= ap * ap;
180  }
181  mOverflow = true;
182  }
183  if (!mOverflow) {
184  // to calculate time
185  float st = a * mexp[i];
186  b0 += st;
187  b1 += st * xi;
188  b2 += st * xi * xi;
189  y2 += a * a;
190  } else { // do not add current point and subtract contributions to amx[] arrays
191  sa0 = ma0[i] - ma0[i - 1]; // can not appear at i<2
192  sa1 = ma1[i] - ma1[i - 1];
193  sa2 = ma2[i] - ma2[i - 1];
194  sa3 = ma3[i] - ma3[i - 1];
195  sa4 = ma4[i] - ma4[i - 1];
196  }
197  } // Scanned full
198 
199  // too small amplitude, assing max to max Amp and time to zero and do not calculate height
200  if (maxSample < mMinTimeCalc) {
201  mAmp = maxSample;
202  mTime = 0.;
203  mChi2 = 0.;
204  return kOK;
205  }
206 
207  if (mOverflow && b0 == 0) { // strong overflow, no reasonable counts, can not extract anything
208  mAmp = 0.;
209  mTime = 0.;
210  mChi2 = 900.;
211  return kOverflow;
212  }
213 
214  // calculate time, amp and chi2
215  double a, b, c, d, e; // Polinomial coefficients
216  if (!mOverflow) {
217  a = ma1[j] * b0 - ma0[j] * b1;
218  b = ma0[j] * b2 + 2. * ma1[j] * b1 - 3. * ma2[j] * b0;
219  c = 3. * (ma3[j] * b0 - ma1[j] * b2);
220  d = 3. * ma2[j] * b2 - ma4[j] * b0 - 2. * ma3[j] * b1;
221  e = ma4[j] * b1 - ma3[j] * b2;
222  } else { // account removed points in overflow
223  a = (ma1[j] - sa1) * b0 - (ma0[j] - sa0) * b1;
224  b = (ma0[j] - sa0) * b2 + 2. * (ma1[j] - sa1) * b1 - 3. * (ma2[j] - sa2) * b0;
225  c = 3. * ((ma3[j] - sa3) * b0 - (ma1[j] - sa1) * b2);
226  d = 3. * (ma2[j] - sa2) * b2 - (ma4[j] - sa4) * b0 - 2. * (ma3[j] - sa4) * b1;
227  e = (ma4[j] - sa4) * b1 - (ma3[j] - sa3) * b2;
228  }
229 
230  // Find zero of 4-order polinomial
231  // first use linear extrapolation to reach correct root of four
232  double z = -1.;
233  if (ma0[j] * b1 - ma1[j] * b0 != 0) {
234  z = (ma1[j] * b1 - ma2[j] * b0) / (ma0[j] * b1 - ma1[j] * b0) - 1.; // linear fit + offset
235  }
236  double q = 0., dq = 0., ddq = 0., lq = 0., dz = 0.1;
237  double z2 = z * z;
238  double z3 = z2 * z;
239  double z4 = z2 * z2;
240  q = a * z4 + b * z3 + c * z2 + d * z + e; // polinomial
241  dq = 4. * a * z3 + 3. * b * z2 + 2. * c * z + d; // Derivative
242  ddq = 12. * a * z2 + 6. * b * z + 2. * c; // Second derivative
243  if (dq != 0.) {
244  lq = q * ddq / (dq * dq);
245  }
246  // dz = -q/dq ; // Newton ~7 terations
247  // dz =-(1+0.5*lq)*q/dq ; // Chebyshev ~3 iterations to reach |q|<1.e-11
248  double ttt = dq * (1. - 0.5 * lq); // Halley’s method ~3 iterations, a bit more precise
249  if (ttt != 0) {
250  dz = -q / ttt;
251  } else {
252  dz = 0.1; // step off saddle point
253  }
254  int it = 0;
255  while (TMath::Abs(q) > 0.0001 && (++it < 15)) {
256  z += dz;
257  z2 = z * z;
258  z3 = z2 * z;
259  z4 = z2 * z2;
260  q = a * z4 + b * z3 + c * z2 + d * z + e;
261  dq = 4. * a * z3 + 3. * b * z2 + 2. * c * z + d;
262  ddq = 12. * a * z2 + 6. * b * z + 2. * c;
263  if (dq != 0) {
264  lq = q * ddq / (dq * dq);
265  ttt = dq * (1. - 0.5 * lq);
266  // dz = -q/dq ; //Newton
267  // dz =-(1+0.5*lq)*q/dq ; //Chebyshev
268  if (ttt != 0) {
269  dz = -q / ttt; // Halley’s
270  } else {
271  dz = -q / dq;
272  }
273  } else {
274  dz = 0.5 * dz; // step off saddle point
275  }
276  }
277 
278  // check that result is reasonable
279  double denom = ma4[j] - 4. * ma3[j] * z + 6. * ma2[j] * z * z - 4. * ma1[j] * z * z * z + ma0[j] * z * z * z * z;
280  if (denom != 0.) {
281  mAmp = 4. * exp(-2 - z) * (b2 - 2. * b1 * z + b0 * z * z) / denom;
282  } else {
283  mAmp = 0.;
284  }
285 
286  if ((TMath::Abs(q) < mQAccuracy) && (mAmp < 1.2 * maxSample)) { // converged and estimated amplitude is not mush larger than Max
287  mTime = z / mDecTime;
288  mChi2 = (y2 - 0.25 * exp(2. + z) * mAmp * (b2 - 2 * b1 * z + b0 * z2)) / nSamples;
289  return kOK;
290  } else { // too big difference, fit failed
291  mAmp = maxSample;
292  mTime = 0; // First count in sample
293  mChi2 = 999.;
294  return kFitFailed;
295  }
296 }
