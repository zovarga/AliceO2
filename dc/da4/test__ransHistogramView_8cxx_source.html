28 struct ReferenceState {
29  ReferenceState(std::vector<int32_t> v) : mV{std::move(v)} {};
30 
31  int32_t getMin() { return *std::min_element(mV.begin(), mV.end()); };
32  int32_t getMax() { return *std::max_element(mV.begin(), mV.end()); };
33  size_t size() { return mV.size(); };
34  int32_t getOffset() { return this->getMin(); };
35  auto begin() { return mV.begin(); };
36  auto end() { return mV.end(); };
37 
38  std::vector<int32_t> mV{};
39 };
