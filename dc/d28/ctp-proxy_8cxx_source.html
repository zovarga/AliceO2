43 InjectorFunction dcs2dpl()
44 {
45 
46  auto timesliceId = std::make_shared<size_t>(0);
47 
48  return [timesliceId](FairMQDevice& device, FairMQParts& parts, ChannelRetriever channelRetriever) {
49  // make sure just 2 messages received
50  if (parts.Size() != 2) {
51  LOG(error) << "received " << parts.Size() << " instead of 2 expected";
52  return;
53  }
54  std::string messageHeader{static_cast<const char*>(parts.At(0)->GetData()), parts.At(0)->GetSize()};
55  size_t dataSize = parts.At(1)->GetSize();
56  std::string messageData{static_cast<const char*>(parts.At(1)->GetData()), parts.At(1)->GetSize()};
57  LOG(info) << "received message " << messageHeader << " of size " << dataSize << " Payload:" << messageData;
58  o2::header::DataHeader hdrF("CTP_COUNTERS", o2::header::gDataOriginCTP, 0);
59  OutputSpec outsp{hdrF.dataOrigin, hdrF.dataDescription, hdrF.subSpecification};
60  auto channel = channelRetriever(outsp, *timesliceId);
61  if (channel.empty()) {
62  LOG(error) << "No output channel found for OutputSpec " << outsp;
63  return;
64  }
65 
66  hdrF.tfCounter = *timesliceId; // this also
67  hdrF.payloadSerializationMethod = o2::header::gSerializationMethodNone;
68  hdrF.splitPayloadParts = 1;
69  hdrF.splitPayloadIndex = 0;
70  hdrF.payloadSize = parts.At(1)->GetSize();
71  hdrF.firstTForbit = 0; // this should be irrelevant for Counters ? Orbit is in payload
72 
73  auto fmqFactory = device.GetChannel(channel).Transport();
74 
75  o2::header::Stack headerStackF{hdrF, DataProcessingHeader{*timesliceId, 1}};
76  auto hdMessageF = fmqFactory->CreateMessage(headerStackF.size(), fair::mq::Alignment{64});
77  auto plMessageF = fmqFactory->CreateMessage(hdrF.payloadSize, fair::mq::Alignment{64});
78  memcpy(hdMessageF->GetData(), headerStackF.data(), headerStackF.size());
79  memcpy(plMessageF->GetData(), parts.At(1)->GetData(), hdrF.payloadSize);
80 
81  FairMQParts outParts;
82  outParts.AddPart(std::move(hdMessageF));
83  outParts.AddPart(std::move(plMessageF));
84  sendOnChannel(device, outParts, channel);
85  // LOG(info) << "Sent CTP counters DPL message";
86  };
87 }

97 WorkflowSpec defineDataProcessing(ConfigContext const& config)
98 {
99  LOG(info) << "Defining data processing";
100  auto setChanName = [](const std::string& chan, const std::string& name) {
101  size_t n = 0;
102  if (std::string(chan).find("name=") != std::string::npos) {
103  n = std::string(chan).find(",");
104  if (n == std::string::npos) {
105  throw std::runtime_error(fmt::format("wrongly formatted channel: {}", chan));
106  }
107  n++;
108  }
109  LOG(info) << "===>inside:" << name << " " << chan;
110  return o2::utils::Str::concat_string("name=", name, ",", chan.substr(n, chan.size()));
111  };
112  const std::string devName = "ctp-proxy";
113  auto chan = config.options().get<std::string>("subscribe-to");
114  if (chan.empty()) {
115  throw std::runtime_error("input channel is not provided");
116  }
117  chan = setChanName(chan, devName);
118  LOG(info) << "name:" << devName << " chan:" << chan;
119  LOG(info) << "Channels setup: " << chan;
120  Outputs ctpCountersOutputs;
121  ctpCountersOutputs.emplace_back("CTP", "CTP_COUNTERS", 0, Lifetime::Timeframe);
122  LOG(info) << "===> Proxy to be set";
123  DataProcessorSpec ctpProxy = specifyExternalFairMQDeviceProxy(
124  devName.c_str(),
125  std::move(ctpCountersOutputs),
126  // this is just default, can be overriden by --ctp-config-proxy '--channel-config..'
127  chan.c_str(),
128  dcs2dpl());
129  LOG(info) << "===> Proxy done";
130  WorkflowSpec workflow;
131  workflow.emplace_back(ctpProxy);
132  return workflow;
133 }
