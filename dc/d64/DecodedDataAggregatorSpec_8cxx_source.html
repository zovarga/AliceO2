38  void init(of::InitContext& ic)

40  auto stop = [this]() {
41  LOG(info) << "Capacities: ROFRecords: " << mAggregator.getROFRecords().capacity() << " Data: " << mAggregator.getData().capacity();
42  double scaleFactor = 1.e6 / mNROFs;
43  LOG(info) << "Processing time / " << mNROFs << " ROFs: full: " << mTimer.count() * scaleFactor << " aggregating: " << mTimerAlgo.count() * scaleFactor << " us";
44  };
45  ic.services().get<of::CallbackService>().set(of::CallbackService::Id::Stop, stop);
46  }

48  void run(of::ProcessingContext& pc)
49  {
50  auto tStart = std::chrono::high_resolution_clock::now();
51 
52  auto msg = pc.inputs().get("mid_decoded");
53  auto data = of::DataRefUtils::as<const ROBoard>(msg);
54 
55  auto msgROF = pc.inputs().get("mid_decoded_rof");
56  auto inROFRecords = of::DataRefUtils::as<const ROFRecord>(msgROF);
57 
58  auto tAlgoStart = std::chrono::high_resolution_clock::now();
59  mAggregator.process(data, inROFRecords);
60  mTimerAlgo += std::chrono::high_resolution_clock::now() - tAlgoStart;
61 
62  for (o2::header::DataHeader::SubSpecificationType subSpec = 0; subSpec < 3; ++subSpec) {
63  EventType evtType = static_cast<EventType>(subSpec);
64  pc.outputs().snapshot(of::Output{o2::header::gDataOriginMID, "DATA", subSpec, of::Lifetime::Timeframe}, mAggregator.getData(evtType));
65  pc.outputs().snapshot(of::Output{o2::header::gDataOriginMID, "DATAROF", subSpec, of::Lifetime::Timeframe}, mAggregator.getROFRecords(evtType));
66  }
67 
68  mTimer += std::chrono::high_resolution_clock::now() - tStart;
69  mNROFs += mAggregator.getROFRecords().size();
70  }
71 
72  private:
73  DecodedDataAggregator mAggregator{};
74  std::chrono::duration<double> mTimer{0};
75  std::chrono::duration<double> mTimerAlgo{0};
76  unsigned int mNROFs{0};
77 };
