60  return;
61  }
62 
63  stream_IT inputIter = inputEnd;
64  source_IT it = outputBegin;
65 
66  // make Iter point to the last last element

74 
75  for (size_t i = 0; i < (messageLength & ~1); i += 2) {

77  const int64_t s1 = this->mReverseLUT[rans1.get()];
78  *it++ = s0;
79  *it++ = s1;
80  inputIter = rans0.advanceSymbol(inputIter, this->mSymbolTable[s0]);
81  inputIter = rans1.advanceSymbol(inputIter, this->mSymbolTable[s1]);
82  }
83 
84  // last byte, if message length was odd
85  if (messageLength & 1) {
86  const int64_t s0 = this->mReverseLUT[rans0.get()];
87  *it = s0;

89  }

92  << "processedBytes: " << messageLength * sizeof(source_T) << ","
93  << " inclusiveTimeMS: " << t.getDurationMS() << ","
94  << " BandwidthMiBPS: " << std::fixed << std::setprecision(2) << (messageLength * sizeof(source_T) * 1.0) / (t.getDurationS() * 1.0 * (1 << 20)) << "}";
