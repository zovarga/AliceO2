124 struct TrackletHCHeader {
125  union {
126  // 10987654321098765432109876543210
127  // uint32_t: 33222222222211111111110000000000
128  // cccccccccccccccX LLL SSSSS
129  // ffff| |y| sss|
130  // | | ||| | |----- 0-4 supermodule
131  // | | ||| |-------- 5-7 stack
132  // | | ||------------ 8-10 layer
133  // | | |------------- 11 always 0x1
134  // | | |------------- 12 side of chamber
135  // | ----------------------------- 13-27 MCM Clock counter
136  // --------------------------------- 28-31 tracklet data format number
137  uint32_t word;
138  struct {
139  uint32_t supermodule : 5;
140  uint32_t stack : 3;
141  uint32_t layer : 3;
142  uint32_t one : 1; //always 1
143  uint32_t side : 1; // side of chamber
144  uint32_t MCLK : 15; // MCM clock counter 120MHz ... for simulation -- incrementing, and uniform across an event
145  uint32_t format : 4;
146  // 0 baseline PID 3 time slices, 7 bit each
147  // 1 DO NOT USE ! reserved for tracklet end marker disambiguation
148  // 14 Tracklet test-pattern mode
149  // 15 Reserved for testing
150  } __attribute__((__packed__));
