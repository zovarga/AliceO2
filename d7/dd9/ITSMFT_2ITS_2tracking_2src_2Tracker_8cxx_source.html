59 void Tracker::clustersToTracks(std::function<void(std::string s)> logger)
60 {
61  double total{0};
62  for (int iteration = 0; iteration < mTrkParams.size(); ++iteration) {
63  mTraits->UpdateTrackingParameters(mTrkParams[iteration]);
64 
65  total += evaluateTask(&Tracker::initialiseTimeFrame, "Timeframe initialisation",
66  logger, iteration, mMemParams[iteration], mTrkParams[iteration]);
67  total += evaluateTask(&Tracker::computeTracklets, "Tracklet finding", logger);
68  total += evaluateTask(&Tracker::computeCells, "Cell finding", logger);
69  total += evaluateTask(&Tracker::findCellsNeighbours, "Neighbour finding", logger, iteration);
70  total += evaluateTask(&Tracker::findRoads, "Road finding", logger, iteration);
71  total += evaluateTask(&Tracker::findTracks, "Track finding", logger);
72  total += evaluateTask(&Tracker::extendTracks, "Extending tracks", logger);
73  }
74 
75  std::stringstream sstream;
76  if (constants::DoTimeBenchmarks) {
77  sstream << std::setw(2) << " - "
78  << "Timeframe " << mTimeFrameCounter++ << " processing completed in: " << total << "ms" << std::endl;
79  }
80  logger(sstream.str());
81 
82  if (mTimeFrame->hasMCinformation()) {
83  computeTracksMClabels();
84  }
85  rectifyClusterIndices();
86 }

98 void Tracker::findCellsNeighbours(int& iteration)
99 {
100 #ifdef OPTIMISATION_OUTPUT
101  std::ofstream off(fmt::format("cellneighs{}.txt", iteration));
102 #endif
103  for (int iLayer{0}; iLayer < mTrkParams[iteration].CellsPerRoad() - 1; ++iLayer) {
104 
105  if (mTimeFrame->getCells()[iLayer + 1].empty() ||
106  mTimeFrame->getCellsLookupTable()[iLayer].empty()) {
107  continue;
108  }
109 
110  int layerCellsNum{static_cast<int>(mTimeFrame->getCells()[iLayer].size())};
111  const int nextLayerCellsNum{static_cast<int>(mTimeFrame->getCells()[iLayer + 1].size())};
112  mTimeFrame->getCellsNeighbours()[iLayer].resize(nextLayerCellsNum);
113 
114  for (int iCell{0}; iCell < layerCellsNum; ++iCell) {
115 
116  const Cell& currentCell{mTimeFrame->getCells()[iLayer][iCell]};
117  const int nextLayerTrackletIndex{currentCell.getSecondTrackletIndex()};
118  const int nextLayerFirstCellIndex{mTimeFrame->getCellsLookupTable()[iLayer][nextLayerTrackletIndex]};
119  const int nextLayerLastCellIndex{mTimeFrame->getCellsLookupTable()[iLayer][nextLayerTrackletIndex + 1]};
120  for (int iNextCell{nextLayerFirstCellIndex}; iNextCell < nextLayerLastCellIndex; ++iNextCell) {
121 
122  Cell& nextCell{mTimeFrame->getCells()[iLayer + 1][iNextCell]};
123  if (nextCell.getFirstTrackletIndex() != nextLayerTrackletIndex) {
124  break;
125  }
126 
127 #ifdef OPTIMISATION_OUTPUT
128  bool good{mTimeFrame->getCellsLabel(iLayer)[iCell] == mTimeFrame->getCellsLabel(iLayer + 1)[iNextCell]};
129  float signedDelta{currentCell.getTanLambda() - nextCell.getTanLambda()};
130  off << fmt::format("{}\t{:d}\t{}\t{}", iLayer, good, signedDelta, signedDelta / mTrkParams[iteration].CellDeltaTanLambdaSigma) << std::endl;
131 #endif
132  mTimeFrame->getCellsNeighbours()[iLayer][iNextCell].push_back(iCell);
133 
134  const int currentCellLevel{currentCell.getLevel()};
135 
136  if (currentCellLevel >= nextCell.getLevel()) {
137  nextCell.setLevel(currentCellLevel + 1);
138  }
139  // }
140  }
141  }

145 void Tracker::findRoads(int& iteration)
146 {
147  for (int iLevel{mTrkParams[iteration].CellsPerRoad()}; iLevel >= mTrkParams[iteration].CellMinimumLevel(); --iLevel) {
148  CA_DEBUGGER(int nRoads = -mTimeFrame->getRoads().size());
149  const int minimumLevel{iLevel - 1};
150 
151  for (int iLayer{mTrkParams[iteration].CellsPerRoad() - 1}; iLayer >= minimumLevel; --iLayer) {
152 
153  const int levelCellsNum{static_cast<int>(mTimeFrame->getCells()[iLayer].size())};
154 
155  for (int iCell{0}; iCell < levelCellsNum; ++iCell) {
156 
157  Cell& currentCell{mTimeFrame->getCells()[iLayer][iCell]};
158 
159  if (currentCell.getLevel() != iLevel) {
160  continue;
161  }
162 
163  mTimeFrame->getRoads().emplace_back(iLayer, iCell);
164 
167  if (iLevel == 1) {
168  continue;
169  }
170 
171  const int cellNeighboursNum{static_cast<int>(
172  mTimeFrame->getCellsNeighbours()[iLayer - 1][iCell].size())};
173  bool isFirstValidNeighbour = true;
174 
175  for (int iNeighbourCell{0}; iNeighbourCell < cellNeighboursNum; ++iNeighbourCell) {
176 
177  const int neighbourCellId = mTimeFrame->getCellsNeighbours()[iLayer - 1][iCell][iNeighbourCell];
178  const Cell& neighbourCell = mTimeFrame->getCells()[iLayer - 1][neighbourCellId];
179 
180  if (iLevel - 1 != neighbourCell.getLevel()) {
181  continue;
182  }
183 
184  if (isFirstValidNeighbour) {
185 
186  isFirstValidNeighbour = false;
187 
188  } else {
189 
190  mTimeFrame->getRoads().emplace_back(iLayer, iCell);
191  }
192 
193  traverseCellsTree(neighbourCellId, iLayer - 1);
194  }
195 
196  // TODO: crosscheck for short track iterations
197  // currentCell.setLevel(0);
198  }
199  }
200 #ifdef CA_DEBUG
201  nRoads += mTimeFrame->getRoads().size();
202  std::cout << "+++ Roads with " << iLevel + 2 << " clusters: " << nRoads << " / " << mTimeFrame->getRoads().size() << std::endl;
203 #endif
204  }

207 void Tracker::findTracks()
208 {
209  std::vector<TrackITSExt> tracks;
210  tracks.reserve(mTimeFrame->getRoads().size());
211 
212  for (auto& road : mTimeFrame->getRoads()) {
213  std::vector<int> clusters(mTrkParams[0].NLayers, constants::its::UnusedIndex);
214  int lastCellLevel = constants::its::UnusedIndex;
215  CA_DEBUGGER(int nClusters = 2);
216  int firstTracklet{constants::its::UnusedIndex};
217  std::vector<int> tracklets(mTrkParams[0].TrackletsPerRoad(), constants::its::UnusedIndex);
218 
219  for (int iCell{0}; iCell < mTrkParams[0].CellsPerRoad(); ++iCell) {
220  const int cellIndex = road[iCell];
221  if (cellIndex == constants::its::UnusedIndex) {
222  continue;
223  } else {
224  if (firstTracklet == constants::its::UnusedIndex) {
225  firstTracklet = iCell;
226  }
227  tracklets[iCell] = mTimeFrame->getCells()[iCell][cellIndex].getFirstTrackletIndex();
228  tracklets[iCell + 1] = mTimeFrame->getCells()[iCell][cellIndex].getSecondTrackletIndex();
229  clusters[iCell] = mTimeFrame->getCells()[iCell][cellIndex].getFirstClusterIndex();
230  clusters[iCell + 1] = mTimeFrame->getCells()[iCell][cellIndex].getSecondClusterIndex();
231  clusters[iCell + 2] = mTimeFrame->getCells()[iCell][cellIndex].getThirdClusterIndex();
232  assert(clusters[iCell] != constants::its::UnusedIndex &&
233  clusters[iCell + 1] != constants::its::UnusedIndex &&
234  clusters[iCell + 2] != constants::its::UnusedIndex);
235  lastCellLevel = iCell;
236  CA_DEBUGGER(nClusters++);
237  }
238  }
239 
240  CA_DEBUGGER(assert(nClusters >= mTrkParams[0].MinTrackLength));
241  int count{1};
242  unsigned short rof{mTimeFrame->getTracklets()[firstTracklet][tracklets[firstTracklet]].rof[0]};
243  for (int iT = firstTracklet; iT < 6; ++iT) {
244  if (tracklets[iT] == constants::its::UnusedIndex) {
245  continue;
246  }
247  if (rof == mTimeFrame->getTracklets()[iT][tracklets[iT]].rof[1]) {
248  count++;
249  } else {
250  if (count == 1) {
251  rof = mTimeFrame->getTracklets()[iT][tracklets[iT]].rof[1];
252  } else {
253  count--;
254  }
255  }
256  }
257 
258  CA_DEBUGGER(assert(nClusters >= mTrkParams[0].MinTrackLength));
259  CA_DEBUGGER(roadCounters[nClusters - 4]++);
260 
261  if (lastCellLevel == constants::its::UnusedIndex) {
262  continue;
263  }
264 
266  for (int iC{0}; iC < clusters.size(); iC++) {
267  if (clusters[iC] != constants::its::UnusedIndex) {
268  clusters[iC] = mTimeFrame->getClusters()[iC][clusters[iC]].clusterId;
269  }
270  }
271 
273  const auto& cluster1_glo = mTimeFrame->getUnsortedClusters()[lastCellLevel + 2].at(clusters[lastCellLevel + 2]);
274  const auto& cluster2_glo = mTimeFrame->getUnsortedClusters()[lastCellLevel + 1].at(clusters[lastCellLevel + 1]);
275  const auto& cluster3_glo = mTimeFrame->getUnsortedClusters()[lastCellLevel].at(clusters[lastCellLevel]);
276 
277  const auto& cluster3_tf = mTimeFrame->getTrackingFrameInfoOnLayer(lastCellLevel).at(clusters[lastCellLevel]);
278 
280  TrackITSExt temporaryTrack{buildTrackSeed(cluster1_glo, cluster2_glo, cluster3_glo, cluster3_tf, mTimeFrame->getPositionResolution(lastCellLevel))};
281  for (size_t iC = 0; iC < clusters.size(); ++iC) {
282  temporaryTrack.setExternalClusterIndex(iC, clusters[iC], clusters[iC] != constants::its::UnusedIndex);
283  }
284  bool fitSuccess = fitTrack(temporaryTrack, mTrkParams[0].NLayers - 4, -1, -1);
285  if (!fitSuccess) {
286  continue;
287  }
288  CA_DEBUGGER(fitCounters[nClusters - 4]++);
289  temporaryTrack.resetCovariance();
290  fitSuccess = fitTrack(temporaryTrack, 0, mTrkParams[0].NLayers, 1, mTrkParams[0].FitIterationMaxChi2[0]);
291  if (!fitSuccess) {
292  continue;
293  }
294  CA_DEBUGGER(backpropagatedCounters[nClusters - 4]++);
295  temporaryTrack.getParamOut() = temporaryTrack;
296  temporaryTrack.resetCovariance();
297  fitSuccess = fitTrack(temporaryTrack, mTrkParams[0].NLayers - 1, -1, -1, mTrkParams[0].FitIterationMaxChi2[1], 50.);
298  if (!fitSuccess) {
299  continue;
300  }
301  // temporaryTrack.setROFrame(rof);
302  tracks.emplace_back(temporaryTrack);
303  }
304 
305  if (mApplySmoothing) {
306  // Smoothing tracks
307  }
308  std::sort(tracks.begin(), tracks.end(),
309  [](TrackITSExt& track1, TrackITSExt& track2) { return track1.isBetter(track2, 1.e6f); });
310 
311  for (auto& track : tracks) {
312  int nShared = 0;
313  for (int iLayer{0}; iLayer < mTrkParams[0].NLayers; ++iLayer) {
314  if (track.getClusterIndex(iLayer) == constants::its::UnusedIndex) {
315  continue;
316  }
317  nShared += int(mTimeFrame->isClusterUsed(iLayer, track.getClusterIndex(iLayer)));
318  }
319 
320  if (nShared > mTrkParams[0].ClusterSharing) {
321  continue;
322  }
323 
324  std::array<int, 3> rofs{INT_MAX, INT_MAX, INT_MAX};
325  for (int iLayer{0}; iLayer < mTrkParams[0].NLayers; ++iLayer) {
326  if (track.getClusterIndex(iLayer) == constants::its::UnusedIndex) {
327  continue;
328  }
329  mTimeFrame->markUsedCluster(iLayer, track.getClusterIndex(iLayer));
330  int currentROF = mTimeFrame->getClusterROF(iLayer, track.getClusterIndex(iLayer));
331  for (int iR{0}; iR < 3; ++iR) {
332  if (rofs[iR] == INT_MAX) {
333  rofs[iR] = currentROF;
334  }
335  if (rofs[iR] == currentROF) {
336  break;
337  }
338  }
339  }
340  if (rofs[2] != INT_MAX) {
341  continue;
342  }
343  if (rofs[1] != INT_MAX) {
344  track.setNextROFbit();
345  }
346  mTimeFrame->getTracks(std::min(rofs[0], rofs[1])).emplace_back(track);
347  }
348 }

354 bool Tracker::fitTrack(TrackITSExt& track, int start, int end, int step, const float chi2cut, const float maxQoverPt)
355 {
356  auto propInstance = o2::base::Propagator::Instance();
357  track.setChi2(0);
358  int nCl{0};
359  for (int iLayer{start}; iLayer != end; iLayer += step) {
360  if (track.getClusterIndex(iLayer) == constants::its::UnusedIndex) {
361  continue;
362  }
363  const TrackingFrameInfo& trackingHit = mTimeFrame->getTrackingFrameInfoOnLayer(iLayer).at(track.getClusterIndex(iLayer));
364 
365  if (!track.rotate(trackingHit.alphaTrackingFrame)) {
366  return false;
367  }
368 
369  if (!propInstance->propagateToX(track, trackingHit.xTrackingFrame, getBz(), o2::base::PropagatorImpl<float>::MAX_SIN_PHI, o2::base::PropagatorImpl<float>::MAX_STEP, mCorrType)) {
370  return false;
371  }
372 
373  if (mCorrType == o2::base::PropagatorF::MatCorrType::USEMatCorrNONE) {
374  float radl = 9.36f; // Radiation length of Si [cm]
375  float rho = 2.33f; // Density of Si [g/cm^3]
376  if (!track.correctForMaterial(mTrkParams[0].LayerxX0[iLayer], mTrkParams[0].LayerxX0[iLayer] * radl * rho, true)) {
377  continue;
378  }
379  }
380 
381  GPUArray<float, 3> cov{trackingHit.covarianceTrackingFrame};
382  cov[0] = std::hypot(cov[0], mTrkParams[0].LayerMisalignment[iLayer]);
383  cov[2] = std::hypot(cov[2], mTrkParams[0].LayerMisalignment[iLayer]);
384  auto predChi2{track.getPredictedChi2(trackingHit.positionTrackingFrame, cov)};
385  if (nCl >= 3 && predChi2 > chi2cut * (nCl * 2 - 5)) {
386  return false;
387  }
388  track.setChi2(track.getChi2() + predChi2);
389  if (!track.o2::track::TrackParCov::update(trackingHit.positionTrackingFrame, cov)) {
390  return false;
391  }
392  nCl++;
393  }
394  return std::abs(track.getQ2Pt()) < maxQoverPt;
395 }

397 void Tracker::traverseCellsTree(const int currentCellId, const int currentLayerId)
398 {
399  Cell& currentCell{mTimeFrame->getCells()[currentLayerId][currentCellId]};
400  const int currentCellLevel = currentCell.getLevel();
401 
402  mTimeFrame->getRoads().back().addCell(currentLayerId, currentCellId);
403 
404  if (currentLayerId > 0 && currentCellLevel > 1) {
405  const int cellNeighboursNum{static_cast<int>(
406  mTimeFrame->getCellsNeighbours()[currentLayerId - 1][currentCellId].size())};
407  bool isFirstValidNeighbour = true;
408 
409  for (int iNeighbourCell{0}; iNeighbourCell < cellNeighboursNum; ++iNeighbourCell) {
410 
411  const int neighbourCellId =
412  mTimeFrame->getCellsNeighbours()[currentLayerId - 1][currentCellId][iNeighbourCell];
413  const Cell& neighbourCell = mTimeFrame->getCells()[currentLayerId - 1][neighbourCellId];
414 
415  if (currentCellLevel - 1 != neighbourCell.getLevel()) {
416  continue;
417  }
418 
419  if (isFirstValidNeighbour) {
420  isFirstValidNeighbour = false;
421  } else {
422  mTimeFrame->getRoads().push_back(mTimeFrame->getRoads().back());
423  }
424 
425  traverseCellsTree(neighbourCellId, currentLayerId - 1);
426  }
427  }

430  // currentCell.setLevel(0);

437  return;

439

441 
442  int roadsNum{static_cast<int>(mTimeFrame->getRoads().size())};

445 
446  Road& currentRoad{mTimeFrame->getRoads()[iRoad]};
447  std::vector<std::pair<MCCompLabel, size_t>> occurrences;
448  bool isFakeRoad{false};

450

458  break;
459  }
460  }
461 
462  const Cell& currentCell{mTimeFrame->getCells()[iCell][currentCellIndex]};
463 
464  if (isFirstRoadCell) {
465 
466  const int cl0index{mTimeFrame->getClusters()[iCell][currentCell.getFirstClusterIndex()].clusterId};
467  auto cl0labs{mTimeFrame->getClusterLabels(iCell, cl0index)};
468  bool found{false};
469  for (size_t iOcc{0}; iOcc < occurrences.size(); ++iOcc) {
470  std::pair<o2::MCCompLabel, size_t>& occurrence = occurrences[iOcc];
471  for (auto& label : cl0labs) {

476  }
477  }
478  }
479  if (!found) {
480  for (auto& label : cl0labs) {

483  }
484 
485  const int cl1index{mTimeFrame->getClusters()[iCell + 1][currentCell.getSecondClusterIndex()].clusterId};
486 
487  const auto& cl1labs{mTimeFrame->getClusterLabels(iCell + 1, cl1index)};
488  found = false;
489  for (size_t iOcc{0}; iOcc < occurrences.size(); ++iOcc) {
490  std::pair<o2::MCCompLabel, size_t>& occurrence = occurrences[iOcc];
491  for (auto& label : cl1labs) {
492  if (label == occurrence.first) {
493  ++occurrence.second;
494  found = true;
495  // break; // uncomment to stop to the first hit
496  }
497  }
498  }
499  if (!found) {
500  for (auto& label : cl1labs) {
501  occurrences.emplace_back(label, 1);
502  }
503  }
504 
505  isFirstRoadCell = false;
506  }
507 
508  const int cl2index{mTimeFrame->getClusters()[iCell + 2][currentCell.getThirdClusterIndex()].clusterId};
509  const auto& cl2labs{mTimeFrame->getClusterLabels(iCell + 2, cl2index)};
510  bool found{false};
511  for (size_t iOcc{0}; iOcc < occurrences.size(); ++iOcc) {
512  std::pair<o2::MCCompLabel, size_t>& occurrence = occurrences[iOcc];
513  for (auto& label : cl2labs) {
514  if (label == occurrence.first) {
515  ++occurrence.second;
516  found = true;
517  // break; // uncomment to stop to the first hit
518  }
519  }
520  }
521  if (!found) {
522  for (auto& label : cl2labs) {
523  occurrences.emplace_back(label, 1);
524  }
525  }
526  }
527 
528  std::sort(occurrences.begin(), occurrences.end(), [](auto e1, auto e2) {
529  return e1.second > e2.second;
530  });
531 
532  auto maxOccurrencesValue = occurrences[0].first;
533  mTimeFrame->setRoadLabel(iRoad, maxOccurrencesValue.getRawValue(), isFakeRoad);
534  }

537 void Tracker::computeTracksMClabels()
538 {
539 
540  for (int iROF{0}; iROF < mTimeFrame->getNrof(); ++iROF) {
541  for (auto& track : mTimeFrame->getTracks(iROF)) {
542  std::vector<std::pair<MCCompLabel, size_t>> occurrences;
543  occurrences.clear();
544  bool isFakeTrack{false};
545 
546  for (int iCluster = 0; iCluster < TrackITSExt::MaxClusters; ++iCluster) {
547  const int index = track.getClusterIndex(iCluster);
548  if (index == constants::its::UnusedIndex) {
549  continue;
550  }
551  auto labels = mTimeFrame->getClusterLabels(iCluster, index);
552  bool found{false};
553  for (size_t iOcc{0}; iOcc < occurrences.size(); ++iOcc) {
554  std::pair<o2::MCCompLabel, size_t>& occurrence = occurrences[iOcc];
555  for (auto& label : labels) {
556  if (label == occurrence.first) {
557  ++occurrence.second;
558  found = true;
559  // break; // uncomment to stop to the first hit
560  }
561  }
562  }
563  if (!found) {
564  for (auto& label : labels) {
565  occurrences.emplace_back(label, 1);
566  }
567  }
568  }
569  std::sort(std::begin(occurrences), std::end(occurrences), [](auto e1, auto e2) {
570  return e1.second > e2.second;
571  });
572 
573  auto maxOccurrencesValue = occurrences[0].first;
574  uint32_t pattern = track.getPattern();
575  // set fake clusters pattern
576  for (int ic{TrackITSExt::MaxClusters}; ic--;) {
577  auto clid = track.getClusterIndex(ic);
578  if (clid != constants::its::UnusedIndex) {
579  auto labelsSpan = mTimeFrame->getClusterLabels(ic, clid);
580  for (auto& currentLabel : labelsSpan) {
581  if (currentLabel == maxOccurrencesValue) {
582  pattern |= 0x1 << (16 + ic); // set bit if correct
583  break;
584  }
585  }
586  }
587  }
588  track.setPattern(pattern);
589  if (occurrences[0].second < track.getNumberOfClusters()) {
590  maxOccurrencesValue.setFakeFlag();
591  }
592  mTimeFrame->getTracksLabel(iROF).emplace_back(maxOccurrencesValue);
593  }
594  }

615 track::TrackParCov Tracker::buildTrackSeed(const Cluster& cluster1, const Cluster& cluster2,
616  const Cluster& cluster3, const TrackingFrameInfo& tf3, float resolution)
617 {
618  const float ca = std::cos(tf3.alphaTrackingFrame), sa = std::sin(tf3.alphaTrackingFrame);
619  const float x1 = cluster1.xCoordinate * ca + cluster1.yCoordinate * sa;
620  const float y1 = -cluster1.xCoordinate * sa + cluster1.yCoordinate * ca;
621  const float z1 = cluster1.zCoordinate;
622  const float x2 = cluster2.xCoordinate * ca + cluster2.yCoordinate * sa;
623  const float y2 = -cluster2.xCoordinate * sa + cluster2.yCoordinate * ca;
624  const float z2 = cluster2.zCoordinate;
625  const float x3 = tf3.xTrackingFrame;
626  const float y3 = tf3.positionTrackingFrame[0];
627  const float z3 = tf3.positionTrackingFrame[1];
628 
629  const float crv = math_utils::computeCurvature(x1, y1, x2, y2, x3, y3);
630  const float x0 = math_utils::computeCurvatureCentreX(x1, y1, x2, y2, x3, y3);
631  const float tgl12 = math_utils::computeTanDipAngle(x1, y1, x2, y2, z1, z2);
632  const float tgl23 = math_utils::computeTanDipAngle(x2, y2, x3, y3, z2, z3);
633 
634  const float fy = 1. / (cluster2.radius - cluster3.radius);
635  const float& tz = fy;
636  float cy = 1.e15f;
637  if (std::abs(getBz()) > o2::constants::math::Almost0) {
638  cy = (math_utils::computeCurvature(x1, y1, x2, y2 + resolution, x3, y3) - crv) /
639  (resolution * getBz() * o2::constants::math::B2C) *
640  20.f; // FIXME: MS contribution to the cov[14] (*20 added)
641  }
642  const float s2 = resolution;
643 
644  return track::TrackParCov(tf3.xTrackingFrame, tf3.alphaTrackingFrame,
645  {y3, z3, crv * (x3 - x0), 0.5f * (tgl12 + tgl23),
646  std::abs(getBz()) < o2::constants::math::Almost0 ? 1.f / o2::track::kMostProbablePt
647  : crv / (getBz() * o2::constants::math::B2C)},
648  {s2, 0.f, s2, s2 * fy, 0.f, s2 * fy * fy, 0.f, s2 * tz, 0.f, s2 * tz * tz, s2 * cy, 0.f,
649  s2 * fy * cy, 0.f, s2 * cy * cy});
650 }
