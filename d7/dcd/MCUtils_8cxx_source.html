72 bool MCTrackNavigator::isPhysicalPrimary(o2::MCTrack const& p, std::vector<o2::MCTrack> const& pcontainer)
73 {
74  // Test if a particle is a physical primary according to the following definition:
75  // Particles produced in the collision including products of strong and
76  // electromagnetic decay and excluding feed-down from weak decays of strange
77  // particles.
78  //
79 
80  const int ist = p.getStatusCode(); // the generator status code
81  const int pdg = std::abs(p.GetPdgCode());
82  //
83  // Initial state particle
84  // Solution for K0L decayed by Pythia6
85  // ->
86  // ist > 1 --> essentially means unstable
87  if ((ist > 1) && (pdg != 130) && p.isPrimary()) {
88  return false;
89  }
90  if ((ist > 1) && p.isSecondary()) {
91  return false;
92  }
93  // <-
94 
95  if (!isStable(pdg)) {
96  return false;
97  }
98 
99  if (p.isPrimary()) {
100  //
101  // Particle produced by generator
102  // Solution for K0L decayed by Pythia6
103  // ->
104  const auto ipm = getMother(p, pcontainer);
105  if (ipm != nullptr) {
106  if (std::abs(ipm->GetPdgCode()) == 130) {
107  return false;
108  }
109  }
110  // <-
111  // check for direct photon in parton shower
112  // ->
113  if (pdg == 22) {
114  const auto ipd = getDaughter(p, pcontainer);
115  if (ipd && ipd->GetPdgCode() == 22) {
116  return false;
117  }
118  }
119  // <-
120  return true;
121  } else {
122  //
123  // Particle produced during transport
124  //
125 
126  auto pm = getMother(p, pcontainer);
127  int mpdg = std::abs(pm->GetPdgCode());
128  // Check for Sigma0
129  if ((mpdg == 3212) && pm->isPrimary()) {
130  return true;
131  }
132 
133  //
134  // Check if it comes from a pi0 decay
135  //
136  if ((mpdg == kPi0) && pm->isPrimary()) {
137  return true;
138  }
139 
140  // Check if this is a heavy flavor decay product
141  int mfl = int(mpdg / std::pow(10, int(std::log10(mpdg))));
142  //
143  // Light hadron
144  if (mfl < 4) {
145  return false;
146  }
147 
148  //
149  // Heavy flavor hadron produced by generator
150  if (pm->isPrimary()) {
151  return true;
152  }
153 
154  // To be sure that heavy flavor has not been produced in a secondary interaction
155  // Loop back to the generated mother
156  while (!pm->isPrimary()) {
157  pm = getMother(*pm, pcontainer);
158  }
159  mpdg = std::abs(pm->GetPdgCode());
160  mfl = int(mpdg / std::pow(10, int(std::log10(mpdg))));
161 
162  if (mfl < 4) {
163  return false;
164  } else {
165  return true;
166  }
167  } // end else branch produced by generator
168 }

171 {
