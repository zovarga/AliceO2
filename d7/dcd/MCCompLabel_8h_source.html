19 // Composed Label to encode MC track id, event it comes from and the source (file)
20 
21 class MCCompLabel
22 {
23  private:
24  static constexpr uint64_t ul0x1 = 0x1;
25  static constexpr uint64_t NotSet = 0xffffffffffffffff;
26  static constexpr uint64_t Noise = 0xfffffffffffffffe;
27  static constexpr uint64_t Fake = ul0x1 << 63;
28  static constexpr int NReservedBits = 1;
29 
30  uint64_t mLabel = NotSet;
31 
32  public:
33  // number of bits reserved for MC track ID, DON'T modify this, since the
34  // track ID might be negative
35  static constexpr int nbitsTrackID = 31; // number of bits reserved for MC track ID
36  static constexpr int nbitsEvID = 19; // number of bits reserved for MC event ID
37  static constexpr int nbitsSrcID = 8; // number of bits reserved for MC source ID
38  // the rest of the bits is reserved at the moment
39 
40  // check if the fields are defined consistently
41  static_assert(nbitsTrackID + nbitsEvID + nbitsSrcID <= sizeof(uint64_t) * 8 - NReservedBits,
42  "Fields cannot be stored in 64 bits");
43 
44  // mask to extract MC track ID
45  static constexpr uint64_t maskTrackID = (ul0x1 << nbitsTrackID) - 1;
46  // mask to extract MC track ID
47  static constexpr uint64_t maskEvID = (ul0x1 << nbitsEvID) - 1;
48  // mask to extract MC track ID
49  static constexpr uint64_t maskSrcID = (ul0x1 << nbitsSrcID) - 1;
50  // mask for all used fields
51  static constexpr uint64_t maskFull = (ul0x1 << (nbitsTrackID + nbitsEvID + nbitsSrcID)) - 1;
52 
53  MCCompLabel(int trackID, int evID, int srcID, bool fake = false) { set(trackID, evID, srcID, fake); }
54  MCCompLabel(bool noise = false)
55  {
56  if (noise) {
57  mLabel = Noise;
58  } else {
59  mLabel = NotSet;
60  }
61  }
62  ~MCCompLabel() = default;
63 
64  // check if label was assigned
65  bool isSet() const { return mLabel != NotSet; }
66  // check if label was not assigned
67  bool isEmpty() const { return mLabel == NotSet; }
68  // check if label corresponds to real particle
69  bool isNoise() const { return mLabel == Noise; }
70  // check if label was assigned as for correctly identified particle
71  bool isValid() const { return isSet() && !isNoise(); }
72 
73  // check if label was assigned as for incorrectly identified particle or not set or noise
74  bool isFake() const { return mLabel & Fake; }
75  // check if label was assigned as for correctly identified particle
76  bool isCorrect() const { return !isFake(); }
77 
78  // return 1 if the tracks are the same and correctly identified
79  // 0 if the tracks are the same but at least one of them is fake
80  // -1 otherwhise
81  int compare(const MCCompLabel& other) const
82  {
83  if (getEventID() != other.getEventID() || getSourceID() != other.getSourceID()) {
84  return -1;
85  }
86  int tr1 = getTrackID(), tr2 = other.getTrackID();
87  return (tr1 == tr2) ? ((isCorrect() && other.isCorrect()) ? 1 : 0) : -1;
88  }
89 
90  // allow to retrieve bare label
91  uint64_t getRawValue() const { return mLabel; }
92 
93  // comparison operator, compares only label, not eventual weight or correctness info
94  bool operator==(const MCCompLabel& other) const { return (mLabel & maskFull) == (other.mLabel & maskFull); }
95  bool operator!=(const MCCompLabel& other) const { return (mLabel & maskFull) != (other.mLabel & maskFull); }
96  // relation operators needed for some sorting methods
97  bool operator<(const MCCompLabel& other) const { return (mLabel & maskFull) < (other.mLabel & maskFull); }
98  bool operator>(const MCCompLabel& other) const { return (mLabel & maskFull) > (other.mLabel & maskFull); }
99 
100  // invalidate
101  void unset() { mLabel = NotSet; }
102  void setNoise() { mLabel = Noise; }
103  void setFakeFlag(bool v = true)
104  {
105  if (v) {
106  mLabel |= Fake;
107  } else {
108  mLabel &= ~Fake;
109  }
110  }
111 
112  void set(unsigned int trackID, int evID, int srcID, bool fake)
113  {
115  mLabel = (maskTrackID & static_cast<uint64_t>(trackID)) |
116  (maskEvID & static_cast<uint64_t>(evID)) << nbitsTrackID |
117  (maskSrcID & static_cast<uint64_t>(srcID)) << (nbitsTrackID + nbitsEvID);
118  if (fake) {
119  setFakeFlag();
120  }
121  }
122 
123  int getTrackID() const { return static_cast<int>(mLabel & maskTrackID); }
124  int getTrackIDSigned() const { return isFake() ? -getTrackID() : getTrackID(); }
125  int getEventID() const { return (mLabel >> nbitsTrackID) & maskEvID; }
126  int getSourceID() const { return (mLabel >> (nbitsTrackID + nbitsEvID)) & maskSrcID; }
127  uint64_t getTrackEventSourceID() const { return static_cast<uint64_t>(mLabel & maskFull); }
128  void get(int& trackID, int& evID, int& srcID, bool& fake)
129  {
131  trackID = getTrackID();
132  evID = getEventID();
133  srcID = getSourceID();
134  fake = isFake();
135  }
136 
137  void print() const;
138 
139  static constexpr int maxSourceID() { return maskSrcID; }
140  static constexpr int maxEventID() { return maskEvID; }
141  static constexpr int maxTrackID() { return maskTrackID; }
142  ClassDefNV(MCCompLabel, 1);
143 };
