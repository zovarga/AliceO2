225 template <typename DataT, int YdimT>
226 class Spline2DSpec<DataT, YdimT, 0>
227  : public Spline2DContainer<DataT>
228 {
229  typedef Spline2DContainer<DataT> TBase;
230 
231  public:
232  typedef typename TBase::SafetyLevel SafetyLevel;
233  typedef typename TBase::Knot Knot;
234 
236 
238  GPUd() void interpolate(DataT x1, DataT x2, GPUgeneric() DataT S[/*mYdim*/]) const
239  {
240  interpolateU<SafetyLevel::kSafe>(mYdim, mParameters, mGridX1.convXtoU(x1), mGridX2.convXtoU(x2), S);
241  }
242 
244  template <SafetyLevel SafeT = SafetyLevel::kSafe>
245  GPUd() void interpolateU(int inpYdim, GPUgeneric() const DataT Parameters[],
246  DataT u1, DataT u2, GPUgeneric() DataT S[/*inpYdim*/]) const
247  {
248 
249  const auto nYdimTmp = SplineUtil::getNdim<YdimT>(inpYdim);
250  const int nYdim = nYdimTmp.get();
251 
252  const auto maxYdim = SplineUtil::getMaxNdim<YdimT>(inpYdim);
253  const int maxYdim4 = 4 * maxYdim.get();
254 
255  const auto nYdim2 = nYdim * 2;
256  const auto nYdim4 = nYdim * 4;
257 
258  const float& u = u1;
259  const float& v = u2;
260  int nu = mGridX1.getNumberOfKnots();
261  int iu = mGridX1.template getLeftKnotIndexForU<SafeT>(u);
262  int iv = mGridX2.template getLeftKnotIndexForU<SafeT>(v);
263 
264  const typename TBase::Knot& knotU = mGridX1.template getKnot<SafetyLevel::kNotSafe>(iu);
265  const typename TBase::Knot& knotV = mGridX2.template getKnot<SafetyLevel::kNotSafe>(iv);
266 
267  const DataT* par00 = Parameters + (nu * iv + iu) * nYdim4; // values { {Y1,Y2,Y3}, {Y1,Y2,Y3}'v, {Y1,Y2,Y3}'u, {Y1,Y2,Y3}''vu } at {u0, v0}
268  const DataT* par10 = par00 + nYdim4; // values { ... } at {u1, v0}
269  const DataT* par01 = par00 + nYdim4 * nu; // values { ... } at {u0, v1}
270  const DataT* par11 = par01 + nYdim4; // values { ... } at {u1, v1}
271 
272  DataT Su0[maxYdim4]; // values { {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v0), {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v1) }, at u0
273  DataT Du0[maxYdim4]; // derivatives {}'_u at u0
274  DataT Su1[maxYdim4]; // values { {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v0), {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v1) }, at u1
275  DataT Du1[maxYdim4]; // derivatives {}'_u at u1
276 
277  for (int i = 0; i < nYdim2; i++) {
278  Su0[i] = par00[i];
279  Su0[nYdim2 + i] = par01[i];
280 
281  Du0[i] = par00[nYdim2 + i];
282  Du0[nYdim2 + i] = par01[nYdim2 + i];
283 
284  Su1[i] = par10[i];
285  Su1[nYdim2 + i] = par11[i];
286 
287  Du1[i] = par10[nYdim2 + i];
288  Du1[nYdim2 + i] = par11[nYdim2 + i];
289  }
290 
291  DataT parU[maxYdim4]; // interpolated values { {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v0), {Y1,Y2,Y3,Y1'v,Y2'v,Y3'v}(v1) } at u
292 
293  typedef Spline1DSpec<DataT, 4 * YdimT, 0> TGridX1;
294  const TGridX1& gridX1 = reinterpret_cast<const TGridX1&>(mGridX1);
295 
296  gridX1.interpolateU(nYdim4, knotU, Su0, Du0, Su1, Du1, u, parU);
297 
298  const DataT* Sv0 = parU + 0;
299  const DataT* Dv0 = parU + nYdim;
300  const DataT* Sv1 = parU + nYdim2;
301  const DataT* Dv1 = parU + nYdim2 + nYdim;
302 
303  typedef Spline1DSpec<DataT, YdimT, 0> TGridX2;
304  const TGridX2& gridX2 = reinterpret_cast<const TGridX2&>(mGridX2);
305  gridX2.interpolateU(nYdim, knotV, Sv0, Dv0, Sv1, Dv1, v, S);
306  }
307 
308  protected:
309  using TBase::mGridX1;
310  using TBase::mGridX2;
311  using TBase::mParameters;
312  using TBase::mYdim;
313  using TBase::TBase; // inherit constructors and hide them
314 };
