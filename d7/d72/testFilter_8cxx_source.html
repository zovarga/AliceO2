38 std::vector<ColumnData> generateData(size_t nData = 10)
39 {
40 
41  std::random_device rd;
42  std::mt19937 mt(rd());
43  std::uniform_int_distribution<uint8_t> deIds(0, 71);
44  std::uniform_int_distribution<uint8_t> colIds(0, 6);
45  std::uniform_int_distribution<uint16_t> patterns(0, 0xFFFF);
46 
47  std::vector<o2::mid::ColumnData> data;
48  for (size_t idata = 0; idata < nData; ++idata) {
49  ColumnData col;
50  col.deId = deIds(mt);
51  col.columnId = colIds(mt);
52  for (int iline = 0; iline < 4; ++iline) {
53  col.setBendPattern(patterns(mt), iline);
54  }
55  col.setNonBendPattern(patterns(mt));
56  data.emplace_back(col);
57  }
58  return data;
59 }

136 BOOST_AUTO_TEST_CASE(FETConversion)
137 {
139 
140  // Use the masks as FET: all channels should answer
141  auto fets = makeDefaultMasks();
142 
143  // We now modify one FET data
144  uint16_t fullPattern = 0xFFFF;
145  for (auto& col : fets) {
146  if (col.deId == 3 && col.columnId == 0) {
147  col.setBendPattern(fullPattern & ~0x1, 0);
148  col.setBendPattern(fullPattern, 1);
149  col.setBendPattern(fullPattern, 2);
150  // The following does not exist in deId 3, col 0
151  col.setBendPattern(fullPattern & ~0x1, 3);
152  col.setNonBendPattern(fullPattern);
153  }
154  }
155 
156  FetToDead fetToDead;
157  auto inverted = fetToDead.process(fets);
158 
159  BOOST_TEST(inverted.size() == 1);
160  BOOST_TEST(inverted.back().deId == 3);
161  BOOST_TEST(inverted.back().columnId == 0);
162  BOOST_TEST(inverted.back().getBendPattern(0) == 1);
163  BOOST_TEST(inverted.back().getBendPattern(1) == 0);
164  BOOST_TEST(inverted.back().getBendPattern(2) == 0);
165  // Test that the non-existing pattern was not converted
166  // Since the mask in the FET conversion takes care of removing it
167  BOOST_TEST(inverted.back().getBendPattern(3) == 0);
168  BOOST_TEST(inverted.back().getNonBendPattern() == 0);
169 }
