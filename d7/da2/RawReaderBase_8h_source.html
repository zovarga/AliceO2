58  //decoding binary data into data blocks
59  template <class DataBlockType>
60  size_t decodeBlocks(const gsl::span<const uint8_t> binaryPayload, std::vector<DataBlockType>& vecDataBlocks)
61  {
62  size_t srcPos = 0;
63  while (srcPos < binaryPayload.size()) {
64  auto& refDataBlock = vecDataBlocks.emplace_back();
65  refDataBlock.decodeBlock(binaryPayload, srcPos);
66  srcPos += refDataBlock.mSize;
67  if (!refDataBlock.isCorrect()) {
68  LOG(warning) << "INCORRECT DATA BLOCK! Byte position: " << srcPos - refDataBlock.mSize << " | Payload size: " << binaryPayload.size() << " | DataBlock size: " << refDataBlock.mSize;
69  refDataBlock.print();
70  vecDataBlocks.pop_back();
71  return srcPos;
72  }
73  if (refDataBlock.getNgbtWords() == 0) {
74  vecDataBlocks.pop_back();
75  continue;
76  }
77  }
78  return srcPos;
79  }
