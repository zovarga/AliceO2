63 void o2::tpc::IDCDrawHelper::drawSide(const IDCDraw& idc, const o2::tpc::Side side, const std::string zAxisTitle, const std::string filename, const float minZ, const float maxZ)
64 {
65  const auto coords = o2::tpc::painter::getPadCoordinatesSector();
66  TH2Poly* poly = o2::tpc::painter::makeSideHist(side);
67  poly->SetContour(255);
68  poly->SetTitle(nullptr);
69  poly->GetXaxis()->SetTitleOffset(1.2f);
70  poly->GetYaxis()->SetTitleOffset(1.3f);
71  poly->GetZaxis()->SetTitleOffset(1.4f);
72  poly->GetZaxis()->SetTitle(zAxisTitle.data());
73  poly->GetZaxis()->SetMaxDigits(3); // force exponential axis
74  poly->SetStats(0);
75  if (minZ < maxZ) {
76  poly->SetMinimum(minZ);
77  poly->SetMaximum(maxZ);
78  }
79 
80  TCanvas* can = new TCanvas("can", "can", 650, 600);
81  can->SetTopMargin(0.04f);
82  can->SetRightMargin(0.14f);
83  can->SetLeftMargin(0.1f);
84  poly->Draw("colz");
85 
86  unsigned int sectorStart = (side == Side::A) ? 0 : o2::tpc::SECTORSPERSIDE;
87  unsigned int sectorEnd = (side == Side::A) ? o2::tpc::SECTORSPERSIDE : Mapper::NSECTORS;
88  for (unsigned int sector = sectorStart; sector < sectorEnd; ++sector) {
89  for (unsigned int region = 0; region < Mapper::NREGIONS; ++region) {
90  for (unsigned int irow = 0; irow < Mapper::ROWSPERREGION[region]; ++irow) {
91  for (unsigned int ipad = 0; ipad < Mapper::PADSPERROW[region][irow]; ++ipad) {
92  const auto padNum = Mapper::getGlobalPadNumber(irow, ipad, region);
93  const float angDeg = 10.f + sector * 20;
94  auto coordinate = coords[padNum];
95  coordinate.rotate(angDeg);
96  const float yPos = static_cast<float>(coordinate.yVals[0] + coordinate.yVals[1] + coordinate.yVals[2] + coordinate.yVals[3]) / 4;
97  const float xPos = static_cast<float>(coordinate.xVals[0] + coordinate.xVals[1] + coordinate.xVals[2] + coordinate.xVals[3]) / 4;
98  const auto padTmp = (side == Side::A) ? ipad : (Mapper::PADSPERROW[region][irow] - ipad - 1); // C-Side is mirrored
99  poly->Fill(xPos, yPos, idc.getIDC(sector, region, irow, padTmp));
100  }
101  }
102  }
103  }
104  if (!filename.empty()) {
105  can->SaveAs(filename.data());
106  delete poly;
107  delete can;
108  }
109 }
