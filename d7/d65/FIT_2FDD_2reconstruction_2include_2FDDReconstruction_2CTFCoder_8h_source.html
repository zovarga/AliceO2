67 template <typename VEC>
68 void CTFCoder::encode(VEC& buff, const gsl::span<const Digit>& digitVec, const gsl::span<const ChannelData>& channelVec)
69 {
70  using MD = o2::ctf::Metadata::OptStore;
71  // what to do which each field: see o2::ctd::Metadata explanation
72  constexpr MD optField[CTF::getNBlocks()] = {
73  MD::EENCODE, // BLC_trigger
74  MD::EENCODE, // BLC_bcInc
75  MD::EENCODE, // BLC_orbitInc
76  MD::EENCODE, // BLC_nChan
77 
78  MD::EENCODE, // BLC_idChan
79  MD::EENCODE, // BLC_time
80  MD::EENCODE, // BLC_charge
81  MD::EENCODE // BLC_feeBits
82  };
83  CompressedDigits cd;
84  compress(cd, digitVec, channelVec);
85 
86  // book output size with some margin
87  auto szIni = estimateCompressedSize(cd);
88  buff.resize(szIni);
89 
90  auto ec = CTF::create(buff);
91  using ECB = CTF::base;
92 
93  ec->setHeader(cd.header);
94  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
95  ec->getANSHeader().majorVersion = 0;
96  ec->getANSHeader().minorVersion = 1;
97  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
98 #define ENCODEFDD(part, slot, bits) CTF::get(buff.data())->encode(part, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
99  // clang-format off
100  ENCODEFDD(cd.trigger, CTF::BLC_trigger, 0);
101  ENCODEFDD(cd.bcInc, CTF::BLC_bcInc, 0);
102  ENCODEFDD(cd.orbitInc, CTF::BLC_orbitInc, 0);
103  ENCODEFDD(cd.nChan, CTF::BLC_nChan, 0);
104 
105  ENCODEFDD(cd.idChan , CTF::BLC_idChan, 0);
106  ENCODEFDD(cd.time, CTF::BLC_time, 0);
107  ENCODEFDD(cd.charge, CTF::BLC_charge, 0);
108  ENCODEFDD(cd.feeBits, CTF::BLC_feeBits, 0);
109  // clang-format on
110  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
111 }

138 template <typename VDIG, typename VCHAN>
139 void CTFCoder::decompress(const CompressedDigits& cd, VDIG& digitVec, VCHAN& channelVec)
140 {
141  digitVec.clear();
142  channelVec.clear();
143  digitVec.reserve(cd.header.nTriggers);
144  channelVec.reserve(cd.idChan.size());
145 
146  uint32_t firstEntry = 0, clCount = 0, chipCount = 0;
147  o2::InteractionRecord ir(cd.header.firstBC, cd.header.firstOrbit);
148 
149  for (uint32_t idig = 0; idig < cd.header.nTriggers; idig++) {
150  // restore ROFRecord
151  if (cd.orbitInc[idig]) { // non-0 increment => new orbit
152  ir.bc = cd.bcInc[idig]; // bcInc has absolute meaning
153  ir.orbit += cd.orbitInc[idig];
154  } else {
155  ir.bc += cd.bcInc[idig];
156  }
157  Triggers trig;
158  trig.triggersignals = cd.trigger[idig];
159 
160  firstEntry = channelVec.size();
161  uint8_t chID = 0;
162  int amplA = 0, amplC = 0, timeA = 0, timeC = 0;
163  for (uint8_t ic = 0; ic < cd.nChan[idig]; ic++) {
164  auto icc = channelVec.size();
165  const auto& chan = channelVec.emplace_back((chID += cd.idChan[icc]), cd.time[icc], cd.charge[icc], cd.feeBits[icc]);
166  //
167  // rebuild digit
168  if (chan.mPMNumber > 7) { // A side
169  amplA += chan.mChargeADC;
170  timeA += chan.mTime;
171  trig.nChanA++;
172 
173  } else {
174  amplC += chan.mChargeADC;
175  timeC += chan.mTime;
176  trig.nChanC++;
177  }
178  }
179  if (trig.nChanA) {
180  trig.timeA = timeA / trig.nChanA;
181  trig.amplA = amplA * 0.125;
182  }
183  if (trig.nChanC) {
184  trig.timeC = timeC / trig.nChanC;
185  trig.amplC = amplC * 0.125;
186  }
187  digitVec.emplace_back(firstEntry, cd.nChan[idig], ir, trig);
188  }
189 }
