31 MapCRU::MapCRU(std::string_view content)
32 {
33  mFeeLink2Solar.fill(std::numeric_limits<uint16_t>::max());
34  std::istringstream input(content.data());
35  std::string s;
36  while (std::getline(input, s)) {
37  if (s.empty()) {
38  continue;
39  }
40  std::istringstream line(s);
41  int f, l, link_id;
42  line >> link_id >> f >> l;
43  auto ix = indexFeeLink(f, l);
44  if (ix < 0) {
45  continue;
46  }
47  mFeeLink2Solar.at(ix) = link_id;
48  }
49 }

62 std::optional<FeeLinkId> MapCRU::operator()(uint16_t solarId) const
63 {
64  auto it = std::find(begin(mFeeLink2Solar),
65  end(mFeeLink2Solar), solarId);
66  if (it == mFeeLink2Solar.end()) {
67  return std::nullopt;
68  }
69  auto d = std::distance(mFeeLink2Solar.begin(), it);
70  int feeId = d / sMaxLinkId;
71  int linkId = d % sMaxLinkId;
72  return FeeLinkId(feeId, linkId);
73 }

75 int MapCRU::indexFeeLink(int feeid, int linkid) const
76 {
77  if (feeid < 0 || feeid >= sMaxFeeId) {
78  return -1;
79  }
80  if (linkid < 0 || linkid >= sMaxLinkId) {
81  return -1;
82  }
83  return feeid * sMaxLinkId + linkid;
84 }
