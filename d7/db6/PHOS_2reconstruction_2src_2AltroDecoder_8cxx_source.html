24 AltroDecoderError::ErrorType_t AltroDecoder::decode(RawReaderMemory& rawreader, CaloRawFitter* rawFitter,
25  std::vector<o2::phos::Cell>& currentCellContainer, std::vector<o2::phos::Cell>& currentTRUContainer)
26 {
27  mOutputHWErrors.clear();
28  mOutputFitChi.clear();
29 
30  try {
31  auto& header = rawreader.getRawHeader();
32  mddl = o2::raw::RDHUtils::getFEEID(header);
33  } catch (...) {
34  return AltroDecoderError::RCU_TRAILER_ERROR;
35  }
36  const std::vector<uint32_t>& payloadwords = rawreader.getPayload().getPayloadWords();
37 
38  if (payloadwords.size() == 0) {
39  return AltroDecoderError::kOK;
40  }
41 
42  try {
43  gsl::span<const uint32_t> tmp(payloadwords.data(), payloadwords.size());
44  mRCUTrailer.constructFromRawPayload(tmp);
45  } catch (RCUTrailer::Error& e) {
46  LOG(error) << "RCU trailer error " << (int)e.getErrorType();
47  mOutputHWErrors.emplace_back(mddl, kGeneralSRUErr, static_cast<char>(e.getErrorType())); //assign general SRU header errors to non-existing FEE 15
48  return AltroDecoderError::RCU_TRAILER_ERROR;
49  }
50 
51  //TODO checkRCUTrailer();
52  try {
53  readChannels(payloadwords, rawFitter, currentCellContainer, currentTRUContainer);
54  } catch (AltroDecoderError::ErrorType_t e) {
55  LOG(error) << "Altro decoding error " << e;
56  mOutputHWErrors.emplace_back(mddl, kGeneralTRUErr, static_cast<char>(e)); //assign general SRU header errors to non-existing FEE 16
57  return e;
58  }
59  return AltroDecoderError::kOK;
60 }

62 void AltroDecoder::readChannels(const std::vector<uint32_t>& buffer, CaloRawFitter* rawFitter,
63  std::vector<o2::phos::Cell>& currentCellContainer, std::vector<o2::phos::Cell>& currentTRUContainer)
64 {
65  int currentpos = 0;
66 
67  int payloadend = buffer.size() - mRCUTrailer.getTrailerSize(); //mRCUTrailer.getPayloadSize() was not updated in case of merged pages.
68  while (currentpos < payloadend) {
69  auto currentword = buffer[currentpos++];
70  ChannelHeader header = {currentword};
71  if (header.mMark != 1) {
72  if (currentword != 0) {
73  LOG(error) << "Channel header mark not found, header word " << currentword;
74  short fec = header.mHardwareAddress >> 7 & 0xf; //try to extract FEE number from header
75  short branch = header.mHardwareAddress >> 11 & 0x1;
76  if (fec > 14) {
77  fec = kGeneralSRUErr;
78  }
79  fec += kGeneralTRUErr * branch;
80  mOutputHWErrors.emplace_back(mddl, fec, 5); //5: channel header error
81  }
82  continue;
83  }
85  int numberofwords = (header.mPayloadSize + 2) / 3;
86  if (numberofwords > payloadend - currentpos) {
87  LOG(error) << "Channel payload " << numberofwords << " larger than left in total " << payloadend - currentpos;
88  short fec = header.mHardwareAddress >> 7 & 0xf; //try to extract FEE number from header
89  short branch = header.mHardwareAddress >> 11 & 0x1;
90  if (fec > 14) {
91  fec = kGeneralSRUErr;
92  }
93  fec += kGeneralTRUErr * branch;
94  mOutputHWErrors.emplace_back(mddl, fec, 6); //6: channel payload error
95  continue;
96  }
97  mBunchwords.clear();
98  int isample = 0;
99  while (isample < header.mPayloadSize) {
100  currentword = buffer[currentpos++];
101  if ((currentword >> 30) != 0) {
102  LOG(error) << "Unexpected end of payload in altro channel payload! FEE=" << mddl
103  << ", Address=0x" << std::hex << header.mHardwareAddress << ", word=0x" << currentword << std::dec;
104  currentpos--;
105  short fec = header.mHardwareAddress >> 7 & 0xf; //try to extract FEE number from header
106  short branch = header.mHardwareAddress >> 11 & 0x1;
107  if (fec > 14) {
108  fec = kGeneralSRUErr;
109  }
110  fec += kGeneralTRUErr * branch;
111  mOutputHWErrors.emplace_back(mddl, fec, 6); //6: channel payload error
112  break;
113  }
114  mBunchwords.push_back((currentword >> 20) & 0x3FF);
115  isample++;
116  if (isample < header.mPayloadSize) {
117  mBunchwords.push_back((currentword >> 10) & 0x3FF);
118  isample++;
119  if (isample < header.mPayloadSize) {
120  mBunchwords.push_back(currentword & 0x3FF);
121  isample++;
122  } else {
123  break;
124  }
125  } else {
126  break;
127  }
128  }
129  short absId;
130  Mapping::CaloFlag caloFlag;
131  if (!hwToAbsAddress(header.mHardwareAddress, absId, caloFlag)) {
132  // do not decode, skip to hext channel
133  short fec = header.mHardwareAddress >> 7 & 0xf; //try to extract FEE number from header
134  short branch = header.mHardwareAddress >> 11 & 0x1;
135  if (fec > 14) {
136  fec = kGeneralSRUErr;
137  }
138  fec += kGeneralTRUErr * branch;
139  mOutputHWErrors.emplace_back(mddl, fec, 7); //7: wrong hw address
140  continue;
141  }
142 
143  //Get time and amplitude
144  if (caloFlag != Mapping::kTRU) { //HighGain or LowGain
145  // decode bunches
146  int currentsample = 0;
147  while (currentsample < header.mPayloadSize) {
148  int bunchlength = mBunchwords[currentsample] - 2, // remove words for bunchlength and starttime
149  starttime = mBunchwords[currentsample + 1];
150  if (bunchlength < 0) { // corrupted data,
151  short fec = header.mHardwareAddress >> 7 & 0xf; //try to extract FEE number from header
152  short branch = header.mHardwareAddress >> 11 & 0x1;
153  fec += kGeneralTRUErr * branch;
154  mOutputHWErrors.emplace_back(mddl, fec, 6); //6: channel payload error
155  break;
156  }
157  //extract sample properties
158  CaloRawFitter::FitStatus fitResult = rawFitter->evaluate(gsl::span<uint16_t>(&mBunchwords[currentsample + 2], std::min((unsigned long)bunchlength, mBunchwords.size() - currentsample - 2)));
159  currentsample += bunchlength + 2;
160  //set output cell
161  // if (fitResult == CaloRawFitter::FitStatus::kNoTime) { //Time evaluation error occured: should we add this err to list?
162  // short fec = header.mHardwareAddress >> 7 & 0xf; //try to extract FEE number from header
163  // short branch = header.mHardwareAddress >> 11 & 0x1;
164  // if (fec > 14) {
165  // fec = kGeneralSRUErr;
166  // }
167  // fec += kGeneralTRUErr * branch;
168  // mOutputHWErrors.emplace_back(mddl, fec, 8); //8: time calculation failed
169  // }
170  if (!rawFitter->isOverflow()) { //Overflow is will show wrong chi2
171  short chiAddr = absId;
172  chiAddr |= caloFlag << 14;
173  mOutputFitChi.emplace_back(chiAddr);
174  mOutputFitChi.emplace_back(short(rawFitter->getChi2()));
175  }
176 
177  if (fitResult == CaloRawFitter::FitStatus::kOK || fitResult == CaloRawFitter::FitStatus::kNoTime) {
178  if (!mPedestalRun) {
179  if (caloFlag == Mapping::kHighGain && !rawFitter->isOverflow()) {
180  currentCellContainer.emplace_back(absId, rawFitter->getAmp(),
181  (rawFitter->getTime() + starttime) * o2::phos::PHOSSimParams::Instance().mTimeTick * 1.e-9, (ChannelType_t)caloFlag);
182  }
183  if (caloFlag == Mapping::kLowGain) {
184  currentCellContainer.emplace_back(absId, rawFitter->getAmp(),
185  (rawFitter->getTime() + starttime) * o2::phos::PHOSSimParams::Instance().mTimeTick * 1.e-9, (ChannelType_t)caloFlag);
186  }
187  } else { //pedestal, to store RMS, scale in by 1.e-7 to fit range
188  currentCellContainer.emplace_back(absId, rawFitter->getAmp(), 1.e-7 * rawFitter->getTime(), (ChannelType_t)caloFlag);
189  }
190  } //Successful fit
191  } //Bunched of a channel
192  } //HG or LG channel
193  else { //TRU channel
194  // Channels in TRU:
195  // There are 112 readout channels and 12 channels reserved for production flags:
196  // Channels 0-111: channel data readout
197  // Channels 112-123: production flags
198  if (Mapping::isTRUReadoutchannel(header.mHardwareAddress)) {
199  Mapping::Instance()->hwToAbsId(mddl, header.mHardwareAddress, absId, caloFlag);
200  readTRUDigits(absId, header.mPayloadSize, currentTRUContainer);
201  } else {
202  readTRUFlags(header.mHardwareAddress, header.mPayloadSize);
203  }
204  } //TRU channel
205  }
206 
207  //Scan Flags and trigger cells and left only good
208  //if trigger cell exists and the trigger flag true -add it
209  bool is4x4Trigger = mTRUFlags[Mapping::NTRUReadoutChannels];
210  for (auto rit = currentTRUContainer.rbegin(); rit != currentTRUContainer.rend(); rit++) {
211  if (mTRUFlags[rit->getTRUId()]) { //there is corresponding flag
212  if (is4x4Trigger) {
213  rit->setType(ChannelType_t::TRU4x4);
214  } else {
215  rit->setType(ChannelType_t::TRU2x2);
216  }
217  } else { //will be removed later
218  rit->setEnergy(0);
219  }
220  }

223 bool AltroDecoder::hwToAbsAddress(short hwAddr, short& absId, Mapping::CaloFlag& caloFlag)
224 {
225  //check hardware address and convert to absId and caloFlag
226 
227  if (mddl < 0 || mddl > o2::phos::Mapping::NDDL) {
228  return (char)4;
229  }
230  // short chan = hwAddr & 0xf;
231  short chip = hwAddr >> 4 & 0x7;
232  short fec = hwAddr >> 7 & 0xf;
233  short branch = hwAddr >> 11 & 0x1;
234 
235  short e2 = 0;
236  if (fec > 14) {
237  e2 = 2;
238  fec = kGeneralSRUErr;
239  mOutputHWErrors.emplace_back(mddl, fec + branch * kGeneralTRUErr, 2);
240  } else {
241  if (fec != 0 && (chip < 0 || chip > 4 || chip == 1)) { //Do not check for TRU (fec=0)
242  e2 = 3;
243  mOutputHWErrors.emplace_back(mddl, fec + branch * kGeneralTRUErr, 3);
244  }
245  }
246 
247  if (e2) {
248  return false;
249  }
250 
251  //correct hw address, try to convert
252  Mapping::ErrorStatus s = Mapping::Instance()->hwToAbsId(mddl, hwAddr, absId, caloFlag);
253  if (s != Mapping::ErrorStatus::kOK) {
254  mOutputHWErrors.emplace_back(mddl, branch * kGeneralTRUErr + kGeneralSRUErr, 4); //4: error in mapping
255  return false;
256  }
257  return true;
258 }

260 void AltroDecoder::readTRUDigits(short absId, int payloadSize, std::vector<o2::phos::Cell>& truContainer)
261 {
262  int currentsample = 0;
263  while (currentsample < payloadSize) {
264  int bunchlength = mBunchwords[currentsample] - 2; // remove words for bunchlength and starttime
265  if (bunchlength < 0) { //corrupted sample: add error and ignore the reast of bunchwords
266  mOutputHWErrors.emplace_back(mddl, kGeneralTRUErr, static_cast<char>(1)); // 1: wrong TRU header
267  return;
268  }
269  int timeBin = mBunchwords[currentsample + 1];
270  int istart = currentsample + 2;
271  int iend = std::min((unsigned long)bunchlength, mBunchwords.size() - currentsample - 2);
272  currentsample += bunchlength + 2;
273  int smax = 0, tmax = 0;
274  // Loop over all the time steps in the signal
275  for (int i = iend - 1; i >= istart; i--) {
276  if (mBunchwords[i] > smax) {
277  smax = mBunchwords[i];
278  tmax = timeBin;
279  }
280  timeBin++;
281  }
282  truContainer.emplace_back(absId + 14337 + 1, smax, tmax * 1.e-9, TRU2x2); //add TRU cells
283  }

285 void AltroDecoder::readTRUFlags(short hwAddress, int payloadSize)
286 {
287  // Production flags:
288  // Production flags are supplied in channels 112 - 123
289  // Each of the channels is 10 bit wide
290  // The bits inside the channel (indexing starting from the first bit of channel 112) is as follows:
291  // Bits 0-111: Trigger flags for corresponding channel index
292  // If using 4x4 algorithm, only 91 first bits are used of these
293  // Bit 112: Marker for 4x4 algorithm (1 active, 0 not active)
294  // Bit 113: Marker for 2x2 algorithm (1 active, 0 not active)
295  // Bit 114: Global L0 OR of all patches in the TRU
296 
297  int currentsample = 0;
298  while (currentsample < payloadSize) {
299  int bunchlength = mBunchwords[currentsample] - 2; // remove words for bunchlength and starttime
300  int timeBin = mBunchwords[currentsample + 1];
301  if (bunchlength < 0) { //corrupted sample: add error and ignore the reast of bunchwords
302  mOutputHWErrors.emplace_back(mddl, kGeneralTRUErr, static_cast<char>(1)); // 1: wrong TRU header
303  return;
304  }
305  int istart = currentsample + 2;
306  int iend = std::min((unsigned long)bunchlength, mBunchwords.size() - currentsample - 2);
307  currentsample += bunchlength + 2;
308 
309  for (int i = iend - 1; i >= istart; i--) {
310  short a = mBunchwords[i];
311  // If bit 112 is 1, we are considering 4x4 algorithm
312  if (hwAddress == Mapping::TRUFinalProductionChannel) {
313  mTRUFlags[Mapping::NTRUReadoutChannels] = (a & (1 << 2)); // Check the bit number 112
314  }
315  const int kWordLength = 10; // Length of one data word in TRU raw data
316 
317  // Assign the bits in the words to corresponding channels
318  for (Int_t bitIndex = 0; bitIndex < kWordLength; bitIndex++) {
319  // Find the correct channel number assuming that
320  // hwAddress 112 = bits 0-9 corresponding trigger flags in channels 0-9
321  // hwAddress 113 = bits 10-19 corresponding trigger flags in channels 10-19
322  // and so on
323  short channel;
324  if (hwAddress < 128) {
325  channel = (hwAddress - Mapping::NTRUBranchReadoutChannels) * kWordLength + bitIndex;
326  } else {
327  channel = 112 + (hwAddress - 2048 - Mapping::NTRUBranchReadoutChannels) * kWordLength + bitIndex; //branch 0
328  }
329  mTRUFlags[channel] = mTRUFlags[channel] | (a & (1 << bitIndex));
330  } // Bits in one word
331  } // Length of signal
332  }
