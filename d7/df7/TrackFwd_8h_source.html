60  Double_t getSnp() const
61  {
62  return o2::math_utils::sin(mParameters(2));
63  }
64 
65  Double_t getCsp2() const
66  {
67  auto snp = o2::math_utils::sin(mParameters(2));
68  Double_t csp;
69  csp = std::sqrt((1. - snp) * (1. + snp));
70  return csp * csp;
71  }
72 
73  void setTanl(Double_t tanl) { mParameters(3) = tanl; }
74  Double_t getTanl() const { return mParameters(3); }
75 
76  Double_t getTgl() const { return mParameters(3); } // for the sake of helixhelper
77 
78  void setInvQPt(Double_t invqpt) { mParameters(4) = invqpt; }
79  Double_t getInvQPt() const { return mParameters(4); } // return Inverse charged pt
80  Double_t getPt() const { return TMath::Abs(1.f / mParameters(4)); }
81  Double_t getInvPt() const { return TMath::Abs(mParameters(4)); }
82  Double_t getPx() const { return TMath::Cos(getPhi()) * getPt(); } // return px
83  Double_t getPy() const { return TMath::Sin(getPhi()) * getPt(); } // return py
84  Double_t getPz() const { return getTanl() * getPt(); } // return pz
85  Double_t getP() const { return getPt() * TMath::Sqrt(1. + getTanl() * getTanl()); } // return total momentum
86  Double_t getInverseMomentum() const { return 1.f / getP(); }
87 
88  Double_t getEta() const { return -TMath::Log(TMath::Tan((TMath::PiOver2() - TMath::ATan(getTanl())) / 2)); } // return total momentum
89 
90  Double_t getCurvature(double b) const
91  {
92  auto invqpt = getInvQPt();
93  return o2::constants::math::B2C * b * invqpt;
94  }
95 
97  Double_t getCharge() const { return TMath::Sign(1., mParameters(4)); }
99  void setCharge(Double_t charge)
100  {
101  if (charge * mParameters(4) < 0.) {
102  mParameters(4) *= -1.;
103  }
104  }
