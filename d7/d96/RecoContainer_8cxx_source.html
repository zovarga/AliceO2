389 void DataRequest::requestTracks(GTrackID::mask_t src, bool useMC)
390 {
391  // request tracks for sources probided by the mask
392  if (src[GTrackID::ITS]) {
393  requestITSTracks(useMC);
394  }
395  if (src[GTrackID::MFT]) {
396  requestMFTTracks(useMC);
397  }
398  if (src[GTrackID::MCH]) {
399  requestMCHTracks(useMC);
400  }
401  if (src[GTrackID::MID]) {
402  requestMIDTracks(useMC);
403  }
404  if (src[GTrackID::TPC]) {
405  requestTPCTracks(useMC);
406  }
407  if (src[GTrackID::ITSTPC] || src[GTrackID::ITSTPCTOF]) {
408  requestITSTPCTracks(useMC);
409  }
410  if (src[GTrackID::MFTMCH]) {
411  requestGlobalFwdTracks(useMC);
412  }
413  if (src[GTrackID::TPCTOF]) {
414  requestTPCTOFTracks(useMC);
415  }
416  if (src[GTrackID::TPCTRD] || src[GTrackID::TPCTRDTOF]) {
417  requestTPCTRDTracks(useMC);
418  }
419  if (src[GTrackID::ITSTPCTRD] || src[GTrackID::ITSTPCTRDTOF]) {
420  requestITSTPCTRDTracks(useMC);
421  }
422  if (src[GTrackID::ITSTPCTRDTOF] || src[GTrackID::ITSTPCTOF] || src[GTrackID::TPCTRDTOF]) {
423  requestTOFMatches(src, useMC);
424  requestTOFClusters(false);
425  }
426  if (src[GTrackID::FT0]) {
427  requestFT0RecPoints(false); // RS FIXME: at the moment does not support MC
428  }
429  if (src[GTrackID::FV0]) {
430  requestFV0RecPoints(false); // RS FIXME: at the moment does not support MC
431  }
432  if (src[GTrackID::FDD]) {
433  requestFDDRecPoints(false); // RS FIXME: at the moment does not support MC
434  }
435  if (src[GTrackID::ZDC]) {
436  requestZDCRecEvents(false); // RS FIXME: at the moment does not support MC
437  }
438  if (GTrackID::includesDet(DetID::CTP, src)) {
439  requestCTPDigits(false); // RS FIXME: at the moment does not support MC
440  }
441 }

447  // clusters needed for refits

478 void RecoContainer::collectData(ProcessingContext& pc, const DataRequest& requests)
479 {
480  auto& reqMap = requests.requestMap;
481 
482  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true));
483  startIR = {0, dh->firstTForbit};
484 
485  auto req = reqMap.find("trackITS");
486  if (req != reqMap.end()) {
487  addITSTracks(pc, req->second);
488  }
489 
490  req = reqMap.find("trackMFT");
491  if (req != reqMap.end()) {
492  addMFTTracks(pc, req->second);
493  }
494 
495  req = reqMap.find("trackMCH");
496  if (req != reqMap.end()) {
497  addMCHTracks(pc, req->second);
498  }
499 
500  req = reqMap.find("trackMID");
501  if (req != reqMap.end()) {
502  addMIDTracks(pc, req->second);
503  }
504 
505  req = reqMap.find("trackTPC");
506  if (req != reqMap.end()) {
507  addTPCTracks(pc, req->second);
508  }
509 
510  req = reqMap.find("trackITSTPC");
511  if (req != reqMap.end()) {
512  addITSTPCTracks(pc, req->second);
513  }
514 
515  req = reqMap.find("fwdtracks");
516  if (req != reqMap.end()) {
517  addGlobalFwdTracks(pc, req->second);
518  }
519 
520  req = reqMap.find("matchMFTMCH");
521  if (req != reqMap.end()) {
522  addMFTMCHMatches(pc, req->second);
523  }
524 
525  req = reqMap.find("matchMCHMID");
526  if (req != reqMap.end()) {
527  addMCHMIDMatches(pc, req->second);
528  }
529 
530  req = reqMap.find("trackITSTPCTRD");
531  if (req != reqMap.end()) {
532  addITSTPCTRDTracks(pc, req->second);
533  }
534
