35  // convert global MCM index into HC, ROB and MCM number
36  static constexpr void convertMcmIdxGlb(int mcmGlb, int& hcid, int& rob, int& mcm)
37  {
38  hcid = mcmGlb / constants::NMCMHCMAX;
39  int side = (hcid % 2) ? 1 : 0;
40  rob = ((mcmGlb % constants::NMCMHCMAX) / constants::NMCMROB) * 2 + side;
41  mcm = (mcmGlb % constants::NMCMHCMAX) % constants::NMCMROB;
42  }
43  // convert HC, ROB and MCM number into a global MCM index
44  static constexpr int getMcmIdxGlb(int hcid, int rob, int mcm) { return hcid * constants::NMCMHCMAX + (rob / 2) * constants::NMCMROB + mcm; }
45 
46  // setters
47  void setIsNoisy(int hcid, int rob, int mcm) { mNoiseFlag.set(hcid * constants::NMCMHCMAX + rob * constants::NMCMROB + mcm); }
48  void setIsNoisy(int mcmIdxGlb) { mNoiseFlag.set(mcmIdxGlb); }
49 
50  // getters
51  bool getIsNoisy(int hcid, int rob, int mcm) const { return mNoiseFlag.test(hcid * constants::NMCMHCMAX + (rob / 2) * constants::NMCMROB + mcm); }
52  bool getIsNoisy(int mcmIdxGlb) const { return mNoiseFlag.test(mcmIdxGlb); }
53  auto getNumberOfNoisyMCMs() const { return mNoiseFlag.count(); }
54  bool isTrackletFromNoisyMCM(const Tracklet64& trklt) const { return getIsNoisy(trklt.getHCID(), trklt.getROB(), trklt.getMCM()); }
