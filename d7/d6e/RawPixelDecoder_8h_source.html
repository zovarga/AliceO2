151 template <class Mapping>
152 template <class DigitContainer, class ROFContainer>
153 int RawPixelDecoder<Mapping>::fillDecodedDigits(DigitContainer& digits, ROFContainer& rofs)
154 {
155  if (mInteractionRecord.isDummy()) {
156  return 0; // nothing was decoded
157  }
158  mTimerFetchData.Start(false);
159  int ref = digits.size();
160  for (unsigned int iru = 0; iru < mRUDecodeVec.size(); iru++) {
161  for (int ic = 0; ic < mRUDecodeVec[iru].nChipsFired; ic++) {
162  const auto& chip = mRUDecodeVec[iru].chipsData[ic];
163  for (const auto& hit : mRUDecodeVec[iru].chipsData[ic].getData()) {
164  digits.emplace_back(chip.getChipID(), hit.getRow(), hit.getCol());
165  }
166  }
167  }
168  int nFilled = digits.size() - ref;
169  rofs.emplace_back(mInteractionRecord, mROFCounter, ref, nFilled);
170  mTimerFetchData.Stop();
171  return nFilled;
172 }

176 template <>
177 template <class DigitContainer, class ROFContainer>
178 int RawPixelDecoder<ChipMappingMFT>::fillDecodedDigits(DigitContainer& digits, ROFContainer& rofs)
179 {
180  if (mInteractionRecord.isDummy()) {
181  return 0; // nothing was decoded
182  }
183  mTimerFetchData.Start(false);
184  int ref = digits.size();
185  for (auto chipData = mOrderedChipsPtr.rbegin(); chipData != mOrderedChipsPtr.rend(); ++chipData) {
186  assert(mLastReadChipID < (*chipData)->getChipID());
187  mLastReadChipID = (*chipData)->getChipID();
188  for (const auto& hit : (*chipData)->getData()) {
189  digits.emplace_back(mLastReadChipID, hit.getRow(), hit.getCol());
190  }
191  }
192  int nFilled = digits.size() - ref;
193  rofs.emplace_back(mInteractionRecord, mROFCounter, ref, nFilled);
194  mTimerFetchData.Stop();
195  return nFilled;
196 }

200 template <class Mapping>
201 template <class CalibContainer>
202 void RawPixelDecoder<Mapping>::fillCalibData(CalibContainer& calib)
203 {
204  if (!mInteractionRecord.isDummy()) {
205  auto curSize = calib.size();
206  calib.resize(curSize + Mapping::getNRUs());
207  for (unsigned int iru = 0; iru < mRUDecodeVec.size(); iru++) {
208  calib[curSize + mRUDecodeVec[iru].ruSWID] = mRUDecodeVec[iru].calibData;
209  }
210  }
