96 void CalibdEdxTrackTopologyPol::construct()
97 {
98  FlatObject::startConstruction();
99 
100  size_t buffSize = 0;
101  size_t offsets1[FFits];
102  size_t offsets2[FFits];
103 
104  for (int index = 0; index < FFits; ++index) {
105  buffSize = alignSize(buffSize, mCalibPolsqTot[index].getBufferAlignmentBytes());
106  offsets1[index] = buffSize;
107  buffSize += mCalibPolsqTot[index].getFlatBufferSize();
108  }
109  for (int index = 0; index < FFits; ++index) {
110  buffSize = alignSize(buffSize, mCalibPolsqMax[index].getBufferAlignmentBytes());
111  offsets2[index] = buffSize;
112  buffSize += mCalibPolsqMax[index].getFlatBufferSize();
113  }
114 
115  FlatObject::finishConstruction(buffSize);
116 
117  for (unsigned int i = 0; i < FFits; i++) {
118  mCalibPolsqTot[i].moveBufferTo(mFlatBufferPtr + offsets1[i]);
119  }
120  for (unsigned int i = 0; i < FFits; i++) {
121  mCalibPolsqMax[i].moveBufferTo(mFlatBufferPtr + offsets2[i]);
122  }
123 }

141 void CalibdEdxTrackTopologyPol::setFromContainer(const CalibdEdxTrackTopologyPolContainer& container)
142 {
143  if (2 * FFits != container.mCalibPols.size()) {
144  LOGP(info, fmt::format("wrong number of polynomials stored! this {} container {}", 2 * FFits, container.mCalibPols.size()));
145  return;
146  }
147 
148  for (int i = 0; i < FFits; ++i) {
149  mCalibPolsqTot[i].setFromContainer(container.mCalibPols[i]);
150  }
151 
152  for (int i = 0; i < FFits; ++i) {
153  mCalibPolsqMax[i].setFromContainer(container.mCalibPols[FFits + i]);
154  }
155 
156  mMaxTanTheta = container.mMaxTanTheta;
157  mMaxSinPhi = container.mMaxSinPhi;
158  mThresholdMin = container.mThresholdMin;
159  mThresholdMax = container.mThresholdMax;
160 }
