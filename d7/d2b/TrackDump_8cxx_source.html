26 void TrackDump::filter(const gsl::span<const TrackTPC> tracks, ClusterNativeAccess const& clusterIndex, const gsl::span<const o2::tpc::TPCClRefElem> clRefs)
27 {
28  if (!mTreeDump && outputFileName.size()) {
29  mTreeDump = std::make_unique<utils::TreeStreamRedirector>(outputFileName.data(), "recreate");
30  }
31 
32  std::vector<TrackInfo> tpcTracks;
33  std::vector<std::vector<ClusterGlobal>> clustersGlobalEvent;
34  std::vector<ClusterGlobal>* clustersGlobal{};
35 
36  GPUCA_NAMESPACE::gpu::GPUTPCGeometry gpuGeom;
37 
38  for (const auto& track : tracks) {
39  const int nCl = track.getNClusterReferences();
40  auto& trackInfo = tpcTracks.emplace_back(track);
41  auto& clInfos = trackInfo.clusters;
42  if (writeGlobal) {
43  clustersGlobal = &clustersGlobalEvent.emplace_back();
44  }
45 
46  for (int j = nCl - 1; j >= 0; j--) {
47  uint8_t sector, padrow;
48  uint32_t clusterIndexInRow;
49  track.getClusterReference(clRefs, j, sector, padrow, clusterIndexInRow);
50  const auto& cl = clusterIndex.clusters[sector][padrow][clusterIndexInRow];
51  auto& clInfo = clInfos.emplace_back(cl);
52  clInfo.sector = sector;
53  clInfo.padrow = padrow;
54 
55  if (clustersGlobal) {
56  auto& clGlobal = clustersGlobal->emplace_back(ClusterGlobal{clInfo.getGx(), clInfo.getGy(), cl.qMax, cl.qTot, sector, padrow});
57  }
58  }
59  }
60 
61  if (mTreeDump) {
62  auto& tree = (*mTreeDump) << "tpcrec";
63  if (writeTracks) {
64  // << "info=" << trackInfos
65  tree << "TPCTracks=" << tpcTracks;
66  }
67  if (writeGlobal) {
68  tree << "cls" << clustersGlobalEvent;
69  }
70  tree << "\n";
71  // << "clusters=" << clInfoVec
72  }
73 }
