73 std::vector<float> o2::tpc::IDCGroup::get1DIDCs(const std::vector<float> idc, const unsigned int nIntervals, const unsigned int nIDCsPerIntegrationInterval, const unsigned int region, const bool normalize)
74 {
75  // integrate IDCs for each interval
76  std::vector<float> idcOne;
77  idcOne.reserve(nIntervals);
78  for (unsigned int i = 0; i < nIntervals; ++i) {
79  // set integration range for one integration interval
80  const auto start = idc.begin() + i * nIDCsPerIntegrationInterval;
81  const auto end = start + nIDCsPerIntegrationInterval;
82  idcOne.emplace_back(std::accumulate(start, end, decltype(idc)::value_type(0)));
83  }
84  // normalize 1D-IDCs to absolute space charge
85  const float norm = normalize ? Mapper::REGIONAREA[region] / nIDCsPerIntegrationInterval : 1.f / nIDCsPerIntegrationInterval;
86  std::transform(idcOne.begin(), idcOne.end(), idcOne.begin(), [norm](auto& val) { return val * norm; });
87 
88  return idcOne;
89 }
