58 template <typename VEC>
59 void CTFCoder::encode(VEC& buff, const gsl::span<const BCData>& trigData, const gsl::span<const ChannelData>& chanData, const gsl::span<const OrbitData>& pedData)
60 {
61  using MD = o2::ctf::Metadata::OptStore;
62  // what to do which each field: see o2::ctd::Metadata explanation
63  constexpr MD optField[CTF::getNBlocks()] = {
64  MD::EENCODE, // _bcIncTrig
65  MD::EENCODE, // _orbitIncTrig,
66  MD::EENCODE, // _moduleTrig,
67  MD::EENCODE, // _channelsHL,
68  MD::EENCODE, // _triggersHL,
69  MD::EENCODE, // _extTriggers,
70  MD::EENCODE, // _nchanTrig,
71  //
72  MD::EENCODE, // _chanID,
73  MD::EENCODE, // _chanData,
74  //
75  MD::EENCODE, // _orbitIncEOD,
76  MD::EENCODE, // _pedData
77  MD::EENCODE, // _sclInc
78  };
79 
80  CTFHelper helper(trigData, chanData, pedData);
81 
82  // book output size with some margin
83  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
84  buff.resize(szIni);
85 
86  auto ec = CTF::create(buff);
87  using ECB = CTF::base;
88 
89  ec->setHeader(helper.createHeader());
90  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
91  ec->getANSHeader().majorVersion = 0;
92  ec->getANSHeader().minorVersion = 1;
93  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
94 #define ENCODEZDC(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
95  // clang-format off
96  ENCODEZDC(helper.begin_bcIncTrig(), helper.end_bcIncTrig(), CTF::BLC_bcIncTrig, 0);
97  ENCODEZDC(helper.begin_orbitIncTrig(), helper.end_orbitIncTrig(), CTF::BLC_orbitIncTrig, 0);
98  ENCODEZDC(helper.begin_moduleTrig(), helper.end_moduleTrig(), CTF::BLC_moduleTrig, 0);
99  ENCODEZDC(helper.begin_channelsHL(), helper.end_channelsHL(), CTF::BLC_channelsHL, 0);
100  ENCODEZDC(helper.begin_triggersHL(), helper.end_triggersHL(), CTF::BLC_triggersHL, 0);
101  ENCODEZDC(helper.begin_extTriggers(), helper.end_extTriggers(), CTF::BLC_extTriggers, 0);
102  ENCODEZDC(helper.begin_nchanTrig(), helper.end_nchanTrig(), CTF::BLC_nchanTrig, 0);
103  //
104  ENCODEZDC(helper.begin_chanID(), helper.end_chanID(), CTF::BLC_chanID, 0);
105  ENCODEZDC(helper.begin_chanData(), helper.end_chanData(), CTF::BLC_chanData, 0);
106  //
107  ENCODEZDC(helper.begin_orbitIncEOD(), helper.end_orbitIncEOD(), CTF::BLC_orbitIncEOD, 0);
108  ENCODEZDC(helper.begin_pedData(), helper.end_pedData(), CTF::BLC_pedData, 0);
109  ENCODEZDC(helper.begin_sclInc(), helper.end_sclInc(), CTF::BLC_sclInc, 0);
110 
111  // clang-format on
112  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
113 }

116 template <typename VTRG, typename VCHAN, typename VPED>
117 void CTFCoder::decode(const CTF::base& ec, VTRG& trigVec, VCHAN& chanVec, VPED& pedVec)
118 {
119  auto header = ec.getHeader();
120  checkDictVersion(static_cast<const o2::ctf::CTFDictHeader&>(header));
121  ec.print(getPrefix(), mVerbosity);
122  std::vector<uint16_t> bcIncTrig, moduleTrig, nchanTrig, chanData, pedData, scalerInc, triggersHL, channelsHL;
123  std::vector<uint32_t> orbitIncTrig, orbitIncEOD;
124  std::vector<uint8_t> extTriggers, chanID;
125 
126 #define DECODEZDC(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
127  // clang-format off
128  DECODEZDC(bcIncTrig, CTF::BLC_bcIncTrig);
129  DECODEZDC(orbitIncTrig, CTF::BLC_orbitIncTrig);
130  DECODEZDC(moduleTrig, CTF::BLC_moduleTrig);
131  DECODEZDC(channelsHL, CTF::BLC_channelsHL);
132  DECODEZDC(triggersHL, CTF::BLC_triggersHL);
133  DECODEZDC(extTriggers, CTF::BLC_extTriggers);
134  DECODEZDC(nchanTrig, CTF::BLC_nchanTrig);
135  //
136  DECODEZDC(chanID, CTF::BLC_chanID);
137  DECODEZDC(chanData, CTF::BLC_chanData);
138  //
139  DECODEZDC(orbitIncEOD, CTF::BLC_orbitIncEOD);
140  DECODEZDC(pedData, CTF::BLC_pedData);
141  DECODEZDC(scalerInc, CTF::BLC_sclInc);
142  // clang-format on
143  //
144  trigVec.clear();
145  chanVec.clear();
146  pedVec.clear();
147  trigVec.reserve(header.nTriggers);
148  chanVec.reserve(header.nChannels);
149  pedVec.reserve(header.nEOData);
150 
151  // triggers and channels
152  uint32_t firstEntry = 0;
153  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
154  auto chanDataIt = chanData.begin();
155  auto chanIdIt = chanID.begin();
156  auto modTrigIt = moduleTrig.begin();
157  auto pedValIt = pedData.begin();
158  auto sclIncIt = scalerInc.begin();
159  auto channelsHLIt = channelsHL.begin();
160  auto triggersHLIt = triggersHL.begin();
161  auto scalers = header.firstScaler;
162 
163  for (uint32_t itrig = 0; itrig < header.nTriggers; itrig++) {
164  // restore TrigRecord
165  if (orbitIncTrig[itrig]) { // non-0 increment => new orbit
166  ir.bc = bcIncTrig[itrig]; // bcInc has absolute meaning
167  ir.orbit += orbitIncTrig[itrig];
168  } else {
169  ir.bc += bcIncTrig[itrig];
170  }
171 
172  auto firstChanEntry = chanVec.size();
173  for (uint16_t ic = 0; ic < nchanTrig[itrig]; ic++) {
174  auto& chan = chanVec.emplace_back();
175  chan.id = *chanIdIt++;
176  std::copy_n(chanDataIt, NTimeBinsPerBC, chan.data.begin());
177  chanDataIt += NTimeBinsPerBC;
178  }
179  uint32_t chHL = (uint32_t(*channelsHLIt++) << 16) + *channelsHLIt++;
180  uint32_t trHL = (uint32_t(*triggersHLIt++) << 16) + *triggersHLIt++;
181 
182  auto& bcTrig = trigVec.emplace_back(firstChanEntry, chanVec.size() - firstChanEntry, ir, chHL, trHL, extTriggers[itrig]);
183  std::copy_n(modTrigIt, NModules, bcTrig.moduleTriggers.begin());
184  modTrigIt += NModules;
185  }
186 
187  // pedestal data
188  ir = {o2::constants::lhc::LHCMaxBunches - 1, header.firstOrbitEOData};
189  for (uint32_t ip = 0; ip < header.nEOData; ip++) {
190  ir.orbit += orbitIncEOD[ip];
191  for (uint32_t ic = 0; ic < NChannels; ic++) {
192  scalers[ic] += *sclIncIt++; // increment scaler
193  }
194  auto& ped = pedVec.emplace_back(OrbitData{ir, {}, scalers});
195  std::copy_n(pedValIt, NChannels, ped.data.begin());
196  pedValIt += NChannels;
197  }
198  // make sure whole data was used
199  assert(chanDataIt == chanData.end());
200  assert(chanIdIt == chanID.end());
201  assert(modTrigIt == moduleTrig.end());
202  assert(pedValIt == pedData.end());
203  assert(channelsHLIt == channelsHL.end());
204  assert(triggersHLIt == triggersHL.end());
205  assert(sclIncIt == scalerInc.end());
206 }
