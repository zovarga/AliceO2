44 NodeColor decideColorForNode(const DeviceInfo& info)
45 {
46  if (info.active == false) {
47  return NodeColor{
48  .normal = PaletteHelpers::RED,
49  .hovered = PaletteHelpers::SHADED_RED};
50  }
51  switch (info.streamingState) {
52  case StreamingState::EndOfStreaming:
53  return NodeColor{
54  .normal = PaletteHelpers::SHADED_YELLOW,
55  .hovered = PaletteHelpers::YELLOW,
56  .title = PaletteHelpers::YELLOW,
57  .title_hovered = PaletteHelpers::DARK_YELLOW};
58  case StreamingState::Idle:
59  return NodeColor{
60  .normal = PaletteHelpers::SHADED_GREEN,
61  .hovered = PaletteHelpers::GREEN,
62  .title = PaletteHelpers::GREEN,
63  .title_hovered = PaletteHelpers::DARK_GREEN};
64  case StreamingState::Streaming:
65  default:
66  return NodeColor{
67  .normal = PaletteHelpers::GRAY,
68  .hovered = PaletteHelpers::LIGHT_GRAY,
69  .title = PaletteHelpers::GRAY,
70  .title_hovered = PaletteHelpers::BLACK};
71  }
72 }

185 // Private helper struct for the layout of the graph
186 struct NodePos {
187  ImVec2 pos;
188  static ImVec2 GetInputSlotPos(ImVector<Node> const& infos, ImVector<NodePos> const& positions, int nodeId, int slot_no)
189  {
190  ImVec2 const& pos = positions[nodeId].pos;
191  ImVec2 const& size = infos[nodeId].Size;
192  float inputsCount = infos[nodeId].InputsCount;
193  return ImVec2(pos.x, pos.y + size.y * ((float)slot_no + 1) / (inputsCount + 1));
194  }
195  static ImVec2 GetOutputSlotPos(ImVector<Node> const& infos, ImVector<NodePos> const& positions, int nodeId, int slot_no)
196  {
197  ImVec2 const& pos = positions[nodeId].pos;
198  ImVec2 const& size = infos[nodeId].Size;
199  float outputsCount = infos[nodeId].OutputsCount;
200  return ImVec2(pos.x + size.x, pos.y + size.y * ((float)slot_no + 1) / (outputsCount + 1));
201  }
202 };

244  auto prepareChannelView = [&specs, &metadata](ImVector<Node>& nodeList, ImVector<Group>& groupList) {
245  struct LinkInfo {
246  int specId;
247  int outputId;
248  };
249  std::map<std::string, LinkInfo> linkToIndex;
250  for (int si = 0; si < specs.size(); ++si) {
251  int oi = 0;
252  for (auto&& output : specs[si].outputChannels) {
253  linkToIndex.insert(std::make_pair(output.name, LinkInfo{si, oi}));
254  oi += 1;
255  }
256  }
257  // Prepare the list of groups.
258  std::string workflow = "Ungrouped";
259  int groupId = 0;
260  for (size_t mi = 0; mi < metadata.size(); ++mi) {
261  auto const& metadatum = metadata[mi];
262  if (metadatum.executable == workflow) {
263  continue;
264  }
265  workflow = metadatum.executable;
266  char* groupBasename = strrchr(workflow.data(), '/');
267  char const* groupName = groupBasename ? groupBasename + 1 : workflow.data();
268  bool hasDuplicate = false;
269  for (size_t gi = 0; gi < groupList.Size; ++gi) {
270  if (strncmp(groupName, groupList[gi].name, MAX_GROUP_NAME_SIZE - 1) == 0) {
271  hasDuplicate = true;
272  break;
273  }
274  }
275  if (hasDuplicate == false) {
276  groupList.push_back(Group(groupId++, groupName, mi));
277  }
278  }
279  // Do matching between inputs and outputs
280  for (int si = 0; si < specs.size(); ++si) {
281  auto& spec = specs[si];
282  int groupId = 0;
283 
284  auto metadatum = std::find_if(metadata.begin(), metadata.end(),
285  [&name = spec.name](DataProcessorInfo const& info) { return info.name == name; });
286 
287  for (size_t gi = 0; gi < groupList.Size; ++gi) {
288  if (metadatum == metadata.end()) {
289  break;
290  }
291  const char* groupName = strrchr(metadatum->executable.data(), '/');
292  if (strncmp(groupList[gi].name, groupName ? groupName + 1 : metadatum->executable.data(), 127) == 0) {
293  groupId = gi;
294  break;
295  }
296  }
297  nodeList.push_back(Node(si, groupId, spec.id.c_str(), 0.5f,
298  ImColor(255, 100, 100),
299  spec.inputChannels.size(),
300  spec.outputChannels.size()));
301  int ii = 0;
302  for (auto& input : spec.inputChannels) {
303  auto const& outName = input.name;
304  auto const& out = linkToIndex.find(input.name);
305  if (out == linkToIndex.end()) {
306  LOG(error) << "Could not find suitable node for " << outName;
307  continue;
308  }
309  links.push_back(NodeLink{out->second.specId, out->second.outputId, si, ii});
310  ii += 1;
311  }
312  }
313 
314  // ImVector does boudary checks, so I bypass the case there is no
315  // edges.
316  std::vector<TopoIndexInfo> sortedNodes = {{0, 0}};
317  if (links.size()) {
318  sortedNodes = WorkflowHelpers::topologicalSort(specs.size(), &(links[0].InputIdx), &(links[0].OutputIdx), sizeof(links[0]), links.size());
319  }
320  // This is to protect for the cases in which there is a loop in the
321  // definition of the inputs and of the outputs due to the
322  // way the forwarding creates hidden dependencies between processes.
323  // This should not happen, but apparently it does.
324  for (auto di = 0; di < specs.size(); ++di) {
325  auto fn = std::find_if(sortedNodes.begin(), sortedNodes.end(), [di](TopoIndexInfo const& info) {
326  return di == info.index;
327  });
328  if (fn == sortedNodes.end()) {
329  sortedNodes.push_back({(int)di, 0});
330  }
331  }
332  assert(specs.size() == sortedNodes.size());
334  std::sort(sortedNodes.begin(), sortedNodes.end());
335 
336  std::vector<int> layerEntries(1024, 0);
337  std::vector<int> layerMax(1024, 0);
338  for (auto& node : sortedNodes) {
339  layerMax[node.layer < 1023 ? node.layer : 1023] += 1;
340  }
341 
342  // FIXME: display nodes using topological sort
343  // Update positions
344  for (int si = 0; si < specs.size(); ++si) {
345  auto& node = sortedNodes[si];
346  assert(node.index == si);
347  int xpos = 40 + 240 * node.layer;
348  int ypos = 300 + (std::max(600, 60 * (int)layerMax[node.layer]) / (layerMax[node.layer] + 1)) * (layerEntries[node.layer] - layerMax[node.layer] / 2);
349  positions.push_back(NodePos{ImVec2(xpos, ypos)});
350  layerEntries[node.layer] += 1;
351  }
352  };

356  inited = true;
357  }
