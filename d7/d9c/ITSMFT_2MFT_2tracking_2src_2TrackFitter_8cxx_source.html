41 template <typename T>
42 bool TrackFitter<T>::fit(T& track, bool outward)
43 {
44  if constexpr (std::is_same<T, o2::mft::TrackLTF>::value) {
47 
48  auto nClusters = track.getNumberOfPoints();
49  auto lastLayer = track.getLayers()[(outward ? 0 : nClusters - 1)];
50 
51  if (mVerbose) {
52  std::cout << "Seed covariances: \n"
53  << track.getCovariances() << std::endl
54  << std::endl;
55  }
56 
57  // recursively compute clusters, updating the track parameters
58  if (!outward) { // Inward for vertexing
59  while (nClusters-- > 0) {
60  if (!computeCluster(track, nClusters, lastLayer)) {
61  return false;
62  }
63  }
64  } else { // Outward for MCH matching
65  int ncl = 0;
66  while (ncl < nClusters) {
67  if (!computeCluster(track, ncl, lastLayer)) {
68  return false;
69  }
70  ncl++;
71  }
72  }
73  if (mVerbose) {
74  std::cout << "Track Chi2 = " << track.getTrackChi2() << std::endl;
75  std::cout << " ***************************** Done fitting *****************************\n";
76  }
77  } else { // Magnet off: linear tracks
78  // Do nothing while there is no alignment
79  }
80 
81  return true;
82 }

85 template <typename T>
86 bool TrackFitter<T>::initTrack(T& track, bool outward)
87 {
88 
89  if constexpr (std::is_same<T, o2::mft::TrackLTF>::value) { // Magnet on
90  // initialize the starting track parameters
91  double sigmainvQPtsq;
92  double chi2invqptquad;
93  auto invQPt0 = invQPtFromFCF(track, mBZField, sigmainvQPtsq);
94  auto nPoints = track.getNumberOfPoints();
95  auto k = TMath::Abs(o2::constants::math::B2C * mBZField);
96  auto Hz = std::copysign(1, mBZField);
97 
98  if (mVerbose) {
99  std::cout << "\n ***************************** Start Fitting new track ***************************** \n";
100  std::cout << "N Clusters = " << nPoints << std::endl;
101  }
102 
103  track.setInvQPtSeed(invQPt0);
104  track.setChi2QPtSeed(chi2invqptquad);
105  track.setInvQPt(invQPt0);
106 
108  int first_cls, next_cls;
109  if (outward) { // MCH matching
110  first_cls = 0;
111  next_cls = nPoints - 1;
112  } else { // Vertexing
113  first_cls = nPoints - 1;
114  next_cls = 0;
115  }
116 
117  auto x0 = track.getXCoordinates()[first_cls];
118  auto y0 = track.getYCoordinates()[first_cls];
119  auto z0 = track.getZCoordinates()[first_cls];
120 
121  // Compute tanl using first two clusters
122  auto deltaX = track.getXCoordinates()[1] - track.getXCoordinates()[0];
123  auto deltaY = track.getYCoordinates()[1] - track.getYCoordinates()[0];
124  auto deltaZ = track.getZCoordinates()[1] - track.getZCoordinates()[0];
125  auto deltaR = TMath::Sqrt(deltaX * deltaX + deltaY * deltaY);
126  auto tanl0 = 0.5 * TMath::Sqrt2() * (deltaZ / deltaR) *
127  TMath::Sqrt(TMath::Sqrt((invQPt0 * deltaR * k) * (invQPt0 * deltaR * k) + 1) + 1);
128 
129  // Compute phi at the last cluster using two last clusters
130  deltaX = track.getXCoordinates()[first_cls] - track.getXCoordinates()[next_cls];
131  deltaY = track.getYCoordinates()[first_cls] - track.getYCoordinates()[next_cls];
132  deltaZ = track.getZCoordinates()[first_cls] - track.getZCoordinates()[next_cls];
133  deltaR = TMath::Sqrt(deltaX * deltaX + deltaY * deltaY);
134  double phi0;
135  if (outward) {
136  phi0 = TMath::ATan2(-deltaY, -deltaX) - 0.5 * Hz * invQPt0 * deltaZ * k / tanl0;
137  } else {
138  phi0 = TMath::ATan2(deltaY, deltaX) - 0.5 * Hz * invQPt0 * deltaZ * k / tanl0;
139  }
140  // The low momentum phi0 correction may be irrelevant and may require a call to o2::math_utils::bringToPMPiGend(phi0);
141 
142  track.setX(x0);
143  track.setY(y0);
144  track.setZ(z0);
145  track.setPhi(phi0);
146  track.setTanl(tanl0);
147 
148  if (mVerbose) {
149  std::cout << " Init " << (track.isCA() ? "CA Track " : "LTF Track") << (outward ? " (outward)" : " (inward)") << std::endl;
150  auto model = (mTrackModel == Helix) ? "Helix" : (mTrackModel == Quadratic) ? "Quadratic"
151  : (mTrackModel == Optimized) ? "Optimized"
152  : "Linear";
153  std::cout << "Track Model: " << model << std::endl;
154  std::cout << " initTrack: X = " << x0 << " Y = " << y0 << " Z = " << z0 << " Tgl = " << tanl0 << " Phi = " << phi0 << " pz = " << track.getPz() << " q/pt = " << track.getInvQPt() << std::endl;
155  }
156 
157  SMatrix55Sym lastParamCov;
158  float qptsigma = TMath::Max(std::abs(track.getInvQPt()), .5);
159  float tanlsigma = TMath::Max(std::abs(track.getTanl()), .5);
160 
161  lastParamCov(0, 0) = 1; // <X,X>
162  lastParamCov(1, 1) = 1; // <Y,X>
163  lastParamCov(2, 2) = TMath::Pi() * TMath::Pi() / 16; // <PHI,X>
164  lastParamCov(3, 3) = 10 * tanlsigma * tanlsigma; // <TANL,X>
165  lastParamCov(4, 4) = 10 * qptsigma * qptsigma; // <INVQPT,X>
166 
167  track.setCovariances(lastParamCov);
168  track.setTrackChi2(0.);
169 
170  } else { // Magnet off: linear tracks
171 
172  // initialize the starting track parameters
173  double chi2invqptquad;
174  auto invQPt0 = 0;
175 
176  auto nPoints = track.getNumberOfPoints();
177 
178  if (mVerbose) {
179  std::cout << "\n ***************************** Start Fitting new track ***************************** \n";
180  std::cout << "N Clusters = " << nPoints << std::endl;
181  }
182 
183  track.setInvQPtSeed(0);
184  track.setChi2QPtSeed(0);
185  track.setInvQPt(invQPt0);
186 
188  outward = !outward; // Temporary fix for misaligned detector.
189  int first_cls, next_cls;
190  if (outward) { // MCH matching
191  first_cls = 0;
192  next_cls = nPoints - 1;
193  } else { // Vertexing
194  first_cls = nPoints - 1;
195  next_cls = 0;
196  }
197 
198  auto x0 = track.getXCoordinates()[first_cls];
199  auto y0 = track.getYCoordinates()[first_cls];
200  auto z0 = track.getZCoordinates()[first_cls];
201 
202  auto deltaX = track.getXCoordinates()[first_cls] - track.getXCoordinates()[next_cls];
203  auto deltaY = track.getYCoordinates()[first_cls] - track.getYCoordinates()[next_cls];
204  auto deltaZ = track.getZCoordinates()[first_cls] - track.getZCoordinates()[next_cls];
205  auto deltaR = TMath::Sqrt(deltaX * deltaX + deltaY * deltaY);
206  auto tanl0 = -std::abs(deltaZ) / deltaR;
207  double phi0;
208  if (outward) {
209  phi0 = TMath::ATan2(-deltaY, -deltaX);
210  } else {
211  phi0 = TMath::ATan2(deltaY, deltaX);
212  }
213 
214  track.setX(x0);
215  track.setY(y0);
216  track.setZ(z0);
217  track.setPhi(phi0);
218  track.setTanl(tanl0);
219 
220  if (mVerbose) {
221  std::cout << " Init " << (track.isCA() ? "CA Track " : "LTF Track") << (outward ? " (outward)" : " (inward)") << std::endl;
222  auto model = "Linear";
223  std::cout << "Track Model: " << model << std::endl;
224  std::cout << " initTrack: X = " << x0 << " Y = " << y0 << " Z = " << z0 << " Tgl = " << tanl0 << " Phi = " << phi0 << " pz = " << track.getPz() << " q/pt = " << track.getInvQPt() << std::endl;
225  }
226 
227  SMatrix55Sym lastParamCov;
228  float qptsigma = TMath::Max(std::abs(track.getInvQPt()), .5);
229  float tanlsigma = TMath::Max(std::abs(track.getTanl()), .5);
230 
231  lastParamCov(0, 0) = 1; // <X,X>
232  lastParamCov(1, 1) = 1; // <Y,X>
233  lastParamCov(2, 2) = TMath::Pi() * TMath::Pi() / 16; // <PHI,X>
234  lastParamCov(3, 3) = 10 * tanlsigma * tanlsigma; // <TANL,X>
235  lastParamCov(4, 4) = 10 * qptsigma * qptsigma; // <INVQPT,X>
236 
237  track.setCovariances(lastParamCov);
238  track.setTrackChi2(0.);
239  }
240 
241  return true;
242 }

245 template <typename T>
246 bool TrackFitter<T>::propagateToZ(T& track, double z)
247 {
248  // Propagate track to the z position of the new cluster
249  if constexpr (std::is_same<T, o2::mft::TrackLTF>::value) { // Magnet on
250 
251  switch (mTrackModel) {
252  case Linear:
253  track.propagateToZlinear(z);
254  break;
255  case Quadratic:
256  track.propagateToZquadratic(z, mBZField);
257  break;
258  case Helix:
259  track.propagateToZhelix(z, mBZField);
260  break;
261  case Optimized:
262  track.propagateToZ(z, mBZField);
263  break;
264  default:
265  std::cout << " Invalid track model.\n";
266  return false;
267  break;
268  }
269  } else {
270  // Do nothing while there is no alignment
271  }
272  return true;
273 }

276 template <typename T>
277 bool TrackFitter<T>::propagateToNextClusterWithMCS(T& track, double z, int& startingLayerID, const int& newLayerID)
278 {
279 
280  // Propagate track to the next cluster z position, adding angular MCS effects at the center of
281  // each disk crossed by the track. This method is valid only for track propagation between
282  // clusters at MFT layers positions.
283 
284  if (startingLayerID == newLayerID) { // Same layer, nothing to do.
285  if (mVerbose) {
286  std::cout << " => Propagate to next cluster with MCS : startingLayerID = " << startingLayerID << " = > "
287  << " newLayerID = " << newLayerID << " (NLayers = " << std::abs(newLayerID - startingLayerID);
288  std::cout << ") ; track.getZ() = " << track.getZ() << " => ";
289  std::cout << "destination cluster z = " << z << " ; => Same layer: no MCS effects." << std::endl;
290  }
291  if (z != track.getZ()) {
292  propagateToZ(track, z);
293  }
294  return true;
295  }
296 
297  using o2::mft::constants::LayerZPosition;
298  auto startingZ = track.getZ();
299 
300  //https://stackoverflow.com/questions/1903954/is-there-a-standard-sign-function-signum-sgn-in-c-c
301  auto signum = [](auto a) {
302  return (0 < a) - (a < 0);
303  };
304  int direction = signum(newLayerID - startingLayerID); // takes values +1, 0, -1
305  auto currentLayer = startingLayerID;
306 
307  if (mVerbose) {
308  std::cout << " => Propagate to next cluster with MCS : startingLayerID = " << startingLayerID << " = > "
309  << " newLayerID = " << newLayerID << " (NLayers = " << std::abs(newLayerID - startingLayerID);
310  std::cout << ") ; track.getZ() = " << track.getZ() << " => ";
311  std::cout << "destination cluster z = " << z << " ; " << std::endl;
312  }
313 
314  // Number of disks crossed by this track segment
315  while (currentLayer != newLayerID) {
316  auto nextlayer = currentLayer + direction;
317  auto nextZ = LayerZPosition[nextlayer];
318 
319  int NDisksMS;
320  if (nextZ - track.getZ() > 0) {
321  NDisksMS = (currentLayer % 2 == 0) ? (currentLayer - nextlayer) / 2 : (currentLayer - nextlayer + 1) / 2;
322  } else {
323  NDisksMS = (currentLayer % 2 == 0) ? (nextlayer - currentLayer + 1) / 2 : (nextlayer - currentLayer) / 2;
324  }
325 
326  if (mVerbose) {
327  std::cout << "currentLayer = " << currentLayer << " ; "
328  << "nextlayer = " << nextlayer << " ; ";
329  std::cout << "track.getZ() = " << track.getZ() << " ; ";
330  std::cout << "nextZ = " << nextZ << " ; ";
331  std::cout << "NDisksMS = " << NDisksMS << std::endl;
332  }
333 
334  if ((NDisksMS * mMFTDiskThicknessInX0) != 0) {
335  track.addMCSEffect(NDisksMS * mMFTDiskThicknessInX0);
336  if (mVerbose) {
337  std::cout << "Track covariances after MCS effects: \n"
338  << track.getCovariances() << std::endl
339  << std::endl;
340  }
341  }
342 
343  if (mVerbose) {
344  std::cout << " BeforeExtrap: X = " << track.getX() << " Y = " << track.getY() << " Z = " << track.getZ() << " Tgl = " << track.getTanl() << " Phi = " << track.getPhi() << " pz = " << track.getPz() << " q/pt = " << track.getInvQPt() << std::endl;
345  }
346 
347  propagateToZ(track, nextZ);
348 
349  currentLayer = nextlayer;
350  }
351  if (z != track.getZ()) {
352  propagateToZ(track, z);
353  }
354  startingLayerID = newLayerID;
355  return true;
356 }

359 template <typename T>
360 bool TrackFitter<T>::computeCluster(T& track, int cluster, int& startingLayerID)
361 {
366 
367  const auto& clx = track.getXCoordinates()[cluster];
368  const auto& cly = track.getYCoordinates()[cluster];
369  const auto& clz = track.getZCoordinates()[cluster];
370  const auto& sigmaX2 = track.getSigmasX2()[cluster];
371  const auto& sigmaY2 = track.getSigmasY2()[cluster];
372  const auto& newLayerID = track.getLayers()[cluster];
373 
374  if (mVerbose) {
375  std::cout << "computeCluster: X = " << clx << " Y = " << cly << " Z = " << clz << " nCluster = " << cluster << " Layer = " << newLayerID << std::endl;
376  }
377 
378  if (!propagateToNextClusterWithMCS(track, clz, startingLayerID, newLayerID)) {
379  return false;
380  }
381 
382  if (mVerbose) {
383  std::cout << " AfterExtrap: X = " << track.getX() << " Y = " << track.getY() << " Z = " << track.getZ() << " Tgl = " << track.getTanl() << " Phi = " << track.getPhi() << " pz = " << track.getPz() << " q/pt = " << track.getInvQPt() << std::endl;
384  std::cout << "Track covariances after extrap: \n"
385  << track.getCovariances() << std::endl
386  << std::endl;
387  }
388 
389  // recompute parameters
390  const std::array<float, 2>& pos = {clx, cly};
391  const std::array<float, 2>& cov = {sigmaX2, sigmaY2};
392 
393  if (track.update(pos, cov)) {
394  if (mVerbose) {
395  std::cout << " New Cluster: X = " << clx << " Y = " << cly << " Z = " << clz << std::endl;
396  std::cout << " AfterKalman: X = " << track.getX() << " Y = " << track.getY() << " Z = " << track.getZ() << " Tgl = " << track.getTanl() << " Phi = " << track.getPhi() << " pz = " << track.getPz() << " q/pt = " << track.getInvQPt() << std::endl;
397  std::cout << std::endl;
398  std::cout << "Track covariances after Kalman update: \n"
399  << track.getCovariances() << std::endl
400  << std::endl;
401  }
402  return true;
403  }
404  return false;
405 }

408 template <typename T>
409 Double_t invQPtFromFCF(const T& track, Double_t bFieldZ, Double_t& sigmainvqptsq)
410 {
411 
412  const std::array<Float_t, constants::mft::LayersNumber>& xPositions = track.getXCoordinates();
413  const std::array<Float_t, constants::mft::LayersNumber>& yPositions = track.getYCoordinates();
414  const std::array<Float_t, constants::mft::LayersNumber>& zPositions = track.getZCoordinates();
415  const std::array<Float_t, constants::mft::LayersNumber>& SigmasX2 = track.getSigmasX2();
416  const std::array<Float_t, constants::mft::LayersNumber>& SigmasY2 = track.getSigmasY2();
417 
418  // Fast Circle Fit (Hansroul, Jeremie, Savard, 1987)
419  auto nPoints = track.getNumberOfPoints();
420  std::vector<double> xVal(nPoints);
421  std::vector<double> yVal(nPoints);
422  std::vector<double> zVal(nPoints);
423  std::vector<double> xErr(nPoints);
424  std::vector<double> yErr(nPoints);
425  std::vector<double> uVal(nPoints - 1);
426  std::vector<double> vVal(nPoints - 1);
427  std::vector<double> vErr(nPoints - 1);
428  std::vector<double> fweight(nPoints - 1);
429  std::vector<double> Rn(nPoints - 1);
430  std::vector<double> Pn(nPoints - 1);
431  Double_t A, Aerr, B, Berr, x2, y2, invx2y2, a, b, r, sigmaRsq, u2, sigma;
432  Double_t F0, F1, F2, F3, F4, SumSRn, SumSPn, SumRn, SumUPn, SumRP;
433 
434  SumSRn = SumSPn = SumRn = SumUPn = SumRP = 0.0;
435  F0 = F1 = F2 = F3 = F4 = 0.0;
436 
437  for (auto np = 0; np < nPoints; np++) {
438  xErr[np] = SigmasX2[np];
439  yErr[np] = SigmasY2[np];
440  if (np > 0) {
441  xVal[np] = xPositions[np] - xVal[0];
442  yVal[np] = yPositions[np] - yVal[0];
443  xErr[np] *= std::sqrt(2.);
444  yErr[np] *= std::sqrt(2.);
445  } else {
446  xVal[np] = 0.;
447  yVal[np] = 0.;
448  }
449  zVal[np] = zPositions[np];
450  }
451  for (int i = 0; i < (nPoints - 1); i++) {
452  x2 = xVal[i + 1] * xVal[i + 1];
453  y2 = yVal[i + 1] * yVal[i + 1];
454  invx2y2 = 1. / (x2 + y2);
455  uVal[i] = xVal[i + 1] * invx2y2;
456  vVal[i] = yVal[i + 1] * invx2y2;
457  vErr[i] = std::sqrt(8. * xErr[i + 1] * xErr[i + 1] * x2 * y2 + 2. * yErr[i + 1] * yErr[i + 1] * (x2 - y2) * (x2 - y2)) * invx2y2 * invx2y2;
458  u2 = uVal[i] * uVal[i];
459  fweight[i] = 1. / vErr[i];
460  F0 += fweight[i];
461  F1 += fweight[i] * uVal[i];
462  F2 += fweight[i] * u2;
463  F3 += fweight[i] * uVal[i] * u2;
464  F4 += fweight[i] * u2 * u2;
465  }
466 
467  double Rn_det1 = F2 * F4 - F3 * F3;
468  double Rn_det2 = F1 * F4 - F2 * F3;
469  double Rn_det3 = F1 * F3 - F2 * F2;
470  double Pn_det1 = Rn_det2;
471  double Pn_det2 = F0 * F4 - F2 * F2;
472  double Pn_det3 = F0 * F3 - F1 * F2;
473 
474  for (int j = 0; j < (nPoints - 1); j++) {
475  Rn[j] = fweight[j] * (Rn_det1 - uVal[j] * Rn_det2 + uVal[j] * uVal[j] * Rn_det3);
476  SumSRn += Rn[j] * Rn[j] * vErr[j] * vErr[j];
477  SumRn += Rn[j];
478 
479  Pn[j] = fweight[j] * (-Pn_det1 + uVal[j] * Pn_det2 - uVal[j] * uVal[j] * Pn_det3);
480  SumSPn += Pn[j] * Pn[j] * vErr[j] * vErr[j];
481  SumUPn += uVal[j] * Pn[j];
482 
483  SumRP += Rn[j] * Pn[j] * vErr[j] * vErr[j] * vErr[j];
484  }
485 
486  Double_t invqpt_fcf;
487  Int_t qfcf;
488  // chi2 = 0.;
489  if (LinearRegression((nPoints - 1), uVal, vVal, vErr, B, Berr, A, Aerr)) {
490  // v = a * u + b
491  // circle passing through (0,0):
492  // (x - rx)^2 + (y - ry)^2 = r^2
493  // ---> a = - rx / ry;
494  // ---> b = 1 / (2 * ry)
495  b = 1. / (2. * A);
496  a = -B * b;
497  r = std::sqrt(a * a + b * b);
498  double_t invR = 1. / r;
499 
500  // pt --->
501  Double_t invpt = 1. / (o2::constants::math::B2C * bFieldZ * r);
502 
503  // sign(q) --->
504  // rotate around the first point (0,0) to bring the last point
505  // on the x axis (y = 0) and check the y sign of the rotated
506  // center of the circle
507  Double_t x = xVal[nPoints - 1], y = yVal[nPoints - 1], z = zVal[nPoints - 1];
508  Double_t slope = TMath::ATan2(y, x);
509  Double_t cosSlope = TMath::Cos(slope);
510  Double_t sinSlope = TMath::Sin(slope);
511  Double_t rxRot = a * cosSlope + b * sinSlope;
512  Double_t ryRot = a * sinSlope - b * cosSlope;
513  qfcf = (ryRot > 0.) ? -1 : +1;
514 
515  Double_t alpha = 2.0 * std::abs(TMath::ATan2(rxRot, ryRot));
516  Double_t x0 = xVal[0], y0 = yVal[0], z0 = zVal[0];
517  Double_t dxyz2 = (x - x0) * (x - x0) + (y - y0) * (y - y0) + (z - z0) * (z - z0);
518  Double_t cst = 1000.;
519  Double_t c_alpha = cst * alpha;
520  Double_t p, pt, pz;
521  pt = 1. / invpt;
522  p = std::sqrt(dxyz2) / c_alpha;
523  pz = std::sqrt(p * p - pt * pt);
524 
525  invqpt_fcf = qfcf * invpt;
526 
527  //error calculations:
528  double invA2 = 1. / (A * A);
529 
530  double sigmaAsq = SumSRn / (SumRn * SumRn);
531  double sigmaBsq = SumSPn / (SumUPn * SumUPn);
532  double sigmaAB = SumRP / (SumRn * SumUPn);
533 
534  double sigmaasq_FCF = TMath::Abs(0.25 * invA2 * invA2 * (B * B * sigmaAsq + A * A * sigmaBsq - A * B * sigmaAB));
535  double sigmabsq_FCF = TMath::Abs(0.25 * invA2 * invA2 * sigmaAsq);
536  double sigma2R = invR * invR * (b * b * sigmaasq_FCF + a * a * sigmabsq_FCF + 2 * a * b * TMath::Sqrt(sigmaasq_FCF) * TMath::Sqrt(sigmabsq_FCF));
537 
538  sigmainvqptsq = sigma2R * invpt * invpt * invR * invR;
539 
540  } else { // the linear regression failed...
541  LOG(warn) << "LinearRegression failed!";
542  invqpt_fcf = 1. / 100.;
543  }
544 
545  return invqpt_fcf;
546 }

549 Bool_t LinearRegression(Int_t nVal, std::vector<double>& xVal, std::vector<double>& yVal, std::vector<double>& yErr, Double_t& B, Double_t& Berr, Double_t& A, Double_t& Aerr)
550 {
551  // linear regression y = B * x + A
552 
553  Double_t S1, SXY, SX, SY, SXX, SsXY, SsXX, SsYY, Xm, Ym, s, delta, difx;
554  Double_t invYErr2;
555 
556  S1 = SXY = SX = SY = SXX = 0.0;
557  SsXX = SsYY = SsXY = Xm = Ym = 0.;
558  difx = 0.;
559  for (Int_t i = 0; i < nVal; i++) {
560  invYErr2 = 1. / (yErr[i] * yErr[i]);
561  S1 += invYErr2;
562  SXY += xVal[i] * yVal[i] * invYErr2;
563  SX += xVal[i] * invYErr2;
564  SY += yVal[i] * invYErr2;
565  SXX += xVal[i] * xVal[i] * invYErr2;
566  if (i > 0) {
567  difx += TMath::Abs(xVal[i] - xVal[i - 1]);
568  }
569  Xm += xVal[i];
570  Ym += yVal[i];
571  SsXX += xVal[i] * xVal[i];
572  SsYY += yVal[i] * yVal[i];
573  SsXY += xVal[i] * yVal[i];
574  }
575  delta = SXX * S1 - SX * SX;
576  if (delta == 0.) {
577  return kFALSE;
578  }
579  B = (SXY * S1 - SX * SY) / delta;
580  A = (SY * SXX - SX * SXY) / delta;
581 
582  Ym /= (Double_t)nVal;
583  Xm /= (Double_t)nVal;
584  SsYY -= (Double_t)nVal * (Ym * Ym);
585  SsXX -= (Double_t)nVal * (Xm * Xm);
586  SsXY -= (Double_t)nVal * (Ym * Xm);
587  Double_t eps = 1.E-24;
588  if ((nVal > 2) && (TMath::Abs(difx) > eps) && ((SsYY - (SsXY * SsXY) / SsXX) > 0.)) {
589  s = TMath::Sqrt((SsYY - (SsXY * SsXY) / SsXX) / (nVal - 2));
590  Aerr = s * TMath::Sqrt(1. / (Double_t)nVal + (Xm * Xm) / SsXX);
591  Berr = s / TMath::Sqrt(SsXX);
592  } else {
593  Aerr = 0.;
594  Berr = 0.;
595  }
596  return kTRUE;
597 }
