41 GPUd() void MEM_LG(GPUTPCGrid)::Create(float yMin, float yMax, float zMin, float zMax, int ny, int nz)
42 {
43  //* Create the grid
44  mYMin = yMin;
45  mZMin = zMin;
46 
47  float sy = CAMath::Max((yMax + 0.1f - yMin) / ny, GPUCA_MIN_BIN_SIZE);
48  float sz = CAMath::Max((zMax + 0.1f - zMin) / nz, GPUCA_MIN_BIN_SIZE);
49 
50  mStepYInv = 1.f / sy;
51  mStepZInv = 1.f / sz;
52 
53  mNy = ny;
54  mNz = nz;
55 
56  mN = mNy * mNz;
57 
58  mYMax = mYMin + mNy * sy;
59  mZMax = mZMin + mNz * sz;
60 }
