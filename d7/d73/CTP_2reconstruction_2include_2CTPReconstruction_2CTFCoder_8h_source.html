58 template <typename VEC>
59 void CTFCoder::encode(VEC& buff, const gsl::span<const CTPDigit>& data)
60 {
61  using MD = o2::ctf::Metadata::OptStore;
62  // what to do which each field: see o2::ctd::Metadata explanation
63  constexpr MD optField[CTF::getNBlocks()] = {
64  MD::EENCODE, // BLC_bcIncTrig
65  MD::EENCODE, // BLC_orbitIncTrig
66  MD::EENCODE, // BLC_bytesInput
67  MD::EENCODE, // BLC_bytesClass
68  };
69 
70  CTFHelper helper(data);
71 
72  // book output size with some margin
73  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
74  buff.resize(szIni);
75 
76  auto ec = CTF::create(buff);
77  using ECB = CTF::base;
78 
79  ec->setHeader(helper.createHeader());
80  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
81  ec->getANSHeader().majorVersion = 0;
82  ec->getANSHeader().minorVersion = 1;
83  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
84 #define ENCODECTP(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
85  // clang-format off
86  ENCODECTP(helper.begin_bcIncTrig(), helper.end_bcIncTrig(), CTF::BLC_bcIncTrig, 0);
87  ENCODECTP(helper.begin_orbitIncTrig(), helper.end_orbitIncTrig(), CTF::BLC_orbitIncTrig, 0);
88 
89  ENCODECTP(helper.begin_bytesInput(), helper.end_bytesInput(), CTF::BLC_bytesInput, 0);
90  ENCODECTP(helper.begin_bytesClass(), helper.end_bytesClass(), CTF::BLC_bytesClass, 0);
91 
92  // clang-format on
93  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
94 }

97 template <typename VTRG>
98 void CTFCoder::decode(const CTF::base& ec, VTRG& data)
99 {
100  auto header = ec.getHeader();
101  checkDictVersion(static_cast<const o2::ctf::CTFDictHeader&>(header));
102  ec.print(getPrefix(), mVerbosity);
103  std::vector<uint16_t> bcInc;
104  std::vector<uint32_t> orbitInc;
105  std::vector<uint8_t> bytesInput, bytesClass;
106 
107 #define DECODECTP(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
108  // clang-format off
109  DECODECTP(bcInc, CTF::BLC_bcIncTrig);
110  DECODECTP(orbitInc, CTF::BLC_orbitIncTrig);
111  DECODECTP(bytesInput, CTF::BLC_bytesInput);
112  DECODECTP(bytesClass, CTF::BLC_bytesClass);
113  // clang-format on
114  //
115  data.clear();
116 
117  uint32_t firstEntry = 0, digCount = 0;
118  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
119  auto itInp = bytesInput.begin();
120  auto itCls = bytesClass.begin();
121 
122  for (uint32_t itrig = 0; itrig < header.nTriggers; itrig++) {
123  // restore TrigRecord
124  if (orbitInc[itrig]) { // non-0 increment => new orbit
125  ir.bc = bcInc[itrig]; // bcInc has absolute meaning
126  ir.orbit += orbitInc[itrig];
127  } else {
128  ir.bc += bcInc[itrig];
129  }
130  auto& dig = data.emplace_back();
131  dig.intRecord = ir;
132  for (int i = 0; i < CTFHelper::CTPInpNBytes; i++) {
133  dig.CTPInputMask |= static_cast<uint64_t>(*itInp++) << (8 * i);
134  }
135  for (int i = 0; i < CTFHelper::CTPClsNBytes; i++) {
136  dig.CTPClassMask |= static_cast<uint64_t>(*itCls++) << (8 * i);
137  }
138  }
139  assert(itInp == bytesInput.end());
140  assert(itCls == bytesClass.end());
141 }
