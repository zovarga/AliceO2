11 /*
12  Digits to RAW data coding. RAW data format - class DataFormat/Detectors/FIT/FT0/RawEventData
13  18 PMs (GBT links) 12 MCPs each and 1 TCM, each stream transmit separately
14 Event header - 80bits
15  uint startDescriptor : 4;
16  uint nGBTWords : 4;
17  uint reservedField : 28;
18  uint orbit : 32;
19  uint bc : 12;
20 
21  Event data 40bits
22  short int time : 12;
23  short int charge : 12;
24  unsigned short int numberADC : 1;
25  bool isDoubleEvent : 1;
26  bool is1TimeLostEvent : 1;
27  bool is2TimeLostEvent : 1;
28  bool isADCinGate : 1;
29  bool isTimeInfoLate : 1;
30  bool isAmpHigh : 1;
31  bool isEventInTVDC : 1;
32  bool isTimeInfoLost : 1;
33  uint channelID : 4;
34 GBT packet:
35 Event header + event data, 2 channels per 1 GBT word;
36 if no data for this PM - only headers.
37 
38 Trigger mode : detector sends data to FLP at each trigger;
39 Continueous mode : for only bunches with data at least in 1 channel.
40 */

101 
102  for (int ilink = 0; ilink < 8; ilink++) {
103  for (int ich = 0; ich < 12; ich++) {
104  LOG(info) << " ep 0 " << ilink << " " << ich << " " << lut.getChannel(ilink, ich + 1, int(0));
105  }
106  }
107  for (int ilink = 0; ilink < 10; ilink++) {
108  for (int ich = 0; ich < 12; ich++) {
109  LOG(info) << " ep 1 " << ilink << " " << ich << " " << lut.getChannel(ilink, ich + 1, int(1));
110  }
111  }
112 
113  // read content of infile
114  long posInFile = 0;
115  while (posInFile < sizeFile - sizeof(mRDH)) {
116 
117  int pos = 0;
118  mFileDest.seekg(posInFile);
119  mFileDest.read(reinterpret_cast<char*>(&mRDH), sizeof(mRDH));
120  RDHUtils::printRDH(mRDH);
121  int nwords = RDHUtils::getMemorySize(mRDH);
122  int npackages = RDHUtils::getPacketCounter(mRDH);
123  int numPage = RDHUtils::getPageCounter(mRDH);
124  int offset = RDHUtils::getOffsetToNext(mRDH);
125  int link = RDHUtils::getLinkID(mRDH);
126  int ep = RDHUtils::getEndPointID(mRDH);
127  if (nwords <= sizeof(mRDH)) {
128  posInFile += RDHUtils::getOffsetToNext(mRDH);
129  LOG(info) << " next RDH";
130  pos = 0;
131  } else {
132  posInFile += offset;
133  pos = int(sizeof(mRDH));
134 
135  while (pos < nwords) {
136  mFileDest.read(reinterpret_cast<char*>(&mEventHeader), sizeof(mEventHeader));
137  pos += sizeof(mEventHeader);
138  LOG(debug) << "read header for " << link << "word " << (int)mEventHeader.nGBTWords << " orbit " << int(mEventHeader.orbit) << " BC " << int(mEventHeader.bc) << " pos " << pos << " posinfile " << posInFile << " endPoint " << int(ep);
139  o2::InteractionRecord intrec{uint16_t(mEventHeader.bc), uint32_t(mEventHeader.orbit)};
140  auto [digitIter, isNew] = mDigitAccum.try_emplace(intrec);
141  if (isNew) {
142  double eventTime = intrec.bc2ns();
143  LOG(info) << "new intrec " << intrec.orbit << " " << intrec.bc << " link " << link << " EP " << ep;
144  o2::ft0::DigitsTemp& digit = digitIter->second;
145  digit.setTime(eventTime);
146  digit.setInteractionRecord(intrec);
147  }

149  if (link == mLinkTCM) {
150  mFileDest.read(reinterpret_cast<char*>(&mTCMdata), sizeof(mTCMdata));
151  pos += sizeof(mTCMdata);
152  digitIter->second.setTriggers(Bool_t(mTCMdata.orA), Bool_t(mTCMdata.orC), Bool_t(mTCMdata.vertex), Bool_t(mTCMdata.sCen), Bool_t(mTCMdata.cen), uint8_t(mTCMdata.nChanA), uint8_t(mTCMdata.nChanC), int32_t(mTCMdata.amplA), int32_t(mTCMdata.amplC), int16_t(mTCMdata.timeA), int16_t(mTCMdata.timeC));
153  LOG(info) << "read TCM " << (int)mEventHeader.nGBTWords << " orbit " << int(mEventHeader.orbit) << " BC " << int(mEventHeader.bc) << " pos " << pos << " posinfile " << posInFile;
154  } else {
155  if (mIsPadded) {
156  pos += CRUWordSize - o2::ft0::RawEventData::sPayloadSize;
157  }
158  for (int i = 0; i < mEventHeader.nGBTWords; ++i) {
159  mFileDest.read(reinterpret_cast<char*>(&mEventData[2 * i]), o2::ft0::RawEventData::sPayloadSizeFirstWord);
160  chDgDataArr->emplace_back(lut.getChannel(link, int(mEventData[2 * i].channelID), ep),
161  int(mEventData[2 * i].time),
162  int(mEventData[2 * i].charge),
163  int(mEventData[2 * i].numberADC));
164 
165  pos += o2::ft0::RawEventData::sPayloadSizeFirstWord;
166  LOG(info) << " read 1st word channelID " << int(mEventData[2 * i].channelID) << " charge " << mEventData[2 * i].charge << " time " << mEventData[2 * i].time << " PM " << link << " lut channel " << lut.getChannel(link, int(mEventData[2 * i].channelID), ep) << " pos " << pos;
167 
168  mFileDest.read(reinterpret_cast<char*>(&mEventData[2 * i + 1]), o2::ft0::RawEventData::sPayloadSizeSecondWord);
169  pos += o2::ft0::RawEventData::sPayloadSizeSecondWord;
170  LOG(info) << "read 2nd word channel " << int(mEventData[2 * i + 1].channelID) << " charge " << int(mEventData[2 * i + 1].charge) << " time " << mEventData[2 * i + 1].time << " PM " << link << " lut channel " << lut.getChannel(link, int(mEventData[2 * i + 1].channelID), ep) << " pos " << pos;
171  if (mEventData[2 * i + 1].charge <= 0 && mEventData[2 * i + 1].channelID <= 0 && mEventData[2 * i + 1].time <= 0) {
172  continue;
173  }
174  chDgDataArr->emplace_back(lut.getChannel(link, int(mEventData[2 * i + 1].channelID), ep),
175  int(mEventData[2 * i + 1].time),
176  int(mEventData[2 * i + 1].charge),
177  int(mEventData[2 * i + 1].numberADC));
178  }
179  }
180  }
181  }
182  }
183  close();
184 }

196 void ReadRaw::writeDigits(std::string fileDataOut)
197 {
198  TFile* mOutFile = new TFile(fileDataOut.data(), "RECREATE");
199  if (!mOutFile || mOutFile->IsZombie()) {
200  LOG(error) << "Failed to open " << fileDataOut << " output file";
201  } else {
202  LOG(info) << "Opened " << fileDataOut << " output file";
203  }
204  TTree* mOutTree = new TTree("o2sim", "o2sim");
205  // retrieve the digits from the input
206  auto inDigits = mDigitAccum;
207  LOG(info) << "RECEIVED DIGITS SIZE " << inDigits.size();
208 
209  // connect this to a particular branch
210 
211  std::vector<o2::ft0::Digit> digitVec;
212  std::vector<o2::ft0::ChannelData> chDataVec;
213  digitVec.reserve(mDigitAccum.size());
214  size_t numberOfChData = 0;
215  for (auto const& [intrec, digit] : mDigitAccum) {
216  numberOfChData += digit.getChDgData().size();
217  }
218  chDataVec.reserve(numberOfChData);
219  for (auto& [intrec, digit] : mDigitAccum) {
220  int first = gsl::narrow_cast<int>(chDataVec.size());
221  auto& chDgData = digit.getChDgData();
222  chDataVec.insert(chDataVec.end(), chDgData.begin(), chDgData.end());
223  mTrigger = digit.getTriggers();
224  o2::ft0::Digit newDigit{first, (int)chDgData.size(), intrec, mTrigger, 0};
225  newDigit.setTriggers(mTrigger);
226  newDigit.printStream(std::cout);
227  digitVec.emplace_back(newDigit);
228  }
229  mDigitAccum.clear();
230  mOutTree->Branch("FT0DIGITSBC", &digitVec);
231  mOutTree->Branch("FT0DIGITSCH", &chDataVec);
232  mOutTree->Fill();
233 
234  mOutFile->cd();
235  mOutTree->Write();
236  mOutFile->Close();
237 }
