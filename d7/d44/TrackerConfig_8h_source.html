74 inline const Int_t TrackerConfig::getBinIndex(const Int_t rIndex, const Int_t phiIndex) const
75 {
76  if (0 <= rIndex && rIndex < mRBins &&
77  0 <= phiIndex && phiIndex < mPhiBins) {
78  return (phiIndex * mRBins + rIndex);
79  }
80  return (mRBins * mPhiBins);
