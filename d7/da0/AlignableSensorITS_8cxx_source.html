43 void AlignableSensorITS::setTrackingFrame()
44 {
45  // define tracking frame of the sensor
46  double tra[3]={0},loc[3],glo[3];
47  // ITS defines tracking frame with origin in sensor, others at 0
48  getMatrixT2L().LocalToMaster(tra,loc);
49  getMatrixL2GIdeal().LocalToMaster(loc,glo);
50  mX = Sqrt(glo[0] * glo[0] + glo[1] * glo[1]);
51  mAlp = ATan2(glo[1], glo[0]);
52  utils::bringToPiPM(mAlp);
53 }
