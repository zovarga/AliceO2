139 template <typename... T>
140 float Tracker::evaluateTask(void (Tracker::*task)(T...), const char* taskName, std::function<void(std::string s)> logger,
141  T&&... args)
142 {
143  float diff{0.f};
144 
145  if (constants::DoTimeBenchmarks) {
146  auto start = std::chrono::high_resolution_clock::now();
147  (this->*task)(std::forward<T>(args)...);
148  auto end = std::chrono::high_resolution_clock::now();
149 
150  std::chrono::duration<double, std::milli> diff_t{end - start};
151  diff = diff_t.count();
152 
153  std::stringstream sstream;
154  if (taskName == nullptr) {
155  sstream << diff << "\t";
156  } else {
157  sstream << std::setw(2) << " - " << taskName << " completed in: " << diff << " ms";
158  }
159  logger(sstream.str());
160  } else {
161  (this->*task)(std::forward<T>(args)...);
162  }
163 
164  return diff;
165 }
