42 void CTFCoder::compress(CompressedDigits& cd, const gsl::span<const Digit>& digitVec, const gsl::span<const ChannelData>& channelVec)
43 {
44  // convert digits/channel to their compressed version
45  cd.clear();
46  if (!digitVec.size()) {
47  return;
48  }
49  const auto& dig0 = digitVec[0];
50  cd.header.det = mDet;
51  cd.header.nTriggers = digitVec.size();
52  cd.header.firstOrbit = dig0.getOrbit();
53  cd.header.firstBC = dig0.getBC();
54  cd.header.triggerGate = DigitizationParameters::Instance().mTime_trg_gate;
55 
56  cd.trigger.resize(cd.header.nTriggers);
57  cd.bcInc.resize(cd.header.nTriggers);
58  cd.orbitInc.resize(cd.header.nTriggers);
59  cd.eventStatus.resize(cd.header.nTriggers);
60  cd.nChan.resize(cd.header.nTriggers);
61 
62  cd.idChan.resize(channelVec.size());
63  cd.qtcChain.resize(channelVec.size());
64  cd.cfdTime.resize(channelVec.size());
65  cd.qtcAmpl.resize(channelVec.size());
66 
67  uint16_t prevBC = cd.header.firstBC;
68  uint32_t prevOrbit = cd.header.firstOrbit;
69  uint32_t ccount = 0;
70  for (uint32_t idig = 0; idig < cd.header.nTriggers; idig++) {
71  const auto& digit = digitVec[idig];
72  const auto chanels = digit.getBunchChannelData(channelVec); // we assume the channels are sorted
73 
74  // fill trigger info
75  cd.trigger[idig] = digit.getTriggers().triggersignals;
76  cd.eventStatus[idig] = digit.getEventStatusWord();
77  if (prevOrbit == digit.getOrbit()) {
78  cd.bcInc[idig] = digit.getBC() - prevBC;
79  cd.orbitInc[idig] = 0;
80  } else {
81  cd.bcInc[idig] = digit.getBC();
82  cd.orbitInc[idig] = digit.getOrbit() - prevOrbit;
83  }
84  prevBC = digit.getBC();
85  prevOrbit = digit.getOrbit();
86  // fill channels info
87  cd.nChan[idig] = chanels.size();
88  if (!cd.nChan[idig]) {
89  LOG(debug) << "Digits with no channels";
90  continue;
91  }
92  uint8_t prevChan = 0;
93  for (uint8_t ic = 0; ic < cd.nChan[idig]; ic++) {
94  assert(prevChan <= chanels[ic].ChId);
95  cd.idChan[ccount] = chanels[ic].ChId - prevChan;
96  cd.qtcChain[ccount] = chanels[ic].ChainQTC;
97  cd.cfdTime[ccount] = chanels[ic].CFDTime;
98  cd.qtcAmpl[ccount] = chanels[ic].QTCAmpl;
99  prevChan = chanels[ic].ChId;
100  ccount++;
101  }
102  }
103 }

126 {

133  sz += ESTSIZE(cd.trigger, CTF::BLC_trigger);
134  sz += ESTSIZE(cd.bcInc, CTF::BLC_bcInc);
135  sz += ESTSIZE(cd.orbitInc, CTF::BLC_orbitInc);
