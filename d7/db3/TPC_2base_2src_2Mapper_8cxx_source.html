88 bool Mapper::readMappingFile(std::string file)
89 {
90  // ===| Mapping file layout |=================================================
91  // Col 0 -> INDEX
92  // Col 1 -> PADROW
93  // Col 2 -> PAD
94  // Col 3 -> X coordinate
95  // Col 4 -> y coordinate
96  // Col 5 -> Connector
97  // Col 6 -> Pin
98  // Col 7 -> Partion
99  // Col 8 -> Region
100  // Col 9 -> FEC
101  // Col 10 -> FEC Connector
102  // Col 11 -> FEC Channel
103  // Col 12 -> SAMPA Chip
104  // Col 13 -> SAMPA Channel
105 
106  // ===| Input variables |=====================================================
107  // pad info
108  GlobalPadNumber padIndex;
109  unsigned int padRow;
110  unsigned int pad;
111  float xPos;
112  float yPos;
113 
114  // pad plane info
115  unsigned int connector;
116  unsigned int pin;
117  unsigned int partion;
118  unsigned int region;
119 
120  // FEC info
121  unsigned int fecIndex;
122  unsigned int fecConnector;
123  unsigned int fecChannel;
124  unsigned int sampaChip;
125  unsigned int sampaChannel;
126 
127  std::string line;
128  std::ifstream infile(file, std::ifstream::in);
129  if (!infile.is_open()) {
130  std::cout << "could not open file " << file << "\n";
131  exit(1);
132  }
133  while (std::getline(infile, line)) {
134  std::stringstream streamLine(line);
135  streamLine
136  // pad info
137  >> padIndex >> padRow >> pad >> xPos >> yPos
138 
139  // pad plane info
140  >> connector >> pin >> partion >> region
141 
142  // FEC info
143  >> fecIndex >> fecConnector >> fecChannel >> sampaChip >> sampaChannel;
144 
145  // the x and y positions are in mm
146  // in the mapping files, the values are given for sector C04 in the global ALICE coordinate system
147  // however, we need it in the local tracking system. Therefore:
148  const float localX = yPos / 10.f;
149  const float localY = -xPos / 10.f;
150  // with the pad counting (looking to C-Side pad 0,0 is bottom left -- pad-side front view)
151  // these values are for the C-Side
152  // For the A-Side the localY position must be mirrored
153 
154  mMapGlobalPadToPadPos[padIndex] = PadPos(padRow, pad);
155  mMapPadPosGlobalPad[PadPos(padRow, pad)] = padIndex;
156  mMapGlobalPadFECInfo[padIndex] = FECInfo(fecIndex, /*fecConnector, fecChannel,*/ sampaChip, sampaChannel);
157  mMapFECIDGlobalPad[FECInfo::globalSAMPAId(fecIndex, sampaChip, sampaChannel)] = padIndex;
158  mMapGlobalPadCentre[padIndex] = PadCentre(localX, localY);
159 
160  //std::cout
161  //<< padIndex<< " "
162  //<< padRow<< " "
163  //<< pad<< " "
164  //<< xPos<< " "
165  //<< yPos<< " "
166  //<< " "
168  //<< connector<< " "
169  //<< pin<< " "
170  //<< partion<< " "
171  //<< region<< " "
172  //<< " "
174  //<< fecIndex<< " "
175  //<< fecConnector<< " "
176  //<< fecChannel<< " "
177  //<< sampaChip<< " "
178  //<< sampaChannel << std::endl;
179  }
180  return true;
181 }

269 void Mapper::setTraceLengths(std::string_view inputFile, std::vector<float>& length)
270 {
271 
272  std::ifstream infile(inputFile.data(), std::ifstream::in);
273  if (!infile.is_open()) {
274  std::cout << "could not open file " << inputFile.data() << "\n";
275  exit(1);
276  }
277 
278  // e.g. IROC file
279  // Col 0 -> INDEX (0 - 5279)
280  // Col 1 -> PADROW (0 - 62)
281  // Col 2 -> PAD (0 - (Np-1))
282  // Col 3 -> Connector (1 - 132)
283  // Col 4 -> Pin (1 - 40)
284  // Col 5 -> Trace length (mm)
285  // Col 6 -> Number of vias
286 
287  unsigned int index{};
288  unsigned int padrow{};
289  unsigned int pad{};
290  unsigned int connector{};
291  unsigned int pin{};
292  float traceLength{};
293  unsigned int numberOfVias{};
294 
295  std::string line;
296  while (std::getline(infile, line)) {
297  std::stringstream streamLine(line);
298  streamLine >> index >> padrow >> pad >> connector >> pin >> traceLength >> numberOfVias;
299 
300  traceLength /= 10.f;
301 
302  length.emplace_back(traceLength);
303  }
304 }
