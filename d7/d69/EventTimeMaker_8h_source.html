94 template <typename trackTypeContainer,
95  typename trackType,
96  bool (*trackFilter)(const trackType&)>
97 eventTimeContainer evTimeMaker(const trackTypeContainer& tracks,
98  float diamond = 6.0 /* spread of primary verdex in cm */, bool isFast = false)
99 {
100  static std::vector<eventTimeTrack> trkWork;
101  trkWork.clear();
102  static std::vector<int> trkIndex; // indexes of working tracks in the track original array
103  trkIndex.clear();
104 
105  static float expt[3], expsigma[3];
106 
107  static eventTimeContainer result = {0, 0};
108 
109  // reset info
110  float sigmaFill = diamond * 33.356409; // move from diamond (cm) to spread on event time (ps)
111  result.weights.clear();
112  result.tracktime.clear();
113  result.eventTime = 0.;
114  result.eventTimeError = sigmaFill;
115  result.sumweights = 0.;
116 
117  for (auto track : tracks) { // Loop on tracks
118  if (trackFilter(track)) { // Select tracks good for T0 computation
119  expt[0] = track.tofExpSignalPi();
120  expt[1] = track.tofExpSignalKa();
121  expt[2] = track.tofExpSignalPr();
122  expsigma[0] = track.tofExpSigmaPi();
123  expsigma[1] = track.tofExpSigmaKa();
124  expsigma[2] = track.tofExpSigmaPr();
125  trkWork.emplace_back(track.tofSignal(), expt, expsigma);
126  trkIndex.push_back(result.weights.size());
127  }
128  result.weights.push_back(0.);
129  result.tracktime.push_back(0.);
130  }
131  if (!isFast) {
132  computeEvTime(trkWork, trkIndex, result);
133  } else {
134  computeEvTimeFast(trkWork, trkIndex, result);
135  }
136  return result;
137 }

139 template <typename trackTypeContainer,
140  typename trackType,
141  bool (*trackFilter)(const trackType&),
142  template <typename T, o2::track::PID::ID> typename response,
143  typename responseParametersType>
144 eventTimeContainer evTimeMakerFromParam(const trackTypeContainer& tracks,
145  const responseParametersType& responseParameters,
146  float diamond = 6.0 /* spread of primary verdex in cm */, bool isFast = false)
147 {
148  static std::vector<eventTimeTrack> trkWork;
149  trkWork.clear();
150  static std::vector<int> trkIndex; // indexes of working tracks in the track original array
151  trkIndex.clear();
152 
153  constexpr auto responsePi = response<trackType, o2::track::PID::Pion>();
154  constexpr auto responseKa = response<trackType, o2::track::PID::Kaon>();
155  constexpr auto responsePr = response<trackType, o2::track::PID::Proton>();
156 
157  static float expt[3], expsigma[3];
158 
159  static eventTimeContainer result = {0, 0};
160 
161  // reset info
162  float sigmaFill = diamond * 33.356409; // move from diamond (cm) to spread on event time (ps)
163  result.weights.clear();
164  result.tracktime.clear();
165  result.eventTime = 0.;
166  result.eventTimeError = sigmaFill;
167  result.sumweights = 0.;
168 
169  for (auto track : tracks) { // Loop on tracks
170  if (trackFilter(track)) { // Select tracks good for T0 computation
171  expt[0] = responsePi.GetExpectedSignal(track);
172  expt[1] = responseKa.GetExpectedSignal(track);
173  expt[2] = responsePr.GetExpectedSignal(track);
174  expsigma[0] = responsePi.GetExpectedSigmaTracking(responseParameters, track);
175  expsigma[1] = responseKa.GetExpectedSigmaTracking(responseParameters, track);
176  expsigma[2] = responsePr.GetExpectedSigmaTracking(responseParameters, track);
177  trkWork.emplace_back(track.tofSignal(), expt, expsigma);
178  trkIndex.push_back(result.weights.size());
179  }
180  result.weights.push_back(0.);
181  result.tracktime.push_back(0.);
182  }
183  if (!isFast) {
184  computeEvTime(trkWork, trkIndex, result);
185  } else {
186  computeEvTimeFast(trkWork, trkIndex, result);
187  }
188  return result;
189 }
