77 void Mille::mille(int NLC, const float* derLc,
78  int NGL, const float* derGl, const int* label,
79  float rMeas, float sigma)
80 {
81  if (sigma <= 0.) {
82  return;
83  }
84  if (myBufferPos == -1) {
85  this->newSet();
86  } // start, e.g. new track
87  if (!this->checkBufferSize(NLC, NGL)) {
88  return;
89  }
90 
91  // first store measurement
92  ++myBufferPos;
93  float* bufferFloat = myBufferFloat.GetArray();
94  int* bufferInt = myBufferInt.GetArray();
95  bufferFloat[myBufferPos] = rMeas;
96  bufferInt[myBufferPos] = 0;
97 
98  // store local derivatives and local 'lables' 1,...,NLC
99  for (int i = 0; i < NLC; ++i) {
100  if (derLc[i] || myWriteZero) { // by default store only non-zero derivatives
101  ++myBufferPos;
102  bufferFloat[myBufferPos] = derLc[i]; // local derivatives
103  bufferInt[myBufferPos] = i + 1; // index of local parameter
104  }
105  }
106 
107  // store uncertainty of measurement in between locals and globals
108  ++myBufferPos;
109  bufferFloat[myBufferPos] = sigma;
110  bufferInt[myBufferPos] = 0;
111 
112  // store global derivatives and their labels
113  for (int i = 0; i < NGL; ++i) {
114  if (derGl[i] || myWriteZero) { // by default store only non-zero derivatives
115  if ((label[i] > 0 || myWriteZero) && label[i] <= myMaxLabel) { // and for valid labels
116  ++myBufferPos;
117  bufferFloat[myBufferPos] = derGl[i]; // global derivatives
118  bufferInt[myBufferPos] = label[i]; // index of global parameter
119  } else {
120  std::cerr << "Mille::mille: Invalid label " << label[i]
121  << " <= 0 or > " << myMaxLabel << std::endl;
122  }
123  }
124  }
125 }

134 void Mille::special(int nSpecial, const float* floatings, const int* integers)
135 {
136  if (nSpecial == 0) {
137  return;
138  }
139  if (myBufferPos == -1) {
140  this->newSet();
141  } // start, e.g. new track
142  if (myHasSpecial) {
143  std::cerr << "Mille::special: Special values already stored for this record."
144  << std::endl;
145  return;
146  }
147  if (!this->checkBufferSize(nSpecial, 0)) {
148  return;
149  }
150  myHasSpecial = true; // after newSet() (Note: MILLSP sets to buffer position...)
151 
152  // myBufferFloat[.] | myBufferInt[.]
153  // ------------------------------------
154  // 0.0 | 0
155  // -float(nSpecial) | 0
156  // The above indicates special data, following are nSpecial floating and nSpecial integer data.
157  //
158  float* bufferFloat = myBufferFloat.GetArray();
159  int* bufferInt = myBufferInt.GetArray();
160  //
161  ++myBufferPos; // zero pair
162  bufferFloat[myBufferPos] = 0.;
163  bufferInt[myBufferPos] = 0;
164 
165  ++myBufferPos; // nSpecial and zero
166  bufferFloat[myBufferPos] = -nSpecial; // automatic conversion to float
167  bufferInt[myBufferPos] = 0;
168 
169  for (int i = 0; i < nSpecial; ++i) {
170  ++myBufferPos;
171  bufferFloat[myBufferPos] = floatings[i];
172  bufferInt[myBufferPos] = integers[i];
173  }
174 }

185 int Mille::end()
186 {
187  int wrote = 0;
188  if (myBufferPos > 0) { // only if anything stored...
189  const int numWordsToWrite = (myBufferPos + 1) * 2;
190  float* bufferFloat = myBufferFloat.GetArray();
191  int* bufferInt = myBufferInt.GetArray();
192 
193  if (myAsBinary) {
194  myOutFile.write(reinterpret_cast<const char*>(&numWordsToWrite),
195  sizeof(numWordsToWrite));
196  myOutFile.write(reinterpret_cast<char*>(bufferFloat),
197  (myBufferPos + 1) * sizeof(bufferFloat[0]));
198  myOutFile.write(reinterpret_cast<char*>(bufferInt),
199  (myBufferPos + 1) * sizeof(bufferInt[0]));
200  } else {
201  myOutFile << numWordsToWrite << "\n";
202  for (int i = 0; i < myBufferPos + 1; ++i) {
203  myOutFile << bufferFloat[i] << " ";
204  }
205  myOutFile << "\n";
206 
207  for (int i = 0; i < myBufferPos + 1; ++i) {
208  myOutFile << bufferInt[i] << " ";
209  }
210  myOutFile << "\n";
211  }
212  wrote = (myBufferPos + 1) * (sizeof(bufferFloat[0]) + sizeof(bufferInt[0])) + sizeof(int);
213  }
214  myBufferPos = -1; // reset buffer for next set of derivatives
215  return wrote;
216 }

235 bool Mille::checkBufferSize(int nLocal, int nGlobal)
236 {
237  if (myBufferPos + nLocal + nGlobal + 2 >= myBufferInt.GetSize()) {
238  ++(myBufferInt[0]); // increase error count
239  std::cerr << "Mille::checkBufferSize: Buffer too short ("
240  << myBufferInt.GetSize() << "),"
241  << "\n need space for nLocal (" << nLocal << ")"
242  << "/nGlobal (" << nGlobal << ") local/global derivatives, "
243  << myBufferPos + 1 << " already stored!"
244  << std::endl;
245  // return false;
246  myBufferInt.Set(myBufferPos + nLocal + nGlobal + 1000);
247  myBufferFloat.Set(myBufferPos + nLocal + nGlobal + 1000);
248  }
249  return true;
250 }
