74 std::map<o2::InteractionRecord, std::vector<int>> groupIR(gsl::span<const o2::InteractionRecord> records, uint32_t width)
75 {
76  if (!std::is_sorted(records.begin(), records.end())) {
77  throw std::invalid_argument("input records must be sorted");
78  }
79  if (width < 1) {
80  throw std::invalid_argument("width must be >=1");
81  }
82  std::map<o2::InteractionRecord, std::vector<int>> binned;
83  auto ir0 = records[0];
84  for (auto i = 0; i < records.size(); ++i) {
85  auto ir = records[i];
86  auto mchIR = ir;
87  mchIR.bc = mchIR.bc - mchIR.bc % width;
88  binned[mchIR].emplace_back(i);
89  }
90  return binned;
91 }
