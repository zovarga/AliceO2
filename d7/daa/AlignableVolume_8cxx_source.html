17 /*
18  Alignment formalism:
19 
20  Vector l in the local frame of the volume_j (assuming hierarchy of nested volumes 0...J
21  from most coarse to the end volume) is transformed to master frame vector
22  g = G_j*l_j
23  Matrix G_j is Local2Global matrix (L2G in the code). If the volume has a parent
24  volume j-1, the global vector g can be transformed to the local volume of j-1 as
25  l_{j-1} = G^-1_{j-1}* g
26  Hence, the transormation from volume j to j-1 is
27  l_{j-1} = G^-1_{j-1}*G_j l_j = R_j*l_j
28 
29  The alignment corrections in general can be defined either as a
30 
31  1) local delta: l'_j = delta_j * l_j
32  hence g' = G_j * delta_j = G'_j*l_j
33  or as
34  2) global Delta: g' = Delta_j * G_j * l_j = G'_j*l_j
35 
36  Hence Delta and delta are linked as
37  Delta_j = G_j delta_j G^-1_j
38  delta_j = G^-1_j Delta_j G_j
39 
40  In case the whole chain of nested volumes is aligned, the corrections pile-up as:
41 
42  G_0*delta_0 ... G^-1_{j-2}*G_{j-1}*delta_{j-1}*G^-1_{j-1}*G_j*delta_j =
43  Delta_0 * Delta_{1} ... Delta_{j-1}*Delta_{j}... * G_j
44 
45  -> Delta_j = G'_{j-1} * G^-1_{j-1} * G_j * G'^-1_j
46  where G and G' are modified and original L2G matrices
47 
48 
49  From this by induction one gets relation between local and global deltas:
50 
51  Delta_j = Z_j * delta_j * Z^-1_j
52 
53  where Z_j = [ Prod_{k=0}^{j-1} (G_k * delta_k * G^-1_k) ] * G_j
54 
55  By convention, aliroot aligment framework stores global Deltas !
56 
57  In case the geometry was already prealigned by PDelta_j matrices, the result
58  of the new incremental alignment Delta_j must be combined with PDelta_j to
59  resulting matrix TDelta_j before writing new alignment object.
60 
61  Derivation: if G_j and IG_j are final and ideal L2G matrices for level j, then
62 
63  G_j = TDelta_j * TDelta_{j-1} ... TDelta_0 * IG_j
64  = (Delta_j * Delta_{j-1} ... Delta_0) * (PDelta_j * PDelta_{j-1} ... PDelta_0) * IG_j
65 
66  Hence:
67  TDelta_j = [Prod_{i=j}^0 Delta_i ] * [Prod_{k=j}^0 PDelta_k ] * [Prod_{l=0}^{j-1} TDelta_l]
68 
69  By induction we get combination rule:
70 
71  TDelta_j = Delta_j * X_{j-1} * PDelta_j * X^-1_{j-1}
72 
73  where X_i = Delta_i * Delta_{i-1} ... Delta_0
74 
75  ---------------------------------
76 
77  This alignment framework internally allows to find geometry corrections either in the
78  volume LOCAL frame or in its TRACKING frame. The latter is defined for sensors as
79  lab frame, rotated by the angle alpha in such a way that the X axis is normal to the
80  sensor plane (note, that for ITS the rotated X axis origin is also moved to the sensor)
81  For the non-sensor volumes the TRACKING frame is defined by rotation of the lab frame
82  with the alpha angle = average angle of centers of its children, seen from the origin.
83 
84  The TRACKING and IDEAL LOCAL (before misalignment) frames are related by the
85  tracking-to-local matrix (T2L in the code), i.e. the vectors in local and tracking frames
86  are related as
87  l = T2L * t
88 
89  The alignment can be done using both frames for different volumes of the same geometry
90  branch.
91  The alignments deltas in local and tracking frames are related as:
92 
93  l' = T2L * delta_t * t
94  l' = delta_l * T2L * t
95  -> delta_l = T2L * delta_t * T2L^-1
96 
97  */

170 void AlignableVolume::getDeltaT2LmodLOC(TGeoHMatrix& matMod, const double* delta) const
171 {
172  // prepare the variation matrix tau in volume TRACKING frame by applying
173  // local delta of modification of LOCAL frame:
174  // tra' = tau*tra = tau*T2L^-1*loc = T2L^-1*loc' = T2L^-1*delta*loc
175  // tau = T2L^-1*delta*T2L
176  delta2Matrix(matMod, delta);
177  matMod.Multiply(&getMatrixT2L());
178  const TGeoHMatrix& t2li = getMatrixT2L().Inverse();
179  matMod.MultiplyLeft(&t2li);
180 }

182 //__________________________________________________________________
183 void AlignableVolume::getDeltaT2LmodLOC(TGeoHMatrix& matMod, const double* delta, const TGeoHMatrix& relMat) const
184 {
185  // prepare the variation matrix tau in volume TRACKING frame by applying
186  // local delta of modification of LOCAL frame of its PARENT;
187  // The relMat is matrix for transformation from child to parent frame: LOC = relMat*loc
188  //
189  // tra' = tau*tra = tau*T2L^-1*loc = T2L^-1*loc' = T2L^-1*relMat^-1*Delta*relMat*loc
190  // tau = (relMat*T2L)^-1*Delta*(relMat*T2L)
191  delta2Matrix(matMod, delta);
192  TGeoHMatrix tmp = relMat;
193  tmp *= getMatrixT2L();
194  matMod.Multiply(&tmp);
195  const TGeoHMatrix& tmpi = tmp.Inverse();
196  matMod.MultiplyLeft(&tmpi);
197 }

210 {
211  // prepare the variation matrix tau in volume TRACKING frame by applying

219  matMod.Multiply(&relMat);
220  const TGeoHMatrix& reli = relMat.Inverse();
221  matMod.MultiplyLeft(&reli);

292 //____________________________________________
293 void AlignableVolume::prepareMatrixL2G(bool reco)
294 {
295  // extract from geometry L2G matrix, depending on reco flag, set it as a reco-time
296  // or current alignment matrix
297  const char* path = getSymName();
298  if (gGeoManager->GetAlignableEntry(path)) {
299  const TGeoHMatrix* l2g = base::GeometryManager::getMatrix(path);
300  if (!l2g) {
301  LOG(fatal) << "Failed to find L2G matrix for alignable " << path;
302  }
303  reco ? setMatrixL2GReco(*l2g) : setMatrixL2G(*l2g);
304  } else { // extract from path
305  if (!gGeoManager->CheckPath(path)) {
306  LOG(fatal) << "Volume path " << path << " is not valid!";
307  }
308  TGeoPhysicalNode* node = (TGeoPhysicalNode*)gGeoManager->GetListOfPhysicalNodes()->FindObject(path);
309  TGeoHMatrix l2g;
310  if (!node) {
311  LOG(warning) << "volume " << path << " was not misaligned, extracting original matrix";
312  if (!base::GeometryManager::getOriginalMatrix(path, l2g)) {
313  LOG(fatal) << "Failed to find ideal L2G matrix for " << path;
314  }
315  } else {
316  l2g = *node->GetMatrix();
317  }
318  reco ? setMatrixL2GReco(l2g) : setMatrixL2G(l2g);
319  }
320 }

333 //____________________________________________
334 void AlignableVolume::prepareMatrixT2L()
335 {
336  // for non-sensors we define the fake tracking frame with the alpha angle being
337  // the average angle of centers of its children
338  //
339  double tot[3] = {0, 0, 0}, loc[3] = {0, 0, 0}, glo[3];
340  int nch = getNChildren();
341  for (int ich = nch; ich--;) {
342  AlignableVolume* vol = getChild(ich);
343  vol->getMatrixL2GIdeal().LocalToMaster(loc, glo);
344  for (int j = 3; j--;) {
345  tot[j] += glo[j];
346  }
347  }
348  if (nch) {
349  for (int j = 3; j--;) {
350  tot[j] /= nch;
351  }
352  }
353  //
354  mAlp = TMath::ATan2(tot[1], tot[0]);
355  utils::bringToPiPM(mAlp);
356  //
357  mX = TMath::Sqrt(tot[0] * tot[0] + tot[1] * tot[1]);
358  //
359  // 1st create Tracking to Global matrix
360  mMatT2L.Clear();
361  mMatT2L.SetDx(mX);
362  mMatT2L.RotateZ(mAlp * RadToDeg());
363  // then convert it to Tracking to Local matrix
364  const TGeoHMatrix& l2gi = getMatrixL2GIdeal().Inverse();
365  mMatT2L.MultiplyLeft(&l2gi);
366  //
367 }

551  // children volume

553  //
554  for (int ich = 0; ich < nch; ich++) {

556  }
557  //
558 }

603 //_________________________________________________________________
604 void AlignableVolume::createPreGloDeltaMatrix(TGeoHMatrix& deltaM) const
605 {
606  // Create prealignment global matrix deltaM from prealigned G and
607  // original GO local-to-global matrices
608  //
609  // From G_j = Delta_j * Delta_{j-1} ... Delta_0 * GO_j
610  // where Delta_j is global prealignment matrix for volume at level j
611  // we get by induction
612  // Delta_j = G_j * GO^-1_j * GO_{j-1} * G^-1_{j-1}
613  //
614  deltaM = getMatrixL2G();
615  deltaM *= getMatrixL2GIdeal().Inverse();
