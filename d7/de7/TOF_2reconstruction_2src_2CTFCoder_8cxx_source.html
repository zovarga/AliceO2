40 void CTFCoder::compress(CompressedInfos& cc,
41  const gsl::span<const ReadoutWindowData>& rofRecVec,
42  const gsl::span<const Digit>& cdigVec,
43  const gsl::span<const uint8_t>& pattVec)
44 {
45  // store in the header the orbit of 1st ROF
46  cc.clear();
47  if (!rofRecVec.size()) {
48  return;
49  }
50  const auto& rofRec0 = rofRecVec[0];
51  int nrof = rofRecVec.size();
52 
53  LOGF(debug, "TOF compress %d ReadoutWindow with %ld digits", nrof, cdigVec.size());
54  cc.header.det = mDet;
55  cc.header.nROFs = nrof;
56  cc.header.firstOrbit = rofRec0.getBCData().orbit;
57  cc.header.firstBC = rofRec0.getBCData().bc;
58  cc.header.nPatternBytes = pattVec.size();
59  cc.header.nDigits = cdigVec.size();
60 
61  cc.bcIncROF.resize(cc.header.nROFs);
62  cc.orbitIncROF.resize(cc.header.nROFs);
63  cc.ndigROF.resize(cc.header.nROFs);
64  cc.ndiaROF.resize(cc.header.nROFs);
65  cc.ndiaCrate.resize(cc.header.nROFs * 72);
66  //
67  cc.timeFrameInc.resize(cc.header.nDigits);
68  cc.timeTDCInc.resize(cc.header.nDigits);
69  cc.stripID.resize(cc.header.nDigits);
70  cc.chanInStrip.resize(cc.header.nDigits);
71  cc.tot.resize(cc.header.nDigits);
72  cc.pattMap.resize(pattVec.size());
73 
74  uint16_t prevBC = cc.header.firstBC;
75  uint32_t prevOrbit = cc.header.firstOrbit;
76 
77  std::vector<Digit> digCopy;
78 
79  for (uint32_t irof = 0; irof < rofRecVec.size(); irof++) {
80  const auto& rofRec = rofRecVec[irof];
81 
82  const auto& intRec = rofRec.getBCData();
83  int64_t rofInBC = intRec.toLong();
84  // define interaction record
85  if (intRec.orbit == prevOrbit) {
86  cc.orbitIncROF[irof] = 0;
87  cc.bcIncROF[irof] = intRec.bc - prevBC; // store increment of BC if in the same orbit
88  } else {
89  cc.orbitIncROF[irof] = intRec.orbit - prevOrbit;
90  cc.bcIncROF[irof] = intRec.bc; // otherwise, store absolute bc
91  prevOrbit = intRec.orbit;
92  }
93  prevBC = intRec.bc;
94  auto ndig = rofRec.size();
95  cc.ndigROF[irof] = ndig;
96  cc.ndiaROF[irof] = rofRec.sizeDia();
97  for (int icrate = 0; icrate < 72; icrate++) {
98  if (rofRec.isEmptyCrate(icrate)) {
99  cc.ndiaCrate[irof * 72 + icrate] = 0;
100  } else {
101  cc.ndiaCrate[irof * 72 + icrate] = rofRec.getDiagnosticInCrate(icrate) + 1; // shifted by one since -1 means crate not available (then to get unsigned int)
102  }
103  }
104 
105  if (!ndig) { // no hits data for this ROF --> not fill
106  continue;
107  }
108 
109  int idigMin = rofRec.first(), idigMax = idigMin + ndig;
110  int idig = idigMin;
111 
112  // make a copy of digits
113  digCopy.clear();
114  for (; idig < idigMax; idig++) {
115  digCopy.emplace_back(cdigVec[idig]);
116  }
117 
118  // sort digits according to time (ascending order)
119  std::sort(digCopy.begin(), digCopy.end(),
120  [](o2::tof::Digit a, o2::tof::Digit b) {
121  if (a.getBC() == b.getBC()) {
122  return a.getTDC() < b.getTDC();
123  } else {
124  return a.getBC() < b.getBC();
125  }
126  });
127 
128  int timeframe = 0;
129  int tdc = 0;
130  idig = idigMin;
131  for (; idig < idigMax; idig++) {
132  const auto& dig = digCopy[idig - idigMin];
133  int deltaBC = dig.getBC() - rofInBC;
134  int ctimeframe = deltaBC / 64;
135  int cTDC = (deltaBC % 64) * 1024 + dig.getTDC();
136  if (ctimeframe == timeframe) {
137  cc.timeFrameInc[idig] = 0;
138  cc.timeTDCInc[idig] = cTDC - tdc;
139  } else {
140  cc.timeFrameInc[idig] = ctimeframe - timeframe;
141  cc.timeTDCInc[idig] = cTDC;
142  timeframe = ctimeframe;
143  }
144  tdc = cTDC;
145 
146  int chan = dig.getChannel();
147  cc.stripID[idig] = chan / Geo::NPADS;
148  cc.chanInStrip[idig] = chan % Geo::NPADS;
149  cc.tot[idig] = dig.getTOT();
150  LOGF(debug, "%d) TOFBC = %d, deltaBC = %d, TDC = %d, CH=%d", irof, rofInBC, deltaBC, cTDC, chan);
151  LOGF(debug, "%d) TF=%d, TDC=%d, STRIP=%d, CH=%d, TOT=%d", idig, cc.timeFrameInc[idig], cc.timeTDCInc[idig], cc.stripID[idig], cc.chanInStrip[idig], cc.tot[idig]);
152  }
153  }
154  memcpy(cc.pattMap.data(), pattVec.data(), cc.header.nPatternBytes); // RSTODO: do we need this?
155 }

180 size_t CTFCoder::estimateCompressedSize(const CompressedInfos& cc)
181 {
182  size_t sz = 0;
183  // clang-format off
184  // RS FIXME this is very crude estimate, instead, an empirical values should be used
185 #define VTP(vec) typename std::remove_reference<decltype(vec)>::type::value_type
186 #define ESTSIZE(vec, slot) mCoders[int(slot)] ? \
187  rans::calculateMaxBufferSize(vec.size(), reinterpret_cast<const o2::rans::LiteralEncoder64<VTP(vec)>*>(mCoders[int(slot)].get())->getAlphabetRangeBits(), sizeof(VTP(vec)) ) : vec.size()*sizeof(VTP(vec))
188 
189  sz += ESTSIZE(cc.bcIncROF, CTF::BLCbcIncROF);
190  sz += ESTSIZE(cc.orbitIncROF, CTF::BLCorbitIncROF);
191  sz += ESTSIZE(cc.ndigROF, CTF::BLCndigROF);
192  sz += ESTSIZE(cc.ndiaROF, CTF::BLCndiaROF);
193  sz += ESTSIZE(cc.ndiaCrate, CTF::BLCndiaCrate);
194  sz += ESTSIZE(cc.timeFrameInc, CTF::BLCtimeFrameInc);
195  sz += ESTSIZE(cc.timeTDCInc, CTF::BLCtimeTDCInc);
196  sz += ESTSIZE(cc.stripID, CTF::BLCstripID);
197  sz += ESTSIZE(cc.chanInStrip, CTF::BLCchanInStrip);
198  sz += ESTSIZE(cc.tot, CTF::BLCtot);
199  sz += ESTSIZE(cc.pattMap, CTF::BLCpattMap);
200  // clang-format on
201  sz *= 2. / 3; // if needed, will be autoexpanded
202  LOG(debug) << "Estimated output size is " << sz << " bytes";
203  return sz;
204 }
