45 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
46 {
47  DetID::mask_t dets = 0;
48  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
49  long run = 0;
50  bool doCTF = true, doDict = false, dictPerDet = false;
51  size_t szMin = 0, szMax = 0;
52  std::string outType{}; // RS FIXME once global/local options clash is solved, --output-type will become device option
53  if (!configcontext.helpOnCommandLine()) {
54  bool noGRP = configcontext.options().get<bool>("no-grp");
55  auto onlyDet = configcontext.options().get<std::string>("onlyDet");
56  if (!noGRP) {
57  std::unique_ptr<o2::parameters::GRPObject> grp(o2::parameters::GRPObject::loadFrom(configcontext.options().get<std::string>("grpfile")));
58  dets = grp->getDetsReadOut(onlyDet, configcontext.options().get<std::string>("skipDet"));
59  run = grp->getRun();
60  } else {
61  dets.set(); // by default read all
62  auto mskOnly = DetID::getMask(configcontext.options().get<std::string>("onlyDet"));
63  auto mskSkip = DetID::getMask(configcontext.options().get<std::string>("skipDet"));
64  if (mskOnly.any()) {
65  dets &= mskOnly;
66  } else {
67  dets ^= mskSkip;
68  }
69  run = 0;
70  }
71  if (dets.none()) {
72  throw std::invalid_argument("Invalid workflow: no detectors found");
73  }
74  outType = configcontext.options().get<std::string>("output-type");
75  }
76  WorkflowSpec specs{o2::ctf::getCTFWriterSpec(dets, run, outType, configcontext.options().get<int>("ctf-writer-verbosity"))};
77  return std::move(specs);
78 }
