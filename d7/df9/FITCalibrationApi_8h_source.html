75 template <typename CalibrationObjectType>
76 FITCalibrationApi::CalibObjWithInfoType FITCalibrationApi::doSerializationAndPrepareObjectInfo(const CalibrationObjectType& calibrationObject, TFType starting, TFType stopping)
77 {
78  static std::map<std::string, std::string> metaData;
79  static std::string dummyStringVariableThatWillBeChangedAnyway;
80  CalibObjWithInfoType result;
81  auto clName = o2::utils::MemFileHelper::getClassName(calibrationObject);
82  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
83  LOG(info) << " clName " << clName << " flName " << flName;
84  result.first = o2::ccdb::CcdbObjectInfo(FITCalibrationApi::getObjectPath<CalibrationObjectType>(), clName, flName, metaData, starting, stopping);
85  stopping = 999999999999;
86  result.second = o2::ccdb::CcdbApi::createObjectImage(&calibrationObject, &result.first);
87  LOG(info) << " FITCalibrationApi::doSerializationAndPrepareObjectInfo"
88  << " start " << starting << " end " << stopping;
89  return result;
90 }

92 template <typename CalibrationObjectType>
93 const CalibrationObjectType& FITCalibrationApi::getCalibrationObjectForGivenTimestamp(long timestamp)
94 {
95 
96  o2::ccdb::BasicCCDBManager::instance().setTimestamp(timestamp);
97  auto calibObjectPtr = o2::ccdb::BasicCCDBManager::instance().get<CalibrationObjectType>(getObjectPath<CalibrationObjectType>());
98  if (nullptr == calibObjectPtr) {
99  throw std::runtime_error("Cannot read requested calibration object");
100  }
101  return *calibObjectPtr;
102 }
