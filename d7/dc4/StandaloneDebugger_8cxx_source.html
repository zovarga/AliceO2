73  }
74 
75  for (auto& combination : comb12) {
76  o2::MCCompLabel lblClus1 = *(event->getClusterLabels(1, clusters[1][combination.secondClusterIndex].clusterId).begin());

94 void StandaloneDebugger::fillTrackletSelectionTree(std::array<std::vector<Cluster>, constants::its::LayersNumberVertexer>& clusters,
95  std::vector<Tracklet> comb01,
96  std::vector<Tracklet> comb12,
97  std::vector<std::array<int, 2>> allowedTracklets,
98  const ROframe* event)
99 {
100  assert(event != nullptr);
101  assert(mTreeStream != nullptr);
102  int id = event->getROFrameId();
103  for (auto& trackletPair : allowedTracklets) {
104  o2::MCCompLabel lblClus0 = *(event->getClusterLabels(0, clusters[0][comb01[trackletPair[0]].firstClusterIndex].clusterId).begin());
105  o2::MCCompLabel lblClus1 = *(event->getClusterLabels(1, clusters[1][comb01[trackletPair[0]].secondClusterIndex].clusterId).begin());
106  o2::MCCompLabel lblClus2 = *(event->getClusterLabels(2, clusters[2][comb12[trackletPair[1]].secondClusterIndex].clusterId).begin());
107  unsigned char isValidated{lblClus0.compare(lblClus1) == 1 && lblClus0.compare(lblClus2) == 1};
108  float deltaPhi{comb01[trackletPair[0]].phi - comb12[trackletPair[1]].phi};
109  float deltaTanLambda{comb01[trackletPair[0]].tanLambda - comb12[trackletPair[1]].tanLambda};
110  mTreeStream->GetDirectory()->cd(); // in case of existing other open files
111  (*mTreeStream)
112  << "selectedTracklets"
113  << "ROframeId=" << id
114  << "deltaTanlambda=" << deltaTanLambda
115  << "deltaPhi=" << deltaPhi
116  << "isValidated=" << isValidated
117  << "cluster0z=" << clusters[0][comb01[trackletPair[0]].firstClusterIndex].zCoordinate
118  << "cluster0r=" << clusters[0][comb01[trackletPair[0]].firstClusterIndex].radius
119  << "cluster0phi=" << clusters[0][comb01[trackletPair[0]].firstClusterIndex].phi
120  << "cluster1z=" << clusters[1][comb01[trackletPair[0]].secondClusterIndex].zCoordinate
121  << "cluster1r=" << clusters[1][comb01[trackletPair[0]].secondClusterIndex].radius
122  << "cluster1phi=" << clusters[1][comb01[trackletPair[0]].secondClusterIndex].phi
123  << "cluster2z=" << clusters[2][comb12[trackletPair[1]].secondClusterIndex].zCoordinate
124  << "cluster2r=" << clusters[2][comb12[trackletPair[1]].secondClusterIndex].radius
125  << "lblClus0=" << lblClus0
126  << "lblClus1=" << lblClus1
127  << "lblClus2=" << lblClus2
128  << "\n";
129  }
130 }

157 void StandaloneDebugger::fillPairsInfoTree(std::vector<Line> lines, const ROframe* event)
158 {
159  assert(event != nullptr);
160  int id = event->getROFrameId();
161  for (unsigned int iLine1{0}; iLine1 < lines.size(); ++iLine1) { // compute centroids for every line pair
162  auto line1 = lines[iLine1];
163  for (unsigned int iLine2{iLine1 + 1}; iLine2 < lines.size(); ++iLine2) {
164  auto line2 = lines[iLine2];
165  ClusterLines cluster{-1, line1, -1, line2};
166  auto vtx = cluster.getVertex();
167  if (std::hypot(vtx[0], vtx[1]) < 1.98) {
168  float dcaPair = Line::getDCA(line1, line2);
169  if (dcaPair < 0.02) {
170  (*mTreeStream)
171  << "pairInfo"
172  << "centroids"
173  << "ROframeId=" << id
174  << "xCoord=" << vtx[0]
175  << "yCoord=" << vtx[1]
176  << "zCoord=" << vtx[2]
177  << "DCApair=" << dcaPair
178  << "\n";
179  }
180  }

248 int StandaloneDebugger::getBinIndex(const float value, const int size, const float min, const float max)
249 {
250  std::vector<double> divisions;
251  const double binsize = (max - min) / size;
252  divisions.resize(size + 1);
253  for (int i{0}; i <= size; ++i) {
254  divisions[i] = min + i * binsize;
255  }
256  return std::distance(divisions.begin(), TMath::BinarySearch(divisions.begin(), divisions.end(), value));
257 }
