145 void GPUDisplayFrontendGlfw::GetKey(int key, int scancode, int mods, int& keyOut, int& keyPressOut)
146 {
147  int specialKey = GetKey(key);
148  const char* str = glfwGetKeyName(key, scancode);
149  char localeKey = str ? str[0] : 0;
150  if ((mods & GLFW_MOD_SHIFT) && localeKey >= 'a' && localeKey <= 'z') {
151  localeKey += 'A' - 'a';
152  }
153  // GPUInfo("Key: key %d (%c) scancode %d -> %d (%c) special %d (%c)", key, (char)key, scancode, (int)localeKey, localeKey, specialKey, (char)specialKey);
154 
155  if (specialKey) {
156  keyOut = keyPressOut = specialKey;
157  } else {
158  keyOut = keyPressOut = localeKey;
159  if (keyPressOut >= 'a' && keyPressOut <= 'z') {
160  keyPressOut += 'A' - 'a';
161  }
162  }
163 }

167 void GPUDisplayFrontendGlfw::key_callback(GLFWwindow* window, int key, int scancode, int action, int mods)
168 {
169  int handleKey = 0, keyPress = 0;
170  GetKey(key, scancode, mods, handleKey, keyPress);
171  if (handleKey < 32) {
172  if (action == GLFW_PRESS) {
173  me->mKeys[keyPress] = true;
174  me->mKeysShift[keyPress] = mods & GLFW_MOD_SHIFT;
175  me->HandleKey(handleKey);
176  } else if (action == GLFW_RELEASE) {
177  me->mKeys[keyPress] = false;
178  me->mKeysShift[keyPress] = false;
179  }
180  } else if (handleKey < 256) {
181  if (action == GLFW_PRESS) {
182  me->mLastKeyDown = handleKey;
183  } else if (action == GLFW_RELEASE) {
184  keyPress = me->mKeyDownMap[handleKey];
185  me->mKeys[keyPress] = false;
186  me->mKeysShift[keyPress] = false;
187  }
188  }
189 }

191 void GPUDisplayFrontendGlfw::char_callback(GLFWwindow* window, unsigned int codepoint)
192 {
193  // GPUInfo("Key (char callback): %d %c - key: %d", codepoint, (char)codepoint, (int)me->mLastKeyDown);
194  int keyPress = codepoint;
195  if (keyPress >= 'a' && keyPress <= 'z') {
196  keyPress += 'A' - 'a';
197  }
198  me->mKeyDownMap[me->mLastKeyDown] = keyPress;
199  me->mKeys[keyPress] = true;
200  me->mKeysShift[keyPress] = me->mKeys[KEY_SHIFT];
201  me->HandleKey(codepoint);
202 }

247 int GPUDisplayFrontendGlfw::FrontendMain()
248 {
249  me = this;
250 
251  if (!glfwInit()) {
252  fprintf(stderr, "Error initializing glfw\n");
253  return (-1);
254  }
255  glfwSetErrorCallback(error_callback);
256 
257  glfwWindowHint(GLFW_MAXIMIZED, 1);
258  glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, GL_MIN_VERSION_MAJOR);
259  glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, GL_MIN_VERSION_MINOR);
260  glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, 0);
261  glfwWindowHint(GLFW_OPENGL_PROFILE, mBackend->CoreProfile() ? GLFW_OPENGL_CORE_PROFILE : GLFW_OPENGL_COMPAT_PROFILE);
262  mWindow = glfwCreateWindow(INIT_WIDTH, INIT_HEIGHT, GL_WINDOW_NAME, nullptr, nullptr);
263  if (!mWindow) {
264  fprintf(stderr, "Error creating glfw window\n");
265  glfwTerminate();
266  return (-1);
267  }
268  glfwMakeContextCurrent(mWindow);
269 
270  glfwSetKeyCallback(mWindow, key_callback);
271  glfwSetCharCallback(mWindow, char_callback);
272  glfwSetMouseButtonCallback(mWindow, mouseButton_callback);
273  glfwSetScrollCallback(mWindow, scroll_callback);
274  glfwSetCursorPosCallback(mWindow, cursorPos_callback);
275  glfwSetWindowSizeCallback(mWindow, resize_callback);
276 
277  pthread_mutex_lock(&mSemLockExit);
278  mGlfwRunning = true;
279  pthread_mutex_unlock(&mSemLockExit);
280 
281  if (mBackend->ExtInit()) {
282  fprintf(stderr, "Error initializing GL extension wrapper\n");
283  return (-1);
284  }
285 
286 #if defined(GPUCA_O2_LIB) && !defined(GPUCA_DISPLAY_GL3W)
287  if (gl3wInit()) {
288  fprintf(stderr, "Error initializing gl3w (2)\n");
289  return (-1); // Hack: We have to initialize gl3w as well, as the DebugGUI uses it.
290  }
291 #endif
292 
293  if (InitDisplay()) {
294  fprintf(stderr, "Error in OpenGL initialization\n");
295  return (1);
296  }
297 
298 #ifdef GPUCA_O2_LIB
299  ImGui_ImplGlfwGL3_Init(mWindow, false);
300  while (o2::framework::pollGUI(mWindow, DisplayLoop)) {
301  }
302 #else
303  while (!glfwWindowShouldClose(mWindow)) {
304  HandleSendKey();
305  if (DrawGLScene()) {
306  fprintf(stderr, "Error drawing GL scene\n");
307  return (1);
308  }
309  glfwSwapBuffers(mWindow);
310  glfwPollEvents();
311  }
312 #endif
313 
314  mDisplayControl = 2;
315  pthread_mutex_lock(&mSemLockExit);
316 #ifdef GPUCA_O2_LIB
317  ImGui_ImplGlfwGL3_Shutdown();
318 #endif
319  glfwDestroyWindow(mWindow);
320  glfwTerminate();
321  mGlfwRunning = false;
322  pthread_mutex_unlock(&mSemLockExit);
323 
324  return 0;
325 }
