37 void AltroDecoder::readRCUTrailer()
38 {
39  try {
40  auto payloadwordsOrig = mRawReader.getPayload().getPayloadWords();
41  gsl::span<const uint32_t> payloadwords(payloadwordsOrig.data(), payloadwordsOrig.size());
42  mRCUTrailer.constructFromRawPayload(payloadwords);

44  AliceO2::InfoLogger::InfoLogger logger;

46  throw AltroDecoderError(AltroDecoderError::ErrorType_t::RCU_TRAILER_ERROR, (boost::format("RCU trailer decoding error: %s") % e.what()).str().data());
47  }

59 void AltroDecoder::readChannels()
60 {
61  mChannelsInitialized = false;
62  mChannels.clear();
63  int currentpos = 0;
64  auto& buffer = mRawReader.getPayload().getPayloadWords();
65  auto maxpayloadsize = buffer.size() - mRCUTrailer.getTrailerSize();
66  while (currentpos < maxpayloadsize) {
67  auto currentword = buffer[currentpos++];
68  if (currentword >> 30 != 1) {
69  continue;
70  }
71  // starting a new channel
72  auto channelheader = currentword;
73  mChannels.emplace_back(currentword & 0xFFF, (currentword >> 16) & 0x3FF);
74  auto& currentchannel = mChannels.back();
75  currentchannel.setBadChannel((currentword >> 29) & 0x1);
76 
78  int numberofwords = (currentchannel.getPayloadSize() + 2) / 3;
79  std::vector<uint16_t> bunchwords;
80  for (int iword = 0; iword < numberofwords; iword++) {
81  if (currentpos >= maxpayloadsize) {
82  mMinorDecodingErrors.emplace_back(MinorAltroDecodingError::ErrorType_t::CHANNEL_PAYLOAD_EXCEED, channelheader, currentword);
83  break; // Must break here in order not to prevent a buffer overrun
84  }
85  currentword = buffer[currentpos++];
86  if ((currentword >> 30) != 0) {
87  mMinorDecodingErrors.emplace_back(MinorAltroDecodingError::ErrorType_t::CHANNEL_END_PAYLOAD_UNEXPECT, channelheader, currentword);
88  currentpos--;
89  continue;
90  }
91  bunchwords.push_back((currentword >> 20) & 0x3FF);
92  bunchwords.push_back((currentword >> 10) & 0x3FF);
93  bunchwords.push_back(currentword & 0x3FF);
94  }
95 
96  // decode bunches
97  int currentsample = 0;
98  while (currentsample < currentchannel.getPayloadSize() && bunchwords.size() > currentsample + 2) {
99  // Check if bunch word is 0 - if yes skip all following bunches as they can no longer be reliably decoded
100  if (bunchwords[currentsample] == 0) {
101  mMinorDecodingErrors.emplace_back(MinorAltroDecodingError::ErrorType_t::BUNCH_HEADER_NULL, channelheader, 0);
102  break;
103  }
104  int bunchlength = bunchwords[currentsample] - 2, // remove words for bunchlength and starttime
105  starttime = bunchwords[currentsample + 1];
106  auto& currentbunch = currentchannel.createBunch(bunchlength, starttime);
107  currentbunch.initFromRange(gsl::span<uint16_t>(&bunchwords[currentsample + 2], std::min((unsigned long)bunchlength, bunchwords.size() - currentsample - 2)));
108  currentsample += bunchlength + 2;
109  }
110  }
111  mChannelsInitialized = true;
112 }

206 std::string MinorAltroDecodingError::what() const noexcept
207 {
208  std::stringstream result;
209  switch (mErrorType) {
210  case ErrorType_t::CHANNEL_END_PAYLOAD_UNEXPECT:
211  result << "Unexpected end of payload in altro channel payload!";
212  break;
213  case ErrorType_t::CHANNEL_PAYLOAD_EXCEED:
214  result << "Trying to access out-of-bound payload!";
215  break;
216  case ErrorType_t::BUNCH_HEADER_NULL:
217  result << "Bunch header 0 or not configured!";
218  break;
219  };
220  auto address = mChannelHeader & 0xFFF,
221  payload = (mChannelHeader >> 16) & 0x3FF;
222  bool good = (mChannelHeader >> 29) & 0x1;
223 
224  result << " Channel header=0x" << std::hex << mChannelHeader
225  << " (Address=0x" << address << ", payload " << std::dec << payload << ", good " << (good ? "yes" : "no") << ")"
226  << ", word=0x" << std::hex << mPayloadWord << std::dec;
227  return result.str();
228 }
