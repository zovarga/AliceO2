17 void MatchGlobalFwd::init()
18 {
19 
20  LOG(info) << "Initializing Global Forward Matcher";
21 
22  auto& matchingParam = GlobalFwdMatchingParam::Instance();
23 
24  mMatchingPlaneZ = matchingParam.matchPlaneZ;
25  LOG(info) << "MFTMCH matchingPlaneZ = " << mMatchingPlaneZ;
26 
27  auto& matchingFcnStr = matchingParam.matchFcn;
28  LOG(info) << "Match function string = " << matchingFcnStr;
29 
30  if (matchingParam.isMatchUpstream()) {
31  LOG(info) << " ==> Setting Upstream matching.";
32  mMatchingType = MATCHINGUPSTREAM;
33  } else if (matchingParam.matchingExternalFunction()) {
34  loadExternalMatchingFunction();
35  mMatchingType = MATCHINGFUNC;
36  } else {
37  if (mMatchingFunctionMap.find(matchingFcnStr) != mMatchingFunctionMap.end()) {
38  mMatchFunc = mMatchingFunctionMap[matchingFcnStr];
39  mMatchingType = MATCHINGFUNC;
40  LOG(info) << " Found built-in matching function " << matchingFcnStr;
41  } else {
42  throw std::invalid_argument("Invalid matching function! Aborting...");
43  }
44  }
45 
46  auto& cutFcnStr = matchingParam.cutFcn;
47  LOG(info) << "MFTMCH pair candidate cut function string = " << cutFcnStr;
48 
49  if (matchingParam.cutExternalFunction()) {
50  loadExternalCutFunction();
51  } else if (mCutFunctionMap.find(cutFcnStr) != mCutFunctionMap.end()) {
52  mCutFunc = mCutFunctionMap[cutFcnStr];
53  LOG(info) << " Found built-in cut function " << cutFcnStr;
54  } else {
55  throw std::invalid_argument("Invalid cut function! Aborting...");
56  }
57 
58  mUseMIDMCHMatch = matchingParam.useMIDMatch;
59  LOG(info) << "UseMIDMCH Matching = " << (mUseMIDMCHMatch ? "true" : "false");
60 
61  mSaveMode = matchingParam.saveMode;
62  LOG(info) << "Save mode MFTMCH candidates = " << mSaveMode;
63 }

66 void MatchGlobalFwd::run(const o2::globaltracking::RecoContainer& inp)
67 {
68 
69  auto& matchingParam = GlobalFwdMatchingParam::Instance();
70 
71  mRecoCont = &inp;
72  mStartIR = inp.startIR;
73 
74  clear();
75 
76  if (!prepareMFTData() || !prepareMCHData() || !processMCHMIDMatches()) {
77  return;
78  }
79 
80  if (matchingParam.MCMatching) { // MC Label matching
81  mMCTruthON ? doMCMatching() : throw std::runtime_error("Label matching requries MC Labels!");
82  } else {
83  switch (mMatchingType) {
84  case MATCHINGFUNC:
85  switch (mSaveMode) {
86  case kBestMatch:
87  doMatching<kBestMatch>();
88  break;
89  case kSaveAll:
90  doMatching<kSaveAll>();
91  break;
92  case kSaveTrainingData:
93  doMatching<kSaveTrainingData>();
94  break;
95  default:
96  LOG(fatal) << "Invalid MFTMCH save mode";
97  }
98  break;
99  case MATCHINGUPSTREAM:
100  loadMatches();
101  break;
102  default:
103  LOG(fatal) << "Invalid MFTMCH matching mode";
104  }
105  }
106 
107  fitTracks();
108  finalize();
109 }

132 bool MatchGlobalFwd::prepareMCHData()
133 {
134  const auto& inp = *mRecoCont;
135 
136  // Load MCH tracks
137  mMCHTracks = inp.getMCHTracks();
138  mMCHTrackROFRec = inp.getMCHTracksROFRecords();
139  if (mMCTruthON) {
140  mMCHTrkLabels = inp.getMCHTracksMCLabels();
141  }
142  int nROFs = mMCHTrackROFRec.size();
143  LOG(info) << "Loaded " << mMCHTracks.size() << " MCH Tracks in " << nROFs << " ROFs";
144  if (mMCHTracks.empty()) {
145  return false;
146  }
147  mMCHWork.reserve(mMCHTracks.size());
148 
149  for (int irof = 0; irof < nROFs; irof++) {
150  const auto& rofRec = mMCHTrackROFRec[irof];
151 
152  int nBC = rofRec.getBCData().differenceInBC(mStartIR);
153  float tMin = nBC * o2::constants::lhc::LHCBunchSpacingMUS;
154  float tMax = (nBC + rofRec.getBCWidth()) * o2::constants::lhc::LHCBunchSpacingMUS;
155 
156  mMCHROFTimes.emplace_back(tMin, tMax); // MCH ROF min/max time
157  LOG(debug) << "MCH ROF # " << irof << " [tMin;tMax] = [" << tMin << ";" << tMax << "]";
158  int trlim = rofRec.getFirstIdx() + rofRec.getNEntries();
159  for (int it = rofRec.getFirstIdx(); it < trlim; it++) {
160  auto& trcOrig = mMCHTracks[it];
161  int nWorkTracks = mMCHWork.size();
162  // working copy MCH track propagated to matching plane and converted to the forward track format
163  o2::mch::TrackParam tempParam(trcOrig.getZ(), trcOrig.getParameters(), trcOrig.getCovariances());
164  if (!o2::mch::TrackExtrap::extrapToVertexWithoutBranson(tempParam, mMatchingPlaneZ)) {
165  LOG(warning) << "MCH track propagation to matching plane failed!";
166  continue;
167  }
168  auto convertedTrack = MCHtoFwd(tempParam);
169  auto& thisMCHTrack = mMCHWork.emplace_back(TrackLocMCH{convertedTrack, {tMin, tMax}});
170  thisMCHTrack.setMCHTrackID(it);
171  }
172  }
173  return true;
174 }

204 bool MatchGlobalFwd::prepareMFTData()
205 {
206  const auto& inp = *mRecoCont;
207 
208  // MFT clusters
209  mMFTClusterROFRec = inp.getMFTClustersROFRecords();
210  mMFTTrackClusIdx = inp.getMFTTracksClusterRefs();
211  const auto clusMFT = inp.getMFTClusters();
212  if (mMFTClusterROFRec.empty() || clusMFT.empty()) {
213  return false;
214  }
215  const auto patterns = inp.getMFTClustersPatterns();
216  auto pattIt = patterns.begin();
217  mMFTClusters.reserve(clusMFT.size());
218  o2::mft::ioutils::convertCompactClusters(clusMFT, pattIt, mMFTClusters, *mMFTDict);
219 
220  // Load MFT tracks
221  mMFTTracks = inp.getMFTTracks();
222  mMFTTrackROFRec = inp.getMFTTracksROFRecords();
223  if (mMCTruthON) {
224  mMFTTrkLabels = inp.getMFTTracksMCLabels();
225  }
226  int nROFs = mMFTTrackROFRec.size();
227 
228  LOG(info) << "Loaded " << mMFTTracks.size() << " MFT Tracks in " << nROFs << " ROFs";
229  if (mMFTTracks.empty()) {
230  return false;
231  }
232  mMFTWork.reserve(mMFTTracks.size());
233 
234  for (int irof = 0; irof < nROFs; irof++) {
235  const auto& rofRec = mMFTTrackROFRec[irof];
236 
237  int nBC = rofRec.getBCData().differenceInBC(mStartIR);
238  float tMin = nBC * o2::constants::lhc::LHCBunchSpacingMUS;
239  float tMax = (nBC + mMFTROFrameLengthInBC) * o2::constants::lhc::LHCBunchSpacingMUS;
240  if (!mMFTTriggered) {
241  auto irofCont = nBC / mMFTROFrameLengthInBC;
242  if (mMFTTrackROFContMapping.size() <= irofCont) { // there might be gaps in the non-empty rofs, this will map continuous ROFs index to non empty ones
243  mMFTTrackROFContMapping.resize((1 + irofCont / 128) * 128, 0);
244  }
245  mMFTTrackROFContMapping[irofCont] = irof;
246  }
247  mMFTROFTimes.emplace_back(tMin, tMax); // MFT ROF min/max time
248  LOG(debug) << "MFT ROF # " << irof << " [tMin;tMax] = [" << tMin << ";" << tMax << "]";
249 
250  int trlim = rofRec.getFirstEntry() + rofRec.getNEntries();
251  for (int it = rofRec.getFirstEntry(); it < trlim; it++) {
252  const auto& trcOrig = mMFTTracks[it];
253 
254  int nWorkTracks = mMFTWork.size();
255  // working copy of outer track param
256  auto& trc = mMFTWork.emplace_back(TrackLocMFT{trcOrig, {tMin, tMax}, irof});
257  trc.setParameters(trcOrig.getOutParam().getParameters());
258  trc.setZ(trcOrig.getOutParam().getZ());
259  trc.setCovariances(trcOrig.getOutParam().getCovariances());
260  trc.setTrackChi2(trcOrig.getOutParam().getTrackChi2());
261  trc.propagateToZ(mMatchingPlaneZ, mBz);
262  }
263  }
264 
265  return true;
266 }

301 template <Int_t saveAllMode>
302 void MatchGlobalFwd::doMatching()
303 {
304  // Range of compatible MCH ROFS for the first MFT track
305  int nMCHROFs = mMCHROFTimes.size();
306 
307  LOG(info) << "Running MCH-MFT Track Matching.";
308  // ROFrame of first MFT track
309  auto firstMFTTrackIdInROF = 0;
310  auto MFTROFId = mMFTWork.front().roFrame;
311  while ((firstMFTTrackIdInROF < mMFTTracks.size()) && (MFTROFId < mMFTTrackROFRec.size())) {
312  auto MFTROFId = mMFTWork[firstMFTTrackIdInROF].roFrame;
313  const auto& thisMFTBracket = mMFTROFTimes[MFTROFId];
314  auto nMFTTracksInROF = mMFTTrackROFRec[MFTROFId].getNEntries();
315  firstMFTTrackIdInROF = mMFTTrackROFRec[MFTROFId].getFirstEntry();
316  LOG(debug) << "MFT ROF = " << MFTROFId << "; interval: [" << thisMFTBracket.getMin() << "," << thisMFTBracket.getMax() << "]";
317  LOG(debug) << "ROF " << MFTROFId << " : firstMFTTrackIdInROF " << firstMFTTrackIdInROF << " ; nMFTTracksInROF = " << nMFTTracksInROF;
318  firstMFTTrackIdInROF += nMFTTracksInROF;
319  int mchROF = 0;
320  while (mchROF < nMCHROFs && (thisMFTBracket.isOutside(mMCHROFTimes[mchROF]))) {
321  mchROF++;
322  }
323  if (mchROF >= nMCHROFs) {
324  continue;
325  }
326  int mchROFMatchFirst = -1;
327  int mchROFMatchLast = -1;
328 
329  if (thisMFTBracket.isOutside(mMCHROFTimes[mchROF]) == 0) {
330  mchROFMatchFirst = mchROF;
331 
332  while (mchROF < nMCHROFs && !(thisMFTBracket < mMCHROFTimes[mchROF])) {
333  mchROF++;
334  }
335  if (mchROF >= nMCHROFs) {
336  continue;
337  }
338  mchROFMatchLast = mchROF - 1;
339  } else {
340  LOG(debug) << "No compatible MCH ROF with MFT ROF " << MFTROFId << std::endl;
341  }
342  if (mchROFMatchFirst >= 0) {
343  ROFMatch<saveAllMode>(MFTROFId, mchROFMatchFirst, mchROFMatchLast);
344  }
345  }
346 
347  if constexpr (saveAllMode == SaveMode::kBestMatch) { // Otherwise output container is filled by ROFMatch()
348  int nFakes = 0, nTrue = 0;
349  for (auto& thisMCHTrack : mMCHWork) {
350  auto bestMFTMatchID = thisMCHTrack.getMFTTrackID();
351  if (bestMFTMatchID >= 0) { // If there is a match, add to output container
352  if (mMCTruthON) {
353  mMatchLabels.push_back(computeLabel(thisMCHTrack.getMCHTrackID(), bestMFTMatchID));
354  mMatchLabels.back().isFake() ? nFakes++ : nTrue++;
355  }
356 
357  thisMCHTrack.setMFTTrackID(bestMFTMatchID);
358  thisMCHTrack.setTimeMUS(thisMCHTrack.tBracket.getMin(), thisMCHTrack.tBracket.delta());
359  LOG(debug) << " thisMCHTrack.getMFTTrackID() = " << thisMCHTrack.getMFTTrackID()
360  << "; thisMCHTrack.getMFTMCHMatchingChi2() = " << thisMCHTrack.getMFTMCHMatchingChi2();
361 
362  mMatchedTracks.emplace_back(thisMCHTrack);
363  mMatchingInfo.emplace_back(thisMCHTrack);
364  }
365  }
366  if (mMCTruthON) {
367  LOG(info) << " MFT-MCH Matching: nFakes = " << nFakes << " nTrue = " << nTrue;
368  }
369  }
370 }

373 template <Int_t saveAllMode>
374 void MatchGlobalFwd::ROFMatch(int MFTROFId, int firstMCHROFId, int lastMCHROFId)
375 {
377  const auto& thisMFTROF = mMFTTrackROFRec[MFTROFId];
378  const auto& firstMCHROF = mMCHTrackROFRec[firstMCHROFId];
379  const auto& lastMCHROF = mMCHTrackROFRec[lastMCHROFId];
380  int nFakes = 0, nTrue = 0;
381 
382  auto firstMFTTrackID = thisMFTROF.getFirstEntry();
383  auto lastMFTTrackID = firstMFTTrackID + thisMFTROF.getNEntries() - 1;
384 
385  auto firstMCHTrackID = firstMCHROF.getFirstIdx();
386  auto lastMCHTrackID = lastMCHROF.getLastIdx();
387  auto nMFTTracks = thisMFTROF.getNEntries();
388  auto nMCHTracks = lastMCHTrackID - firstMCHTrackID + 1;
389 
390  auto& matchAllChi2 = mMatchingFunctionMap["matchALL"];
391 
392  LOG(debug) << "Matching MFT ROF " << MFTROFId << " with MCH ROFs [" << firstMCHROFId << "->" << lastMCHROFId << "]";
393  LOG(debug) << " firstMFTTrackID = " << firstMFTTrackID << " ; lastMFTTrackID = " << lastMFTTrackID;
394  LOG(debug) << " firstMCHTrackID = " << firstMCHTrackID << " ; lastMCHTrackID = " << lastMCHTrackID << std::endl;
395 
396  // loop over all MCH tracks
397  for (auto MCHId = firstMCHTrackID; MCHId <= lastMCHTrackID; MCHId++) {
398  auto& thisMCHTrack = mMCHWork[MCHId];
399  o2::MCCompLabel matchLabel;
400  for (auto MFTId = firstMFTTrackID; MFTId <= lastMFTTrackID; MFTId++) {
401  auto& thisMFTTrack = mMFTWork[MFTId];
402  if (mMCTruthON) {
403  matchLabel = computeLabel(MCHId, MFTId);
404  }
405  if (mCutFunc(thisMCHTrack, thisMFTTrack)) {
406  thisMCHTrack.countMFTCandidate();
407  if (mMCTruthON) {
408  if (matchLabel.isCorrect()) {
409  thisMCHTrack.setCloseMatch();
410  }
411  }
412  auto score = mMatchFunc(thisMCHTrack, thisMFTTrack);
413  if (score < thisMCHTrack.getMFTMCHMatchingScore()) {
414  thisMCHTrack.setMFTTrackID(MFTId);
415  auto chi2 = matchAllChi2(thisMCHTrack, thisMFTTrack); // Matching chi2 is stored independently
416  thisMCHTrack.setMFTMCHMatchingScore(score);
417  thisMCHTrack.setMFTMCHMatchingChi2(chi2);
418  }
419  if constexpr (saveAllMode == SaveMode::kSaveAll) { // In saveAllmode save all pairs to output container
420  thisMCHTrack.setMFTTrackID(MFTId);
421  thisMCHTrack.setTimeMUS(thisMCHTrack.tBracket.getMin(), thisMCHTrack.tBracket.delta());
422  mMatchedTracks.emplace_back(thisMCHTrack);
423  mMatchingInfo.emplace_back(thisMCHTrack);
424  if (mMCTruthON) {
425  mMatchLabels.push_back(matchLabel);
426  mMatchLabels.back().isFake() ? nFakes++ : nTrue++;
427  }
428  }
429 
430  if constexpr (saveAllMode == SaveMode::kSaveTrainingData) { // In save training data mode store track parameters at matching plane
431  thisMCHTrack.setMFTTrackID(MFTId);
432  thisMCHTrack.setTimeMUS(thisMCHTrack.tBracket.getMin(), thisMCHTrack.tBracket.delta());
433  mMatchingInfo.emplace_back(thisMCHTrack);
434  mMCHMatchPlaneParams.emplace_back(thisMCHTrack);
435  mMFTMatchPlaneParams.emplace_back(static_cast<o2::mft::TrackMFT>(thisMFTTrack));
436 
437  if (mMCTruthON) {
438  mMatchLabels.push_back(matchLabel);
439  mMatchLabels.back().isFake() ? nFakes++ : nTrue++;
440  }
441  }
442  }
443  }
444  auto bestMFTMatchID = thisMCHTrack.getMFTTrackID();
445  LOG(debug) << " Matching MCHId = " << MCHId << " ==> bestMFTMatchID = " << thisMCHTrack.getMFTTrackID() << " ; thisMCHTrack.getMFTMCHMatchingChi2() = " << thisMCHTrack.getMFTMCHMatchingChi2();
446  LOG(debug) << " MCH COV<X,X> = " << thisMCHTrack.getSigma2X() << " ; COV<Y,Y> = " << thisMCHTrack.getSigma2Y() << " ; pt = " << thisMCHTrack.getPt();
447 
448  } // /loop over MCH tracks seeds
449 
450  LOG(debug) << "Finished matching MFT ROF " << MFTROFId << ": " << nMFTTracks << " MFT tracks and " << nMCHTracks << " MCH Tracks.";
451  if (mMCTruthON) {
452  LOG(debug) << " nFakes = " << nFakes << " nTrue = " << nTrue;
453  }
454 }

470 void MatchGlobalFwd::doMCMatching()
471 {
472  int nFakes = 0, nTrue = 0;
473 
474  // loop over all MCH tracks
475  for (auto MCHId = 0; MCHId < mMCHWork.size(); MCHId++) {
476  auto& thisMCHTrack = mMCHWork[MCHId];
477  const o2::MCCompLabel& thisMCHLabel = mMCHTrkLabels[MCHId];
478 
479  LOG(debug) << " MCH Track # " << MCHId << " Label: " << thisMCHLabel;
480  if (!((thisMCHLabel).isSet())) {
481  continue;
482  }
483  for (auto MFTId = 0; MFTId < mMFTWork.size(); MFTId++) {
484  auto& thisMFTTrack = mMFTWork[MFTId];
485  o2::MCCompLabel matchLabel = computeLabel(MCHId, MFTId);
486 
487  if (matchLabel.isCorrect()) {
488  nTrue++;
489  thisMCHTrack.setCloseMatch();
490  auto chi2 = mMatchFunc(thisMCHTrack, thisMFTTrack);
491  thisMCHTrack.setMFTTrackID(MFTId);
492  thisMCHTrack.setMFTMCHMatchingChi2(chi2);
493  thisMCHTrack.setTimeMUS(thisMCHTrack.tBracket.getMin(), thisMCHTrack.tBracket.delta());
494  mMatchedTracks.emplace_back(thisMCHTrack);
495  mMatchingInfo.emplace_back(thisMCHTrack);
496  mMatchLabels.push_back(matchLabel);
497  auto bestMFTMatchID = thisMCHTrack.getMFTTrackID();
498  LOG(debug) << " Matching MCHId = " << MCHId << " ==> bestMFTMatchID = " << thisMCHTrack.getMFTTrackID() << " ; thisMCHTrack.getMFTMCHMatchingChi2() = " << thisMCHTrack.getMFTMCHMatchingChi2();
499  LOG(debug) << " MCH COV<X,X> = " << thisMCHTrack.getSigma2X() << " ; COV<Y,Y> = " << thisMCHTrack.getSigma2Y() << " ; pt = " << thisMCHTrack.getPt();
500  LOG(debug) << " Label: " << matchLabel;
501  break;
502  }
503  }
504 
505  } // /loop over MCH tracks seeds
506 
507  auto nMFTTracks = mMFTWork.size();
508  auto nMCHTracks = mMCHWork.size();
509 
510  LOG(info) << " Done MC matching of " << nMFTTracks << " MFT tracks with " << nMCHTracks << " MCH Tracks. nFakes = " << nFakes << " nTrue = " << nTrue;
511 }

668 o2::dataformats::GlobalFwdTrack MatchGlobalFwd::MCHtoFwd(const o2::mch::TrackParam& mchParam)
669 {
670  // Convert a MCH Track parameters and covariances matrix to the
671  // Forward track format. Must be called after propagation though the absorber
672 
673  o2::dataformats::GlobalFwdTrack convertedTrack;
674 
675  // Parameter conversion
676  double alpha1, alpha3, alpha4, x2, x3, x4;
677 
678  alpha1 = mchParam.getNonBendingSlope();
679  alpha3 = mchParam.getBendingSlope();
680  alpha4 = mchParam.getInverseBendingMomentum();
681 
682  x2 = TMath::ATan2(-alpha3, -alpha1);
683  x3 = -1. / TMath::Sqrt(alpha3 * alpha3 + alpha1 * alpha1);
684  x4 = alpha4 * -x3 * TMath::Sqrt(1 + alpha3 * alpha3);
685 
686  auto K = alpha1 * alpha1 + alpha3 * alpha3;
687  auto K32 = K * TMath::Sqrt(K);
688  auto L = TMath::Sqrt(alpha3 * alpha3 + 1);
689 
690  // Covariances matrix conversion
691  SMatrix55Std jacobian;
692  SMatrix55Sym covariances;
693 
694  if (0) {
695 
696  std::cout << " MCHtoGlobal - MCH Covariances:\n";

700  << mchParam.getCovariances()(2, 2) << std::endl;
701  }
702  covariances(0, 0) = mchParam.getCovariances()(0, 0);
703  covariances(0, 1) = mchParam.getCovariances()(0, 1);
704  covariances(0, 2) = mchParam.getCovariances()(0, 2);
705  covariances(0, 3) = mchParam.getCovariances()(0, 3);
706  covariances(0, 4) = mchParam.getCovariances()(0, 4);
707 
708  covariances(1, 1) = mchParam.getCovariances()(1, 1);
709  covariances(1, 2) = mchParam.getCovariances()(1, 2);
710  covariances(1, 3) = mchParam.getCovariances()(1, 3);
711  covariances(1, 4) = mchParam.getCovariances()(1, 4);

713  covariances(2, 2) = mchParam.getCovariances()(2, 2);

717  covariances(3, 3) = mchParam.getCovariances()(3, 3);

723

725

731

733  jacobian(4, 3) = alpha3 * alpha4 * (1 / (TMath::Sqrt(K) * L) - L / K32);
734  jacobian(4, 4) = L / TMath::Sqrt(K);

741  convertedTrack.setY(mchParam.getBendingCoor());
742  convertedTrack.setZ(mchParam.getZ());
743  convertedTrack.setPhi(x2);
744  convertedTrack.setTanl(x3);
745  convertedTrack.setInvQPt(x4);
746  convertedTrack.setCharge(mchParam.getCharge());
747  convertedTrack.setCovariances(covariances);
748 
749  return convertedTrack;
