45 arrow::Status getSliceFor(
46  int value,
47  char const* key,
48  std::shared_ptr<arrow::Table> const& input,
49  std::shared_ptr<arrow::Table>& output,
50  uint64_t& offset)
51 {
52  std::shared_ptr<arrow::NumericArray<arrow::Int32Type>> values;
53  std::shared_ptr<arrow::NumericArray<arrow::Int64Type>> counts;
54 
55  auto status = getSlices(key, input, values, counts);
56 
57  for (auto slice = 0; slice < values->length(); ++slice) {
58  if (values->Value(slice) == value) {
59  output = input->Slice(offset, counts->Value(slice));
60  return arrow::Status::OK();
61  }
62  offset += counts->Value(slice);
63  }
64  output = input->Slice(offset, 0);
65  return arrow::Status::OK();
66 }

68 void sliceByColumnGeneric(
69  char const* key,
70  char const* target,
71  std::shared_ptr<arrow::Table> const& input,
72  int32_t fullSize,
73  ListVector* groups,
74  ListVector* unassigned)
75 {
76  groups->resize(fullSize);
77  auto column = input->GetColumnByName(key);
78  int32_t row = 0;
79  for (auto iChunk = 0; iChunk < column->num_chunks(); ++iChunk) {
80  auto chunk = static_cast<arrow::NumericArray<arrow::Int32Type>>(column->chunk(iChunk)->data());
81  for (auto iElement = 0; iElement < chunk.length(); ++iElement) {
82  auto v = chunk.Value(iElement);
83  if (v >= 0) {
84  if (v >= groups->size()) {
85  throw runtime_error_f("Table %s has an entry with index (%d) that is larger than the grouping table size (%d)", target, v, fullSize);
86  }
87  (*groups)[v].push_back(row);
88  } else if (unassigned != nullptr) {
89  auto av = std::abs(v);
90  if (unassigned->size() < av + 1) {
91  unassigned->resize(av + 1);
92  }
93  (*unassigned)[av].push_back(row);
94  }
95  ++row;
96  }
97  }
98 }

100 arrow::Status sliceByColumn(
101  char const* key,
102  char const* target,
103  std::shared_ptr<arrow::Table> const& input,
104  int32_t fullSize,
105  std::vector<arrow::Datum>* slices,
106  std::vector<uint64_t>* offsets,
107  std::vector<int>* sizes,
108  std::vector<arrow::Datum>* unassignedSlices,
109  std::vector<uint64_t>* unassignedOffsets)
110 {
111  arrow::Datum value_counts;
112  auto column = input->GetColumnByName(key);
113  auto array0 = static_cast<arrow::NumericArray<arrow::Int32Type>>(column->chunk(0)->data());
114  int32_t prev = 0;
115  int32_t cur = array0.Value(0);
116  int32_t lastNeg = cur < 0 ? cur : 0;
117  int32_t lastPos = cur < 0 ? -1 : cur;
118  for (auto i = 0; i < column->num_chunks(); ++i) {
119  auto array = static_cast<arrow::NumericArray<arrow::Int32Type>>(column->chunk(i)->data());
120  for (auto e = 0; e < array.length(); ++e) {
121  prev = cur;
122  if (prev >= 0) {
123  lastPos = prev;
124  } else {
125  lastNeg = prev;
126  }
127  cur = array.Value(e);
128  if (cur >= 0) {
129  if (lastPos > cur) {
130  throw runtime_error_f("Table %s index %s is not sorted: next value %d < previous value %d!", target, key, cur, lastPos);
131  }
132  if (lastPos == cur && prev < 0) {
133  throw runtime_error_f("Table %s index %s has a group with index %d that is split by %d", target, key, cur, prev);
134  }
135  } else {
136  if (lastNeg < cur) {
137  throw runtime_error_f("Table %s index %s is not sorted: next negative value %d > previous negative value %d!", target, key, cur, lastNeg);
138  }
139  if (lastNeg == cur && prev >= 0) {
140  throw runtime_error_f("Table %s index %s has a group with index %d that is split by %d", target, key, cur, prev);
141  }
142  }
143  }
144  }
145  std::shared_ptr<arrow::NumericArray<arrow::Int32Type>> values;
146  std::shared_ptr<arrow::NumericArray<arrow::Int64Type>> counts;
147 
148  auto status = getSlices(key, input, values, counts);
149 
150  // create slices and offsets
151  uint64_t offset = 0;
152  int64_t count = 0;
153  auto size = values->length();
154 
155  auto makeSlice = [&](uint64_t offset_, int64_t count_) {
156  slices->emplace_back(arrow::Datum{input->Slice(offset_, count_)});
157  if (offsets) {
158  offsets->emplace_back(offset_);
159  }
160  if (sizes) {
161  sizes->emplace_back(count_);
162  }
163  };
164 
165  auto makeUnassignedSlice = [&](uint64_t offset_, int64_t count_) {
166  if (unassignedSlices) {
167  unassignedSlices->emplace_back(arrow::Datum{input->Slice(offset_, count_)});
168  }
169  if (unassignedOffsets) {
170  unassignedOffsets->emplace_back(offset_);
171  }
172  };
173 
174  auto v = 0;
175  auto vprev = v;
176  auto nzeros = 0;
177 
178  for (auto i = 0; i < size; ++i) {
179  count = counts->Value(i);
180  if (v >= 0) {
181  vprev = v;
182  }
183  v = values->Value(i);
184  if (v < 0) {
185  makeUnassignedSlice(offset, count);
186  offset += count;
187  continue;
188  }
189  nzeros = v - vprev - ((i == 0 || slices->empty() == true) ? 0 : 1);
190  for (auto z = 0; z < nzeros; ++z) {
191  makeSlice(offset, 0);
192  }
193  makeSlice(offset, count);
194  offset += count;
195  }
196  v = values->Value(size - 1);
197  if (v >= 0) {
198  vprev = v;
199  }
200  if (vprev < fullSize - 1) {
201  for (auto v = vprev + 1; v < fullSize; ++v) {
202  makeSlice(offset, 0);
203  }
204  }
205 
206  return arrow::Status::OK();
207 }
