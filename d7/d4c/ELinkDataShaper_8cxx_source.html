70 EventType ELinkDataShaper::processSelfTriggered(uint16_t localClock, InteractionRecord& ir)
71 {
73 
74  // This is a self-triggered events.
75  // The physics data arrives with a delay compared to the BC,
76  // which is due to the travel time of muons up to the MID chambers
77  // plus the travel time of the signal to the readout electronics.
78  // In the case of regional cards, a further delay is expected
79  // since this card needs to wait for the tracklet decision of each local card.
80  // For simplicity, this delay is added to the BCToLocal in the constructor.
81  // In both cases, we need to correct for the delay in order to go back to the real BC.
82  if (ir.bc < mElectronicsDelay.BCToLocal) {
83  // If the bc is smaller than the delay, it means that the local clock was reset
84  // This events therefore belongs to the previous orbit.
85  // We therefore add the value of the last BC (+1 to account for the reset)
86  // and we decrease the orbit by 1.
87  ir.bc += mLastClock + 1;
88  --ir.orbit;
89  }
90  // We can now safely subtract the delay, which, thanks to the above protection,
91  // will not result in a negative number
92  ir.bc -= mElectronicsDelay.BCToLocal;
93  if (mReceivedCalibration && (localClock == mExpectedFETClock)) {
94  // Reset the calibration flag for this e-link
95  mReceivedCalibration = false;
96  return EventType::FET;
97  }
98  return EventType::Standard;
99 }

109 void ELinkDataShaper::processOrbitTrigger(uint16_t localClock, uint8_t triggerWord)
110 {
112 
113  // The local clock is reset: we are now in synch with the new HB
114  // We have two ways to account for the orbit change:
115  // - increase the orbit counter by 1 for this e-link
116  // (CAVEAT: synch is lost if we lose some orbit)
117  // - set the orbit to the one found in RDH
118  // (CAVEAT: synch is lost if we have lot of data, spanning over two orbits)
119  // FIXME: pick one of the two
120  // ++mIR.orbit; // orbit increase
121  mIR.orbit = mRDHOrbit; // reset to RDH
122  if ((triggerWord & raw::sSOX) == 0) {
123  mLastClock = localClock;
124  }
125  // The orbit trigger resets the clock.
126  // If we received a calibration trigger, we need to change the value of the expected clock accordingly
127  if (mReceivedCalibration) {
128  mExpectedFETClock -= (localClock + 1);
129  }
130 }

132 bool ELinkDataShaper::processTrigger(const ELinkDecoder& decoder, EventType& eventType, InteractionRecord& ir)
133 {
137  auto localClock = decoder.getCounter();
138 
139  // FIXME: So far the bc information is not used
140  // since it seems that it is always equal to 0
141  // (at least in the RDH of the first page, which is what we need).
142  // In this case the local clock and the BC coincide.
143  // If this is not the case, the bc of the orbit trigger should be correctly taken into account
144  ir.bc = localClock;
145  // ir.bc = mIR.bc + localClock;
146  ir.orbit = mIR.orbit;
147 
148  if (decoder.getTriggerWord() == 0) {
149  // This is a self-triggered event
150  eventType = processSelfTriggered(localClock, ir);
151  return true;
152  }
153 
154  // From here we treat triggered events
155  bool goOn = false;
156  eventType = EventType::Standard;
157  if (decoder.getTriggerWord() & raw::sCALIBRATE) {
158  // This is an answer to a calibration trigger
159  eventType = processCalibrationTrigger(localClock);
160  goOn = true;
161  }
162 
163  if (decoder.getTriggerWord() & raw::sORB) {
164  // This is the answer to an orbit trigger
165  processOrbitTrigger(localClock, decoder.getTriggerWord());
166  }
167 
168  return goOn;
169 }

171 void ELinkDataShaper::addLoc(const ELinkDecoder& decoder, EventType eventType, InteractionRecord ir, std::vector<ROBoard>& data, std::vector<ROFRecord>& rofs)
172 {
174  auto firstEntry = data.size();
175  data.push_back({decoder.getStatusWord(), decoder.getTriggerWord(), mUniqueId, decoder.getInputs()});
176  rofs.emplace_back(ir, eventType, firstEntry, 1);
177  for (int ich = 0; ich < 4; ++ich) {
178  if ((data.back().firedChambers & (1 << ich))) {
179  data.back().patternsBP[ich] = decoder.getPattern(0, ich);
180  data.back().patternsNBP[ich] = decoder.getPattern(1, ich);
181  }
182  }
183 }
