171 void getChildData(int infd, DeviceInfo& outinfo)
172 {
173  char buffer[1024 * 16];
174  int bytes_read;
175  // NOTE: do not quite understand read ends up blocking if I read more than
176  // once. Oh well... Good enough for now.
177  O2_SIGNPOST_START(DriverStatus::ID, DriverStatus::BYTES_READ, outinfo.pid, infd, 0);
178  while (true) {
179  bytes_read = read(infd, buffer, 1024 * 16);
180  if (bytes_read == 0) {
181  O2_SIGNPOST_END(DriverStatus::ID, DriverStatus::BYTES_READ, bytes_read, 0, 0);
182  return;
183  }
184  if (bytes_read < 0) {
185  switch (errno) {
186  case EWOULDBLOCK:
187  O2_SIGNPOST_END(DriverStatus::ID, DriverStatus::BYTES_READ, bytes_read, 0, 0);
188  return;
189  default:
190  O2_SIGNPOST_END(DriverStatus::ID, DriverStatus::BYTES_READ, bytes_read, 0, 0);
191  return;
192  }
193  }
194  assert(bytes_read > 0);
195  outinfo.unprinted.append(buffer, bytes_read);
196  }
197 }

240 int calculateExitCode(DriverInfo& driverInfo, DeviceSpecs& deviceSpecs, DeviceInfos& infos)
241 {
242  std::regex regexp(R"(^\[([\d+:]*)\]\[\w+\] )");
243  if (!driverInfo.lastError.empty()) {
244  LOGP(error, "SEVERE: DPL driver encountered an error while running.\n{}",
245  driverInfo.lastError);
246  return 1;
247  }
248  for (size_t di = 0; di < deviceSpecs.size(); ++di) {
249  auto& info = infos[di];
250  auto& spec = deviceSpecs[di];
251  if (info.maxLogLevel >= driverInfo.minFailureLevel) {
252  LOGP(error, "SEVERE: Device {} ({}) had at least one message above severity {}: {}",
253  spec.name,
254  info.pid,
255  info.minFailureLevel,
256  std::regex_replace(info.firstSevereError, regexp, ""));
257  return 1;
258  }
259  if (info.exitStatus != 0) {
260  LOGP(error, "SEVERE: Device {} ({}) returned with {}",
261  spec.name,
262  info.pid,
263  info.exitStatus);
264  return info.exitStatus;
265  }
266  }
267  return 0;
268 }

271 void createPipes(int* pipes)
272 {
273  auto p = pipe(pipes);
274 
275  if (p == -1) {
276  std::cerr << "Unable to create PIPE: ";
277  switch (errno) {
278  case EFAULT:
279  assert(false && "EFAULT while reading from pipe");
280  break;

283  break;
284  case ENFILE:
285  std::cerr << "System file table is full";
286  break;
287  default:
288  std::cerr << "Unknown PIPE" << std::endl;
289  };
290  // Kill immediately both the parent and all the children
291  kill(-1 * getpid(), SIGKILL);
292  }

304 {
305  if (graceful_exit == false) {
306  graceful_exit = true;
307  } else {
308  forceful_exit = true;

314  // did not gracefully exited.
315  double_sigint = true;

687  unsigned parentNode)
688 {
689  // FIXME: this might not work when more than one DPL driver on the same
690  // machine. Hopefully we do not care.
691  // Not how the first port is actually used to broadcast clients.
692  auto& spec = specs[ref.index];
693  auto& control = controls[ref.index];
694  auto& execution = executions[ref.index];
695 
696  driverInfo.tracyPort++;
697 
698  for (auto& service : spec.services) {
699  if (service.preFork != nullptr) {
700  service.preFork(serviceRegistry, varmap);
701  }
702  }
703  // If we have a framework id, it means we have already been respawned
704  // and that we are in a child. If not, we need to fork and re-exec, adding
705  // the framework-id as one of the options.
706  pid_t id = 0;
707  id = fork();
708  // We are the child: prepare options and reexec.
709  if (id == 0) {
710  // We allow being debugged and do not terminate on SIGTRAP
711  signal(SIGTRAP, SIG_IGN);
712 
713  // This is the child.
714  // For stdout / stderr, we close the read part of the pipe, the
715  // old descriptor, and then replace it with the write part of the pipe.
716  // For stdin, we close the write part of the pipe, the old descriptor,
717  // and then we replace it with the read part of the pipe.
718  // We also close all the filedescriptors for our sibilings.
719  struct rlimit rlim;
720  getrlimit(RLIMIT_NOFILE, &rlim);
721  // We close all FD, but the one which are actually
722  // used to communicate with the driver.
723  for (size_t i = 0; i < rlim.rlim_cur; ++i) {
724  if (childFds[ref.index].childstdin[0] == i) {
725  continue;
726  }
727  if (childFds[ref.index].childstdout[1] == i) {
728  continue;
729  }
730  close(i);
731  }
732  dup2(childFds[ref.index].childstdin[0], STDIN_FILENO);
733  dup2(childFds[ref.index].childstdout[1], STDOUT_FILENO);
734  dup2(childFds[ref.index].childstdout[1], STDERR_FILENO);
735 
736  auto portS = std::to_string(driverInfo.tracyPort);
737  setenv("TRACY_PORT", portS.c_str(), 1);
738  for (auto& service : spec.services) {
739  if (service.postForkChild != nullptr) {
740  service.postForkChild(serviceRegistry);
741  }
742  }
743  for (auto& env : execution.environ) {
744  char* formatted = strdup(fmt::format(env,
745  fmt::arg("timeslice0", spec.inputTimesliceId),
746  fmt::arg("timeslice1", spec.inputTimesliceId + 1),
747  fmt::arg("timeslice4", spec.inputTimesliceId + 4))
748  .c_str());
749  putenv(formatted);
750  }
751  execvp(execution.args[0], execution.args.data());
752  }
753  close(childFds[ref.index].childstdin[0]);
754  close(childFds[ref.index].childstdout[1]);
755  if (varmap.count("post-fork-command")) {
756  auto templateCmd = varmap["post-fork-command"];
757  auto cmd = fmt::format(templateCmd.as<std::string>(),
758  fmt::arg("pid", id),
759  fmt::arg("id", spec.id),
760  fmt::arg("cpu", parentCPU),
761  fmt::arg("node", parentNode),
762  fmt::arg("name", spec.name),
763  fmt::arg("timeslice0", spec.inputTimesliceId),
764  fmt::arg("timeslice1", spec.inputTimesliceId + 1),
765  fmt::arg("rank0", spec.rank),
766  fmt::arg("maxRank0", spec.nSlots));
767  int err = system(cmd.c_str());
768  if (err) {
769  LOG(error) << "Post fork command `" << cmd << "` returned with status " << err;
770  }
771  LOG(debug) << "Successfully executed `" << cmd;
772  }
773  // This is the parent. We close the write end of
774  // the child pipe and and keep track of the fd so
775  // that we can later select on it.
776  for (auto& service : spec.services) {
777  if (service.postForkParent != nullptr) {
778  service.postForkParent(serviceRegistry);
779  }
780  }
781 
782  LOG(info) << "Starting " << spec.id << " on pid " << id;
783  DeviceInfo info;
784  info.pid = id;
785  info.active = true;
786  info.readyToQuit = false;
787  info.historySize = 1000;
788  info.historyPos = 0;
789  info.maxLogLevel = LogParsingHelpers::LogLevel::Debug;
790  info.minFailureLevel = driverInfo.minFailureLevel;
791  info.dataRelayerViewIndex = Metric2DViewIndex{"data_relayer", 0, 0, {}};
792  info.variablesViewIndex = Metric2DViewIndex{"matcher_variables", 0, 0, {}};
793  info.queriesViewIndex = Metric2DViewIndex{"data_queries", 0, 0, {}};
794  info.outputsViewIndex = Metric2DViewIndex{"output_matchers", 0, 0, {}};
795  info.tracyPort = driverInfo.tracyPort;
796  info.lastSignal = uv_hrtime() - 10000000;
797 
798  deviceInfos.emplace_back(info);
799  // Let's add also metrics information for the given device
800  gDeviceMetricsInfos.emplace_back(DeviceMetricsInfo{});
801 }

811 LogProcessingState processChildrenOutput(DriverInfo& driverInfo,
812  DeviceInfos& infos,
813  DeviceSpecs const& specs,
814  DeviceControls& controls,
815  std::vector<DeviceMetricsInfo>& metricsInfos)
816 {
817  // Display part. All you need to display should actually be in
818  // `infos`.
819  // TODO: split at \n
820  // TODO: update this only once per 1/60 of a second or
821  // things like this.
822  // TODO: have multiple display modes
823  // TODO: graphical view of the processing?
824  assert(infos.size() == controls.size());
825  std::smatch match;
826  ParsedMetricMatch metricMatch;
827  ParsedConfigMatch configMatch;
828  const std::string delimiter("\n");
829  bool hasNewMetric = false;
830  LogProcessingState result;
831 
832  for (size_t di = 0, de = infos.size(); di < de; ++di) {
833  DeviceInfo& info = infos[di];
834  DeviceControl& control = controls[di];
835  DeviceMetricsInfo& metrics = metricsInfos[di];
836  assert(specs.size() == infos.size());
837  DeviceSpec const& spec = specs[di];
838 
839  if (info.unprinted.empty()) {
840  continue;
841  }
842 
843  O2_SIGNPOST_START(DriverStatus::ID, DriverStatus::BYTES_PROCESSED, info.pid, 0, 0);
844 
845  std::string_view s = info.unprinted;
846  size_t pos = 0;
847  info.history.resize(info.historySize);
848  info.historyLevel.resize(info.historySize);
849 
850  auto updateMetricsViews =
851  Metric2DViewIndex::getUpdater({&info.dataRelayerViewIndex,
852  &info.variablesViewIndex,
853  &info.queriesViewIndex,
854  &info.outputsViewIndex});
855 
856  auto newMetricCallback = [&updateMetricsViews, &driverInfo, &metricsInfos, &hasNewMetric](std::string const& name, MetricInfo const& metric, int value, size_t metricIndex) {
857  updateMetricsViews(name, metric, value, metricIndex);
858  hasNewMetric = true;
859  };
860 
861  while ((pos = s.find(delimiter)) != std::string::npos) {
862  std::string token{s.substr(0, pos)};
863  auto logLevel = LogParsingHelpers::parseTokenLevel(token);
864 
865  // Check if the token is a metric from SimpleMetricsService
866  // if yes, we do not print it out and simply store it to be displayed
867  // in the GUI.
868  // Then we check if it is part of our Poor man control system
869  // if yes, we execute the associated command.
870  if (DeviceMetricsHelper::parseMetric(token, metricMatch)) {
871  // We use this callback to cache which metrics are needed to provide a
872  // the DataRelayer view.
873  DeviceMetricsHelper::processMetric(metricMatch, metrics, newMetricCallback);
874  result.didProcessMetric = true;
875  } else if (logLevel == LogParsingHelpers::LogLevel::Info && ControlServiceHelpers::parseControl(token, match)) {
876  ControlServiceHelpers::processCommand(infos, info.pid, match[1].str(), match[2].str());
877  result.didProcessControl = true;
878  } else if (logLevel == LogParsingHelpers::LogLevel::Info && DeviceConfigHelper::parseConfig(token, configMatch)) {
879  DeviceConfigHelper::processConfig(configMatch, info);
880  result.didProcessConfig = true;
881  } else if (!control.quiet && (token.find(control.logFilter) != std::string::npos) &&
882  logLevel >= control.logLevel) {
883  assert(info.historyPos >= 0);
884  assert(info.historyPos < info.history.size());
885  info.history[info.historyPos] = token;
886  info.historyLevel[info.historyPos] = logLevel;
887  info.historyPos = (info.historyPos + 1) % info.history.size();
888  fmt::print("[{}:{}]: {}\n", info.pid, spec.id, token);
889  result.didProcessLog = true;
890  }
891  // We keep track of the maximum log error a
892  // device has seen.
893  bool maxLogLevelIncreased = false;
894  if (logLevel > info.maxLogLevel && logLevel > LogParsingHelpers::LogLevel::Info &&
895  logLevel != LogParsingHelpers::LogLevel::Unknown) {
896  info.maxLogLevel = logLevel;
897  maxLogLevelIncreased = true;
898  }
899  if (logLevel >= driverInfo.minFailureLevel) {
900  info.lastError = token;
901  if (info.firstSevereError.empty() || maxLogLevelIncreased) {
902  info.firstSevereError = token;
903  }
904  }
905  s.remove_prefix(pos + delimiter.length());
906  }
907  size_t oldSize = info.unprinted.size();
908  info.unprinted = std::string(s);
909  O2_SIGNPOST_END(DriverStatus::ID, DriverStatus::BYTES_PROCESSED, oldSize - info.unprinted.size(), 0, 0);
910  }
911  result.hasNewMetric = hasNewMetric;
912  if (hasNewMetric) {
913  hasNewMetric = false;
914  DeviceMetricsHelper::updateMetricsNames(driverInfo, metricsInfos);
915  }
916  return result;
917 }
