32 struct Triggers {
33  enum {
34  bitMinBias,
35  bitMinBiasInner, // experimental
36  bitMinBiasOuter, // experimental
37  bitHighMult,
38  bitDummy // non-defined yet, placeholder
39  };
40  uint8_t triggerSignals = 0; // V0 trigger signals
41  int8_t nChanA = 0; // number of fired channels [A side]
42  int32_t amplA = -1000; // sum amplitude [A side]
43  Triggers() = default;
44  Triggers(uint8_t signals, int8_t chanA, int32_t amplASum)
45  {
46  triggerSignals = signals;
47  nChanA = chanA;
48  amplA = amplASum;
49  }
50 
51  bool getIsMinBias() const { return (triggerSignals & (1 << bitMinBias)) != 0; }
52  bool getIsMinBiasInner() const { return (triggerSignals & (1 << bitMinBiasInner)) != 0; }
53  bool getIsMinBiasOuter() const { return (triggerSignals & (1 << bitMinBiasOuter)) != 0; }
54  bool getIsHighMult() const { return (triggerSignals & (1 << bitHighMult)) != 0; }
55  bool getIsDummy() const { return (triggerSignals & (1 << bitDummy)) != 0; }
56  void setTriggers(Bool_t isMinBias, Bool_t isMinBiasInner, Bool_t isMinBiasOuter, Bool_t isHighMult, Bool_t isDummy, int8_t chanA, int32_t amplASum)
57  {
58  triggerSignals = (isMinBias << bitMinBias) | (isMinBiasInner << bitMinBiasInner) | (isMinBiasOuter << bitMinBiasOuter) | (isHighMult << bitHighMult) | (isDummy << bitDummy);
59  nChanA = chanA;
60  amplA = amplASum;
61  }
62  bool operator==(Triggers const& other) const
63  {
64  //Will be implemented later
65  //return std::tie(triggersignals, nChanA, nChanC, amplA, amplC, timeA, timeC) ==
66  // std::tie(other.triggersignals, other.nChanA, other.nChanC, other.amplA, other.amplC, other.timeA, other.timeC);
67  return std::tie(triggerSignals, nChanA, amplA) ==
68  std::tie(other.triggerSignals, other.nChanA, other.amplA);
69  }
70  void printLog() const;
71  ClassDefNV(Triggers, 1);
72 };

100  Triggers mTriggers;
101  BCData() = default;
102  BCData(int first, int ne, o2::InteractionRecord iRec, const Triggers& chTrig)
103  {
104  ref.setFirstEntry(first);
105  ref.setEntries(ne);
106  ir = iRec;
107  mTriggers = chTrig;
108  }
109  typedef DetTrigInput DetTrigInput_t;
110  gsl::span<const ChannelData> getBunchChannelData(const gsl::span<const ChannelData> tfdata) const;
111  const o2::InteractionRecord& getIntRecord() const { return ir; };
112  Triggers getTriggers() const { return mTriggers; }
113  void setIntRecord(const o2::InteractionRecord& intRec) { ir = intRec; }
114  void setTriggers(Triggers triggers) { mTriggers = triggers; };
115  void print() const;
116  bool operator==(const BCData& other) const
117  {
118  return std::tie(ref, mTriggers, ir) == std::tie(other.ref, other.mTriggers, other.ir);
119  }
120  void printLog() const;
121  DetTrigInput makeTrgInput() const { return DetTrigInput{ir, mTriggers.getIsMinBias(), mTriggers.getIsMinBiasInner(), mTriggers.getIsMinBiasOuter(), mTriggers.getIsHighMult(), mTriggers.getIsDummy()}; }
122  void fillTrgInputVec(std::vector<DetTrigInput>& vecTrgInput) const
123  {
124  vecTrgInput.emplace_back(ir, mTriggers.getIsMinBias(), mTriggers.getIsMinBiasInner(), mTriggers.getIsMinBiasOuter(), mTriggers.getIsHighMult(), mTriggers.getIsDummy());
125  }
