53 Bool_t GeneratorPythia6::Init()
54 {
58  Generator::Init();
59 
61  if (!mConfig.empty()) {
62  std::ifstream fin(mConfig.c_str());
63  if (!fin.is_open()) {
64  LOG(fatal) << "cannot open configuration file: " << mConfig;
65  return kFALSE;
66  }
68  std::string whitespace = " \t\f\v\n\r";
69  std::string comment = "#";
70  for (std::string line; getline(fin, line);) {
72  line = line.substr(0, line.find_first_of(comment));
73  if (line.size() == 0) {
74  continue;
75  }
77  const auto line_begin = line.find_first_not_of(whitespace);
78  const auto line_end = line.find_last_not_of(whitespace);
79  if (line_begin == std::string::npos ||
80  line_end == std::string::npos) {
81  continue;
82  }
83  const auto line_range = line_end - line_begin + 1;
84  line = line.substr(line_begin, line_range);
85  if (line.size() == 0) {
86  continue;
87  }
89  readString(line.c_str());
90  }
91  fin.close();
92  }
93 
95  TPythia6::Instance()->Pytune(mTune);
96 
98  TPythia6::Instance()->Initialize(mFrame.c_str(), mBeam.c_str(), mTarget.c_str(), mWin);
99 
101  return kTRUE;
102 }
