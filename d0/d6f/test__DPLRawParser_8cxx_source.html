37  // we create message for the 3 input routes, the messages have different page size

49 BOOST_AUTO_TEST_CASE(test_DPLRawParser)
50 {
51  auto dataset = createData();
52  InputRecord& inputs = dataset.record;
53  BOOST_REQUIRE(dataset.messages.size() > 0);
54  BOOST_REQUIRE(dataset.messages[0].at(0) != nullptr);
55  BOOST_REQUIRE(inputs.size() > 0);
56  BOOST_CHECK((*inputs.begin()).header == dataset.messages[0].at(0)->data());
57  DPLRawParser parser(inputs);
58  int count = 0;
59  o2::header::DataHeader const* last = nullptr;
60  for (auto it = parser.begin(), end = parser.end(); it != end; ++it, ++count) {
61  LOG(info) << "data " << count << " " << *((int*)it.data());
62  // now check the iterator API
63  // retrieving RDH
64  auto const* rdh = it.get_if<test::RAWDataHeader>();
65  // retrieving the raw pointer of the page
66  auto const* raw = it.raw();
67  // retrieving payload pointer of the page
68  auto const* payload = it.data();
69  // size of payload
70  size_t payloadSize = it.size();
71  // offset of payload in the raw page
72  size_t offset = it.offset();
73  BOOST_REQUIRE(rdh != nullptr);
74  BOOST_REQUIRE(offset == sizeof(test::RAWDataHeader));
75  BOOST_REQUIRE(payload == raw + offset);
76  BOOST_REQUIRE(*reinterpret_cast<int const*>(payload) == dataset.values[count]);
77  BOOST_REQUIRE(payloadSize == PAGESIZE - sizeof(test::RAWDataHeader));
78  auto const* dh = it.o2DataHeader();
79  if (last != dh) {
80  // this is a special wrapper to print the RDU info and table header, this will
81  // be extended
82  std::cout << DPLRawParser::RDHInfo(it) << std::endl;
83  last = dh;
84  }
85  std::cout << it << " payload size " << it.size() << std::endl;
86  }
87 
88  // test the parser with filter on data specs, this will filter out the first input
89  // route with 5 raw pages in the payload, so we start checking at count 5
90  DPLRawParser filteredparser(inputs, o2::framework::select("its:ITS/RAWDATA"));
91  count = 5;
92  for (auto it = filteredparser.begin(), end = filteredparser.end(); it != end; ++it, ++count) {
93  LOG(info) << "data " << count << " " << *((int*)it.data());
94  BOOST_REQUIRE(*reinterpret_cast<int const*>(it.data()) == dataset.values[count]);
95  }
96 
97  // test with filter not matching any input route
98  DPLRawParser nomatchingparser(inputs, o2::framework::select("nmatch:NO/MATCH"));

100  for (auto it = nomatchingparser.begin(), end = nomatchingparser.end(); it != end; ++it, ++count) {
101  LOG(info) << "data " << count << " " << *((int*)it.data());
102  }
103  BOOST_CHECK(count == 0);
104 }
