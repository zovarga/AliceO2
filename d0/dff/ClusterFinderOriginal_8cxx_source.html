125 void ClusterFinderOriginal::findClusters(gsl::span<const Digit> digits)
126 {
129 
130  // skip preclusters with only 1 digit
131  if (digits.size() < 2) {
132  return;
133  }
134 
135  // set the Mathieson function to be used
136  mMathieson = (digits[0].getDetID() < 300) ? &mMathiesons[0] : &mMathiesons[1];
137 
138  // reset the current precluster being processed
139  resetPreCluster(digits);
140 
141  // try to simplify the precluster by removing pads if possible (sent back to preclustering)
142  std::vector<int> removedDigits{};
143  simplifyPreCluster(removedDigits);
144 
145  if (mPreCluster->multiplicity() > 1) {
146 
147  // extract clusters from the precluster
148  int iNewCluster = mClusters.size();
149  processPreCluster();
150 
151  if (mClusters.size() > iNewCluster) {
152 
153  // copy the digits associated to the new clusters (if any) in the list of used digits
154  int iFirstNewDigit = mUsedDigits.size();
155  for (const auto& pad : *mPreCluster) {
156  if (pad.isReal()) {
157  mUsedDigits.emplace_back(digits[pad.digitIndex()]);
158  }
159  }
160  int nNewDigits = mUsedDigits.size() - iFirstNewDigit;
161 
162  // give the new clusters a unique ID, make them point to these digits then set their resolution
163  for (; iNewCluster < mClusters.size(); ++iNewCluster) {
164  mClusters[iNewCluster].uid = Cluster::buildUniqueId(digits[0].getDetID() / 100 - 1, digits[0].getDetID(), iNewCluster);
165  mClusters[iNewCluster].firstDigit = iFirstNewDigit;
166  mClusters[iNewCluster].nDigits = nNewDigits;
167  setClusterResolution(mClusters[iNewCluster]);
168  }
169  }
170  }
171 
172  if (!removedDigits.empty()) {
173 
174  // load the released digits (if any) in the preclusterizer
175  mPreClusterFinder.reset();
176  for (auto iDigit : removedDigits) {
177  mPreClusterFinder.loadDigit(digits[iDigit]);
178  }
179 
180  // preclusterize and get the new preclusters and associated digits
181  std::vector<PreCluster> preClusters{};
182  std::vector<Digit> usedDigits{};
183  int nPreClusters = mPreClusterFinder.run();
184  preClusters.reserve(nPreClusters);
185  usedDigits.reserve(removedDigits.size());
186  mPreClusterFinder.getPreClusters(preClusters, usedDigits);
187 
188  // clusterize every new preclusters
189  for (const auto& preCluster : preClusters) {
190  findClusters({&usedDigits[preCluster.firstDigit], preCluster.nDigits});
191  }
192  }
193 }

196 void ClusterFinderOriginal::resetPreCluster(gsl::span<const Digit>& digits)
197 {
199 
200  mPreCluster->clear();
201 
202  mSegmentation = &mapping::segmentation(digits[0].getDetID());
203 
204  for (int iDigit = 0; iDigit < digits.size(); ++iDigit) {
205 
206  const auto& digit = digits[iDigit];
207  int padID = digit.getPadID();
208 
209  double x = mSegmentation->padPositionX(padID);
210  double y = mSegmentation->padPositionY(padID);
211  double dx = mSegmentation->padSizeX(padID) / 2.;
212  double dy = mSegmentation->padSizeY(padID) / 2.;
213  double charge = mADCToCharge(digit.getADC());
214  bool isSaturated = digit.isSaturated();
215  int plane = mSegmentation->isBendingPad(padID) ? 0 : 1;
216 
217  if (charge <= 0.) {
218  throw std::runtime_error("The precluster contains a digit with charge <= 0");
219  }
220 
221  mPreCluster->addPad(x, y, dx, dy, charge, isSaturated, plane, iDigit, PadOriginal::kZero);
222  }

226 void ClusterFinderOriginal::simplifyPreCluster(std::vector<int>& removedDigits)
227 {
232 
233  // discard small clusters (leftovers from splitting or noise)
234  if (mPreCluster->multiplicity() < 3 && mPreCluster->charge() < mLowestClusterCharge) {
235  mPreCluster->clear();
236  return;
237  }
238 
239  // the following is only for two-cathode preclusters
240  if (mPreCluster->multiplicity(0) == 0 || mPreCluster->multiplicity(1) == 0) {
241  return;
242  }
243 
244  // tag every pad that overlap with another on the other plane
245  std::vector<bool> overlap(mPreCluster->multiplicity(), false);
246  for (int i = 0; i < mPreCluster->multiplicity(); ++i) {
247  const auto& padi = mPreCluster->pad(i);
248  for (int j = i + 1; j < mPreCluster->multiplicity(); ++j) {
249  const auto& padj = mPreCluster->pad(j);
250  if (padi.plane() == padj.plane() || (overlap[i] && overlap[j])) {
251  continue;
252  }
253  if (areOverlapping(padi, padj, SDistancePrecision)) {
254  overlap[i] = true;
255  overlap[j] = true;
256  }
257  }
258  }
259 
260  // count the number of pads that do not overlap
261  int nNotOverlapping(0);
262  for (int i = 0; i < mPreCluster->multiplicity(); ++i) {
263  if (!overlap[i]) {
264  ++nNotOverlapping;
265  }
266  }
267 
268  // discard pads with no overlap (unless it is at the edge or there is only one with low charge)
269  // loop over pads in decreasing index order to do not shift the indices while removing
270  if (nNotOverlapping > 0) {
271  const mapping::CathodeSegmentation* cathSeg[2] = {&mSegmentation->bending(), &mSegmentation->nonBending()};
272  for (int i = mPreCluster->multiplicity() - 1; i >= 0; --i) {
273  if (overlap[i]) {
274  continue;
275  }
276  const auto& pad = mPreCluster->pad(i);
277  if (nNotOverlapping == 1 && pad.charge() < mLowestPadCharge) {
278  break; // there is only one
279  }
280  int cathPadIdx = cathSeg[1 - pad.plane()]->findPadByPosition(pad.x(), pad.y());
281  if (!cathSeg[1 - pad.plane()]->isValid(cathPadIdx)) {
282  continue;
283  }
284  removedDigits.push_back(pad.digitIndex());
285  mPreCluster->removePad(i);
286  }
287  }
288 
289  // now adresses the case of large charge asymmetry between the two planes
290  if (!mPreCluster->isSaturated() && mPreCluster->chargeAsymmetry() > 0.5) {
291 
292  // get the pads with minimum and maximum charge on the plane with maximum integrated charge
293  int plane = mPreCluster->maxChargePlane();
294  double chargeMin(std::numeric_limits<double>::max()), chargeMax(-1.);
295  int iPadMin(-1), iPadMax(-1);
296  for (int i = 0; i < mPreCluster->multiplicity(); ++i) {
297  const auto& pad = mPreCluster->pad(i);
298  if (pad.plane() == plane) {
299  if (pad.charge() < chargeMin) {
300  chargeMin = pad.charge();
301  iPadMin = i;
302  }
303  if (pad.charge() > chargeMax) {
304  chargeMax = pad.charge();
305  iPadMax = i;
306  }
307  }
308  }
309  if (iPadMin < 0 || iPadMax < 0) {
310  throw std::runtime_error("This plane should contain at least 1 pad!?");
311  }
312 
313  // distance of the pad with minimum charge to the pad with maximum charge
314  const auto& padMin = mPreCluster->pad(iPadMin);
315  const auto& padMax = mPreCluster->pad(iPadMax);
316  double dxOfMin = (padMin.x() - padMax.x()) / padMax.dx() / 2.;
317  double dyOfMin = (padMin.y() - padMax.y()) / padMax.dy() / 2.;
318  double distOfMin = TMath::Sqrt(dxOfMin * dxOfMin + dyOfMin * dyOfMin);
319 
320  // arrange pads of this plane according to their distance to the pad with maximum charge normalized to its size
321  double precision = SDistancePrecision / 2. * TMath ::Sqrt((1. / padMax.dx() / padMax.dx() + 1. / padMax.dy() / padMax.dy()) / 2.);
322  auto cmp = [precision](double dist1, double dist2) { return dist1 < dist2 - precision; };
323  std::multimap<double, int, decltype(cmp)> distIndices(cmp);
324  for (int i = 0; i < mPreCluster->multiplicity(); ++i) {
325  if (i == iPadMax) {
326  distIndices.emplace(0., i);
327  } else {
328  const auto& pad = mPreCluster->pad(i);
329  if (pad.plane() == plane) {
330  double dx = (pad.x() - padMax.x()) / padMax.dx() / 2.;
331  double dy = (pad.y() - padMax.y()) / padMax.dy() / 2.;
332  distIndices.emplace(TMath::Sqrt(dx * dx + dy * dy), i);
333  }
334  }
335  }
336 
337  // try to extract from this plane the cluster centered around the pad with maximum charge
338  double previousDist(std::numeric_limits<float>::max());
339  double previousChargeMax(-1.);
340  std::set<int, std::greater<>> padsToRemove{};
341  for (const auto& distIndex : distIndices) {
342  const auto& pad = mPreCluster->pad(distIndex.second);
343 
344  // try not to overstep the pad with minimum charge
345  if (distIndex.first > distOfMin + precision) {
346  double ddx = (pad.x() - padMax.x()) / padMax.dx() / 2. * dxOfMin;
347  double ddy = (pad.y() - padMax.y()) / padMax.dy() / 2. * dyOfMin;
348  if ((ddx > -precision && ddy > -precision) ||
349  (TMath::Abs(ddx) > TMath::Abs(ddy) + precision && ddx > -precision) ||
350  (TMath::Abs(ddy) > TMath::Abs(ddx) + precision && ddy > -precision)) {
351  continue;
352  }
353  }
354 
355  // stop if their is a gap between this pad and the last one tagged for removal
356  if (distIndex.first > previousDist + 1. + precision) {
357  break;
358  }
359 
360  // update the maximum charge if we reach another ring
361  if (TMath::Abs(distIndex.first - previousDist) >= precision) {
362  previousChargeMax = chargeMax;
363  }
364 
365  // tag all pads at this distance to be removed if the maximum charge decreases
366  if (pad.charge() <= previousChargeMax) {
367  if (TMath::Abs(distIndex.first - previousDist) < precision) {
368  chargeMax = TMath::Max(pad.charge(), chargeMax);
369  } else {
370  chargeMax = pad.charge();
371  }
372  previousDist = distIndex.first;
373  padsToRemove.insert(distIndex.second);
374  }
375  }
376 
377  // remove the tagged pads (in decreasing index order to do not invalidate them while removing)
378  for (auto iPad : padsToRemove) {
379  removedDigits.push_back(mPreCluster->pad(iPad).digitIndex());
380  mPreCluster->removePad(iPad);
381  }
382  }
383 }

385 //_________________________________________________________________________________________________
386 void ClusterFinderOriginal::processPreCluster()
387 {
389 
390  buildPixArray();
391  if (mPixels.empty()) {
392  return;
393  }
394 
395  // switch between different clustering methods depending on the complexity of the precluster
396  std::pair<int, int> nPadsXY = mPreCluster->sizeInPads(PadOriginal::kZero);
397  if (nPadsXY.first < 4 && nPadsXY.second < 4) {
398 
399  // simple precluster
400  processSimple();
401 
402  } else {
403 
404  // find the local maxima in the pixel array
405  std::unique_ptr<TH2D> histAnode(nullptr);
406  std::multimap<double, std::pair<int, int>, std::greater<>> localMaxima{};
407  findLocalMaxima(histAnode, localMaxima);
408  if (localMaxima.empty()) {
409  return;
410  }
411 
412  if (localMaxima.size() == 1 || mPreCluster->multiplicity() <= 50) {
413 
414  // precluster of reasonable size --> treat it in one piece
415  process();
416 
417  } else {
418 
419  // too large precluster --> split it and treat every pieces separately
420  for (const auto& localMaximum : localMaxima) {
421 
422  // select the part of the precluster that is around the local maximum
423  restrictPreCluster(*histAnode, localMaximum.second.first, localMaximum.second.second);
424 
425  // treat it
426  process();
427  }
428  }
429  }
430 }

432 //_________________________________________________________________________________________________
433 void ClusterFinderOriginal::buildPixArray()
434 {
436 
437  mPixels.clear();
438 
439  // pixel half size is the minimum pad half size on both plane
440  std::pair<double, double> dim = mPreCluster->minPadDimensions(-1, false);
441  double width[2] = {dim.first, dim.second};
442 
443  // to make sure the grid is aligned with the smallest pad in both direction
444  double xy0[2] = {0., 0.};
445  bool found[2] = {false, false};
446  for (const auto& pad : *mPreCluster) {
447  for (int ixy = 0; ixy < 2; ++ixy) {
448  if (!found[ixy] && TMath::Abs(pad.dxy(ixy) - width[ixy]) < SDistancePrecision) {
449  xy0[ixy] = pad.xy(ixy);
450  found[ixy] = true;
451  }
452  }
453  if (found[0] && found[1]) {
454  break;
455  }
456  }
457 
458  // to deal with mono-cathod clusters
459  int plane0 = 0, plane1 = 1;
460  if (mPreCluster->multiplicity(0) == 0) {
461  plane0 = 1;
462  } else if (mPreCluster->multiplicity(1) == 0) {
463  plane1 = 0;
464  }
465 
466  // grid size is the intersect of the cluster areas on both planes
467  double area[2][2] = {0.};
468  mPreCluster->area(plane0, area);
469  if (plane1 != plane0) {
470  double area2[2][2] = {0.};
471  mPreCluster->area(plane1, area2);
472  area[0][0] = TMath::Max(area[0][0], area2[0][0]);
473  area[0][1] = TMath::Min(area[0][1], area2[0][1]);
474  area[1][0] = TMath::Max(area[1][0], area2[1][0]);
475  area[1][1] = TMath::Min(area[1][1], area2[1][1]);
476  }
477 
478  // abort if the areas do not intersect
479  if (area[0][1] - area[0][0] < SDistancePrecision || area[1][1] - area[1][0] < SDistancePrecision) {
480  return;
481  }
482 
483  // adjust limits
484  int nbins[2] = {0, 0};
485  for (int ixy = 0; ixy < 2; ++ixy) {
486  double precision = SDistancePrecision / width[ixy] / 2.;
487  double dist = (area[ixy][0] - xy0[ixy]) / width[ixy] / 2.;
488  area[ixy][0] = xy0[ixy] + (TMath::Nint(dist + precision) - 0.5) * width[ixy] * 2.;
489  nbins[ixy] = TMath::Ceil((area[ixy][1] - area[ixy][0]) / width[ixy] / 2. - precision);
490  area[ixy][1] = area[ixy][0] + nbins[ixy] * width[ixy] * 2.;
491  }
492 
493  // book pixel histograms and fill them
494  TH2D hCharges("Charges", "", nbins[0], area[0][0], area[0][1], nbins[1], area[1][0], area[1][1]);
495  TH2I hEntries("Entries", "", nbins[0], area[0][0], area[0][1], nbins[1], area[1][0], area[1][1]);
496  for (const auto& pad : *mPreCluster) {
497  ProjectPadOverPixels(pad, hCharges, hEntries);
498  }
499 
500  // store fired pixels with an entry from both planes if both planes are fired
501  for (int i = 1; i <= nbins[0]; ++i) {
502  double x = hCharges.GetXaxis()->GetBinCenter(i);
503  for (int j = 1; j <= nbins[1]; ++j) {
504  int entries = hEntries.GetBinContent(i, j);
505  if (entries == 0 || (plane0 != plane1 && (entries < 1000 || entries % 1000 < 1))) {
506  continue;
507  }
508  double y = hCharges.GetYaxis()->GetBinCenter(j);
509  double charge = hCharges.GetBinContent(i, j);
510  mPixels.emplace_back(x, y, width[0], width[1], charge);
511  }
512  }
513 
514  // split the pixel into 2 if there is only one
515  if (mPixels.size() == 1) {
516  auto& pixel = mPixels.front();
517  pixel.setdx(width[0] / 2.);
518  pixel.setx(pixel.x() - width[0] / 2.);
519  mPixels.emplace_back(pixel.x() + width[0], pixel.y(), width[0] / 2., width[1], pixel.charge());
520  }
521 
522  // sort and remove pixels with the lowest signal if there are too many
523  size_t nPads = mPreCluster->multiplicity();
524  if (mPixels.size() > nPads) {
525  std::stable_sort(mPixels.begin(), mPixels.end(), [](const PadOriginal& pixel1, const PadOriginal& pixel2) {
526  return pixel1.charge() > pixel2.charge();
527  });
528  mPixels.erase(mPixels.begin() + nPads, mPixels.end());
529  }
530 }

533 void ClusterFinderOriginal::ProjectPadOverPixels(const PadOriginal& pad, TH2D& hCharges, TH2I& hEntries) const
534 {
536 
537  const TAxis* xaxis = hCharges.GetXaxis();
538  const TAxis* yaxis = hCharges.GetYaxis();
539 
540  int iMin = TMath::Max(1, xaxis->FindBin(pad.x() - pad.dx() + SDistancePrecision));
541  int iMax = TMath::Min(hCharges.GetNbinsX(), xaxis->FindBin(pad.x() + pad.dx() - SDistancePrecision));
542  int jMin = TMath::Max(1, yaxis->FindBin(pad.y() - pad.dy() + SDistancePrecision));
543  int jMax = TMath::Min(hCharges.GetNbinsY(), yaxis->FindBin(pad.y() + pad.dy() - SDistancePrecision));
544 
545  double charge = pad.charge();
546  int entry = 1 + pad.plane() * 999;
547 
548  for (int i = iMin; i <= iMax; ++i) {
549  for (int j = jMin; j <= jMax; ++j) {
550  int entries = hEntries.GetBinContent(i, j);
551  hCharges.SetBinContent(i, j, (entries > 0) ? TMath::Min(hCharges.GetBinContent(i, j), charge) : charge);
552  hEntries.SetBinContent(i, j, entries + entry);
553  }
554  }
555 }

557 //_________________________________________________________________________________________________
558 void ClusterFinderOriginal::findLocalMaxima(std::unique_ptr<TH2D>& histAnode,
559  std::multimap<double, std::pair<int, int>, std::greater<>>& localMaxima)
560 {
564 
565  // create a 2D histogram from the pixel array
566  double xMin(std::numeric_limits<double>::max()), xMax(-std::numeric_limits<double>::max());
567  double yMin(std::numeric_limits<double>::max()), yMax(-std::numeric_limits<double>::max());
568  double dx(mPixels.front().dx()), dy(mPixels.front().dy());
569  for (const auto& pixel : mPixels) {
570  xMin = TMath::Min(xMin, pixel.x());
571  xMax = TMath::Max(xMax, pixel.x());
572  yMin = TMath::Min(yMin, pixel.y());
573  yMax = TMath::Max(yMax, pixel.y());
574  }
575  int nBinsX = TMath::Nint((xMax - xMin) / dx / 2.) + 1;
576  int nBinsY = TMath::Nint((yMax - yMin) / dy / 2.) + 1;
577  histAnode = std::make_unique<TH2D>("anode", "anode", nBinsX, xMin - dx, xMax + dx, nBinsY, yMin - dy, yMax + dy);
578  for (const auto& pixel : mPixels) {
579  histAnode->Fill(pixel.x(), pixel.y(), pixel.charge());
580  }
581 
582  // find the local maxima
583  std::vector<std::vector<int>> isLocalMax(nBinsX, std::vector<int>(nBinsY, 0));
584  for (int j = 1; j <= nBinsY; ++j) {
585  for (int i = 1; i <= nBinsX; ++i) {
586  if (isLocalMax[i - 1][j - 1] == 0 && histAnode->GetBinContent(i, j) >= mLowestPixelCharge) {
587  flagLocalMaxima(*histAnode, i, j, isLocalMax);
588  }
589  }
590  }
591 
592  // store local maxima and tag corresponding pixels
593  TAxis* xAxis = histAnode->GetXaxis();
594  TAxis* yAxis = histAnode->GetYaxis();
595  for (int j = 1; j <= nBinsY; ++j) {
596  for (int i = 1; i <= nBinsX; ++i) {
597  if (isLocalMax[i - 1][j - 1] > 0) {

697 void ClusterFinderOriginal::processSimple()
698 {
702 
703  // add virtual pads if necessary
704  addVirtualPad();
705 
706  // calculate pad-pixel coupling coefficients and pixel visibilities
707  std::vector<double> coef(0);
708  std::vector<double> prob(0);
709  computeCoefficients(coef, prob);
710 
711  // discard "invisible" pixels
712  for (int ipix = 0; ipix < mPixels.size(); ++ipix) {
713  if (prob[ipix] < 0.01) {
714  mPixels[ipix].setCharge(0);
715  }
716  }
717 
718  // run the MLEM algorithm with 15 iterations
719  double qTot = mlem(coef, prob, 15);
720 
721  // abort if the total charge of the pixels is too low
722  if (qTot < 1.e-4 || (mPreCluster->multiplicity() < 3 && qTot < mLowestClusterCharge)) {
723  return;
724  }
725 
726  // fit all pads but the ones saturated
727  for (auto& pad : *mPreCluster) {
728  if (!pad.isSaturated()) {
729  pad.setStatus(PadOriginal::kUseForFit);
730  }
731  }
732 
733  // use all the pixels as a single cluster seed
734  std::vector<int> pixels(mPixels.size());
735  std::iota(pixels.begin(), pixels.end(), 0);
736 
737  // set the fit range based on the limits of the pixel array
738  double fitRange[2][2] = {{std::numeric_limits<double>::max(), -std::numeric_limits<double>::max()},
739  {std::numeric_limits<double>::max(), -std::numeric_limits<double>::max()}};
740  for (const auto& pixel : mPixels) {
741  fitRange[0][0] = TMath::Min(fitRange[0][0], pixel.x() - 3. * pixel.dx());
742  fitRange[0][1] = TMath::Max(fitRange[0][1], pixel.x() + 3. * pixel.dx());
743  fitRange[1][0] = TMath::Min(fitRange[1][0], pixel.y() - 3. * pixel.dy());
744  fitRange[1][1] = TMath::Max(fitRange[1][1], pixel.y() + 3. * pixel.dy());
745  }
746 
747  // do the fit
748  double fitParam[SNFitParamMax + 1] = {0.};
749  fit({&pixels}, fitRange, fitParam);

752 //_________________________________________________________________________________________________
753 void ClusterFinderOriginal::process()
754 {
759 
760  // add virtual pads if necessary
761  addVirtualPad();
762 
763  // number of pads to be considered in the precluster
764  size_t npadOK(0);
765  for (const auto& pad : *mPreCluster) {
766  if (pad.status() == PadOriginal::kZero) {
767  ++npadOK;
768  }
769  }
770 
771  // compute the limits of the histogram based on the current pixel array
772  double xMin(std::numeric_limits<double>::max()), xMax(-std::numeric_limits<double>::max());
773  double yMin(std::numeric_limits<double>::max()), yMax(-std::numeric_limits<double>::max());
774  for (const auto& pixel : mPixels) {
775  xMin = TMath::Min(xMin, pixel.x());
776  xMax = TMath::Max(xMax, pixel.x());
777  yMin = TMath::Min(yMin, pixel.y());
778  yMax = TMath::Max(yMax, pixel.y());
779  }
780 
781  std::vector<double> coef(0);
782  std::vector<double> prob(0);
783  std::unique_ptr<TH2D> histMLEM(nullptr);
784  while (true) {
785 
786  // calculate pad-pixel coupling coefficients and pixel visibilities
787  computeCoefficients(coef, prob);
788 
789  // discard "invisible" pixels
790  for (int ipix = 0; ipix < mPixels.size(); ++ipix) {
791  if (prob[ipix] < 0.01) {
792  mPixels[ipix].setCharge(0);
793  }
794  }
795 
796  // run the MLEM algorithm with 15 iterations
797  double qTot = mlem(coef, prob, 15);
798 
799  // abort if the total charge of the pixels is too low
800  if (qTot < 1.e-4 || (npadOK < 3 && qTot < mLowestClusterCharge)) {
801  return;
802  }
803 
804  // create a 2D histogram from the pixel array
805  double dx(mPixels.front().dx()), dy(mPixels.front().dy());
806  int nBinsX = TMath::Nint((xMax - xMin) / dx / 2.) + 1;
807  int nBinsY = TMath::Nint((yMax - yMin) / dy / 2.) + 1;
808  histMLEM.reset(nullptr); // delete first to avoid "Replacing existing TH1: mlem (Potential memory leak)"
809  histMLEM = std::make_unique<TH2D>("mlem", "mlem", nBinsX, xMin - dx, xMax + dx, nBinsY, yMin - dy, yMax + dy);
810  for (const auto& pixel : mPixels) {
811  histMLEM->Fill(pixel.x(), pixel.y(), pixel.charge());
812  }
813 
814  // stop here if the pixel size is small enough
815  if ((dx < 0.07 || dy < 0.07) && dy < dx) {
816  break;
817  }
818 
819  // calculate the position of the center-of-gravity around the pixel with maximum charge
820  double xyCOG[2] = {0., 0.};
821  findCOG(*histMLEM, xyCOG);
822 
823  // decrease the pixel size and align the array with the position of the center-of-gravity
824  refinePixelArray(xyCOG, npadOK, xMin, xMax, yMin, yMax);
825  }
826 
827  // discard pixels with low visibility by moving their charge to their nearest neighbour (cuts are empirical !!!)
828  double threshold = TMath::Min(TMath::Max(histMLEM->GetMaximum() / 100., 2.0 * mLowestPixelCharge), 100.0 * mLowestPixelCharge);
829  cleanPixelArray(threshold, prob);
830 
831  // re-run the MLEM algorithm with 2 iterations
832  double qTot = mlem(coef, prob, 2);
833 
834  // abort if the total charge of the pixels is too low
835  if (qTot < 2. * mLowestPixelCharge) {
836  return;
837  }
838 
839  // update the histogram
840  for (const auto& pixel : mPixels) {
841  histMLEM->SetBinContent(histMLEM->GetXaxis()->FindBin(pixel.x()), histMLEM->GetYaxis()->FindBin(pixel.y()), pixel.charge());
842  }
843 
844  // split the precluster into clusters
845  split(*histMLEM, coef);
846 }

848 //_________________________________________________________________________________________________
849 void ClusterFinderOriginal::addVirtualPad()
850 {
853 
854  const mapping::CathodeSegmentation* cathSeg[2] = {&mSegmentation->bending(), &mSegmentation->nonBending()};
855 
856  // decide what plane to consider for x and y directions
857  int iPlaneXY[2] = {1, 0}; // 0 = bending plane and 1 = non-bending plane as defined in resetPreCluster(...)
858 
859  // check if virtual pads are needed in each direction
860  // if the minimum pad size in y is the same on both planes, check also the other plane
861  std::pair<double, double> dim0 = mPreCluster->minPadDimensions(0, PadOriginal::kZero, true);
862  std::pair<double, double> dim1 = mPreCluster->minPadDimensions(1, PadOriginal::kZero, true);
863  bool sameSizeY = TMath::Abs(dim0.second - dim1.second) < SDistancePrecision;
864  bool addVirtualPad[2] = {false, false};
865  std::pair<int, int> nPadsXY0 = mPreCluster->sizeInPads(iPlaneXY[0], PadOriginal::kZero);
866  std::pair<int, int> nPadsXY1 = mPreCluster->sizeInPads(iPlaneXY[1], PadOriginal::kZero);
867  if (nPadsXY0.first == 2 && (!sameSizeY || nPadsXY1.first <= 2)) {
868  addVirtualPad[0] = true;
869  }
870  if (nPadsXY1.second == 2 && (!sameSizeY || nPadsXY0.second <= 2)) {
871  addVirtualPad[1] = true;
872  }
873 
874  double chargeReduction[2] = {100., 15.};
875  for (int ixy = 0; ixy < 2; ++ixy) {
876 
877  // no need to add virtual pads in this direction
878  if (!addVirtualPad[ixy]) {
879  continue;
880  }
881 
882  // find pads with maximum and next-to-maximum charges on the plane of interest
883  // find min and max dimensions of the precluster in this direction on that plane
884  long iPadMax[2] = {-1, -1};
885  double chargeMax[2] = {0., 0.};
886  double limits[2] = {std::numeric_limits<double>::max(), -std::numeric_limits<double>::max()};
887  for (int i = 0; i < mPreCluster->multiplicity(); ++i) {
888  const auto& pad = mPreCluster->pad(i);
889  if (pad.plane() != iPlaneXY[ixy]) {
890  continue;
891  }
892  if (pad.charge() > chargeMax[0]) {
893  chargeMax[1] = chargeMax[0];
894  chargeMax[0] = pad.charge();
895  iPadMax[1] = iPadMax[0];
896  iPadMax[0] = i;
897  } else if (pad.charge() > chargeMax[1]) {
898  chargeMax[1] = pad.charge();
899  iPadMax[1] = i;
900  }
901  double xy = pad.xy(ixy);
902  if (xy < limits[0]) {
903  limits[0] = xy;
904  }
905  if (xy > limits[1]) {
906  limits[1] = xy;
907  }
908  }
909 
910  // try to add a virtual pad next to the max pad then, if done, next to the next-to-max pads
911  // do not try to add a second virtual pad if the next-to-max charge is too low
912  int n = (chargeMax[1] / chargeMax[0] < 0.5) ? 1 : 2;
913  int sideDone(0);
914  for (int i = 0; i < n; ++i) {
915 
916  if (iPadMax[i] < 0) {
917  throw std::runtime_error("This plane should contain at least 2 pads!?");
918  }
919 
920  // check if the pad is at the cluster limit and on which side to add a virtual pad
921  const auto& pad = mPreCluster->pad(iPadMax[i]);
922  int side(0);
923  double xy = pad.xy(ixy);
924  if (TMath::Abs(xy - limits[0]) < SDistancePrecision) {
925  side = -1;
926  } else if (TMath::Abs(xy - limits[1]) < SDistancePrecision) {
927  side = 1;
928  } else {
929  break;
930  }
931 
932  // do not add 2 virtual pads on the same side
933  if (side == sideDone) {
934  break;
935  }
936 
937  // find the pad to add in the mapping
938  double pos[2] = {pad.x(), pad.y()};
939  pos[ixy] += side * (pad.dxy(ixy) + SDistancePrecision);
940  pos[1 - ixy] += SDistancePrecision; // pickup always the same in case we fall at the border between 2 pads
941  int cathPadIdx = cathSeg[iPlaneXY[ixy]]->findPadByPosition(pos[0], pos[1]);
942  if (!cathSeg[iPlaneXY[ixy]]->isValid(cathPadIdx)) {
943  break;
944  }
945 
946  // add the virtual pad
947  double charge = TMath::Max(TMath::Min(chargeMax[i] / chargeReduction[ixy], mLowestPadCharge), 2. * mLowestPixelCharge);
948  mPreCluster->addPad(cathSeg[iPlaneXY[ixy]]->padPositionX(cathPadIdx), cathSeg[iPlaneXY[ixy]]->padPositionY(cathPadIdx),
949  cathSeg[iPlaneXY[ixy]]->padSizeX(cathPadIdx) / 2., cathSeg[iPlaneXY[ixy]]->padSizeY(cathPadIdx) / 2.,
950  charge, false, iPlaneXY[ixy], -1, pad.status());
951 
952  sideDone = side;
953  }
954  }
955 }

987 //_________________________________________________________________________________________________
988 double ClusterFinderOriginal::mlem(const std::vector<double>& coef, const std::vector<double>& prob, int nIter)
989 {
992 
993  double qTot(0.);
994  double maxProb = *std::max_element(prob.begin(), prob.end());
995  std::vector<double> padSum(mPreCluster->multiplicity(), 0.);
996 
997  for (int iter = 0; iter < nIter; ++iter) {
998 
999  // calculate expectations, ignoring the pads that must not be considered
1000  int iCoef(0);
1001  for (int iPad = 0; iPad < mPreCluster->multiplicity(); ++iPad) {
1002  const auto& pad = mPreCluster->pad(iPad);
1003  if (pad.status() != PadOriginal::kZero) {
1004  iCoef += mPixels.size();
1005  continue;
1006  }
1007  padSum[iPad] = 0.;
1008  for (const auto& pixel : mPixels) {
1009  padSum[iPad] += pixel.charge() * coef[iCoef++];
1010  }
1011  }
1012 
1013  qTot = 0.;
1014  for (int iPix = 0; iPix < mPixels.size(); ++iPix) {
1015 
1016  // skip "invisible" pixel
1017  if (prob[iPix] < 0.01) {
1018  mPixels[iPix].setCharge(0.);
1019  continue;
1020  }
1021 
1022  double pixelSum(0.);
1023  double pixelNorm(maxProb);
1024  for (int iPad = 0; iPad < mPreCluster->multiplicity(); ++iPad) {
1025 
1026  // ignore the pads that must not be considered
1027  const auto& pad = mPreCluster->pad(iPad);
1028  if (pad.status() != PadOriginal::kZero) {
1029  continue;
1030  }
1031 
1032  // correct for pad charge overflows
1033  int iCoef = iPad * mPixels.size() + iPix;
1034  if (pad.isSaturated() && padSum[iPad] > pad.charge()) {
1035  pixelNorm -= coef[iCoef];
1036  continue;
1037  }
1038 
1039  if (padSum[iPad] > 1.e-6) {
1040  pixelSum += pad.charge() * coef[iCoef] / padSum[iPad];
1041  }
1042  }
1043 
1044  // correct the pixel charge
1045  if (pixelNorm > 1.e-6) {
1046  mPixels[iPix].setCharge(mPixels[iPix].charge() * pixelSum / pixelNorm);
1047  qTot += mPixels[iPix].charge();
1048  } else {
1049  mPixels[iPix].setCharge(0.);
1050  }
1051  }
1052 
1053  // can happen in clusters with large number of overflows - speeding up
1054  if (qTot < 1.e-6) {
1055  return qTot;
1056  }
1057  }
1058 
1059  return qTot;
1060 }

1063 void ClusterFinderOriginal::findCOG(const TH2D& histMLEM, double xy[2]) const
1064 {
1066 
1067  // define the range of pixels and the minimum charge to consider
1068  int ix0(0), iy0(0), iz0(0);
1069  double chargeThreshold = histMLEM.GetBinContent(histMLEM.GetMaximumBin(ix0, iy0, iz0)) / 10.;
1070  int ixMin = TMath::Max(1, ix0 - 1);
1071  int ixMax = TMath::Min(histMLEM.GetNbinsX(), ix0 + 1);
1072  int iyMin = TMath::Max(1, iy0 - 1);
1073  int iyMax = TMath::Min(histMLEM.GetNbinsY(), iy0 + 1);
1074

1077  const TAxis* yAxis = histMLEM.GetYaxis();
1078  double xq(0.), yq(0.), q(0.);
1079  bool onePixelWidthX(true), onePixelWidthY(true);
1080  for (int iy = iyMin; iy <= iyMax; ++iy) {
1081  for (int ix = ixMin; ix <= ixMax; ++ix) {
1082  double charge = histMLEM.GetBinContent(ix, iy);
1083  if (charge >= chargeThreshold) {
1084  xq += xAxis->GetBinCenter(ix) * charge;
1085  yq += yAxis->GetBinCenter(iy) * charge;
1086  q += charge;
1087  if (ix != ix0) {
1088  onePixelWidthX = false;
1089  }
1090  if (iy != iy0) {
1091  onePixelWidthY = false;
1092  }
1093  }
1094  }
1095  }
1096 
1097  // if all pixels used so far are aligned with iy0, add one more, with the highest charge, in y direction
1098  if (onePixelWidthY) {
1099  double xPixel(0.), yPixel(0.), chargePixel(0.);
1100  int ixPixel(ix0);
1101  for (int iy = iyMin; iy <= iyMax; ++iy) {
1102  if (iy != iy0) {
1103  for (int ix = ixMin; ix <= ixMax; ++ix) {
1104  double charge = histMLEM.GetBinContent(ix, iy);
1105  if (charge > chargePixel) {
1106  xPixel = xAxis->GetBinCenter(ix);
1107  yPixel = yAxis->GetBinCenter(iy);
1108  chargePixel = charge;
1109  ixPixel = ix;
1110  }
1111  }
1112  }
1113  }
1114  xq += xPixel * chargePixel;
1115  yq += yPixel * chargePixel;
1116  q += chargePixel;
1117  if (ixPixel != ix0) {
1118  onePixelWidthX = false;
1119  }
1120  }
1121 
1122  // if all pixels used so far are aligned with ix0, add one more, with the highest charge, in x direction
1123  if (onePixelWidthX) {
1124  double xPixel(0.), yPixel(0.), chargePixel(0.);
1125  for (int ix = ixMin; ix <= ixMax; ++ix) {
1126  if (ix != ix0) {
1127  for (int iy = iyMin; iy <= iyMax; ++iy) {
1128  double charge = histMLEM.GetBinContent(ix, iy);
1129  if (charge > chargePixel) {
1130  xPixel = xAxis->GetBinCenter(ix);
1131  yPixel = yAxis->GetBinCenter(iy);
1132  chargePixel = charge;
1133  }
1134  }
1135  }
1136  }
1137  xq += xPixel * chargePixel;
1138  yq += yPixel * chargePixel;
1139  q += chargePixel;
1140  }
1141 
1142  // compute the position of the centrer-of-gravity
1143  xy[0] = xq / q;
1144  xy[1] = yq / q;
1145 }

1147 //_________________________________________________________________________________________________
1148 void ClusterFinderOriginal::refinePixelArray(const double xyCOG[2], size_t nPixMax, double& xMin, double& xMax, double& yMin, double& yMax)
1149 {
1155 
1156  xMin = std::numeric_limits<double>::max();
1157  xMax = -std::numeric_limits<double>::max();
1158  yMin = std::numeric_limits<double>::max();
1159  yMax = -std::numeric_limits<double>::max();
1160 
1161  // sort pixels according to the charge and move all pixels that must be kept at the beginning
1162  std::stable_sort(mPixels.begin(), mPixels.end(), [](const PadOriginal& pixel1, const PadOriginal& pixel2) {
1163  return (pixel1.status() == PadOriginal::kMustKeep && pixel2.status() != PadOriginal::kMustKeep) ||
1164  (pixel1.status() == pixel2.status() && pixel1.charge() > pixel2.charge());
1165  });
1166  double pixMinCharge = TMath::Min(0.01 * mPixels.front().charge(), 100. * mLowestPixelCharge);
1167 
1168  // define the half-size and shift of the new pixels depending on the direction of splitting
1169  double width[2] = {mPixels.front().dx(), mPixels.front().dy()};
1170  double shift[2] = {0., 0.};
1171  if (width[0] > width[1]) {
1172  width[0] /= 2.;
1173  shift[0] = -width[0];
1174  } else {
1175  width[1] /= 2.;
1176  shift[1] = -width[1];
1177  }
1178

1458 double ClusterFinderOriginal::fit(double currentParam[SNFitParamMax + 2],
1459  const double parmin[SNFitParamMax], const double parmax[SNFitParamMax],
1460  int nParamUsed, int& nTrials) const
1461 {
1464 
1465  // default step size in x, y and charge fraction
1466  static const double defaultShift[SNFitParamMax] = {0.01, 0.002, 0.02, 0.01, 0.002, 0.02, 0.01, 0.002};
1467 
1468  double shift[SNFitParamMax] = {0.};
1469  for (int i = 0; i < nParamUsed; ++i) {
1470  shift[i] = defaultShift[i];
1471  }
1472 
1473  // copy of current and best parameters and associated first derivatives and chi2
1474  double param[2][SNFitParamMax] = {{0.}, {0.}};
1475  double deriv[2][SNFitParamMax] = {{0.}, {0.}};
1476  double chi2[2] = {0., std::numeric_limits<float>::max()};
1477  int iBestParam(1);
1478 
1479  double shiftSave(0.);
1480  int nSimilarSteps[SNFitParamMax] = {0};
1481  int nLoop(0), nFail(0);
1482 
1483  while (true) {
1484  ++nLoop;
1485 
1486  // keep the best results from the previous step and save the new ones in the other slot
1487  int iCurrentParam = 1 - iBestParam;
1488 
1489  // get the chi2 of the fit with the current parameters
1490  chi2[iCurrentParam] = computeChi2(currentParam, nParamUsed);
1491  ++nTrials;
1492 
1493  // compute first and second chi2 derivatives w.r.t. each parameter
1494  double deriv2nd[SNFitParamMax] = {0.};
1495  for (int i = 0; i < nParamUsed; ++i) {
1496  param[iCurrentParam][i] = currentParam[i];
1497  currentParam[i] += defaultShift[i] / 10.;
1498  double chi2Shift = computeChi2(currentParam, nParamUsed);
1499  ++nTrials;
1500  deriv[iCurrentParam][i] = (chi2Shift - chi2[iCurrentParam]) / defaultShift[i] * 10;
1501  deriv2nd[i] = param[0][i] != param[1][i] ? (deriv[0][i] - deriv[1][i]) / (param[0][i] - param[1][i]) : 0;
1502  currentParam[i] -= defaultShift[i] / 10.;
1503  }
1504 
1505  // abort if we exceed the maximum number of trials (integrated over the fits with 1, 2 and 3 clusters)
1506  if (nTrials > 2000) {
1507  break;
1508  }
1509 
1510  // choose the best parameters between the current ones and the best ones from the previous step
1511  iBestParam = chi2[0] < chi2[1] ? 0 : 1;
1512  nFail = iBestParam == iCurrentParam ? 0 : nFail + 1;
1513 
1514  // stop here if we reached the maximum number of iterations
1515  if (nLoop > 150) {
1516  break;
1517  }
1518 
1519  double stepMax(0.), derivMax(0.);
1520  int iDerivMax(0);
1521  for (int i = 0; i < nParamUsed; ++i) {
1522 
1523  // estimate the shift to perform of this parameter to reach the minimum chi2
1524  double previousShift = shift[i];
1525  if (nLoop == 1) {
1526  // start with the default step size
1527  shift[i] = TMath::Sign(defaultShift[i], -deriv[iCurrentParam][i]);
1528  } else if (TMath::Abs(deriv[0][i]) < 1.e-3 && TMath::Abs(deriv[1][i]) < 1.e-3) {
1529  // stay there if the minimum is reached w.r.t. to this parameter
1530  shift[i] = 0;
1531  } else if ((deriv[iBestParam][i] * deriv[1 - iBestParam][i] > 0. &&
1532  TMath::Abs(deriv[iBestParam][i]) > TMath::Abs(deriv[1 - iBestParam][i])) ||
1533  TMath::Abs(deriv[0][i] - deriv[1][i]) < 1.e-3 ||
1534  TMath::Abs(deriv2nd[i]) < 1.e-6) {
1535  // same size of shift if the first derivative increases or remain constant
1536  shift[i] = -TMath::Sign(shift[i], (chi2[0] - chi2[1]) * (param[0][i] - param[1][i]));
1537  // move faster if we already did >= 2 steps like this and the chi2 improved
1538  if (iBestParam == iCurrentParam) {
1539  if (nSimilarSteps[i] > 1) {
1540  shift[i] *= 2.;
1541  }
1542  ++nSimilarSteps[i];
1543  }
1544  } else {
1545  // adjust the shift otherwise
1546  shift[i] = deriv2nd[i] != 0. ? -deriv[iBestParam][i] / deriv2nd[i] : 0.;
1547  nSimilarSteps[i] = 0;
1548  }
1549 
1550  // maximum shift normalized to the default step size and maximum first derivative
1551  stepMax = TMath::Max(stepMax, TMath::Abs(shift[i]) / defaultShift[i]);
1552  if (TMath::Abs(deriv[iBestParam][i]) > derivMax) {
1553  iDerivMax = i;
1554  derivMax = TMath::Abs(deriv[iBestParam][i]);
1555  }
1556 
1557  // limit the shift to 10 times the default step size
1558  if (TMath::Abs(shift[i]) / defaultShift[i] > 10.) {
1559  shift[i] = TMath::Sign(10., shift[i]) * defaultShift[i];
1560  }
1561 
1562  // reset the current parameter and adjust the shift if the chi2 did not improve
1563  if (iBestParam != iCurrentParam) {
1564  nSimilarSteps[i] = 0;
1565  currentParam[i] = param[iBestParam][i];
1566  if (TMath::Abs(shift[i] + previousShift) > 0.1 * defaultShift[i]) {
1567  shift[i] = (shift[i] + previousShift) / 2.;
1568  } else {
1569  shift[i] /= -2.;
1570  }
1571  }
1572 
1573  // reduce the shift if the step is too big
1574  if (TMath::Abs(shift[i] * deriv[iBestParam][i]) > chi2[iBestParam]) {
1575  shift[i] = TMath::Sign(chi2[iBestParam] / deriv[iBestParam][i], shift[i]);
1576  }
1577 
1578  // introduce step relaxation factor
1579  if (nSimilarSteps[i] < 3) {
1580  double scMax = 1. + 4. / TMath::Max(nLoop / 2., 1.);
1581  if (TMath::Abs(previousShift) > 0. && TMath::Abs(shift[i] / previousShift) > scMax) {
1582  shift[i] = TMath::Sign(previousShift * scMax, shift[i]);
1583  }
1584  }
1585 
1586  // shift the current parameter and make sure we do not overstep its limits
1587  currentParam[i] += shift[i];
1588  if (currentParam[i] < parmin[i]) {
1589  shift[i] = parmin[i] - currentParam[i];
1590  currentParam[i] = parmin[i];
1591  } else if (currentParam[i] > parmax[i]) {
1592  shift[i] = parmax[i] - currentParam[i];
1593  currentParam[i] = parmax[i];
1594  }
1595  }
1596 
1597  // stop here if the minimum was found
1598  if (stepMax < 1. && derivMax < 2.) {
1599  break;
1600  }
1601 
1602  // check for small step
1603  if (shift[iDerivMax] == 0.) {
1604  shift[iDerivMax] = defaultShift[iDerivMax] / 10.;
1605  currentParam[iDerivMax] += shift[iDerivMax];
1606  continue;
1607  }
1608 
1609  // further adjustment...
1610  if (nSimilarSteps[iDerivMax] == 0 && derivMax > 0.5 && nLoop > 9) {
1611  if (deriv2nd[iDerivMax] != 0. && TMath::Abs(deriv[iBestParam][iDerivMax] / deriv2nd[iDerivMax] / shift[iDerivMax]) > 10.) {
1612  if (iBestParam == iCurrentParam) {
1613  deriv2nd[iDerivMax] = -deriv2nd[iDerivMax];
1614  }
1615  shift[iDerivMax] = -deriv[iBestParam][iDerivMax] / deriv2nd[iDerivMax] / 10.;
1616  currentParam[iDerivMax] += shift[iDerivMax];
1617  if (iBestParam == iCurrentParam) {
1618  shiftSave = shift[iDerivMax];
1619  }
1620  }
1621  if (nFail > 10) {
1622  currentParam[iDerivMax] -= shift[iDerivMax];
1623  shift[iDerivMax] = 4. * shiftSave * (gRandom->Rndm(0) - 0.5);
1624  currentParam[iDerivMax] += shift[iDerivMax];
1625  }
1626  }
1627  }
1628 
1629  // return the best parameters and associated chi2
1630  for (int i = 0; i < nParamUsed; ++i) {
1631  currentParam[i] = param[iBestParam][i];
1632  }
1633  return chi2[iBestParam];
1634 }

1874 void ClusterFinderOriginal::extractLeastCoupledClusters(std::vector<int>& coupledClusters, std::vector<int>& clustersForFit,

1876 {

1880  double minCoupling(DBL_MAX);
1881  int leastCoupledClusters[3] = {-1, -1, -1};
1882 
1883  // compute the coupling of each cluster with all the others and find the least coupled
1884  std::vector<double> coupling1(coupledClusters.size(), 0.);
1885  for (int i = 0; i < coupledClusters.size(); ++i) {
1886  for (int j = i + 1; j < coupledClusters.size(); ++j) {
1887  coupling1[i] += couplingClCl[coupledClusters[i]][coupledClusters[j]];
1888  coupling1[j] += couplingClCl[coupledClusters[i]][coupledClusters[j]];
1889  }
1890  if (coupling1[i] < minCoupling) {
1891  leastCoupledClusters[0] = i;
1892  minCoupling = coupling1[i];
1893  }
1894  }
1895 
1896  if (SNFitClustersMax > 1) {
1897 
1898  bool tryTierce = SNFitClustersMax > 2 && coupledClusters.size() > 5;
1899 
1900  for (int i = 0; i < coupledClusters.size(); ++i) {
1901  for (int j = i + 1; j < coupledClusters.size(); ++j) {
1902 
1903  // look for a lower coupling with the others by grouping clusters by pair
1904  double coupling2 = coupling1[i] + coupling1[j] - 2. * couplingClCl[coupledClusters[i]][coupledClusters[j]];
1905  if (coupling2 < minCoupling) {
1906  leastCoupledClusters[0] = i;
1907  leastCoupledClusters[1] = j;
1908  leastCoupledClusters[2] = -1;
1909  minCoupling = coupling2;
1910  }
1911 
1912  // look for a lower coupling with the others by grouping clusters by tierce
1913  if (tryTierce) {
1914  for (int k = j + 1; k < coupledClusters.size(); ++k) {
1915  double coupling3 = coupling2 + coupling1[k] -
1916  2. * (couplingClCl[coupledClusters[i]][coupledClusters[k]] +
1917  couplingClCl[coupledClusters[j]][coupledClusters[k]]);
1918  if (coupling3 < minCoupling) {
1919  leastCoupledClusters[0] = i;
1920  leastCoupledClusters[1] = j;
1921  leastCoupledClusters[2] = k;
1922  minCoupling = coupling3;
1923  }
1924  }
1925  }
1926  }
1927  }
1928  }
1929

1931  // take into account the shift of indices each time a cluster is remove
1932  int idxShift(0);
1933  for (int i = 0; i < 3 && leastCoupledClusters[i] >= 0; ++i) {
1934  clustersForFit.push_back(coupledClusters[leastCoupledClusters[i] - idxShift]);
1935  coupledClusters.erase(coupledClusters.begin() + leastCoupledClusters[i] - idxShift);
1936  ++idxShift;
1937  }
