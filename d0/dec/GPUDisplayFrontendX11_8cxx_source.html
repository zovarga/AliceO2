171 int GPUDisplayFrontendX11::FrontendMain()
172 {
173  XSetWindowAttributes windowAttributes;
174  XVisualInfo* visualInfo = nullptr;
175  XEvent event;
176  Colormap colorMap;
177  GLXContext glxContext;
178  int errorBase;
179  int eventBase;
180 
181  // Open a connection to the X server
182  mDisplay = XOpenDisplay(nullptr);
183 
184  if (mDisplay == nullptr) {
185  GPUError("could not open display");
186  return (-1);
187  }
188 
189  // Make sure OpenGL's GLX extension supported
190  if (!glXQueryExtension(mDisplay, &errorBase, &eventBase)) {
191  GPUError("X server has no OpenGL GLX extension");
192  return (-1);
193  }
194 
195  const char* glxExt = glXQueryExtensionsString(mDisplay, DefaultScreen(mDisplay));
196  if (strstr(glxExt, "GLX_EXT_swap_control") == nullptr) {
197  GPUError("No vsync support!");
198  vsync_supported = false;
199  } else {
200  vsync_supported = true;
201  }
202 
203  // Require MSAA, double buffering, and Depth buffer
204  int attribs[] = {GLX_X_RENDERABLE, True, GLX_DRAWABLE_TYPE, GLX_WINDOW_BIT, GLX_RENDER_TYPE, GLX_RGBA_BIT, GLX_X_VISUAL_TYPE, GLX_TRUE_COLOR, GLX_RED_SIZE, 8, GLX_GREEN_SIZE, 8, GLX_BLUE_SIZE, 8, GLX_ALPHA_SIZE, 8, GLX_DEPTH_SIZE, 24, GLX_STENCIL_SIZE, 8, GLX_DOUBLEBUFFER, True,
205  // GLX_SAMPLE_BUFFERS , 1, //Disable MSAA here, we do it by rendering to offscreenbuffer
206  // GLX_SAMPLES , MSAA_SAMPLES,
207  None};
208 
209  GLXFBConfig fbconfig = nullptr;
210  int fbcount;
211  GLXFBConfig* fbc = glXChooseFBConfig(mDisplay, DefaultScreen(mDisplay), attribs, &fbcount);
212  if (fbc == nullptr || fbcount == 0) {
213  GPUError("Failed to get MSAA GLXFBConfig");
214  return (-1);
215  }
216  fbconfig = fbc[0];
217  XFree(fbc);
218  visualInfo = glXGetVisualFromFBConfig(mDisplay, fbconfig);
219 
220  if (visualInfo == nullptr) {
221  GPUError("no RGB visual with depth buffer");
222  return (-1);
223  }
224 
225  // Create an OpenGL rendering context
226  glXCreateContextAttribsARBProc glXCreateContextAttribsARB = (glXCreateContextAttribsARBProc)glXGetProcAddressARB((const GLubyte*)"glXCreateContextAttribsARB");
227  if (glXCreateContextAttribsARB) {
228  int context_attribs[] = {
229  GLX_CONTEXT_MAJOR_VERSION_ARB, GL_MIN_VERSION_MAJOR,
230  GLX_CONTEXT_MINOR_VERSION_ARB, GL_MIN_VERSION_MINOR,
231  GLX_CONTEXT_PROFILE_MASK_ARB, mBackend->CoreProfile() ? GLX_CONTEXT_CORE_PROFILE_BIT_ARB : GLX_CONTEXT_COMPATIBILITY_PROFILE_BIT_ARB,
232  None};
233  glxContext = glXCreateContextAttribsARB(mDisplay, fbconfig, nullptr, GL_TRUE, context_attribs);
234  } else {
235  glxContext = glXCreateContext(mDisplay, visualInfo, nullptr, GL_TRUE);
236  }
237  if (glxContext == nullptr) {
238  GPUError("could not create rendering context");
239  return (-1);
240  }
241 
242  Window win = RootWindow(mDisplay, visualInfo->screen);
243  colorMap = XCreateColormap(mDisplay, win, visualInfo->visual, AllocNone);
244  windowAttributes.colormap = colorMap;
245  windowAttributes.border_pixel = 0;
246  windowAttributes.event_mask = ExposureMask | VisibilityChangeMask | KeyPressMask | KeyReleaseMask | ButtonPressMask | ButtonReleaseMask | PointerMotionMask | StructureNotifyMask | SubstructureNotifyMask | FocusChangeMask;
247 
248  // Create an X window with the selected visual
249  mWindow = XCreateWindow(mDisplay, win, 50, 50, INIT_WIDTH, INIT_HEIGHT, // Position / Width and height of window
250  0, visualInfo->depth, InputOutput, visualInfo->visual, CWBorderPixel | CWColormap | CWEventMask, &windowAttributes);
251  XSetStandardProperties(mDisplay, mWindow, GL_WINDOW_NAME, GL_WINDOW_NAME, None, nullptr, 0, nullptr);
252  glXMakeCurrent(mDisplay, mWindow, glxContext);
253  XMapWindow(mDisplay, mWindow);
254 
255  // Maximize window
256  ToggleMaximized(true);
257 
258  // Receive signal when window closed
259  Atom WM_DELETE_WINDOW = XInternAtom(mDisplay, "WM_DELETE_WINDOW", False);
260  XSetWMProtocols(mDisplay, mWindow, &WM_DELETE_WINDOW, 1);
261 #ifndef GPUCA_DISPLAY_OPENGL_CORE
262  // Prepare fonts
263  mFontBase = glGenLists(256);
264  if (!glIsList(mFontBase)) {
265  GPUError("Out of display lists.");
266  return (-1);
267  }
268  const char* f = "fixed";
269  XFontStruct* font_info = XLoadQueryFont(mDisplay, f);
270  if (!font_info) {
271  GPUError("XLoadQueryFont failed.");
272  return (-1);
273  } else {
274  int first = font_info->min_char_or_byte2;
275  int last = font_info->max_char_or_byte2;
276  glXUseXFont(font_info->fid, first, last - first + 1, mFontBase + first);
277  }
278 #endif
279 
280  // Init OpenGL...
281  if (mBackend->ExtInit()) {
282  fprintf(stderr, "Error initializing GL extension wrapper\n");
283  return (-1);
284  }
285 
286  XMapWindow(mDisplay, mWindow);
287  XFlush(mDisplay);
288  int x11_fd = ConnectionNumber(mDisplay);
289 
290  // Enable vsync
291  if (vsync_supported) {
292  mGlXSwapIntervalEXT = (PFNGLXSWAPINTERVALEXTPROC)glXGetProcAddressARB((const GLubyte*)"glXSwapIntervalEXT");
293  if (mGlXSwapIntervalEXT == nullptr) {
294  GPUError("Cannot enable vsync");
295  return (-1);
296  }
297  mGlXSwapIntervalEXT(mDisplay, glXGetCurrentDrawable(), 0);
298  }
299 
300  if (InitDisplay()) {
301  return (1);
302  }
303 
304  pthread_mutex_lock(&mSemLockExit);
305  mDisplayRunning = true;
306  pthread_mutex_unlock(&mSemLockExit);
307 
308  while (1) {
309  int num_ready_fds;
310  struct timeval tv;
311  fd_set in_fds;
312  int waitCount = 0;
313  do {
314  FD_ZERO(&in_fds);
315  FD_SET(x11_fd, &in_fds);
316  tv.tv_usec = 10000;
317  tv.tv_sec = 0;
318  num_ready_fds = mMaxFPSRate || XPending(mDisplay) || select(x11_fd + 1, &in_fds, nullptr, nullptr, &tv);
319  if (num_ready_fds < 0) {
320  GPUError("Error (num_ready_fds)");
321  }
322  if (mDisplayControl == 2) {
323  break;
324  }
325  if (mSendKey) {
326  mNeedUpdate = 1;
327  }
328  if (waitCount++ != 100) {
329  mNeedUpdate = 1;
330  }
331  } while (!(num_ready_fds || mNeedUpdate));
332  mNeedUpdate = 0;
333 
334  do {
335  if (mDisplayControl == 2) {
336  break;
337  }
338  HandleSendKey();
339  if (!XPending(mDisplay)) {
340  event.type = Expose;
341  } else {
342  XNextEvent(mDisplay, &event);
343  }
344  switch (event.type) {
345  case ButtonPress: {
346  if (event.xbutton.button == 4) {
347  mMouseWheel += 100;
348  } else if (event.xbutton.button == 5) {
349  mMouseWheel -= 100;
350  } else {
351  if (event.xbutton.button == 1) {
352  mMouseDn = true;
353  } else {
354  mMouseDnR = true;
355  }
356  mMouseDnX = event.xmotion.x;
357  mMouseDnY = event.xmotion.y;
358  }
359  break;
360  }
361 
362  case ButtonRelease: {
363  if (event.xbutton.button != 4 && event.xbutton.button != 5) {
364  if (event.xbutton.button == 1) {
365  mMouseDn = false;
366  } else {
367  mMouseDnR = false;
368  }
369  }
370  break;
371  }
372 
373  case KeyPress: {
374  int handleKey = 0, keyPress = 0;
375  GetKey(event, handleKey, keyPress);
376  mKeysShift[keyPress] = mKeys[KEY_SHIFT];
377  mKeys[keyPress] = true;
378  HandleKey(handleKey);
379  break;
380  }
381 
382  case KeyRelease: {
383  int handleKey = 0, keyPress = 0;
384  GetKey(event, handleKey, keyPress);
385  mKeys[keyPress] = false;
386  mKeysShift[keyPress] = false;
387  break;
388  }
389 
390  case MotionNotify: {
391  mouseMvX = event.xmotion.x;
392  mouseMvY = event.xmotion.y;
393  break;
394  }
395 
396  case Expose: {
397  break;
398  }
399 
400  case ConfigureNotify: {
401  ReSizeGLScene(event.xconfigure.width, event.xconfigure.height);
402  break;
403  }
404 
405  case ClientMessage: {
406  if (event.xclient.message_type == XInternAtom(mDisplay, "_NET_WM_STATE", False)) {
407  XFlush(mDisplay);
408  } else {
409  mDisplayControl = 2;
410  }
411  break;
412  }
413  }
414  } while (XPending(mDisplay)); // Loop to compress events
415  if (mDisplayControl == 2) {
416  break;
417  }
418 
419  DrawGLScene();
420  glXSwapBuffers(mDisplay, mWindow); // Buffer swap does implicit glFlush
421  }
422 
423 #ifndef GPUCA_DISPLAY_OPENGL_CORE
424  glDeleteLists(mFontBase, 256);
425  XUnloadFont(mDisplay, font_info->fid);
426 #endif
427  ExitDisplay();
428  glXDestroyContext(mDisplay, glxContext);
429  XFree(visualInfo);
430  XDestroyWindow(mDisplay, mWindow);
431  XCloseDisplay(mDisplay);
432 
433  pthread_mutex_lock(&mSemLockExit);
434  mDisplayRunning = false;
435  pthread_mutex_unlock(&mSemLockExit);
436 
437  return (0);
438 }

452 void GPUDisplayFrontendX11::SwitchFullscreen(bool set)
453 {
454  XEvent xev;
455  memset(&xev, 0, sizeof(xev));
456  xev.type = ClientMessage;
457  xev.xclient.window = mWindow;
458  xev.xclient.message_type = XInternAtom(mDisplay, "_NET_WM_STATE", False);
459  xev.xclient.format = 32;
460  xev.xclient.data.l[0] = 2; // _NET_WM_STATE_TOGGLE

462  xev.xclient.data.l[2] = 0;
463  XSendEvent(mDisplay, DefaultRootWindow(mDisplay), False, SubstructureNotifyMask, &xev);
