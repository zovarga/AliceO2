117 void GPUTPCGMMerger::CheckMergedTracks()
118 {
119  std::vector<bool> trkUsed(SliceTrackInfoLocalTotal());
120  for (int i = 0; i < SliceTrackInfoLocalTotal(); i++) {
121  trkUsed[i] = false;
122  }
123 
124  for (int itr = 0; itr < SliceTrackInfoLocalTotal(); itr++) {
125  GPUTPCGMSliceTrack& track = mSliceTrackInfos[itr];
126  if (track.PrevSegmentNeighbour() >= 0) {
127  continue;
128  }
129  if (track.PrevNeighbour() >= 0) {
130  continue;
131  }
132  int leg = 0;
133  GPUTPCGMSliceTrack *trbase = &track, *tr = &track;
134  while (true) {
135  int iTrk = tr - mSliceTrackInfos;
136  if (trkUsed[iTrk]) {
137  GPUError("FAILURE: double use");
138  }
139  trkUsed[iTrk] = true;
140 
141  int jtr = tr->NextSegmentNeighbour();
142  if (jtr >= 0) {
143  tr = &(mSliceTrackInfos[jtr]);
144  continue;
145  }
146  jtr = trbase->NextNeighbour();
147  if (jtr >= 0) {
148  trbase = &(mSliceTrackInfos[jtr]);
149  tr = trbase;
150  if (tr->PrevSegmentNeighbour() >= 0) {
151  break;
152  }
153  leg++;
154  continue;
155  }
156  break;
157  }
158  }
159  for (int i = 0; i < SliceTrackInfoLocalTotal(); i++) {
160  if (trkUsed[i] == false) {
161  GPUError("FAILURE: trk missed");
162  }
163  }
164 }

223 void GPUTPCGMMerger::PrintMergeGraph(const GPUTPCGMSliceTrack* trk, std::ostream& out)
224 {
225  const GPUTPCGMSliceTrack* orgTrack = trk;
226  while (trk->PrevSegmentNeighbour() >= 0) {
227  trk = &mSliceTrackInfos[trk->PrevSegmentNeighbour()];
228  }
229  const GPUTPCGMSliceTrack* orgTower = trk;
230  while (trk->PrevNeighbour() >= 0) {
231  trk = &mSliceTrackInfos[trk->PrevNeighbour()];
232  }
233 
234  int nextId = trk - mSliceTrackInfos;
235  out << "Graph of track %d" << (orgTrack - mSliceTrackInfos) << "\n";
236  while (nextId >= 0) {
237  trk = &mSliceTrackInfos[nextId];
238  if (trk->PrevSegmentNeighbour() >= 0) {
239  out << "TRACK TREE INVALID!!! " << trk->PrevSegmentNeighbour() << " --> " << nextId << "\n";
240  }
241  out << (trk == orgTower ? "--" : " ");
242  while (nextId >= 0) {
243  GPUTPCGMSliceTrack* trk2 = &mSliceTrackInfos[nextId];
244  if (trk != trk2 && (trk2->PrevNeighbour() >= 0 || trk2->NextNeighbour() >= 0)) {
245  out << " (TRACK TREE INVALID!!! " << trk2->PrevNeighbour() << " <-- " << nextId << " --> " << trk2->NextNeighbour() << ") ";
246  }
247  char tmp[128];
248  snprintf(tmp, 128, " %s%5d(%5.2f)", trk2 == orgTrack ? "!" : " ", nextId, trk2->QPt());
249  out << tmp;
250  nextId = trk2->NextSegmentNeighbour();
251  }
252  out << "\n";
253  nextId = trk->NextNeighbour();
254  }

259 void* GPUTPCGMMerger::SetPointersMerger(void* mem)
260 {
261  computePointerWithAlignment(mem, mSliceTrackInfos, mNMaxSliceTracks);
262  computePointerWithAlignment(mem, mSliceTrackInfoIndex, NSLICES * 2 + 1);
263  if (mRec->GetParam().rec.nonConsecutiveIDs) {
264  computePointerWithAlignment(mem, mGlobalClusterIDs, mNMaxOutputTrackClusters);
265  }
266 
267  void* memBase = mem;
268  computePointerWithAlignment(mem, mBorderMemory, 2 * mNMaxSliceTracks); // MergeBorders & Resolve
269  computePointerWithAlignment(mem, mBorderRangeMemory, 2 * mNMaxSliceTracks);
270  int nTracks = 0;
271  for (int iSlice = 0; iSlice < NSLICES; iSlice++) {
272  const int n = mRec->GetParam().rec.tpc.mergerReadFromTrackerDirectly ? *mRec->GetConstantMem().tpcTrackers[iSlice].NTracks() : mkSlices[iSlice]->NTracks();
273  mBorder[iSlice] = mBorderMemory + 2 * nTracks;
274  mBorder[NSLICES + iSlice] = mBorderMemory + 2 * nTracks + n;
275  mBorderRange[iSlice] = mBorderRangeMemory + 2 * nTracks;
276  nTracks += n;
277  }
278  computePointerWithAlignment(mem, mTrackLinks, mNMaxSliceTracks);
279  computePointerWithAlignment(mem, mTrackCCRoots, mNMaxSliceTracks);
280  void* memMax = mem;
281  mem = memBase;
282  computePointerWithAlignment(mem, mTrackIDs, mNMaxTracks); // UnpackResetIds - RefitSliceTracks - UnpackSliceGlobal
283  memMax = (void*)std::max((size_t)mem, (size_t)memMax);
284  mem = memBase;
285  computePointerWithAlignment(mem, mTrackSort, mNMaxTracks); // PrepareClustersForFit0 - SortTracksQPt - PrepareClustersForFit1 - PrepareClustersForFit1 / Finalize0 - Finalize2
286  computePointerWithAlignment(mem, mSharedCount, mNMaxClusters);
287  memMax = (void*)std::max((size_t)mem, (size_t)memMax);
288  mem = memBase;
289  computePointerWithAlignment(mem, mLoopData, mNMaxTracks); // GPUTPCGMMergerTrackFit - GPUTPCGMMergerFollowLoopers
290  computePointerWithAlignment(mem, mRetryRefitIds, mNMaxTracks); // Reducing mNMaxTracks for mLoopData / mRetryRefitIds does not save memory, since the other parts are larger anyway
291  memMax = (void*)std::max((size_t)mem, (size_t)memMax);
292  mem = memBase;
293  computePointerWithAlignment(mem, mLooperCandidates, mNMaxLooperMatches); // MergeLoopers 1-3
294  memMax = (void*)std::max((size_t)mem, (size_t)memMax);
295  return memMax;
296 }

442 GPUd() int GPUTPCGMMerger::RefitSliceTrack(GPUTPCGMSliceTrack& sliceTrack, const GPUTPCTrack* inTrack, float alpha, int slice)
443 {
444  GPUTPCGMPropagator prop;
445  prop.SetMaterialTPC();
446  prop.SetMaxSinPhi(GPUCA_MAX_SIN_PHI);
447  prop.SetToyMCEventsFlag(false);
448  prop.SetSeedingErrors(true); // Larger errors for seeds, better since we don't start with good hypothesis
449  prop.SetFitInProjections(false);
450  prop.SetPolynomialField(&Param().polynomialField);
451  GPUTPCGMTrackParam trk;
452  trk.X() = inTrack->Param().GetX();
453  trk.Y() = inTrack->Param().GetY();
454  trk.Z() = inTrack->Param().GetZ();
455  trk.SinPhi() = inTrack->Param().GetSinPhi();
456  trk.DzDs() = inTrack->Param().GetDzDs();
457  trk.QPt() = inTrack->Param().GetQPt();
458  trk.TZOffset() = Param().par.earlyTpcTransform ? inTrack->Param().GetZOffset() : GetConstantMem()->calibObjects.fastTransform->convZOffsetToVertexTime(slice, inTrack->Param().GetZOffset(), Param().par.continuousMaxTimeBin);
459  trk.ShiftZ(this, slice, sliceTrack.ClusterZT0(), sliceTrack.ClusterZTN());
460  sliceTrack.SetX2(0.f);
461  for (int way = 0; way < 2; way++) {
462  if (way) {
463  prop.SetFitInProjections(true);
464  prop.SetPropagateBzOnly(true);
465  }
466  trk.ResetCovariance();
467  prop.SetTrack(&trk, alpha);
468  int start = way ? inTrack->NHits() - 1 : 0;
469  int end = way ? 0 : (inTrack->NHits() - 1);
470  int incr = way ? -1 : 1;
471  for (int i = start; i != end; i += incr) {
472  float x, y, z;
473  int row, flags;
474  if (Param().rec.tpc.mergerReadFromTrackerDirectly) {
475  const GPUTPCTracker& tracker = GetConstantMem()->tpcTrackers[slice];
476  const GPUTPCHitId& ic = tracker.TrackHits()[inTrack->FirstHitID() + i];
477  int clusterIndex = tracker.Data().ClusterDataIndex(tracker.Data().Row(ic.RowIndex()), ic.HitIndex());
478  row = ic.RowIndex();
479  const ClusterNative& cl = GetConstantMem()->ioPtrs.clustersNative->clustersLinear[GetConstantMem()->ioPtrs.clustersNative->clusterOffset[slice][0] + clusterIndex];
480  flags = cl.getFlags();
481  if (Param().par.earlyTpcTransform) {
482  x = tracker.Data().ClusterData()[clusterIndex].x;
483  y = tracker.Data().ClusterData()[clusterIndex].y;
484  z = tracker.Data().ClusterData()[clusterIndex].z - trk.TZOffset();
