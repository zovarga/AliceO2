262  void EventManagerFrame::checkMemory()
263  {
264  const long memoryLimit = Options::Instance()->memoryLimit();
265  if (memoryLimit != -1) {
266  const char* statmPath = "/proc/self/statm";
267  long size = -1;
268  FILE* f = fopen(statmPath, "r");
269  if (f != nullptr) { // could not read file => no check
270  int success = fscanf(f, "%ld", &size);
271  fclose(f);
272  if (success == 1) { // properly readed
273  size = 4 * size / 1024; // in MB
274  LOG(info) << "Memory used: " << size << " memory allowed: " << memoryLimit;
275  if (size > memoryLimit) {
276  LOG(error) << "Memory used: " << size << " exceeds memory allowed: "
277  << memoryLimit;
278  exit(-1);
279  }
280  }
281  }

378  bool EventManagerFrame::CopyImage(TASImage* dst, TASImage* src, Int_t x_dst, Int_t y_dst, Int_t x_src, Int_t y_src,
379  UInt_t w_src, UInt_t h_src)
380  {
381 
382  if (!dst) {
383  return false;
384  }
385  if (!src) {
386  return false;
387  }
388 
389  int x = 0;
390  int y = 0;
391  int idx_src = 0;
392  int idx_dst = 0;
393  x_src = x_src < 0 ? 0 : x_src;
394  y_src = y_src < 0 ? 0 : y_src;
395 
396  if ((x_src >= (int)src->GetWidth()) || (y_src >= (int)src->GetHeight())) {
397  return false;
398  }
399 
400  w_src = x_src + w_src > src->GetWidth() ? src->GetWidth() - x_src : w_src;
401  h_src = y_src + h_src > src->GetHeight() ? src->GetHeight() - y_src : h_src;
402  UInt_t yy = (y_src + y) * src->GetWidth();
403 
404  src->BeginPaint(false);
405  dst->BeginPaint(false);
406 
407  UInt_t* dst_image_array = dst->GetArgbArray();
408  UInt_t* src_image_array = src->GetArgbArray();
409 
410  if (!dst_image_array || !src_image_array) {
411  return false;
412  }
413 
414  for (y = 0; y < (int)h_src; y++) {
415  for (x = 0; x < (int)w_src; x++) {
416 
417  idx_src = yy + x + x_src;
418  idx_dst = (y_dst + y) * dst->GetWidth() + x + x_dst;
419 
420  if ((x + x_dst < 0) || (y_dst + y < 0) ||
421  (x + x_dst >= (int)dst->GetWidth()) || (y + y_dst >= (int)dst->GetHeight())) {
422  continue;
423  }
424 
425  dst_image_array[idx_dst] = src_image_array[idx_src];
426  }
427  yy += src->GetWidth();
428  }
429 
430  return true;
431  }

433  TASImage* EventManagerFrame::ScaleImage(TASImage* image, UInt_t desiredWidth, UInt_t desiredHeight)
434  {
435  if (!image) {
436  return nullptr;
437  }
438  if (desiredWidth == 0 || desiredHeight == 0) {
439  return nullptr;
440  }
441 
442  const char* backgroundColor = "#000000";
443 
444  UInt_t scaleWidth = desiredWidth;
445  UInt_t scaleHeight = desiredHeight;
446  UInt_t offsetWidth = 0;
447  UInt_t offsetHeight = 0;
448 
449  float aspectRatio = (float)image->GetWidth() / (float)image->GetHeight();
450 
451  if (desiredWidth >= aspectRatio * desiredHeight) {
452  scaleWidth = (UInt_t)(aspectRatio * desiredHeight);
453  offsetWidth = (desiredWidth - scaleWidth) / 2.0f;
454  } else {
455  scaleHeight = (UInt_t)((1.0f / aspectRatio) * desiredWidth);
456  offsetHeight = (desiredHeight - scaleHeight) / 2.0f;
457  }
458 
459  TASImage* scaledImage = new TASImage(desiredWidth, desiredHeight);
460  scaledImage->FillRectangle(backgroundColor, 0, 0, desiredWidth, desiredHeight);
461 
462  image->Scale(scaleWidth, scaleHeight);
463 
464  CopyImage(scaledImage, image, offsetWidth, offsetHeight, 0, 0, scaleWidth, scaleHeight);
465 
466  return scaledImage;
467  }
