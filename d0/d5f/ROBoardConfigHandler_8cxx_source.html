58 bool ROBoardConfigHandler::load(const char* filename)
59 {
60  std::ifstream inFile(filename);
61  if (!inFile.is_open()) {
62  return false;
63  }
64  std::vector<ROBoardConfig> configurations;
65  std::string line, token;
66  while (std::getline(inFile, line)) {
67  if (std::count(line.begin(), line.end(), ' ') < 5) {
68  continue;
69  }
70  if (line.find('#') < line.find(' ')) {
71  continue;
72  }
73  ROBoardConfig cfg;
74  std::stringstream ss;
75  ss << line;
76  std::getline(ss, token, ' ');
77  cfg.boardId = static_cast<uint8_t>(std::strtol(token.c_str(), nullptr, 16));
78  std::getline(ss, token, ' ');
79  cfg.configWord = static_cast<uint32_t>(std::strtol(token.c_str(), nullptr, 16));
80  for (int ich = 0; ich < 4; ++ich) {
81  std::getline(ss, token, ' ');
82  auto mask = static_cast<uint32_t>(std::strtol(token.c_str(), nullptr, 16));
83  cfg.masksBP[ich] = (mask >> 16) & 0xFFFF;
84  cfg.masksNBP[ich] = (mask & 0xFFFF);
85  }
86  configurations.emplace_back(cfg);
87  inFile.close();
88  }
89  set(configurations);
90  return true;
91 }

93 void ROBoardConfigHandler::write(const char* filename) const
94 {
96  std::vector<ROBoardConfig> configs;
97  for (auto& cfgIt : mROBoardConfigs) {
98  configs.emplace_back(cfgIt.second);
99  }
100 
101  std::sort(configs.begin(), configs.end(), [](const ROBoardConfig& cfg1, const ROBoardConfig& cfg2) { return cfg1.boardId < cfg2.boardId; });
102 
103  std::ofstream outFile(filename);
104  for (auto& cfg : configs) {
105  outFile << cfg << std::endl;
106  }
107  outFile.close();
108 }

118 void ROBoardConfigHandler::updateMasks(const std::vector<ROBoard>& masks)
119 {
120  for (auto& mask : masks) {
121  auto cfgIt = mROBoardConfigs.find(mask.boardId);
122 
123  // First we check if some patterns has zeros.
124  // When set xORy for boards with no Y input.
125  // So in this case we explicitly mask Y.
126  bool isMasked = ((cfgIt->second.configWord & crateconfig::sXorY) != 0);
127  for (int ich = 0; ich < 4; ++ich) {
128  if (mask.patternsBP[ich] != 0xFFFF || mask.patternsNBP[ich] != 0xFFFF) {
129  isMasked = true;
130  }
131  }
132 
133  if (isMasked) {
134  cfgIt->second.configWord |= crateconfig::sMonmoff;
135  cfgIt->second.masksBP = mask.patternsBP;
136  if ((cfgIt->second.configWord & crateconfig::sXorY) == 0) {
137  cfgIt->second.masksNBP = mask.patternsNBP;
138  }
139  }
140  }
141 }

143 std::vector<ROBoardConfig> makeDefaultROBoardConfig(uint16_t gbtUniqueId)
144 {
145  // FIXME: in the current configuration, when an Y strip covers several local boards
146  // the signal is sent to the first board only and no copy of the signal is sent to the others
147  // In this case we cannot implement zero suppression to boards that only receive the X signal.
148  // Originally we applied zero suppression to the board receiving both X and Y signals,
149  // but this leads to a bias.
150  // When the neighbour board is fired, indeed we have that:
151  // - the board with no Y signal transmits its X signal
152  // - the board with Y signal has no X signal. If we require X AND Y we lose the Y signal.
153  // For the moment we decide to require X OR Y by default to all board.
154  // This is equivalent to no zero suppression applied.
155  // In the future, we might apply X AND Y to cases were the Y strip belong to 1 local board only.
156  std::vector<ROBoardConfig> configurations;
157  CrateMapper crateMapper;
158  auto locIds = crateMapper.getROBoardIds(gbtUniqueId);
159  for (auto& locId : locIds) {
160  ROBoardConfig cfg;
161  cfg.configWord = crateconfig::sTxDataMask | crateconfig::sXorY;
162  cfg.boardId = locId;
163  configurations.emplace_back(cfg);
164  }
165  return configurations;
166 }

168 std::vector<ROBoardConfig> makeNoZSROBoardConfig(uint16_t gbtUniqueId)
169 {
170  // FIXME: notice that the current default implies no zero suppression
171  // so this is equivalent to the default
172  // We still keep this in case we change the default behaviour in the future
173  auto configurations = makeDefaultROBoardConfig(gbtUniqueId);
174  for (auto& cfg : configurations) {
175  cfg.configWord |= crateconfig::sXorY;
176  }
177  return configurations;
