64 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
65 {
66  WorkflowSpec specs;
67 
68  // Update the (declared) parameters if changed from the command line
69  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
70  // write the configuration used for the workflow
71  o2::conf::ConfigurableParam::writeINI("o2eventtime-check-tof-workflow_configuration.ini");
72 
73  auto useMC = !configcontext.options().get<bool>("disable-mc");
74  auto disableRootIn = configcontext.options().get<bool>("disable-root-input");
75  auto disableRootOut = configcontext.options().get<bool>("disable-root-output");
76 
77  auto spread = configcontext.options().get<float>("event-time-spread");
78  auto phase = configcontext.options().get<float>("lhc-phase");
79  auto minEta = configcontext.options().get<float>("eta-min");
80  auto maxEta = configcontext.options().get<float>("eta-max");
81  auto fillscheme = configcontext.options().get<std::string>("fill-mask");
82 
83  o2::tof::Utils::mEventTimeSpread = spread;
84  o2::tof::Utils::mLHCPhase = phase;
85  o2::tof::Utils::mEtaMin = minEta;
86  o2::tof::Utils::mEtaMax = maxEta;
87 
88  while (fillscheme.size()) {
89  int bc;
90  int res = sscanf(fillscheme.c_str(), "%d", &bc);
91  if (res) {
92  o2::tof::Utils::addInteractionBC(bc);
93  } else {
94  fillscheme.clear();
95  }
96  int next = fillscheme.find(",");
97  if (next < fillscheme.size()) {
98  fillscheme.erase(0, next + 1);
99  } else {
100  fillscheme.clear();
101  }
102  }
103 
104  if (!o2::tof::Utils::hasFillScheme()) {
105  // check collision context
106  auto mcReader = std::make_unique<o2::steer::MCKinematicsReader>("collisioncontext.root");
107  auto context = mcReader->getDigitizationContext();
108  if (!context) {
109  auto bcf = context->getBunchFilling();
110  std::bitset<3564> isInBC = bcf.getBCPattern();
111  for (int i = 0; i < isInBC.size(); i++) {
112  if (isInBC.test(i)) {
113  o2::tof::Utils::addInteractionBC(i, true);
114  }
115  }
116  }
117  }
118 
119  LOG(debug) << "TOF EVENTTIME CHECKER WORKFLOW configuration";
120  LOG(debug) << "TOF track inputs = " << configcontext.options().get<std::string>("track-sources");
121  LOG(debug) << "TOF disable-mc = " << configcontext.options().get<std::string>("disable-mc");
122  LOG(debug) << "TOF disable-root-input = " << disableRootIn;
123  LOG(debug) << "TOF disable-root-output = " << disableRootOut;
124 
125  GID::mask_t alowedSources = GID::getSourcesMask("ITS-TPC-TOF,TPC-TOF,ITS-TPC-TRD-TOF,TPC-TRD-TOF,ITS-TPC,TPC,ITS-TPC-TRD,TPC-TRD");
126 
127  GID::mask_t src = alowedSources & GID::getSourcesMask(configcontext.options().get<std::string>("track-sources"));
128  GID::mask_t mcmaskcl;
129  GID::mask_t nonemask = GID::getSourcesMask(GID::NONE);
130  GID::mask_t clustermask = GID::getSourcesMask("TOF");
131 
132  if (useMC) {
133  mcmaskcl |= GID::getSourceMask(GID::TOF);
134  }
135  // pass strict flag to fetch eventual TPC-TRD input with correct subspec
136  o2::globaltracking::InputHelper::addInputSpecs(configcontext, specs, clustermask, src, src, useMC, mcmaskcl, GID::getSourcesMask(GID::ALL), false);
137 
138  specs.emplace_back(o2::globaltracking::getTOFEventTimeCheckerSpec(src, useMC)); // doTPCrefit not yet supported (need to load TPC clusters?)
139 
140  o2::raw::HBFUtilsInitializer hbfIni(configcontext, specs);
141 
142  return std::move(specs);
143 }
