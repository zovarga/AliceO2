36 struct Triggers {
37  enum { bitA,
38  bitC,
39  bitVertex,
40  bitCen,
41  bitSCen,
42  bitLaser };
43  uint8_t triggersignals = 0; // T0 trigger signals
44  int8_t nChanA = 0; // number of fired channels A side
45  int8_t nChanC = 0; // number of fired channels A side
46  int32_t amplA = -5000; // sum amplitude A side
47  int32_t amplC = -5000; // sum amplitude C side
48  int16_t timeA = -5000; // average time A side
49  int16_t timeC = -5000; // average time C side
50  Triggers() = default;
51  Triggers(uint8_t signals, int8_t chanA, int8_t chanC, int32_t aamplA, int32_t aamplC, int16_t atimeA, int16_t atimeC)
52  {
53  triggersignals = signals;
54  nChanA = chanA;
55  nChanC = chanC;
56  amplA = aamplA;
57  amplC = aamplC;
58  timeA = atimeA;
59  timeC = atimeC;
60  }
61  bool getOrA() const { return (triggersignals & (1 << bitA)) != 0; }
62  bool getOrC() const { return (triggersignals & (1 << bitC)) != 0; }
63  bool getVertex() const { return (triggersignals & (1 << bitVertex)) != 0; }
64  bool getCen() const { return (triggersignals & (1 << bitCen)) != 0; }
65  bool getSCen() const { return (triggersignals & (1 << bitSCen)) != 0; }
66  bool getLaserBit() const { return (triggersignals & (1 << bitLaser)) != 0; }
67 
68  void setTriggers(Bool_t isA, Bool_t isC, Bool_t isVrtx, Bool_t isCnt, Bool_t isSCnt, int8_t chanA, int8_t chanC, int32_t aamplA,
69  int32_t aamplC, int16_t atimeA, int16_t atimeC, Bool_t isLaser = kFALSE)
70  {
71  triggersignals = (isA << bitA) | (isC << bitC) | (isVrtx << bitVertex) | (isCnt << bitCen) | (isSCnt << bitSCen) | (isLaser << bitLaser);
72  nChanA = chanA;
73  nChanC = chanC;
74  amplA = aamplA;
75  amplC = aamplC;
76  timeA = atimeA;
77  timeC = atimeC;
78  }
79  void cleanTriggers()
80  {
81  triggersignals = 0;
82  nChanA = nChanC = 0;
83  amplA = amplC = -5000;
84  timeA = timeC = -5000;
85  }
86  bool operator==(Triggers const& other) const
87  {
88  return std::tie(triggersignals, nChanA, nChanC, amplA, amplC, timeA, timeC) ==
89  std::tie(other.triggersignals, other.nChanA, other.nChanC, other.amplA, other.amplC, other.timeA, other.timeC);
90  }
91  void printLog() const;
92  ClassDefNV(Triggers, 3);
93 };
