34 void MatLayerCylSet::addLayer(float rmin, float rmax, float zmax, float dz, float drphi)
35 {
36  // add new layer checking for overlaps
37  assert(mConstructionMask != Constructed);
38  assert(rmin < rmax && zmax > 0 && dz > 0 && drphi > 0);
39  mConstructionMask = InProgress;
40  int nlr = getNLayers();
41  if (!nlr) {
42  // book local storage
43  auto sz = sizeof(MatLayerCylSetLayout);
44  o2::gpu::resizeArray(mFlatBufferContainer, 0, sz);
45  mFlatBufferPtr = mFlatBufferContainer;
46  mFlatBufferSize = sz;
47  //--------------????
48  get()->mRMin = 1.e99;
49  get()->mRMax = 0.;
50  }
51 
52  for (int il = 0; il < nlr; il++) {
53  const auto& lr = getLayer(il);
54  if (lr.getRMax() > rmin && rmax > lr.getRMin()) {
55  LOG(fatal) << "new layer overlaps with layer " << il;
56  }
57  }
58  auto* oldLayers = o2::gpu::resizeArray(get()->mLayers, nlr, nlr + 1);
59  // dynamyc buffers of old layers were used in new ones, detach them
60  for (int i = nlr; i--;) {
61  oldLayers[i].clearInternalBufferPtr();
62  }
63  delete[] oldLayers;
64  get()->mLayers[nlr].initSegmentation(rmin, rmax, zmax, dz, drphi);
65  get()->mNLayers++;
66  get()->mRMin = get()->mRMin > rmin ? rmin : get()->mRMin;
67  get()->mRMax = get()->mRMax < rmax ? rmax : get()->mRMax;
68  get()->mZMax = get()->mZMax < zmax ? zmax : get()->mZMax;
69  get()->mRMin2 = get()->mRMin * get()->mRMin;
70  get()->mRMax2 = get()->mRMax * get()->mRMax;
71 }

74 void MatLayerCylSet::populateFromTGeo(int ntrPerCell)
75 {
77  assert(mConstructionMask == InProgress);
78 
79  int nlr = getNLayers();
80  if (!nlr) {
81  LOG(error) << "The LUT is not yet initialized";
82  return;
83  }
84  if (get()->mR2Intervals) {
85  LOG(error) << "The LUT is already populated";
86  return;
87  }
88  for (int i = 0; i < nlr; i++) {
89  printf("Populating with %d trials Lr %3d ", ntrPerCell, i);
90  get()->mLayers[i].print();
91  get()->mLayers[i].populateFromTGeo(ntrPerCell);
92  }
93  // build layer search structures
94  int nR2Int = 2 * (nlr + 1);
95  o2::gpu::resizeArray(get()->mR2Intervals, 0, nR2Int);
96  o2::gpu::resizeArray(get()->mInterval2LrID, 0, nR2Int);
97  get()->mR2Intervals[0] = get()->mRMin2;
98  get()->mR2Intervals[1] = get()->mRMax2;
99  get()->mInterval2LrID[0] = 0;
100  auto& nRIntervals = get()->mNRIntervals;
101  nRIntervals = 1;
102 
103  for (int i = 1; i < nlr; i++) {
104  const auto& lr = getLayer(i);
105  if (std::sqrt(lr.getRMin2()) > std::sqrt(get()->mR2Intervals[nRIntervals] + Ray::Tiny)) {
106  // register gap
107  get()->mInterval2LrID[nRIntervals] = -1;
108  get()->mR2Intervals[++nRIntervals] = lr.getRMin2();
109  }
110  get()->mInterval2LrID[nRIntervals] = i;
111  get()->mR2Intervals[++nRIntervals] = lr.getRMax2();
112  }
113  delete[] o2::gpu::resizeArray(get()->mInterval2LrID, nR2Int, nRIntervals); // rebook with precise size
114  delete[] o2::gpu::resizeArray(get()->mR2Intervals, nR2Int, ++nRIntervals); // rebook with precise size
115  //
116 }

119 void MatLayerCylSet::dumpToTree(const std::string& outName) const
120 {
122 
123  o2::utils::TreeStreamRedirector dump(outName.data(), "recreate");
124  for (int i = 0; i < getNLayers(); i++) {
125  const auto& lr = getLayer(i);
126  float r = 0.5 * (lr.getRMin() + lr.getRMax());
127  // per cell dump
128  int nphib = lr.getNPhiBins();
129  for (int ip = 0; ip < nphib; ip++) {
130  float phi = 0.5 * (lr.getPhiBinMin(ip) + lr.getPhiBinMax(ip));
131  float sn, cs;
132  int ips = lr.phiBin2Slice(ip);
133  char merge = 0; // not mergeable
134  if (ip + 1 < nphib) {
135  int ips1 = lr.phiBin2Slice(ip + 1);
136  merge = ips == ips1 ? -1 : lr.canMergePhiSlices(ips, ips1); // -1 for already merged
137  } else {
138  merge = -2; // last one
139  }
140  o2::math_utils::sincos(phi, sn, cs);
141  float x = r * cs, y = r * sn;
142  for (int iz = 0; iz < lr.getNZBins(); iz++) {
143  float z = 0.5 * (lr.getZBinMin(iz) + lr.getZBinMax(iz));
144  auto cell = lr.getCellPhiBin(ip, iz);
145  dump << "cell"
146  << "ilr=" << i << "r=" << r << "phi=" << phi << "x=" << x << "y=" << y << "z=" << z << "ip=" << ip << "ips=" << ips << "iz=" << iz
147  << "mrgnxt=" << merge << "val=" << cell << "\n";
148  }
149  }
150  //
151  // statistics per layer
152  MatCell mean, rms;
153  lr.getMeanRMS(mean, rms);
154  dump << "lay"
155  << "ilr=" << i << "r=" << r << "mean=" << mean << "rms=" << rms << "\n";
156  }
157 }

248 GPUd() MatBudget MatLayerCylSet::getMatBudget(float x0, float y0, float z0, float x1, float y1, float z1) const
249 {
250  // get material budget traversed on the line between point0 and point1
251  MatBudget rval;
252  Ray ray(x0, y0, z0, x1, y1, z1);
253  short lmin, lmax; // get innermost and outermost relevant layer
254  if (ray.isTooShort() || !getLayersRange(ray, lmin, lmax)) {
255  rval.length = ray.getDist();
256  return rval;
257  }
258  short lrID = lmax;
259  while (lrID >= lmin) { // go from outside to inside
260  const auto& lr = getLayer(lrID);
261  int nc = ray.crossLayer(lr);
262  for (int ic = nc; ic--;) {
263  float cross1, cross2;
264  ray.getCrossParams(ic, cross1, cross2); // tmax,tmin of crossing the layer
265  auto phi0 = ray.getPhi(cross1), phi1 = ray.getPhi(cross2), dPhi = phi0 - phi1;
266  auto phiID = lr.getPhiSliceID(phi0), phiIDLast = lr.getPhiSliceID(phi1);
267  // account for eventual wrapping around 0
268  if (dPhi > 0.f) {
269  if (dPhi > o2::constants::math::PI) { // wraps around phi=0
270  phiIDLast += lr.getNPhiSlices();
271  }
272  } else {
273  if (dPhi < -o2::constants::math::PI) { // wraps around phi=0
274  phiID += lr.getNPhiSlices();
275  }
276  }
277  int stepPhiID = phiID > phiIDLast ? -1 : 1;
278  bool checkMorePhi = true;
279  auto tStartPhi = cross1, tEndPhi = 0.f;
280  do {
281  // get the path in the current phi slice
282  if (phiID == phiIDLast) {
283  tEndPhi = cross2;
284  checkMorePhi = false;
285  } else { // last phi slice still not reached
286  tEndPhi = ray.crossRadial(lr, (stepPhiID > 0 ? phiID + 1 : phiID) % lr.getNPhiSlices());
287  if (tEndPhi == Ray::InvalidT) {
288  break; // ray parallel to radial line, abandon check for phi bin change
289  }
290  }
291  auto zID = lr.getZBinID(ray.getZ(tStartPhi));
292  auto zIDLast = lr.getZBinID(ray.getZ(tEndPhi));
293  // check if Zbins are crossed
294 
295 #ifdef _DBG_LOC_
296  printf("-- Zdiff (%3d : %3d) mode: t: %+e %+e\n", zID, zIDLast, tStartPhi, tEndPhi);
297 #endif
298 
299  if (zID != zIDLast) {
300  auto stepZID = zID < zIDLast ? 1 : -1;
301  bool checkMoreZ = true;
302  auto tStartZ = tStartPhi, tEndZ = 0.f;
303  do {
304  if (zID == zIDLast) {
305  tEndZ = tEndPhi;
306  checkMoreZ = false;
307  } else {
308  tEndZ = ray.crossZ(lr.getZBinMin(stepZID > 0 ? zID + 1 : zID));
309  if (tEndZ == Ray::InvalidT) { // track normal to Z axis, abandon Zbin change test
310  break;
311  }
312  }
313  // account materials of this step
314  float step = tEndZ > tStartZ ? tEndZ - tStartZ : tStartZ - tEndZ; // the real step is ray.getDist(tEnd-tStart), will rescale all later
315  const auto& cell = lr.getCell(phiID, zID);
316  rval.meanRho += cell.meanRho * step;
317  rval.meanX2X0 += cell.meanX2X0 * step;
318  rval.length += step;
319 
320 #ifdef _DBG_LOC_
321  float pos0[3] = {ray.getPos(tStartZ, 0), ray.getPos(tStartZ, 1), ray.getPos(tStartZ, 2)};
322  float pos1[3] = {ray.getPos(tEndZ, 0), ray.getPos(tEndZ, 1), ray.getPos(tEndZ, 2)};
323  printf(
324  "Lr#%3d / cross#%d : account %f<t<%f at phiSlice %d | Zbin: %3d (%3d) |[%+e %+e +%e]:[%+e %+e %+e] "
325  "Step: %.3e StrpCor: %.3e\n",
326  lrID, ic, tEndZ, tStartZ, phiID % lr.getNPhiSlices(), zID, zIDLast,
327  pos0[0], pos0[1], pos0[2], pos1[0], pos1[1], pos1[2], step, ray.getDist(step));
328 #endif
329 
330  tStartZ = tEndZ;
331  zID += stepZID;
332  } while (checkMoreZ);
333  } else {
334  float step = tEndPhi > tStartPhi ? tEndPhi - tStartPhi : tStartPhi - tEndPhi; // the real step is |ray.getDist(tEnd-tStart)|, will rescale all later
335  const auto& cell = lr.getCell(phiID, zID);
336  rval.meanRho += cell.meanRho * step;
337  rval.meanX2X0 += cell.meanX2X0 * step;
338  rval.length += step;
339 
340 #ifdef _DBG_LOC_
341  float pos0[3] = {ray.getPos(tStartPhi, 0), ray.getPos(tStartPhi, 1), ray.getPos(tStartPhi, 2)};
342  float pos1[3] = {ray.getPos(tEndPhi, 0), ray.getPos(tEndPhi, 1), ray.getPos(tEndPhi, 2)};
343  printf(
344  "Lr#%3d / cross#%d : account %f<t<%f at phiSlice %d | Zbin: %3d ----- |[%+e %+e +%e]:[%+e %+e %+e]"
345  "Step: %.3e StrpCor: %.3e\n",
346  lrID, ic, tEndPhi, tStartPhi, phiID % lr.getNPhiSlices(), zID,
347  pos0[0], pos0[1], pos0[2], pos1[0], pos1[1], pos1[2], step, ray.getDist(step));
348 #endif
349  }
350  //
351  tStartPhi = tEndPhi;
352  phiID += stepPhiID;
353 
354  } while (checkMorePhi);
355  }
356  lrID--;
357  } // loop over layers
358 
359  if (rval.length != 0.f) {
360  rval.meanRho /= rval.length; // average
361  rval.meanX2X0 *= ray.getDist(); // normalize
362  }
363  rval.length = ray.getDist();
364 
365 #ifdef _DBG_LOC_
366  printf("<rho> = %e, x2X0 = %e | step = %e\n", rval.meanRho, rval.meanX2X0, rval.length);
367 #endif
368  return rval;
369 }

372 GPUd() bool MatLayerCylSet::getLayersRange(const Ray& ray, short& lmin, short& lmax) const
373 {
374  // get range of layers corresponding to rmin/rmax
375  //
376  lmin = lmax = -1;
377  float rmin2, rmax2;
378  ray.getMinMaxR2(rmin2, rmax2);
379 
380  if (rmin2 >= getRMax2() || rmax2 <= getRMin2()) {
381  return false;
382  }
383  int lmxInt, lmnInt;
384  lmxInt = rmax2 < getRMax2() ? searchSegment(rmax2, 0) : get()->mNRIntervals - 2;
385  lmnInt = rmin2 >= getRMin2() ? searchSegment(rmin2, 0, lmxInt + 1) : 0;
386  const auto* interval2LrID = get()->mInterval2LrID;
387  lmax = interval2LrID[lmxInt];
388  lmin = interval2LrID[lmnInt];
389  // make sure lmnInt and/or lmxInt are not in the gap
390  if (lmax < 0) {
391  lmax = interval2LrID[--lmxInt]; // rmax2 is in the gap, take highest layer below rmax2
392  }
393  if (lmin < 0) {
394  lmin = interval2LrID[++lmnInt]; // rmin2 is in the gap, take lowest layer above rmin2
395  }
396  return lmin <= lmax; // valid if both are not in the same gap
397 }

399 GPUd() int MatLayerCylSet::searchSegment(float val, int low, int high) const
400 {
402  if (low < 0) {
403  low = 0;
404  }
405  if (high < 0) {
406  high = get()->mNRIntervals;
407  }
408  int mid = (low + high) >> 1;
409  const auto* r2Intervals = get()->mR2Intervals;
410  while (mid != low) {
411  if (val < r2Intervals[mid]) {
412  high = mid;
413  } else {
414  low = mid;
415  }
416  mid = (low + high) >> 1;
417  }
418  return mid;
419 }

423 void MatLayerCylSet::flatten()
424 {
425  // make object flat: move all content to single internally allocated buffer
426  assert(mConstructionMask == InProgress);
427 
428  int sz = estimateFlatBufferSize();
429  // create new internal buffer with total size and copy data
430  delete[] o2::gpu::resizeArray(mFlatBufferContainer, mFlatBufferSize, sz);
431  mFlatBufferPtr = mFlatBufferContainer;
432  mFlatBufferSize = sz;
433  int nLr = getNLayers();
434 
435  auto offs = alignSize(sizeof(MatLayerCylSetLayout), getBufferAlignmentBytes()); // account for the alignment
436  // move array of layer pointers to the flat array
437  auto* oldLayers = o2::gpu::resizeArray(get()->mLayers, nLr, nLr, (MatLayerCyl*)(mFlatBufferPtr + offs));
438  // dynamyc buffers of old layers were used in new ones, detach them
439  for (int i = nLr; i--;) {
440  oldLayers[i].clearInternalBufferPtr();
441  }
442  delete[] oldLayers;
443  offs = alignSize(offs + nLr * sizeof(MatLayerCyl), MatLayerCyl::getClassAlignmentBytes()); // account for the alignment
444 
445  // move array of R2 boundaries to the flat array
446  delete[] o2::gpu::resizeArray(get()->mR2Intervals, nLr + 1, nLr + 1, (float*)(mFlatBufferPtr + offs));
447  offs = alignSize(offs + (nLr + 1) * sizeof(float), getBufferAlignmentBytes()); // account for the alignment
448 
449  // move array of R2 boundaries to the flat array
450  delete[] o2::gpu::resizeArray(get()->mInterval2LrID, nLr, nLr, (int*)(mFlatBufferPtr + offs));
451  offs = alignSize(offs + nLr * sizeof(int), getBufferAlignmentBytes()); // account for the alignment
452 
453  for (int il = 0; il < nLr; il++) {
454  MatLayerCyl& lr = get()->mLayers[il];
455  lr.flatten(mFlatBufferPtr + offs);
456  offs = alignSize(offs + lr.getFlatBufferSize(), getBufferAlignmentBytes()); // account for the alignment
457  }
458  mConstructionMask = Constructed;
459 }
