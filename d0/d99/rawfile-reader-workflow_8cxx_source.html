42  options.push_back(ConfigParamSpec{"configKeyValues", VariantType::String, "", {"semicolon separated key=value strings"}});
43  options.push_back(ConfigParamSpec{"hbfutils-config", VariantType::String, std::string(o2::base::NameConf::DIGITIZATIONCONFIGFILE), {"configKeyValues ini file for HBFUtils (used if exists)"}});
44  // options for error-check suppression
45

64  rinp.bufferSize = uint64_t(configcontext.options().get<int64_t>("buffer-size"));
65  rinp.spSize = uint64_t(configcontext.options().get<int64_t>("super-page-size"));
66  rinp.partPerSP = configcontext.options().get<bool>("part-per-sp");
67  rinp.cache = configcontext.options().get<bool>("cache-data");
68  rinp.autodetectTF0 = configcontext.options().get<bool>("detect-tf0");
69  rinp.preferCalcTF = configcontext.options().get<bool>("calculate-tf-start");
70  rinp.rawChannelConfig = configcontext.options().get<std::string>("raw-channel-config");
71  rinp.delay_us = uint32_t(1e6 * configcontext.options().get<float>("delay")); // delay in microseconds
72  rinp.dropTF = configcontext.options().get<std::string>("drop-tf");
73  rinp.verbosity = configcontext.options().get<int>("verbosity-level");
74  rinp.errMap = 0;
75  for (int i = RawFileReader::NErrorsDefined; i--;) {
76  auto ei = RawFileReader::ErrTypes(i);
77  bool defOpt = RawFileReader::ErrCheckDefaults[i];
78  if (configcontext.options().get<bool>(RawFileReader::nochk_opt(ei).c_str()) ? !defOpt : defOpt) { // cmdl option inverts default!
79  rinp.errMap |= 0x1 << i;
80  }
81  }
82  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
83  auto hbfini = configcontext.options().get<std::string>("hbfutils-config");
84  if (!hbfini.empty() && o2::utils::Str::pathExists(hbfini)) {
85  o2::conf::ConfigurableParam::updateFromFile(hbfini, "HBFUtils", true); // update only those values which were not touched yet (provenance == kCODE)
86  }
87  return std::move(o2::raw::getRawFileReaderWorkflow(rinp));
88 }
