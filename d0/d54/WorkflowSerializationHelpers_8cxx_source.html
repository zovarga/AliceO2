240  bool StartObject()
241  {
242  enter("START_OBJECT");
243  if (in(State::IN_START)) {
244  push(State::IN_EXECUTION);
245  } else if (in(State::IN_DATAPROCESSORS)) {
246  push(State::IN_DATAPROCESSOR);
247  dataProcessors.push_back(DataProcessorSpec{});
248  } else if (in(State::IN_DATAPROCESSOR)) {
249  dataProcessors.push_back(DataProcessorSpec{});
250  } else if (in(State::IN_INPUTS)) {
251  push(State::IN_INPUT);
252  inputHasDescription = false;
253  inputHasSubSpec = false;
254  } else if (in(State::IN_OUTPUTS)) {
255  push(State::IN_OUTPUT);
256  outputHasSubSpec = false;
257  } else if (in(State::IN_OPTIONS)) {
258  push(State::IN_OPTION);
259  } else if (in(State::IN_INPUT_OPTIONS)) {
260  push(State::IN_OPTION);
261  } else if (in(State::IN_OUTPUT_OPTIONS)) {
262  push(State::IN_OPTION);
263  } else if (in(State::IN_WORKFLOW_OPTIONS)) {
264  push(State::IN_OPTION);
265  } else if (in(State::IN_METADATA)) {
266  push(State::IN_METADATUM);
267  metadata.push_back(DataProcessorInfo{});
268  } else if (in(State::IN_METADATUM)) {
269  metadata.push_back(DataProcessorInfo{});
270  } else if (in(State::IN_COMMAND)) {
271  command = CommandInfo{};
272  }
273  return true;
274  }

276  bool EndObject(SizeType memberCount)
277  {
278  enter("END_OBJECT");
279  if (in(State::IN_INPUT)) {
280  if (!inputHasDescription && !inputHasSubSpec) {
281 
282  DataDescriptorMatcher expectedMatcher00{
283  DataDescriptorMatcher::Op::And,
284  OriginValueMatcher{origin.str},
285  std::make_unique<DataDescriptorMatcher>(
286  DataDescriptorMatcher::Op::And,
287  DescriptionValueMatcher{ContextRef{1}},
288  std::make_unique<DataDescriptorMatcher>(
289  DataDescriptorMatcher::Op::And,
290  SubSpecificationTypeValueMatcher{ContextRef{2}},
291  std::make_unique<DataDescriptorMatcher>(DataDescriptorMatcher::Op::Just,
292  StartTimeValueMatcher{ContextRef{0}})))};
293 
294  dataProcessors.back().inputs.push_back(InputSpec({binding}, std::move(expectedMatcher00)));
295  } else if (inputHasDescription) {
296  if (inputHasSubSpec) {
297  dataProcessors.back().inputs.push_back(InputSpec(binding, origin, description, subspec, lifetime, inputOptions));
298  } else {
299  dataProcessors.back().inputs.push_back(InputSpec(binding, {origin, description}, lifetime, inputOptions));
300  }
301  } else {
302  LOG(error) << "Input w/o description but with subspec is not supported";
303  }
304 
305  inputOptions.clear();
306  inputHasDescription = false;
307  inputHasSubSpec = false;
308  } else if (in(State::IN_OUTPUT)) {
309  if (outputHasSubSpec) {
310  dataProcessors.back().outputs.push_back(OutputSpec({binding}, origin, description, subspec, lifetime));
311  } else {
312  dataProcessors.back().outputs.push_back(OutputSpec({binding}, {origin, description}, lifetime));
313  }
314  outputHasSubSpec = false;
315  } else if (in(State::IN_OPTION)) {
316  std::unique_ptr<ConfigParamSpec> opt{nullptr};
317 
318  using HelpString = ConfigParamSpec::HelpString;
319  std::stringstream is;
320  is.str(optionDefault);
321  switch (optionType) {
322  case VariantType::String:
323  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, optionDefault.c_str(), HelpString{optionHelp}, optionKind);
324  break;
325  case VariantType::Int:
326  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, std::stoi(optionDefault, nullptr), HelpString{optionHelp}, optionKind);
327  break;
328  case VariantType::UInt8:
329  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, static_cast<uint8_t>(std::stoul(optionDefault, nullptr)), HelpString{optionHelp}, optionKind);
330  break;
331  case VariantType::UInt16:
332  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, static_cast<uint16_t>(std::stoul(optionDefault, nullptr)), HelpString{optionHelp}, optionKind);
333  break;
334  case VariantType::UInt32:
335  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, static_cast<uint32_t>(std::stoul(optionDefault, nullptr)), HelpString{optionHelp}, optionKind);
336  break;
337  case VariantType::UInt64:
338  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, std::stoul(optionDefault, nullptr), HelpString{optionHelp}, optionKind);
339  break;
340  case VariantType::Int64:
341  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, std::stol(optionDefault, nullptr), HelpString{optionHelp}, optionKind);
342  break;
343  case VariantType::Float:
344  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, std::stof(optionDefault, nullptr), HelpString{optionHelp}, optionKind);
345  break;
346  case VariantType::Double:
347  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, std::stod(optionDefault, nullptr), HelpString{optionHelp}, optionKind);
348  break;
349  case VariantType::Bool:
350  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, (bool)std::stoi(optionDefault, nullptr), HelpString{optionHelp}, optionKind);
351  break;
352  case VariantType::ArrayInt:
353  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::ArrayInt>(is), HelpString{optionHelp}, optionKind);
354  break;
355  case VariantType::ArrayFloat:
356  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::ArrayFloat>(is), HelpString{optionHelp}, optionKind);
357  break;
358  case VariantType::ArrayDouble:
359  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::ArrayDouble>(is), HelpString{optionHelp}, optionKind);
360  break;
361  // case VariantType::ArrayBool:
362  // opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::ArrayBool>(is), HelpString{optionHelp}, optionKind);
363  // break;
364  case VariantType::ArrayString:
365  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::ArrayString>(is), HelpString{optionHelp}, optionKind);
366  break;
367  case VariantType::Array2DInt:
368  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::Array2DInt>(is), HelpString{optionHelp}, optionKind);
369  break;
370  case VariantType::Array2DFloat:
371  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::Array2DFloat>(is), HelpString{optionHelp}, optionKind);
372  break;
373  case VariantType::Array2DDouble:
374  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::Array2DDouble>(is), HelpString{optionHelp}, optionKind);
375  break;
376  case VariantType::LabeledArrayInt:
377  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::LabeledArrayInt>(is), HelpString{optionHelp}, optionKind);
378  break;
379  case VariantType::LabeledArrayFloat:
380  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::LabeledArrayFloat>(is), HelpString{optionHelp}, optionKind);
381  break;
382  case VariantType::LabeledArrayDouble:
383  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, VariantJSONHelpers::read<VariantType::LabeledArrayDouble>(is), HelpString{optionHelp}, optionKind);
384  break;
385  case VariantType::Dict:
386  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, emptyDict(), HelpString{optionHelp}, optionKind);
387  break;
388  default:
389  opt = std::make_unique<ConfigParamSpec>(optionName, optionType, optionDefault, HelpString{optionHelp}, optionKind);
390  }
391  // Depending on the previous state, push options to the right place.
392  if (previousIs(State::IN_OPTIONS)) {
393  dataProcessors.back().options.push_back(*opt);
394  } else if (previousIs(State::IN_WORKFLOW_OPTIONS)) {
395  metadata.back().workflowOptions.push_back(*opt);
396  } else if (previousIs(State::IN_INPUT_OPTIONS)) {
397  inputOptions.push_back(*opt);
398  } else if (previousIs(State::IN_OUTPUT_OPTIONS)) {
399  outputOptions.push_back(*opt);
400  } else {
401  assert(false);
402  }
403  }
404  pop();
405  return true;
406  }

531  bool String(const Ch* str, SizeType length, bool copy)
532  {
533  enter("STRING");
534  enter(str);
535  auto s = std::string(str, length);
536  if (in(State::IN_DATAPROCESSOR_NAME)) {
537  assert(dataProcessors.size());
538  dataProcessors.back().name = s;

688 bool WorkflowSerializationHelpers::import(std::istream& s,
689  std::vector<DataProcessorSpec>& workflow,
690  std::vector<DataProcessorInfo>& metadata,
691  CommandInfo& command)
692 {
693  // Skip any line which does not start with '{'
694  // If we do not find a starting {, we simply assume that no workflow
695  // was actually passed on the PIPE.
696  // FIXME: not particularly resilient, but works for now.
697  // FIXME: this will fail if { is found at char 1024.
698  char buf[1024];
699  bool hasFatalImportError = false;
700  while (s.peek() != '{') {
701  if (s.eof()) {
702  return !hasFatalImportError;
703  }
704  if (s.fail() || s.bad()) {
705  throw std::runtime_error("Malformatted input workflow");
706  }
707  s.getline(buf, 1024, '\n');
708  // FairLogger messages (starting with [) simply get forwarded.
709  // Other messages we consider them as ERRORs since they
710  // were printed out without FairLogger.
711  if (buf[0] == '[') {
712  if (strncmp(buf, "[ERROR] invalid workflow in", strlen("[ERROR] invalid workflow in")) == 0 ||
713  strncmp(buf, "[ERROR] error while setting up workflow", strlen("[ERROR] error while setting up workflow")) == 0 ||
714  strncmp(buf, "[ERROR] error parsing options of", strlen("[ERROR] error parsing options of")) == 0) {
715  hasFatalImportError = true;
716  }
717  std::cout << buf << std::endl;
718  } else {
719  LOG(error) << buf;
720  }
721  }
722  if (hasFatalImportError) {
723  return false;
724  }
725  rapidjson::Reader reader;
726  rapidjson::IStreamWrapper isw(s);
727  WorkflowImporter importer{workflow, metadata, command};
728  bool ok = reader.Parse(isw, importer);
729  if (ok == false) {
730  throw std::runtime_error("Error while parsing serialised workflow");
731  }
732  return true;
733 }
