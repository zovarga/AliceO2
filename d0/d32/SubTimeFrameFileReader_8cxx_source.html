75 std::size_t SubTimeFrameFileReader::getHeaderStackSize() // throws ios_base::failure
76 {
77  // Expect valid Stack in the file.
78  // First Header must be DataHeader. The size is unknown since there are multiple versions.
79  // Each header in the stack extends BaseHeader
80 
81  // Read first the base header then the rest of the extended header. Keep going until the next flag is set.
82  // reset the file pointer to the original incoming position, so the complete Stack can be read in
83 
84  bool readNextHeader = true;
85  std::size_t lStackSize = 0;
86  DataHeader lBaseHdr; // Use DataHeader since the BaseHeader has no default contructor.
87 
88  const auto lFilePosStart = position();
89 
90  const auto cMaxHeaders = 16; /* make sure we don't loop forever */
91  auto lNumHeaders = 0;
92  while (readNextHeader && (++lNumHeaders <= cMaxHeaders)) {
93  // read BaseHeader only!
94  const auto lBaseHdrPos = position();
95  if (!read_advance(&lBaseHdr, sizeof(BaseHeader))) {
96  return 0;
97  }
98 
99  // go back, and read the whole O2 header (Base+Derived)
100  set_position(lBaseHdrPos);
101  if (!ignore_nbytes(lBaseHdr.size())) {
102  return 0;
103  }
104 
105  lStackSize += lBaseHdr.size();
106  readNextHeader = (lBaseHdr.next() != nullptr);
107  }
108  // reset the file pointer
109  set_position(lFilePosStart);
110 
111  if (lNumHeaders >= cMaxHeaders) {
112  LOGP(error, "FileReader: Reached max number of headers allowed: {}.", cMaxHeaders);
113  return 0;
114  }
115 
116  return lStackSize;
117 }

119 Stack SubTimeFrameFileReader::getHeaderStack(std::size_t& pOrigsize)
120 {
121  const auto lStackSize = getHeaderStackSize();
122  pOrigsize = lStackSize;
123 
124  if (lStackSize < sizeof(BaseHeader)) {
125  // error in the stream
126  pOrigsize = 0;
127  return Stack{};
128  }
129 
130  std::byte* lStackMem = reinterpret_cast<std::byte*>(peek());
131  if (!ignore_nbytes(lStackSize)) {
132  // error in the stream
133  pOrigsize = 0;
134  return Stack{};
135  }
136 
137  // This must handle different versions of DataHeader
138  // check if DataHeader needs an upgrade by looking at the version number
139  const BaseHeader* lBaseOfDH = BaseHeader::get(lStackMem);
140  if (!lBaseOfDH) {
141  return Stack{};
142  }
143 
144  if (lBaseOfDH->headerVersion < DataHeader::sVersion) {
145  DataHeader lNewDh;
146 
147  // Write over the new DataHeader. We need to update some of the BaseHeader values.
148  assert(sizeof(DataHeader) > lBaseOfDH->size()); // current DataHeader must be larger
149  std::memcpy(&lNewDh, (void*)lBaseOfDH->data(), lBaseOfDH->size());
150 
151  // make sure to bump the version in the BaseHeader.
152  // TODO: Is there a better way?
153  lNewDh.headerSize = sizeof(DataHeader);
154  lNewDh.headerVersion = DataHeader::sVersion;
155 
156  if (lBaseOfDH->headerVersion == 1 || lBaseOfDH->headerVersion == 2) {
157  /* nothing to do for the upgrade */
158  } else {
159  LOGP(error, "FileReader: DataHeader v{} read from file is not upgraded to the current version {}",
160  lBaseOfDH->headerVersion, DataHeader::sVersion);
161  LOGP(error, "Try using a newer version of DataDistribution or file a BUG");
162  }
163 
164  if (lBaseOfDH->size() == lStackSize) {
165  return Stack(lNewDh);
166  } else {
167  assert(lBaseOfDH->size() < lStackSize);
168 
169  return Stack(
170  lNewDh,
171  Stack(lStackMem + lBaseOfDH->size()));
172  }
173  }
174 
175  return Stack(lStackMem);
176 }

180 std::unique_ptr<MessagesPerRoute> SubTimeFrameFileReader::read(FairMQDevice* device, const std::vector<o2f::OutputRoute>& outputRoutes,
181  const std::string& rawChannel, int verbosity)
182 {
183  std::unique_ptr<MessagesPerRoute> messagesPerRoute = std::make_unique<MessagesPerRoute>();
184  auto& msgMap = *messagesPerRoute.get();
185  assert(device);
186  std::unordered_map<o2::header::DataHeader, std::pair<std::string, bool>> channelsMap;
187  auto findOutputChannel = [&outputRoutes, &rawChannel, &channelsMap](const o2::header::DataHeader* h) -> const std::string& {
188  if (!rawChannel.empty()) {
189  return rawChannel;
190  }
191  auto& chFromMap = channelsMap[*h];
192  if (chFromMap.first.empty() && !chFromMap.second) { // search for channel which is enountered for the 1st time
193  chFromMap.second = true; // flag that it was already checked
194  for (auto& oroute : outputRoutes) {
195  LOG(debug) << "comparing with matcher to route " << oroute.matcher << " TSlice:" << oroute.timeslice;
196  if (o2f::DataSpecUtils::match(oroute.matcher, h->dataOrigin, h->dataDescription, h->subSpecification) && ((h->tfCounter % oroute.maxTimeslices) == oroute.timeslice)) {
197  LOG(debug) << "picking the route:" << o2f::DataSpecUtils::describe(oroute.matcher) << " channel " << oroute.channel;
198  chFromMap.first = oroute.channel;
199  break;
200  }
201  }
202  }
203  return chFromMap.first;
204  };

206  auto addPart = [&msgMap](FairMQMessagePtr hd, FairMQMessagePtr pl, const std::string& fairMQChannel) {
207  FairMQParts* parts = nullptr;
208  parts = msgMap[fairMQChannel].get(); // FairMQParts*
209  if (!parts) {
210  msgMap[fairMQChannel] = std::make_unique<FairMQParts>();
211  parts = msgMap[fairMQChannel].get();
212  }
213  parts->AddPart(std::move(hd));
214  parts->AddPart(std::move(pl));
215  };
216 
217  // record current position
218  const auto lTfStartPosition = position();
219 
220  if (lTfStartPosition == size() || !mFileMap.is_open() || eof()) {
221  return nullptr;
222  }
223  auto tfID = sStfId++;
224  std::size_t lMetaHdrStackSize = 0;
225  const DataHeader* lStfMetaDataHdr = nullptr;
226  SubTimeFrameFileMeta lStfFileMeta;
227 
228  auto printStack = [tfID](const o2::header::Stack& st) {
229  auto dph = o2::header::get<o2f::DataProcessingHeader*>(st.data());
230  auto dh = o2::header::get<o2::header::DataHeader*>(st.data());
231  LOGP(info, "TF#{} Header for {}/{}/{} @ timeslice {} run {} | {} of {} size {}, TForbit {} | DPH: {}/{}/{}", tfID,
232  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->runNumber,
233  dh->splitPayloadIndex, dh->splitPayloadParts, dh->payloadSize, dh->firstTForbit,
234  dph ? dph->startTime : 0, dph ? dph->duration : 0, dph ? dph->creation : 0);
235  };
236 
237  // Read DataHeader + SubTimeFrameFileMeta
238  auto lMetaHdrStack = getHeaderStack(lMetaHdrStackSize);
239  if (lMetaHdrStackSize == 0) {
240  LOG(error) << "Failed to read the TF file header. The file might be corrupted.";
241  mFileMap.close();
242  return nullptr;
243  }
244 
245  lStfMetaDataHdr = o2::header::DataHeader::Get(lMetaHdrStack.first());
246  if (!read_advance(&lStfFileMeta, sizeof(SubTimeFrameFileMeta))) {
247  return nullptr;
248  }
249 
250  // verify we're actually reading the correct data in
251  if (!(SubTimeFrameFileMeta::getDataHeader().dataDescription == lStfMetaDataHdr->dataDescription)) {
252  LOGP(warning, "Reading bad data: SubTimeFrame META header");
253  mFileMap.close();
254  return nullptr;
255  }
256 
257  // prepare to read the TF data
258  const auto lStfSizeInFile = lStfFileMeta.mStfSizeInFile;
259  if (lStfSizeInFile == (sizeof(DataHeader) + sizeof(SubTimeFrameFileMeta))) {
260  LOGP(warning, "Reading an empty TF from file. Only meta information present");
261  mFileMap.close();
262  return nullptr;
263  }
264 
265  // check there's enough data in the file
266  if ((lTfStartPosition + lStfSizeInFile) > this->size()) {
267  LOGP(warning, "Not enough data in file for this TF. Required: {}, available: {}", lStfSizeInFile, (this->size() - lTfStartPosition));
268  mFileMap.close();
269  return nullptr;
270  }
271 
272  // Index
273  std::size_t lStfIndexHdrStackSize = 0;
274  const DataHeader* lStfIndexHdr = nullptr;
275 
276  // Read DataHeader + SubTimeFrameFileMeta
277  auto lStfIndexHdrStack = getHeaderStack(lStfIndexHdrStackSize);
278  if (lStfIndexHdrStackSize == 0) {
279  mFileMap.close();
280  return nullptr;
281  }
282  lStfIndexHdr = o2::header::DataHeader::Get(lStfIndexHdrStack.first());
283  if (!lStfIndexHdr) {
284  LOG(error) << "Failed to read the TF index structure. The file might be corrupted.";
285  return nullptr;
286  }
287 
288  if (!ignore_nbytes(lStfIndexHdr->payloadSize)) {
289  return nullptr;
290  }
291 #ifdef _RUN_TIMING_MEASUREMENT_
292  TStopwatch readSW, findChanSW, msgSW, addPartSW;
293  findChanSW.Stop();
294  msgSW.Stop();
295  addPartSW.Stop();
296 #endif
297  // Remaining data size of the TF:
298  // total size in file - meta (hdr+struct) - index (hdr + payload)
299  const auto lStfDataSize = lStfSizeInFile - (lMetaHdrStackSize + sizeof(SubTimeFrameFileMeta)) - (lStfIndexHdrStackSize + lStfIndexHdr->payloadSize);
300 
301  std::int64_t lLeftToRead = lStfDataSize;
302  STFHeader stfHeader{tfID, -1u, -1u};
303 
304  // read <hdrStack + data> pairs
305  while (lLeftToRead > 0) {
306 
307  // allocate and read the Headers
308  std::size_t lDataHeaderStackSize = 0;
309  Stack lDataHeaderStack = getHeaderStack(lDataHeaderStackSize);
310  if (lDataHeaderStackSize == 0) {
311  mFileMap.close();
312  return nullptr;
313  }
314  const DataHeader* lDataHeader = o2::header::DataHeader::Get(lDataHeaderStack.first());
315  if (!lDataHeader) {
316  LOG(error) << "Failed to read the TF HBF DataHeader structure. The file might be corrupted.";
317  mFileMap.close();
318  return nullptr;
319  }
320 
321  o2::header::Stack headerStack{*lDataHeader, o2f::DataProcessingHeader{tfID, 1, lStfFileMeta.mWriteTimeMs}};
322  if (stfHeader.runNumber == -1) {
323  stfHeader.id = lDataHeader->tfCounter;
324  stfHeader.runNumber = lDataHeader->runNumber;
325  stfHeader.firstOrbit = lDataHeader->firstTForbit;
326  }
327 
328  const std::uint64_t lDataSize = lDataHeader->payloadSize;
329  // do we accept these data?
330  auto detOrigStatus = mDetOrigMap.find(lDataHeader->dataOrigin);
331  if (detOrigStatus != mDetOrigMap.end() && !detOrigStatus->second) { // this is a detector data and we don't want to read it
332  if (!ignore_nbytes(lDataSize)) {
333  return nullptr;
334  }
335  lLeftToRead -= (lDataHeaderStackSize + lDataSize); // update the counter
336  continue;
337  }
338 #ifdef _RUN_TIMING_MEASUREMENT_
339  findChanSW.Start(false);
340 #endif
341  const auto& fmqChannel = findOutputChannel(lDataHeader);
342 #ifdef _RUN_TIMING_MEASUREMENT_
343  findChanSW.Stop();
344 #endif
345  if (fmqChannel.empty()) { // no output channel
346  if (!ignore_nbytes(lDataSize)) {
347  return nullptr;
348  }
349  lLeftToRead -= (lDataHeaderStackSize + lDataSize); // update the counter
350  continue;
351  //mFileMap.close();
352  //return nullptr;
353  }
354  // read the data
355 
356  auto fmqFactory = device->GetChannel(fmqChannel, 0).Transport();
357 #ifdef _RUN_TIMING_MEASUREMENT_
358  msgSW.Start(false);
359 #endif
360  auto lHdrStackMsg = fmqFactory->CreateMessage(headerStack.size(), fair::mq::Alignment{64});
361  auto lDataMsg = fmqFactory->CreateMessage(lDataSize, fair::mq::Alignment{64});
362 #ifdef _RUN_TIMING_MEASUREMENT_
363  msgSW.Stop();
364 #endif
365  memcpy(lHdrStackMsg->GetData(), headerStack.data(), headerStack.size());
366 
367  if (!read_advance(lDataMsg->GetData(), lDataSize)) {
368  return nullptr;
369  }
370  if (verbosity > 0) {
371  if (verbosity > 1 || lDataHeader->splitPayloadIndex == 0) {
372  printStack(headerStack);
373  if (o2::raw::RDHUtils::checkRDH(lDataMsg->GetData()) && verbosity > 2) {
374  o2::raw::RDHUtils::printRDH(lDataMsg->GetData());
375  }
376  }
377  }
378 #ifdef _RUN_TIMING_MEASUREMENT_
379  addPartSW.Start(false);
380 #endif
381  addPart(std::move(lHdrStackMsg), std::move(lDataMsg), fmqChannel);
382 #ifdef _RUN_TIMING_MEASUREMENT_
383  addPartSW.Stop();
384 #endif
385  // update the counter
386  lLeftToRead -= (lDataHeaderStackSize + lDataSize);
387  }
388 
389  if (lLeftToRead < 0) {
390  LOG(error) << "FileRead: Read more data than it is indicated in the META header!";
391  return nullptr;
392  }
393 
394  // add TF acknowledge part
395  o2::header::DataHeader stfDistDataHeader(o2::header::gDataDescriptionDISTSTF, o2::header::gDataOriginFLP, 0, sizeof(STFHeader), 0, 1);
396  stfDistDataHeader.payloadSerializationMethod = o2::header::gSerializationMethodNone;
397  stfDistDataHeader.firstTForbit = stfHeader.firstOrbit;
398  stfDistDataHeader.runNumber = stfHeader.runNumber;
399  stfDistDataHeader.tfCounter = stfHeader.id;
400  stfHeader.id = tfID;
401  const auto fmqChannel = findOutputChannel(&stfDistDataHeader);
402  if (!fmqChannel.empty()) { // no output channel
403  auto fmqFactory = device->GetChannel(fmqChannel, 0).Transport();
404  o2::header::Stack headerStackSTF{stfDistDataHeader, o2f::DataProcessingHeader{tfID, 1, lStfFileMeta.mWriteTimeMs}};
405  if (verbosity > 0) {
406  printStack(headerStackSTF);
407  }
408  auto hdMessageSTF = fmqFactory->CreateMessage(headerStackSTF.size(), fair::mq::Alignment{64});
409  auto plMessageSTF = fmqFactory->CreateMessage(stfDistDataHeader.payloadSize, fair::mq::Alignment{64});
410  memcpy(hdMessageSTF->GetData(), headerStackSTF.data(), headerStackSTF.size());
411  memcpy(plMessageSTF->GetData(), &stfHeader, sizeof(STFHeader));
412 #ifdef _RUN_TIMING_MEASUREMENT_
413  addPartSW.Start(false);
414 #endif
415  addPart(std::move(hdMessageSTF), std::move(plMessageSTF), fmqChannel);
416 #ifdef _RUN_TIMING_MEASUREMENT_
417  addPartSW.Stop();
418 #endif
419  }
420 #ifdef _RUN_TIMING_MEASUREMENT_
421  readSW.Stop();
422  LOG(info) << "TF creation time: CPU: " << readSW.CpuTime() << " Wall: " << readSW.RealTime() << " s";
423  LOG(info) << "AddPart Timer CPU: " << addPartSW.CpuTime() << " Wall: " << addPartSW.RealTime() << " s";
424  LOG(info) << "CreMsg Timer CPU: " << msgSW.CpuTime() << " Wall: " << msgSW.RealTime() << " s";
425  LOG(info) << "FndChan Timer CPU: " << findChanSW.CpuTime() << " Wall: " << findChanSW.RealTime() << " s";
426 #endif
427  return messagesPerRoute;
428 }
