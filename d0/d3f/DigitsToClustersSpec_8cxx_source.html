59 // Splits a string in float array for string delimiter, TODO: Move this in a HMPID common library
60 void DigitsToClustersTask::strToFloatsSplit(std::string s, std::string delimiter, float* res, int maxElem)
61 {
62  int index = 0;
63  size_t pos_start = 0;
64  size_t pos_end;
65  size_t delim_len = delimiter.length();
66  std::string token;
67  while ((pos_end = s.find(delimiter, pos_start)) != std::string::npos) {
68  token = s.substr(pos_start, pos_end - pos_start);
69  pos_start = pos_end + delim_len;
70  res[index++] = std::stof(token);
71  if (index == maxElem) {
72  return;
73  }
74  }
75  res[index++] = (std::stof(s.substr(pos_start)));
76  return;
77 }

95 void DigitsToClustersTask::run(framework::ProcessingContext& pc)
96 {
97  LOG(info) << "[HMPID DClusterization - run() ] Enter ...";
98  std::vector<o2::hmpid::Trigger> triggers;
99  std::vector<o2::hmpid::Digit> digits;
100  std::vector<o2::hmpid::Digit> oneEventDigits;
101  std::vector<o2::hmpid::Cluster> oneEventClusters;
102  std::vector<o2::hmpid::Cluster> clusters;
103  std::vector<o2::hmpid::Trigger> clustersTrigger;
104 
105  for (auto const& ref : InputRecordWalker(pc.inputs())) {
106  if (DataRefUtils::match(ref, {"check", ConcreteDataTypeMatcher{gDataOriginHMP, "INTRECORDS"}})) {
107  triggers = pc.inputs().get<std::vector<o2::hmpid::Trigger>>(ref);
108  LOG(info) << "We receive triggers =" << triggers.size();
109  }
110  if (DataRefUtils::match(ref, {"check", ConcreteDataTypeMatcher{gDataOriginHMP, "DIGITS"}})) {
111  digits = pc.inputs().get<std::vector<o2::hmpid::Digit>>(ref);
112  LOG(info) << "The size of the vector =" << digits.size();
113  mDigitsReceived += digits.size();
114  }
115 
116  clustersTrigger.clear();
117  for (int i = 0; i < triggers.size(); i++) {
118  oneEventDigits.clear();
119  LOG(info) << "Trigger Event Orbit = " << triggers[i].getOrbit() << " BC = " << triggers[i].getBc();
120  for (int j = triggers[i].getFirstEntry(); j <= triggers[i].getLastEntry(); j++) {
121  oneEventDigits.push_back(digits[j]);
122  }
123  // ------ Now we have the Digit Array for one Trigger event
124  oneEventClusters.clear();
125  mRec->Dig2Clu(&oneEventDigits, &oneEventClusters, mSigmaCut, true);
126  if (oneEventClusters.size() > 0) { // we have something to store
127  clustersTrigger.push_back(triggers[i]); // copy the Interaction record in the new triggers vector
128  clustersTrigger.back().setDataRange(clusters.size(), oneEventClusters.size()); // set the data range to the chunch of clusters
129  for (int j = 0; j < oneEventClusters.size(); j++) {
130  clusters.push_back(oneEventClusters[j]);
131  } // append clusters
132  }
133  LOG(info) << "Clusters for event = " << oneEventClusters.size();
134  }
135 
136  pc.outputs().snapshot(o2::framework::Output{"HMP", "CLUSTERS", 0, o2::framework::Lifetime::Timeframe}, clusters);
137  pc.outputs().snapshot(o2::framework::Output{"HMP", "INTRECORDS1", 0, o2::framework::Lifetime::Timeframe}, clustersTrigger);
138  }
139 
140  mExTimer.elapseMes("Clusterization of Digits received = " + std::to_string(mDigitsReceived));
141  return;
142 }
