42 void GPUTPCGMMerger::DumpSliceTracks(std::ostream& out)
43 {
44  std::streamsize ss = out.precision();
45  out << std::setprecision(2);
46  out << "\nTPC Merger Slice Tracks\n";
47  trackOrder.resize(mSliceTrackInfoIndex[2 * NSLICES]);
48  trackOrderReverse.resize(mSliceTrackInfoIndex[2 * NSLICES]);
49  std::iota(trackOrder.begin(), trackOrder.end(), 0);
50  for (int iSlice = 0; iSlice < NSLICES; iSlice++) {
51  out << "Slice Track Info Index " << (mSliceTrackInfoIndex[iSlice + 1] - mSliceTrackInfoIndex[iSlice]) << " / " << (mSliceTrackInfoIndex[NSLICES + iSlice + 1] - mSliceTrackInfoIndex[NSLICES + iSlice]) << "\n";
52  for (int iGlobal = 0; iGlobal < 2; iGlobal++) {
53  std::sort(&trackOrder[mSliceTrackInfoIndex[iSlice + NSLICES * iGlobal]], &trackOrder[mSliceTrackInfoIndex[iSlice + NSLICES * iGlobal + 1]], [this](const int& aa, const int& bb) {
54  const GPUTPCGMSliceTrack& a = mSliceTrackInfos[aa];
55  const GPUTPCGMSliceTrack& b = mSliceTrackInfos[bb];
56  return (a.X() != b.X()) ? (a.X() < b.X()) : (a.Y() != b.Y()) ? (a.Y() < b.Y()) : (a.Z() < b.Z());
57  });
58  out << " Track type " << iGlobal << "\n";
59  for (int j = mSliceTrackInfoIndex[iSlice + NSLICES * iGlobal]; j < mSliceTrackInfoIndex[iSlice + NSLICES * iGlobal + 1]; j++) {
60  trackOrderReverse[trackOrder[j]] = j;
61  const auto& trk = mSliceTrackInfos[trackOrder[j]];
62  out << " Track " << j << ": X " << trk.X() << " A " << trk.Alpha() << " Y " << trk.Y() << " Z " << trk.Z() << " Phi " << trk.SinPhi() << " Tgl " << trk.DzDs() << " QPt " << trk.QPt() << "\n";
63  }
64  }
65  }
66  out << std::setprecision(ss);
67 }

98 void GPUTPCGMMerger::DumpCollected(std::ostream& out)
99 {
100  trackOrder.resize(mMemory->nOutputTracks);
101  trackOrderReverse.resize(mMemory->nOutputTracks);
102  std::iota(trackOrder.begin(), trackOrder.end(), 0);
103  std::sort(trackOrder.begin(), trackOrder.end(), [this](const int& aa, const int& bb) {
104  const GPUTPCGMMergedTrack& a = mOutputTracks[aa];
105  const GPUTPCGMMergedTrack& b = mOutputTracks[bb];
106  return (a.GetAlpha() != b.GetAlpha()) ? (a.GetAlpha() < b.GetAlpha()) : (a.GetParam().GetX() != b.GetParam().GetX()) ? (a.GetParam().GetX() < b.GetParam().GetX()) : (a.GetParam().GetY() != b.GetParam().GetY()) ? (a.GetParam().GetY() < b.GetParam().GetY()) : (a.GetParam().GetZ() < b.GetParam().GetZ());
107  });
108 
109  std::streamsize ss = out.precision();
110  out << std::setprecision(2);
111  out << "\nTPC Merger Collected Tracks\n";
112  for (unsigned int i = 0; i < mMemory->nOutputTracks; i++) {
113  const auto& trk = mOutputTracks[trackOrder[i]];
114  const auto& p = trk.GetParam();
115  out << " Track " << i << ": Loop " << trk.Looper() << " Alpha " << trk.GetAlpha() << " X " << p.GetX() << " Y " << p.GetY() << " Z " << p.GetZ() << " SPhi " << p.GetSinPhi() << " Tgl " << p.GetDzDs() << " QPt " << p.GetQPt() << " NCl " << trk.NClusters() << "\n";
