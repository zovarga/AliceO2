115 {
117  for (auto loc : locs) {
118  // The board patterns coincide with the masks ("overwritten" mode)

141 bool GBTRawDataChecker::checkRegLocConsistency(const std::vector<ROBoard>& regs, const std::vector<ROBoard>& locs, const InteractionRecord& ir)
142 {
144  uint8_t regFired{0};
145  bool isTrig = false;
146  for (auto& reg : regs) {
147  uint8_t ireg = raw::getLocId(reg.boardId) % 2;
148  if (reg.triggerWord == 0) {
149  // Self-triggered event: check the decision
150  regFired |= (reg.firedChambers << (4 * ireg));
151  } else {
152  // Triggered event: all active boards must answer
153  regFired |= (mCrateMask & (0xF << (4 * ireg)));
154  isTrig = true;
155  }
156  }
157  uint8_t locFired{0};
158  for (auto& loc : locs) {
159  auto linkId = getElinkId(loc);
160  uint8_t mask = (1 << linkId);
161  if (loc.triggerWord == 0) {
162  // Self-triggered event: check the decision
163  if (loc.firedChambers) {
164  locFired |= mask;
165  }
166  } else {
167  // Triggered event: all active boards must answer
168  locFired |= mask;
169  isTrig = true;
170  }
171  }
172 
173  // The XOR returns 1 in case of a difference
174  uint8_t problems = (regFired ^ locFired);
175 
176  if (problems) {
177  // It can be that a busy signal was raised by one of the board in previous events
178  // If the board is still busy it will not answer.
179  uint8_t busy{0};
180  for (uint8_t iboard = 0; iboard < crateparams::sNELinksPerGBT; ++iboard) {
181  auto rawIr = getRawIR(iboard, isTrig, ir);
182  bool isBusy = false;
183  for (auto& busyInfo : mBusyPeriods[iboard]) {
184  if (busyInfo.interactionRecord > rawIr) {
185  break;
186  }
187  isBusy = busyInfo.isBusy;
188  }
189  if (isBusy) {
190  busy |= (iboard < crateparams::sMaxNBoardsInLink) ? (1 << iboard) : (0xF << (4 * (iboard % 2)));
191  }
192  }
193 
194  if (problems & ~busy) {
195  std::stringstream ss;
196  ss << fmt::format("loc-reg inconsistency: fired locals ({:08b}) != expected from reg ({:08b});\n", locFired, regFired);
197  ss << printBoards(regs);
198  ss << printBoards(locs);
199  mEventDebugMsg += ss.str();
200  return false;
201  }
202  }
203  return true;
204 }

267 void GBTRawDataChecker::clearChecked(bool isTriggered, bool clearTrigEvents)
268 {
270 
271  auto& boards = isTriggered ? mBoardsTrig : mBoardsSelfTrig;
272  auto& lastIndexes = isTriggered ? mLastIndexTrig : mLastIndexSelfTrig;
273  // Create a new board map with the checked events stripped
274  for (auto& lastIdxItem : lastIndexes) {
275  auto firstIdx = lastIdxItem.second + 1;
276  auto& boardVec = boards[lastIdxItem.first];
277  boards[lastIdxItem.first].erase(boardVec.begin(), boardVec.begin() + firstIdx);
278  }
279 
280  if (clearTrigEvents) {
281  // Clears the map with the processed triggers
282  auto& lastCompleteTrigIR = isTriggered ? mLastCompleteIRTrig : mLastCompleteIRSelfTrig;
283  auto up = mTrigEvents.upper_bound(lastCompleteTrigIR);
284  mTrigEvents.erase(mTrigEvents.begin(), up);
285  for (auto& busyInfoVec : mBusyPeriods) {
286  auto upBusy = std::upper_bound(busyInfoVec.begin(), busyInfoVec.end(), lastCompleteTrigIR, [](const InteractionRecord& ir, const BusyInfo& busyInfo) { return ir <= busyInfo.interactionRecord; });
287  if (upBusy != busyInfoVec.begin()) {
288  --upBusy;
289  }
290  busyInfoVec.erase(busyInfoVec.begin(), upBusy);
291  }

295 bool GBTRawDataChecker::isCompleteSelfTrigEvent(const o2::InteractionRecord& ir) const
296 {
298 
299  // The regional board information in self-triggered events is delayed
300  // compared to triggered events.
301  // So, we expect information from a previous orbit after having received an orbit trigger.
302  // Let us check that we have all boards with the same orbit
303 
304  bool isIncluded = false;
305  for (uint8_t ireg = 8; ireg < crateparams::sNELinksPerGBT; ++ireg) {
306  auto& boards = mBoardsSelfTrig[ireg];
307  if (!boards.empty()) {
308  if (boards.back().interactionRecord.orbit == ir.orbit) {
309  return false;
310  }
311  if (boards.front().interactionRecord.orbit <= ir.orbit) {
312  isIncluded = true;
313  }
314  }
315  }
316  return isIncluded;
317 }

319 unsigned int GBTRawDataChecker::getLastCompleteTrigEvent()
320 {
324 
325  // The information for an event comes at different times for different boards,
326  // depending on the length of the self-triggered event for that board.
327  // So, before testing the consistency of the event,
328  // we must wait to have received the information from all boards.
329  // This can be checked in triggered events, since all boards should be present.
330 
331  unsigned int completeMask = 0;
332 
333  // Check if we have a triggered event with the information from all active boards
334  mLastCompleteIRSelfTrig = o2::InteractionRecord();
335  uint16_t fullMask = (3 << 8) | mCrateMask;
336  auto trigEventIt = mTrigEvents.rbegin();
337  auto end = mTrigEvents.rend();
338  for (; trigEventIt != end; ++trigEventIt) {
339  if ((trigEventIt->second & fullMask) == fullMask) {
340  // The trigger events contain the unprocessed events for both triggered and self-triggered events
341  // These might not be synchronized (typically the latest complete self-triggered events lie behind)
342  // If the latest IR in memory is more recent than the current complete event found,
343  // then it means that we need to wait for more HBs.
344  if (mLastCompleteIRTrig.isDummy() || mLastCompleteIRTrig < trigEventIt->first) {
345  completeMask |= 1;
346  mLastCompleteIRTrig = trigEventIt->first;
347  }
348  auto trIt = trigEventIt;
349  while (trIt != end) {
350  if (isCompleteSelfTrigEvent(trIt->first)) {
351  completeMask |= (1 << 1);
352  mLastCompleteIRSelfTrig = trIt->first;
353  break;
354  }
355  ++trIt;
356  }
357 
358  return completeMask;
359  }
360  }
361 
362  return completeMask;
363 }

365 bool GBTRawDataChecker::runCheckEvents(unsigned int completeMask)
366 {
368 
369  bool isOk = true;
370 
371  if (completeMask & 0x1) {
372  sortEvents(true);
373  isOk &= checkEvents(true);
374  // This is needed to clear vectors in runs with no self-triggered events
375  bool clearTrigger = true;
376  for (auto infos : mBoardsSelfTrig) {
377  if (!infos.empty()) {
378  clearTrigger = false;
379  break;
380  }
381  }
382  clearChecked(true, clearTrigger);
383  }
384 
385  if (completeMask & 0x2) {
386  sortEvents(false);
387  isOk &= checkEvents(false);
388  clearChecked(false, true);
389  }
390 
391  return isOk;
392 }

394 void GBTRawDataChecker::sortEvents(bool isTriggered)
395 {
397  auto& orderedIndexes = isTriggered ? mOrderedIndexesTrig : mOrderedIndexesSelfTrig;
398  auto& lastIndexes = isTriggered ? mLastIndexTrig : mLastIndexSelfTrig;
399  auto& boards = isTriggered ? mBoardsTrig : mBoardsSelfTrig;
400  auto& lastCompleteTrigEventIR = isTriggered ? mLastCompleteIRTrig : mLastCompleteIRSelfTrig;
401  orderedIndexes.clear();
402  lastIndexes.clear();
403  for (uint8_t ilink = 0; ilink < crateparams::sNELinksPerGBT; ++ilink) {
404  long int lastIdx = -1;
405  for (auto boardIt = boards[ilink].begin(), end = boards[ilink].end(); boardIt != end; ++boardIt) {
406  if (boardIt->interactionRecord > lastCompleteTrigEventIR) {
407  break;
408  }
409  lastIdx = std::distance(boards[ilink].begin(), boardIt);
410  orderedIndexes[boardIt->interactionRecord.toLong()].emplace_back(ilink, lastIdx);
411  }
412  lastIndexes[ilink] = lastIdx;
413  }
414 }

416 bool GBTRawDataChecker::checkEvents(bool isTriggered)
417 {
419  bool isOk = true;
420  auto& boards = isTriggered ? mBoardsTrig : mBoardsSelfTrig;
421  auto& orderedIndexes = isTriggered ? mOrderedIndexesTrig : mOrderedIndexesSelfTrig;
422  // Loop on the event indexes
423  o2::InteractionRecord ir;
424  for (auto& evtIdxItem : orderedIndexes) {
425  // All of these boards have the same timestamp
426  GBT gbtEvent;
427  // bool busyRaised = false;
428  for (auto& evtPair : evtIdxItem.second) {
429  auto& boardInfo = boards[evtPair.first][evtPair.second];
430  uint8_t triggerId = boardInfo.board.triggerWord;
431  auto elinkId = getElinkId(boardInfo.board);
432 
433  if (raw::isLoc(boardInfo.board.statusWord)) {
434  gbtEvent.locs.push_back(boardInfo.board);
435  } else {
436  gbtEvent.regs.push_back(boardInfo.board);
437  }
438  if (boardInfo.page >= 0) {
439  if (std::find(gbtEvent.pages.begin(), gbtEvent.pages.end(), boardInfo.page) == gbtEvent.pages.end()) {
440  gbtEvent.pages.push_back(boardInfo.page);
441  }
442  }
443  }
444  ++mStatistics[0];
445  ir.setFromLong(evtIdxItem.first);
446  if (!checkEvent(isTriggered, gbtEvent.regs, gbtEvent.locs, ir)) {
447  std::stringstream ss;
448  ss << fmt::format("BCid: 0x{:x} Orbit: 0x{:x}", ir.bc, ir.orbit);
449  if (!gbtEvent.pages.empty()) {
450  ss << " [in";
451  for (auto& page : gbtEvent.pages) {
452  ss << std::dec << " page: " << page << " (line: " << 512 * page + 1 << ") ";
453  }
454  ss << "]";
455  }
456  ss << "\n";
457  isOk = false;
458  ss << mEventDebugMsg << "\n";
459  mDebugMsg += ss.str();
460  ++mStatistics[1];
461  }
462  }
463 
464  return isOk;
465 }

467 bool GBTRawDataChecker::process(gsl::span<const ROBoard> localBoards, gsl::span<const ROFRecord> rofRecords, gsl::span<const ROFRecord> pageRecords)
468 {
470  mDebugMsg.clear();
471 
472  // Fill board information
473  for (auto rofIt = rofRecords.begin(); rofIt != rofRecords.end(); ++rofIt) {
474  if (rofIt->interactionRecord.orbit == 0xffffffff) {
475  // Protection for event with orbit 0
476  continue;
477  }
478  for (auto locIt = localBoards.begin() + rofIt->firstEntry; locIt != localBoards.begin() + rofIt->firstEntry + rofIt->nEntries; ++locIt) {
479  // Find what page this event corresponds to.
480  // This is useful for debugging.
481  long int page = -1;
482  for (auto& rofPage : pageRecords) {
483  if (rofIt->firstEntry >= rofPage.firstEntry && rofIt->firstEntry < rofPage.firstEntry + rofPage.nEntries) {
484  page = rofPage.interactionRecord.orbit;
485  break;
486  }
487  }
488 
489  // Store the information per local board.
490  // The information should be already ordered in time
491  auto id = getElinkId(*locIt);
492  auto& elinkVec = (locIt->triggerWord == 0) ? mBoardsSelfTrig[id] : mBoardsTrig[id];
493  elinkVec.push_back({*locIt, rofIt->interactionRecord, page});
494 
495  auto& busyVec = mBusyPeriods[id];
496  bool wasBusy = !busyVec.empty() && busyVec.back().isBusy;
497  bool isBusy = locIt->statusWord & raw::sREJECTING;
498  if (isBusy != wasBusy) {
499  if (isBusy) {
500  ++mStatistics[2];
501  }
502  busyVec.push_back({isBusy, getRawIR(id, locIt->triggerWord != 0, rofIt->interactionRecord)});
503  }
504 
505  if (locIt->triggerWord == 0) {
506  continue;
507  }
508 
509  // Keep track of the trigger chosen for synchronisation
510  if (locIt->triggerWord & mSyncTrigger) {
511  mTrigEvents[rofIt->interactionRecord] |= (1 << id);
512  }
513  if (locIt->triggerWord & raw::sORB) {
514  mResetVal = rofIt->interactionRecord.bc + 1;
515  }
516 
517  // Compute the masks
518  if (locIt->triggerWord & raw::sSOX) {
519  if (raw::isLoc(locIt->statusWord)) {
520  auto maskItem = mMasks.find(locIt->boardId);
521  // Check if we have already a mask for this
522  if (maskItem == mMasks.end()) {
523  // If not, read the map
524  auto& mask = mMasks[locIt->boardId];
525  for (int ich = 0; ich < 4; ++ich) {
526  mask.patternsBP[ich] = locIt->patternsBP[ich];
527  mask.patternsNBP[ich] = locIt->patternsNBP[ich];
528  }
529  }
530  }
531  }
532  } // loop on local boards
533  } // loop on ROF records
534 
535  return runCheckEvents(getLastCompleteTrigEvent());
536 }
