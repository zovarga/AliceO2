228 template <typename T, typename Iterator, typename BinCenterView>
229 std::vector<double> fitGaus(Iterator first, Iterator last, BinCenterView axisfirst, const bool ignoreUnderOverflowBin = true)
230 {
231 
232  if (ignoreUnderOverflowBin) {
233  first++;
234  last--;
235  }
236 
237  static TLinearFitter fitter(3, "pol2");
238  static TMatrixD mat(3, 3);
239  static Double_t kTol = mat.GetTol();
240  fitter.StoreData(kFALSE);
241  fitter.ClearPoints();
242  TVectorD par(3);
243  TMatrixD A(3, 3);
244  TMatrixD b(3, 1);
245  T rms = TMath::RMS(first, last);
246  // Markus: return type of std::max_element is an iterator, cannot cast implicitly to double
247  // better use auto
248  // pointer needs to be dereferenced afterwards
249  auto xMax = std::max_element(first, last);
250  auto xMin = std::min_element(first, last);
251  auto nbins = last - first;
252  const double binWidth = double(*xMax - *xMin) / double(nbins);
253 
254  Float_t meanCOG = 0;
255  Float_t rms2COG = 0;
256  Float_t sumCOG = 0;
257 
258  Float_t entries = 0;
259  Int_t nfilled = 0;
260 
261  for (auto iter = first; iter != last; iter++) {
262  entries += *iter;
263  if (*iter > 0) {
264  nfilled++;
265  }
266  }
267 
268  if (*xMax < 4) {
269  LOG(error) << "Gaus fit failed! xMax < 4";
270  throw FitGausError_t::FIT_ERROR;
271  }
272  if (entries < 12) {
273  LOG(error) << "Gaus fit failed! entries < 12";
274  throw FitGausError_t::FIT_ERROR;
275  }
276  if (rms < kTol) {
277  LOG(error) << "Gaus fit failed! rms < kTol";
278  throw FitGausError_t::FIT_ERROR;
279  }
280 
281  // create the result, first fill it with all 0's
282  std::vector<double> result;
283  for (int r = 0; r < 5; r++) {
284  result.push_back(0);
285  }
286  // then set the third parameter to entries
287  result.at(3) = entries;
288 
289  Int_t npoints = 0;
290  // in this loop: increase iter and axisiter (iterator for bin center and bincontent)
291  auto axisiter = axisfirst;
292  for (auto iter = first; iter != last; iter++, axisiter++) {
293  if (nbins > 1) {
294  // dont take bins with 0 entries or bins with nan into account
295  // if y-value (*iter) is 1, log(*iter) will be 0. Exclude these cases
296  if (isnan(*axisiter) || isinf(*axisiter) || *iter <= 0 || *iter == 1) {
297  continue;
298  }
299  double x = *axisiter;
300  // take logarith of gaussian in order to obtain a pol2
301  double y = std::log(*iter);
302  // first order taylor series of log(x) to approimate the errors df(x)/dx * err(f(x))
303  double ey = std::sqrt(fabs(*iter)) / fabs(*iter);
304 
305  fitter.AddPoint(&x, y, ey);
306  if (npoints < 3) {
307  A(npoints, 0) = 1;
308  A(npoints, 1) = x;
309  A(npoints, 2) = x * x;
310  b(npoints, 0) = y;
311  meanCOG += x * nbins;
312  rms2COG += x * nbins * x;
313  sumCOG += nbins;
314  }
315  npoints++;
316  }
317  }
318  Double_t chi2 = 0;
319  if (npoints >= 3) {
320  if (npoints == 3) {
321  // analytic calculation of the parameters for three points
322  A.Invert();
323  TMatrixD res(1, 3);
324  res.Mult(A, b);
325  par[0] = res(0, 0);
326  par[1] = res(0, 1);
327  par[2] = res(0, 2);
328  chi2 = -3.;
329  } else {
330  // use fitter for more than three points
331  fitter.Eval();
332  fitter.GetParameters(par);
333  fitter.GetCovarianceMatrix(mat);
334  result.at(4) = (fitter.GetChisquare() / Double_t(npoints));
335  }
336 
337  if (TMath::Abs(par[1]) < kTol) {
338  LOG(error) << "Gaus fit failed! TMath::Abs(par[1]) < kTol";
339  throw FitGausError_t::FIT_ERROR;
340  }
341  if (TMath::Abs(par[2]) < kTol) {
342  LOG(error) << "Gaus fit failed! TMath::Abs(par[2]) < kTol";
343  throw FitGausError_t::FIT_ERROR;
344  }
345 
346  // calculate parameters for gaus from pol2 fit
347  T param1 = T(par[1] / (-2. * par[2]));
348  result.at(1) = param1;
349  result.at(2) = T(1. / TMath::Sqrt(TMath::Abs(-2. * par[2])));
350  auto lnparam0 = par[0] - par[1] * par[1] / (4 * par[2]);
351  if (lnparam0 > 307) {
352  LOG(error) << "Gaus fit failed! lnparam0 > 307";
353  throw FitGausError_t::FIT_ERROR;
354  }
355 
356  result.at(0) = T(TMath::Exp(lnparam0));
357  return result;
358  }
359 
360  if (npoints == 2) {
361  // use center of gravity for 2 points
362  meanCOG /= sumCOG;
363  rms2COG /= sumCOG;
364  /*
365  result.setParameter<0>(xMax);
366  result.setParameter<1>(meanCOG);
367  result.setParameter<2>(TMath::Sqrt(TMath::Abs(meanCOG * meanCOG - rms2COG)));
368  result.setChi2(-2);
369  */
370  result.at(0) = *xMax;
371  result.at(1) = meanCOG;
372  result.at(2) = TMath::Sqrt(TMath::Abs(meanCOG * meanCOG - rms2COG));
373  result.at(4) = -2;
374  }
375  if (npoints == 1) {
376  meanCOG /= sumCOG;
377  /*
378  result.setParameter<0>(xMax);
379  result.setParameter<1>(meanCOG);
380  result.setParameter<2>(binWidth / TMath::Sqrt(12));
381  result.setChi2(-1);
382  */
383  result.at(0) = *xMax;
384  result.at(1) = meanCOG;
385  result.at(2) = binWidth / TMath::Sqrt(12);
386  result.at(4) = -1;
387  }
388 
389  return result;
390 }

410 template <typename... axes>
411 auto ProjectBoostHistoX(boost::histogram::histogram<axes...>& hist2d, const int binLow, const int binHigh)
412 {
413  using namespace boost::histogram::literals; // enables _c suffix needed for projection
414 
415  unsigned int nbins = hist2d.axis(0).size();
416  // make reduced histo in range that we want to project
417  auto reducedHisto2d = boost::histogram::algorithm::reduce(hist2d, boost::histogram::algorithm::shrink(hist2d.axis(0).bin(0).lower(), hist2d.axis(0).bin(nbins - 1).upper()), boost::histogram::algorithm::shrink(binLow, binHigh));
418 
419  // set under and overflow bin to 0 such that they will not be used in the projection
420  for (int i = 0; i < reducedHisto2d.axis(0).size(); ++i) {
421  reducedHisto2d.at(i, -1) = 0;
422  reducedHisto2d.at(i, reducedHisto2d.axis(1).size()) = 0;
423  }
424  // make the projection onto the x-axis (0_c) of the reduced histogram
425  auto histoProj = boost::histogram::algorithm::project(reducedHisto2d, 0_c);
426 
427  return histoProj;
428 }

436 template <typename... axes>
437 auto ProjectBoostHistoXFast(boost::histogram::histogram<axes...>& hist2d, const int binLow, const int binHigh)
438 {
439  unsigned int nbins = hist2d.axis(0).size();
440  double binStartX = hist2d.axis(0).bin(0).lower();
441  double binEndX = hist2d.axis(0).bin(nbins - 1).upper();
442  auto histoProj = boost::histogram::make_histogram(boost::histogram::axis::regular<>(nbins, binStartX, binEndX));
443 
444  // Now rewrite the bin content of the 1d histogram to get the summed bin content in the specified range
445  for (int x = 0; x < nbins; ++x) {
446  histoProj.at(x) = 0;
447  for (int y = binLow; y < binHigh; ++y) {
448  histoProj.at(x) = histoProj.at(x) + hist2d.at(x, y);
449  }
450  }
451 
452  return histoProj;
453 }
