90 void compare(const o2::dataformats::MCLabelContainer& output, const o2::dataformats::MCLabelContainer& expected)
91 {
92  BOOST_CHECK_EQUAL(output.getIndexedSize(), expected.getIndexedSize());
93 
94  for (int i = 0; i < output.getIndexedSize(); ++i) {
95  const auto outputLabels = output.getLabels(i);
96  const auto expectedLabels = expected.getLabels(i);

98  expectedLabels.begin(), expectedLabels.end());
