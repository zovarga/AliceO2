41 void CTFCoder::compress(CompressedClusters& cc,
42  const gsl::span<const ROFRecord>& rofRecVec,
43  const gsl::span<const CompClusterExt>& cclusVec,
44  const gsl::span<const unsigned char>& pattVec)
45 {
46  // store in the header the orbit of 1st ROF
47  cc.clear();
48  if (!rofRecVec.size()) {
49  return;
50  }
51  const auto& rofRec0 = rofRecVec[0];
52  cc.header.det = mDet;
53  cc.header.nROFs = rofRecVec.size();
54  cc.header.firstOrbit = rofRec0.getBCData().orbit;
55  cc.header.firstBC = rofRec0.getBCData().bc;
56  cc.header.nPatternBytes = pattVec.size();
57  cc.header.nClusters = cclusVec.size();
58  cc.header.nChips = 0; // this is the version with chipInc stored once per new chip
59 
60  cc.firstChipROF.resize(cc.header.nROFs);
61  cc.bcIncROF.resize(cc.header.nROFs);
62  cc.orbitIncROF.resize(cc.header.nROFs);
63  cc.nclusROF.resize(cc.header.nROFs);
64  //
65  cc.row.resize(cc.header.nClusters);
66  cc.colInc.resize(cc.header.nClusters);
67  // cc.chipInc.resize(cc.header.nClusters); // this is the version with chipInc stored for every pixel
68  cc.chipInc.reserve(1000); // this is the version with chipInc stored once per new chip
69  cc.chipMul.reserve(1000); // this is the version with chipInc stored once per new chip
70  cc.pattID.resize(cc.header.nClusters);
71  cc.pattMap.resize(cc.header.nPatternBytes);
72 
73  uint16_t prevBC = cc.header.firstBC;
74  uint32_t prevOrbit = cc.header.firstOrbit;
75 
76  for (uint32_t irof = 0; irof < cc.header.nROFs; irof++) {
77  const auto& rofRec = rofRecVec[irof];
78  const auto& intRec = rofRec.getBCData();
79  if (intRec.orbit == prevOrbit) {
80  cc.orbitIncROF[irof] = 0;
81 #ifdef _CHECK_INCREMENTES_
82  if (intRec.bc < prevBC) {
83  LOG(warning) << "Negative BC increment " << intRec.bc << " -> " << prevBC;
84  }
85 #endif
86  cc.bcIncROF[irof] = intRec.bc - prevBC; // store increment of BC if in the same orbit
87  } else {
88  cc.orbitIncROF[irof] = intRec.orbit - prevOrbit;
89 #ifdef _CHECK_INCREMENTES_
90  if (intRec.orbit < prevOrbit) {
91  LOG(warning) << "Negative Orbit increment " << intRec.orbit << " -> " << prevOrbit;
92  }
93 #endif
94  cc.bcIncROF[irof] = intRec.bc; // otherwise, store absolute bc
95  prevOrbit = intRec.orbit;
96  }
97  prevBC = intRec.bc;
98  auto ncl = rofRec.getNEntries();
99  cc.nclusROF[irof] = ncl;
100  if (!ncl) { // no hits data for this ROF
101  cc.firstChipROF[irof] = 0;
102  continue;
103  }
104  cc.firstChipROF[irof] = cclusVec[rofRec.getFirstEntry()].getChipID();
105  int icl = rofRec.getFirstEntry(), iclMax = icl + ncl;
106 
107  uint16_t prevChip = cc.firstChipROF[irof], prevCol = 0;
108  if (icl != iclMax) { // there are still clusters to store
109  cc.chipMul.push_back(0);
110  cc.chipInc.push_back(0);
111  }
112  for (; icl < iclMax; icl++) { // clusters within a chip are stored in increasing column number
113  const auto& cl = cclusVec[icl];
114  cc.row[icl] = cl.getRow(); // row is practically random, store it as it is
115  cc.pattID[icl] = cl.getPatternID();
116  if (cl.getChipID() == prevChip) { // for the same chip store column increment
117  // cc.chipInc[icl] = 0; // this is the version with chipInc stored for every pixel
118  cc.chipMul.back()++; // this is the version with chipInc stored once per new chip
119 #ifdef _CHECK_INCREMENTES_ // RS FIXME with current clusterization column increment can be slightly negative
120 // if (cl.getCol() < prevCol) {
121 // LOG(warning) << "Negative Column increment " << cl.getCol() << " -> " << prevCol << " in chip " << prevChip << " of ROF " << irof;
122 // }
123 #endif
124  cc.colInc[icl] = cl.getCol() - prevCol;
125  prevCol = cl.getCol();
126  } else { // for new chips store chipID increment and abs. column
127  // cc.chipInc[icl] = cl.getChipID() - prevChip; // this is the version with chipInc stored for every pixel
128  cc.chipInc.push_back(cl.getChipID() - prevChip); // this is the version with chipInc stored once per new chip
129 #ifdef _CHECK_INCREMENTES_
130  if (cl.getChipID() < prevChip) {
131  LOG(warning) << "Negative Chip increment " << cl.getChipID() << " -> " << prevChip;
132  }
133 #endif
134  cc.chipMul.push_back(1); // this is the version with chipInc stored once per new chip
135  prevCol = cc.colInc[icl] = cl.getCol();
136  prevChip = cl.getChipID();
137  }
138  }
139  }
140  cc.header.nChips = cc.chipMul.size();
141  // store explicit patters as they are
142  memcpy(cc.pattMap.data(), pattVec.data(), cc.header.nPatternBytes); // RSTODO: do we need this?
143 }

167 size_t CTFCoder::estimateCompressedSize(const CompressedClusters& cc)
168 {
169  size_t sz = 0;
170  // clang-format off
171  // RS FIXME this is very crude estimate, instead, an empirical values should be used
172 #define VTP(vec) typename std::remove_reference<decltype(vec)>::type::value_type
173 #define ESTSIZE(vec, slot) mCoders[int(slot)] ? \
174  rans::calculateMaxBufferSize(vec.size(), reinterpret_cast<const o2::rans::LiteralEncoder64<VTP(vec)>*>(mCoders[int(slot)].get())->getAlphabetRangeBits(), sizeof(VTP(vec)) ) : vec.size()*sizeof(VTP(vec))
175  sz += ESTSIZE(cc.firstChipROF, CTF::BLCfirstChipROF);
176  sz += ESTSIZE(cc.bcIncROF, CTF::BLCbcIncROF );
177  sz += ESTSIZE(cc.orbitIncROF, CTF::BLCorbitIncROF );
178  sz += ESTSIZE(cc.nclusROF, CTF::BLCnclusROF );
179  //
180  sz += ESTSIZE(cc.chipInc, CTF::BLCchipInc );
181  sz += ESTSIZE(cc.chipMul, CTF::BLCchipMul );
182  sz += ESTSIZE(cc.row, CTF::BLCrow );
183  sz += ESTSIZE(cc.colInc, CTF::BLCcolInc );
184  sz += ESTSIZE(cc.pattID, CTF::BLCpattID );
185  sz += ESTSIZE(cc.pattMap, CTF::BLCpattMap );
186 
187  // clang-format on
188  sz *= 2. / 3; // if needed, will be autoexpanded
189  LOG(info) << "Estimated output size is " << sz << " bytes";
190  return sz;
191 }

213  // clang-format on
