145 template <typename T>
146 GPUd() int GPUTRDTrack_t<T>::getNlayersFindable() const
147 {
148  // returns number of layers in which the track is in active area of TRD
149  int retVal = 0;
150  for (int iLy = 0; iLy < kNLayers; iLy++) {
151  if ((mFlags >> iLy) & 0x1) {
152  ++retVal;
153  }

171 template <typename T>
172 GPUd() int GPUTRDTrack_t<T>::getNmissingConsecLayers(int iLayer) const
173 {
174  // returns number of consecutive layers in which the track was
175  // inside the deadzone up to (and including) the given layer
176  int retVal = 0;
177  while (!getIsFindable(iLayer)) {
178  ++retVal;
179  --iLayer;
180  if (iLayer < 0) {
181  break;
182  }
183  }
184  return retVal;
185 }
