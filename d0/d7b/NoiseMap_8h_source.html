56  float getNoiseLevel(int chip, int row, int col) const
57  {
58  assert(chip > (int)mNoisyPixels.size());
59  const auto keyIt = mNoisyPixels[chip].find(getKey(row, col));

72  int dumpAboveThreshold(int t = 3) const
73  {
74  int n = 0;
75  auto chipID = mNoisyPixels.size();
76  while (chipID--) {
77  const auto& map = mNoisyPixels[chipID];
78  for (const auto& pair : map) {
79  if (pair.second <= t) {
80  continue;
81  }
82  n++;
83  auto key = pair.first;
84  auto row = key2Row(key);
85  auto col = key2Col(key);
86  std::cout << "chip, row, col, noise: " << chipID << ' ' << row << ' ' << col << ' ' << pair.second << '\n';
87  }
88  }
89  return n;
90  }
91  int dumpAboveProbThreshold(float p = 1e-7) const
92  {
93  return dumpAboveThreshold(p * mNumOfStrobes);
94  }
95 
96  void applyProbThreshold(float t, long int n)
97  {
98  // Remove from the maps all pixels with the firing probability below the threshold
99  mProbThreshold = t;
100  mNumOfStrobes = n;
101  for (auto& map : mNoisyPixels) {
102  for (auto it = map.begin(); it != map.end();) {
103  float prob = float(it->second) / mNumOfStrobes;
104  if (prob < mProbThreshold) {
105  it = map.erase(it);
106  } else {
107  ++it;
108  }
109  }
110  }
111  }
112  float getProbThreshold() const { return mProbThreshold; }
113  long int getNumOfStrobes() const { return mNumOfStrobes; }

115  bool isNoisy(int chip, int row, int col) const
