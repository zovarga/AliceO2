93 framework::WorkflowSpec getWorkflow(CompletionPolicyData* policyData, std::vector<int> const& tpcSectors, unsigned long tpcSectorMask, std::vector<int> const& laneConfiguration,
94  bool propagateMC, unsigned nLanes, std::string const& cfgInput, std::string const& cfgOutput, bool disableRootInput,
95  int caClusterer, int zsOnTheFly, bool askDISTSTF)
96 {
97  InputType inputType;
98  try {
99  inputType = InputMap.at(cfgInput);
100  } catch (std::out_of_range&) {
101  throw std::invalid_argument(std::string("invalid input type: ") + cfgInput);
102  }
103  std::vector<OutputType> outputTypes;
104  try {
105  outputTypes = RangeTokenizer::tokenize<OutputType>(cfgOutput, [](std::string const& token) { return OutputMap.at(token); });
106  } catch (std::out_of_range&) {
107  throw std::invalid_argument(std::string("invalid output type: ") + cfgOutput);
108  }
109  auto isEnabled = [&outputTypes](OutputType type) {
110  return std::find(outputTypes.begin(), outputTypes.end(), type) != outputTypes.end();
111  };
112 
113  bool decompressTPC = inputType == InputType::CompClustersCTF || inputType == InputType::CompClusters;
114  // Disable not applicable settings depending on TPC input, no need to disable manually
115  if (decompressTPC && (isEnabled(OutputType::Clusters) || isEnabled(OutputType::Tracks))) {
116  caClusterer = false;
117  zsOnTheFly = false;
118  propagateMC = false;
119  }
120  if (inputType == InputType::ZSRaw || inputType == InputType::CompClustersFlat) {
121  caClusterer = true;
122  zsOnTheFly = false;
123  propagateMC = false;
124  }
125  if (inputType == InputType::PassThrough || inputType == InputType::ClustersHardware || inputType == InputType::Clusters) {
126  caClusterer = false;
127  }
128  if (!caClusterer) {
129  zsOnTheFly = false;
130  }
131 
132  if (inputType == InputType::ClustersHardware && isEnabled(OutputType::Digits)) {
133  throw std::invalid_argument("input/output type mismatch, can not produce 'digits' from 'clustershardware'");
134  }
135  if (inputType == InputType::Clusters && (isEnabled(OutputType::Digits) || isEnabled(OutputType::ClustersHardware))) {
136  throw std::invalid_argument("input/output type mismatch, can not produce 'digits', nor 'clustershardware' from 'clusters'");
137  }
138  if (inputType == InputType::ZSRaw && isEnabled(OutputType::ClustersHardware)) {
139  throw std::invalid_argument("input/output type mismatch, can not produce 'clustershardware' from 'zsraw'");
140  }
141  if (caClusterer && (inputType == InputType::Clusters || inputType == InputType::ClustersHardware)) {
142  throw std::invalid_argument("ca-clusterer requires digits as input");
143  }
144  if (caClusterer && (isEnabled(OutputType::ClustersHardware))) {
145  throw std::invalid_argument("ca-clusterer cannot produce clustershardware output");
146  }
147 
148  WorkflowSpec specs;
149 
150  // We provide a special publishing method for labels which have been stored in a split format and need
151  // to be transformed into a contiguous shareable container before publishing. For other branches/types this returns
152  // false and the generic RootTreeWriter publishing proceeds
153  static Reader::SpecialPublishHook hook{[](std::string_view name, ProcessingContext& context, o2::framework::Output const& output, char* data) -> bool {
154  if (TString(name.data()).Contains("TPCDigitMCTruth") || TString(name.data()).Contains("TPCClusterHwMCTruth") || TString(name.data()).Contains("TPCClusterNativeMCTruth")) {
155  auto storedlabels = reinterpret_cast<o2::dataformats::IOMCTruthContainerView const*>(data);
156  o2::dataformats::ConstMCTruthContainer<o2::MCCompLabel> flatlabels;
157  storedlabels->copyandflatten(flatlabels);
158  //LOG(info) << "PUBLISHING CONST LABELS " << flatlabels.getNElements();
159  context.outputs().snapshot(output, flatlabels);
160  return true;
161  }
162  return false;
163  }};
164 
165  if (!disableRootInput || inputType == InputType::PassThrough) {
166  // The OutputSpec of the PublisherSpec is configured depending on the input
167  // type. Note that the configuration of the dispatch trigger in the main file
168  // needs to be done in accordance. This means, if a new input option is added
169  // also the dispatch trigger needs to be updated.
170  if (inputType == InputType::Digits) {
171  using Type = std::vector<o2::tpc::Digit>;
172 
173  specs.emplace_back(o2::tpc::getPublisherSpec<Type>(PublisherConf{
174  "tpc-digit-reader",
175  "tpcdigits.root",
176  "o2sim",
177  {"digitbranch", "TPCDigit", "Digit branch"},
178  {"mcbranch", "TPCDigitMCTruth", "MC label branch"},
179  OutputSpec{"TPC", "DIGITS"},
180  OutputSpec{"TPC", "DIGITSMCTR"},
181  tpcSectors,
182  laneConfiguration,
183  &hook},
184  propagateMC));
185  } else if (inputType == InputType::ClustersHardware) {
186  specs.emplace_back(o2::tpc::getPublisherSpec(PublisherConf{
187  "tpc-clusterhardware-reader",
188  "tpc-clusterhardware.root",
189  "tpcclustershardware",
190  {"databranch", "TPCClusterHw", "Branch with TPC ClustersHardware"},
191  {"mcbranch", "TPCClusterHwMCTruth", "MC label branch"},
192  OutputSpec{"TPC", "CLUSTERHW"},
193  OutputSpec{"TPC", "CLUSTERHWMCLBL"},
194  tpcSectors,
195  laneConfiguration,
196  &hook},
197  propagateMC));
198  } else if (inputType == InputType::Clusters) {
199  specs.emplace_back(o2::tpc::getClusterReaderSpec(propagateMC, &tpcSectors, &laneConfiguration));
200  } else if (inputType == InputType::CompClusters) {
201  // TODO: need to check if we want to store the MC labels alongside with compressed clusters
202  // for the moment reading of labels is disabled (last parameter is false)
203  // TODO: make a different publisher spec for only one output spec, for now using the
204  // PublisherSpec with only sector 0, '_0' is thus appended to the branch name
205  specs.emplace_back(o2::tpc::getPublisherSpec(PublisherConf{
206  "tpc-compressed-cluster-reader",
207  "tpc-compclusters.root",
208  "tpcrec",
209  {"clusterbranch", "TPCCompClusters", "Branch with TPC compressed clusters"},
210  {"", "", ""}, // No MC labels
211  OutputSpec{"TPC", "COMPCLUSTERS"},
212  OutputSpec{"", ""}, // No MC labels
213  std::vector<int>(1, 0),
214  std::vector<int>(1, 0),
215  &hook},
216  false));
217  }
218  }

220  // output matrix
221  // Note: the ClusterHardware format is probably a deprecated legacy format and also the
222  // ClusterDecoderRawSpec
223  bool produceCompClusters = isEnabled(OutputType::CompClusters);
224  bool produceTracks = isEnabled(OutputType::Tracks);
225  bool runTracker = (produceTracks || produceCompClusters || (isEnabled(OutputType::Clusters) && caClusterer)) && inputType != InputType::CompClustersFlat;
226  bool runHWDecoder = !caClusterer && (runTracker || isEnabled(OutputType::Clusters));
227  bool runClusterer = !caClusterer && (runHWDecoder || isEnabled(OutputType::ClustersHardware));
228  bool zsDecoder = inputType == InputType::ZSRaw;
229  bool runClusterEncoder = isEnabled(OutputType::EncodedClusters);
230 
231  // input matrix
232  runClusterer &= inputType == InputType::Digitizer || inputType == InputType::Digits;
233  runHWDecoder &= runClusterer || inputType == InputType::ClustersHardware;
234  runTracker &= caClusterer || runHWDecoder || inputType == InputType::Clusters || decompressTPC;
235 
236  bool outRaw = inputType == InputType::Digits && isEnabled(OutputType::ZSRaw) && !isEnabled(OutputType::DisableWriter);
237  //bool runZSDecode = inputType == InputType::ZSRaw;
238  bool zsToDigit = inputType == InputType::ZSRaw && isEnabled(OutputType::Digits);
239 
240  if (inputType == InputType::PassThrough) {
241  runTracker = runHWDecoder = runClusterer = runClusterEncoder = zsToDigit = false;
242  }

246  //
247  // clusterer process(es)
248  //
249  //
250  if (runClusterer) {
251  parallelProcessors.push_back(o2::tpc::getClustererSpec(propagateMC));
252  }
253 
255  //
256  // cluster decoder process(es)
257  //
258  //
259  if (runHWDecoder) {
260  parallelProcessors.push_back(o2::tpc::getClusterDecoderRawSpec(propagateMC));

264  //
265  // set up parallel TPC lanes
266  //
267  // the parallelPipeline helper distributes the subspec ids from the lane configuration
268  // among the pipelines. All inputs and outputs of processors of one pipeline will be
269  // cloned by the number of subspecs served by this pipeline and amended with the subspecs
270  parallelProcessors = parallelPipeline(
271  parallelProcessors, nLanes,
272  [&laneConfiguration]() { return laneConfiguration.size(); },
273  [&laneConfiguration](size_t index) { return laneConfiguration[index]; });
274  specs.insert(specs.end(), parallelProcessors.begin(), parallelProcessors.end());

277  //
278  // generation of processor specs for various types of outputs
279  // based on generic RootTreeWriter and MakeRootTreeWriterSpec generator
280  //
281  // -------------------------------------------------------------------------------------------
282  // the callbacks for the RootTreeWriter
283  //
284  // The generic writer needs a way to associate incoming data with the individual branches for
285  // the TPC sectors. The sector number is transmitted as part of the sector header, the callback
286  // finds the corresponding index in the vector of configured sectors
287  auto getIndex = [tpcSectors](o2::framework::DataRef const& ref) {
288  auto const* tpcSectorHeader = o2::framework::DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
289  if (!tpcSectorHeader) {
290  throw std::runtime_error("TPC sector header missing in header stack");
291  }
292  if (tpcSectorHeader->sector() < 0) {
293  // special data sets, don't write
294  return ~(size_t)0;
295  }
296  size_t index = 0;
297  for (auto const& sector : tpcSectors) {
298  if (sector == tpcSectorHeader->sector()) {
299  return index;
300  }
301  ++index;
302  }
303  throw std::runtime_error("sector " + std::to_string(tpcSectorHeader->sector()) + " not configured for writing");
304  };
305  auto getName = [tpcSectors](std::string base, size_t index) {
306  return base + "_" + std::to_string(tpcSectors.at(index));
307  };
308 
309  // -------------------------------------------------------------------------------------------
310  // helper to create writer specs for different types of output
311  auto fillLabels = [](TBranch& branch, std::vector<char> const& labelbuffer, DataRef const& /*ref*/) {
312  o2::dataformats::ConstMCTruthContainerView<o2::MCCompLabel> labels(labelbuffer);
313  o2::dataformats::IOMCTruthContainerView outputcontainer;
314  auto ptr = &outputcontainer;
315  auto br = framework::RootTreeWriter::remapBranch(branch, &ptr);
316  outputcontainer.adopt(labelbuffer);
317  br->Fill();
318  br->ResetAddress();
319  };

321  auto makeWriterSpec = [tpcSectors, laneConfiguration, propagateMC, getIndex, getName](const char* processName,
322  const char* defaultFileName,
323  const char* defaultTreeName,
324  auto&& databranch,
325  auto&& mcbranch,
326  bool singleBranch = false) {
327  if (tpcSectors.size() == 0) {
328  throw std::invalid_argument(std::string("writer process configuration needs list of TPC sectors"));
329  }
330 
331  auto amendInput = [tpcSectors, laneConfiguration](InputSpec& input, size_t index) {
332  input.binding += std::to_string(laneConfiguration[index]);
333  DataSpecUtils::updateMatchingSubspec(input, laneConfiguration[index]);
334  };
335  auto amendBranchDef = [laneConfiguration, amendInput, tpcSectors, getIndex, getName, singleBranch](auto&& def, bool enableMC = true) {
336  if (!singleBranch) {
337  def.keys = mergeInputs(def.keys, laneConfiguration.size(), amendInput);
338  // the branch is disabled if set to 0
339  def.nofBranches = enableMC ? tpcSectors.size() : 0;
340  def.getIndex = getIndex;
341  def.getName = getName;
342  } else {
343  // instead of the separate sector branches only one is going to be written
344  def.nofBranches = enableMC ? 1 : 0;
345  }
346  return std::move(def);
347  };
348 
349  return std::move(MakeRootTreeWriterSpec(processName, defaultFileName, defaultTreeName,
350  std::move(amendBranchDef(databranch)),
351  std::move(amendBranchDef(mcbranch, propagateMC)))());
352  };

357  //

368  "TPCDigitMCTruth",

370  }

374  // a writer process for hardware clusters
375  //
376  // selected by output type 'clustershardware'

378  specs.push_back(makeWriterSpec("tpc-clusterhardware-writer",
379  inputType == InputType::ClustersHardware ? "tpc-filtered-clustershardware.root" : "tpc-clustershardware.root",
380  "tpcclustershardware",
381  BranchDefinition<const char*>{InputSpec{"data", "TPC", "CLUSTERHW", 0},
382  "TPCClusterHw",
383  "databranch"},
384  BranchDefinition<MCLabelContainer>{InputSpec{"mc", "TPC", "CLUSTERHWMCLBL", 0},
385  "TPCClusterHwMCTruth",
386  "mcbranch"}));
387  }

391  // a writer process for TPC native clusters
392  //
393  // selected by output type 'clusters'
394  if (isEnabled(OutputType::Clusters) && !isEnabled(OutputType::DisableWriter)) {
395  // if the caClusterer is enabled, only one data set with the full TPC is produced, and the writer
396  // is configured to write one single branch
397  specs.push_back(makeWriterSpec("tpc-native-cluster-writer",
398  inputType == InputType::Clusters ? "tpc-filtered-native-clusters.root" : "tpc-native-clusters.root",
399  "tpcrec",
400  BranchDefinition<const char*>{InputSpec{"data", ConcreteDataTypeMatcher{"TPC", "CLUSTERNATIVE"}},
401  "TPCClusterNative",

403  BranchDefinition<std::vector<char>>{InputSpec{"mc", ConcreteDataTypeMatcher{"TPC", "CLNATIVEMCLBL"}},
404  "TPCClusterNativeMCTruth",
405  "mcbranch", fillLabels},
406  (caClusterer || decompressTPC || inputType == InputType::PassThrough) && !isEnabled(OutputType::SendClustersPerSector)));
407  }

409  if (zsOnTheFly) {
410  specs.emplace_back(o2::tpc::getZSEncoderSpec(tpcSectors, outRaw, tpcSectorMask));
411  }

414  specs.emplace_back(o2::tpc::getZStoDigitsSpec(tpcSectors));

419  // tracker process
420  //
421  // selected by output type 'tracks'
422  if (runTracker) {
423  o2::gpu::gpuworkflow::Config cfg;
424  cfg.decompressTPC = decompressTPC;

435  cfg.processMC = propagateMC;

444  //

486  std::move(mcdef))());

491  // a writer process for compressed clusters container
492  //
493  // selected by output type 'compressed-clusters'

496  //
497  // defaults

499  const char* defaultFileName = "tpc-compclusters.root";
500  const char* defaultTreeName = "tpcrec";

509  }
