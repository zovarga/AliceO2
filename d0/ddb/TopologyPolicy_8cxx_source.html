39 TopologyPolicy::DependencyChecker TopologyPolicyHelpers::dataDependency()
40 {
41  return [](DataProcessorSpec const& a, DataProcessorSpec const& b) {
42  for (size_t ii = 0; ii < a.inputs.size(); ++ii) {
43  for (size_t oi = 0; oi < b.outputs.size(); ++oi) {
44  try {
45  if (DataSpecUtils::match(a.inputs[ii], b.outputs[oi])) {
46  return true;
47  }
48  } catch (...) {
49  continue;
50  }
51  }
52  }
53  return false;
54  };
55 }
