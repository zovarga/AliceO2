33 #define GBTLINK_DECODE_ERRORCHECK(errRes, errEval) \
34  errRes = errEval; \
35  if ((errRes)&uint8_t(ErrorPrinted)) { \
36  ruPtr->linkHBFToDump[(uint64_t(subSpec) << 32) + hbfEntry] = irHBF.orbit; \
37  errRes &= ~uint8_t(ErrorPrinted); \
38  } \
39  if ((errRes)&uint8_t(Abort)) { \
40  discardData(); \
41  return AbortedOnError; \
42  }

193 GBTLink::CollectedDataStatus GBTLink::collectROFCableData(const Mapping& chmap)
194 {
195  int nw = 0;
196  status = None;
197  auto* currRawPiece = rawData.currentPiece();
198  uint8_t errRes = uint8_t(GBTLink::NoError);
199  bool expectPacketDone = false;
200  ir.clear();
201  while (currRawPiece) { // we may loop over multiple CRU page
202  if (dataOffset >= currRawPiece->size) {
203  dataOffset = 0; // start of the RDH
204  if (!(currRawPiece = rawData.nextPiece())) { // fetch next CRU page
205  break; // Data chunk (TF?) is done
206  }
207  }
208  if (!dataOffset) { // here we always start with the RDH
209  auto hbfEntrySav = hbfEntry;
210  hbfEntry = 0xffffffff; // in case of problems with RDH, dump full TF
211  const auto* rdh = reinterpret_cast<const RDH*>(&currRawPiece->data[dataOffset]);
212  if (verbosity >= VerboseHeaders) {
213  RDHUtils::printRDH(rdh);
214  }
215  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsRDH(*rdh)); // make sure we are dealing with RDH
216  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsRDHStop(*rdh)); // if new HB starts, the lastRDH must have stop
217  // GBTLINK_DECODE_ERRORCHECK(checkErrorsRDHStopPageEmpty(*rdh)); // end of HBF should be an empty page with stop
218  hbfEntry = hbfEntrySav; // critical check of RDH passed
219  lastRDH = rdh;
220  statistics.nPackets++;
221  if (RDHUtils::getPageCounter(*rdh) == 0) {
222  irHBF = RDHUtils::getHeartBeatIR(*rdh);
223  hbfEntry = rawData.currentPieceID();
224  }
225  dataOffset += sizeof(RDH);
226  auto psz = RDHUtils::getMemorySize(*rdh);
227  if (psz == sizeof(RDH)) {
228  continue; // filter out empty page
229  }
230  if (format == NewFormat && RDHUtils::getStop(*rdh)) { // only diagnostic word can be present after the stop
231  auto gbtDiag = reinterpret_cast<const GBTDiagnostic*>(&currRawPiece->data[dataOffset]);
232  if (verbosity >= VerboseHeaders) {
233  printDiagnostic(gbtDiag);
234  }
235  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsDiagnosticWord(gbtDiag));
236  dataOffset += RDHUtils::getOffsetToNext(*rdh) - sizeof(RDH);
237  continue;
238  }
239 
240  // data must start with the GBTHeader
241  auto gbtH = reinterpret_cast<const GBTDataHeader*>(&currRawPiece->data[dataOffset]); // process GBT header
242  dataOffset += GBTPaddedWordLength;
243  if (verbosity >= VerboseHeaders) {
244  printHeader(gbtH);
245  }
246  if (format == OldFormat) {
247  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsHeaderWord(reinterpret_cast<const GBTDataHeaderL*>(gbtH)));
248  lanesActive = reinterpret_cast<const GBTDataHeaderL*>(gbtH)->activeLanesL; // TODO do we need to update this for every page?
249  } else {
250  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsHeaderWord(gbtH));
251  lanesActive = gbtH->activeLanes; // TODO do we need to update this for every page?
252  }
253 
254  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsActiveLanes(chmap.getCablesOnRUType(ruPtr->ruInfo->ruType)));
255  if (format == OldFormat && reinterpret_cast<const GBTDataHeaderL*>(gbtH)->packetIdx == 0) { // reset flags in case of 1st page of new ROF (old format: judge by RDH)
256  lanesStop = 0;
257  lanesWithData = 0;
258  }
259 
260  continue;
261  }
262 
263  ruPtr->nCables = ruPtr->ruInfo->nCables; // RSTODO is this needed? TOREMOVE
264 
265  // then we expect GBT trigger word (unless we work with old format)
266  const GBTTrigger* gbtTrg = nullptr;
267  if (format == NewFormat) {
268  gbtTrg = reinterpret_cast<const GBTTrigger*>(&currRawPiece->data[dataOffset]); // process GBT trigger
269  dataOffset += GBTPaddedWordLength;
270  if (verbosity >= VerboseHeaders) {
271  printTrigger(gbtTrg);
272  }
273  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsTriggerWord(gbtTrg));
274  statistics.nTriggers++;
275  if (gbtTrg->noData) { // emtpy trigger
276  return status;
277  }
278  lanesStop = 0;
279  lanesWithData = 0;
280  ir.bc = gbtTrg->bc;
281  ir.orbit = gbtTrg->orbit;
282  trigger = gbtTrg->triggerType;
283  }
284  if (format == NewFormat) { // at the moment just check if calibration word is there
285  auto gbtC = reinterpret_cast<const o2::itsmft::GBTCalibration*>(&currRawPiece->data[dataOffset]);
286  if (gbtC->isCalibrationWord()) {
287  if (verbosity >= VerboseHeaders) {
288  printCalibrationWord(gbtC);
289  }
290  dataOffset += GBTPaddedWordLength;
291  LOGP(debug, "SetCalibData for RU:{} at bc:{}/orb:{} : [{}/{}]", ruPtr->ruSWID, gbtTrg->bc, gbtTrg->orbit, gbtC->calibCounter, gbtC->calibUserField);
292  ruPtr->calibData = {gbtC->calibCounter, gbtC->calibUserField};
293  }
294  }
295  auto gbtD = reinterpret_cast<const o2::itsmft::GBTData*>(&currRawPiece->data[dataOffset]);
296  expectPacketDone = true;
297  while (!gbtD->isDataTrailer()) { // start reading real payload
298  nw++;
299  if (verbosity >= VerboseData) {
300  gbtD->printX();
301  }
302  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsGBTDataID(gbtD));
303  if (errRes != uint8_t(GBTLink::Skip)) {
304  int cableHW = gbtD->getCableID(), cableSW = chmap.cableHW2SW(ruPtr->ruInfo->ruType, cableHW);
305  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsCableID(gbtD, cableSW));
306  if (errRes != uint8_t(GBTLink::Skip)) {
307  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsGBTData(chmap.cableHW2Pos(ruPtr->ruInfo->ruType, cableHW)));
308  ruPtr->cableData[cableSW].add(gbtD->getW8(), 9);
309  ruPtr->cableHWID[cableSW] = cableHW;
310  ruPtr->cableLinkID[cableSW] = idInRU;
311  ruPtr->cableLinkPtr[cableSW] = this;
312  }
313  }
314  dataOffset += GBTPaddedWordLength;
315  gbtD = reinterpret_cast<const o2::itsmft::GBTData*>(&currRawPiece->data[dataOffset]);
316  } // we are at the trailer, packet is over, check if there are more data on the next page
317 
318  auto gbtT = reinterpret_cast<const o2::itsmft::GBTDataTrailer*>(&currRawPiece->data[dataOffset]); // process GBT trailer
319  dataOffset += GBTPaddedWordLength;
320  if (verbosity >= VerboseHeaders) {
321  printTrailer(gbtT);
322  }
323  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsTrailerWord(gbtT));
324  // we finished the GBT page, but there might be continuation on the next CRU page
325  if (!gbtT->packetDone) {
326  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsPacketDoneMissing(gbtT, dataOffset < currRawPiece->size));
327  continue; // keep reading next CRU page
328  }
329  if (format == OldFormat) {
330  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsLanesStops());
331  }
332  // accumulate packet states
333  statistics.packetStates[gbtT->getPacketState()]++;
334  if (format != NewFormat) {
335  ir = RDHUtils::getTriggerIR(*lastRDH);
336  trigger = RDHUtils::getTriggerType(*lastRDH);
337  }
338 
339  return (status = DataSeen);
340  }
341 
342  if (expectPacketDone) { // no trailer with packet done was encountered, register error
343  GBTLINK_DECODE_ERRORCHECK(errRes, checkErrorsPacketDoneMissing(nullptr, false));
344  return (status = DataSeen);
345  }
346  return (status = StoppedOnEndOfData);
347 }
