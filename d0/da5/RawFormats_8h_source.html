69 class CpvHeader
70 {
71  public:
72  CpvHeader() = default;
73  CpvHeader(std::vector<char>::const_iterator b, std::vector<char>::const_iterator e)
74  { //reading header from file
75  for (int i = 0; i < 16 && b != e; i++, b++) { //read up to 16 mBytes
76  mBytes[i] = *b;
77  }
78  }
79  CpvHeader(InteractionRecord orbitBC, bool isNoDataExpected, bool isDataContinued)
80  { //writing header
81  //header is 128-bit word.
82  //|127-120|119-112|111-104|103-96|95-88 |87-80 |79-72|71-64|63-56|55-48|47-40|39-32|31-24|23-16|15-8 |7-0 |
83  // byte15 byte14 byte13 byte12 byte11 byte10 byte9 byte8 byte7 byte6 byte5 byte4 byte3 byte2 byte1 byte0
84  //byte = |76543210|
85  mBytes[0] = (0x010 & 0x0ff); //bits 11 - 0 trigger id (0x010 = physics trigger)
86  mBytes[1] = ((0x010 & 0xf00) >> 8) //bits 11 - 0 trigger id (0x010 = physics trigger)
87  + 0b00100000 * isNoDataExpected + 0b0100000 * isDataContinued; //bit 13 (no data for this trigger) + bit 14 (payload continues from previous page)
88  mBytes[2] = (orbitBC.bc & 0x00ff); //bits 27 - 16 bunch crossing
89  mBytes[3] = (orbitBC.bc & 0x0f00) >> 8; //bits 27 - 16 bunch crossing
90  mBytes[4] = (orbitBC.orbit & 0x000000ff); //bits 63 - 32 orbit
91  mBytes[5] = (orbitBC.orbit & 0x0000ff00) >> 8; //bits 63 - 32 orbit
92  mBytes[6] = (orbitBC.orbit & 0x00ff0000) >> 16; //bits 63 - 32 orbit
93  mBytes[7] = (orbitBC.orbit & 0xff000000) >> 24; //bits 63 - 32 orbit
94  mBytes[8] = 0x00; //bits 64-71 reserved
95  mBytes[9] = 0xe0; //word ID of cpv header (bits 79 - 72)
96  for (int i = 10; i < 16; i++) {
97  mBytes[i] = 0; //bits 127-80 must be zeros
98  }
99  }
100  ~CpvHeader() = default;
101  bool isOK() const { return (mBytes[9] == 0xe0) && (mBytes[10] == 0) && (mBytes[11] == 0) && (mBytes[12] == 0) && (mBytes[13] == 0) && (mBytes[14] == 0) && (mBytes[15] == 0); }
102  bool isNoDataExpected() const { return mBytes[1] & 0b00100000; }
103  bool isDataContinued() const { return mBytes[1] & 0b0100000; }
104  uint16_t bc() const { return static_cast<uint16_t>(mBytes[2]) + static_cast<uint16_t>((mBytes[3] & 0x0f) << 8); }
105  uint32_t orbit() const { return mBytes[4] + (mBytes[5] << 8) + (mBytes[6] << 16) + (mBytes[7] << 24); }
106 
107  public:
108  unsigned char mBytes[16] = {0}; //0 - 127 bits (16 bytes)
109 };

111 class CpvTrailer
112 {
113  public:
114  CpvTrailer() = default;
115  CpvTrailer(std::vector<char>::const_iterator b, std::vector<char>::const_iterator e)
116  { //reading
117  for (int i = 0; i < 16 && b != e; i++, b++) { //read up to 16 mBytes
118  mBytes[i] = *b;
119  }
120  }
121  CpvTrailer(unsigned short wordCounter, uint16_t bunchCrossing, bool isAllDataSent)
122  { //writing
123  mBytes[0] = bunchCrossing & 0x00ff; //bits 11 - 0 bunch crossing
124  mBytes[1] = ((bunchCrossing & 0x0f00) >> 8) //bits 11 - 0 bunch crossing
125  + ((wordCounter & 0x0f) << 4); //bits 20 - 12 wordCounter
126  mBytes[2] = (wordCounter & 0b111110000) >> 4; //bits 20 - 12 wordCounter
127  for (int i = 3; i < 8; i++) {
128  mBytes[i] = 0; //bits 70 - 21 reserved
129  }
130  mBytes[8] = isAllDataSent * 0b10000000; //bit 71 all data is sent for current trigger
131  mBytes[9] = char(0xf0); //word ID of cpv trailer
132  for (int i = 10; i < 16; i++) {
133  mBytes[i] = 0;
134  }
135  }
136  ~CpvTrailer() = default;
137  bool isOK() const { return (mBytes[9] == 0xf0) && (mBytes[10] == 0) && (mBytes[11] == 0) && (mBytes[12] == 0) && (mBytes[13] == 0) && (mBytes[14] == 0) && (mBytes[15] == 0); }
138  uint16_t wordCounter() const { return (mBytes[1] >> 4) + ((mBytes[2] & 0b00011111) << 4); }
139  bool isAllDataSent() const { return (mBytes[8] & 0b10000000); }
140  uint16_t bc() const { return mBytes[0] + ((mBytes[1] & 0x0f) << 8); }

143  unsigned char mBytes[16] = {0};
