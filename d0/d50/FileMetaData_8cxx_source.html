22 bool FileMetaData::fillFileData(const std::string& fname)
23 {
24  try {
25  lurl = std::filesystem::canonical(fname).string();
26  size = std::filesystem::file_size(lurl);
27  std::unique_ptr<TMD5> md5ptr{TMD5::FileChecksum(fname.c_str())};
28  md5 = md5ptr->AsString();
29  ctime = std::chrono::time_point_cast<std::chrono::milliseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
30  } catch (std::exception const& e) {
31  LOG(error) << "Failed to fill metadata for file " << fname << ", reason: " << e.what();
32  return false;
33  }
34  return true;
35 }
