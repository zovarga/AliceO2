75  DataT compressValue(float idcDelta, const o2::tpc::Side side) const
76  {
77  idcDelta = (std::clamp(idcDelta, mCompressionFactor.mFactors[side].first, mCompressionFactor.mFactors[side].second) - mCompressionFactor.mFactors[side].first) * std::numeric_limits<DataT>::max() / (mCompressionFactor.mFactors[side].second - mCompressionFactor.mFactors[side].first);

169 {

172 
176  static IDCDelta<DataT> getCompressedIDCs(const IDCDelta<float>& idcDeltaUncompressed)
177  {
178  IDCDelta<DataT> idcCompressed{};
179  compress(idcDeltaUncompressed, idcCompressed, o2::tpc::Side::A);
180  compress(idcDeltaUncompressed, idcCompressed, o2::tpc::Side::C);
181  return idcCompressed;
182  }
183 
184  private:
185  static void compress(const IDCDelta<float>& idcDeltaUncompressed, IDCDelta<DataT>& idcCompressed, const o2::tpc::Side side)
186  {
187  idcCompressed.getIDCDelta(side).reserve(idcDeltaUncompressed.getIDCDelta(side).size());
188  const auto minmaxIDC = std::minmax_element(std::begin(idcDeltaUncompressed.getIDCDelta(side)), std::end(idcDeltaUncompressed.getIDCDelta(side)));
189  const auto& paramIDCGroup = ParameterIDCCompression::Instance();
190  const float minIDCDelta = std::clamp(*minmaxIDC.first, paramIDCGroup.minIDCDeltaValue, paramIDCGroup.maxIDCDeltaValue);
191  const float maxIDCDelta = std::clamp(*minmaxIDC.second, paramIDCGroup.minIDCDeltaValue, paramIDCGroup.maxIDCDeltaValue);
192  idcCompressed.setCompressionFactor(side, minIDCDelta, maxIDCDelta);
193  for (auto& idc : idcDeltaUncompressed.getIDCDelta(side)) {
194  idcCompressed.emplace_back(idc, side);
195  }
196  }

312 class OneDIDCAggregator
313 {
314  public:
317  OneDIDCAggregator(const unsigned int nTimeFrames = 1) : mOneDIDCAgg(nTimeFrames), mWeight(nTimeFrames){};
318 
324  void aggregate1DIDCs(const o2::tpc::Side side, const std::vector<float>& idc, const unsigned int timeframe, const unsigned int region)
325  {
326  if (mOneDIDCAgg[timeframe].mOneDIDC[side].empty()) {
327  mOneDIDCAgg[timeframe].mOneDIDC[side] = idc;
328  mWeight[timeframe][side] = Mapper::REGIONAREA[region];
329  } else {
330  std::transform(mOneDIDCAgg[timeframe].mOneDIDC[side].begin(), mOneDIDCAgg[timeframe].mOneDIDC[side].end(), idc.begin(), mOneDIDCAgg[timeframe].mOneDIDC[side].begin(), std::plus<>());
331  mWeight[timeframe][side] += Mapper::REGIONAREA[region];
332  }
333  }
334 
336  OneDIDC getAggregated1DIDCs()
337  {
338  OneDIDC oneDIDCTmp;
339  for (int iside = 0; iside < o2::tpc::SIDES; ++iside) {
340  const o2::tpc::Side side = iside == 0 ? o2::tpc::Side::A : o2::tpc::Side::C;
341  for (unsigned int i = 0; i < mOneDIDCAgg.size(); ++i) {
342  std::transform(mOneDIDCAgg[i].mOneDIDC[side].begin(), mOneDIDCAgg[i].mOneDIDC[side].end(), mOneDIDCAgg[i].mOneDIDC[side].begin(), [norm = mWeight[i][side]](auto& val) { return val / norm; });
343  oneDIDCTmp.mOneDIDC[side].insert(oneDIDCTmp.mOneDIDC[side].end(), mOneDIDCAgg[i].mOneDIDC[side].begin(), mOneDIDCAgg[i].mOneDIDC[side].end());
344  mOneDIDCAgg[i].mOneDIDC[side].clear();
345  }
346  }
347  return oneDIDCTmp;
348  }
349 
350  const auto& get() const { return mOneDIDCAgg; }
351  const auto& getWeight() const { return mWeight; }
352 
354  std::vector<unsigned int> getIntegrationIntervalsPerTF(const o2::tpc::Side side = o2::tpc::Side::A) const
355  {
356  std::vector<unsigned int> integrationIntervalsPerTF;
357  integrationIntervalsPerTF.reserve(mOneDIDCAgg.size());
358  for (const auto& tf : mOneDIDCAgg) {
359  integrationIntervalsPerTF.emplace_back(tf.getNIDCs(side));
360  }
361  return integrationIntervalsPerTF;
362  }

364  private:

366  std::vector<std::array<float, o2::tpc::SIDES>> mWeight{};
367 };
