51  template <typename I, typename D, typename T, int M = 1>
52  class _Iter
53  {
54  public:
55  using difference_type = int64_t;
56  using value_type = T;
57  using pointer = const T*;
58  using reference = const T&;
59  using iterator_category = std::random_access_iterator_tag;
60 
61  _Iter(const gsl::span<const D>& data, bool end = false) : mData(data), mIndex(end ? M * data.size() : 0){};
62  _Iter() = default;
63 
64  const I& operator++()
65  {
66  ++mIndex;
67  return (I&)(*this);
68  }
69 
70  const I& operator--()
71  {
72  mIndex--;
73  return (I&)(*this);
74  }
75 
76  difference_type operator-(const I& other) const { return mIndex - other.mIndex; }
77 
78  difference_type operator-(size_t idx) const { return mIndex - idx; }
79 
80  const I& operator-(size_t idx)
81  {
82  mIndex -= idx;
83  return (I&)(*this);
84  }
85 
86  bool operator!=(const I& other) const { return mIndex != other.mIndex; }
87  bool operator==(const I& other) const { return mIndex == other.mIndex; }
88  bool operator>(const I& other) const { return mIndex > other.mIndex; }
89  bool operator<(const I& other) const { return mIndex < other.mIndex; }
90 
91  protected:
92  gsl::span<const D> mData{};
93  size_t mIndex = 0;
94  };

97  // BC difference wrt previous if in the same orbit, otherwise the abs.value.
98  // For the very 1st entry return 0 (diff wrt 1st BC in the CTF header)
99  class Iter_bcIncROF : public _Iter<Iter_bcIncROF, ROFRecord, uint16_t>
100  {
101  public:
102  using _Iter<Iter_bcIncROF, ROFRecord, uint16_t>::_Iter;
103  value_type operator*() const
104  {
105  if (mIndex) {
106  if (mData[mIndex].getBCData().orbit == mData[mIndex - 1].getBCData().orbit) {
107  return mData[mIndex].getBCData().bc - mData[mIndex - 1].getBCData().bc;
108  } else {
109  return mData[mIndex].getBCData().bc;
110  }
111  }
112  return 0;
113  }
114  };
