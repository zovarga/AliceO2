153 int FeeParam::getMCMfromSharedPad(int irow, int icol)
154 {
155  //
156  // Return on which MCM this pad is directry connected.
157  // Return -1 for error.
158  //
159 
160  if (irow < 0 || icol < 0 || irow > NROWC1 || icol > NCOLUMN + 8 * 3) {
161  return -1;
162  }
163 
164  int adc = 20 - (icol % 18) - 1;
165  switch (adc) {
166  case 2:
167  icol += 5;
168  break;
169  case 18:
170  icol -= 5;
171  break;
172  case 19:
173  icol -= 5;
174  break;
175  default:
176  icol += 0;
177  break;
178  }
179 
180  return (icol % (NCOLUMN / 2)) / NCOLMCM + NMCMROBINCOL * (irow % NMCMROBINROW);
181 }

234 int FeeParam::extAliToAli(unsigned int dest, unsigned short linkpair, unsigned short rocType, int* mcmList, int listSize)
235 {
236  // Converts an extended ALICE ID which identifies a single MCM or a group of MCMs to
237  // the corresponding list of MCMs. Only broadcasts (127) are encoded as 127
238  // The return value is the number of MCMs in the list
239 
240  mcmList[0] = -1;
241 
242  short nmcm = 0;
243  unsigned int mcm, rob, robAB;
244  unsigned int cmA = 0, cmB = 0; // Chipmask for each A and B side
245 
246  // Default chipmask for 4 linkpairs (each bit correponds each alice-mcm)
247  const unsigned int gkChipmaskDefLp[4] = {0x1FFFF, 0x1FFFF, 0x3FFFF, 0x1FFFF};
248 
249  rob = dest >> 7; // Extract ROB pattern from dest.
250  mcm = dest & 0x07F; // Extract MCM pattern from dest.
251  robAB = getRobAB(rob, linkpair); // Get which ROB sides are selected.
252 
253  // Abort if no ROB is selected
254  if (robAB == 0) {
255  return 0;
256  }
257 
258  // Special case
259  if (mcm == 127) {
260  if (robAB == 3) { // This is very special 127 can stay only if two ROBs are selected
261  mcmList[0] = 127; // broadcase to ALL
262  mcmList[1] = -1;
263  return 1;
264  }
265  cmA = cmB = 0x3FFFF;
266  } else if ((mcm & 0x40) != 0) { // If top bit is 1 but not 127, this is chip group.
267  if ((mcm & 0x01) != 0) {
268  cmA |= 0x04444;
269  cmB |= 0x04444;
270  } // chip_cmrg
271  if ((mcm & 0x02) != 0) {
272  cmA |= 0x10000;
273  cmB |= 0x10000;
274  } // chip_bmrg
275  if ((mcm & 0x04) != 0 && rocType == 0) {
276  cmA |= 0x20000;
277  cmB |= 0x20000;
278  } // chip_hm3
279  if ((mcm & 0x08) != 0 && rocType == 1) {
280  cmA |= 0x20000;
281  cmB |= 0x20000;
282  } // chip_hm4
283  if ((mcm & 0x10) != 0) {
284  cmA |= 0x01111;
285  cmB |= 0x08888;
286  } // chip_edge
287  if ((mcm & 0x20) != 0) {
288  cmA |= 0x0aaaa;
289  cmB |= 0x03333;
290  } // chip_norm
291  } else { // Otherwise, this is normal chip ID, turn on only one chip.
292  cmA = 1 << mcm;
293  cmB = 1 << mcm;
294  }
295 
296  // Mask non-existing MCMs
297  cmA &= gkChipmaskDefLp[linkpair];
298  cmB &= gkChipmaskDefLp[linkpair];
299  // Remove if only one side is selected
300  if (robAB == 1) {
301  cmB = 0;
302  }
303  if (robAB == 2) {
304  cmA = 0;
305  }
306  if (robAB == 4 && linkpair != 2) {
307  cmA = cmB = 0; // Restrict to only T3A and T3B
308  }
309 
310  // Finally convert chipmask to list of slaves
311  nmcm = chipmaskToMCMlist(cmA, cmB, linkpair, mcmList, listSize);
312 
313  return nmcm;
314 }

316 short FeeParam::getRobAB(unsigned short robsel, unsigned short linkpair)
317 {
318  // Converts the ROB part of the extended ALICE ID to robs
319 
320  if ((robsel & 0x8) != 0) { // 1000 .. direct ROB selection. Only one of the 8 ROBs are used.
321  robsel = robsel & 7;
322  if ((robsel % 2) == 0 && (robsel / 2) == linkpair) {
323  return 1; // Even means A side (position 0,2,4,6)
324  }
325  if ((robsel % 2) == 1 && (robsel / 2) == linkpair) {
326  return 2; // Odd means B side (position 1,3,5,7)
327  }
328  return 0;
329  }
330 
331  // ROB group
332  if (robsel == 0) {
333  return 3;
334  } // Both ROB
335  if (robsel == 1) {
336  return 1;
337  } // A-side ROB
338  if (robsel == 2) {
339  return 2;
340  } // B-side ROB
341  if (robsel == 3) {
342  return 3;
343  } // Both ROB
344  if (robsel == 4) {
345  return 4;
346  } // Only T3A and T3B
347  // Other number 5 to 7 are ignored (not defined)
348 
349  return 0;
350 }

438 short FeeParam::chipmaskToMCMlist(unsigned int cmA, unsigned int cmB, unsigned short linkpair, int* mcmList, int listSize)
439 {
440  // Converts the chipmask to a list of MCMs
441 
442  short nmcm = 0;
443  short i;
444  for (i = 0; i < 18; i++) { // 18: number of MCMs on a ROB
445  if ((cmA & (1 << i)) != 0 && nmcm < listSize) {
446  mcmList[nmcm] = ((linkpair * 2) << 7) | i;
447  ++nmcm;
448  }
449  if ((cmB & (1 << i)) != 0 && nmcm < listSize) {
450  mcmList[nmcm] = ((linkpair * 2 + 1) << 7) | i;
451  ++nmcm;
452  }
453  }
454 
455  mcmList[nmcm] = -1;
456  return nmcm;
457 }

494 int FeeParam::getDyCorrection(int det, int rob, int mcm) const
495 {
496  // calculate the correction of the deflection
497  // i.e. Lorentz angle and tilt correction (if active)
498 
499  int layer = det % NLAYER;
500 
501  float dyTilt = (mDriftLength * std::tan(mTiltingAngle[layer] * M_PI / 180.) *
502  getLocalZ(det, rob, mcm) * mInvX[layer]);
503 
504  // calculate Lorentz correction
505  float dyCorr = -mOmegaTau * mDriftLength;
506 
507  if (mTiltCorr) {
508  dyCorr += dyTilt; // add tilt correction
509  }
510 
511  return (int)TMath::Nint(dyCorr * mScalePad * mInvWidthPad[layer]);
512 }

514 void FeeParam::getDyRange(int det, int rob, int mcm, int ch,
515  int& dyMinInt, int& dyMaxInt) const
516 {
517  // calculate the deflection range in which tracklets are accepted
518 
519  dyMinInt = mDyMin;
520  dyMaxInt = mDyMax;
521 
522  // deflection cut is considered for |B| > 0.1 T only
523  if (std::abs(mMagField) < 0.1) {
524  return;
525  }
526 
527  float e = 0.30;
528 
529  float maxDeflTemp = getPerp(det, rob, mcm, ch) / 2. * // Sekante/2 (cm)
530  (e * 1e-2 * std::abs(mMagField) * mInvPtMin); // 1/R (1/cm)
531 
532  float phi = getPhi(det, rob, mcm, ch);
533  if (maxDeflTemp < std::cos(phi)) {
534  float maxDeflAngle = std::asin(maxDeflTemp);
535 
536  float dyMin = (mDriftLength *
537  std::tan(phi - maxDeflAngle));
538 
539  dyMinInt = int(dyMin / mBinDy);
540  // clipping to allowed range
541  if (dyMinInt < mDyMin) {
542  dyMinInt = mDyMin;
543  } else if (dyMinInt > mDyMax) {
544  dyMinInt = mDyMax;
545  }
546 
547  float dyMax = (mDriftLength *
548  std::tan(phi + maxDeflAngle));
549 
550  dyMaxInt = int(dyMax / mBinDy);
551  // clipping to allowed range
552  if (dyMaxInt > mDyMax) {
553  dyMaxInt = mDyMax;
554  } else if (dyMaxInt < mDyMin) {
555  dyMaxInt = mDyMin;
556  }
557  } else if (maxDeflTemp < 0.) {
558  // this must not happen
559  printf("Inconsistent calculation of sin(alpha): %f\n", maxDeflTemp);
560  } else {
561  // TRD is not reached at the given pt threshold
562  // max range
563  }
564 
565  if ((dyMaxInt - dyMinInt) <= 0) {
566  LOG(debug) << "strange dy range: [" << dyMinInt << "," << dyMaxInt << "], using max range now";
567  dyMaxInt = mDyMax;
568  dyMinInt = mDyMin;
569  }
570 }

572 float FeeParam::getElongation(int det, int rob, int mcm, int ch) const
573 {
574  // calculate the ratio of the distance to the primary vertex and the
575  // distance in x-direction for the given ADC channel
576 
577  int layer = det % NLAYER;
578 
579  float elongation = std::abs(getDist(det, rob, mcm, ch) * mInvX[layer]);

583  elongation = 1.;
584  }

632 float FeeParam::getLocalZ(int det, int rob, int mcm) const
633 {
634  // get local z-position w.r.t. to the chamber boundary
635 
636  int stack = (det % NCHAMBERPERSEC) / NLAYER;
637  int layer = det % NLAYER;
638  int row = (rob / 2) * 4 + mcm / 4;
639 
640  if (stack == 2) {
641  if (row == 0) {
642  return (mZrow[layer * NLAYER + stack] - 0.5 * mLengthOuterPadC0);
643  } else if (row == 11) {
644  return (mZrow[layer * NLAYER + stack] - 1.5 * mLengthOuterPadC0 - (row - 1) * mLengthInnerPadC0);
645  } else {
646  return (mZrow[layer * NLAYER + stack] - mLengthOuterPadC0 - (row - 0.5) * mLengthInnerPadC0);
647  }
648  } else {
649  if (row == 0) {
650  return (mZrow[layer * NLAYER + stack] - 0.5 * mLengthOuterPadC1[layer]);
651  } else if (row == 15) {
652  return (mZrow[layer * NLAYER + stack] - 1.5 * mLengthOuterPadC1[layer] - (row - 1) * mLengthInnerPadC1[layer]);
653  } else {
654  return (mZrow[layer * NLAYER + stack] - mLengthOuterPadC1[layer] - (row - 0.5) * mLengthInnerPadC1[layer]);
655  }
656  }
657 }
