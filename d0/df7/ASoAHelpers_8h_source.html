65 template <typename T2, typename ARRAY, typename T>
66 std::vector<std::pair<uint64_t, uint64_t>> doGroupTable(const T& table, const std::string& categoryColumnName, int minCatSize, const T2& outsider)
67 {
68  auto columnIndex = table.asArrowTable()->schema()->GetFieldIndex(categoryColumnName);
69  auto chunkedArray = table.asArrowTable()->column(columnIndex);
70 
71  uint64_t ind = 0;
72  uint64_t selInd = 0;
73  gsl::span<int64_t const> selectedRows;
74  std::vector<std::pair<uint64_t, uint64_t>> groupedIndices;
75 
76  // Separate check to account for Filtered size different from arrow table
77  if (table.size() == 0) {
78  return groupedIndices;
79  }
80 
81  if constexpr (soa::is_soa_filtered_t<T>::value) {
82  selectedRows = table.getSelectedRows(); // vector<int64_t>
83  }
84 
85  for (uint64_t ci = 0; ci < chunkedArray->num_chunks(); ++ci) {
86  auto chunk = chunkedArray->chunk(ci);
87  if constexpr (soa::is_soa_filtered_t<T>::value) {
88  if (selectedRows[ind] >= selInd + chunk->length()) {
89  selInd += chunk->length();
90  continue; // Go to the next chunk, no value selected in this chunk
91  }
92  }
93 
94  T2 const* data = std::static_pointer_cast<ARRAY>(chunk)->raw_values();
95  uint64_t ai = 0;
96  while (ai < chunk->length()) {
97  if constexpr (soa::is_soa_filtered_t<T>::value) {
98  ai += selectedRows[ind] - selInd;
99  selInd = selectedRows[ind];
100  }
101 
102  if (data[ai] != outsider) {
103  groupedIndices.emplace_back(data[ai], ind);
104  }
105  ind++;
106 
107  if constexpr (soa::is_soa_filtered_t<T>::value) {
108  if (ind >= selectedRows.size()) {
109  break;
110  }
111  } else {
112  ai++;
113  }
114  }
115 
116  if constexpr (soa::is_soa_filtered_t<T>::value) {
117  if (ind == selectedRows.size()) {
118  break;
119  }
120  }
121  }
122 
123  // Do a stable sort so that same categories entries are
124  // grouped together.
125  std::stable_sort(groupedIndices.begin(), groupedIndices.end());
126 
127  // Remove categories of too small size
128  if (minCatSize > 1) {
129  auto catBegin = groupedIndices.begin();
130  while (catBegin != groupedIndices.end()) {
131  auto catEnd = std::upper_bound(catBegin, groupedIndices.end(), *catBegin, sameCategory);
132  if (std::distance(catBegin, catEnd) < minCatSize) {
133  catEnd = groupedIndices.erase(catBegin, catEnd);
134  }
135  catBegin = catEnd;
136  }
137  }
138 
139  return groupedIndices;
140 }

142 // TODO: With arrow table we lose the filtered information!
143 // Can we group directly on T? Otherwise we need to extract the selection vector from T
144 template <typename T, typename T2>
145 auto groupTable(const T& table, const std::string& categoryColumnName, int minCatSize, const T2& outsider)
146 {
147  auto columnIndex = table.asArrowTable()->schema()->GetFieldIndex(categoryColumnName);
148  auto dataType = table.asArrowTable()->column(columnIndex)->type();
149  if (dataType->id() == arrow::Type::UINT64) {
150  return doGroupTable<uint64_t, arrow::UInt64Array>(table, categoryColumnName, minCatSize, outsider);
151  }
152  if (dataType->id() == arrow::Type::INT64) {
153  return doGroupTable<int64_t, arrow::Int64Array>(table, categoryColumnName, minCatSize, outsider);
154  }
155  if (dataType->id() == arrow::Type::UINT32) {
156  return doGroupTable<uint32_t, arrow::UInt32Array>(table, categoryColumnName, minCatSize, outsider);
157  }
158  if (dataType->id() == arrow::Type::INT32) {
159  return doGroupTable<int32_t, arrow::Int32Array>(table, categoryColumnName, minCatSize, outsider);
160  }
161  if (dataType->id() == arrow::Type::FLOAT) {
162  return doGroupTable<float, arrow::FloatArray>(table, categoryColumnName, minCatSize, outsider);
163  }
164  // FIXME: Should we support other types as well?
165  throw o2::framework::runtime_error("Combinations: category column must be of integral type");
166 }

168 // Synchronize categories so as groupedIndices contain elements only of categories common to all tables
169 template <std::size_t K>
170 void syncCategories(std::array<std::vector<std::pair<uint64_t, uint64_t>>, K>& groupedIndices)
171 {
172  std::vector<std::pair<uint64_t, uint64_t>> firstCategories;
173  std::vector<std::pair<uint64_t, uint64_t>> commonCategories;
174  std::unique_copy(groupedIndices[0].begin(), groupedIndices[0].end(), std::back_inserter(firstCategories), diffCategory);
175 
176  for (auto& cat : firstCategories) {
177  bool isCommon = true;
178  for (int i = 1; i < K; i++) {
179  if (!std::binary_search(groupedIndices[i].begin(), groupedIndices[i].end(), cat, sameCategory)) {
180  isCommon = false;
181  break;
182  }
183  }
184  if (isCommon) {
185  commonCategories.push_back(cat);
186  }
187  }
188 
189  for (int i = 0; i < K; i++) {
190  auto nonCatBegin = groupedIndices[i].begin();
191  for (auto& cat : commonCategories) {
192  auto nonCatEnd = std::lower_bound(nonCatBegin, groupedIndices[i].end(), cat, sameCategory);
193  nonCatEnd = groupedIndices[i].erase(nonCatBegin, nonCatEnd);
194  nonCatBegin = std::upper_bound(nonCatEnd, groupedIndices[i].end(), cat, sameCategory);
195  }
196  }
197 }

539  void setRanges()
540  {
541  constexpr auto k = sizeof...(Ts);
542  for_<k>([&, this](auto i) {
543  auto catBegin = this->mGroupedIndices[i.value].begin() + std::get<i.value>(this->mCurrentIndices);
544  auto range = std::equal_range(catBegin, this->mGroupedIndices[i.value].end(), *catBegin, sameCategory);
545  std::get<i.value>(this->mBeginIndices) = std::distance(this->mGroupedIndices[i.value].begin(), range.first);
546  std::get<i.value>(this->mCurrent).setCursor(range.first->second);
547  std::get<i.value>(this->mMaxOffset) = std::distance(this->mGroupedIndices[i.value].begin(), range.second);
548  });
549  }

551  void addOne()
552  {
553  constexpr auto k = sizeof...(Ts);
554  bool modify = true;
555  bool nextCatAvailable = true;
556  for_<k - 1>([&, this](auto i) {
557  if (modify) {
558  constexpr auto curInd = k - i.value - 1;
559  std::get<curInd>(this->mCurrentIndices)++;
560  uint64_t curGroupedInd = std::get<curInd>(this->mCurrentIndices);
561  uint64_t maxForWindow = std::get<curInd>(this->mBeginIndices) + this->mSlidingWindowSize;
562 
563  // If we remain within the same sliding window
564  if (curGroupedInd < maxForWindow && curGroupedInd < std::get<curInd>(this->mMaxOffset)) {
565  std::get<curInd>(this->mCurrent).setCursor(this->mGroupedIndices[curInd][curGroupedInd].second);
566  modify = false;
567  for_<i.value>([&, this](auto j) {
568  constexpr auto curJ = k - i.value + j.value;
569  if (std::get<curJ - 1>(this->mCurrentIndices) < std::get<curJ>(this->mMaxOffset)) {
570  std::get<curJ>(this->mCurrentIndices) = std::get<curJ - 1>(this->mCurrentIndices);
571  uint64_t curGroupedJ = std::get<curJ>(this->mCurrentIndices);
572  std::get<curJ>(this->mCurrent).setCursor(this->mGroupedIndices[curJ][curGroupedJ].second);
573  } else {
574  modify = true;
575  }
576  });
577  }
578  }
579  });
580 
581  // First iterator processed separately
582  if (modify) {
583  std::get<0>(this->mCurrentIndices)++;
584  std::get<0>(this->mBeginIndices)++;
585  uint64_t curGroupedInd = std::get<0>(this->mCurrentIndices);
586 
587  // If we remain within the same category - slide window
588  if (curGroupedInd < std::get<0>(this->mMaxOffset)) {
