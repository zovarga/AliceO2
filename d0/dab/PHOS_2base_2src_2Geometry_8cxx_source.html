51 short Geometry::relToAbsId(char moduleNumber, int strip, int cell)
52 {
53  // calculates absolute cell Id from moduleNumber, strip (number) and cell (number)
54  // PHOS layout parameters:
55  const short nStrpZ = 28; // Number of strips along z-axis
56  const short nCrystalsInModule = 56 * 64; // Total number of crystals in module
57  const short nCellsXInStrip = 8; // Number of crystals in strip unit along x-axis
58  const short nZ = 56; // nStripZ * nCellsZInStrip
59 
60  short row = nStrpZ - (strip - 1) % nStrpZ;
61  short col = (int)std::ceil((float)strip / (nStrpZ)) - 1;
62 
63  return (moduleNumber - 1) * nCrystalsInModule + row * 2 + (col * nCellsXInStrip + (cell - 1) / 2) * nZ -
64  (cell & 1 ? 1 : 0);
65 }

67 bool Geometry::absToRelNumbering(short absId, char* relid)
68 {
69  // Converts the absolute numbering into the following array
70  // relid[0] = PHOS Module number 1:fNModules
71  // relid[1] = Row number inside a PHOS module (Z coordinate)
72  // relid[2] = Column number inside a PHOS module (Phi coordinate)
73  const short nZ = 56; // nStripZ * nCellsZInStrip
74  const short nPhi = 64; // nStripZ * nCellsZInStrip
75  absId--;
76  short phosmodulenumber = absId / (nZ * nPhi);
77 
78  relid[0] = phosmodulenumber + 1;
79  absId -= phosmodulenumber * nPhi * nZ;
80  relid[1] = 1 + absId / nZ;
81  relid[2] = absId - (relid[1] - 1) * nZ + 1;
82 
83  return true;
84 }

125 int Geometry::areNeighbours(short absId1, short absId2)
126 {
127 
128  // Gives the neighbourness of two digits = 0 are not neighbour but continue searching
129  // = 1 are neighbour
130  // = 2 are not neighbour but do not continue searching
131  // =-1 are not neighbour, continue searching, but do not look before d2 next
132  // time
133  // neighbours are defined as digits having at least a common vertex
134  // The order of d1 and d2 is important: first (d1) should be a digit already in a cluster
135  // which is compared to a digit (d2) not yet in a cluster
136 
137  char relid1[3];
138  absToRelNumbering(absId1, relid1);
139 
140  char relid2[3];
141  absToRelNumbering(absId2, relid2);
142 
143  if (relid1[0] == relid2[0]) { // inside the same PHOS module
144  char rowdiff = TMath::Abs(relid1[1] - relid2[1]);
145  char coldiff = TMath::Abs(relid1[2] - relid2[2]);
146 
147  if ((coldiff <= 1) && (rowdiff <= 1)) { // At least common vertex
148  return 1;
149  } else {
150  if ((relid2[1] > relid1[1]) && (relid2[2] > relid1[2] + 1)) {
151  return 2; // Difference in row numbers is too large to look further
152  }
153  }
154  return 0;
155 
156  } else {
157  if (relid1[0] > relid2[0]) { // we switched to the next module
158  return -1;
159  }
160  return 2;
161  }
162  return 0;
163 }

209 bool Geometry::impactOnPHOS(const TVector3& vtx, const TVector3& p,
210  short& module, float& z, float& x) const
211 {
212  // calculates the impact coordinates on PHOS of a neutral particle
213  // emitted in the vertex vtx with 3-momentum p
214  constexpr float shiftY = -1.26; //Depth-optimized
215  constexpr float moduleXhalfSize = 72.16; // 18.04 / 2 * 8
216  constexpr float moduleZhalfSize = 64.14; // 4.51 / 2 * 28
217 
218  for (short mod = 1; mod < 5; mod++) {
219  //create vector from (0,0,0) to center of crystal surface of imod module
220  double tmp[3] = {0., 0., shiftY};
221  double posG[3] = {0., 0., 0.};
222  mPHOS[mod].LocalToMaster(tmp, posG);
223  TVector3 n(posG[0], posG[1], posG[2]);
224  double direction = n.Dot(p);
225  if (direction <= 0.) {
226  continue; //momentum directed FROM module
227  }
228  double fr = (n.Mag2() - n.Dot(vtx)) / direction;
229  //Calculate direction in module plane
230  n -= vtx + fr * p;
231  n *= -1.;
232  if (TMath::Abs(n.Z()) < moduleZhalfSize && n.Pt() < moduleXhalfSize) {
233  module = mod;
234  z = n.Z();
235  x = TMath::Sign(n.Pt(), n.X());
236  //no need to return to local system since we calcilated distance from module center
237  //and tilts can not be significant.
238  return true;
239  }
240  }
241  //Not in acceptance
242  x = 0;
243  z = 0;
244  module = 0;
245  return false;
246 }
