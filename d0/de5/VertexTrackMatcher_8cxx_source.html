36 void VertexTrackMatcher::updateTimeDependentParams()
37 {
38  // tpc time bin in microseconds
39  if (mMaxTPCDriftTimeMUS == 0) {
40  auto& gasParam = o2::tpc::ParameterGas::Instance();
41  auto& elParam = o2::tpc::ParameterElectronics::Instance();
42  auto& detParam = o2::tpc::ParameterDetector::Instance();
43  mTPCBin2MUS = elParam.ZbinWidth;
44  mMaxTPCDriftTimeMUS = detParam.TPClength / gasParam.DriftV;
45  }
46  if (mITSROFrameLengthMUS == 0) {
47  std::unique_ptr<o2::parameters::GRPObject> grp{o2::parameters::GRPObject::loadFrom()};
48  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
49  mITSROFrameLengthMUS = grp->isDetContinuousReadOut(o2::detectors::DetID::ITS) ? alpParams.roFrameLengthInBC * o2::constants::lhc::LHCBunchSpacingMUS : alpParams.roFrameLengthTrig * 1.e-3;
50  LOG(info) << "VertexTrackMatcher::ITSROFrameLengthMUS = " << mITSROFrameLengthMUS;
51  }
52  if (mMFTROFrameLengthMUS == 0) {
53  std::unique_ptr<o2::parameters::GRPObject> grp{o2::parameters::GRPObject::loadFrom()};
54  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::MFT>::Instance();
55  mMFTROFrameLengthMUS = grp->isDetContinuousReadOut(o2::detectors::DetID::MFT) ? alpParams.roFrameLengthInBC * o2::constants::lhc::LHCBunchSpacingMUS : alpParams.roFrameLengthTrig * 1.e-3;
56  LOG(info) << "VertexTrackMatcher::MFTROFrameLengthMUS = " << mMFTROFrameLengthMUS;
57  }
58 }

60 void VertexTrackMatcher::process(const o2::globaltracking::RecoContainer& recoData,
61  std::vector<VTIndex>& trackIndex,
62  std::vector<VRef>& vtxRefs)
63 {
64  updateTimeDependentParams();
65 
66  auto vertices = recoData.getPrimaryVertices();
67  auto v2tfitIDs = recoData.getPrimaryVertexContributors();
68  auto v2tfitRefs = recoData.getPrimaryVertexContributorsRefs();
69 
70  int nv = vertices.size(), nv1 = nv + 1;
71  TmpMap tmpMap(nv1);
72  auto& orphans = tmpMap.back(); // in the last element we store unassigned track indices
73 
74  // register vertex contributors
75  std::unordered_map<GIndex, bool> vcont;
76  std::vector<VtxTBracket> vtxOrdBrack; // vertex indices and brackets sorted in tmin
77  float maxVtxSpan = 0;
78  for (int iv = 0; iv < nv; iv++) {
79  int idMin = v2tfitRefs[iv].getFirstEntry(), idMax = idMin + v2tfitRefs[iv].getEntries();
80  auto& vtxIds = tmpMap[iv]; // global IDs of contibuting tracks
81  vtxIds.reserve(v2tfitRefs[iv].getEntries());
82  for (int id = idMin; id < idMax; id++) {
83  auto gid = v2tfitIDs[id];
84  vtxIds.emplace_back(gid).setPVContributor();
85  vcont[gid] = true;
86  }
87  const auto& vtx = vertices[iv];
88  const auto& vto = vtxOrdBrack.emplace_back(VtxTBracket{
89  {float((vtx.getIRMin().differenceInBC(recoData.startIR) - 0.5f) * o2::constants::lhc::LHCBunchSpacingMUS - mPVParams->timeMarginVertexTime),
90  float((vtx.getIRMax().differenceInBC(recoData.startIR) + 0.5f) * o2::constants::lhc::LHCBunchSpacingMUS + mPVParams->timeMarginVertexTime)},
91  iv});
92  if (vto.tBracket.delta() > maxVtxSpan) {
93  maxVtxSpan = vto.tBracket.delta();
94  }
95  }
96  // sort vertices in tmin
97  std::sort(vtxOrdBrack.begin(), vtxOrdBrack.end(), [](const VtxTBracket& a, const VtxTBracket& b) { return a.tBracket.getMin() < b.tBracket.getMin(); });
98 
99  extractTracks(recoData, vcont); // extract all track t-brackets, excluding those tracks which contribute to vertex (already attached)
100 
101  int ivStart = 0, nAssigned = 0, nAmbiguous = 0;
102  std::vector<int> vtxList; // list of vertices which match to checked track
103  for (const auto& tro : mTBrackets) {
104  vtxList.clear();
105  for (int iv = ivStart; iv < nv; iv++) {
106  const auto& vto = vtxOrdBrack[iv];
107  auto res = tro.tBracket.isOutside(vto.tBracket);
108  if (res == TBracket::Below) { // vertex preceeds the track
109  if (tro.tBracket.getMin() > vto.tBracket.getMin() + maxVtxSpan) { // all following vertices will be preceeding all following tracks times
110  ivStart = ++iv;
111  break;
112  }
113  continue; // following vertex with longer span might still match this track
114  }
115  if (res == TBracket::Above) { // track preceeds the vertex, so will preceed also all following vertices
116  break;
117  }
118  // track matches to vertex, register
119  vtxList.push_back(vto.origID); // flag matching vertex
120  }
121  if (vtxList.size()) {
122  nAssigned++;
123  bool ambig = vtxList.size() > 1;
124  for (auto v : vtxList) {
125  auto& ref = tmpMap[v].emplace_back(tro.origID);
126  if (ambig) {
127  ref.setAmbiguous();
128  }
129  }
130  if (ambig) { // did track match to multiple vertices?
131  nAmbiguous++;
132  }
133  } else {
134  orphans.emplace_back(tro.origID); // register unassigned track
135  }
136  }
137 
138  // build final vector of global indices
139  trackIndex.clear();
140  vtxRefs.clear();
141 
142  for (int iv = 0; iv < nv1; iv++) {
143  auto& trvec = tmpMap[iv];
144  // sort entries in each vertex track indices list according to the source
145  std::sort(trvec.begin(), trvec.end(), [](VTIndex a, VTIndex b) { return a.getSource() < b.getSource(); });
146 
147  auto entry0 = trackIndex.size(); // start of entries for this vertex
148  auto& vr = vtxRefs.emplace_back();
149  vr.setVtxID(iv < nv ? iv : -1); // flag table for unassigned tracks by VtxID = -1
150  int oldSrc = -1;
151  for (const auto gid0 : trvec) {
152  int src = gid0.getSource();
153  while (oldSrc < src) {
154  oldSrc++;
155  vr.setFirstEntryOfSource(oldSrc, trackIndex.size()); // register start of new source
156  }
157  trackIndex.push_back(gid0);
158  }
159  while (++oldSrc < GIndex::NSources) {
160  vr.setFirstEntryOfSource(oldSrc, trackIndex.size());
161  }
162  vr.setEnd(trackIndex.size());
163  LOG(info) << vr;
164  }
165  LOG(info) << "Assigned " << nAssigned << " (" << nAmbiguous << " ambiguously) out of " << mTBrackets.size() << " non-contributor tracks + " << vcont.size() << " contributors";
166 }

169 void VertexTrackMatcher::extractTracks(const o2::globaltracking::RecoContainer& data, const std::unordered_map<GIndex, bool>& vcont)
170 {
171  // Scan all inputs and create tracks
172 
173  mTBrackets.clear();
174 
175  auto creator = [this, &vcont](auto& _tr, GIndex _origID, float t0, float terr) {
176  if (vcont.find(_origID) != vcont.end()) { // track is contributor to vertex, already accounted
177  return true;
178  }
179  if constexpr (isTPCTrack<decltype(_tr)>()) {
180  // unconstrained TPC track, with t0 = TrackTPC.getTime0+0.5*(DeltaFwd-DeltaBwd) and terr = 0.5*(DeltaFwd+DeltaBwd) in TimeBins
181  t0 *= this->mTPCBin2MUS;
182  terr *= this->mTPCBin2MUS;
183  } else if constexpr (isITSTrack<decltype(_tr)>()) {
184  t0 += 0.5 * this->mITSROFrameLengthMUS; // ITS time is supplied in \mus as beginning of ROF
185  terr *= this->mITSROFrameLengthMUS; // error is supplied as a half-ROF duration, convert to \mus
186  } else if constexpr (isMFTTrack<decltype(_tr)>()) { // Same for MFT
187  t0 += 0.5 * this->mMFTROFrameLengthMUS;
188  terr *= this->mMFTROFrameLengthMUS;
189  } else if constexpr (isGlobalFwdTrack<decltype(_tr)>()) {
190  t0 = _tr.getTimeMUS().getTimeStamp();
191  terr = _tr.getTimeMUS().getTimeStampError() * mPVParams->nSigmaTimeTrack; // gaussian errors must be scaled by requested n-sigma
192  } else {
193  terr *= mPVParams->nSigmaTimeTrack; // gaussian errors must be scaled by requested n-sigma
194  }
195  // for all other tracks the time is in \mus with gaussian error
196  terr += mPVParams->timeMarginTrackTime;
197  mTBrackets.emplace_back(TrackTBracket{{t0 - terr, t0 + terr}, _origID});
198  return true;
199  };
200 
201  data.createTracksVariadic(creator);
202 
203  // sort in increasing min.time
204  std::sort(mTBrackets.begin(), mTBrackets.end(), [](const TrackTBracket& a, const TrackTBracket& b) { return a.tBracket.getMin() < b.tBracket.getMin(); });
205 
206  LOG(info) << "collected " << mTBrackets.size() << " non-contributor and " << vcont.size() << " contributor seeds";
207 }
