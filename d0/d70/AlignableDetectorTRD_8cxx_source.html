60 void AlignableDetectorTRD::defineVolumes()
61 {
62  // define TRD volumes
63  //
64  const int kNSect = 18, kNStacks = 5, kNLayers = 6;
65  AlignableSensorTRD* chamb = 0;
66  //
67  int labDet = getDetLabel();
68  // AddVolume( volTRD = new AlignableVolume("TRD") ); // no main volume, why?
69  AlignableVolume* sect[kNSect] = {0};
70  //
71  for (int ilr = 0; ilr < kNLayers; ilr++) { // layer
72  for (int ich = 0; ich < kNStacks * kNSect; ich++) { // chamber
73  int isector = ich / AliTRDgeometry::Nstack();
74  int istack = ich % AliTRDgeometry::Nstack();
75  //int lid = AliTRDgeometry::GetDetector(ilr,istack,isector);
76  int iid = labDet + (1 + ilr) * 10000 + (1 + isector) * 100 + (1 + istack);
77  const char* symname = Form("TRD/sm%02d/st%d/pl%d", isector, istack, ilr);
78  if (!gGeoManager->GetAlignableEntry(symname))
79  continue;
80  uint16_t vid = AliGeomManager::LayerToVolUID(AliGeomManager::kTRD1 + ilr, ich);
81  addVolume(chamb = new AlignableSensorTRD(symname, vid, iid /*lid*/, isector));
82  iid = labDet + (1 + isector) * 100;
83  if (!sect[isector])
84  sect[isector] = new AlignableVolume(Form("TRD/sm%02d", isector), iid);
85  chamb->setParent(sect[isector]);
86  } // chamber
87  } // layer
88  //
89  for (int isc = 0; isc < kNSect; isc++) {
90  if (sect[isc])
91  addVolume(sect[isc]);
92  }
93  //
94 }

123 void AlignableDetectorTRD::writePedeInfo(FILE* parOut, const Option_t* opt) const
124 {
125  // contribute to params and constraints template files for PEDE
126  AlignableDetector::writePedeInfo(parOut, opt);
127  //
128  // write calibration parameters
129  enum { kOff,
130  kOn,
131  kOnOn };
132  const char* comment[3] = {" ", "! ", "!!"};
133  const char* kKeyParam = "parameter";
134  //
135  fprintf(parOut, "%s%s %s\t %d calibraction params for %s\n", comment[kOff], kKeyParam, comment[kOnOn],
136  getNCalibDOFs(), GetName());
137  //
138  for (int ip = 0; ip < getNCalibDOFs(); ip++) {
139  int cmt = isCondDOF(ip) ? kOff : kOn;
140  fprintf(parOut, "%s %9d %+e %+e\t%s %s p%d\n", comment[cmt], getParLab(ip),
141  getParVal(ip), getParErr(ip), comment[kOnOn], isFreeDOF(ip) ? " " : "FX", ip);
142  }
143  //
144 }
