57 void DataSampling::DoGenerateInfrastructure(Dispatcher& dispatcher, WorkflowSpec& workflow, const boost::property_tree::ptree& policiesTree, size_t threads, const std::string& host)
58 {
59  LOG(debug) << "Generating Data Sampling infrastructure...";
60 
61  for (auto&& policyConfig : policiesTree) {
62 
63  // We don't want the Dispatcher to exit due to one faulty Policy
64  try {
65  auto policy = DataSamplingPolicy::fromConfiguration(policyConfig.second);
66  std::vector<std::string> machines;
67  if (policyConfig.second.count("machines") > 0) {
68  for (const auto& machine : policyConfig.second.get_child("machines")) {
69  machines.emplace_back(machine.second.get<std::string>(""));
70  }
71  }
72  if (host.empty() || machines.empty() || std::find(machines.begin(), machines.end(), host) != machines.end()) {
73  dispatcher.registerPolicy(std::make_unique<DataSamplingPolicy>(std::move(policy)));
74  }
75  } catch (const std::exception& ex) {
76  LOG(warn) << "Could not load the Data Sampling Policy '"
77  << policyConfig.second.get_optional<std::string>("id").value_or("") << "', because: " << ex.what();
78  continue;
79  } catch (...) {
80  LOG(warn) << "Could not load the Data Sampling Policy '"
81  << policyConfig.second.get_optional<std::string>("id").value_or("") << "'";
82  continue;
83  }
84  }
85 
86  if (dispatcher.numberOfPolicies() > 0) {
87  DataProcessorSpec spec;
88  spec.name = dispatcher.getName();
89  spec.inputs = dispatcher.getInputSpecs();
90  spec.outputs = dispatcher.getOutputSpecs();
91  spec.maxInputTimeslices = threads;
92  spec.labels = {{"DataSampling"}, {"Dispatcher"}};
93  spec.options = dispatcher.getOptions();
94  spec.algorithm = adaptFromTask<Dispatcher>(std::move(dispatcher));
95 
96  workflow.emplace_back(std::move(spec));
97  } else {
98  LOG(debug) << "No input to this dispatcher, it won't be added to the workflow.";
99  }
100 }

114 std::vector<framework::InputSpec> DataSampling::InputSpecsForPolicy(const boost::property_tree::ptree& policiesTree, const std::string& policyName)
115 {
116  std::vector<InputSpec> inputs;
117  for (auto&& policyConfig : policiesTree) {
118  if (policyConfig.second.get<std::string>("id") == policyName) {
119  auto policy = DataSamplingPolicy::fromConfiguration(policyConfig.second);
120  for (const auto& path : policy.getPathMap()) {
121  InputSpec input = DataSpecUtils::matchingInput(path.second);
122  inputs.push_back(input);
123  }
124  break;
125  }
126  }
127  return inputs;
128 }
