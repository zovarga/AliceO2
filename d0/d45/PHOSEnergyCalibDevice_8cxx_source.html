25 void PHOSEnergyCalibDevice::init(o2::framework::InitContext& ic)
26 {
27  mCalibrator.reset(new PHOSEnergyCalibrator());
28 
29  //Configure output Digits file
30  mCalibrator->setOutDigitsFile(mdigitsfilename);
31 
32  //read calibration and bad map objects and send them to calibrator
33  if (!mBadMap) {
34  if (mUseCCDB) {
35  LOG(info) << "Retrieving BadMap from CCDB";
36  o2::ccdb::CcdbApi ccdb;
37  ccdb.init(mCCDBPath); // or http://localhost:8080 for a local installation
38  std::map<std::string, std::string> metadata;
39  mBadMap.reset(ccdb.retrieveFromTFileAny<BadChannelsMap>("PHS/Calib/BadChannels", metadata, mRunStartTime));
40 
41  if (!mBadMap) { //was not read from CCDB, but expected
42  LOG(fatal) << "Can not read BadMap from CCDB, you may use --not-use-ccdb option to create default bad map";
43  }
44  //same for calibration
45 
46  mCalibParams.reset(ccdb.retrieveFromTFileAny<CalibParams>("PHS/Calib/CalibParams", metadata, mRunStartTime));
47  if (!mCalibParams) { //was not read from CCDB, but expected
48  LOG(fatal) << "Can not read current CalibParams from ccdb";
49  }
50  } else {
51  LOG(info) << "Do not use CCDB, create default BadMap and calibration";
52  mBadMap.reset(new BadChannelsMap(1));
53  mCalibParams.reset(new CalibParams(1));
54  }
55  }
56  mCalibrator->setBadMap(*mBadMap);
57  mCalibrator->setCalibration(*mCalibParams);
58  mCalibrator->setCuts(mPtMin, mEminHGTime, mEminLGTime);
59 
60  //Create geometry instance (inclusing reading mis-alignement)
61  //instance will be pick up by Calibrator
62  Geometry::GetInstance("Run3");
63 }

65 void PHOSEnergyCalibDevice::run(o2::framework::ProcessingContext& pc)
66 {
67  //TODO! extract vertex information and send to Calibrator
68  auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("clusters").header)->startTime; // is this the timestamp of the current TF?
69  const gsl::span<const Cluster>& clusters = pc.inputs().get<gsl::span<Cluster>>("clusters");
70  const gsl::span<const CluElement>& cluelements = pc.inputs().get<gsl::span<CluElement>>("cluelements");
71  const gsl::span<const TriggerRecord>& cluTR = pc.inputs().get<gsl::span<TriggerRecord>>("clusterTriggerRecords");
72 
73  LOG(info) << "[PHOSEnergyCalibDevice - run] Received " << clusters.size() << " clusters and " << clusters.size() << " clusters, running calibration";
74  if (mRunStartTime == 0) {
75  mRunStartTime = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("clusterTriggerRecords").header)->startTime;
76  }
77 
78  mCalibrator->process(tfcounter, clusters, cluelements, cluTR);
79 }

91 void PHOSEnergyCalibDevice::postHistosCCDB(o2::framework::EndOfStreamContext& ec)
92 {
93  // prepare all info to be sent to CCDB
94  auto flName = o2::ccdb::CcdbApi::generateFileName("TimeEnHistos");
95  std::map<std::string, std::string> md;
96  o2::ccdb::CcdbObjectInfo info("PHS/Calib/TimeEnHistos", "TimeEnHistos", flName, md, mRunStartTime, 99999999999999);
97  info.setMetaData(md);
98  auto image = o2::ccdb::CcdbApi::createObjectImage(mCalibrator->getCollectedHistos(), &info);
99 
100  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName()
101  << " of size " << image->size()
102  << " bytes, valid for " << info.getStartValidityTimestamp()
103  << " : " << info.getEndValidityTimestamp();
104 
105  header::DataHeader::SubSpecificationType subSpec{(header::DataHeader::SubSpecificationType)0};
106  ec.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "PHOS_TEHistos", subSpec}, *image.get());
107  ec.outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "PHOS_TEHistos", subSpec}, info);
108 }
