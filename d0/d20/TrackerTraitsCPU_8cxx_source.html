262  const int currentLayerTrackletsNum{static_cast<int>(tf->getTracklets()[iLayer].size())};
263 
264  for (int iTracklet{0}; iTracklet < currentLayerTrackletsNum; ++iTracklet) {
265 
266  const Tracklet& currentTracklet{tf->getTracklets()[iLayer][iTracklet]};
267  const int nextLayerClusterIndex{currentTracklet.secondClusterIndex};
268  const int nextLayerFirstTrackletIndex{
269  tf->getTrackletsLookupTable()[iLayer][nextLayerClusterIndex]};
270  const int nextLayerLastTrackletIndex{
271  tf->getTrackletsLookupTable()[iLayer][nextLayerClusterIndex + 1]};
272 
273  if (nextLayerFirstTrackletIndex == nextLayerLastTrackletIndex) {
274  continue;
275  }
276 
277  for (int iNextTracklet{nextLayerFirstTrackletIndex}; iNextTracklet < nextLayerLastTrackletIndex; ++iNextTracklet) {
278  if (tf->getTracklets()[iLayer + 1][iNextTracklet].firstClusterIndex != nextLayerClusterIndex) {
279  break;
280  }
281  const Tracklet& nextTracklet{tf->getTracklets()[iLayer + 1][iNextTracklet]};
282  const float deltaTanLambda{std::abs(currentTracklet.tanLambda - nextTracklet.tanLambda)};
283  const float tanLambda{(currentTracklet.tanLambda + nextTracklet.tanLambda) * 0.5f};
284 
285 #ifdef OPTIMISATION_OUTPUT
286  bool good{tf->getTrackletsLabel(iLayer)[iTracklet] == tf->getTrackletsLabel(iLayer + 1)[iNextTracklet]};
287  float signedDelta{currentTracklet.tanLambda - nextTracklet.tanLambda};
288  off << fmt::format("{}\t{:d}\t{}\t{}\t{}\t{}", iLayer, good, signedDelta, signedDelta / (mTrkParams.CellDeltaTanLambdaSigma), tanLambda, resolution) << std::endl;
289 #endif
290 
291  if (deltaTanLambda / mTrkParams.CellDeltaTanLambdaSigma < mTrkParams.NSigmaCut) {
292 
293  if (iLayer > 0 && tf->getCellsLookupTable()[iLayer - 1].size() <= iTracklet) {
294  tf->getCellsLookupTable()[iLayer - 1].resize(iTracklet + 1, tf->getCells()[iLayer].size());
295  }
296 
297  tf->getCells()[iLayer].emplace_back(
298  currentTracklet.firstClusterIndex, nextTracklet.firstClusterIndex, nextTracklet.secondClusterIndex,
299  iTracklet, iNextTracklet, tanLambda);
300  }
301  }
302  }
303  if (iLayer > 0) {
304  tf->getCellsLookupTable()[iLayer - 1].resize(currentLayerTrackletsNum + 1, tf->getCells()[iLayer].size());
305  }
306  }
307 
309  if (tf->hasMCinformation()) {
310  for (int iLayer{0}; iLayer < mTrkParams.CellsPerRoad(); ++iLayer) {
311  for (auto& cell : tf->getCells()[iLayer]) {
312  MCCompLabel currentLab{tf->getTrackletsLabel(iLayer)[cell.getFirstTrackletIndex()]};
313  MCCompLabel nextLab{tf->getTrackletsLabel(iLayer + 1)[cell.getSecondTrackletIndex()]};

315  }
316  }
317  }
318 
319  if constexpr (debugLevel) {

321  std::cout << "Cells on layer " << iLayer << " " << tf->getCells()[iLayer].size() << std::endl;
