73  inline void addInfo(uint8_t ch, uint16_t code)
74  {
75  if (ch >= NChannels && ch != 0x1f) {
76  LOGF(error, "Adding info (0x%x) for not existent channel %u", code, ch);
77  return;

87  void addInfo(const std::array<bool, NChannels>& vec, const uint16_t code)
88  {
89  // Prepare list of channels interested by this message
90  int cnt = 0;
91  std::array<int, NChannels> active;
92  for (uint8_t ich = 0; ich < NChannels; ich++) {
93  if (vec[ich] == true) {
94  active[cnt] = ich;
95  cnt++;
96  }
97  }
98  if (cnt == 0) {
99  return;
100  }
101 #ifdef O2_ZDC_DEBUG
102  printf("addInfo(");
103  for (uint8_t ich = 0; ich < NChannels; ich++) {
104  if (vec[ich] == true) {
105  printf("1");
106  } else {
107  printf("0");
108  }
109  }
110  printf(", code=%u \"%s\") %d active.\n", code, MsgText[code].data(), cnt);
111 #endif
112  if (cnt <= 3) {
113  // Transmission of single channels
114  for (uint8_t i = 0; i < cnt; i++) {
115  addInfo(active[i], code);
116  }
117  } else {
118  // Transmission of channel pattern
119  uint16_t ch = 0x1f;
120  addInfo(ch, code);
121  uint16_t info = 0x8000;
122  uint8_t i = 0;
123  for (; i < cnt && active[i] < 15; i++) {
124  info = info | (0x1 << active[i]);
125  }
126  addInfo(info);
127  info = 0x8000;
128  for (; i < cnt; i++) {
129  info = info | (0x1 << (active[i] - 15));
130  }
131  addInfo(info);
132  }
133  }
