33 o2::header::RDHAny RawReaderMemory::decodeRawHeader(const void* payloadwords)
34 {
35  auto headerversion = RDHDecoder::getVersion(payloadwords);
36  if (headerversion == 4) {
37  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV4*>(payloadwords));
38  } else if (headerversion == 5) {
39  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV5*>(payloadwords));
40  } else if (headerversion == 6) {
41  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV6*>(payloadwords));
42  }
43  throw RawDecodingError(RawDecodingError::ErrorType_t::HEADER_DECODING, mCurrentFEE);
44 }

54 void RawReaderMemory::next()
55 {
56  mRawPayload.reset();
57  mCurrentTrailer.reset();
58  bool isDataTerminated = false;
59  do {
60  nextPage(false);
61  if (hasNext()) {
62  auto nextheader = decodeRawHeader(mRawMemoryBuffer.data() + mCurrentPosition);
70  auto nextpagecounter = RDHDecoder::getPageCounter(nextheader);
71  if (nextpagecounter == 0) {
72  isDataTerminated = true;
73  } else {
74  isDataTerminated = false;
75  }
76  } else {
77  isDataTerminated = true;
78  }
79  // Check if the data continues
80  } while (!isDataTerminated);
81  // add combined trailer to payload (in case the FEE is a SRU DDL)
82  if (mCurrentTrailer.isInitialized()) {
83  mRawPayload.appendPayloadWords(mCurrentTrailer.encode());
84  }
85 }

87 void RawReaderMemory::nextPage(bool doResetPayload)
88 {
89  if (!hasNext()) {
90  throw RawDecodingError(RawDecodingError::ErrorType_t::PAGE_NOTFOUND, mCurrentFEE);
91  }
92  if (doResetPayload) {
93  mRawPayload.reset();
94  }
95  mRawHeaderInitialized = false;
96  mPayloadInitialized = false;
97  // Read header
98  try {
99  mRawHeader = decodeRawHeader(mRawMemoryBuffer.data() + mCurrentPosition);
100  auto feeID = RDHDecoder::getFEEID(mRawHeader);
101  if (mCurrentFEE < 0 || mCurrentFEE != feeID) {
102  // update current FEE ID
103  mCurrentFEE = feeID;
104  }
105  //RDHDecoder::printRDH(mRawHeader);
106  if (RDHDecoder::getOffsetToNext(mRawHeader) == RDHDecoder::getHeaderSize(mRawHeader)) {
107  // No Payload - jump to next rawheader
108  // This will eventually move, depending on whether for events without payload in the SRU we send the RCU trailer
109  mCurrentPosition += RDHDecoder::getHeaderSize(mRawHeader);
110  mRawHeader = decodeRawHeader(mRawMemoryBuffer.data() + mCurrentPosition);
111  feeID = RDHDecoder::getFEEID(mRawHeader);
112  if (mCurrentFEE < 0 || mCurrentFEE != feeID) {
113  // update current FEE ID
114  mCurrentFEE = feeID;
115  }
116  //RDHDecoder::printRDH(mRawHeader);
117  }
118  mRawHeaderInitialized = true;
119  } catch (...) {
120  throw RawDecodingError(RawDecodingError::ErrorType_t::HEADER_DECODING, mCurrentFEE);
121  }
122  if (mCurrentPosition + RDHDecoder::getMemorySize(mRawHeader) > mRawMemoryBuffer.size()) {
123  // Payload incomplete
124  throw RawDecodingError(RawDecodingError::ErrorType_t::PAYLOAD_DECODING, mCurrentFEE);
125  } else if (mCurrentPosition + RDHDecoder::getHeaderSize(mRawHeader) > mRawMemoryBuffer.size()) {
126  // Start position of the payload is outside the payload range
127  throw RawDecodingError(RawDecodingError::ErrorType_t::PAGE_START_INVALID, mCurrentFEE);
128  } else {
129  mRawBuffer.readFromMemoryBuffer(gsl::span<const char>(mRawMemoryBuffer.data() + mCurrentPosition + RDHDecoder::getHeaderSize(mRawHeader), RDHDecoder::getMemorySize(mRawHeader) - RDHDecoder::getHeaderSize(mRawHeader)));
130 
131  gsl::span<const uint32_t> payloadWithoutTrailer;
132  auto feeID = RDHDecoder::getFEEID(mRawHeader);
133  if (feeID >= mMinSRUDDL && feeID <= mMaxSRUDDL) {
134  // Read off and chop trailer (if required)
135  //
136  // In case every page gets a trailer (intermediate format). The trailers from the single
137  // pages need to be removed. There will be a combined trailer which keeps the sum of the
138  // payloads for all trailers. This will be appended to the chopped payload.
139  //
140  // Trailer only at the last page (new format): Only last page gets trailer. The trailer is
141  // also chopped from the payload as it will be added later again.
142  //
143  // The trailer is only decoded if the DDL is in the range of SRU DDLs. STU pages are propagated
144  // 1-1 without trailer parsing
145  auto lastword = *(mRawBuffer.getDataWords().rbegin());
146  if (lastword >> 30 == 3) {
147  // lastword is a trailer word
148  // decode trailer and chop
149  auto trailer = RCUTrailer::constructFromPayloadWords(mRawBuffer.getDataWords());
150  if (!mCurrentTrailer.isInitialized()) {
151  mCurrentTrailer = trailer;
152  } else {
153  mCurrentTrailer.setPayloadSize(mCurrentTrailer.getPayloadSize() + trailer.getPayloadSize());
154  }
155  payloadWithoutTrailer = gsl::span<const uint32_t>(mRawBuffer.getDataWords().data(), mRawBuffer.getDataWords().size() - trailer.getTrailerSize());
156  } else {
157  // Not a trailer word = copy page as it is
158  payloadWithoutTrailer = mRawBuffer.getDataWords(); // No trailer to be chopped
159  }
160  } else {
161  payloadWithoutTrailer = mRawBuffer.getDataWords(); // No trailer to be chopped
162  }
163 
164  mRawPayload.appendPayloadWords(payloadWithoutTrailer);
165  mRawPayload.increasePageCount();
166  }
167  mCurrentPosition += RDHDecoder::getOffsetToNext(mRawHeader);
168 }
