24 void FastMultEst::fillNClPerLayer(const gsl::span<const o2::itsmft::CompClusterExt>& clusters)
25 {
26  int lr = FastMultEst::NLayers - 1, nchAcc = o2::itsmft::ChipMappingITS::getNChips() - o2::itsmft::ChipMappingITS::getNChipsPerLr(lr);
27  std::memset(&nClPerLayer[0], 0, sizeof(int) * FastMultEst::NLayers);
28  for (int i = clusters.size(); i--;) { // profit from clusters being ordered in chip increasing order
29  while (clusters[i].getSensorID() < nchAcc) {
30  assert(lr >= 0);
31  nchAcc -= o2::itsmft::ChipMappingITS::getNChipsPerLr(--lr);
32  }
33  nClPerLayer[lr]++;
34  }
35 }

39 float FastMultEst::processNoiseFree(const std::array<int, NLayers> ncl)
40 {
41  // we assume that on the used layers the observed number of clusters is defined by the
42  // the noise ~ nu * Nchips and contribution from the signal tracks Ntr*mAccCorr
43  const auto& conf = FastMultEstConfig::Instance();
44 
45  float mat[3] = {0}, b[2] = {0};
46  nLayersUsed = 0;
47  for (int il = conf.firstLayer; il <= conf.lastLayer; il++) {
48  if (ncl[il] > 0) {
49  int nch = o2::itsmft::ChipMappingITS::getNChipsPerLr(il);
50  float err2i = 1. / ncl[il];
51  float m2n = nch * err2i;
52  mat[0] += err2i * conf.accCorr[il] * conf.accCorr[il];
53  mat[2] += nch * m2n;
54  mat[1] += conf.accCorr[il] * m2n; // non-diagonal element
55  b[0] += conf.accCorr[il];
56  b[1] += nch;
57  nLayersUsed++;
58  }
59  }
60  mult = noisePerChip = chi2 = -1;
61  float det = mat[0] * mat[2] - mat[1] * mat[1];
62  if (nLayersUsed < 2 || std::abs(det) < 1e-15) {
63  return -1;
64  }
65  float detI = 1. / det;
66  mult = detI * (b[0] * mat[2] - b[1] * mat[1]);
67  noisePerChip = detI * (b[1] * mat[0] - b[0] * mat[1]);
68  cov[0] = mat[2] * detI;
69  cov[2] = mat[0] * detI;
70  cov[1] = -mat[1] * detI;
71  chi2 = 0.;
72  for (int il = conf.firstLayer; il <= conf.lastLayer; il++) {
73  if (ncl[il] > 0) {
74  int nch = o2::itsmft::ChipMappingITS::getNChipsPerLr(il);
75  float err2i = 1. / ncl[il];
76  float diff = mult * conf.accCorr[il] + nch * noisePerChip - ncl[il];
77  chi2 += diff * diff / ncl[il];
78  }
79  }
80  chi2 = nLayersUsed > 2 ? chi2 / (nLayersUsed - 2) : 0.;
81  return mult > 0 ? mult : 0;
82 }

86 float FastMultEst::processNoiseImposed(const std::array<int, NLayers> ncl)
87 {
88  // we assume that on the used layers the observed number of clusters is defined by the
89  // the noise ~ nu * Nchips and contribution from the signal tracks Ntr*conf.accCorr
90  //
91  // minimize the form sum_lr (noise_i - mu nchips_i)^2 / (mu nchips_i) + lambda_i * (noise_i + mult*acc_i - ncl_i)
92  // whith noise_i being estimate of the noise clusters in nchips_i of layer i, mu is the mean noise per chip,
93  // mult is the number of signal clusters on the ref. (1st) layer and the acc_i is the acceptance of layer i wrt 1st.
94  // The lambda_i is hust a Lagrange multiplier.
95 
96  const auto& conf = FastMultEstConfig::Instance();
97  float w2sum = 0., wnsum = 0., wsum = 0.;
98  nLayersUsed = 0;
99  for (int il = conf.firstLayer; il <= conf.lastLayer; il++) {
100  if (ncl[il] > 0) {
101  float nchInv = 1. / o2::itsmft::ChipMappingITS::getNChipsPerLr(il);
102  w2sum += conf.accCorr[il] * conf.accCorr[il] * nchInv;
103  wnsum += ncl[il] * nchInv * conf.accCorr[il];
104  wsum += conf.accCorr[il];
105  nLayersUsed++;
106  }
107  }
108  mult = 0;
109  chi2 = -1;
110  noisePerChip = conf.imposeNoisePerChip;
111  if (nLayersUsed < 1) {
112  return -1;
113  }
114  auto w2sumI = 1. / w2sum;
115  mult = (wnsum - noisePerChip * wsum) * w2sumI;
116  cov[0] = wnsum * w2sumI;
117  cov[2] = 0.;
118  cov[1] = 0.;
119 
120  chi2 = 0.;
121  for (int il = conf.firstLayer; il <= conf.lastLayer; il++) {
122  if (ncl[il] > 0) {
123  int nch = o2::itsmft::ChipMappingITS::getNChipsPerLr(il);
124  float noise = ncl[il] - mult * conf.accCorr[il], estNoise = o2::itsmft::ChipMappingITS::getNChipsPerLr(il) * noisePerChip;
125  float diff = noise - estNoise;
126  chi2 += diff * diff / estNoise;
127  }
128  }
129  chi2 = nLayersUsed > 2 ? chi2 / (nLayersUsed - 2) : 0.;
130  return mult > 0 ? mult : 0;
131 }
