89 WorkflowSpec defineDataProcessing(ConfigContext const& config)
90 {
91 
92  using namespace o2::tpc;
93 
94  // set up configuration
95  o2::conf::ConfigurableParam::updateFromFile(config.options().get<std::string>("configFile"));
96  o2::conf::ConfigurableParam::updateFromString(config.options().get<std::string>("configKeyValues"));
97  o2::conf::ConfigurableParam::writeINI("o2tpccalibration_configuration.ini");
98 
99  const std::string outputFile = config.options().get<std::string>("outputFile");
100 
101  const auto tpcSectors = o2::RangeTokenizer::tokenize<int>(config.options().get<std::string>("sectors"));
102  const auto nSectors = (int)tpcSectors.size();
103  const auto nLanes = std::min(config.options().get<int>("lanes"), nSectors);
104 
105  WriterType writerType;
106  try {
107  writerType = WriterMap.at(config.options().get<std::string>("writer-type"));
108  } catch (std::out_of_range&) {
109  throw std::invalid_argument(std::string("invalid writer-type type: ") + config.options().get<std::string>("writer-type"));
110  }
111 
112  WorkflowSpec workflow;
113 
114  if (nLanes <= 0) {
115  return workflow;
116  }
117 
118  std::vector<int> laneConfiguration = tpcSectors; // Currently just a copy of the tpcSectors, why?
119 
120  gTpcSectorMask = 0;
121  for (auto s : tpcSectors) {
122  gTpcSectorMask |= (1ul << s);
123  }
124  gPolicyData.emplace_back(o2::framework::InputSpec{"data", o2::framework::ConcreteDataTypeMatcher{"TPC", "KRCLUSTERS"}});
125 
126  WorkflowSpec parallelProcessors;
127  parallelProcessors.emplace_back(getKryptonClustererSpec());
128 
129  parallelProcessors = parallelPipeline(
130  parallelProcessors, nLanes,
131  [&laneConfiguration]() { return laneConfiguration.size(); },
132  [&laneConfiguration](size_t index) { return laneConfiguration[index]; });
133  workflow.insert(workflow.end(), parallelProcessors.begin(), parallelProcessors.end());
134 
135  if (writerType == WriterType::Local) {
137  //
138  // generation of processor specs for various types of outputs
139  // based on generic RootTreeWriter and MakeRootTreeWriterSpec generator
140  //
141  // -------------------------------------------------------------------------------------------
142  // the callbacks for the RootTreeWriter
143  //
144  // The generic writer needs a way to associate incoming data with the individual branches for
145  // the TPC sectors. The sector number is transmitted as part of the sector header, the callback
146  // finds the corresponding index in the vector of configured sectors
147  auto getIndex = [tpcSectors](o2::framework::DataRef const& ref) {
148  auto const* tpcSectorHeader = o2::framework::DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
149  if (!tpcSectorHeader) {
150  throw std::runtime_error("TPC sector header missing in header stack");
151  }
152  if (tpcSectorHeader->sector() < 0) {
153  // special data sets, don't write
154  return ~(size_t)0;
155  }
156  size_t index = 0;
157  for (auto const& sector : tpcSectors) {
158  if (sector == tpcSectorHeader->sector()) {
159  return index;
160  }
161  ++index;
162  }
163  throw std::runtime_error("sector " + std::to_string(tpcSectorHeader->sector()) + " not configured for writing");
164  };
165  auto getName = [tpcSectors](std::string base, size_t index) {
166  return base + "_" + std::to_string(tpcSectors.at(index));
167  };
168 
169  auto makeWriterSpec = [tpcSectors, laneConfiguration, getIndex, getName](const char* processName,
170  const char* defaultFileName,
171  const char* defaultTreeName,
172  auto&& databranch,
173  bool singleBranch = false) {
174  if (tpcSectors.size() == 0) {
175  throw std::invalid_argument(std::string("writer process configuration needs list of TPC sectors"));
176  }
177 
178  auto amendInput = [tpcSectors, laneConfiguration](InputSpec& input, size_t index) {
179  input.binding += std::to_string(laneConfiguration[index]);
180  DataSpecUtils::updateMatchingSubspec(input, laneConfiguration[index]);
181  };
182  auto amendBranchDef = [laneConfiguration, amendInput, tpcSectors, getIndex, getName, singleBranch](auto&& def, bool enableMC = true) {
183  if (!singleBranch) {
184  def.keys = mergeInputs(def.keys, laneConfiguration.size(), amendInput);
185  // the branch is disabled if set to 0
186  def.nofBranches = enableMC ? tpcSectors.size() : 0;
187  def.getIndex = getIndex;
188  def.getName = getName;
189  } else {
190  // instead of the separate sector branches only one is going to be written
191  def.nofBranches = enableMC ? 1 : 0;
192  }
193  return std::move(def);
194  };
195 
196  return std::move(MakeRootTreeWriterSpec(processName, defaultFileName, defaultTreeName,
197  std::move(amendBranchDef(databranch)))());
198  };
199 
201  //
202  // a writer process for digits
203  //
204  // selected by output type 'difits'
205  using KrClusterOutputType = std::vector<o2::tpc::KrCluster>;
206  workflow.push_back(makeWriterSpec("tpc-krcluster-writer",
207  outputFile.data(),
208  "Clusters",
209  BranchDefinition<KrClusterOutputType>{InputSpec{"data", "TPC", "KRCLUSTERS", 0},
210  "TPCBoxCluster",
211  "boxcluster-branch-name"}));
212  } else if (writerType == WriterType::EPN) {
213  workflow.push_back(getFileWriterSpec<KrCluster>("data:TPC/KRCLUSTERS", BranchType::Krypton));
214  }
215 
216  return workflow;
217 }
