293  template <SafetyLevel SafeT = SafetyLevel::kSafe>
294  GPUd() void interpolateU(int inpYdim, GPUgeneric() const DataT Parameters[],
295  DataT u, GPUgeneric() DataT S[/*nYdim*/]) const
296  {
297  const auto nYdimTmp = SplineUtil::getNdim<YdimT>(inpYdim);
298  const auto nYdim = nYdimTmp.get();
299  int iknot = TBase::template getLeftKnotIndexForU<SafeT>(u);
300  const DataT* d = Parameters + (2 * nYdim) * iknot;
301  interpolateU(nYdim, getKnots()[iknot], &(d[0]), &(d[nYdim]), &(d[2 * nYdim]), &(d[3 * nYdim]), u, S);
302  }
303 
307  template <typename T>
308  GPUd() void interpolateU(int inpYdim, const Knot& knotL,
309  GPUgeneric() const T Sl[/*mYdim*/], GPUgeneric() const T Dl[/*mYdim*/],
310  GPUgeneric() const T Sr[/*mYdim*/], GPUgeneric() const T Dr[/*mYdim*/],
311  DataT u, GPUgeneric() T S[/*mYdim*/]) const
312  {
313  const auto nYdimTmp = SplineUtil::getNdim<YdimT>(inpYdim);
314  const auto nYdim = nYdimTmp.get();
315  T uu = T(u - knotL.u);
316  T li = T(knotL.Li);
317  T v = uu * li; // scaled u
318  for (int dim = 0; dim < nYdim; ++dim) {
319  T df = (Sr[dim] - Sl[dim]) * li;
320  T a = Dl[dim] + Dr[dim] - df - df;
321  T b = df - Dl[dim] - a;
322  S[dim] = ((a * v + b) * v + Dl[dim]) * uu + Sl[dim];
323  }
324  /*
325  another way to calculate f(u):
326  T uu = T(u - knotL.u);
327  T v = uu * T(knotL.Li); // scaled u
328  T vm1 = v-1;
329  T v2 = v * v;
330  float cSr = v2*(3-2*v);
331  float cSl = 1-cSr;
332  float cDl = v*vm1*vm1*knotL.L;
333  float cDr = v2*vm1*knotL.L;
334  return cSl*Sl + cSr*Sr + cDl*Dl + cDr*Dr;
335  */
336  }
337 
338  using TBase::convXtoU;
339  using TBase::getKnot;
340  using TBase::getKnots;
341  using TBase::getNumberOfKnots;
342 
343  protected:
344  using TBase::mParameters;
345  using TBase::mYdim;
346  using TBase::TBase; // inherit constructors and hide them
347 #ifndef GPUCA_ALIROOT_LIB
348  ClassDefNV(Spline1DSpec, 0);
349 #endif
350 };
