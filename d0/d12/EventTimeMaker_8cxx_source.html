35 void computeEvTime(const std::vector<eventTimeTrack>& tracks, const std::vector<int>& trkIndex, eventTimeContainer& evtime)
36 {
37  static constexpr int maxNumberOfSets = 200;
38  static constexpr float weightLimit = 1E-6; // Limit in the weights
39 
40  const int ntracks = tracks.size();
41  LOG(debug) << "For the collision time using " << ntracks << " tracks";
42 
43  if (ntracks < 2) { // at least 2 tracks required
44  LOG(debug) << "Skipping event because at least 2 tracks are required";
45  return;
46  }
47 
48  int hypo[MAXNTRACKINSET];
49 
50  int nmaxtracksinset = ntracks > 22 ? 6 : MAXNTRACKINSET; // max number of tracks in a set for event time computation
51  LOG(debug) << "nmaxtracksinset " << nmaxtracksinset;
52  int ntracksinset = std::min(ntracks, nmaxtracksinset);
53  LOG(debug) << "ntracksinset " << ntracksinset;
54 
55  int nset = ((ntracks - 1) / ntracksinset) + 1;
56  LOG(debug) << "nset " << nset;
57  int ntrackUsed = ntracks;
58  LOG(debug) << "ntrackUsed " << ntrackUsed;
59 
60  if (nset > maxNumberOfSets) {
61  nset = maxNumberOfSets;
62  LOG(debug) << "resetting nset " << nset;
63  ntrackUsed = nmaxtracksinset * nset;
64  LOG(debug) << "resetting ntrackUsed " << ntrackUsed;
65  }
66 
67  // list of tracks in set
68  std::vector<int> trackInSet[maxNumberOfSets];
69 
70  for (int i = 0; i < ntrackUsed; i++) {
71  int iset = i % nset;
72 
73  trackInSet[iset].push_back(i);
74  }
75 
76  int status;
77  // compute event time for each set
78  for (int iset = 0; iset < nset; iset++) {
79  LOG(debug) << "iset " << iset << " has size " << trackInSet[iset].size();
80  unsigned long bestComb = 0;
81  while (!(status = getStartTimeInSet(tracks, trackInSet[iset], bestComb))) {
82  ;
83  }
84  if (status == 1) {
85  int ntracks = trackInSet[iset].size();
86  // set the best in set
87  for (int itrk = 0; itrk < ntracks; itrk++) {
88  hypo[itrk] = bestComb % 3;
89  bestComb /= 3;
90 
91  int index = trkIndex[trackInSet[iset][itrk]];
92  const eventTimeTrack& ctrack = tracks[trackInSet[iset][itrk]];
93  LOG(debug) << "Using hypothesis: " << hypo[itrk] << " tofSignal: " << ctrack.mSignal << " exp. time: " << ctrack.expTimes[hypo[itrk]] << " exp. sigma: " << ctrack.expSigma[hypo[itrk]];
94  LOG(debug) << "0= " << ctrack.expTimes[0] << " +- " << ctrack.expSigma[0] << " 1= " << ctrack.expTimes[1] << " +- " << ctrack.expSigma[1] << " 2= " << ctrack.expTimes[2] << " +- " << ctrack.expSigma[2];
95 
96  evtime.weights[index] = 1. / (ctrack.expSigma[hypo[itrk]] * ctrack.expSigma[hypo[itrk]]);
97  evtime.tracktime[index] = ctrack.mSignal - ctrack.expTimes[hypo[itrk]];
98  }
99  }
100  LOG(debug) << "iset " << iset << " did not have good status";
101  } // end loop in set
102 
103  // do average among all tracks
104  float finalTime = 0, allweights = 0;
105  for (int i = 0; i < evtime.weights.size(); i++) {
106  if (evtime.weights[i] < weightLimit) {
107  continue;
108  }
109  allweights += evtime.weights[i];
110  finalTime += evtime.tracktime[i] * evtime.weights[i];
111  }
112 
113  if (allweights < weightLimit) {
114  LOG(debug) << "Skipping because allweights " << allweights << " are lower than " << weightLimit;
115  return;
116  }
117 
118  evtime.eventTime = finalTime / allweights;
119  evtime.eventTimeError = sqrt(1. / allweights);
120  evtime.eventTimeMultiplicity = ntracks;
121 }

203 int getStartTimeInSet(const std::vector<eventTimeTrack>& tracks, std::vector<int>& trackInSet, unsigned long& bestComb)
204 {
205  float chi2, chi2best, averageBest = 0;
206  int hypo[MAXNTRACKINSET];
207  float starttime[MAXNTRACKINSET], weighttime[MAXNTRACKINSET];
208 
209  chi2best = 10000;
210  int ntracks = trackInSet.size();
211  LOG(debug) << "Computing the start time in set with " << ntracks << " tracks";
212 
213  if (ntracks < 3) {
214  LOG(debug) << "Not enough tracks!";
215  return 2; // no event time in the set
216  }
217 
218  unsigned long ncomb = combinatorial[ntracks];
219  for (unsigned long comb = 0; comb < ncomb; comb++) {
220  unsigned long curr = comb;
221 
222  int ngood = 0;
223  float average = 0;
224  float sumweights = 0;
225  // get track info in the set for current combination
226  for (int itrk = 0; itrk < ntracks; itrk++) {
227  hypo[itrk] = curr % 3;
228  curr /= 3;
229  const eventTimeTrack& ctrack = tracks[trackInSet[itrk]];
230  starttime[itrk] = ctrack.mSignal - ctrack.expTimes[hypo[itrk]];
231  weighttime[itrk] = 1. / (ctrack.expSigma[hypo[itrk]] * ctrack.expSigma[hypo[itrk]]);
232 
233  average += starttime[itrk] * weighttime[itrk];
234  sumweights += weighttime[itrk];
235  ngood++;
236  }
237 
238  average /= sumweights;
239 
240  // compute chi2
241  chi2 = 0;
242  float deltat;
243  for (int itrk = 0; itrk < ntracks; itrk++) {
244  deltat = starttime[itrk] - average;
245  chi2 += deltat * deltat * weighttime[itrk];
246  }
247  chi2 /= (ngood - 1);
248 
249  if (chi2 < chi2best) {
250  bestComb = comb;
251  chi2best = chi2;
252  averageBest = average;
253  }
254  } // end loop in combinations
255 
256  int worse = -1;
257  float errworse = 4;
258  // check the best combination
259  unsigned long curr = bestComb;
260  for (int itrk = 0; itrk < ntracks; itrk++) {
261  hypo[itrk] = curr % 3;
262  curr /= 3;
263 
264  const eventTimeTrack& ctrack = tracks[trackInSet[itrk]];
265  float err = ctrack.mSignal - ctrack.expTimes[hypo[itrk]] - averageBest;
266  err /= ctrack.expSigma[hypo[itrk]];
267  err = fabs(err);
268  if (err > errworse) {
269  errworse = err;
270  worse = itrk;
271  }
272  }
273 
274  if (worse > -1) {
275  const eventTimeTrack& ctrack = tracks[trackInSet[worse]];
276  // remove the track and try again
277  trackInSet.erase(trackInSet.begin() + worse);
278  return 0;
279  }
280 
281  return 1; // good event time in the set
282 }

357 void generateEvTimeTracks(std::vector<eventTimeTrackTest>& tracks, int ntracks, float evTime)
358 {
359  eventTimeTrackTest track;
360  constexpr float masses[3] = {0.13957000, 0.49367700, 0.93827200};
361  constexpr float kCSPEED = TMath::C() * 1.0e2f * 1.0e-12f;
362  float energy = 0.f;
363  float betas[3] = {0.f};
364 
365  float pMismatch = ntracks * 0.00005;
366 
367  for (int i = 0; i < ntracks; i++) {
368  track.mTOFChi2 = 1.f;
369  track.mP = gRandom->Exp(1);
370  track.mPt = track.mP;
371  track.mLength = 400.;
372  track.mHypo = gRandom->Exp(1);
373  if (track.mHypo > 2) {
374  track.mHypo = 2;
375  }
376  for (int j = 0; j < 3; j++) {
377  energy = sqrt(masses[j] * masses[j] + track.mP * track.mP);
378  betas[j] = track.mP / energy;
379  track.expTimes[j] = track.mLength / (betas[j] * kCSPEED);
380  track.expSigma[j] = 100.f;
381  if (j == track.mHypo) {
382  track.mSignal = track.expTimes[j] + gRandom->Gaus(0.f, track.expSigma[j]);
383 
384  if (gRandom->Rndm() < pMismatch) { // assign time from a different particle
385  float p = gRandom->Exp(1);
386  float l = 400;
387  int hypo = gRandom->Exp(1);
388  if (hypo > 2) {
389  hypo = 2;
390  }
391  energy = sqrt(masses[hypo] * masses[hypo] + p * p);
392  float beta = p / energy;
393  track.mSignal = l / (beta * kCSPEED) + gRandom->Gaus(0.f, track.expSigma[j]);
394  }
395  }
396  }
397  tracks.push_back(track);
398  }
399 }
