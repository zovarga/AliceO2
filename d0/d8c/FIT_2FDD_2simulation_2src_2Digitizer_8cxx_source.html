34 void Digitizer::process(const std::vector<o2::fdd::Hit>& hits,
35  std::vector<o2::fdd::Digit>& digitsBC,
36  std::vector<o2::fdd::ChannelData>& digitsCh,
37  std::vector<o2::fdd::DetTrigInput>& digitsTrig,
38  o2::dataformats::MCTruthContainer<o2::fdd::MCLabel>& labels)
39 {
40  // loop over all hits and produce digits
41  //LOG(info) << "Processing IR = " << mIntRecord << " | NHits = " << hits.size();
42 
43  flush(digitsBC, digitsCh, digitsTrig, labels); // flush cached signal which cannot be affect by new event
44 
45  auto sorted_hits{hits};
46  std::sort(sorted_hits.begin(), sorted_hits.end(), [](o2::fdd::Hit const& a, o2::fdd::Hit const& b) {
47  return a.GetTrackID() < b.GetTrackID();
48  });
49  //LOG(info) << "Pulse";
50  //Conversion of hits to the analogue pulse shape
51  for (auto& hit : sorted_hits) {
52  if (hit.GetTime() > 20e3) {
53  const int maxWarn = 10;
54  static int warnNo = 0;
55  if (warnNo < maxWarn) {
56  LOG(warning) << "Ignoring hit with time_in_event = " << hit.GetTime() << " ns"
57  << ((++warnNo < maxWarn) ? "" : " (suppressing further warnings)");
58  }
59  continue;
60  }
61 
62  std::array<o2::InteractionRecord, NBC2Cache> cachedIR;
63  int iChannel = hit.GetDetectorID();
64  int nPhotoElectrons = simulateLightYield(iChannel, hit.GetNphot());
65 
66  double delayScintillator = mRndScintDelay.getNextValue();
67  double timeHit = delayScintillator + hit.GetTime();
68 
69  timeHit -= getTOFCorrection(int(iChannel / 4)); // account for TOF to detector
70  timeHit += mIntRecord.getTimeNS();
71  o2::InteractionRecord irHit(timeHit); // BC in which the hit appears (might be different from interaction BC for slow particles)
72 
73  int nCachedIR = 0;
74  for (int i = BCCacheMin; i < BCCacheMax + 1; i++) {
75  double tNS = timeHit + o2::constants::lhc::LHCBunchSpacingNS * i;
76  cachedIR[nCachedIR].setFromNS(tNS);
77  if (tNS < 0 && cachedIR[nCachedIR] > irHit) {
78  continue; // don't go to negative BC/orbit (it will wrap)
79  }
80  setBCCache(cachedIR[nCachedIR++]); // ensure existence of cached container
81  }
82  // if digit for this sector does not exist, create one otherwise add to it
83  createPulse(nPhotoElectrons, hit.GetTrackID(), timeHit, cachedIR, nCachedIR, iChannel);
84  } //hit loop
85 }

88 void Digitizer::createPulse(int nPhE, int parID, double timeHit, std::array<o2::InteractionRecord, NBC2Cache> const& cachedIR, int nCachedIR, int channel)
89 {
90  auto const roundVc = [&](int i) -> int {
91  return (i / Vc::float_v::Size) * Vc::float_v::Size;
92  };
93 
94  double time0 = cachedIR[0].bc2ns(); // start time of the 1st cashed BC
95  float timeDiff = time0 - timeHit;
96  if (channel < 9) {
97  timeDiff += parameters.TimeDelayFDC;
98  } else {
99  timeDiff += parameters.TimeDelayFDA;
100  }
101 
102  //LOG(info) <<"Ch = "<<channel<<" NphE = " << nPhE <<" timeDiff "<<timeDiff;
103  float charge = TMath::Qe() * parameters.PmGain * mBinSize / (mPmtTimeIntegral * ChargePerADC);
104 
105  Bool_t added[nCachedIR];
106  for (int ir = 0; ir < nCachedIR; ir++) {
107  added[ir] = kFALSE;
108  }
109 
110  constexpr float BinSizeInv = 1.0 / mBinSize;
111  for (int iPhE = 0; iPhE < nPhE; ++iPhE) {
112  float tPhE = timeDiff + mRndSignalShape.getNextValue();
113  int const firstBin = roundVc(TMath::Max((int)0, (int)((tPhE - PMTransitTime) * BinSizeInv)));
114  int const lastBin = TMath::Min((int)NBC2Cache * NTimeBinsPerBC - 1, (int)((tPhE + 2.0 * PMTransitTime) * BinSizeInv));
115  //LOG(info) << "firstBin = "<<firstBin<<" lastbin "<<lastBin;
116 
117  float const tempT = mBinSize * (0.5f + firstBin) - tPhE;
118  long iStart = std::lround((tempT + 2.0f * PMTransitTime) * BinSizeInv);
119  float const offset = tempT + 2.0f * PMTransitTime - float(iStart) * mBinSize;
120  long const iOffset = std::lround(offset * BinSizeInv * float(parameters.NResponseTables - 1));
121  if (iStart < 0) { // this should not happen
122  LOG(error) << "FDDDigitizer: table lookup failure";
123  }
124  iStart = roundVc(std::max(long(0), iStart));
125 
126  Vc::float_v workVc;
127  Vc::float_v pmtVc;
128  float const* q = mPMResponseTables[parameters.NResponseTables / 2 + iOffset].data() + iStart;
129  float const* qEnd = &mPMResponseTables[parameters.NResponseTables / 2 + iOffset].back();
130 
131  for (int ir = firstBin / NTimeBinsPerBC; ir <= lastBin / NTimeBinsPerBC; ir++) {
132  int localFirst = (ir == firstBin / NTimeBinsPerBC) ? firstBin : 0;
133  int localLast = (ir < lastBin / NTimeBinsPerBC) ? NTimeBinsPerBC : (lastBin - ir * NTimeBinsPerBC);
134  auto bcCache = getBCCache(cachedIR[ir]);
135  auto& analogSignal = (*bcCache).pulse[channel];
136  float* p = analogSignal.data() + localFirst;
137 
138  for (int localBin = localFirst, iEnd = roundVc(localLast); q < qEnd && localBin < iEnd; localBin += Vc::float_v::Size) {
139  pmtVc.load(q);
140  q += Vc::float_v::Size;
141  Vc::prefetchForOneRead(q);
142  workVc.load(p);
143  workVc += mRndGainVar.getNextValueVc<Vc::float_v>() * charge * pmtVc;
144  workVc.store(p);
145  p += Vc::float_v::Size;
146  Vc::prefetchForOneRead(p);
147  }
148  added[ir] = kTRUE;
149  }
150  }
151  for (int ir = 0; ir < nCachedIR; ir++) {
152  if (added[ir]) {
153  auto bcCache = getBCCache(cachedIR[ir]);
154  (*bcCache).labels.emplace_back(parID, mEventID, mSrcID, channel);
155  }
156  }
157 }

159 void Digitizer::flush(std::vector<o2::fdd::Digit>& digitsBC,
160  std::vector<o2::fdd::ChannelData>& digitsCh,
161  std::vector<o2::fdd::DetTrigInput>& digitsTrig,
162  o2::dataformats::MCTruthContainer<o2::fdd::MCLabel>& labels)
163 {
164 
165  // do we have something to flush? We can do this only for cached BC data which is distanced from currently processed BC by NBCReadOut
166  int nCached = mCache.size();
167  if (nCached < 1) {
168  return;
169  }
170  if (mIntRecord.differenceInBC(mCache.back()) > -BCCacheMin) {
171  LOG(debug) << "Generating new pedestal BL fluct. for BC range " << mCache.front() << " : " << mCache.back();
172  //generatePedestal();
173  } else {
174  return;
175  }
176  //o2::InteractionRecord ir0(mCache.front());
177  //int cacheSpan = 1 + mCache.back().differenceInBC(ir0);
178  //LOG(info) << "Cache spans " << cacheSpan << " with " << nCached << " BCs cached";
179 
180  for (int ibc = 0; ibc < nCached; ibc++) { // digitize BCs which might not be affected by future events
181  auto& bc = mCache[ibc];
182  storeBC(bc, digitsBC, digitsCh, digitsTrig, labels);
183  }
184  // clean cache for BCs which are not needed anymore
185  //LOG(info) << "Cleaning cache";
186  mCache.erase(mCache.begin(), mCache.end());

189 void Digitizer::storeBC(const BCCache& bc,
190  std::vector<o2::fdd::Digit>& digitsBC, std::vector<o2::fdd::ChannelData>& digitsCh,
191  std::vector<o2::fdd::DetTrigInput>& digitsTrig,
192  o2::dataformats::MCTruthContainer<o2::fdd::MCLabel>& labels)
193 {
194  //LOG(info) << "Storing BC " << bc;
195 
196  int first = digitsCh.size(), nStored = 0;
197  for (int ic = 0; ic < Nchannels; ic++) {
198  float chargeADC = integrateCharge(bc.pulse[ic]);
199  if (chargeADC != 0) {
200  digitsCh.emplace_back(ic, int(simulateTimeCFD(bc.pulse[ic])), int(chargeADC), std::rand() % (1 << 8));
201  nStored++;
202  }
203  }
204  //bc.print();
205 
206  if (nStored != 0) {
207  int nBC = digitsBC.size();
208  digitsBC.emplace_back(first, nStored, bc, mTriggers);
209  digitsTrig.emplace_back(bc, 0, 0, 0, 0, 0);
210 
211  for (const auto& lbl : bc.labels) {
212  labels.addElement(nBC, lbl);
213  }
214  }
215 }

218 float Digitizer::integrateCharge(const ChannelBCDataF& pulse)
219 {
220  float chargeADC = 0;
221  for (int iBin = 0; iBin < NTimeBinsPerBC; ++iBin) {
222  //pulse[iBin] /= ChargePerADC;
223  chargeADC += pulse[iBin];
224  }
225  //saturation
226  if (chargeADC > 4095) {
227  chargeADC = 4095;
228  }
229 
230  //LOG(info) <<" Charge " << chargeADC;
231  return std::lround(chargeADC);
232 }

234 float Digitizer::simulateTimeCFD(const ChannelBCDataF& pulse)
235 {
236 
237  std::fill(mTimeCFD.begin(), mTimeCFD.end(), 0);
238  float timeCFD = -1024;
239  int binShift = TMath::Nint(parameters.TimeShiftCFD / mBinSize);
240  for (int iBin = 0; iBin < NTimeBinsPerBC; ++iBin) {
241  //if (mTime[channel][iBin] != 0) std::cout << mTime[channel][iBin] / parameters.mChargePerADC << ", ";
242  if (iBin >= binShift) {
243  mTimeCFD[iBin] = 5.0 * pulse[iBin - binShift] - pulse[iBin];
244  } else {
245  mTimeCFD[iBin] = -1.0 * pulse[iBin];
246  }
247  }
248  for (int iBin = 1; iBin < NTimeBinsPerBC; ++iBin) {
249  if (mTimeCFD[iBin - 1] < 0 && mTimeCFD[iBin] >= 0) {
250  timeCFD = mBinSize * float(iBin);
251  break;
252  }
253  }
254  //LOG(info) <<" Time " << timeCFD;
255  timeCFD *= TimePerTDC; //ns -> counts
256  return std::lround(timeCFD);
257 }

259 o2::fdd::Digitizer::BCCache& Digitizer::setBCCache(const o2::InteractionRecord& ir)
260 {
261  if (mCache.empty() || mCache.back() < ir) {
262  mCache.emplace_back();
263  auto& cb = mCache.back();
264  cb = ir;
265  return cb;
266  }
267  if (mCache.front() > ir) {
268  mCache.emplace_front();
269  auto& cb = mCache.front();
270  cb = ir;
271  return cb;
272  }
273 
274  for (auto cb = mCache.begin(); cb != mCache.end(); cb++) {
275  if ((*cb) == ir) {
276  return *cb;
277  }
278  if (ir < (*cb)) {
279  auto cbnew = mCache.emplace(cb); // insert new element before cb
280  (*cbnew) = ir;
281  return (*cbnew);
282  }
283  }
284  return mCache.front();
285 }

303 void Digitizer::init()
304 {
305  auto const roundVc = [&](int i) -> int {
306  return (i / Vc::float_v::Size) * Vc::float_v::Size;
307  };
308  // set up PMT response tables
309  float offset = -0.5f * mBinSize; // offset \in [-0.5..0.5] * mBinSize
310  int const nBins = roundVc(std::lround(4.0f * PMTransitTime / mBinSize));
311  for (auto& table : mPMResponseTables) {
312  table.resize(nBins);
313  float t = -2.0f * PMTransitTime + offset; // t \in offset + [-2 2] * DP::mPmtTransitTime
314  for (int j = 0; j < nBins; ++j) {
315  table[j] = Digitizer::PMResponse(t);
316  t += mBinSize;
317  }
318  offset += mBinSize / float(parameters.NResponseTables - 1);
319  }
320 
321  TF1 scintDelayFn("fScintDelay", "gaus", -6.0f * IntTimeRes, +6.0f * IntTimeRes);
322  scintDelayFn.SetParameters(1, 0, IntTimeRes);
323  mRndScintDelay.initialize(scintDelayFn);
324 
325  // Initialize function describing the PMT time response
326  TF1 pmtResponseFn("mPmtResponseFn", &Digitizer::PMResponse, -1.0f * PMTransitTime, +2.0f * PMTransitTime, 0);
327  pmtResponseFn.SetNpx(100);
328  mPmtTimeIntegral = pmtResponseFn.Integral(-1.0f * PMTransitTime, +2.0f * PMTransitTime);
329 
330  // Initialize function describing PMT response to the single photoelectron
331  TF1 singlePhESpectrumFn("mSinglePhESpectrum",
332  &Digitizer::SinglePhESpectrum, 0, 30, 0);
333  float const meansPhE = singlePhESpectrumFn.Mean(0, 30);
334  mRndGainVar.initialize([&]() -> float {
335  return singlePhESpectrumFn.GetRandom(0, 30) / meansPhE;
336  });
337 
338  TF1 signalShapeFn("signalShape", "crystalball", 0, 300);
339  signalShapeFn.SetParameters(1, parameters.ShapeSigma, parameters.ShapeSigma, parameters.ShapeAlpha, parameters.ShapeN);
340  mRndSignalShape.initialize([&]() -> float {
341  return signalShapeFn.GetRandom(0, 200);
342  });
343 }

348 int Digitizer::simulateLightYield(int pmt, int nPhot)
349 {
350  const float p = parameters.LightYield * PhotoCathodeEfficiency;
351  if (p == 1.0f || nPhot == 0) {
352  return nPhot;
353  }
354  const int n = int(nPhot < 100 ? gRandom->Binomial(nPhot, p) : gRandom->Gaus(p * nPhot + 0.5, TMath::Sqrt(p * (1 - p) * nPhot)));
355  return n;
356 }

363 Double_t Digitizer::PMResponse(Double_t x)
364 {
365  // this function describes the PM time response to a single photoelectron
366  Double_t y = x + PMTransitTime;
367  return y * y * TMath::Exp(-y * y / (PMTransitTime * PMTransitTime));
368 }

370 Double_t Digitizer::SinglePhESpectrum(Double_t* x, Double_t*)
371 {
372  // this function describes the PM amplitude response to a single photoelectron
373  Double_t y = x[0];
374  if (y < 0) {
375  return 0;
376  }
377  return (TMath::Poisson(y, PMNbOfSecElec) + PMTransparency * TMath::Poisson(y, 1.0));
378 }
