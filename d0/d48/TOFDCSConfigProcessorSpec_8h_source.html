54  void run(o2::framework::ProcessingContext& pc) final
55  {
56  auto configBuff = pc.inputs().get<gsl::span<char>>("inputConfig");
57  auto configFileName = pc.inputs().get<std::string>("inputConfigFileName");
58  auto timer = std::chrono::duration_cast<std::chrono::milliseconds>(HighResClock::now().time_since_epoch()).count();
59  LOG(info) << "got input file " << configFileName << " of size " << configBuff.size();
60  mTOFFEElightReader.loadFEElightConfig(configBuff);
61  mTOFFEElightReader.parseFEElightConfig(mVerbose);
62  //auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().getFirstValid(true).header)->startTime;
63  sendOutput(pc.outputs(), timer);
64  }

73  void sendOutput(DataAllocator& output, long tf)
74  {
75  // sending output to CCDB
76 
77  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
78 
79  using clbUtils = o2::calibration::Utils;
80  const auto& payload = mTOFFEElightReader.getTOFFEElightInfo();
81  auto clName = o2::utils::MemFileHelper::getClassName(payload);
82  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
83  std::map<std::string, std::string> md;
84  md.emplace("created by", "dpl");
85  o2::ccdb::CcdbObjectInfo info("TOF/Calib/FEELIGHT", clName, flName, md, tf, INFINITE_TF);
86  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
87  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
88  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
89  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "TOF_FEELIGHT", 0}, *image.get()); // vector<char>
90  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "TOF_FEELIGHT", 0}, info); // root-serialized
91  }
