37 int GPUO2Interface::Initialize(const GPUO2InterfaceConfiguration& config)
38 {
39  if (mInitialized) {
40  return (1);
41  }
42  mConfig.reset(new GPUO2InterfaceConfiguration(config));
43  mContinuous = mConfig->configGRP.continuousMaxTimeBin != 0;
44  mRec.reset(GPUReconstruction::CreateInstance(mConfig->configDeviceBackend));
45  if (mRec == nullptr) {
46  GPUError("Error obtaining instance of GPUReconstruction");
47  return 1;
48  }
49  mChain = mRec->AddChain<GPUChainTracking>(mConfig->configInterface.maxTPCHits, mConfig->configInterface.maxTRDTracklets);
50  mChain->mConfigDisplay = &mConfig->configDisplay;
51  mChain->mConfigQA = &mConfig->configQA;
52  if (mConfig->configWorkflow.inputs.isSet(GPUDataTypes::InOutType::TPCRaw)) {
53  mConfig->configGRP.needsClusterer = 1;
54  }
55  mRec->SetSettings(&mConfig->configGRP, &mConfig->configReconstruction, &mConfig->configProcessing, &mConfig->configWorkflow);
56  mChain->SetCalibObjects(mConfig->configCalib);
57  mOutputRegions.reset(new GPUTrackingOutputs);
58  if (mConfig->configInterface.outputToExternalBuffers) {
59  for (unsigned int i = 0; i < mOutputRegions->count(); i++) {
60  mChain->SetSubOutputControl(i, &mOutputRegions->asArray()[i]);
61  }
62  GPUOutputControl dummy;
63  dummy.set([](size_t size) -> void* {throw std::runtime_error("invalid output memory request, no common output buffer set"); return nullptr; });
64  mRec->SetOutputControl(dummy);
65  }
66 
67  if (mRec->Init()) {
68  return (1);
69  }
70  if (!mRec->IsGPU() && mRec->GetProcessingSettings().memoryAllocationStrategy == GPUMemoryResource::ALLOCATION_INDIVIDUAL) {
71  mRec->MemoryScalers()->factor *= 2;
72  }
73  mInitialized = true;
74  return (0);
75 }

86 int GPUO2Interface::RunTracking(GPUTrackingInOutPointers* data, GPUInterfaceOutputs* outputs)
87 {
88  if (!mInitialized) {
89  return (1);
90  }
91  if (mConfig->configInterface.dumpEvents) {
92  static int nEvent = 0;
93  mChain->ClearIOPointers();
94  mChain->mIOPtrs = *data;
95 
96  char fname[1024];
97  sprintf(fname, "event.%d.dump", nEvent);
98  mChain->DumpData(fname);
99  if (nEvent == 0) {
100  mRec->DumpSettings();
101 #ifdef GPUCA_BUILD_QA
102  if (mConfig->configProcessing.runMC) {
103  mChain->ForceInitQA();
104  sprintf(fname, "mc.%d.dump", nEvent);
105  mChain->GetQA()->DumpO2MCData(fname);
106  }
107 #endif
108  }
109  nEvent++;
110  if (mConfig->configInterface.dumpEvents >= 2) {
111  return 0;
112  }
113  }
114 
115  mChain->mIOPtrs = *data;
116  if (mConfig->configInterface.outputToExternalBuffers) {
117  for (unsigned int i = 0; i < mOutputRegions->count(); i++) {
118  if (outputs->asArray()[i].allocator) {
119  mOutputRegions->asArray()[i].set(outputs->asArray()[i].allocator);
120  } else if (outputs->asArray()[i].ptrBase) {
121  mOutputRegions->asArray()[i].set(outputs->asArray()[i].ptrBase, outputs->asArray()[i].size);
122  } else {
123  mOutputRegions->asArray()[i].reset();
124  }
125  }
126  }
127 
128  int retVal = mRec->RunChains();
129  if (retVal == 2) {
130  retVal = 0; // 2 signals end of event display, ignore
131  }
132  if (retVal) {
133  mRec->ClearAllocatedMemory();
134  return retVal;
135  }
136  if (mConfig->configQA.shipToQC) {
137  outputs->qa.hist1 = &mChain->GetQA()->getHistograms1D();
138  outputs->qa.hist2 = &mChain->GetQA()->getHistograms2D();
139  outputs->qa.hist3 = &mChain->GetQA()->getHistograms1Dd();
140  }
141  *data = mChain->mIOPtrs;
142 
143  return 0;
144 }
