134 void Trap2CRU::sortDataToLinks()
135 {
136  //sort the digits array TODO refactor this intoa vector index sort and possibly generalise past merely digits.
137  auto sortstart = std::chrono::high_resolution_clock::now();
138  //build indexes
139  // digits first
140  mDigitsIndex.resize(mDigits.size());
141  std::iota(mDigitsIndex.begin(), mDigitsIndex.end(), 0);
142 
143  for (auto& trig : mTrackletTriggerRecords) {
144  if (trig.getNumberOfTracklets() > 0) {
145  if (mVerbosity) {
146  LOG(debug) << " sorting digits from : " << trig.getFirstTracklet() << " till " << trig.getFirstTracklet() + trig.getNumberOfTracklets();
147  }
148  // sort to link order *NOT* hcid order ...
149  // link is defined by stack,layer,halfchamberside.
150  // tracklet data we have hcid,padrow,colum.
151  // hcid/2 = detector, detector implies stack and layer, and hcid odd/even gives side.
152  std::stable_sort(std::begin(mTracklets) + trig.getFirstTracklet(), std::begin(mTracklets) + trig.getNumberOfTracklets() + trig.getFirstTracklet(),
153  [this](auto&& t1, auto&& t2) {
154  int link1 = HelperMethods::getLinkIDfromHCID(t1.getHCID());
155  int link2 = HelperMethods::getLinkIDfromHCID(t2.getHCID());
156  if (link1 != link2) {
157  return link1 < link2;
158  }
159  if (t1.getPadRow() != t2.getPadRow()) {
160  return t1.getPadRow() < t2.getPadRow();
161  }
162  return t1.getMCM() < t2.getMCM();
163  });
164  }
165  if (trig.getNumberOfDigits() != 0) {
166  if (mVerbosity) {
167  LOG(debug) << " sorting digits from : " << trig.getFirstDigit() << " till " << trig.getFirstDigit() + trig.getNumberOfDigits();
168  }
169  std::stable_sort(mDigitsIndex.begin() + trig.getFirstDigit(), mDigitsIndex.begin() + trig.getNumberOfDigits() + trig.getFirstDigit(),
170  [this](const uint32_t i, const uint32_t j) {
171  int link1=HelperMethods::getLinkIDfromHCID(mDigits[i].getDetector() * 2 + (mDigits[i].getROB() % 2));
172  int link2=HelperMethods::getLinkIDfromHCID(mDigits[j].getDetector() * 2 + (mDigits[j].getROB() % 2));
173  if(link1!=link2){return link1<link2;}
174  if(mDigits[i].getROB() != mDigits[j].getROB()){return (mDigits[i].getROB() < mDigits[j].getROB());}
175  if(mDigits[i].getMCM() != mDigits[j].getMCM()){return (mDigits[i].getMCM() < mDigits[j].getMCM());}
176  return (mDigits[i].getChannel() < mDigits[j].getChannel()); });
177  }
178  }
179 
180  std::chrono::duration<double> duration = std::chrono::high_resolution_clock::now() - sortstart;
181  if (mVerbosity) {
182  LOG(info) << "TRD Digit/Tracklet Sorting took " << duration.count() << " s";
183  int triggercount = 0;
184  for (auto& trig : mTrackletTriggerRecords) {
185 
186  LOG(info) << "Trigger: " << triggercount << " with T" << trig.getBCData().asString();
187  LOG(info) << "Tracklets from:" << trig.getFirstTracklet() << " with " << trig.getNumberOfTracklets();
188  LOG(info) << "Digits from:" << trig.getFirstDigit() << " with " << trig.getNumberOfDigits();
189  if (trig.getNumberOfTracklets() > 0) {
190  int firsttracklet = trig.getFirstTracklet();
191  int numtracklets = trig.getNumberOfTracklets();
192  for (int trackletcount = firsttracklet; trackletcount < firsttracklet + numtracklets; ++trackletcount) {
193  LOG(info) << "Tracklet : " << trackletcount << " details : supermodule:" << std::dec << mTracklets[trackletcount].getHCID() << std::hex << " tracklet:" << mTracklets[trackletcount];
194  }
195  } else {
196  LOG(info) << "No Tracklets for this trigger";
197  }
198  if (trig.getNumberOfDigits() != 0) {
199  int firstdigit = trig.getFirstDigit();
200  int numdigits = trig.getNumberOfDigits();
201  for (int digitcount = firstdigit; digitcount < firstdigit + numdigits; ++digitcount) {
202  LOG(info) << "Digit indexed: " << digitcount << " digit index : " << mDigitsIndex[digitcount] << " details : hcid=" << mDigits[mDigitsIndex[digitcount]].getHCId()
203  << " calculated hcid=" << (mDigits[mDigitsIndex[digitcount]].getDetector() * 2 + (mDigits[mDigitsIndex[digitcount]].getROB() % 2))
204  << " det=" << mDigits[mDigitsIndex[digitcount]].getDetector()
205  << " mcm=" << mDigits[mDigitsIndex[digitcount]].getMCM()
206  << " rob=" << mDigits[mDigitsIndex[digitcount]].getROB()
207  << " channel=" << mDigits[mDigitsIndex[digitcount]].getChannel()
208  << " col=" << mDigits[mDigitsIndex[digitcount]].getPadRow()
209  << " pad=" << mDigits[mDigitsIndex[digitcount]].getPadCol()
210  << " adcsum=" << mDigits[mDigitsIndex[digitcount]].getADCsum()
211  << " hcid=" << mDigits[mDigitsIndex[digitcount]].getHCId();
212  }
213 
214  } else {
215  LOG(error) << "No Digits for this trigger <----- this should NEVER EVER HAPPEN";
216  }
217  triggercount++;
218  }
219  LOG(info) << "end of pre sort tracklets then digits";
220  } // if verbose

242 void Trap2CRU::readTrapData()
243 {
244  //set things up, read the file and then deligate to convertTrapdata to do the conversion.
245  //
246  mRawData.reserve(1024 * 1024); //TODO take out the hardcoded 1MB its supposed to come in from the options
247  if (mVerbosity) {
248  LOG(info) << "Trap2CRU::readTrapData";
249  }
250  // data comes in index by event (triggerrecord)
251  // first 15 links go to cru0a, second 15 links go to cru0b, 3rd 15 links go to cru1a ... first 90 links to flp0 and then repeat for 12 flp
252  // then do next event
253 
254  // lets register our links
255  std::string prefix = mOutputDir;
256  if (!prefix.empty() && prefix.back() != '/') {
257  prefix += '/';
258  }
259 
260  for (int link = 0; link < o2::trd::constants::NHALFCRU; link++) {
261  // FeeID *was* 0xFEED, now is indicates the cru Supermodule, side (A/C) and endpoint. See RawData.cxx for details.
262  int supermodule = link / 4;
263  int endpoint = link % 2;
264  int cru = link / 2;
265  int side = cru % 2; // A or C, 0 or 1 respectively:
266  mFeeID = buildTRDFeeID(supermodule, side, endpoint);
267  LOG(info) << "FEEID;" << std::hex << mFeeID;
268  mCruID = link / 2;
269  mEndPointID = endpoint;
270  mLinkID = o2::trd::constants::TRDLINKID;
271 
272  std::string outFileLink;
273  std::string outPrefix = "TRD_";
274  outPrefix += "alio2-cr1-flp";
275  std::string outSuffix = ".raw";
276  // filename structure of trd_cru_[CRU#]_[upper/lower].raw
277  auto flpid = trdHWMap[mCruID].flpid;
278  auto cruhwid = trdHWMap[mCruID].cruHWID;
279  if (mFilePer == "all") { // single file for all links
280  outFileLink = o2::utils::Str::concat_string(mOutputDir, "/", outPrefix, outSuffix);
281  } else if (mFilePer == "sm") {
282  int sm = link / 4;
283  std::stringstream ss;
284  ss << std::setw(2) << std::setfill('0') << sm;
285  std::string supermodule = ss.str();
286  outFileLink = o2::utils::Str::concat_string(mOutputDir, "/", outPrefix, "_sm_", supermodule, outSuffix);
287  } else if (mFilePer == "cru") {
288  outFileLink = o2::utils::Str::concat_string(mOutputDir, "/", outPrefix, std::to_string(flpid), "_cru", std::to_string(cruhwid), "_", std::to_string(mEndPointID), outSuffix);
289  } else if (mFilePer == "halfcru") {
290  outFileLink = o2::utils::Str::concat_string(mOutputDir, "/", outPrefix, std::to_string(flpid), "_cru", std::to_string(cruhwid), "_", std::to_string(mEndPointID), outSuffix);
291  } else {
292  throw std::runtime_error("invalid option provided for file grouping");
293  }
294 
295  //std::string outputFilelink = o2::utils::Str::concat_string(prefix, "trd_cru_", std::to_string(mCruID), "_", trdside, "_", whichrun, ".raw");
296  LOG(info) << "registering links";
297 
298  mWriter.registerLink(mFeeID, mCruID, mLinkID, mEndPointID, outFileLink);
299  }
300 
301  openInputFiles();
302 
303  if (mTrackletsTree->GetEntries() != mDigitsTree->GetEntries()) {
304  LOG(fatal) << "Entry counts in mTrackletsTree and Digits Tree dont match " << mTrackletsTree->GetEntries() << "!=" << mDigitsTree->GetEntries();
305  }
306  uint32_t totaltracklets = 0;
307  uint32_t totaldigits = 0;
308  int triggercount = 42; // triggercount is here so that we can span timeframes. The actual number is of no consequence,but must increase.
309  for (int entry = 0; entry < mTrackletsTree->GetEntries(); entry++) {
310  mTrackletsTree->GetEntry(entry);
311  mDigitsTree->GetEntry(entry);
312  totaltracklets += mTracklets.size();
313  totaldigits += mDigits.size();
314  //migrate digit trigger information into the tracklettrigger (historical)
315  sortDataToLinks();
316  // each entry is a timeframe
317  uint32_t linkcount = 0;
318  for (auto tracklettrigger : mTrackletTriggerRecords) {
319  convertTrapData(tracklettrigger, triggercount); // tracklettrigger assumed to be authoritive, TODO check for blanks digits range
320  triggercount++;
321  mEventDigitCount++;
322  }
323  }
324  LOG(info) << " Total digits : " << totaldigits;
325  LOG(info) << " Total tracklets : " << totaltracklets;
326 }

328 void Trap2CRU::linkSizePadding(uint32_t linksize, uint32_t& crudatasize, uint32_t& padding)
329 {
330 
331  // if zero the whole 256 bit must be padded (empty link)
332  // crudatasize is the size to be stored in the cruheader, i.e. units of 256bits.
333  // linksize is the incoming link size from the linkrecord,
334  // padding is of course the amount of padding in 32bit words.
335  uint32_t rem = 0;
336  if (linksize != 0) {
337  //data, so figure out padding cru word, the other case is simple, full padding if size=0
338  rem = linksize % 8;
339  if (rem != 0) {
340  crudatasize = linksize / 8 + 1;
341  padding = 8 - rem;
342  } else {
343 
344  crudatasize = linksize / 8; // 32 bit word to 256 bit word.
345  padding = 0;
346  }
347  if (mVerbosity) {
348  LOG(info) << "We have data with linkdatasize=" << linksize << " with size number in header of:" << crudatasize << " padded with " << padding << " 32bit words";
349  }
350  } else {
351  //linksize is zero so no data, pad fully.
352  crudatasize = 1;
353  padding = 8;
354  if (mVerbosity) {
355  LOG(info) << "We have data with linkdatasize=" << linksize << " with size number in header of:" << crudatasize << " padded with " << padding << " 32bit words";
356  }
357  }
358  if (mVerbosity) {
359  LOG(debug) << __func__ << " leaving linkSizePadding : CRUDATASIZE : " << crudatasize;
360  }
361 }

403 int Trap2CRU::buildDigitRawData(const int digitstartindex, const int digitendindex, const int mcm, const int rob, const uint32_t triggerrecordcount)
404 {
405  //this is not zero suppressed.
406  int digitwordswritten = 0;
407  int digitswritten = 0;
408  // Digit
409  DigitMCMHeader header;
410  DigitMCMData data;
411  int startdet = mDigits[mDigitsIndex[digitstartindex]].getDetector();
412  int startrob = mDigits[mDigitsIndex[digitstartindex]].getROB();
413  int startmcm = mDigits[mDigitsIndex[digitstartindex]].getMCM();
414  if (startrob != rob && startmcm != mcm) {
415  LOG(error) << "buildDigitRawData but startmcm and rob are not correct : " << startrob << "!=" << rob << " and mcm: " << startmcm << "!=" << mcm;
416  }
417  int digitcounter = 0;
418  header.res = 0xc; //1100
419  header.mcm = startmcm;
420  header.rob = startrob;
421  header.yearflag = 1; // >10.2007
422  header.eventcount = triggerrecordcount;
423  memcpy(mRawDataPtr, (char*)&header, sizeof(DigitMCMHeader)); // uint32 -- 4 bytes.
424  DigitMCMHeader* headerptr = (DigitMCMHeader*)mRawDataPtr;
425  //LOG(info) << "Digt Header word: 0x" << std::hex << headerptr->word;
426  mRawDataPtr += 4;
427  digitwordswritten++;
428  //we are writing zero suppressed so
429  DigitMCMADCMask adcmask;
430  adcmask = buildBlankADCMask();
431  memcpy(mRawDataPtr, (char*)&adcmask, sizeof(DigitMCMADCMask));
432  DigitMCMADCMask* adcmaskptr = (DigitMCMADCMask*)mRawDataPtr;
433  mRawDataPtr += 4;
434  digitwordswritten++;
435  //LOG(info) << "writing data to digit stream of " << std::hex << header.word;
436  for (int digitindex = digitstartindex; digitindex < digitendindex; ++digitindex) {
437  //LOG(info) << "digit index of : " << digitindex; //<<
438  Digit* d = &mDigits[mDigitsIndex[digitindex]];
439  ArrayADC adcdata = d->getADC();
440  //write these 2 now as we only have it now.
441  if (startmcm != d->getMCM()) {
442  LOG(error) << " we are on the wrong mcm:" << startmcm << "!=" << d->getMCM();
443  }
444  if (startrob != d->getROB()) {
445  LOG(error) << " we are on the wrong rob:" << startrob << "!=" << d->getROB();
446  }
447  int channel = d->getChannel();
448  //set adcmask for the channel we currently have.
449  adcmaskptr->adcmask |= 1UL << channel;
450  for (int timebin = 0; timebin < o2::trd::constants::TIMEBINS; timebin += 3) {
451  data.x = adcdata[timebin];
452  data.y = adcdata[timebin + 1];
453  data.z = adcdata[timebin + 2];
454  data.c = (channel % 2 == 0) ? 0x3 : 0x2; // 3 for even channel 2 for odd channel
455  memcpy(mRawDataPtr, (char*)&data, sizeof(DigitMCMData)); // uint32 -- 4 bytes.
456  mRawDataPtr += sizeof(DigitMCMData);
457  digitwordswritten++;
458  }
459  if (mVerbosity) {
460  LOG(info) << "DDDD " << d->getDetector() << ":" << d->getROB() << ":" << d->getMCM() << ":" << d->getChannel() << ":" << d->getADCsum() << ":" << d->getADC()[0] << ":" << d->getADC()[1] << ":" << d->getADC()[2] << "::" << d->getADC()[27] << ":" << d->getADC()[28] << ":" << d->getADC()[29];
461  }
462  if (d->getMCM() != startmcm) {
463  LOG(fatal) << "digit getmcm = " << d->getMCM() << " while startmcm=" << startmcm;
464  }
465  digitswritten++;
466  }
467  if (digitswritten != digitendindex - digitstartindex) {
468  LOG(error) << " something wrong the number of digitswritten does not correspond to the the loop count";
469  }
470  if (digitwordswritten != (digitswritten * 10 + 2)) {
471  LOG(error) << "something wrong with writing the digits the following should be equal " << digitwordswritten << "==" << (digitswritten * 10 + 2) << " with digitswritten=" << digitswritten;
472  LOG(error) << "digit start index distance to digit end index :" << digitendindex - digitstartindex;
473  }
474  return digitwordswritten;
475 }

477 int Trap2CRU::buildTrackletRawData(const int trackletindex, const int linkid)
478 {
479  TrackletMCMHeader header;
480  bool destroytracklets = false;
481  std::array<TrackletMCMData, 3> trackletdata;
482 
483  header.col = mTracklets[trackletindex].getColumn();
484  header.padrow = mTracklets[trackletindex].getPadRow();
485  header.onea = 1;
486  header.oneb = 1;
487  header.pid0 = 0xff;
488  header.pid1 = 0xff;
489  header.pid2 = 0xff;
490  unsigned int trackletcounter = 0;
491  if (mVerbosity) {
492  LOG(info) << "After instantiation header is : 0x" << header.word << " " << header;
493  }
494  while (linkid == HelperMethods::getLinkIDfromHCID(mTracklets[trackletindex + trackletcounter].getHCID()) && header.col == mTracklets[trackletindex + trackletcounter].getColumn() && header.padrow == mTracklets[trackletindex + trackletcounter].getPadRow()) {
495  int trackletoffset = trackletindex + trackletcounter;
496  buildTrackletMCMData(trackletdata[trackletcounter], mTracklets[trackletoffset].getSlope(),

529  LOG(warn) << ">3 tracklets when building the Tracklet raw data stream for hcid=" << mTracklets[trackletindex + trackletcounter].getHCID() << " col:" << mTracklets[trackletindex + trackletcounter].getColumn() << " padrow:" << mTracklets[trackletindex + trackletcounter].getPadRow();

531  break;
