74 template <typename T>
75 constexpr auto selectArrowType()
76 {
77  if constexpr (std::is_same_v<T, int>) {
78  return atype::INT32;
79  } else if constexpr (std::is_same_v<T, bool>) {
80  return atype::BOOL;
81  } else if constexpr (std::is_same_v<T, float>) {
82  return atype::FLOAT;
83  } else if constexpr (std::is_same_v<T, double>) {
84  return atype::DOUBLE;
85  } else if constexpr (std::is_same_v<T, uint8_t>) {
86  return atype::UINT8;
87  } else if constexpr (std::is_same_v<T, int8_t>) {
88  return atype::INT8;
89  } else if constexpr (std::is_same_v<T, uint16_t>) {
90  return atype::UINT16;
91  } else if constexpr (std::is_same_v<T, int16_t>) {
92  return atype::INT16;
93  } else if constexpr (std::is_same_v<T, int64_t>) {
94  return atype::INT64;
95  } else if constexpr (std::is_same_v<T, uint32_t>) {
96  return atype::UINT32;
97  } else if constexpr (std::is_same_v<T, uint64_t>) {
98  return atype::UINT64;
99  } else {
100  return atype::NA;
101  }
102  O2_BUILTIN_UNREACHABLE();
103 }
