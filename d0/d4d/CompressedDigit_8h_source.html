27 // Compressed Digit class for TRD
28 // Notes:
29 // This is to simplify the handling of raw data that comes in in the same 3 timebins per 32 bit integer format.
30 
31 class CompressedDigit
32 {
33  public:
34  CompressedDigit() = default;
35  ~CompressedDigit() = default;
36  CompressedDigit(const int det, const int rob, const int mcm, const int channel, const std::array<uint16_t, constants::TIMEBINS>& adc);
37  CompressedDigit(const int det, const int rob, const int mcm, const int channel); // add adc data in a seperate step

39  // Copy
40  CompressedDigit(const CompressedDigit&) = default;
41  // Assignment
42  CompressedDigit& operator=(const CompressedDigit&) = default;
43  // Modifiers
44  void setChannel(int channel)
45  {
46  mHeader &= ~(0x003f);
47  mHeader |= (channel)&0x003f;
48  }
49  void setMCM(int mcm)
50  {
51  mHeader &= ~(0x3c0);
52  mHeader |= (mcm)&0x3c0;
53  }
54  void setROB(int rob)
55  {
56  mHeader &= ~(0x1c00);
57  mHeader |= (rob)&0x1c00;
58  }
59  void setDetector(int det)
60  {
61  mHeader &= ~(0x7fe000);
62  mHeader |= (det << 12) & 0x7fe000;
63  }
64  void setADC(std::array<uint16_t, constants::TIMEBINS> const& adcs)
65  {
66  int adcindex = 0;
67  for (auto adc : adcs) {
68  int rem = adcindex % 3;
69  // LOG(info) << "adc index :" << adcindex << " rem:" << rem << " adcindex/3=" << adcindex/3;
70  mADC[adcindex / 3] &= ~((0x3ff) << (rem * 10));
71  // LOG(info) << "mADC[adcindex/3] after &= :" << std::hex << mADC[adcindex/3] << rem;
72  mADC[adcindex / 3] |= (adcs[adcindex] << (rem * 10));
73  // LOG(info) << "mADC[adcindex/3] after != :" << std::hex << mADC[adcindex/3] << rem;
74  adcindex++;
75  }
76  }
77  void setADCi(int index, uint16_t adcvalue)
78  {
79  int rem = index % 3;
80  mADC[index / 3] &= ~((0x3ff) << (rem * 10));
81  mADC[index / 3] |= (adcvalue << (rem * 10));
82  }
83  // Get methods
84  int getChannel() const { return mHeader & 0x3f; }
85  int getMCM() const { return (mHeader & 0x3c0) >> 6; }
86  int getROB() const { return (mHeader & 0x1c00) >> 10; }
87  int getDetector() const { return (mHeader & 0x7fe000) >> 12; }
88  bool isSharedCompressedDigit() const;
89 
90  uint16_t operator[](const int index) { return mADC[index / 3] >> ((index % 3) * 10); }
91  uint32_t getADCsum() const
92  {
93  uint32_t sum = 0;
94  for (int i = 0; i < constants::TIMEBINS; ++i) {
95  sum += (mADC[i / 3] >> ((i % 3) * 10));
96  }
97  return sum;
98  }
99 
100  private:
101  uint32_t mHeader;
102  // 3322 2222 2222 1111 1111 1100 0000 0000
103  // 1098 7654 3210 9876 5432 1098 7654 3210
104  // uint32_t: 0000 0000 0DDD DDDD DDDR RRMM MMCC CCCC
105  // C= channel[5 bits 0-21], M=MCM [4bits 0-15], R=ROB[3bits 0-7], D=Detector[10 bits 0-540]
106  std::array<uint32_t, 10> mADC; // ADC vector (30 time-bins) packed into 10 32bit integers.
107  ClassDefNV(CompressedDigit, 1);
108 };
