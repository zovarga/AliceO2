54  void run(o2::framework::ProcessingContext& pc) final
55  {
56  const auto& mapper = Mapper::instance();
57  const size_t MAXROWS = 152;
58  const size_t MAXPADS = 138;
59  const size_t MAXTIMEBINS = 450;
60 
61  for (auto const& inputRef : InputRecordWalker(pc.inputs())) {
62  auto const* sectorHeader = DataRefUtils::getHeader<TPCSectorHeader*>(inputRef);
63  if (sectorHeader == nullptr) {
64  LOGP(error, "sector header missing on header stack for input on ", inputRef.spec->binding);
65  continue;
66  }
67 
68  const int sector = sectorHeader->sector();
69  auto inDigitsO = pc.inputs().get<gsl::span<o2::tpc::Digit>>(inputRef);
70  LOGP(debug, "processing sector {} with {} input digits", sector, inDigitsO.size());
71 
72  // ===| flatten digits for simple access and register interesting digit positions |===
73  // flat digit structure
74  std::vector<float> inDigits(MAXROWS * MAXPADS * MAXTIMEBINS);
75 
76  // counter how often a pad triggered to filter extremely noisy or struck pads
77  std::vector<size_t> nTriggered(MAXROWS * MAXPADS);
78 
79  struct DigitInfo {
80  int row{};
81  int pad{};
82  int time{};
83  };
84  std::vector<DigitInfo> maxDigits;
85 
86  for (const auto& digit : inDigitsO) {
87  const auto time = digit.getTimeStamp();
88  if (time > MAXTIMEBINS) {
89  continue;
90  }
91 
92  const auto pad = digit.getPad();
93  const auto row = digit.getRow();
94  const auto charge = digit.getChargeFloat();
95 
96  const size_t offset = row * MAXPADS * MAXTIMEBINS + pad * MAXTIMEBINS;
97  inDigits[offset + time] = charge;
98 
99  const auto noiseThreshold = mNoise ? mThresholdSigma * mNoise->getValue(Sector(sector), row, pad) : 0.f;
100  if ((charge > mThresholdMax) && (charge > noiseThreshold)) {
101  if (!(mSkipIROCTSensors && (row == 47) && (pad > 40))) {
102  if ((time > mTimeBinsBefore) && (time < (MAXTIMEBINS - mTimeBinsAfter))) {
103  maxDigits.emplace_back(DigitInfo{row, pad, time});
104  ++nTriggered[row * MAXPADS + pad];
105  }
106  }
107  }
108  }
109 
110  // ===| find local maxima |===
111 
112  // simple local maximum detection in time and pad direction
113  auto isLocalMaximum = [&inDigits](const int row, const int pad, const int time) {
114  const size_t offset = row * MAXPADS * MAXTIMEBINS + pad * MAXTIMEBINS + time;
115  const auto testCharge = inDigits[offset];
116 
117  // look in time direction
118  if (!(testCharge > inDigits[offset + 1])) {
119  return false;
120  }
121  if (testCharge < inDigits[offset - 1]) {
122  return false;
123  }
124 
125  // look in pad direction
126  if (!(testCharge > inDigits[offset + MAXTIMEBINS])) {
127  return false;
128  }
129  if (testCharge < inDigits[offset - MAXTIMEBINS]) {
130  return false;
131  }
132 
133  // check diagonals
134  if (!(testCharge > inDigits[offset + MAXTIMEBINS + 1])) {
135  return false;
136  }
137  if (!(testCharge > inDigits[offset + MAXTIMEBINS - 1])) {
138  return false;
139  }
140  if (testCharge < inDigits[offset - MAXTIMEBINS + 1]) {
141  return false;
142  }
143  if (testCharge < inDigits[offset - MAXTIMEBINS - 1]) {
144  return false;
145  }
146 
147  return true;
148  };
149 
150  // digit filter
151  std::vector<Digit> digits;
152 
153  // fill digits and return true if the neigbouring pad should be filtered as well
154  auto fillDigits = [&digits, &inDigits, &sector, this](const size_t row, const size_t pad, const size_t time) {
155  int cru = Mapper::REGION[row] + sector * Mapper::NREGIONS;
156  const size_t offset = row * MAXPADS * MAXTIMEBINS + pad * MAXTIMEBINS;
157 
158  const auto chargeMax = inDigits[offset + time];
159 
160  for (int iTime = time - mTimeBinsBefore; iTime < time + mTimeBinsAfter; ++iTime) {
161  const auto charge = inDigits[offset + iTime];
162  if (charge < -999.f) { // avoid double copy
163  continue;
164  }
165  digits.emplace_back(Digit(cru, charge, row, pad, iTime));
166  inDigits[offset + iTime] = -1000.f;
167  }
168 
169  const auto noiseThreshold = mNoise ? mThresholdSigma * mNoise->getValue(Sector(sector), row, pad) : 0.f;
170  return (chargeMax > mThresholdAbs) && (chargeMax > noiseThreshold);
171  };
172 
173  // filter digits
174  for (const auto& maxDigit : maxDigits) {
175  const auto row = maxDigit.row;
176  const auto pad = maxDigit.pad;
177  // skip extremely noisy pads
178  if (nTriggered[row * MAXPADS + pad] > MAXTIMEBINS / 4) {
179  continue;
180  }
181  const auto time = maxDigit.time;
182  const int nPads = mapper.getNumberOfPadsInRowSector(row);
183  // skip edge pads
184  if ((pad == 0) || (pad == nPads - 1)) {
185  continue;
186  }
187 
188  if (isLocalMaximum(row, pad, time)) {
189  auto padRef = pad;
190  while ((padRef >= 0) && fillDigits(row, padRef, time)) {
191  --padRef;
192  }
193  padRef = pad + 1;
194  while ((padRef < nPads) && fillDigits(row, padRef, time)) {
195  ++padRef;
196  }
197  }
198  }
199 
200  snapshot(pc.outputs(), digits, sector);
201 
202  LOGP(info, "processed sector {} with {} input and {} filtered digits", sector, inDigitsO.size(), digits.size());
203  }
204 
205  ++mProcessedTFs;
206  LOGP(info, "Number of processed time frames: {}", mProcessedTFs);
207  }
208 
209  private:
210  float mThresholdMax{50.f};
211  float mThresholdAbs{20.f};
212  float mThresholdSigma{10.f};
213  int mTimeBinsBefore{10};
214  int mTimeBinsAfter{100};
215  uint32_t mProcessedTFs{0};
216  bool mSkipIROCTSensors{true};
217  CalPad* mNoise{nullptr};
218 
219  //____________________________________________________________________________
220  void snapshot(DataAllocator& output, std::vector<Digit>& digits, int sector)
221  {
222  o2::tpc::TPCSectorHeader header{sector};
223  header.activeSectors = (0x1 << sector);
224  output.snapshot(Output{gDataOriginTPC, "FILTERDIG", static_cast<SubSpecificationType>(sector), Lifetime::Timeframe, header}, digits);
225  }

228 o2::framework::DataProcessorSpec getKryptonRawFilterSpec(CalPad* noise)
229 {
230  using device = o2::tpc::KrRawFilterDevice;
231 
232  std::vector<InputSpec> inputs{
233  InputSpec{"digits", gDataOriginTPC, "DIGITS", 0, Lifetime::Timeframe},
234  };
235 
236  std::vector<OutputSpec> outputs;
237  outputs.emplace_back(gDataOriginTPC, "FILTERDIG", 0, Lifetime::Timeframe);
238 
239  return DataProcessorSpec{
240  "tpc-krypton-raw-filter",
241  inputs,
242  outputs,
243  AlgorithmSpec{adaptFromTask<device>(noise)},
244  Options{
245  {"threshold-max", VariantType::Float, 50.f, {"threshold in absolute ADC counts for the maximum digit"}},
246  {"threshold-abs", VariantType::Float, 20.f, {"threshold in absolute ADC counts above which time sequences will be written out"}},
247  {"threshold-sigma", VariantType::Float, 10.f, {"threshold in sigma noise above which time sequences will be written out"}},
248  {"time-bins-before", VariantType::Int, 10, {"time bins before trigger digit to be written"}},
249  {"time-bins-after", VariantType::Int, 100, {"time bins after trigger digit to be written"}},
250  {"skip-iroc-tsen", VariantType::Bool, true, {"skip IROC T-Sensor pads in maxima detection"}},
251  } // end Options
252  }; // end DataProcessorSpec
