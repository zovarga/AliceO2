168  void makeConstrainedTPCTracks(V& container)
169  {
170  checkRefitter();
171  int nmatched = mMatchedTracks[trkType::TPC].size(), nconstrained = 0;
172  container.resize(nmatched);
173  for (unsigned i = 0; i < nmatched; i++) {
174  if (makeConstrainedTPCTrack(i, container[nconstrained])) {
175  nconstrained++;
176  }
177  }
178  container.resize(nconstrained);
179  }
