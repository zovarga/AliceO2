58  auto makeBoxGen = [](int pdgid, int mult, double etamin, double etamax, double pmin, double pmax, double phimin, double phimax, bool debug = false) {
59  auto gen = new FairBoxGenerator(pdgid, mult);
60  gen->SetEtaRange(etamin, etamax);
61  gen->SetPRange(pmin, pmax);
62  gen->SetPhiRange(phimin, phimax);
63  gen->SetDebug(debug);
64  return gen;

69  auto gen = new o2::eventgen::GeneratorPythia8();
