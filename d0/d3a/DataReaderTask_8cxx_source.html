225 bool DataReaderTask::isTimeFrameEmpty(ProcessingContext& pc)
226 {
227  constexpr auto origin = header::gDataOriginTRD;
228  o2::framework::InputSpec dummy{"dummy", framework::ConcreteDataMatcher{origin, header::gDataDescriptionRawData, 0xDEADBEEF}};
229  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload.
230  // frame detected we have no data and send this instead
231  // send empty output so as to not block workflow
232  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
233  for (const auto& ref : o2::framework::InputRecordWalker(pc.inputs(), {dummy})) {
234  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
235  if (dh->payloadSize == 0) {
236  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
237  if (++contDeadBeef <= maxWarn) {
238  LOGP(warning, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
239  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize,
240  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
241  }
242  return true;
243  }
244  }
245  contDeadBeef = 0; // if good data, reset the counter
246  return false;
247 }

249 void DataReaderTask::run(ProcessingContext& pc)
250 {
251  //NB this is run per time frame on the epn.
252  LOG(info) << "TRD Translator Task run";
253  auto dataReadStart = std::chrono::high_resolution_clock::now();
254 
255  if (isTimeFrameEmpty(pc)) {
256  sendData(pc, true); //send the empty tf data.
257  return;
258  }
259  uint64_t total1 = 0, total2 = 0;
260 
261  std::vector<InputSpec> sel{InputSpec{"filter", ConcreteDataTypeMatcher{"TRD", "RAWDATA"}}};
262  uint64_t tfCount = 0;
263  for (const auto& ref : InputRecordWalker(pc.inputs(), sel)) {
264  auto inputprocessingstart = std::chrono::high_resolution_clock::now(); // measure total processing time
265  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
266  tfCount = dh->tfCounter;
267  if (mVerbose) {
268  LOGP(info, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : ",
269  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize);
270  }
271  auto payloadIn = ref.payload;
272  auto payloadInSize = dh->payloadSize;
273  if (!mCompressedData) { //we have raw data coming in from flp
274  if (mVerbose) {
275  LOG(info) << " parsing non compressed data in the data reader task with a payload of " << payloadInSize << " payload size";
276  }
277  // LOG(info) << "start of data is at ref.payload=0x"<< std::hex << " dh->payloadSize:0x" << dh->payloadSize <<" dh->headerSize:0x" <<dh->headerSize;
278  total1 += dh->payloadSize;
279  total2 += dh->headerSize;
280  // LOG(info) << "start of data is at ref.payload=0x"<< std::hex << " total1:0x" << total1 <<" total2:0x" <<total2;
281  mReader.setDataBuffer(payloadIn);
282  mReader.setDataBufferSize(payloadInSize);
283  mReader.configure(mTrackletHCHeaderState, mHalfChamberWords, mHalfChamberMajor, mOptions);
284  //mReader.setStats(&mTimeFrameStats);
285  mReader.run();
286  mWordsRead += mReader.getWordsRead();
287  mWordsRejected += mReader.getWordsRejected();
288  if (mVerbose) {
289  LOG(info) << "relevant vectors to read : " << mReader.sumTrackletsFound() << " tracklets and " << mReader.sumDigitsFound() << " compressed digits";
290  }
291  } else { // we have compressed data coming in from flp.
292  mCompressedReader.setDataBuffer(payloadIn);
293  mCompressedReader.setDataBufferSize(payloadInSize);
294  mCompressedReader.configure(mOptions);
295  mCompressedReader.run();
296  }
297  }
298 
299  sendData(pc, false);
300  std::chrono::duration<double, std::milli> dataReadTime = std::chrono::high_resolution_clock::now() - dataReadStart;
301  if (mRootOutput) {
302  mTimeFrameTime->Fill((int)std::chrono::duration_cast<std::chrono::milliseconds>(dataReadTime).count());
303  }
304  LOGP(info, "Digits: {}, Tracklets: {}, DataRead in: {:.3f} MB, Rejected: {:.3f} MB for TF {} in {} ms",
305  mReader.getDigitsFound(), mReader.getTrackletsFound(), (float)mWordsRead * 4 / 1024.0 / 1024.0, (float)mWordsRejected * 4 / 1024.0 / 1024.0, tfCount,
306  std::chrono::duration_cast<std::chrono::milliseconds>(dataReadTime).count());
307 }
