47 void GRPECSObject::setDetROMode(o2::detectors::DetID id, ROMode status)
48 {
50  if (!(status & PRESENT)) {
51  remDetReadOut(id);
52  return;
53  }
54  addDetReadOut(id);
55  if ((status & CONTINUOUS) == CONTINUOUS) {
56  addDetContinuousReadOut(id);
57  } else {
58  remDetContinuousReadOut(id);
59  }
60  if ((status & TRIGGERING) == TRIGGERING) {
61  addDetTrigger(id);
62  } else {
63  remDetTrigger(id);
64  }
65 }

68 GRPECSObject::ROMode GRPECSObject::getDetROMode(o2::detectors::DetID id) const
69 {
70  GRPECSObject::ROMode status = ABSENT;
71  if (isDetReadOut(id)) {
72  status = PRESENT;
73  } else {
74  if (isDetContinuousReadOut(id)) {
75  status = GRPECSObject::ROMode(status | CONTINUOUS);
76  }
77  if (isDetTriggers(id)) {
78  status = GRPECSObject::ROMode(status | TRIGGERING);
79  }
80  }
81  return status;
82 }

85 GRPECSObject* GRPECSObject::loadFrom(const std::string& grpecsFileName)
86 {
87  // load object from file
88  auto fname = o2::base::NameConf::getGRPFileName(grpecsFileName);
89  TFile flGRP(fname.c_str());
90  if (flGRP.IsZombie()) {
91  LOG(error) << "Failed to open " << fname;
92  throw std::runtime_error("Failed to open GRPECS file");
93  }
94  auto grp = reinterpret_cast<o2::parameters::GRPECSObject*>(flGRP.GetObjectChecked(o2::base::NameConf::CCDBOBJECT.data(), Class()));
95  if (!grp) {
96  throw std::runtime_error(fmt::format("Failed to load GRPECS object from {}", fname));
97  }
98  return grp;
