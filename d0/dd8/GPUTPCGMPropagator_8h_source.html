107  GPUd() int RejectCluster(float chiY, float chiZ, unsigned char clusterState)
108  {
109  if (chiY > 9.f || chiZ > 9.f) {
110  return 2;
111  }
112  if ((chiY > 6.25f || chiZ > 6.25f) && (clusterState & (GPUTPCGMMergedTrackHit::flagSplit | GPUTPCGMMergedTrackHit::flagShared))) {
113  return 2;
114  }
115  if ((chiY > 1.f || chiZ > 6.25f) && (clusterState & (GPUTPCGMMergedTrackHit::flagEdge | GPUTPCGMMergedTrackHit::flagSingle))) {
116  return 2;
117  }
118  return 0;
119  }
