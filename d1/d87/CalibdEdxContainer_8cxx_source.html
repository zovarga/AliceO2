168 void CalibdEdxContainer::setDefaultZeroSupresssionThreshold()
169 {
170  const float defaultVal = getMinZeroSupresssionThreshold();
171  mThresholdMap.setMinCorrectionFactor(defaultVal - 0.1f);
172  mThresholdMap.setMaxCorrectionFactor(defaultVal + 0.1f);
173  for (int sector = 0; sector < o2::tpc::constants::MAXSECTOR; ++sector) {
174  for (unsigned short globPad = 0; globPad < TPC_PADS_IN_SECTOR; ++globPad) {
175  mThresholdMap.setGainCorrection(sector, globPad, defaultVal);
176  }
177  }
178 }

180 template <class Type>
181 void CalibdEdxContainer::loadTopologyCorrectionFromFile(std::string_view fileName, Type*& obj)
182 {
183  FlatObject::startConstruction();
184 
185  // load and set-up container
186  Type calibTrackTopologyTmp(fileName.data());
187 
188  // get size of the flat buffer of the splines
189  const std::size_t flatbufferSize = calibTrackTopologyTmp.getFlatBufferSize();
190 
191  // size of the dEdx container without taking flat buffer into account
192  const std::size_t objSize = sizeOfCalibdEdxTrackTopologyObj<Type>();
193 
194  // create mFlatBuffer with correct size
195  const std::size_t totalSize = flatbufferSize + objSize;
196  FlatObject::finishConstruction(totalSize);
197 
198  // setting member of CalibdEdxTrackTopologyPol to correct buffer address
199  obj = reinterpret_cast<Type*>(mFlatBufferPtr);
200 
201  // deep copy of CalibdEdxTrackTopologyPol to buffer without moving the flat buffer to correct address
202  obj->cloneFromObject(calibTrackTopologyTmp, nullptr);
203 
204  // seting the buffer of the splines to current buffer
205  obj->moveBufferTo(objSize + mFlatBufferPtr);
206 }
