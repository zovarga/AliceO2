81 static void patchPage(gsl::span<const std::byte> rdhBuffer, bool verbose)
82 {
83  auto& rdhAny = *reinterpret_cast<RDH*>(const_cast<std::byte*>(&(rdhBuffer[0])));
84 
85  auto existingFeeId = o2::raw::RDHUtils::getFEEID(rdhAny);
86  if (existingFeeId == 0) {
87  // early versions of raw data did not set the feeId
88  // which we need to select the right decoder
89 
90  auto cruId = o2::raw::RDHUtils::getCRUID(rdhAny) & 0xFF;
91  auto endpoint = o2::raw::RDHUtils::getEndPointID(rdhAny);

93 
94  uint32_t feeId = cruId * 2 + endpoint + flags;
95  o2::raw::RDHUtils::setFEEID(rdhAny, feeId);

157 bool DataDecoder::TimeFrameStartRecord::update(uint32_t orbit, uint32_t bunchCrossing, bool verbose)
158 {
159  if (verbose) {
160  LOGP(info, "[TimeFrameStartRecord::update()] new {}/{} current {}/{} prev {}/{} valid {}",
161  orbit, bunchCrossing, mOrbit, mBunchCrossing, mOrbitPrev, mBunchCrossingPrev, mValid);
162  }
163 
164  // if this is the first occurence, simply initialize the orbit and binch crossing and mark the start recortd as valid
165  if (mOrbit < 0) {
166  mOrbit = orbit;
167  mBunchCrossing = bunchCrossing;
168 
169  mValid = true;
170  return true;
171  }
172 
173  if (mOrbit == orbit) {
174  if (mValid) {
175  // there is already one valid record for this orbit, so the current one is discarded
176  return false;
177  } else {
178  // there is already one record for this orbit, but it is invalid
179  // check if the current one is compatible with the previous, if yes replace the existing one
180  if (check(orbit, bunchCrossing, mOrbitPrev, mBunchCrossingPrev, verbose)) {
181  mOrbit = orbit;
182  mBunchCrossing = bunchCrossing;
183 
184  mValid = true;
185  }
186  }
187  } else {
188  // we received an HB packet for a new TF, check if it is compatible with the last stored one
189  bool valid = check(orbit, bunchCrossing, mOrbit, mBunchCrossing, verbose);
190 
191  bool replace = valid || (mOrbitPrev < 0) || (mValid == false);
192 
193  if (replace) {
194  mOrbitPrev = mOrbit;
195  mBunchCrossingPrev = mBunchCrossing;
196 
197  mOrbit = orbit;
198  mBunchCrossing = bunchCrossing;
199 
200  mValid = valid;
201  }
202  }
203 
204  if (verbose) {
205  LOGP(info, "[TimeFrameStartRecord::update()] set to {}/{} prev {}/{} valid {}",
206  mOrbit, mBunchCrossing, mOrbitPrev, mBunchCrossingPrev, mValid);
207  }
208 
209  return mValid;
210 }

212 bool DataDecoder::TimeFrameStartRecord::check(int32_t orbit, uint32_t bc, int32_t orbitRef, uint32_t bcRef, bool verbose)
213 {
214  if (verbose) {
215  LOGP(info, "[TimeFrameStartRecord::check()] current {}/{} ref {}/{}", orbit, bc, orbitRef, bcRef);
216  }
217 
218  if (orbitRef < 0) {
219  return true;
220  }
221 
222  int64_t dOrbit = orbit - orbitRef;
223  if (verbose) {
224  LOGP(info, "[TimeFrameStartRecord::check()] dOrbit{}", dOrbit);
225  }
226  int64_t bcExpected = dOrbit * bcInOrbit + bcRef;
227 
228  uint64_t bcExpected20bits = bcExpected & twentyBitsAtOne;
229 
230  bool result = (bcExpected20bits == bc);
231 
232  if (verbose) {
233  LOGP(info, " dOrbit {} expected {} expected20bits {} bc {} valid {}", dOrbit, bcExpected, bcExpected20bits, bc, result);
234  }
235 
236  return result;
237 }

267 void DataDecoder::decodeBuffer(gsl::span<const std::byte> buf)
268 {
269  if (mDebug) {
270  std::cout << "\n\n============================\nStart of new buffer\n";
271  }
272  size_t bufSize = buf.size();
273  size_t pageStart = 0;
274  while (bufSize > pageStart) {
275  RDH* rdh = reinterpret_cast<RDH*>(const_cast<std::byte*>(&(buf[pageStart])));
276  if (mDebug) {
277  if (pageStart == 0) {
278  std::cout << "+++\n[decodeBuffer]" << std::endl;
279  } else {
280  std::cout << "---\n[decodeBuffer]" << std::endl;
281  }
282  o2::raw::RDHUtils::printRDH(rdh);
283  }
284  auto rdhVersion = o2::raw::RDHUtils::getVersion(rdh);
285  auto rdhHeaderSize = o2::raw::RDHUtils::getHeaderSize(rdh);
286  if (rdhHeaderSize != 64) {
287  break;
288  }
289  auto pageSize = o2::raw::RDHUtils::getOffsetToNext(rdh);
290 
291  gsl::span<const std::byte> page(reinterpret_cast<const std::byte*>(rdh), pageSize);
292  decodePage(page);
293 
294  pageStart += pageSize;
295  }
296 
297  if (mDebug) {
298  std::cout << "[decodeBuffer] mOrbits size: " << mOrbits.size() << std::endl;
299  dumpOrbits(mOrbits);
300  std::cout << "[decodeBuffer] mDigits size: " << mDigits.size() << std::endl;
301  dumpDigits();
302  }
303 }

307 void DataDecoder::dumpDigits()
308 {
309  for (size_t di = 0; di < mDigits.size(); di++) {
310  auto& d = mDigits[di];
311  auto detID = d.getDetID();
312  auto padID = d.getPadID();
313  if (padID < 0) {
314  continue;
315  }
316  const Segmentation& segment = segmentation(detID);
317  bool bend = segment.isBendingPad(padID);
318  float X = segment.padPositionX(padID);
319  float Y = segment.padPositionY(padID);
320  uint32_t orbit = d.getOrbit();
321  uint32_t bunchCrossing = d.getBunchCrossing();
322  uint32_t sampaTime = d.getSampaTime();
323  std::cout << fmt::format(" DE {:4d} PAD {:5d} ADC {:6d} TIME ({} {} {:4d})",
324  detID, padID, d.getADC(), orbit, bunchCrossing, sampaTime);
325  std::cout << fmt::format("\tC {} PAD_XY {:+2.2f} , {:+2.2f}", (bend ? (int)0 : (int)1), X, Y);
326  std::cout << std::endl;
327  }

342 bool DataDecoder::getMergerChannelId(const DsElecId& dsElecId, DualSampaChannelId channel, uint32_t& chId, uint32_t& boardId)
343 {
344  static constexpr uint32_t sChannelsInOneDs = 64;
345  static constexpr uint32_t sDsInOneSolar = 40;
346  static constexpr uint32_t sChannelsInOneSolar = sChannelsInOneDs * sDsInOneSolar;
347  auto solarId = dsElecId.solarId();
348  uint32_t dsId = static_cast<uint32_t>(dsElecId.elinkGroupId()) * 5 + dsElecId.elinkIndexInGroup();
349  if (solarId > DataDecoder::sMaxSolarId || dsId >= 40 || channel >= 64) {
350  return false;
351  }
352 
353  boardId = solarId * sDsInOneSolar + dsId;
354  chId = boardId * sChannelsInOneDs + channel;
355  return true;

375 bool DataDecoder::mergeDigits(uint32_t mergerChannelId, uint32_t mergerBoardId, uint64_t mergerChannelBitmask, o2::mch::raw::SampaCluster& sc)
376 {
377  static constexpr uint32_t BCROLLOVER = (1 << 20);
378  static constexpr uint32_t ONEADCCLOCK = 4;
379  static constexpr uint32_t MAXNOFSAMPLES = 0x3FF;
380  static constexpr uint32_t TWENTYBITSATONE = 0xFFFFF;
381 
382  // only digits that start at the beginning of the SAMPA window can be merged
383  if (sc.sampaTime != 0) {
384  return false;
385  }
386 
387  // if there is not previous digit for this channel then no merging is possible
388  if ((mMergerRecordsReady[mergerBoardId] & mergerChannelBitmask) == 0) {
389  return false;
390  }
391 
392  auto& mergerCh = mMergerRecords[mergerChannelId];
393 
394  // time stamp of the digits to be merged
395  uint32_t bcStart = sc.bunchCrossing;
396  // correct for bunch crossing counter rollover if needed
397  if (bcStart < mergerCh.bcEnd) {
398  bcStart += BCROLLOVER;
399  }
400 
401  // if the new digit starts just one ADC clock cycle after the end of the previous,
402  // the it can be merged into the existing one
403  if ((bcStart - mergerCh.bcEnd) != ONEADCCLOCK) {
404  return false;
405  }
406 
407  // add total charge and number of samples to existing digit
408  auto& digit = mDigits[mergerCh.digitId].digit;
409 
410  digit.setADC(digit.getADC() + sc.sum());
411  uint32_t newNofSamples = digit.getNofSamples() + sc.nofSamples();
412  if (newNofSamples > MAXNOFSAMPLES) {
413  newNofSamples = MAXNOFSAMPLES;
414  }
415  digit.setNofSamples(newNofSamples);
416 
417  // update the time stamp of the signal's end
418  mergerCh.bcEnd = (bcStart & TWENTYBITSATONE) + (sc.nofSamples() - 1) * 4;
419 
420  return true;
421 }

443 {

477 bool DataDecoder::addDigit(const DsElecId& dsElecId, DualSampaChannelId channel, const o2::mch::raw::SampaCluster& sc)
478 {
479  int deId, dsIddet, padId;
480  if (!getPadMapping(dsElecId, channel, deId, dsIddet, padId)) {
481  return false;
482  }
483 
484  uint32_t digitadc = sc.sum();
485 
486  if (mDebug) {
487  auto s = asString(dsElecId);
488  auto ch = fmt::format("{}-CH{:02d}", s, channel);
489  LOG(info) << ch << " "
490  << fmt::format("PAD ({:04d} {:04d} {:04d})\tADC {:06d} TIME ({} {} {:02d}) SIZE {} END {}",
491  deId, dsIddet, padId, digitadc, mOrbit, sc.bunchCrossing, sc.sampaTime, sc.nofSamples(), (sc.sampaTime + sc.nofSamples() - 1))
492  << (((sc.sampaTime + sc.nofSamples() - 1) >= 98) ? " *" : "");
493  }
494 
495  // skip channels not associated to any pad
496  if (padId < 0) {
497  LOGP(alarm, "got invalid padId from dsElecId={} dualSampaId={} channel={}", asString(dsElecId), dsIddet, channel);
498  return false;
499  }
500 
501  RawDigit digit;
502  digit.digit = o2::mch::Digit(deId, padId, digitadc, 0, sc.nofSamples());
503  digit.info.chip = channel / 32;
504  digit.info.ds = dsElecId.elinkId();
505  digit.info.solar = dsElecId.solarId();
506  digit.info.sampaTime = sc.sampaTime;
507  digit.info.bunchCrossing = sc.bunchCrossing;
508  digit.info.orbit = mOrbit;
509 
510  mDigits.emplace_back(digit);
511 
512  if (mDebug) {
513  RawDigit& lastDigit = mDigits.back();
514  LOGP(info, "DIGIT STORED: ORBIT {} ADC {} DE {} PADID {} TIME {} BXCOUNT {}",
515  mOrbit, lastDigit.getADC(), lastDigit.getDetID(), lastDigit.getPadID(),
516  lastDigit.getSampaTime(), lastDigit.getBunchCrossing());
517  }
518  return true;
519 }

551 void DataDecoder::decodePage(gsl::span<const std::byte> page)
552 {
553  uint8_t isStopRDH = 0;
554  uint32_t orbit;
555  uint32_t feeId;
556  uint32_t linkId;
557 
558  auto heartBeatHandler = [&](DsElecId dsElecId, uint8_t chip, uint32_t bunchCrossing) {
559  auto ds = dsElecId.elinkId();
560  auto solar = dsElecId.solarId();
561  uint64_t chipId = getChipId(solar, ds, chip);
562 
563  uint32_t bcTF = subtractBcOffset(bunchCrossing, mSampaTimeOffset);
564 
565  if (mDebug) {
566  auto s = asString(dsElecId);
567  LOGP(info, "HeartBeat: {}-CHIP{} -> {}/{} offset {} -> bcTF {}",
568  s, chip, mFirstOrbitInTF, bunchCrossing, mSampaTimeOffset, bcTF);
569  }
570 
571  if (chipId >= DataDecoder::sReadoutChipsNum) {
572  return;
573  }
574 
575  if (!mTimeFrameStartRecords[chipId].update(mFirstOrbitInTF, bcTF)) {
576  if (mErrorCount < MCH_DECODER_MAX_ERROR_COUNT) {
577  auto s = asString(dsElecId);
578  LOGP(warning, "Bad HeartBeat packet received: {}-CHIP{} {}/{} (last {}/{})",
579  s, chip, mFirstOrbitInTF, bcTF, mTimeFrameStartRecords[chipId].mOrbitPrev, mTimeFrameStartRecords[chipId].mBunchCrossingPrev);
580  mErrorCount += 1;
581  }
582  }
583  };
584 
585  auto channelHandler = [&](DsElecId dsElecId, DualSampaChannelId channel,
586  o2::mch::raw::SampaCluster sc) {
587  if (mChannelHandler) {
588  mChannelHandler(dsElecId, channel, sc);
589  }
590 
591  if (mDs2manu) {
592  LOGP(error, "using ds2manu");
593  channel = ds2manu(int(channel));
594  }
595 
596  uint32_t mergerChannelId;
597  uint32_t mergerBoardId;
598  if (!getMergerChannelId(dsElecId, channel, mergerChannelId, mergerBoardId)) {
599  LOGP(error, "dsElecId={} is out-of-bounds", asString(dsElecId));
600  return;
601  }
602  uint64_t mergerChannelBitmask = getMergerChannelBitmask(channel);
603 
604  if (mergeDigits(mergerChannelId, mergerBoardId, mergerChannelBitmask, sc)) {
605  return;
606  }
607 
608  if (!addDigit(dsElecId, channel, sc)) {
609  return;
610  }
611 
612  updateMergerRecord(mergerChannelId, mergerBoardId, mergerChannelBitmask, mDigits.size() - 1);

651 int32_t DataDecoder::getDigitTime(uint32_t orbitStart, uint32_t bcStart, uint32_t orbitDigit, uint32_t bcDigit)
652 {
653  // We use the difference of orbits values to estimate the minimum and maximum allowed
654  // difference in bunch crossings
655  int64_t dOrbit = static_cast<int64_t>(orbitDigit) - static_cast<int64_t>(orbitStart);
656 
657  // Digits might be sent out later than the orbit in which they were recorded.
658  // We account for this by allowing an extra -3 / +10 orbits when converting the
659  // difference from orbit numbers to bunch crossings.
660  int64_t dBcMin = (dOrbit - 50) * bcInOrbit;
661  int64_t dBcMax = (dOrbit + 3) * bcInOrbit;
662 
663  // Difference in bunch crossing values
664  int64_t dBc = static_cast<int64_t>(bcDigit) - static_cast<int64_t>(bcStart);
665 
666  if (dBc < dBcMin) {
667  // the difference is too small, so we assume that it needs to be
668  // incremented by one rollover factor
669  dBc += bcRollOver;
670  } else if (dBc > dBcMax) {
671  // the difference is too big, so we assume that it needs to be
672  // decremented by one rollover factor
673  dBc -= bcRollOver;
674  }
675 
676  return static_cast<int32_t>(dBc);
677 }

681 bool DataDecoder::getTimeFrameStartRecord(const RawDigit& digit, uint32_t& orbitTF, uint32_t& bcTF)
682 {
683  static constexpr uint32_t bcInOrbit = o2::constants::lhc::LHCMaxBunches;
684  static constexpr uint32_t twentyBitsAtOne = 0xFFFFF;
685 
686  // first orbit of the current TF
687  orbitTF = mFirstOrbitInTF;
688 
689  auto& d = digit.digit;
690  auto& info = digit.info;
691 
692  auto chipId = getChipId(info.solar, info.ds, info.chip);
693  auto& tfStart = mTimeFrameStartRecords[chipId];
694 
695  if (tfStart.mOrbit < 0) {
696  if (mErrorCount < MCH_DECODER_MAX_ERROR_COUNT) {
697  LOGP(alarm, "Missing TF start record for S{}-J{}-DS{}-CHIP{}", info.solar, info.ds / 5 + 1, info.ds % 5, info.chip);
698  mErrorCount += 1;
699  }
700  return false;
701  }
702 
703  if (tfStart.mValid == false) {
704  if (mErrorCount < MCH_DECODER_MAX_ERROR_COUNT) {
705  LOGP(alarm, "Invalid TF start record for S{}-J{}-DS{}-CHIP{}", info.solar, info.ds / 5 + 1, info.ds % 5, info.chip);
706  mErrorCount += 1;
707  }
708  }
709 
710  // orbit and BC from the last received HB packet
711  uint32_t orbitHBP = tfStart.mOrbit;
712  // SAMPA BC at the beginning of the current TF
713  bcTF = tfStart.mBunchCrossing;
714 
715  if (orbitHBP != orbitTF) {
716  // we correct the BC from the last received HB packet, if it was recorded from an older TF
717  bcTF += (orbitTF - orbitHBP) * bcInOrbit;
718  // only keep 20 bits
719  bcTF &= twentyBitsAtOne;
720 
721  // update the time frame start information for this chip, to speed-up the computations
722  // in case another digit from the same chip is found in the same time frame.
723  tfStart.update(orbitTF, bcTF);
724  }
725 
726  return true;
727 }
