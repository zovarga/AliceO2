36 //_____________________________________________
37 void LHCClockDataHisto::fill(const gsl::span<const o2::dataformats::CalibInfoTOF> data)
38 {
39  // fill container
40  for (int i = data.size(); i--;) {
41  auto ch = data[i].getTOFChIndex();
42  auto dt = data[i].getDeltaTimePi();
43  auto tot = data[i].getTot();
44  auto corr = calibApi->getTimeCalibration(ch, tot); // we take into account LHCphase, offsets and time slewing
45  dt -= corr;
46 
47  // printf("ch=%d - tot=%f - corr=%f -> dtcorr = %f (range=%f, bin=%d)\n",ch,tot,corr,dt,range,int((dt+range)*v2Bin));
48 
49  dt += range;
50  if (dt > 0 && dt < 2 * range) {
51  histo[int(dt * v2Bin)]++;
52  entries++;
53  }
54  }
55 }

84 //_____________________________________________
85 void LHCClockCalibrator::finalizeSlot(Slot& slot)
86 {
87  static const double TFlength = 1E-3 * o2::raw::HBFUtils::Instance().getNOrbitsPerTF() * o2::constants::lhc::LHCOrbitMUS; // in ms
88 
89  // Extract results for the single slot
90  o2::tof::LHCClockDataHisto* c = slot.getContainer();
91  LOG(info) << "Finalize slot " << slot.getTFStart() << " <= TF <= " << slot.getTFEnd() << " with "
92  << c->getEntries() << " entries";
93  std::array<double, 3> fitValues;
94  double fitres = fitGaus(c->nbins, c->histo.data(), -(c->range), c->range, fitValues, nullptr, 2., true);
95  if (fitres >= 0) {
96  LOG(info) << "Fit result " << fitres << " Mean = " << fitValues[1] << " Sigma = " << fitValues[2];
97  } else {
98  LOG(error) << "Fit failed with result = " << fitres;
99  }
100 
101  // TODO: the timestamp is now given with the TF index, but it will have
102  // to become an absolute time. This is true both for the lhc phase object itself
103  // and the CCDB entry
104  std::map<std::string, std::string> md;
105  LHCphase l;
106  l.addLHCphase(0, fitValues[1]);
107  l.addLHCphase(999999999, fitValues[1]);
108  auto clName = o2::utils::MemFileHelper::getClassName(l);
109  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
110 
111  uint64_t starting = slot.getTFStart() * TFlength;
112  uint64_t stopping = slot.getTFEnd() * TFlength;
113  LOG(info) << "starting = " << starting * 1E-3 << " - stopping = " << stopping * 1E-3 << " -> phase = " << fitValues[1] << " ps";
114 
115  mInfoVector.emplace_back("TOF/Calib/LHCphase", clName, flName, md, starting, stopping);
116  mLHCphaseVector.emplace_back(l);
117 
118  slot.print();
119 }
