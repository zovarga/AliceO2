27 template <typename T>
28 FlatHisto2D<T>::FlatHisto2D(uint32_t nbx, T xmin, T xmax, uint32_t nby, T ymin, T ymax)
29 {
30  assert(nbx > 0 && xmin < xmax);
31  assert(nby > 0 && ymin < ymax);
32  mContainer.resize(nbx * nby + NServiceSlots, 0.);
33  mContainer[NBinsX] = nbx;
34  mContainer[NBinsY] = nby;
35  mContainer[XMin] = xmin;
36  mContainer[XMax] = xmax;
37  mContainer[YMin] = ymin;
38  mContainer[YMax] = ymax;
39  mContainer[BinSizeX] = (xmax - xmin) / nbx;
40  mContainer[BinSizeY] = (ymax - ymin) / nby;
41  init(gsl::span<const T>(mContainer.data(), mContainer.size()));
42 }

96 template <typename T>
97 void FlatHisto2D<T>::init(const gsl::span<const T> ext)
98 { // when reading from file, need to call this method to make it operational
99  assert(ext.size() > NServiceSlots);
100  mContainerView = ext;
101  mDataPtr = const_cast<T*>(&ext[NServiceSlots]);
102  mNBinsX = (uint32_t)ext[NBinsX];
103  mNBinsY = (uint32_t)ext[NBinsY];
104  mXMin = ext[XMin];
105  mXMax = ext[XMax];
106  mYMin = ext[YMin];
107  mYMax = ext[YMax];
108  mBinSizeX = ext[BinSizeX];
109  mBinSizeY = ext[BinSizeY];
110  mBinSizeXInv = 1. / mBinSizeX;
111  mBinSizeYInv = 1. / mBinSizeY;

114 template <typename T>
115 std::unique_ptr<TH2F> FlatHisto2D<T>::createTH2F(const std::string& name)
116 {
117  auto h = std::make_unique<TH2F>(name.c_str(), name.c_str(), getNBinsX(), getXMin(), getXMax(), getNBinsY(), getYMin(), getYMax());
118  for (uint32_t i = getNBinsX(); i--;) {
119  for (uint32_t j = getNBinsY(); j--;) {
120  auto w = getBinContent(i, j);
121  if (w) {
122  h->SetBinContent(i + 1, j + 1, w);
123  }
124  }
125  }
126  return std::move(h);
127 }
