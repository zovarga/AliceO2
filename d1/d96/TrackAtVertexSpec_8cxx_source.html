111  void run(framework::ProcessingContext& pc)
112  {
114 
115  // get the ROFs, tracks and vertices
116  auto rofs = pc.inputs().get<gsl::span<ROFRecord>>("rofs");
117  auto tracks = pc.inputs().get<gsl::span<TrackMCH>>("tracks");
118  auto vertices = pc.inputs().get<gsl::span<math_utils::Point3D<double>>>("vertices");
119 
120  if (vertices.size() != rofs.size()) {
121  throw length_error("number of vertices different from number of events");
122  }
123 
124  // for each event, propagate the tracks to the vertex
125  mTracksAtVtx.clear();
126  int iVertex(-1);
127  int nTracksTot(0);
128  for (const auto& rof : rofs) {
129  auto tStart = std::chrono::high_resolution_clock::now();
130  nTracksTot += extrapTracksToVertex(tracks.subspan(rof.getFirstIdx(), rof.getNEntries()), vertices[++iVertex]);
131  auto tEnd = std::chrono::high_resolution_clock::now();
132  mElapsedTime += tEnd - tStart;
133  }
134 
135  // create the output message
136  auto msgOut = pc.outputs().make<char>(Output{"MCH", "TRACKSATVERTEX", 0, Lifetime::Timeframe},
137  mTracksAtVtx.size() * sizeof(int) + nTracksTot * sizeof(TrackAtVtxStruct));
138 
139  // write the tracks
140  writeTracks(msgOut.data());
141  }
142 
143  private:
144  //_________________________________________________________________________________________________
145  int extrapTracksToVertex(gsl::span<const TrackMCH> tracks, const math_utils::Point3D<double>& vertex)
146  {
149 
150  auto& tracksAtVtx = mTracksAtVtx.emplace_back();
151  int trackIdx(-1);
152 
153  for (const auto& track : tracks) {
154 
155  // create a new track at vertex pointing to the current track (index within the current event)
156  auto& trackAtVtx = tracksAtVtx.emplace_back();
157  trackAtVtx.mchTrackIdx = ++trackIdx;
158 
159  // extrapolate to vertex
160  TrackParam trackParamAtVertex(track.getZ(), track.getParameters());
161  if (!TrackExtrap::extrapToVertex(trackParamAtVertex, vertex.x(), vertex.y(), vertex.z(), 0., 0.)) {
162  tracksAtVtx.pop_back();
163  continue;
164  }
165  trackAtVtx.paramAtVertex.x = trackParamAtVertex.getNonBendingCoor();
166  trackAtVtx.paramAtVertex.y = trackParamAtVertex.getBendingCoor();
167  trackAtVtx.paramAtVertex.z = trackParamAtVertex.getZ();
168  trackAtVtx.paramAtVertex.px = trackParamAtVertex.px();
169  trackAtVtx.paramAtVertex.py = trackParamAtVertex.py();
170  trackAtVtx.paramAtVertex.pz = trackParamAtVertex.pz();
171  trackAtVtx.paramAtVertex.sign = trackParamAtVertex.getCharge();
172 
173  // extrapolate to DCA
174  TrackParam trackParamAtDCA(track.getZ(), track.getParameters());
175  if (!TrackExtrap::extrapToVertexWithoutBranson(trackParamAtDCA, vertex.z())) {
176  tracksAtVtx.pop_back();
177  continue;
178  }
179  double dcaX = trackParamAtDCA.getNonBendingCoor() - vertex.x();
180  double dcaY = trackParamAtDCA.getBendingCoor() - vertex.y();
181  trackAtVtx.dca = TMath::Sqrt(dcaX * dcaX + dcaY * dcaY);
182 
183  // extrapolate to the end of the absorber
184  TrackParam trackParamAtRAbs(track.getZ(), track.getParameters());
185  if (!TrackExtrap::extrapToZ(trackParamAtRAbs, -505.)) {
186  tracksAtVtx.pop_back();
187  continue;
188  }
189  double xAbs = trackParamAtRAbs.getNonBendingCoor();
190  double yAbs = trackParamAtRAbs.getBendingCoor();
191  trackAtVtx.rAbs = TMath::Sqrt(xAbs * xAbs + yAbs * yAbs);
192  }
193 
194  return tracksAtVtx.size();
195  }

197  //_________________________________________________________________________________________________
198  void writeTracks(char* bufferPtr) const
199  {
201 
202  for (const auto& tracksAtVtx : mTracksAtVtx) {
203 
204  // write the number of tracks
205  int nTracks = tracksAtVtx.size();
206  memcpy(bufferPtr, &nTracks, sizeof(int));
207  bufferPtr += sizeof(int);
208 
209  // write the tracks
210  if (nTracks > 0) {
211  memcpy(bufferPtr, tracksAtVtx.data(), nTracks * sizeof(TrackAtVtxStruct));
212  bufferPtr += nTracks * sizeof(TrackAtVtxStruct);
213  }
214  }
215  }
216 
217  std::vector<std::vector<TrackAtVtxStruct>> mTracksAtVtx{};
218  std::chrono::duration<double> mElapsedTime{};
219 };
