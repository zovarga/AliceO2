51  void init(o2::framework::InitContext& ic) final
52  {
53  // parse command line arguments
54  mMaxEvents = static_cast<uint32_t>(ic.options().get<int>("max-events"));
55  mUseOldSubspec = ic.options().get<bool>("use-old-subspec");
56  const bool createOccupancyMaps = ic.options().get<bool>("create-occupancy-maps");
57  mForceQuit = ic.options().get<bool>("force-quit");
58  mCheckDuplicates = ic.options().get<bool>("check-for-duplicates");
59  mRemoveDuplicates = ic.options().get<bool>("remove-duplicates");
60  if (mSendCEdigits) {
61  mRemoveCEdigits = true;
62  } else {
63  mRemoveCEdigits = ic.options().get<bool>("remove-ce-digits");
64  }
65 
66  if (mUseOldSubspec) {
67  LOGP(info, "Using old subspecification (CruId << 16) | ((LinkId + 1) << (CruEndPoint == 1 ? 8 : 0))");
68  }
69 
70  // set up ADC value filling
71  mRawReader.createReader("");
72 
73  if (!ic.options().get<bool>("ignore-grp")) {
74  const auto inputGRP = o2::base::NameConf::getGRPFileName();
75  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
76  if (grp) {
77  const auto nhbf = (int)grp->getNHBFPerTF();
78  const int lastTimeBin = nhbf * 891 / 2;
79  mDigitDump.setTimeBinRange(0, lastTimeBin);
80  LOGP(info, "Using GRP NHBF = {} to set last time bin to {}, might be overwritte via --configKeyValues", nhbf, lastTimeBin);
81  }
82  }
83 
84  mDigitDump.init();
85  mDigitDump.setInMemoryOnly();
86  const auto pedestalFile = ic.options().get<std::string>("pedestal-url");
87  if (pedestalFile.length()) {
88  if (pedestalFile.find("ccdb") != std::string::npos) {
89  LOGP(info, "Loading pedestals from ccdb: {}", pedestalFile);
90  auto& cdb = o2::ccdb::BasicCCDBManager::instance();
91  cdb.setURL(pedestalFile);
92  if (cdb.isHostReachable()) {
93  auto pedestalNoise = cdb.get<std::unordered_map<std::string, CalPad>>("TPC/Calib/PedestalNoise");
94  CalPad* pedestal = nullptr;
95  if (pedestalNoise) {
96  pedestal = &pedestalNoise->at("Pedestals");
97  }
98  if (pedestal) {
99  mDigitDump.setPedestals(pedestal);
100  } else {
101  LOGP(error, "could not load pedestals from {}", pedestalFile);
102  }
103  } else {
104  LOGP(error, "ccdb access to {} requested, but host is not reachable. Cannot load Pedestals", pedestalFile);
105  }
106  } else {
107  LOGP(info, "Setting pedestal file: {}", pedestalFile);
108  mDigitDump.setPedestalAndNoiseFile(pedestalFile);
109  }
110  }
111 
112  // set up cluster qc if requested
113  if (createOccupancyMaps) {
114  mClusterQC = std::make_unique<qc::Clusters>();
115  }
116 
117  mRawReader.setADCDataCallback([this](const PadROCPos& padROCPos, const CRU& cru, const gsl::span<const uint32_t> data) -> int {
118  const int timeBins = mDigitDump.update(padROCPos, cru, data);
119  mDigitDump.setNumberOfProcessedTimeBins(std::max(mDigitDump.getNumberOfProcessedTimeBins(), size_t(timeBins)));
120  return timeBins;
121  });
122 
123  mRawReader.setLinkZSCallback([this](int cru, int rowInSector, int padInRow, int timeBin, float adcValue) -> bool {
124  CRU cruID(cru);
125  const PadRegionInfo& regionInfo = Mapper::instance().getPadRegionInfo(cruID.region());
126  mDigitDump.updateCRU(cruID, rowInSector - regionInfo.getGlobalRowOffset(), padInRow, timeBin, adcValue);
127  if (mClusterQC) {
128  mClusterQC->fillADCValue(cru, rowInSector, padInRow, timeBin, adcValue);
129  }
130  return true;
131  });
132  }

134  void run(o2::framework::ProcessingContext& pc) final
135  {
136  // in case the maximum number of events was reached don't do further processing
137  if (mReadyToQuit) {
138  return;
139  }
140 
141  auto& reader = mRawReader.getReaders()[0];
142  mActiveSectors = calib_processing_helper::processRawData(pc.inputs(), reader, mUseOldSubspec);
143 
144  mDigitDump.incrementNEvents();
145  if (mClusterQC) {
146  mClusterQC->endTF();
147  }
148  LOGP(info, "Number of processed events: {} ({})", mDigitDump.getNumberOfProcessedEvents(), mMaxEvents);
149 
150  snapshotDigits(pc.outputs());
151 
152  if (mMaxEvents && (mDigitDump.getNumberOfProcessedEvents() >= mMaxEvents)) {
153  LOGP(info, "Maximm number of events reached ({}), no more processing will be done", mMaxEvents);
154  mReadyToQuit = true;
155  if (mForceQuit) {
156  pc.services().get<ControlService>().endOfStream();
157  pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
158  } else {
159  // pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
160  }
161  }
162  }
163 
164  void endOfStream(o2::framework::EndOfStreamContext& ec) final
165  {
166  LOGP(info, "endOfStream");
167  if (mActiveSectors) {
168  snapshotDigits(ec.outputs());
169  }
170  ec.services().get<ControlService>().readyToQuit(QuitRequest::Me);
171 
172  if (mClusterQC) {
173  dumpClusterQC();
174  }
175  }

177  private:
178  DigitDump mDigitDump;
179  std::unique_ptr<qc::Clusters> mClusterQC;
180  rawreader::RawReaderCRUManager mRawReader;
181  uint32_t mMaxEvents{0};
182  bool mReadyToQuit{false};
183  bool mCalibDumped{false};
184  bool mUseOldSubspec{false};
185  bool mForceQuit{false};
186  bool mCheckDuplicates{false};
187  bool mRemoveDuplicates{false};
188  bool mRemoveCEdigits{false};
189  bool mSendCEdigits{false};
190  uint64_t mActiveSectors{0};
191  std::vector<int> mSectors{};

193  //____________________________________________________________________________
194  void snapshotDigits(DataAllocator& output)
195  {
196  if (mCheckDuplicates || mRemoveDuplicates) {
197  // iplicityly sorts
198  mDigitDump.checkDuplicates(mRemoveDuplicates);
199  } else {
200  mDigitDump.sortDigits();
201  }
202 
203  std::array<std::vector<Digit>, Sector::MAXSECTOR> ceDigits;
204  if (mRemoveCEdigits) {
205  mDigitDump.removeCEdigits(10, 100, &ceDigits);
206  }
207 
208  for (auto isector : mSectors) {
209  o2::tpc::TPCSectorHeader header{isector};
210  header.activeSectors = mActiveSectors;
211  // digit for now are transported per sector, not per lane
212  output.snapshot(Output{"TPC", "DIGITS", static_cast<SubSpecificationType>(isector), Lifetime::Timeframe, header},
213  mDigitDump.getDigits(isector));
214  if (mSendCEdigits) {
215  output.snapshot(Output{"TPC", "CEDIGITS", static_cast<SubSpecificationType>(isector), Lifetime::Timeframe, header},
216  ceDigits[isector]);
217  }
218  }
219  mDigitDump.clearDigits();
220  mActiveSectors = 0;
221  }
222 
223  //____________________________________________________________________________
224  void dumpClusterQC()
225  {
226  mClusterQC->normalize();
227  mClusterQC->dumpToFile("ClusterQC.root", 2);
228  }

231 DataProcessorSpec getRawToDigitsSpec(int channel, const std::string inputSpec, std::vector<int> const& tpcSectors, bool sendCEdigits)
232 {
233  using device = o2::tpc::TPCDigitDumpDevice;
234 
235  std::vector<OutputSpec> outputs;
236  for (auto isector : tpcSectors) {
237  outputs.emplace_back("TPC", "DIGITS", static_cast<SubSpecificationType>(isector), Lifetime::Timeframe);
238  if (sendCEdigits) {
239  outputs.emplace_back("TPC", "CEDIGITS", static_cast<SubSpecificationType>(isector), Lifetime::Timeframe);
240  }
241  }
242 
243  return DataProcessorSpec{
244  fmt::format("tpc-raw-to-digits-{}", channel),
245  select(inputSpec.data()),
246  outputs,
247  AlgorithmSpec{adaptFromTask<device>(tpcSectors, sendCEdigits)},
248  Options{
249  {"max-events", VariantType::Int, 0, {"maximum number of events to process"}},
250  {"use-old-subspec", VariantType::Bool, false, {"use old subsecifiation definition"}},
251  {"force-quit", VariantType::Bool, false, {"force quit after max-events have been reached"}},
252  {"pedestal-url", VariantType::String, "", {"file with pedestals and noise or ccdb url for zero suppression"}},
253  {"create-occupancy-maps", VariantType::Bool, false, {"create occupancy maps and store them to local root file for debugging"}},
254  {"check-for-duplicates", VariantType::Bool, false, {"check if duplicate digits exist and only report them"}},
255  {"remove-duplicates", VariantType::Bool, false, {"check if duplicate digits exist and remove them"}},
256  {"remove-ce-digits", VariantType::Bool, false, {"find CE position and remove digits around it"}},
257  {"ignore-grp", VariantType::Bool, false, {"ignore GRP file"}},
258  } // end Options
259  }; // end DataProcessorSpec
