53  void run(o2::framework::ProcessingContext& pc) final
54  {
55  auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("digits").header)->startTime;
56  auto&& digits = pc.inputs().get<gsl::span<o2::cpv::Digit>>("digits");
57  auto&& trigrecs = pc.inputs().get<gsl::span<o2::cpv::TriggerRecord>>("trigrecs");
58  LOG(info) << "Processing TF " << tfcounter << " with " << digits.size() << " digits in " << trigrecs.size() << " trigger records.";
59  auto& slotTF = mCalibrator->getSlotForTF(tfcounter);
60 
61  for (auto trigrec = trigrecs.begin(); trigrec != trigrecs.end(); trigrec++) { //event loop
62  // here we're filling TimeSlot event by event
63  // and when last event is reached we call mCalibrator->process() to finalize the TimeSlot
64  auto&& digitsInOneEvent = digits.subspan((*trigrec).getFirstEntry(), (*trigrec).getNumberOfObjects());
65  if ((trigrec + 1) == trigrecs.end()) { //last event in current TF, let's process corresponding TimeSlot
66  //LOG(info) << "last event, I call mCalibrator->process()";
67  mCalibrator->process(tfcounter, digitsInOneEvent); //fill TimeSlot with digits from 1 event and check slots for finalization
68  } else {
69  slotTF.getContainer()->fill(digitsInOneEvent); //fill TimeSlot with digits from 1 event
70  }
71  }
72 
73  auto infoVecSize = mCalibrator->getCcdbInfoPedestalsVector().size();
74  auto pedsVecSize = mCalibrator->getPedestalsVector().size();
75  if (infoVecSize > 0) {
76  LOG(info) << "Created " << infoVecSize << " ccdb infos and " << pedsVecSize << " pedestal objects for TF " << tfcounter;
77  }
78  sendOutput(pc.outputs());
79  }

117  template <class Payload>
118  bool sendOutputWhat(const std::vector<Payload>& payloadVec, std::vector<o2::ccdb::CcdbObjectInfo>&& infoVec, header::DataDescription what, DataAllocator& output)
119  {
120  assert(payloadVec.size() == infoVec.size());
121  if (!payloadVec.size()) {
122  return false;

124

126  auto& w = infoVec[i];
