41 // All we do is to printout
42 void ControlService::readyToQuit(QuitRequest what)
43 {
44  std::scoped_lock lock(mMutex);
45  if (mOnce == true) {
46  return;
47  }
48  mOnce = true;
49  switch (what) {
50  case QuitRequest::All:
51  mDeviceState.quitRequested = true;
52  mDriverClient.tell("CONTROL_ACTION: READY_TO_QUIT_ALL");
53  break;
54  case QuitRequest::Me:
55  mDeviceState.quitRequested = true;
56  mDriverClient.tell("CONTROL_ACTION: READY_TO_QUIT_ME");
57  break;
58  }
59  mDriverClient.flushPending();
60 }

62 void ControlService::notifyStreamingState(StreamingState state)
63 {
64  std::scoped_lock lock(mMutex);
65  switch (state) {
66  case StreamingState::Idle:
67  mDriverClient.tell("CONTROL_ACTION: NOTIFY_STREAMING_STATE IDLE");
68  break;
69  case StreamingState::Streaming:
70  mDriverClient.tell("CONTROL_ACTION: NOTIFY_STREAMING_STATE STREAMING");
71  break;
72  case StreamingState::EndOfStreaming:
73  mDriverClient.tell("CONTROL_ACTION: NOTIFY_STREAMING_STATE EOS");
74  break;
75  default:
76  throw std::runtime_error("Unknown streaming state");
77  }
78  mDriverClient.flushPending();
79 }
