73  auto doParseConfig = [](std::string const& token, ParsedConfigMatch& configMatch, DeviceInfo& info) -> bool {
74  auto ts = " " + token;
75  if (DeviceConfigHelper::parseConfig(ts, configMatch)) {
76  DeviceConfigHelper::processConfig(configMatch, info);

91  } else if (doParseConfig(token, configMatch, (*mContext.infos)[mIndex]) && mContext.infos) {
92  LOG(debug2) << "Found configuration information for pid " << mPid;
93  } else {

114  void endChunk() override
