61 void Detector::buildFT3FromFile(std::string configFileName)
62 {
63  // Geometry description from file. One line per disk
64  // z_layer r_in r_out Layerx2X0
65  // This simple file reader is not failproof. Do not add empty lines!
66 
67  /*
68  # Sample MFT configuration
69  # z_layer r_in r_out Layerx2X0
70  -45.3 2.5 9.26 0.0042
71  -46.7 2.5 9.26 0.0042
72  -48.6 2.5 9.8 0.0042
73  -50.0 2.5 9.8 0.0042
74  -52.4 2.5 10.43 0.0042
75  -53.8 2.5 10.43 0.0042
76  -67.7 3.82 13.01 0.0042
77  -69.1 3.82 13.01 0.0042
78  -76.1 3.92 14.35 0.0042
79  -77.5 3.92 14.35 0.0042
80  */
81 
82  mLayerName.clear();
83  mLayers.clear();
84  mLayerID.clear();
85  mLayerName.resize(1);
86  mLayers.resize(1);
87 
88  LOG(info) << "Building FT3 Detector: From file";
89  LOG(info) << " FT3 detector configuration: " << configFileName;
90  std::ifstream ifs(configFileName.c_str());
91  if (!ifs.good()) {
92  LOG(fatal) << " Invalid FT3Base.configFile!";
93  }
94  std::string tempstr;
95  float z_layer, r_in, r_out, Layerx2X0;
96  char delimiter;
97  int layerNumber = 0;
98  while (std::getline(ifs, tempstr)) {
99  if (tempstr[0] == '#') {
100  LOG(info) << " Comment: " << tempstr;
101  continue;
102  }
103  std::istringstream iss(tempstr);
104  iss >> z_layer;
105  iss >> r_in;
106  iss >> r_out;
107  iss >> Layerx2X0;
108 
109  std::string layerName = GeometryTGeo::getFT3LayerPattern() + std::string("_") + std::to_string(layerNumber);
110  mLayerName[0].push_back(layerName);
111  LOG(info) << "Adding Layer " << layerName << " at z = " << z_layer << " ; r_in = " << r_in << " ; r_out = " << r_out << " x/X0 = " << Layerx2X0;
112  auto& thisLayer = mLayers[0].emplace_back(0, layerNumber, layerName, z_layer, r_in, r_out, Layerx2X0);
113  layerNumber++;
114  }
115 
116  mNumberOfLayers = layerNumber;
117  LOG(info) << " Loaded FT3 Detector with " << mNumberOfLayers << " layers";
118 }

143 //_________________________________________________________________________________________________
144 void Detector::buildBasicFT3(const FT3BaseParam& param)
145 {
146  // Build a basic parametrized FT3 detector with nLayers equally spaced between z_first and z_first+z_length
147  // Covering pseudo rapidity [etaIn,etaOut]. Silicon thinkness computed to match layer x/X0
148 
149  LOG(info) << "Building FT3 Detector: Conical Telescope";
150 
151  auto z_first = param.z0;
152  auto z_length = param.zLength;
153  auto etaIn = param.etaIn;
154  auto etaOut = param.etaOut;
155  auto Layerx2X0 = param.Layerx2X0;
156  mNumberOfLayers = param.nLayers;
157  mLayerName.resize(2);
158  mLayerName[0].resize(mNumberOfLayers);
159  mLayerName[1].resize(mNumberOfLayers);
160  mLayerID.clear();
161  mLayers.resize(2);
162 
163  for (Int_t direction : {0, 1}) {
164  for (Int_t layerNumber = 0; layerNumber < mNumberOfLayers; layerNumber++) {
165  std::string layerName = GeometryTGeo::getFT3LayerPattern() + std::to_string(layerNumber + mNumberOfLayers * direction);
166  mLayerName[direction][layerNumber] = layerName;
167 
168  // Adds evenly spaced layers
169  Float_t layerZ = z_first + (layerNumber * z_length / (mNumberOfLayers - 1)) * std::copysign(1, z_first);
170  Float_t rIn = std::abs(layerZ * std::tan(2.f * std::atan(std::exp(-etaIn))));
171  Float_t rOut = std::abs(layerZ * std::tan(2.f * std::atan(std::exp(-etaOut))));
172  auto& thisLayer = mLayers[direction].emplace_back(direction, layerNumber, layerName, layerZ, rIn, rOut, Layerx2X0);
173  }
174  }
175 }

178 void Detector::buildFT3V1()
179 {
180  //Build FT3 detector according to
181  //https://indico.cern.ch/event/992488/contributions/4174473/attachments/2168881/3661331/tracker_parameters_werner_jan_11_2021.pdf
182 
183  LOG(info) << "Building FT3 Detector: V1";
184 
185  mNumberOfLayers = 10;
186  Float_t sensorThickness = 30.e-4;
187  Float_t layersx2X0 = 1.e-2;
188  std::vector<std::array<Float_t, 5>> layersConfig{
189  {26., .5, 3., 0.1f * layersx2X0}, // {z_layer, r_in, r_out, Layerx2X0}
190  {30., .5, 3., 0.1f * layersx2X0},
191  {34., .5, 3., 0.1f * layersx2X0},
192  {77., 3.5, 35., layersx2X0},
193  {100., 3.5, 35., layersx2X0},
194  {122., 3.5, 35., layersx2X0},
195  {150., 3.5, 100., layersx2X0},
196  {180., 3.5, 100., layersx2X0},
197  {220., 3.5, 100., layersx2X0},
198  {279., 3.5, 100., layersx2X0}};
199 
200  mLayerName.resize(2);
201  mLayerName[0].resize(mNumberOfLayers);
202  mLayerName[1].resize(mNumberOfLayers);
203  mLayerID.clear();
204  mLayers.resize(2);
205 
206  for (auto direction : {0, 1}) {
207  for (int layerNumber = 0; layerNumber < mNumberOfLayers; layerNumber++) {
208  std::string directionName = std::to_string(direction);
209  std::string layerName = GeometryTGeo::getFT3LayerPattern() + directionName + std::string("_") + std::to_string(layerNumber);
210  mLayerName[direction][layerNumber] = layerName;
211  auto& z = layersConfig[layerNumber][0];
212 
213  auto& rIn = layersConfig[layerNumber][1];
214  auto& rOut = layersConfig[layerNumber][2];
215  auto& x0 = layersConfig[layerNumber][3];
216 
217  LOG(info) << "Adding Layer " << layerName << " at z = " << z;
218  // Add layers
219  auto& thisLayer = mLayers[direction].emplace_back(direction, layerNumber, layerName, z, rIn, rOut, x0);
220  }
221  }
222 }

225 Detector::Detector(Bool_t active)
226  : o2::base::DetImpl<Detector>("FT3", active),
227  mTrackData(),
228  mHits(o2::utils::createSimVector<o2::itsmft::Hit>())
229 {
230 
231  // FT3 Base configuration parameters
232  auto& ft3BaseParam = FT3BaseParam::Instance();
233 
234  if (ft3BaseParam.configFile != "") {
235  LOG(info) << "FT3 Geometry configuration file provided. Overriding FT3Base.geoModel configuration.";
236  buildFT3FromFile(ft3BaseParam.configFile);
237 
238  } else {
239  switch (ft3BaseParam.geoModel) {
240  case Default:
241  buildFT3V1(); // FT3V1
242  break;
243  case Telescope:
244  buildBasicFT3(ft3BaseParam); // BasicFT3 = Parametrized telescopic detector (equidistant layers)
245  break;
246  default:
247  LOG(fatal) << "Invalid Geometry.\n";
248  break;
249  }
250  }
251  exportLayout();
252 }

318 //_________________________________________________________________________________________________
319 Bool_t Detector::ProcessHits(FairVolume* vol)
320 {
321  // This method is called from the MC stepping
322  if (!(fMC->TrackCharge())) {
323  return kFALSE;
324  }
325 
326  Int_t lay = 0, volID = vol->getMCid();
327  while ((lay <= mLayerID.size()) && (volID != mLayerID[lay])) {
328  ++lay;
329  }
330 
331  auto stack = (o2::data::Stack*)fMC->GetStack();
332 
333  bool startHit = false, stopHit = false;
334  unsigned char status = 0;
335  if (fMC->IsTrackEntering()) {
336  status |= Hit::kTrackEntering;
337  }
338  if (fMC->IsTrackInside()) {
339  status |= Hit::kTrackInside;
340  }
341  if (fMC->IsTrackExiting()) {
342  status |= Hit::kTrackExiting;
343  }
344  if (fMC->IsTrackOut()) {
345  status |= Hit::kTrackOut;
346  }
347  if (fMC->IsTrackStop()) {
348  status |= Hit::kTrackStopped;
349  }
350  if (fMC->IsTrackAlive()) {
351  status |= Hit::kTrackAlive;
352  }
353 
354  // track is entering or created in the volume
355  if ((status & Hit::kTrackEntering) || (status & Hit::kTrackInside && !mTrackData.mHitStarted)) {
356  startHit = true;
357  } else if ((status & (Hit::kTrackExiting | Hit::kTrackOut | Hit::kTrackStopped))) {
358  stopHit = true;
359  }
360 
361  // increment energy loss at all steps except entrance
362  if (!startHit) {
363  mTrackData.mEnergyLoss += fMC->Edep();
364  }
365  if (!(startHit | stopHit)) {
366  return kFALSE; // do noting
367  }
368  if (startHit) {
369  mTrackData.mEnergyLoss = 0.;
370  fMC->TrackMomentum(mTrackData.mMomentumStart);
371  fMC->TrackPosition(mTrackData.mPositionStart);
372  mTrackData.mTrkStatusStart = status;
373  mTrackData.mHitStarted = true;
374  }
375  if (stopHit) {
376  TLorentzVector positionStop;
377  fMC->TrackPosition(positionStop);
378  // Retrieve the indices with the volume path
379  int chipindex = lay;
380 
381  Hit* p = addHit(stack->GetCurrentTrackNumber(), chipindex, mTrackData.mPositionStart.Vect(), positionStop.Vect(),
382  mTrackData.mMomentumStart.Vect(), mTrackData.mMomentumStart.E(), positionStop.T(),
383  mTrackData.mEnergyLoss, mTrackData.mTrkStatusStart, status);
384  // p->SetTotalEnergy(vmc->Etot());
385 
386  // RS: not sure this is needed
387  // Increment number of Detector det points in TParticle
388  stack->addHit(GetDetId());
389  }
390 
391  return kTRUE;
392 }

411  Float_t stminAir = 0.0; // cm "Default value used"

459 //_________________________________________________________________________________________________
460 void Detector::createGeometry()
461 {
462 
463  mGeometryTGeo = GeometryTGeo::Instance();
464 
465  TGeoVolume* volFT3 = new TGeoVolumeAssembly(GeometryTGeo::getFT3VolPattern());
466  TGeoVolume* volIFT3 = new TGeoVolumeAssembly(GeometryTGeo::getFT3InnerVolPattern());
467 
468  LOG(info) << "GeometryBuilder::buildGeometry volume name = " << GeometryTGeo::getFT3VolPattern();
469 
470  TGeoVolume* vALIC = gGeoManager->GetVolume("barrel");
471  if (!vALIC) {
472  LOG(fatal) << "Could not find the top volume";
473  }
474 
475  TGeoVolume* A3IPvac = gGeoManager->GetVolume("OUT_PIPEVACUUM");
476  if (!A3IPvac) {
477  LOG(info) << "Running simulation with no beam pipe.";
478  }
479 
480  LOG(debug) << "FT3 createGeometry: "
481  << Form("gGeoManager name is %s title is %s", gGeoManager->GetName(), gGeoManager->GetTitle());
482 
483  if (mLayers.size() == 2) { // V1 and telescope
484  if (!A3IPvac) {
485  for (Int_t direction : {0, 1}) { // Backward layers at mLayers[0]; Forward layers at mLayers[1]
486  std::string directionString = direction ? "Forward" : "Backward";
487  LOG(info) << "Creating FT3 " << directionString << " layers:";
488  for (Int_t iLayer = 0; iLayer < mLayers[direction].size(); iLayer++) {
489  mLayers[direction][iLayer].createLayer(volFT3);
490  }
491  }
492  vALIC->AddNode(volFT3, 2, new TGeoTranslation(0., 30., 0.));
493  } else { // If beampipe is enabled append inner disks to beampipe filling volume, this should be temporary.
494  for (Int_t direction : {0, 1}) {
495  std::string directionString = direction ? "Forward" : "Backward";
496  LOG(info) << "Creating FT3 " << directionString << " layers:";
497  for (Int_t iLayer = 0; iLayer < mLayers[direction].size(); iLayer++) {
498  if (iLayer < 3) {
499  mLayers[direction][iLayer].createLayer(volIFT3);
500  } else {
501  mLayers[direction][iLayer].createLayer(volFT3);
502  }
503  }
504  }
505  A3IPvac->AddNode(volIFT3, 2, new TGeoTranslation(0., 0., 0.));
506  vALIC->AddNode(volFT3, 2, new TGeoTranslation(0., 30., 0.));
507  }
508 
509  for (auto direction : {0, 1}) {
510  std::string directionString = direction ? "Forward" : "Backward";
511  LOG(info) << "Registering FT3 " << directionString << " LayerIDs:";
512  for (int iLayer = 0; iLayer < mLayers[direction].size(); iLayer++) {
513  auto layerID = gMC ? TVirtualMC::GetMC()->VolId(Form("%s_%d_%d", GeometryTGeo::getFT3SensorPattern(), direction, iLayer)) : 0;
514  mLayerID.push_back(layerID);
515  LOG(info) << " " << directionString << " layer " << iLayer << " LayerID " << layerID;
516  }
517  }
518  }
519 
520  if (mLayers.size() == 1) { // All layers registered at mLayers[0], used when building from file
521  LOG(info) << "Creating FT3 layers:";
522  if (A3IPvac) {
523  for (Int_t iLayer = 0; iLayer < mLayers[0].size(); iLayer++) {
524  if (std::abs(mLayers[0][iLayer].getZ()) < 25) {
525  mLayers[0][iLayer].createLayer(volIFT3);
526  } else {
527  mLayers[0][iLayer].createLayer(volFT3);
528  }
529  }
530  A3IPvac->AddNode(volIFT3, 2, new TGeoTranslation(0., 0., 0.));
531  vALIC->AddNode(volFT3, 2, new TGeoTranslation(0., 30., 0.));
532  } else {
533  for (Int_t iLayer = 0; iLayer < mLayers[0].size(); iLayer++) {
534  mLayers[0][iLayer].createLayer(volFT3);
535  }
536  vALIC->AddNode(volFT3, 2, new TGeoTranslation(0., 30., 0.));
537  }
538  LOG(info) << "Registering FT3 LayerIDs:";
539  for (int iLayer = 0; iLayer < mLayers[0].size(); iLayer++) {
540  auto layerID = gMC ? TVirtualMC::GetMC()->VolId(Form("%s_%d_%d", GeometryTGeo::getFT3SensorPattern(), 0, iLayer)) : 0;
541  mLayerID.push_back(layerID);
542  LOG(info) << " mLayerID[" << iLayer << "] = " << layerID;
543  }
544  }
545 }

548 void Detector::defineSensitiveVolumes()
549 {
550  TGeoManager* geoManager = gGeoManager;
551  TGeoVolume* v;
552 
553  TString volumeName;
554  LOG(info) << "Adding FT3 Sensitive Volumes";

564  }
565  }
566  }
567 
568  if (mLayers.size() == 1) {
569  for (Int_t iLayer = 0; iLayer < mLayers[0].size(); iLayer++) {
570  volumeName = o2::ft3::GeometryTGeo::getFT3SensorPattern() + std::to_string(iLayer);
571  v = geoManager->GetVolume(Form("%s_%d_%d", GeometryTGeo::getFT3SensorPattern(), 0, iLayer));
572  LOG(info) << "Adding FT3 Sensitive Volume => " << v->GetName();
573  AddSensitiveVolume(v);
574  }
575  }
