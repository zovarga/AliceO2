36 void RecoConfigZDC::setIntegration(uint32_t ich, int beg, int end, int beg_ped, int end_ped)
37 {
38  int sig_l = 0;
39  int sig_h = NTimeBinsPerBC - 1;
40  int ped_l = -NTimeBinsPerBC;
41  int ped_h = NTimeBinsPerBC - 1;
42 
43  if (ich >= 0 && ich < NChannels) {
44  if (beg < sig_l || beg > sig_h) {
45  LOG(fatal) << "Integration start = " << beg << " for signal " << ich << " (" << ChannelNames[ich] << ") not in allowed range [" << sig_l << "-" << sig_h << "]";
46  return;
47  }
48  if (end < sig_l || end > sig_h) {
49  LOG(fatal) << "Integration end = " << beg << " for signal " << ich << " (" << ChannelNames[ich] << ") not in allowed range [" << sig_l << "-" << sig_h << "]";
50  return;
51  }
52  if (end < beg) {
53  LOG(fatal) << "Inconsistent integration range for signal " << ich << " (" << ChannelNames[ich] << "): [" << beg << "-" << end << "]";
54  return;
55  }
56  if (beg_ped < ped_l || beg_ped > ped_h) {
57  LOG(fatal) << "Pedestal integration start = " << beg_ped << " for signal " << ich << " (" << ChannelNames[ich] << ") not in allowed range [" << ped_l << "-" << ped_h << "]";
58  return;
59  }
60  if (end_ped < ped_l || end_ped > ped_h) {
61  LOG(fatal) << "Pedestal integration end = " << end_ped << " for signal " << ich << " (" << ChannelNames[ich] << ") not in allowed range [" << ped_l << "-" << ped_h << "]";
62  return;
63  }
64  if (end_ped < beg_ped) {
65  LOG(fatal) << "Inconsistent integration range for pedestal " << ich << " (" << ChannelNames[ich] << "): [" << beg_ped << "-" << end_ped << "]";
66  return;
67  }
68  beg_int[ich] = beg;
69  end_int[ich] = end;
70  beg_ped_int[ich] = beg_ped;
71  end_ped_int[ich] = end_ped;
72  } else {
73  LOG(fatal) << __func__ << " channel " << ich << " not in allowed range";
74  }
75 }
