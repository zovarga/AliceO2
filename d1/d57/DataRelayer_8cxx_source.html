100 DataRelayer::ActivityStats DataRelayer::processDanglingInputs(std::vector<ExpirationHandler> const& expirationHandlers,
101  ServiceRegistry& services, bool createNew)
102 {
103  std::scoped_lock<LockableBase(std::recursive_mutex)> lock(mMutex);
104 
105  ActivityStats activity;
107  if (expirationHandlers.empty()) {
108  return activity;
109  }
110  // Create any slot for the time based fields
111  std::vector<TimesliceSlot> slotsCreatedByHandlers;
112  if (createNew) {
113  for (auto& handler : expirationHandlers) {
114  slotsCreatedByHandlers.push_back(handler.creator(mTimesliceIndex));
115  }
116  }
117  if (slotsCreatedByHandlers.empty() == false) {
118  activity.newSlots++;
119  }
120  // Outer loop, we process all the records because the fact that the record
121  // expires is independent from having received data for it.
122  for (size_t ti = 0; ti < mTimesliceIndex.size(); ++ti) {
123  TimesliceSlot slot{ti};
124  if (mTimesliceIndex.isValid(slot) == false) {
125  continue;
126  }
127  assert(mDistinctRoutesIndex.empty() == false);
128  auto& variables = mTimesliceIndex.getVariablesForSlot(slot);
129  auto timestamp = VariableContextHelpers::getTimeslice(variables);
130  // We iterate on all the hanlders checking if they need to be expired.
131  for (size_t ei = 0; ei < expirationHandlers.size(); ++ei) {
132  auto& expirator = expirationHandlers[ei];
133  // We check that no data is already there for the given cell
134  // it is enough to check the first element
135  auto& part = mCache[ti * mDistinctRoutesIndex.size() + expirator.routeIndex.value];
136  if (part.size() > 0 && part.header(0) != nullptr) {
137  continue;
138  }
139  if (part.size() > 0 && part.payload(0) != nullptr) {
140  continue;
141  }
142  // We check that the cell can actually be expired.
143  if (!expirator.checker) {
144  continue;
145  }
146  if (slotsCreatedByHandlers[ei] != slot) {
147  continue;
148  }
149  if (expirator.checker(services, timestamp.value) == false) {
150  continue;
151  }
152 
153  assert(ti * mDistinctRoutesIndex.size() + expirator.routeIndex.value < mCache.size());
154  assert(expirator.handler);
155  PartRef newRef;
156  expirator.handler(services, newRef, variables);
157  part.reset(std::move(newRef));
158  activity.expiredSlots++;
159 
160  mTimesliceIndex.markAsDirty(slot, true);
161  assert(part.header(0) != nullptr);
162  assert(part.payload(0) != nullptr);
163  }
164  }
165  return activity;
166 }

171 size_t matchToContext(void const* data,
172  std::vector<DataDescriptorMatcher> const& matchers,
173  std::vector<size_t> const& index,
174  VariableContext& context)
175 {
176  for (size_t ri = 0, re = index.size(); ri < re; ++ri) {
177  auto& matcher = matchers[index[ri]];
178 
179  if (matcher.match(reinterpret_cast<char const*>(data), context)) {
180  context.commit();
181  return ri;
182  }
183  context.discard();
184  }
185  return INVALID_INPUT;
186 }

210 DataRelayer::RelayChoice
211  DataRelayer::relay(void const* rawHeader,
212  std::unique_ptr<FairMQMessage>* messages,
213  size_t nMessages,
214  size_t nPayloads)
215 {
216  std::scoped_lock<LockableBase(std::recursive_mutex)> lock(mMutex);
217  DataProcessingHeader const* dph = o2::header::get<DataProcessingHeader*>(rawHeader);
218  // STATE HOLDING VARIABLES
219  // This is the class level state of the relaying. If we start supporting
220  // multithreading this will have to be made thread safe before we can invoke
221  // relay concurrently.
222  auto& index = mTimesliceIndex;
223 
224  auto& cache = mCache;
225  auto const& readonlyCache = mCache;
226  auto& metrics = mMetrics;
227  auto numInputTypes = mDistinctRoutesIndex.size();
228 
229  // IMPLEMENTATION DETAILS
230  //
231  // This returns true if a given slot is available for the current number of lanes
232  auto isSlotInLane = [currentLane = dph->startTime, maxLanes = mMaxLanes](TimesliceSlot slot) {
233  return (slot.index % maxLanes) == (currentLane % maxLanes);
234  };
235  // This returns the identifier for the given input. We use a separate
236  // function because while it's trivial now, the actual matchmaking will
237  // become more complicated when we will start supporting ranges.
238  auto getInputTimeslice = [&matchers = mInputMatchers,
239  &distinctRoutes = mDistinctRoutesIndex,
240  &rawHeader,
241  &index](VariableContext& context)
242  -> std::tuple<int, TimesliceId> {
245  auto input = matchToContext(rawHeader, matchers, distinctRoutes, context);
246 
247  if (input == INVALID_INPUT) {
248  return {
249  INVALID_INPUT,
250  TimesliceId{TimesliceId::INVALID},
251  };
252  }
255  if (auto pval = std::get_if<uint64_t>(&context.get(0))) {
256  TimesliceId timeslice{*pval};
257  return {input, timeslice};
258  }
259  // If we get here it means we need to push something out of the cache.
260  return {
261  INVALID_INPUT,
262  TimesliceId{TimesliceId::INVALID},
263  };
264  };
265 
266  // We need to prune the cache from the old stuff, if any. Otherwise we
267  // simply store the payload in the cache and we mark relevant bit in the
268  // hence the first if.
269  auto pruneCache = [&cache,
270  &cachedStateMetrics = mCachedStateMetrics,
271  &numInputTypes,
272  &index,
273  &metrics](TimesliceSlot slot) {
274  assert(cache.empty() == false);
275  assert(index.size() * numInputTypes == cache.size());
276  // Prune old stuff from the cache, hopefully deleting it...
277  // We set the current slot to the timeslice value, so that old stuff
278  // will be ignored.
279  assert(numInputTypes * slot.index < cache.size());
280  for (size_t ai = slot.index * numInputTypes, ae = ai + numInputTypes; ai != ae; ++ai) {
281  cache[ai].clear();
282  cachedStateMetrics[ai] = CacheEntryStatus::EMPTY;
283  }
284  };
285 
286  // Actually save the header / payload in the slot
287  auto saveInSlot = [&cachedStateMetrics = mCachedStateMetrics,
288  &messages,
289  &nMessages,
290  &nPayloads,
291  &cache,
292  &numInputTypes,
293  &metrics](TimesliceId timeslice, int input, TimesliceSlot slot) {
294  auto cacheIdx = numInputTypes * slot.index + input;
295  MessageSet& target = cache[cacheIdx];
296  cachedStateMetrics[cacheIdx] = CacheEntryStatus::PENDING;
297  // TODO: make sure that multiple parts can only be added within the same call of
298  // DataRelayer::relay
299  assert(nPayloads > 0);
300  for (size_t mi = 0; mi < nMessages; ++mi) {
301  assert(mi + nPayloads < nMessages);
302  target.add([&messages, &mi](size_t i) -> FairMQMessagePtr& { return messages[mi + i]; }, nPayloads + 1);
303  mi += nPayloads;
304  }
305  };
306 
307  auto updateStatistics = [& stats = mStats](TimesliceIndex::ActionTaken action) {
308  // Update statistics for what happened
309  switch (action) {
310  case TimesliceIndex::ActionTaken::DropObsolete:
311  stats.droppedIncomingMessages++;
312  break;
313  case TimesliceIndex::ActionTaken::DropInvalid:
314  stats.malformedInputs++;
315  stats.droppedIncomingMessages++;
316  break;
317  case TimesliceIndex::ActionTaken::ReplaceUnused:
318  stats.relayedMessages++;
319  break;
320  case TimesliceIndex::ActionTaken::ReplaceObsolete:
321  stats.droppedComputations++;
322  stats.relayedMessages++;
323  break;
324  case TimesliceIndex::ActionTaken::Wait:
325  break;
326  }
327  };
328 
329  // OUTER LOOP
330  //
331  // This is the actual outer loop processing input as part of a given
332  // timeslice. All the other implementation details are hidden by the lambdas
333  auto input = INVALID_INPUT;
334  auto timeslice = TimesliceId{TimesliceId::INVALID};
335  auto slot = TimesliceSlot{TimesliceSlot::INVALID};
336 
337  bool needsCleaning = false;
338  // First look for matching slots which already have some
339  // partial match.
340  for (size_t ci = 0; ci < index.size(); ++ci) {
341  slot = TimesliceSlot{ci};
342  if (!isSlotInLane(slot)) {
343  continue;
344  }
345  if (index.isValid(slot) == false) {
346  continue;
347  }
348  std::tie(input, timeslice) = getInputTimeslice(index.getVariablesForSlot(slot));
349  if (input != INVALID_INPUT) {
350  break;
351  }
352  }
353 
354  // If we did not find anything, look for slots which
355  // are invalid.
356  if (input == INVALID_INPUT) {
357  for (size_t ci = 0; ci < index.size(); ++ci) {
358  slot = TimesliceSlot{ci};
359  if (index.isValid(slot) == true) {
360  continue;
361  }
362  if (!isSlotInLane(slot)) {
363  continue;
364  }
365  std::tie(input, timeslice) = getInputTimeslice(index.getVariablesForSlot(slot));
366  if (input != INVALID_INPUT) {
367  needsCleaning = true;
368  break;
369  }
370  }
371  }
372 
374  if (input != INVALID_INPUT && TimesliceId::isValid(timeslice) && TimesliceSlot::isValid(slot)) {
375  O2_SIGNPOST(O2_PROBE_DATARELAYER, timeslice.value, 0, 0, 0);
376  if (needsCleaning) {
377  pruneCache(slot);
378  }
379  saveInSlot(timeslice, input, slot);
380  index.publishSlot(slot);
381  index.markAsDirty(slot, true);
382  mStats.relayedMessages++;
383  return WillRelay;
384  }
385 
388  VariableContext pristineContext;
389  std::tie(input, timeslice) = getInputTimeslice(pristineContext);
390 
391  auto DataHeaderInfo = [&rawHeader]() {
392  std::string error;
393  // extract header from message model
394  const auto* dh = o2::header::get<o2::header::DataHeader*>(rawHeader);
395  if (dh) {
396  error += fmt::format("{}/{}/{}", dh->dataOrigin, dh->dataDescription, dh->subSpecification);
397  } else {
398  error += "invalid header";
399  }
400  return error;
401  };
402 
403  if (input == INVALID_INPUT) {
404  LOG(error) << "Could not match incoming data to any input route: " << DataHeaderInfo();
405  mStats.malformedInputs++;
406  mStats.droppedIncomingMessages++;
407  for (size_t pi = 0; pi < nMessages; ++pi) {
408  messages[pi].reset(nullptr);
409  }
410  return Invalid;
411  }
412 
413  if (TimesliceId::isValid(timeslice) == false) {
414  LOG(error) << "Could not determine the timeslice for input: " << DataHeaderInfo();
415  mStats.malformedInputs++;
416  mStats.droppedIncomingMessages++;
417  for (size_t pi = 0; pi < nMessages; ++pi) {
418  messages[pi].reset(nullptr);
419  }
420  return Invalid;
421  }
422 
423  TimesliceIndex::ActionTaken action;
424  std::tie(action, slot) = index.replaceLRUWith(pristineContext, timeslice);
425 
426  updateStatistics(action);
427 
428  switch (action) {
429  case TimesliceIndex::ActionTaken::Wait:
430  return Backpressured;
431  case TimesliceIndex::ActionTaken::DropObsolete:
432  static std::atomic<size_t> obsoleteCount = 0;
433  static std::atomic<size_t> mult = 1;
434  if ((obsoleteCount++ % (1 * mult)) == 0) {
435  LOGP(warning, "Over {} incoming messages are already obsolete, not relaying.", obsoleteCount);
436  if (obsoleteCount > mult * 10) {
437  mult = mult * 10;
438  }
439  }
440  return Dropped;
441  case TimesliceIndex::ActionTaken::DropInvalid:
442  LOG(warning) << "Incoming data is invalid, not relaying.";
443  mStats.malformedInputs++;
444  mStats.droppedIncomingMessages++;
445  for (size_t pi = 0; pi < nMessages; ++pi) {
446  messages[pi].reset(nullptr);
447  }
448  return Invalid;
449  case TimesliceIndex::ActionTaken::ReplaceUnused:
450  case TimesliceIndex::ActionTaken::ReplaceObsolete:
451  // At this point the variables match the new input but the
452  // cache still holds the old data, so we prune it.
453  pruneCache(slot);
454  saveInSlot(timeslice, input, slot);
455  index.publishSlot(slot);
456  index.markAsDirty(slot, true);
457  return WillRelay;
458  }
459  O2_BUILTIN_UNREACHABLE();
460 }

462 void DataRelayer::getReadyToProcess(std::vector<DataRelayer::RecordAction>& completed)
463 {
464  std::scoped_lock<LockableBase(std::recursive_mutex)> lock(mMutex);
465 
466  // THE STATE
467  const auto& cache = mCache;
468  const auto numInputTypes = mDistinctRoutesIndex.size();
469  //
470  // THE IMPLEMENTATION DETAILS
471  //
472  // We use this to bail out early from the check as soon as we find something
473  // which we know is not complete.
474  auto getPartialRecord = [&cache, &numInputTypes](int li) -> gsl::span<MessageSet const> {
475  auto offset = li * numInputTypes;
476  assert(cache.size() >= offset + numInputTypes);
477  auto const start = cache.data() + offset;
478  auto const end = cache.data() + offset + numInputTypes;
479  return gsl::span<MessageSet const>(start, end);
480  };
481 
482  // These two are trivial, but in principle the whole loop could be parallelised
483  // or vectorised so "completed" could be a thread local variable which needs
484  // merging at the end.
485  auto updateCompletionResults = [&completed](TimesliceSlot li, CompletionPolicy::CompletionOp op) {
486  completed.emplace_back(RecordAction{li, op});
487  };
488 
489  // THE OUTER LOOP
490  //
491  // To determine if a line is complete, we iterate on all the arguments
492  // and check if they are ready. We do it this way, because in the end
493  // the number of inputs is going to be small and having a more complex
494  // structure will probably result in a larger footprint in any case.
495  // Also notice that ai == inputsNumber only when we reach the end of the
496  // iteration, that means we have found all the required bits.
497  //
498  // Notice that the only time numInputTypes is 0 is when we are a dummy
499  // device created as a source for timers / conditions.
500  if (numInputTypes == 0) {
501  return;
502  }
503  size_t cacheLines = cache.size() / numInputTypes;
504  assert(cacheLines * numInputTypes == cache.size());
505 
506  for (int li = cacheLines - 1; li >= 0; --li) {
507  TimesliceSlot slot{(size_t)li};
508  // We only check the cachelines which have been updated by an incoming
509  // message.
510  if (mTimesliceIndex.isDirty(slot) == false) {
511  continue;
512  }
513  auto partial = getPartialRecord(li);
514  // TODO: get the data ref from message model
515  auto getter = [&partial](size_t idx, size_t part) {
516  if (partial[idx].size() > 0 && partial[idx].header(part).get()) {
517  auto header = partial[idx].header(part).get();
518  auto payload = partial[idx].payload(part).get();
519  return DataRef{nullptr,
520  reinterpret_cast<const char*>(header->GetData()),
521  reinterpret_cast<char const*>(payload ? payload->GetData() : nullptr),
522  payload ? payload->GetSize() : 0};
523  }
524  return DataRef{};
525  };
526  auto nPartsGetter = [&partial](size_t idx) {
527  return partial[idx].size();
528  };
529  InputSpan span{getter, nPartsGetter, static_cast<size_t>(partial.size())};
530  CompletionPolicy::CompletionOp action;
531  if (mCompletionPolicy.callback) {
532  action = mCompletionPolicy.callback(span);
533  } else if (mCompletionPolicy.callbackFull) {
534  action = mCompletionPolicy.callbackFull(span, mInputs);
535  } else {
536  throw std::runtime_error("No completion policy found");
537  }
538  switch (action) {
539  case CompletionPolicy::CompletionOp::Consume:
540  case CompletionPolicy::CompletionOp::ConsumeExisting:
541  case CompletionPolicy::CompletionOp::Process:
542  case CompletionPolicy::CompletionOp::Discard:
543  updateCompletionResults(slot, action);
544  break;
545  case CompletionPolicy::CompletionOp::Wait:
546  break;
547  }
548  // Given we have created an action for this cacheline, we need to wait for
549  // a new message before we look again into the given cacheline.
550  mTimesliceIndex.markAsDirty(slot, false);
551  }
552 }

554 void DataRelayer::updateCacheStatus(TimesliceSlot slot, CacheEntryStatus oldStatus, CacheEntryStatus newStatus)
555 {
556  std::scoped_lock<LockableBase(std::recursive_mutex)> lock(mMutex);
557  const auto numInputTypes = mDistinctRoutesIndex.size();
558  auto& index = mTimesliceIndex;

560  auto markInputDone = [&cachedStateMetrics = mCachedStateMetrics,
561  &index, &numInputTypes](TimesliceSlot s, size_t arg, CacheEntryStatus oldStatus, CacheEntryStatus newStatus) {
562  auto cacheId = s.index * numInputTypes + arg;
563  if (cachedStateMetrics[cacheId] == oldStatus) {

566  };

569  markInputDone(slot, ai, oldStatus, newStatus);
570  }

573 std::vector<o2::framework::MessageSet> DataRelayer::consumeAllInputsForTimeslice(TimesliceSlot slot)
574 {
575  std::scoped_lock<LockableBase(std::recursive_mutex)> lock(mMutex);
576 
577  const auto numInputTypes = mDistinctRoutesIndex.size();
578  // State of the computation
579  std::vector<MessageSet> messages(numInputTypes);
580  auto& cache = mCache;
581  auto& index = mTimesliceIndex;
582  auto& metrics = mMetrics;
583 
584  // Nothing to see here, this is just to make the outer loop more understandable.
585  auto jumpToCacheEntryAssociatedWith = [](TimesliceSlot) {
586  return;
587  };
588 
589  // We move ownership so that the cache can be reused once the computation is
590  // finished. We mark the given cache slot invalid, so that it can be reused
591  // This means we can still handle old messages if there is still space in the
592  // cache where to put them.
593  auto moveHeaderPayloadToOutput = [&messages,
594  &cachedStateMetrics = mCachedStateMetrics,
595  &cache, &index, &numInputTypes, &metrics](TimesliceSlot s, size_t arg) {
596  auto cacheId = s.index * numInputTypes + arg;
597  cachedStateMetrics[cacheId] = CacheEntryStatus::RUNNING;
598  // TODO: in the original implementation of the cache, there have been only two messages per entry,
599  // check if the 2 above corresponds to the number of messages.
600  if (cache[cacheId].size() > 0) {
601  messages[arg] = std::move(cache[cacheId]);
602  }
603  index.markAsInvalid(s);
604  };
605 
606  // An invalid set of arguments is a set of arguments associated to an invalid
607  // timeslice, so I can simply do that. I keep the assertion there because in principle
608  // we should have dispatched the timeslice already!
609  // FIXME: what happens when we have enough timeslices to hit the invalid one?
610  auto invalidateCacheFor = [&numInputTypes, &cachedStateMetrics = mCachedStateMetrics, &index, &cache](TimesliceSlot s) {
611  for (size_t ai = s.index * numInputTypes, ae = ai + numInputTypes; ai != ae; ++ai) {
612  assert(std::accumulate(cache[ai].messages.begin(), cache[ai].messages.end(), true, [](bool result, auto const& element) { return result && element.get() == nullptr; }));
613  cache[ai].clear();
614  }
615  index.markAsInvalid(s);
616  };
617 
618  // Outer loop here.
619  jumpToCacheEntryAssociatedWith(slot);
620  for (size_t ai = 0, ae = numInputTypes; ai != ae; ++ai) {
621  moveHeaderPayloadToOutput(slot, ai);
622  }
623  invalidateCacheFor(slot);
624 
625  return std::move(messages);
626 }

628 std::vector<o2::framework::MessageSet> DataRelayer::consumeExistingInputsForTimeslice(TimesliceSlot slot)
629 {
630  std::scoped_lock<LockableBase(std::recursive_mutex)> lock(mMutex);
631 
632  const auto numInputTypes = mDistinctRoutesIndex.size();
633  // State of the computation
634  std::vector<MessageSet> messages(numInputTypes);
635  auto& cache = mCache;
636  auto& index = mTimesliceIndex;
637  auto& metrics = mMetrics;
638 
639  // Nothing to see here, this is just to make the outer loop more understandable.
640  auto jumpToCacheEntryAssociatedWith = [](TimesliceSlot) {
641  return;
642  };
643 
644  // We move ownership so that the cache can be reused once the computation is
645  // finished. We mark the given cache slot invalid, so that it can be reused
646  // This means we can still handle old messages if there is still space in the
647  // cache where to put them.
648  auto copyHeaderPayloadToOutput = [&messages,
649  &cachedStateMetrics = mCachedStateMetrics,
650  &cache, &index, &numInputTypes, &metrics](TimesliceSlot s, size_t arg) {
651  auto cacheId = s.index * numInputTypes + arg;
652  cachedStateMetrics[cacheId] = CacheEntryStatus::RUNNING;
653  // TODO: in the original implementation of the cache, there have been only two messages per entry,
654  // check if the 2 above corresponds to the number of messages.
655  for (size_t pi = 0; pi < cache[cacheId].size(); pi++) {
656  auto& header = cache[cacheId].header(pi);
657  auto&& newHeader = header->GetTransport()->CreateMessage();
658  newHeader->Copy(*header);
659  messages[arg].add(PartRef{std::move(newHeader), std::move(cache[cacheId].payload(pi))});
660  }
661  };
662 
663  // Outer loop here.
664  jumpToCacheEntryAssociatedWith(slot);
665  for (size_t ai = 0, ae = numInputTypes; ai != ae; ++ai) {
666  copyHeaderPayloadToOutput(slot, ai);
667  }
668 
669  return std::move(messages);
670 }
