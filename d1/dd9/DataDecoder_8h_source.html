62  struct SampaInfo {
63  union {
64  uint32_t id = 0;
65  struct {
66  uint32_t chip : 1;

68  uint32_t solar : 16;
69  uint32_t unused : 9;

71  };

73  union {
74  // default value
75  uint64_t time = 0x0000000000000000;
76  struct {
77  uint32_t sampaTime : 10;
78  uint32_t bunchCrossing : 20;
79  uint32_t reserved : 2;
80  uint32_t orbit;
81  };
82  };
83  uint32_t getBXTime() const
84  {
85  return (bunchCrossing + (sampaTime * 4));
86  }
87  int32_t tfTime;
88  bool timeValid() const
89  {
90  return (tfTime != tfTimeInvalid);
91  }
92 
93  bool operator==(const SampaInfo&) const;
94  bool operator<(const SampaInfo& rhs) const
95  {
96  if (id < rhs.id) {
97  return true;
98  } else if (time < rhs.time) {
99  return true;
100  }
101  return false;
102  }
103  };
