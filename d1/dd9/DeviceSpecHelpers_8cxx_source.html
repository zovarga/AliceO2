62 {
63  // We simply wake up the event loop. Nothing to be done here.

341 void DeviceSpecHelpers::processOutEdgeActions(std::vector<DeviceSpec>& devices,
342  std::vector<DeviceId>& deviceIndex,
343  std::vector<DeviceConnectionId>& connections,
344  ResourceManager& resourceManager,
345  const std::vector<size_t>& outEdgeIndex,
346  const std::vector<DeviceConnectionEdge>& logicalEdges,
347  const std::vector<EdgeAction>& actions, const WorkflowSpec& workflow,
348  const std::vector<OutputSpec>& outputsMatchers,
349  const std::vector<ChannelConfigurationPolicy>& channelPolicies,
350  std::string const& channelPrefix,
351  ComputingOffer const& defaultOffer)
352 {
353  // The topology cannot be empty or not connected. If that is the case, than
354  // something before this went wrong.
355  // FIXME: is that really true???
356  assert(!workflow.empty());
357 
358  // Edges are navigated in order for each device, so the device associaited to
359  // an edge is always the last one created.
360  auto deviceForEdge = [&actions, &workflow, &devices,
361  &logicalEdges, &resourceManager,
362  &defaultOffer, &channelPrefix](size_t ei, ComputingOffer& acceptedOffer) {
363  auto& edge = logicalEdges[ei];
364  auto& action = actions[ei];
365 
366  if (action.requiresNewDevice == false) {
367  assert(devices.empty() == false);
368  return devices.size() - 1;
369  }
370  if (acceptedOffer.hostname != "") {
371  resourceManager.notifyAcceptedOffer(acceptedOffer);
372  }
373 
374  auto processor = workflow[edge.producer];
375 
376  acceptedOffer.cpu = defaultOffer.cpu;
377  acceptedOffer.memory = defaultOffer.memory;
378  for (auto offer : resourceManager.getAvailableOffers()) {
379  if (offer.cpu < acceptedOffer.cpu) {
380  continue;
381  }
382  if (offer.memory < acceptedOffer.memory) {
383  continue;
384  }
385  acceptedOffer.hostname = offer.hostname;
386  acceptedOffer.startPort = offer.startPort;
387  acceptedOffer.rangeSize = 0;
388  break;
389  }
390 
391  DeviceSpec device;
392  device.name = processor.name;
393  device.id = processor.name;
394  device.channelPrefix = channelPrefix;
395  if (processor.maxInputTimeslices != 1) {
396  device.id = processor.name + "_t" + std::to_string(edge.producerTimeIndex);
397  }
398  device.algorithm = processor.algorithm;
399  device.services = processor.requiredServices;
400  device.options = processor.options;
401  device.rank = processor.rank;
402  device.nSlots = processor.nSlots;
403  device.inputTimesliceId = edge.producerTimeIndex;
404  device.maxInputTimeslices = processor.maxInputTimeslices;
405  device.resource = {acceptedOffer};
406  device.labels = processor.labels;
407  devices.push_back(device);
408  return devices.size() - 1;
409  };

411  auto channelFromDeviceEdgeAndPort = [&connections, &workflow, &channelPolicies](const DeviceSpec& device,
412  ComputingResource& deviceResource,
413  ComputingOffer& acceptedOffer,
414  const DeviceConnectionEdge& edge) {
415  OutputChannelSpec channel;
416  auto& consumer = workflow[edge.consumer];
417  std::string consumerDeviceId = consumer.name;
418  if (consumer.maxInputTimeslices != 1) {
419  consumerDeviceId += "_t" + std::to_string(edge.timeIndex);
420  }
421  channel.name = device.channelPrefix + "from_" + device.id + "_to_" + consumerDeviceId;
422  channel.port = acceptedOffer.startPort + acceptedOffer.rangeSize;
423  channel.hostname = acceptedOffer.hostname;
424  deviceResource.usedPorts += 1;
425  acceptedOffer.rangeSize += 1;
426 
427  for (auto& policy : channelPolicies) {
428  if (policy.match(device.id, consumerDeviceId)) {
429  policy.modifyOutput(channel);
430  break;
431  }
432  }
433  DeviceConnectionId id{edge.producer, edge.consumer, edge.timeIndex, edge.producerTimeIndex, channel.port};
434  connections.push_back(id);
435  return std::move(channel);
436  };
437 
438  auto isDifferentDestinationDeviceReferredBy = [&actions](size_t ei) { return actions[ei].requiresNewChannel; };
439 
440  // This creates a new channel for a given edge, if needed. Notice that we
441  // navigate edges in a per device fashion (creating those if they are not
442  // alredy there) and create a new channel only if it connects two new
443  // devices. Whether or not this is the case was previously computed
444  // in the action.requiresNewChannel field.
445  auto createChannelForDeviceEdge = [&devices, &logicalEdges, &channelFromDeviceEdgeAndPort,
446  &outputsMatchers, &deviceIndex,
447  &workflow](size_t di, size_t ei, ComputingOffer& offer) {
448  auto& device = devices[di];
449  auto& edge = logicalEdges[ei];
450 
451  deviceIndex.emplace_back(DeviceId{edge.producer, edge.producerTimeIndex, di});
452 
453  OutputChannelSpec channel = channelFromDeviceEdgeAndPort(device, device.resource, offer, edge);
454 
455  device.outputChannels.push_back(channel);
456  return device.outputChannels.size() - 1;
457  };
458 
459  // Notice how we need to behave in two different ways depending
460  // whether this is a real OutputRoute or if it's a forward from
461  // a previous consumer device.
462  // FIXME: where do I find the InputSpec for the forward?
463  auto appendOutputRouteToSourceDeviceChannel = [&outputsMatchers, &workflow, &devices, &logicalEdges](
464  size_t ei, size_t di, size_t ci) {
465  assert(ei < logicalEdges.size());
466  assert(di < devices.size());
467  assert(ci < devices[di].outputChannels.size());
468  auto& edge = logicalEdges[ei];
469  auto& device = devices[di];
470  assert(edge.consumer < workflow.size());
471  auto& consumer = workflow[edge.consumer];
472  auto& channel = devices[di].outputChannels[ci];
473  assert(edge.outputGlobalIndex < outputsMatchers.size());
474 
475  if (edge.isForward == false) {
476  OutputRoute route{
477  edge.timeIndex,
478  consumer.maxInputTimeslices,
479  outputsMatchers[edge.outputGlobalIndex],
480  channel.name};
481  device.outputs.emplace_back(route);
482  } else {
483  ForwardRoute route{
484  edge.timeIndex,
485  consumer.maxInputTimeslices,
486  workflow[edge.consumer].inputs[edge.consumerInputIndex],
487  channel.name};
488  device.forwards.emplace_back(route);
489  }
490  };
491 
492  auto sortDeviceIndex = [&deviceIndex]() { std::sort(deviceIndex.begin(), deviceIndex.end()); };
493 
494  auto lastChannelFor = [&devices](size_t di) {
495  assert(di < devices.size());
496  assert(devices[di].outputChannels.empty() == false);
497  return devices[di].outputChannels.size() - 1;
498  };
499 
500  //
501  // OUTER LOOP
502  //
503  // We need to create all the channels going out of a device, and associate
504  // routes to them for this reason
505  // we iterate over all the edges (which are per-datatype exchanged) and
506  // whenever we need to connect to a new device we create the channel. `device`
507  // here refers to the source device. This loop will therefore not create the
508  // devices which acts as sink, which are done in the preocessInEdgeActions
509  // function.
510  ComputingOffer acceptedOffer;
511  for (auto edge : outEdgeIndex) {
512  auto device = deviceForEdge(edge, acceptedOffer);
513  size_t channel = -1;
514  if (isDifferentDestinationDeviceReferredBy(edge)) {
515  channel = createChannelForDeviceEdge(device, edge, acceptedOffer);
516  } else {
517  channel = lastChannelFor(device);
518  }
519  appendOutputRouteToSourceDeviceChannel(edge, device, channel);
520  }
521  if (std::string(acceptedOffer.hostname) != "") {
522  resourceManager.notifyAcceptedOffer(acceptedOffer);
523  }
524  sortDeviceIndex();
525 }

527 void DeviceSpecHelpers::processInEdgeActions(std::vector<DeviceSpec>& devices,
528  std::vector<DeviceId>& deviceIndex,
529  const std::vector<DeviceConnectionId>& connections,
530  ResourceManager& resourceManager,
531  const std::vector<size_t>& inEdgeIndex,
532  const std::vector<DeviceConnectionEdge>& logicalEdges,
533  const std::vector<EdgeAction>& actions, const WorkflowSpec& workflow,
534  std::vector<LogicalForwardInfo> const& availableForwardsInfo,
535  std::vector<ChannelConfigurationPolicy> const& channelPolicies,
536  std::string const& channelPrefix,
537  ComputingOffer const& defaultOffer)
538 {
539  auto const& constDeviceIndex = deviceIndex;
540 
541  auto findProducerForEdge = [&logicalEdges, &constDeviceIndex](size_t ei) {
542  auto& edge = logicalEdges[ei];
543 
544  DeviceId pid{edge.producer, edge.producerTimeIndex, 0};
545  auto deviceIt = std::lower_bound(constDeviceIndex.cbegin(), constDeviceIndex.cend(), pid);
546  // By construction producer should always be there
547  assert(deviceIt != constDeviceIndex.end());
548  assert(deviceIt->processorIndex == pid.processorIndex && deviceIt->timeslice == pid.timeslice);
549  return deviceIt->deviceIndex;
550  };
551 
552  auto findConsumerForEdge = [&logicalEdges, &constDeviceIndex](size_t ei) {
553  auto& edge = logicalEdges[ei];
554  if (!std::is_sorted(constDeviceIndex.cbegin(), constDeviceIndex.cend())) {
555  throw o2::framework::runtime_error("Needs a sorted vector to be correct");
556  }
557 
558  DeviceId pid{edge.consumer, edge.timeIndex, 0};
559  auto deviceIt = std::lower_bound(constDeviceIndex.cbegin(), constDeviceIndex.cend(), pid);
560  // We search for a consumer only if we know it's is already there.
561  assert(deviceIt != constDeviceIndex.end());
562  assert(deviceIt->processorIndex == pid.processorIndex && deviceIt->timeslice == pid.timeslice);
563  return deviceIt->deviceIndex;
564  };
565 
566  // Notice that to start with, consumer exists only if they also are
567  // producers, so we need to create one if it does not exist. Given this is
568  // stateful, we keep an eye on what edge was last searched to make sure we
569  // are not screwing up.
570  //
571  // Notice this is not thread safe.
572  decltype(deviceIndex.begin()) lastConsumerSearch;
573  size_t lastConsumerSearchEdge;
574  auto hasConsumerForEdge = [&lastConsumerSearch, &lastConsumerSearchEdge, &deviceIndex,
575  &logicalEdges](size_t ei) -> int {
576  auto& edge = logicalEdges[ei];
577  DeviceId cid{edge.consumer, edge.timeIndex, 0};
578  lastConsumerSearchEdge = ei; // This will invalidate the cache

788  // previously.

810  } else {

814  }

816  resourceManager.notifyAcceptedOffer(acceptedOffer);
