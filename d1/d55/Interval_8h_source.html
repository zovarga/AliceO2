34  Interval(T b = {}, T e = {}) : mBegin(b), mEnd(e)

39  }

42  {
43  return (i.begin() < mBegin || areEqual(i.begin(), mBegin)) && (mEnd < i.end() || areEqual(i.end(), mEnd));
44  }
