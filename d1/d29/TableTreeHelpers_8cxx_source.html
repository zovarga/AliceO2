27 auto arrowTypeFromROOT(EDataType type, int size)
28 {
29  auto typeGenerator = [](std::shared_ptr<arrow::DataType>&& type, int size) -> std::shared_ptr<arrow::DataType> {
30  switch (size) {
31  case -1:
32  return arrow::list(type);
33  case 1:
34  return std::move(type);
35  default:
36  return arrow::fixed_size_list(type, size);
37  }
38  };
39 
40  switch (type) {
41  case EDataType::kBool_t:
42  return typeGenerator(arrow::boolean(), size);
43  case EDataType::kUChar_t:
44  return typeGenerator(arrow::uint8(), size);
45  case EDataType::kUShort_t:
46  return typeGenerator(arrow::uint16(), size);
47  case EDataType::kUInt_t:
48  return typeGenerator(arrow::uint32(), size);
49  case EDataType::kULong64_t:
50  return typeGenerator(arrow::uint64(), size);
51  case EDataType::kChar_t:
52  return typeGenerator(arrow::int8(), size);
53  case EDataType::kShort_t:
54  return typeGenerator(arrow::int16(), size);
55  case EDataType::kInt_t:
56  return typeGenerator(arrow::int32(), size);
57  case EDataType::kLong64_t:
58  return typeGenerator(arrow::int64(), size);
59  case EDataType::kFloat_t:
60  return typeGenerator(arrow::float32(), size);
61  case EDataType::kDouble_t:
62  return typeGenerator(arrow::float64(), size);
63  default:
64  throw runtime_error_f("Unsupported branch type: %d", static_cast<int>(type));
65  }
66 }

165 void swapCopy(unsigned char* dest, char* source, int size, int typeSize) noexcept
166 {
167  switch (typeSize) {
168  case 1:
169  return (void)std::memcpy(dest, source, size);
170  case 2:
171  return doSwapCopy_<uint16_t>(dest, source, size);
172  case 4:
173  return doSwapCopy_<uint32_t>(dest, source, size);
174  case 8:
175  return doSwapCopy_<uint64_t>(dest, source, size);

179 std::pair<std::shared_ptr<arrow::ChunkedArray>, std::shared_ptr<arrow::Field>> BranchToColumn::read(TBuffer* buffer)
180 {
181  auto totalEntries = mBranch->GetEntries();
182  arrow::Status status;
183  int readEntries = 0;
184  buffer->Reset();
185  std::shared_ptr<arrow::Array> array;
186 
187  if (mType == EDataType::kBool_t) {
188  // boolean array special case: we need to use builder to create the bitmap
189  status = mValueBuilder->Reserve(totalEntries * mListSize);
190  if (mListSize > 1) {
191  status &= mListBuilder->Reserve(totalEntries);
192  }
193  if (!status.ok()) {
194  throw runtime_error("Failed to reserve memory for array builder");
195  }
196  while (readEntries < totalEntries) {
197  auto readLast = mBranch->GetBulkRead().GetBulkEntries(readEntries, *buffer);
198  readEntries += readLast;
199  status &= static_cast<arrow::BooleanBuilder*>(mValueBuilder)->AppendValues(reinterpret_cast<uint8_t const*>(buffer->GetCurrent()), readLast * mListSize);
200  }
201  if (mListSize > 1) {
202  status &= static_cast<arrow::FixedSizeListBuilder*>(mListBuilder.get())->AppendValues(readEntries);
203  }
204  if (!status.ok()) {
205  throw runtime_error("Failed to append values to array");
206  }
207  if (mListSize > 1) {
208  status &= mListBuilder->Finish(&array);
209  } else {
210  status &= mValueBuilder->Finish(&array);
211  }
212  if (!status.ok()) {
213  throw runtime_error("Failed to create array");
214  }
215  } else {
216  // other types: use serialized read to build arrays directly
217  auto&& result = arrow::AllocateResizableBuffer(mBranch->GetTotBytes(), mPool);
218  if (!result.ok()) {
219  throw runtime_error("Cannot allocate values buffer");
220  }
221  std::shared_ptr<arrow::Buffer> arrowValuesBuffer = std::move(result).ValueUnsafe();
222  auto ptr = arrowValuesBuffer->mutable_data();
223  if (ptr == nullptr) {
224  throw runtime_error("Invalid buffer");
225  }
226 
227  auto typeSize = TDataType::GetDataType(mType)->Size();
228  std::unique_ptr<TBufferFile> offsetBuffer = nullptr;
229 
230  uint32_t offset = 0;
231  int count = 0;
232  std::shared_ptr<arrow::Buffer> arrowOffsetBuffer;
233  gsl::span<int> offsets;
234  int size = 0;
235  uint32_t totalSize = 0;
236  TBranch* mSizeBranch = nullptr;
237  if (mVLA) {
238  mSizeBranch = mBranch->GetTree()->GetBranch((std::string{mBranch->GetName()} + TableTreeHelpers::sizeBranchSuffix).c_str());
239  offsetBuffer = std::make_unique<TBufferFile>(TBuffer::EMode::kWrite, 4 * 1024 * 1024);
240  result = arrow::AllocateResizableBuffer((totalEntries + 1) * (int64_t)sizeof(int), mPool);
241  if (!result.ok()) {
242  throw runtime_error("Cannot allocate offset buffer");
243  }
244  arrowOffsetBuffer = std::move(result).ValueUnsafe();
245  unsigned char* ptrOffset = arrowOffsetBuffer->mutable_data();
246  auto* tPtrOffset = reinterpret_cast<int*>(ptrOffset);
247  offsets = gsl::span<int>{tPtrOffset, tPtrOffset + totalEntries + 1};
248 
249  // read sizes first
250  while (readEntries < totalEntries) {
251  auto readLast = mSizeBranch->GetBulkRead().GetEntriesSerialized(readEntries, *offsetBuffer);
252  readEntries += readLast;
253  for (auto i = 0; i < readLast; ++i) {
254  offsets[count++] = (int)offset;
255  offset += swap32_(reinterpret_cast<uint32_t*>(offsetBuffer->GetCurrent())[i]);
256  }
257  }
258  offsets[count] = (int)offset;
259  totalSize = offset;
260  readEntries = 0;
261  }
262 
263  while (readEntries < totalEntries) {
264  auto readLast = mBranch->GetBulkRead().GetEntriesSerialized(readEntries, *buffer);
265  if (mVLA) {
266  size = offsets[readEntries + readLast] - offsets[readEntries];
267  } else {
268  size = readLast * mListSize;
269  }
270  readEntries += readLast;
271  swapCopy(ptr, buffer->GetCurrent(), size, typeSize);
272  ptr += (ptrdiff_t)(size * typeSize);
273  }
274  if (!mVLA) {
275  totalSize = readEntries * mListSize;
276  }
277  std::shared_ptr<arrow::PrimitiveArray> varray;
278  switch (mListSize) {
279  case -1:
280  varray = std::make_shared<arrow::PrimitiveArray>(mArrowType->field(0)->type(), totalSize, arrowValuesBuffer);
281  array = std::make_shared<arrow::ListArray>(mArrowType, readEntries, arrowOffsetBuffer, varray);
282  break;
283  case 1:
284  array = std::make_shared<arrow::PrimitiveArray>(mArrowType, readEntries, arrowValuesBuffer);
285  break;
286  default:
287  varray = std::make_shared<arrow::PrimitiveArray>(mArrowType->field(0)->type(), totalSize, arrowValuesBuffer);
288  array = std::make_shared<arrow::FixedSizeListArray>(mArrowType, readEntries, varray);
289  }
290  }
291 
292  auto fullArray = std::make_shared<arrow::ChunkedArray>(array);
293  auto field = std::make_shared<arrow::Field>(mBranch->GetName(), mArrowType);
294 
295  mBranch->SetStatus(false);
296  mBranch->DropBaskets("all");
297  mBranch->Reset();
298  mBranch->GetTransientBuffer(0)->Expand(0);
299 
300  return std::make_pair(fullArray, field);
301 }

303 ColumnToBranch::ColumnToBranch(TTree* tree, std::shared_ptr<arrow::ChunkedArray> const& column, std::shared_ptr<arrow::Field> const& field)
304  : mBranchName{field->name()},
305  mColumn{column.get()}
306 {
307  std::string leafList;
308  std::string sizeLeafList;
309  auto arrowType = field->type();
310  mFieldType = arrowType->id();
311  switch (mFieldType) {
312  case arrow::Type::FIXED_SIZE_LIST:
313  mListSize = std::static_pointer_cast<arrow::FixedSizeListType>(arrowType)->list_size();
314  arrowType = arrowType->field(0)->type();
315  mElementType = basicROOTTypeFromArrow(arrowType->id());
316  leafList = mBranchName + "[" + std::to_string(mListSize) + "]" + mElementType.suffix;
317  break;
318  case arrow::Type::LIST:
319  arrowType = arrowType->field(0)->type();
320  mElementType = basicROOTTypeFromArrow(arrowType->id());
321  leafList = mBranchName + "[" + mBranchName + TableTreeHelpers::sizeBranchSuffix + "]" + mElementType.suffix;
322  sizeLeafList = mBranchName + TableTreeHelpers::sizeBranchSuffix + "/I";
323  break;
324  default:
325  mElementType = basicROOTTypeFromArrow(arrowType->id());
326  leafList = mBranchName + mElementType.suffix;
327  break;
328  }
329  if (!sizeLeafList.empty()) {
330  mSizeBranch = tree->GetBranch((mBranchName + TableTreeHelpers::sizeBranchSuffix).c_str());
331  if (mSizeBranch == nullptr) {
332  mSizeBranch = tree->Branch((mBranchName + TableTreeHelpers::sizeBranchSuffix).c_str(), (char*)nullptr, sizeLeafList.c_str());
333  }
334  }
335  mBranch = tree->GetBranch(mBranchName.c_str());
336  if (mBranch == nullptr) {
337  mBranch = tree->Branch(mBranchName.c_str(), (char*)nullptr, leafList.c_str());
338  }
339  if (mElementType.type == EDataType::kBool_t) {
340  cache.resize(mListSize);
341  }
342  accessChunk();
343 }

345 void ColumnToBranch::at(const int64_t* pos)
346 {
347  if (O2_BUILTIN_UNLIKELY(*pos - mFirstIndex >= mChunkLength)) {
348  nextChunk();
349  }
350  if (mElementType.type == EDataType::kBool_t) {
351  auto boolArray = std::static_pointer_cast<arrow::BooleanArray>(mCurrentArray);
352  for (auto i = 0; i < mListSize; ++i) {
353  cache[i] = boolArray->Value((*pos - mFirstIndex) * mListSize + i);
354  }
355  mBranch->SetAddress((void*)(cache.data()));
356  return;
357  }
358  uint8_t const* buffer;
359  switch (mFieldType) {
360  case arrow::Type::LIST: {
361  auto list = std::static_pointer_cast<arrow::ListArray>(mCurrentArray);
362  mListSize = list->value_length((*pos - mFirstIndex));
363  buffer = std::static_pointer_cast<arrow::PrimitiveArray>(list->values())->values()->data() + mCurrentArray->offset() + list->value_offset((*pos - mFirstIndex)) * mElementType.size;
364  mBranch->SetAddress((void*)buffer);
365  mSizeBranch->SetAddress(&mListSize);
366  };
367  break;
368  case arrow::Type::FIXED_SIZE_LIST:
369  default: {
370  buffer = std::static_pointer_cast<arrow::PrimitiveArray>(mCurrentArray)->values()->data() + mCurrentArray->offset() + (*pos - mFirstIndex) * mListSize * mElementType.size;
371  mBranch->SetAddress((void*)buffer);
372  };
373  }
374 }

376 void ColumnToBranch::accessChunk()
377 {
378  auto array = mColumn->chunk(mCurrentChunk);
379  switch (mFieldType) {
380  case arrow::Type::FIXED_SIZE_LIST: {
381  auto list = std::static_pointer_cast<arrow::FixedSizeListArray>(array);
382  mChunkLength = list->length();
383  mCurrentArray = list->values();
384  };
385  break;
386  case arrow::Type::LIST: {
387  auto list = std::static_pointer_cast<arrow::ListArray>(array);
388  mChunkLength = list->length();
389  mCurrentArray = list;
390  };
391  break;
392  default:
393  mCurrentArray = array;
394  mChunkLength = mCurrentArray->length();
395  }
396 }

477 void TreeToTable::addAllColumns(TTree* tree, std::vector<std::string>&& names)
478 {
479  auto branches = tree->GetListOfBranches();
480  auto n = branches->GetEntries();
481  if (n == 0) {
482  throw runtime_error("Tree has no branches");
483  }
484 
485  std::vector<BranchInfo> branchInfos;
486  for (auto i = 0; i < n; ++i) {
487  auto branch = static_cast<TBranch*>(branches->At(i));
488  auto name = std::string{branch->GetName()};
489  auto pos = name.find(TableTreeHelpers::sizeBranchSuffix);
490  if (pos != std::string::npos) {
491  name.erase(pos);
492  branchInfos.emplace_back(BranchInfo{name, (TBranch*)nullptr, true});
493  } else {
494  auto lookup = std::find_if(branchInfos.begin(), branchInfos.end(), [&](BranchInfo const& bi) {
495  return bi.name == name;
496  });
497  if (lookup == branchInfos.end()) {
498  branchInfos.emplace_back(BranchInfo{name, branch, false});
499  } else {
500  lookup->ptr = branch;
501  }
502  }
503  }
504 
505  if (names.empty()) {
506  for (auto& bi : branchInfos) {

509  } else {
