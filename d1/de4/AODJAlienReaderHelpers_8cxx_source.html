47 struct RuntimeWatchdog {
48  int numberTimeFrames;
49  uint64_t startTime;
50  uint64_t lastTime;
51  double runTime;
52  uint64_t runTimeLimit;
53 
54  RuntimeWatchdog(Long64_t limit)
55  {
56  numberTimeFrames = -1;
57  startTime = uv_hrtime();
58  lastTime = startTime;
59  runTime = 0.;
60  runTimeLimit = limit;
61  }
62 
63  bool update()
64  {
65  numberTimeFrames++;
66  if (runTimeLimit <= 0) {
67  return true;
68  }
69 
70  auto nowTime = uv_hrtime();
71 
72  // time spent to process the time frame
73  double time_spent = numberTimeFrames < 1 ? (double)(nowTime - lastTime) / 1.E9 : 0.;
74  runTime += time_spent;
75  lastTime = nowTime;
76 
77  return ((double)(lastTime - startTime) / 1.E9 + runTime / (numberTimeFrames + 1)) < runTimeLimit;
78  }
79 
80  void printOut()
81  {
82  LOGP(info, "RuntimeWatchdog");
83  LOGP(info, " run time limit: {}", runTimeLimit);
84  LOGP(info, " number of time frames: {}", numberTimeFrames);
85  LOGP(info, " estimated run time per time frame: {}", (numberTimeFrames >= 0) ? runTime / (numberTimeFrames + 1) : 0.);
86  LOGP(info, " estimated total run time: {}", (double)(lastTime - startTime) / 1.E9 + ((numberTimeFrames >= 0) ? runTime / (numberTimeFrames + 1) : 0.));
87  }
88 };

135 void AODJAlienReaderHelpers::dumpFileMetrics(Monitoring& monitoring, TFile* currentFile, uint64_t startedAt, uint64_t ioTime, int dfPerFile, int dfRead)
136 {
137  if (currentFile == nullptr) {
138  return;
139  }
140  std::string monitoringInfo(fmt::format("lfn={},size={},total_df={},read_df={},read_bytes={},read_calls={},io_time={:.1f},wait_time={:.1f}", currentFile->GetName(),
141  currentFile->GetSize(), dfPerFile, dfRead, currentFile->GetBytesRead(), currentFile->GetReadCalls(),
142  ((float)ioTime / 1e9), ((float)(uv_hrtime() - startedAt - ioTime) / 1e9)));
143 #if __has_include(<TJAlienFile.h>)
144  auto alienFile = dynamic_cast<TJAlienFile*>(currentFile);
145  if (alienFile) {
146  monitoringInfo += fmt::format(",se={},open_time={:.1f}", alienFile->GetSE(), alienFile->GetElapsed());
147  }
148 #endif
149  monitoring.send(Metric{monitoringInfo, "aod-file-read-info"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
150  LOGP(info, "Read info: {}", monitoringInfo);
151 }

153 AlgorithmSpec AODJAlienReaderHelpers::rootFileReaderCallback()
154 {
155  auto callback = AlgorithmSpec{adaptStateful([](ConfigParamRegistry const& options,
156  DeviceSpec const& spec,
157  Monitoring& monitoring) {
158  monitoring.send(Metric{(uint64_t)0, "arrow-bytes-created"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
159  monitoring.send(Metric{(uint64_t)0, "arrow-messages-created"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
160  monitoring.send(Metric{(uint64_t)0, "arrow-bytes-destroyed"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
161  monitoring.send(Metric{(uint64_t)0, "arrow-messages-destroyed"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
162  monitoring.send(Metric{(uint64_t)0, "arrow-bytes-expired"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
163  monitoring.flushBuffer();
164 
165  if (!options.isSet("aod-file")) {
166  LOGP(fatal, "No input file defined!");
167  throw std::runtime_error("Processing is stopped!");
168  }
169 
170  auto filename = options.get<std::string>("aod-file");
171 
172  // create a DataInputDirector
173  auto didir = std::make_shared<DataInputDirector>(filename);
174  if (options.isSet("aod-reader-json")) {
175  auto jsonFile = options.get<std::string>("aod-reader-json");
176  if (!didir->readJson(jsonFile)) {
177  LOGP(error, "Check the JSON document! Can not be properly parsed!");
178  }
179  }
180 
181  // get the run time watchdog
182  auto* watchdog = new RuntimeWatchdog(options.get<int64_t>("time-limit"));
183 
184  // selected the TFN input and
185  // create list of requested tables
186  header::DataHeader TFNumberHeader;
187  std::vector<OutputRoute> requestedTables;
188  std::vector<OutputRoute> routes(spec.outputs);
189  for (auto route : routes) {
190  if (DataSpecUtils::partialMatch(route.matcher, header::DataOrigin("TFN"))) {
191  auto concrete = DataSpecUtils::asConcreteDataMatcher(route.matcher);
192  TFNumberHeader = header::DataHeader(concrete.description, concrete.origin, concrete.subSpec);
193  } else {
194  requestedTables.emplace_back(route);
195  }
196  }
197 
198  auto fileCounter = std::make_shared<int>(0);
199  auto numTF = std::make_shared<int>(-1);
200  return adaptStateless([TFNumberHeader,
201  requestedTables,
202  fileCounter,
203  numTF,
204  watchdog,
205  didir](Monitoring& monitoring, DataAllocator& outputs, ControlService& control, DeviceSpec const& device) {
206  // Each parallel reader device.inputTimesliceId reads the files fileCounter*device.maxInputTimeslices+device.inputTimesliceId
207  // the TF to read is numTF
208  assert(device.inputTimesliceId < device.maxInputTimeslices);
209  uint64_t timeFrameNumber = 0;
210  int fcnt = (*fileCounter * device.maxInputTimeslices) + device.inputTimesliceId;
211  int ntf = *numTF + 1;
212  static int currentFileCounter = -1;
213  static int filesProcessed = 0;
214  if (currentFileCounter != *fileCounter) {
215  currentFileCounter = *fileCounter;
216  monitoring.send(Metric{(uint64_t)++filesProcessed, "files-opened"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
217  }
218 
219  // loop over requested tables
220  bool first = true;
221  static size_t totalSizeUncompressed = 0;
222  static size_t totalSizeCompressed = 0;
223  static TFile* currentFile = nullptr;
224  static int tfCurrentFile = -1;
225  static auto currentFileStartedAt = uv_hrtime();
226  static uint64_t currentFileIOTime = 0;
227  static uint64_t totalDFSent = 0;
228 
229  // check if RuntimeLimit is reached
230  if (!watchdog->update()) {
231  LOGP(info, "Run time exceeds run time limit of {} seconds. Exiting gracefully...", watchdog->runTimeLimit);
232  LOGP(info, "Stopping reader {} after time frame {}.", device.inputTimesliceId, watchdog->numberTimeFrames - 1);
233  dumpFileMetrics(monitoring, currentFile, currentFileStartedAt, currentFileIOTime, tfCurrentFile, ntf);
234  monitoring.flushBuffer();
235  didir->closeInputFiles();
236  control.endOfStream();
237  control.readyToQuit(QuitRequest::Me);
238  return;
239  }
240 
241  auto ioStart = uv_hrtime();
242 
243  for (auto& route : requestedTables) {
244  if ((device.inputTimesliceId % route.maxTimeslices) != route.timeslice) {
245  continue;
246  }
247 
248  // create header
249  auto concrete = DataSpecUtils::asConcreteDataMatcher(route.matcher);
250  auto dh = header::DataHeader(concrete.description, concrete.origin, concrete.subSpec);
251 
252  // create a TreeToTable object
253  TTree* tr = didir->getDataTree(dh, fcnt, ntf);
254  if (!tr) {
255  if (first) {
256  // dump metrics of file which is done for reading
257  dumpFileMetrics(monitoring, currentFile, currentFileStartedAt, currentFileIOTime, tfCurrentFile, ntf);
258  currentFile = nullptr;
259  currentFileStartedAt = uv_hrtime();
260  currentFileIOTime = 0;
261 
262  // check if there is a next file to read
263  fcnt += device.maxInputTimeslices;
264  if (didir->atEnd(fcnt)) {
265  LOGP(info, "No input files left to read for reader {}!", device.inputTimesliceId);
266  didir->closeInputFiles();
267  control.endOfStream();
268  control.readyToQuit(QuitRequest::Me);
269  return;
270  }
271  // get first folder of next file
272  ntf = 0;
273  tr = didir->getDataTree(dh, fcnt, ntf);
274  if (!tr) {
275  LOGP(fatal, "Can not retrieve tree for table {}: fileCounter {}, timeFrame {}", concrete.origin, fcnt, ntf);
276  throw std::runtime_error("Processing is stopped!");
277  }
278  } else {
279  LOGP(fatal, "Can not retrieve tree for table {}: fileCounter {}, timeFrame {}", concrete.origin, fcnt, ntf);
280  throw std::runtime_error("Processing is stopped!");
281  }
282  }
283 
284  if (first) {
285  timeFrameNumber = didir->getTimeFrameNumber(dh, fcnt, ntf);
286  auto o = Output(TFNumberHeader);
287  outputs.make<uint64_t>(o) = timeFrameNumber;
288  }
289 
290  // create table output
291  auto o = Output(dh);
292  auto& t2t = outputs.make<TreeToTable>(o);
293 
294  // add branches to read
295  // fill the table
296  auto colnames = getColumnNames(dh);
297  t2t.setLabel(tr->GetName());
298  if (colnames.size() == 0) {
299  totalSizeCompressed += tr->GetZipBytes();
300  totalSizeUncompressed += tr->GetTotBytes();
301  t2t.addAllColumns(tr);
302  } else {
303  for (auto& colname : colnames) {
304  TBranch* branch = tr->GetBranch(colname.c_str());
305  totalSizeCompressed += branch->GetZipBytes("*");
306  totalSizeUncompressed += branch->GetTotBytes("*");
307  }
308  t2t.addAllColumns(tr, std::move(colnames));
309  }
310  t2t.fill(tr);
311  delete tr;
312 
313  // needed for metrics dumping (upon next file read, or terminate due to watchdog)
314  if (currentFile == nullptr) {
315  currentFile = didir->getFileFolder(dh, fcnt, ntf).file;
316  tfCurrentFile = didir->getTimeFramesInFile(dh, fcnt);
317  }
318 
319  first = false;
320  }
321  totalDFSent++;
322  monitoring.send(Metric{(uint64_t)totalDFSent, "df-sent"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
323  monitoring.send(Metric{(uint64_t)totalSizeUncompressed / 1000, "aod-bytes-read-uncompressed"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
324  monitoring.send(Metric{(uint64_t)totalSizeCompressed / 1000, "aod-bytes-read-compressed"}.addTag(Key::Subsystem, monitoring::tags::Value::DPL));
325 
326  // save file number and time frame
327  *fileCounter = (fcnt - device.inputTimesliceId) / device.maxInputTimeslices;
328  *numTF = ntf;
329  currentFileIOTime += (uv_hrtime() - ioStart);
330  });
331  })};
332 
333  return callback;
334 }
