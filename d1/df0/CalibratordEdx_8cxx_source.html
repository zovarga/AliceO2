34 void CalibratordEdx::finalizeSlot(Slot& slot)
35 {
36  LOGP(info, "Finalizing slot {} <= TF <= {}", slot.getTFStart(), slot.getTFEnd());
37  slot.print();
38 
39  // compute calibration values from histograms
40  CalibdEdx* container = slot.getContainer();
41  container->finalize();
42  mCalibs.push_back(container->getCalib());
43 
44  TFType startTF = slot.getTFStart();
45  TFType endTF = slot.getTFEnd();
46  mIntervals.emplace_back(startTF, endTF);
47 
48  if (mDebugOutputStreamer) {
49  LOGP(info, "Dumping time slot data to file");
50  auto calibCopy = container->getCalib();
51  *mDebugOutputStreamer << "CalibdEdx"
52  << "startTF=" << startTF // Initial time frame of time slot
53  << "endTF=" << endTF // Final time frame of time slot
54  << "correction=" << calibCopy // dE/dx corretion
55  << "\n";
56  }
57 }
