40 void PHOSPedestalCalibDevice::run(o2::framework::ProcessingContext& ctx)
41 {
42  // scan Cells stream, collect mean and RMS then calculateaverage and post
43  if (mRunStartTime == 0) {
44  mRunStartTime = o2::header::get<o2::framework::DataProcessingHeader*>(ctx.inputs().get("cellTriggerRecords").header)->startTime;
45  }
46 
47  auto cells = ctx.inputs().get<gsl::span<o2::phos::Cell>>("cells");
48  LOG(debug) << "[PHOSPedestalCalibDevice - run] Received " << cells.size() << " cells, running calibration ...";
49  auto cellsTR = ctx.inputs().get<gsl::span<o2::phos::TriggerRecord>>("cellTriggerRecords");
50  for (const auto& tr : cellsTR) {
51  int firstCellInEvent = tr.getFirstEntry();
52  int lastCellInEvent = firstCellInEvent + tr.getNumberOfObjects();
53  for (int i = firstCellInEvent; i < lastCellInEvent; i++) {
54  const Cell c = cells[i];
55  if (c.getHighGain()) {
56  mMeanHG->Fill(c.getAbsId(), c.getEnergy());
57  mRMSHG->Fill(c.getAbsId(), 1.e+7 * c.getTime());
58  } else {
59  mMeanLG->Fill(c.getAbsId(), c.getEnergy());
60  mRMSLG->Fill(c.getAbsId(), 1.e+7 * c.getTime());
61  }
62  }
63  }
64 }

102 void PHOSPedestalCalibDevice::calculatePedestals()
103 {
104  mPedestals.reset(new Pedestals());
105 
106  //Calculate mean of pedestal distributions
107  for (unsigned short i = mMeanHG->GetNbinsX(); i > 0; i--) {
108  TH1D* pr = mMeanHG->ProjectionY(Form("proj%d", i), i, i);
109  float a = pr->GetMean();
110  short cellId = static_cast<short>(mMeanHG->GetXaxis()->GetBinCenter(i));
111  mPedestals->setHGPedestal(cellId, std::min(255, int(a)));
112  pr->Delete();
113  pr = mMeanLG->ProjectionY(Form("projLG%d", i), i, i);
114  a = pr->GetMean();
115  mPedestals->setLGPedestal(cellId, std::min(255, int(a)));
116  pr->Delete();
117  pr = mRMSHG->ProjectionY(Form("projRMS%d", i), i, i);
118  a = pr->GetMean();
119  mPedestals->setHGRMS(cellId, a);
120  pr->Delete();
121  pr = mRMSLG->ProjectionY(Form("projRMSLG%d", i), i, i);
122  a = pr->GetMean();
123  mPedestals->setLGRMS(cellId, a);
124  pr->Delete();
125  }
126 }

128 void PHOSPedestalCalibDevice::checkPedestals()
129 {
130  if (!mUseCCDB) {
131  mUpdateCCDB = true;
132  return;
133  }
134  LOG(info) << "Retrieving current Pedestals from CCDB";
135  //Read current pedestals for comparison
136  o2::ccdb::CcdbApi ccdb;
137  ccdb.init(mCCDBPath); // or http://localhost:8080 for a local installation
138  std::map<std::string, std::string> metadata;
139  auto* currentPedestals = ccdb.retrieveFromTFileAny<Pedestals>("PHS/Calib/Pedestals", metadata, mRunStartTime);
140 
141  if (!currentPedestals) { //was not read from CCDB, but expected
142  mUpdateCCDB = true;
143  return;
144  }
145 
146  LOG(info) << "Got current Pedestals from CCDB";
147 
148  //Compare to current
149  int nChanged = 0;
150  for (short i = o2::phos::Mapping::NCHANNELS; i > 1792; i--) {
151  short dp = mPedestals->getHGPedestal(i) - currentPedestals->getHGPedestal(i);
152  mPedDiff[i] = dp;
153  if (abs(dp) > 1) { //not a fluctuation
154  nChanged++;
155  }
156  dp = mPedestals->getLGPedestal(i) - currentPedestals->getLGPedestal(i);
157  mPedDiff[i + o2::phos::Mapping::NCHANNELS] = dp;
158  if (abs(dp) > 1) { //not a fluctuation
159  nChanged++;
160  }
161  }
162  LOG(info) << nChanged << " channels changed more that 1 ADC channel";
163  if (nChanged > kMinorChange) { //serious change, do not update CCDB automatically, use "force" option to overwrite
164  LOG(error) << "too many channels changed: " << nChanged << " (threshold not more than " << kMinorChange << ")";
165  if (!mForceUpdate) {
166  LOG(error) << "you may use --forceupdate option to force updating ccdb";
167  }
168  mUpdateCCDB = false;
169  } else {
170  mUpdateCCDB = true;
171  }
172 }
