66  //--------------????

69  //--------------????

123 //________________________________________________________________________________
124 void MatLayerCyl::populateFromTGeo(int ip, int iz, int ntrPerCell)
125 {
127 
128  float zmn = getZBinMin(iz), phmn = getPhiBinMin(ip), sn, cs, rMin = getRMin(), rMax = getRMax();
129  double meanRho = 0., meanX2X0 = 0., lgt = 0.;
130  ;
131  float dz = getDZ() / ntrPerCell;
132  for (int isz = ntrPerCell; isz--;) {
133  float zs = zmn + (isz + 0.5) * dz;
134  float dzt = zs > 0.f ? 0.25 * dz : -0.25 * dz; // to avoid 90 degree polar angle
135  for (int isp = ntrPerCell; isp--;) {
136  o2::math_utils::sincos(phmn + (isp + 0.5) * getDPhi() / ntrPerCell, sn, cs);
137  auto bud = o2::base::GeometryManager::meanMaterialBudget(rMin * cs, rMin * sn, zs - dzt, rMax * cs, rMax * sn, zs + dzt);
138  if (bud.length > 0.) {
139  meanRho += bud.length * bud.meanRho;
140  meanX2X0 += bud.meanX2X0; // we store actually not X2X0 but 1./X0
141  lgt += bud.length;
142  }
143  }
144  }
145  if (lgt > 0.) {
146  auto& cell = mCells[getCellIDPhiBin(ip, iz)];
147  cell.meanRho = meanRho / lgt; // mean rho
148  cell.meanX2X0 = meanX2X0 / lgt; // mean 1./X0 seen in this cell
149  }
150 }

153 bool MatLayerCyl::canMergePhiSlices(int i, int j, float maxRelDiff, int maxDifferent) const
154 {
155  if (std::abs(i - j) > 1 || i == j || std::max(i, j) >= getNPhiSlices()) {
156  LOG(error) << "Only existing " << getNPhiSlices() << " slices with diff. of 1 can be merged, input is " << i << " and " << j;
157  return false;
158  }
159  int ndiff = 0; // number of different cells
160  for (int iz = getNZBins(); iz--;) {
161  const auto& cellI = getCellPhiBin(i, iz);
162  const auto& cellJ = getCellPhiBin(j, iz);
163  if (cellsDiffer(cellI, cellJ, maxRelDiff)) {
164  if (++ndiff > maxDifferent) {
165  return false;
166  }
167  }
168  }
169  return true;
170 }

172 //________________________________________________________________________________
173 bool MatLayerCyl::cellsDiffer(const MatCell& cellA, const MatCell& cellB, float maxRelDiff) const
174 {
176  float rav = 0.5 * (cellA.meanRho + cellB.meanRho), xav = 0.5 * (cellA.meanX2X0 + cellB.meanX2X0);
177  float rdf = 0.5 * (cellA.meanRho - cellB.meanRho), xdf = 0.5 * (cellA.meanX2X0 - cellB.meanX2X0);
178  if (rav > 0 && std::abs(rdf / rav) > maxRelDiff) {
179  return true;
180  }
181  if (xav > 0 && std::abs(xdf / xav) > maxRelDiff) {
182  return true;
183  }
184  return false;
185 }

188 void MatLayerCyl::optimizePhiSlices(float maxRelDiff)
189 {
190  // merge compatible phi slices
191  if (getNPhiSlices() < getNPhiBins()) {
192  LOG(error) << getNPhiBins() << " phi bins were already merged to " << getNPhiSlices() << " slices";
193  return;
194  }
195  int newSl = 0;
196  for (int is = 1; is < getNPhiSlices(); is++) {
197  if (!canMergePhiSlices(is - 1, is, maxRelDiff)) {
198  newSl++;
199  }
200  mPhiBin2Slice[is] = newSl;
201  }
202  LOG(info) << newSl + 1 << " slices out of " << getNPhiBins();
203  if (newSl + 1 == getNPhiSlices()) {
204  return;
205  }
206  newSl = 0;
207  int slMin = 0, slMax = 0, is = 0;
208  while (is++ < getNPhiSlices()) {
209  while (is < getNPhiSlices() && mPhiBin2Slice[is] == newSl) { // select similar slices
210  slMax++;
211  is++;
212  }
213  if (slMax > slMin || newSl != slMin) { // merge or shift slices
214  mSliceCos[newSl] = mSliceCos[slMin];
215  mSliceSin[newSl] = mSliceSin[slMin];
216  float norm = 1.f / (1.f + slMax - slMin);
217  for (int iz = getNZBins(); iz--;) {
218  int iDest = newSl * getNZBins() + iz, iSrc = slMin * getNZBins() + iz;
219  mCells[iDest] = mCells[iSrc];
220  for (int ism = slMin + 1; ism <= slMax; ism++) {
221  iSrc = ism * getNZBins() + iz;
222  mCells[iDest].meanX2X0 += mCells[iSrc].meanX2X0;
223  mCells[iDest].meanRho += mCells[iSrc].meanRho;
224  }
225  mCells[iDest].scale(norm);
226  }
227  LOG(info) << "mapping " << slMin << ":" << slMax << " to new slice " << newSl;
228  }
229  newSl++;
230  slMin = slMax = is;
231  }
232  mNPhiSlices = newSl;
233 
234  // relocate arrays to avoid spaces after optimization
235  // mSliceCos pointer does not change, but sliceSin needs to be relocated
236  auto offs = alignSize(newSl * sizeof(float), getBufferAlignmentBytes());
237  char* dst = ((char*)mSliceCos) + offs; // account for alignment
238  o2::gpu::resizeArray(mSliceSin, getNPhiBins(), newSl, reinterpret_cast<float*>(dst));
239  // adjust mCells array
240  dst = ((char*)mSliceSin) + offs; // account for alignment
241  o2::gpu::resizeArray(mCells, getNPhiBins() * getNZBins(), newSl * getNZBins(), reinterpret_cast<MatCell*>(dst));
242  mFlatBufferSize = estimateFlatBufferSize();
243  LOG(info) << "Updated Nslices = " << getNPhiSlices();
244 }

247 void MatLayerCyl::getMeanRMS(MatCell& mean, MatCell& rms) const
248 {
249  // mean and RMS over layer
250  mean.meanRho = rms.meanRho = 0.f;
251  mean.meanX2X0 = rms.meanX2X0 = 0.f;
252  for (int ip = getNPhiBins(); ip--;) {
253  for (int iz = getNZBins(); iz--;) {
254  const auto& cell = getCellPhiBin(ip, iz);
255  mean.meanRho += cell.meanRho;
256  mean.meanX2X0 += cell.meanX2X0;
257  rms.meanRho += cell.meanRho * cell.meanRho;
258  rms.meanX2X0 += cell.meanX2X0 * cell.meanX2X0;
259  }
260  }
261  int nc = getNPhiBins() * getNZBins();
262  mean.meanRho /= nc;
263  mean.meanX2X0 /= nc;
264  rms.meanRho /= nc;
265  rms.meanX2X0 /= nc;
266  rms.meanRho -= mean.meanRho * mean.meanRho;
267  rms.meanX2X0 -= mean.meanX2X0 * mean.meanX2X0;
268  rms.meanRho = rms.meanRho > 0.f ? std::sqrt(rms.meanRho) : 0.f;
269  rms.meanX2X0 = rms.meanX2X0 > 0.f ? std::sqrt(rms.meanX2X0) : 0.f;
270 }

327 GPUd() int MatLayerCyl::getNPhiBinsInSlice(int iSlice, int& binMin, int& binMax) const
328 {
329  // slow method to get number of phi bins for given phi slice
330  int nb = 0;
331  binMin = binMax = -1;
332  for (int ib = getNPhiBins(); ib--;) {
333  if (phiBin2Slice(ib) == iSlice) {
334  binMax < 0 ? binMin = binMax = ib : binMin = ib;
335  nb++;
336  continue;
337  }
338  if (binMax >= 0) {
339  break; // no more bins since they are consecutive
340  }
341  }
342  return nb;
