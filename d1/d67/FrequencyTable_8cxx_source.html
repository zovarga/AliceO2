57 inline double_t computeEntropy(const FrequencyTable& table)
58 {
59  double_t entropy = std::accumulate(table.begin(), table.end(), 0, [&table](double_t entropy, count_t frequency) {
60  const double_t p = static_cast<double_t>(frequency) / static_cast<double_t>(table.getNumSamples());
61  const double_t length = p == 0 ? 0 : std::log2(p);
62  return entropy -= p * length;
63  });
64  entropy += [&table]() {
65  const double_t p = static_cast<double_t>(table.getIncompressibleSymbolFrequency()) / static_cast<double_t>(table.getNumSamples());
66  return p * (-std::log2(p) + table.getAlphabetRangeBits());
67  }();
68 
69  return entropy;
70 };

72 count_t computeRenormingPrecision(const FrequencyTable& frequencyTable)
73 {
74  const uint8_t minBits = std::ceil(std::log2(frequencyTable.getNUsedAlphabetSymbols()));
75  const uint8_t estimate = minBits * 3u / 2u;
76  const uint8_t maxThreshold = std::max(minBits, MaxRenormThreshold);
77  const uint8_t minThreshold = std::max(estimate, MinRenormThreshold);
78 
79  return std::min(minThreshold, maxThreshold);
80 };
