35 template <typename T>
36 void readFile(const std::string& filename, std::vector<T>* tokens)
37 {
38  std::ifstream is(filename, std::ios_base::binary | std::ios_base::in);
39  if (is) {
40  // get length of file:
41  is.seekg(0, is.end);
42  size_t length = is.tellg();
43  is.seekg(0, is.beg);
44 
45  // reserve size of tokens
46  if (!tokens) {
47  throw std::runtime_error("Cannot read file into nonexistent vector");
48  }
49 
50  if (length % sizeof(T)) {
51  throw std::runtime_error("Filesize is not a multiple of datatype.");
52  }
53  // size the vector appropriately
54  size_t num_elems = length / sizeof(T);
55  tokens->resize(num_elems);
56 
57  // read data as a block:
58  is.read(reinterpret_cast<char*>(tokens->data()), length);
59  is.close();
60  }
61 }
