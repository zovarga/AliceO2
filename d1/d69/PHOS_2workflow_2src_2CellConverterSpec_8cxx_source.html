33 void CellConverterSpec::run(framework::ProcessingContext& ctx)
34 {
35  // LOG(debug) << "[PHOSCellConverter - run] called";
36  LOG(info) << "[PHOSCellConverter - run] called";
37  auto dataref = ctx.inputs().get("digits");
38  auto const* phosheader = o2::framework::DataRefUtils::getHeader<o2::phos::PHOSBlockHeader*>(dataref);
39  if (!phosheader->mHasPayload) {
40  mOutputCells.clear();
41  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLS", 0, o2::framework::Lifetime::Timeframe}, mOutputCells);
42  mOutputCellTrigRecs.clear();
43  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLTRIGREC", 0, o2::framework::Lifetime::Timeframe}, mOutputCellTrigRecs);
44  if (mPropagateMC) {
45  mOutputTruthCont.clear();
46  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLSMCTR", 0, o2::framework::Lifetime::Timeframe}, mOutputTruthCont);
47  }
48  return;
49  }
50  mOutputCells.clear();
51  mOutputCellTrigRecs.clear();
52 
53  auto digits = ctx.inputs().get<std::vector<o2::phos::Digit>>("digits");
54  auto digitsTR = ctx.inputs().get<std::vector<o2::phos::TriggerRecord>>("digitTriggerRecords");
55  std::unique_ptr<const o2::dataformats::MCTruthContainer<o2::phos::MCLabel>> truthcont(nullptr);
56  if (mPropagateMC) {
57  truthcont = ctx.inputs().get<o2::dataformats::MCTruthContainer<o2::phos::MCLabel>*>("digitsmctr");
58  mOutputTruthCont.clear();
59  }
60  if (mPropagateMC) {
61  LOG(info) << "[PHOSCellConverter - run] Received " << digits.size() << " digits and " << digitsTR.size() << " TriggerRecords" << truthcont->getNElements() << " MC labels";
62  } else {
63  LOG(info) << "[PHOSCellConverter - run] Received " << digits.size() << " digits and " << digitsTR.size() << " TriggerRecords";
64  }
65 
66  //Get TimeStamp from TriggerRecord
67  if (!mBadMap) {
68  if (o2::phos::PHOSSimParams::Instance().mCCDBPath.compare("localtest") == 0) {
69  mBadMap = new BadChannelsMap(1); // test default map
70  LOG(info) << "[PHOSCellConverter - run] No reading BadMap from ccdb requested, set default";
71  } else {
72  LOG(info) << "[PHOSCellConverter - run] getting BadMap object from ccdb";
73  o2::ccdb::CcdbApi ccdb;
74  std::map<std::string, std::string> metadata; // do we want to store any meta data?
75  ccdb.init(o2::base::NameConf::getCCDBServer()); // or http://localhost:8080 for a local installation
76  long bcTime = -1; //TODO!!! Convert BC time to time o2::InteractionRecord bcTime = digitsTR.front().getBCData() ;
77  mBadMap = ccdb.retrieveFromTFileAny<o2::phos::BadChannelsMap>("PHOS/BadMap", metadata, bcTime);
78  if (!mBadMap) {
79  LOG(fatal) << "[PHOSCellConverter - run] can not get Bad Map";
80  }
81  }
82  }
83  //TODO!!! Should we check if BadMap should be updated/validity range still valid???
84  mOutputCells.reserve(digits.size()); // most of digits will be copied
85  int icell = 0;
86  int labelIndex = 0;
87  for (const auto& tr : digitsTR) {
88  int iFirstDigit = tr.getFirstEntry();
89  int iLastDigit = iFirstDigit + tr.getNumberOfObjects();
90  int indexStart = mOutputCells.size();
91  for (int i = iFirstDigit; i < iLastDigit; i++) {
92  const auto& dig = digits.at(i);
93 
94  if (dig.isTRU()) {
95  ChannelType_t chantype;
96  if (dig.isHighGain()) {
97  chantype = ChannelType_t::TRU2x2;
98  } else {
99  chantype = ChannelType_t::TRU4x4;
100  }
101  mOutputCells.emplace_back(dig.getAbsId(), dig.getAmplitude(), dig.getTime(), chantype);
102  } else {
103  //apply filter
104  if (!mBadMap->isChannelGood(dig.getAbsId())) {
105  continue;
106  }
107 
108  ChannelType_t chantype;
109  if (dig.isHighGain()) {
110  chantype = ChannelType_t::HIGH_GAIN;
111  } else {
112  chantype = ChannelType_t::LOW_GAIN;
113  }
114  mOutputCells.emplace_back(dig.getAbsId(), dig.getAmplitude(), dig.getTime(), chantype);
115  if (mPropagateMC) { //copy MC info,
116  int iLab = dig.getLabel();
117  if (iLab > -1) {
118  mOutputTruthCont.addElements(icell, truthcont->getLabels(iLab));
119  } else {
120  MCLabel label(0, 0, 0, true, 0);
121  label.setNoise();
122  mOutputTruthCont.addElement(icell, label);
123  }
124  icell++;
125  }
126  }
127  }
128  mOutputCellTrigRecs.emplace_back(tr.getBCData(), indexStart, mOutputCells.size() - indexStart);
129  }
130  LOG(info) << "[PHOSCellConverter - run] Writing " << mOutputCells.size() << " cells, " << mOutputCellTrigRecs.size() << " Trig Records " << mOutputTruthCont.getNElements() << " PHOS labels ";
131  ;
132  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLS", 0, o2::framework::Lifetime::Timeframe}, mOutputCells);
133  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLTRIGREC", 0, o2::framework::Lifetime::Timeframe}, mOutputCellTrigRecs);
134  if (mPropagateMC) {
135  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLSMCTR", 0, o2::framework::Lifetime::Timeframe}, mOutputTruthCont);
136  }
137 }

150  }
151  return o2::framework::DataProcessorSpec{"PHOSCellConverterSpec",
152  inputs,
153  outputs,
154  o2::framework::adaptFromTask<o2::phos::reco_workflow::CellConverterSpec>(propagateMC)};
