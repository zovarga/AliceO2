37 // helper function to publish a message to an outside subscriber
38 bool publishMessage(FairMQChannel& channel, std::string const& message)
39 {
40  if (channel.IsValid()) {
41  auto text = new std::string(message);
42  std::unique_ptr<FairMQMessage> payload(channel.NewMessage(
43  const_cast<char*>(text->c_str()),
44  text->length(), [](void* data, void* hint) { delete static_cast<std::string*>(hint); }, text));
45  if (channel.Send(payload) > 0) {
46  return true;
47  }
48  } else {
49  LOG(error) << "CHANNEL NOT VALID";
50  }
51  return false;
52 }

54 // make channel (transport factory needs to be injected)
55 FairMQChannel createPUBChannel(std::string const& address,
56  std::string const& type = "pub")
57 {
58  auto factory = FairMQTransportFactory::CreateTransportFactory("zeromq");
59  static int i = 0;
60  std::stringstream str;
61  str << "channel" << i++;
62  auto externalpublishchannel = FairMQChannel{str.str(), type, factory};
63  externalpublishchannel.Init();
64  if ((type.compare("pub") == 0) || (type.compare("pull") == 0)) {
65  externalpublishchannel.Bind(address);
66  LOG(info) << "BINDING TO ADDRESS " << address << " type " << type;
67  } else {
68  externalpublishchannel.Connect(address);
69  LOG(info) << "CONNECTING TO ADDRESS " << address << " type " << type;
70  }
71  externalpublishchannel.Validate();
72  return externalpublishchannel;
73 }
