49  void initDigitizerTask(framework::InitContext& ic) override
50  {
51  LOG(info) << "Initializing ZDC digitization";
52 
53  auto& dopt = o2::conf::DigiParams::Instance();
54 
55  mDigitizer.setCCDBServer(o2::base::NameConf::getCCDBServer());
56  auto enableHitInfo = ic.options().get<bool>("enable-hit-info");
57  mDigitizer.setMaskTriggerBits(!enableHitInfo);
58  mDigitizer.setSkipMCLabels(not mUseMC);
59  mDigitizer.init();
60  mROMode = mDigitizer.isContinuous() ? o2::parameters::GRPObject::CONTINUOUS : o2::parameters::GRPObject::PRESENT;
61  }

63  void run(framework::ProcessingContext& pc)
64  {
65  LOG(info) << "Doing ZDC digitization";
66 
67  // TODO: this should eventually come from the framework and depend on the TF timestamp
68  mDigitizer.refreshCCDB();
69 
70  // read collision context from input
71  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
72  context->initSimChains(o2::detectors::DetID::ZDC, mSimChains);
73 
74  const auto& grp = context->getGRP();
75  mDigitizer.setTimeStamp(grp.getTimeStart());
76  if (mDigitizer.getNEmptyBunches() < 0) {
77  mDigitizer.findEmptyBunches(context->getBunchFilling().getPattern());
78  }
79  auto& irecords = context->getEventRecords();
80  auto& eventParts = context->getEventParts();
81  if (irecords.empty()) {
82  return;
83  }
84  // loop over all composite collisions given from context
85  // (aka loop over all the interaction records)
86  std::vector<o2::zdc::Hit> hits;
87 
88  for (int collID = 0; collID < irecords.size(); ++collID) {
89 
90  const auto& irec = irecords[collID];
91  mDigitizer.setInteractionRecord(irec);
92 
93  for (auto& part : eventParts[collID]) {
94 
95  context->retrieveHits(mSimChains, "ZDCHit", part.sourceID, part.entryID, &hits);
96  LOG(info) << "For collision " << collID << " eventID " << part.entryID << " found ZDC " << hits.size() << " hits ";
97 
98  mDigitizer.setEventID(part.entryID);
99  mDigitizer.setSrcID(part.sourceID);
100 
101  mDigitizer.process(hits, mDigitsBC, mDigitsCh, mLabels);
102  }
103  }
104 
105  o2::InteractionTimeRecord terminateIR;
106  terminateIR.orbit = 0xffffffff; // supply IR in the infinite future to flush all cached BC
107  mDigitizer.setInteractionRecord(terminateIR);
108  mDigitizer.flush(mDigitsBC, mDigitsCh, mLabels);
109 
110  for (uint32_t ib = 0; ib < mDigitsBC.size(); ib++) {
111  mDigitizer.assignTriggerBits(ib, mDigitsBC);
112  }
113 
114  const auto &irFirst = irecords.front(), irLast = irecords.back();
115  o2::InteractionRecord irPed(o2::constants::lhc::LHCMaxBunches - 1, irFirst.orbit);
116  int norbits = irLast.orbit - irFirst.orbit + 1;
117  mOrbitData.resize(norbits);
118  for (int i = 0; i < norbits; i++) {
119  mDigitizer.updatePedestalReference(mOrbitData[i]);
120  mOrbitData[i].ir = irPed;
121  irPed.orbit++;
122  }
123 
124  mDigitizer.Finalize(mDigitsBC, mOrbitData);
125 
126  // send out to next stage
127  pc.outputs().snapshot(Output{"ZDC", "DIGITSBC", 0, Lifetime::Timeframe}, mDigitsBC);
128  pc.outputs().snapshot(Output{"ZDC", "DIGITSCH", 0, Lifetime::Timeframe}, mDigitsCh);
129  pc.outputs().snapshot(Output{"ZDC", "DIGITSPD", 0, Lifetime::Timeframe}, mOrbitData);
130  if (pc.outputs().isAllowed({"ZDC", "DIGITSLBL", 0})) {
131  pc.outputs().snapshot(Output{"ZDC", "DIGITSLBL", 0, Lifetime::Timeframe}, mLabels);
132  }
133 
134  LOG(info) << "ZDC: Sending ROMode= " << mROMode << " to GRPUpdater";
135  pc.outputs().snapshot(Output{"ZDC", "ROMode", 0, Lifetime::Timeframe}, mROMode);
136 
137  // we should be only called once; tell DPL that this process is ready to exit
138  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
139  }
140 
141  private:
142  Digitizer mDigitizer;
143  std::vector<TChain*> mSimChains;
144  std::vector<o2::zdc::ChannelData> mDigitsCh;
145  std::vector<o2::zdc::BCData> mDigitsBC;
146  std::vector<o2::zdc::OrbitData> mOrbitData;
147  o2::dataformats::MCTruthContainer<o2::zdc::MCLabel> mLabels; // labels which get filled
148 
149  // RS: at the moment using hardcoded flag for continuous readout
150  o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::CONTINUOUS; // readout mode
151 };

153 o2::framework::DataProcessorSpec getZDCDigitizerSpec(int channel, bool mctruth)
154 {
155  // create the full data processor spec using
156  // a name identifier
157  // input description
158  // algorithmic description (here a lambda getting called once to setup the actual processing function)
159  // options that can be used for this processor (here: input file names where to take the hits)
160  std::vector<OutputSpec> outputs;
161  outputs.emplace_back("ZDC", "DIGITSBC", 0, Lifetime::Timeframe);
162  outputs.emplace_back("ZDC", "DIGITSCH", 0, Lifetime::Timeframe);
163  outputs.emplace_back("ZDC", "DIGITSPD", 0, Lifetime::Timeframe);
164  if (mctruth) {
165  outputs.emplace_back("ZDC", "DIGITSLBL", 0, Lifetime::Timeframe);

168 
169  return DataProcessorSpec{
170  "ZDCDigitizer",
171  Inputs{InputSpec{"collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe}},
172  outputs,
173  AlgorithmSpec{adaptFromTask<ZDCDPLDigitizerTask>(mctruth)},
174  Options{{"enable-hit-info", o2::framework::VariantType::Bool, false, {"enable hit info of unread channels"}}}};
