59 // create histogram from specification and insert it into the registry
60 HistPtr HistogramRegistry::insert(const HistogramSpec& histSpec)
61 {
62  validateHistName(histSpec.name, histSpec.hash);
63  const uint32_t idx = imask(histSpec.hash);
64  for (auto i = 0u; i < MAX_REGISTRY_SIZE; ++i) {
65  TObject* rawPtr = nullptr;
66  std::visit([&](const auto& sharedPtr) { rawPtr = sharedPtr.get(); }, mRegistryValue[imask(idx + i)]);
67  if (!rawPtr) {
68  registerName(histSpec.name);
69  mRegistryKey[imask(idx + i)] = histSpec.hash;
70  mRegistryValue[imask(idx + i)] = HistFactory::createHistVariant(histSpec);
71  lookup += i;
72  return mRegistryValue[imask(idx + i)];
73  }
74  }
75  LOGF(fatal, R"(Internal array of HistogramRegistry "%s" is full.)", mName);
76  return HistPtr();
77 }

79 // helper function that checks if histogram name can be used in registry
80 void HistogramRegistry::validateHistName(const std::string& name, const uint32_t hash)
81 {
82  // validate that hash is unique
83  auto it = std::find(mRegistryKey.begin(), mRegistryKey.end(), hash);
84  if (it != mRegistryKey.end()) {
85  auto idx = it - mRegistryKey.begin();
86  std::string collidingName{};
87  std::visit([&](const auto& hist) { collidingName = hist->GetName(); }, mRegistryValue[idx]);
88  LOGF(fatal, R"(Hash collision in HistogramRegistry "%s"! Please rename histogram "%s" or "%s".)", mName, name, collidingName);
89  }
90 
91  // validate that name contains only allowed characters
92  if (!std::regex_match(name, std::regex("([a-zA-Z0-9])(([\\/_])?[a-zA-Z0-9])*"))) {
93  LOGF(fatal, R"(Histogram name "%s" contains invalid characters.)", name);
94  }
95 }

112 // store a copy of an existing histogram (or group of histograms) under a different name
113 void HistogramRegistry::addClone(const std::string& source, const std::string& target)
114 {
115  auto doInsertClone = [&](const auto& sharedPtr) {
116  if (!sharedPtr.get()) {
117  return;
118  }
119  std::string sourceName{((TNamed*)sharedPtr.get())->GetName()};
120  // search for histograms starting with source_ substring
121  if (sourceName.rfind(source, 0) == 0) {
122  // when cloning groups of histograms source_ and target_ must end with "/"
123  if (sourceName.size() != source.size() && (source.back() != '/' || target.back() != '/')) {
124  return;
125  }
126  // when cloning a single histogram the specified target_ must not be a group name
127  if (sourceName.size() == source.size() && target.back() == '/') {
128  LOGF(fatal, "Cannot turn histogram into folder!");
129  }
130  std::string targetName{target};
131  targetName += sourceName.substr(sourceName.find(source) + source.size());
132  insertClone(targetName.data(), sharedPtr);
133  }
134  };
135 
136  for (auto& histVariant : mRegistryValue) {
137  std::visit(doInsertClone, histVariant);
138  }
139 }

141 // function to query if name is already in use
142 bool HistogramRegistry::contains(const HistName& histName)
143 {
144  // check for all occurances of the hash
145  auto iter = mRegistryKey.begin();
146  while ((iter = std::find(iter, mRegistryKey.end(), histName.hash)) != mRegistryKey.end()) {
147  const char* curName = nullptr;
148  std::visit([&](auto&& hist) { if(hist) { curName = hist->GetName(); } }, mRegistryValue[iter - mRegistryKey.begin()]);
149  // if hash is the same, make sure that name is indeed the same
150  if (strcmp(curName, histName.str) == 0) {
151  return true;
152  }
153  }
154  return false;
155 }

175 // print some useful meta-info about the stored histograms
176 void HistogramRegistry::print(bool showAxisDetails)
177 {
178  std::vector<double> fillFractions{0.1, 0.25, 0.5};
179  std::vector<double> totalSizes(fillFractions.size());
180 
181  uint32_t nHistos{};
182  bool containsSparseHist{};
183  auto printHistInfo = [&](auto&& hist) {
184  if (hist) {
185  using T = std::decay_t<decltype(*hist)>;
186  bool isSparse{};
187  if (hist->InheritsFrom(THnSparse::Class())) {
188  isSparse = true;
189  containsSparseHist = true;
190  }
191  ++nHistos;
192  std::vector<double> sizes;
193  std::string sizeInfo{};
194  if (isSparse) {
195  std::transform(std::begin(fillFractions), std::end(fillFractions), std::back_inserter(sizes), [&hist](auto& fraction) { return HistFiller::getSize(hist, fraction); });
196  for (int i = 0; i < fillFractions.size(); ++i) {
197  sizeInfo += fmt::format("{:.2f} kB ({:.0f} %)", sizes[i] * 1024, fillFractions[i] * 100);
198  if (i != fillFractions.size() - 1) {
199  sizeInfo += ", ";
200  }
201  }
202  } else {
203  double size = HistFiller::getSize(hist);
204  sizes.resize(fillFractions.size(), size);
205  sizeInfo = fmt::format("{:.2f} kB", sizes[0] * 1024);
206  }
207  std::transform(totalSizes.begin(), totalSizes.end(), sizes.begin(), totalSizes.begin(), std::plus<double>());
208  LOGF(info, "Hist %03d: %-35s %-19s [%s]", nHistos, hist->GetName(), hist->IsA()->GetName(), sizeInfo);
209 
210  if (showAxisDetails) {
211  int nDim = 0;
212  if constexpr (std::is_base_of_v<THnBase, T>) {
213  nDim = hist->GetNdimensions();
214  } else if constexpr (std::is_base_of_v<TH1, T>) {
215  nDim = hist->GetDimension();
216  }
217  TAxis* axis{nullptr};
218  for (int d = 0; d < nDim; ++d) {
219  if constexpr (std::is_base_of_v<THnBase, T> || std::is_base_of_v<StepTHn, T>) {
220  axis = hist->GetAxis(d);
221  } else {
222  if (d == 0) {
223  axis = hist->GetXaxis();
224  } else if (d == 1) {
225  axis = hist->GetYaxis();
226  } else if (d == 2) {
227  axis = hist->GetZaxis();
228  }
229  }
230  LOGF(info, "- Axis %d: %-20s (%d bins)", d, axis->GetTitle(), axis->GetNbins());
231  }
232  }
233  }
234  };
235 
236  std::string titleString{"======================== HistogramRegistry ========================"};
237  LOGF(info, "");
238  LOGF(info, "%s", titleString);
239  LOGF(info, "%s\"%s\"", std::string((int)(0.5 * titleString.size() - (1 + 0.5 * mName.size())), ' '), mName);
240  for (auto& curHistName : mRegisteredNames) {
241  std::visit(printHistInfo, mRegistryValue[getHistIndex(HistName{curHistName.data()})]);
242  }
243  std::string totalSizeInfo{};
244  if (containsSparseHist) {
245  for (int i = 0; i < totalSizes.size(); ++i) {
246  totalSizeInfo += fmt::format("{:.2f} MB ({:.0f} %)", totalSizes[i], fillFractions[i] * 100);
247  if (i != totalSizes.size() - 1) {
248  totalSizeInfo += ", ";
249  }
250  }
251  } else {
252  totalSizeInfo = fmt::format("{:.2f} MB", totalSizes[0]);
253  }
254  LOGF(info, "%s", std::string(titleString.size(), '='), titleString);
255  LOGF(info, "Total: %d histograms, ca. %s", nHistos, totalSizeInfo);
256  if (lookup) {
257  LOGF(info, "Due to index collisions, histograms were shifted by %d registry slots in total.", lookup);
258  }
259  LOGF(info, "%s", std::string(titleString.size(), '='), titleString);
260  LOGF(info, "");
261 }

263 // create output structure will be propagated to file-sink
264 TList* HistogramRegistry::operator*()
265 {
266  TList* list = new TList();
267  list->SetName(mName.data());
268 
269  if (mSortHistos) {
270  auto caseInsensitiveCompare = [](const std::string& s1, const std::string& s2) {
271  return std::lexicographical_compare(s1.begin(), s1.end(), s2.begin(), s2.end(),
272  [](char c1, char c2) { return std::tolower(static_cast<unsigned char>(c1)) < std::tolower(static_cast<unsigned char>(c2)); });
273  };
274  std::sort(mRegisteredNames.begin(), mRegisteredNames.end(), caseInsensitiveCompare);
275  }
276 
277  for (auto& curHistName : mRegisteredNames) {
278  TNamed* rawPtr = nullptr;
279  std::visit([&](const auto& sharedPtr) { rawPtr = (TNamed*)sharedPtr.get(); }, mRegistryValue[getHistIndex(HistName{curHistName.data()})]);
280  if (rawPtr) {
281  std::deque<std::string> path = splitPath(rawPtr->GetName());
282  std::string name = path.back();
283  path.pop_back();
284  TList* targetList{getSubList(list, path)};
285  if (targetList) {
286  rawPtr->SetName(name.data());
287  targetList->Add(rawPtr);
288  } else {
289  LOGF(fatal, "Specified subfolder could not be created.");
290  }
291  }
292  }
293 
294  // place lists always at the top
295  std::function<void(TList*)> moveListsToTop;
296  moveListsToTop = [&](TList* list) {
297  TIter next(list);
298  TNamed* subList = nullptr;
299  std::vector<TObject*> subLists;
300  while ((subList = (TNamed*)next())) {
301  if (subList->InheritsFrom(TList::Class())) {
302  subLists.push_back(subList);
303  moveListsToTop((TList*)subList);
304  }
305  }
306  std::reverse(subLists.begin(), subLists.end());
307  for (auto curList : subLists) {
308  list->Remove(curList);
309  list->AddFirst(curList);
310  }
311  };
312  moveListsToTop(list);
313 
314  // create dedicated directory containing all of the registrys histograms
315  if (mCreateRegistryDir) {
316  // propagate this to the writer by adding a 'flag' to the output list
317  list->AddLast(new TNamed("createFolder", ""));
318  }
319  return list;
320 }

322 // helper function to create resp. find the subList defined by path
323 TList* HistogramRegistry::getSubList(TList* list, std::deque<std::string>& path)
324 {
325  if (path.empty()) {
326  return list;
327  }
328  TList* targetList{nullptr};
329  std::string nextList = path[0];
330  path.pop_front();
331  if (auto subList = (TList*)list->FindObject(nextList.data())) {
332  if (subList->InheritsFrom(TList::Class())) {
333  targetList = getSubList((TList*)subList, path);
334  } else {
335  return nullptr;
336  }
337  } else {
338  subList = new TList();
339  subList->SetName(nextList.data());
340  list->Add(subList);
341  targetList = getSubList(subList, path);
342  }
343  return targetList;
344 }

358 // helper function that checks if name of histogram is reasonable and keeps track of names already in use
359 void HistogramRegistry::registerName(const std::string& name)
360 {
361  if (name.empty() || name.back() == '/') {
362  LOGF(fatal, "Invalid name for a histogram.");
363  }
364  std::deque<std::string> path = splitPath(name);
365  std::string cumulativeName{};
366  int depth = path.size();
367  for (auto& step : path) {
368  if (step.empty()) {
369  LOGF(fatal, R"(Found empty group name in path for histogram "%s".)", name);
370  }
371  cumulativeName += step;
372  for (auto& curName : mRegisteredNames) {
373  // there is already a histogram where we want to put a folder or histogram
374  if (cumulativeName == curName) {
375  LOGF(fatal, R"(Histogram name "%s" is not compatible with existing names.)", name);
376  }
377  // for the full new histogram name we need to check that none of the existing histograms already uses this as a group name
378  if (depth == 1) {
379  if (curName.rfind(cumulativeName, 0) == 0 && curName.size() > cumulativeName.size() && curName.at(cumulativeName.size()) == '/') {
380  LOGF(fatal, R"(Histogram name "%s" is not compatible with existing names.)", name);
381  }
382  }
383  }
384  --depth;
385  cumulativeName += "/";
386  }
387  mRegisteredNames.push_back(name);
388 }
