55  void run(o2::framework::ProcessingContext& pc) final
56  {
57 
58  auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime; // is this the timestamp of the current TF?
59  auto data = pc.inputs().get<gsl::span<o2::dataformats::CalibInfoTOF>>("input");
60  LOG(info) << "Processing TF " << tfcounter << " with " << data.size() << " tracks";
61  mCollector->process(tfcounter, data);
62  sendOutput(pc.outputs());
63  }

65  void endOfStream(o2::framework::EndOfStreamContext& ec) final
66  {
67  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
68  mCollector->checkSlotsToFinalize(INFINITE_TF);
69  // we force finalizing slot zero (unless everything was already finalized), no matter how many entries we had
70  if (mCollector->getNSlots() != 0) {
71  mCollector->finalizeSlot(mCollector->getSlot(0));
72  }
73  sendOutput(ec.outputs());
74  }

80  //________________________________________________________________
81  void sendOutput(DataAllocator& output)
82  {
83  // in output we send the calibration tree
84  auto& collectedInfo = mCollector->getCollectedCalibInfo();
85  LOG(debug) << "In CollectorSpec sendOutput: size = " << collectedInfo.size();
86  if (collectedInfo.size()) {
87  auto entries = collectedInfo.size();
88  // this means that we are ready to send the output
89  auto entriesPerChannel = mCollector->getEntriesPerChannel();
90  output.snapshot(Output{o2::header::gDataOriginTOF, "COLLECTEDINFO", 0, Lifetime::Timeframe}, collectedInfo);
91  output.snapshot(Output{o2::header::gDataOriginTOF, "ENTRIESCH", 0, Lifetime::Timeframe}, entriesPerChannel);
92  mCollector->initOutput(); // reset the output for the next round
93  }
94  }
