44 void Dispatcher::init(InitContext& ctx)
45 {
46  LOG(debug) << "Reading Data Sampling Policies...";
47  boost::property_tree::ptree policiesTree;
48 
49  if (mReconfigurationSource.empty() == false) {
50  std::unique_ptr<ConfigurationInterface> cfg = ConfigurationFactory::getConfiguration(mReconfigurationSource);
51  policiesTree = cfg->getRecursive("dataSamplingPolicies");
52  mPolicies.clear();
53  } else if (ctx.options().isSet("sampling-config-ptree")) {
54  policiesTree = ctx.options().get<boost::property_tree::ptree>("sampling-config-ptree");
55  mPolicies.clear();
56  } else {
57  ; // we use policies declared during workflow init.
58  }
59 
60  for (auto&& policyConfig : policiesTree) {
61  // we don't want the Dispatcher to exit due to one faulty Policy
62  try {
63  mPolicies.emplace_back(std::make_shared<DataSamplingPolicy>(DataSamplingPolicy::fromConfiguration(policyConfig.second)));
64  } catch (std::exception& ex) {
65  LOG(warn) << "Could not load the Data Sampling Policy '"
66  << policyConfig.second.get_optional<std::string>("id").value_or("") << "', because: " << ex.what();
67  } catch (...) {
68  LOG(warn) << "Could not load the Data Sampling Policy '"
69  << policyConfig.second.get_optional<std::string>("id").value_or("") << "'";
70  }
71  }
72 
73  auto spec = ctx.services().get<const DeviceSpec>();
74  mDeviceID.runtimeInit(spec.id.substr(0, DataSamplingHeader::deviceIDTypeSize).c_str());
75 }

77 void Dispatcher::run(ProcessingContext& ctx)
78 {
79  // todo: consider matching (and deciding) in completion policy to save some time
80  // it is not trivial though, we would have to share state with the customize() method,
81  // which is not possible atm.
82 
83  for (auto inputIt = ctx.inputs().begin(); inputIt != ctx.inputs().end(); inputIt++) {
84 
85  const DataRef& firstPart = inputIt.getByPos(0);
86  if (firstPart.header == nullptr) {
87  continue;
88  }
89  const auto* firstInputHeader = DataRefUtils::getHeader<header::DataHeader*>(firstPart);
90  ConcreteDataMatcher inputMatcher{firstInputHeader->dataOrigin, firstInputHeader->dataDescription, firstInputHeader->subSpecification};
91 
92  for (auto& policy : mPolicies) {
93  // fixme: in principle matching could be broken by having query "TST/RAWDATA/0" and having parts with just
94  // the first subspec == 0, but others could be different. However, we trust that DPL does necessary checks
95  // during workflow validation and when passing messages (e.g. query "TST/RAWDATA/0" should not match
96  // a "TST/RAWDATA/*" output.
97  if (auto route = policy->match(inputMatcher); route != nullptr && policy->decide(firstPart)) {
98  auto routeAsConcreteDataType = DataSpecUtils::asConcreteDataTypeMatcher(*route);
99  auto dsheader = prepareDataSamplingHeader(*policy);
100  for (const auto& part : inputIt) {
101  if (part.header != nullptr) {
102  // We copy every header which is not DataHeader or DataProcessingHeader,
103  // so that custom data-dependent headers are passed forward,
104  // and we add a DataSamplingHeader.
105  header::Stack headerStack{
106  std::move(extractAdditionalHeaders(part.header)),
107  dsheader};
108  const auto* partInputHeader = DataRefUtils::getHeader<header::DataHeader*>(part);
109 
110  Output output{
111  routeAsConcreteDataType.origin,
112  routeAsConcreteDataType.description,
113  partInputHeader->subSpecification,
114  part.spec->lifetime,
115  std::move(headerStack)};
116  send(ctx.outputs(), part, output);
117  }
118  }
119  }
120  }
121  }
122 
123  if (ctx.inputs().isValid("timer-stats")) {
124  reportStats(ctx.services().get<Monitoring>());
125  }
126 }

128 void Dispatcher::reportStats(Monitoring& monitoring) const
129 {
130  uint64_t dispatcherTotalEvaluatedMessages = 0;
131  uint64_t dispatcherTotalAcceptedMessages = 0;
132 
133  for (const auto& policy : mPolicies) {
134  dispatcherTotalEvaluatedMessages += policy->getTotalEvaluatedMessages();
135  dispatcherTotalAcceptedMessages += policy->getTotalAcceptedMessages();
136  }
137 
138  monitoring.send(Metric{dispatcherTotalEvaluatedMessages, "Dispatcher_messages_evaluated"}.addTag(tags::Key::Subsystem, tags::Value::DataSampling));
139  monitoring.send(Metric{dispatcherTotalAcceptedMessages, "Dispatcher_messages_passed"}.addTag(tags::Key::Subsystem, tags::Value::DataSampling));
140 }

153 header::Stack Dispatcher::extractAdditionalHeaders(const char* inputHeaderStack) const
154 {
155  header::Stack headerStack;
156 
157  const auto* first = header::BaseHeader::get(reinterpret_cast<const std::byte*>(inputHeaderStack));
158  for (const auto* current = first; current != nullptr; current = current->next()) {
159  if (current->description != header::DataHeader::sHeaderType &&
160  current->description != DataProcessingHeader::sHeaderType) {
161  headerStack = std::move(header::Stack{std::move(headerStack), *current});
162  }
163  }
164 
165  return headerStack;
166 }

184 Inputs Dispatcher::getInputSpecs()
185 {
186  Inputs declaredInputs;
187 
188  // Add data inputs. Avoid duplicates and inputs which include others (e.g. "TST/DATA" includes "TST/DATA/1".
189  for (const auto& policy : mPolicies) {
190  for (const auto& path : policy->getPathMap()) {
191  auto& potentiallyNewInput = path.first;
192 
193  // The idea is that we remove all existing inputs which are covered by the potentially new input.
194  // If there are none which are broader than the new one, then we add it.
195  auto newInputIsBroader = [&potentiallyNewInput](const InputSpec& other) {
196  return DataSpecUtils::includes(potentiallyNewInput, other);
197  };
198  declaredInputs.erase(std::remove_if(declaredInputs.begin(), declaredInputs.end(), newInputIsBroader), declaredInputs.end());
199 
200  auto declaredInputIsBroader = [&potentiallyNewInput](const InputSpec& other) {
201  return DataSpecUtils::includes(other, potentiallyNewInput);
202  };
203  if (std::none_of(declaredInputs.begin(), declaredInputs.end(), declaredInputIsBroader)) {
204  declaredInputs.push_back(potentiallyNewInput);
205  }
206  }
207  }
208 
209  // add timer input
210  header::DataDescription timerDescription;
211  timerDescription.runtimeInit(("TIMER-" + mName).substr(0, 16).c_str());
212  declaredInputs.emplace_back(InputSpec{"timer-stats", "DS", timerDescription, 0, Lifetime::Timer});
213 
214  return declaredInputs;
215 }

217 Outputs Dispatcher::getOutputSpecs()
218 {
219  Outputs declaredOutputs;
220  for (const auto& policy : mPolicies) {
221  for (const auto& [_policyInput, policyOutput] : policy->getPathMap()) {
222  (void)_policyInput;
223  // In principle Data Sampling Policies should have different outputs.
224  // We may add a check to be very gentle with users.
225  declaredOutputs.push_back(policyOutput);
226  }
227  }
228  return declaredOutputs;
