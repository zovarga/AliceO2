102  template <typename T, typename... Args>
103  decltype(auto) make(const Output& spec, Args... args)
104  {
105  if constexpr (is_specialization_v<T, UninitializedVector>) {
106  // plain buffer as polymorphic spectator std::vector, which does not run constructors / destructors
107  using ValueType = typename T::value_type;
108  auto& timingInfo = mRegistry->get<TimingInfo>();
109  std::string const& channel = matchDataHeader(spec, timingInfo.timeslice);
110  auto& context = mRegistry->get<MessageContext>();
111 
112  // Note: initial payload size is 0 and will be set by the context before sending
113  FairMQMessagePtr headerMessage = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodNone, 0);
114  return context.add<MessageContext::VectorObject<ValueType, MessageContext::ContainerRefObject<std::vector<ValueType, o2::pmr::NoConstructAllocator<ValueType>>>>>(
115  std::move(headerMessage), channel, 0, std::forward<Args>(args)...)
116  .get();
117  } else if constexpr (is_specialization_v<T, std::vector> && has_messageable_value_type<T>::value) {
118  // this catches all std::vector objects with messageable value type before checking if is also
119  // has a root dictionary, so non-serialized transmission is preferred
120  using ValueType = typename T::value_type;
121  auto& timingInfo = mRegistry->get<TimingInfo>();
122  std::string const& channel = matchDataHeader(spec, timingInfo.timeslice);
123  auto& context = mRegistry->get<MessageContext>();
124 
125  // Note: initial payload size is 0 and will be set by the context before sending
126  FairMQMessagePtr headerMessage = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodNone, 0);
127  return context.add<MessageContext::VectorObject<ValueType>>(std::move(headerMessage), channel, 0, std::forward<Args>(args)...).get();
128  } else if constexpr (has_root_dictionary<T>::value == true && is_messageable<T>::value == false) {
129  // Extended support for types implementing the Root ClassDef interface, both TObject
130  // derived types and others
131  auto& timingInfo = mRegistry->get<TimingInfo>();
132  std::string const& channel = matchDataHeader(spec, timingInfo.timeslice);
133  auto& context = mRegistry->get<MessageContext>();
134 
135  // Note: initial payload size is 0 and will be set by the context before sending
136  FairMQMessagePtr headerMessage = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodROOT, 0);
137  return context.add<MessageContext::RootSerializedObject<T>>(std::move(headerMessage), channel, std::forward<Args>(args)...).get();
138  } else if constexpr (std::is_base_of_v<std::string, T>) {
139  std::string* s = new std::string(args...);
140  adopt(spec, s);
141  return *s;
142  } else if constexpr (std::is_base_of_v<struct TableBuilder, T>) {
143  TableBuilder* tb = nullptr;
144  call_if_defined<struct TableBuilder>([&](auto* p) {
145  tb = new std::decay_t<decltype(*p)>(args...);
146  adopt(spec, tb);
147  });
148  return *tb;
149  } else if constexpr (std::is_base_of_v<struct TreeToTable, T>) {
150  void* t2tr = nullptr;

152  auto t2t = new std::decay_t<decltype(*p)>(args...);
153  adopt(spec, t2t);
154  t2tr = t2t;
155  });
156  return *reinterpret_cast<TreeToTable*>(t2tr);
157  } else if constexpr (sizeof...(Args) == 0) {
158  constexpr bool isBoostSerializable = framework::is_boost_serializable<T>::value;
159  if constexpr (is_messageable<T>::value == true) {
160  return *reinterpret_cast<T*>(newChunk(spec, sizeof(T)).data());
161  } else if constexpr (is_specialization_v<T, BoostSerialized> == true) {
162  return make_boost<typename T::wrapped_type>(std::move(spec));
163  } else if constexpr (is_specialization_v<T, BoostSerialized> == false && isBoostSerializable == true && std::is_base_of<std::string, T>::value == false) {
164  return make_boost<T>(std::move(spec));
165  } else {
166  static_assert(always_static_assert_v<T>, ERROR_STRING);
167  }
168  } else if constexpr (sizeof...(Args) == 1) {
169  using FirstArg = typename std::tuple_element<0, std::tuple<Args...>>::type;
170  if constexpr (std::is_integral_v<FirstArg>) {
171  if constexpr (is_messageable<T>::value == true) {

173  auto size = nElements * sizeof(T);
174  auto& timingInfo = mRegistry->get<TimingInfo>();
175  std::string const& channel = matchDataHeader(spec, timingInfo.timeslice);
176  auto& context = mRegistry->get<MessageContext>();

178  FairMQMessagePtr headerMessage = headerMessageFromOutput(spec, channel, o2::header::gSerializationMethodNone, size);
179  return context.add<MessageContext::SpanObject<T>>(std::move(headerMessage), channel, 0, nElements).get();

181  } else if constexpr (std::is_same_v<FirstArg, std::shared_ptr<arrow::Schema>>) {

189  return make_boost<FirstArg>(std::move(spec));

191  static_assert(always_static_assert_v<T>, ERROR_STRING);
192  }
193  } else {
194  static_assert(always_static_assert_v<T>, ERROR_STRING);
195  }

278  template <typename T>
279  void snapshot(const Output& spec, T const& object)
280  {
281  auto proxy = mRegistry->get<MessageContext>().proxy();
282  FairMQMessagePtr payloadMessage;
283  auto serializationType = o2::header::gSerializationMethodNone;
284  if constexpr (is_messageable<T>::value == true) {
285  // Serialize a snapshot of a trivially copyable, non-polymorphic object,
286  payloadMessage = proxy.createMessage(sizeof(T));
287  memcpy(payloadMessage->GetData(), &object, sizeof(T));
288 
289  serializationType = o2::header::gSerializationMethodNone;
290  } else if constexpr (is_specialization_v<T, std::vector> == true ||
291  (gsl::details::is_span<T>::value && has_messageable_value_type<T>::value)) {
292  using ElementType = typename std::remove_pointer<typename T::value_type>::type;
293  if constexpr (is_messageable<ElementType>::value) {
294  // Serialize a snapshot of a std::vector of trivially copyable, non-polymorphic elements
295  // Note: in most cases it is better to use the `make` function und work with the provided
296  // reference object
297  constexpr auto elementSizeInBytes = sizeof(ElementType);
298  auto sizeInBytes = elementSizeInBytes * object.size();
299  payloadMessage = proxy.createMessage(sizeInBytes);
300 
301  if constexpr (std::is_pointer<typename T::value_type>::value == false) {
302  // vector of elements
303  if (object.data() && sizeInBytes) {
304  memcpy(payloadMessage->GetData(), object.data(), sizeInBytes);
305  }
306  } else {
307  // serialize vector of pointers to elements
308  auto target = reinterpret_cast<unsigned char*>(payloadMessage->GetData());
309  for (auto const& pointer : object) {
310  memcpy(target, pointer, elementSizeInBytes);
311  target += elementSizeInBytes;
312  }
313  }
314 
315  serializationType = o2::header::gSerializationMethodNone;
316  } else if constexpr (has_root_dictionary<ElementType>::value) {
317  return snapshot(spec, ROOTSerialized<T const>(object));
318  } else {
319  static_assert(always_static_assert_v<T>,
320  "value type of std::vector not supported by API, supported types:"
321  "\n - messageable tyeps (trivially copyable, non-polymorphic structures)"
322  "\n - pointers to those"
323  "\n - types with ROOT dictionary and implementing ROOT ClassDef interface");
324  }
325  } else if constexpr (has_root_dictionary<T>::value == true || is_specialization_v<T, ROOTSerialized> == true) {
326  // Serialize a snapshot of an object with root dictionary
327  payloadMessage = proxy.createMessage();
328  if constexpr (is_specialization_v<T, ROOTSerialized> == true) {
329  // Explicitely ROOT serialize a snapshot of object.
330  // An object wrapped into type `ROOTSerialized` is explicitely marked to be ROOT serialized
331  // and is expected to have a ROOT dictionary. Availability can not be checked at compile time
332  // for all cases.
333  using WrappedType = typename T::wrapped_type;
334  static_assert(std::is_same<typename T::hint_type, const char>::value ||
335  std::is_same<typename T::hint_type, TClass>::value ||
336  std::is_void<typename T::hint_type>::value,
337  "class hint must be of type TClass or const char");
338 
339  const TClass* cl = nullptr;
340  if (object.getHint() == nullptr) {
341  // get TClass info by wrapped type
342  cl = TClass::GetClass(typeid(WrappedType));
343  } else if (std::is_same<typename T::hint_type, TClass>::value) {
344  // the class info has been passed directly
345  cl = reinterpret_cast<const TClass*>(object.getHint());
346  } else if (std::is_same<typename T::hint_type, const char>::value) {
347  // get TClass info by optional name
348  cl = TClass::GetClass(reinterpret_cast<const char*>(object.getHint()));
349  }
350  if (has_root_dictionary<WrappedType>::value == false && cl == nullptr) {
351  if (std::is_same<typename T::hint_type, const char>::value) {
352  throw runtime_error_f("ROOT serialization not supported, dictionary not found for type %s",
353  reinterpret_cast<const char*>(object.getHint()));
354  } else {
355  throw runtime_error_f("ROOT serialization not supported, dictionary not found for type %s",
356  typeid(WrappedType).name());
357  }
358  }
359  TMessageSerializer().Serialize(*payloadMessage, &object(), cl);
360  } else {
361  TMessageSerializer().Serialize(*payloadMessage, &object, TClass::GetClass(typeid(T)));
362  }
363  serializationType = o2::header::gSerializationMethodROOT;
364  } else {
365  static_assert(always_static_assert_v<T>,
366  "data type T not supported by API, \n specializations available for"
367  "\n - trivially copyable, non-polymorphic structures"
368  "\n - std::vector of messageable structures or pointers to those"
369  "\n - types with ROOT dictionary and implementing ROOT ClassDef interface");
370  }
371  addPartToContext(std::move(payloadMessage), spec, serializationType);
372  }

495 template <typename ContainerT>
496 DataAllocator::CacheId DataAllocator::adoptContainer(const Output& spec, ContainerT&& container, bool cache, header::SerializationMethod method)
497 {
498  // Find a matching channel, extract the message for it form the container
499  // and put it in the queue to be sent at the end of the processing
500  auto& timingInfo = mRegistry->get<TimingInfo>();
501  std::string const& channel = matchDataHeader(spec, timingInfo.timeslice);
502 
503  auto& context = mRegistry->get<MessageContext>();
504  FairMQMessagePtr payloadMessage = o2::pmr::getMessage(std::forward<ContainerT>(container), *context.proxy().getTransport(channel));
505 
506  FairMQMessagePtr headerMessage = headerMessageFromOutput(spec, channel, //
507  method, //
508  payloadMessage->GetSize() //
509  );
510 
511  CacheId cacheId{0}; //
512  if (cache) {
513  // The message will be shallow cloned in the cache. Since the
514  // clone is indistinguishable from the original, we can keep sending
515  // the original.
516  cacheId.value = context.addToCache(payloadMessage);
517  }
518 
519  context.add<MessageContext::TrivialObject>(std::move(headerMessage), std::move(payloadMessage), channel);
520  return cacheId;
521 }
