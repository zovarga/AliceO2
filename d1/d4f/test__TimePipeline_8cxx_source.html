66 void someDataProducerAlgorithm(ProcessingContext& ctx)
67 {
68  uint32_t index = ctx.services().get<ParallelContext>().index1D();
69  std::this_thread::sleep_for(std::chrono::seconds(1));
70  // Creates a new message of size collectionChunkSize which
71  // has "TPC" as data origin and "CLUSTERS" as data description.
72  auto& tpcClusters = ctx.outputs().make<FakeCluster>(Output{"TPC", "CLUSTERS", index}, collectionChunkSize);
73  int i = 0;
74 
75  for (auto& cluster : tpcClusters) {
76  assert(i < collectionChunkSize);
77  cluster.x = index;

79  cluster.z = i;

81  i++;
