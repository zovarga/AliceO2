36 template <typename TruthElement>
37 class ConstMCTruthContainer : public std::vector<char>
38 {
39  public:
40  // (unfortunately we need these constructors for DPL)
41  using std::vector<char>::vector;
42  ConstMCTruthContainer() = default;
43 
44  // const data access
45  // get individual const "view" container for a given data index
46  // the caller can't do modifications on this view
47  MCTruthHeaderElement const& getMCTruthHeader(uint32_t dataindex) const
48  {
49  return getHeaderStart()[dataindex];
50  }
51 
52  gsl::span<const TruthElement> getLabels(uint32_t dataindex) const
53  {
54  if (dataindex >= getIndexedSize()) {
55  return gsl::span<const TruthElement>();
56  }
57  const auto start = getMCTruthHeader(dataindex).index;
58  const auto labelsptr = getLabelStart();
59  return gsl::span<const TruthElement>(&labelsptr[start], getSize(dataindex));
60  }
61 
62  // return the number of original data indexed here
63  size_t getIndexedSize() const { return size() >= sizeof(FlatHeader) ? getHeader().nofHeaderElements : 0; }
64 
65  // return the number of labels managed in this container
66  size_t getNElements() const { return size() >= sizeof(FlatHeader) ? getHeader().nofTruthElements : 0; }
67 
68  private:
69  using FlatHeader = typename MCTruthContainer<TruthElement>::FlatHeader;
70 
71  size_t getSize(uint32_t dataindex) const
72  {
73  // calculate size / number of labels from a difference in pointed indices
74  const auto size = (dataindex < getIndexedSize() - 1)
75  ? getMCTruthHeader(dataindex + 1).index - getMCTruthHeader(dataindex).index
76  : getNElements() - getMCTruthHeader(dataindex).index;
77  return size;
78  }

83  TruthElement const* getLabelStart() const
84  {
85  auto* source = &(*this)[0];
86  auto flatheader = getHeader();
87  source += sizeof(FlatHeader);
88  const size_t headerSize = flatheader.sizeofHeaderElement * flatheader.nofHeaderElements;
89  source += headerSize;

91  }
92 
93  FlatHeader const& getHeader() const
94  {
95  const auto* source = &(*this)[0];

97  return flatheader;
98  }
