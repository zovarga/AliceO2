49 template <typename RDH>
50 bool DecoderBaseT<RDH>::processHBF()
51 {
52 
53 #ifdef DECODER_VERBOSE
54  if (mDecoderVerbose) {
55  std::cout << colorBlue
56  << "--- PROCESS HBF"
57  << colorReset
58  << std::endl;
59  }
60 #endif
61 
62  if (mDecoderBufferSize <= 0) {
63  std::cout << colorRed
64  << " got an empty buffer, do nothing"
65  << colorReset
66  << std::endl;
67  return true;
68  }
69 
70  mDecoderRDH = reinterpret_cast<const RDH*>(mDecoderPointer);
71  auto rdh = mDecoderRDH;
72 
73  int iterations = 0;
75  while (!RDHUtils::getStop(*rdh)) {
76  iterations++;
77  if (iterations > 5) {
78  return true;
79  }
80 #ifdef DECODER_VERBOSE
81  if (mDecoderVerbose) {
82  std::cout << colorBlue
83  << "--- RDH open/continue detected"
84  << colorReset
85  << std::endl;
86  o2::raw::RDHUtils::printRDH(*rdh);
87  }
88 #endif
89 
91  rdhHandler(rdh);
92 
93  auto headerSize = RDHUtils::getHeaderSize(*rdh);
94  auto memorySize = RDHUtils::getMemorySize(*rdh);
95  auto offsetToNext = RDHUtils::getOffsetToNext(*rdh);
96  auto drmPayload = memorySize - headerSize;
97 
98  bool isValidRDH = RDHUtils::checkRDH(*rdh, false);
99 
101  if (isValidRDH && drmPayload > 0) {
102  std::memcpy(mDecoderSaveBuffer + mDecoderSaveBufferDataSize, reinterpret_cast<const char*>(rdh) + headerSize, drmPayload);
103  mDecoderSaveBufferDataSize += drmPayload;
104  }
105 #ifdef DECODER_VERBOSE
106  else {
107  if (mDecoderVerbose) {
108  RDHUtils::checkRDH(*rdh); // verbose
109  }
110  }
111 #endif
112 
114  rdh = reinterpret_cast<const RDH*>(reinterpret_cast<const char*>(rdh) + offsetToNext);
115 
117  if (reinterpret_cast<const char*>(rdh) < mDecoderBuffer + mDecoderBufferSize) {
118  continue;
119  }
120 
122  return true;
123  }
124 
125 #ifdef DECODER_VERBOSE
126  if (mDecoderVerbose) {
127  std::cout << colorBlue
128  << "--- RDH close detected"
129  << colorReset
130  << std::endl;
131  o2::raw::RDHUtils::printRDH(*rdh);
132  }
133 #endif
134 
136  mDecoderPointer = reinterpret_cast<const uint32_t*>(mDecoderSaveBuffer);
137  mDecoderPointerMax = reinterpret_cast<const uint32_t*>(mDecoderSaveBuffer + mDecoderSaveBufferDataSize);
138  while (mDecoderPointer < mDecoderPointerMax) {
139  if (processDRM()) {
140  break;
141  }
142  }
143  mDecoderSaveBufferDataSize = 0;
144 
146  rdhHandler(rdh);
147 
148 #ifdef DECODER_VERBOSE
149  if (mDecoderVerbose) {
150  std::cout << colorBlue
151  << "--- END PROCESS HBF"
152  << colorReset
153  << std::endl;
154  }
155 #endif
156 
158  mDecoderPointer = reinterpret_cast<const uint32_t*>(reinterpret_cast<const char*>(rdh) + RDHUtils::getOffsetToNext(*rdh));
159 
161  if (reinterpret_cast<const char*>(mDecoderPointer) < mDecoderBuffer + mDecoderBufferSize) {
162  return false;
163  }
164 
166  return true;
167 }

169 template <typename RDH>
170 bool DecoderBaseT<RDH>::processDRM()
171 {
172 
173 #ifdef DECODER_VERBOSE
174  if (mDecoderVerbose) {
175  std::cout << colorBlue << "--- PROCESS DRM"
176  << colorReset
177  << std::endl;
178  }
179 #endif
180 
181  if ((*mDecoderPointer & 0x80000000) != 0x80000000) {
182 #ifdef DECODER_VERBOSE
183  if (mDecoderVerbose) {
184  printf(" %08x [ERROR] \n ", *mDecoderPointer);
185  }
186 #endif
187  return true;
188  }
189 
191  auto crateHeader = reinterpret_cast<const CrateHeader_t*>(mDecoderPointer);
192 #ifdef DECODER_VERBOSE
193  if (mDecoderVerbose) {
194  printf(" %08x CrateHeader (drmID=%d) \n ", *mDecoderPointer, crateHeader->drmID);
195  }
196 #endif
197  mDecoderPointer++;
198 
200  auto crateOrbit = reinterpret_cast<const CrateOrbit_t*>(mDecoderPointer);
201 #ifdef DECODER_VERBOSE
202  if (mDecoderVerbose) {
203  printf(" %08x CrateOrbit (orbit=0x%08x) \n ", *mDecoderPointer, crateOrbit->orbitID);
204  }
205 #endif
206  mDecoderPointer++;
207 
209  headerHandler(crateHeader, crateOrbit);
210 
211  while (true) {
212 
214  if (*mDecoderPointer & 0x80000000) {
215  auto crateTrailer = reinterpret_cast<const CrateTrailer_t*>(mDecoderPointer);
216 #ifdef DECODER_VERBOSE
217  if (mDecoderVerbose) {
218  printf(" %08x CrateTrailer (numberOfDiagnostics=%d, numberOfErrors=%d) \n ", *mDecoderPointer, crateTrailer->numberOfDiagnostics, crateTrailer->numberOfErrors);
219  }
220 #endif
221  mDecoderPointer++;
222  auto diagnostics = reinterpret_cast<const Diagnostic_t*>(mDecoderPointer);
223 #ifdef DECODER_VERBOSE
224  if (mDecoderVerbose) {
225  for (int i = 0; i < crateTrailer->numberOfDiagnostics; ++i) {
226  auto diagnostic = reinterpret_cast<const Diagnostic_t*>(mDecoderPointer + i);
227  printf(" %08x Diagnostic (slotId=%d) \n ", *(mDecoderPointer + i), diagnostic->slotID);
228  }
229  }
230 #endif
231  mDecoderPointer += crateTrailer->numberOfDiagnostics;
232  auto errors = reinterpret_cast<const Error_t*>(mDecoderPointer);
233 #ifdef DECODER_VERBOSE
234  if (mDecoderVerbose) {
235  for (int i = 0; i < crateTrailer->numberOfErrors; ++i) {
236  auto error = reinterpret_cast<const Error_t*>(mDecoderPointer + i);
237  printf(" %08x Error (slotId=%d) \n ", *(mDecoderPointer + i), error->slotID);
238  }
239  }
240 #endif
241  mDecoderPointer += crateTrailer->numberOfErrors;
242 
244  trailerHandler(crateHeader, crateOrbit, crateTrailer, diagnostics, errors);
245 
246  return false;
247  }
248 
250  auto frameHeader = reinterpret_cast<const FrameHeader_t*>(mDecoderPointer);
251 #ifdef DECODER_VERBOSE
252  if (mDecoderVerbose) {
253  printf(" %08x FrameHeader (numberOfHits=%d) \n ", *mDecoderPointer, frameHeader->numberOfHits);
254  }
255 #endif
256  mDecoderPointer++;
257  auto packedHits = reinterpret_cast<const PackedHit_t*>(mDecoderPointer);
258 #ifdef DECODER_VERBOSE
259  if (mDecoderVerbose) {
260  for (int i = 0; i < frameHeader->numberOfHits; ++i) {
261  auto packedHit = reinterpret_cast<const PackedHit_t*>(mDecoderPointer + 1);
262  printf(" %08x PackedHit (tdcID=%d) \n ", *(mDecoderPointer + 1), packedHit->tdcID);
263  packedHits++;
264  }
265  }
266 #endif
267  mDecoderPointer += frameHeader->numberOfHits;
268 
270  frameHandler(crateHeader, crateOrbit, frameHeader, packedHits);
271  }
272 
274  return false;
275 }
