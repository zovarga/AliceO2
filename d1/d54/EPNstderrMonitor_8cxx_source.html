114 void EPNMonitor::sendLog(const std::string& file, const std::string& message)
115 {
116  if (mInfoLoggerActive) {
117  mLoggerContext->setField(InfoLogger::InfoLoggerContext::FieldName::Facility, ("stderr/" + file).substr(0, 31));
118  mLoggerContext->setField(InfoLogger::InfoLoggerContext::FieldName::Run, mRunNUmber == 0 ? std::to_string(mRunNUmber) : "unspecified");
119  static const InfoLogger::InfoLogger::InfoLoggerMessageOption opt = {InfoLogger::InfoLogger::Severity::Error, 3, InfoLogger::InfoLogger::undefinedMessageOption.errorCode, InfoLogger::InfoLogger::undefinedMessageOption.sourceFile, InfoLogger::InfoLogger::undefinedMessageOption.sourceLine};
120  mLogger->log(opt, *mLoggerContext, "stderr: %s", message.c_str());
121  } else {
122  printf("stderr: %s: %s\n", file.c_str(), message.c_str());
123  }
124 }

126 void EPNMonitor::thread()
127 {
128  printf("EPN stderr Monitor active\n");
129 
130  int fd;
131  int wd;
132  static constexpr size_t BUFFER_SIZE = 64 * 1024;
133  std::vector<char> evt_buffer(BUFFER_SIZE);
134  std::vector<char> text_buffer(8192);
135  fd = inotify_init();
136  wd = inotify_add_watch(fd, mPath.c_str(), IN_CREATE);
137  if (fd < 0) {
138  throw std::runtime_error(std::string("Error initializing inotify ") + std::to_string(fd) + " " + std::to_string(wd));
139  }
140  pollfd pfd = {fd, POLLIN, 0};
141 
142  for (const auto& entry : std::filesystem::directory_iterator(mPath)) {
143  if (entry.is_regular_file()) {
144  check_add_file(entry.path().filename());
145  }
146  }
147 
148  auto lastTime = std::chrono::system_clock::now();
149  while (!mTerminate) {
150  if (poll(&pfd, 1, 50) > 0) {
151  int l = read(fd, evt_buffer.data(), BUFFER_SIZE);
152  if (l < 0) {
153  throw std::runtime_error(std::string("Error waiting for inotify event ") + std::to_string(l));
154  }
155  for (int i = 0; i < l; i += sizeof(inotify_event)) {
156  inotify_event* event = (inotify_event*)&evt_buffer[i];
157  if (event->len && (event->mask & IN_CREATE) && !(event->mask & IN_ISDIR)) {
158  check_add_file(event->name);
159  }
160  i += event->len;
161  }
162  }
163  auto curTime = std::chrono::system_clock::now();
164  if (std::chrono::duration_cast<std::chrono::milliseconds>(curTime - lastTime).count() >= 1000) {
165  char* ptr = text_buffer.data();
166  std::string line;
167  for (auto fit = mFiles.begin(); fit != mFiles.end(); fit++) {
168  auto& f = fit->second;
169  auto& file = f.file;
170  file.clear();
171  do {
172  std::getline(file, line);
173  if (line.size()) {
174  bool filterLine = false;
175  for (const auto& filter : mFilters) {
176  if (std::regex_search(line, filter)) {
177  filterLine = true;
178  break;
179  }
180  }
181  if (filterLine) {
182  continue;
183  }
184  f.nLines++;
185  f.nBytes += line.size();
186  nLines++;
187  nBytes += line.size();
188  if (f.nLines >= MAX_LINES_FILE || f.nBytes >= MAX_BYTES_FILE) {
189  sendLog(f.name, "Exceeded log size for process " + f.name + " (" + std::to_string(f.nLines) + " lines, " + std::to_string(f.nBytes) + " bytes), not reporting any more errors from this file...");
190  fit = mFiles.erase(fit);
191  break;
192  }
193  if (nLines >= MAX_LINES_TOTAL || nBytes >= MAX_BYTES_TOTAL) {
194  break;
195  }
196  sendLog(f.name, line);
197  }
198  } while (!file.eof());
199  }
200  lastTime = curTime;
201  }
202  if (nLines >= MAX_LINES_TOTAL || nBytes >= MAX_BYTES_TOTAL) {
203  sendLog("", "Max total stderr log size exceeded (" + std::to_string(nLines) + " lines, " + std::to_string(nBytes) + "), not sending any more stderr logs from this node...");
204  break;
205  }
206 
207  usleep(50000);
208  }
209 
210  inotify_rm_watch(fd, wd);
211  close(fd);
212 
213  printf("EPN stderr Monitor terminating\n");
214 }
