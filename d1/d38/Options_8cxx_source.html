51 bool Options::processCommandLine(int argc, char* argv[])
52 {
53  bool save = false;
54 
55  bpo::options_description eveOptions("o2Eve options");
56 
57  eveOptions.add_options()(
58  "help,h", "produce help message")(
59  "datafolder,d", bpo::value<decltype(this->mDataFolder)>()->default_value("./json"), "name of the data folder")(
60  "filename,f", bpo::value<decltype(this->mFileName)>()->default_value("data.root"), "name of the data file")(
61  "json,j", bpo::value<decltype(this->mJSON)>()->zero_tokens()->default_value(false), "use json files as a source")(
62  "memorylimit,m", bpo::value<decltype(this->mMemoryLimit)>()->default_value(-1), "memory usage limit (MB) - app will terminate if it is exceeded (pass -1 for no limit)")(
63  "online,o", bpo::value<decltype(this->mOnline)>()->zero_tokens()->default_value(false), "use online json files as a source")(
64  "optionsfilename,p", bpo::value<std::string>()->default_value("o2eve.json"), "name of the options file")(
65  "randomtracks,r", bpo::value<decltype(this->mRandomTracks)>()->zero_tokens()->default_value(false), "use random tracks")(
66  "saveddatafolder,s", bpo::value<decltype(this->mSavedDataFolder)>()->default_value(""), "name of the saved data folder")(
67  "hidedplgui", bpo::value<decltype(this->mHideDplGUI)>()->zero_tokens()->default_value(false), "hide DPL GUI when processing AODs")(
68  "aodconverter,a", bpo::value<decltype(this->mAODConverterPath)>()->default_value("o2-eve-aodconverter"), "AOD converter path");
69 
70  using namespace bpo::command_line_style;
71  auto style = (allow_short | short_allow_adjacent | short_allow_next | allow_long | long_allow_adjacent | long_allow_next | allow_sticky | allow_dash_for_short);
72  bpo::variables_map varmap;
73  try {
74  bpo::store(
75  bpo::command_line_parser(argc, argv)
76  .options(eveOptions)
77  .style(style)
78  .run(),
79  varmap);
80  } catch (std::exception const& e) {
81  LOGP(error, "error parsing options of {}: {}", argv[0], e.what());
82  exit(1);
83  }
84 
85  if (varmap.count("help")) {
86  LOG(info) << eveOptions << std::endl
87  << " default values are always taken from o2eve.json in current folder if present" << std::endl;
88  return false;
89  }
90 
91  this->mDataFolder = varmap["datafolder"].as<decltype(this->mDataFolder)>();
92  this->mFileName = varmap["filename"].as<decltype(this->mFileName)>();
93  this->mJSON = varmap["json"].as<decltype(this->mJSON)>();
94  this->mMemoryLimit = varmap["memorylimit"].as<decltype(this->mMemoryLimit)>();
95  this->mOnline = varmap["online"].as<decltype(this->mOnline)>();
96  auto optionsFileName = varmap["optionsfilename"].as<std::string>();
97  this->mRandomTracks = varmap["randomtracks"].as<decltype(this->mRandomTracks)>();
98  this->mSavedDataFolder = varmap["saveddatafolder"].as<decltype(this->mSavedDataFolder)>();
99  this->mHideDplGUI = varmap["hidedplgui"].as<decltype(this->mHideDplGUI)>();
100  this->mAODConverterPath = varmap["aodconverter"].as<decltype(this->mAODConverterPath)>();
101 
102  if (this->mOnline && varmap["datafolder"].defaulted()) {
103  LOGP(error, "If online mode is enabled, the --datafolder option must be specified!");
104  return false;
105  }
106 
107  if (save) {
108  this->saveToJSON("o2eve.json");
109  return false;
110  }
111 
112  return true;
113 }

115 bool Options::saveToJSON(std::string filename)
116 {
117  rapidjson::Value dataFolder;
118  rapidjson::Value savedDataFolder;
119  rapidjson::Value fileName;
120  rapidjson::Value json(rapidjson::kNumberType);
121  rapidjson::Value online(rapidjson::kNumberType);
122  rapidjson::Value randomTracks(rapidjson::kNumberType);
123 
124  dataFolder.SetString(rapidjson::StringRef(this->dataFolder().c_str()));
125  savedDataFolder.SetString(rapidjson::StringRef(this->savedDataFolder().c_str()));
126  fileName.SetString(rapidjson::StringRef(this->fileName().c_str()));
127  json.SetBool(this->json());
128  online.SetBool(this->online());
129  randomTracks.SetBool(this->randomTracks());
130 
131  rapidjson::Document tree(rapidjson::kObjectType);
132  rapidjson::Document::AllocatorType& allocator = tree.GetAllocator();
133  tree.AddMember("dataFolder", dataFolder, allocator);
134  tree.AddMember("savedDataFolder", savedDataFolder, allocator);
135  tree.AddMember("fileName", fileName, allocator);
136  tree.AddMember("json", json, allocator);
137  tree.AddMember("online", online, allocator);
138  tree.AddMember("randomTracks", randomTracks, allocator);
139 
140  rapidjson::StringBuffer buffer;
141  rapidjson::PrettyWriter<rapidjson::StringBuffer> writer(buffer);
142  tree.Accept(writer);
143 
144  std::ofstream out(filename);
145  out << buffer.GetString();
146  out.close();
147  return true;
148 }
