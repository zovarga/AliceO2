32 double FitFunctor::calculateDeltaAlphaSim(double vdFit, double laFit, double impactAng) const
33 {
34 
35  double trdAnodePlane = .0335; // distance of the TRD anode plane from the drift cathodes in [m]
36 
37  auto xDir = TMath::Cos(impactAng);
38  auto yDir = TMath::Sin(impactAng);
39  double slope = (TMath::Abs(xDir) < 1e-7) ? 1e7 : yDir / xDir;
40  auto laTan = TMath::Tan(laFit);
41  double lorentzSlope = (TMath::Abs(laTan) < 1e-7) ? 1e7 : 1. / laTan;
42 
43  // hit point of incoming track with anode plane
44  double xAnodeHit = trdAnodePlane / slope;
45  double yAnodeHit = trdAnodePlane;
46 
47  // hit point at anode plane of Lorentz angle shifted cluster from the entrance -> independent of true drift velocity
48  double xLorentzAnodeHit = trdAnodePlane / lorentzSlope;
49  double yLorentzAnodeHit = trdAnodePlane;
50 
51  // cluster location within drift cell of cluster from entrance after drift velocity ratio is applied
52  double xLorentzDriftHit = xLorentzAnodeHit;
53  double yLorentzDriftHit = trdAnodePlane - trdAnodePlane * (vdPreCorr / vdFit);
54 
55  // reconstructed hit of first cluster at chamber entrance after pre Lorentz angle correction
56  double xLorentzDriftHitPreCorr = xLorentzAnodeHit - (trdAnodePlane - yLorentzDriftHit) * TMath::Tan(laPreCorr);
57  double yLorentzDriftHitPreCorr = trdAnodePlane - trdAnodePlane * (vdPreCorr / vdFit);
58 
59  double impactAngleSim = TMath::ATan2(yAnodeHit, xAnodeHit);
60 
61  double deltaXLorentzDriftHit = xAnodeHit - xLorentzDriftHitPreCorr;
62  double deltaYLorentzDriftHit = yAnodeHit - yLorentzDriftHitPreCorr;
63  double impactAngleRec = TMath::ATan2(deltaYLorentzDriftHit, deltaXLorentzDriftHit);
64 
65  double deltaAngle = (impactAngleRec - impactAngleSim); // * TMath::RadToDeg());
66 
67  return deltaAngle;
68 }

70 double FitFunctor::operator()(const double* par) const
71 {
72  double sum = 0; // this value is minimized
73  for (int iBin = 1; iBin <= profiles[currDet]->GetNbinsX(); ++iBin) {
74  auto impactAngle = (profiles[currDet]->GetBinCenter(iBin) + 90) * TMath::DegToRad();
75  auto deltaAlpha = profiles[currDet]->GetBinContent(iBin) * TMath::DegToRad();
76  if (TMath::Abs(deltaAlpha) < 1e-7) {
77  continue;
78  }
79  if (impactAngle < lowerBoundAngleFit || impactAngle > upperBoundAngleFit) {
80  continue;
81  }
82  auto deltaAlphaSim = calculateDeltaAlphaSim(par[CalibratorVdExB::ParamIndex::VD], par[CalibratorVdExB::ParamIndex::LA], impactAngle);
83  sum += TMath::Power(deltaAlphaSim - deltaAlpha, 2);
84  }
85  return sum;
86 }

96 void CalibratorVdExB::initProcessing()
97 {
98  if (mInitDone) {
99  return;
100  }
101  mFitFunctor.lowerBoundAngleFit = 80 * TMath::DegToRad();
102  mFitFunctor.upperBoundAngleFit = 100 * TMath::DegToRad();
103  mFitFunctor.vdPreCorr = 1.546; // TODO: will be taken from CCDB in the future
104  mFitFunctor.laPreCorr = 0.; // TODO: will be taken from CCDB in the future
105  for (int iDet = 0; iDet < MAXCHAMBER; ++iDet) {
106  mFitFunctor.profiles[iDet] = std::make_unique<TProfile>(Form("profAngleDiff_%i", iDet), Form("profAngleDiff_%i", iDet), NBINSANGLEDIFF, -MAXIMPACTANGLE, MAXIMPACTANGLE);
107  }
108  mInitDone = true;
109 }

111 void CalibratorVdExB::finalizeSlot(Slot& slot)
112 {
113  // do actual calibration for the data provided in the given slot
114  TStopwatch timer;
115  timer.Start();
116  std::array<float, MAXCHAMBER> laFitResults{};
117  std::array<float, MAXCHAMBER> vdFitResults{};
118  auto residHists = slot.getContainer();
119  initProcessing();
120  for (int iDet = 0; iDet < MAXCHAMBER; ++iDet) {
121  mFitFunctor.profiles[iDet]->Reset();
122  mFitFunctor.currDet = iDet;
123  for (int iBin = 0; iBin < NBINSANGLEDIFF; ++iBin) {
124  // fill profiles
125  auto angleDiffSum = residHists->getHistogramEntry(iDet * NBINSANGLEDIFF + iBin);
126  auto nEntries = residHists->getBinCount(iDet * NBINSANGLEDIFF + iBin);
127  if (nEntries > 0) {
128  mFitFunctor.profiles[iDet]->Fill(2 * iBin - MAXIMPACTANGLE, angleDiffSum / nEntries, nEntries);
129  }
130  }
131  ROOT::Fit::Fitter fitter;
132  double paramsStart[2];
133  paramsStart[ParamIndex::LA] = 0. * TMath::DegToRad();
134  paramsStart[ParamIndex::VD] = 1.;
135  fitter.SetFCN<FitFunctor>(2, mFitFunctor, paramsStart);
136  fitter.Config().ParSettings(ParamIndex::LA).SetLimits(-0.7, 0.7);
137  fitter.Config().ParSettings(ParamIndex::LA).SetStepSize(.01);
138  fitter.Config().ParSettings(ParamIndex::VD).SetLimits(0., 3.);
139  fitter.Config().ParSettings(ParamIndex::VD).SetStepSize(.01);
140  ROOT::Math::MinimizerOptions opt;
141  opt.SetMinimizerType("Minuit2");
142  opt.SetMinimizerAlgorithm("Migrad");
143  opt.SetPrintLevel(0);
144  opt.SetMaxFunctionCalls(1'000);
145  opt.SetTolerance(.001);
146  fitter.Config().SetMinimizerOptions(opt);
147  fitter.FitFCN();
148  auto fitResult = fitter.Result();
149  laFitResults[iDet] = fitResult.Parameter(ParamIndex::LA);
150  vdFitResults[iDet] = fitResult.Parameter(ParamIndex::VD);
151  LOGF(debug, "Fit result for chamber %i: vd=%f, la=%f", iDet, vdFitResults[iDet], laFitResults[iDet] * TMath::RadToDeg());
152  }
153  timer.Stop();
154  LOGF(info, "Done fitting angular residual histograms. CPU time: %f, real time: %f", timer.CpuTime(), timer.RealTime());
155 
156  // write results to CCDB
157  o2::ccdb::CcdbApi ccdb;
158  ccdb.init(o2::base::NameConf::getCCDBServer());
159  // ccdb.init("http://localhost:8080");
160  std::map<std::string, std::string> metadata; // TODO: do we want to store any meta data?
161  CalVdriftExB calObject;
162  for (int iDet = 0; iDet < MAXCHAMBER; ++iDet) {
163  // OS: what about chambers for which we had no data in this slot? should we use the initial parameters or something else?
164  // maybe it is better not to overwrite an older result if we don't have anything better?
165  calObject.setVdrift(iDet, vdFitResults[iDet]);
166  calObject.setExB(iDet, laFitResults[iDet]);
167  }
168  auto timeStamp = std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count();
169  auto timeStampEnd = timeStamp;
170  timeStampEnd += 1e3 * 60 * 60 * 24 * 7; // set validity of 7 days
171  ccdb.storeAsTFileAny(&calObject, "TRD/Calib/CalVdriftExB", metadata, timeStamp, timeStampEnd);
172 }
