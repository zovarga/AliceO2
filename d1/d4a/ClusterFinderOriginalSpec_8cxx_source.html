55  //_________________________________________________________________________________________________
56  void init(framework::InitContext& ic)
57  {
59  LOG(info) << "initializing cluster finder";
60 
61  auto config = ic.options().get<std::string>("mch-config");
62  if (!config.empty()) {
63  o2::conf::ConfigurableParam::updateFromFile(config, "MCHClustering", true);
64  }
65  bool run2Config = ic.options().get<bool>("run2-config");
66  mClusterFinder.init(run2Config);
67 
68  mAttachInitalPrecluster = ic.options().get<bool>("attach-initial-precluster");
69 
71  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, [this]() {
72  LOG(info) << "cluster finder duration = " << mTimeClusterFinder.count() << " s";
73  this->mClusterFinder.deinit();
74  });
75  }

77  //_________________________________________________________________________________________________
78  void run(framework::ProcessingContext& pc)
79  {
81 
82  // get the input preclusters and associated digits
83  auto preClusterROFs = pc.inputs().get<gsl::span<ROFRecord>>("preclusterrofs");
84  auto preClusters = pc.inputs().get<gsl::span<PreCluster>>("preclusters");
85  auto digits = pc.inputs().get<gsl::span<Digit>>("digits");
86 
87  // create the output messages for clusters and attached digits
88  auto& clusterROFs = pc.outputs().make<std::vector<ROFRecord>>(OutputRef{"clusterrofs"});
89  auto& clusters = pc.outputs().make<std::vector<Cluster>>(OutputRef{"clusters"});
90  auto& usedDigits = pc.outputs().make<std::vector<Digit>>(OutputRef{"clusterdigits"});
91 
92  clusterROFs.reserve(preClusterROFs.size());
93  for (const auto& preClusterROF : preClusterROFs) {
94 
95  // prepare to clusterize the current ROF
96  auto clusterOffset = clusters.size();
97  mClusterFinder.reset();
98 
99  for (const auto& preCluster : preClusters.subspan(preClusterROF.getFirstIdx(), preClusterROF.getNEntries())) {
100 
101  auto preclusterDigits = digits.subspan(preCluster.firstDigit, preCluster.nDigits);
102  auto firstClusterIdx = mClusterFinder.getClusters().size();
103 
104  // clusterize the current precluster
105  auto tStart = std::chrono::high_resolution_clock::now();
106  mClusterFinder.findClusters(preclusterDigits);
107  auto tEnd = std::chrono::high_resolution_clock::now();
108  mTimeClusterFinder += tEnd - tStart;
109 
110  if (mAttachInitalPrecluster) {
111  // store the new clusters and associate them to all the digits of the precluster
112  writeClusters(preclusterDigits, firstClusterIdx, clusters, usedDigits);
113  }
114  }
115 
116  if (!mAttachInitalPrecluster) {
117  // store all the clusters of the current ROF and the associated digits actually used in the clustering
118  writeClusters(clusters, usedDigits);
119  }
120 
121  // create the cluster ROF
122  clusterROFs.emplace_back(preClusterROF.getBCData(), clusterOffset, clusters.size() - clusterOffset,
123  preClusterROF.getBCWidth());
124  }
125 
126  LOGP(info, "Found {:4d} clusters from {:4d} preclusters in {:2d} ROFs",
127  clusters.size(), preClusters.size(), preClusterROFs.size());
128  }
129 
130  private:
131  //_________________________________________________________________________________________________
132  void writeClusters(const gsl::span<const Digit>& preclusterDigits, size_t firstClusterIdx,
133  std::vector<Cluster, o2::pmr::polymorphic_allocator<Cluster>>& clusters,
134  std::vector<Digit, o2::pmr::polymorphic_allocator<Digit>>& usedDigits) const
135  {
138 
139  if (firstClusterIdx == mClusterFinder.getClusters().size()) {
140  return;
141  }
142 
143  auto clusterOffset = clusters.size();
144  clusters.insert(clusters.end(), mClusterFinder.getClusters().begin() + firstClusterIdx, mClusterFinder.getClusters().end());
145 
146  auto digitOffset = usedDigits.size();
147  usedDigits.insert(usedDigits.end(), preclusterDigits.begin(), preclusterDigits.end());
148 
149  for (auto itCluster = clusters.begin() + clusterOffset; itCluster < clusters.end(); ++itCluster) {
150  itCluster->firstDigit = digitOffset;
151  itCluster->nDigits = preclusterDigits.size();
152  }
153  }
154 
155  //_________________________________________________________________________________________________
156  void writeClusters(std::vector<Cluster, o2::pmr::polymorphic_allocator<Cluster>>& clusters,
157  std::vector<Digit, o2::pmr::polymorphic_allocator<Digit>>& usedDigits) const
158  {
161 
162  auto clusterOffset = clusters.size();
163  clusters.insert(clusters.end(), mClusterFinder.getClusters().begin(), mClusterFinder.getClusters().end());
164 
165  auto digitOffset = usedDigits.size();
166  usedDigits.insert(usedDigits.end(), mClusterFinder.getUsedDigits().begin(), mClusterFinder.getUsedDigits().end());
167 
168  for (auto itCluster = clusters.begin() + clusterOffset; itCluster < clusters.end(); ++itCluster) {
169  itCluster->firstDigit += digitOffset;
170  }
171  }
172 
173  bool mAttachInitalPrecluster = false;
174  ClusterFinderOriginal mClusterFinder{};
175  std::chrono::duration<double> mTimeClusterFinder{};
176 };
