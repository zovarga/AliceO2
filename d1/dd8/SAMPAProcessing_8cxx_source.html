42 void SAMPAProcessing::getShapedSignal(float ADCsignal, float driftTime, std::vector<float>& signalArray) const
43 {
44  const float timeBinTime = getTimeBinTime(driftTime);
45  const float offset = driftTime - timeBinTime;
46  for (float bin = 0; bin < mEleParam->NShapedPoints; bin += Vc::float_v::Size) {
47  Vc::float_v binvector;
48  for (int i = 0; i < Vc::float_v::Size; ++i) {
49  binvector[i] = bin + i;

51  Vc::float_v time = timeBinTime + binvector * mEleParam->ZbinWidth;

53  for (int i = 0; i < Vc::float_v::Size; ++i) {
54  signalArray[bin + i] = signal[i];
