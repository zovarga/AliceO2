40 class DigitSamplerTask : public io::DigitIOBaseTask
41 {
42  private:
43  std::unique_ptr<io::DigitReader> mDigitReader;
44  std::ifstream mInput;
45  bool mReadIsOk = true;
46  size_t mMaxNofROFs;
47  size_t mNofProcessedROFs{0};
48  size_t mMinNumberOfROFsPerTF{1};
49  std::vector<ROFRecord> mROFs;
50  std::vector<Digit> mDigits;
51 
52  public:
53  void init(InitContext& ic)
54  {
55  io::DigitIOBaseTask::init(ic); // init common options
56  auto inputFileName = ic.options().get<std::string>(OPTNAME_INFILE);
57  mInput.open(inputFileName);
58  mDigitReader = std::make_unique<io::DigitReader>(mInput);
59  mNofProcessedTFs = 0;
60  mMaxNofROFs = ic.options().get<int>(OPTNAME_MAX_NOF_ROFS);
61  mMinNumberOfROFsPerTF = ic.options().get<int>(OPTNAME_REPACK_ROFS);
62  }
63 
64  void outputAndClear(DataAllocator& out)
65  {
66  printSummary(mDigits, mROFs, "-> to output");
67  out.snapshot(OutputRef{"rofs"}, mROFs);
68  out.snapshot(OutputRef{"digits"}, mDigits);
69  mDigits.clear();
70  mROFs.clear();
71  }
72 
73  bool shouldEnd() const
74  {
75  bool maxTFreached = mNofProcessedTFs >= mMaxNofTimeFrames;
76  bool maxROFreached = mNofProcessedROFs >= mMaxNofROFs;
77  return !mReadIsOk || maxTFreached || maxROFreached;
78  }
79 
80  void run(ProcessingContext& pc)
81  {
82  if (shouldEnd()) {
83  // output remaining data if any
84  if (mROFs.size() > 0) {
85  --mTFid;
86  outputAndClear(pc.outputs());
87  }
88  pc.services().get<ControlService>().endOfStream();
89  return;
90  }
91 
92  std::vector<ROFRecord> rofs;
93  std::vector<Digit> digits;
94  mReadIsOk = mDigitReader->read(digits, rofs);
95  if (!mReadIsOk) {
96  return;
97  }
98 
99  if (shouldProcess()) {
100  incNofProcessedTFs();
101  mNofProcessedROFs += rofs.size();
102  // append rofs to mROFs, but shift the indices by the amount of digits
103  // we have read so far.
104  auto offset = mDigits.size();
105  std::transform(rofs.begin(), rofs.end(), std::back_inserter(mROFs),
106  [offset](ROFRecord r) {
107  r.setDataRef(r.getFirstIdx() + offset, r.getNEntries());
108  return r;
109  });
110  mDigits.insert(mDigits.end(), digits.begin(), digits.end());
111  printSummary(mDigits, mROFs);
112  printFull(mDigits, mROFs);
113  }
114 
115  // output if we've accumulated enough ROFs
116  if (mROFs.size() >= mMinNumberOfROFsPerTF) {
117  outputAndClear(pc.outputs());
118  }
119 
120  incTFid();
121  }
122 };

124 o2::framework::DataProcessorSpec getDigitsFileReaderSpec(const char* specName, bool run2)
125 {
126  std::string spec = fmt::format("digits:MCH/DIGITS{}/0", run2 ? "R2" : "");
127  InputSpec itmp = o2::framework::select(spec.c_str())[0];
128 
129  auto commonOptions = o2::mch::io::getCommonOptions();
130  auto options = Options{
131  {OPTNAME_INFILE, VariantType::String, "", {"input file name"}},
132  {OPTNAME_MAX_NOF_ROFS, VariantType::Int, std::numeric_limits<int>::max(), {"max number of ROFs to process"}},
133  {OPTNAME_REPACK_ROFS, VariantType::Int, 1, {"number of rofs to repack into a timeframe (aka min number of rofs per timeframe"}}};
134  options.insert(options.end(), commonOptions.begin(), commonOptions.end());
135 
136  return DataProcessorSpec{
137  specName,
138  Inputs{},
139  Outputs{{DataSpecUtils::asOutputSpec(itmp)},
140  OutputSpec{{"rofs"}, "MCH", "DIGITROFS", 0, Lifetime::Timeframe}},
141  AlgorithmSpec{adaptFromTask<DigitSamplerTask>()},
142  options};
143 }
