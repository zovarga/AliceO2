66 class TPCFLPIDCDevice : public o2::framework::Task

68  public:
69  TPCFLPIDCDevice(const int lane, const std::vector<uint32_t>& crus, const unsigned int rangeIDC, const bool debug = false, const bool loadFromFile = false)
70  : mLane{lane}, mCRUs{crus}, mRangeIDC{rangeIDC}, mDebug{debug}, mLoadFromFile{loadFromFile}, mOneDIDCs(rangeIDC)
71  {
72  if constexpr (std::is_same_v<Type, TPCFLPIDCDeviceGroup>) {
73  auto& paramIDCGroup = ParameterIDCGroup::Instance();
74  for (const auto& cru : mCRUs) {
75  const CRU cruTmp(cru);
76  const unsigned int reg = cruTmp.region();
77  mIDCStruct.mIDCs.emplace(cru, IDCAverageGroup<IDCAverageGroupCRU>(paramIDCGroup.groupPads[reg], paramIDCGroup.groupRows[reg], paramIDCGroup.groupLastRowsThreshold[reg], paramIDCGroup.groupLastPadsThreshold[reg], paramIDCGroup.groupPadsSectorEdges, reg, cruTmp.sector()));
78  }
79  }
80 
81  for (const auto& cru : mCRUs) {
82  mBuffer1DIDCs.emplace(cru, std::deque<std::vector<float>>(std::ceil(static_cast<float>(mRangeIDC) / mMinIDCsPerTF), std::vector<float>(mMinIDCsPerTF, 0)));
83  }
84  }

86  void init(o2::framework::InitContext& ic) final
87  {
88  if constexpr (std::is_same_v<Type, TPCFLPIDCDeviceGroup>) {
89  if (mLoadFromFile) {
90  const char* fName = "IDCGroup.root";
91  TFile fInp(fName, "READ");
92  for (TObject* keyAsObj : *fInp.GetListOfKeys()) {
93  const auto key = dynamic_cast<TKey*>(keyAsObj);
94  const char* name = key->GetName();
95  LOGP(info, "Key name: {} Type: {}", name, key->GetClassName());
96 
97  if (std::strcmp(o2::tpc::IDCAverageGroup<IDCAverageGroupCRU>::Class()->GetName(), key->GetClassName()) != 0) {
98  LOGP(info, "skipping object. wrong class.");
99  continue;
100  }
101  IDCAverageGroup<IDCAverageGroupCRU>* idcavg = (IDCAverageGroup<IDCAverageGroupCRU>*)fInp.Get(name);
102  unsigned int cru = idcavg->getSector() * Mapper::NREGIONS + idcavg->getRegion();
103  // check cru
104  if (std::find(mCRUs.begin(), mCRUs.end(), cru) != mCRUs.end()) {
105  mIDCStruct.mIDCs[cru].processIDCs();
106  mIDCStruct.mIDCs[cru].setFromFile(fName, name);
107  }
108  delete idcavg;
109  }
110  }
111  }
112  }

114  void run(o2::framework::ProcessingContext& pc) final
115  {
116  if constexpr (std::is_same_v<Type, TPCFLPIDCDeviceGroup>) {
117  LOGP(info, "averaging and grouping IDCs for TF {} for CRUs {} to {} using {} threads", getCurrentTF(pc), mCRUs.front(), mCRUs.back(), mIDCStruct.mIDCs.begin()->second.getNThreads());
118  } else {
119  LOGP(info, "skipping grouping of IDCs for TF {} for CRUs {} to {}", getCurrentTF(pc), mCRUs.front(), mCRUs.back());
120  }
121 
122  if (!mLoadFromFile) {
123  for (int i = 0; i < mCRUs.size(); ++i) {
124  const DataRef ref = pc.inputs().getByPos(i);
125  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
126  const int cru = tpcCRUHeader->subSpecification >> 7;
127  if constexpr (std::is_same_v<Type, TPCFLPIDCDeviceGroup>) {
128  mIDCStruct.mIDCs[cru].setIDCs(pc.inputs().get<std::vector<float>>(ref));
129  mIDCStruct.mIDCs[cru].processIDCs();
130  } else {
131  mIDCStruct.mIDCs[cru] = pc.inputs().get<std::vector<float>>(ref);
132  }
133 
134  // send the output for one CRU for one TF
135  sendOutput(pc.outputs(), cru);
136  }
137  } else {
138  for (int i = 0; i < mCRUs.size(); ++i) {
139  sendOutput(pc.outputs(), mCRUs[i]);
140  }
141  }
142 
143  if (mDebug) {
144  TFile fOut(fmt::format("IDCGroup_{}_tf_{}.root", mLane, getCurrentTF(pc)).data(), "RECREATE");
145  for (int i = 0; i < mCRUs.size(); ++i) {
146  const DataRef ref = pc.inputs().getByPos(i);
147  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
148  const int cru = tpcCRUHeader->subSpecification >> 7;
149  fOut.WriteObject(&mIDCStruct.mIDCs[cru], fmt::format("CRU_{}", cru).data());
150  }
151  }
152  }

189  void sendOutput(DataAllocator& output, const uint32_t cru)
190  {
191  const header::DataHeader::SubSpecificationType subSpec{cru << 7};
192 
193  std::vector<float> vec1DIDCs;
194  if constexpr (std::is_same_v<Type, TPCFLPIDCDeviceGroup>) {
195  // calculate 1D-IDCs = sum over 2D-IDCs as a function of the integration interval
196  vec1DIDCs = mIDCStruct.mIDCs[cru].getIDCGroup().get1DIDCs();
197  } else {
198  CRU cruTmp(cru);
199  vec1DIDCs = IDCGroup::get1DIDCsUngrouped(mIDCStruct.mIDCs[cru], cruTmp.region());
200  }
201  output.snapshot(Output{gDataOriginTPC, getDataDescription1DIDC(), subSpec, Lifetime::Timeframe}, vec1DIDCs);
202 
203  // TODO use this and fix #include <boost/container/pmr/polymorphic_allocator.hpp> in ROOT CINT
204  // output.adoptContainer(Output{gDataOriginTPC, getDataDescriptionIDCGroup(), subSpec, Lifetime::Timeframe}, std::move(mIDCs[cru]).getIDCGroupData());
205  output.snapshot(Output{gDataOriginTPC, getDataDescriptionIDCGroup(), subSpec, Lifetime::Timeframe}, mIDCStruct.getData(cru));
206 
207  mBuffer1DIDCs[cru].emplace_back(std::move(vec1DIDCs));
208  mBuffer1DIDCs[cru].pop_front(); // removing oldest 1D-IDCs
209 
210  fill1DIDCs(cru);
211  output.snapshot(Output{gDataOriginTPC, getDataDescription1DIDCEPN(), subSpec, Lifetime::Timeframe}, mOneDIDCs);
212  }

214  void fill1DIDCs(const uint32_t cru)
215  {
216  // fill 1D-IDC vector with mRangeIDC buffered values
217  unsigned int i = mRangeIDC;
218  for (auto it = mBuffer1DIDCs[cru].crbegin(); it != mBuffer1DIDCs[cru].crend(); ++it) {
219  for (auto idc = it->crbegin(); idc != it->crend(); ++idc) {
220  mOneDIDCs[--i] = *idc;
221  if (i == 0) {
222  return;
223  }
224  }
225  }
226  }

229 template <class Type>
230 DataProcessorSpec getTPCFLPIDCSpec(const int ilane, const std::vector<uint32_t>& crus, const unsigned int rangeIDC, const bool debug, const bool loadFromFile)
231 {
232  std::vector<OutputSpec> outputSpecs;
233  std::vector<InputSpec> inputSpecs;
234  outputSpecs.reserve(crus.size());
235  inputSpecs.reserve(crus.size());
236 
237  for (const auto& cru : crus) {
238  const header::DataHeader::SubSpecificationType subSpec{cru << 7};
239  if (!loadFromFile) {
240  inputSpecs.emplace_back(InputSpec{"idcs", gDataOriginTPC, TPCIntegrateIDCDevice::getDataDescription(TPCIntegrateIDCDevice::IDCFormat::Sim), subSpec, Lifetime::Timeframe});
241  }
242  outputSpecs.emplace_back(ConcreteDataMatcher{gDataOriginTPC, TPCFLPIDCDevice<Type>::getDataDescriptionIDCGroup(), subSpec});
243  outputSpecs.emplace_back(ConcreteDataMatcher{gDataOriginTPC, TPCFLPIDCDevice<Type>::getDataDescription1DIDC(), subSpec});
244  outputSpecs.emplace_back(ConcreteDataMatcher{gDataOriginTPC, TPCFLPIDCDevice<Type>::getDataDescription1DIDCEPN(), subSpec});
245  }
246 
247  const auto id = fmt::format("tpc-flp-idc-{:02}", ilane);
248  return DataProcessorSpec{
249  id.data(),
250  inputSpecs,
251  outputSpecs,
252  AlgorithmSpec{adaptFromTask<TPCFLPIDCDevice<Type>>(ilane, crus, rangeIDC, debug, loadFromFile)},
253  }; // end DataProcessorSpec
254 }
