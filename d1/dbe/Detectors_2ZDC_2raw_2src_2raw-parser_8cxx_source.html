45 WorkflowSpec defineDataProcessing(ConfigContext const& config)
46 {
47  WorkflowSpec workflow;
48  o2::conf::ConfigurableParam::updateFromString(config.options().get<std::string>("configKeyValues"));
49  workflow.emplace_back(DataProcessorSpec{
50  "zdc-raw-parser",
51  select(config.options().get<std::string>("input-spec").c_str()),
52  Outputs{},
53  AlgorithmSpec{[](InitContext& setup) {
54  auto loglevel = setup.options().get<int>("log-level");
55  return adaptStateless([loglevel](InputRecord& inputs, DataAllocator& outputs) {
56  o2::zdc::DumpRaw zdc_dr;
57  zdc_dr.init();
58  zdc_dr.setVerbosity(loglevel);
59  DPLRawParser parser(inputs);
60  o2::header::DataHeader const* lastDataHeader = nullptr;
61  std::stringstream rdhprintout;
62  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
63  // retrieving RDH v4
64  auto const* rdh = it.get_if<o2::header::RAWDataHeaderV4>();
65  // retrieving the raw pointer of the page
66  auto const* raw = it.raw();
67  // retrieving payload pointer of the page
68  auto const* payload = it.data();
69  // size of payload
70  size_t payloadSize = it.size();
71  // offset of payload in the raw page
72  size_t offset = it.offset();
73  // Note: the following code is only for printing out raw page information
74  const auto* dh = it.o2DataHeader();
75  if (loglevel > 0) {
76  if (dh != lastDataHeader) {
77  // print the DataHeader information only for the first part or if we have high verbosity
78  if (loglevel > 1 || dh->splitPayloadIndex == 0) {
79  rdhprintout << fmt::format("{}/{}/{}", dh->dataOrigin, dh->dataDescription, dh->subSpecification)
80  << " ";
81  // at high verbosity print part number, otherwise only the total number of parts
82  if (loglevel > 1) {
83  rdhprintout << "part " + std::to_string(dh->splitPayloadIndex) + " of " + std::to_string(dh->splitPayloadParts);
84  } else {
85  rdhprintout << " " + std::to_string(dh->splitPayloadParts) + " part(s)";
86  }
87  rdhprintout << " payload size " << dh->payloadSize << std::endl;
88  }
89  if (!rdhprintout.str().empty()) {
90  LOG(info) << rdhprintout.str();
91  rdhprintout.str(std::string());
92  }
93  }
94  if (payload != nullptr) {
95  for (int32_t ip = 0; ip < payloadSize; ip += 16) {
96  //o2::zdc::Digits2Raw::print_gbt_word((const uint32_t*)&payload[ip]);
97  zdc_dr.processWord((const uint32_t*)&payload[ip]);
98  }
99  }
100  }
101  lastDataHeader = dh;
102  }
103  if (loglevel > 0) {
104  LOG(info) << rdhprintout.str();
105  }
106  zdc_dr.write();
107  }); }},
108  Options{
109  {"log-level", VariantType::Int, 1, {"Logging level [0-2]"}}}});
110  return workflow;
111 }
