41 // a helper class to do the checking at the end of the program when
42 // the destructor of the class is called.
43 class StaticChecker
44 {
45  public:
46  StaticChecker() = default;
47  // have to throw an exception in the destructor if checking fails
48  // this is ok in this cae because no other instances which would require proper
49  // cleanup are expected
50  ~StaticChecker() noexcept(false)
51  {
52  // the check in the desctructor makes sure that the workflow has been run at all
53  if (mChecks.size() > 0) {
54  throw std::runtime_error("Workflow error: Checks have not been executed");
55  }
56  }
57 
58  struct Attributes {
59  std::string fileName;
60  int nEntries = 0;
61  int nBranches = 0;
62  };
63 
64  void runChecks()
65  {
66  for (auto const& check : mChecks) {
67  TFile* file = TFile::Open(check.fileName.c_str());
68  if (file == nullptr) {
69  setError(std::string("missing file ") + check.fileName.c_str());
70  continue;
71  }
72  TTree* tree = reinterpret_cast<TTree*>(file->GetObjectChecked("testtree", "TTree"));
73  if (tree == nullptr) {
74  setError(std::string("can not find tree 'testtree' in file ") + check.fileName.c_str());
75  } else if (tree->GetEntries() != check.nEntries) {
76  setError(std::string("inconsistent number of entries in 'testtree' of file ") + check.fileName + " expecting " + std::to_string(check.nEntries) + " got " + std::to_string(tree->GetEntries()));
77  } else if (tree->GetNbranches() != check.nBranches) {
78  setError(std::string("inconsistent number of branches in 'testtree' of file ") + check.fileName + " expecting " + std::to_string(check.nBranches) + " got " + std::to_string(tree->GetNbranches()));
79  }
80  file->Close();
81  std::filesystem::remove(check.fileName.c_str());
82  }
83  mChecks.clear();
84  if (mErrorMessage.empty() == false) {
85  throw std::runtime_error(mErrorMessage);
86  }
87  }
88 
89  void addCheck(std::string filename, int entries, int branches = 0)
90  {
91  mChecks.emplace_back(Attributes{filename, entries, branches});
92  }
93 
94  template <typename T>
95  void setError(T const& message)
96  {
97  if (mErrorMessage.empty()) {
98  mErrorMessage = message;
99  }
100  }
101 
102  void clear()
103  {
104  mChecks.clear();
105  mErrorMessage.clear();
106  }
107 
108  private:
109  std::vector<Attributes> mChecks;
110  std::string mErrorMessage;
111 };

130 static constexpr int sTreeSize = 10; // elements to send and write
131 DataProcessorSpec getSourceSpec()
132 {
133  auto initFct = [](InitContext& ic) {
134  auto counter = std::make_shared<int>();
135  *counter = 0;
136 
137  auto processingFct = [counter](ProcessingContext& pc) {
138  if (*counter >= sTreeSize) {
139  // don't publish more
140  return;
141  }
142  o2::test::Polymorphic a(*counter);
143  pc.outputs().snapshot(OutputRef{"output", 0}, a);
144  pc.outputs().snapshot(OutputRef{"output", 1}, a);
145  int& metadata = pc.outputs().make<int>(Output{"TST", "METADATA", 0, Lifetime::Timeframe});
146  metadata = *counter;
147  *counter = *counter + 1;
148  if (*counter >= sTreeSize) {
149  pc.services().get<ControlService>().endOfStream();
150  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
151  }
152  };
153 
154  return processingFct;
155  };
156  return DataProcessorSpec{"source", // name of the processor
157  {},
158  {OutputSpec{{"output"}, "TST", "SOMEOBJECT", 0, Lifetime::Timeframe},
159  OutputSpec{{"output"}, "TST", "SOMEOBJECT", 1, Lifetime::Timeframe},
160  OutputSpec{{"meta"}, "TST", "METADATA", 0, Lifetime::Timeframe}},
161  AlgorithmSpec(initFct)};
162 }

173  // the first writer is configured with number of events (1)
174  // it receives two routes and saves those to two branches
175  // a third route is disabled (and not published by the source)
176  sChecker.addCheck(fileName, 1, 2);
177  // the second writer uses a check function to determine when its ready
178  // the first definition creates two branches, input data comes in over the
179  // same route, together with the second definition its three branches
180  sChecker.addCheck(altFileName, sTreeSize, 3);
181 
182  auto checkReady = [counter = std::make_shared<int>(0)](auto) -> bool {
183  *counter = *counter + 1;
184  // processing function checks the callback for two inputs -> factor 2
185  // the two last calls have to return true to signal on both inputs
186  return (*counter + 1) >= (2 * sTreeSize);
187  };
