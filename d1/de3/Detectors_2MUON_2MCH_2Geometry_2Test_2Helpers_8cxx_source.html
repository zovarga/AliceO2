34 TGeoVolume* createAirVacuumCave(const char* name)
35 {
36  // create the air medium (only used for the geometry test)
37  auto& mgr = o2::base::MaterialManager::Instance();
38 
39  const int nAir = 4;
40  Float_t aAir[nAir] = {12.0107, 14.0067, 15.9994, 39.948};
41  Float_t zAir[nAir] = {6., 7., 8., 18.};
42  Float_t wAir[nAir] = {0.000124, 0.755267, 0.231781, 0.012827};
43  Float_t dAirVacuum = 1.20479E-10;
44  const int kID = 90; // to avoid conflicts with definitions of other MCH materials
45 
46  mgr.Mixture("MCH", kID, "Air", aAir, zAir, dAirVacuum, nAir, wAir);
47  mgr.Medium("MCH", kID, "Air", kID,
48  false, /* isvol */
49  0, /* ifield */
50  -1.0, /* fieldm */
51  -1.0, /* tmaxfd */
52  -1.0, /* stemax */
53  -1.0, /* deemax */
54  -1.0, /* epsil */
55  -1.0 /* stmin */);
56  return gGeoManager->MakeBox(name, gGeoManager->GetMedium("MCH_Air"), 2000.0, 2000.0, 3000.0);
57 }

59 void dump(std::ostream& out, const TGeoNode& n, int level, int maxdepth, std::string prefix)
60 {
61  if (level >= maxdepth) {
62  return;
63  }
64 
65  if (level == 0) {
66  out << n.GetName() << "\n";
67  }
68 
69  if (level < maxdepth) {
70  for (int i = 0; i < n.GetNdaughters(); i++) {
71  TGeoNode* d = n.GetDaughter(i);
72  if (i == n.GetNdaughters() - 1) {
73  out << prefix + "└──" << d->GetName()
74  << "\n";
75  dump(out, *d, level + 1, maxdepth, prefix + " ");
76  } else {
77  out << prefix + "├──" << d->GetName()
78  << "\n";
79  dump(out, *d, level + 1, maxdepth, prefix + "│ ");
80  }
81  }
82  }
83 }

206 math_utils::Vector3D<double> getNormalVector(const o2::math_utils::Transform3D& t)
207 {
208  math_utils::Point3D<double> px, py, po;
209  t.LocalToMaster(math_utils::Point3D<double>{0, 1, 0}, py);
210  t.LocalToMaster(math_utils::Point3D<double>{1, 0, 0}, px);
211  t.LocalToMaster(math_utils::Point3D<double>{0, 0, 0}, po);
212  math_utils::Vector3D<double> a{px - po};
213  math_utils::Vector3D<double> b{py - po};
214  return a.Cross(b).Unit();
215 }

217 TH2* getRadio(int detElemId, float xmin, float ymin, float xmax, float ymax, float xstep, float ystep, float thickness)
218 {
219  if (xmin >= xmax || ymin >= ymax) {
220  std::cerr << "incorrect limits\n";
221  return nullptr;
222  }
223  TH2* hmatb = new TH2F("hmatb", "hmatb", (int)((xmax - xmin) / xstep), xmin, xmax, (int)((ymax - ymin) / ystep), ymin, ymax);
224 
225  auto transformation = o2::mch::geo::transformationFromTGeoManager(*gGeoManager);
226  auto t = transformation(detElemId);
227 
228  auto normal = getNormalVector(t);
229 
230  for (auto x = xmin; x < xmax; x += xstep) {
231  for (auto y = ymin; y < ymax; y += ystep) {
232  auto matb = getMatBudgetExt(t, normal, x, y, thickness);
233  if (std::isfinite(matb.meanX2X0)) {
234  hmatb->Fill(x, y, matb.meanX2X0);
235  }
236  }
237  }
238  return hmatb;
239 }
