28 // a structure describing EventPart
29 // (an elementary constituent of a collision)
30 
31 constexpr static int QEDSOURCEID = 99;
32 
33 struct EventPart {
34  EventPart() = default;
35  EventPart(int s, int e) : sourceID(s), entryID(e) {}
36  int sourceID = 0; // the ID of the source (0->backGround; > 1 signal source)
37  // the sourceID should correspond to the chain ID
38  int entryID = 0; // the event/entry ID inside the chain corresponding to sourceID
39 
40  static bool isSignal(EventPart e) { return e.sourceID > 1 && e.sourceID != QEDSOURCEID; }
41  static bool isBackGround(EventPart e) { return !isSignal(e); }
42  static bool isQED(EventPart e) { return e.sourceID == QEDSOURCEID; }
43  ClassDefNV(EventPart, 1);
44 };

141 inline void DigitizationContext::retrieveHits(std::vector<TChain*> const& chains,
142  const char* brname,
143  int sourceID,
144  int entryID,
145  std::vector<T>* hits) const
146 {
147  auto br = chains[sourceID]->GetBranch(brname);
148  if (!br) {
149  LOG(error) << "No branch found with name " << brname;
150  return;
151  }
152  br->SetAddress(&hits);
153  br->GetEntry(entryID);
154 }
