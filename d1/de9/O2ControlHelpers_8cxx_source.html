138 std::string_view extractValueFromChannelConfig(std::string_view string, std::string_view token)
139 {
140  size_t tokenStart = string.find(token);
141  if (tokenStart == std::string_view::npos) {
142  return {};
143  }
144  size_t valueStart = tokenStart + token.size();
145  if (valueStart >= string.size()) {
146  return {};
147  }
148  size_t valueEnd = string.find(',', valueStart);
149  return valueEnd == std::string_view::npos
150  ? string.substr(valueStart, string.size() - valueStart)
151  : string.substr(valueStart, valueEnd - valueStart);
152 }

157 std::vector<RawChannel> extractRawChannels(const DeviceSpec& spec, const DeviceExecution& execution)
158 {
159  std::vector<std::string> dplChannels;
160  for (const auto& channel : spec.inputChannels) {
161  dplChannels.emplace_back(channel.name);
162  }
163  for (const auto& channel : spec.outputChannels) {
164  dplChannels.emplace_back(channel.name);
165  }
166 
167  std::vector<RawChannel> rawChannels;
168  for (size_t i = 0; i < execution.args.size(); i++) {
169  if (execution.args[i] != nullptr && strcmp(execution.args[i], "--channel-config") == 0 && i + 1 < execution.args.size()) {
170  auto channelConfig = execution.args[i + 1];
171  auto channelName = extractValueFromChannelConfig(channelConfig, "name=");
172  if (std::find(dplChannels.begin(), dplChannels.end(), channelName) == dplChannels.end()) {
173  // "name=readout-proxy,type=pair,method=connect,address=ipc:///tmp/readout-pipe-0,rateLogging=1,transport=shmem"
174  rawChannels.push_back({channelName,
175  extractValueFromChannelConfig(channelConfig, "type="),
176  extractValueFromChannelConfig(channelConfig, "method="),
177  extractValueFromChannelConfig(channelConfig, "address="),
178  extractValueFromChannelConfig(channelConfig, "rateLogging="),
179  extractValueFromChannelConfig(channelConfig, "transport="),
180  extractValueFromChannelConfig(channelConfig, "sndBufSize="),
181  extractValueFromChannelConfig(channelConfig, "rcvBufSize=")});
182  }
183  }
184  }
185  return rawChannels;
186 }

245  dumpOut << indLevel << indScheme << "- \"--shm-throw-bad-alloc\"\n";

247  dumpOut << indLevel << indScheme << "- \"--resources-monitoring\"\n";
248  dumpOut << indLevel << indScheme << "- \"'{{ resources_monitoring }}'\"\n";
249 
250  for (size_t ai = 1; ai < execution.args.size(); ++ai) {
251  const char* option = execution.args[ai];
252  const char* value = nullptr; // no value by default (i.e. a boolean)
253  // If the subsequent option exists and does not start with -, we assume
254  // it is an argument to the previous one.
255  // ...that is unless it is a "-1" for example.
256  if (ai + 1 < execution.args.size() && execution.args[ai + 1][0] != '-') {
257  value = execution.args[ai + 1];
258  ai++;
259  }
260  if (!option) {
261  break;
262  }
263 
264  static const std::set<std::string> omitOptions = {
265  "--channel-config", "--o2-control", "--control", "--session", "--monitoring-backend",
266  "-b", "--color", "--infologger-severity", "--infologger-mode", "--driver-client-backend",
267  "--shm-segment-size", "--shm-throw-bad-alloc", "--resources-monitoring"};
268  if (omitOptions.find(option) != omitOptions.end()) {
269  continue;
270  }
271  // todo: possible improvement - do not print if default values are used
272  // todo: check if '' are there already.
273  dumpOut << indLevel << indScheme << R"(- ")" << option << "\"\n";
274  if (value) {
275  dumpOut << indLevel << indScheme << R"(- "')" << value << "'\"\n";

280 std::string findBinder(const std::vector<DeviceSpec>& specs, const std::string& channel)
281 {
282  // fixme: it is not crucial to be fast here, but ideally we should check only input OR output channels.
283  for (const auto& spec : specs) {
284  for (const auto& inputChannel : spec.inputChannels) {
285  if (inputChannel.method == ChannelMethod::Bind && inputChannel.name == channel) {
286  return spec.id;
287  }
288  }
289  for (const auto& outputChannel : spec.outputChannels) {
290  if (outputChannel.method == ChannelMethod::Bind && outputChannel.name == channel) {
291  return spec.id;
292  }
293  }
294  }
295  throw std::runtime_error("Could not find a device which binds the '" + channel + "' channel.");
296 }

298 void dumpRole(std::ostream& dumpOut, const std::string& taskName, const DeviceSpec& spec, const std::vector<DeviceSpec>& allSpecs, const DeviceExecution& execution, const std::string indLevel)
299 {
300  dumpOut << indLevel << "- name: \"" << spec.id << "\"\n";
301 
302  dumpOut << indLevel << indScheme << "connect:\n";
303 
304  for (const auto& outputChannel : spec.outputChannels) {
305  if (outputChannel.method == ChannelMethod::Connect) {
306  dumpChannelConnect(dumpOut, outputChannel, findBinder(allSpecs, outputChannel.name), indLevel + indScheme);
307  }
308  }
309  for (const auto& inputChannel : spec.inputChannels) {
310  if (inputChannel.method == ChannelMethod::Connect) {
311  dumpChannelConnect(dumpOut, inputChannel, findBinder(allSpecs, inputChannel.name), indLevel + indScheme);
312  }
313  }
314  bool uniqueProxy = isUniqueProxy(spec);
315  bool preserveRawChannels = shouldPreserveRawChannels(spec);
316  bool bindsRawChannels = false;
317  auto rawChannels = extractRawChannels(spec, execution);
318  for (const auto& rawChannel : rawChannels) {
319  if (rawChannel.method == "connect") {
320  dumpRawChannelConnect(dumpOut, rawChannel, uniqueProxy, preserveRawChannels, indLevel + indScheme);
321  } else if (rawChannel.method == "bind") {
322  bindsRawChannels = true;
323  }
324  }
325 
326  // for the time being we have to publish global bound channels also in WFT
327  if (bindsRawChannels) {
328  dumpOut << indLevel << indScheme << "bind:\n";
329  for (const auto& rawChannel : rawChannels) {
330  if (rawChannel.method == "bind") {
331  dumpRawChannelBind(dumpOut, rawChannel, uniqueProxy, preserveRawChannels, indLevel + indScheme);
332  }
333  }
334  }
335 
336  dumpOut << indLevel << indScheme << "task:\n";
337  dumpOut << indLevel << indScheme << indScheme << "load: " << taskName << "\n";
338 }

340 std::string removeO2ControlArg(std::string_view command)
341 {
342  const char* o2ControlArg = " --o2-control ";
343  size_t o2ControlArgStart = command.find(o2ControlArg);
344  if (o2ControlArgStart == std::string_view::npos) {
345  return std::string(command);
346  }
347  size_t o2ControlArgEnd = command.find(" ", o2ControlArgStart + std::strlen(o2ControlArg));
348  auto result = std::string(command.substr(0, o2ControlArgStart));
349  if (o2ControlArgEnd != std::string_view::npos) {
350  result += command.substr(o2ControlArgEnd);
351  }
352  return result;
353 }
