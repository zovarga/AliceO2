137 template <typename T>
138 void mergeHelper(const char* brprefix, std::vector<int> const& tpcsectors, uint64_t activeSectors,
139  TFile& originfile, framework::ProcessingContext& pc)
140 {
141  auto keyslist = originfile.GetListOfKeys();
142  for (int i = 0; i < keyslist->GetEntries(); ++i) {
143  auto key = keyslist->At(i);
144  int sector = atoi(key->GetName());
145  if (std::find(tpcsectors.begin(), tpcsectors.end(), sector) == tpcsectors.end()) {
146  // do nothing if sector not wanted
147  continue;
148  }
149 
150  auto oldtree = (TTree*)originfile.Get(key->GetName());
151  assert(oldtree);
152  std::stringstream digitbrname;
153  digitbrname << brprefix << key->GetName();
154  auto br = oldtree->GetBranch(digitbrname.str().c_str());
155  if (!br) {
156  continue;
157  }
158  T* chunk = nullptr;
159  br->SetAddress(&chunk);
160 
161  using AccumType = std::decay_t<decltype(makePublishBuffer<T>(pc, sector, activeSectors))>;
162  AccumType accum;
163 #pragma omp critical
164  accum = makePublishBuffer<T>(pc, sector, activeSectors);
165 
166  for (auto e = 0; e < br->GetEntries(); ++e) {
167  br->GetEntry(e);
168  copyHelper(*chunk, *accum);
169  delete chunk;
170  chunk = nullptr;
171  }
172  br->ResetAddress();
173  br->DropBaskets("all");
174  delete oldtree;
175 
176  // some data (labels are published slightly differently)
177  publishBuffer(pc, sector, activeSectors, accum);
178  }
179 }

181 void publishMergedTimeframes(std::vector<int> const& lanes, std::vector<int> const& tpcsectors, bool domctruth, framework::ProcessingContext& pc)
182 {
183  uint64_t activeSectors = 0;
184  for (auto s : tpcsectors) {
185  activeSectors |= (uint64_t)0x1 << s;
186  }
187 
188  ROOT::EnableThreadSafety();
189  // we determine the exact input list of files
190  auto digitfilelist = o2::utils::listFiles("tpc_driftime_digits_lane.*.root$");
191 #ifdef WITH_OPENMP
192  omp_set_num_threads(std::min(lanes.size(), digitfilelist.size()));
193  LOG(info) << "Running digit publisher with OpenMP enabled";
194 #pragma omp parallel for schedule(dynamic)
195 #endif
196  for (size_t fi = 0; fi < digitfilelist.size(); ++fi) {
197  auto& filename = digitfilelist[fi];
198  LOG(debug) << "MERGING CHUNKED DIGITS FROM FILE " << filename;
199  auto originfile = new TFile(filename.c_str(), "OPEN");
200  assert(originfile);
201 
202  //data definitions
203  using DigitsType = std::vector<o2::tpc::Digit>;
204  using LabelType = o2::dataformats::MCTruthContainer<o2::MCCompLabel>;
205  mergeHelper<DigitsType>("TPCDigit_", tpcsectors, activeSectors, *originfile, pc);
206  if (domctruth) {
207  mergeHelper<LabelType>("TPCDigitMCTruth_", tpcsectors, activeSectors, *originfile, pc);
208  }
209  originfile->Close();
210  delete originfile;
211  }

249 DataProcessorSpec getSpec(std::vector<int> const& laneConfiguration, std::vector<int> const& tpcsectors, bool mctruth, bool publish = true)
250 {
251  //data definitions
252  using DigitsOutputType = std::vector<o2::tpc::Digit>;
253  using CommonModeOutputType = std::vector<o2::tpc::CommonMode>;
254 
255  std::vector<OutputSpec> outputs; // define channel by triple of (origin, type id of data to be sent on this channel, subspecification)
256  if (publish) {
257  // effectively the input expects one sector per subspecification
258  for (int s = 0; s < 36; ++s) {
259  OutputLabel binding{std::to_string(s)};
260  outputs.emplace_back(/*binding,*/ "TPC", "DIGITS", static_cast<SubSpecificationType>(s), Lifetime::Timeframe);
261  if (mctruth) {
262  outputs.emplace_back(/*binding,*/ "TPC", "DIGITSMCTR", static_cast<SubSpecificationType>(s), Lifetime::Timeframe);
263  }
264  }
265  }
266 
267  return DataProcessorSpec{
268  "TPCDigitMerger", {}, outputs, AlgorithmSpec{o2::framework::adaptFromTask<Task>(laneConfiguration, tpcsectors, mctruth)}, Options{}};
269 }
