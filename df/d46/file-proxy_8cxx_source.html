90 long FileProxyTask::readFLPv4()
91 {
93  char* pointer = mBuffer;
94  if (!mFile.read(pointer, 64)) {
95  std::cout << " --- Cannot read input file: " << strerror(errno) << std::endl;
96  mStatus = true;
97  return 0;
98  }
99  long payload = 64;
100  auto rdh = reinterpret_cast<o2::header::RAWDataHeaderV4*>(pointer);
101  while (!rdh->stop) {
102  if (!mFile.read(pointer + rdh->headerSize, rdh->offsetToNext - rdh->headerSize)) {
103  std::cout << " --- Cannot read input file: " << strerror(errno) << std::endl;
104  mStatus = true;
105  return 0;
106  }
107  payload += (rdh->offsetToNext - rdh->headerSize);
108  pointer += rdh->offsetToNext;
109  if (!mFile.read(pointer, 64)) {
110  std::cout << " --- Cannot read input file: " << strerror(errno) << std::endl;
111  mStatus = true;
112  return 0;
113  }
114  payload += 64;
115  rdh = reinterpret_cast<o2::header::RAWDataHeaderV4*>(pointer);
116  }
117 
118  return payload;
119 }

152 long FileProxyTask::readCONET()
153 {
154 
155  uint32_t word;
156  auto current = mFile.tellg();
157 
159  if (!mFile.read((char*)&word, 4)) {
160  std::cout << " --- Cannot read input file: " << strerror(errno) << std::endl;
161  mStatus = true;
162  return 0;
163  }
164  auto bytePayload = word * 4;
165  std::cout << " --- tofBuffer: " << word << " (" << bytePayload << " bytes) at " << current << std::endl;
166 
168  if (!mFile.read(mBuffer, bytePayload)) {
169  std::cout << " --- Cannot read input file: " << strerror(errno) << std::endl;
170  mStatus = true;
171  return 0;
172  }
173  return bytePayload;
174 }

176 void FileProxyTask::run(ProcessingContext& pc)
177 {
178 
180  if (mStatus) {
181  mFile.close();
182  pc.services().get<ControlService>().endOfStream();
183  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
184  return;
185  }
186 
188  int payload = 0;
189  if (mCONET) {
190  payload = readCONET();
191  } else {
192  payload = readFLP();
193  }
194  if (payload == 0) {
195  mStatus = true;
196  return;
197  }
198 
199  if (mDumpData) {
200  std::cout << " --- dump data: " << payload << " bytes" << std::endl;
201  uint32_t* word = reinterpret_cast<uint32_t*>(mBuffer);
202  for (int i = 0; i < payload / 4; ++i) {
203  printf(" 0x%08x \n", *word);
204  word++;
205  }
206  std::cout << " --- end of dump data " << std::endl;
207  }
208 
210  auto device = pc.services().get<o2::framework::RawDeviceService>().device();
211  auto outputRoutes = pc.services().get<o2::framework::RawDeviceService>().spec().outputs;
212  auto fairMQChannel = outputRoutes.at(0).channel;
213  auto payloadMessage = device->NewMessage(payload);
214  std::memcpy(payloadMessage->GetData(), mBuffer, payload);
215  o2::header::DataHeader header("RAWDATA", "TOF", 0);
216  header.payloadSize = payload;
217  o2::framework::DataProcessingHeader dataProcessingHeader{0};
218  o2::header::Stack headerStack{header, dataProcessingHeader};
219  auto headerMessage = device->NewMessage(headerStack.size());
220  std::memcpy(headerMessage->GetData(), headerStack.data(), headerStack.size());
221 
223  FairMQParts parts;
224  parts.AddPart(std::move(headerMessage));
225  parts.AddPart(std::move(payloadMessage));
226  device->Send(parts, fairMQChannel);
227 
229  if (mFile.eof()) {
230  std::cout << " --- End of file " << std::endl;
231  mStatus = true;
232  }
233 };
