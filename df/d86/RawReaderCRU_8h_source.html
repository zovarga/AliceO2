842 inline void GBTFrame::updateSyncCheck(SyncArray& syncArray)
843 {
844  const auto offset = mPrevHWpos ^ 4;
845 
846  for (int s = 0; s < 5; s++) {
847  for (int h = 0; h < 4; h++) {
848  const auto hPos = h + offset; // set position of last filled HW
849  // shift in a 1 if the halfword is 0x15
850  if (mFrameHalfWords[s][hPos] == 0x15) {
851  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1) | 1;
852  // shift in a 0 if the halfword is 0xA
853  } else if (mFrameHalfWords[s][hPos] == 0xA) {
854  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1);
855  // otherwise reset the register to 0
856  } else {
857  mSyncCheckRegister[s] = 0;
858  }
859  // std::cout << " SReg : " << s << " : " << std::hex << mSyncCheckRegister[s] << std::endl;
860  // check the register content for the SYNC pattern
861  // Patterh is : 1100.1100.1100.1100.1111.0000.1111.0000 = 0xCCCCF0F0
862  // alternative Pattern for first corrupted SYNC word : 0100.1100.1100.1100.1111.0000.1111.0000 = 4CCC.F0F0
863  if (mSyncCheckRegister[s] == 0xCCCCF0F0 or mSyncCheckRegister[s] == 0x4CCCF0F0 or mSyncCheckRegister[s] == 0x0CCCF0F0) {
864  syncArray[s].setPos(mPacketNum, mFrameNum, mFilePos, h);
865  };
866  }
867  };
868 }

876  const auto hPos = h + offset; // set position of last filled HW
877  // shift in a 1 if the halfword is 0x15
878  if (mFrameHalfWords[s][hPos] == 0x15) {
879  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1) | 1;
880  // shift in a 0 if the halfword is 0xA
881  } else if (mFrameHalfWords[s][hPos] == 0xA) {
882  mSyncCheckRegister[s] = (mSyncCheckRegister[s] << 1);
883  // otherwise reset the register to 0
884  } else {
885  mSyncCheckRegister[s] = 0;
886  }
887  // std::cout << " SReg : " << s << " : " << std::hex << mSyncCheckRegister[s] << std::endl;
888  // check the register content for the SYNC pattern
889  // Patterh is : 1100.1100.1100.1100.1111.0000.1111.0000 = 0xCCCCF0F0
890  // alternative Pattern for first corrupted SYNC word : 0100.1100.1100.1100.1111.0000.1111.0000 = 4CCC.F0F0
891  if (mSyncCheckRegister[s] == 0xCCCCF0F0 or mSyncCheckRegister[s] == 0x4CCCF0F0 or mSyncCheckRegister[s] == 0x0CCCF0F0) {
892  mSyncPos[s].setPos(mPacketNum, mFrameNum, mFilePos, h);
893  if (verbose) {
894  std::cout << s << " : " << mSyncPos[s];
895  }
896  }

904 inline void GBTFrame::getFrameHalfWords()
905 {
906  constexpr adc_t P[5][4] = {{19, 18, 17, 16}, {39, 38, 37, 36}, {63, 62, 61, 60}, {83, 82, 81, 80}, {107, 106, 105, 104}};
907  // i = Stream, j = Halfword
908  for (int i = 0; i < 5; i++) {
909  for (int j = 0; j < 4; j++) {
910  mFrameHalfWords[i][j + mPrevHWpos] = bit(P[i][j], adc_t(4)) |
911  bit(adc_t(P[i][j] - 4), adc_t(3)) |
912  bit(adc_t(P[i][j] - 8), adc_t(2)) |
913  bit(adc_t(P[i][j] - 12), adc_t(1)) |
914  bit(adc_t(P[i][j] - 16), adc_t(0));
915  // std::cout << " S : " << i << " H : " << j << " : " << std::hex << res << std::endl;
916  };
917  }
918  mPrevHWpos ^= 4; // toggle position of previous HW position
919 }

922 inline void GBTFrame::getAdcValues(ADCRawData& rawData)
923 {
924  // loop over all the 5 data streams
925  for (int s = 0; s < 5; s++) {
926  if (!mSyncPos[s].synched()) {
927  continue;
928  }
929 
930  // assemble the 2 ADC words from 4 half-words. Which halfwords are
931  // used depends on the position of the SYNC pattern.
932  const uint32_t pos = mSyncPos[s].getHalfWordPosition();
933  const uint32_t v0 = (mFrameHalfWords[s][(pos + 2 + mPrevHWpos) & 7] << 5) | mFrameHalfWords[s][(pos + 1 + mPrevHWpos) & 7];
934  const uint32_t v1 = (mFrameHalfWords[s][(pos + 4 + mPrevHWpos) & 7] << 5) | mFrameHalfWords[s][(pos + 3 + mPrevHWpos) & 7];
935 
936  // add the two ADC values for the current processed stream
937  rawData.add(s, v0, v1);
938  };
