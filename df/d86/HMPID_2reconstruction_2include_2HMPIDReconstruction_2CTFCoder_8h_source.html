58 template <typename VEC>
59 void CTFCoder::encode(VEC& buff, const gsl::span<const Trigger>& trigData, const gsl::span<const Digit>& digData)
60 {
61  using MD = o2::ctf::Metadata::OptStore;
62  // what to do which each field: see o2::ctd::Metadata explanation
63  constexpr MD optField[CTF::getNBlocks()] = {
64  MD::EENCODE, // BLC_bcIncTrig
65  MD::EENCODE, // BLC_orbitIncTrig
66  MD::EENCODE, // BLC_entriesDig
67  MD::EENCODE, // BLC_ChID
68  MD::EENCODE, // BLC_Q
69  MD::EENCODE, // BLC_Ph
70  MD::EENCODE, // BLC_X
71  MD::EENCODE // BLC_Y
72  };
73 
74  CTFHelper helper(trigData, digData);
75 
76  // book output size with some margin
77  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
78  buff.resize(szIni);
79 
80  auto ec = CTF::create(buff);
81  using ECB = CTF::base;
82 
83  ec->setHeader(helper.createHeader());
84  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
85  ec->getANSHeader().majorVersion = 0;
86  ec->getANSHeader().minorVersion = 1;
87  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
88 #define ENCODEHMP(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
89  // clang-format off
90  ENCODEHMP(helper.begin_bcIncTrig(), helper.end_bcIncTrig(), CTF::BLC_bcIncTrig, 0);
91  ENCODEHMP(helper.begin_orbitIncTrig(), helper.end_orbitIncTrig(), CTF::BLC_orbitIncTrig, 0);
92  ENCODEHMP(helper.begin_entriesDig(), helper.end_entriesDig(), CTF::BLC_entriesDig, 0);
93 
94  ENCODEHMP(helper.begin_ChID(), helper.end_ChID(), CTF::BLC_ChID, 0);
95  ENCODEHMP(helper.begin_Q(), helper.end_Q(), CTF::BLC_Q, 0);
96  ENCODEHMP(helper.begin_Ph(), helper.end_Ph(), CTF::BLC_Ph, 0);
97  ENCODEHMP(helper.begin_X(), helper.end_X(), CTF::BLC_X, 0);
98  ENCODEHMP(helper.begin_Y(), helper.end_Y(), CTF::BLC_Y, 0);
99 
100  // clang-format on
101  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
102 }

105 template <typename VTRG, typename VDIG>
106 void CTFCoder::decode(const CTF::base& ec, VTRG& trigVec, VDIG& digVec)
107 {
108  auto header = ec.getHeader();
109  checkDictVersion(static_cast<const o2::ctf::CTFDictHeader&>(header));
110  ec.print(getPrefix(), mVerbosity);
111  std::vector<uint16_t> bcInc, q;
112  std::vector<uint32_t> orbitInc, entriesDig;
113  std::vector<uint8_t> chID, ph, x, y;
114 
115 #define DECODEHMP(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
116  // clang-format off
117  DECODEHMP(bcInc, CTF::BLC_bcIncTrig);
118  DECODEHMP(orbitInc, CTF::BLC_orbitIncTrig);
119  DECODEHMP(entriesDig, CTF::BLC_entriesDig);
120 
121  DECODEHMP(chID, CTF::BLC_ChID);
122  DECODEHMP(q, CTF::BLC_Q);
123  DECODEHMP(ph, CTF::BLC_Ph);
124  DECODEHMP(x, CTF::BLC_X);
125  DECODEHMP(y, CTF::BLC_Y);
126  // clang-format on
127  //
128  trigVec.clear();
129  digVec.clear();
130  trigVec.reserve(header.nTriggers);
131  digVec.reserve(header.nDigits);
132 
133  uint32_t digCount = 0;
134  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
135 
136  for (uint32_t itrig = 0; itrig < header.nTriggers; itrig++) {
137  // restore TrigRecord
138  if (orbitInc[itrig]) { // non-0 increment => new orbit
139  ir.bc = bcInc[itrig]; // bcInc has absolute meaning
140  ir.orbit += orbitInc[itrig];
141  } else {
142  ir.bc += bcInc[itrig];
143  }
144 
145  uint32_t firstEntryDig = digVec.size();
146  int8_t chid = 0;
147  for (uint32_t id = 0; id < entriesDig[itrig]; id++) {
148  chid += chID[digCount]; // 1st digit of trigger was encoded with abs ChID, then increments
149  auto& dig = digVec.emplace_back(chid, ph[digCount], x[digCount], y[digCount], q[digCount]);
150  digCount++;
151  }
152 
153  trigVec.emplace_back(ir, firstEntryDig, entriesDig[itrig]);
154  }
155  assert(digCount == header.nDigits);
156 }
