111 void FIT_CALIBRATION_DEVICE_TYPE::_sendOutputs(o2::framework::DataAllocator& outputs)
112 {
113  using clbUtils = o2::calibration::Utils;
114  const auto& objectsToSend = mCalibrator->getStoredCalibrationObjects();
115 
116  uint32_t iSendChannel = 0;
117  for (const auto& [ccdbInfo, calibObject] : objectsToSend) {
118  outputs.snapshot(o2::framework::Output{clbUtils::gDataOriginCDBPayload, "FIT_CALIB", iSendChannel}, *calibObject);
119  outputs.snapshot(o2::framework::Output{clbUtils::gDataOriginCDBWrapper, "FIT_CALIB", iSendChannel}, ccdbInfo);
120  LOG(info) << "_sendOutputs " << ccdbInfo.getStartValidityTimestamp();
121  ++iSendChannel;
122  }
123  mCalibrator->initOutput();
124 }
