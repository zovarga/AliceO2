257 inline void AlignmentPoint::diagonalizeResiduals(double rY, double rZ, double& resU, double& resV) const
258 {
259  // rotate residuals to frame where their error matrix is diagonal
260  resU = mCosDiagErr * rY - mSinDiagErr * rZ;
261  resV = mSinDiagErr * rY + mCosDiagErr * rZ;
262  //
263 }
264 
265 //_____________________________________
266 inline void AlignmentPoint::getResidualsDiag(const double* pos, double& resU, double& resV) const
267 {
268  // calculate residuals in the frame where the errors are diagonal, given the position
269  // of the track in the standard tracking frame
270  diagonalizeResiduals(pos[0] - mXYZTracking[1], pos[1] - mXYZTracking[2], resU, resV);
271  //
272 }
