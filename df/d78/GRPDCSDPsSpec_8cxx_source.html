43 void GRPDCSDPsDataProcessor::init(o2::framework::InitContext& ic)
44 {
45 
46  std::vector<DPID> vect;
47  mDPsUpdateInterval = ic.options().get<int64_t>("DPs-update-interval");
48  if (mDPsUpdateInterval == 0) {
49  LOG(error) << "GRP DPs update interval set to zero seconds --> changed to 60";
50  mDPsUpdateInterval = 60;
51  }
52  bool useCCDBtoConfigure = ic.options().get<bool>("use-ccdb-to-configure");
53  if (useCCDBtoConfigure) {
54  LOG(info) << "Configuring via CCDB";
55  std::string ccdbpath = ic.options().get<std::string>("ccdb-path");
56  auto& mgr = CcdbManager::instance();
57  mgr.setURL(ccdbpath);
58  CcdbApi api;
59  api.init(mgr.getURL());
60  long ts = std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count();
61  std::unordered_map<DPID, std::string>* dpid2DataDesc = mgr.getForTimeStamp<std::unordered_map<DPID, std::string>>("GRP/Config/DCSDPconfig", ts);
62  for (auto& i : *dpid2DataDesc) {
63  vect.push_back(i.first);
64  }
65  } else {
66  LOG(info) << "Configuring via hardcoded strings";
67  std::vector<std::string> aliasesBFieldDouble = {"L3Current", "DipoleCurrent"};
68  std::vector<std::string> aliasesBFieldBool = {"L3Polarity", "DipolePolarity"};
69  std::vector<std::string> aliasesEnvVar = {"CavernTemperature", "CavernAtmosPressure", "SurfaceAtmosPressure", "CavernAtmosPressure2"};
70  std::vector<std::string> compactAliasesLHCIFDouble = {"LHC_IntensityBeam[1..2]_totalIntensity", "ALI_Background[1..3]",
71  "ALI_Lumi_Total_Inst",
72  "BPTX_deltaT_B1_B2", "BPTX_deltaTRMS_B1_B2",
73  "BPTX_Phase_B[1..2]", "BPTX_PhaseRMS_B[1..2]", "BPTX_Phase_Shift_B[1..2]"};
74  std::vector<std::string> aliasesLHCIFDouble = o2::dcs::expandAliases(compactAliasesLHCIFDouble);
75  std::vector<std::string> aliasesLHCIFString = {"ALI_Lumi_Source_Name", "MACHINE_MODE", "BEAM_MODE"};
76  std::vector<std::string> aliasesLHCIFCollimators = {"LHC_CollimatorPos_TCLIA_4R2_lvdt_gap_downstream", "LHC_CollimatorPos_TCLIA_4R2_lvdt_gap_upstream",
77  "LHC_CollimatorPos_TCLIA_4R2_lvdt_left_downstream", "LHC_CollimatorPos_TCLIA_4R2_lvdt_left_upstream",
78  "LHC_CollimatorPos_TCLIA_4R2_lvdt_right_downstream", "LHC_CollimatorPos_TCLIA_4R2_lvdt_right_upstream"};
79 
80  for (const auto& i : aliasesBFieldDouble) {
81  vect.emplace_back(i, o2::dcs::RAW_DOUBLE);
82  }
83  for (const auto& i : aliasesBFieldBool) {
84  vect.emplace_back(i, o2::dcs::RAW_BOOL);
85  }
86  for (const auto& i : aliasesEnvVar) {
87  vect.emplace_back(i, o2::dcs::RAW_DOUBLE);
88  }
89  for (const auto& i : aliasesLHCIFDouble) {
90  vect.emplace_back(i, o2::dcs::RAW_DOUBLE);
91  }
92  for (const auto& i : aliasesLHCIFCollimators) {
93  vect.emplace_back(i, o2::dcs::RAW_DOUBLE);
94  }
95  for (const auto& i : aliasesLHCIFString) {
96  vect.emplace_back(i, o2::dcs::RAW_STRING);
97  }
98  }
99 
100  LOG(info) << "Listing Data Points for GRP:";
101  for (auto& i : vect) {
102  LOG(info) << i;
103  }
104 
105  mProcessor = std::make_unique<o2::grp::GRPDCSDPsProcessor>();
106  bool useVerboseMode = ic.options().get<bool>("use-verbose-mode");
107  LOG(info) << " ************************* Verbose?" << useVerboseMode;
108  if (useVerboseMode) {
109  mProcessor->useVerboseMode();
110  }
111  mProcessor->init(vect);
112  mTimer = HighResClock::now();
113 }

116 void GRPDCSDPsDataProcessor::run(o2::framework::ProcessingContext& pc)
117 {
118  auto startValidity = DataRefUtils::getHeader<DataProcessingHeader*>(pc.inputs().getFirstValid(true))->creation;
119  auto dps = pc.inputs().get<gsl::span<DPCOM>>("input");
120  auto timeNow = HighResClock::now();
121  if (startValidity == 0xffffffffffffffff) { // it means it is not set
122  startValidity = std::chrono::duration_cast<std::chrono::milliseconds>(timeNow.time_since_epoch()).count(); // in ms
123  }
124  mProcessor->setStartValidity(startValidity);
125  mProcessor->process(dps);
126  Duration elapsedTime = timeNow - mTimer; // in seconds
127  if (elapsedTime.count() >= mDPsUpdateInterval || mProcessor->isLHCIFInfoUpdated()) {
128  sendLHCIFDPsoutput(pc.outputs());
129  mProcessor->resetLHCIFDPs();
130  mTimer = timeNow;
131  }
132  if (mProcessor->isMagFieldUpdated()) {
133  sendMagFieldDPsoutput(pc.outputs());
134  }
135  if (mProcessor->isCollimatorsUpdated()) {
136  sendCollimatorsDPsoutput(pc.outputs());
137  }
138  if (mProcessor->isEnvVarsUpdated()) {
139  sendEnvVarsDPsoutput(pc.outputs());
140  }
141 }

159 void GRPDCSDPsDataProcessor::sendLHCIFDPsoutput(DataAllocator& output)
160 {
161  // filling CCDB with LHCIF DPs object
162 
163  const auto& payload = mProcessor->getLHCIFObj();
164  auto& info = mProcessor->getccdbLHCIFInfo();
165  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
166  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
167  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
168  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "GRP_LHCIF_DPs", 0}, *image.get());
169  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "GRP_LHCIF_DPs", 0}, info);
