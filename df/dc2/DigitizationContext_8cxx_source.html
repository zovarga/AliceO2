67 bool DigitizationContext::initSimChains(o2::detectors::DetID detid, std::vector<TChain*>& simchains) const
68 {
69  if (!(simchains.size() == 0)) {
70  // nothing to do ... already setup
71  return false;
72  }
73 
74  simchains.emplace_back(new TChain("o2sim"));
75  // add the main (background) file
76  simchains.back()->AddFile(o2::base::DetectorNameConf::getHitsFileName(detid, mSimPrefixes[0].data()).c_str());
77 
78  for (int source = 1; source < mSimPrefixes.size(); ++source) {
79  simchains.emplace_back(new TChain("o2sim"));
80  // add signal files
81  simchains.back()->AddFile(o2::base::DetectorNameConf::getHitsFileName(detid, mSimPrefixes[source].data()).c_str());
82  }
83 
84  // QED part
85  if (mEventRecordsWithQED.size() > 0) {
86  if (mSimPrefixes.size() >= QEDSOURCEID) {
87  LOG(fatal) << "Too many signal chains; crashes with QED source ID";
88  }
89 
90  // it might be better to use an unordered_map for the simchains but this requires interface changes
91  simchains.resize(QEDSOURCEID + 1, nullptr);
92  simchains[QEDSOURCEID] = new TChain("o2sim");
93  simchains[QEDSOURCEID]->AddFile(o2::base::DetectorNameConf::getHitsFileName(detid, mQEDSimPrefix).c_str());
94  }
95 
96  return true;
97 }

121 bool DigitizationContext::checkVertexCompatibility(bool verbose) const
122 {
123  if (mMaxPartNumber == 1) {
124  return true;
125  }
126 
127  auto checkVertexPair = [](math_utils::Point3D<double> const& p1, math_utils::Point3D<double> const& p2) -> bool {
128  return (p2 - p1).Mag2() < 1E-6;
129  };
130 
131  std::vector<TChain*> kinematicschain;
132  std::vector<TBranch*> headerbranches;
133  std::vector<o2::dataformats::MCEventHeader*> headers;
134  std::vector<math_utils::Point3D<double>> vertices;
135  initSimKinematicsChains(kinematicschain);
136  bool consistent = true;
137  if (kinematicschain.size() > 0) {
138  headerbranches.resize(kinematicschain.size(), nullptr);
139  headers.resize(kinematicschain.size(), nullptr);
140  // loop over all collisions in this context
141  int collisionID = 0;
142  for (auto& collision : getEventParts()) {
143  collisionID++;
144  vertices.clear();
145  for (auto& part : collision) {
146  const auto source = part.sourceID;
147  const auto entry = part.entryID;
148  auto chain = kinematicschain[source];
149  if (!headerbranches[source]) {
150  headerbranches[source] = chain->GetBranch("MCEventHeader.");
151  headerbranches[source]->SetAddress(&headers[source]);
152  }
153  // get the MCEventHeader to read out the vertex
154  headerbranches[source]->GetEntry(entry);
155  auto header = headers[source];
156  vertices.emplace_back(header->GetX(), header->GetY(), header->GetZ());
157  }
158  // analyse vertex matching
159  bool thiscollision = true;
160  const auto& p1 = vertices[0];
161  for (int j = 1; j < vertices.size(); ++j) {
162  const auto& p2 = vertices[j];
163  bool thischeck = checkVertexPair(p1, p2);
164  thiscollision &= thischeck;
165  }
166  if (verbose && !thiscollision) {
167  std::stringstream text;
168  text << "Found inconsistent vertices for digit collision ";
169  text << collisionID << " : ";
170  for (auto& p : vertices) {
171  text << p << " ";
172  }
173  LOG(error) << text.str();
174  }
175  consistent &= thiscollision;
176  }
177  }
178  return consistent;
179 }

212 void DigitizationContext::fillQED(std::string_view QEDprefix, std::vector<o2::InteractionTimeRecord> const& irecord)
213 {
214  mQEDSimPrefix = QEDprefix;
215 
216  std::vector<std::vector<o2::steer::EventPart>> qedEventParts;
217 
218  // we need to fill the QED parts (using a simple round robin scheme)
219  auto qedKinematicsName = o2::base::NameConf::getMCKinematicsFileName(mQEDSimPrefix);
220  // find out how many events are stored
221  TFile f(qedKinematicsName.c_str(), "OPEN");
222  auto t = (TTree*)f.Get("o2sim");
223  if (!t) {
224  LOG(error) << "No QED kinematics found";
225  throw std::runtime_error("No QED kinematics found");
226  }
227  auto numberQEDevents = t->GetEntries();
228  int eventID = 0;
229  for (auto& tmp : irecord) {
230  std::vector<EventPart> qedpart;
231  qedpart.emplace_back(QEDSOURCEID, eventID++);
232  qedEventParts.push_back(qedpart);
233  if (eventID == numberQEDevents) {
234  eventID = 0;
235  }
236  }
237 
238  // we need to do the interleaved event records for detectors consuming both
239  // normal and QED events
240  // --> merge both; sort first according to times and sort second one according to same order
241  auto combinedrecords = mEventRecords;
242  combinedrecords.insert(combinedrecords.end(), irecord.begin(), irecord.end());
243  auto combinedparts = mEventParts;
244  combinedparts.insert(combinedparts.end(), qedEventParts.begin(), qedEventParts.end());
245 
246  // get sorted index vector based on event records
247  std::vector<size_t> idx(combinedrecords.size());
248  std::iota(idx.begin(), idx.end(), 0);
249 
250  std::stable_sort(idx.begin(), idx.end(),
251  [&combinedrecords](size_t i1, size_t i2) { return combinedrecords[i1] < combinedrecords[i2]; });
252 
253  mEventRecordsWithQED.clear();
254  mEventPartsWithQED.clear();
255  for (int i = 0; i < idx.size(); ++i) {
256  mEventRecordsWithQED.push_back(combinedrecords[idx[i]]);
257  mEventPartsWithQED.push_back(combinedparts[idx[i]]);
258  }
259 }
