56  void run(o2::framework::ProcessingContext& pc) final
57  {
58  int nSlots = pc.inputs().getNofParts(0);
59  assert(pc.inputs().getNofParts(1) == nSlots);
60  auto runNoFromDH = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true))->runNumber;
61  std::string runNoStr;
62  if (runNoFromDH > 0) {
63  runNoStr = std::to_string(runNoFromDH);
64  }
65 
66  std::map<std::string, std::string> metadata;
67  for (int isl = 0; isl < nSlots; isl++) {
68  auto refWrp = pc.inputs().get("clbWrapper", isl);
69  auto refPld = pc.inputs().get("clbPayload", isl);
70  if (mSSpecMin >= 0 && mSSpecMin <= mSSpecMax) { // there is a selection
71  auto ss = std::int64_t(o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(refWrp)->subSpecification);
72  if (ss < mSSpecMin || ss > mSSpecMax) {
73  continue;
74  }
75  }
76  const auto wrp = pc.inputs().get<CcdbObjectInfo*>(refWrp);
77  const auto pld = pc.inputs().get<gsl::span<char>>(refPld); // this is actually an image of TMemFile
78  const auto* md = &wrp->getMetaData();
79  if (runNoFromDH > 0 && md->find(o2::base::NameConf::CCDBRunTag.data()) == md->end()) { // if valid run number is provided and it is not filled in the metadata, add it to the clone
80  metadata = *md; // clone since the md from the message is const
81  metadata[o2::base::NameConf::CCDBRunTag.data()] = runNoStr;
82  md = &metadata;
83  }
84 
85  LOG(info) << "Storing in ccdb " << wrp->getPath() << "/" << wrp->getFileName() << " of size " << pld.size()
86  << " Valid for " << wrp->getStartValidityTimestamp() << " : " << wrp->getEndValidityTimestamp();
87  mAPI.storeAsBinaryFile(&pld[0], pld.size(), wrp->getFileName(), wrp->getObjectType(), wrp->getPath(),
88  *md, wrp->getStartValidityTimestamp(), wrp->getEndValidityTimestamp());
89  }
90  }
91 
92  private:
93  CcdbApi mAPI;
94  std::int64_t mSSpecMin = -1; // min subspec to accept
95  std::int64_t mSSpecMax = -1; // max subspec to accept
96  std::string mCCDBpath = "http://ccdb-test.cern.ch:8080"; // CCDB path
97 };
