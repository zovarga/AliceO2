45 int InputRecord::getPos(const char* binding) const
46 {
47  auto inputIndex = 0;
48  for (size_t i = 0; i < mInputsSchema.size(); ++i) {
49  auto& route = mInputsSchema[i];
50  if (route.timeslice != 0) {
51  continue;
52  }
53  if (route.matcher.binding == binding) {
54  return inputIndex;
55  }
56  ++inputIndex;
57  }
58  return -1;
59 }

66 DataRef InputRecord::getByPos(int pos, int part) const
67 {
68  if (pos >= mSpan.size() || pos < 0) {
69  throw runtime_error_f("Unknown message requested at position %d", pos);
70  }
71  if (part > 0 && part >= getNofParts(pos)) {
72  throw runtime_error_f("Invalid message part index at %d:%d", pos, part);
73  }
74  if (pos >= mInputsSchema.size()) {
75  throw runtime_error_f("Unknown schema at position %d", pos);
76  }
77  auto ref = mSpan.get(pos, part);
78  auto inputIndex = 0;
79  auto schemaIndex = 0;
80  for (size_t i = 0; i < mInputsSchema.size(); ++i) {
81  schemaIndex = i;
82  auto& route = mInputsSchema[i];
83  if (route.timeslice != 0) {
84  continue;
85  }
86  if (inputIndex == pos) {
87  break;
88  }
89  ++inputIndex;
90  }
91  ref.spec = &mInputsSchema[schemaIndex].matcher;
92  return ref;
93 }

95 DataRef InputRecord::getFirstValid(bool throwOnFailure) const
96 {

102  }
103  }
104  if (throwOnFailure) {

106  }
107  return {};
