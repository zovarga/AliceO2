56 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
57 {
58  WorkflowSpec specs;
59 
60  auto disableRootOutput = configcontext.options().get<bool>("disable-root-output");
61  auto disableRootInput = configcontext.options().get<bool>("disable-root-input");
62  auto digits = configcontext.options().get<bool>("digits");
63  auto triggered = configcontext.options().get<bool>("triggered");
64  auto useMC = !configcontext.options().get<bool>("disable-mc");
65 
66  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
67 
68  if (!disableRootInput) {
69  specs.emplace_back(o2::mch::getDigitReaderSpec(useMC, "mch-sim-digit-reader"));
70  }
71 
72  specs.emplace_back(o2::mch::getDigitFilteringSpec(useMC, "mch-digit-filtering",
73  "DIGITS", "F-DIGITS",
74  "DIGITROFS", "F-DIGITROFS",
75  "DIGITLABELS", "F-DIGITLABELS"));
76 
77  if (triggered) {
78  specs.emplace_back(o2::mch::getEventFinderSpec(useMC, "mch-event-finder",
79  "F-DIGITS", "E-F-DIGITS",
80  "F-DIGITROFS", "E-F-DIGITROFS",
81  "F-DIGITLABELS", "E-F-DIGITLABELS"));
82  } else if (!useMC) {
83  specs.emplace_back(o2::mch::getTimeClusterFinderSpec("mch-time-cluster-finder",
84  "F-DIGITS",
85  "F-DIGITROFS",
86  "TC-F-DIGITROFS"));
87  }
88 
89  specs.emplace_back(o2::mch::getPreClusterFinderSpec("mch-precluster-finder",
90  triggered ? "E-F-DIGITS" : "F-DIGITS",
91  triggered ? "E-F-DIGITROFS" : (useMC ? "F-DIGITROFS" : "TC-F-DIGITROFS")));
92 
93  specs.emplace_back(o2::mch::getClusterFinderOriginalSpec("mch-cluster-finder"));
94  specs.emplace_back(o2::mch::getClusterTransformerSpec());
95  specs.emplace_back(o2::mch::getTrackFinderSpec("mch-track-finder", digits));
96  if (useMC) {
97  specs.emplace_back(o2::mch::getTrackMCLabelFinderSpec("mch-track-mc-label-finder",
98  triggered ? "E-F-DIGITROFS" : "F-DIGITROFS",
99  triggered ? "E-F-DIGITLABELS" : "F-DIGITLABELS"));
100  }
101 
102  if (!disableRootOutput) {
103  specs.emplace_back(o2::mch::getTrackWriterSpec(useMC, "mch-track-writer", "mchtracks.root", digits));
104  }
105 
106  // configure dpl timer to inject correct firstTFOrbit: start from the 1st orbit of TF containing 1st sampled orbit
107  o2::raw::HBFUtilsInitializer hbfIni(configcontext, specs);
108 
109  // write the configuration used for the reco workflow
110  o2::conf::ConfigurableParam::writeINI("o2mchrecoflow_configuration.ini");
111 
112  return specs;
113 }
