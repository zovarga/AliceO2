209 template <class T, class S>
210 GPUd() int GPUTrackingRefit::RefitTrack(T& trkX, bool outward, bool resetCov)
211 {
212  CADEBUG(int ii; printf("\nRefitting track\n"));
213  typename refitTrackTypes<S>::propagator prop;
214  S trk;
215  float TrackParCovChi2 = 0.f;
216  convertTrack<S, T, typename refitTrackTypes<S>::propagator>(trk, trkX, prop, &TrackParCovChi2);
217  int begin = 0, count;
218  float tOffset;
219  if constexpr (std::is_same_v<T, GPUTPCGMMergedTrack>) {
220  count = trkX.NClusters();
221  if (trkX.Looper()) {
222  int leg = mPtrackHits[trkX.FirstClusterRef() + trkX.NClusters() - 1].leg;
223  for (int i = trkX.NClusters() - 2; i > 0; i--) {
224  if (mPtrackHits[trkX.FirstClusterRef() + i].leg != leg) {
225  begin = i + 1;
226  break;
227  }
228  }
229  }
230  tOffset = trkX.GetParam().GetTZOffset();
231  } else if constexpr (std::is_same_v<T, TrackTPC>) {
232  count = trkX.getNClusters();
233  tOffset = trkX.getTime0();
234  } else if constexpr (std::is_same_v<T, TrackParCovWithArgs>) {
235  count = trkX.clusRef.getEntries();
236  tOffset = trkX.time0;
237  } else {
238  static_assert("Invalid template");
239  }
240  int direction = outward ? -1 : 1;
241  int start = outward ? count - 1 : begin;
242  int stop = outward ? begin - 1 : count;
243  const ClusterNative* cl = nullptr;
244  uint8_t sector, row = 0, currentSector = 0, currentRow = 0;
245  short clusterState = 0, nextState;
246  int nFitted = 0;
247  for (int i = start; i != stop; i += cl ? 0 : direction) {
248  float x, y, z, charge = 0.f;
249  int clusters = 0;
250  while (true) {
251  if (!cl) {
252  CADEBUG(ii = i);
253  if constexpr (std::is_same_v<T, GPUTPCGMMergedTrack>) {
254  const auto& hit = mPtrackHits[trkX.FirstClusterRef() + i];
255  cl = &mPclusterNative->clustersLinear[hit.num];
256  if (hit.state & (GPUTPCGMMergedTrackHit::flagReject | GPUTPCGMMergedTrackHit::flagNotFit)) {
257  cl = nullptr;
258  if (i + direction != stop) {
259  i += direction;
260  continue;
261  }
262  break;
263  }
264  row = hit.row;
265  sector = hit.slice;
266  nextState = mPclusterState[hit.num];
267  } else if constexpr (std::is_same_v<T, TrackTPC>) {
268  cl = &trkX.getCluster(mPtrackHitReferences, i, *mPclusterNative, sector, row);
269  nextState = mPclusterState[cl - mPclusterNative->clustersLinear];
270  } else if constexpr (std::is_same_v<T, TrackParCovWithArgs>) {
271  cl = &TrackTPC::getCluster(mPtrackHitReferences, i, *mPclusterNative, sector, row, trkX.clusRef);
272  nextState = mPclusterState[cl - mPclusterNative->clustersLinear];
273  } else {
274  static_assert("Invalid template");
275  }
276  }
277  if (clusters == 0 || (row == currentRow && sector == currentSector)) {
278  if (clusters == 1) {
279  x *= charge;
280  y *= charge;
281  z *= charge;
282  }
283  if (clusters == 0) {
284  mPfastTransform->Transform(sector, row, cl->getPad(), cl->getTime(), x, y, z, tOffset);
285  CADEBUG(printf("\tHit %3d/%3d Row %3d: Cluster Alpha %8.3f %3d, X %8.3f - Y %8.3f, Z %8.3f\n", ii, count, row, mPparam->Alpha(sector), (int)sector, x, y, z));
286  currentRow = row;
287  currentSector = sector;
288  charge = cl->qTot;
289  clusterState = nextState;
290  } else {
291  float xx, yy, zz;
292  mPfastTransform->Transform(sector, row, cl->getPad(), cl->getTime(), xx, yy, zz, tOffset);
293  CADEBUG(printf("\tHit %3d/%3d Row %3d: Cluster Alpha %8.3f %3d, X %8.3f - Y %8.3f, Z %8.3f\n", ii, count, row, mPparam->Alpha(sector), (int)sector, xx, yy, zz));
294  x += xx * cl->qTot;
295  y += yy * cl->qTot;
296  z += zz * cl->qTot;
297  charge += cl->qTot;
298  clusterState |= nextState;
299  }
300  cl = nullptr;
301  clusters++;
302  if (i + direction != stop) {
303  i += direction;
304  continue;
305  }
306  }
307  break;
308  }
309  if (clusters == 0) {
310  continue;
311  } else if (clusters > 1) {
312  x /= charge;
313  y /= charge;
314  z /= charge;
315  CADEBUG(printf("\tMerged Hit Row %3d: Cluster Alpha %8.3f %3d, X %8.3f - Y %8.3f, Z %8.3f\n", row, mPparam->Alpha(sector), (int)sector, x, y, z));
316  }
317 
318  if constexpr (std::is_same_v<S, GPUTPCGMTrackParam>) {
319  if (prop.PropagateToXAlpha(x, mPparam->Alpha(currentSector), !outward)) {
320  IgnoreErrors(trk.GetSinPhi());
321  return -2;
322  }
323  if (resetCov) {
324  trk.ResetCovariance();
325  }
326  CADEBUG(printf("\t%21sPropaga Alpha %8.3f , X %8.3f - Y %8.3f, Z %8.3f - QPt %7.2f (%7.2f), SP %5.2f (%5.2f) --- Res %8.3f %8.3f --- Cov sY %8.3f sZ %8.3f sSP %8.3f sPt %8.3f - YPt %8.3f\n", "", prop.GetAlpha(), x, trk.Par()[0], trk.Par()[1], trk.Par()[4], prop.GetQPt0(), trk.Par()[2], prop.GetSinPhi0(), trk.Par()[0] - y, trk.Par()[1] - z, sqrtf(trk.Cov()[0]), sqrtf(trk.Cov()[2]), sqrtf(trk.Cov()[5]), sqrtf(trk.Cov()[14]), trk.Cov()[10]));
327  if (prop.Update(y, z, row, *mPparam, clusterState, 0, nullptr, true)) {
328  IgnoreErrors(trk.GetSinPhi());
329  return -3;
330  }
331  CADEBUG(printf("\t%21sFit Alpha %8.3f , X %8.3f - Y %8.3f, Z %8.3f - QPt %7.2f (%7.2f), SP %5.2f (%5.2f), DzDs %5.2f %16s --- Cov sY %8.3f sZ %8.3f sSP %8.3f sPt %8.3f - YPt %8.3f\n", "", prop.GetAlpha(), x, trk.Par()[0], trk.Par()[1], trk.Par()[4], prop.GetQPt0(), trk.Par()[2], prop.GetSinPhi0(), trk.Par()[3], "", sqrtf(trk.Cov()[0]), sqrtf(trk.Cov()[2]), sqrtf(trk.Cov()[5]), sqrtf(trk.Cov()[14]), trk.Cov()[10]));
332  } else if constexpr (std::is_same_v<S, TrackParCov>) {
333  if (!trk.rotate(mPparam->Alpha(currentSector))) {
334  IgnoreErrors(trk.getSnp());
335  return -1;
336  }
337  if (!prop->PropagateToXBxByBz(trk, x, GPUCA_MAX_SIN_PHI_LOW)) {
338  IgnoreErrors(trk.getSnp());
339  return -2;
340  }
341  if (resetCov) {
342  trk.resetCovariance();
343  TrackParCovChi2 = 0.f;
344  }
345  CADEBUG(printf("\t%21sPropaga Alpha %8.3f , X %8.3f - Y %8.3f, Z %8.3f - QPt %7.2f (%7.2f), SP %5.2f (%5.2f) --- Res %8.3f %8.3f --- Cov sY %8.3f sZ %8.3f sSP %8.3f sPt %8.3f - YPt %8.3f\n", "", trk.getAlpha(), x, trk.getParams()[0], trk.getParams()[1], trk.getParams()[4], trk.getParams()[4], trk.getParams()[2], trk.getParams()[2], trk.getParams()[0] - y, trk.getParams()[1] - z, sqrtf(trk.getCov()[0]), sqrtf(trk.getCov()[2]), sqrtf(trk.getCov()[5]), sqrtf(trk.getCov()[14]), trk.getCov()[10]));
346  gpu::gpustd::array<float, 2> p = {y, z};
347  gpu::gpustd::array<float, 3> c = {0, 0, 0};
348  mPparam->GetClusterErrors2(currentRow, z, getPar(trk)[2], getPar(trk)[3], c[0], c[2]);
349  mPparam->UpdateClusterError2ByState(clusterState, c[0], c[2]);
350  TrackParCovChi2 += trk.getPredictedChi2(p, c);
351  if (!trk.update(p, c)) {
352  IgnoreErrors(trk.getSnp());
353  return -3;
354  }
355  CADEBUG(printf("\t%21sFit Alpha %8.3f , X %8.3f - Y %8.3f, Z %8.3f - QPt %7.2f (%7.2f), SP %5.2f (%5.2f), DzDs %5.2f %16s --- Cov sY %8.3f sZ %8.3f sSP %8.3f sPt %8.3f - YPt %8.3f\n", "", trk.getAlpha(), x, trk.getParams()[0], trk.getParams()[1], trk.getParams()[4], trk.getParams()[4], trk.getParams()[2], trk.getParams()[2], trk.getParams()[3], "", sqrtf(trk.getCov()[0]), sqrtf(trk.getCov()[2]), sqrtf(trk.getCov()[5]), sqrtf(trk.getCov()[14]), trk.getCov()[10]));
356  } else {
357  static_assert("Invalid template");
358  }
359  resetCov = false;
360  nFitted++;
361  }
362  if constexpr (std::is_same_v<S, GPUTPCGMTrackParam>) {
363  float alpha = prop.GetAlpha();
364  trk.MoveToReference(prop, *mPparam, alpha);
365  trk.NormalizeAlpha(alpha);
366  prop.SetAlpha(alpha);
367  } else if constexpr (std::is_same_v<S, TrackParCov>) {
368  static constexpr float kDeg2Rad = M_PI / 180.f;
369  static constexpr float kSectAngle = 2 * M_PI / 18.f;
370  if (mPparam->rec.tpc.trackReferenceX <= 500) {
371  if (prop->PropagateToXBxByBz(trk, mPparam->rec.tpc.trackReferenceX)) {
372  if (CAMath::Abs(trk.getY()) > trk.getX() * CAMath::Tan(kSectAngle / 2.f)) {
373  float newAlpha = trk.getAlpha() + floor(CAMath::ATan2(trk.getY(), trk.getX()) / kDeg2Rad / 20.f + 0.5f) * kSectAngle;
374  GPUTPCGMTrackParam::NormalizeAlpha(newAlpha);
375  trk.rotate(newAlpha) && prop->PropagateToXBxByBz(trk, mPparam->rec.tpc.trackReferenceX);
376  }
377  }
378  }
379  } else {
380  static_assert("Invalid template");
381  }
382 
383  convertTrack<T, S, typename refitTrackTypes<S>::propagator>(trkX, trk, prop, &TrackParCovChi2);
384  return nFitted;
385 }
