540 void updateInputs(std::string type, bool value, std::vector<InputSpec>& inputs, InputSpec& spec)
541 {
542  auto locate = std::find_if(inputs.begin(), inputs.end(), [&](InputSpec& input) { return input.binding == spec.binding; });
543  if (locate != inputs.end()) {
544  // amend entry
545  auto& entryMetadata = locate->metadata;
546  entryMetadata.push_back(ConfigParamSpec{std::string{"control:"} + type, VariantType::Bool, value, {"\"\""}});
547  std::sort(entryMetadata.begin(), entryMetadata.end(), [](ConfigParamSpec const& a, ConfigParamSpec const& b) { return a.name < b.name; });
548  auto new_end = std::unique(entryMetadata.begin(), entryMetadata.end(), [](ConfigParamSpec const& a, ConfigParamSpec const& b) { return a.name == b.name; });
549  entryMetadata.erase(new_end, entryMetadata.end());
550  } else {
551  //add entry
552  spec.metadata.push_back(ConfigParamSpec{std::string{"control:"} + type, VariantType::Bool, value, {"\"\""}});
553  inputs.emplace_back(spec);
554  }
555 }
