48 void Initializer::setup()
49 {
50  TEnv settings;
51  ConfigurationManager::getInstance().getConfig(settings);
52 
53  const bool fullscreen = settings.GetValue("fullscreen.mode", false); // hide left and bottom tabs
54  const string ocdbStorage = settings.GetValue("OCDB.default.path", o2::base::NameConf::getCCDBServer().c_str()); // default path to OCDB
55  LOG(info) << "Initializer -- OCDB path:" << ocdbStorage;
56 
57  auto& eventManager = EventManager::getInstance();
58  eventManager.setCdbPath(ocdbStorage);
59 
60  auto const options = Options::Instance();
61 
62  if (options->online()) {
63  eventManager.setDataSource(new DataSourceOnline(options->dataFolder()));
64  } else {
65  eventManager.setDataSource(new DataSourceOffline(options->AODConverterPath(), options->dataFolder(), options->fileName(), options->hideDplGUI()));
66  }
67 
68  eventManager.getDataSource()->registerReader(new DataReaderJSON());
69 
70  setupGeometry();
71  gSystem->ProcessEvents();
72  gEve->Redraw3D(true);
73 
74  setupBackground();
75 
76  // Setup windows size, fullscreen and focus
77  TEveBrowser* browser = gEve->GetBrowser();
78  browser->GetTabRight()->SetTab(1);
79  browser->MoveResize(0, 0, gClient->GetDisplayWidth(), gClient->GetDisplayHeight() - 32);
80 
81  browser->StartEmbedding(TRootBrowser::kBottom);
82  EventManagerFrame* frame = new EventManagerFrame(eventManager);
83  browser->StopEmbedding("EventCtrl");
84 
85  if (fullscreen) {
86  ((TGWindow*)gEve->GetBrowser()->GetTabLeft()->GetParent())->Resize(1, 0);
87  ((TGWindow*)gEve->GetBrowser()->GetTabBottom()->GetParent())->Resize(0, 1);
88  }
89  gEve->GetBrowser()->Layout();
90  gSystem->ProcessEvents();
91 
92  setupCamera();
93 
94  // Temporary:
95  // Later this will be triggered by button, and finally moved to configuration.
96  gEve->AddEvent(&EventManager::getInstance());
97  // eventManager.getDataSource()->refresh();
98 
99  if (Options::Instance()->online()) {
100  frame->StartTimer();
101  } else {
102  eventManager.getDataSource()->refresh();
103  frame->DoFirstEvent();
104  }
105  gApplication->Connect("TEveBrowser", "CloseWindow()", "o2::event_visualisation::EventManagerFrame", frame, "DoTerminate()");
106 }

108 void Initializer::setupGeometry()
109 {
110  // read path to geometry files from config file
111  TEnv settings;
112  ConfigurationManager::getInstance().getConfig(settings);
113 
114  // get geometry from Geometry Manager and register in multiview
115  auto multiView = MultiView::getInstance();
116 
117  // auto geometry_enabled = GeometryManager::getInstance().getR2Geometry()? R2Visualisation:R3Visualisation;
118  for (int iDet = 0; iDet < NvisualisationGroups; ++iDet) {
119  if (!R3Visualisation[iDet]) {
120  continue;
121  }
122  EVisualisationGroup det = static_cast<EVisualisationGroup>(iDet);
123  string detName = gVisualisationGroupName[det];
124  LOG(info) << detName;
125 
126  if (settings.GetValue((detName + ".draw").c_str(), false)) {
127  if (detName == "TPC" || detName == "MCH" || detName == "MID" || detName == "MFT") { // don't load MUON+MFT and AD and standard TPC to R-Phi view
128 
129  multiView->drawGeometryForDetector(detName, true, false);
130  } else if (detName == "RPH") { // special TPC geom from R-Phi view
131 
132  multiView->drawGeometryForDetector(detName, false, true, false);
133  } else { // default
134  if (detName != "ACO") {
135  multiView->drawGeometryForDetector(detName);
136  }
137  }
138  }
139  }

142 void Initializer::setupCamera()
143 {
144  // move and rotate sub-views
145  TEnv settings;
146  ConfigurationManager::getInstance().getConfig(settings);
147 
148  // read settings from config file
149  const double angleHorizontal = settings.GetValue("camera.3D.rotation.horizontal", -0.4);
150  const double angleVertical = settings.GetValue("camera.3D.rotation.vertical", 1.0);
151 
152  double zoom[MultiView::NumberOfViews];
153  zoom[MultiView::View3d] = settings.GetValue("camera.3D.zoom", 1.0);
154  zoom[MultiView::ViewRphi] = settings.GetValue("camera.R-Phi.zoom", 1.0);
155  zoom[MultiView::ViewZrho] = settings.GetValue("camera.Rho-Z.zoom", 1.0);
156 
157  // get necessary elements of the multiview and set camera position
158  auto multiView = MultiView::getInstance();
159 
160  for (int viewIter = 0; viewIter < MultiView::NumberOfViews; ++viewIter) {
161  TGLViewer* glView = multiView->getView(static_cast<MultiView::EViews>(viewIter))->GetGLViewer();
162  glView->CurrentCamera().Reset();
163 
164  if (viewIter == 0) {
165  glView->CurrentCamera().RotateRad(angleHorizontal, angleVertical);
166  }
167  glView->CurrentCamera().Dolly(zoom[viewIter], kFALSE, kTRUE);
168  }
169 }
