67 class TPCSectorCompletionPolicy
68 {
69  public:
70  using CompletionPolicyData = std::vector<framework::InputSpec>;
71 
72  enum struct Config {
73  // require data on all other inputs in addition to the ones checked for the sector completion
74  RequireAll,
75  };
76  TPCSectorCompletionPolicy() = delete;
77  template <typename... Args>
78  TPCSectorCompletionPolicy(const char* processorName, Args&&... args)
79  : mProcessorName(processorName), mInputMatchers()
80  {
81  init(std::forward<Args>(args)...);
82  }
83 
84  o2::framework::CompletionPolicy operator()()
85  {
86  constexpr static size_t NSectors = o2::tpc::Sector::MAXSECTOR;
87 
88  auto matcher = [expression = mProcessorName](framework::DeviceSpec const& device) -> bool {
89  return std::regex_match(device.name.begin(), device.name.end(), std::regex(expression.c_str()));
90  };
91 
92  auto callback = [bRequireAll = mRequireAll, inputMatchers = mInputMatchers, externalInputMatchers = mExternalInputMatchers, pTpcSectorMask = mTpcSectorMask](framework::InputSpan const& inputs) -> framework::CompletionPolicy::CompletionOp {
93  unsigned long tpcSectorMask = pTpcSectorMask ? *pTpcSectorMask : 0xFFFFFFFFF;
94  std::bitset<NSectors> validSectors = 0;
95  bool haveMatchedInput = false;
96  uint64_t activeSectors = 0;
97  std::vector<uint64_t> validSectorsExternal(externalInputMatchers ? externalInputMatchers->size() : 0);
98  size_t nActiveInputRoutes = 0;
99  size_t nMaxPartsPerRoute = 0;
100  int inputType = -1;
101  for (auto it = inputs.begin(), end = inputs.end(); it != end; ++it) {
102  nMaxPartsPerRoute = it.size() > nMaxPartsPerRoute ? it.size() : nMaxPartsPerRoute;
103  bool haveActivePart = false;
104  for (auto const& ref : it) {
105  if (!framework::DataRefUtils::isValid(ref)) {
106  continue;
107  }
108  haveActivePart = true;
109  auto const* dh = framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
110  // check if the O2 message matches on of the input specs to be matched and
111  // if it matches, check for the sector header, retrieve the active sector information
112  // and mark the sector as valid, we require to match exactly one of the inputs in the list
113  for (size_t idx = 0, end = inputMatchers.size(); idx < end; idx++) {
114  auto const& spec = inputMatchers[idx];
115  if (framework::DataRefUtils::match(ref, spec)) {
116  haveMatchedInput = true;
117  if (inputType == -1) {
118  // we bind to the index of the first match and require all other inputs to match the same spec
119  inputType = idx;
120  } else if (inputType != idx) {
121  std::stringstream error;
122  error << fmt::format("routing error, input messages must all be of the same type previously bound to {} {}/{}/{}",
123  inputMatchers[inputType],
124  dh->dataOrigin,
125  dh->dataDescription, dh->subSpecification);
126  throw std::runtime_error(error.str());
127  }
128  auto const* sectorHeader = framework::DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
129  if (sectorHeader == nullptr) {
130  throw std::runtime_error("TPC sector header missing on header stack");
131  }
132  activeSectors |= (sectorHeader->activeSectors & tpcSectorMask);
133  validSectors |= (sectorHeader->sectorBits & tpcSectorMask);
134  break;
135  }
136  }
137 
138  // We require to match all inputs in the external list
139  if (externalInputMatchers) {
140  for (size_t idx = 0, end = externalInputMatchers->size(); idx < end; idx++) {
141  auto const& spec = (*externalInputMatchers)[idx];
142  if (framework::DataRefUtils::match(ref, spec)) {
143  auto const* sectorHeader = framework::DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
144  if (sectorHeader == nullptr) {
145  throw std::runtime_error("TPC sector header missing on header stack");
146  }
147  activeSectors |= (sectorHeader->activeSectors & tpcSectorMask);
148  validSectorsExternal[idx] |= (sectorHeader->sectorBits & tpcSectorMask);
149  break;
150  }
151  }
152  }
153  }
154  if (haveActivePart) {
155  ++nActiveInputRoutes;
156  }
157  }
158 
159  if (externalInputMatchers) {
160  // We require all external matchers to have all sectors present, if not we wait
161  for (size_t idx = 0, end = externalInputMatchers->size(); idx < end; idx++) {
162  if (validSectorsExternal[idx] == 0 || validSectorsExternal[idx] != activeSectors) {
163  return framework::CompletionPolicy::CompletionOp::Wait;
164  }
165  }
166  }
167 
168  // If the flag Config::RequireAll is set in the constructor arguments we require
169  // data from all inputs in addition to the sector matching condition
170  // To be fully correct we would need to require data from all inputs not going
171  // into the TPC policy, but that is not possible for the moment. That's why there is a possibly
172  // unhandled case if multiple TPC input routes are defined but a complete data set is coming over
173  // one of them. Not likely to be a use case, though.
174  if ((inputMatchers.size() == 0 || (haveMatchedInput && activeSectors == validSectors.to_ulong())) &&
175  (!bRequireAll || nActiveInputRoutes == inputs.size())) {
176  // we can process if there is input for all sectors, the required sectors are
177  // transported as part of the sector header
178  return framework::CompletionPolicy::CompletionOp::Consume;
179  } else if (activeSectors == 0 && nActiveInputRoutes == inputs.size()) {
180  // no sector header is transmitted, this is the case for e.g. the ZS raw data
181  // we simply require input on all routes, this is also the default of DPL DataRelayer
182  // Because DPL can not do more without knowing how many parts are required for a complete
183  // data set, the workflow should be such that exactly one O2 message arrives per input route.
184  // Currently, the workflow has multiple O2 messages per input route, but they all come in
185  // a single multipart message. So it works fine, and we disable the warning below, but there
186  // is a potential problem. Need to fix this on the level of the workflow.
187  //if (nMaxPartsPerRoute > 1) {
188  // LOG(warning) << "No sector information is provided with the data, data set is complete with data on all input routes. But there are multiple parts on at least one route and this policy might not be complete, no check possible if other parts on some routes are still missing. It is adviced to add a custom policy.";
189  //}
190  return framework::CompletionPolicy::CompletionOp::Consume;
191  }
192 
193  return framework::CompletionPolicy::CompletionOp::Wait;
194  };
195  return framework::CompletionPolicy{"TPCSectorCompletionPolicy", matcher, callback};
196  }
197 
198  private:
200  template <typename Arg, typename... Args>
201  void init(Arg&& arg, Args&&... args)
202  {
203  using Type = std::decay_t<Arg>;
204  if constexpr (std::is_same_v<Type, framework::InputSpec>) {
205  mInputMatchers.emplace_back(std::move(arg));
206  } else if constexpr (std::is_same_v<Type, TPCSectorCompletionPolicy::Config>) {
207  switch (arg) {
208  case Config::RequireAll:
209  mRequireAll = true;
210  break;
211  }
212  } else if constexpr (std::is_same_v<Type, std::vector<o2::framework::InputSpec>*>) {
213  mExternalInputMatchers = arg;
214  } else if constexpr (std::is_same_v<Type, unsigned long*> || std::is_same_v<Type, const unsigned long*>) {
215  mTpcSectorMask = arg;
216  } else {
217  static_assert(framework::always_static_assert_v<Type>);
218  }
219  if constexpr (sizeof...(args) > 0) {
220  init(std::forward<Args>(args)...);
221  }
222  }
223 
224  std::string mProcessorName;
225  std::vector<framework::InputSpec> mInputMatchers;
226  // The external input matchers behave as the internal ones with the following differences:
227  // - They are controlled externally and the external entity can modify them, e.g. after parsing command line arguments.
228  // - They are all matched independently, it is not sufficient that one of them is present for all sectors
229  const std::vector<framework::InputSpec>* mExternalInputMatchers = nullptr;
230  const unsigned long* mTpcSectorMask = nullptr;
231  bool mRequireAll = false;
232 };
