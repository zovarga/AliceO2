60 int TOFDCSProcessor::process(const gsl::span<const DPCOM> dps)
61 {
62 
63  // first we check which DPs are missing - if some are, it means that
64  // the delta map was sent
65  if (mVerbose) {
66  LOG(info) << "\n\n\nProcessing new DCS DP map\n-----------------";
67  }
68  if (!mFirstTimeSet) {
69  mFirstTime = mStartValidity;
70  mFirstTimeSet = true;
71  }
72 
73  std::unordered_map<DPID, DPVAL> mapin;
74  for (auto& it : dps) {
75  mapin[it.id] = it.data;
76  }
77  for (auto& it : mPids) {
78  const auto& el = mapin.find(it.first);
79  if (el == mapin.end()) {
80  LOG(debug) << "DP " << it.first << " not found in map";
81  } else {
82  LOG(debug) << "DP " << it.first << " found in map";
83  }
84  }
85 
86  mUpdateFeacStatus = false; // by default, we do not foresee a new entry in the CCDB for the FEAC
87  mUpdateHVStatus = false; // by default, we do not foresee a new entry in the CCDB for the HV
88 
89  // now we process all DPs, one by one
90  for (const auto& it : dps) {
91  // we process only the DPs defined in the configuration
92  const auto& el = mPids.find(it.id);
93  if (el == mPids.end()) {
94  LOG(info) << "DP " << it.id << " not found in TOFDCSProcessor, we will not process it";
95  continue;
96  }
97  processDP(it);
98  mPids[it.id] = true;
99  }
100 
101  if (mUpdateFeacStatus) {
102  updateFEACCCDB();
103  }
104 
105  if (mUpdateHVStatus) {
106  updateHVCCDB();
107  }
108 
109  return 0;
110 }

114 int TOFDCSProcessor::processDP(const DPCOM& dpcom)
115 {
116 
117  // processing single DP
118 
119  auto& dpid = dpcom.id;
120  const auto& type = dpid.get_type();
121  auto& val = dpcom.data;
122  if (mVerbose) {
123  if (type == RAW_DOUBLE) {
124  LOG(info);
125  LOG(info) << "Processing DP = " << dpcom << ", with value = " << o2::dcs::getValue<double>(dpcom);
126  } else if (type == RAW_INT) {
127  LOG(info);
128  LOG(info) << "Processing DP = " << dpcom << ", with value = " << o2::dcs::getValue<int32_t>(dpcom);
129  }
130  }
131  auto flags = val.get_flags();
132  if (processFlags(flags, dpid.get_alias()) == 0) {
133  // now I need to access the correct element
134  if (type == RAW_DOUBLE) {
135  // for these DPs, we will store the first, last, mid value, plus the value where the maximum variation occurred
136  auto& dvect = mDpsdoublesmap[dpid];
137  LOG(debug) << "mDpsdoublesmap[dpid].size() = " << dvect.size();
138  auto etime = val.get_epoch_time();
139  if (dvect.size() == 0 ||
140  etime != dvect.back().get_epoch_time()) { // we check
141  // that we did not get the
142  // same timestamp as the
143  // latest one
144  dvect.push_back(val);
145  }
146  }
147 
148  if (type == RAW_INT) {
149  // for these DPs, we need some processing
150  if (std::strstr(dpid.get_alias(), "FEACSTATUS") != nullptr) { // DP is FEACSTATUS
151  std::string aliasStr(dpid.get_alias());
152  // extracting DDL number (regex is quite slow, using this way)
153  const auto offs = std::strlen("TOF_FEACSTATUS_");
154  std::size_t const nn = aliasStr.find_first_of("0123456789", offs);
155  std::size_t const mm = aliasStr.find_first_not_of("0123456789", nn);
156  std::string ddlStr = aliasStr.substr(nn, mm != std::string::npos ? mm - nn : mm);
157  auto iddl = std::stoi(ddlStr);
158  std::bitset<8> feacstatus(o2::dcs::getValue<int32_t>(dpcom));
159  if (mVerbose) {
160  LOG(info) << "DDL: " << iddl << ": Prev FEAC = " << mPrevFEACstatus[iddl] << ", new = " << feacstatus;
161  }
162  if (feacstatus == mPrevFEACstatus[iddl]) {
163  if (mVerbose) {
164  LOG(info) << "Same FEAC status as before, we do nothing";
165  }
166  return 0;
167  }
168  if (mVerbose) {
169  LOG(info) << "Something changed in LV for DDL " << iddl << ", we need to check what";
170  }
171  mUpdateFeacStatus = true;
172  int plate = -1, strip = -1;
173  int det[5] = {iddl / 4, -1, -1, -1, -1};
174  for (auto ifeac = 0; ifeac < NFEACS; ++ifeac) { // we have one bit per FEAC
175  auto singlefeacstatus = feacstatus[ifeac];
176  for (int istrip = 0; istrip < 6; ++istrip) {
177  if (mFeacInfo[iddl][ifeac].stripInSM[istrip] == -1) {
178  continue;
179  }
180  for (int ipadz = 0; ipadz < Geo::NPADZ; ++ipadz) {
181  for (int ipadx = mFeacInfo[iddl][ifeac].firstPadX; ipadx <= mFeacInfo[iddl][ifeac].lastPadX; ++ipadx) {
182  if (mVerbose) {
183  LOG(info) << "mFeacInfo[" << iddl << "][" << ifeac << "].stripInSM[" << istrip << "] = " << mFeacInfo[iddl][ifeac].stripInSM[istrip];
184  }
185  Geo::getStripAndModule(mFeacInfo[iddl][ifeac].stripInSM[istrip], plate, strip);
186  det[1] = plate;
187  det[2] = strip;
188  det[3] = ipadz;
189  det[4] = ipadx;
190  if (mVerbose) {
191  LOG(info) << "det[0] = " << det[0] << ", det[1] = " << det[1] << ", det[2] = " << det[2] << ", det[3] = " << det[3] << ", det[4] = " << det[4];
192  }
193  int channelIdx = Geo::getIndex(det);
194  if (mFeac[channelIdx] != singlefeacstatus) {
195  mFeac[channelIdx] = singlefeacstatus;
196  }
197  }
198  }
199  }
200  } // end loop on FEACs
201  if (mVerbose) {
202  LOG(info) << "Updating previous FEAC status for DDL " << iddl;
203  }
204  mPrevFEACstatus[iddl] = feacstatus;
205  } // end processing current DP, when it is of type FEACSTATUS
206 
207  if (std::strstr(dpid.get_alias(), "HVSTATUS") != nullptr) { // DP is HVSTATUS
208  std::string aliasStr(dpid.get_alias());
209  // extracting SECTOR and PLATE number (regex is quite slow, using this way)
210  const auto offs = std::strlen("TOF_HVSTATUS_SM");
211  std::size_t const nn = aliasStr.find_first_of("0123456789", offs);
212  std::size_t const mm = aliasStr.find_first_not_of("0123456789", nn);
213  std::size_t const oo = aliasStr.find_first_of("0123456789", mm);
214  std::size_t const pp = aliasStr.find_first_not_of("0123456789", oo);
215  std::string sectorStr = aliasStr.substr(nn, mm != std::string::npos ? mm - nn : mm);
216  auto isect = std::stoi(sectorStr);
217  std::string plateStr = aliasStr.substr(oo, pp != std::string::npos ? pp - oo : pp);
218  auto iplat = std::stoi(plateStr);
219  std::bitset<19> hvstatus(o2::dcs::getValue<int32_t>(dpcom));
220  if (mVerbose) {
221  LOG(info) << "Sector: " << isect << ", plate = " << iplat << ": Prev HV = "
222  << mPrevHVstatus[iplat][isect] << ", new = " << hvstatus;
223  }
224  if (hvstatus == mPrevHVstatus[iplat][isect]) {
225  if (mVerbose) {
226  LOG(info) << "Same HV status as before, we do nothing";
227  }
228  return 0;
229  }
230  if (mVerbose) {
231  LOG(info) << "Something changed in HV for Sect " << isect << " and plate "
232  << iplat << ", we need to check what";
233  }
234  mUpdateHVStatus = true;
235  int det[5] = {isect, iplat, -1, -1, -1};
236  auto nStrips = (iplat == 2 ? Geo::NSTRIPA : (iplat == 0 || iplat == 4) ? Geo::NSTRIPC : Geo::NSTRIPB);
237  for (auto istrip = 0; istrip < nStrips; ++istrip) {
238  auto singlestripHV = hvstatus[istrip];
239  for (int ipadz = 0; ipadz < Geo::NPADZ; ++ipadz) {
240  for (int ipadx = 0; ipadx < Geo::NPADX; ++ipadx) {
241  det[2] = istrip;
242  det[3] = ipadz;
243  det[4] = ipadx;
244  int channelIdx = Geo::getIndex(det);
245  if (mHV[channelIdx] != singlestripHV) {
246  mHV[channelIdx] = singlestripHV;
247  }
248  }
249  }
250  } // end loop on strips
251  if (mVerbose) {
252  LOG(info) << "Updating previous HV status for Sector: " << isect << ", plate = " << iplat;
253  }
254  mPrevHVstatus[iplat][isect] = hvstatus;
255  } //end processing current DP, when it is of type HVSTATUS
256  }
257  }
258  return 0;
259 }

325 void TOFDCSProcessor::updateDPsCCDB()
326 {
327 
328  // here we create the object to then be sent to CCDB
329  LOG(info) << "Finalizing";
330  union Converter {
331  uint64_t raw_data;
332  double double_value;
333  } converter0, converter1;
334 
335  for (const auto& it : mPids) {
336  const auto& type = it.first.get_type();
337  if (type == o2::dcs::RAW_DOUBLE) {
338  auto& tofdcs = mTOFDCS[it.first];
339  if (it.second == true) { // we processed the DP at least 1x
340  auto& dpvect = mDpsdoublesmap[it.first];
341  tofdcs.firstValue.first = dpvect[0].get_epoch_time();
342  converter0.raw_data = dpvect[0].payload_pt1;
343  tofdcs.firstValue.second = converter0.double_value;
344  tofdcs.lastValue.first = dpvect.back().get_epoch_time();
345  converter0.raw_data = dpvect.back().payload_pt1;
346  tofdcs.lastValue.second = converter0.double_value;
347  // now I will look for the max change
348  if (dpvect.size() > 1) {
349  auto deltatime = dpvect.back().get_epoch_time() - dpvect[0].get_epoch_time();
350  if (deltatime < 60000) {
351  // if we did not cover at least 1 minute,
352  // max variation is defined as the difference between first and last value
353  converter0.raw_data = dpvect[0].payload_pt1;
354  converter1.raw_data = dpvect.back().payload_pt1;
355  double delta = std::abs(converter0.double_value - converter1.double_value);
356  tofdcs.maxChange.first = deltatime; // is it ok to do like this, as in Run 2?
357  tofdcs.maxChange.second = delta;
358  } else {
359  for (auto i = 0; i < dpvect.size() - 1; ++i) {
360  for (auto j = i + 1; j < dpvect.size(); ++j) {
361  auto deltatime = dpvect[j].get_epoch_time() - dpvect[i].get_epoch_time();
362  if (deltatime >= 60000) { // we check every min; epoch_time in ms
363  converter0.raw_data = dpvect[i].payload_pt1;
364  converter1.raw_data = dpvect[j].payload_pt1;
365  double delta = std::abs(converter0.double_value - converter1.double_value);
366  if (delta > tofdcs.maxChange.second) {
367  tofdcs.maxChange.first = deltatime; // is it ok to do like this, as in Run 2?
368  tofdcs.maxChange.second = delta;
369  }
370  }
371  }
372  }
373  }
374  // mid point
375  auto midIdx = dpvect.size() / 2 - 1;
376  tofdcs.midValue.first = dpvect[midIdx].get_epoch_time();
377  converter0.raw_data = dpvect[midIdx].payload_pt1;
378  tofdcs.midValue.second = converter0.double_value;
379  } else {
380  tofdcs.maxChange.first = dpvect[0].get_epoch_time();
381  converter0.raw_data = dpvect[0].payload_pt1;
382  tofdcs.maxChange.second = converter0.double_value;
383  tofdcs.midValue.first = dpvect[0].get_epoch_time();
384  converter0.raw_data = dpvect[0].payload_pt1;
385  tofdcs.midValue.second = converter0.double_value;
386  }
387  }
388  if (mVerbose) {
389  LOG(info) << "PID = " << it.first.get_alias();
390  tofdcs.print();
391  }
392  }
393  }
394  std::map<std::string, std::string> md;
395  md["responsible"] = "Chiara Zampolli";
396  o2::calibration::Utils::prepareCCDBobjectInfo(mTOFDCS, mccdbDPsInfo, "TOF/Calib/DCSDPs", md, mStartValidity, o2::calibration::Utils::INFINITE_TIME);
397 
398  return;
399 }

447  case 0:
448  info.firstPadX = 0;

509  case 2:
510  info.firstPadX = Geo::NPADX / 2;

521  } else if (nFEAC == 4) {

525  } else if (nFEAC == 5) {

540  case 3:
541  info.firstPadX = 0;
542  info.lastPadX = Geo::NPADX / 2 - 1;

544  if (nFEAC <= 2) {
