86 class ConcreteRawParser
87 {
88  public:
89  using buffer_type = unsigned char;

98  template <typename T>
99  ConcreteRawParser(T const* buffer, size_t size)
100  : mRawBuffer(reinterpret_cast<buffer_type const*>(buffer)), mSize(size)
101  {
102  static_assert(sizeof(T) == sizeof(buffer_type), "buffer required to be byte-type");
103  if (size < sizeof(header_type)) {
104  std::runtime_error("buffer too small to fit at least the page header");
105  }
106  next();
107  }
108  // FIXME: add constructor for container type buffer
109 
113  ConcreteRawParser(ConcreteRawParser const& other) = default;
114 
116  header_type const& header() const
117  {
118  return *reinterpret_cast<header_type const*>(mPosition);
119  }
120 
122  size_t size() const
123  {
124  if (mPosition == mRawBuffer + mSize) {
125  return 0;
126  }
127  header_type const& h = header();
128  if (h.memorySize >= h.headerSize) {
129  return h.memorySize - h.headerSize;
130  }
131  return max_size - h.headerSize;
132  }
133 
135  buffer_type const* data() const
136  {
137  size_t size = this->size();
138  if (size == 0) {
139  return nullptr;
140  }
141  header_type const& h = header();
142  if (mPosition + size + h.headerSize > mRawBuffer + mSize) {
143  throw std::runtime_error("not enough data at position " + std::to_string(mPosition - mRawBuffer));
144  }
145  return mPosition + h.headerSize;
146  }
147 
149  buffer_type const* raw() const
150  {
151  if (mPosition < mRawBuffer + mSize) {
152  return mPosition;
153  }
154  return nullptr;
155  }
156 
158  size_t offset() const
159  {
160  if (mPosition < mRawBuffer + mSize) {
161  header_type const& h = header();
162  return h.headerSize;
163  }
164  return 0;
165  }

171  template <typename Processor>
172  void parse(Processor&& processor)
173  {
174  reset();
175  //auto deleter = [](buffer_type*) {};
176  do {
177  processor(data(), size());
178  //processor(std::unique_ptr<buffer_type, decltype(deleter)>(data(), deleter), size());
179  } while (next());
180  }
181 
183  bool next()
184  {
185  if (mPosition == nullptr) {
186  mPosition = mRawBuffer;
187  } else {
188  auto offset = header().offsetToNext;
189  if ((mPosition + offset + sizeof(header_type) > mRawBuffer + mSize) || (offset < sizeof(header_type))) {
190  // FIXME: should check if there is unread data left of if we just reached
191  // the end of the buffer by parsing all pages
192  mPosition = mRawBuffer + mSize;
193  return false;
194  }
195  mPosition += offset;
196  }
197  // FIXME: check page header validity: version, size
198  return true;
199  }
200 
202  void reset()
203  {
204  mPosition = mRawBuffer;
205  }
206 
210  void advance(int step)
211  {
212  if (step < 0) {
213  // set beyond the buffer
214  mPosition = mRawBuffer + mSize;
215  } else {
216  while ((step-- > 0) && next()) {
217  };
218  }
219  }
220 
223  template <typename T = self_type>
224  bool operator==(T const& other) const
225  {
226  if constexpr (std::is_same<T, self_type>::value == true) {
227  return mRawBuffer == other.mRawBuffer && mPosition == other.mPosition;
228  } else {
229  throw std::runtime_error(std::string("incompatible types for comparison ") + typeid(T).name() + "/" + typeid(self_type).name());
230  }
231  }

233  void format(std::ostream& os, FormatSpec choice = FormatSpec::Entry, const char* delimiter = "\n") const
234  {
235  RDHFormatter<header_type>::apply(os, header(), choice, delimiter);
236  }

238  private:
239  buffer_type const* mRawBuffer;
240  buffer_type const* mPosition = nullptr;
241  size_t mSize;

273

289 template <size_t N, typename T, typename P>
290 void walk_parse(T& instances, P&& processor, size_t index)
291 {
292  if constexpr (N > 0) {
293  if (index == N - 1) {
294  std::get<N - 1>(instances).parse(processor);
295  }
296 
297  walk_parse<N - 1>(instances, processor, index);
298  }
299 }

301 template <typename U, typename T, size_t N = std::variant_size_v<T>>
302 U const* get_if(T& instances)
303 {
304  if constexpr (N > 0) {
305  auto* parser = std::get_if<N - 1>(&instances);
306  if (parser) {
307  // we are in the active instance, return header if type matches
308  using parser_type = typename std::variant_alternative<N - 1, T>::type;
309  using header_type = typename parser_type::header_type;
310  if constexpr (std::is_same<U, header_type>::value == true) {
311  return &(parser->header());
312  }
313  } else {
314  // continue walking through instances until active one is found
315  return get_if<U, T, N - 1>(instances);
316  }
317  }
318  return nullptr;
319 }
