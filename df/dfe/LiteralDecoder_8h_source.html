55 template <typename stream_IT, typename source_IT, std::enable_if_t<internal::isCompatibleIter_v<stream_T, stream_IT>, bool>>
56 void LiteralDecoder<coder_T, stream_T, source_T>::process(stream_IT inputEnd, source_IT outputBegin, size_t messageLength, std::vector<source_T>& literals) const
57 {
58  using namespace internal;
59  LOG(trace) << "start decoding";
60  RANSTimer t;
61 
62 #ifdef O2_RANS_PRINT_PROCESSED_DATA
63  JSONArrayLogger<source_T> arrayLogger{};
64 #endif
65 
66  t.start();
67 
68  if (messageLength == 0) {
69  LOG(warning) << "Empty message passed to decoder, skipping decode process";
70  return;
71  }
72 
73  stream_IT inputIter = inputEnd;
74  source_IT it = outputBegin;
75 
76  auto decode = [&, this](ransDecoder_t& decoder) {
77  const auto cumul = decoder.get();
78  const auto streamSymbol = (this->mReverseLUT)[cumul];
79  source_T symbol = streamSymbol;
80  if (this->mSymbolTable.isEscapeSymbol(streamSymbol)) {
81  symbol = literals.back();
82  literals.pop_back();
83  }
84 
85 #ifdef O2_RANS_PRINT_PROCESSED_DATA
86  arrayLogger << symbol;
87 #endif
88 
89  return std::make_tuple(symbol, decoder.advanceSymbol(inputIter, (this->mSymbolTable)[streamSymbol]));
90  };
91 
92  // make Iter point to the last last element
93  --inputIter;
94 
95  ransDecoder_t rans0{this->mSymbolTable.getPrecision()};
96  ransDecoder_t rans1{this->mSymbolTable.getPrecision()};
97  inputIter = rans0.init(inputIter);
98  inputIter = rans1.init(inputIter);
99 
100  for (size_t i = 0; i < (messageLength & ~1); i += 2) {
101  std::tie(*it++, inputIter) = decode(rans0);
102  std::tie(*it++, inputIter) = decode(rans1);
103  }
104 
105  // last byte, if message length was odd
106  if (messageLength & 1) {
107  std::tie(*it++, inputIter) = decode(rans0);
108  }
109  t.stop();
110 
111 #ifdef O2_RANS_PRINT_PROCESSED_DATA
112  LOG(info) << "decoderOutput:" << arrayLogger;
113 #endif
114 
115  LOG(debug1) << "Decoder::" << __func__ << " { DecodedSymbols: " << messageLength << ","
116  << "processedBytes: " << messageLength * sizeof(source_T) << ","
117  << " inclusiveTimeMS: " << t.getDurationMS() << ","
118  << " BandwidthMiBPS: " << std::fixed << std::setprecision(2) << (messageLength * sizeof(source_T) * 1.0) / (t.getDurationS() * 1.0 * (1 << 20)) << "}";
119 
120  LOG(trace) << "done decoding";
121 }
