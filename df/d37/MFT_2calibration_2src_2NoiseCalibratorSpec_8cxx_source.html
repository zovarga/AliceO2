42 void NoiseCalibratorSpec::init(InitContext& ic)
43 {
44  auto probT = ic.options().get<float>("prob-threshold");
45  LOG(info) << "Setting the probability threshold to " << probT;
46  std::string dictPath = o2::itsmft::ClustererParam<o2::detectors::DetID::MFT>::Instance().dictFilePath;
47  std::string dictFile = o2::base::DetectorNameConf::getAlpideClusterDictionaryFileName(o2::detectors::DetID::MFT, dictPath);
48  if (o2::utils::Str::pathExists(dictFile)) {
49  mCalibrator->loadDictionary(dictFile);
50  LOG(info) << "MFT NoiseCalibrator is running with a provided dictionary: " << dictFile;
51  } else {
52  LOG(info) << "Dictionary " << dictFile
53  << " is absent, MFT NoiseCalibrator expects cluster patterns for all clusters";
54  }
55 
56  mPath = ic.options().get<std::string>("path");
57  mMeta = ic.options().get<std::string>("meta");
58  mStart = ic.options().get<int64_t>("tstart");
59  mEnd = ic.options().get<int64_t>("tend");
60 
61  mCalibrator = std::make_unique<CALIBRATOR>(probT);
62 }

64 void NoiseCalibratorSpec::run(ProcessingContext& pc)
65 {
66  if (mDigits) {
67  const auto digits = pc.inputs().get<gsl::span<o2::itsmft::Digit>>("digits");
68  const auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("digitsROF");
69  const auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("digitsROF").header)->startTime;
70 
71  if (mCalibrator->processTimeFrame(tfcounter, digits, rofs)) {
72  LOG(info) << "Minimum number of noise counts has been reached !";
73  sendOutput(pc.outputs());
74  pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
75  }
76  } else {
77  const auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
78  gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
79  const auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
80  const auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("ROframes").header)->startTime;
81 
82  if (mCalibrator->processTimeFrame(tfcounter, compClusters, patterns, rofs)) {
83  LOG(info) << "Minimum number of noise counts has been reached !";
84  sendOutput(pc.outputs());
85  pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
86  }
87  }
88 }

90 void NoiseCalibratorSpec::sendOutput(DataAllocator& output)
91 {
92  mCalibrator->finalize();
93 
94  long tstart = mStart;
95  if (tstart == -1) {
96  tstart = o2::ccdb::getCurrentTimestamp();
97  }
98  long tend = mEnd;
99  if (tend == -1) {
100  constexpr long SECONDSPERYEAR = 365 * 24 * 60 * 60;
101  tend = o2::ccdb::getFutureTimestamp(SECONDSPERYEAR);
102  }
103 
104  std::map<std::string, std::string> meta;
105  auto toKeyValPairs = [&meta](std::vector<std::string> const& tokens) {
106  for (auto& token : tokens) {
107  auto keyval = Str::tokenize(token, '=', false);
108  if (keyval.size() != 2) {
109  LOG(error) << "Illegal command-line key/value string: " << token;
110  continue;
111  }
112  Str::trim(keyval[1]);
113  meta[keyval[0]] = keyval[1];
114  }
115  };
116  toKeyValPairs(Str::tokenize(mMeta, ';', true));
117 
118  long startTF, endTF;
119 
120  const auto& payload = mCalibrator->getNoiseMap();
121  // const auto& payload = mCalibrator->getNoiseMap(starTF, endTF); //For TimeSlot calibration
122 
123  o2::ccdb::CcdbObjectInfo info(mPath, "NoiseMap", "noise.root", meta, tstart, tend);
124  auto flName = o2::ccdb::CcdbApi::generateFileName("noise");
125  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
126  info.setFileName(flName);
127  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName()
128  << " of size " << image->size()
129  << " bytes, valid for " << info.getStartValidityTimestamp()
130  << " : " << info.getEndValidityTimestamp();
131 
132  using clbUtils = o2::calibration::Utils;
133  output.snapshot(Output{clbUtils::gDataOriginCDBPayload, "MFT_NoiseMap", 0}, *image.get());
134  output.snapshot(Output{clbUtils::gDataOriginCDBWrapper, "MFT_NoiseMap", 0}, info);
135 }

142 DataProcessorSpec getNoiseCalibratorSpec(bool useDigits)
143 {
144  o2::header::DataOrigin detOrig = o2::header::gDataOriginMFT;
145  std::vector<InputSpec> inputs;
146  if (useDigits) {
147  inputs.emplace_back("digits", detOrig, "DIGITS", 0, Lifetime::Timeframe);
148  inputs.emplace_back("digitsROF", detOrig, "DIGITSROF", 0, Lifetime::Timeframe);
149  } else {
150  inputs.emplace_back("compClusters", detOrig, "COMPCLUSTERS", 0, Lifetime::Timeframe);
151  inputs.emplace_back("patterns", detOrig, "PATTERNS", 0, Lifetime::Timeframe);
152  inputs.emplace_back("ROframes", detOrig, "CLUSTERSROF", 0, Lifetime::Timeframe);
153  }
154 
155  using clbUtils = o2::calibration::Utils;
156  std::vector<OutputSpec> outputs;
157  outputs.emplace_back(ConcreteDataTypeMatcher{clbUtils::gDataOriginCDBPayload, "MFT_NoiseMap"}, Lifetime::Sporadic);
158  outputs.emplace_back(ConcreteDataTypeMatcher{clbUtils::gDataOriginCDBWrapper, "MFT_NoiseMap"}, Lifetime::Sporadic);
159 
160  return DataProcessorSpec{
161  "mft-noise-calibrator",
162  inputs,
163  outputs,
164  AlgorithmSpec{adaptFromTask<NoiseCalibratorSpec>(useDigits)},
165  Options{
166  {"prob-threshold", VariantType::Float, 1.e-6f, {"Probability threshold for noisy pixels"}},
167  {"tstart", VariantType::Int64, -1ll, {"Start of validity timestamp"}},
168  {"tend", VariantType::Int64, -1ll, {"End of validity timestamp"}},
169  {"path", VariantType::String, "/MFT/Calib/NoiseMap", {"Path to write to in CCDB"}},
170  {"meta", VariantType::String, "", {"meta data to write in CCDB"}},
171  {"hb-per-tf", VariantType::Int, 256, {"Number of HBF per TF"}}}};
172 }
