18 void BCData::print(uint32_t triggerMask, int diff) const
19 {
20  if (diff == 0) {

22  } else {
23  printf("%9u.%04u ", ir.orbit, ir.bc);
24  }
25  printf("Read:");

27  if (ic % NChPerModule == 0) {
28  if (ic == 0) {

30  } else {
31  printf("] %d[", ic / NChPerModule);
32  }
33  }

35  printf("R");
36  } else {
37  printf(" ");
38  }
39  }
40  printf("]\n");
41  if (diff) {
42  printf("%9u.%04u ", ir.orbit, ir.bc);
43  }
44  printf("Hits:");

46  if (ic % NChPerModule == 0) {

56  if (is_hit) {
57  printf("T");
58  } else {
59  printf(".");
60  }
61  } else {
62  if (is_hit) {

65  printf(" ");
66  }
67  }
68  }
69  printf("]\n");

71  printf("%9u.%04u ", ir.orbit, ir.bc);
72  }

75  std::bitset<10> bb(moduleTriggers[i]);
76  printf(" %d %s%s%s%s%s", i, bb[8] ? "3" : "-", bb[7] ? "2" : "-", bb[6] ? "1" : "-", bb[5] ? "0" : "-", bb[4] ? "M" : "-");
77  }
