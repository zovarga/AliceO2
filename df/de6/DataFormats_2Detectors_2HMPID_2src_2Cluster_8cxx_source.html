96 // Calculates naive cluster position as a center of gravity of its digits.
97 void Cluster::coG()
98 {
99  int minPadX = 999;
100  int minPadY = 999;
101  int maxPadX = -1;
102  int maxPadY = -1; // for box finding
103 
104  if (mDigs.size() == 0) {
105  return;
106  } // no digits in this cluster
107  mXX = mYY = mQRaw = 0; // init summable parameters
108  mCh = -1; // init chamber
109  int maxQpad = -1;
110  int maxQ = -1; // to calculate the pad with the highest charge
111 
112  o2::hmpid::Digit* pDig = nullptr;
113  for (int iDig = 0; iDig < mDigs.size(); iDig++) { // digits loop
114  int x, y, mod;
115  int padId = mDigs[iDig]->getPadID();
116  o2::hmpid::Digit::pad2Absolute(padId, &mod, &x, &y);
117  if (x > maxPadX) {
118  maxPadX = x;
119  } // find the minimum box that contain the cluster MaxX
120  if (y > maxPadY) {
121  maxPadY = y;
122  } // MaxY
123  if (x < minPadX) {
124  minPadX = x;
125  } // MinX
126  if (y < minPadY) {
127  minPadY = y;
128  } // MinY
129 
130  float q = mDigs[iDig]->mQ; // get QDC
131  mXX += o2::hmpid::Digit::lorsX(padId) * q;
132  mYY += o2::hmpid::Digit::lorsY(padId) * q; // add digit center weighted by QDC
133  mQRaw += q; // increment total charge
134  if (q > maxQ) {
135  maxQpad = padId;
136  maxQ = (int)q;
137  } // to find pad with highest charge
138  mCh = mod; // initialize chamber number
139  } // digits loop
140 
141  mBox = (maxPadX - minPadX + 1) * 100 + maxPadY - minPadY + 1; // dimension of the box: format Xdim*100+Ydim
142  if (mQRaw != 0) {
143  mXX /= mQRaw;
144  mYY /= mQRaw;
145  } // final center of gravity
146 
147  if (mDigs.size() > 1 && fgDoCorrSin) {
148  corrSin();
149  } // correct it by sinoid
150 
151  mQ = mQRaw; // Before starting fit procedure, Q and QRaw must be equal
152  mMaxQpad = maxQpad;
153  mMaxQ = maxQ; // store max charge pad to the field
154  mChi2 = 0; // no Chi2 to find
155  mNlocMax = 0; // proper status from this method
156  mSt = o2::hmpid::Cluster::kCoG;
157  return;
158  // setClusterParams(mXX, mYY, mCh); //need to fill the AliCluster3D part
159 } // CoG()

162 // Correction of cluster x position due to sinoid, see HMPID TDR page 30
163 void Cluster::corrSin()
164 {
165  int pc;
166  int px;
167  int py;
168  o2::hmpid::Param::lors2Pad(mXX, mYY, pc, px, py); // tmp digit to get it center
169  double x = mXX - mParam->lorsX(pc, px); // diff between cluster x and center of the pad contaning this cluster
170  mXX += 3.31267e-2 * sin(2 * M_PI / 0.8 * x) - 2.66575e-3 * sin(4 * M_PI / 0.8 * x) + 2.80553e-3 * sin(6 * M_PI / 0.8 * x) + 0.0070;
171  return;
172 }

182 void Cluster::fitFunc(int& iNpars, double* deriv, double& chi2, double* par, int iflag)
183 {
184  // Cluster fit function
185  // par[0]=x par[1]=y par[2]=q for the first Mathieson shape
186  // par[3]=x par[4]=y par[5]=q for the second Mathieson shape and so on up to iNpars/3 Mathieson shapes
187  // For each pad of the cluster calculates the difference between actual pad charge and the charge induced to this pad by all Mathieson distributions
188  // Then the chi2 is calculated as the sum of this value squared for all pad in the cluster.
189  // Arguments: iNpars - number of parameters which is number of local maxima of cluster * 3
190  // chi2 - function result to be minimised
191  // par - parameters array of size iNpars
192  // Returns: none
193  Cluster* pClu = (Cluster*)TVirtualFitter::GetFitter()->GetObjectFit();
194  int nPads = pClu->mSi;
195  chi2 = 0;
196  int iNshape = iNpars / 3;
197  for (int i = 0; i < nPads; i++) { // loop on all pads of the cluster
198  double dQpadMath = 0;
199  for (int j = 0; j < iNshape; j++) { // Mathiesons loop as all of them may contribute to this pad
200  double fracMathi = o2::hmpid::Digit::intMathieson(par[3 * j], par[3 * j + 1], pClu->dig(i)->getPadID());
201  dQpadMath += par[3 * j + 2] * fracMathi; // par[3*j+2] is charge par[3*j] is x par[3*j+1] is y of current Mathieson
202  }
203  if (dQpadMath > 0 && pClu->dig(i)->mQ > 0) {
204  chi2 += std::pow((pClu->dig(i)->mQ - dQpadMath), 2.0) / pClu->dig(i)->mQ; // chi2 function to be minimized
205  }
206  }
207  //---calculate gradients...
208  if (iflag == 2) {
209  float** derivPart;
210  derivPart = new float*[iNpars];
211  for (int j = 0; j < iNpars; j++) {
212  deriv[j] = 0;
213  derivPart[j] = new float[nPads];
214  for (int i = 0; i < nPads; i++) {
215  derivPart[j][i] = 0;
216  }
217  }
218  for (int i = 0; i < nPads; i++) { // loop on all pads of the cluster
219  for (int j = 0; j < iNshape; j++) { // Mathiesons loop as all of them may contribute to this pad
220  double fracMathi = o2::hmpid::Digit::intMathieson(par[3 * j], par[3 * j + 1], pClu->dig(i)->getPadID());
221  double lx = o2::hmpid::Digit::lorsX(pClu->dig(i)->getPadID());
222  double ly = o2::hmpid::Digit::lorsY(pClu->dig(i)->getPadID());
223  derivPart[3 * j][i] += par[3 * j + 2] * (o2::hmpid::Digit::mathiesonX(par[3 * j] - lx - 0.5 * o2::hmpid::Param::sizePadX()) - o2::hmpid::Digit::mathiesonX(par[3 * j] - lx + 0.5 * o2::hmpid::Param::sizePadX())) *
224  o2::hmpid::Digit::intPartMathiY(par[3 * j + 1], pClu->dig(i)->getPadID());
225  derivPart[3 * j + 1][i] += par[3 * j + 2] * (o2::hmpid::Digit::mathiesonY(par[3 * j + 1] - ly - 0.5 * o2::hmpid::Param::sizePadY()) - o2::hmpid::Digit::mathiesonY(par[3 * j + 1] - ly + 0.5 * o2::hmpid::Param::sizePadY())) *
226  o2::hmpid::Digit::intPartMathiX(par[3 * j], pClu->dig(i)->getPadID());
227  derivPart[3 * j + 2][i] += fracMathi;
228  }
229  }
230  // loop on all pads of the cluster
231  for (int i = 0; i < nPads; i++) { // loop on all pads of the cluster
232  double dQpadMath = 0; // pad charge collector
233  for (int j = 0; j < iNshape; j++) { // Mathiesons loop as all of them may contribute to this pad
234  float fracMathi = o2::hmpid::Digit::intMathieson(par[3 * j], par[3 * j + 1], pClu->dig(i)->getPadID());
235  dQpadMath += par[3 * j + 2] * fracMathi;
236  if (dQpadMath > 0 && pClu->dig(i)->mQ > 0) {
237  deriv[3 * j] += 2 / pClu->dig(i)->mQ * (pClu->dig(i)->mQ - dQpadMath) * derivPart[3 * j][i];
238  deriv[3 * j + 1] += 2 / pClu->dig(i)->mQ * (pClu->dig(i)->mQ - dQpadMath) * derivPart[3 * j + 1][i];
239  deriv[3 * j + 2] += 2 / pClu->dig(i)->mQ * (pClu->dig(i)->mQ - dQpadMath) * derivPart[3 * j + 2][i];
240  }
241  }
242  }
243  // delete array...
244  for (int i = 0; i < iNpars; i++) {
245  delete[] derivPart[i];
246  }
247  delete[] derivPart;
248  } //---gradient calculations ended
249  // fit ended. Final calculations
250 } // FitFunction()

314 int Cluster::solve(std::vector<o2::hmpid::Cluster>* pCluLst, float* pSigmaCut, bool isTryUnfold)
315 {
316  // This methode is invoked when the cluster is formed to solve it. Solve the cluster means to try to unfold the cluster
317  // into the local maxima number of clusters. This methode is invoked by AliHMPIDRconstructor::Dig2Clu() on cluster by cluster basis.
318  // At this point, cluster contains a list of digits, cluster charge and size is precalculated in AddDigit(), position is preset to (-1,-1) in ctor,
319  // status is preset to kFormed in AddDigit(), chamber-sector info is preseted to actual values in AddDigit()
320  // Method first finds number of local maxima and if it's more then one tries to unfold this cluster into local maxima number of clusters
321  // Arguments: pCluLst - cluster list pointer where to add new cluster(s)
322  // isTryUnfold - flag to switch on/off unfolding
323  // Returns: number of local maxima of original cluster
324  const int kMaxLocMax = 6; // max allowed number of loc max for fitting
325  coG(); // First calculate CoG for the given cluster
326  int iCluCnt = pCluLst->size(); // get current number of clusters already stored in the list by previous operations
327  int rawSize = mSi; // get current raw cluster size
328  if (rawSize > 100) {
329  mSt = kBig;
330  } else if (isTryUnfold == false) {
331  mSt = kNot;
332  } else if (rawSize == 1) {
333  mSt = kSi1;
334  }
335  if (rawSize > 100 || isTryUnfold == false || rawSize == 1) { // No deconv if: 1 - big cluster (also avoid no zero suppression!)
336  // setClusterParams(mXX, mYY, mCh); // 2 - flag is set to FALSE
337  // new ((*pCluLst)[iCluCnt++]) Cluster(*this); // 3 - size = 1
338  pCluLst->push_back(o2::hmpid::Cluster(*this));
339  return 1; // add this raw cluster
340  }
341 
342  // Phase 0. Initialise Fitter
343  double arglist[10];
344  float ierflg = 0.;
345  TVirtualFitter* fitter = TVirtualFitter::Fitter((TObject*)this, 3 * 6); // initialize Fitter
346  arglist[0] = -1;
347  ierflg = fitter->ExecuteCommand("SET PRI", arglist, 1); // no printout
348  ierflg = fitter->ExecuteCommand("SET NOW", arglist, 0); // no warning messages
349  arglist[0] = 1;
350  ierflg = fitter->ExecuteCommand("SET GRA", arglist, 1); // force Fitter to use my gradient
351  fitter->SetFCN(Cluster::fitFunc);
352  // Phase 1. Find number of local maxima. Strategy is to check if the current pad has QDC more then all neigbours. Also find the box contaning the cluster
353  mNlocMax = 0;
354  for (int iDig1 = 0; iDig1 < rawSize; iDig1++) { // first digits loop
355  o2::hmpid::Digit* pDig1 = mDigs.at(iDig1); // take next digit
356  int iCnt = 0; // counts how many neighbouring pads has QDC more then current one
357  for (int iDig2 = 0; iDig2 < rawSize; iDig2++) { // loop on all digits again
358  if (iDig1 == iDig2) {
359  continue;
360  } // the same digit, no need to compare
361  o2::hmpid::Digit* pDig2 = mDigs.at(iDig2); // take second digit to compare with the first one
362  int dist = TMath::Sign(int(pDig1->mX - pDig2->mX), 1) + TMath::Sign(int(pDig1->mY - pDig2->mY), 1); // distance between pads
363  if (dist == 1) { // means dig2 is a neighbour of dig1
364  if (pDig2->mQ >= pDig1->mQ) {
365  iCnt++; // count number of pads with Q more then Q of current pad
366  }
367  }
368  } // second digits loop
369  if (iCnt == 0 && mNlocMax < kMaxLocMax) { // this pad has Q more then any neighbour so it's local maximum
370  float xStart = o2::hmpid::Digit::lorsX(pDig1->getPadID());
371  float yStart = o2::hmpid::Digit::lorsY(pDig1->getPadID());
372  float xMin = xStart - mParam->sizePadX();
373  float xMax = xStart + mParam->sizePadX();
374  float yMin = yStart - mParam->sizePadY();
375  float yMax = yStart + mParam->sizePadY();
376  ierflg = fitter->SetParameter(3 * mNlocMax, Form("x%i", mNlocMax), xStart, 0.1, xMin, xMax); // X,Y,Q initial values of the loc max pad
377  ierflg = fitter->SetParameter(3 * mNlocMax + 1, Form("y%i", mNlocMax), yStart, 0.1, yMin, yMax); // X, Y constrained to be near the loc max
378  ierflg = fitter->SetParameter(3 * mNlocMax + 2, Form("q%i", mNlocMax), pDig1->mQ, 0.1, 0, 10000); // Q constrained to be positive
379  mNlocMax++;
380  } // if this pad is local maximum
381  } // first digits loop
382 
383  // Phase 2. Fit loc max number of Mathiesons or add this current cluster to the list
384  // case 1 -> no loc max found
385  if (mNlocMax == 0) { // case of no local maxima found: pads with same charge...
386  mNlocMax = 1;
387  mSt = kNoLoc;
388  // setClusterParams(mXX, mYY, mCh); //need to fill the AliCluster3D part
389  pCluLst->push_back(o2::hmpid::Cluster(*this)); // add new unfolded cluster pCluLst->push_back(o2::hmpid::Cluster(*this));
390  return mNlocMax;
391  }
392 
393  // case 2 -> loc max found. Check # of loc maxima
394  if (mNlocMax >= kMaxLocMax) {
395  // setClusterParams(mXX, mYY, mCh); // if # of local maxima exceeds kMaxLocMax...
396  mSt = kMax;
397  pCluLst->push_back(o2::hmpid::Cluster(*this)); //...add this raw cluster
398  } else { // or resonable number of local maxima to fit and user requested it
399  // Now ready for minimization step
400  arglist[0] = 500; // number of steps and sigma on pads charges
401  arglist[1] = 1.; //
402  ierflg = fitter->ExecuteCommand("SIMPLEX", arglist, 2); // start fitting with Simplex
403  if (!ierflg) {
404  fitter->ExecuteCommand("MIGRAD", arglist, 2); // fitting improved by Migrad
405  }
406  if (ierflg) {
407  double strategy = 2.;
408  ierflg = fitter->ExecuteCommand("SET STR", &strategy, 1); // change level of strategy
409  if (!ierflg) {
410  ierflg = fitter->ExecuteCommand("SIMPLEX", arglist, 2); // start fitting with Simplex
411  if (!ierflg) {
412  fitter->ExecuteCommand("MIGRAD", arglist, 2); // fitting improved by Migrad
413  }
414  }
415  }
416  if (ierflg) {
417  mSt = kAbn; // no convergence of the fit...
418  }
419  double dummy;
420  char sName[80]; // vars to get results from Minuit
421  double edm;
422  double errdef;
423  int nvpar;
424  int nparx;
425  for (int i = 0; i < mNlocMax; i++) { // store the local maxima parameters
426  fitter->GetParameter(3 * i, sName, mXX, mErrX, dummy, dummy); // X
427  fitter->GetParameter(3 * i + 1, sName, mYY, mErrY, dummy, dummy); // Y
428  fitter->GetParameter(3 * i + 2, sName, mQ, mErrQ, dummy, dummy); // Q
429  fitter->GetStats(mChi2, edm, errdef, nvpar, nparx); // get fit infos
430  if (mNlocMax > 1) {
431  findClusterSize(i, pSigmaCut); // find clustersize for deconvoluted clusters
432  // after this call, fSi temporarly is the calculated size. Later is set again
433  // to its original value
434  }
435  if (mSt != kAbn) {
436  if (mNlocMax != 1) {
437  mSt = kUnf; // if unfolded
438  }
439  if (mNlocMax == 1 && mSt != kNoLoc) {
440  mSt = kLo1; // if only 1 loc max
441  }
442  if (!isInPc()) {
443  mSt = kEdg; // if Out of Pc
444  }
445  if (mSt == kNoLoc) {
446  mNlocMax = 0; // if with no loc max (pads with same charge..)
447  }
448  }
449  // setClusterParams(mXX, mYY, mCh); //need to fill the AliCluster3D part
450  pCluLst->push_back(o2::hmpid::Cluster(*this)); // add new unfolded cluster
451  if (mNlocMax > 1) {
452  setSize(rawSize); // Original raw size is set again to its proper value
453  }
454  }
455  }
456  return mNlocMax;
457 } // Solve()

466  int iCh = pDig->mCh;
467  double qPad = mQ * o2::hmpid::Digit::intMathieson(x(), y(), pDig->getPadID()); // pad charge pDig->
468  // AliDebug(1,Form("Chamber %i X %i Y %i SigmaCut %i pad %i qpadMath %8.2f qPadRaw %8.2f Qtotal %8.2f cluster n.%i",
469  // iCh, o2::hmpid::Digit::a2X(pDig->getPadID()), o2::hmpid::Digit::a2Y(pDig->getPadID()),
470  // pSigmaCut[iCh],iDig,qPad,pDig->mQ,mQRaw,i));
471  if (qPad > pSigmaCut[iCh]) {
472  size++;
473  }
474  }
475  // AliDebug(1,Form(" Calculated size %i",size));

477  setSize(size); // in case of size == 0, original raw clustersize used
478  }

481 Bool_t Cluster::isInPc()
482 {
483  // Check if (X,Y) position is inside the PC limits
484  // Arguments:
485  // Returns: True or False
