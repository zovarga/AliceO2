38 template <typename RawReader>
39 class RawDataReaderSpec : public Task
40 {
41  public:
42  RawDataReaderSpec(const RawReader& rawReader) : mRawReader(rawReader) {}
43  RawDataReaderSpec() = default;
44  ~RawDataReaderSpec() override = default;
45  void init(InitContext& ic) final { o2::fdd::SingleLUT::Instance().printFullMap(); }
46  void run(ProcessingContext& pc) final
47  {
48  DPLRawParser parser(pc.inputs());
49  mRawReader.clear();
50  LOG(info) << "FDD RawDataReaderSpec";
51  uint64_t count = 0;
52  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
53  //Proccessing each page
54  count++;
55  auto rdhPtr = it.get_if<o2::header::RAWDataHeader>();
56  gsl::span<const uint8_t> payload(it.data(), it.size());
57  mRawReader.process(payload, rdhPtr->linkID, int(0));
58  }
59  LOG(info) << "Pages: " << count;
60  mRawReader.accumulateDigits();
61  mRawReader.makeSnapshot(pc);
62  }
63  RawReader mRawReader;
64 };
