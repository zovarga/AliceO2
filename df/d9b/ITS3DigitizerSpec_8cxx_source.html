94  virtual void setDigitizationOptions()
95  {
96  auto& dopt = o2::itsmft::DPLDigitizerParam<o2::detectors::DetID::ITS>::Instance();
97  auto& aopt = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
98  auto& digipar = mDigitizer.getParams();
99  digipar.setContinuous(dopt.continuous);
100  if (dopt.continuous) {
101  auto frameNS = aopt.roFrameLengthInBC * o2::constants::lhc::LHCBunchSpacingNS;
102  digipar.setROFrameLengthInBC(aopt.roFrameLengthInBC);
103  digipar.setROFrameLength(frameNS); // RO frame in ns
104  digipar.setStrobeDelay(aopt.strobeDelay); // Strobe delay wrt beginning of the RO frame, in ns
105  digipar.setStrobeLength(aopt.strobeLengthCont > 0 ? aopt.strobeLengthCont : frameNS - aopt.strobeDelay); // Strobe length in ns
106  } else {
107  digipar.setROFrameLength(aopt.roFrameLengthTrig); // RO frame in ns
108  digipar.setStrobeDelay(aopt.strobeDelay); // Strobe delay wrt beginning of the RO frame, in ns
109  digipar.setStrobeLength(aopt.strobeLengthTrig); // Strobe length in ns
110  }
111  // parameters of signal time response: flat-top duration, max rise time and q @ which rise time is 0
112  digipar.getSignalShape().setParameters(dopt.strobeFlatTop, dopt.strobeMaxRiseTime, dopt.strobeQRiseTime0);
113  digipar.setChargeThreshold(dopt.chargeThreshold); // charge threshold in electrons
114  digipar.setNoisePerPixel(dopt.noisePerPixel); // noise level
115  digipar.setTimeOffset(dopt.timeOffset);
116  digipar.setNSimSteps(dopt.nSimSteps);
117  }

119  void run(framework::ProcessingContext& pc)
120  {
121  if (mFinished) {
122  return;
123  }
124  std::string detStr = mID.getName();
125  // read collision context from input
126  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
127  context->initSimChains(mID, mSimChains);
128  const bool withQED = context->isQEDProvided() && !mDisableQED;
129  auto& timesview = context->getEventRecords(withQED);
130  LOG(info) << "GOT " << timesview.size() << " COLLISSION TIMES";
131  LOG(info) << "SIMCHAINS " << mSimChains.size();
132 
133  // if there is nothing to do ... return
134  if (timesview.size() == 0) {
135  return;
136  }
137  TStopwatch timer;
138  timer.Start();
139  LOG(info) << " CALLING ITS3 DIGITIZATION ";
140 
141  mDigitizer.setDigits(&mDigits);
142  mDigitizer.setROFRecords(&mROFRecords);
143  mDigitizer.setMCLabels(&mLabels);
144 
145  // digits are directly put into DPL owned resource
146  auto& digitsAccum = pc.outputs().make<std::vector<itsmft::Digit>>(Output{mOrigin, "DIGITS", 0, Lifetime::Timeframe});
147 
148  auto accumulate = [this, &digitsAccum]() {
149  // accumulate result of single event processing, called after processing every event supplied
150  // AND after the final flushing via digitizer::fillOutputContainer
151  if (!mDigits.size()) {
152  return; // no digits were flushed, nothing to accumulate
153  }
154  static int fixMC2ROF = 0; // 1st entry in mc2rofRecordsAccum to be fixed for ROFRecordID
155  auto ndigAcc = digitsAccum.size();
156  std::copy(mDigits.begin(), mDigits.end(), std::back_inserter(digitsAccum));
157 
158  // fix ROFrecords references on ROF entries
159  auto nROFRecsOld = mROFRecordsAccum.size();
160 
161  for (int i = 0; i < mROFRecords.size(); i++) {
162  auto& rof = mROFRecords[i];
163  rof.setFirstEntry(ndigAcc + rof.getFirstEntry());
164  rof.print();
165 
166  if (mFixMC2ROF < mMC2ROFRecordsAccum.size()) { // fix ROFRecord entry in MC2ROF records
167  for (int m2rid = mFixMC2ROF; m2rid < mMC2ROFRecordsAccum.size(); m2rid++) {
168  // need to register the ROFRecors entry for MC event starting from this entry
169  auto& mc2rof = mMC2ROFRecordsAccum[m2rid];
170  if (rof.getROFrame() == mc2rof.minROF) {
171  mFixMC2ROF++;
172  mc2rof.rofRecordID = nROFRecsOld + i;
173  mc2rof.print();
174  }
175  }
176  }
177  }
178 
179  std::copy(mROFRecords.begin(), mROFRecords.end(), std::back_inserter(mROFRecordsAccum));
180  if (mWithMCTruth) {
181  mLabelsAccum.mergeAtBack(mLabels);
182  }
183  LOG(info) << "Added " << mDigits.size() << " digits ";
184  // clean containers from already accumulated stuff
185  mLabels.clear();
186  mDigits.clear();
187  mROFRecords.clear();
188  }; // and accumulate lambda
189 
190  auto& eventParts = context->getEventParts(withQED);
191  // loop over all composite collisions given from context (aka loop over all the interaction records)
192  for (int collID = 0; collID < timesview.size(); ++collID) {
193  const auto& irt = timesview[collID];
194 
195  mDigitizer.setEventTime(irt);
196  mDigitizer.resetEventROFrames(); // to estimate min/max ROF for this collID
197  // for each collision, loop over the constituents event and source IDs
198  // (background signal merging is basically taking place here)
199  for (auto& part : eventParts[collID]) {
200 
201  // get the hits for this event and this source
202  mHits.clear();
203  context->retrieveHits(mSimChains, o2::detectors::SimTraits::DETECTORBRANCHNAMES[mID][0].c_str(), part.sourceID, part.entryID, &mHits);
204 
205  if (mHits.size() > 0) {
206  LOG(debug) << "For collision " << collID << " eventID " << part.entryID
207  << " found " << mHits.size() << " hits ";
208  mDigitizer.process(&mHits, part.entryID, part.sourceID); // call actual digitization procedure
209  }
210  }
211  mMC2ROFRecordsAccum.emplace_back(collID, -1, mDigitizer.getEventROFrameMin(), mDigitizer.getEventROFrameMax());
212  accumulate();
213  }
214  mDigitizer.fillOutputContainer();
215  accumulate();
216 
217  // here we have all digits and labels and we can send them to consumer (aka snapshot it onto output)
218 
219  pc.outputs().snapshot(Output{mOrigin, "DIGITSROF", 0, Lifetime::Timeframe}, mROFRecordsAccum);
220  if (mWithMCTruth) {
221  pc.outputs().snapshot(Output{mOrigin, "DIGITSMC2ROF", 0, Lifetime::Timeframe}, mMC2ROFRecordsAccum);
222  auto& sharedlabels = pc.outputs().make<o2::dataformats::ConstMCTruthContainer<o2::MCCompLabel>>(Output{mOrigin, "DIGITSMCTR", 0, Lifetime::Timeframe});
223  mLabelsAccum.flatten_to(sharedlabels);
224  // free space of existing label containers
225  mLabels.clear_andfreememory();
226  mLabelsAccum.clear_andfreememory();
227  }
228  LOG(info) << mID.getName() << ": Sending ROMode= " << mROMode << " to GRPUpdater";
229  pc.outputs().snapshot(Output{mOrigin, "ROMode", 0, Lifetime::Timeframe}, mROMode);
230 
231  timer.Stop();
232  LOG(info) << "Digitization took " << timer.CpuTime() << "s";
233 
234  // we should be only called once; tell DPL that this process is ready to exit
235  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
236 
237  mFinished = true;
238  }

265 {
266  std::string detStr = o2::detectors::DetID::getName(ITS3DPLDigitizerTask::DETID);
267  auto detOrig = ITS3DPLDigitizerTask::DETOR;
268  std::stringstream parHelper;

271  << "\n or " << o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::getParamName().data() << ".<param>=value;... with"

275  Inputs{InputSpec{"collisioncontext", "SIM", "COLLISIONCONTEXT",
276  static_cast<SubSpecificationType>(channel), Lifetime::Timeframe}},
277  makeOutChannels(detOrig, mctruth),
278  AlgorithmSpec{adaptFromTask<ITS3DPLDigitizerTask>(mctruth)},
279  Options{
280  {"disable-qed", o2::framework::VariantType::Bool, false, {"disable QED handling"}}
