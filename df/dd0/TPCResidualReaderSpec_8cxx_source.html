68 void TPCResidualReader::run(ProcessingContext& pc)
69 {
70  mTrackResiduals.createOutputFile(); // FIXME remove when map output is handled properly
71  for (int iSec = 0; iSec < SECTORSPERSIDE * SIDES; ++iSec) {
72  auto brStats = mTreeStats->GetBranch(Form("sec%d", iSec));
73  brStats->SetAddress(mTrackResiduals.getVoxStatPtr());
74  brStats->GetEntry(0); // there is only one entry for the statistics tree
75  auto brResid = mTreeResiduals->GetBranch(Form("sec%d", iSec));
76  brResid->SetAddress(&mResidualsPtr);
77  for (int iEntry = 0; iEntry < brResid->GetEntries(); ++iEntry) {
78  brResid->GetEntry(iEntry);
79  LOGF(debug, "Pushing %lu TPC residuals at entry %i for sector %i", mResiduals.size(), iEntry, iSec);
80  for (const auto& res : mResiduals) {
81  LOGF(debug, "Adding residual from Voxel %i-%i-%i. dy(%i), dz(%i), tg(%i)", res.bvox[0], res.bvox[1], res.bvox[2], res.dy, res.dz, res.tgSlp);
82  }
83  mTrackResiduals.getLocalResVec().insert(mTrackResiduals.getLocalResVec().end(), mResiduals.begin(), mResiduals.end());
84  }
85  // do processing
86  mTrackResiduals.processSectorResiduals(iSec);
87  // do cleanup
88  mTrackResiduals.getLocalResVec().clear();
89  }
90 
91  mTrackResiduals.closeOutputFile(); // FIXME remove when map output is handled properly
92 
93  // const auto& voxResArray = mTrackResiduals.getVoxelResults(); // array with one vector of results per sector
94  // pc.outputs().snapshot(Output{"GLO", "VOXELRESULTS", 0, Lifetime::Timeframe}, voxResArray); // send results as one large vector?
95 
96  pc.services().get<ControlService>().endOfStream();
97  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
98 }

100 void TPCResidualReader::connectTree(const std::string& filename)
101 {
102  mTreeResiduals.reset(nullptr); // in case it was already loaded
103  mTreeStats.reset(nullptr); // in case it was already loaded
104  mFile.reset(TFile::Open(filename.c_str()));
105  assert(mFile && !mFile->IsZombie());
106  mTreeResiduals.reset((TTree*)mFile->Get("resid"));
107  assert(mTreeResiduals);
108  mTreeStats.reset((TTree*)mFile->Get("stats"));
109  assert(mTreeStats);
110  assert(mTreeStats->GetEntries() == 1); // expect exactly one entry for statistics
111 
112  LOG(info) << "Loaded tree from " << filename << " with " << mTreeResiduals->GetEntries() << " entries";
113 }
