139 uint8_t GBTLink::checkErrorsRDH(const RDH& rdh)
140 {
141  uint8_t err = uint8_t(NoError);
142  if (!RDHUtils::checkRDH(rdh, true)) {
143  statistics.errorCounts[GBTLinkDecodingStat::ErrNoRDHAtStart]++;
144  if (needToPrintError(statistics.errorCounts[GBTLinkDecodingStat::ErrNoRDHAtStart])) {
145  err |= uint8_t(ErrorPrinted);
146  LOG(important) << describe() << ' ' << irHBF << ' ' << statistics.ErrNames[GBTLinkDecodingStat::ErrNoRDHAtStart];
147  }
148  errorBits |= 0x1 << int(GBTLinkDecodingStat::ErrNoRDHAtStart);
149  err |= uint8_t(Abort);
150  return err; // fatal error
151  }
152  if (format == OldFormat && RDHUtils::getVersion(rdh) > 4) {
153  if (verbosity >= VerboseErrors) {
154  LOG(important) << "Requested old format requires data with RDH version 3 or 4, RDH version "
155  << RDHUtils::getVersion(rdh) << " is found";
156  err |= uint8_t(ErrorPrinted);
157  }
158  err |= uint8_t(Abort);
159  return err;
160  }
161  if ((RDHUtils::getPacketCounter(rdh) > packetCounter + 1) && packetCounter >= 0) {
162  statistics.errorCounts[GBTLinkDecodingStat::ErrPacketCounterJump]++;
163  if (needToPrintError(statistics.errorCounts[GBTLinkDecodingStat::ErrPacketCounterJump])) {
164  LOG(important) << describe() << ' ' << irHBF << ' ' << statistics.ErrNames[GBTLinkDecodingStat::ErrPacketCounterJump]
165  << " : jump from " << int(packetCounter) << " to " << int(RDHUtils::getPacketCounter(rdh));
166  err |= uint8_t(ErrorPrinted);
167  }
168  errorBits |= 0x1 << int(GBTLinkDecodingStat::ErrPacketCounterJump);
169  err |= uint8_t(Warning);
170  }
171  packetCounter = RDHUtils::getPacketCounter(rdh);
172  return err;
173 }

259 uint8_t GBTLink::checkErrorsHeaderWord(const GBTDataHeaderL* gbtH)
260 {
261  uint8_t err = uint8_t(NoError);
262  if (!gbtH->isDataHeader()) { // check header word
263  statistics.errorCounts[GBTLinkDecodingStat::ErrMissingGBTHeader]++;
264  if (verbosity >= VerboseErrors) {
265  gbtH->printX();
266  LOG(important) << describe() << ' ' << irHBF << ' ' << statistics.ErrNames[GBTLinkDecodingStat::ErrMissingGBTHeader];
267  err |= uint8_t(ErrorPrinted);
268  }
269  errorBits |= 0x1 << int(GBTLinkDecodingStat::ErrMissingGBTHeader);
270  err |= uint8_t(Abort);
271  return err;
272  }
273  int cnt = RDHUtils::getPageCounter(*lastRDH);
274  // RSTODO: this makes sense only for old format, where every trigger has its RDH
275  if (gbtH->packetIdx != cnt) {
276  statistics.errorCounts[GBTLinkDecodingStat::ErrRDHvsGBTHPageCnt]++;
277  if (needToPrintError(statistics.errorCounts[GBTLinkDecodingStat::ErrRDHvsGBTHPageCnt])) {
278  LOG(important) << describe() << ' ' << irHBF << ' ' << statistics.ErrNames[GBTLinkDecodingStat::ErrRDHvsGBTHPageCnt] << ": diff in GBT header "
279  << gbtH->packetIdx << " and RDH page " << cnt << " counters";
280  err |= uint8_t(ErrorPrinted);
281  }
282  errorBits |= 0x1 << int(GBTLinkDecodingStat::ErrRDHvsGBTHPageCnt);
283  err |= uint8_t(Warning);
284  return err;
285  }
286  // RSTODO CHECK
287  if (lanesActive == lanesStop) { // all lanes received their stop, new page 0 expected
288  //if (cnt) { // makes sens for old format only
289  if (gbtH->packetIdx) {
290  statistics.errorCounts[GBTLinkDecodingStat::ErrNonZeroPageAfterStop]++;
291  if (needToPrintError(statistics.errorCounts[GBTLinkDecodingStat::ErrNonZeroPageAfterStop])) {
292  LOG(important) << describe() << ' ' << irHBF << ' ' << statistics.ErrNames[GBTLinkDecodingStat::ErrNonZeroPageAfterStop]
293  << ": Non-0 page counter (" << cnt << ") while all lanes were stopped";
294  err |= uint8_t(ErrorPrinted);
295  }
296  errorBits |= 0x1 << int(GBTLinkDecodingStat::ErrNonZeroPageAfterStop);
297  err |= uint8_t(Warning);
298  }
299  }
300  return err;
301 }

404 uint8_t GBTLink::checkErrorsLanesStops()
405 {
406  // make sure all lane stops for finished page are received
407  uint8_t err = uint8_t(NoError);
408  if ((lanesActive & ~lanesStop)) {
409  if (RDHUtils::getTriggerType(*lastRDH) != o2::trigger::SOT) { // only SOT trigger allows unstopped lanes?
410  statistics.errorCounts[GBTLinkDecodingStat::ErrUnstoppedLanes]++;
411  if (needToPrintError(statistics.errorCounts[GBTLinkDecodingStat::ErrUnstoppedLanes])) {
412  std::bitset<32> active(lanesActive), stopped(lanesStop);
413  LOG(important) << describe() << ' ' << irHBF << ' ' << statistics.ErrNames[GBTLinkDecodingStat::ErrUnstoppedLanes]
414  << " | active: " << active << " stopped: " << stopped;
415  err |= uint8_t(ErrorPrinted);
416  }
417  errorBits |= 0x1 << int(GBTLinkDecodingStat::ErrUnstoppedLanes);
418  }
419  err |= uint8_t(Warning);
420  }
421  // make sure all active lanes (except those in time-out) have sent some data
422  if ((~lanesWithData & lanesActive) != lanesTimeOut) {
423  statistics.errorCounts[GBTLinkDecodingStat::ErrNoDataForActiveLane]++;
424  if (needToPrintError(statistics.errorCounts[GBTLinkDecodingStat::ErrNoDataForActiveLane])) {
425  std::bitset<32> withData(lanesWithData), active(lanesActive), timeOut(lanesTimeOut);
426  LOG(important) << describe() << ' ' << irHBF << ' ' << statistics.ErrNames[GBTLinkDecodingStat::ErrNoDataForActiveLane]
427  << " | with data: " << withData << " active: " << active << " timeOut: " << timeOut;
428  err |= uint8_t(ErrorPrinted);

430  errorBits |= 0x1 << int(GBTLinkDecodingStat::ErrNoDataForActiveLane);
