97 {
98  // fill vector with interaction records
99  dest.clear();
100  for (int i = dest.capacity(); i--;) {

106 inline void InteractionSampler::nextCollidingBC(int n)
107 {
109  if ((mCurrBCIdx += n) >= (int)mInteractingBCs.size()) {
110  mIR.orbit += mCurrBCIdx / mInteractingBCs.size();
111  mCurrBCIdx %= mInteractingBCs.size();
112  }
113  mIR.bc = mInteractingBCs[mCurrBCIdx];
114 }
