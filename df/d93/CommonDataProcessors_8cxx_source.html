86 DataProcessorSpec CommonDataProcessors::getOutputObjHistSink(std::vector<OutputObjectInfo> const& objmap, std::vector<OutputTaskInfo> const& tskmap)
87 {
88  auto writerFunction = [objmap, tskmap](InitContext& ic) -> std::function<void(ProcessingContext&)> {
89  auto& callbacks = ic.services().get<CallbackService>();
90  auto inputObjects = std::make_shared<std::vector<std::pair<InputObjectRoute, InputObject>>>();
91 
92  auto endofdatacb = [inputObjects](EndOfStreamContext& context) {
93  LOG(debug) << "Writing merged objects and histograms to file";
94  if (inputObjects->empty()) {
95  LOG(error) << "Output object map is empty!";
96  context.services().get<ControlService>().readyToQuit(QuitRequest::Me);
97  return;
98  }
99  std::string currentDirectory = "";
100  std::string currentFile = "";
101  TFile* f[OutputObjHandlingPolicy::numPolicies];
102  for (auto i = 0u; i < OutputObjHandlingPolicy::numPolicies; ++i) {
103  f[i] = nullptr;
104  }
105  for (auto& [route, entry] : *inputObjects) {
106  auto file = ROOTfileNames.find(route.policy);
107  if (file != ROOTfileNames.end()) {
108  auto filename = file->second;
109  if (f[route.policy] == nullptr) {
110  f[route.policy] = TFile::Open(filename.c_str(), "RECREATE");
111  }
112  auto nextDirectory = route.directory;
113  if ((nextDirectory != currentDirectory) || (filename != currentFile)) {
114  if (!f[route.policy]->FindKey(nextDirectory.c_str())) {
115  f[route.policy]->mkdir(nextDirectory.c_str());
116  }
117  currentDirectory = nextDirectory;
118  currentFile = filename;
119  }
120 
121  // translate the list-structure created by the registry into a directory structure within the file
122  std::function<void(TList*, TDirectory*)> writeListToFile;
123  writeListToFile = [&](TList* list, TDirectory* parentDir) {
124  TIter next(list);
125  TNamed* object = nullptr;
126  while ((object = (TNamed*)next())) {
127  if (object->InheritsFrom(TList::Class())) {
128  writeListToFile((TList*)object, parentDir->mkdir(object->GetName(), object->GetName(), true));
129  } else {
130  parentDir->WriteObjectAny(object, object->Class(), object->GetName());
131  list->Remove(object);
132  }
133  }
134  };
135 
136  TDirectory* currentDir = f[route.policy]->GetDirectory(currentDirectory.c_str());
137  if (route.sourceType == OutputObjSourceType::HistogramRegistrySource) {
138  TList* outputList = (TList*)entry.obj;
139  outputList->SetOwner(false);
140 
141  // if registry should live in dedicated folder a TNamed object is appended to the list
142  if (outputList->Last()->IsA() == TNamed::Class()) {
143  delete outputList->Last();
144  outputList->RemoveLast();
145  currentDir = currentDir->mkdir(outputList->GetName(), outputList->GetName(), true);
146  }
147 
148  writeListToFile(outputList, currentDir);
149  outputList->SetOwner();
150  delete outputList;
151  entry.obj = nullptr;
152  } else {
153  currentDir->WriteObjectAny(entry.obj, entry.kind, entry.name.c_str());
154  }
155  }
156  }
157  for (auto i = 0u; i < OutputObjHandlingPolicy::numPolicies; ++i) {
158  if (f[i] != nullptr) {
159  f[i]->Close();
160  }
161  }
162  LOG(debug) << "All outputs merged in their respective target files";
163  context.services().get<ControlService>().readyToQuit(QuitRequest::Me);
164  };
165 
166  callbacks.set(CallbackService::Id::EndOfStream, endofdatacb);
167  return [inputObjects, objmap, tskmap](ProcessingContext& pc) mutable -> void {
168  auto const& ref = pc.inputs().get("x");
169  if (!ref.header) {
170  LOG(error) << "Header not found";
171  return;
172  }
173  if (!ref.payload) {
174  LOG(error) << "Payload not found";
175  return;
176  }
177  auto datah = o2::header::get<o2::header::DataHeader*>(ref.header);
178  if (!datah) {
179  LOG(error) << "No data header in stack";
180  return;
181  }
182 
183  auto objh = o2::header::get<o2::framework::OutputObjHeader*>(ref.header);
184  if (!objh) {
185  LOG(error) << "No output object header in stack";
186  return;
187  }
188 
189  FairTMessage tm(const_cast<char*>(ref.payload), static_cast<int>(datah->payloadSize));
190  InputObject obj;
191  obj.kind = tm.GetClass();
192  if (obj.kind == nullptr) {
193  LOG(error) << "Cannot read class info from buffer.";
194  return;
195  }
196 
197  auto policy = objh->mPolicy;
198  auto sourceType = objh->mSourceType;
199  auto hash = objh->mTaskHash;
200 
201  obj.obj = tm.ReadObjectAny(obj.kind);
202  TNamed* named = static_cast<TNamed*>(obj.obj);
203  obj.name = named->GetName();
204  auto hpos = std::find_if(tskmap.begin(), tskmap.end(), [&](auto&& x) { return x.id == hash; });
205  if (hpos == tskmap.end()) {
206  LOG(error) << "No task found for hash " << hash;
207  return;
208  }
209  auto taskname = hpos->name;
210  auto opos = std::find_if(objmap.begin(), objmap.end(), [&](auto&& x) { return x.id == hash; });
211  if (opos == objmap.end()) {
212  LOG(error) << "No object list found for task " << taskname << " (hash=" << hash << ")";
213  return;
214  }
215  auto objects = opos->bindings;
216  if (std::find(objects.begin(), objects.end(), obj.name) == objects.end()) {
217  LOG(error) << "No object " << obj.name << " in map for task " << taskname;
218  return;
219  }
220  auto nameHash = compile_time_hash(obj.name.c_str());
221  InputObjectRoute key{obj.name, nameHash, taskname, hash, policy, sourceType};
222  auto existing = std::find_if(inputObjects->begin(), inputObjects->end(), [&](auto&& x) { return (x.first.uniqueId == nameHash) && (x.first.taskHash == hash); });
223  if (existing == inputObjects->end()) {
224  inputObjects->push_back(std::make_pair(key, obj));
225  return;
226  }
227  auto merger = existing->second.kind->GetMerge();
228  if (!merger) {
229  LOG(error) << "Already one unmergeable object found for " << obj.name;
230  return;
231  }
232 
233  TList coll;
234  coll.Add(static_cast<TObject*>(obj.obj));
235  merger(existing->second.obj, &coll, nullptr);
236  };
237  };

240  DataProcessorSpec spec{
241  .name = name,
242  .inputs = {InputSpec("x", DataSpecUtils::dataDescriptorMatcherFrom(header::DataOrigin{"ATSK"}))},
243  .algorithm = {writerFunction},
244  };
245 
246  return spec;

254 // add sink for the AODs
255 DataProcessorSpec
256  CommonDataProcessors::getGlobalAODSink(std::shared_ptr<DataOutputDirector> dod,
257  std::vector<InputSpec> const& outputInputs)
258 {
259 
260  auto writerFunction = [dod, outputInputs](InitContext& ic) -> std::function<void(ProcessingContext&)> {
261  LOGP(debug, "======== getGlobalAODSink::Init ==========");
262 
263  // find out if any table needs to be saved
264  bool hasOutputsToWrite = false;
265  for (auto& outobj : outputInputs) {
266  auto ds = dod->getDataOutputDescriptors(outobj);
267  if (ds.size() > 0) {
268  hasOutputsToWrite = true;
269  break;
270  }
271  }
272 
273  // if nothing needs to be saved then return a trivial functor
274  // this happens when nothing needs to be saved but there are dangling outputs
275  if (!hasOutputsToWrite) {
276  return [](ProcessingContext&) mutable -> void {
277  static bool once = false;
278  if (!once) {
279  LOG(info) << "No AODs to be saved.";
280  once = true;
281  }
282  };
283  }
284 
285  // end of data functor is called at the end of the data stream
286  auto endofdatacb = [dod](EndOfStreamContext& context) {
287  dod->closeDataFiles();
288  context.services().get<ControlService>().readyToQuit(QuitRequest::Me);
289  };
290 
291  auto& callbacks = ic.services().get<CallbackService>();
292  callbacks.set(CallbackService::Id::EndOfStream, endofdatacb);
293 
294  // prepare map<uint64_t, uint64_t>(startTime, tfNumber)
295  std::map<uint64_t, uint64_t> tfNumbers;
296 
297  // this functor is called once per time frame
298  return [dod, tfNumbers](ProcessingContext& pc) mutable -> void {
299  LOGP(debug, "======== getGlobalAODSink::processing ==========");
300  LOGP(debug, " processing data set with {} entries", pc.inputs().size());
301 
302  // return immediately if pc.inputs() is empty. This should never happen!
303  if (pc.inputs().size() == 0) {
304  LOGP(info, "No inputs available!");
305  return;
306  }
307 
308  // update tfNumbers
309  uint64_t startTime = 0;
310  uint64_t tfNumber = 0;
311  auto ref = pc.inputs().get("tfn");
312  if (ref.spec && ref.payload) {
313  startTime = DataRefUtils::getHeader<DataProcessingHeader*>(ref)->startTime;
314  tfNumber = pc.inputs().get<uint64_t>("tfn");
315  tfNumbers.insert(std::pair<uint64_t, uint64_t>(startTime, tfNumber));
316  }
317 
318  // loop over the DataRefs which are contained in pc.inputs()
319  for (const auto& ref : pc.inputs()) {
320  if (!ref.spec) {
321  LOGP(debug, "Invalid input will be skipped!");
322  continue;
323  }
324 
325  // skip non-AOD refs
326  if (!DataSpecUtils::partialMatch(*ref.spec, header::DataOrigin("AOD"))) {
327  continue;
328  }
329  startTime = DataRefUtils::getHeader<DataProcessingHeader*>(ref)->startTime;
330 
331  // does this need to be saved?
332  auto dh = DataRefUtils::getHeader<header::DataHeader*>(ref);
333  auto tableName = dh->dataDescription.as<std::string>();
334  auto ds = dod->getDataOutputDescriptors(*dh);
335  if (ds.empty()) {
336  continue;
337  }
338 
339  // get TF number fro startTime
340  auto it = tfNumbers.find(startTime);
341  if (it != tfNumbers.end()) {
342  tfNumber = (it->second / dod->getNumberTimeFramesToMerge()) * dod->getNumberTimeFramesToMerge();
343  } else {
344  LOGP(fatal, "No time frame number found for output with start time {}", startTime);
345  throw std::runtime_error("Processing is stopped!");
346  }
347 
348  // get the TableConsumer and corresponding arrow table
349  auto msg = pc.inputs().get(ref.spec->binding);
350  if (msg.header == nullptr) {
351  LOGP(error, "No header for message {}:{}", ref.spec->binding, *ref.spec);
352  continue;
353  }
354  auto s = pc.inputs().get<TableConsumer>(ref.spec->binding);
355  auto table = s->asArrowTable();
356  if (!table->Validate().ok()) {
357  LOGP(warning, "The table \"{}\" is not valid and will not be saved!", tableName);
358  continue;
359  }
360  if (table->schema()->fields().empty()) {
361  LOGP(debug, "The table \"{}\" is empty but will be saved anyway!", tableName);
362  }
363 
364  // loop over all DataOutputDescriptors
365  // a table can be saved in multiple ways
366  // e.g. different selections of columns to different files
367  for (auto d : ds) {
368  auto fileAndFolder = dod->getFileFolder(d, tfNumber);
369  auto treename = fileAndFolder.folderName + d->treename;
370  TableToTree ta2tr(table,
371  fileAndFolder.file,
372  treename.c_str());
373 
374  if (!d->colnames.empty()) {
375  for (auto& cn : d->colnames) {
376  auto idx = table->schema()->GetFieldIndex(cn);
377  auto col = table->column(idx);
378  auto field = table->schema()->field(idx);
379  if (idx != -1) {
380  ta2tr.addBranch(col, field);
381  }
382  }
383  } else {
384  ta2tr.addAllBranches();
385  }
386  ta2tr.process();
387  }
388  }
389  };
390  }; // end of writerFunction
391 
392  // the command line options relevant for the writer are global
393  // see runDataProcessing.h
394  DataProcessorSpec spec{
395  "internal-dpl-aod-writer",
396  outputInputs,
397  Outputs{},
398  AlgorithmSpec(writerFunction),
399  {}};
400 
401  return spec;
402 }

404 DataProcessorSpec
405  CommonDataProcessors::getGlobalFileSink(std::vector<InputSpec> const& danglingOutputInputs,
406  std::vector<InputSpec>& unmatched)
407 {
408  auto writerFunction = [danglingOutputInputs](InitContext& ic) -> std::function<void(ProcessingContext&)> {
409  auto filename = ic.options().get<std::string>("outfile");
410  auto keepString = ic.options().get<std::string>("keep");
411 
412  if (filename.empty()) {
413  throw runtime_error("output file missing");
414  }
415 
416  bool hasOutputsToWrite = false;
417  auto [variables, outputMatcher] = DataDescriptorQueryBuilder::buildFromKeepConfig(keepString);
418  VariableContext context;
419  for (auto& spec : danglingOutputInputs) {
420  auto concrete = DataSpecUtils::asConcreteDataTypeMatcher(spec);
421  if (outputMatcher->match(concrete, context)) {
422  hasOutputsToWrite = true;
423  }
424  }
425  if (hasOutputsToWrite == false) {
426  return [](ProcessingContext&) mutable -> void {
427  static bool once = false;
428  if (!once) {
429  LOG(debug) << "No dangling output to be dumped.";
430  once = true;
431  }
432  };
433  }
434  auto output = std::make_shared<std::ofstream>(filename.c_str(), std::ios_base::binary);
435  return [output, matcher = outputMatcher](ProcessingContext& pc) mutable -> void {
436  VariableContext matchingContext;
437  LOG(debug) << "processing data set with " << pc.inputs().size() << " entries";
438  for (const auto& entry : pc.inputs()) {
439  LOG(debug) << " " << *(entry.spec);
440  auto header = DataRefUtils::getHeader<header::DataHeader*>(entry);
441  auto dataProcessingHeader = DataRefUtils::getHeader<DataProcessingHeader*>(entry);
442  if (matcher->match(*header, matchingContext) == false) {
443  continue;
444  }
445  output->write(reinterpret_cast<char const*>(header), sizeof(header::DataHeader));
446  output->write(reinterpret_cast<char const*>(dataProcessingHeader), sizeof(DataProcessingHeader));
447  output->write(entry.payload, o2::framework::DataRefUtils::getPayloadSize(entry));
448  LOG(debug) << "wrote data, size " << o2::framework::DataRefUtils::getPayloadSize(entry);
449  }
450  };
451  };
452 
453  std::vector<InputSpec> validBinaryInputs;
454  auto onlyTimeframe = [](InputSpec const& input) {
455  return (DataSpecUtils::partialMatch(input, o2::header::DataOrigin("TFN")) == false) &&
456  input.lifetime == Lifetime::Timeframe;
457  };
458 
459  auto noTimeframe = [](InputSpec const& input) {
460  return (DataSpecUtils::partialMatch(input, o2::header::DataOrigin("TFN")) == true) ||
461  input.lifetime != Lifetime::Timeframe;
462  };
463 
464  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
465  std::back_inserter(validBinaryInputs), onlyTimeframe);
466  std::copy_if(danglingOutputInputs.begin(), danglingOutputInputs.end(),
467  std::back_inserter(unmatched), noTimeframe);
468 
469  DataProcessorSpec spec{
470  "internal-dpl-injected-global-binary-file-sink",
471  validBinaryInputs,
472  Outputs{},
473  AlgorithmSpec(writerFunction),
474  {{"outfile", VariantType::String, "dpl-out.bin", {"Name of the output file"}},
475  {"keep", VariantType::String, "", {"Comma separated list of ORIGIN/DESCRIPTION/SUBSPECIFICATION to save in outfile"}}}};
476 
477  return spec;
478 }
