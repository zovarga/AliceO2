93 void convertDigitsToZSfinal(std::string_view digitsFile, std::string_view outputPath, std::string_view fileFor,
94  bool sectorBySector, uint32_t rdhV, bool stopPage, bool padding, bool createParentDir)
95 {
96 
97  // ===| open file and get tree |==============================================
98  std::unique_ptr<TFile> o2simDigits(TFile::Open(digitsFile.data()));
99  if (!o2simDigits || !o2simDigits->IsOpen() || o2simDigits->IsZombie()) {
100  LOGP(error, "Could not open file {}", digitsFile.data());
101  exit(1);
102  }
103  auto treeSim = (TTree*)o2simDigits->Get("o2sim");
104  if (!treeSim) {
105  LOGP(error, "Could not read digits tree from file {}", digitsFile.data());
106  exit(1);
107  }
108 
109  gROOT->cd();
110 
111  // ===| set up output directory |=============================================
112  std::string outDir{outputPath};
113  if (outDir.empty()) {
114  outDir = "./";
115  }
116  if (outDir.back() != '/') {
117  outDir += '/';
118  }
119 
120  // if needed, create output directory
121  if (!std::filesystem::exists(outDir)) {
122  if (createParentDir) {
123  o2::raw::assertOutputDirectory(outDir);
124  } else {
125  LOGP(error, "Requested output directory '{}' does not exists, consider removing '-n'", outDir);
126  exit(1);
127  }
128  }
129 
130  // ===| set up raw writer |===================================================
131  std::string inputGRP = o2::base::NameConf::getGRPFileName();
132  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
133 
134  o2::raw::RawFileWriter writer{"TPC"}; // to set the RDHv6.sourceID if V6 is used
135  writer.useRDHVersion(rdhV);
136  writer.setAddSeparateHBFStopPage(stopPage);
137  writer.setContinuousReadout(grp->isDetContinuousReadOut(o2::detectors::DetID::TPC)); // must be set explicitly
138  const unsigned int defaultLink = rdh_utils::UserLogicLinkID;
139 
140  for (unsigned int i = 0; i < NSectors; i++) {
141  for (unsigned int j = 0; j < NEndpoints; j++) {
142  const unsigned int cruInSector = j / 2;
143  const unsigned int cruID = i * 10 + cruInSector;
144  const rdh_utils::FEEIDType feeid = rdh_utils::getFEEID(cruID, j & 1, defaultLink);
145  std::string outfname;
146  if (fileFor == "all") { // single file for all links
147  outfname = fmt::format("{}tpc_all.raw", outDir);
148  } else if (fileFor == "sector") {
149  outfname = fmt::format("{}tpc_sector{}.raw", outDir, i);
150  } else if (fileFor == "link" || fileFor == "cru") {
151  outfname = fmt::format("{}TPC_{}_cru{}_{}.raw", outDir, CRU_FLPS[cruID], cruID, j & 1);
152  } else {
153  throw std::runtime_error("invalid option provided for file grouping");
154  }
155  writer.registerLink(feeid, cruID, defaultLink, j & 1, outfname);
156  }
157  }
158  if (fileFor != "link") { // in case >1 link goes to the file, we must cache to preserve the TFs ordering
159  writer.useCaching();
160  }
161  writer.doLazinessCheck(false); // LazinessCheck is not thread-safe
162 
163  // ===| set up branch addresses |=============================================
164  std::vector<Digit>* vDigitsPerSectorCollection[Sector::MAXSECTOR] = {nullptr}; // container that keeps Digits per sector
165 
166  treeSim->SetBranchStatus("*", 0);
167  treeSim->SetBranchStatus("TPCDigit_*", 1);
168 
169  ProcessAttributes attr;
170  attr.padding = padding;
171 
172  for (int iSecBySec = 0; iSecBySec < Sector::MAXSECTOR; ++iSecBySec) {
173  treeSim->ResetBranchAddresses();
174  for (int iSec = 0; iSec < Sector::MAXSECTOR; ++iSec) {
175  if (sectorBySector) {
176  iSec = iSecBySec;
177  }
178  vDigitsPerSectorCollection[iSec] = nullptr;
179  treeSim->SetBranchAddress(TString::Format("TPCDigit_%d", iSec), &vDigitsPerSectorCollection[iSec]);
180  if (sectorBySector) {
181  break;
182  }
183  }
184  for (Long64_t ievent = 0; ievent < treeSim->GetEntries(); ++ievent) {
185  DigitArray inputDigits;
186  if (sectorBySector) {
187  treeSim->GetBranch(TString::Format("TPCDigit_%d", iSecBySec))->GetEntry(ievent);
188  } else {
189  treeSim->GetEntry(ievent);
190  }
191 
192  for (int iSec = 0; iSec < Sector::MAXSECTOR; ++iSec) {
193  if (sectorBySector) {
194  iSec = iSecBySec;
195  }
196  inputDigits[iSec] = *vDigitsPerSectorCollection[iSec]; //????
197  if (sectorBySector) {
198  break;
199  }
200  }
201  convert(inputDigits, &attr, writer);
202  for (int iSec = 0; iSec < Sector::MAXSECTOR; ++iSec) {
203  delete vDigitsPerSectorCollection[iSec];
204  vDigitsPerSectorCollection[iSec] = nullptr;
205  }
206  }
207  if (!sectorBySector) {
208  break;
209  }
210  }
211  // for further use we write the configuration file for the output
212  writer.writeConfFile("TPC", "RAWDATA", fmt::format("{}tpcraw.cfg", outDir));
213 }

215 void convert(DigitArray& inputDigits, ProcessAttributes* processAttributes, o2::raw::RawFileWriter& writer)
216 {
217  const auto zsThreshold = processAttributes->zsThreshold;
218  const auto zs12bit = processAttributes->zs12bit;
219  GPUParam _GPUParam;
220  _GPUParam.SetDefaults(5.00668);
221  const GPUParam mGPUParam = _GPUParam;
222 
223  o2::InteractionRecord ir = o2::raw::HBFUtils::Instance().getFirstSampledTFIR();
224  ir.bc = 0; // By convention the TF starts at BC = 0
225  o2::gpu::GPUReconstructionConvert::RunZSEncoder<o2::tpc::Digit>(inputDigits, nullptr, nullptr, &writer, &ir, mGPUParam, zs12bit, false, zsThreshold, processAttributes->padding);
226 }
