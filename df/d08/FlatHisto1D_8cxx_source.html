24 template <typename T>
25 FlatHisto1D<T>::FlatHisto1D(uint32_t nb, T xmin, T xmax)
26 {
27  assert(nb > 0 && xmin < xmax);
28  mContainer.resize(nb + NServiceSlots, 0.);
29  mContainer[NBins] = nb;
30  mContainer[XMin] = xmin;
31  mContainer[XMax] = xmax;
32  mContainer[BinSize] = (xmax - xmin) / nb;
33  init(gsl::span<const T>(mContainer.data(), mContainer.size()));
34 }
