113 template <typename T, typename... Args>
114 AlgorithmSpec adaptFromTask(Args&&... args)
115 {
116  return AlgorithmSpec::InitCallback{[=](InitContext& ic) {
117  auto task = std::make_shared<T>(args...);
118  if constexpr (has_endOfStream<T>::value) {
119  auto& callbacks = ic.services().get<CallbackService>();
120  callbacks.set(CallbackService::Id::EndOfStream, [task](EndOfStreamContext& eosContext) {
121  task->endOfStream(eosContext);
122  });
123  }
124  if constexpr (has_finaliseCCDB<T>::value) {
125  auto& callbacks = ic.services().get<CallbackService>();
126  callbacks.set(CallbackService::Id::CCDBDeserialised, [task](ConcreteDataMatcher& matcher, void* obj) {
127  task->finaliseCCDB(matcher, obj);
128  });
129  }
130  if constexpr (has_stop<T>::value) {
131  auto& callbacks = ic.services().get<CallbackService>();
132  callbacks.set(CallbackService::Id::Stop, [task]() {
133  task->stop();
134  });
135  }
136  task->init(ic);
137  return [task](ProcessingContext& pc) {
138  task->run(pc);
139  };
140  }};
141 }
