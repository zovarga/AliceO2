29 BOOST_AUTO_TEST_CASE(TestInputRecord)
30 {
31  // Create the routes we want for the InputRecord
32  InputSpec spec1{"x", "TPC", "CLUSTERS", 0, Lifetime::Timeframe};
33  InputSpec spec2{"y", "ITS", "CLUSTERS", 0, Lifetime::Timeframe};
34  InputSpec spec3{"z", "TST", "EMPTY", 0, Lifetime::Timeframe};
35 
36  size_t i = 0;
37  auto createRoute = [&i](const char* source, InputSpec& spec) {
38  return InputRoute{
39  spec,
40  i++,
41  source};
42  };
43 
45  std::vector<InputRoute> schema = {
46  createRoute("x_source", spec1),
47  createRoute("y_source", spec2),
48  createRoute("z_source", spec3)};
49  // First of all we test if an empty registry behaves as expected, raising a
50  // bunch of exceptions.
51  InputSpan span{
52  [](size_t) { return DataRef{nullptr, nullptr, nullptr}; },
53  0};
54  ServiceRegistry registry;
55  InputRecord emptyRecord(schema, span, registry);
56 
57  BOOST_CHECK_EXCEPTION(emptyRecord.get("x"), RuntimeErrorRef, any_exception);
58  BOOST_CHECK_EXCEPTION(emptyRecord.get("y"), RuntimeErrorRef, any_exception);
59  BOOST_CHECK_EXCEPTION(emptyRecord.get("z"), RuntimeErrorRef, any_exception);
60  BOOST_CHECK_EXCEPTION(emptyRecord.getByPos(0), RuntimeErrorRef, any_exception);
61  BOOST_CHECK_EXCEPTION(emptyRecord.getByPos(1), RuntimeErrorRef, any_exception);
62  BOOST_CHECK_EXCEPTION(emptyRecord.getByPos(2), RuntimeErrorRef, any_exception);
63  // Then we actually check with a real set of inputs.
64 
65  std::vector<void*> inputs;
66 
67  auto createMessage = [&inputs](DataHeader& dh, int value) {
68  DataProcessingHeader dph{0, 1};
69  Stack stack{dh, dph};
70  void* header = malloc(stack.size());
71  void* payload = malloc(sizeof(int));
72  memcpy(header, stack.data(), stack.size());
73  memcpy(payload, &value, sizeof(int));
74  inputs.emplace_back(header);
75  inputs.emplace_back(payload);
76  };
77 
78  auto createEmpty = [&inputs]() {
79  inputs.emplace_back(nullptr);
80  inputs.emplace_back(nullptr);
81  };
82 
83  DataHeader dh1;
84  dh1.dataDescription = "CLUSTERS";
85  dh1.dataOrigin = "TPC";
86  dh1.subSpecification = 0;
87  dh1.payloadSerializationMethod = o2::header::gSerializationMethodNone;
88  DataHeader dh2;
89  dh2.dataDescription = "CLUSTERS";
90  dh2.dataOrigin = "ITS";
91  dh2.subSpecification = 0;
92  dh2.payloadSerializationMethod = o2::header::gSerializationMethodNone;
93  createMessage(dh1, 1);
94  createMessage(dh2, 2);
95  createEmpty();
96  InputSpan span2{[&inputs](size_t i) { return DataRef{nullptr, static_cast<char const*>(inputs[2 * i]), static_cast<char const*>(inputs[2 * i + 1])}; }, inputs.size() / 2};
97  InputRecord record{schema, span2, registry};
98 
99  // Checking we can get the whole ref by name
100  BOOST_CHECK_NO_THROW(record.get("x"));
101  BOOST_CHECK_NO_THROW(record.get("y"));
102  BOOST_CHECK_NO_THROW(record.get("z"));
103  auto ref00 = record.get("x");
104  auto ref10 = record.get("y");
105  auto ref20 = record.get("z");
106  BOOST_CHECK_EXCEPTION(record.get("err"), RuntimeErrorRef, any_exception);
107 
108  // Or we can get it positionally
109  BOOST_CHECK_NO_THROW(record.get("x"));
110  auto ref01 = record.getByPos(0);
111  auto ref11 = record.getByPos(1);
112  BOOST_CHECK_EXCEPTION(record.getByPos(10), RuntimeErrorRef, any_exception);
113 
114  // This should be exactly the same pointers
115  BOOST_CHECK_EQUAL(ref00.header, ref01.header);
116  BOOST_CHECK_EQUAL(ref00.payload, ref01.payload);
117  BOOST_CHECK_EQUAL(ref10.header, ref11.header);
118  BOOST_CHECK_EQUAL(ref10.payload, ref11.payload);
119 
120  BOOST_CHECK_EQUAL(record.isValid("x"), true);
121  BOOST_CHECK_EQUAL(record.isValid("y"), true);
122  BOOST_CHECK_EQUAL(record.isValid("z"), false);
123  BOOST_CHECK_EQUAL(record.size(), 3);
124  BOOST_CHECK_EQUAL(record.countValidInputs(), 2);
125 
126  BOOST_CHECK_EQUAL(record.isValid(0), true);
127  BOOST_CHECK_EQUAL(record.isValid(1), true);
128  BOOST_CHECK_EQUAL(record.isValid(2), false);
129  // This by default is a shortcut for
130  //
131  // *static_cast<int const *>(record.get("x").payload);
132  //
133  BOOST_CHECK_EQUAL(record.get<int>("x"), 1);
134  BOOST_CHECK_EQUAL(record.get<int>("y"), 2);
135  // A few more time just to make sure we are not stateful..
136  BOOST_CHECK_EQUAL(record.get<int>("x"), 1);
137  BOOST_CHECK_EQUAL(record.get<int>("x"), 1);
138 
139  // test the iterator
140  int position = 0;
141  for (auto input = record.begin(), end = record.end(); input != end; input++, position++) {
142  if (position == 0) {
143  BOOST_CHECK(input.matches("TPC") == true);
144  BOOST_CHECK(input.matches("TPC", "CLUSTERS") == true);
145  BOOST_CHECK(input.matches("ITS", "CLUSTERS") == false);
146  }
147  if (position == 1) {
148  BOOST_CHECK(input.matches("ITS") == true);
149  BOOST_CHECK(input.matches("ITS", "CLUSTERS") == true);
150  BOOST_CHECK(input.matches("TPC", "CLUSTERS") == false);
151  }
152  // check if invalid slots are filtered out by the iterator
153  BOOST_CHECK(position != 2);
154  }
155 
156  // the 2-level iterator to access inputs and their parts
157  // all inputs have 1 part, we check the first input
158  BOOST_CHECK(record.begin().size() == 1);
159  // the end-instance of the inputs has no parts
160  BOOST_CHECK(record.end().size() == 0);
161  // thus there is no element and begin == end
162  BOOST_CHECK(record.end().begin() == record.end().end());
163 }
