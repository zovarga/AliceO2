59  // Defines for GPU Architecture types (using the SM version to determine the # of cores per SM
60  typedef struct

69  {0x21, 48}, // Fermi Generation (SM 2.1) GF10x class
70  {0x30, 192}, // Kepler Generation (SM 3.0) GK10x class
71  {0x32, 192}, // Kepler Generation (SM 3.2) GK10x class
72  {0x35, 192}, // Kepler Generation (SM 3.5) GK11x class
73  {0x37, 192}, // Kepler Generation (SM 3.7) GK21x class
74  {0x50, 128}, // Maxwell Generation (SM 5.0) GM10x class
75  {0x52, 128}, // Maxwell Generation (SM 5.2) GM20x class
76  {0x53, 128}, // Maxwell Generation (SM 5.3) GM20x class
77  {0x60, 64}, // Pascal Generation (SM 6.0) GP100 class
78  {0x61, 128}, // Pascal Generation (SM 6.1) GP10x class
79  {0x62, 128}, // Pascal Generation (SM 6.2) GP10x class
80  {0x70, 64}, // Volta Generation (SM 7.0) GV100 class
81  {0x72, 64}, // Volta Generation (SM 7.2) GV10B class

83  {-1, -1}};

85  int index = 0;
