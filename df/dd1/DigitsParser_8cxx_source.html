37 inline void DigitsParser::swapByteOrder(unsigned int& word)
38 {
39  word = (word >> 24) |
40  ((word << 8) & 0x00FF0000) |
41  ((word >> 8) & 0x0000FF00) |
42  (word << 24);
43 }

69 void DigitsParser::OutputIncomingData()
70 {
71 
72  LOG(info) << "Data buffer to parse for Digits begin " << mStartParse << ":" << mEndParse;
73  int wordcount = 0;
74  std::stringstream outputstring;
75  auto word = mStartParse;
76  outputstring << "digit 0x" << std::hex << std::setfill('0') << std::setw(6) << 0 << " :: ";
77  while (word <= mEndParse) { // loop over the entire data buffer (a complete link of tracklets and digits)
78 
79  if (wordcount != 0 && (wordcount % 8 == 0 || word == mEndParse)) {
80  LOG(info) << outputstring.str();
81  outputstring.str("");
82  outputstring << "digit 0x" << std::hex << std::setfill('0') << std::setw(6) << wordcount << " :: ";
83  }
84  if (wordcount == 0) {
85  outputstring << " 0x" << std::hex << std::setfill('0') << std::setw(8) << HelperMethods::swapByteOrderreturn(*word);
86  } else {
87  outputstring << " 0x" << std::hex << std::setfill('0') << std::setw(8) << HelperMethods::swapByteOrderreturn(*word);
88  }
89  word++;
90  wordcount++;
91  }
92  LOG(info) << "Data buffer to parse for Digits end";
93 }

95 int DigitsParser::Parse(bool verbose)
96 {
97 
98  //we are handed the buffer payload of an rdh and need to parse its contents.
99  //producing a vector of digits.
100  mVerbose = verbose;
101 
102  mState = StateDigitMCMHeader;
103  mDataWordsParsed = 0; // count of data wordsin data that have been parsed in current call to parse.
104  mWordsDumped = 0;
105  mDigitsFound = 0; // tracklets found in the data block, mostly used for debugging.
106  mBufferLocation = 0;
107  mPaddingWordsCounter = 0;
108  int bitsinmask = 0;
109  int overchannelcount = 0;
110  int lastmcmread = 0;
111  int lastrobread = 0;
112  int lasteventcounterread = 0;
113  if (mHeaderVerbose) {
114  OutputIncomingData();
115  }
116  int mcmdatacount = 0;
117  int digittimebinoffset = 0;
118  //mData holds a buffer containing digits parse placing the read digits where they need to be
119  // due to the nature of the incoming data, there will *never* straggling digits or for that matter trap outputs spanning a boundary.
120  // data starts with a DigitHCHeader, so pull that off first to simplify looping
121 
122  for (auto word = mStartParse; word < mEndParse; ++word) { // loop over the entire data buffer (a complete link of digits)
123  //loop over all the words
124  //check for digit end marker
125  if (mByteOrderFix) {
126  // byte swap if needed.
127  swapByteOrder(*word);
128  }
129  auto nextword = std::next(word, 1);
130  if ((*word) == 0x0 && (*nextword == 0x0)) { // no need to byte swap nextword
131  // end of digits marker.
132  if (mHeaderVerbose) {
133  LOG(info) << "*** DigitEndMarker :0x" << std::hex << *word << "::0x" << *nextword << " at offset " << std::distance(mStartParse, word);
134  }
135  //state *should* be StateDigitMCMData check that it is
136  if (mState == StateDigitMCMData || mState == StateDigitEndMarker || mState == StateDigitHCHeader || mState == StateDigitMCMHeader) {
137  } else {
138  incParsingError(TRDParsingDigitEndMarkerWrongState);
139  }
140  //only thing that can remain is the padding.
141  //now read padding words till end.
142  //no need to byteswap its uniform.
143  mBufferLocation += 2; // 2 words forward.
144  mDataWordsParsed += 2;
145  mState = StateDigitEndMarker;
146  std::advance(word, 1); // advance over the second word of the endmarker
147  if (word < mEndParse) {
148  mWordsDumped += std::distance(word, mEndParse) - 1;
149  }
150  word = mEndParse;
151  } else {
152  if ((*word & 0xf) == 0xc && (mState == StateDigitMCMHeader || mState == StateDigitMCMData)) { //marker for DigitMCMHeader.
153  //read the header
154  mcmdatacount = 0;
155  mCurrentADCChannel = 0;
156  mDigitMCMHeader = (DigitMCMHeader*)(word);
157  if (mHeaderVerbose) {
158  printDigitMCMHeader(*mDigitMCMHeader);
159  }
160  if (!digitMCMHeaderSanityCheck(mDigitMCMHeader)) {
161  incParsingError(TRDParsingDigitMCMHeaderSanityCheckFailure);
162  // we dump the remainig data pending better options.
163  // we can try a 16 bit bitshift...
164  mWordsDumped = std::distance(word, mEndParse) - 1;
165  word = mEndParse;
166  continue;
167  }
168  // now check mcm/rob are read in correct order
169  if (mDigitMCMHeader->rob > lastrobread) {
170  lastmcmread = 0;
171  } else {
172  if (mDigitMCMHeader->rob < lastrobread) {
173  incParsingError(TRDParsingDigitROBDecreasing);
174  // we dump the remainig data pending better options.
175  // we can try a 16 bit bitshift...
176  mWordsDumped += std::distance(word, mEndParse) - 1;
177  word = mEndParse;
178  }
179  //the case of rob not changing we ignore, error condition handled by mcm # increasing.
180  }
181  if (mDigitMCMHeader->mcm < lastmcmread && mDigitMCMHeader->rob == lastrobread) {
182  incParsingError(TRDParsingDigitMCMNotIncreasing);
183  }
184  lastmcmread = mDigitMCMHeader->mcm;
185  lastrobread = mDigitMCMHeader->rob;
186  lasteventcounterread = mDigitMCMHeader->eventcount;
187  if (mDigitHCHeader.major & 0x20) {
188  //zero suppressed
189  //so we have an adcmask next
190  std::advance(word, 1);
191  if (mByteOrderFix) {
192  // byte swap if needed.
193  swapByteOrder(*word);
194  }
195  mDigitMCMADCMask = (DigitMCMADCMask*)(word);
196  mADCMask = mDigitMCMADCMask->adcmask;
197  if (mHeaderVerbose) {
198  LOG(info) << "**DigitADCMask is " << std::hex << mDigitMCMADCMask->adcmask << " raw form : 0x" << std::hex << mDigitMCMADCMask->word << " at offset " << std::distance(mStartParse, word);
199  }
200  if (word == mEndParse) {
201  incParsingError(TRDParsingDigitADCMaskAdvanceToEnd);
202  }
203  std::bitset<21> adcmask(mADCMask);
204  bitsinmask = adcmask.count();
205  //check ADCMask:
206  if (!digitMCMADCMaskSanityCheck(*mDigitMCMADCMask, bitsinmask)) {
207  incParsingError(TRDParsingDigitADCMaskMismatch);
208  mWordsDumped += std::distance(word, mEndParse) - 1;
209  word = mEndParse;
210  }
211  overchannelcount = 0;
212  //output all the adc data for the described adc mask, i.e 10 32 bit words per bit in mask.
213  }
214  mBufferLocation++;
215  //new header so digit word count becomes zero
216  mDigitWordCount = 0;
217  mState = StateDigitMCMData;
218  mMCM = mDigitMCMHeader->mcm;
219  mROB = mDigitMCMHeader->rob;
220  //TOOD does it match the feeid which ncodes this information as well.
221  //
222  mEventCounter = mDigitMCMHeader->eventcount;
223  mDataWordsParsed++; // header
224  if (mDigitHCHeader.major & 0x20) {
225  //zero suppressed digits
226  mDataWordsParsed++; // adc mask
227  }
228  mCurrentADCChannel = 0;
229  mADCValues.fill(0);
230  digittimebinoffset = 0;
231  if (!mReturnVector) {
232  //returning the raw "compressed" data stream.
233  //build the digit header and add to outgoing buffer;
234  uint32_t* header = &(*mData)[mReturnVectorPos];
235  //build header.
236  }
237  // we dont care about the year flag, we are >2007 already.
238  } else {
239  if (mState == StateDigitMCMHeader) { //safety check for some weird data occurances
240  unsigned int lastbit = (*word) & 0xf;
241  if (mHeaderVerbose) {
242  LOG(info) << " we bypassed the mcmheader block but the state is MCMHeader ... 0x" << std::hex << *word << " " << lastbit << " should == 0xc";
243  }
244  incParsingError(TRDParsingDigitMCMHeaderBypassButStateMCMHeader);
245  //dump data
246  mBufferLocation++;
247  mWordsDumped++;
248  // carry on parsing we might find an mcm header again
249  } else if (*word == o2::trd::constants::CRUPADDING32) {
250  if (mHeaderVerbose) {
251  LOG(info) << "***CRUPADDING32 word : 0x" << std::hex << *word << " state is:" << mState;
252  }
253  //another pointer with padding.
254  mBufferLocation++;
255  mPaddingWordsCounter++;
256  mState = StatePadding;
257  // mDataWordsParsed++;
258  mDataWordsParsed += std::distance(word, mEndParse);
259  //dump the rest of the received buffer its the rdh with e's
260  word = mEndParse;
261  mcmdatacount = 0;
262  // this is supposed to carry on till the end of the buffer, hence the term padding.
263  } else { // all we are left with is digitmcmdata words.
264  if (mState == StateDigitEndMarker) {
265 
266  if (mHeaderVerbose) {
267  LOG(info) << "***DigitEndMarker State : " << std::hex << *word << " at offset " << std::distance(mStartParse, word);
268  //we are at the end
269  // do nothing.
270  }
271  incParsingError(TRDParsingDigitEndMarkerStateButReadingMCMADCData);
272  mDataWordsParsed += std::distance(word, mEndParse) - 1;
273  word = mEndParse;
274  } else {
275  //for the case of on flp build a vector of tracklets, then pack them into a data stream with a header.
276  //for dpl build a vector and connect it with a triggerrecord.
277  if (mHeaderVerbose) {
278  LOG(info) << "***DigitMCMWord : " << std::hex << *word << " channel:" << std::dec << mCurrentADCChannel << " wordcount:" << mDigitWordCount << " at offset " << std::hex << std::distance(mStartParse, word);
279  }
280  if (mDigitWordCount == 0 || mDigitWordCount == constants::TIMEBINS / 3) {
281  //new adc expected so set channel accord to bitpattern or sequential depending on zero suppressed or not.
282  if (mDigitHCHeader.major & 0x20) { // zero suppressed
283  //zero suppressed, so channel must be extracted from next available bit in adcmask
284  mCurrentADCChannel = nextmcmadc(mADCMask, mCurrentADCChannel);
285  if (mCurrentADCChannel == 21) {
286  incParsingError(TRDParsingDigitADCChannel21);
287  }
288  if (mCurrentADCChannel > 22) {
289  LOG(error) << "invalid bitpattern (read a zero) for this mcm 0x" << std::hex << mADCMask << " at offset " << std::distance(mStartParse, word);
290  incParsingError(TRDParsingDigitADCChannelGT22);
291  mCurrentADCChannel = 100 * bitsinmask + overchannelcount++;
292  if (mHeaderVerbose) {
293  LOG(info) << "EEE " << mDetector << ":" << mROB << ":" << mMCM << ":" << mCurrentADCChannel
294  << " supermodule:stack:layer:side : " << mDigitHCHeader.supermodule << ":" << mDigitHCHeader.stack << ":" << mDigitHCHeader.layer << ":" << mDigitHCHeader.side;
295  }
296  }
297  if (mADCMask == 0) {
298  //no more adc for zero suppression.
299  //now we should either have another MCMHeader, or End marker
300  if (*word != 0 && *(std::next(word)) != 0) { // end marker is a sequence of 32 bit 2 zeros.
301  mState = StateDigitMCMHeader;
302  } else {
303  mState = StateDigitEndMarker;
304  }
305  }
306  }
307  }
308  if (mDigitWordCount > constants::TIMEBINS / 3) {
309  incParsingError(TRDParsingDigitGT10ADCs);
310  }
311  mDigitMCMData = (DigitMCMData*)word;
312  mBufferLocation++;
313  mDataWordsParsed++;
314  mcmdatacount++;
315  // digit sanity check
316  if (!digitMCMWordSanityCheck(mDigitMCMData, mCurrentADCChannel)) {
317  incParsingError(TRDParsingDigitSanityCheck);
318  mWordsDumped++;
319  mDataWordsParsed--; // we incremented it above the if loop so now transfer the count to dumped instead of parsed;
320  } else {
321  mState = StateDigitMCMData;
322  mDigitWordCount++;
323  mADCValues[digittimebinoffset++] = mDigitMCMData->z;
324  mADCValues[digittimebinoffset++] = mDigitMCMData->y;
325  mADCValues[digittimebinoffset++] = mDigitMCMData->x;
326 
327  if (digittimebinoffset > constants::TIMEBINS) {
328  incParsingError(TRDParsingDigitSanityCheck);
329  //bale out TODO
330  mWordsDumped += std::distance(word, mEndParse) - 1;
331  word = mEndParse;
332  }
333  if (mVerbose || mDataVerbose) {
334  LOG(info) << "digit word count is : " << mDigitWordCount << " digittimebinoffset = " << digittimebinoffset;
335  }
336  if (mDigitWordCount == constants::TIMEBINS / 3) {
337  //write out adc value to vector
338  //zero digittimebinoffset
339  mEventRecord->getDigits().emplace_back(mDetector, mROB, mMCM, mCurrentADCChannel, mADCValues); // outgoing parsed digits
340  mEventRecord->incDigitsFound(1);
341  //mEventRecords->incMCMDigitCount(mDetector, mROB, mMCM, 1);
342  if (mDataVerbose) {
343  LOG(info) << "DDD " << mDetector << ":" << mROB << ":" << mMCM << ":" << mCurrentADCChannel
344  << " supermodule:stack:layer:side : " << mDigitHCHeader.supermodule << ":" << mDigitHCHeader.stack << ":" << mDigitHCHeader.layer << ":" << mDigitHCHeader.side;
345  }
346  mDigitsFound++;
347  digittimebinoffset = 0;
348  mDigitWordCount = 0; // end of the digit.
349  if (!(mDigitHCHeader.major & 0x2)) {
350  mCurrentADCChannel++; // we count channels as all 21 channels are present, no way to check this.
351  }
352  } // mDigitWordCount == timebins/3
353  }
354  } //else digitendmarker if statement
355  } // else of if crupadding word
356  } // else of if digitMCMheader
357  } // else of if digitendmarker
358  //accounting
359  // mCurrentLinkDataPosition256++;
360  // mCurrentHalfCRUDataPosition256++;
361  // mTotalHalfCRUDataLength++;
362  //end of data so
363  } // for loop over word
364  if (!(mState == StateDigitMCMHeader || mState == StatePadding || mState == StateDigitEndMarker)) {
365  incParsingError(TRDParsingDigitParsingExitInWrongState);
366  }
367  if (std::distance(mStartParse, mEndParse) != mDataWordsParsed && mHeaderVerbose) {
368  if (mHeaderVerbose) {
369  LOG(info) << " we rejected " << mWordsDumped << " word and parse " << mDataWordsParsed << " % loss rate of " << (double)mWordsDumped / (double)mDataWordsParsed * 100.0;
370  }
371  }
372  return mDataWordsParsed;
373 }
