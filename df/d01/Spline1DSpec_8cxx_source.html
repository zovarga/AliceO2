57 template <class DataT>
58 void Spline1DContainer<DataT>::recreate(int nYdim, int numberOfKnots, const int inputKnots[])
59 {
69 
70  FlatObject::startConstruction();
71 
72  mYdim = (nYdim >= 0) ? nYdim : 0;
73 
74  std::vector<int> knotU;
75 
76  { // sort knots
77  std::vector<int> tmp;
78  for (int i = 0; i < numberOfKnots; i++) {
79  tmp.push_back(inputKnots[i]);
80  }
81  std::sort(tmp.begin(), tmp.end());
82 
83  knotU.push_back(0); // first knot at 0
84 
85  for (unsigned int i = 1; i < tmp.size(); ++i) {
86  int u = tmp[i] - tmp[0];
87  if (knotU.back() < u) { // remove duplicated knots
88  knotU.push_back(u);
89  }
90  }
91  if (knotU.back() < 1) { // there is only one knot at u=0, add the second one at u=1
92  knotU.push_back(1);
93  }
94  }
95 
96  mNumberOfKnots = knotU.size();
97  mUmax = knotU.back();
98  mXmin = 0.;
99  mXtoUscale = 1.;
100 
101  const int uToKnotMapOffset = mNumberOfKnots * sizeof(Knot);
102  int parametersOffset = uToKnotMapOffset + (mUmax + 1) * sizeof(int);
103  int bufferSize = parametersOffset;
104  if (mYdim > 0) {
105  parametersOffset = alignSize(bufferSize, getParameterAlignmentBytes());
106  bufferSize = parametersOffset + getSizeOfParameters();
107  }
108 
109  FlatObject::finishConstruction(bufferSize);
110 
111  mUtoKnotMap = reinterpret_cast<int*>(mFlatBufferPtr + uToKnotMapOffset);
112  mParameters = reinterpret_cast<DataT*>(mFlatBufferPtr + parametersOffset);
113 
114  for (int i = 0; i < getNumberOfParameters(); i++) {
115  mParameters[i] = 0;
116  }
117 
118  Knot* s = getKnots();
119 
120  for (int i = 0; i < mNumberOfKnots; i++) {
121  s[i].u = knotU[i];
122  }
123 
124  for (int i = 0; i < mNumberOfKnots - 1; i++) {
125  s[i].Li = 1. / (s[i + 1].u - s[i].u); // do division in double
126  }
127 
128  s[mNumberOfKnots - 1].Li = 0.; // the value will not be used, we define it for consistency
129 
130  // Set up the map (integer U) -> (knot index)
131 
132  int* map = getUtoKnotMap();
133 
134  const int iKnotMax = mNumberOfKnots - 2;
135 
136  //
137  // With iKnotMax=nKnots-2 we map the U==Umax coordinate to the last [nKnots-2, nKnots-1] segment.
138  // This trick allows one to avoid a special condition for this edge case.
139  // Any U from [0,Umax] is mapped to some knot_i such, that the next knot_i+1 always exist
140  //
141 
142  for (int u = 0, iKnot = 0; u <= mUmax; u++) {
143  if ((knotU[iKnot + 1] == u) && (iKnot < iKnotMax)) {
144  iKnot = iKnot + 1;
145  }
146  map[u] = iKnot;
147  }
148 }
