38 class CalibratordEdxDevice : public Task
39 {
40  public:
41  void init(framework::InitContext& ic) final
42  {
43  const int slotLength = ic.options().get<int>("tf-per-slot");
44  const int maxDelay = ic.options().get<int>("max-delay");
45  const int minEntries = ic.options().get<int>("min-entries");
46 
47  const int minEntriesSector = ic.options().get<int>("min-entries-sector");
48  const int minEntries1D = ic.options().get<int>("min-entries-1d");
49  const int minEntries2D = ic.options().get<int>("min-entries-2d");
50 
51  const int dEdxBins = ic.options().get<int>("dedxbins");
52  const int zBins = ic.options().get<int>("zbins");
53  const int angularBins = ic.options().get<int>("angularbins");
54  const float mindEdx = ic.options().get<float>("min-dedx");
55  const float maxdEdx = ic.options().get<float>("max-dedx");
56 
57  const bool dumpData = ic.options().get<bool>("file-dump");
58  float field = ic.options().get<float>("field");
59 
60  if (field <= -10.f) {
61  const auto inputGRP = o2::base::NameConf::getGRPFileName();
62  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
63  if (grp != nullptr) {
64  field = 5.00668f * grp->getL3Current() / 30000.;
65  LOGP(info, "Using GRP file to set the magnetic field to {} kG", field);
66  }
67  }
68 
69  mCalibrator = std::make_unique<tpc::CalibratordEdx>();
70  mCalibrator->setHistParams(mindEdx, maxdEdx, dEdxBins, zBins, angularBins);
71  mCalibrator->setApplyCuts(false);
72  mCalibrator->setFitCuts({minEntriesSector, minEntries1D, minEntries2D});
73  mCalibrator->setField(field);
74  mCalibrator->setMinEntries(minEntries);
75 
76  mCalibrator->setSlotLength(slotLength);
77  mCalibrator->setMaxSlotsDelay(maxDelay);
78 
79  if (dumpData) {
80  mCalibrator->enableDebugOutput("calibratordEdx.root");
81  }
82  }
83 
84  void run(ProcessingContext& pc) final
85  {
86  const auto tfcounter = o2::header::get<DataProcessingHeader*>(pc.inputs().get("tracks").header)->startTime;
87  const auto tracks = pc.inputs().get<gsl::span<tpc::TrackTPC>>("tracks");
88 
89  LOGP(info, "Processing TF {} with {} tracks", tfcounter, tracks.size());
90 
91  mCalibrator->process(tfcounter, tracks);
92  sendOutput(pc.outputs());
93 
94  const auto& infoVec = mCalibrator->getTFinterval();
95  LOGP(info, "Created {} objects for TF {}", infoVec.size(), tfcounter);
96  }
97 
98  void endOfStream(EndOfStreamContext& eos) final
99  {
100  LOGP(info, "Finalizing calibration");
101  constexpr calibration::TFType INFINITE_TF = 0xffffffffffffffff;
102  mCalibrator->checkSlotsToFinalize(INFINITE_TF);
103  sendOutput(eos.outputs());
104 
105  if (mCalibrator->hasDebugOutput()) {
106  mCalibrator->finalizeDebugOutput();
107  }
108  }
109 
110  private:
111  void sendOutput(DataAllocator& output)
112  {
113  using clbUtils = o2::calibration::Utils;
114  const auto& calibrations = mCalibrator->getCalibs();
115  auto& intervals = mCalibrator->getTFinterval();
116  const long timeEnd = 99999999999999;
117 
118  for (unsigned int i = 0; i < calibrations.size(); i++) {
119  const auto& object = calibrations[i];
120  o2::ccdb::CcdbObjectInfo info;
121  auto image = o2::ccdb::CcdbApi::createObjectImage(&object, &info);
122 
123  info.setPath("TPC/Calib/dEdx");
124  info.setStartValidityTimestamp(intervals[i].first);
125  info.setEndValidityTimestamp(timeEnd);
126 
127  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", info.getPath(), info.getFileName(), image->size(), info.getStartValidityTimestamp(), info.getEndValidityTimestamp());
128 
129  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "TPC_CalibdEdx", i}, *image.get()); // vector<char>
130  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "TPC_CalibdEdx", i}, info); // root-serialized
131  }
132  mCalibrator->initOutput(); // empty the outputs after they are send
133  }
134 
135  std::unique_ptr<CalibratordEdx> mCalibrator;
136 };

138 DataProcessorSpec getCalibratordEdxSpec()
139 {
140  std::vector<OutputSpec> outputs;
141  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "TPC_CalibdEdx"}, Lifetime::Sporadic);
142  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "TPC_CalibdEdx"}, Lifetime::Sporadic);
143 
144  return DataProcessorSpec{
145  "tpc-calibrator-dEdx",
146  Inputs{
147  InputSpec{"tracks", "TPC", "MIPS"},
148  },
149  outputs,
150  adaptFromTask<CalibratordEdxDevice>(),
151  Options{
152  {"tf-per-slot", VariantType::Int, 100, {"number of TFs per calibration time slot"}},
153  {"max-delay", VariantType::Int, 3, {"number of slots in past to consider"}},
154  {"min-entries", VariantType::Int, 50, {"minimum entries per stack to fit a single time slot"}},
155 
156  {"min-entries-sector", VariantType::Int, 1000, {"bellow this number of entries per stack, higher dimensional fits will be perform only for GEM stacks types (IROC, OROC1, ...). The mean is still corrected for every stack"}},
157  {"min-entries-1d", VariantType::Int, 500, {"minimum entries per stack to fit 1D correction"}},
158  {"min-entries-2d", VariantType::Int, 2500, {"minimum entries per stack to fit 2D correction"}},
159 
160  {"dedxbins", VariantType::Int, 100, {"number of dEdx bins"}},
161  {"zbins", VariantType::Int, 20, {"number of Z bins"}},
162  {"angularbins", VariantType::Int, 18, {"number of bins for angular data, like Tgl and Snp"}},
163  {"min-dedx", VariantType::Float, 5.0f, {"minimum value for the dEdx histograms"}},
164  {"max-dedx", VariantType::Float, 100.0f, {"maximum value for the dEdx histograms"}},
165 
166  {"field", VariantType::Float, -100.f, {"magnetic field"}},
167  {"file-dump", VariantType::Bool, false, {"directly dump calibration to file"}}}};
168 }
