58 void Digitizer::process(const std::vector<itsmft::Hit>* hits, int evID, int srcID)
59 {
60  // digitize single event, the time must have been set beforehand
61 
62  LOG(info) << "Digitizing " << mGeometry->getName() << " hits of entry " << evID << " from source "
63  << srcID << " at time " << mEventTime << " ROFrame= " << mNewROFrame << ")"
64  << " cont.mode: " << isContinuous()
65  << " Min/Max ROFrames " << mROFrameMin << "/" << mROFrameMax;
66 
67  // is there something to flush ?
68  if (mNewROFrame > mROFrameMin) {
69  fillOutputContainer(mNewROFrame - 1); // flush out all frame preceding the new one
70  }
71 
72  int nHits = hits->size();
73  std::vector<int> hitIdx(nHits);
74  std::iota(std::begin(hitIdx), std::end(hitIdx), 0);
75  // sort hits to improve memory access
76  std::sort(hitIdx.begin(), hitIdx.end(),
77  [hits](auto lhs, auto rhs) {
78  return (*hits)[lhs].GetDetectorID() < (*hits)[rhs].GetDetectorID();
79  });
80  for (int i : hitIdx) {
81  processHit((*hits)[i], mROFrameMax, evID, srcID);
82  }
83  // in the triggered mode store digits after every MC event
84  // TODO: in the real triggered mode this will not be needed, this is actually for the
85  // single event processing only
86  if (!mParams.isContinuous()) {
87  fillOutputContainer(mROFrameMax);
88  }
89 }

91 void Digitizer::setEventTime(const o2::InteractionTimeRecord& irt)
92 {
93  // assign event time in ns
94  mEventTime = irt;
95  if (!mParams.isContinuous()) {
96  mROFrameMin = 0; // in triggered mode reset the frame counters
97  mROFrameMax = 0;
98  }
99  // RO frame corresponding to provided time
100  mCollisionTimeWrtROF = mEventTime.timeInBCNS; // in triggered mode the ROF starts at BC (is there a delay?)
101  if (mParams.isContinuous()) {
102  auto nbc = mEventTime.differenceInBC(mIRFirstSampledTF);
103  if (mCollisionTimeWrtROF < 0 && nbc > 0) {
104  nbc--;
105  }
106  mNewROFrame = nbc / mParams.getROFrameLengthInBC();
107  // in continuous mode depends on starts of periodic readout frame
108  mCollisionTimeWrtROF += (nbc % mParams.getROFrameLengthInBC()) * o2::constants::lhc::LHCBunchSpacingNS;
109  } else {
110  mNewROFrame = 0;
111  }
112 
113  if (mNewROFrame < mROFrameMin) {
114  LOG(error) << "New ROFrame " << mNewROFrame << " (" << irt << ") precedes currently cashed " << mROFrameMin;
115  throw std::runtime_error("deduced ROFrame precedes already processed one");
116  }
117 
118  if (mParams.isContinuous() && mROFrameMax < mNewROFrame) {
119  mROFrameMax = mNewROFrame - 1; // all frames up to this are finished
120  }
121 }

123 void Digitizer::fillOutputContainer(uint32_t frameLast)
124 {
125  // fill output with digits from min.cached up to requested frame, generating the noise beforehand
126  if (frameLast > mROFrameMax) {
127  frameLast = mROFrameMax;
128  }
129  // make sure all buffers for extra digits are created up to the maxFrame
130  getExtraDigBuffer(mROFrameMax);
131 
132  LOG(info) << "Filling " << mGeometry->getName() << " digits output for RO frames " << mROFrameMin << ":"
133  << frameLast;
134 
135  o2::itsmft::ROFRecord rcROF;
136 
137  // we have to write chips in RO increasing order, therefore have to loop over the frames here
138  for (; mROFrameMin <= frameLast; mROFrameMin++) {
139  rcROF.setROFrame(mROFrameMin);
140  rcROF.setFirstEntry(mDigits->size()); // start of current ROF in digits
141 
142  auto& extra = *(mExtraBuff.front().get());
143  for (int iChip{0}; iChip < mChips.size(); ++iChip) {
144  auto& chip = mChips[iChip];
145  if (iChip < SegmentationSuperAlpide::NLayers) {
146  chip.addNoise(mROFrameMin, mROFrameMin, &mParams, mSuperSegmentations[iChip].NRows, mSuperSegmentations[iChip].NCols);
147  } else {
148  chip.addNoise(mROFrameMin, mROFrameMin, &mParams);
149  }
150  auto& buffer = chip.getPreDigits();
151  if (buffer.empty()) {
152  continue;
153  }
154  auto itBeg = buffer.begin();
155  auto iter = itBeg;
156  ULong64_t maxKey = chip.getOrderingKey(mROFrameMin + 1, 0, 0) - 1; // fetch digits with key below that
157  for (; iter != buffer.end(); ++iter) {
158  if (iter->first > maxKey) {
159  break; // is the digit ROFrame from the key > the max requested frame
160  }
161  auto& preDig = iter->second; // preDigit
162  if (preDig.charge >= mParams.getChargeThreshold()) {
163  int digID = mDigits->size();
164  mDigits->emplace_back(chip.getChipIndex(), preDig.row, preDig.col, preDig.charge);
165  mMCLabels->addElement(digID, preDig.labelRef.label);
166  auto& nextRef = preDig.labelRef; // extra contributors are in extra array
167  while (nextRef.next >= 0) {
168  nextRef = extra[nextRef.next];
169  mMCLabels->addElement(digID, nextRef.label);
170  }
171  }
172  }
173  buffer.erase(itBeg, iter);
174  }
175  // finalize ROF record
176  rcROF.setNEntries(mDigits->size() - rcROF.getFirstEntry()); // number of digits
177  if (isContinuous()) {
178  rcROF.getBCData().setFromLong(mIRFirstSampledTF.toLong() + mROFrameMin * mParams.getROFrameLengthInBC());
179  } else {
180  rcROF.getBCData() = mEventTime; // RSTODO do we need to add trigger delay?
181  }
182  if (mROFRecords) {
183  mROFRecords->push_back(rcROF);
184  }
185  extra.clear(); // clear container for extra digits of the mROFrameMin ROFrame
186  // and move it as a new slot in the end
187  mExtraBuff.emplace_back(mExtraBuff.front().release());
188  mExtraBuff.pop_front();
189  }

192 void Digitizer::processHit(const o2::itsmft::Hit& hit, uint32_t& maxFr, int evID, int srcID)
193 {
194  // convert single hit to digits
195  float timeInROF = hit.GetTime() * sec2ns;
196  if (timeInROF > 20e3) {
197  const int maxWarn = 10;
198  static int warnNo = 0;
199  if (warnNo < maxWarn) {
200  LOG(warning) << "Ignoring hit with time_in_event = " << timeInROF << " ns"
201  << ((++warnNo < maxWarn) ? "" : " (suppressing further warnings)");
202  }
203  return;
204  }
205  if (isContinuous()) {
206  timeInROF += mCollisionTimeWrtROF;
207  }
208  // calculate RO Frame for this hit
209  if (timeInROF < 0) {
210  timeInROF = 0.;
211  }
212  float tTot = mParams.getSignalShape().getMaxDuration();
213  // frame of the hit signal start wrt event ROFrame
214  int roFrameRel = int(timeInROF * mParams.getROFrameLengthInv());
215  // frame of the hit signal end wrt event ROFrame: in the triggered mode we read just 1 frame
216  uint32_t roFrameRelMax = mParams.isContinuous() ? (timeInROF + tTot) * mParams.getROFrameLengthInv() : roFrameRel;
217  int nFrames = roFrameRelMax + 1 - roFrameRel;
218  uint32_t roFrameMax = mNewROFrame + roFrameRelMax;
219  if (roFrameMax > maxFr) {
220  maxFr = roFrameMax; // if signal extends beyond current maxFrame, increase the latter
221  }
222 
223  // here we start stepping in the depth of the sensor to generate charge diffision
224  float nStepsInv = mParams.getNSimStepsInv();
225  int nSteps = mParams.getNSimSteps();
226  short detID{hit.GetDetectorID()};
227  const auto& matrix = mGeometry->getMatrixL2G(detID);
228  bool innerBarrel{detID < SegmentationSuperAlpide::NLayers};
229  auto startPos = hit.GetPosStart();
230  float startPhi{std::atan2(-startPos.Y(), -startPos.X())};
231  auto endPos = hit.GetPos();
232  float endPhi{std::atan2(-endPos.Y(), -endPos.X())};
233  math_utils::Vector3D<float> xyzLocS, xyzLocE;
234  if (innerBarrel) {
235  xyzLocS = {SegmentationSuperAlpide::Radii[detID] * startPhi, 0.f, startPos.Z()};
236  xyzLocE = {SegmentationSuperAlpide::Radii[detID] * endPhi, 0.f, endPos.Z()};
237  } else {
238  xyzLocS = matrix ^ (hit.GetPosStart());
239  xyzLocE = matrix ^ (hit.GetPos());
240  }
241 
242  math_utils::Vector3D<float> step(xyzLocE);
243  step -= xyzLocS;
244  step *= nStepsInv; // position increment at each step
245  // the electrons will injected in the middle of each step
246  math_utils::Vector3D<float> stepH(step * 0.5);
247  xyzLocS += stepH;
248  xyzLocE -= stepH;
249 
250  int rowS = -1, colS = -1, rowE = -1, colE = -1, nSkip = 0;
251  if (innerBarrel) {
252  // get entrance pixel row and col
253  while (!mSuperSegmentations[detID].localToDetector(xyzLocS.X(), xyzLocS.Z(), rowS, colS)) { // guard-ring ?
254  if (++nSkip >= nSteps) {
255  return; // did not enter to sensitive matrix
256  }
257  xyzLocS += step;
258  }
259  // get exit pixel row and col
260  while (!mSuperSegmentations[detID].localToDetector(xyzLocE.X(), xyzLocE.Z(), rowE, colE)) { // guard-ring ?
261  if (++nSkip >= nSteps) {
262  return; // did not enter to sensitive matrix
263  }
264  xyzLocE -= step;
265  }
266  } else {
267  // get entrance pixel row and col
268  while (!Segmentation::localToDetector(xyzLocS.X(), xyzLocS.Z(), rowS, colS)) { // guard-ring ?
269  if (++nSkip >= nSteps) {
270  return; // did not enter to sensitive matrix
271  }
272  xyzLocS += step;
273  }
274  // get exit pixel row and col
275  while (!Segmentation::localToDetector(xyzLocE.X(), xyzLocE.Z(), rowE, colE)) { // guard-ring ?
276  if (++nSkip >= nSteps) {
277  return; // did not enter to sensitive matrix
278  }
279  xyzLocE -= step;
280  }
281  }
282  // estimate the limiting min/max row and col where the non-0 response is possible
283  if (rowS > rowE) {
284  std::swap(rowS, rowE);
285  }
286  if (colS > colE) {
287  std::swap(colS, colE);
288  }
289  rowS -= AlpideRespSimMat::NPix / 2;
290  rowE += AlpideRespSimMat::NPix / 2;
291  if (rowS < 0) {
292  rowS = 0;
293  }
294 
295  int maxNrows{innerBarrel ? mSuperSegmentations[detID].NRows : Segmentation::NRows};
296  int maxNcols{innerBarrel ? mSuperSegmentations[detID].NCols : Segmentation::NCols};
297  if (rowE >= maxNrows) {
298  rowE = maxNrows - 1;
299  }
300  colS -= AlpideRespSimMat::NPix / 2;
301  colE += AlpideRespSimMat::NPix / 2;
302  if (colS < 0) {
303  colS = 0;
304  }
305  if (colE >= maxNcols) {
306  colE = maxNcols - 1;
307  }
308  int rowSpan = rowE - rowS + 1, colSpan = colE - colS + 1; // size of plaquet where some response is expected
309  float respMatrix[rowSpan][colSpan]; // response accumulated here
310  std::fill(&respMatrix[0][0], &respMatrix[0][0] + rowSpan * colSpan, 0.f);
311 
312  float nElectrons = hit.GetEnergyLoss() * mParams.getEnergyToNElectrons(); // total number of deposited electrons
313  nElectrons *= nStepsInv; // N electrons injected per step
314  if (nSkip) {
315  nSteps -= nSkip;
316  }
317  //
318  int rowPrev = -1, colPrev = -1, row, col;
319  float cRowPix = 0.f, cColPix = 0.f; // local coordinated of the current pixel center
320 
321  const o2::itsmft::AlpideSimResponse* resp = mParams.getAlpSimResponse();
322 
323  // take into account that the AlpideSimResponse depth defintion has different min/max boundaries
324  // although the max should coincide with the surface of the epitaxial layer, which in the chip
325  // local coordinates has Y = +SensorLayerThickness/2
326 
327  xyzLocS.SetY(xyzLocS.Y() + resp->getDepthMax() - Segmentation::SensorLayerThickness / 2.);
328 
329  // collect charge in evey pixel which might be affected by the hit
330  for (int iStep = nSteps; iStep--;) {
331  // Get the pixel ID
332  if (innerBarrel) {
333  mSuperSegmentations[detID].localToDetector(xyzLocS.X(), xyzLocS.Z(), row, col);
334  } else {
335  Segmentation::localToDetector(xyzLocS.X(), xyzLocS.Z(), row, col);
336  }
337  if (row != rowPrev || col != colPrev) { // update pixel and coordinates of its center
338  if (innerBarrel) {
339  if (!mSuperSegmentations[detID].detectorToLocal(row, col, cRowPix, cColPix)) {
340  continue;
341  }
342  } else if (!Segmentation::detectorToLocal(row, col, cRowPix, cColPix)) {
343  continue; // should not happen
344  }
345  rowPrev = row;
346  colPrev = col;
347  }
348  bool flipCol, flipRow;
349  // note that response needs coordinates along column row (locX) (locZ) then depth (locY)
350  float rowMax{0.5f * (innerBarrel ? mSuperSegmentations[detID].PitchRow : Segmentation::PitchRow)};
351  float colMax{0.5f * (innerBarrel ? mSuperSegmentations[detID].PitchCol : Segmentation::PitchCol)};
352  auto rspmat = resp->getResponse(xyzLocS.X() - cRowPix, xyzLocS.Z() - cColPix, xyzLocS.Y(), flipRow, flipCol, rowMax, colMax);
353 
354  xyzLocS += step;
355  if (!rspmat) {
356  continue;
357  }
358 
359  for (int irow = AlpideRespSimMat::NPix; irow--;) {
360  int rowDest = row + irow - AlpideRespSimMat::NPix / 2 - rowS; // destination row in the respMatrix
361  if (rowDest < 0 || rowDest >= rowSpan) {
362  continue;
363  }
364  for (int icol = AlpideRespSimMat::NPix; icol--;) {
365  int colDest = col + icol - AlpideRespSimMat::NPix / 2 - colS; // destination column in the respMatrix
366  if (colDest < 0 || colDest >= colSpan) {
367  continue;
368  }
369  respMatrix[rowDest][colDest] += rspmat->getValue(irow, icol, flipRow, flipCol);
370  }
371  }
372  }
373 
374  // fire the pixels assuming Poisson(n_response_electrons)
375  o2::MCCompLabel lbl(hit.GetTrackID(), evID, srcID, false);
376  auto& chip = mChips[detID];
377  auto roFrameAbs = mNewROFrame + roFrameRel;
378  for (int irow = rowSpan; irow--;) {
379  uint16_t rowIS = irow + rowS;
380  for (int icol = colSpan; icol--;) {
381  float nEleResp = respMatrix[irow][icol];
382  if (nEleResp <= 1.e-36) {
383  continue;
384  }
385  int nEle = gRandom->Poisson(nElectrons * nEleResp); // total charge in given pixel
386  // ignore charge which have no chance to fire the pixel
387  if (nEle < mParams.getMinChargeToAccount()) {
388  continue;
389  }
390  uint16_t colIS = icol + colS;
391  //
392  registerDigits(chip, roFrameAbs, timeInROF, nFrames, rowIS, colIS, nEle, lbl);
393  }
394  }
395 }

397 void Digitizer::registerDigits(o2::itsmft::ChipDigitsContainer& chip, uint32_t roFrame, float tInROF, int nROF,
398  uint16_t row, uint16_t col, int nEle, o2::MCCompLabel& lbl)
399 {
400  // Register digits for given pixel, accounting for the possible signal contribution to
401  // multiple ROFrame. The signal starts at time tInROF wrt the start of provided roFrame
402  // In every ROFrame we check the collected signal during strobe
403 
404  float tStrobe = mParams.getStrobeDelay() - tInROF; // strobe start wrt signal start
405  for (int i = 0; i < nROF; i++) {
406  uint32_t roFr = roFrame + i;
407  int nEleROF = mParams.getSignalShape().getCollectedCharge(nEle, tStrobe, tStrobe + mParams.getStrobeLength());
408  tStrobe += mParams.getROFrameLength(); // for the next ROF
409 
410  // discard too small contributions, they have no chance to produce a digit
411  if (nEleROF < mParams.getMinChargeToAccount()) {
412  continue;
413  }
414  if (roFr > mEventROFrameMax) {
415  mEventROFrameMax = roFr;
416  }
417  if (roFr < mEventROFrameMin) {
418  mEventROFrameMin = roFr;
419  }
420  auto key = chip.getOrderingKey(roFr, row, col);
421  PreDigit* pd = chip.findDigit(key);
422  if (!pd) {
423  chip.addDigit(key, roFr, row, col, nEleROF, lbl);
424  } else { // there is already a digit at this slot, account as PreDigitExtra contribution
425  pd->charge += nEleROF;
426  if (pd->labelRef.label == lbl) { // don't store the same label twice
427  continue;
428  }
429  ExtraDig* extra = getExtraDigBuffer(roFr);
430  int& nxt = pd->labelRef.next;
431  bool skip = false;
432  while (nxt >= 0) {
433  if ((*extra)[nxt].label == lbl) { // don't store the same label twice
434  skip = true;
435  break;
436  }
437  nxt = (*extra)[nxt].next;
438  }
439  if (skip) {
440  continue;
441  }
442  // new predigit will be added in the end of the chain
443  nxt = extra->size();
444  extra->emplace_back(lbl);
445  }
446  }
