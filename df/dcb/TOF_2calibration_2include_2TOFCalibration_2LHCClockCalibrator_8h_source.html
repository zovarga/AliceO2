30 struct LHCClockDataHisto {
31  float range = o2::tof::Geo::BC_TIME_INPS * 0.5;
32  int nbins = 1000;
33  float v2Bin = nbins / (2 * range);
34  int entries = 0;
35  o2::tof::CalibTOFapi* calibApi;
36  std::vector<float> histo{0};
37 
38  LHCClockDataHisto();
39 
40  LHCClockDataHisto(int nb, float r, o2::tof::CalibTOFapi* api) : nbins(nb), range(r), v2Bin(0), calibApi(api)
41  {
42  if (r <= 0. || nb < 1) {
43  throw std::runtime_error("Wrong initialization of the histogram");
44  }
45  v2Bin = nbins / (2 * range);
46  histo.resize(nbins, 0.);
47  }
48 
49  size_t getEntries() const { return entries; }
50  void print() const;
51  void fill(const gsl::span<const o2::dataformats::CalibInfoTOF> data);
52  void merge(const LHCClockDataHisto* prev);
53 
54  ClassDefNV(LHCClockDataHisto, 1);
55 };
