50  const auto grp = o2::parameters::GRPObject::loadFrom();
51  if (grp) {
52  mQAConfig->configGRP.solenoidBz = 5.00668f * grp->getL3Current() / 30000.;
53  mQAConfig->configGRP.continuousMaxTimeBin = grp->isDetContinuousReadOut(o2::detectors::DetID::TPC) ? -1 : 0;

55  throw std::runtime_error("Failed to initialize run parameters from GRP");
56  }
57  mQAConfig->ReadConfigurableParam();
58  mQAConfig->configQA.shipToQCAsCanvas = mOutputMode == outputLayout;
59  mQA = std::make_unique<GPUO2InterfaceQA>(mQAConfig.get());
