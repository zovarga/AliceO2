38 void RawFileWriter::close()
39 {
40  // finalize all links
41  if (mFName2File.empty()) {
42  return;
43  }
44  if (mCachingStage) {
45  fillFromCache();
46  }
47  if (mDoLazinessCheck) {
48  IR newIR = mDetLazyCheck.ir;
49  mDetLazyCheck.completeLinks(this, ++newIR); // make sure that all links for previously called IR got their addData call
50  mDoLazinessCheck = false;
51  }
52 
53  if (!mFirstIRAdded.isDummy()) { // flushing and completing the last HBF makes sense only if data was added.
54  auto irmax = getIRMax();
55  // for CRU detectors link.updateIR and hence the irmax points on the last IR with data + 1 orbit
56  if (isCRUDetector()) {
57  irmax.orbit -= 1;
58  }
59  for (auto& lnk : mSSpec2Link) {
60  lnk.second.close(irmax);
61  lnk.second.print();
62  }
63  }
64  //
65  // close all files
66  for (auto& flh : mFName2File) {
67  LOG(info) << "Closing output file " << flh.first;
68  fclose(flh.second.handler);
69  flh.second.handler = nullptr;
70  }
71  mFName2File.clear();
72  if (mDetLazyCheck.completeCount) {
73  LOG(warning) << "RawFileWriter forced " << mDetLazyCheck.completeCount << " dummy addData calls in "
74  << mDetLazyCheck.irSeen << " IRs for links which did not receive data";
75  }
76  mTimer.Stop();
77  mTimer.Print();
78 }

81 void RawFileWriter::fillFromCache()
82 {
83  LOG(info) << "Filling links from cached trees";
84  mCachingStage = false;
85  for (const auto& cache : mCacheMap) {
86  for (const auto& entry : cache.second) {
87  auto& link = getLinkWithSubSpec(entry.first);
88  link.cacheTree->GetEntry(entry.second);
89  if (mDoLazinessCheck) {
90  mDetLazyCheck.completeLinks(this, cache.first); // make sure that all links for previously called IR got their addData call
91  mDetLazyCheck.acknowledge(link.subspec, cache.first, link.cacheBuffer.preformatted, link.cacheBuffer.trigger, link.cacheBuffer.detField);
92  }
93  link.addData(cache.first, link.cacheBuffer.payload, link.cacheBuffer.preformatted, link.cacheBuffer.trigger, link.cacheBuffer.detField);
94  }
95  }
96  mCacheFile->cd();
97  for (auto& linkEntry : mSSpec2Link) {
98  if (linkEntry.second.cacheTree) {
99  linkEntry.second.cacheTree->Write();
100  linkEntry.second.cacheTree.reset(nullptr);
101  }
102  }
103  std::string cacheName{mCacheFile->GetName()};
104  mCacheFile->Close();
105  mCacheFile.reset(nullptr);
106  unlink(cacheName.c_str());
107 }

136 RawFileWriter::LinkData& RawFileWriter::registerLink(uint16_t fee, uint16_t cru, uint8_t link, uint8_t endpoint, std::string_view outFileNameV)
137 {
138  // register the GBT link and its output file
139  std::string outFileName{outFileNameV};
140  auto sspec = RDHUtils::getSubSpec(cru, link, endpoint, fee);
141  auto& linkData = mSSpec2Link[sspec];
142  auto& file = mFName2File[std::string(outFileName)];
143  if (!file.handler) {
144  if (!(file.handler = fopen(outFileName.c_str(), "wb"))) { // if file does not exist, create it
145  LOG(error) << "Failed to open output file " << outFileName;
146  throw std::runtime_error(std::string("cannot open link output file ") + outFileName);
147  }
148  }
149  if (!linkData.fileName.empty()) { // this link was already declared and associated with a file
150  if (linkData.fileName == outFileName) {
151  LOGF(info, "Link 0x%ux was already declared with same output, do nothing", sspec);
152  return linkData;
153  } else {
154  LOGF(error, "Link 0x%ux was already connected to different output file %s", sspec, linkData.fileName);
155  throw std::runtime_error("redifinition of the link output file");
156  }
157  }
158  linkData.fileName = outFileName;
159  linkData.subspec = sspec;
160  RDHUtils::setVersion(linkData.rdhCopy, mUseRDHVersion);
161  RDHUtils::setFEEID(linkData.rdhCopy, fee);
162  RDHUtils::setCRUID(linkData.rdhCopy, cru);
163  RDHUtils::setLinkID(linkData.rdhCopy, link);
164  RDHUtils::setEndPointID(linkData.rdhCopy, endpoint);
165  if (mUseRDHVersion >= 6) {
166  RDHUtils::setSourceID(linkData.rdhCopy, o2::header::DAQID::O2toDAQ(mOrigin));
167  }
168  linkData.writer = this;
169  linkData.updateIR = mHBFUtils.getFirstIR();
170  linkData.buffer.reserve(mSuperPageSize);
171  RDHUtils::printRDH(linkData.rdhCopy);
172  LOGF(info, "Registered %s with output to %s", linkData.describe(), outFileName);
173  return linkData;
174 }

177 void RawFileWriter::addData(uint16_t feeid, uint16_t cru, uint8_t lnk, uint8_t endpoint, const IR& ir, const gsl::span<char> data, bool preformatted, uint32_t trigger, uint32_t detField)
178 {
179  // add payload to relevant links
180  auto sspec = RDHUtils::getSubSpec(cru, lnk, endpoint, feeid);
181  auto& link = getLinkWithSubSpec(sspec);
182  if (mVerbosity > 10) {
183  LOGP(info, "addData for {} on IR BCid:{} Orbit: {}, payload: {}, preformatted: {}, trigger: {}, detField: {}", link.describe(), ir.bc, ir.orbit, data.size(), preformatted, trigger, detField);
184  }
185  if (isCRUDetector() && (data.size() % RDHUtils::GBTWord)) {
186  LOG(error) << "provided payload size " << data.size() << " is not multiple of GBT word size";
187  throw std::runtime_error("payload size is not mutiple of GBT word size");
188  }
189  if (ir < mHBFUtils.getFirstIR()) {
190  LOG(warning) << "provided " << ir << " precedes first TF " << mHBFUtils.getFirstIR() << " | discarding data for " << link.describe();
191  return;
192  }
193  if (link.discardData || ir.orbit - mHBFUtils.orbitFirst >= mHBFUtils.maxNOrbits) {
194  if (!link.discardData) {
195  link.discardData = true;
196  LOG(info) << "Orbit " << ir.orbit << ": max. allowed orbit " << mHBFUtils.orbitFirst + mHBFUtils.maxNOrbits - 1 << " exceeded, " << link.describe() << " will discard further data";
197  }
198  return;
199  }
200  if (ir < mFirstIRAdded) {
201  mHBFUtils.checkConsistency(); // done only once
202  mFirstIRAdded = ir;
203  }
204  if (mDoLazinessCheck && !mCachingStage) {
205  mDetLazyCheck.completeLinks(this, ir); // make sure that all links for previously called IR got their addData call
206  mDetLazyCheck.acknowledge(sspec, ir, preformatted, trigger, detField);
207  }
208  link.addData(ir, data, preformatted, trigger, detField);
209 }

265 void RawFileWriter::useCaching()
266 {
267  // impose preliminary caching of data to the tree, used in case of async. data input
268  if (!mFirstIRAdded.isDummy()) {
269  throw std::runtime_error("caching must be requested before feeding the data");
270  }
271  mCachingStage = true;
272  if (mCacheFile) {
273  return; // already done
274  }
275  auto cachename = o2::utils::Str::concat_string("_rawWriter_cache_", mOrigin.str, ::getpid(), ".root");
276  mCacheFile.reset(TFile::Open(cachename.c_str(), "recreate"));
277  LOG(info) << "Switched caching ON";
278 }

283 void RawFileWriter::LinkData::cacheData(const IR& ir, const gsl::span<char> data, bool preformatted, uint32_t trigger, uint32_t detField)
284 {
285  // cache data to temporary tree
286  std::lock_guard<std::mutex> lock(writer->mCacheFileMtx);
287  if (!cacheTree) {
288  writer->mCacheFile->cd();
289  cacheTree = std::make_unique<TTree>(o2::utils::Str::concat_string("lnk", std::to_string(subspec)).c_str(), "cache");
290  cacheTree->Branch("cache", &cacheBuffer);
291  }
292  cacheBuffer.preformatted = preformatted;
293  cacheBuffer.trigger = trigger;
294  cacheBuffer.detField = detField;
295  cacheBuffer.payload.resize(data.size());
296  if (!data.empty()) {
297  memcpy(cacheBuffer.payload.data(), data.data(), data.size());
298  }
299  writer->mCacheMap[ir].emplace_back(subspec, cacheTree->GetEntries());
300  cacheTree->Fill();
301  return;
302 }

312 //___________________________________________________________________________________
313 void RawFileWriter::LinkData::addDataInternal(const IR& ir, const gsl::span<char> data, bool preformatted, uint32_t trigger, uint32_t detField, bool checkEmpty)
314 {
315  // add payload corresponding to IR
316  LOG(debug) << "Adding " << data.size() << " bytes in IR " << ir << " to " << describe() << " checkEmpty=" << checkEmpty;
317  if (writer->mCachingStage) {
318  cacheData(ir, data, preformatted, trigger, detField);
319  return;
320  }
321 
322  if (startOfRun && writer->isRORCDetector()) { // in RORC mode we write separate RDH with SOX in the very beginning of the run
323  writer->mHBFUtils.updateRDH<RDHAny>(rdhCopy, writer->mHBFUtils.getFirstIR(), false);
324  RDHUtils::setTriggerType(rdhCopy, 0);
325  openHBFPage(rdhCopy); // open new HBF just to report the SOX
326  // closeHBFPage();
327  }
328 
329  int dataSize = data.size();
330  if (ir >= updateIR && checkEmpty) { // new IR exceeds or equal IR of next HBF to open, insert missed HBFs if needed
331  fillEmptyHBHs(ir, true);
332  }
333  // we are guaranteed to be under the valid RDH + possibly some data
334 
335  if (trigger) {
336  auto& rdh = *getLastRDH();
337  RDHUtils::setTriggerType(rdh, RDHUtils::getTriggerType(rdh) | trigger);
338  }
339  if (detField) {
340  auto& rdh = *getLastRDH();
341  RDHUtils::setDetectorField(rdh, detField);
342  }
343 
344  if (!dataSize) {
345  return;
346  }
347  if (preformatted) { // in case detectors wants to add new CRU page of predefined size
348  addPreformattedCRUPage(data);
349  return;
350  }
351 
352  // if we are at the beginning of the page, detector may want to add some header
353  if (isNewPage() && writer->newRDHFunc) {
354  std::vector<char> newPageHeader;
355  writer->newRDHFunc(getLastRDH(), false, newPageHeader);
356  pushBack(newPageHeader.data(), newPageHeader.size());
357  }
358 
359  const char* ptr = &data[0];
360  // in case particular detector CRU pages need to be self-consistent, when carrying-over
361  // large payload to new CRU page we may need to write optional trailer and header before
362  // and after the new RDH.
363  bool carryOver = false, wasSplit = false, lastSplitPart = false;
364  int splitID = 0;
365  std::vector<char> carryOverHeader;
366  while (dataSize > 0) {
367 
368  if (carryOver) { // check if there is carry-over header to write in the buffer
369  addHBFPage(); // start new CRU page, if needed, the completed superpage is flushed
370  if (writer->newRDHFunc) {
371  std::vector<char> newPageHeader;
372  writer->newRDHFunc(getLastRDH(), false, newPageHeader);
373  pushBack(newPageHeader.data(), newPageHeader.size());
374  }
375 
376  // for sure after the carryOver we have space on the CRU page, no need to check
377  LOG(debug) << "Adding carryOverHeader " << carryOverHeader.size()
378  << " bytes in IR " << ir << " to " << describe();
379  pushBack(carryOverHeader.data(), carryOverHeader.size());
380  carryOverHeader.clear();
381  carryOver = false;
382  }
383  int sizeLeftSupPage = writer->mSuperPageSize - buffer.size();
384  int sizeLeftCRUPage = RDHUtils::MAXCRUPage - (int(buffer.size()) - lastRDHoffset);
385  int sizeLeft = sizeLeftCRUPage < sizeLeftSupPage ? sizeLeftCRUPage : sizeLeftSupPage;
386  if (!sizeLeft) { // this page is just over, open a new one
387  addHBFPage(); // start new CRU page, if needed, the completed superpage is flushed
388  if (writer->newRDHFunc) {
389  std::vector<char> newPageHeader;
390  writer->newRDHFunc(getLastRDH(), false, newPageHeader);
391  pushBack(newPageHeader.data(), newPageHeader.size());
392  }
393  continue;
394  }
395 
396  if (dataSize <= sizeLeft) {
397  if (wasSplit && writer->mApplyCarryOverToLastPage) {
398  lastSplitPart = true;
399  carryOver = true;
400  }
401  } else {
402  carryOver = true;
403  wasSplit = true;
404  }
405 
406  if (!carryOver) { // add all remaining data
407  LOG(debug) << "Adding payload " << dataSize << " bytes in IR " << ir << " (carryover=" << carryOver << " ) to " << describe();
408  pushBack(ptr, dataSize);
409  dataSize = 0;
410  } else { // need to carryOver payload, determine 1st wsize bytes to write starting from ptr
411  if (sizeLeft > dataSize) {
412  sizeLeft = dataSize;
413  }
414  int sizeActual = sizeLeft;
415  std::vector<char> carryOverTrailer;
416  if (writer->carryOverFunc) {
417  sizeActual = writer->carryOverFunc(&rdhCopy, data, ptr, sizeLeft, splitID++, carryOverTrailer, carryOverHeader);
418  }
419  LOG(debug) << "Adding carry-over " << splitID - 1 << " fitted payload " << sizeActual << " bytes in IR " << ir << " to " << describe();
420  if (sizeActual < 0 || (!lastSplitPart && (sizeActual + carryOverTrailer.size() > sizeLeft))) {
421  throw std::runtime_error(std::string("wrong carry-over data size provided by carryOverMethod") + std::to_string(sizeActual));
422  }
423  // if there is carry-over trailer at the very last chunk, it must overwrite existing trailer
424  int trailerOffset = 0;
425  if (lastSplitPart) {
426  trailerOffset = carryOverTrailer.size();
427  if (sizeActual - trailerOffset < 0) {
428  throw std::runtime_error("trailer size of last split chunk cannot exceed actual size as it overwrites the existing trailer");
429  }
430  }
431  pushBack(ptr, sizeActual - trailerOffset); // write payload fitting to this page
432  dataSize -= sizeActual;
433  ptr += sizeActual;
434  LOG(debug) << "Adding carryOverTrailer " << carryOverTrailer.size() << " bytes in IR " << ir << " to " << describe();
435  pushBack(carryOverTrailer.data(), carryOverTrailer.size());
436  }
437  }
438 }

440 //___________________________________________________________________________________
441 void RawFileWriter::LinkData::addPreformattedCRUPage(const gsl::span<char> data)
442 {
443  // add preformatted CRU page w/o any attempt of splitting
444 
445  // we are guaranteed to have a page with RDH open
446  int sizeLeftSupPage = writer->mSuperPageSize - buffer.size();
447  if (sizeLeftSupPage < data.size()) { // we are not allowed to split this payload
448  flushSuperPage(true); // flush all but the last added RDH
449  }
450  if (data.size() > RDHUtils::MAXCRUPage - sizeof(RDHAny)) {
451  LOG(error) << "Preformatted payload size of " << data.size() << " bytes for " << describe()
452  << " exceeds max. size " << RDHUtils::MAXCRUPage - sizeof(RDHAny);
453  throw std::runtime_error("preformatted payload exceeds max size");
454  }
455  if (int(buffer.size()) - lastRDHoffset > sizeof(RDHAny)) { // we must start from empty page
456  addHBFPage(); // start new CRU page
457  }
458  pushBack(&data[0], data.size());
459 }

461 //___________________________________________________________________________________
462 void RawFileWriter::LinkData::addHBFPage(bool stop)
463 {
465 
466  if (lastRDHoffset < 0) {
467  return; // no page was open
468  }
469  // finalize last RDH
470  auto& lastRDH = *getLastRDH();
471  int psize = getCurrentPageSize(); // set the size for the previous header RDH
472  RDHUtils::setOffsetToNext(lastRDH, psize);
473  RDHUtils::setMemorySize(lastRDH, psize);
474 
475  rdhCopy = lastRDH;
476  bool add = true;
477  if (stop && !writer->mAddSeparateHBFStopPage) {
478  if (writer->isRDHStopUsed()) {
479  RDHUtils::setStop(lastRDH, stop);
480  }
481  add = false;
482  }
483  if (writer->mVerbosity > 2) {
484  RDHUtils::printRDH(lastRDH);
485  }
486  if (add) { // if we are in stopping HBF and new page is needed, add it
487  // check if the superpage reached the size where it hase to be flushed
488  int left = writer->mSuperPageSize - buffer.size();
489  if (left <= MarginToFlush) {
490  flushSuperPage();
491  }
492  RDHUtils::setPacketCounter(rdhCopy, packetCounter++);
493  RDHUtils::setPageCounter(rdhCopy, pageCnt++);
494  RDHUtils::setStop(rdhCopy, stop);
495  std::vector<char> userData;
496  int sz = sizeof(RDHAny);
497  if (stop && writer->newRDHFunc) { // detector may want to write something in closing page
498  writer->newRDHFunc(&rdhCopy, psize == sizeof(RDHAny), userData);
499  sz += userData.size();
500  }
501  RDHUtils::setOffsetToNext(rdhCopy, sz);
502  RDHUtils::setMemorySize(rdhCopy, sz);
503  lastRDHoffset = pushBack(rdhCopy); // entry of the new RDH
504  if (!userData.empty()) {
505  pushBack(userData.data(), userData.size());
506  }
507  }
508  if (stop) {
509  if (RDHUtils::getTriggerType(rdhCopy) & o2::trigger::TF) {
510  nTFWritten++;
511  }
512  if (writer->mVerbosity > 2 && add) {
513  RDHUtils::printRDH(rdhCopy);
514  }
515  lastRDHoffset = -1; // after closing, the previous RDH is not valid anymore
516  startOfRun = false; // signal that we are definitely not in the beginning of the run
517  }
518  //
519 }

521 //___________________________________________________________________________________
522 void RawFileWriter::LinkData::closeHBFPage()
523 {
524  // close the HBF page, if it is empty and detector has a special treatment of empty pages
525  // invoke detector callback method
526  if (lastRDHoffset < 0) {
527  return; // no page was open
528  }
529  bool emptyPage = getCurrentPageSize() == sizeof(RDHAny);
530  if (emptyPage && writer->emptyHBFFunc) { // we are closing an empty page, does detector want to add something?
531  std::vector<char> emtyHBFFiller; // working space for optional empty HBF filler
532  const auto rdh = getLastRDH();
533  writer->emptyHBFFunc(rdh, emtyHBFFiller);
534  if (!emtyHBFFiller.empty()) {
535  auto ir = RDHUtils::getTriggerIR(rdh);
536  LOG(debug) << "Adding empty HBF filler of size " << emtyHBFFiller.size() << " for " << describe();
537  addDataInternal(ir, emtyHBFFiller, false, 0, 0, false); // add filler w/o new check for empty HBF
538  }
539  }
540  addHBFPage(true);
541 }

543 //___________________________________________________________________________________
544 void RawFileWriter::LinkData::openHBFPage(const RDHAny& rdhn, uint32_t trigger)
545 {
547  bool forceNewPage = false;
548  // for RORC detectors the TF flag is absent, instead the 1st trigger after the start of TF will define the 1st be interpreted as 1st TF
549  if ((RDHUtils::getTriggerType(rdhn) & o2::trigger::TF) ||
550  (writer->isRORCDetector() && writer->mHBFUtils.getTF(updateIR - 1) < writer->mHBFUtils.getTF(RDHUtils::getTriggerIR(rdhn)))) {
551  if (writer->mVerbosity > -10) {
552  LOGF(info, "Starting new TF for link FEEId 0x%04x", RDHUtils::getFEEID(rdhn));
553  }
554  if (writer->mStartTFOnNewSPage && nTFWritten) { // don't flush if 1st TF
555  forceNewPage = true;
556  }
557  }
558  int left = writer->mSuperPageSize - buffer.size();
559  if (forceNewPage || left <= MarginToFlush) {
560  flushSuperPage();
561  }
562  pageCnt = 0;
563  lastRDHoffset = pushBack(rdhn);
564  auto& newrdh = *getLastRDH(); // dress new RDH with correct counters
565  RDHUtils::setPacketCounter(newrdh, packetCounter++);
566  RDHUtils::setPageCounter(newrdh, pageCnt++);
567  RDHUtils::setStop(newrdh, 0);
568  RDHUtils::setMemorySize(newrdh, sizeof(RDHAny));
569  RDHUtils::setOffsetToNext(newrdh, sizeof(RDHAny));
570 
571  if (startOfRun && writer->isReadOutModeSet()) {
572  auto trg = RDHUtils::getTriggerType(newrdh) | (writer->isContinuousReadout() ? o2::trigger::SOC : o2::trigger::SOT);
573  RDHUtils::setTriggerType(newrdh, trg);
574  }
575  rdhCopy = newrdh;
576 }

578 //___________________________________________________________________________________
579 void RawFileWriter::LinkData::flushSuperPage(bool keepLastPage)
580 {
581  // write link superpage data to file (if requested, only up to the last page)
582  size_t pgSize = (lastRDHoffset < 0 || !keepLastPage) ? buffer.size() : lastRDHoffset;
583  if (writer->mVerbosity) {
584  LOGF(info, "Flushing super page of %u bytes for %s", pgSize, describe());
585  }
586  writer->mFName2File.find(fileName)->second.write(buffer.data(), pgSize);
587  auto toMove = buffer.size() - pgSize;
588  if (toMove) { // is there something left in the buffer, move it to the beginning of the buffer
589  if (toMove > pgSize) {
590  memcpy(buffer.data(), &buffer[pgSize], toMove);
591  } else {
592  memmove(buffer.data(), &buffer[pgSize], toMove);
593  }
594  buffer.resize(toMove);
595  lastRDHoffset -= pgSize;
596  } else {
597  buffer.clear();
598  lastRDHoffset = -1;
599  }
600 }

603 void RawFileWriter::LinkData::close(const IR& irf)
604 {
605  // finalize link data
606  // close open HBF, write empty HBFs until the end of the TF corresponding to irfin and detach from the stream
607  if (writer->mFName2File.find(fileName) == writer->mFName2File.end()) {
608  return; // already closed
609  }
610  if (writer->isCRUDetector()) { // finalize last TF
611  int tf = writer->mHBFUtils.getTF(irf);
612  auto finalIR = writer->mHBFUtils.getIRTF(tf + 1) - 1; // last IR of the current TF
613  fillEmptyHBHs(finalIR, false);
614  }
615  closeHBFPage(); // close last HBF
616  flushSuperPage();
617 }

619 //___________________________________________________________________________________
620 void RawFileWriter::LinkData::fillEmptyHBHs(const IR& ir, bool dataAdded)
621 {
622  // fill HBFs from last processed one to requested ir
623 
624  if (writer->isCRUDetector()) {
625  std::vector<o2::InteractionRecord> irw;
626  if (!writer->mHBFUtils.fillHBIRvector(irw, updateIR, ir)) {
627  return;
628  }
629  for (const auto& irdummy : irw) {
630  if (writer->mDontFillEmptyHBF &&
631  writer->mHBFUtils.getTFandHBinTF(irdummy).second != 0 &&
632  (!dataAdded || irdummy.orbit < ir.orbit)) {
633  // even if requested, we skip empty HBF filling only if
634  // 1) we are not at the new TF start
635  // 2) method was called from addData and the current IR orbit is the one for which it was called (then it is not empty HB/trigger!)
636  continue;
637  }
638  if (writer->mVerbosity > 2) {
639  LOG(info) << "Adding HBF " << irdummy << " for " << describe();
640  }
641  closeHBFPage(); // close current HBF: add RDH with stop and update counters
642  RDHUtils::setTriggerType(rdhCopy, 0); // reset to avoid any detector specific flags in the dummy HBFs
643  writer->mHBFUtils.updateRDH<RDHAny>(rdhCopy, irdummy, writer->isCRUDetector()); // update HBF orbit/bc and trigger flags
644  openHBFPage(rdhCopy); // open new HBF
645  }
646  updateIR = irw.back() + o2::constants::lhc::LHCMaxBunches; // new HBF will be generated at >= this IR
647  } else { // RORC detector
648  if (writer->mVerbosity > 2) {
649  LOG(info) << "Adding HBF " << ir << " for " << describe();
650  }
651  closeHBFPage(); // close current HBF: add RDH with stop and update counters
652  RDHUtils::setTriggerType(rdhCopy, 0); // reset to avoid any detector specific flags in the dummy HBFs
653  writer->mHBFUtils.updateRDH<RDHAny>(rdhCopy, ir, false); // update HBF orbit/bc and trigger flags
654  openHBFPage(rdhCopy); // open new HBF
655  updateIR = ir + 1; // new Trigger in RORC detector will be generated at >= this IR
656  }
657 }
