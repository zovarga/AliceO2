33 void ReadRaw::readRawData(const LookUpTable& lut)
34 {
35  LOG(info) << "o2::fdd::ReadRaw::readRawData():Start.";
36  constexpr int CRUWordSize = 16;
37  o2::header::RAWDataHeader mRDH;
38  ChannelData chData; // output to store digits
39  EventHeader eventHeader; // raw data container
40  TCMdata mTCMdata; // raw data container
41  EventData eventData[NChPerMod]; // raw data container
42 
43  // Get input RAW file size
44  mRawFileIn.seekg(0, mRawFileIn.end);
45  long rawFileSize = mRawFileIn.tellg();
46  mRawFileIn.seekg(0);
47 
48  long posInFile = 0;
49  while (posInFile < rawFileSize - sizeof(mRDH)) {
50  // Read contents of the next RDH
51  mRawFileIn.seekg(posInFile);
52  mRawFileIn.read(reinterpret_cast<char*>(&mRDH), sizeof(mRDH));
53  int nwords = RDHUtils::getMemorySize(mRDH);
54  int offset = RDHUtils::getOffsetToNext(mRDH);
55  int link = RDHUtils::getLinkID(mRDH);
56  posInFile += offset; // posInFile is now prepared for the next iteration of the main loop
57 
58  int posPayload = 0;
59  if (nwords > sizeof(mRDH)) {
60  // RDHUtils::printRDH(mRDH);
61  // Read the payload following the RDH (only if there is anything to read)
62  posPayload = int(sizeof(mRDH));
63 
64  while (posPayload < nwords) {
65  mRawFileIn.read(reinterpret_cast<char*>(&eventHeader), sizeof(eventHeader));
66  posPayload += sizeof(eventHeader);
67  LOG(debug) << " Read internal EventHeader for link: " << link
68  << " nWords: " << (int)eventHeader.nGBTWords
69  << " orbit: " << int(eventHeader.orbit)
70  << " BC: " << int(eventHeader.bc)
71  << " posInFile: " << posInFile
72  << " posPayload: " << posPayload;
73  o2::InteractionRecord intrec{uint16_t(eventHeader.bc), uint32_t(eventHeader.orbit)};
74 
75  if (link == lut.getTcmLink()) { // is TCM payload
76  mRawFileIn.read(reinterpret_cast<char*>(&mTCMdata), sizeof(mTCMdata));
77  posPayload += sizeof(mTCMdata);
78  LOG(debug) << " Read TCM: posPayload: " << posPayload
79  << " posInFile: " << posInFile;
80  } else { // is PM payload
81  posPayload += CRUWordSize - o2::fdd::RawEventData::sPayloadSize; // padding is enabled
82  for (int i = 0; i < eventHeader.nGBTWords; ++i) {
83  mRawFileIn.read(reinterpret_cast<char*>(&eventData[2 * i]), o2::fdd::RawEventData::sPayloadSizeFirstWord);
84  posPayload += o2::fdd::RawEventData::sPayloadSizeFirstWord;
85  chData = {static_cast<uint8_t>(lut.getChannel(link, int(eventData[2 * i].channelID))),
86  int(eventData[2 * i].time),
87  int(eventData[2 * i].charge), 0};
88  mDigitAccum[intrec].emplace_back(chData);
89  LOG(debug) << " Read 1st half-word: (PMchannel, globalChannel, Q, T, posPayload) = "
90  << std::setw(3) << int(eventData[2 * i].channelID)
91  << std::setw(4) << lut.getChannel(link, int(eventData[2 * i].channelID))
92  << std::setw(5) << int(eventData[2 * i].charge)
93  << std::setw(5) << int(eventData[2 * i].time)
94  << std::setw(5) << posPayload;
95 
96  Short_t channelIdFirstHalfWord = chData.mPMNumber;
97 
98  mRawFileIn.read(reinterpret_cast<char*>(&eventData[2 * i + 1]), o2::fdd::RawEventData::sPayloadSizeSecondWord);
99  posPayload += o2::fdd::RawEventData::sPayloadSizeSecondWord;
100  chData = {static_cast<uint8_t>(lut.getChannel(link, int(eventData[2 * i + 1].channelID))),
101  int(eventData[2 * i + 1].time),
102  int(eventData[2 * i + 1].charge), 0};
103  if (chData.mPMNumber <= channelIdFirstHalfWord) {
104  // Don't save the second half-word if it is only filled with zeroes (empty-data)
105  // TODO: Verify if it works correctly with real data from readout
106  continue;
107  }
108  mDigitAccum[intrec].emplace_back(chData);
109  LOG(debug) << " Read 2nd half-word: (PMchannel, globalChannel, Q, T, posPayload) = "
110  << std::setw(3) << int(eventData[2 * i + 1].channelID)
111  << std::setw(4) << lut.getChannel(link, int(eventData[2 * i + 1].channelID))
112  << std::setw(5) << int(eventData[2 * i + 1].charge)
113  << std::setw(5) << int(eventData[2 * i + 1].time)
114  << std::setw(5) << posPayload;
115  }
116  }
117  }
118  }
119  }
120  close();
121  LOG(info) << "o2::fdd::ReadRaw::readRawData():Finished.";
122 }

131 void ReadRaw::writeDigits(const std::string& outputDigitsFilePath)
132 {
133  TFile* outFile = new TFile(outputDigitsFilePath.data(), "RECREATE");
134  if (!outFile || outFile->IsZombie()) {
135  LOG(error) << "Failed to open " << outputDigitsFilePath << " output file";
136  } else {
137  LOG(info) << "o2::fdd::ReadRaw::writeDigits(): Opened output file: " << outputDigitsFilePath;
138  }
139  TTree* outTree = new TTree("o2sim", "o2sim");
140  std::vector<ChannelData> chDataVecTree;
141  std::vector<Digit> chBcVecTree;
142 
143  for (auto& digit : mDigitAccum) {
144  LOG(debug) << " IR (" << digit.first << ") (i, PMT, Q, T):";
145  for (uint16_t i = 0; i < digit.second.size(); i++) {
146  ChannelData* chd = &(digit.second.at(i));
147  LOG(debug) << " " << std::setw(3) << i
148  << std::setw(4) << chd->mPMNumber
149  << std::setw(5) << chd->mChargeADC
150  << std::setw(5) << chd->mTime;
151  }
152 
153  size_t nStored = 0;
154  size_t first = chDataVecTree.size();
155  for (auto& sec : digit.second) {
156  chDataVecTree.emplace_back(sec.mPMNumber, sec.mTime, sec.mChargeADC, sec.mFEEBits);
157  nStored++;
158  }
159  chBcVecTree.emplace_back(first, nStored, digit.first, Triggers());
160  }
161 
162  outTree->Branch("FDDDigit", &chBcVecTree);
163  outTree->Branch("FDDDigitCh", &chDataVecTree);
164  outTree->Fill();
165 
166  outFile->cd();
167  outTree->Write();
168  outFile->Close();
169  LOG(info) << "o2::fdd::ReadRaw::writeDigits(): Finished converting " << chBcVecTree.size() << " events.";
170 }
