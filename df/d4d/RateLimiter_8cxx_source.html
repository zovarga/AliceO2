21 void RateLimiter::check(ProcessingContext& ctx, int maxInFlight, size_t minSHM)
22 {
23  if (!maxInFlight && !minSHM) {
24  return;
25  }
26  auto device = ctx.services().get<RawDeviceService>().device();
27  if (maxInFlight && device->fChannels.count("metric-feedback")) {
28  int waitMessage = 0;
29  int recvTimeot = 0;
30  while ((mSentTimeframes - mConsumedTimeframes) >= maxInFlight) {
31  if (recvTimeot == -1 && waitMessage == 0) {
32  LOG(alarm) << "Maximum number of TF in flight reached (" << maxInFlight << ": published " << mSentTimeframes << " - finished " << mConsumedTimeframes << "), waiting";
33  waitMessage = 1;
34  }
35  auto msg = device->NewMessageFor("metric-feedback", 0, 0);
36 
37  auto count = device->Receive(msg, "metric-feedback", 0, recvTimeot);
38  if (count <= 0) {
39  recvTimeot = -1;
40  continue;
41  }
42  assert(msg->GetSize() == 8);
43  mConsumedTimeframes = *(int64_t*)msg->GetData();
44  }
45  if (waitMessage) {
46  LOG(important) << (mSentTimeframes - mConsumedTimeframes) << " / " << maxInFlight << " TF in flight, continuing to publish";
47  }
48  }
49  if (minSHM) {
50  int waitMessage = 0;
51  auto& runningWorkflow = ctx.services().get<RunningWorkflowInfo const>();
52  while (true) {
53  uint64_t freeSHM = fair::mq::shmem::Monitor::GetFreeMemory(fair::mq::shmem::SessionId{device->fConfig->GetProperty<std::string>("session")}, runningWorkflow.shmSegmentId);
54  if (freeSHM > minSHM) {
55  if (waitMessage) {
56  LOG(important) << "Sufficient SHM memory free (" << freeSHM << " >= " << minSHM << "), continuing to publish";
57  }
58  break;
59  }
60  if (waitMessage == 0) {
61  LOG(alarm) << "Free SHM memory too low: " << freeSHM << " < " << minSHM << ", waiting";
62  waitMessage = 1;
63  }
64  }
65  }

67 }
