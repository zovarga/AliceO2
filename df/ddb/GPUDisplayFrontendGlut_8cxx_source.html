168 void GPUDisplayFrontendGlut::mouseFunc(int button, int state, int x, int y)
169 {
170  if (button == 3) {
171  me->mMouseWheel += 100;
172  } else if (button == 4) {
173  me->mMouseWheel -= 100;
174  } else if (state == GLUT_DOWN) {
175  if (button == GLUT_LEFT_BUTTON) {
176  me->mMouseDn = true;
177  } else if (button == GLUT_RIGHT_BUTTON) {
178  me->mMouseDnR = true;
179  }
180  me->mMouseDnX = x;
181  me->mMouseDnY = y;

184  me->mMouseDn = false;

186  me->mMouseDnR = false;

188  }

199 int GPUDisplayFrontendGlut::FrontendMain()
200 {
201  me = this;
202  int nopts = 2;
203  char opt1[] = "progname";
204  char opt2[] = "-direct";
205  char* opts[] = {opt1, opt2};
206  glutInit(&nopts, opts);
207  glutInitContextVersion(GL_MIN_VERSION_MAJOR, GL_MIN_VERSION_MINOR);
208  glutInitContextProfile(mBackend->CoreProfile() ? GLUT_CORE_PROFILE : GLUT_COMPATIBILITY_PROFILE);
209  glutInitDisplayMode(GLUT_DOUBLE | GLUT_RGB);
210  glutInitWindowSize(INIT_WIDTH, INIT_HEIGHT);
211  glutCreateWindow(GL_WINDOW_NAME);
212  glutSetOption(GLUT_ACTION_ON_WINDOW_CLOSE, GLUT_ACTION_GLUTMAINLOOP_RETURNS);
213 
214  if (mBackend->ExtInit()) {
215  fprintf(stderr, "Error initializing GL extension wrapper\n");
216  return (-1);
217  }
218  if (InitDisplay()) {
219  fprintf(stderr, "Error in OpenGL initialization\n");
220  return (1);
221  }
222 
223  glutDisplayFunc(displayFunc);
224  glutIdleFunc(glutLoopFunc);
225  glutReshapeFunc(ReSizeGLSceneWrapper);
226  glutKeyboardFunc(keyboardDownFunc);
227  glutKeyboardUpFunc(keyboardUpFunc);
228  glutSpecialFunc(specialDownFunc);
229  glutSpecialUpFunc(specialUpFunc);
230  glutMouseFunc(mouseFunc);
231  glutMotionFunc(mouseMoveFunc);
232  glutPassiveMotionFunc(mouseMoveFunc);
233  glutMouseWheelFunc(mMouseWheelFunc);
234  ToggleMaximized(true);
235 
236  pthread_mutex_lock(&mSemLockExit);
237  mGlutRunning = true;
238  pthread_mutex_unlock(&mSemLockExit);
239  glutMainLoop();
240  mDisplayControl = 2;
241  pthread_mutex_lock(&mSemLockExit);
242  mGlutRunning = false;
243  pthread_mutex_unlock(&mSemLockExit);
244 
245  return 0;
246 }
