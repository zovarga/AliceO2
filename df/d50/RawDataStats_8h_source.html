34 enum ParsingErrors { TRDParsingNoError,
35  TRDParsingUnrecognisedVersion,
36  TRDParsingBadDigt,
37  TRDParsingBadTracklet,
38  TRDParsingDigitEndMarkerWrongState, // read a end marker but we were expecting something else due to
39  TRDParsingDigitMCMHeaderSanityCheckFailure, //essentially we did not see an MCM header see RawData.h for requirement
40  TRDParsingDigitROBDecreasing, // sequential headers must have the same or increasing rob number
41  TRDParsingDigitMCMNotIncreasing, // sequential headers must have increasing mcm number
42  TRDParsingDigitADCMaskMismatch, // mask adc count does not match # of 1s in bitpattern
43  TRDParsingDigitADCMaskAdvanceToEnd, // in advancing to adcmask we have reached the end of the buffer
44  TRDParsingDigitMCMHeaderBypassButStateMCMHeader, // we are reading mcmadc data but the state is mcmheader
45  TRDParsingDigitEndMarkerStateButReadingMCMADCData, // read the endmarker while expecting to read the mcmadcdata
46  TRDParsingDigitADCChannel21, // ADCMask is zero but we are still on a digit.
47  TRDParsingDigitADCChannelGT22, // error allocating digit, so digit channel has error value
48  TRDParsingDigitGT10ADCs, // more than 10 adc data words seen
49  TRDParsingDigitSanityCheck, // adc failed sanity check see RawData.cxx for faiulre reasons
50  TRDParsingDigitExcessTimeBins, // ADC has more than 30 timebins (10 adc words)
51  TRDParsingDigitParsingExitInWrongState, // exiting parsing in the wrong state ... got to the end of the buffer in wrong state.
52  TRDParsingDigitStackMismatch, // mismatch between rdh and hcheader stack calculation/value
53  TRDParsingDigitLayerMismatch, // mismatch between rdh and hcheader stack calculation/value
54  TRDParsingDigitSectorMismatch, // mismatch between rdh and hcheader stack calculation/value
55  TRDParsingTrackletCRUPaddingWhileParsingTracklets, // reading a padding word while expecting tracklet data
56  TRDParsingTrackletBit11NotSetInTrackletHCHeader, // bit 11 not set in hc header for tracklets.
57  TRDParsingTrackletHCHeaderSanityCheckFailure, // HCHeader sanity check failure, see RawData.cxx for reasons.
58  TRDParsingTrackletMCMHeaderSanityCheckFailure, // MCMHeader sanity check failure, see RawData.cxx for reasons.
59  TRDParsingTrackletMCMHeaderButParsingMCMData, // state is still MCMHeader but we are parsing MCMData
60  TRDParsingTrackletStateMCMHeaderButParsingMCMData,
61  TRDParsingTrackletTrackletCountGTThatDeclaredInMCMHeader, //mcmheader tracklet count does not match that in we have parsed.
62  TRDParsingTrackletInvalidTrackletCount, // invalid tracklet count in header vs data
63  TRDParsingTrackletPadRowIncreaseError, // subsequent padrow can not be less than previous one.
64  TRDParsingTrackletColIncreaseError, // subsequent col can not be less than previous one
65  TRDParsingTrackletNoTrackletEndMarker, // got to the end of the buffer with out finding a tracklet end marker.
66  TRDParsingTrackletExitingNoTrackletEndMarker, // got to the end of the buffer exiting tracklet parsing with no tracklet end marker
67  TRDParsingDigitHeaderCountGT3, // digital half chamber header had more than 3 additional words expected by header. most likely corruption above somewhere.
68  TRDParsingDigitHeaderWrong1, // expected header word1 but wrong ending marker
69  TRDParsingDigitHeaderWrong2, // expected header word2 but wrong ending marker
70  TRDParsingDigitHeaderWrong3, // expected header word3 but wrong ending marker
71  TRDParsingDigitHeaderWrong4, // expected header word but have no idea what we are looking at default of switch statement
72  TRDParsingDigitDataStillOnLink, // got to the end of digit parsing and there is still data on link, normally not advancing far enough when dumping data.
73  TRDParsingTrackletIgnoringDataTillEndMarker, // for some reason we are bouncing to the end word by word, this counts those words
74  TRDLastParsingError // here as place holder for trivial sizing of structures.
75 };

111 class TRDDataCountersPerTimeFrame
112 { //thisis on a per event basis

116  std::array<uint16_t, o2::trd::constants::NSECTOR * 60> mLinkWords{}; //units of 256bits, read from the cru half chamber header

118  std::array<uint16_t, o2::trd::constants::NSECTOR * 60> mLinkWordsRejected{}; // units of 32 bits the data dumped due to some or other error
119  // std::array<uint16_t, o2::trd::constants::MAXMCMCOUNT> mLinkMCMsWithData{}; // and its corresponding volume of data.
120  std::array<uint16_t, TRDLastParsingError> mParsingErrors{}; // errors in parsing, indexed by enum above of ParsingErrors
121  std::array<uint32_t, o2::trd::constants::NSECTOR * 60 * TRDLastParsingError> mParsingErrorsByLink{}; // errors in parsing, indexed by enum above of ParsingErrors
122  // std::array<uint16_t, constants::MAXMCMCOUNT> mMCMDigitsFound{}; can be reprocessed in qc rather, save work and space,
123  // std::array<uint16_t, constants::MAXMCMCOUNT> mMCMTrackletsFound{}; sim as above
124  uint16_t mDigitsPerEvent; // average digits found per event
125  uint16_t mTrackletsPerEvent; // average tracklets found per event
126  // std::map<uint16_t, uint16_t> mRdhSize; // sizes of rdhs read i think tf sizes are plotted some where so not needed anymore i think.
127  double mTimeTaken; // time taken to process the entire timeframe [ms].
128  double mTimeTakenForDigits; // time take to process tracklet data blocks [ms].
129  double mTimeTakenForTracklets; // time take to process digit data blocks [ms].
130  uint32_t mDigitsFound; // digit found in the time frame.
131  uint32_t mTrackletsFound; // tracklets found in the time frame.
132  std::array<uint64_t, 256> mDataFormatRead{}; // We just keep the major version number
133  void clear()
134  {
135  mLinkNoData.fill(0);
136  mLinkWords.fill(0);
137  mLinkWordsRead.fill(0);
138  mLinkWordsRejected.fill(0);
139  // mLinkMCMsWithData.fill(0);
140  mParsingErrors.fill(0);
141  mParsingErrorsByLink.fill(0);
142  mDigitsPerEvent = 0;
143  mTrackletsPerEvent = 0;
144  mTimeTaken = 0;
145  mTimeTakenForDigits = 0;
146  mTimeTakenForTracklets = 0;
147  mDigitsFound = 0;
148  mTrackletsFound = 0; //tracklets found in timeframe.
149  mDataFormatRead.fill(0);
150  };
151  ClassDefNV(TRDDataCountersPerTimeFrame, 1); // primarily for serialisation so we can send this as a message in o2
152 };
