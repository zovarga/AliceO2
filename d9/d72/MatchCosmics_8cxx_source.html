42 //________________________________________________________
43 void MatchCosmics::process(const o2::globaltracking::RecoContainer& data)
44 {
45  updateTimeDependentParams();
46  mRecords.clear();
47  mWinners.clear();
48  mCosmicTracks.clear();
49  mCosmicTracksLbl.clear();
50 
51  createSeeds(data);
52  int ntr = mSeeds.size();
53 
54  // propagate to DCA to origin
55  const o2::math_utils::Point3D<float> v{0., 0., 0};
56  for (int i = 0; i < ntr; i++) {
57  auto& trc = mSeeds[i];
58  if (!o2::base::Propagator::Instance()->propagateToDCABxByBz(v, trc, mMatchParams->maxStep, mMatchParams->matCorr)) {
59  trc.matchID = Reject; // reject track
60  }
61  }
62 
63  // sort in time bracket lower edge
64  std::vector<int> sortID(ntr);
65  std::iota(sortID.begin(), sortID.end(), 0);
66  std::sort(sortID.begin(), sortID.end(), [this](int a, int b) { return mSeeds[a].tBracket.getMin() < mSeeds[b].tBracket.getMin(); });
67 
68  for (int i = 0; i < ntr; i++) {
69  for (int j = i + 1; j < ntr; j++) {
70  if (checkPair(sortID[i], sortID[j]) == RejTime) {
71  break;
72  }
73  }
74  }
75 
76  selectWinners();
77  refitWinners(data);
78 
79  mTFCount++;
80 }

83 void MatchCosmics::refitWinners(const o2::globaltracking::RecoContainer& data)
84 {
85  LOG(info) << "Refitting " << mWinners.size() << " winner matches";
86  int count = 0;
87  auto tpcTBinMUSInv = 1. / mTPCTBinMUS;
88  if (!mTPCTransform) { // eventually, should be updated at every TF?
89  mTPCTransform = o2::tpc::TPCFastTransformHelperO2::instance()->create(0);
90  }
91  const auto& tpcClusRefs = data.getTPCTracksClusterRefs();
92  const auto& tpcClusShMap = data.clusterShMapTPC;
93  std::unique_ptr<o2::gpu::GPUO2InterfaceRefit> tpcRefitter;
94  if (data.inputsTPCclusters) {
95  tpcRefitter = std::make_unique<o2::gpu::GPUO2InterfaceRefit>(&data.inputsTPCclusters->clusterIndex,
96  mTPCTransform.get(), mBz,
97  tpcClusRefs.data(), tpcClusShMap.data(),
98  nullptr, o2::base::Propagator::Instance());
99  }
100 
101  const auto& itsClusters = prepareITSClusters(data);
102  // RS FIXME: this is probably a temporary solution, since ITS tracking over boundaries will likely change the TrackITS format
103  std::vector<int> itsTracksROF;
104 
105  const auto& itsTracksROFRec = data.getITSTracksROFRecords();
106  itsTracksROF.resize(data.getITSTracks().size());
107  for (unsigned irf = 0, cnt = 0; irf < itsTracksROFRec.size(); irf++) {
108  int ntr = itsTracksROFRec[irf].getNEntries();
109  for (int itr = 0; itr < ntr; itr++) {
110  itsTracksROF[cnt++] = irf;
111  }
112  }
113 
114  auto refitITSTrack = [this, &data, &itsTracksROF, &itsClusters](o2::track::TrackParCov& trFit, GTrackID gidx, float& chi2, bool inward = false) {
115  const auto& itsTrOrig = data.getITSTrack(gidx);
116  int nclRefit = 0, ncl = itsTrOrig.getNumberOfClusters(), rof = itsTracksROF[gidx.getIndex()];
117  const auto& itsTrackClusRefs = data.getITSTracksClusterRefs();
118  int clEntry = itsTrOrig.getFirstClusterEntry();
119  const auto propagator = o2::base::Propagator::Instance();
120  const auto geomITS = o2::its::GeometryTGeo::Instance();
121  int from = ncl - 1, to = -1, step = -1;
122  if (inward) {
123  from = 0;
124  to = ncl;
125  step = 1;
126  }
127  for (int icl = from; icl != to; icl += step) { // ITS clusters are referred in layer decreasing order
128  const auto& clus = itsClusters[itsTrackClusRefs[clEntry + icl]];
129  float alpha = geomITS->getSensorRefAlpha(clus.getSensorID()), x = clus.getX();
130  if (!trFit.rotate(alpha) || !propagator->propagateToX(trFit, x, propagator->getNominalBz(), this->mMatchParams->maxSnp, this->mMatchParams->maxStep, this->mMatchParams->matCorr)) {
131  break;
132  }
133  chi2 += trFit.getPredictedChi2(clus);
134  if (!trFit.update(clus)) {
135  break;
136  }
137  nclRefit++;
138  }
139  return nclRefit == ncl ? ncl : -1;
140  };
141 
142  for (auto winRID : mWinners) {
143  const auto& rec = mRecords[winRID];
144  int poolEntryID[2] = {rec.id0, rec.id1};
145  const o2::track::TrackParCov outerLegs[2] = {data.getTrackParamOut(mSeeds[rec.id0].origID), data.getTrackParamOut(mSeeds[rec.id1].origID)};
146  auto tOverlap = mSeeds[rec.id0].tBracket.getOverlap(mSeeds[rec.id1].tBracket);
147  float t0 = tOverlap.mean(), dt = tOverlap.delta() * 0.5;
148  auto pnt0 = outerLegs[0].getXYZGlo(), pnt1 = outerLegs[1].getXYZGlo();
149  int btm = 0, top = 1;
150  // we fit topward from bottom
151  if (pnt0.Y() > pnt1.Y()) {
152  btm = 1;
153  top = 0;
154  }
155  LOG(debug) << "Winner " << count++ << " Record " << winRID << " Partners:"
156  << " B: " << mSeeds[poolEntryID[btm]].origID << "/" << mSeeds[poolEntryID[btm]].origID.getSourceName()
157  << " U: " << mSeeds[poolEntryID[top]].origID << "/" << mSeeds[poolEntryID[top]].origID.getSourceName()
158  << " | T:" << tOverlap.asString();
159 
160  float chi2 = 0;
161  int nclTot = 0;
162 
163  // Start from bottom leg inward refit
164  o2::track::TrackParCov trCosm(mSeeds[poolEntryID[btm]]); // copy of the btm track
165  // The bottom leg needs refit only if it is an unconstrained TPC track, otherwise it is already refitted as inner param
166  if (mSeeds[poolEntryID[btm]].origID.getSource() == GTrackID::TPC) {
167  const auto& tpcTrOrig = data.getTPCTrack(mSeeds[poolEntryID[btm]].origID);
168  trCosm = outerLegs[btm];
169  int retVal = tpcRefitter->RefitTrackAsTrackParCov(trCosm, tpcTrOrig.getClusterRef(), t0 * tpcTBinMUSInv, &chi2, false, true); // inward refit, reset
170  if (retVal < 0) { // refit failed
171  LOG(debug) << "Inward refit of btm TPC track failed.";
172  continue;
173  }
174  nclTot += retVal;
175  LOG(debug) << "chi2 after btm TPC refit with " << retVal << " clusters : " << chi2 << " orig.chi2 was " << tpcTrOrig.getChi2();
176  } else { // just collect NClusters and chi2
177  auto gidxListBtm = data.getSingleDetectorRefs(mSeeds[poolEntryID[btm]].origID);
178  if (gidxListBtm[GTrackID::TPC].isIndexSet()) {
179  const auto& tpcTrOrig = data.getTPCTrack(gidxListBtm[GTrackID::TPC]);
180  nclTot += tpcTrOrig.getNClusters();
181  chi2 += tpcTrOrig.getChi2();
182  }
183  if (gidxListBtm[GTrackID::ITS].isIndexSet()) {
184  const auto& itsTrOrig = data.getITSTrack(gidxListBtm[GTrackID::ITS]);
185  nclTot += itsTrOrig.getNClusters();
186  chi2 += itsTrOrig.getChi2();
187  }
188  }
189  trCosm.invert();
190  if (!trCosm.rotate(mSeeds[poolEntryID[top]].getAlpha()) ||
191  !o2::base::Propagator::Instance()->PropagateToXBxByBz(trCosm, mSeeds[poolEntryID[top]].getX(), mMatchParams->maxSnp, mMatchParams->maxStep, mMatchParams->matCorr)) {
192  LOG(debug) << "Rotation/propagation of btm-track to top-track frame failed.";
193  continue;
194  }
195  // save bottom parameter at merging point
196  auto trCosmBtm = trCosm;
197  int nclBtm = nclTot;
198 
199  // Continue with top leg outward refit
200  auto gidxListTop = data.getSingleDetectorRefs(mSeeds[poolEntryID[top]].origID);
201 
202  // is there ITS sub-track?
203  if (gidxListTop[GTrackID::ITS].isIndexSet()) {
204  auto nclfit = refitITSTrack(trCosm, gidxListTop[GTrackID::ITS], chi2, false);
205  if (nclfit < 0) {
206  continue;
207  }
208  LOG(debug) << "chi2 after top ITS refit with " << nclfit << " clusters : " << chi2 << " orig.chi2 was " << data.getITSTrack(gidxListTop[GTrackID::ITS]).getChi2();
209  nclTot += nclfit;
210  } // ITS refit
211  //
212  if (gidxListTop[GTrackID::TPC].isIndexSet()) { // outward refit in TPC
213  // go to TPC boundary, if needed
214  if (trCosm.getX() * trCosm.getX() + trCosm.getY() * trCosm.getY() <= o2::constants::geom::XTPCInnerRef * o2::constants::geom::XTPCInnerRef) {
215  float xtogo = 0;
216  if (!trCosm.getXatLabR(o2::constants::geom::XTPCInnerRef, xtogo, mBz, o2::track::DirOutward) ||
217  !o2::base::Propagator::Instance()->PropagateToXBxByBz(trCosm, xtogo, mMatchParams->maxSnp, mMatchParams->maxStep, mMatchParams->matCorr)) {
218  LOG(debug) << "Propagation to inner TPC boundary X=" << xtogo << " failed";
219  continue;
220  }
221  }
222  const auto& tpcTrOrig = data.getTPCTrack(gidxListTop[GTrackID::TPC]);
223  int retVal = tpcRefitter->RefitTrackAsTrackParCov(trCosm, tpcTrOrig.getClusterRef(), t0 * tpcTBinMUSInv, &chi2, true, false); // outward refit, no reset
224  if (retVal < 0) { // refit failed
225  LOG(debug) << "Outward refit of top TPC track failed.";
226  continue;
227  } // outward refit in TPC
228  LOG(debug) << "chi2 after top TPC refit with " << retVal << " clusters : " << chi2 << " orig.chi2 was " << tpcTrOrig.getChi2();
229  nclTot += retVal;
230  }
231 
232  // inward refit of top leg for evaluation in DCA
233  float chi2Dummy = 0;
234  auto trCosmTop = outerLegs[top];
235  if (gidxListTop[GTrackID::TPC].isIndexSet()) { // inward refit in TPC
236  const auto& tpcTrOrig = data.getTPCTrack(gidxListTop[GTrackID::TPC]);
237  int retVal = tpcRefitter->RefitTrackAsTrackParCov(trCosmTop, tpcTrOrig.getClusterRef(), t0 * tpcTBinMUSInv, &chi2Dummy, false, true); // inward refit, reset
238  if (retVal < 0) { // refit failed
239  LOG(debug) << "Outward refit of top TPC track failed.";
240  continue;
241  } // inward refit in TPC
242  }
243  // is there ITS sub-track ?
244  if (gidxListTop[GTrackID::ITS].isIndexSet()) {
245  auto nclfit = refitITSTrack(trCosmTop, gidxListTop[GTrackID::ITS], chi2Dummy, true);
246  if (nclfit < 0) {
247  continue;
248  }
249  nclTot += nclfit;
250  } // ITS refit
251  // propagate to bottom param
252  if (!trCosmTop.rotate(trCosmBtm.getAlpha()) ||
253  !o2::base::Propagator::Instance()->PropagateToXBxByBz(trCosmTop, trCosmBtm.getX(), mMatchParams->maxSnp, mMatchParams->maxStep, mMatchParams->matCorr)) {
254  LOG(debug) << "Rotation/propagation of top-track to bottom-track frame failed.";
255  continue;
256  }
257  // calculate weighted average of 2 legs and chi2
258  o2::track::TrackParCov::MatrixDSym5 cov5;
259  float chi2Match = trCosmBtm.getPredictedChi2(trCosmTop, cov5);
260  if (!trCosmBtm.update(trCosmTop, cov5)) {
261  LOG(debug) << "Top/Bottom update failed";
262  continue;
263  }
264  // create final track
265  mCosmicTracks.emplace_back(mSeeds[poolEntryID[btm]].origID, mSeeds[poolEntryID[top]].origID, trCosmBtm, trCosmTop, chi2, chi2Match, nclTot, t0, dt);
266  if (mUseMC) {
267  o2::MCCompLabel lbl[2] = {data.getTrackMCLabel(mSeeds[poolEntryID[btm]].origID), data.getTrackMCLabel(mSeeds[poolEntryID[top]].origID)};
268  auto& tlb = mCosmicTracksLbl.emplace_back((nclBtm > nclTot - nclBtm ? lbl[0] : lbl[1]));
269  tlb.setFakeFlag(lbl[0] != lbl[1]);
270  }
271  }
272  LOG(info) << "Validated " << mCosmicTracks.size() << " top-bottom tracks in TF# " << mTFCount;
273 }

276 void MatchCosmics::selectWinners()
277 {
278  // select mutually best matches
279  int ntr = mSeeds.size(), iter = 0, nValidated = 0;
280  mWinners.reserve(mRecords.size() / 2); // there are 2 records per match candidate
281  do {
282  nValidated = 0;
283  int nRemaining = 0;
284  for (int i = 0; i < ntr; i++) {
285  if (mSeeds[i].matchID < 0 || mRecords[mSeeds[i].matchID].next == Validated) { // either have no match or already validated
286  continue;
287  }
288  nRemaining++;
289  if (validateMatch(i)) {
290  mWinners.push_back(mSeeds[i].matchID);
291  nValidated++;
292  continue;
293  }
294  }
295  LOGF(info, "iter %d Validated %d of %d remaining matches", iter, nValidated, nRemaining);
296  iter++;
297  } while (nValidated);
298 }

301 bool MatchCosmics::validateMatch(int partner0)
302 {
303  // make sure that the best partner of seed_i has also seed_i as a best partner
304  auto& matchRec = mRecords[mSeeds[partner0].matchID];
305  auto partner1 = matchRec.id1;
306  auto& patnerRec = mRecords[mSeeds[partner1].matchID];
307  if (patnerRec.next == Validated) { // partner1 was already validated with other partner0
308  return false;
309  }
310  if (patnerRec.id1 == partner0) { // mutually best
311  // unlink winner partner0 from all other mathes
312  auto next0 = matchRec.next;
313  while (next0 > MinusOne) {
314  auto& nextRec = mRecords[next0];
315  suppressMatch(partner0, nextRec.id1);
316  next0 = nextRec.next;
317  }
318  matchRec.next = Validated;
319 
320  // unlink winner partner1 from all other matches
321  auto next1 = patnerRec.next;
322  while (next1 > MinusOne) {
323  auto& nextRec = mRecords[next1];
324  suppressMatch(partner1, nextRec.id1);
325  next1 = nextRec.next;
326  }
327  patnerRec.next = Validated;
328  return true;
329  }
330  return false;
331 }

334 void MatchCosmics::suppressMatch(int partner0, int partner1)
335 {
336  // suppress reference to partner0 from partner1 match record
337  if (mSeeds[partner1].matchID < 0 || mRecords[mSeeds[partner1].matchID].next == Validated) {
338  LOG(warning) << "Attempt to remove null or validated partner match " << mSeeds[partner1].matchID;
339  return;
340  }
341  int topID = MinusOne, next = mSeeds[partner1].matchID;
342  while (next > MinusOne) {
343  auto& matchRec = mRecords[next];
344  if (matchRec.id1 == partner0) {
345  if (topID < 0) { // best match
346  mSeeds[partner1].matchID = matchRec.next; // exclude best match link
347  } else { // not the 1st link in the chain
348  mRecords[topID].next = matchRec.next;
349  }
350  return;
351  }
352  topID = next;
353  next = matchRec.next;
354  }
355 }

358 MatchCosmics::RejFlag MatchCosmics::checkPair(int i, int j)
359 {
360  // if validated with given chi2, register match
361  RejFlag rej = RejOther;
362  auto& seed0 = mSeeds[i];
363  auto& seed1 = mSeeds[j];
364  if (seed0.matchID == Reject) {
365  return rej;
366  }
367  if (seed1.matchID == Reject) {
368  return rej;
369  }
370 
371  LOG(debug) << "Seed " << i << " [" << seed0.tBracket.getMin() << " : " << seed0.tBracket.getMax() << "] | "
372  << "Seed " << j << " [" << seed1.tBracket.getMin() << " : " << seed1.tBracket.getMax() << "] | ";
373  LOG(debug) << seed0.origID << " | " << seed0.o2::track::TrackPar::asString();
374  LOG(debug) << seed1.origID << " | " << seed1.o2::track::TrackPar::asString();
375 
376  if (seed1.tBracket > seed0.tBracket) {
377  return (rej = RejTime); // since the brackets are sorted in tmin, all following tbj will also exceed tbi
378  }
379  float chi2 = 1.e9f;
380 
381  // check
382  // 1) crude check on tgl and q/pt (if B!=0). Note: back-to-back tracks will have mutually params (see TrackPar::invertParam)
383  while (1) {
384  auto dTgl = seed0.getTgl() + seed1.getTgl();
385  if (dTgl * dTgl > (mMatchParams->systSigma2[o2::track::kTgl] + seed0.getSigmaTgl2() + seed1.getSigmaTgl2()) * mMatchParams->crudeNSigma2Cut[o2::track::kTgl]) {
386  rej = RejTgl;
387  break;
388  }
389  if (mFieldON) {
390  auto dQ2Pt = seed0.getQ2Pt() + seed1.getQ2Pt();
391  if (dQ2Pt * dQ2Pt > (mMatchParams->systSigma2[o2::track::kQ2Pt] + seed0.getSigma1Pt2() + seed1.getSigma1Pt2()) * mMatchParams->crudeNSigma2Cut[o2::track::kQ2Pt]) {
392  rej = RejQ2Pt;
393  break;
394  }
395  }
396  o2::track::TrackParCov seed1Inv = seed1;
397  seed1Inv.invert();
398  for (int i = 0; i < o2::track::kNParams; i++) { // add systematic error
399  seed1Inv.updateCov(mMatchParams->systSigma2[i], o2::track::DiagMap[i]);
400  }
401 
402  if (!seed1Inv.rotate(seed0.getAlpha()) ||
403  !o2::base::Propagator::Instance()->PropagateToXBxByBz(seed1Inv, seed0.getX(), mMatchParams->maxSnp, mMatchParams->maxStep, mMatchParams->matCorr)) {
404  rej = RejProp;
405  break;
406  }
407  auto dSnp = seed0.getSnp() - seed1Inv.getSnp();
408  if (dSnp * dSnp > (seed0.getSigmaSnp2() + seed1Inv.getSigmaSnp2()) * mMatchParams->crudeNSigma2Cut[o2::track::kSnp]) {
409  rej = RejSnp;
410  break;
411  }
412  auto dY = seed0.getY() - seed1Inv.getY();
413  if (dY * dY > (seed0.getSigmaY2() + seed1Inv.getSigmaY2()) * mMatchParams->crudeNSigma2Cut[o2::track::kY]) {
414  rej = RejY;
415  break;
416  }
417  bool ignoreZ = seed0.origID.getSource() == o2d::GlobalTrackID::TPC || seed1.origID.getSource() == o2d::GlobalTrackID::TPC;
418  if (!ignoreZ) { // cut on Z makes no sense for TPC only tracks
419  auto dZ = seed0.getZ() - seed1Inv.getZ();
420  if (dZ * dZ > (seed0.getSigmaZ2() + seed1Inv.getSigmaZ2()) * mMatchParams->crudeNSigma2Cut[o2::track::kZ]) {
421  rej = RejZ;
422  break;
423  }
424  } else { // inflate Z error
425  seed1Inv.setCov(250. * 250., o2::track::DiagMap[o2::track::kZ]);
426  seed1Inv.setCov(0., o2::track::CovarMap[o2::track::kZ][o2::track::kY]); // set all correlation terms for Z error to 0
427  seed1Inv.setCov(0., o2::track::CovarMap[o2::track::kZ][o2::track::kSnp]);
428  seed1Inv.setCov(0., o2::track::CovarMap[o2::track::kZ][o2::track::kTgl]);
429  seed1Inv.setCov(0., o2::track::CovarMap[o2::track::kZ][o2::track::kQ2Pt]);
430  }
431  // calculate chi2 (expensive)
432  chi2 = seed0.getPredictedChi2(seed1Inv);
433  if (chi2 > mMatchParams->crudeChi2Cut) {
434  rej = RejChi2;
435  break;
436  }
437  rej = Accept;
438  registerMatch(i, j, chi2);
439  registerMatch(j, i, chi2); // the reverse reference can be also done in a separate loop
440  LOG(debug) << "Chi2 = " << chi2 << " NMatches " << mRecords.size();
441  break;
442  }
443 
444 #ifdef _ALLOW_DEBUG_TREES_
445  if (mDBGOut && ((rej == Accept && isDebugFlag(MatchTreeAccOnly)) || isDebugFlag(MatchTreeAll))) {
446  auto seed1I = seed1;
447  seed1I.invert();
448  if (seed1I.rotate(seed0.getAlpha()) && o2::base::Propagator::Instance()->PropagateToXBxByBz(seed1I, seed0.getX(), mMatchParams->maxSnp, mMatchParams->maxStep, mMatchParams->matCorr)) {
449  int rejI = int(rej);
450  (*mDBGOut) << "match"
451  << "tf=" << mTFCount << "seed0=" << seed0 << "seed1=" << seed1I << "chi2Match=" << chi2 << "rej=" << rejI << "\n";
452  }
453  }
454 #endif
455 
456  return rej;
457 }

460 void MatchCosmics::registerMatch(int i, int j, float chi2)
461 {
463  int newRef = mRecords.size();
464  auto& matchRec = mRecords.emplace_back(MatchRecord{i, j, chi2, MinusOne});
465  auto* best = &mSeeds[i].matchID;
466  while (*best > MinusOne) {
467  auto& oldMatchRec = mRecords[*best];
468  if (oldMatchRec.chi2 > chi2) { // insert new match in front of the old one
469  matchRec.next = *best; // new record will refer to the one it is superseding
470  *best = newRef; // the reference on the superseded record should now refer to new one
471  break;
472  }
473  best = &oldMatchRec.next;
474  }
475  if (matchRec.next == MinusOne) { // did not supersed any other record
476  *best = newRef;
477  }
478 }

481 void MatchCosmics::createSeeds(const o2::globaltracking::RecoContainer& data)
482 {
483  // Scan all inputs and create seeding tracks
484 
485  mSeeds.clear();
486 
487  auto creator = [this](auto& _tr, GTrackID _origID, float t0, float terr) {
488  if constexpr (std::is_base_of_v<o2::track::TrackParCov, std::decay_t<decltype(_tr)>>) {
489  if (std::abs(_tr.getQ2Pt()) > this->mQ2PtCutoff) {
490  return true;
491  }
492  if constexpr (isTPCTrack<decltype(_tr)>()) {
493  // unconstrained TPC track, with t0 = TrackTPC.getTime0+0.5*(DeltaFwd-DeltaBwd) and terr = 0.5*(DeltaFwd+DeltaBwd) in TimeBins
494  t0 *= this->mTPCTBinMUS;
495  terr *= this->mTPCTBinMUS;
496  } else if (isITSTrack<decltype(_tr)>()) {
497  t0 += 0.5 * this->mITSROFrameLengthMUS; // time 0 is supplied as beginning of ROF in \mus
498  terr *= this->mITSROFrameLengthMUS; // error is supplied a half-ROF duration, convert to \mus
499  } else { // all other tracks are provided with time and its gaussian error in \mus
500  terr *= this->mMatchParams->nSigmaTError;
501  }
502  terr += this->mMatchParams->timeToleranceMUS;
503  mSeeds.emplace_back(TrackSeed{_tr, {t0 - terr, t0 + terr}, _origID, MinusOne});
504  return true;
505  } else {
506  return false;
507  }
508  };
509 
510  data.createTracksVariadic(creator);
511 
512  LOG(info) << "collected " << mSeeds.size() << " seeds";
513 }
