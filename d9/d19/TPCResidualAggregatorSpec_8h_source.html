35 class ResidualAggregatorDevice : public o2::framework::Task
36 {
37  public:
38  void init(o2::framework::InitContext& ic) final
39  {
40  int minEnt = ic.options().get<int>("min-entries");
41  long slotLength = ic.options().get<long>("tf-per-slot");
42  if (slotLength == -1) {
43  slotLength = std::numeric_limits<long>::max();
44  }
45  int updateInterval = ic.options().get<int>("updateInterval");
46  int delay = ic.options().get<int>("max-delay");
47  mAggregator = std::make_unique<o2::tpc::ResidualAggregator>(minEnt);
48  // TODO mAggregator should get an option to set the binning externally (expose TrackResiduals::setBinning methods to user? as command line option?)
49  mAggregator->setSlotLength(slotLength);
50  mAggregator->setMaxSlotsDelay(delay);
51  mAggregator->setCheckIntervalInfiniteSlot(updateInterval);
52  }
53 
54  void run(o2::framework::ProcessingContext& pc) final
55  {
56  auto tfcounter = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime;
57  auto data = pc.inputs().get<gsl::span<o2::tpc::TrackResiduals::UnbinnedResid>>("input");
58  LOG(debug) << "Processing TF " << tfcounter << " with " << data.size() << " unbinned residuals";
59  mAggregator->process(tfcounter, data);
60  }
61 
62  void endOfStream(o2::framework::EndOfStreamContext& ec) final
63  {
64  LOG(info) << "Finalizing calibration for end of stream";
65  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
66  mAggregator->checkSlotsToFinalize(INFINITE_TF);
67  mAggregator.reset(); // must invoke destructor manually here, otherwise we get a segfault
68  }
69 
70  private:
71  std::unique_ptr<o2::tpc::ResidualAggregator> mAggregator;
72 };
