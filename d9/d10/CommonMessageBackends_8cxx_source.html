47 o2::framework::ServiceSpec CommonMessageBackends::fairMQBackendSpec()
48 {
49  return ServiceSpec{
50  .name = "fairmq-backend",
51  .init = [](ServiceRegistry& services, DeviceState&, fair::mq::ProgOptions&) -> ServiceHandle {
52  auto& device = services.get<RawDeviceService>();
53  auto context = new MessageContext(FairMQDeviceProxy{device.device()});
54  auto& spec = services.get<DeviceSpec const>();
55 
56  auto dispatcher = [&device](FairMQParts&& parts, std::string const& channel, unsigned int index) {
57  device.device()->Send(parts, channel, index);
58  };
59 
60  auto matcher = [policy = spec.dispatchPolicy](o2::header::DataHeader const& header) {
61  if (policy.triggerMatcher == nullptr) {
62  return true;
63  }
64  return policy.triggerMatcher(Output{header});
65  };
66 
67  if (spec.dispatchPolicy.action == DispatchPolicy::DispatchOp::WhenReady) {
68  context->init(DispatchControl{dispatcher, matcher});
69  }
70  return ServiceHandle{TypeIdHelpers::uniqueId<MessageContext>(), context};
71  },
72  .configure = CommonServices::noConfiguration(),
73  .preProcessing = CommonMessageBackendsHelpers<MessageContext>::clearContext(),
74  .postProcessing = CommonMessageBackendsHelpers<MessageContext>::sendCallback(),
75  .preEOS = CommonMessageBackendsHelpers<MessageContext>::clearContextEOS(),
76  .postEOS = CommonMessageBackendsHelpers<MessageContext>::sendCallbackEOS(),
77  .kind = ServiceKind::Serial};
78 }
