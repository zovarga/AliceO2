94 inline void TrackLTF::setPoint(const Cluster& cl, const Int_t layer, const Int_t clusterId, const MCCompLabel label, const Int_t extClsIndex)
95 {
96  auto nPoints = getNumberOfPoints();
97  if (nPoints > 0) {
98  if (mZ[nPoints - 1] == cl.getZ()) {
99  LOG(warn) << "MFT TrackLTF: skipping setPoint (1 cluster per layer!)";
100  return;
101  }
102  }
103  if (nPoints > constants::mft::LayersNumber) {
104  LOG(warn) << "MFT TrackLTF Overflow";
105  return;
106  }
107  mX[nPoints] = cl.getX();
108  mY[nPoints] = cl.getY();
109  mZ[nPoints] = cl.getZ();
110  mSigmaX2[nPoints] = cl.sigmaX2;
111  mSigmaY2[nPoints] = cl.sigmaY2;
112  mLayer[nPoints] = layer;
113  mClusterId[nPoints] = clusterId;
114  mMCCompLabels[nPoints] = label;
115  setExternalClusterIndex(nPoints, extClsIndex);
116  setNumberOfPoints(nPoints + 1);
117 }

120 inline void TrackLTF::sort()
121 {
122  // Orders elements along z position
123  struct ClusterData {
124  Float_t x;
125  Float_t y;
126  Float_t z;
127  Float_t sigmaX2;
128  Float_t sigmaY2;
129  Int_t layer;
130  Int_t clusterId;
131  MCCompLabel label;
132  Int_t extClsIndex;
133  };
134  std::vector<ClusterData> points;
135 
136  // Loading cluster data
137  for (Int_t point = 0; point < getNumberOfPoints(); ++point) {
138  auto& somepoint = points.emplace_back();
139  somepoint.x = mX[point];
140  somepoint.y = mY[point];
141  somepoint.z = mZ[point];
142  somepoint.sigmaX2 = mSigmaX2[point];
143  somepoint.sigmaY2 = mSigmaY2[point];
144  somepoint.layer = mLayer[point];
145  somepoint.clusterId = mClusterId[point];
146  somepoint.label = mMCCompLabels[point];
147  somepoint.extClsIndex = mExtClsIndex[point];
148  }
149 
150  // Sorting cluster data
151  std::sort(points.begin(), points.end(), [](ClusterData a, ClusterData b) { return a.z > b.z; });
152 
153  // Storing sorted cluster data
154  for (Int_t point = 0; point < getNumberOfPoints(); ++point) {
155  mX[point] = points[point].x;
156  mY[point] = points[point].y;
157  mZ[point] = points[point].z;
158  mSigmaX2[point] = points[point].sigmaX2;
159  mSigmaY2[point] = points[point].sigmaY2;
160  mLayer[point] = points[point].layer;
161  mClusterId[point] = points[point].clusterId;
162  mMCCompLabels[point] = points[point].label;
163  mExtClsIndex[point] = points[point].extClsIndex;
164  }
165 }
