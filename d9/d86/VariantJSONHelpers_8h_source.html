31 struct VariantReader : public rapidjson::BaseReaderHandler<rapidjson::UTF8<>, VariantReader<V>> {
32  using Ch = rapidjson::UTF8<>::Ch;
33  using SizeType = rapidjson::SizeType;
34 
35  enum struct State {
36  IN_START,
37  IN_STOP,
38  IN_DATA,
39  IN_KEY,
40  IN_ARRAY,
41  IN_ROW,
42  IN_ERROR
43  };
44 
45  VariantReader()
46  : states{},
47  rows{0},
48  cols{0}
49  {
50  debug << "Start" << std::endl;
51  states.push(State::IN_START);
52  }
53 
54  bool Null()
55  {
56  debug << "Null value encountered" << std::endl;
57  return true;
58  }

60  bool Int(int i)
61  {
62  debug << "Int(" << i << ")" << std::endl;
63  if (states.top() == State::IN_ERROR) {
64  debug << "In ERROR state" << std::endl;
65  return false;
66  }
67  if constexpr (!std::is_same_v<int, variant_array_element_type_t<V>>) {
68  states.push(State::IN_ERROR);
69  return true;
70  } else {
71  if (states.top() == State::IN_ARRAY || states.top() == State::IN_ROW) {
72  debug << "added to array" << std::endl;
73  accumulatedData.push_back(i);
74  return true;
75  }
76  }
77  states.push(State::IN_ERROR);
78  return true;
79  }
80 
81  bool Uint(unsigned i)
82  {
83  debug << "Uint -> Int" << std::endl;
84  return Int(static_cast<int>(i));
85  }
86 
87  bool Int64(int64_t i)
88  {
89  debug << "Int64 -> Int" << std::endl;
90  return Int(static_cast<int>(i));
91  }
92 
93  bool Uint64(uint64_t i)
94  {
95  debug << "Uint64 -> Int" << std::endl;
96  return Int(static_cast<int>(i));
97  }
98 
99  bool Double(double d)
100  {
101  debug << "Double(" << d << ")" << std::endl;
102  if (states.top() == State::IN_ERROR) {
103  debug << "In ERROR state" << std::endl;
104  return false;
105  }
106  if constexpr (!(std::is_same_v<float, variant_array_element_type_t<V>> || std::is_same_v<double, variant_array_element_type_t<V>>)) {
107  states.push(State::IN_ERROR);
108  return true;
109  }
110  if (states.top() == State::IN_ARRAY || states.top() == State::IN_ROW) {
111  if constexpr (std::is_same_v<double, variant_array_element_type_t<V>>) {
112  debug << "added to array as double" << std::endl;
113  accumulatedData.push_back(d);
114  return true;
115  } else if constexpr (std::is_same_v<float, variant_array_element_type_t<V>>) {
116  debug << "added to array as float" << std::endl;
117  accumulatedData.push_back(static_cast<float>(d));
118  return true;
119  }
120  }
121  states.push(State::IN_ERROR);
122  return true;
123  }
124 
125  bool Bool(bool b)
126  {
127  debug << "Bool(" << b << ")" << std::endl;
128  if (states.top() == State::IN_ERROR) {
129  debug << "In ERROR state" << std::endl;
130  return false;
131  }
132  if constexpr (!std::is_same_v<bool, variant_array_element_type_t<V>>) {
133  states.push(State::IN_ERROR);
134  return false;
135  } else {
136  if (states.top() == State::IN_ARRAY) {
137  debug << "added to array" << std::endl;
138  accumulatedData.push_back(b);
139  return true;
140  }
141  states.push(State::IN_ERROR);
142  return true;
143  }
144  }
145 
146  bool String(const Ch* str, SizeType, bool)
147  {
148  debug << "String(" << str << ")" << std::endl;
149  if (states.top() == State::IN_ERROR) {
150  debug << "In ERROR state" << std::endl;
151  return false;
152  }
153  if constexpr (!(V == VariantType::ArrayString || isLabeledArray<V>())) {
154  states.push(State::IN_ERROR);
155  return true;
156  } else {
157  if (states.top() == State::IN_ARRAY) {
158  debug << "added to array" << std::endl;
159  if constexpr (isLabeledArray<V>()) {
160  if (currentKey == labels_rows_str) {
161  labels_rows.push_back(str);
162  return true;
163  } else if (currentKey == labels_cols_str) {
164  labels_cols.push_back(str);
165  return true;
166  } else {
167  states.push(State::IN_ERROR);
168  return true;
169  }
170  } else {
171  accumulatedData.push_back(str);
172  }
173  return true;
174  }
175  states.push(State::IN_ERROR);
176  return true;
177  }
178  }
179 
180  bool StartObject()
181  {
182  debug << "StartObject()" << std::endl;
183  if (states.top() == State::IN_ERROR) {
184  debug << "In ERROR state" << std::endl;
185  return false;
186  }
187  if (states.top() == State::IN_START) {
188  states.push(State::IN_DATA);
189  return true;
190  }
191  states.push(State::IN_ERROR);
192  return true;
193  }
194 
195  bool Key(const Ch* str, SizeType, bool)
196  {
197  debug << "Key(" << str << ")" << std::endl;
198  if (states.top() == State::IN_ERROR) {
199  debug << "In ERROR state" << std::endl;
200  currentKey = str;
201  return false;
202  }
203  if (states.top() == State::IN_DATA) {
204  //no previous keys
205  states.push(State::IN_KEY);
206  currentKey = str;
207  return true;
208  }
209  if (states.top() == State::IN_KEY) {
210  currentKey = str;
211  if constexpr (!isLabeledArray<V>()) {
212  debug << "extra keys in a single-key variant" << std::endl;
213  states.push(State::IN_ERROR);
214  return true;
215  }
216  return true;
217  }
218  currentKey = str;
219  states.push(State::IN_ERROR);
220  return true;
221  }
222 
223  bool EndObject(SizeType)
224  {
225  debug << "EndObject()" << std::endl;
226  if (states.top() == State::IN_ERROR) {
227  debug << "In ERROR state" << std::endl;
228  return false;
229  }
230  if (states.top() == State::IN_KEY) {
231  if constexpr (isArray<V>()) {
232  debug << "creating 1d-array variant" << std::endl;
233  result = Variant(accumulatedData);
234  } else if constexpr (isArray2D<V>()) {
235  debug << "creating 2d-array variant" << std::endl;
236  assert(accumulatedData.size() == rows * cols);
237  result = Variant(Array2D{accumulatedData, rows, cols});
238  } else if constexpr (isLabeledArray<V>()) {
239  debug << "creating labeled array variant" << std::endl;
240  assert(accumulatedData.size() == rows * cols);
241  if (labels_rows.empty() == false) {
242  assert(labels_rows.size() == rows);
243  }
244  if (labels_cols.empty() == false) {
245  assert(labels_cols.size() == cols);
246  }
247  result = Variant(LabeledArray{Array2D{accumulatedData, rows, cols}, labels_rows, labels_cols});
248  }
249  states.push(State::IN_STOP);
250  return true;
251  }
252  states.push(State::IN_ERROR);
253  return true;
254  }
255 
256  bool StartArray()
257  {
258  debug << "StartArray()" << std::endl;
259  if (states.top() == State::IN_ERROR) {
260  debug << "In ERROR state" << std::endl;
261  return false;
262  }
263  if (states.top() == State::IN_KEY) {
264  states.push(State::IN_ARRAY);
265  return true;
266  } else if (states.top() == State::IN_ARRAY) {
267  if constexpr (isArray2D<V>() || isLabeledArray<V>()) {
268  states.push(State::IN_ROW);
269  return true;
270  }
271  }
272  states.push(State::IN_ERROR);
273  return true;
274  }
275 
276  bool EndArray(SizeType elementCount)
277  {
278  debug << "EndArray()" << std::endl;
279  if (states.top() == State::IN_ERROR) {
280  debug << "In ERROR state" << std::endl;
281  return false;
282  }
283  if (states.top() == State::IN_ARRAY) {
284  //finish up array
285  states.pop();
286  if constexpr (isArray2D<V>() || isLabeledArray<V>()) {
287  rows = elementCount;
288  }
289  return true;
290  } else if (states.top() == State::IN_ROW) {
291  //finish up row
292  states.pop();
293  if constexpr (isArray2D<V>() || isLabeledArray<V>()) {
294  cols = elementCount;
295  }
296  return true;
297  }
298  states.push(State::IN_ERROR);
299  return true;
300  }
301 
302  std::stack<State> states;
303  std::ostringstream debug;
304 
305  uint32_t rows;
306  uint32_t cols;
307  std::string currentKey;
308  std::vector<variant_array_element_type_t<V>> accumulatedData;
309  std::vector<std::string> labels_rows;
310  std::vector<std::string> labels_cols;
311  Variant result;
312 };

314 template <VariantType V>
315 void writeVariant(std::ostream& o, Variant const& v)
316 {
317  if constexpr (isArray<V>() || isArray2D<V>() || isLabeledArray<V>()) {
318  using type = variant_array_element_type_t<V>;
319  rapidjson::OStreamWrapper osw(o);
320  rapidjson::Writer<rapidjson::OStreamWrapper> w(osw);
321 
322  auto writeArray = [&](auto* values, size_t size) {
323  using T = std::remove_pointer_t<decltype(values)>;
324  w.StartArray();
325  for (auto i = 0u; i < size; ++i) {
326  if constexpr (std::is_same_v<int, T>) {
327  w.Int(values[i]);
328  } else if constexpr (std::is_same_v<float, T> || std::is_same_v<double, T>) {
329  w.Double(values[i]);
330  } else if constexpr (std::is_same_v<bool, T>) {
331  w.Bool(values[i]);
332  } else if constexpr (std::is_same_v<std::string, T>) {
333  w.String(values[i].c_str());
334  }
335  }
336  w.EndArray();
337  };
338 
339  auto writeVector = [&](auto&& vector) {
340  return writeArray(vector.data(), vector.size());
341  };
342 
343  auto writeArray2D = [&](auto&& array2d) {
344  using T = typename std::decay_t<decltype(array2d)>::element_t;
345  w.StartArray();
346  for (auto i = 0u; i < array2d.rows; ++i) {
347  w.StartArray();
348  for (auto j = 0u; j < array2d.cols; ++j) {
349  if constexpr (std::is_same_v<int, T>) {
350  w.Int(array2d(i, j));
351  } else if constexpr (std::is_same_v<float, T> || std::is_same_v<double, T>) {
352  w.Double(array2d(i, j));
353  }
354  }
355  w.EndArray();
356  }
357  w.EndArray();
358  };
359 
360  auto writeLabeledArray = [&](auto&& array) {
361  w.Key(labels_rows_str);
362  writeVector(array.getLabelsRows());
363  w.Key(labels_cols_str);
364  writeVector(array.getLabelsCols());
365  w.Key("values");
366  writeArray2D(array.getData());
367  };
368 
369  w.StartObject();
370  if constexpr (isArray<V>()) {
371  w.Key("values");
372  writeArray(v.get<type*>(), v.size());
373  } else if constexpr (isArray2D<V>()) {
374  w.Key("values");
375  writeArray2D(v.get<Array2D<type>>());
376  } else if constexpr (isLabeledArray<V>()) {
377  writeLabeledArray(v.get<LabeledArray<type>>());
378  } else if constexpr (V == VariantType::Dict) {
379  // nothing to do for dicts
380  }
381  w.EndObject();
382  }
383 }

386 struct VariantJSONHelpers {
387  template <VariantType V>
388  static Variant read(std::istream& s)
389  {
390  rapidjson::Reader reader;
391  rapidjson::IStreamWrapper isw(s);
392  VariantReader<V> vreader;
393  bool ok = reader.Parse(isw, vreader);
394 
395  if (ok == false) {
396  std::stringstream error;
397  error << "Cannot parse serialized Variant, error: " << rapidjson::GetParseError_En(reader.GetParseErrorCode()) << " at offset: " << reader.GetErrorOffset();
398  throw std::runtime_error(error.str());
399  }
400  return vreader.result;
401  }
402 
403  static void write(std::ostream& o, Variant const& v)
404  {
405  switch (v.type()) {
406  case VariantType::ArrayInt:
407  writeVariant<VariantType::ArrayInt>(o, v);
408  break;
409  case VariantType::ArrayFloat:
410  writeVariant<VariantType::ArrayFloat>(o, v);
411  break;
412  case VariantType::ArrayDouble:
413  writeVariant<VariantType::ArrayDouble>(o, v);
414  break;
415  case VariantType::ArrayBool:
416  throw std::runtime_error("Bool vectors not implemented yet");
417  // writeVariant<VariantType::ArrayBool>(o, v);
418  break;
419  case VariantType::ArrayString:
420  writeVariant<VariantType::ArrayString>(o, v);
421  break;
422  case VariantType::Array2DInt:
423  writeVariant<VariantType::Array2DInt>(o, v);
424  break;
425  case VariantType::Array2DFloat:
426  writeVariant<VariantType::Array2DFloat>(o, v);
427  break;
428  case VariantType::Array2DDouble:
429  writeVariant<VariantType::Array2DDouble>(o, v);
430  break;
431  case VariantType::LabeledArrayInt:
432  writeVariant<VariantType::LabeledArrayInt>(o, v);
433  break;
434  case VariantType::LabeledArrayFloat:
435  writeVariant<VariantType::LabeledArrayFloat>(o, v);
436  break;
437  case VariantType::LabeledArrayDouble:
438  writeVariant<VariantType::LabeledArrayDouble>(o, v);
439  break;
440  case VariantType::Dict:
441  writeVariant<VariantType::Dict>(o, v);
442  default:
443  break;
444  }
445  }
