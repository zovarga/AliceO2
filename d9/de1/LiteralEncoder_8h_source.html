54 stream_IT LiteralEncoder<coder_T, stream_T, source_T>::process(source_IT inputBegin, source_IT inputEnd, stream_IT outputBegin, std::vector<source_T>& literals) const
55 {
56  using namespace internal;
57  LOG(trace) << "start encoding";
58  RANSTimer t;
59 
60 #ifdef O2_RANS_PRINT_PROCESSED_DATA
61  JSONArrayLogger<source_T> arrayLogger{true};
62 #endif
63 
64  t.start();
65 
66  if (inputBegin == inputEnd) {
67  LOG(warning) << "passed empty message to encoder, skip encoding";
68  return outputBegin;
69  }
70 
71  ransCoder_t rans0{this->mSymbolTable.getPrecision()};
72  ransCoder_t rans1{this->mSymbolTable.getPrecision()};
73 
74  stream_IT outputIter = outputBegin;
75  source_IT inputIT = inputEnd;
76 
77  const auto inputBufferSize = std::distance(inputBegin, inputEnd);
78 
79  auto encode = [&, this](source_IT symbolIter, stream_IT outputIter, ransCoder_t& coder) {
80  const source_T symbol = *symbolIter;
81  const auto& encoderSymbol = (this->mSymbolTable)[symbol];
82  if (this->mSymbolTable.isEscapeSymbol(symbol)) {
83  literals.push_back(symbol);
84  }
85 
86 #ifdef O2_RANS_PRINT_PROCESSED_DATA
87  arrayLogger << symbol;
88 #endif
89 
90  return coder.putSymbol(outputIter, encoderSymbol);
91  };
92 
93  // odd number of bytes?
94  if (inputBufferSize & 1) {
95  outputIter = encode(--inputIT, outputIter, rans0);
96  }
97 
98  while (inputIT != inputBegin) { // NB: working in reverse!
99  outputIter = encode(--inputIT, outputIter, rans1);
100  outputIter = encode(--inputIT, outputIter, rans0);
101  }
102  outputIter = rans1.flush(outputIter);
103  outputIter = rans0.flush(outputIter);
104  // first iterator past the range so that sizes, distances and iterators work correctly.
105  ++outputIter;
106 
107  t.stop();
108 
109 #ifdef O2_RANS_PRINT_PROCESSED_DATA
110  LOG(info) << "encoderInput:" << arrayLogger;
111 #endif
112 
113  LOG(debug1) << "Encoder::" << __func__ << " {ProcessedBytes: " << inputBufferSize * sizeof(source_T) << ","
114  << " inclusiveTimeMS: " << t.getDurationMS() << ","
115  << " BandwidthMiBPS: " << std::fixed << std::setprecision(2) << (inputBufferSize * sizeof(source_T) * 1.0) / (t.getDurationS() * 1.0 * (1 << 20)) << "}";
116 
117 // advanced diagnostics for debug builds
118 #if !defined(NDEBUG)
119 
120  const auto inputBufferSizeB = inputBufferSize * sizeof(source_T);
121 
122  LOG(debug2) << "EncoderProperties: {"
123  << "sourceTypeB: " << sizeof(source_T) << ", "
124  << "streamTypeB: " << sizeof(stream_T) << ", "
125  << "coderTypeB: " << sizeof(coder_T) << ", "
126  << "symbolTablePrecision: " << this->mSymbolTable.getPrecision() << ", "
127  << "inputBufferSizeB: " << inputBufferSizeB << "}";
128 #endif
129 
130  LOG(trace) << "done encoding";
131 
132  return outputIter;
133 };
