52 template <typename value_T, size_t shift>
53 class ShiftFunctor
54 {
55  public:
56  template <typename iterA_T, typename iterB_T>
57  inline value_T operator()(iterA_T iterA, iterB_T iterB) const
58  {
59  return *iterB + (static_cast<value_T>(*iterA) << shift);
60  };
61 
62  template <typename iterA_T, typename iterB_T>
63  inline void operator()(iterA_T iterA, iterB_T iterB, value_T value) const
64  {
65  *iterA = value >> shift;
66  *iterB = value & ((0x1 << shift) - 0x1);
67  };
68 };

70 template <typename iterA_T, typename iterB_T, typename F>
71 auto makeInputIterators(iterA_T iterA, iterB_T iterB, size_t nElements, F functor)
72 {
73  using namespace o2::rans::utils;
74 
75  auto advanceIter = [](auto iter, size_t nElements) {
76  auto tmp = iter;
77  std::advance(tmp, nElements);
78  return tmp;
79  };
80 
81  return std::make_tuple(CombinedInputIterator{iterA, iterB, functor},
82  CombinedInputIterator{advanceIter(iterA, nElements), advanceIter(iterB, nElements), functor});
83 };

163 void CTFCoder::encode(VEC& buff, const CompressedClusters& ccl)
164 {
165  using MD = o2::ctf::Metadata::OptStore;
166  using namespace detail;
167  // what to do which each field: see o2::ctf::Metadata explanation
168  constexpr MD optField[CTF::getNBlocks()] = {
169  MD::EENCODE, // qTotA
170  MD::EENCODE, // qMaxA
171  MD::EENCODE, // flagsA
172  MD::EENCODE, // rowDiffA
173  MD::EENCODE, // sliceLegDiffA
174  MD::EENCODE, // padResA
175  MD::EENCODE, // timeResA
176  MD::EENCODE, // sigmaPadA
177  MD::EENCODE, // sigmaTimeA
178  MD::EENCODE, // qPtA
179  MD::EENCODE, // rowA
180  MD::EENCODE, // sliceA
181  MD::EENCODE, // timeA
182  MD::EENCODE, // padA
183  MD::EENCODE, // qTotU
184  MD::EENCODE, // qMaxU
185  MD::EENCODE, // flagsU
186  MD::EENCODE, // padDiffU
187  MD::EENCODE, // timeDiffU
188  MD::EENCODE, // sigmaPadU
189  MD::EENCODE, // sigmaTimeU
190  MD::EENCODE, // nTrackClusters
191  MD::EENCODE // nSliceRowClusters
192  };
193 
194  // book output size with some margin
195  auto szIni = estimateCompressedSize(ccl);
196  buff.resize(szIni);
197 
198  auto ec = CTF::create(buff);
199  uint32_t flags = 0;
200  if (mCombineColumns) {
201  flags |= CTFHeader::CombinedColumns;
202  }
203  ec->setHeader(CTFHeader{o2::detectors::DetID::TPC, 0, 1, 0, // dummy timestamp, version 1.0
204  ccl, flags});
205  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
206  ec->getANSHeader().majorVersion = 0;
207  ec->getANSHeader().minorVersion = 1;
208 
209  auto encodeTPC = [&buff, &optField, &coders = mCoders, mfc = this->getMemMarginFactor()](auto begin, auto end, CTF::Slots slot, size_t probabilityBits) {
210  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
211  const auto slotVal = static_cast<int>(slot);
212  CTF::get(buff.data())->encode(begin, end, slotVal, probabilityBits, optField[slotVal], &buff, coders[slotVal].get(), mfc);
213  };
214 
215  if (mCombineColumns) {
216  const auto [begin, end] = makeInputIterators(ccl.qTotA, ccl.qMaxA, ccl.nAttachedClusters,
217  ShiftFunctor<combinedType_t<CTF::NBitsQTot, CTF::NBitsQMax>, CTF::NBitsQMax>{});
218  encodeTPC(begin, end, CTF::BLCqTotA, 0);
219  } else {
220  encodeTPC(ccl.qTotA, ccl.qTotA + ccl.nAttachedClusters, CTF::BLCqTotA, 0);
221  }
222  encodeTPC(ccl.qMaxA, ccl.qMaxA + (mCombineColumns ? 0 : ccl.nAttachedClusters), CTF::BLCqMaxA, 0);
223 
224  encodeTPC(ccl.flagsA, ccl.flagsA + ccl.nAttachedClusters, CTF::BLCflagsA, 0);
225 
226  if (mCombineColumns) {
227  const auto [begin, end] = makeInputIterators(ccl.rowDiffA, ccl.sliceLegDiffA, ccl.nAttachedClustersReduced,
228  ShiftFunctor<combinedType_t<CTF::NBitsRowDiff, CTF::NBitsSliceLegDiff>, CTF::NBitsSliceLegDiff>{});
229  encodeTPC(begin, end, CTF::BLCrowDiffA, 0);
230  } else {
231  encodeTPC(ccl.rowDiffA, ccl.rowDiffA + ccl.nAttachedClustersReduced, CTF::BLCrowDiffA, 0);
232  }
233  encodeTPC(ccl.sliceLegDiffA, ccl.sliceLegDiffA + (mCombineColumns ? 0 : ccl.nAttachedClustersReduced), CTF::BLCsliceLegDiffA, 0);
234 
235  encodeTPC(ccl.padResA, ccl.padResA + ccl.nAttachedClustersReduced, CTF::BLCpadResA, 0);
236  encodeTPC(ccl.timeResA, ccl.timeResA + ccl.nAttachedClustersReduced, CTF::BLCtimeResA, 0);
237 
238  if (mCombineColumns) {
239  const auto [begin, end] = makeInputIterators(ccl.sigmaPadA, ccl.sigmaTimeA, ccl.nAttachedClusters,
240  ShiftFunctor<combinedType_t<CTF::NBitsSigmaPad, CTF::NBitsSigmaTime>, CTF::NBitsSigmaTime>{});
241  encodeTPC(begin, end, CTF::BLCsigmaPadA, 0);
242  } else {
243  encodeTPC(ccl.sigmaPadA, ccl.sigmaPadA + ccl.nAttachedClusters, CTF::BLCsigmaPadA, 0);
244  }
245  encodeTPC(ccl.sigmaTimeA, ccl.sigmaTimeA + (mCombineColumns ? 0 : ccl.nAttachedClusters), CTF::BLCsigmaTimeA, 0);
246 
247  encodeTPC(ccl.qPtA, ccl.qPtA + ccl.nTracks, CTF::BLCqPtA, 0);
248  encodeTPC(ccl.rowA, ccl.rowA + ccl.nTracks, CTF::BLCrowA, 0);
249  encodeTPC(ccl.sliceA, ccl.sliceA + ccl.nTracks, CTF::BLCsliceA, 0);
250  encodeTPC(ccl.timeA, ccl.timeA + ccl.nTracks, CTF::BLCtimeA, 0);
251  encodeTPC(ccl.padA, ccl.padA + ccl.nTracks, CTF::BLCpadA, 0);
252 
253  if (mCombineColumns) {
254  const auto [begin, end] = makeInputIterators(ccl.qTotU, ccl.qMaxU, ccl.nUnattachedClusters,
255  ShiftFunctor<combinedType_t<CTF::NBitsQTot, CTF::NBitsQMax>, CTF::NBitsQMax>{});
256  encodeTPC(begin, end, CTF::BLCqTotU, 0);
257  } else {
258  encodeTPC(ccl.qTotU, ccl.qTotU + ccl.nUnattachedClusters, CTF::BLCqTotU, 0);
259  }
260  encodeTPC(ccl.qMaxU, ccl.qMaxU + (mCombineColumns ? 0 : ccl.nUnattachedClusters), CTF::BLCqMaxU, 0);
261 
262  encodeTPC(ccl.flagsU, ccl.flagsU + ccl.nUnattachedClusters, CTF::BLCflagsU, 0);
263  encodeTPC(ccl.padDiffU, ccl.padDiffU + ccl.nUnattachedClusters, CTF::BLCpadDiffU, 0);
264  encodeTPC(ccl.timeDiffU, ccl.timeDiffU + ccl.nUnattachedClusters, CTF::BLCtimeDiffU, 0);
265 
266  if (mCombineColumns) {
267  const auto [begin, end] = makeInputIterators(ccl.sigmaPadU, ccl.sigmaTimeU, ccl.nUnattachedClusters,
268  ShiftFunctor<combinedType_t<CTF::NBitsSigmaPad, CTF::NBitsSigmaTime>, CTF::NBitsSigmaTime>{});
269  encodeTPC(begin, end, CTF::BLCsigmaPadU, 0);
270  } else {
271  encodeTPC(ccl.sigmaPadU, ccl.sigmaPadU + ccl.nUnattachedClusters, CTF::BLCsigmaPadU, 0);
272  }
273  encodeTPC(ccl.sigmaTimeU, ccl.sigmaTimeU + (mCombineColumns ? 0 : ccl.nUnattachedClusters), CTF::BLCsigmaTimeU, 0);
274 
275  encodeTPC(ccl.nTrackClusters, ccl.nTrackClusters + ccl.nTracks, CTF::BLCnTrackClusters, 0);
276  encodeTPC(ccl.nSliceRowClusters, ccl.nSliceRowClusters + ccl.nSliceRows, CTF::BLCnSliceRowClusters, 0);
277  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
278 }

281 template <typename VEC>
282 void CTFCoder::decode(const CTF::base& ec, VEC& buffVec)
283 {
284  using namespace detail;
285  CompressedClusters cc;
286  CompressedClustersCounters& ccCount = cc;
287  auto& header = ec.getHeader();
288  checkDictVersion(static_cast<const o2::ctf::CTFDictHeader&>(header));
289  checkDataDictionaryConsistency(header);
290  ccCount = static_cast<const CompressedClustersCounters&>(header);
291  CompressedClustersFlat* ccFlat = nullptr;
292  size_t sizeCFlatBody = alignSize(ccFlat);
293  size_t sz = sizeCFlatBody + estimateSize(cc); // total size of the buffVec accounting for the alignment
294  size_t vsz = sizeof(typename std::remove_reference<decltype(buffVec)>::type::value_type); // size of the element of the buffer
295  buffVec.resize(sz / vsz);
296  ccFlat = reinterpret_cast<CompressedClustersFlat*>(buffVec.data()); // RS? do we need to align this pointer, or PMR vector will be already aligned?
297  auto buff = reinterpret_cast<void*>(reinterpret_cast<char*>(buffVec.data()) + sizeCFlatBody); // will be the start of the CompressedClustersFlat payload
298 
299  setCompClusAddresses(cc, buff);
300  ccFlat->set(sz, cc); // set offsets
301  ec.print(getPrefix(), mVerbosity);
302 
303  // decode encoded data directly to destination buff
304  auto decodeTPC = [&ec, &coders = mCoders](auto begin, CTF::Slots slot) {
305  const auto slotVal = static_cast<int>(slot);
306  ec.decode(begin, slotVal, coders[slotVal].get());
307  };
308 
309  if (mCombineColumns) {
310  detail::MergedColumnsDecoder<CTF::NBitsQTot, CTF::NBitsQMax>::decode(cc.qTotA, cc.qMaxA, CTF::BLCqTotA, decodeTPC);
311  } else {
312  decodeTPC(cc.qTotA, CTF::BLCqTotA);
313  decodeTPC(cc.qMaxA, CTF::BLCqMaxA);
314  }
315 
316  decodeTPC(cc.flagsA, CTF::BLCflagsA);
317 
318  if (mCombineColumns) {
319  detail::MergedColumnsDecoder<CTF::NBitsRowDiff, CTF::NBitsSliceLegDiff>::decode(cc.rowDiffA, cc.sliceLegDiffA, CTF::BLCrowDiffA, decodeTPC);
320  } else {
321  decodeTPC(cc.rowDiffA, CTF::BLCrowDiffA);
322  decodeTPC(cc.sliceLegDiffA, CTF::BLCsliceLegDiffA);
323  }
324 
325  decodeTPC(cc.padResA, CTF::BLCpadResA);
326  decodeTPC(cc.timeResA, CTF::BLCtimeResA);
327 
328  if (mCombineColumns) {
329  detail::MergedColumnsDecoder<CTF::NBitsSigmaPad, CTF::NBitsSigmaTime>::decode(cc.sigmaPadA, cc.sigmaTimeA, CTF::BLCsigmaPadA, decodeTPC);
330  } else {
331  decodeTPC(cc.sigmaPadA, CTF::BLCsigmaPadA);
332  decodeTPC(cc.sigmaTimeA, CTF::BLCsigmaTimeA);
333  }
334 
335  decodeTPC(cc.qPtA, CTF::BLCqPtA);
336  decodeTPC(cc.rowA, CTF::BLCrowA);
337  decodeTPC(cc.sliceA, CTF::BLCsliceA);
338  decodeTPC(cc.timeA, CTF::BLCtimeA);
339  decodeTPC(cc.padA, CTF::BLCpadA);
340 
341  if (mCombineColumns) {
342  detail::MergedColumnsDecoder<CTF::NBitsQTot, CTF::NBitsQMax>::decode(cc.qTotU, cc.qMaxU, CTF::BLCqTotU, decodeTPC);
343  } else {
344  decodeTPC(cc.qTotU, CTF::BLCqTotU);
345  decodeTPC(cc.qMaxU, CTF::BLCqMaxU);
346  }
347 
348  decodeTPC(cc.flagsU, CTF::BLCflagsU);
349  decodeTPC(cc.padDiffU, CTF::BLCpadDiffU);
350  decodeTPC(cc.timeDiffU, CTF::BLCtimeDiffU);
351 
352  if (mCombineColumns) {
353  detail::MergedColumnsDecoder<CTF::NBitsSigmaPad, CTF::NBitsSigmaTime>::decode(cc.sigmaPadU, cc.sigmaTimeU, CTF::BLCsigmaPadU, decodeTPC);
354  } else {
355  decodeTPC(cc.sigmaPadU, CTF::BLCsigmaPadU);
356  decodeTPC(cc.sigmaTimeU, CTF::BLCsigmaTimeU);
357  }
358 
359  decodeTPC(cc.nTrackClusters, CTF::BLCnTrackClusters);
360  decodeTPC(cc.nSliceRowClusters, CTF::BLCnSliceRowClusters);
361 }
