31 o2::ft0::RecPoints CollisionTimeRecoTask::process(o2::ft0::Digit const& bcd,
32  gsl::span<const o2::ft0::ChannelData> inChData,
33  gsl::span<o2::ft0::ChannelDataFloat> outChData)
34 {
35  LOG(debug) << "Running reconstruction on new event";
36 
37  Int_t ndigitsC = 0, ndigitsA = 0;
38 
39  constexpr Int_t nMCPsA = 4 * Geometry::NCellsA;
40 
41  Float_t sideAtime = 0, sideCtime = 0;
42 
43  int nch = inChData.size();
44  const auto parInv = DigitizationParameters::Instance().mMV_2_NchannelsInverse;
45  for (int ich = 0; ich < nch; ich++) {
46  int offsetChannel = getOffset(int(inChData[ich].ChId), inChData[ich].QTCAmpl);
47  outChData[ich] = o2::ft0::ChannelDataFloat{inChData[ich].ChId,
48  (inChData[ich].CFDTime - offsetChannel) * Geometry::ChannelWidth,
49  (float)inChData[ich].QTCAmpl,
50  inChData[ich].ChainQTC};
51 
52  // only signals with amplitude participate in collision time
53  if (outChData[ich].QTCAmpl > DigitizationParameters::Instance().mAmpThresholdForReco && std::abs(outChData[ich].CFDTime) < DigitizationParameters::Instance().mTimeThresholdForReco) {
54  if (outChData[ich].ChId < nMCPsA) {
55  sideAtime += outChData[ich].CFDTime;
56  ndigitsA++;
57  } else {
58  sideCtime += outChData[ich].CFDTime;
59  LOG(debug) << "cfd " << outChData[ich].ChId << " dig " << 13.2 * inChData[ich].CFDTime << " rec " << outChData[ich].CFDTime << " amp " << (float)inChData[ich].QTCAmpl << " offset " << offsetChannel;
60  ndigitsC++;
61  }
62  }
63  }
64  auto sDummyCollissionTime = o2::ft0::RecPoints::sDummyCollissionTime;
65  std::array<short, 4> mCollisionTime = {sDummyCollissionTime, sDummyCollissionTime, sDummyCollissionTime, sDummyCollissionTime};
66  // !!!! tobe done::should be fix with ITS vertex
67  mCollisionTime[TimeA] = (ndigitsA > 0) ? sideAtime / ndigitsA : sDummyCollissionTime; // 2 * o2::InteractionRecord::DummyTime;
68  mCollisionTime[TimeC] = (ndigitsC > 0) ? sideCtime / ndigitsC : sDummyCollissionTime; //2 * o2::InteractionRecord::DummyTime;
69 
70  if (ndigitsA > 0 && ndigitsC > 0) {
71  mCollisionTime[Vertex] = (mCollisionTime[TimeA] - mCollisionTime[TimeC]) / 2.;
72  mCollisionTime[TimeMean] = (mCollisionTime[TimeA] + mCollisionTime[TimeC]) / 2.;
73  } else {
74  mCollisionTime[TimeMean] = std::min(mCollisionTime[TimeA], mCollisionTime[TimeC]);
75  }
76  LOG(debug) << " Nch " << nch << " Collision time " << mCollisionTime[TimeA] << " " << mCollisionTime[TimeC] << " " << mCollisionTime[TimeMean] << " " << mCollisionTime[Vertex];
77  return RecPoints{
78  mCollisionTime, bcd.ref.getFirstEntry(), bcd.ref.getEntries(), bcd.mIntRecord, bcd.mTriggers};
79 }
