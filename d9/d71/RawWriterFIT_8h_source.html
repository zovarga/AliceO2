64  void convertDigitsToRaw(const std::string& outputDir, const std::string& filenameDigits)
65  {
66  LOG(info) << "Converting Digits to Raw data...";
67  mWriter.setCarryOverCallBack(this);
68  LookupTable_t::Instance().printFullMap();
69  //Preparing topo2FEEmetadata map
70  mMapTopo2FEEmetadata.clear();
71  mMapTopo2FEEmetadata = LookupTable_t::Instance().template makeMapFEEmetadata<o2::header::RAWDataHeader, RDHUtils>();
72  //Preparing filenames
73  std::string detName = LookupTable_t::sDetectorName;
74  auto makeFilename = [&](const o2::header::RAWDataHeader& rdh) -> std::string {
75  std::string maskName{};
76  if (mFileFor != "all") { // single file for all links
77  maskName += fmt::format("_{}", mFlpName);
78  if (mFileFor != "flp") {
79  maskName += fmt::format("_cru{}_{}", RDHUtils::getCRUID(rdh), RDHUtils::getEndPointID(rdh));
80  if (mFileFor != "cru") {
81  maskName += fmt::format("_lnk{}_feeid{}", RDHUtils::getLinkID(rdh), RDHUtils::getFEEID(rdh));
82  if (mFileFor != "link") {
83  throw std::runtime_error("invalid option provided for file grouping");
84  }
85  }
86  }
87  }
88  std::string outputFilename = o2::utils::Str::concat_string(outputDir, detName, maskName, ".raw");
89  return outputFilename;
90  };
91  //Registering links
92  for (const auto& metadataPair : mMapTopo2FEEmetadata) {
93  const auto& rdh = metadataPair.second;
94  const auto outputFilename = makeFilename(rdh);
95  mWriter.registerLink(RDHUtils::getFEEID(rdh), RDHUtils::getCRUID(rdh), RDHUtils::getLinkID(rdh), RDHUtils::getEndPointID(rdh), outputFilename);
96  }
97  //Processing digits into raw data
98  TFile* inputFile = TFile::Open(filenameDigits.c_str());
99  assert(inputFile != nullptr);
100  LOG(info) << "Source file: " << filenameDigits;
101  TTree* inputTree = dynamic_cast<TTree*>(inputFile->Get("o2sim"));
102  DigitBlockFIT_t::processDigitBlocks(inputTree, *this);
103  delete inputTree;
104  inputFile->Close();
105  delete inputFile;
106  }
107  void processDigitBlockPerTF(const std::vector<DigitBlockFIT_t>& vecDigitBlock) // Is used in DigitBlockFIT_t::processDigitBlocks for each TF (TTree entry)
108  {
109  for (const auto& digitBlock : vecDigitBlock) {
110  //Processing PM data
111  auto mapDataBlockPM = digitBlock.template decomposeDigits<DataBlockPM_t>();
112  if (mVerbosity > 0) {
113  digitBlock.print();
114  }
115  for (const auto& dataBlockPair : mapDataBlockPM) {
116  const auto& topo = dataBlockPair.first;
117  const auto& dataBlock = dataBlockPair.second;
118  const auto itRdh = mMapTopo2FEEmetadata.find(topo);
119  if (itRdh == mMapTopo2FEEmetadata.end()) {
120  LOG(warning) << "No CRU entry in map! Data block: ";
121  dataBlock.print();
122  continue;
123  }
124  if (mVerbosity > 0) {
125  dataBlock.print();
126  }
127  const auto& rdh = itRdh->second;
128  auto data = dataBlock.serialize();
129  mWriter.addData(RDHUtils::getFEEID(rdh), RDHUtils::getCRUID(rdh), RDHUtils::getLinkID(rdh), RDHUtils::getEndPointID(rdh), dataBlock.getInteractionRecord(), data);
130  }
131  //Processing TCM data
132  const auto dataBlockPair = digitBlock.template decomposeDigits<DataBlockTCM_t>();
133  const auto& topo = dataBlockPair.first;
134  const auto& dataBlock = dataBlockPair.second;
135  const auto itRdh = mMapTopo2FEEmetadata.find(topo);
136  if (itRdh == mMapTopo2FEEmetadata.end()) {
137  LOG(warning) << "No CRU entry in map! Data block: ";
