170 class GBTBareDecoderImplV1
171 {
172  public:
173  GBTBareDecoderImplV1(uint16_t feeId, bool isDebugMode = false, uint8_t mask = 0xFF, const ElectronicsDelay& electronicsDelay = ElectronicsDelay()) : mIsDebugMode(isDebugMode), mMask(mask)
174  {
176  mElinkManager.init(feeId, isDebugMode, true, electronicsDelay);
177  auto crateId = crateparams::getCrateIdFromGBTUniqueId(feeId);
178  auto offset = 8 * crateparams::getGBTIdInCrate(feeId);
179  for (int ilink = 0; ilink < 10; ++ilink) {
180  mBoardUniqueIds.emplace_back(raw::makeUniqueLocID(crateId, offset + ilink % 8));
181  }
182  }
183 
184  void operator()(gsl::span<const uint8_t> payload, uint32_t orbit, std::vector<ROBoard>& data, std::vector<ROFRecord>& rofs)
185  {
187  mElinkManager.set(orbit);
188  mData = &data;
189  mROFs = &rofs;
190 
191  uint8_t byte = 0;
192  size_t ilink = 0, linkMask = 0, byteOffset = 0;
193 
194  for (int ireg = 0; ireg < 2; ++ireg) {
195  byteOffset = 5 * ireg;
196  if (mIsDebugMode) {
197  ilink = 8 + ireg;
198  linkMask = (1 << ilink);
199  for (size_t idx = byteOffset + 4; idx < payload.size(); idx += 16) {
200  byte = payload[idx];
201  if ((mIsFeeding & linkMask) || byte) {
202  processRegDebug(ilink, byte);
203  }
204  }
205  }
206  for (int ib = 0; ib < 4; ++ib) {
207  ilink = ib + 4 * ireg;
208  linkMask = (1 << ilink);
209  if (mMask & linkMask) {
210  for (size_t idx = byteOffset + ib; idx < payload.size(); idx += 16) {
211  byte = payload[idx];
212  if ((mIsFeeding & linkMask) || byte) {
213  processLoc(ilink, byte);
214  }
215  }
216  }
217  }
218  }
219  }
220 
221  private:
222  void processLoc(size_t ilink, uint8_t byte)
223  {
225  auto& decoder = mElinkManager.getDecoder(mBoardUniqueIds[ilink], true);
226  if (decoder.getNBytes() > 0) {
227  decoder.addAndComputeSize(byte);
228  if (decoder.isComplete()) {
229  mElinkManager.onDone(decoder, mBoardUniqueIds[ilink], *mData, *mROFs);
230  decoder.reset();
231  mIsFeeding &= (~(1 << ilink));
232  }
233  } else if ((byte & (raw::sSTARTBIT | raw::sCARDTYPE)) == (raw::sSTARTBIT | raw::sCARDTYPE)) {
234  decoder.add(byte);
235  mIsFeeding |= (1 << ilink);
236  }
237  }
238 
239  void processRegDebug(size_t ilink, uint8_t byte)
240  {
242  auto& decoder = mElinkManager.getDecoder(mBoardUniqueIds[ilink], false);
243  if (decoder.getNBytes() > 0) {
244  decoder.add(byte);
245  if (decoder.isComplete()) {
246  mElinkManager.onDone(decoder, mBoardUniqueIds[ilink], *mData, *mROFs);
247  decoder.reset();
248  mIsFeeding &= (~(1 << ilink));
249  }
250  } else if (byte & raw::sSTARTBIT) {
251  decoder.add(byte);
252  mIsFeeding |= (1 << ilink);
253  }
254  }

257  uint8_t mMask{0xFF};
258  uint16_t mIsFeeding{0};

381 
382  for (int ireg = 0; ireg < 2; ++ireg) {
383  size_t byteOffset = 5 * ireg;
384 
385  if (mIsDebugMode) {
386  // Treat regional cards
387  size_t ilink = 8 + ireg;
388  mLinkPayload.clear();
389  for (size_t idx = byteOffset + 4, end = payload.size(); idx < end; idx += 16) {
390  mLinkPayload.emplace_back(payload[idx]);

402  }
403  }
404  }
405 
406  // Treat local cards

408  size_t ilink = ib + 4 * ireg;
409  if (mMask & (1 << ilink)) {
410  mLinkPayload.clear();
411  for (size_t idx = byteOffset + ib, end = payload.size(); idx < end; idx += 16) {
412  mLinkPayload.emplace_back(payload[idx]);

415  auto end = mLinkPayload.end();
