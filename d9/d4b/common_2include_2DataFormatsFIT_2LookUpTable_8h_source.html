58 struct HasherCRU {
59  // Hash-function without any collisions due to technical bit size of fields:
60  // RDH::EndPointID : 4 bits
61  // RDH::LinkID : 8 bits
62  std::size_t operator()(const EntryCRU& entryCRU) const
63  {
64  return (entryCRU.mLinkID << 4) | entryCRU.mEndPointID;
65  }

94 struct HasherPM {
95  // Hash-function without any collisions due to technical bit size of fields:
96  // RDH::EndPointID : 4 bits
97  // EventData::ChannelID : 4 bits
98  // RDH::LinkID : 8 bits
99  std::size_t operator()(const EntryPM& entryPM) const
100  {
101  return (entryPM.mEntryCRU.mLinkID << 8) | (entryPM.mLocalChannelID << 4) | (entryPM.mEntryCRU.mEndPointID);
102  }
