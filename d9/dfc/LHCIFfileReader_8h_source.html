44 template <typename T>
45 void LHCIFfileReader::readValue(const std::string& alias, std::string& type, int& nele, int& nmeas, std::vector<std::pair<long, std::vector<T>>>& meas)
46 {
47  // look for value 'value' in the string from the LHC
48 
49  auto posStart = mFileBuffStr.find(alias);
50  if (posStart == std::string::npos) {
51  LOG(info) << alias << " not found in LHC IF file";
52  return;
53  }
54  auto posEnd = mFileBuffStr.find("\n", posStart);
55  LOG(debug) << "posStart = " << posStart << ", posEnd = " << posEnd;
56  if (posEnd == std::string::npos) {
57  posEnd = mFileBuffStr.size();
58  }
59  std::string subStr = mFileBuffStr.substr(posStart, posEnd - posStart);
60  LOG(debug) << "subStr = " << subStr;
61  auto tokensStr = o2::utils::Str::tokenize(subStr, '\t');
62  LOG(debug) << "size of tokensStr = " << tokensStr.size();
63  if (tokensStr.size() < 5) {
64  LOG(fatal) << "Number of tokens too small: " << tokensStr.size() << ", should be at 5 (alias, type, nelements, value(s), timestamp(s)";
65  }
66  auto tokensStr_type = o2::utils::Str::tokenize(tokensStr[1], ':');
67  LOG(debug) << "size of tokensStr_type = " << tokensStr_type.size();
68 
69  type = tokensStr_type[0];
70  LOG(debug) << "type = " << type;
71 
72  nele = std::stoi(tokensStr_type[1]); // number of elements per measurement
73  nmeas = std::stoi(tokensStr[2]); // number of measurements
74  LOG(debug) << "nele = " << nele << ", nmeas = " << nmeas;
75  int shift = 3; // number of tokens that are not measurments (alias, type, number of measurements)
76  if ((tokensStr.size() - shift) != (nele + 1) * nmeas) { // +1 to account for the timestamp
77  LOG(fatal) << "Wrong number of pairs (value(s), timestamp): " << tokensStr.size() - 3 << ", should be " << (nele + 1) * nmeas;
78  }
79  meas.reserve(nmeas);
80 
81  for (int idx = 0; idx < nmeas; ++idx) {
82  std::vector<T> vect;
83  vect.reserve(nele);
84  if constexpr (std::is_same<T, int32_t>::value) {
85  if (type == "i" || type == "b") {
86  for (int iele = 0; iele < nele; ++iele) {
87  LOG(debug) << alias << ": value int/bool = " << tokensStr[shift + iele];
88  vect.emplace_back(std::stoi(tokensStr[shift + iele]));
89  }
90  } else {
91  LOG(fatal) << "templated function called with wrong type, should be int32_t or bool, but it is " << type;
92  }
93  } else if constexpr (std::is_same<T, float>::value) {
94  if (type == "f") {
95  for (int iele = 0; iele < nele; ++iele) {
96  LOG(debug) << alias << ": value float = " << tokensStr[shift + iele];
97  vect.emplace_back(std::stof(tokensStr[shift + iele]));
98  }
99  } else {
100  LOG(fatal) << "templated function called with wrong type, should be float";
101  }
102  }
103 
104  else if constexpr (std::is_same<T, std::string>::value) {
105  if (type == "s") {
106  for (int iele = 0; iele < nele; ++iele) {
107  LOG(debug) << alias << ": value string = " << tokensStr[shift + iele];
108  vect.emplace_back(tokensStr[shift + iele]);
109  }
110  } else {
111  LOG(fatal) << "templated function called with wrong type, should be string";
112  }
113  }
114 
115  LOG(debug) << "timestamp = " << std::stof(tokensStr[shift + nele]);
116  meas.emplace_back(std::stol(tokensStr[shift + nele]) * 1e6, vect); // measurement comes in seconds, we want it in microseconds
117  }
118 }
