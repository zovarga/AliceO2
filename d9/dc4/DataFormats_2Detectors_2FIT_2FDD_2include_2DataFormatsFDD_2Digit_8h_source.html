32 struct Triggers {
33  enum { bitA,
34  bitC,
35  bitVertex,
36  bitCen,
37  bitSCen };
38  uint8_t triggersignals = 0; // FDD trigger signals
39  int8_t nChanA = 0; // number of fired channels A side
40  int8_t nChanC = 0; // number of fired channels A side
41  int32_t amplA = -1024; // sum amplitude A side
42  int32_t amplC = -1024; // sum amplitude C side
43  int16_t timeA = 0; // average time A side
44  int16_t timeC = 0; // average time C side
45  Triggers() = default;
46  Triggers(uint8_t signals, int8_t chanA, int8_t chanC, int32_t aamplA, int32_t aamplC, int16_t atimeA, int16_t atimeC)
47  {
48  triggersignals = signals;
49  nChanA = chanA;
50  nChanC = chanC;
51  amplA = aamplA;
52  amplC = aamplC;
53  timeA = atimeA;
54  timeC = atimeC;
55  }
56 
57  bool getOrA() const { return (triggersignals & (1 << bitA)) != 0; }
58  bool getOrC() const { return (triggersignals & (1 << bitC)) != 0; }
59  bool getVertex() const { return (triggersignals & (1 << bitVertex)) != 0; }
60  bool getCen() const { return (triggersignals & (1 << bitCen)) != 0; }
61  bool getSCen() const { return (triggersignals & (1 << bitSCen)) != 0; }
62 
63  void cleanTriggers()
64  {
65  triggersignals = 0;
66  nChanA = nChanC = 0;
67  amplA = amplC = 0;
68  timeA = timeC = 0;
69  }
70  Triggers getTriggers();
71  bool operator==(Triggers const& other) const
72  {
73  return std::tie(triggersignals, nChanA, nChanC, amplA, amplC, timeA, timeC) ==
74  std::tie(other.triggersignals, other.nChanA, other.nChanC, other.amplA, other.amplC, other.timeA, other.timeC);
75  }
76  void printLog() const
77  {
78  LOG(info) << "mTrigger: " << static_cast<uint16_t>(triggersignals);
79  LOG(info) << "nChanA: " << static_cast<uint16_t>(nChanA) << " | nChanC: " << static_cast<uint16_t>(nChanC);
80  LOG(info) << "amplA: " << amplA << " | amplC: " << amplC;
81  LOG(info) << "timeA: " << timeA << " | timeC: " << timeC;
82  }
83  ClassDefNV(Triggers, 1);
84 };

105 struct Digit {
106  static constexpr char sChannelNameDPL[] = "DIGITSBC";
107  static constexpr char sDigitName[] = "Digit";
108  static constexpr char sDigitBranchName[] = "FDDDigit";
109  o2::dataformats::RangeRefComp<5> ref;
110  Triggers mTriggers; // pattern of triggers in this BC
111  o2::InteractionRecord mIntRecord; // Interaction record (orbit, bc)
112  Digit() = default;
113  Digit(int first, int ne, o2::InteractionRecord iRec, Triggers chTrig) : ref(first, ne), mIntRecord(iRec), mTriggers(chTrig) {}
114  typedef DetTrigInput DetTrigInput_t;
115  uint32_t getOrbit() const { return mIntRecord.orbit; }
116  uint16_t getBC() const { return mIntRecord.bc; }
117  o2::InteractionRecord getIntRecord() const { return mIntRecord; };
118  void setIntRecord(const o2::InteractionRecord& intRec) { mIntRecord = intRec; }
119  gsl::span<const ChannelData> getBunchChannelData(const gsl::span<const ChannelData> tfdata) const
120  {
121  return ref.getEntries() ? gsl::span<const ChannelData>(&tfdata[ref.getFirstEntry()], ref.getEntries()) : gsl::span<const ChannelData>();
122  }
123  DetTrigInput makeTrgInput() const { return DetTrigInput{mIntRecord, mTriggers.getOrA(), mTriggers.getOrC(), mTriggers.getVertex(), mTriggers.getCen(), mTriggers.getSCen()}; }
124  void fillTrgInputVec(std::vector<DetTrigInput>& vecTrgInput) const
125  {
126  vecTrgInput.emplace_back(mIntRecord, mTriggers.getOrA(), mTriggers.getOrC(), mTriggers.getVertex(), mTriggers.getCen(), mTriggers.getSCen());
127  }
128  bool operator==(const Digit& other) const
129  {
130  return std::tie(ref, mTriggers, mIntRecord) == std::tie(other.ref, other.mTriggers, other.mIntRecord);
131  }
132  void printLog() const
133  {
134  LOG(info) << "______________DIGIT DATA____________";
135  LOG(info) << "BC: " << mIntRecord.bc << "| ORBIT: " << mIntRecord.orbit;
136  LOG(info) << "Ref first: " << ref.getFirstEntry() << "| Ref entries: " << ref.getEntries();
137  mTriggers.printLog();
138  }
139  ClassDefNV(Digit, 3);
140 };
