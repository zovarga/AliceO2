56  void init(InitContext& ic)
57  {
62  DigitIOBaseTask::init(ic);
63 
64  mNoFile = ic.options().get<bool>(OPTNAME_NO_FILE);
65  mBinary = not ic.options().get<bool>(OPTNAME_TXT);
66 
67  if (!mNoFile) {
68  auto outputFileName = ic.options().get<std::string>(OPTNAME_OUTFILE);
69  mStream = std::make_unique<std::fstream>(outputFileName, mBinary ? std::ios::out | std::ios::binary : std::ios::out);
70  if (mBinary) {
71  auto binaryFileFormat = ic.options().get<int>(OPTNAME_BINARY_FORMAT);
72  if (binaryFileFormat >= o2::mch::io::digitFileFormats.size()) {
73  throw std::invalid_argument(fmt::format("file version {} is unknown", binaryFileFormat));
74  }
75  auto maxsize = ic.options().get<int>(OPTNAME_MAX_SIZE);
76  LOGP(warn,
77  "Will dump binary information (version {}) up to a maximum size of {} KB",
78  binaryFileFormat, maxsize);
79  mWriter = std::make_unique<io::DigitWriter>(*mStream,
80  io::digitFileFormats[binaryFileFormat],
81  static_cast<size_t>(maxsize));
82  } else {
83  LOGP(warn, "Will dump textual information");
84  mWriter = std::make_unique<io::DigitWriter>(*mStream);
85  }
86  }
87  }

89  void writeOrbits(gsl::span<const o2::mch::OrbitInfo> orbits)
90  {
91  if (orbits.size() && !mBinary) {
92  std::set<OrbitInfo> orderedOrbits(orbits.begin(), orbits.end());
93  for (auto o : orderedOrbits) {
94  (*mStream) << " FEEID " << o.getFeeID() << " LINK " << (int)o.getLinkID() << " ORBIT " << o.getOrbit() << std::endl;
95  }
96  }
97  }

110  void run(ProcessingContext& pc)
111  {
112  gsl::span<o2::mch::OrbitInfo> voidOrbitInfos;
113 
114  auto digits = pc.inputs().get<gsl::span<Digit>>("digits");
115  auto rofs = pc.inputs().get<gsl::span<o2::mch::ROFRecord>>("rofs");
116  auto orbits = mWithOrbits ? pc.inputs().get<gsl::span<o2::mch::OrbitInfo>>("orbits") : voidOrbitInfos;
117 
118  if (shouldProcess()) {
119  incNofProcessedTFs();
120  printSummary(digits, rofs, fmt::format("{:4d} orbits", orbits.size()).c_str());
121  printFull(digits, rofs);
122  write(digits, rofs, orbits);
123  }
124  incTFid();
125  }

132  bool mBinary = true; // output is a binary file

163  auto commonOptions = o2::mch::io::getCommonOptions();
164  auto options = Options{

166  {OPTNAME_NO_FILE, VariantType::Bool, false, {"no output to file"}},
167  {OPTNAME_BINARY_FORMAT, VariantType::Int, 0, {"digit binary format to use"}},
168  {OPTNAME_TXT, VariantType::Bool, false, {"output digits in text format"}},
169  {OPTNAME_MAX_SIZE, VariantType::Int, std::numeric_limits<int>::max(), {"max output size (in KB)"}}};
170  options.insert(options.end(), commonOptions.begin(), commonOptions.end());
171 
172  DataProcessorSpec producer{
173  "mch-digits-writer",
174  Inputs{o2::framework::select(input.c_str())},
175  Outputs{},
176  AlgorithmSpec{adaptFromTask<DigitsSinkTask>(withOrbits)},
177  options};
178  specs.push_back(producer);
179 
180  return specs;
