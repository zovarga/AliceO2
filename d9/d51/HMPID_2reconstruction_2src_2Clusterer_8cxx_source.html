33 void Clusterer::Dig2Clu(std::vector<o2::hmpid::Digit>* mDigs, std::vector<o2::hmpid::Cluster>* mClus, float* pUserCut, bool isTryUnfold)
34 {
35  // Finds all clusters for a given digits list provided not empty. Currently digits list is a list of all digits for a single chamber.
36  // Puts all found clusters in separate lists, one per clusters.
37  // Arguments: pDigAll - list of digits for all chambers
38  // pCluAll - list of clusters for all chambers
39  // isTryUnfold - flag to choose between CoG and Mathieson fitting
40  // Returns: none
41 
42  TMatrixF padMap(Param::kMinPx, Param::kMaxPcx, Param::kMinPy, Param::kMaxPcy); // pads map for single chamber 0..159 x 0..143
43 
44  int pUsedDig = -1;
45  int padChX = 0, padChY = 0, module = 0;
46  o2::hmpid::Cluster* clu; // tmp cluster to be used as current
47 
48  for (int iCh = Param::kMinCh; iCh <= Param::kMaxCh; iCh++) { // chambers loop
49  padMap = (Float_t)-1; // reset map to -1 (means no digit for this pad)
50  for (int iDig = 0; iDig < mDigs->size(); iDig++) { // digits loop for current chamber
51  o2::hmpid::Digit::pad2Absolute(mDigs->at(iDig).getPadID(), &module, &padChX, &padChY);
52  if (module != iCh) {
53  continue;
54  }
55  padMap(padChX, padChY) = iDig; // fill the map for the given chamber, (padx,pady) cell takes digit index
56  } // digits loop for current chamber
57 
58  for (int iDig = 0; iDig < mDigs->size(); iDig++) { // digits loop for current chamber
59  o2::hmpid::Digit::pad2Absolute(mDigs->at(iDig).getPadID(), &module, &padChX, &padChY);
60  if (module != iCh) {
61  continue;
62  }
63  pUsedDig = UseDig(padChX, padChY, &padMap);
64  if (pUsedDig == -1) {
65  continue;
66  } // this digit is already taken in FormClu(), go after next digit
67  clu = new o2::hmpid::Cluster;
68  clu->setCh(iCh);
69  FormClu(clu, pUsedDig, mDigs, &padMap); // form cluster starting from this digit by recursion
70  clu->solve(mClus, pUserCut, isTryUnfold); // solve this cluster and add all unfolded clusters to provided list
71  // clu.reset(); //empty current cluster
72  } // digits loop for current chamber
73  } // chambers loop
74  return;
75 } // Dig2Clu()

77 void Clusterer::FormClu(Cluster* pClu, int pDig, std::vector<o2::hmpid::Digit>* pDigList, TMatrixF* pDigMap)
78 {
79  // Forms the initial cluster as a combination of all adjascent digits. Starts from the given digit then calls itself recursevly for all neighbours.
80  // Arguments: pClu - pointer to cluster being formed
81  // Returns: none
82  pClu->digAdd(&pDigList->at(pDig)); // take this digit in cluster
83  int cnt = 0;
84  int cx[4];
85  int cy[4];
86  int padChX = 0;
87  int padChY = 0;
88  int module = 0;
89  o2::hmpid::Digit::pad2Absolute(pDigList->at(pDig).getPadID(), &module, &padChX, &padChY);
90 
91  if (padChX > Param::kMinPx) {
92  cx[cnt] = padChX - 1;
93  cy[cnt] = padChY;
94  cnt++;
95  } // left
96  if (padChX < Param::kMaxPcx) {
97  cx[cnt] = padChX + 1;
98  cy[cnt] = padChY;
99  cnt++;
100  } // right
101  if (padChY > Param::kMinPy) {
102  cx[cnt] = padChX;
103  cy[cnt] = padChY - 1;
104  cnt++;
105  } // down
106  if (padChY < Param::kMaxPcy) {
107  cx[cnt] = padChX;
108  cy[cnt] = padChY + 1;
109  cnt++;
110  } // up
111 
112  for (int i = 0; i < cnt; i++) { // neighbours loop
113  pDig = UseDig(cx[i], cy[i], pDigMap);
114  if (pDig != -1) {
115  FormClu(pClu, pDig, pDigList, pDigMap);
116  } // check if this neighbour pad fired and mark it as taken
117  } // neighbours loop
118 } // FormClu()
