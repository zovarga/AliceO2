58  }

91 void RawToCellConverterSpec::run(framework::ProcessingContext& ctx)
92 {
93  // Cache cells from bunch crossings as the component reads timeframes from many links consecutively
94  std::vector<o2::InteractionRecord> irList;
95  std::vector<std::array<short, 56>> cellTRURanges; // start/end points for cells in mTmpCells[ddl] arrays
96  for (int iddl = 14; iddl--;) {
97  mTmpCells[iddl].clear();
98  mTmpTRU[iddl].clear();
99  }
100  mOutputHWErrors.clear();
101  if (mFillChi2) {
102  mOutputFitChi.clear();
103  }
104 
105  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload this means that the "delayed message"
106  // mechanism created it in absence of real data from upstream. Processor should send empty output to not block the workflow
107  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
108  std::vector<o2::framework::InputSpec> dummy{o2::framework::InputSpec{"dummy", o2::framework::ConcreteDataMatcher{"PHS", o2::header::gDataDescriptionRawData, 0xDEADBEEF}}};
109  for (const auto& ref : framework::InputRecordWalker(ctx.inputs(), dummy)) {
110  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
111  if (dh->payloadSize == 0) { // send empty output
112  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
113  if (++contDeadBeef <= maxWarn) {
114  LOGP(warning, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
115  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize,
116  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
117  }
118  mOutputCells.clear();
119  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLS", 0, o2::framework::Lifetime::Timeframe}, mOutputCells);
120  mOutputTriggerRecords.clear();
121  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLTRIGREC", 0, o2::framework::Lifetime::Timeframe}, mOutputTriggerRecords);
122  mOutputHWErrors.clear();
123  ctx.outputs().snapshot(o2::framework::Output{"PHS", "RAWHWERRORS", 0, o2::framework::Lifetime::Timeframe}, mOutputHWErrors);
124  if (mFillChi2) {
125  mOutputFitChi.clear();
126  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLFITQA", 0, o2::framework::Lifetime::QA}, mOutputFitChi);
127  }
128  return; //empty TF, nothing to process
129  }
130  }
131  contDeadBeef = 0; // if good data, reset the counter
132 
133  std::vector<o2::framework::InputSpec> inputFilter{o2::framework::InputSpec{"filter", o2::framework::ConcreteDataTypeMatcher{"PHS", "RAWDATA"}, o2::framework::Lifetime::Timeframe}};
134  for (const auto& rawData : framework::InputRecordWalker(ctx.inputs(), inputFilter)) {
135 
136  const gsl::span<const char>& rawmemory = o2::framework::DataRefUtils::as<const char>(rawData);
137  if (rawmemory.size() == 0) {
138  continue;
139  }
140 
141  o2::phos::RawReaderMemory rawreader(rawmemory);
142 
143  // loop over all the DMA pages
144  while (rawreader.hasNext()) {
145  try {
146  rawreader.next();
147  } catch (RawDecodingError::ErrorType_t e) {
148  // LOG(error) << "Raw decoding error " << (int)e;
149  //add error list
150  mOutputHWErrors.emplace_back(14, (int)e, 1); //Put general errors to non-existing DDL14
151  //if problem in header, abandon this page
152  if (e == RawDecodingError::ErrorType_t::PAGE_NOTFOUND ||
153  e == RawDecodingError::ErrorType_t::HEADER_DECODING ||
154  e == RawDecodingError::ErrorType_t::HEADER_INVALID) {
155  break;
156  }
157  //if problem in payload, try to continue
158  continue;
159  }
160  auto& header = rawreader.getRawHeader();
161  auto triggerBC = o2::raw::RDHUtils::getTriggerBC(header);
162  auto triggerOrbit = o2::raw::RDHUtils::getTriggerOrbit(header);
163  auto ddl = o2::raw::RDHUtils::getFEEID(header);
164 
165  if (ddl > o2::phos::Mapping::NDDL || ddl < 0) { //only 14 correct DDLs
166  LOG(error) << "DDL=" << ddl;
167  mOutputHWErrors.emplace_back(14, 16, char(ddl)); //Add non-existing DDL as DDL 15
168  continue; //skip STU ddl
169  }
170 
171  o2::InteractionRecord currentIR(triggerBC, triggerOrbit);
172  auto irIter = irList.rbegin();
173  auto rangeIter = cellTRURanges.rbegin();
174  while (irIter != irList.rend() && *irIter != currentIR) {
175  irIter++;
176  rangeIter++;
177  }
178  if (irIter != irList.rend()) { //found
179  (*rangeIter)[2 * ddl] = mTmpCells[ddl].size(); //start of the cell list
180  (*rangeIter)[28 + 2 * ddl] = mTmpTRU[ddl].size(); //start of the tru list
181  } else { //create new entry
182  irList.push_back(currentIR);
183  cellTRURanges.emplace_back();
184  cellTRURanges.back().fill(0);
185  cellTRURanges.back()[2 * ddl] = mTmpCells[ddl].size();
186  cellTRURanges.back()[28 + 2 * ddl] = mTmpTRU[ddl].size();
187  rangeIter = cellTRURanges.rbegin();
188  }
189  std::vector<Cell>& currentCellContainer = mTmpCells[ddl];
190  std::vector<Cell>& currentTRUContainer = mTmpTRU[ddl];
191 
192  // use the altro decoder to decode the raw data, and extract the RCU trailer
193  mDecoder->decode(rawreader, mRawFitter.get(), currentCellContainer, currentTRUContainer);
194  const std::vector<o2::phos::RawReaderError>& errs = mDecoder->hwerrors();
195  for (auto a : errs) {
196  mOutputHWErrors.emplace_back(a);
197  }
198  // Sort cells according to cell ID
199  (*rangeIter)[2 * ddl + 1] = currentCellContainer.size();
200  auto itBegin = currentCellContainer.begin() + (*rangeIter)[2 * ddl];
201  std::sort(itBegin, currentCellContainer.end(), [](o2::phos::Cell& lhs, o2::phos::Cell& rhs) { return lhs.getAbsId() < rhs.getAbsId(); });
202  auto itTrBegin = currentTRUContainer.begin() + (*rangeIter)[28 + 2 * ddl];
203  (*rangeIter)[28 + 2 * ddl + 1] = currentTRUContainer.size();
204  std::sort(itTrBegin, currentTRUContainer.end(), [](o2::phos::Cell& lhs, o2::phos::Cell& rhs) { return lhs.getAbsId() < rhs.getAbsId(); });
205  } //RawReader::hasNext
206  }
207 
208  // Loop over BCs, sort cells with increasing cell ID and write to output containers
209  mOutputCells.clear();
210  if (mLastSize > 0) {
211  mOutputCells.reserve(mLastSize);
212  }
213  mOutputTriggerRecords.clear();
214  auto rangeIter = cellTRURanges.begin();
215  for (auto irIter = irList.begin(); irIter != irList.end(); ++irIter, ++rangeIter) {
216  // find all DDLs for current BC
217  // sort separately then concatenate
218  int prevCellSize = mOutputCells.size();
219  for (int iddl = 0; iddl < 14; iddl++) {
220  auto cbegin = mTmpCells[iddl].begin() + (*rangeIter)[2 * iddl];
221  auto cend = mTmpCells[iddl].begin() + (*rangeIter)[2 * iddl + 1];
222 
223  if (mCombineGHLG && !mPedestalRun) { // combine for normal data, do not combine e.g. for LED run and pedestal
224  //Combine HG and LG sells
225  //Should be next to each other after sorting
226  auto it1 = cbegin;
227  auto it2 = cbegin;
228  it2++;
229  while (it1 != cend) {
230  if (it2 != cend) {
231  if ((*it1).getAbsId() == (*it2).getAbsId()) { //HG and LG channels, if both, copy only HG as more precise
232  if ((*it1).getType() == o2::phos::HIGH_GAIN) {
233  mOutputCells.push_back(*it1);
234  } else {
235  mOutputCells.push_back(*it2);
236  }
237  ++it1; //yes increase twice
238  ++it2;
239  } else { //no double cells, copy this one
240  mOutputCells.push_back(*it1);
241  }
242  } else { //just copy last one
243  mOutputCells.push_back(*it1);
244  }
245  ++it1;
246  ++it2;
247  }
248  } else {
249  mOutputCells.insert(mOutputCells.end(), cbegin, cend);
250  }
251  } //all readout cells
252  for (int iddl = 0; iddl < 14; iddl++) {
253  auto trbegin = mTmpTRU[iddl].begin() + (*rangeIter)[28 + 2 * iddl];
254  auto trend = mTmpTRU[iddl].begin() + (*rangeIter)[28 + 2 * iddl + 1];
255  //Move trigger cells
256  for (auto tri = trbegin; tri != trend; tri++) {
257  if (tri->getEnergy() > 0) {
258  mOutputCells.emplace_back(tri->getAbsId(), tri->getEnergy(), tri->getTime(), tri->getType());
259  }
260  }
261  }
262 
263  mOutputTriggerRecords.emplace_back(*irIter, prevCellSize, mOutputCells.size() - prevCellSize);
264  }
265 
266  mLastSize = 1.1 * mOutputCells.size();
267 
268  LOG(debug) << "[PHOSRawToCellConverter - run] Writing " << mOutputCells.size() << " cells ...";
269  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLS", 0, o2::framework::Lifetime::Timeframe}, mOutputCells);
270  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLTRIGREC", 0, o2::framework::Lifetime::Timeframe}, mOutputTriggerRecords);
271  ctx.outputs().snapshot(o2::framework::Output{"PHS", "RAWHWERRORS", 0, o2::framework::Lifetime::Timeframe}, mOutputHWErrors);
272  if (mFillChi2) {
273  ctx.outputs().snapshot(o2::framework::Output{"PHS", "CELLFITQA", 0, o2::framework::Lifetime::QA}, mOutputFitChi);
274  }
275 }

277 o2::framework::DataProcessorSpec o2::phos::reco_workflow::getRawToCellConverterSpec(int flpId)
278 {
279  std::vector<o2::framework::InputSpec> inputs;
280  inputs.emplace_back("RAWDATA", o2::framework::ConcreteDataTypeMatcher{"PHS", "RAWDATA"}, o2::framework::Lifetime::Optional);
281  //receive at least 1 guaranteed input (which will allow to acknowledge the TF)
282  inputs.emplace_back("STFDist", "FLP", "DISTSUBTIMEFRAME", 0, o2::framework::Lifetime::Timeframe);
283 
284  std::vector<o2::framework::OutputSpec> outputs;
285  outputs.emplace_back("PHS", "CELLS", flpId, o2::framework::Lifetime::Timeframe);
286  outputs.emplace_back("PHS", "CELLTRIGREC", flpId, o2::framework::Lifetime::Timeframe);
287  outputs.emplace_back("PHS", "RAWHWERRORS", flpId, o2::framework::Lifetime::Timeframe);
288  outputs.emplace_back("PHS", "CELLFITQA", flpId, o2::framework::Lifetime::QA);
289 
290  return o2::framework::DataProcessorSpec{"PHOSRawToCellConverterSpec",
291  inputs, // o2::framework::select("A:PHS/RAWDATA"),
292  outputs,
293  o2::framework::adaptFromTask<o2::phos::reco_workflow::RawToCellConverterSpec>(),
294  o2::framework::Options{
295  {"fitmethod", o2::framework::VariantType::String, "default", {"Fit method (default or semigaus)"}},
296  {"mappingpath", o2::framework::VariantType::String, "", {"Path to mapping files"}},
297  {"fillchi2", o2::framework::VariantType::String, "off", {"Fill sample qualities on/off"}},
298  {"keepHGLG", o2::framework::VariantType::String, "off", {"keep HighGain and Low Gain signals on/off"}},
299  {"pedestal", o2::framework::VariantType::String, "off", {"Analyze as pedestal run on/off"}}}};
