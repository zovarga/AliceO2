69  auto processSectorFunction = [processAttributes](ProcessingContext& pc, DataRef const& ref, DataRef const& mclabelref) {
70  auto& decoder = processAttributes->decoder;
71  auto& verbosity = processAttributes->verbosity;
72  auto& activeInputs = processAttributes->activeInputs;
73  // this will return a span of TPC clusters
74  auto size = o2::framework::DataRefUtils::getPayloadSize(ref);
75  auto const* dataHeader = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
76  o2::header::DataHeader::SubSpecificationType fanSpec = dataHeader->subSpecification;
77 
78  // init the stacks for forwarding the sector header
79  // FIXME check if there is functionality in the DPL to forward the stack
80  // FIXME make one function
81  o2::header::Stack rawHeaderStack;
82  o2::header::Stack mcHeaderStack;
83  o2::tpc::TPCSectorHeader const* sectorHeaderMC = nullptr;
84  if (DataRefUtils::isValid(mclabelref)) {
85  sectorHeaderMC = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(mclabelref);
86  if (sectorHeaderMC) {
87  o2::header::Stack actual{*sectorHeaderMC};
88  std::swap(mcHeaderStack, actual);
89  if (sectorHeaderMC->sector() < 0) {
90  pc.outputs().snapshot(Output{gDataOriginTPC, DataDescription("CLNATIVEMCLBL"), fanSpec, Lifetime::Timeframe, std::move(mcHeaderStack)}, fanSpec);
91  }
92  }
93  }
94  auto const* sectorHeader = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(ref);
95  if (sectorHeader) {
96  o2::header::Stack actual{*sectorHeader};
97  std::swap(rawHeaderStack, actual);
98  if (sectorHeader->sector() < 0) {
99  pc.outputs().snapshot(Output{gDataOriginTPC, DataDescription("CLUSTERNATIVE"), fanSpec, Lifetime::Timeframe, std::move(rawHeaderStack)}, fanSpec);
100  return;
101  }
102  }
103  assert(sectorHeaderMC == nullptr || sectorHeader->sector() == sectorHeaderMC->sector());
104 
105  // input to the decoder is a vector of raw pages description ClusterHardwareContainer,
106  // each specified as a pair of pointer to ClusterHardwareContainer and the number
107  // of pages in that buffer
108  // FIXME: better description of the raw page
109  size_t nPages = size / 8192;
110  std::vector<std::pair<const ClusterHardwareContainer*, std::size_t>> inputList;
111  if (verbosity > 0 && !DataRefUtils::isValid(mclabelref)) {
112  LOG(info) << "Decoder input: " << size << ", " << nPages << " pages for sector " << sectorHeader->sector();
113  }
114 
115  // MC labels are received as one container of labels in the sequence matching clusters
116  // in the raw pages
117  std::vector<ConstMCLabelContainer> mcinCopiesFlat;
118  std::vector<ConstMCLabelContainerView> mcinCopiesFlatView;
119  ConstMCLabelContainerView mcin;
120  if (DataRefUtils::isValid(mclabelref)) {
121  mcin = pc.inputs().get<gsl::span<char>>(mclabelref);
122  mcinCopiesFlat.resize(nPages);
123  mcinCopiesFlatView.reserve(nPages);
124  if (verbosity > 0) {
125  LOG(info) << "Decoder input: " << size << ", " << nPages << " pages, " << mcin.getIndexedSize() << " MC label sets for sector " << sectorHeader->sector();
126  }
127  }
128 
129  // FIXME: the decoder takes vector of MCLabelContainers as input and the retreived
130  // input can not be moved because the input is const, so we have to copy
131  // Furthermore, the current decoder implementation supports handling of MC labels
132  // only for single 8kb pages. So we have to add the raw pages individually and create
133  // MC label containers for the corresponding clusters.
134  size_t mcinPos = 0;
135  size_t totalNumberOfClusters = 0;
136  for (size_t page = 0; page < nPages; page++) {
137  MCLabelContainer mcinCopy;
138  inputList.emplace_back(reinterpret_cast<const ClusterHardwareContainer*>(ref.payload + page * 8192), 1);
139  const ClusterHardwareContainer& container = *(inputList.back().first);
140  if (verbosity > 1) {
141  LOG(info) << "Decoder input in page " << std::setw(2) << page << ": " //
142  << "CRU " << std::setw(3) << container.CRU << " " //
143  << std::setw(3) << container.numberOfClusters << " cluster(s)"; //
144  }
145  totalNumberOfClusters += container.numberOfClusters;
146  if (mcin.getBuffer().size()) {
147  for (size_t mccopyPos = 0;
148  mccopyPos < container.numberOfClusters && mcinPos < mcin.getIndexedSize();
149  mccopyPos++, mcinPos++) {
150  for (auto const& label : mcin.getLabels(mcinPos)) {
151  mcinCopy.addElement(mccopyPos, label);
152  }
153  }
154  }
155  mcinCopy.flatten_to(mcinCopiesFlat[page]);
156  mcinCopiesFlatView.emplace_back(mcinCopiesFlat[page]);
157  }
158  // FIXME: introduce error handling policy: throw, ignore, warn
159  //assert(!mcin || mcinPos == mcin->getIndexedSize());
160  if (mcin.getBuffer().size() && mcinPos != totalNumberOfClusters) {
161  LOG(error) << "inconsistent number of MC label objects processed"
162  << ", expecting MC label objects for " << totalNumberOfClusters << " cluster(s)"
163  << ", got " << mcin.getIndexedSize();
164  }
165  // output of the decoder is sorted in (sector,globalPadRow) coordinates, individual
166  // containers are created for clusters and MC labels per (sector,globalPadRow) address
167  char* outputBuffer = nullptr;
168  auto outputAllocator = [&pc, &fanSpec, &outputBuffer, &rawHeaderStack](size_t size) -> char* {
169  outputBuffer = pc.outputs().newChunk(Output{gDataOriginTPC, DataDescription("CLUSTERNATIVE"), fanSpec, Lifetime::Timeframe, std::move(rawHeaderStack)}, size).data();
170  return outputBuffer;
171  };
172  MCLabelContainer mcout;
173  decoder->decodeClusters(inputList, outputAllocator, (mcin.getBuffer().size() ? &mcinCopiesFlatView : nullptr), &mcout);
174 
175  // TODO: reestablish the logging messages on the raw buffer
176  // if (verbosity > 1) {
177  // LOG(info) << "decoder " << std::setw(2) << sectorHeader->sector() //
178  // << ": decoded " << std::setw(4) << coll.clusters.size() << " clusters on sector " //
179  // << std::setw(2) << (int)coll.sector << "[" << (int)coll.globalPadRow << "]"; //
180  // }
181 
182  if (DataRefUtils::isValid(mclabelref)) {
183  if (verbosity > 0) {
184  LOG(info) << "sending " << mcout.getIndexedSize()
185  << " label object(s)" << std::endl;
186  }
187  // serialize the complete list of MC label containers
188  ConstMCLabelContainer labelsFlat;
189  mcout.flatten_to(labelsFlat);
190  pc.outputs().snapshot(Output{gDataOriginTPC, DataDescription("CLNATIVEMCLBL"), fanSpec, Lifetime::Timeframe, std::move(mcHeaderStack)}, labelsFlat);
191  }

194  auto processingFct = [processAttributes, processSectorFunction](ProcessingContext& pc) {
195  struct SectorInputDesc {
196  DataRef dataref;
197  DataRef mclabelref;
198  };
199  // loop over all inputs and their parts and associate data with corresponding mc truth data
200  // by the subspecification
201  std::map<int, SectorInputDesc> inputs;
202  std::vector<InputSpec> filter = {
203  {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "CLUSTERHW"}, Lifetime::Timeframe},
204  {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "CLUSTERHWMCLBL"}, Lifetime::Timeframe},
205  };
206  for (auto const& inputRef : InputRecordWalker(pc.inputs(), filter)) {
207  auto const* sectorHeader = DataRefUtils::getHeader<o2::tpc::TPCSectorHeader*>(inputRef);
208  if (sectorHeader == nullptr) {
209  LOG(error) << "sector header missing on header stack for input on " << inputRef.spec->binding;
210  continue;
211  }
212  const int sector = sectorHeader->sector();
213  if (DataRefUtils::match(inputRef, {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "CLUSTERHW"}})) {
214  inputs[sector].dataref = inputRef;
215  }
216  if (DataRefUtils::match(inputRef, {"check", ConcreteDataTypeMatcher{gDataOriginTPC, "CLUSTERHWMCLBL"}})) {
217  inputs[sector].mclabelref = inputRef;
218  }
219  }
220  for (auto const& input : inputs) {
221  if (processAttributes->sendMC && !DataRefUtils::isValid(input.second.mclabelref)) {
222  throw std::runtime_error("missing the required MC label data for sector " + std::to_string(input.first));
223  }
224  processSectorFunction(pc, input.second.dataref, input.second.mclabelref);
225  }
226  };
227 
228  // return the actual processing function as a lambda function using variables
229  // of the init function
230  return processingFct;
231  };

236  };
237  if (makeMcInput) {

239  constexpr o2::header::DataDescription datadesc("CLUSTERHWMCLBL");
240  inputSpecs.emplace_back(InputSpec{"mclblin", gDataOriginTPC, datadesc, 0, Lifetime::Timeframe});
241  }
242  return std::move(inputSpecs);
243  };
244 
245  auto createOutputSpecs = [](bool makeMcOutput) {
246  std::vector<OutputSpec> outputSpecs{
247  OutputSpec{{"clusterout"}, gDataOriginTPC, "CLUSTERNATIVE", 0, Lifetime::Timeframe},
248  };
249  if (makeMcOutput) {
250  OutputLabel label{"mclblout"};
251  // have to use a new data description, routing is only based on origin and decsription
252  constexpr o2::header::DataDescription datadesc("CLNATIVEMCLBL");
253  outputSpecs.emplace_back(label, gDataOriginTPC, datadesc, 0, Lifetime::Timeframe);
254  }
255  return std::move(outputSpecs);
256  };
257 
258  return DataProcessorSpec{processorName,
259  {createInputSpecs(sendMC)},
260  {createOutputSpecs(sendMC)},
261  AlgorithmSpec(initFunction)};
