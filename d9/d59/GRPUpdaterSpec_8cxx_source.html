56  void run(framework::ProcessingContext& pc)
57  {
58  const std::string grpName{o2::base::NameConf::CCDBOBJECT};
59 
60  // a standardized semaphore convention --> taking the current execution path should be enough
61  // (the user enables this via O2_USEGRP_SEMA environment)
62  bool use_sema = false;
63  boost::interprocess::named_semaphore* sem = nullptr;
64  std::string semhashedstring;
65  try {
66  const auto semname = std::filesystem::current_path().string() + mGRPFileName;
67  std::hash<std::string> hasher;
68  semhashedstring = "alice_grp_" + std::to_string(hasher(semname)).substr(0, 16);
69  sem = new boost::interprocess::named_semaphore(boost::interprocess::open_or_create_t{}, semhashedstring.c_str(), 1);
70  } catch (std::exception e) {
71  LOG(warn) << "Could not setup GRP semaphore; Continuing without";
72  sem = nullptr;
73  }
74  try {
75  if (sem) {
76  sem->wait(); // wait until we can enter (no one else there)
77  }
78 
79  auto postSem = [sem, &semhashedstring] {
80  if (sem) {
81  sem->post();
82  if (sem->try_wait()) {
83  // if nobody else is waiting remove the semaphore resource
84  sem->post();
85  boost::interprocess::named_semaphore::remove(semhashedstring.c_str());
86  }
87  delete sem;
88  }
89  };
90 
91  TFile flGRP(mGRPFileName.c_str(), "update");
92  if (flGRP.IsZombie()) {
93  LOG(error) << "Failed to open in update mode " << mGRPFileName;
94  postSem();
95  return;
96  }
97  std::unique_ptr<GRP> grp(static_cast<GRP*>(flGRP.GetObjectChecked(grpName.c_str(), GRP::Class())));
98  for (auto det : sDetList) { // get readout mode data from different detectors
99  auto roMode = pc.inputs().get<o2::parameters::GRPObject::ROMode>(det.getName());
100  if (!(roMode & o2::parameters::GRPObject::PRESENT)) {
101  LOG(error) << "Detector " << det.getName() << " is read out while processor set ABSENT";
102  continue;
103  }
104  grp->setDetROMode(det, roMode);
105  }
106  grp->setIsMC(true);
107  grp->setFirstOrbit(o2::raw::HBFUtils::Instance().orbitFirst);
108  grp->setNHBFPerTF(o2::raw::HBFUtils::Instance().nHBFPerTF);
109  LOG(info) << "Updated GRP in " << mGRPFileName << " for detectors RO mode and 1st orbit of the run";
110  grp->print();
111  flGRP.WriteObjectAny(grp.get(), grp->Class(), grpName.c_str());
112  flGRP.Close();
113 
114  postSem();
115  } catch (boost::interprocess::interprocess_exception e) {
116  LOG(error) << "Caught semaphore exception " << e.what();
117  }
118  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
119  }
120 
121  private:
122  std::string mGRPFileName = "o2sim_grp.root";
123 };
