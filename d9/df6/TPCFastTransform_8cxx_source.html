157 int TPCFastTransform::writeToFile(std::string outFName, std::string name)
158 {
160  assert(isConstructed());
161 
162  if (outFName.empty()) {
163  outFName = "tpcFastTransform.root";
164  }
165  if (name.empty()) {
166  name = "TPCFastTransform";
167  }
168  TFile outf(outFName.data(), "recreate");
169  if (outf.IsZombie()) {
170  LOG(error) << "Failed to open output file " << outFName;
171  return -1;
172  }
173 
174  bool isBufferExternal = !isBufferInternal();
175  if (isBufferExternal) {
176  adoptInternalBuffer(mFlatBufferPtr);
177  }
178  outf.WriteObjectAny(this, Class(), name.data());
179  outf.Close();
180  if (isBufferExternal) {
181  clearInternalBufferPtr();
182  }
183  return 0;
184 }

186 TPCFastTransform* TPCFastTransform::loadFromFile(std::string inpFName, std::string name)
187 {
189 
190  if (inpFName.empty()) {
191  inpFName = "tpcFastTransform.root";
192  }
193  if (name.empty()) {
194  name = "TPCFastTransform";
195  }
196  TFile inpf(inpFName.data());
197  if (inpf.IsZombie()) {
198  LOG(error) << "Failed to open input file " << inpFName;
199  return nullptr;
200  }
201  TPCFastTransform* transform = reinterpret_cast<TPCFastTransform*>(inpf.GetObjectChecked(name.data(), TPCFastTransform::Class()));
202  if (!transform) {
203  LOG(error) << "Failed to load " << name << " from " << inpFName;
204  return nullptr;
205  }
206  if (transform->mFlatBufferSize > 0 && transform->mFlatBufferContainer == nullptr) {
207  LOG(error) << "Failed to load " << name << " from " << inpFName << ": empty flat buffer container";
208  return nullptr;
209  }
210  transform->setActualBufferAddress(transform->mFlatBufferContainer);
211  return transform;
212 }
