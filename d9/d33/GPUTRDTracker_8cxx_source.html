121 template <class TRDTRK, class PROP>
122 void GPUTRDTracker_t<TRDTRK, PROP>::InitializeProcessor()
123 {
124  //--------------------------------------------------------------------
125  // Initialise tracker
126  //--------------------------------------------------------------------
127  mGeo = (TRD_GEOMETRY_CONST GPUTRDGeometry*)GetConstantMem()->calibObjects.trdGeometry;
128  if (!mGeo) {
129  GPUError("TRD geometry must be provided externally");
130  }
131 
132  float Bz = Param().par.bzkG;
133  GPUInfo("Initializing with B-field: %f kG", Bz);
134  if (CAMath::Abs(CAMath::Abs(Bz) - 2) < 0.1) {
135  // magnetic field +-0.2 T
136  if (Bz > 0) {
137  GPUInfo("Loading error parameterization for Bz = +2 kG");
138  mRPhiA2 = 1.6e-3f, mRPhiB = -1.43e-2f, mRPhiC2 = 4.55e-2f;
139  mDyA2 = 1.225e-3f, mDyB = -9.8e-3f, mDyC2 = 3.88e-2f;
140  mAngleToDyA = -0.1f, mAngleToDyB = 1.89f, mAngleToDyC = -0.4f;
141  } else {
142  GPUInfo("Loading error parameterization for Bz = -2 kG");
143  mRPhiA2 = 1.6e-3f, mRPhiB = 1.43e-2f, mRPhiC2 = 4.55e-2f;
144  mDyA2 = 1.225e-3f, mDyB = 9.8e-3f, mDyC2 = 3.88e-2f;
145  mAngleToDyA = 0.1f, mAngleToDyB = 1.89f, mAngleToDyC = 0.4f;
146  }
147  } else if (CAMath::Abs(CAMath::Abs(Bz) - 5) < 0.1) {
148  // magnetic field +-0.5 T
149  if (Bz > 0) {
150  GPUInfo("Loading error parameterization for Bz = +5 kG");
151  mRPhiA2 = 1.6e-3f, mRPhiB = 0.125f, mRPhiC2 = 0.0961f;
152  mDyA2 = 1.681e-3f, mDyB = 0.15f, mDyC2 = 0.1849f;
153  mAngleToDyA = 0.13f, mAngleToDyB = 2.43f, mAngleToDyC = -0.58f;
154  } else {
155  GPUInfo("Loading error parameterization for Bz = -5 kG");
156  mRPhiA2 = 1.6e-3f, mRPhiB = -0.14f, mRPhiC2 = 0.1156f;
157  mDyA2 = 2.209e-3f, mDyB = -0.15f, mDyC2 = 0.2025f;
158  mAngleToDyA = -0.15f, mAngleToDyB = 2.34f, mAngleToDyC = 0.56f;
159  }
160  } else {
161  // magnetic field 0 T or another value which is not covered by the error parameterizations
162  GPUError("No error parameterization available for Bz = %.2f kG", Bz);
163  }
164 
165 #ifdef GPUCA_ALIROOT_LIB
166  for (int iCandidate = 0; iCandidate < mNCandidates * 2 * mMaxThreads; ++iCandidate) {
167  new (&mCandidates[iCandidate]) TRDTRK;
168  }
169 #endif
170 
171  // obtain average radius of TRD chambers
172  float x0[kNLayers] = {300.2f, 312.8f, 325.4f, 338.0f, 350.6f, 363.2f}; // used as default value in case no transformation matrix can be obtained
173  auto* matrix = mGeo->GetClusterMatrix(0);
174  My_Float loc[3] = {mGeo->AnodePos(), 0.f, 0.f};
175  My_Float glb[3] = {0.f, 0.f, 0.f};
176  for (int iDet = 0; iDet < kNChambers; ++iDet) {
177  matrix = mGeo->GetClusterMatrix(iDet);
178  if (!matrix) {
179  mR[iDet] = x0[mGeo->GetLayer(iDet)];
180  continue;
181  }
182  matrix->LocalToMaster(loc, glb);
183  mR[iDet] = glb[0];
184  }
185 
186  mDebug->ExpandVectors();
187  mIsInitialized = true;
188 }

199 template <class TRDTRK, class PROP>
200 void GPUTRDTracker_t<TRDTRK, PROP>::PrepareTracking(GPUChainTracking* chainTracking)
201 {
202  //--------------------------------------------------------------------
203  // Prepare tracklet index array and if requested calculate space points
204  // in part duplicated from DoTracking() method to allow for calling
205  // this function on the host prior to GPU processing
206  //--------------------------------------------------------------------
207  for (unsigned int iColl = 0; iColl < GetConstantMem()->ioPtrs.nTRDTriggerRecords; ++iColl) {
208  if (GetConstantMem()->ioPtrs.trdTrigRecMask && GetConstantMem()->ioPtrs.trdTrigRecMask[iColl] == 0) {
209  // this trigger is masked as there is no ITS information available for it
210  continue;
211  }
212  int nTrklts = 0;
213  int idxOffset = 0;
214  if (mProcessPerTimeFrame) {
215  idxOffset = GetConstantMem()->ioPtrs.trdTrackletIdxFirst[iColl];
216  nTrklts = (iColl < GetConstantMem()->ioPtrs.nTRDTriggerRecords - 1) ? GetConstantMem()->ioPtrs.trdTrackletIdxFirst[iColl + 1] - GetConstantMem()->ioPtrs.trdTrackletIdxFirst[iColl] : GetConstantMem()->ioPtrs.nTRDTracklets - GetConstantMem()->ioPtrs.trdTrackletIdxFirst[iColl];
217  } else {
218  nTrklts = GetConstantMem()->ioPtrs.nTRDTracklets;
219  }
220  const GPUTRDTrackletWord* tracklets = &((GetConstantMem()->ioPtrs.trdTracklets)[idxOffset]);
221  int* trkltIndexArray = &mTrackletIndexArray[iColl * (kNChambers + 1) + 1];
222  trkltIndexArray[-1] = 0;
223  int currDet = 0;
224  int nextDet = 0;
225  int trkltCounter = 0;
226  for (int iTrklt = 0; iTrklt < nTrklts; ++iTrklt) {
227  if (tracklets[iTrklt].GetDetector() > currDet) {
228  nextDet = tracklets[iTrklt].GetDetector();
229  for (int iDet = currDet; iDet < nextDet; ++iDet) {
230  trkltIndexArray[iDet] = trkltCounter;
231  }
232  currDet = nextDet;
233  }
234  ++trkltCounter;
235  }
236  for (int iDet = currDet; iDet <= kNChambers; ++iDet) {
237  trkltIndexArray[iDet] = trkltCounter;
238  }
239  if (mGenerateSpacePoints) {
240  if (!CalculateSpacePoints(iColl)) {
241  GPUError("Space points for at least one chamber could not be calculated (for interaction %i)", iColl);
242  break;
243  }
244  }
245  }
246  if (mGenerateSpacePoints) {
247  chainTracking->mIOPtrs.trdSpacePoints = mSpacePoints;
248  }
249  mNEvents++;
250 }

361 template <class TRDTRK, class PROP>
362 GPUd() int GPUTRDTracker_t<TRDTRK, PROP>::GetCollisionIDs(int iTrk, int* collisionIds) const
363 {
364  //--------------------------------------------------------------------
365  // Check which TRD trigger times possibly match given input track.
366  // If ITS-TPC matches or CE-crossing TPC tracks the time is precisely
367  // known and max 1 trigger time can be assigned.
368  // For TPC-only tracks the collision IDs are stored in collisionIds array
369  // and the number of valid entries in the array is returned
370  //--------------------------------------------------------------------
371  int nColls = 0;
372  for (unsigned int iColl = 0; iColl < GetConstantMem()->ioPtrs.nTRDTriggerRecords; ++iColl) {
373  if (GetConstantMem()->ioPtrs.trdTrigRecMask && GetConstantMem()->ioPtrs.trdTrigRecMask[iColl] == 0) {
374  continue;
375  }
376  if (GetConstantMem()->ioPtrs.trdTriggerTimes[iColl] > mTrackAttribs[iTrk].GetTimeMin() && GetConstantMem()->ioPtrs.trdTriggerTimes[iColl] < mTrackAttribs[iTrk].GetTimeMax()) {
377  if (nColls == 20) {
378  GPUError("Found too many collision candidates for track with tMin(%f) and tMax(%f)", mTrackAttribs[iTrk].GetTimeMin(), mTrackAttribs[iTrk].GetTimeMax());
379  return nColls;
380  }
381  collisionIds[nColls++] = iColl;
382  }
383  }
384  return nColls;
385 }

387 template <class TRDTRK, class PROP>
388 GPUd() void GPUTRDTracker_t<TRDTRK, PROP>::DoTrackingThread(int iTrk, int threadId)
389 {
390  //--------------------------------------------------------------------
391  // perform the tracking for one track (must be threadsafe)
392  //--------------------------------------------------------------------
393  int collisionIds[20] = {0}; // due to the dead time there will never exist more possible TRD triggers for a single track
394  int nCollisionIds = 1; // initialize with 1 for AliRoot compatibility
395  if (mProcessPerTimeFrame) {
396  nCollisionIds = GetCollisionIDs(iTrk, collisionIds);
397  if (nCollisionIds == 0) {
398  if (ENABLE_INFO) {
399  GPUInfo("Did not find TRD data for track %i with t=%f. tMin(%f), tMax(%f)", iTrk, mTrackAttribs[iTrk].mTime, mTrackAttribs[iTrk].GetTimeMin(), mTrackAttribs[iTrk].GetTimeMax());
400  }
401  // no TRD data available for the bunch crossing this track originates from
402  return;
403  }
404  }
405  PROP prop(getPropagatorParam());
406  mTracks[iTrk].setChi2(Param().rec.trd.penaltyChi2); // TODO check if this should not be higher
407  auto trkStart = mTracks[iTrk];
408  for (int iColl = 0; iColl < nCollisionIds; ++iColl) {
409  // do track following for each collision candidate and keep best track
410  auto trkCopy = trkStart;
411  prop.setTrack(&trkCopy);
412  prop.setFitInProjections(true);
413  if (!FollowProlongation(&prop, &trkCopy, iTrk, threadId, collisionIds[iColl])) {
414  // track following failed
415  continue;
416  }
417  if (trkCopy.getReducedChi2() < mTracks[iTrk].getReducedChi2()) {
418  mTracks[iTrk] = trkCopy; // copy back the resulting track
419  }
420  }
421 }

454 template <class TRDTRK, class PROP>
455 GPUd() bool GPUTRDTracker_t<TRDTRK, PROP>::CalculateSpacePoints(int iCollision)
456 {
457  //--------------------------------------------------------------------
458  // Calculates TRD space points in sector tracking coordinates
459  // from online tracklets
460  //--------------------------------------------------------------------
461 
462  bool result = true;
463  int idxOffset = iCollision * (kNChambers + 1); // offset for accessing mTrackletIndexArray for collision iCollision
464 
465  const GPUTRDTrackletWord* tracklets = GetConstantMem()->ioPtrs.trdTracklets;
466 
467  for (int iDet = 0; iDet < kNChambers; ++iDet) {
468  int iFirstTrackletInDet = mTrackletIndexArray[idxOffset + iDet];
469  int iFirstTrackletInNextDet = mTrackletIndexArray[idxOffset + iDet + 1];
470  int nTrackletsInDet = iFirstTrackletInNextDet - iFirstTrackletInDet;
471  if (nTrackletsInDet == 0) {
472  continue;
473  }
474  if (!mGeo->ChamberInGeometry(iDet)) {
475  GPUError("Found TRD tracklets in chamber %i which is not included in the geometry", iDet);
476  return false;
477  }
478  auto* matrix = mGeo->GetClusterMatrix(iDet);
479  if (!matrix) {
480  GPUError("No cluster matrix available for chamber %i. Skipping it...", iDet);
481  result = false;
482  continue;
483  }
484  const GPUTRDpadPlane* pp = mGeo->GetPadPlane(iDet);
485 
486  int trkltIdxOffset = (mProcessPerTimeFrame) ? GetConstantMem()->ioPtrs.trdTrackletIdxFirst[iCollision] : 0; // global index of first tracklet in iCollision
487  int trkltIdxStart = trkltIdxOffset + iFirstTrackletInDet;
488  for (int trkltIdx = trkltIdxStart; trkltIdx < trkltIdxStart + nTrackletsInDet; ++trkltIdx) {
489  int trkltZbin = tracklets[trkltIdx].GetZbin();
490  My_Float xTrkltDet[3] = {0.f}; // trklt position in chamber coordinates
491  My_Float xTrkltSec[3] = {0.f}; // trklt position in sector coordinates
492  xTrkltDet[0] = mGeo->AnodePos() + sRadialOffset;
493  xTrkltDet[1] = tracklets[trkltIdx].GetY();
494  xTrkltDet[2] = pp->GetRowPos(trkltZbin) - pp->GetRowSize(trkltZbin) / 2.f - pp->GetRowPos(pp->GetNrows() / 2);
495  //GPUInfo("Space point local %i: x=%f, y=%f, z=%f", trkltIdx, xTrkltDet[0], xTrkltDet[1], xTrkltDet[2]);
496  matrix->LocalToMaster(xTrkltDet, xTrkltSec);
497  mSpacePoints[trkltIdx].setX(xTrkltSec[0]);
498  mSpacePoints[trkltIdx].setY(xTrkltSec[1]);

513  // tracklet(s) on the way
514  // -> returns false if prolongation could not be executed fully
515  // or track does not fullfill threshold conditions

567  //

569  //
570  // --------------------------------------------------------------------------------

573  int det[nMaxChambersToSearch] = {-1, -1, -1, -1}; // TRD chambers to be searched for tracklets

576  trkWork = &mCandidates[2 * iCandidate + currIdx];

583  isOK = true;
584  continue;
585  }

587  // propagate track to average radius of TRD layer iLayer (sector 0, stack 2 is chosen as a reference)
588  if (!prop->propagateToX(mR[2 * kNLayers + iLayer], .8f, 2.f)) {
589  if (ENABLE_INFO) {

591  }
592  continue;
593  }

596  if (!AdjustSector(prop, trkWork)) {
597  if (ENABLE_INFO) {
598  GPUInfo("Adjusting sector failed for track %i candidate %i in layer %i", iTrk, iCandidate, iLayer);
599  }

603  // check if track is findable
604  if (IsGeoFindable(trkWork, iLayer, prop->getAlpha(), zShiftTrk)) {
605  trkWork->setIsFindable(iLayer);

607

617  continue;

623  // track debug information to be stored in case no matching tracklet can be found
624  mDebug->SetTrackParameter(*trkWork, iLayer);
625 
626  // look for tracklets in chamber(s)
627  for (int iDet = 0; iDet < nMaxChambersToSearch; iDet++) {
628  int currDet = det[iDet];
629  if (currDet == -1) {
630  continue;
631  }
632  pad = mGeo->GetPadPlane(currDet);
633  int currSec = mGeo->GetSector(currDet);
634  if (currSec != GetSector(prop->getAlpha())) {
635  if (!prop->rotate(GetAlphaOfSector(currSec))) {
636  if (ENABLE_WARNING) {
637  GPUWarning("Track could not be rotated in tracklet coordinate system");
638  }
639  break;
640  }
641  }
642  if (currSec != GetSector(prop->getAlpha())) {
643  GPUError("Track is in sector %i and sector %i is searched for tracklets", GetSector(prop->getAlpha()), currSec);
644  continue;
645  }
646  // propagate track to radius of chamber
647  if (!prop->propagateToX(mR[currDet], .8f, .2f)) {
648  if (ENABLE_WARNING) {
649  GPUWarning("Track parameter for track %i, x=%f at chamber %i x=%f in layer %i cannot be retrieved", iTrk, trkWork->getX(), currDet, mR[currDet], iLayer);
650  }
651  }
652  // first propagate track to x of tracklet
653  for (int trkltIdx = glbTrkltIdxOffset + mTrackletIndexArray[trkltIdxOffset + currDet]; trkltIdx < glbTrkltIdxOffset + mTrackletIndexArray[trkltIdxOffset + currDet + 1]; ++trkltIdx) {

655  // skip tracklets which are too far away
656  // although the radii of space points and tracks may differ by ~ few mm the roads are large enough to allow no efficiency loss by this cut
657  continue;
658  }
659  float projY, projZ;
660  prop->getPropagatedYZ(spacePoints[trkltIdx].getX(), projY, projZ);
661  // correction for tilted pads (only applied if deltaZ < lPad && track z err << lPad)
662  float tiltCorr = tilt * (spacePoints[trkltIdx].getZ() - projZ);
663  float lPad = pad->GetRowSize(tracklets[trkltIdx].GetZbin());
664  if (!((CAMath::Abs(spacePoints[trkltIdx].getZ() - projZ) < lPad) && (trkWork->getSigmaZ2() < (lPad * lPad / 12.f)))) {
665  tiltCorr = 0.f; // will be zero also for TPC tracks which are shifted in z
666  }
667  // correction for mean z position of tracklet (is not the center of the pad if track eta != 0)
668  float zPosCorr = spacePoints[trkltIdx].getZ() + mZCorrCoefNRC * trkWork->getTgl();
669  float yPosCorr = spacePoints[trkltIdx].getY() - tiltCorr;
670  zPosCorr -= zShiftTrk; // shift tracklet instead of track in order to avoid having to do a re-fit for each collision
671  float deltaY = yPosCorr - projY;
672  float deltaZ = zPosCorr - projZ;
673  My_Float trkltPosTmpYZ[2] = {yPosCorr, zPosCorr};
674  My_Float trkltCovTmp[3] = {0.f};
675  if ((CAMath::Abs(deltaY) < roadY) && (CAMath::Abs(deltaZ) < roadZ)) { // TODO: check if this is still necessary after the cut before propagation of track
676  // tracklet is in windwow: get predicted chi2 for update and store tracklet index if best guess
677  RecalcTrkltCov(tilt, trkWork->getSnp(), pad->GetRowSize(tracklets[trkltIdx].GetZbin()), trkltCovTmp);

679  // TODO cut on angular pull should be made stricter when proper v-drift calibration for the TRD tracklets is implemented
680  if ((chi2 > Param().rec.trd.maxChi2) || (Param().rec.trd.applyDeflectionCut && CAMath::Abs(GetAngularPull(spacePoints[trkltIdx].getDy(), trkWork->getSnp())) > 4)) {
681  continue;

686  } // tracklet loop
687  } // chamber loop
688 
689  // add no update to hypothesis list
690  Hypothesis hypoNoUpdate(trkWork->getNlayersFindable(), iCandidate, -1, trkWork->getChi2() + Param().rec.trd.penaltyChi2);
691  InsertHypothesis(hypoNoUpdate, nCurrHypothesis, hypothesisIdxOffset);
692  isOK = true;
693  } // end candidate loop

699  //

701  //
702  // --------------------------------------------------------------------------------
703  // GPUInfo("nCurrHypothesis=%i, nCandidates=%i", nCurrHypothesis, nCandidates);
704  // for (int idx=0; idx<10; ++idx) { GPUInfo("mHypothesis[%i]: candidateId=%i, nLayers=%i, trackletId=%i, chi2=%f", idx, mHypothesis[idx].mCandidateId, mHypothesis[idx].mLayers, mHypothesis[idx].mTrackletId, mHypothesis[idx].mChi2); }

706  if (mHypothesis[iUpdate + hypothesisIdxOffset].mCandidateId == -1) {
707  // no more candidates
708  if (iUpdate == 0) {
709  return false; // no valid candidates for this track (probably propagation failed)
710  }
711  break; // go to next layer
712  }
713  nCandidates = iUpdate + 1;
714  if (mNCandidates > 1) {
715  mCandidates[2 * iUpdate + nextIdx] = mCandidates[2 * mHypothesis[iUpdate + hypothesisIdxOffset].mCandidateId + currIdx];
716  trkWork = &mCandidates[2 * iUpdate + nextIdx];
717  }
718  if (mHypothesis[iUpdate + hypothesisIdxOffset].mTrackletId == -1) {
719  // no matching tracklet found
720  if (trkWork->getIsFindable(iLayer)) {
721  if (trkWork->getNmissingConsecLayers(iLayer) > Param().rec.trd.stopTrkAfterNMissLy) {
722  trkWork->setIsStopped();
723  }
724  trkWork->setChi2(trkWork->getChi2() + Param().rec.trd.penaltyChi2);
725  }
726  if (iUpdate == 0 && mNCandidates > 1) { // TODO: is thie really necessary????? CHECK!
727  *t = mCandidates[2 * iUpdate + nextIdx];
728  }
729  continue;
730  }
731  // matching tracklet found
732  if (mNCandidates > 1) {
733  prop->setTrack(trkWork);
734  }
735  int trkltSec = mGeo->GetSector(tracklets[mHypothesis[iUpdate + hypothesisIdxOffset].mTrackletId].GetDetector());
736  if (trkltSec != GetSector(prop->getAlpha())) {
737  // if after a matching tracklet was found another sector was searched for tracklets the track needs to be rotated back
738  prop->rotate(GetAlphaOfSector(trkltSec));
739  }
740  if (!prop->propagateToX(spacePoints[mHypothesis[iUpdate + hypothesisIdxOffset].mTrackletId].getX(), .8f, 2.f)) {
741  if (ENABLE_WARNING) {
742  GPUWarning("Final track propagation for track %i update %i in layer %i failed", iTrk, iUpdate, iLayer);
743  }
744  trkWork->setChi2(trkWork->getChi2() + Param().rec.trd.penaltyChi2);
745  if (trkWork->getIsFindable(iLayer)) {
746  if (trkWork->getNmissingConsecLayers(iLayer) >= Param().rec.trd.stopTrkAfterNMissLy) {
747  trkWork->setIsStopped();
748  }
749  }

751  *t = mCandidates[2 * iUpdate + nextIdx];
752  }
753  continue;
754  }

758  float zPosCorrUp = spacePoints[mHypothesis[iUpdate + hypothesisIdxOffset].mTrackletId].getZ() + mZCorrCoefNRC * trkWork->getTgl();
759  zPosCorrUp -= zShiftTrk;
760  float padLength = pad->GetRowSize(tracklets[mHypothesis[iUpdate + hypothesisIdxOffset].mTrackletId].GetZbin());
761  if (!((trkWork->getSigmaZ2() < (padLength * padLength / 12.f)) && (CAMath::Abs(spacePoints[mHypothesis[iUpdate + hypothesisIdxOffset].mTrackletId].getZ() - trkWork->getZ()) < padLength))) {
762  tiltCorrUp = 0.f;
763  }
764  My_Float trkltPosUp[2] = {spacePoints[mHypothesis[iUpdate + hypothesisIdxOffset].mTrackletId].getY() - tiltCorrUp, zPosCorrUp};

766  RecalcTrkltCov(tilt, trkWork->getSnp(), pad->GetRowSize(tracklets[mHypothesis[iUpdate + hypothesisIdxOffset].mTrackletId].GetZbin()), trkltCovUp);

770  prop->rotate(GetAlphaOfSector(trkltSec));
771  //prop->propagateToX(spacePoints[mHypothesis[iUpdate + hypothesisIdxOffset].mTrackletId].getX(), .8f, 2.f);

773  prop->setTrack(trkWork);
774 #endif

779 #endif

785  wasTrackStored = true;
786  }
787 
788  if (!prop->update(trkltPosUp, trkltCovUp)) {
789  if (ENABLE_WARNING) {
790  GPUWarning("Failed to update track %i with space point in layer %i", iTrk, iLayer);
791  }
792  trkWork->setChi2(trkWork->getChi2() + Param().rec.trd.penaltyChi2);
793  if (trkWork->getIsFindable(iLayer)) {
794  if (trkWork->getNmissingConsecLayers(iLayer) >= Param().rec.trd.stopTrkAfterNMissLy) {
795  trkWork->setIsStopped();
796  }
797  }
798  if (iUpdate == 0 && mNCandidates > 1) {
799  *t = mCandidates[2 * iUpdate + nextIdx];
800  }
801  continue;
802  }
803  if (!trkWork->CheckNumericalQuality()) {
804  if (ENABLE_INFO) {
805  GPUInfo("Track %i has invalid covariance matrix. Aborting track following\n", iTrk);
806  }
807  return false;
808  }
809  trkWork->addTracklet(iLayer, mHypothesis[iUpdate + hypothesisIdxOffset].mTrackletId);
810  trkWork->setChi2(mHypothesis[iUpdate + hypothesisIdxOffset].mChi2);
811  trkWork->setIsFindable(iLayer);
812  trkWork->setCollisionId(collisionId);
813  if (iUpdate == 0 && mNCandidates > 1) {
814  *t = mCandidates[2 * iUpdate + nextIdx];
815  }
816  } // end update loop
817 
818  if (!isOK) {
819  if (ENABLE_INFO) {
820  GPUInfo("Track %i cannot be followed. Stopped in layer %i", iTrk, iLayer);
821  }
822  return false;
823  }
824  } // end layer loop
825 
826  // --------------------------------------------------------------------------------
827  // add some debug information (compare labels of attached tracklets to track label)
828  // and store full track information
829  // --------------------------------------------------------------------------------
830  if (mDebugOutput) {
831  mDebug->SetTrack(*t);
832  mDebug->Output();
833  }
834  if (ENABLE_INFO) {
835  GPUInfo("Ended track following for track %i at x=%f with pt=%f. Attached %i tracklets", t->getRefGlobalTrackIdRaw(), t->getX(), t->getPt(), t->getNtracklets());
836  }
837  if (nCurrHypothesis > 1) {
838  if (CAMath::Abs(mHypothesis[hypothesisIdxOffset + 1].GetReducedChi2() - mHypothesis[hypothesisIdxOffset].GetReducedChi2()) < Param().rec.trd.chi2SeparationCut) {
839  t->setIsAmbiguous();
840  }
841  }
842  return true;
843 }

845 template <class TRDTRK, class PROP>
846 GPUd() void GPUTRDTracker_t<TRDTRK, PROP>::InsertHypothesis(Hypothesis hypo, int& nCurrHypothesis, int idxOffset)
847 {
848  // Insert hypothesis into the array. If the array is full and the reduced chi2 is worse
849  // than the worst hypothesis in the array it is dropped.
850  // The hypothesis array is always sorted.
851 
852  if (nCurrHypothesis == 0) {
853  // this is the first hypothesis in the array
854  mHypothesis[idxOffset] = hypo;
855  ++nCurrHypothesis;
856  } else if (nCurrHypothesis > 0 && nCurrHypothesis < mNCandidates) {
857  // insert the hypothesis into the right position and shift all worse hypothesis to the right
858  for (int i = idxOffset; i < nCurrHypothesis + idxOffset; ++i) {
859  if (hypo.GetReducedChi2() < mHypothesis[i].GetReducedChi2()) {
860  for (int k = nCurrHypothesis + idxOffset; k > i; --k) {
861  mHypothesis[k] = mHypothesis[k - 1];
862  }
863  mHypothesis[i] = hypo;
864  ++nCurrHypothesis;
865  return;
866  }
867  }
868  mHypothesis[nCurrHypothesis + idxOffset] = hypo;
869  ++nCurrHypothesis;
870  return;
871  } else {
872  // array is already full, check if new hypothesis should be inserted
873  int i = nCurrHypothesis + idxOffset - 1;
874  for (; i >= idxOffset; --i) {
875  if (mHypothesis[i].GetReducedChi2() < hypo.GetReducedChi2()) {
876  break;
877  }
878  }
879  if (i < (nCurrHypothesis + idxOffset - 1)) {
880  // new hypothesis should be inserted into the array
881  for (int k = nCurrHypothesis + idxOffset - 1; k > i + 1; --k) {
882  mHypothesis[k] = mHypothesis[k - 1];
883  }
884  mHypothesis[i + 1] = hypo;
885  }
886  }
887 }

905 template <class TRDTRK, class PROP>
906 GPUd() bool GPUTRDTracker_t<TRDTRK, PROP>::AdjustSector(PROP* prop, TRDTRK* t) const
907 {
908  //--------------------------------------------------------------------
909  // rotate track in new sector if necessary and
910  // propagate to previous x afterwards
911  // cancel if track crosses two sector boundaries
912  //--------------------------------------------------------------------
913  float alpha = mGeo->GetAlpha();
914  float xTmp = t->getX();
915  float y = t->getY();
916  float yMax = t->getX() * CAMath::Tan(0.5f * alpha);
917  float alphaCurr = t->getAlpha();
918 
919  if (CAMath::Abs(y) > 2.f * yMax) {
920  if (ENABLE_INFO) {
921  GPUInfo("AdjustSector: Track %i with pT = %f crossing two sector boundaries at x = %f", t->getRefGlobalTrackIdRaw(), t->getPt(), t->getX());
922  }
923  return false;
924  }
925 
926  int nTries = 0;
927  while (CAMath::Abs(y) > yMax) {
928  if (nTries >= 2) {
929  return false;
930  }
931  int sign = (y > 0) ? 1 : -1;
932  float alphaNew = alphaCurr + alpha * sign;
933  if (alphaNew > CAMath::Pi()) {
934  alphaNew -= 2 * CAMath::Pi();
935  } else if (alphaNew < -CAMath::Pi()) {
936  alphaNew += 2 * CAMath::Pi();
937  }
938  if (!prop->rotate(alphaNew)) {
939  return false;
940  }
941  if (!prop->propagateToX(xTmp, .8f, 2.f)) {
942  return false;
943  }
944  y = t->getY();
945  ++nTries;
946  }
947  return true;
948 }

1071  }

1078  }
