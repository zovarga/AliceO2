100 {
101  auto result = aliases();

103  constexpr size_t maxLen{62};
104  for (auto& a : result) {
105  sizes[a.size()]++;
106  if (a.size() > maxLen) {
107  std::cout << fmt::format("Alias is too long : {:2d} characters, while {:2d} max are allowed : {}\n",
108  a.size(), maxLen, a);
109  }
110  }
111  size_t len{0};
112 
113  for (auto p : sizes) {
114  std::cout << fmt::format("{:3d} aliases of size {:2d}\n",
115  p.second, p.first);
116  len = std::max(len, p.first);
117  }
118  BOOST_CHECK(len <= 62);
119 }
