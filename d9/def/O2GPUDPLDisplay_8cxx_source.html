62  mConfig->configGRP.solenoidBz = 5.00668f * grp->getL3Current() / 30000.;
63  mConfig->configGRP.continuousMaxTimeBin = grp->isDetContinuousReadOut(o2::detectors::DetID::TPC) ? -1 : 0; // Number of timebins in timeframe if continuous, 0 otherwise

65

67  mConfig->configCalib.fastTransform = mFastTransform.get();

73  mConfig->configCalib.trdGeometry = mTrdGeo.get();
74 
75  mITSDict = std::make_unique<o2::itsmft::TopologyDictionary>();
76  mConfig->configCalib.itsPatternDict = mITSDict.get();
77 
78  mConfig->configProcessing.runMC = mUseMC;
79 
80  o2::tof::Geo::Init();
81 
82  o2::its::GeometryTGeo::Instance()->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2GRot, o2::math_utils::TransformType::T2G, o2::math_utils::TransformType::L2G, o2::math_utils::TransformType::T2L));

109 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
110 {
111  WorkflowSpec specs;
112 
113  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
114 
115  bool useMC = cfgc.options().get<bool>("enable-mc") && !cfgc.options().get<bool>("disable-mc");
116  GlobalTrackID::mask_t srcTrk = GlobalTrackID::getSourcesMask(cfgc.options().get<std::string>("display-tracks"));
117  GlobalTrackID::mask_t srcCl = GlobalTrackID::getSourcesMask(cfgc.options().get<std::string>("display-clusters"));
118  if (!srcTrk.any() && !srcCl.any()) {
119  throw std::runtime_error("No input configured");
120  }
121  std::shared_ptr<DataRequest> dataRequest = std::make_shared<DataRequest>();
122  dataRequest->requestTracks(srcTrk, useMC);
123  dataRequest->requestClusters(srcCl, useMC);
124 
125  if (cfgc.options().get<bool>("read-from-files")) {
126  InputHelper::addInputSpecs(cfgc, specs, srcCl, srcTrk, srcTrk, useMC);
127  }
128 
129  specs.emplace_back(DataProcessorSpec{
130  "o2-gpu-display",
131  dataRequest->inputs,
132  {},
133  AlgorithmSpec{adaptFromTask<O2GPUDPLDisplaySpec>(useMC, srcTrk, srcCl, dataRequest)}});
134 
135  return std::move(specs);
136 }
