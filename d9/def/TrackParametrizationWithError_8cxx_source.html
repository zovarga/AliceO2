48 template <typename value_T>
49 GPUd() void TrackParametrizationWithError<value_T>::invert()
50 {
51  // Transform this track to the local coord. system rotated by 180 deg.
52  this->invertParam();
53  // since the fP1 and fP2 are not inverted, their covariances with others change sign
54  mC[kSigZY] = -mC[kSigZY];
55  mC[kSigSnpY] = -mC[kSigSnpY];
56  mC[kSigTglZ] = -mC[kSigTglZ];
57  mC[kSigTglSnp] = -mC[kSigTglSnp];
58  mC[kSigQ2PtZ] = -mC[kSigQ2PtZ];
59  mC[kSigQ2PtSnp] = -mC[kSigQ2PtSnp];
60 }

63 template <typename value_T>
64 GPUd() bool TrackParametrizationWithError<value_T>::propagateTo(value_t xk, value_t b)
65 {
66  //----------------------------------------------------------------
67  // propagate this track to the plane X=xk (cm) in the field "b" (kG)
68  //----------------------------------------------------------------
69  value_t dx = xk - this->getX();
70  if (gpu::CAMath::Abs(dx) < constants::math::Almost0) {
71  return true;
72  }
73  value_t crv = this->getCurvature(b);
74  value_t x2r = crv * dx;
75  value_t f1 = this->getSnp(), f2 = f1 + x2r;
76  if ((gpu::CAMath::Abs(f1) > constants::math::Almost1) || (gpu::CAMath::Abs(f2) > constants::math::Almost1)) {
77  return false;
78  }
79  value_t r1 = gpu::CAMath::Sqrt((1.f - f1) * (1.f + f1));
80  if (gpu::CAMath::Abs(r1) < constants::math::Almost0) {
81  return false;
82  }
83  value_t r2 = gpu::CAMath::Sqrt((1.f - f2) * (1.f + f2));
84  if (gpu::CAMath::Abs(r2) < constants::math::Almost0) {
85  return false;
86  }
87  this->setX(xk);
88  double dy2dx = (f1 + f2) / (r1 + r2);
89  params_t dP{0.f};
90  dP[kY] = dx * dy2dx;
91  dP[kSnp] = x2r;
92  if (gpu::CAMath::Abs(x2r) < 0.05f) {
93  dP[kZ] = dx * (r2 + f2 * dy2dx) * this->getTgl();
94  } else {
95  // for small dx/R the linear apporximation of the arc by the segment is OK,
96  // but at large dx/R the error is very large and leads to incorrect Z propagation
97  // angle traversed delta = 2*asin(dist_start_end / R / 2), hence the arc is: R*deltaPhi
98  // The dist_start_end is obtained from sqrt(dx^2+dy^2) = x/(r1+r2)*sqrt(2+f1*f2+r1*r2)
99  // double chord = dx*TMath::Sqrt(1+dy2dx*dy2dx); // distance from old position to new one
100  // double rot = 2*TMath::ASin(0.5*chord*crv); // angular difference seen from the circle center
101  // mP1 += rot/crv*mP3;
102  //
103  value_t rot = gpu::CAMath::ASin(r1 * f2 - r2 * f1); // more economic version from Yura.
104  if (f1 * f1 + f2 * f2 > 1.f && f1 * f2 < 0.f) { // special cases of large rotations or large abs angles
105  if (f2 > 0.f) {
106  rot = constants::math::PI - rot; //
107  } else {
108  rot = -constants::math::PI - rot;
109  }
110  }
111  dP[kZ] = this->getTgl() / crv * rot;
112  }
113 
114  this->updateParams(dP); // apply corrections
115 
116  value_t &c00 = mC[kSigY2], &c10 = mC[kSigZY], &c11 = mC[kSigZ2], &c20 = mC[kSigSnpY], &c21 = mC[kSigSnpZ],
117  &c22 = mC[kSigSnp2], &c30 = mC[kSigTglY], &c31 = mC[kSigTglZ], &c32 = mC[kSigTglSnp], &c33 = mC[kSigTgl2],
118  &c40 = mC[kSigQ2PtY], &c41 = mC[kSigQ2PtZ], &c42 = mC[kSigQ2PtSnp], &c43 = mC[kSigQ2PtTgl],
119  &c44 = mC[kSigQ2Pt2];
120 
121  // evaluate matrix in double prec.
122  double rinv = 1. / r1;
123  double r3inv = rinv * rinv * rinv;
124  double f24 = dx * b * constants::math::B2C; // x2r/mP[kQ2Pt];
125  double f02 = dx * r3inv;
126  double f04 = 0.5 * f24 * f02;
127  double f12 = f02 * this->getTgl() * f1;
128  double f14 = 0.5 * f24 * f12; // 0.5*f24*f02*getTgl()*f1;
129  double f13 = dx * rinv;
130 
131  // b = C*ft
132  double b00 = f02 * c20 + f04 * c40, b01 = f12 * c20 + f14 * c40 + f13 * c30;
133  double b02 = f24 * c40;
134  double b10 = f02 * c21 + f04 * c41, b11 = f12 * c21 + f14 * c41 + f13 * c31;
135  double b12 = f24 * c41;
136  double b20 = f02 * c22 + f04 * c42, b21 = f12 * c22 + f14 * c42 + f13 * c32;
137  double b22 = f24 * c42;
138  double b40 = f02 * c42 + f04 * c44, b41 = f12 * c42 + f14 * c44 + f13 * c43;
139  double b42 = f24 * c44;
140  double b30 = f02 * c32 + f04 * c43, b31 = f12 * c32 + f14 * c43 + f13 * c33;
141  double b32 = f24 * c43;
142 
143  // a = f*b = f*C*ft
144  double a00 = f02 * b20 + f04 * b40, a01 = f02 * b21 + f04 * b41, a02 = f02 * b22 + f04 * b42;
145  double a11 = f12 * b21 + f14 * b41 + f13 * b31, a12 = f12 * b22 + f14 * b42 + f13 * b32;
146  double a22 = f24 * b42;
147 
148  // F*C*Ft = C + (b + bt + a)
149  c00 += b00 + b00 + a00;
150  c10 += b10 + b01 + a01;
151  c20 += b20 + b02 + a02;
152  c30 += b30;
153  c40 += b40;
154  c11 += b11 + b11 + a11;
155  c21 += b21 + b12 + a12;
156  c31 += b31;
157  c41 += b41;
158  c22 += b22 + b22 + a22;
159  c32 += b32;
160  c42 += b42;
161 
162  checkCovariance();
163 
164  return true;
165 }

168 template <typename value_T>
169 GPUd() bool TrackParametrizationWithError<value_T>::rotate(value_t alpha)
170 {
171  // rotate to alpha frame
172  if (gpu::CAMath::Abs(this->getSnp()) > constants::math::Almost1) {
173  LOGP(debug, "Precondition is not satisfied: |sin(phi)|>1 ! {:f}", this->getSnp());
174  return false;
175  }
176  //
177  math_utils::detail::bringToPMPi<value_t>(alpha);
178  //
179  value_t ca = 0, sa = 0;
180  math_utils::detail::sincos(alpha - this->getAlpha(), sa, ca);
181  value_t snp = this->getSnp(), csp = gpu::CAMath::Sqrt((1.f - snp) * (1.f + snp)); // Improve precision
182  // RS: check if rotation does no invalidate track model (cos(local_phi)>=0, i.e. particle
183  // direction in local frame is along the X axis
184  if ((csp * ca + snp * sa) < 0) {
185  //LOGP(warning,"Rotation failed: local cos(phi) would become {:.2f}", csp * ca + snp * sa);
186  return false;
187  }
188  //
189 
190  value_t updSnp = snp * ca - csp * sa;
191  if (gpu::CAMath::Abs(updSnp) > constants::math::Almost1) {
192  LOGP(debug, "Rotation failed: new snp {:.2f}", updSnp);
193  return false;
194  }
195  value_t xold = this->getX(), yold = this->getY();
196  this->setAlpha(alpha);
197  this->setX(xold * ca + yold * sa);
198  this->setY(-xold * sa + yold * ca);
199  this->setSnp(updSnp);
200 
201  if (gpu::CAMath::Abs(csp) < constants::math::Almost0) {
202  LOGP(debug, "Too small cosine value {:f}", csp);
203  csp = constants::math::Almost0;
204  }
205 
206  value_t rr = (ca + snp / csp * sa);
207 
208  mC[kSigY2] *= (ca * ca);
209  mC[kSigZY] *= ca;
210  mC[kSigSnpY] *= ca * rr;
211  mC[kSigSnpZ] *= rr;
212  mC[kSigSnp2] *= rr * rr;
213  mC[kSigTglY] *= ca;
214  mC[kSigTglSnp] *= rr;
215  mC[kSigQ2PtY] *= ca;
216  mC[kSigQ2PtSnp] *= rr;
217 
218  checkCovariance();
219  return true;
220 }

223 template <typename value_T>
224 GPUd() bool TrackParametrizationWithError<value_T>::propagateToDCA(const o2::dataformats::VertexBase& vtx, value_t b, o2::dataformats::DCA* dca, value_t maxD)
225 {
226  // propagate track to DCA to the vertex
227  value_t sn, cs, alp = this->getAlpha();
228  o2::math_utils::detail::sincos(alp, sn, cs);
229  value_t x = this->getX(), y = this->getY(), snp = this->getSnp(), csp = gpu::CAMath::Sqrt((1.f - snp) * (1.f + snp));
230  value_t xv = vtx.getX() * cs + vtx.getY() * sn, yv = -vtx.getX() * sn + vtx.getY() * cs, zv = vtx.getZ();
231  x -= xv;
232  y -= yv;
233  //Estimate the impact parameter neglecting the track curvature
234  value_t d = gpu::CAMath::Abs(x * snp - y * csp);
235  if (d > maxD) {
236  return false;
237  }
238  value_t crv = this->getCurvature(b);
239  value_t tgfv = -(crv * x - snp) / (crv * y + csp);
240  sn = tgfv / gpu::CAMath::Sqrt(1.f + tgfv * tgfv);
241  cs = gpu::CAMath::Sqrt((1.f - sn) * (1.f + sn));
242  cs = (gpu::CAMath::Abs(tgfv) > constants::math::Almost0) ? sn / tgfv : constants::math::Almost1;
243 
244  x = xv * cs + yv * sn;
245  yv = -xv * sn + yv * cs;
246  xv = x;
247 
248  auto tmpT(*this); // operate on the copy to recover after the failure
249  alp += gpu::CAMath::ASin(sn);
250  if (!tmpT.rotate(alp) || !tmpT.propagateTo(xv, b)) {
251 #ifndef GPUCA_ALIGPUCODE
252  LOG(debug) << "failed to propagate to alpha=" << alp << " X=" << xv << vtx << " | Track is: " << tmpT.asString();
253 #else
254  LOG(debug) << "failed to propagate to alpha=" << alp << " X=" << xv << vtx;
255  ;
256 #endif
257  return false;
258  }
259  *this = tmpT;
260  if (dca) {
261  o2::math_utils::detail::sincos(alp, sn, cs);
262  auto s2ylocvtx = vtx.getSigmaX2() * sn * sn + vtx.getSigmaY2() * cs * cs - 2. * vtx.getSigmaXY() * cs * sn;
263  dca->set(this->getY() - yv, this->getZ() - zv, getSigmaY2() + s2ylocvtx, getSigmaZY(), getSigmaZ2() + vtx.getSigmaZ2());
264  }
265  return true;
266 }

278 template <typename value_T>
279 GPUd() void TrackParametrizationWithError<value_T>::set(const dim3_t& xyz, const dim3_t& pxpypz,
280  const gpu::gpustd::array<value_t, kLabCovMatSize>& cv, int charge, bool sectorAlpha, const PID pid)
281 {
282  // set track param and covariance from kinematics and lab errors
283 
284  // Alpha of the frame is defined as:
285  // sectorAlpha == false : -> angle of pt direction
286  // sectorAlpha == true : -> angle of the sector from X,Y coordinate for r>1
287  // angle of pt direction for r==0
288  //
289  //
290  constexpr value_t kSafe = 1e-5f;
291  value_t radPos2 = xyz[0] * xyz[0] + xyz[1] * xyz[1];
292  value_t alp = 0;
293  if (sectorAlpha || radPos2 < 1) {
294  alp = gpu::CAMath::ATan2(pxpypz[1], pxpypz[0]);
295  } else {
296  alp = gpu::CAMath::ATan2(xyz[1], xyz[0]);
297  }
298  if (sectorAlpha) {
299  alp = math_utils::detail::angle2Alpha<value_t>(alp);
300  }
301  //
302  value_t sn, cs;
303  math_utils::detail::sincos(alp, sn, cs);
304  // protection against cosp<0
305  if (cs * pxpypz[0] + sn * pxpypz[1] < 0) {
306  LOG(debug) << "alpha from phiPos() will invalidate this track parameters, overriding to alpha from phi()";
307  alp = gpu::CAMath::ATan2(pxpypz[1], pxpypz[0]);
308  if (sectorAlpha) {
309  alp = math_utils::detail::angle2Alpha<value_t>(alp);
310  }
311  math_utils::detail::sincos(alp, sn, cs);
312  }
313  // protection: avoid alpha being too close to 0 or +-pi/2
314  if (gpu::CAMath::Abs(sn) < 2.f * kSafe) {
315  if (alp > 0) {
316  alp += alp < constants::math::PIHalf ? 2.f * kSafe : -2.f * kSafe;
317  } else {
318  alp += alp > -constants::math::PIHalf ? -2.f * kSafe : 2.f * kSafe;
319  }
320  math_utils::detail::sincos(alp, sn, cs);
321  } else if (gpu::CAMath::Abs(cs) < 2.f * kSafe) {
322  if (alp > 0) {
323  alp += alp > constants::math::PIHalf ? 2.f * kSafe : -2.f * kSafe;
324  } else {
325  alp += alp > -constants::math::PIHalf ? 2.f * kSafe : -2.f * kSafe;
326  }
327  math_utils::detail::sincos(alp, sn, cs);
328  }
329  // get the vertex of origin and the momentum
330  dim3_t ver{xyz[0], xyz[1], xyz[2]};
331  dim3_t mom{pxpypz[0], pxpypz[1], pxpypz[2]};
332  //
333  // Rotate to the local coordinate system
334  math_utils::detail::rotateZ<value_t>(ver, -alp);
335  math_utils::detail::rotateZ<value_t>(mom, -alp);
336  //
337  value_t pt = gpu::CAMath::Sqrt(mom[0] * mom[0] + mom[1] * mom[1]);
338  value_t ptI = 1.f / pt;
339  this->setX(ver[0]);
340  this->setAlpha(alp);
341  this->setY(ver[1]);
342  this->setZ(ver[2]);
343  this->setSnp(mom[1] * ptI); // cos(phi)
344  this->setTgl(mom[2] * ptI); // tg(lambda)
345  this->setAbsCharge(gpu::CAMath::Abs(charge));
346  this->setQ2Pt(charge ? ptI * charge : ptI);
347  this->setPID(pid);
348  //
349  if (gpu::CAMath::Abs(1.f - this->getSnp()) < kSafe) {
350  this->setSnp(1.f - kSafe); // Protection
351  } else if (gpu::CAMath::Abs(-1.f - this->getSnp()) < kSafe) {
352  this->setSnp(-1.f + kSafe); // Protection
353  }
354  //
355  // Covariance matrix (formulas to be simplified)
356  value_t r = mom[0] * ptI; // cos(phi)
357  value_t cv34 = gpu::CAMath::Sqrt(cv[3] * cv[3] + cv[4] * cv[4]);
358  //
359  int special = 0;
360  value_t sgcheck = r * sn + this->getSnp() * cs;
361  if (gpu::CAMath::Abs(sgcheck) > 1 - kSafe) { // special case: lab phi is +-pi/2
362  special = 1;
363  sgcheck = sgcheck < 0 ? -1.f : 1.f;
364  } else if (gpu::CAMath::Abs(sgcheck) < kSafe) {
365  sgcheck = cs < 0 ? -1.0f : 1.0f;
366  special = 2; // special case: lab phi is 0
367  }
368  //
369  mC[kSigY2] = cv[0] + cv[2];
370  mC[kSigZY] = (-cv[3] * sn) < 0 ? -cv34 : cv34;
371  mC[kSigZ2] = cv[5];
372  //
373  value_t ptI2 = ptI * ptI;
374  value_t tgl2 = this->getTgl() * this->getTgl();
375  if (special == 1) {
376  mC[kSigSnpY] = cv[6] * ptI;
377  mC[kSigSnpZ] = -sgcheck * cv[8] * r * ptI;
378  mC[kSigSnp2] = gpu::CAMath::Abs(cv[9] * r * r * ptI2);
379  mC[kSigTglY] = (cv[10] * this->getTgl() - sgcheck * cv[15]) * ptI / r;
380  mC[kSigTglZ] = (cv[17] - sgcheck * cv[12] * this->getTgl()) * ptI;
381  mC[kSigTglSnp] = (-sgcheck * cv[18] + cv[13] * this->getTgl()) * r * ptI2;
382  mC[kSigTgl2] = gpu::CAMath::Abs(cv[20] - 2 * sgcheck * cv[19] * mC[4] + cv[14] * tgl2) * ptI2;
383  mC[kSigQ2PtY] = cv[10] * ptI2 / r * charge;
384  mC[kSigQ2PtZ] = -sgcheck * cv[12] * ptI2 * charge;
385  mC[kSigQ2PtSnp] = cv[13] * r * ptI * ptI2 * charge;
386  mC[kSigQ2PtTgl] = (-sgcheck * cv[19] + cv[14] * this->getTgl()) * r * ptI2 * ptI;
387  mC[kSigQ2Pt2] = gpu::CAMath::Abs(cv[14] * ptI2 * ptI2);
388  } else if (special == 2) {
389  mC[kSigSnpY] = -cv[10] * ptI * cs / sn;
390  mC[kSigSnpZ] = cv[12] * cs * ptI;
391  mC[kSigSnp2] = gpu::CAMath::Abs(cv[14] * cs * cs * ptI2);
392  mC[kSigTglY] = (sgcheck * cv[6] * this->getTgl() - cv[15]) * ptI / sn;
393  mC[kSigTglZ] = (cv[17] - sgcheck * cv[8] * this->getTgl()) * ptI;
394  mC[kSigTglSnp] = (cv[19] - sgcheck * cv[13] * this->getTgl()) * cs * ptI2;
395  mC[kSigTgl2] = gpu::CAMath::Abs(cv[20] - 2 * sgcheck * cv[18] * this->getTgl() + cv[9] * tgl2) * ptI2;
396  mC[kSigQ2PtY] = sgcheck * cv[6] * ptI2 / sn * charge;
397  mC[kSigQ2PtZ] = -sgcheck * cv[8] * ptI2 * charge;
398  mC[kSigQ2PtSnp] = -sgcheck * cv[13] * cs * ptI * ptI2 * charge;
399  mC[kSigQ2PtTgl] = (-sgcheck * cv[18] + cv[9] * this->getTgl()) * ptI2 * ptI * charge;
400  mC[kSigQ2Pt2] = gpu::CAMath::Abs(cv[9] * ptI2 * ptI2);
401  } else {
402  double m00 = -sn; // m10=cs;
403  double m23 = -pt * (sn + this->getSnp() * cs / r), m43 = -pt * pt * (r * cs - this->getSnp() * sn);
404  double m24 = pt * (cs - this->getSnp() * sn / r), m44 = -pt * pt * (r * sn + this->getSnp() * cs);
405  double m35 = pt, m45 = -pt * pt * this->getTgl();
406  //
407  if (charge) { // RS: this is a hack, proper treatment to be implemented
408  m43 *= charge;
409  m44 *= charge;
410  m45 *= charge;
411  }
412  //
413  double a1 = cv[13] - cv[9] * (m23 * m44 + m43 * m24) / m23 / m43;
414  double a2 = m23 * m24 - m23 * (m23 * m44 + m43 * m24) / m43;
415  double a3 = m43 * m44 - m43 * (m23 * m44 + m43 * m24) / m23;
416  double a4 = cv[14] + 2. * cv[9];
417  double a5 = m24 * m24 - 2. * m24 * m44 * m23 / m43;
418  double a6 = m44 * m44 - 2. * m24 * m44 * m43 / m23;
419  //
420  mC[kSigSnpY] = (cv[10] * m43 - cv[6] * m44) / (m24 * m43 - m23 * m44) / m00;
421  mC[kSigQ2PtY] = (cv[6] / m00 - mC[kSigSnpY] * m23) / m43;
422  mC[kSigTglY] = (cv[15] / m00 - mC[kSigQ2PtY] * m45) / m35;
423  mC[kSigSnpZ] = (cv[12] * m43 - cv[8] * m44) / (m24 * m43 - m23 * m44);
424  mC[kSigQ2PtZ] = (cv[8] - mC[kSigSnpZ] * m23) / m43;
425  mC[kSigTglZ] = cv[17] / m35 - mC[kSigQ2PtZ] * m45 / m35;
426  mC[kSigSnp2] = gpu::CAMath::Abs((a4 * a3 - a6 * a1) / (a5 * a3 - a6 * a2));
427  mC[kSigQ2Pt2] = gpu::CAMath::Abs((a1 - a2 * mC[kSigSnp2]) / a3);
428  mC[kSigQ2PtSnp] = (cv[9] - mC[kSigSnp2] * m23 * m23 - mC[kSigQ2Pt2] * m43 * m43) / m23 / m43;
429  double b1 = cv[18] - mC[kSigQ2PtSnp] * m23 * m45 - mC[kSigQ2Pt2] * m43 * m45;
430  double b2 = m23 * m35;
431  double b3 = m43 * m35;
432  double b4 = cv[19] - mC[kSigQ2PtSnp] * m24 * m45 - mC[kSigQ2Pt2] * m44 * m45;
433  double b5 = m24 * m35;
434  double b6 = m44 * m35;
435  mC[kSigTglSnp] = (b4 - b6 * b1 / b3) / (b5 - b6 * b2 / b3);
436  mC[kSigQ2PtTgl] = b1 / b3 - b2 * mC[kSigTglSnp] / b3;
437  mC[kSigTgl2] = gpu::CAMath::Abs((cv[20] - mC[kSigQ2Pt2] * (m45 * m45) - mC[kSigQ2PtTgl] * 2.f * m35 * m45) / (m35 * m35));
438  }
439  checkCovariance();

443 template <typename value_T>
444 GPUd() bool TrackParametrizationWithError<value_T>::propagateTo(value_t xk, const dim3_t& b)
445 {
446  //----------------------------------------------------------------
447  // Extrapolate this track to the plane X=xk in the field b[].
448  //
449  // X [cm] is in the "tracking coordinate system" of this track.
450  // b[]={Bx,By,Bz} [kG] is in the Global coordidate system.
451  //----------------------------------------------------------------
452 
453  value_t dx = xk - this->getX();
454  if (gpu::CAMath::Abs(dx) < constants::math::Almost0) {
455  return true;
456  }
457  // Do not propagate tracks outside the ALICE detector
458  if (gpu::CAMath::Abs(dx) > 1e5 || gpu::CAMath::Abs(this->getY()) > 1e5 || gpu::CAMath::Abs(this->getZ()) > 1e5) {
459  LOGP(warning, "Anomalous track, target X:{:f}", xk);
460  // print();
461  return false;
462  }
463  value_t crv = (gpu::CAMath::Abs(b[2]) < constants::math::Almost0) ? 0.f : this->getCurvature(b[2]);
464  if (gpu::CAMath::Abs(crv) < constants::math::Almost0) {
465  return propagateTo(xk, 0.);
466  }
467  value_t x2r = crv * dx;
468  value_t f1 = this->getSnp(), f2 = f1 + x2r;
469  if ((gpu::CAMath::Abs(f1) > constants::math::Almost1) || (gpu::CAMath::Abs(f2) > constants::math::Almost1)) {
470  return false;
471  }
472  value_t r1 = gpu::CAMath::Sqrt((1.f - f1) * (1.f + f1));
473  if (gpu::CAMath::Abs(r1) < constants::math::Almost0) {
474  return false;
475  }
476  value_t r2 = gpu::CAMath::Sqrt((1.f - f2) * (1.f + f2));
477  if (gpu::CAMath::Abs(r2) < constants::math::Almost0) {
478  return false;
479  }
480 
481  value_t dy2dx = (f1 + f2) / (r1 + r2);
482  value_t step = (gpu::CAMath::Abs(x2r) < 0.05f) ? dx * gpu::CAMath::Abs(r2 + f2 * dy2dx) // chord
483  : 2.f * gpu::CAMath::ASin(0.5f * dx * gpu::CAMath::Sqrt(1.f + dy2dx * dy2dx) * crv) / crv; // arc
484  step *= gpu::CAMath::Sqrt(1.f + this->getTgl() * this->getTgl());
485  //
486  // get the track x,y,z,px/p,py/p,pz/p,p,sinAlpha,cosAlpha in the Global System
487  gpu::gpustd::array<value_t, 9> vecLab{0.f};
488  if (!this->getPosDirGlo(vecLab)) {
489  return false;
490  }
491  //
492  // matrix transformed with Bz component only
493  value_t &c00 = mC[kSigY2], &c10 = mC[kSigZY], &c11 = mC[kSigZ2], &c20 = mC[kSigSnpY], &c21 = mC[kSigSnpZ],
494  &c22 = mC[kSigSnp2], &c30 = mC[kSigTglY], &c31 = mC[kSigTglZ], &c32 = mC[kSigTglSnp], &c33 = mC[kSigTgl2],
495  &c40 = mC[kSigQ2PtY], &c41 = mC[kSigQ2PtZ], &c42 = mC[kSigQ2PtSnp], &c43 = mC[kSigQ2PtTgl],
496  &c44 = mC[kSigQ2Pt2];
497  // evaluate matrix in double prec.
498  double rinv = 1. / r1;
499  double r3inv = rinv * rinv * rinv;
500  double f24 = dx * b[2] * constants::math::B2C; // x2r/track[kQ2Pt];
501  double f02 = dx * r3inv;
502  double f04 = 0.5 * f24 * f02;
503  double f12 = f02 * this->getTgl() * f1;
504  double f14 = 0.5 * f24 * f12; // 0.5*f24*f02*getTgl()*f1;
505  double f13 = dx * rinv;
506 
507  // b = C*ft
508  double b00 = f02 * c20 + f04 * c40, b01 = f12 * c20 + f14 * c40 + f13 * c30;
509  double b02 = f24 * c40;
510  double b10 = f02 * c21 + f04 * c41, b11 = f12 * c21 + f14 * c41 + f13 * c31;
511  double b12 = f24 * c41;
512  double b20 = f02 * c22 + f04 * c42, b21 = f12 * c22 + f14 * c42 + f13 * c32;
513  double b22 = f24 * c42;
514  double b40 = f02 * c42 + f04 * c44, b41 = f12 * c42 + f14 * c44 + f13 * c43;
515  double b42 = f24 * c44;
516  double b30 = f02 * c32 + f04 * c43, b31 = f12 * c32 + f14 * c43 + f13 * c33;
517  double b32 = f24 * c43;
518 
519  // a = f*b = f*C*ft
520  double a00 = f02 * b20 + f04 * b40, a01 = f02 * b21 + f04 * b41, a02 = f02 * b22 + f04 * b42;
521  double a11 = f12 * b21 + f14 * b41 + f13 * b31, a12 = f12 * b22 + f14 * b42 + f13 * b32;
522  double a22 = f24 * b42;
523 
524  // F*C*Ft = C + (b + bt + a)
525  c00 += b00 + b00 + a00;
526  c10 += b10 + b01 + a01;
527  c20 += b20 + b02 + a02;
528  c30 += b30;
529  c40 += b40;
530  c11 += b11 + b11 + a11;
531  c21 += b21 + b12 + a12;
532  c31 += b31;
533  c41 += b41;
534  c22 += b22 + b22 + a22;
535  c32 += b32;
536  c42 += b42;
537 
538  checkCovariance();
539 
540  // Rotate to the system where Bx=By=0.
541  value_t bxy2 = b[0] * b[0] + b[1] * b[1];
542  value_t bt = gpu::CAMath::Sqrt(bxy2);
543  value_t cosphi = 1.f, sinphi = 0.f;
544  if (bt > constants::math::Almost0) {
545  cosphi = b[0] / bt;
546  sinphi = b[1] / bt;
547  }
548  value_t bb = gpu::CAMath::Sqrt(bxy2 + b[2] * b[2]);
549  value_t costet = 1., sintet = 0.;
550  if (bb > constants::math::Almost0) {
551  costet = b[2] / bb;
552  sintet = bt / bb;
553  }
554  gpu::gpustd::array<value_t, 7> vect{costet * cosphi * vecLab[0] + costet * sinphi * vecLab[1] - sintet * vecLab[2],
555  -sinphi * vecLab[0] + cosphi * vecLab[1],
556  sintet * cosphi * vecLab[0] + sintet * sinphi * vecLab[1] + costet * vecLab[2],
557  costet * cosphi * vecLab[3] + costet * sinphi * vecLab[4] - sintet * vecLab[5],
558  -sinphi * vecLab[3] + cosphi * vecLab[4],
559  sintet * cosphi * vecLab[3] + sintet * sinphi * vecLab[4] + costet * vecLab[5],
560  vecLab[6]};
561 
562  // Do the helix step
563  value_t sgn = this->getSign();
564  g3helx3(sgn * bb, step, vect);
565 
566  // Rotate back to the Global System
567  vecLab[0] = cosphi * costet * vect[0] - sinphi * vect[1] + cosphi * sintet * vect[2];
568  vecLab[1] = sinphi * costet * vect[0] + cosphi * vect[1] + sinphi * sintet * vect[2];
569  vecLab[2] = -sintet * vect[0] + costet * vect[2];
570 
571  vecLab[3] = cosphi * costet * vect[3] - sinphi * vect[4] + cosphi * sintet * vect[5];
572  vecLab[4] = sinphi * costet * vect[3] + cosphi * vect[4] + sinphi * sintet * vect[5];
573  vecLab[5] = -sintet * vect[3] + costet * vect[5];
574 
575  // Rotate back to the Tracking System
576  value_t sinalp = -vecLab[7], cosalp = vecLab[8];
577  value_t t = cosalp * vecLab[0] - sinalp * vecLab[1];
578  vecLab[1] = sinalp * vecLab[0] + cosalp * vecLab[1];
579  vecLab[0] = t;
580  t = cosalp * vecLab[3] - sinalp * vecLab[4];
581  vecLab[4] = sinalp * vecLab[3] + cosalp * vecLab[4];
582  vecLab[3] = t;
583 
584  // Do the final correcting step to the target plane (linear approximation)
585  value_t x = vecLab[0], y = vecLab[1], z = vecLab[2];
586  if (gpu::CAMath::Abs(dx) > constants::math::Almost0) {
587  if (gpu::CAMath::Abs(vecLab[3]) < constants::math::Almost0) {
588  return false;
589  }
590  dx = xk - vecLab[0];
591  x += dx;
592  y += vecLab[4] / vecLab[3] * dx;
593  z += vecLab[5] / vecLab[3] * dx;
594  }
595 
596  // Calculate the track parameters
597  t = 1.f / gpu::CAMath::Sqrt(vecLab[3] * vecLab[3] + vecLab[4] * vecLab[4]);
598  this->setX(xk);
599  this->setY(y);
600  this->setZ(z);
601  this->setSnp(vecLab[4] * t);
602  this->setTgl(vecLab[5] * t);
603  this->setQ2Pt(sgn * t / vecLab[6]);
604 
605  return true;
606 }

609 template <typename value_T>
610 GPUd() void TrackParametrizationWithError<value_T>::checkCovariance()
611 {
612  // This function forces the diagonal elements of the covariance matrix to be positive.
613  // In case the diagonal element is bigger than the maximal allowed value, it is set to
614  // the limit and the off-diagonal elements that correspond to it are set to zero.
615 
616  mC[kSigY2] = gpu::CAMath::Abs(mC[kSigY2]);
617  if (mC[kSigY2] > kCY2max) {
618  value_t scl = gpu::CAMath::Sqrt(kCY2max / mC[kSigY2]);
619  mC[kSigY2] = kCY2max;
620  mC[kSigZY] *= scl;
621  mC[kSigSnpY] *= scl;
622  mC[kSigTglY] *= scl;
623  mC[kSigQ2PtY] *= scl;
624  }
625  mC[kSigZ2] = gpu::CAMath::Abs(mC[kSigZ2]);
626  if (mC[kSigZ2] > kCZ2max) {
627  value_t scl = gpu::CAMath::Sqrt(kCZ2max / mC[kSigZ2]);
628  mC[kSigZ2] = kCZ2max;
629  mC[kSigZY] *= scl;
630  mC[kSigSnpZ] *= scl;
631  mC[kSigTglZ] *= scl;
632  mC[kSigQ2PtZ] *= scl;
633  }
634  mC[kSigSnp2] = gpu::CAMath::Abs(mC[kSigSnp2]);
635  if (mC[kSigSnp2] > kCSnp2max) {
636  value_t scl = gpu::CAMath::Sqrt(kCSnp2max / mC[kSigSnp2]);
637  mC[kSigSnp2] = kCSnp2max;
638  mC[kSigSnpY] *= scl;
639  mC[kSigSnpZ] *= scl;
640  mC[kSigTglSnp] *= scl;
641  mC[kSigQ2PtSnp] *= scl;
642  }
643  mC[kSigTgl2] = gpu::CAMath::Abs(mC[kSigTgl2]);
644  if (mC[kSigTgl2] > kCTgl2max) {
645  value_t scl = gpu::CAMath::Sqrt(kCTgl2max / mC[kSigTgl2]);
646  mC[kSigTgl2] = kCTgl2max;
647  mC[kSigTglY] *= scl;
648  mC[kSigTglZ] *= scl;
649  mC[kSigTglSnp] *= scl;
650  mC[kSigQ2PtTgl] *= scl;
651  }
652  mC[kSigQ2Pt2] = gpu::CAMath::Abs(mC[kSigQ2Pt2]);
653  if (mC[kSigQ2Pt2] > kC1Pt2max) {
654  value_t scl = gpu::CAMath::Sqrt(kC1Pt2max / mC[kSigQ2Pt2]);
655  mC[kSigQ2Pt2] = kC1Pt2max;
656  mC[kSigQ2PtY] *= scl;
657  mC[kSigQ2PtZ] *= scl;
658  mC[kSigQ2PtSnp] *= scl;
659  mC[kSigQ2PtTgl] *= scl;
660  }
661 }

702 template <typename value_T>
703 GPUd() auto TrackParametrizationWithError<value_T>::getPredictedChi2(const value_t* p, const value_t* cov) const -> value_t
704 {
705  // Estimate the chi2 of the space point "p" with the cov. matrix "cov"
706  auto sdd = static_cast<double>(getSigmaY2()) + static_cast<double>(cov[0]);
707  auto sdz = static_cast<double>(getSigmaZY()) + static_cast<double>(cov[1]);
708  auto szz = static_cast<double>(getSigmaZ2()) + static_cast<double>(cov[2]);
709  auto det = sdd * szz - sdz * sdz;
710 
711  if (gpu::CAMath::Abs(det) < constants::math::Almost0) {
712  return constants::math::VeryBig;
713  }
714 
715  value_t d = this->getY() - p[0];
716  value_t z = this->getZ() - p[1];
717 
718  return (d * (szz * d - sdz * z) + z * (sdd * z - d * sdz)) / det;
719 }

754 template <typename value_T>
755 auto TrackParametrizationWithError<value_T>::getPredictedChi2(const TrackParametrizationWithError<value_T>& rhs, MatrixDSym5& covToSet) const -> value_t
756 {
757  // get chi2 wrt other track, which must be defined at the same parameters X,alpha
758  // Supplied non-initialized covToSet matrix is filled by inverse combined matrix for further use
759 
760  if (gpu::CAMath::Abs(this->getAlpha() - rhs.getAlpha()) > FLT_EPSILON) {
761  LOG(error) << "The reference Alpha of the tracks differ: " << this->getAlpha() << " : " << rhs.getAlpha();
762  return 2.f * HugeF;
763  }
764  if (gpu::CAMath::Abs(this->getX() - rhs.getX()) > FLT_EPSILON) {
765  LOG(error) << "The reference X of the tracks differ: " << this->getX() << " : " << rhs.getX();
766  return 2.f * HugeF;
767  }
768  buildCombinedCovMatrix(rhs, covToSet);
769  if (!covToSet.Invert()) {
770  LOG(warning) << "Cov.matrix inversion failed: " << covToSet;
771  return 2.f * HugeF;
772  }
773  double chi2diag = 0., chi2ndiag = 0., diff[kNParams];
774  for (int i = kNParams; i--;) {
775  diff[i] = this->getParam(i) - rhs.getParam(i);
776  chi2diag += diff[i] * diff[i] * covToSet(i, i);
777  }
778  for (int i = kNParams; i--;) {
779  for (int j = i; j--;) {
780  chi2ndiag += diff[i] * diff[j] * covToSet(i, j);
781  }
782  }
783  return chi2diag + 2. * chi2ndiag;
784 }

787 template <typename value_T>
788 bool TrackParametrizationWithError<value_T>::update(const TrackParametrizationWithError<value_T>& rhs, const MatrixDSym5& covInv)
789 {
790  // update track with other track, the inverted combined cov matrix should be supplied
791 
792  // consider skipping this check, since it is usually already done upstream
793  if (gpu::CAMath::Abs(this->getAlpha() - rhs.getAlpha()) > FLT_EPSILON) {
794  LOG(error) << "The reference Alpha of the tracks differ: " << this->getAlpha() << " : " << rhs.getAlpha();
795  return false;
796  }
797  if (gpu::CAMath::Abs(this->getX() - rhs.getX()) > FLT_EPSILON) {
798  LOG(error) << "The reference X of the tracks differ: " << this->getX() << " : " << rhs.getX();
799  return false;
800  }
801 
802  // gain matrix K = Cov0*H*(Cov0+Cov0)^-1 (for measurement matrix H=I)
803  MatrixDSym5 matC0;
804  matC0(kY, kY) = getSigmaY2();
805  matC0(kZ, kY) = getSigmaZY();
806  matC0(kZ, kZ) = getSigmaZ2();
807  matC0(kSnp, kY) = getSigmaSnpY();
808  matC0(kSnp, kZ) = getSigmaSnpZ();
809  matC0(kSnp, kSnp) = getSigmaSnp2();
810  matC0(kTgl, kY) = getSigmaTglY();
811  matC0(kTgl, kZ) = getSigmaTglZ();
812  matC0(kTgl, kSnp) = getSigmaTglSnp();
813  matC0(kTgl, kTgl) = getSigmaTgl2();
814  matC0(kQ2Pt, kY) = getSigma1PtY();
815  matC0(kQ2Pt, kZ) = getSigma1PtZ();
816  matC0(kQ2Pt, kSnp) = getSigma1PtSnp();
817  matC0(kQ2Pt, kTgl) = getSigma1PtTgl();
818  matC0(kQ2Pt, kQ2Pt) = getSigma1Pt2();
819  MatrixD5 matK = matC0 * covInv;
820 
821  // updated state vector: x = K*(x1-x0)
822  // RS: why SMatix, SVector does not provide multiplication operators ???
823  double diff[kNParams];
824  for (int i = kNParams; i--;) {
825  diff[i] = rhs.getParam(i) - this->getParam(i);
826  }
827  for (int i = kNParams; i--;) {
828  for (int j = kNParams; j--;) {
829  this->updateParam(matK(i, j) * diff[j], i);
830  }
831  }
832 
833  // updated covariance: Cov0 = Cov0 - K*Cov0
834  matK *= ROOT::Math::SMatrix<double, kNParams, kNParams, ROOT::Math::MatRepStd<double, kNParams>>(matC0);
835  mC[kSigY2] -= matK(kY, kY);
836  mC[kSigZY] -= matK(kZ, kY);
837  mC[kSigZ2] -= matK(kZ, kZ);
838  mC[kSigSnpY] -= matK(kSnp, kY);
839  mC[kSigSnpZ] -= matK(kSnp, kZ);
840  mC[kSigSnp2] -= matK(kSnp, kSnp);
841  mC[kSigTglY] -= matK(kTgl, kY);
842  mC[kSigTglZ] -= matK(kTgl, kZ);
843  mC[kSigTglSnp] -= matK(kTgl, kSnp);
844  mC[kSigTgl2] -= matK(kTgl, kTgl);
845  mC[kSigQ2PtY] -= matK(kQ2Pt, kY);
846  mC[kSigQ2PtZ] -= matK(kQ2Pt, kZ);
847  mC[kSigQ2PtSnp] -= matK(kQ2Pt, kSnp);
848  mC[kSigQ2PtTgl] -= matK(kQ2Pt, kTgl);
849  mC[kSigQ2Pt2] -= matK(kQ2Pt, kQ2Pt);

953 template <typename value_T>
954 GPUd() bool TrackParametrizationWithError<value_T>::correctForMaterial(value_t x2x0, value_t xrho, bool anglecorr, value_t dedx)
955 {
956  //------------------------------------------------------------------
957  // This function corrects the track parameters for the crossed material.
958  // "x2x0" - X/X0, the thickness in units of the radiation length.
959  // "xrho" - is the product length*density (g/cm^2).
960  // It should be passed as negative when propagating tracks
961  // from the intreaction point to the outside of the central barrel.
962  // "dedx" - mean enery loss (GeV/(g/cm^2), if <=kCalcdEdxAuto : calculate on the fly
963  // "anglecorr" - switch for the angular correction
964  //------------------------------------------------------------------
965  constexpr value_t kMSConst2 = 0.0136f * 0.0136f;
966  constexpr value_t kMaxELossFrac = 0.3f; // max allowed fractional eloss
967  constexpr value_t kMinP = 0.01f; // kill below this momentum
968 
969  value_t& fC22 = mC[kSigSnp2];
970  value_t& fC33 = mC[kSigTgl2];
971  value_t& fC43 = mC[kSigQ2PtTgl];
972  value_t& fC44 = mC[kSigQ2Pt2];
973  //
974  value_t csp2 = (1.f - this->getSnp()) * (1.f + this->getSnp()); // cos(phi)^2
975  value_t cst2I = (1.f + this->getTgl() * this->getTgl()); // 1/cos(lambda)^2
976  // Apply angle correction, if requested
977  if (anglecorr) {
978  value_t angle = gpu::CAMath::Sqrt(cst2I / csp2);
979  x2x0 *= angle;
980  xrho *= angle;
981  }
982  value_t p = this->getP();
983  value_t p2 = p * p;
984  value_t e2 = p2 + this->getPID().getMass2();
985  value_t beta2 = p2 / e2;
986 
987  // Calculating the multiple scattering corrections******************
988  value_t cC22(0.f), cC33(0.f), cC43(0.f), cC44(0.f);
989  if (x2x0 != 0.f) {
990  value_t theta2 = kMSConst2 / (beta2 * p2) * gpu::CAMath::Abs(x2x0);
991  if (this->getAbsCharge() != 1) {
992  theta2 *= this->getAbsCharge() * this->getAbsCharge();
993  }
994  if (theta2 > constants::math::PI * constants::math::PI) {
995  return false;
996  }
997  value_t fp34 = this->getTgl() * this->getCharge2Pt();
998  value_t t2c2I = theta2 * cst2I;
999  cC22 = t2c2I * csp2;
1000  cC33 = t2c2I * cst2I;
1001  cC43 = t2c2I * fp34;
1002  cC44 = theta2 * fp34 * fp34;
1003  // optimes this
1004  // cC22 = theta2*((1.-getSnp())*(1.+getSnp()))*(1. + this->getTgl()*getTgl());
1005  // cC33 = theta2*(1. + this->getTgl()*getTgl())*(1. + this->getTgl()*getTgl());
1006  // cC43 = theta2*getTgl()*this->getQ2Pt()*(1. + this->getTgl()*getTgl());
1007  // cC44 = theta2*getTgl()*this->getQ2Pt()*getTgl()*this->getQ2Pt();
1008  }
1009 
1010  // Calculating the energy loss corrections************************
1011  value_t cP4 = 1.f;
1012  if ((xrho != 0.f) && (beta2 < 1.f)) {
1013  if (dedx < kCalcdEdxAuto + constants::math::Almost1) { // request to calculate dedx on the fly
1014  dedx = BetheBlochSolid(p / this->getPID().getMass());
1015  if (this->getAbsCharge() != 1) {
1016  dedx *= this->getAbsCharge() * this->getAbsCharge();
1017  }
1018  }
1019 
1020  value_t dE = dedx * xrho;
1021  value_t e = gpu::CAMath::Sqrt(e2);
1022  if (gpu::CAMath::Abs(dE) > kMaxELossFrac * e) {
1023  return false; // 30% energy loss is too much!
1024  }
1025  value_t eupd = e + dE;
1026  value_t pupd2 = eupd * eupd - this->getPID().getMass2();
1027  if (pupd2 < kMinP * kMinP) {
1028  return false;
1029  }
1030  cP4 = p / gpu::CAMath::Sqrt(pupd2);
1031  //
1032  // Approximate energy loss fluctuation (M.Ivanov)
1033  constexpr value_t knst = 0.07f; // To be tuned.
1034  value_t sigmadE = knst * gpu::CAMath::Sqrt(gpu::CAMath::Abs(dE)) * e / p2 * this->getCharge2Pt();
1035  cC44 += sigmadE * sigmadE;
1036  }
1037 
1038  // Applying the corrections*****************************
1039  fC22 += cC22;
1040  fC33 += cC33;
1041  fC43 += cC43;
1042  fC44 += cC44;
1043  this->setQ2Pt(this->getQ2Pt() * cP4);
1044 
1045  checkCovariance();
1046 
1047  return true;
1048 }

1052 GPUd() bool TrackParametrizationWithError<value_T>::getCovXYZPxPyPzGlo(gpu::gpustd::array<value_t, kLabCovMatSize>& cv) const
1053 {
1054  //---------------------------------------------------------------------
1055  // This function returns the global covariance matrix of the track params
1056  //
1057  // Cov(x,x) ... : cv[0]
1058  // Cov(y,x) ... : cv[1] cv[2]
1059  // Cov(z,x) ... : cv[3] cv[4] cv[5]
1060  // Cov(px,x)... : cv[6] cv[7] cv[8] cv[9]
1061  // Cov(py,x)... : cv[10] cv[11] cv[12] cv[13] cv[14]
1062  // Cov(pz,x)... : cv[15] cv[16] cv[17] cv[18] cv[19] cv[20]
1063  //
1064  // Results for (nearly) straight tracks are meaningless !
1065  //---------------------------------------------------------------------
1066  if (gpu::CAMath::Abs(this->getQ2Pt()) <= constants::math::Almost0 || gpu::CAMath::Abs(this->getSnp()) > constants::math::Almost1) {
1067  for (int i = 0; i < 21; i++) {
1068  cv[i] = 0.;
1069  }
1070  return false;
1071  }
1072 
1073  auto pt = this->getPt();
1074  value_t sn, cs;
1075  o2::math_utils::detail::sincos(this->getAlpha(), sn, cs);
1076  auto r = gpu::CAMath::Sqrt((1. - this->getSnp()) * (1. + this->getSnp()));
1077  auto m00 = -sn, m10 = cs;
1078  auto m23 = -pt * (sn + this->getSnp() * cs / r), m43 = -pt * pt * (r * cs - this->getSnp() * sn);
1079  auto m24 = pt * (cs - this->getSnp() * sn / r), m44 = -pt * pt * (r * sn + this->getSnp() * cs);
1080  auto m35 = pt, m45 = -pt * pt * this->getTgl();
1081 
1082  if (this->getSign() < 0) {
1083  m43 = -m43;
1084  m44 = -m44;
1085  m45 = -m45;
1086  }
1087 
1088  cv[0] = mC[0] * m00 * m00;
1089  cv[1] = mC[0] * m00 * m10;
1090  cv[2] = mC[0] * m10 * m10;
1091  cv[3] = mC[1] * m00;
1092  cv[4] = mC[1] * m10;
1093  cv[5] = mC[2];
1094  cv[6] = m00 * (mC[3] * m23 + mC[10] * m43);
1095  cv[7] = m10 * (mC[3] * m23 + mC[10] * m43);
1096  cv[8] = mC[4] * m23 + mC[11] * m43;
1097  cv[9] = m23 * (mC[5] * m23 + mC[12] * m43) + m43 * (mC[12] * m23 + mC[14] * m43);
1098  cv[10] = m00 * (mC[3] * m24 + mC[10] * m44);
1099  cv[11] = m10 * (mC[3] * m24 + mC[10] * m44);
1100  cv[12] = mC[4] * m24 + mC[11] * m44;
1101  cv[13] = m23 * (mC[5] * m24 + mC[12] * m44) + m43 * (mC[12] * m24 + mC[14] * m44);
1102  cv[14] = m24 * (mC[5] * m24 + mC[12] * m44) + m44 * (mC[12] * m24 + mC[14] * m44);
1103  cv[15] = m00 * (mC[6] * m35 + mC[10] * m45);
1104  cv[16] = m10 * (mC[6] * m35 + mC[10] * m45);
1105  cv[17] = mC[7] * m35 + mC[11] * m45;
1106  cv[18] = m23 * (mC[8] * m35 + mC[12] * m45) + m43 * (mC[13] * m35 + mC[14] * m45);
1107  cv[19] = m24 * (mC[8] * m35 + mC[12] * m45) + m44 * (mC[13] * m35 + mC[14] * m45);
1108  cv[20] = m35 * (mC[9] * m35 + mC[13] * m45) + m45 * (mC[13] * m35 + mC[14] * m45);
1109 
1110  return true;
