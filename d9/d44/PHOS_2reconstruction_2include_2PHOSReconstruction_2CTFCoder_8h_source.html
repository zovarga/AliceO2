58 template <typename VEC>
59 void CTFCoder::encode(VEC& buff, const gsl::span<const TriggerRecord>& trigData, const gsl::span<const Cell>& cellData)
60 {
61  using MD = o2::ctf::Metadata::OptStore;
62  // what to do which each field: see o2::ctd::Metadata explanation
63  constexpr MD optField[CTF::getNBlocks()] = {
64  MD::EENCODE, // BLC_bcIncTrig
65  MD::EENCODE, // BLC_orbitIncTrig
66  MD::EENCODE, // BLC_entriesTrig
67  MD::EENCODE, // BLC_packedID
68  MD::EENCODE, // BLC_time
69  MD::EENCODE, // BLC_energy
70  MD::EENCODE // BLC_status
71  };
72 
73  CTFHelper helper(trigData, cellData);
74 
75  // book output size with some margin
76  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
77  buff.resize(szIni);
78 
79  auto ec = CTF::create(buff);
80  using ECB = CTF::base;
81 
82  ec->setHeader(helper.createHeader());
83  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
84  ec->getANSHeader().majorVersion = 0;
85  ec->getANSHeader().minorVersion = 1;
86  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
87 #define ENCODEPHS(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
88  // clang-format off
89  ENCODEPHS(helper.begin_bcIncTrig(), helper.end_bcIncTrig(), CTF::BLC_bcIncTrig, 0);
90  ENCODEPHS(helper.begin_orbitIncTrig(), helper.end_orbitIncTrig(), CTF::BLC_orbitIncTrig, 0);
91  ENCODEPHS(helper.begin_entriesTrig(), helper.end_entriesTrig(), CTF::BLC_entriesTrig, 0);
92 
93  ENCODEPHS(helper.begin_packedID(), helper.end_packedID(), CTF::BLC_packedID, 0);
94  ENCODEPHS(helper.begin_time(), helper.end_time(), CTF::BLC_time, 0);
95  ENCODEPHS(helper.begin_energy(), helper.end_energy(), CTF::BLC_energy, 0);
96  ENCODEPHS(helper.begin_status(), helper.end_status(), CTF::BLC_status, 0);
97  // clang-format on
98  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
99 }

102 template <typename VTRG, typename VCELL>
103 void CTFCoder::decode(const CTF::base& ec, VTRG& trigVec, VCELL& cellVec)
104 {
105  const auto& header = ec.getHeader();
106  checkDictVersion(static_cast<const o2::ctf::CTFDictHeader&>(header));
107  ec.print(getPrefix(), mVerbosity);
108  std::vector<uint16_t> bcInc, entries, energy, cellTime, packedID;
109  std::vector<uint32_t> orbitInc;
110  std::vector<uint8_t> status;
111 
112 #define DECODEPHOS(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
113  // clang-format off
114  DECODEPHOS(bcInc, CTF::BLC_bcIncTrig);
115  DECODEPHOS(orbitInc, CTF::BLC_orbitIncTrig);
116  DECODEPHOS(entries, CTF::BLC_entriesTrig);
117  DECODEPHOS(packedID, CTF::BLC_packedID);
118 
119  DECODEPHOS(cellTime, CTF::BLC_time);
120  DECODEPHOS(energy, CTF::BLC_energy);
121  DECODEPHOS(status, CTF::BLC_status);
122  // clang-format on
123  //
124  trigVec.clear();
125  cellVec.clear();
126  trigVec.reserve(header.nTriggers);
127  status.reserve(header.nCells);
128 
129  uint32_t firstEntry = 0, cellCount = 0;
130  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
131 
132  Cell cell;
133  for (uint32_t itrig = 0; itrig < header.nTriggers; itrig++) {
134  // restore TrigRecord
135  if (orbitInc[itrig]) { // non-0 increment => new orbit
136  ir.bc = bcInc[itrig]; // bcInc has absolute meaning
137  ir.orbit += orbitInc[itrig];
138  } else {
139  ir.bc += bcInc[itrig];
140  }
141 
142  firstEntry = cellVec.size();
143  for (uint16_t ic = 0; ic < entries[itrig]; ic++) {
144  cell.setPacked(packedID[cellCount], cellTime[cellCount], energy[cellCount], status[cellCount]);
145  cellVec.emplace_back(cell);
146  cellCount++;
147  }
148  trigVec.emplace_back(ir, firstEntry, entries[itrig]);
149  }
150  assert(cellCount == header.nCells);
151 }
