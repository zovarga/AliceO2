57 bool ComputingQuotaEvaluator::selectOffer(int task, ComputingQuotaRequest const& selector, uint64_t now)
58 {
59  auto selectOffer = [&offers = this->mOffers, &infos = this->mInfos, task](int ref, uint64_t now) {
60  auto& selected = offers[ref];
61  auto& info = infos[ref];
62  selected.user = task;
63  if (info.firstUsed == 0) {
64  info.firstUsed = now;
65  }
66  info.lastUsed = now;
67  };
68 
69  ComputingQuotaOffer accumulated;
70  static QuotaEvaluatorStats stats;
71 
72  stats.invalidOffers.clear();
73  stats.otherUser.clear();
74  stats.unexpiring.clear();
75  stats.selectedOffers.clear();
76  stats.expired.clear();
77 
78  auto summarizeWhatHappended = [](bool enough, std::vector<int> const& result, ComputingQuotaOffer const& totalOffer, QuotaEvaluatorStats& stats) -> bool {
79  if (result.size() == 1 && result[0] == 0) {
80  // LOG(LOGLEVEL) << "No particular resource was requested, so we schedule task anyways";
81  return enough;
82  }
83  if (enough) {
84  LOGP(LOGLEVEL, "{} offers were selected for a total of: cpu {}, memory {}, shared memory {}", result.size(), totalOffer.cpu, totalOffer.memory, totalOffer.sharedMemory);
85  LOGP(LOGLEVEL, " The following offers were selected for computation: {} ", fmt::join(result, ","));
86  } else {
87  LOG(LOGLEVEL) << "No offer was selected";
88  if (result.size()) {
89  LOGP(LOGLEVEL, " The following offers were selected for computation but not enough: {} ", fmt::join(result, ","));
90  }
91  }
92  if (stats.invalidOffers.size()) {
93  LOGP(LOGLEVEL, " The following offers were invalid: {}", fmt::join(stats.invalidOffers, ", "));
94  }
95  if (stats.otherUser.size()) {
96  LOGP(LOGLEVEL, " The following offers were owned by other users: {}", fmt::join(stats.otherUser, ", "));
97  }
98  if (stats.expired.size()) {
99  LOGP(LOGLEVEL, " The following offers are expired: {}", fmt::join(stats.expired, ", "));
100  }
101  if (stats.unexpiring.size() > 1) {
102  LOGP(LOGLEVEL, " The following offers will never expire: {}", fmt::join(stats.unexpiring, ", "));
103  }
104 
105  return enough;
106  };
107 
108  bool enough = false;
109 
110  for (int i = 0; i != mOffers.size(); ++i) {
111  auto& offer = mOffers[i];
112  auto& info = mInfos[i];
113  if (enough) {
114  break;
115  }
116  // Ignore:
117  // - Invalid offers
118  // - Offers which belong to another task
119  // - Expired offers
120  if (offer.valid == false) {
121  stats.invalidOffers.push_back(i);
122  continue;
123  }
124  if (offer.user != -1 && offer.user != task) {
125  stats.otherUser.push_back(i);
126  continue;
127  }
128  if (offer.runtime < 0) {
129  stats.unexpiring.push_back(i);
130  } else if (offer.runtime + info.received < now) {
131  LOGP(LOGLEVEL, "Offer {} expired since {} milliseconds and holds {}MB", i, now - offer.runtime - info.received, offer.sharedMemory / 1000000);
132  mExpiredOffers.push_back(ComputingQuotaOfferRef{i});
133  stats.expired.push_back(i);
134  continue;
135  } else {
136  LOGP(LOGLEVEL, "Offer {} still valid for {} milliseconds, providing {}MB", i, offer.runtime + info.received - now, offer.sharedMemory / 1000000);
137  }
139  assert(offer.sharedMemory >= 0);
140  auto tmp = accumulated;
141  tmp.cpu += offer.cpu;
142  tmp.memory += offer.memory;
143  tmp.sharedMemory += offer.sharedMemory;
144  offer.score = selector(offer, tmp);
145  switch (offer.score) {
146  case OfferScore::Unneeded:
147  continue;
148  case OfferScore::Unsuitable:
149  continue;
150  case OfferScore::More:
151  selectOffer(i, now);
152  accumulated = tmp;
153  stats.selectedOffers.push_back(i);
154  continue;
155  case OfferScore::Enough:
156  selectOffer(i, now);
157  accumulated = tmp;
158  stats.selectedOffers.push_back(i);
159  enough = true;
160  break;
161  };
162  }
163  // If we get here it means we never got enough offers, so we return false.
164  return summarizeWhatHappended(enough, stats.selectedOffers, accumulated, stats);
165 }

175 void ComputingQuotaEvaluator::dispose(int taskId)
176 {
177  for (int oi = 0; oi < mOffers.size(); ++oi) {
178  auto& offer = mOffers[oi];
179  if (offer.user != taskId) {
180  continue;
181  }
182  offer.user = -1;
183  // Disposing the offer so that the resource can be recyled.
186  if (oi == 0) {
187  return;
188  }
189  if (offer.valid == false) {
190  continue;
191  }
192  if (offer.sharedMemory <= 0) {
193  offer.valid = false;
194  offer.score = OfferScore::Unneeded;
195  }
196  }
197 }

200 void ComputingQuotaEvaluator::updateOffers(std::vector<ComputingQuotaOffer>& pending, uint64_t now)
201 {
202  for (size_t oi = 0; oi < mOffers.size(); oi++) {
203  auto& storeOffer = mOffers[oi];
204  auto& info = mInfos[oi];
205  if (pending.empty()) {
206  return;
207  }
208  if (storeOffer.valid == true) {
209  continue;
210  }
211  info.received = now;
212  auto& offer = pending.back();
213  storeOffer = offer;
214  storeOffer.valid = true;
215  pending.pop_back();
216  }
217 }

219 void ComputingQuotaEvaluator::handleExpired(std::function<void(ComputingQuotaOffer const&, ComputingQuotaStats const& stats)> expirator)
220 {
221  static int nothingToDoCount = mExpiredOffers.size();
222  if (mExpiredOffers.size()) {
223  LOGP(LOGLEVEL, "Handling {} expired offers", mExpiredOffers.size());
224  nothingToDoCount = 0;
225  } else {
226  if (nothingToDoCount == 0) {
227  nothingToDoCount++;
228  LOGP(LOGLEVEL, "No expired offers");
229  }
230  }
233  for (auto& ref : mExpiredOffers) {
234  auto& offer = mOffers[ref.index];
235  if (offer.sharedMemory < 0) {
236  LOGP(LOGLEVEL, "Offer {} does not have any more memory. Marking it as invalid.", ref.index);
237  offer.valid = false;
238  offer.score = OfferScore::Unneeded;
239  continue;
240  }
241  // FIXME: offers should go through the driver client, not the monitoring
242  // api.
243  LOGP(LOGLEVEL, "Offer {} expired. Giving back {}MB and {} cores", ref.index, offer.sharedMemory / 1000000, offer.cpu);
244  assert(offer.sharedMemory >= 0);
245  mStats.totalExpiredBytes += offer.sharedMemory;
246  mStats.totalExpiredOffers++;
247  expirator(offer, mStats);
248  //driverClient.tell("expired shmem {}", offer.sharedMemory);
249  //driverClient.tell("expired cpu {}", offer.cpu);
250  offer.sharedMemory = -1;
251  offer.valid = false;
252  offer.score = OfferScore::Unneeded;
253  }
254  mExpiredOffers.clear();
255 }
