40 void Geo::Init()
41 {
42  if (!mToBeInit) {
43  return;
44  }
45  LOG(info) << "tof::Geo: Initialization of TOF rotation parameters";
46 
47  if (!gGeoManager) {
48  LOG(info) << " no TGeo! Loading it";
49  o2::base::GeometryManager::loadGeometry();
50  }
51 
52  int det[5];
53  for (Int_t isector = 0; isector < NSECTORS; isector++) {
54  det[0] = isector;
55  for (Int_t iplate = 0; iplate < NPLATES; iplate++) {
56  det[1] = iplate;
57 
58  if (iplate == 2 && (isector == 13 || isector == 14 || isector == 15)) {
59  continue; // PHOS HOLES
60  }
61 
62  for (Int_t istrip = 0; istrip < NSTRIPC; istrip++) { // maximum number of strip is 19 for plate B and C
63  det[2] = istrip;
64  if (!(iplate == 2 && istrip >= NSTRIPA)) { // the middle plate (A) has only 15 strips
65  for (Int_t ipadz = 0; ipadz < NPADZ; ipadz++) {
66  det[3] = ipadz;
67  for (Int_t ipadx = 0; ipadx < NPADX; ipadx++) {
68  det[4] = ipadx;
69  gGeoManager->cd(getVolumePath(det).c_str());
70  TGeoHMatrix global;
71  global = *gGeoManager->GetCurrentMatrix();
72  const Double_t* tr = global.GetTranslation();
73  mPadPosition[isector][iplate][istrip][ipadz][ipadx][0] = tr[0];
74  mPadPosition[isector][iplate][istrip][ipadz][ipadx][1] = tr[1];
75  mPadPosition[isector][iplate][istrip][ipadz][ipadx][2] = tr[2];
76  }
77  }
78  }
79  }
80  }
81  }
82 
83  Double_t rotationAngles[6] =
84  {90., 90. /*+ (isector + 0.5) * PHISEC*/, 0., 0., 90., 0 /* + (isector + 0.5) * PHISEC*/};
85  for (Int_t ii = 0; ii < 6; ii++) {
86  rotationAngles[ii] *= TMath::DegToRad();
87  }
88 
89  for (Int_t isector = 0; isector < NSECTORS; isector++) {
90  rotationAngles[5] = ((isector + 0.5) * PHISEC) * TMath::DegToRad();
91  rotationAngles[1] = 90. * TMath::DegToRad() + rotationAngles[5];
92 
93  for (Int_t ii = 0; ii < 3; ii++) {
94  mRotationMatrixSector[isector][ii][0] =
95  TMath::Sin(rotationAngles[2 * ii]) * TMath::Cos(rotationAngles[2 * ii + 1]);
96  mRotationMatrixSector[isector][ii][1] =
97  TMath::Sin(rotationAngles[2 * ii]) * TMath::Sin(rotationAngles[2 * ii + 1]);
98  mRotationMatrixSector[isector][ii][2] = TMath::Cos(rotationAngles[2 * ii]);
99  }
100  }
101 
102  rotationAngles[0] = 90. * TMath::DegToRad();
103  rotationAngles[1] = 0.;
104  rotationAngles[2] = 0.;
105  rotationAngles[3] = 0.;
106  rotationAngles[4] = 90. * TMath::DegToRad();
107  rotationAngles[5] = 270. * TMath::DegToRad();
108  for (Int_t ii = 0; ii < 3; ii++) {
109  mRotationMatrixSector[NSECTORS][ii][0] =
110  TMath::Sin(rotationAngles[2 * ii]) * TMath::Cos(rotationAngles[2 * ii + 1]);
111  mRotationMatrixSector[NSECTORS][ii][1] =
112  TMath::Sin(rotationAngles[2 * ii]) * TMath::Sin(rotationAngles[2 * ii + 1]);
113  mRotationMatrixSector[NSECTORS][ii][2] = TMath::Cos(rotationAngles[2 * ii]);
114  }
115 
116  for (Int_t iplate = 0; iplate < NPLATES; iplate++) {
117  for (Int_t istrip = 0; istrip < NMAXNSTRIP; istrip++) {
118  if (getAngles(iplate, istrip) > 0.) {
119  rotationAngles[0] = 90. * TMath::DegToRad();
120  rotationAngles[1] = 0.;
121  rotationAngles[3] = 90. * TMath::DegToRad();
122  rotationAngles[4] = getAngles(iplate, istrip) * TMath::DegToRad();
123  rotationAngles[5] = 90. * TMath::DegToRad();
124  rotationAngles[2] = 90. * TMath::DegToRad() + rotationAngles[4];
125  } else if (getAngles(iplate, istrip) == 0.) {
126  rotationAngles[0] = 90. * TMath::DegToRad();
127  rotationAngles[1] = 0.;
128  rotationAngles[2] = 90. * TMath::DegToRad();
129  rotationAngles[3] = 90. * TMath::DegToRad();
130  rotationAngles[4] = 0;
131  rotationAngles[5] = 0.;
132  } else if (getAngles(iplate, istrip) < 0.) {
133  rotationAngles[0] = 90. * TMath::DegToRad();
134  rotationAngles[1] = 0.;
135  rotationAngles[3] = 90. * TMath::DegToRad();
136  rotationAngles[4] = -getAngles(iplate, istrip) * TMath::DegToRad();
137  rotationAngles[5] = 270. * TMath::DegToRad();
138  rotationAngles[2] = 90. * TMath::DegToRad() - rotationAngles[4];
139  }
140 
141  for (Int_t ii = 0; ii < 3; ii++) {
142  mRotationMatrixPlateStrip[iplate][istrip][ii][0] =
143  TMath::Sin(rotationAngles[2 * ii]) * TMath::Cos(rotationAngles[2 * ii + 1]);
144  mRotationMatrixPlateStrip[iplate][istrip][ii][1] =
145  TMath::Sin(rotationAngles[2 * ii]) * TMath::Sin(rotationAngles[2 * ii + 1]);
146  mRotationMatrixPlateStrip[iplate][istrip][ii][2] = TMath::Cos(rotationAngles[2 * ii]);
147  }
148  }
149  }
150  InitIndices();
151  mToBeInit = kFALSE;
152 }

154 void Geo::InitIndices()
155 {
156  if (!mToBeInitIndexing) {
157  return;
158  }
159  mToBeInitIndexing = kFALSE;
160  // initialization of some indices arrays
161 
162  for (Int_t istrip = 0; istrip < NSTRIPXSECTOR; ++istrip) {
163  if (istrip < NSTRIPC) {
164  mPlate[istrip] = 0;
165  mStripInPlate[istrip] = istrip;
166  } else if (istrip >= NSTRIPC && istrip < NSTRIPC + NSTRIPB) {
167  mPlate[istrip] = 1;
168  mStripInPlate[istrip] = istrip - NSTRIPC;
169  } else if (istrip >= NSTRIPC + NSTRIPB && istrip < NSTRIPC + NSTRIPB + NSTRIPA) {
170  mPlate[istrip] = 2;
171  mStripInPlate[istrip] = istrip - NSTRIPC - NSTRIPB;
172  } else if (istrip >= NSTRIPC + NSTRIPB + NSTRIPA && istrip < NSTRIPC + NSTRIPB + NSTRIPA + NSTRIPB) {
173  mPlate[istrip] = 3;
174  mStripInPlate[istrip] = istrip - NSTRIPC - NSTRIPB - NSTRIPA;
175  } else if (istrip >= NSTRIPC + NSTRIPB + NSTRIPA + NSTRIPB && istrip < NSTRIPXSECTOR) {
176  mPlate[istrip] = 4;
177  mStripInPlate[istrip] = istrip - NSTRIPC - NSTRIPB - NSTRIPA - NSTRIPB;
178  }
179  }
180 
181  Int_t nstrips = NSTRIPC;
182  for (int iplate = 0; iplate < 5; ++iplate) {
183  if (iplate == 1 || iplate == 3) {
184  nstrips = NSTRIPB;
185  } else if (iplate == 2) {
186  nstrips = NSTRIPA;
187  }
188  mDistances[iplate].reserve(nstrips);
189  for (int i = 0; i < nstrips; ++i) {
190  mDistances[iplate].push_back(DISTANCES[iplate][nstrips - i - 1]);
191  }
192  }
193 }

195 std::string Geo::getVolumePath(const Int_t* ind)
196 {
197  //--------------------------------------------------------------------
198  // This function returns the volume path of a given pad
199  //--------------------------------------------------------------------
200  Int_t sector = ind[0];
201 
202  Int_t icopy = -1;
203  icopy = sector;
204 
205  auto string1 = fmt::format("/cave_1/barrel_1/B077_1/BSEGMO{:d}_1/BTOF{:d}_1", icopy, icopy);
206 
207  Bool_t fgHoles = kTRUE;
208 
209  Int_t iplate = ind[1];
210  Int_t istrip = ind[2];
211  if (iplate == 0) {
212  icopy = istrip;
213  }
214  if (iplate == 1) {
215  icopy = istrip + NSTRIPC;
216  }
217  if (iplate == 2) {
218  icopy = istrip + NSTRIPC + NSTRIPB;
219  }
220  if (iplate == 3) {
221  icopy = istrip + NSTRIPC + NSTRIPB + NSTRIPA;
222  }
223  if (iplate == 4) {
224  icopy = istrip + NSTRIPC + 2 * NSTRIPB + NSTRIPA;
225  }
226  icopy++;
227  auto string2 = fmt::format("FTOA_0/FLTA_0/FSTR_{:d}", icopy);
228  if (fgHoles && (sector == 13 || sector == 14 || sector == 15)) {
229  if (iplate < 2) {
230  string2 = fmt::format("FTOB_0/FLTB_0/FSTR_{:d}", icopy);
231  }
232  if (iplate > 2) {
233  string2 = fmt::format("FTOC_0/FLTC_0/FSTR_{:d}", icopy);
234  }
235  }
236 
237  Int_t padz = ind[3] + 1;
238  Int_t padx = ind[4] + 1;
239  return o2::utils::Str::concat_string(string1, '/', string2, '/', fmt::format("FPCB_1/FSEN_1/FSEZ_{:d}/FPAD_{:d}", padz, padx));
240 }

314 Int_t Geo::getIndex(const Int_t* detId)
315 {
316  //Retrieve calibration channel index
317  Int_t isector = detId[0];
318  if (isector >= NSECTORS) {
319  printf("Wrong sector number in TOF (%d) !\n", isector);
320  return -1;
321  }
322  Int_t iplate = detId[1];
323  if (iplate >= NPLATES) {
324  printf("Wrong plate number in TOF (%d) !\n", iplate);
325  return -1;
326  }
327  Int_t istrip = detId[2];
328  Int_t stripOffset = getStripNumberPerSM(iplate, istrip);
329  if (stripOffset == -1) {
330  printf("Wrong strip number per SM in TOF (%d) !\n", stripOffset);
331  return -1;
332  }
333 
334  Int_t ipadz = detId[3];
335  Int_t ipadx = detId[4];
336 
337  Int_t idet = ((2 * (NSTRIPC + NSTRIPB) + NSTRIPA) * NPADZ * NPADX) * isector +
338  (stripOffset * NPADZ * NPADX) +
339  (NPADX)*ipadz +
340  ipadx;
341  return idet;
342 }

344 Int_t Geo::getStripNumberPerSM(Int_t iplate, Int_t istrip)
345 {
346  //
347  // Get the serial number of the TOF strip number istrip [0,14/18],
348  // in the module number iplate [0,4].
349  // This number will range in [0,90].
350  //
351 
352  Int_t index = -1;
353 
354  Bool_t check = ((iplate < 0 || iplate >= NPLATES) ||
355  ((iplate == 2 && (istrip < 0 || istrip >= NSTRIPA)) ||
356  (iplate != 2 && (istrip < 0 || istrip >= NSTRIPC))));
357 
358  if (iplate < 0 || iplate >= NPLATES) {
359  LOG(error) << "getStripNumberPerSM : "
360  << "Wrong plate number in TOF (" << iplate << ")!\n";
361  }
362 
363  if (
364  (iplate == 2 && (istrip < 0 || istrip >= NSTRIPA)) ||
365  (iplate != 2 && (istrip < 0 || istrip >= NSTRIPC))) {
366  LOG(error) << "getStripNumberPerSM : "
367  << " Wrong strip number in TOF (strip=" << istrip << " in the plate= " << iplate << ")!\n";
368  }
369 
370  Int_t stripOffset = 0;
371  switch (iplate) {
372  case 0:
373  stripOffset = 0;
374  break;
375  case 1:
376  stripOffset = NSTRIPC;
377  break;
378  case 2:
379  stripOffset = NSTRIPC + NSTRIPB;
380  break;
381  case 3:
382  stripOffset = NSTRIPC + NSTRIPB + NSTRIPA;
383  break;
384  case 4:
385  stripOffset = NSTRIPC + NSTRIPB + NSTRIPA + NSTRIPB;
386  break;
387  };
388 
389  if (!check) {
390  index = stripOffset + istrip;
391  }
392 
393  return index;
394 }

413 Int_t Geo::fromPlateToStrip(Float_t* pos, Int_t iplate)
414 {
415  if (iplate == -1) {
416  return -1;
417  }
418 
419  Int_t nstrips = 0;
420  switch (iplate) {
421  case 0:
422  case 4:
423  nstrips = NSTRIPC;
424  break;
425  case 1:
426  case 3:
427  nstrips = NSTRIPB;
428  break;
429  case 2:
430  nstrips = NSTRIPA;
431  break;
432  }
433 
434  constexpr Float_t HGLFY = HFILIY + 2 * HGLASSY; // heigth of GLASS+FISHLINE Layer
435  constexpr Float_t HSTRIPY = 2. * HHONY + 2. * HPCBY + 4. * HRGLY + 2. * HGLFY + HCPCBY; // 3.11
436 
437  Float_t step[3];
438 
439  // FTOA/B/C = FLTA/B/C reference frame -> FSTR reference frame
440 
441  // we restrict the search of the strip to a more reasonable range, considering that the
442  // DISTANCES are never larger than 10 between consecutive strips
443  auto ilower = std::lower_bound(mDistances[iplate].begin(), mDistances[iplate].end(), -pos[2]);
444  int stripFound = mDistances[iplate].size() - 1 - std::distance(mDistances[iplate].begin(), ilower);
445  int firstStripToCheck = stripFound;
446  int lastStripToCheck = stripFound;
447  if (stripFound != 0) {
448  while (std::abs(pos[2] + DISTANCES[iplate][firstStripToCheck - 1]) < 10) {
449  --firstStripToCheck;
450  }
451  }
452  if (stripFound != nstrips - 1) {
453  while (std::abs(pos[2] + DISTANCES[iplate][lastStripToCheck + 1]) < 10) {
454  ++lastStripToCheck;
455  }
456  }
457 
458  for (Int_t istrip = firstStripToCheck; istrip <= lastStripToCheck; ++istrip) {
459  Float_t posLoc2[3] = {pos[0], pos[1], pos[2]};
460 
461  step[0] = 0.;
462  step[1] = getHeights(iplate, istrip);
463  step[2] = -getDistances(iplate, istrip);
464  translate(posLoc2[0], posLoc2[1], posLoc2[2], step);
465 
466  if (fabs(posLoc2[1]) > 10) {
467  continue;
468  }
469  if (fabs(posLoc2[2]) > 10) {
470  continue;
471  }
472 
473  float distanceSquared = posLoc2[1] * posLoc2[1] + posLoc2[2] * posLoc2[2];
474 
475  if (distanceSquared > 45) {
476  continue;
477  }
478 
479  rotateToStrip(posLoc2, iplate, istrip);
480 
481  if ((TMath::Abs(posLoc2[0]) <= STRIPLENGTH * 0.5) && (TMath::Abs(posLoc2[1]) <= HSTRIPY * 0.5) &&
482  (TMath::Abs(posLoc2[2]) <= WCPCBZ * 0.5)) {
483  step[0] = -0.5 * NPADX * XPAD;
484  step[1] = 0.;
485  step[2] = -0.5 * NPADZ * ZPAD;
486  //translate(posLoc2, step);
487  translate(posLoc2[0], posLoc2[1], posLoc2[2], step);
488 
489  for (Int_t jj = 0; jj < 3; ++jj) {
490  pos[jj] = posLoc2[jj];
491  }
492 
493  return istrip;
494  }
495  }
496  return -1;
497 }

572 Int_t Geo::getPlate(const Float_t* pos)
573 {
574  //
575  // Returns the Plate index
576  //
577
