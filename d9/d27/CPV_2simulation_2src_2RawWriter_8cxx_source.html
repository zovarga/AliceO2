91 void RawWriter::digitsToRaw(gsl::span<o2::cpv::Digit> digitsbranch, gsl::span<o2::cpv::TriggerRecord> triggerbranch)
92 {
93  if (triggerbranch.begin() == triggerbranch.end()) { //do we have any data?
94  return;
95  }
96 
97  //process digits which belong to same orbit
98  int iFirstTrgInCurrentOrbit = 0;
99  int currentOrbit = triggerbranch[0].getBCData().orbit;
100  int nTrgsInCurrentOrbit = 1;
101  for (int iTrg = 1; iTrg < triggerbranch.size(); iTrg++) {
102  if (triggerbranch[iTrg].getBCData().orbit != currentOrbit) { //if orbit changed, write previous orbit to file
103  processOrbit(digitsbranch, triggerbranch.subspan(iFirstTrgInCurrentOrbit, nTrgsInCurrentOrbit));
104  iFirstTrgInCurrentOrbit = iTrg; //orbit changed
105  nTrgsInCurrentOrbit = 1;
106  currentOrbit = triggerbranch[iTrg].getBCData().orbit;
107  } else {
108  nTrgsInCurrentOrbit++;
109  }
110  }
111  processOrbit(digitsbranch, triggerbranch.subspan(iFirstTrgInCurrentOrbit, nTrgsInCurrentOrbit)); //process last orbit
112 }

115 bool RawWriter::processOrbit(const gsl::span<o2::cpv::Digit> digitsbranch, const gsl::span<o2::cpv::TriggerRecord> trgs)
116 {
117  static int nMaxGbtWordsPerPage = o2::raw::RDHUtils::MAXCRUPage / o2::raw::RDHUtils::GBTWord - 4; //512*16/16 - 4 = 508;
118  //4 gbt words are reserved for RDH
119 
120  //clear payloads of all links
121  for (auto& payload : mPayload) {
122  payload.clear();
123  }
124 
125  //we're going to prepare preformatted pages
126  bool preformatted = true;
127 
128  int gbtWordCounter[kNGBTLinks] = {0, 0, 0};
129  int gbtWordCounterBeforeCPVTrailer[kNGBTLinks] = {0, 0, 0};
130  bool isHeaderClosedWithTrailer[kNGBTLinks] = {false, false, false};
131  for (auto& trg : trgs) {
132  LOG(debug) << "RawWriter::processOrbit() : "
133  << "I start to process trigger record (orbit = " << trg.getBCData().orbit
134  << ", BC = " << trg.getBCData().bc << ")";
135  LOG(debug) << "First entry = " << trg.getFirstEntry() << ", Number of objects = " << trg.getNumberOfObjects();
136 
137  //Clear array which is used to store digits
138  for (int i = kNcc; i--;) {
139  for (int j = kNDilogic; j--;) {
140  for (int k = kNGasiplex; k--;) {
141  mPadCharge[i][j][k].clear();
142  }
143  }
144  }
145 
146  //make payload for current trigger
147  int nDigsInTrg[kNGBTLinks] = {0, 0, 0};
148  for (auto& dig : gsl::span(digitsbranch.data() + trg.getFirstEntry(), trg.getNumberOfObjects())) {
149 
150  short absId = dig.getAbsId();
151  short ccId, dil, gas, pad;
152  o2::cpv::Geometry::absIdToHWaddress(absId, ccId, dil, gas, pad);
153 
154  //Convert Amp to ADC counts
155  short charge = std::round(dig.getAmplitude() / mCalibParams->getGain(absId) + mPedestals->getPedestal(absId));
156  if (charge > 4095) {
157  charge = 4095;
158  }
159  mPadCharge[ccId][dil][gas].emplace_back(charge, pad);
160  nDigsInTrg[ccId / (kNcc / kNGBTLinks)]++; //linkId = ccId/8 or absId/7680
161  }
162  LOG(debug) << "I produced " << nDigsInTrg << " digits for this trigger record";
163 
164  //we need to write header + at least 1 payload word + trailer
165  for (int iLink = 0; iLink < kNGBTLinks; iLink++) { //looping links
166  gbtWordCounterBeforeCPVTrailer[iLink] = 0;
167  if (nMaxGbtWordsPerPage - gbtWordCounter[iLink] < 3) { //otherwise flush already prepared data to file
168  LOG(debug) << "RawWriter::processOrbit() : before header: adding preformatted dma page of size " << mPayload[iLink].size();
169  mRawWriter->addData(links[iLink].feeId, links[iLink].cruId, links[iLink].linkId, links[iLink].endPointId, trg.getBCData(),
170  gsl::span<char>(mPayload[iLink].data(), mPayload[iLink].size()), preformatted);
171  mPayload[iLink].clear();
172  gbtWordCounter[iLink] = 0;
173  gbtWordCounterBeforeCPVTrailer[iLink] = 0;
174  }
175 
176  //first, header goes
177  CpvHeader header(trg.getBCData(), false, false);
178  for (int i = 0; i < 16; i++) {
179  mPayload[iLink].push_back(header.mBytes[i]);
180  }
181  isHeaderClosedWithTrailer[iLink] = false;
182  LOG(debug) << "RawWriter::processOrbit() : "
183  << "I wrote cpv header for orbit = " << trg.getBCData().orbit
184  << " and BC = " << trg.getBCData().bc;
185 
186  gbtWordCounter[iLink]++;
187  gbtWordCounterBeforeCPVTrailer[iLink]++;
188 
189  int nDigsToWriteLeft = nDigsInTrg[iLink];
190 
191  for (char ccId = iLink * (kNcc / kNGBTLinks); (ccId < (iLink + 1) * (kNcc / kNGBTLinks)) && (ccId < kNcc); ccId++) {
192  int ccWordCounter = 0;
193  for (char dil = 0; dil < kNDilogic; dil++) {
194  for (char gas = 0; gas < kNGasiplex; gas++) {
195  for (padCharge& pc : mPadCharge[ccId][dil][gas]) {
196  // Generate 3 CC words, add CC header and empty bits to complete 128 bits;
197  PadWord currentword = {0};
198  currentword.charge = pc.charge;
199  currentword.address = pc.pad;
200  currentword.gas = gas;
201  currentword.dil = dil;
202  mPayload[iLink].push_back(currentword.mBytes[0]);
203  mPayload[iLink].push_back(currentword.mBytes[1]);
204  mPayload[iLink].push_back(currentword.mBytes[2]);
205  ccWordCounter++;
206  nDigsToWriteLeft--;
207  if (ccWordCounter % 3 == 0) { // complete 3 channels (72 bit) + CC index (8 bits) + 6 empty bits = Generate 128 bits of data
208  mPayload[iLink].push_back(ccId);
209  for (int i = 6; i--;) {
210  mPayload[iLink].push_back(char(0));
211  }
212  gbtWordCounter[iLink]++;
213  gbtWordCounterBeforeCPVTrailer[iLink]++;
214  if (nMaxGbtWordsPerPage - gbtWordCounter[iLink] == 1) { //the only space for trailer left on current page
215  CpvTrailer tr(gbtWordCounterBeforeCPVTrailer[iLink], trg.getBCData().bc, nDigsToWriteLeft == 0); //add trailer and flush page to file
216  for (int i = 0; i < 16; i++) {
217  mPayload[iLink].push_back(tr.mBytes[i]);
218  }
219  isHeaderClosedWithTrailer[iLink] = true;
220  LOG(debug) << "RawWriter::processOrbit() : middle of payload: adding preformatted dma page of size " << mPayload[iLink].size();
221  mRawWriter->addData(links[iLink].feeId, links[iLink].cruId, links[iLink].linkId, links[iLink].endPointId, trg.getBCData(),
222  gsl::span<char>(mPayload[iLink].data(), mPayload[iLink].size()), preformatted);
223 
224  mPayload[iLink].clear();
225  gbtWordCounter[iLink] = 0;
226  gbtWordCounterBeforeCPVTrailer[iLink] = 0;
227  if (nDigsToWriteLeft) { //some digits left for writing
228  CpvHeader newHeader(trg.getBCData(), false, true);
229  for (int i = 0; i < 16; i++) { //so put a new header and continue
230  mPayload[iLink].push_back(newHeader.mBytes[i]);
231  }
232  isHeaderClosedWithTrailer[iLink] = false;
233  gbtWordCounter[iLink]++;
234  gbtWordCounterBeforeCPVTrailer[iLink]++;
235  }
236  }
237  }
238  }
239  }
240  } //end of dil cycle
241  if (ccWordCounter % 3 != 0) {
242  while (ccWordCounter % 3 != 0) {
243  mPayload[iLink].push_back(char(255));
244  mPayload[iLink].push_back(char(255));
245  mPayload[iLink].push_back(char(255));
246  ccWordCounter++;
247  }
248  mPayload[iLink].push_back(ccId);
249  for (int i = 6; i--;) {
250  mPayload[iLink].push_back(char(0));
251  }
252  gbtWordCounter[iLink]++;
253  gbtWordCounterBeforeCPVTrailer[iLink]++;
254  if (nMaxGbtWordsPerPage - gbtWordCounter[iLink] == 1) { //the only space for trailer left on current page
255  CpvTrailer tr(gbtWordCounterBeforeCPVTrailer[iLink], trg.getBCData().bc, nDigsToWriteLeft == 0); //add trailer and flush page to file
256  for (int i = 0; i < 16; i++) {
257  mPayload[iLink].push_back(tr.mBytes[i]);
258  }
259  isHeaderClosedWithTrailer[iLink] = true;
260  LOG(debug) << "RawWriter::processOrbit() : middle of payload (after filling empty words): adding preformatted dma page of size " << mPayload[iLink].size();
261  mRawWriter->addData(links[iLink].feeId, links[iLink].cruId, links[iLink].linkId, links[iLink].endPointId, trg.getBCData(),
262  gsl::span<char>(mPayload[iLink].data(), mPayload[iLink].size()), preformatted);
263  mPayload[iLink].clear();
264  gbtWordCounter[iLink] = 0;
265  gbtWordCounterBeforeCPVTrailer[iLink] = 0;
266  if (nDigsToWriteLeft) { //some digits left for writing
267  for (int i = 0; i < 16; i++) { //so put a new header and continue
268  mPayload[iLink].push_back(header.mBytes[i]);
269  }
270  isHeaderClosedWithTrailer[iLink] = false;
271  gbtWordCounter[iLink]++;
272  gbtWordCounterBeforeCPVTrailer[iLink]++;
273  }
274  }
275  }
276  } //end of ccId cycle
277  if (!isHeaderClosedWithTrailer[iLink]) {
278  CpvTrailer tr(gbtWordCounterBeforeCPVTrailer[iLink], trg.getBCData().bc, true);
279  for (int i = 0; i < 16; i++) {
280  mPayload[iLink].push_back(tr.mBytes[i]);
281  }
282  isHeaderClosedWithTrailer[iLink] = true;
283  gbtWordCounterBeforeCPVTrailer[iLink] = 0;
284  gbtWordCounter[iLink]++;
285  }
286  } //end of iLink cycle
287  } //end of "for (auto& trg : trgs)""
288 
289  //flush payload to file (if any)
290  for (int iLink = 0; iLink < kNGBTLinks; iLink++) {
291  if (mPayload[iLink].size()) {
292  LOG(debug) << "RawWriter::processOrbit() : final payload: adding preformatted dma page of size " << mPayload[iLink].size();
293  mRawWriter->addData(links[iLink].feeId, links[iLink].cruId, links[iLink].linkId, links[iLink].endPointId,
294  trgs.back().getBCData(), gsl::span<char>(mPayload[iLink].data(), mPayload[iLink].size()), preformatted);
295  mPayload[iLink].clear();
296  }
297  }
298  return true;
299 }

301 int RawWriter::carryOverMethod(const header::RDHAny* rdh, const gsl::span<char> data,
302  const char* ptr, int maxSize, int splitID,
303  std::vector<char>& trailer, std::vector<char>& header) const
304 {
305 
306  constexpr int cpvTrailerSize = 36;
307  int offs = ptr - &data[0]; // offset wrt the head of the payload
308  assert(offs >= 0 && size_t(offs + maxSize) <= data.size()); // make sure ptr and end of the suggested block are within the payload
309  int leftBefore = data.size() - offs; // payload left before this splitting
310  int leftAfter = leftBefore - maxSize; // what would be left after the suggested splitting
311  int actualSize = maxSize;
312  if (leftAfter && leftAfter <= cpvTrailerSize) { // avoid splitting the trailer or writing only it.
313  actualSize -= (cpvTrailerSize - leftAfter) + 4; // (as we work with int, not char in decoding)
314  }
315  return actualSize;
316 }
