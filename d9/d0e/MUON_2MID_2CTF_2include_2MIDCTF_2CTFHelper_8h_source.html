110  // BC difference wrt previous if in the same orbit, otherwise the abs.value.
111  // For the very 1st entry return 0 (diff wrt 1st BC in the CTF header)
112  class Iter_bcIncROF : public _Iter<Iter_bcIncROF, ROFRecord, uint16_t>
113  {
114  public:
115  using _Iter<Iter_bcIncROF, ROFRecord, uint16_t>::_Iter;
116  value_type operator*() const
117  {
118  const auto ir = (*mData)[mOrder[mIndex].getSource()][mOrder[mIndex].getIndex()].interactionRecord;
119  if (mIndex) {
120  const auto irP = (*mData)[mOrder[mIndex - 1].getSource()][mOrder[mIndex - 1].getIndex()].interactionRecord;
121  if (ir.orbit == irP.orbit) {
122  return ir.bc - irP.bc;
123  } else {
124  return ir.bc;
125  }
126  }
127  return 0;
128  }
129  };

133  class Iter_orbitIncROF : public _Iter<Iter_orbitIncROF, ROFRecord, uint32_t>

135  public:
