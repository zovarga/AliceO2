82 StepTHnT<TemplateArray>::StepTHnT(const Char_t* name, const Char_t* title, const Int_t nSteps, const Int_t nAxes,
83  Int_t* nBins, std::vector<Double_t> binEdges[], const char** axisTitles) : StepTHn(name, title, nSteps, nAxes)
84 {

87  mNBins *= nBins[i];
88  }
89  mPrototype = new THnSparseT<TemplateArray>(Form("%s_sparse", name), title, nAxes, nBins);
90 
91  for (Int_t i = 0; i < mNVars; i++) {
92  if (nBins[i] + 1 == binEdges[i].size()) { // variable-width binning
93  mPrototype->GetAxis(i)->Set(nBins[i], &(binEdges[i])[0]);
94  } else if (binEdges[i].size() == 2) { // equidistant binning
95  mPrototype->GetAxis(i)->Set(nBins[i], binEdges[i][0], binEdges[i][1]);
96  } else {
97  LOGF(fatal, "Invalid binning information for axis %d with %d bins and %d entries for bin edges", i, nBins[i], binEdges[i].size());
98  }
99  LOGF(debug, "Histogram %s Axis %d created with %d bins and %d edges", name, i, nBins[i], binEdges[i].size());
100  mPrototype->GetAxis(i)->SetTitle(axisTitles[i]);
101  }
102 }

218 template <class TemplateArray>
219 Long64_t StepTHnT<TemplateArray>::Merge(TCollection* list)
220 {
221  // Merge a list of StepTHn objects with this (needed for PROOF).
222  // Returns the number of merged objects (including this).
223 
224  if (!list) {
225  return 0;
226  }
227 
228  if (list->IsEmpty()) {
229  return 1;
230  }
231 
232  TIterator* iter = list->MakeIterator();
233  TObject* obj;
234 
235  Int_t count = 0;
236  while ((obj = iter->Next())) {
237 
238  StepTHnT<TemplateArray>* entry = dynamic_cast<StepTHnT<TemplateArray>*>(obj);
239  if (entry == nullptr) {
240  continue;
241  }
242 
243  for (Int_t i = 0; i < mNSteps; i++) {
244  if (entry->mValues[i]) {
245  if (!mValues[i]) {
246  mValues[i] = createArray();
247  }
248 
249  auto source = dynamic_cast<TemplateArray*>(entry->mValues[i])->GetArray();
250  auto target = dynamic_cast<TemplateArray*>(mValues[i])->GetArray();
251  for (Long64_t l = 0; l < mNBins; l++) {
252  target[l] += source[l];
253  }
254  }
255 
256  if (entry->mSumw2[i]) {
257  if (!mSumw2[i]) {
258  mSumw2[i] = createArray();
259  }
260 
261  auto source = dynamic_cast<TemplateArray*>(entry->mSumw2[i])->GetArray();
262  auto target = dynamic_cast<TemplateArray*>(mSumw2[i])->GetArray();
263  for (Long64_t l = 0; l < mNBins; l++) {
264  target[l] += source[l];
265  }
266  }
267  }
268 
269  count++;
270  }
271 
272  return count + 1;
273 }

275 Long64_t StepTHn::getGlobalBinIndex(const Int_t* binIdx)
276 {
277  // calculates global bin index
278  // binIdx contains TAxis bin indexes
279  // here bin count starts at 0 because we do not have over/underflow bins
280 
281  Long64_t bin = 0;

290 void StepTHn::createTarget(Int_t step, Bool_t sparse)
291 {
292  // fills the information stored in the buffer in this class into the target THn
293 
294  if (!mValues[step]) {
295  LOGF(fatal, "Histogram request for step %d which is empty.", step);
296  return;
297  }
298 
299  if (!mTarget) {
300  mTarget = new THnBase*[mNSteps];
301  for (Int_t i = 0; i < mNSteps; i++) {
302  mTarget[i] = nullptr;
303  }
304  }
305 
306  if (mTarget[step]) {
307  return;
308  }
309 
310  TArray* source = mValues[step];
311  // if mSumw2 is not stored, the sqrt of the number of bin entries in source is filled below; otherwise we use mSumw2
312  TArray* sourceSumw2 = source;
313  if (mSumw2[step]) {
314  sourceSumw2 = mSumw2[step];
315  }
316 
317  if (sparse) {
318  mTarget[step] = THnSparse::CreateSparse(Form("%s_%d", GetName(), step), Form("%s_%d", GetTitle(), step), mPrototype);
319  } else {
320  mTarget[step] = THn::CreateHn(Form("%s_%d", GetName(), step), Form("%s_%d", GetTitle(), step), mPrototype);
321  }
322 
323  THnBase* target = mTarget[step];
324 
325  Int_t* binIdx = new Int_t[mNVars];
326  Int_t* nBins = new Int_t[mNVars];
327  for (Int_t j = 0; j < mNVars; j++) {
328  binIdx[j] = 1;
329  nBins[j] = target->GetAxis(j)->GetNbins();
330  }
331 
332  Long64_t count = 0;
333 
334  while (1) {
335  // for (Int_t j=0; j<mNVars; j++)
336  // printf("%d ", binIdx[j]);
337 
338  Long64_t globalBin = getGlobalBinIndex(binIdx);
339  // Printf(" --> %lld", globalBin);
340 
341  // TODO probably slow
342  double value = source->GetAt(globalBin);
343  if (value != 0) {
344  target->SetBinContent(binIdx, value);
345  target->SetBinError(binIdx, TMath::Sqrt(sourceSumw2->GetAt(globalBin)));
346 
347  count++;
348  }
349 
350  binIdx[mNVars - 1]++;
351 
352  for (Int_t j = mNVars - 1; j > 0; j--) {
353  if (binIdx[j] > nBins[j]) {
354  binIdx[j] = 1;
355  binIdx[j - 1]++;
356  }
357  }
358 
359  if (binIdx[0] > nBins[0]) {
360  break;
361  }
362  }
363 
364  LOGF(info, "Step %d: copied %lld entries out of %lld bins", step, count, getGlobalBinIndex(binIdx));
365 
366  delete[] binIdx;
367  delete[] nBins;
368 
369  delete mValues[step];
370  mValues[step] = nullptr;
371 }

373 void StepTHn::Fill(int iStep, int nParams, double positionAndWeight[])
374 {
375  if (iStep >= mNSteps) {
376  LOGF(fatal, "Selected step for filling is not in range of StepTHn.");
377  }
378 
379  double weight = 1.0;
380  if (nParams == mNVars + 1) {
381  weight = positionAndWeight[mNVars];
382  } else if (nParams != mNVars) {
383  LOGF(fatal, "Fill called with invalid number of parameters (%d vs %d)", mNVars, nParams);
384  }
385 
386  // fill axis cache
387  if (!mAxisCache) {
388  mAxisCache = new TAxis*[mNVars];
389  mNbinsCache = new Int_t[mNVars];
390  for (Int_t i = 0; i < mNVars; i++) {
391  mAxisCache[i] = mPrototype->GetAxis(i);
392  mNbinsCache[i] = mAxisCache[i]->GetNbins();
393  }
394 
395  mLastVars = new Double_t[mNVars];
396  mLastBins = new Int_t[mNVars];
397 
398  // initial values to prevent checking for 0 below
399  for (Int_t i = 0; i < mNVars; i++) {
400  mLastVars[i] = positionAndWeight[i];
401  mLastBins[i] = mAxisCache[i]->FindBin(mLastVars[i]);
402  }
403  }
404 
405  // calculate global bin index
406  Long64_t bin = 0;
407  for (Int_t i = 0; i < mNVars; i++) {
408  bin *= mNbinsCache[i];
409 
410  Int_t tmpBin = 0;
411  if (mLastVars[i] == positionAndWeight[i]) {
412  tmpBin = mLastBins[i];
413  } else {
414  tmpBin = mAxisCache[i]->FindBin(positionAndWeight[i]);
415  mLastBins[i] = tmpBin;
416  mLastVars[i] = positionAndWeight[i];
417  }
418  //Printf("%d", tmpBin);
419 
420  // under/overflow not supported
421  if (tmpBin < 1 || tmpBin > mNbinsCache[i]) {
422  return;
423  }
424 
425  // bins start from 0 here
426  bin += tmpBin - 1;
427  // Printf("%lld", bin);
428  }
429 
430  if (!mValues[iStep]) {
431  mValues[iStep] = createArray();
432  LOGF(info, "Created values container for step %d", iStep);
433  }
434 
435  if (weight != 1.) {
436  // initialize with already filled entries (which have been filled with weight == 1), in this case mSumw2 := mValues
437  if (!mSumw2[iStep]) {
438  mSumw2[iStep] = createArray();
439  LOGF(info, "Created sumw2 container for step %d", iStep);
440  }
441  }
442 
443  // TODO probably slow; add StepTHnT::add ?
444  mValues[iStep]->SetAt(mValues[iStep]->GetAt(bin) + weight, bin);
445  if (mSumw2[iStep]) {
446  mSumw2[iStep]->SetAt(mSumw2[iStep]->GetAt(bin) + weight, bin);
447  }
448 }
