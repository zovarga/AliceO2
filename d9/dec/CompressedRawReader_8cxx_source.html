35 uint32_t CompressedRawReader::processHBFs()
36 {
37 
38  LOG(debug) << "PROCESS HBF starting at " << (void*)mDataPointer;
39 
40  mDataRDH = reinterpret_cast<const o2::header::RDHAny*>(mDataPointer);
41  //mEncoderRDH = reinterpret_cast<o2::header::RDHAny*>(mEncoderPointer);
42  auto rdh = mDataRDH;
43  /* loop until RDH stop header */
44  while (!o2::raw::RDHUtils::getStop(rdh)) { // carry on till the end of the event.
45  auto headerSize = o2::raw::RDHUtils::getHeaderSize(rdh);
46  auto memorySize = o2::raw::RDHUtils::getMemorySize(rdh);
47  auto offsetToNext = o2::raw::RDHUtils::getOffsetToNext(rdh);
48  auto dataPayload = memorySize - headerSize;
49  mFEEID = o2::raw::RDHUtils::getFEEID(rdh); //TODO change this and just carry around the curreht RDH
50  mCRUEndpoint = o2::raw::RDHUtils::getEndPointID(rdh); // the upper or lower half of the currently parsed cru 0-14 or 15-29
51  mCRUID = o2::raw::RDHUtils::getCRUID(rdh);
52  int packetCount = o2::raw::RDHUtils::getPacketCounter(rdh);
53  if (mHeaderVerbose) {
54  LOG(info) << "--- RDH open/continue detected";
55  LOG(info) << "FEEID : " << mFEEID << " Packet: " << packetCount << " sizes : header" << headerSize << " memorysize:" << memorySize << " offsettonext:" << offsetToNext << " datapayload:" << dataPayload;
56  o2::raw::RDHUtils::printRDH(rdh);
57  }
58  /* copy CRU payload to save buffer */
59  // std::memcpy(mSaveBuffer + mSaveBufferDataSize, (char*)(rdh) + headerSize, drmPayload);
60  // mSaveBufferDataSize += drmPayload;
61  //
62  // we will "simply" parse on the fly with a basic state machine.
63  mDataPointer = (uint32_t*)((char*)rdh + headerSize);
64  mDataEndPointer = (const uint32_t*)((char*)rdh + offsetToNext);
65  mIR = o2::raw::RDHUtils::getTriggerIR(rdh);
66  while ((void*)mDataPointer < (void*)mDataEndPointer) { // loop to handle the case where a halfcru ends/begins mid rdh data block
67  mEventCounter++;
68  if (processBlock()) { // at this point the entire payload is in mSaveBuffer, TODO parse this incrementally, less mem foot print.
69  LOG(warn) << "processBlock return flase";
70  if (mVerbose) {
71  LOG(info) << "processBlock return flase";
72  }
73  break; // end of CRU
74  }
75  //buildCRUPayLoad(); // the rest of the hbf for the subsequent cruhalfchamber payload.
76  // TODO is this even possible if hbfs upto the stop is for one event and each cru header is for 1 event?
77  }
78  /* move to next RDH */
79  rdh = (o2::header::RDHAny*)((char*)(rdh) + offsetToNext);
80  if (mVerbose) {
81  LOG(info) << " rdh is now at 0x" << (void*)rdh << " offset to next : " << offsetToNext;
82  }
83  }
84 
85  if (o2::raw::RDHUtils::getStop(rdh)) {
86  if (mDataPointer != (const uint32_t*)((char*)rdh + o2::raw::RDHUtils::getOffsetToNext(rdh))) {
87  LOG(warn) << " at end of parsing loop and mDataPointer is on next rdh";
88  }
89  mDataPointer = (const uint32_t*)((char*)rdh + o2::raw::RDHUtils::getOffsetToNext(rdh));
90  // make sure mDataPointer is in the correct place.
91  } else {
92  mDataPointer = (const uint32_t*)((char*)rdh);
93  }
94  if (mVerbose) {
95  LOGF(debug, " at exiting processHBF after advancing to next rdh mDataPointer is %p ?= %p", (void*)mDataPointer, (void*)mDataEndPointer);
96  }
97  o2::raw::RDHUtils::printRDH(rdh);
98 
99  if (mVerbose) {
100  LOG(info) << "--- RDH close detected";
101  LOG(info) << "--- END PROCESS HBF";
102  }
103  /* move to next RDH */
104  // mDataPointer = (uint32_t*)((char*)(rdh) + o2::raw::RDHUtils::getOffsetToNext(rdh));
105 
106  /* otherwise return */
107 
108  return mDataEndPointer - mDataPointer;
109 }

111 bool CompressedRawReader::processBlock()
112 {
113  /* process a FeeID/halfcru, 15 links */
114  // LOG(debug) << "--- PROCESS BLOCK FeeID:" << mFEEID;
115  mCurrentLinkDataPosition = 0;
116 
117  if (mVerbose) {
118  LOG(info) << "--- END PROCESS HalfCRU with state: " << mState;
119  }
120  //this is essentially the inverse function of CruCompressorTask::buildOutput
121  //tracklet headers.
122  CompressedRawHeader header;
123  memcpy((char*)&header, &mDataBuffer, sizeof(CompressedRawHeader));
124  mDataPointer += sizeof(CompressedRawHeader); //bytes
125  mDataReadIn += sizeof(CompressedRawHeader);
126  //tracklets
127  int numberoftracklets = header.size;
128  o2::InteractionRecord ir(header.bc, header.orbit);
129  Tracklet64* trackletptr = (Tracklet64*)mDataPointer; //payload is supposed to be a block of tracklet64 objects.
130  std::copy(trackletptr, trackletptr + numberoftracklets, std::back_inserter(mEventTracklets));
131  mDataPointer += numberoftracklets * sizeof(Tracklet64); //bytes
132  mDataReadIn += numberoftracklets * sizeof(Tracklet64);
133 
134  //digits headers.
135  CompressedRawTrackletDigitSeperator tracklettrailer;
136  memcpy((char*)&tracklettrailer, &mDataBuffer, sizeof(CompressedRawTrackletDigitSeperator));
137  mDataBuffer += sizeof(CompressedRawTrackletDigitSeperator);
138  mDataReadIn += sizeof(CompressedRawTrackletDigitSeperator);
139 
140  //digits
141  int numberofdigits = tracklettrailer.digitcount;
142  if (mHeaderVerbose || mVerbose) { //sanity check
143  bool badheader = false;
144  //check compressedrawtrackletdigitseperator for the pre and postfix 0xe's
145  if ((tracklettrailer.pad1 & 0xffffff) == 0xeeeeee) {
146  badheader = true;
147  }
148  if ((tracklettrailer.pad2 & 0xffffff) == 0xeeeeee) {
149  badheader = true;
150  }
151  if (badheader) {
152  LOG(info) << "Bad compressed raw header seperator digitsize :" << tracklettrailer.digitcount << " padding1 : " << std::hex << tracklettrailer.pad1 << " padding2:" << std::hex << tracklettrailer.pad2;
153  }
154  if (tracklettrailer.digitcount > 1000) { // TODO probably a better value, but this isfine for now, its for a single half cru count of digits
155  LOG(info) << "Digit count seems unusually high : " << tracklettrailer.digitcount;
156  }
157  }
158 
159  CompressedDigit* digitptr = (CompressedDigit*)mDataPointer;
160  std::copy(digitptr, digitptr + numberofdigits, std::back_inserter(mCompressedEventDigits));
161  mDataPointer += numberofdigits + sizeof(CompressedDigit);
162  mDataReadIn += numberofdigits + sizeof(CompressedDigit);
163  //convert compresed digits to proper digits TODO put this in a copy operator of Compressed Digit class.
164  for (int digitcounter = 0; digitcounter < numberofdigits; ++digitcounter) {
165  ArrayADC timebins;
166  //TODO This already pre supposes o2::trd::constants::TIMEBINS is 30 from other places, figure something out.
167  for (int adc = 0; adc < o2::trd::constants::TIMEBINS; adc++) {
168  timebins[adc] = mCompressedEventDigits[digitcounter][adc];
169  }
170  mEventDigits.emplace_back(mCompressedEventDigits[digitcounter].getDetector(), mCompressedEventDigits[digitcounter].getROB(),
171  mCompressedEventDigits[digitcounter].getMCM(), mCompressedEventDigits[digitcounter].getChannel(), timebins);
172  }
173  // now we *should* have a CompressedRawDigitEndMarker or more commonly known as a lot of 0xe
174  if ((uint32_t)*mDataPointer != 0xeeeeeeee) {
175  LOG(warn) << std::hex << *mDataPointer << " We should be seeing CompressedRawDigitEndMarker which is the same as a o2::trd::constants::CRUPADDING32";
176  }
177  mDataPointer += 4;
178  mDataReadIn += 4;
179  if (mDataReadIn % 8 != 0) {
180  // we have an extra padding word to make it to a full 64bit data buffer.
181  if ((uint32_t)*mDataPointer != 0xeeeeeeee) {
182  LOG(warn) << std::hex << *mDataPointer << " We should be seeing CompressedRawDigitEndMarker which is the same as a o2::trd::constants::CRUPADDING32";
183  }
184  mDataPointer += 4;
185  mDataReadIn += 4;
186  }
187  auto lasttrigger = mEventTriggers.size() - 1;
188  int lastdigit = mEventTriggers[lasttrigger].getFirstDigit() + mEventTriggers[lasttrigger].getNumberOfDigits();
189  int lasttracklet = mEventTriggers[lasttrigger].getFirstTracklet() + mEventTriggers[lasttrigger].getNumberOfTracklets();
190  mEventTriggers.emplace_back(mIR, lastdigit, numberofdigits, lasttracklet, numberoftracklets);
191 
192  // either 1 or more depending on padding requirements.
193  if (mVerbose) {
194  LOG(info) << "Tracklets in block : " << numberoftracklets << " vector has size:" << mEventTracklets.size();
195  LOG(info) << "Digits in block : " << numberofdigits << " vector has size:" << mEventDigits.size();
196  }
197  return true;
198 }
