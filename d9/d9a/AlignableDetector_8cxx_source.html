213 //_________________________________________________________
214 void AlignableDetector::defineMatrices()
215 {
216  // define transformation matrices. Detectors may override this method
217  //
218  TGeoHMatrix mtmp;
219  //
220  TIter next(&mVolumes);
221  AlignableVolume* vol(nullptr);
222  while ((vol = (AlignableVolume*)next())) {
223  // modified global-local matrix
224  vol->prepareMatrixL2G();
225  // ideal global-local matrix
226  vol->prepareMatrixL2GIdeal();
227  //
228  }
229  // Now set tracking-local matrix (MUST be done after ALL L2G matrices are done!)
230  // Attention: for sensor it is a real tracking matrix extracted from
231  // the geometry but for container alignable volumes the tracking frame
232  // is used for as the reference for the alignment parameters only,
233  // see its definition in the AlignableVolume::PrepateMatrixT2L
234  next.Reset();
235  while ((vol = (AlignableVolume*)next())) {
236  vol->prepareMatrixT2L();
237  if (vol->isSensor()) {
238  ((AlignableSensor*)vol)->prepareMatrixClAlg();
239  } // alignment matrix
240  }
241  //
242 }

263 int AlignableDetector::initGeom()
264 {
265  // define hiearchy, initialize matrices, return number of global parameters
266  if (getInitGeomDone()) {
267  return 0;
268  }
269  //

271  sortSensors(); // VolID's must be in increasing order
272  defineMatrices();
273  //
274  // calculate number of global parameters
275  int nvol = getNVolumes();
276  mNDOFs = 0;
277  for (int iv = 0; iv < nvol; iv++) {
278  AlignableVolume* vol = getVolume(iv);

281  //
282  mNDOFs += mNCalibDOFs;
283  setInitGeomDone();
284  return mNDOFs;

311 void AlignableDetector::initDOFs()
312 {

333  //
334  setInitDOFsDone();
335  return;

338 //_________________________________________________________
339 int AlignableDetector::volID2SID(int vid) const
340 {
341  // find SID corresponding to VolID
342  int mn(0), mx(mNSensors - 1);
343  while (mx >= mn) {
344  int md((mx + mn) >> 1), vids(getSensor(md)->getVolID());
345  if (vid < vids) {
346  mx = md - 1;
347  } else if (vid > vids) {
348  mn = md + 1;
349  } else {
350  return md;
351  }
352  }
353  return -1;
354 }

642 //________________________________________
643 void AlignableDetector::constrainOrphans(const double* sigma, const char* match)
644 {
645  // additional constraint on volumes w/o parents (optionally containing "match" in symname)
646  // sigma<0 : dof is not contrained
647  // sigma=0 : dof constrained exactly (Lagrange multiplier)
648  // sigma>0 : dof constrained by gaussian constraint
649  //
650  TString mts = match, syms;
651  GeometricalConstraint* constr = new GeometricalConstraint();
652  for (int i = 0; i < AlignableVolume::kNDOFGeom; i++) {
653  if (sigma[i] >= 0) {
654  constr->constrainDOF(i);
655  } else {
656  constr->unConstrainDOF(i);
657  }
658  constr->setSigma(i, sigma[i]);
659  }
660  for (int i = getNVolumes(); i--;) {
661  AlignableVolume* vol = getVolume(i);
662  if (vol->getParent()) {
663  continue;
664  } // wrong level
665  if (!mts.IsNull() && !(syms = vol->getSymName()).Contains(mts)) {
666  continue;
667  } //wrong name
668  constr->addChild(vol);
669  }
670  //
671  if (!constr->getNChildren()) {
672  LOG(info) << "No volume passed filter " << match;
673  delete constr;
674  } else {
675  ((TObjArray*)mController->getConstraints())->Add(constr);
676  }
677 }
