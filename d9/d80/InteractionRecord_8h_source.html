29  static constexpr uint16_t DummyBC = 0xffff;

65  static double bc2ns(int bc, unsigned int orbit)
66  {
67  return bc * o2::constants::lhc::LHCBunchSpacingNS + orbit * o2::constants::lhc::LHCOrbitNS;
68  }
69 
70  static int ns2bc(double ns, unsigned int& orb)
71  {
72  orb = ns > 0 ? ns / o2::constants::lhc::LHCOrbitNS : 0;
73  ns -= orb * o2::constants::lhc::LHCOrbitNS;
74  return std::round(ns / o2::constants::lhc::LHCBunchSpacingNS);
75  }
76 
77  double bc2ns() const
78  {
79  return bc2ns(bc, orbit);

82  bool operator==(const InteractionRecord& other) const
83  {
84  return (bc == other.bc) && (orbit == other.orbit);
85  }
86 
87  bool operator!=(const InteractionRecord& other) const
88  {
89  return (bc != other.bc) || (orbit != other.orbit);
90  }
91 
92  int64_t differenceInBC(const InteractionRecord& other) const
93  {
94  // return difference in bunch-crossings
95  int64_t diffBC = int(bc) - other.bc;
96  if (orbit != other.orbit) {
97  diffBC += (int64_t(orbit) - other.orbit) * o2::constants::lhc::LHCMaxBunches;
98  }
99  return diffBC;
100  }
101 
102  float differenceInBCNS(const InteractionRecord& other) const
103  {
104  // return difference in bunch-crossings in ns
105  return differenceInBC(other) * o2::constants::lhc::LHCBunchSpacingNS;
106  }
107 
108  float differenceInBCMS(const InteractionRecord& other) const
109  {
110  // return difference in bunch-crossings in ms
111  return differenceInBC(other) * o2::constants::lhc::LHCBunchSpacingMUS;
112  }
113 
114  int64_t toLong() const
115  {
116  // return as single long number
117  return (int64_t(orbit) * o2::constants::lhc::LHCMaxBunches) + bc;
118  }
119 
120  void setFromLong(int64_t l)
121  {

147  InteractionRecord operator--()
148  {
149  // prefix decrement operator, no check for orbit wrap
150  if (!bc--) {
151  orbit--;
152  bc = o2::constants::lhc::LHCMaxBunches - 1;

154  return InteractionRecord(*this);

157  InteractionRecord operator--(int)

159  // postfix decrement operator, no check for orbit wrap
160  InteractionRecord tmp(*this);
161  if (!bc--) {
162  orbit--;
163  bc = o2::constants::lhc::LHCMaxBunches - 1;

168  InteractionRecord operator++()
169  {
170  // prefix increment operator,no check for orbit wrap
171  if ((++bc) == o2::constants::lhc::LHCMaxBunches) {
172  orbit++;
173  bc = 0;
174  }
175  return InteractionRecord(*this);
176  }

178  InteractionRecord operator++(int)
179  {
180  // postfix increment operator, no check for orbit wrap
181  InteractionRecord tmp(*this);
182  if ((++bc) == o2::constants::lhc::LHCMaxBunches) {
183  orbit++;
184  bc = 0;
185  }
186  return tmp;
187  }
188 
189  InteractionRecord& operator+=(int64_t dbc)
190  {
191  // bc self-addition operator, no check for orbit wrap
192  auto l = toLong() + dbc;
193  bc = l % o2::constants::lhc::LHCMaxBunches;
194  orbit = l / o2::constants::lhc::LHCMaxBunches;
195  return *this;
196  }
197 
198  InteractionRecord& operator-=(int64_t dbc)
199  {
200  // bc self-subtraction operator, no check for orbit wrap

202  }
