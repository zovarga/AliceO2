30 // Parses a metric in the form
31 //
32 // [METRIC] <name>,<type> <value> <timestamp> [<tag>,<tag>]
33 bool DeviceMetricsHelper::parseMetric(std::string_view const s, ParsedMetricMatch& match)
34 {
37  constexpr size_t PREFIXSIZE = 9;
38  if (s.size() < PREFIXSIZE) {
39  return false;
40  }
41  if (memcmp(s.data(), "[METRIC] ", 9) != 0) {
42  return false;
43  }
44  char const* comma; // first comma
45  char const* spaces[256]; // list of spaces
46  char const** space = spaces; // first element to fill
47 
48  comma = (char const*)memchr((void*)s.data(), ',', s.size());
49  if (comma == nullptr) {
50  return false;
51  }
52  // Find all spaces
53  char const* nextSpace = s.data();
54  while (space - spaces < 256) {
55  *space = strchr(nextSpace, ' ');
56  if (*space == nullptr) {
57  break;
58  }
59  nextSpace = *space + 1;
60  space++;
61  }
62  // First space should always come before comma
63  if (spaces[0] > comma) {
64  return false;
65  }
66  match.beginKey = spaces[0] + 1;
67  match.endKey = comma;
68  // type is alway 1 char after the comma, followed by space
69  if ((spaces[1] - comma) != 2) {
70  return false;
71  }
72  char* ep = nullptr;
73  match.type = static_cast<MetricType>(strtol(comma + 1, &ep, 10));
74  if (ep != spaces[1]) {
75  return false;
76  }
77  // We need at least 4 spaces
78  if (space - spaces < 4) {
79  return false;
80  }
81  // Value is between the second and the last but one space
82  switch (match.type) {
83  case MetricType::Int:
84  match.intValue = strtol(spaces[1] + 1, &ep, 10);
85  if (ep != *(space - 2)) {
86  return false;
87  }
88  match.floatValue = (float)match.intValue;
89  break;
90  case MetricType::Float:
91  match.floatValue = strtof(spaces[1] + 1, &ep);
92  if (ep != *(space - 2)) {
93  return false;
94  }
95  break;
96  case MetricType::String:
97  match.beginStringValue = spaces[1] + 1;
98  match.endStringValue = *(space - 2);
99  match.floatValue = 0;
100  break;
101  case MetricType::Uint64:
102  match.uint64Value = strtoul(spaces[1] + 1, &ep, 10);
103  if (ep != *(space - 2)) {
104  return false;
105  }
106  match.floatValue = (float)match.uint64Value;
107  break;
108 
109  default:
110  return false;
111  }
112  // Timestamp is between the last but one and the last space
113  match.timestamp = strtol(*(space - 2) + 1, &ep, 10);
114  if (ep != *(space - 1)) {
115  return false;
116  }
117  return true;
118 }

120 size_t DeviceMetricsHelper::bookMetricInfo(DeviceMetricsInfo& info, char const* name)
121 {
122  // Add the index by name in the correct position
123  // this will require moving the tail of the index,
124  // but inserting should happen only once for each metric,
125  // so who cares.
126  // Add the the actual Metric info to the store
127  MetricLabel metricLabel;
128  strncpy(metricLabel.label, name, MetricLabel::MAX_METRIC_LABEL_SIZE - 1);
129  metricLabel.label[MetricLabel::MAX_METRIC_LABEL_SIZE - 1] = '\0';
130  metricLabel.size = strlen(metricLabel.label);
131 
132  // Find the insertion point for the sorted index
133  auto cmpFn = [namePtr = metricLabel.label,
134  &labels = info.metricLabels,
135  nameSize = metricLabel.size](MetricLabelIndex const& a, MetricLabelIndex const& b)
136  -> bool {
137  return strncmp(labels[a.index].label, namePtr, nameSize) < 0;
138  };
139 
140  auto mi = std::lower_bound(info.metricLabelsAlphabeticallySortedIdx.begin(),
141  info.metricLabelsAlphabeticallySortedIdx.end(),
142  MetricLabelIndex{},
143  cmpFn);
144 
145  // If it was already there, return the old index.
146  if (mi != info.metricLabelsAlphabeticallySortedIdx.end() && (strncmp(info.metricLabels[mi->index].label, metricLabel.label, std::min((size_t)metricLabel.size, (size_t)MetricLabel::MAX_METRIC_LABEL_SIZE - 1)) == 0)) {
147  return mi->index;
148  }
149 
150  // Add the the actual Metric info to the store
151  auto metricIndex = info.metrics.size();
152 
153  // Insert the sorted location where it belongs to.
154  MetricLabelIndex metricLabelIdx{metricIndex};
155  info.metricLabelsAlphabeticallySortedIdx.insert(mi, metricLabelIdx);
156  info.metrics.push_back(MetricInfo{});
157 
158  // Create a new metric
159  auto& metricInfo = info.metrics.back();
160  metricInfo.pos = 0;
161  metricInfo.filledMetrics = 0;
162 
163  // Add the timestamp buffer for it
164  info.timestamps.emplace_back(std::array<size_t, 1024>{});
165  info.max.push_back(std::numeric_limits<float>::lowest());
166  info.min.push_back(std::numeric_limits<float>::max());
167  info.average.push_back(0);
168  info.maxDomain.push_back(std::numeric_limits<size_t>::lowest());
169  info.minDomain.push_back(std::numeric_limits<size_t>::max());
170  info.changed.push_back(true);
171 
172  info.metricLabels.push_back(metricLabel);
173 
174  return metricIndex;
175 }

177 bool DeviceMetricsHelper::processMetric(ParsedMetricMatch& match,
178  DeviceMetricsInfo& info,
179  DeviceMetricsHelper::NewMetricCallback newMetricsCallback)
180 {
181  // get the type
182  size_t metricIndex = -1;
183 
184  StringMetric stringValue;
185  switch (match.type) {
186  case MetricType::Float:
187  case MetricType::Int:
188  case MetricType::Uint64:
189  break;
190  case MetricType::String: {
191  auto lastChar = std::min(match.endStringValue - match.beginStringValue, StringMetric::MAX_SIZE - 1);
192  memcpy(stringValue.data, match.beginStringValue, lastChar);
193  stringValue.data[lastChar] = '\0';
194  } break;
195  default:
196  return false;
197  break;
198  };
199 
200  // Find the metric based on the label. Create it if not found.
201  auto cmpFn = [namePtr = match.beginKey,
202  &labels = info.metricLabels,
203  nameSize = match.endKey - match.beginKey](MetricLabelIndex const& a, MetricLabelIndex const& b)
204  -> bool {
205  return strncmp(labels[a.index].label, namePtr, nameSize) < 0;
206  };
207  auto mi = std::lower_bound(info.metricLabelsAlphabeticallySortedIdx.begin(),
208  info.metricLabelsAlphabeticallySortedIdx.end(),
209  MetricLabelIndex{},
210  cmpFn);
211 
212  // We could not find the metric, lets insert a new one.
213  auto matchSize = match.endKey - match.beginKey;
214  if (mi == info.metricLabelsAlphabeticallySortedIdx.end() || (strncmp(info.metricLabels[mi->index].label, match.beginKey, std::min(matchSize, (long)MetricLabel::MAX_METRIC_LABEL_SIZE - 1)) != 0)) {
215  // Create a new metric
216  MetricInfo metricInfo;
217  metricInfo.pos = 0;
218  metricInfo.type = match.type;
219  metricInfo.filledMetrics = 0;
220  // Add a new empty buffer for it of the correct kind
221  switch (match.type) {
222  case MetricType::Int:
223  metricInfo.storeIdx = info.intMetrics.size();
224  info.intMetrics.emplace_back(std::array<int, 1024>{});
225  break;
226  case MetricType::String:
227  metricInfo.storeIdx = info.stringMetrics.size();
228  info.stringMetrics.emplace_back(std::array<StringMetric, 32>{});
229  break;
230  case MetricType::Float:
231  metricInfo.storeIdx = info.floatMetrics.size();
232  info.floatMetrics.emplace_back(std::array<float, 1024>{});
233  break;
234  case MetricType::Uint64:
235  metricInfo.storeIdx = info.uint64Metrics.size();
236  info.uint64Metrics.emplace_back(std::array<uint64_t, 1024>{});
237  break;
238 
239  default:
240  return false;
241  };
242  // Add the timestamp buffer for it
243  info.timestamps.emplace_back(std::array<size_t, 1024>{});
244  info.max.push_back(std::numeric_limits<float>::lowest());
245  info.min.push_back(std::numeric_limits<float>::max());
246  info.average.push_back(0);
247  info.maxDomain.push_back(std::numeric_limits<size_t>::lowest());
248  info.minDomain.push_back(std::numeric_limits<size_t>::max());
249  info.changed.push_back(false);
250 
251  // Add the index by name in the correct position
252  // this will require moving the tail of the index,
253  // but inserting should happen only once for each metric,
254  // so who cares.
255  MetricLabel metricLabel;
256  auto lastChar = std::min(match.endKey - match.beginKey, (ptrdiff_t)MetricLabel::MAX_METRIC_LABEL_SIZE - 1);
257  memcpy(metricLabel.label, match.beginKey, lastChar);
258  metricLabel.label[lastChar] = '\0';
259  metricLabel.size = lastChar;
260  MetricLabelIndex metricLabelIdx;
261  metricLabelIdx.index = info.metrics.size();
262  info.metricLabels.push_back(metricLabel);
263  info.metricLabelsAlphabeticallySortedIdx.insert(mi, metricLabelIdx);
264  // Add the the actual Metric info to the store
265  metricIndex = info.metrics.size();
266  assert(metricInfo.storeIdx != -1);
267  assert(metricLabel.label[0] != '\0');
268  if (newMetricsCallback != nullptr) {
269  newMetricsCallback(metricLabel.label, metricInfo, match.intValue, metricIndex);
270  }
271  info.metrics.push_back(metricInfo);
272  } else {
273  metricIndex = mi->index;
274  }
275  assert(metricIndex != -1);
276  // We are now guaranteed our metric is present at metricIndex.
277  MetricInfo& metricInfo = info.metrics[metricIndex];
278 
279  // auto mod = info.timestamps[metricIndex].size();
280  auto sizeOfCollection = 0;
281  switch (metricInfo.type) {
282  case MetricType::Int: {
283  info.intMetrics[metricInfo.storeIdx][metricInfo.pos] = match.intValue;
284  sizeOfCollection = info.intMetrics[metricInfo.storeIdx].size();
285  } break;
286  case MetricType::String: {
287  info.stringMetrics[metricInfo.storeIdx][metricInfo.pos] = stringValue;
288  sizeOfCollection = info.stringMetrics[metricInfo.storeIdx].size();
289  } break;
290  case MetricType::Float: {
291  info.floatMetrics[metricInfo.storeIdx][metricInfo.pos] = match.floatValue;
292  sizeOfCollection = info.floatMetrics[metricInfo.storeIdx].size();
293  } break;
294  case MetricType::Uint64: {
295  info.uint64Metrics[metricInfo.storeIdx][metricInfo.pos] = match.uint64Value;
296  sizeOfCollection = info.uint64Metrics[metricInfo.storeIdx].size();
297  } break;
298  default:
299  return false;
300  break;
301  };
302  // We do all the updates here, so that not update timestamps for broken metrics
303  // Notice how we always fill floatValue with the float equivalent of the metric
304  // regardless of it's type.
305  info.minDomain[metricIndex] = std::min(info.minDomain[metricIndex], (size_t)match.timestamp);
306  info.maxDomain[metricIndex] = std::max(info.maxDomain[metricIndex], (size_t)match.timestamp);
307  info.max[metricIndex] = std::max(info.max[metricIndex], match.floatValue);
308  info.min[metricIndex] = std::min(info.min[metricIndex], match.floatValue);
309  auto onlineAverage = [](float nextValue, float previousAverage, float previousCount) {
310  return previousAverage + (nextValue - previousAverage) / (previousCount + 1);
311  };
312  info.average[metricIndex] = onlineAverage(match.floatValue, info.average[metricIndex], metricInfo.filledMetrics);
313  info.timestamps[metricIndex][metricInfo.pos] = match.timestamp;
314  // We point to the next metric
315  metricInfo.pos = (metricInfo.pos + 1) % sizeOfCollection;
316  ++metricInfo.filledMetrics;
317  // Note that we updated a given metric.
318  info.changed[metricIndex] = true;
319  return true;
320 }

322 size_t DeviceMetricsHelper::metricIdxByName(const std::string& name, const DeviceMetricsInfo& info)
323 {
324  size_t i = 0;
325  while (i < info.metricLabels.size()) {
326  auto& metricName = info.metricLabels[i];
327  // We check the size first and then the last character because that's
328  // likely to be different for multi-index metrics
329  if (metricName.size == name.size() && metricName.label[metricName.size - 1] == name[metricName.size - 1] && memcmp(metricName.label, name.c_str(), metricName.size) == 0) {
330  return i;
331  }
332  ++i;
333  }
334  return i;
335 }

337 void DeviceMetricsHelper::updateMetricsNames(DriverInfo& driverInfo, std::vector<DeviceMetricsInfo> const& metricsInfos)
338 {
339  // Calculate the unique set of metrics, as available in the metrics service
340  static std::unordered_set<std::string> allMetricsNames;
341  for (const auto& metricsInfo : metricsInfos) {
342  for (const auto& labelsPairs : metricsInfo.metricLabels) {
343  allMetricsNames.insert(std::string(labelsPairs.label));
344  }
345  }
346  for (const auto& labelsPairs : driverInfo.metrics.metricLabels) {
347  allMetricsNames.insert(std::string(labelsPairs.label));
348  }
349  std::vector<std::string> result(allMetricsNames.begin(), allMetricsNames.end());
350  std::sort(result.begin(), result.end());
351  driverInfo.availableMetrics.swap(result);
352 }
