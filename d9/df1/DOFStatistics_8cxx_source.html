30 std::unique_ptr<TH1F> DOFStatistics::buildHistogram(Controller* controller) const
31 {
32  // create histo with stat. If steer object is supplied, build labels
33  auto histogram = std::make_unique<TH1F>("DOFstat", "statistics per DOF", getNDOFs(), 0, getNDOFs());
34  for (size_t i = 0; i < getNDOFs(); ++i) {
35  // Bin 0 is underflow bin
36  histogram->SetBinContent(i + 1, mStat[i]);
37  if (controller != nullptr) {
38  histogram->GetXaxis()->SetBinLabel(i + 1, controller->getDOFLabelTxt(i));
39  }
40  }
41  return histogram;
42 }

45 int64_t DOFStatistics::merge(TCollection* list)
46 {
47  // merge statistics
48  int nMerged = 0;
49  TIter next{list};
50  TObject* obj = nullptr;
51  while ((obj = next()) != nullptr) {
52  DOFStatistics* otherStats = dynamic_cast<DOFStatistics*>(obj);
53  if (!otherStats) {
54  continue;
55  }
56  assert(getNDOFs() == otherStats->getNDOFs());
57  std::transform(std::begin(otherStats->mStat), std::end(otherStats->mStat), std::begin(mStat), std::begin(mStat), std::plus<>{});
58  mNMerges += otherStats->mNMerges;
59  nMerged++;
60  }
61  return nMerged;
62 }
