59  template <typename I, typename D, typename T, int M = 1>
60  class _Iter
61  {
62  public:
63  using difference_type = int64_t;
64  using value_type = T;
65  using pointer = const T*;
66  using reference = const T&;
67  using iterator_category = std::random_access_iterator_tag;
68 
69  _Iter(const gsl::span<const D>& data, bool end = false) : mData(data), mIndex(end ? M * data.size() : 0){};
70  _Iter() = default;
71 
72  const I& operator++()
73  {
74  ++mIndex;
75  return (I&)(*this);
76  }
77 
78  const I& operator--()
79  {
80  mIndex--;
81  return (I&)(*this);
82  }
83 
84  difference_type operator-(const I& other) const { return mIndex - other.mIndex; }
85 
86  difference_type operator-(size_t idx) const { return mIndex - idx; }
87 
88  const I& operator-(size_t idx)
89  {
90  mIndex -= idx;
91  return (I&)(*this);
92  }
93 
94  bool operator!=(const I& other) const { return mIndex != other.mIndex; }
95  bool operator==(const I& other) const { return mIndex == other.mIndex; }
96  bool operator>(const I& other) const { return mIndex > other.mIndex; }
97  bool operator<(const I& other) const { return mIndex < other.mIndex; }
98 
99  protected:
100  gsl::span<const D> mData{};
101  size_t mIndex = 0;
102  };

105  // BC difference wrt previous if in the same orbit, otherwise the abs.value.
106  // For the very 1st entry return 0 (diff wrt 1st BC in the CTF header)
107  class Iter_bcIncTrig : public _Iter<Iter_bcIncTrig, BCData, uint16_t>
108  {
109  public:
110  using _Iter<Iter_bcIncTrig, BCData, uint16_t>::_Iter;
111  value_type operator*() const
112  {
113  if (mIndex) {
114  if (mData[mIndex].ir.orbit == mData[mIndex - 1].ir.orbit) {
115  return mData[mIndex].ir.bc - mData[mIndex - 1].ir.bc;
116  } else {
117  return mData[mIndex].ir.bc;
118  }
119  }
120  return 0;
121  }
122  };
