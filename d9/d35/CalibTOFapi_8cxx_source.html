105 void CalibTOFapi::readDiagnosticFrequencies()
106 {
107  static const int NCH_PER_CRATE = Geo::NSTRIPXSECTOR * Geo::NPADS;
108  // getting the Diagnostic Frequency calibration
109  // needed for simulation
110 
111  memset(mIsNoisy, false, Geo::NCHANNELS);
112 
113  auto& mgr = CcdbManager::instance();
114  long timems = long(mTimeStamp) * 1000;
115  mDiaFreq = mgr.getForTimeStamp<Diagnostic>("TOF/Calib/Diagnostic", timems);
116 
117  // mDiaFreq->print();
118 
119  resetDia();
120 
121  if (!mDiaFreq->getFrequencyROW()) {
122  return;
123  }
124 
125  float nrow = (float)mDiaFreq->getFrequencyROW();
126  mEmptyTOF = mDiaFreq->getFrequencyEmptyTOF() / nrow;
127 
128  nrow -= mDiaFreq->getFrequencyEmptyTOF();
129 
130  if (nrow < 1) {
131  return;
132  }
133 
134  // fill empty crates
135  int ncrate[Geo::kNCrate];
136  for (int i = 0; i < Geo::kNCrate; i++) {
137  ncrate[i] = mDiaFreq->getFrequencyEmptyCrate(i) - mDiaFreq->getFrequencyEmptyTOF(); // counts of empty crate for non-empty event
138  mEmptyCrateProb[i] = ncrate[i] / nrow;
139  }
140 
141  const auto vectorDia = mDiaFreq->getVector();
142  // fill TRM errors and noisy
143  for (auto pair : vectorDia) {
144  auto key = pair.first;
145  int slot = mDiaFreq->getSlot(key);
146 
147  if (slot < 13 && slot > 2) { // is TRM
148  int icrate = mDiaFreq->getCrate(key);
149  int crateslot = icrate * 100 + slot;
150  mTRMerrorProb.push_back(std::make_pair(crateslot, pair.second / (nrow - ncrate[icrate])));
151  mTRMmask.push_back(key - mDiaFreq->getTRMKey(icrate, slot)); // remove crate and slot from the key (28 bit errors remaining)
152  continue;
153  }
154 
155  int channel = mDiaFreq->getChannel(key);
156  if (channel > -1) { // noisy
157  int crate = channel / NCH_PER_CRATE;
158  float prob = pair.second / (nrow - ncrate[crate]);
159  mNoisy.push_back(std::make_pair(channel, prob));
160  continue;
161  }
162  }
163 
164  std::sort(mTRMerrorProb.begin(), mTRMerrorProb.end(), [](const auto& a, const auto& b) {
165  return a.first < b.first;
166  });
167 
168  std::sort(mNoisy.begin(), mNoisy.end(), [](const auto& a, const auto& b) {
169  return a.first < b.first;
170  });
171 
172  int ich = -1;
173  float prob = 0;
174  for (auto [ch, p] : mNoisy) {
175  if (ch != ich) { // new channel
176  if (ich != -1 && prob > 0.5) {
177  mIsNoisy[ich] = true;
178  }
179  ich = ch;
180  prob = p;
181  } else {
182  prob += p;
183  }
184  }
185  if (ich != -1 && prob > 0.5) {
186  mIsNoisy[ich] = true;
187  }
188 }

247 float CalibTOFapi::getTimeCalibration(int ich, float tot)
248 {
249 
250  // time calibration to correct measured TOF times
251 
252  float corr = 0;
253  if (!mLHCphase || !mSlewParam) {
254  LOG(warning) << "Either LHC phase or slewing object null: mLHCphase = " << mLHCphase << ", mSlewParam = " << mSlewParam;
255  return corr;
256  }
257  // printf("LHC phase apply\n");
258  // LHCphase
259  corr += mLHCphase->getLHCphase(mTimeStamp); // timestamp that we use in LHCPhase is in seconds
260  // time slewing + channel offset
261  //printf("eval time sleweing calibration: ch=%d tot=%f (lhc phase = %f)\n",ich,tot,corr);
262  corr += mSlewParam->evalTimeSlewing(ich, tot);
263  //printf("corr = %f\n",corr);
264  return corr;
265 }

290 void CalibTOFapi::processError(int crate, int trm, int mask)
291 {
292  if (checkTRMPolicy(mask)) { // check the policy of TRM -> true=good TRM
293  return;
294  }
295  int ech = (crate << 12) + ((trm - 3) << 8);
296  for (int i = ech; i < ech + 256; i++) {
297  int channel = Geo::getCHFromECH(i);
298  if (channel == -1) {
299  continue;
300  }
301  mIsErrorCh[channel] = true;
302  mFillErrChannel.push_back(channel);
303  }
304 }
