43 void EntropyEncoderSpec::run(ProcessingContext& pc)
44 {
45  auto cput = mTimer.CpuTime();
46  mTimer.Start(false);
47  auto digits = pc.inputs().get<gsl::span<CTPDigit>>("digits");
48 
49  auto& buffer = pc.outputs().make<std::vector<o2::ctf::BufferType>>(Output{"CTP", "CTFDATA", 0, Lifetime::Timeframe});
50  mCTFCoder.encode(buffer, digits);
51  auto eeb = CTF::get(buffer.data()); // cast to container pointer
52  eeb->compactify(); // eliminate unnecessary padding
53  buffer.resize(eeb->size()); // shrink buffer to strictly necessary size
54  // eeb->print();
55  mTimer.Stop();
56  LOG(info) << "Created encoded data of size " << eeb->size() << " for CTP in " << mTimer.CpuTime() - cput << " s";
57 }
