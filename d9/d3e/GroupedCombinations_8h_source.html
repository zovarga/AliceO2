59 template <typename T1, typename GroupingPolicy, typename H, typename G, typename... Us, typename... As>
60 struct GroupedCombinationsGenerator<T1, GroupingPolicy, H, G, pack<Us...>, As...> {
61  using joinIterator = typename soa::Join<H, G>::table_t::iterator;
62  using GroupedIteratorType = pack_to_tuple_t<interleaved_pack_t<repeated_type_pack_t<joinIterator, sizeof...(As)>, pack<As...>>>;

72  GroupedIterator(const GroupingPolicy& groupingPolicy, const H& hashes, const G& grouping, const std::shared_ptr<GroupSlicer<G, Us...>>&& slicer_ptr) : GroupingPolicy(groupingPolicy), mSlicer{std::move(slicer_ptr)}, mGrouping{std::make_shared<G>(std::vector{grouping.asArrowTable()})}
73  {
74  GroupingPolicy::setTables(join(hashes, grouping), join(hashes, grouping));
75  if (!this->mIsEnd) {
76  setCurrentGroupedCombination();

84  void setTables(const H& hashes, const G& grouping, std::shared_ptr<GroupSlicer<G, Us...>> slicer_ptr)
85  {
86  mGrouping = std::make_shared<G>(std::vector{grouping.asArrowTable()});
87  mSlicer = slicer_ptr;
88  setMultipleGroupingTables<sizeof...(As)>(join(hashes, grouping));

94  template <std::size_t N, typename T, typename... Args>
95  void setMultipleGroupingTables(const T& param, const Args&... args)
96  {
97  if constexpr (N == 1) {

99  } else {
100  setMultipleGroupingTables<N - 1>(param, param, args...);
101  }

104  void moveToEnd()
105  {
106  GroupingPolicy::moveToEnd();
107  }

109  // prefix increment
110  GroupedIterator& operator++()
111  {
112  if (!this->mIsEnd) {
113  this->addOne();
114  setCurrentGroupedCombination();
115  }
116  return *this;
117  }
118  // postfix increment
119  GroupedIterator operator++(int /*unused*/)
120  {
121  GroupedIterator copy(*this);
122  operator++();
123  return copy;
124  }
125  // return reference
126  reference operator*()
127  {
128  return *mCurrentGrouped;
129  }
130  bool operator==(const GroupedIterator& rh)
131  {
132  return (this->mIsEnd && rh.mIsEnd) || (this->mCurrent == rh.mCurrent);
133  }
134  bool operator!=(const GroupedIterator& rh)
135  {
136  return !(*this == rh);
137  }

139  private:
140  std::tuple<As...> getAssociatedTables()
141  {
142  auto& currentGrouping = GroupingPolicy::mCurrent;
143  constexpr auto k = sizeof...(As);
144  auto slicerIterators = functionToTuple<k>(&GroupSlicer<G, Us...>::begin, *mSlicer);
145  o2::soa::for_<k>([&](auto i) {
146  auto col = std::get<i.value>(currentGrouping);
147  for (auto& slice : *mSlicer) {
148  if (slice.groupingElement().globalIndex() == col.globalIndex()) {
149  std::get<i.value>(slicerIterators) = slice;
150  break;
151  }
152  }
153  });
154 
155  return getSlices(slicerIterators, std::index_sequence_for<As...>());
156  }
157 
158  template <std::size_t I, typename T, typename... Ts>
159  auto getSliceAt(std::tuple<Ts...>& t)
160  {
161  auto it = std::get<I>(t); // Get the tables corresponding to the grouping at index I
162  auto associatedType = it.template prepareArgument<T>();
163  return associatedType;
164  }
165 
166  template <typename... Ts, std::size_t... Is>
167  std::tuple<As...> getSlices(std::tuple<Ts...>& t, std::index_sequence<Is...> is)
168  {
169  return std::make_tuple(getSliceAt<Is, As, Ts...>(t)...);
170  }
171 
172  void setCurrentGroupedCombination()
173  {
174  std::tuple<As...> initAssociatedTables = getAssociatedTables();
175  constexpr auto k = sizeof...(As);
176  bool moveForward = false;
177  o2::soa::for_<k>([&](auto i) {
178  if (std::get<i.value>(initAssociatedTables).size() == 0) {
179  moveForward = true;
180  }
181  });
182  while (!this->mIsEnd && moveForward) {
183  GroupingPolicy::addOne();
184  std::tuple<As...> temp = getAssociatedTables();
185  moveForward = false;
186  o2::soa::for_<k>([&](auto i) {
187  if (std::get<i.value>(temp).size() == 0) {
188  moveForward = true;
189  }
190  });
191  }
192  std::tuple<As...> associatedTables = getAssociatedTables();
193 
194  if (!this->mIsEnd) {
195  auto& currentGrouping = GroupingPolicy::mCurrent;
196  o2::soa::for_<k>([&](auto i) {
197  std::get<i.value>(associatedTables).bindExternalIndices(mGrouping.get());
198  });
199 
200  mCurrentGrouped.emplace(interleaveTuples(currentGrouping, associatedTables));
201  }
202  }
203 
204  std::shared_ptr<GroupSlicer<G, Us...>> mSlicer = nullptr;
205  std::shared_ptr<G> mGrouping;
206  std::optional<GroupedIteratorType> mCurrentGrouped;
207  };

238  void setTables(H& hashes, G& grouping, std::tuple<Us...>& associated)
239  {
240  if (mSlicer == nullptr) {
241  mSlicer = std::make_shared<GroupSlicer<G, Us...>>(grouping, associated);
242  mBegin.setTables(hashes, grouping, mSlicer);
243  mEnd.setTables(hashes, grouping, mSlicer);
244  mEnd.moveToEnd();
245  }
246  }
