38 void CalibPadGainTracks::processTrack(o2::tpc::TrackTPC track)
39 {
40  // make momentum cut
41  const float mom = track.getP();
42  const int nClusters = track.getNClusterReferences();
43  if (mom < mMomMin || mom > mMomMax || std::abs(track.getEta()) > mEtaMax || nClusters < mMinClusters) {
44  return;
45  }
46 
47  // clearing memory
48  mDEdxIROC.clear();
49  mDEdxOROC.clear();
50  mCLNat.clear();
51  mClTrk.clear();
52 
53  float effectiveLengthLastCluster = 0; // set the effective length of the track over pad for the last cluster
54 
55  for (int iCl = 0; iCl < nClusters; iCl++) { //loop over cluster
56  const float effectiveLength = getTrackTopologyCorrection(track, iCl);
57  mCLNat.emplace_back(track.getCluster(*mTPCTrackClIdxVecInput, iCl, *mClusterIndex));
58  unsigned char sectorIndex = 0;
59  unsigned char rowIndex = 0;
60  unsigned int clusterIndexNumb = 0;
61 
62  // this function sets sectorIndex, rowIndex, clusterIndexNumb
63  track.getClusterReference(*mTPCTrackClIdxVecInput, iCl, sectorIndex, rowIndex, clusterIndexNumb);
64  const float charge = mCLNat[static_cast<unsigned int>(iCl)].qMax;
65  const unsigned char pad = static_cast<unsigned char>(mCLNat[static_cast<unsigned int>(iCl)].getPad() + 0.5f); // the left side of the pad ist defined at e.g. 3.5 and the right side at 4.5
66 
67  // propagateTo delivers sometimes wrong values! break if the effLength doesnt change
68  // TODO check this
69  if (effectiveLength - effectiveLengthLastCluster == 0) {
70  break;
71  }
72 
73  // fill IROC dedx
74  if (rowIndex < mapper.getNumberOfRowsROC(0)) {
75  mDEdxIROC.emplace_back(charge / effectiveLength);
76  } else {
77  mDEdxOROC.emplace_back(charge / effectiveLength);
78  }
79  mClTrk.emplace_back(std::make_tuple(sectorIndex, rowIndex, pad, charge / effectiveLength)); // fill with dummy dedx value
80  effectiveLengthLastCluster = effectiveLength;
81  }
82 
83  // use dedx from track as reference
84  if (mMode == DedxTrack) {
85  const float dedxIROC = getTruncMean(mDEdxIROC);
86  const float dedxOROC = getTruncMean(mDEdxOROC);
87 
88  // set the dEdx
89  for (auto& x : mClTrk) {
90  const unsigned char globRow = std::get<1>(x);
91  const unsigned char pad = std::get<2>(x);
92 
93  // get globalPadNumber (index)
94  const auto rowsIROC = mapper.getNumberOfRowsROC(0);
95  const auto roctype = (globRow < rowsIROC) ? RocType::IROC : RocType::OROC;
96  const float fillVal = (roctype == RocType::IROC) ? (std::get<3>(x) / dedxIROC) : (std::get<3>(x) / dedxOROC);
97  const ROC roc(std::get<0>(x), roctype);
98  const o2::tpc::PadSubset sub = getHistos()->getPadSubset();
99  const int num = getHistos()->getCalArray(roc.getRoc()).getPadSubsetNumber();
100  const int rowinROC = roctype == RocType::IROC ? std::get<1>(x) : std::get<1>(x) - rowsIROC;
101  const auto index = static_cast<unsigned int>(getIndex(sub, num, rowinROC, pad)); // index=globalpadnumber
102  //fill the normalizes charge in pad histogram
103  fillPadByPadHistogram(roc.getRoc(), index, fillVal);
104  }
105  } else {
106  }
107 }

109 float CalibPadGainTracks::getTruncMean(std::vector<float>& vCharge, float low, float high) const
110 {
111  // returns the truncated mean for input vector
112  std::sort(vCharge.begin(), vCharge.end()); //sort the vector for performing truncated mean
113 
114  const int nClustersUsed = static_cast<int>(vCharge.size());
115  const int startInd = static_cast<int>(low * nClustersUsed);
116  const int endInd = static_cast<int>(high * nClustersUsed);
117 
118  if (endInd <= startInd) {
119  return 0;
120  }
121 
122  const float dEdx = std::accumulate(vCharge.begin() + startInd, vCharge.begin() + endInd, 0.f);
123  const int nClustersTrunc = endInd - startInd; // count number of clusters
124  return dEdx / nClustersTrunc;
125 }

127 float CalibPadGainTracks::getTrackTopologyCorrection(o2::tpc::TrackTPC& track, int iCl)
128 {
129  unsigned char sectorIndex = 0;
130  unsigned char rowIndex = 0;
131  unsigned int clusterIndexNumb = 0;
132  // this function sets sectorIndex, rowIndex, clusterIndexNumb
133  track.getClusterReference(*mTPCTrackClIdxVecInput, iCl, sectorIndex, rowIndex, clusterIndexNumb);
134 
135  const PadRegionInfo& region = mapper.getPadRegionInfo(Mapper::REGION[rowIndex]);
136  const float padLength = region.getPadHeight();
137 
138  // to correct the cluster charge for the track topology, the track parameters have to be propagated to the x position if the cluster
139  const float xPosition = mapper.getPadCentre(PadPos(rowIndex, 0)).X();
140  track.propagateTo(xPosition, mField); // propagate this track to the plane X=xk (cm) in the field "b" (kG)
141  const float sinPhi = track.getSnp();
142  const float tgl = track.getTgl();
143  const float snp2 = sinPhi * sinPhi;
144  const float effectiveLength = padLength * std::sqrt((1 + tgl * tgl) / (1 - snp2)); // calculate the trace length of the track over the pad
145  return effectiveLength;
146 }
