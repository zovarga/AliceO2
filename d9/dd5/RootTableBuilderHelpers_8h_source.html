62 // When reading from a ROOT file special care must happen
63 // because uint64_t is platform specific while ULong64_t is
64 // always long long unsigned int (same for the signed version).
65 // By using this traits we make sure that any 64 bit quantity
66 // read from a root file uses the ROOT datatype, not the platform one.

143 struct BulkExtractor {
144  template <typename T>
145  static auto deref(ReaderHolder<T>& holder, size_t maxSize)
146  {
147  holder.buffer[holder.pos % PREBUFFER_SIZE] = **holder.reader;
148  holder.pos++;
149  if (holder.pos == maxSize) {
150  return BulkInfo<Remap64Bit_t<T> const*>{holder.buffer, maxSize % PREBUFFER_SIZE};
151  }
152  // We flush only after PREBUFFER_SIZE items have been inserted
153  if ((holder.pos % PREBUFFER_SIZE) != 0) {
154  return BulkInfo<Remap64Bit_t<T> const*>{nullptr, 0};
155  }
156  return BulkInfo<Remap64Bit_t<T> const*>{holder.buffer, PREBUFFER_SIZE};
157  }
158 
159  template <typename T, int N>
160  static auto deref(ReaderHolder<T[N]>& holder, size_t maxSize)
161  {
162  memcpy(&holder.buffer[(holder.pos % PREBUFFER_SIZE) * N], &((*holder.reader)[0]), N * sizeof(T));
163  holder.pos++;
164  if (holder.pos == maxSize) {
165  return BulkInfo<Remap64Bit_t<T> const*>{holder.buffer, maxSize % PREBUFFER_SIZE};
166  }
167  // We flush only after PREBUFFER_SIZE items have been inserted
168  if ((holder.pos % PREBUFFER_SIZE) != 0) {
169  return BulkInfo<Remap64Bit_t<T> const*>{nullptr, 0};
170  }
171  return BulkInfo<Remap64Bit_t<T> const*>{reinterpret_cast<T const*>(holder.buffer), PREBUFFER_SIZE};
172  }
173 };

201 struct RootTableBuilderHelpers {
203  template <typename... T>
204  static void convertTTree(TableBuilder& builder,
205  TTreeReader& reader,
206  ReaderHolder<T>... holders)
207  {
208  std::vector<std::string> branchNames = {holders.reader->GetBranchName()...};
209  TTree* tree = reader.GetTree();
210  size_t maxExtries = reader.GetEntries(true);
211  tree->SetCacheSize(maxExtries * (holders.itemSize + ...));
212  (tree->AddBranchToCache(tree->GetBranch(holders.reader->GetBranchName()), true), ...);
213  tree->StopCacheLearningPhase();
214 
215  auto filler = builder.bulkPersistChunked<Remap64Bit_t<typename std::decay_t<decltype(holders)>::Type>...>(branchNames, maxExtries);
216  while (reader.Next()) {
217  filler(0, BulkExtractor::deref(holders, maxExtries)...);
218  }
219  }
220 
221  template <typename... C>
222  static void convertASoAColumns(TableBuilder& builder, TTreeReader& reader, pack<C...>)
223  {
224  return convertTTree(builder, reader, ColumnReaderTrait<C>::createReader(reader)...);
225  }
226 
227  template <typename T>
228  static void convertASoA(TableBuilder& builder, TTreeReader& reader)
229  {
230  return convertASoAColumns(builder, reader, typename T::persistent_columns_t{});
231  }
