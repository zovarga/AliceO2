28 GPUdii() void GPUTPCCompressionKernels::Thread<GPUTPCCompressionKernels::step0attached>(int nBlocks, int nThreads, int iBlock, int iThread, GPUsharedref() GPUSharedMemory& smem, processorType& processors)
29 {
30  const GPUTrackingInOutPointers& GPUrestrict() ioPtrs = processors.ioPtrs;
31  const o2::tpc::ClusterNativeAccess* GPUrestrict() clusters = ioPtrs.clustersNative;
32  GPUTPCCompression& GPUrestrict() compressor = processors.tpcCompressor;
33  const GPUParam& GPUrestrict() param = processors.param;
34 
35  char lastLeg = 0;
36  int myTrack = 0;
37  for (unsigned int i = get_global_id(0); i < ioPtrs.nMergedTracks; i += get_global_size(0)) {
38  GPUbarrierWarp();
39  const GPUTPCGMMergedTrack& GPUrestrict() trk = ioPtrs.mergedTracks[i];
40  if (!trk.OK()) {
41  continue;
42  }
43  bool rejectTrk = CAMath::Abs(trk.GetParam().GetQPt()) > processors.param.rec.tpc.rejectQPt || trk.MergedLooper();
44  unsigned int nClustersStored = 0;
45  CompressedClustersPtrs& GPUrestrict() c = compressor.mPtrs;
46  unsigned int lastRow = 0, lastSlice = 0; // BUG: These should be unsigned char, but then CUDA breaks
47  GPUTPCCompressionTrackModel track;
48  float zOffset = 0;
49  for (int k = trk.NClusters() - 1; k >= 0; k--) {
50  const GPUTPCGMMergedTrackHit& GPUrestrict() hit = ioPtrs.mergedTrackHits[trk.FirstClusterRef() + k];
51  if (hit.state & GPUTPCGMMergedTrackHit::flagReject) {
52  continue;
53  }
54 
55  int hitId = hit.num;
56  int attach = ioPtrs.mergedTrackHitAttachment[hitId];
57  if ((attach & gputpcgmmergertypes::attachTrackMask) != i) {
58  continue; // Main attachment to different track
59  }
60  bool rejectCluster = processors.param.rec.tpc.trackFitRejectMode && (rejectTrk || GPUTPCClusterRejection::GetIsRejected(attach));
61  if (rejectCluster) {
62  compressor.mClusterStatus[hitId] = 1; // Cluster rejected, do not store
63  continue;
64  }
65 
66  if (!(param.rec.tpc.compressionTypeMask & GPUSettings::CompressionTrackModel)) {
67  continue; // No track model compression
68  }
69  const ClusterNative& GPUrestrict() orgCl = clusters->clusters[hit.slice][hit.row][hit.num - clusters->clusterOffset[hit.slice][hit.row]];
70  float x = param.tpcGeometry.Row2X(hit.row);
71  float y = param.tpcGeometry.LinearPad2Y(hit.slice, hit.row, orgCl.getPad());
72  float z = param.tpcGeometry.LinearTime2Z(hit.slice, orgCl.getTime());
73  if (nClustersStored) {
74  if ((hit.slice < GPUCA_NSLICES) ^ (lastSlice < GPUCA_NSLICES)) {
75  break;
76  }
77  if (lastLeg != hit.leg && track.Mirror()) {
78  break;
79  }
80  if (track.Propagate(param.tpcGeometry.Row2X(hit.row), param.SliceParam[hit.slice].Alpha)) {
81  break;
82  }
83  }
84 
85  compressor.mClusterStatus[hitId] = 1; // Cluster compressed in track model, do not store as difference
86 
87  int cidx = trk.FirstClusterRef() + nClustersStored++;
88  if (nClustersStored == 1) {
89  unsigned char qpt = fabs(trk.GetParam().GetQPt()) < 20.f ? (trk.GetParam().GetQPt() * (127.f / 20.f) + 127.5f) : (trk.GetParam().GetQPt() > 0 ? 254 : 0);
90  zOffset = z;
91  track.Init(x, y, z - zOffset, param.SliceParam[hit.slice].Alpha, qpt, param);
92 
93  myTrack = CAMath::AtomicAdd(&compressor.mMemory->nStoredTracks, 1u);
94  compressor.mAttachedClusterFirstIndex[myTrack] = trk.FirstClusterRef();
95  lastLeg = hit.leg;
96  c.qPtA[myTrack] = qpt;
97  c.rowA[myTrack] = hit.row;
98  c.sliceA[myTrack] = hit.slice;
99  c.timeA[myTrack] = orgCl.getTimePacked();
100  c.padA[myTrack] = orgCl.padPacked;
101  } else {
102  unsigned int row = hit.row;
103  unsigned int slice = hit.slice;
104 
105  if (param.rec.tpc.compressionTypeMask & GPUSettings::CompressionDifferences) {
106  if (lastRow > row) {
107  row += GPUCA_ROW_COUNT;
108  }
109  row -= lastRow;
110  if (lastSlice > slice) {
111  slice += compressor.NSLICES;
112  }
113  slice -= lastSlice;
114  }
115  c.rowDiffA[cidx] = row;
116  c.sliceLegDiffA[cidx] = (hit.leg == lastLeg ? 0 : compressor.NSLICES) + slice;
117  float pad = CAMath::Max(0.f, CAMath::Min((float)param.tpcGeometry.NPads(GPUCA_ROW_COUNT - 1), param.tpcGeometry.LinearY2Pad(hit.slice, hit.row, track.Y())));
118  c.padResA[cidx] = orgCl.padPacked - orgCl.packPad(pad);
119  float time = CAMath::Max(0.f, param.tpcGeometry.LinearZ2Time(hit.slice, track.Z() + zOffset));
120  c.timeResA[cidx] = (orgCl.getTimePacked() - orgCl.packTime(time)) & 0xFFFFFF;
121  lastLeg = hit.leg;
122  }
123  unsigned short qtot = orgCl.qTot, qmax = orgCl.qMax;
124  unsigned char sigmapad = orgCl.sigmaPadPacked, sigmatime = orgCl.sigmaTimePacked;
125  if (param.rec.tpc.compressionTypeMask & GPUSettings::CompressionTruncate) {
126  compressor.truncateSignificantBitsChargeMax(qmax, param);
127  compressor.truncateSignificantBitsCharge(qtot, param);
128  compressor.truncateSignificantBitsWidth(sigmapad, param);
129  compressor.truncateSignificantBitsWidth(sigmatime, param);
130  }
131  c.qTotA[cidx] = qtot;
132  c.qMaxA[cidx] = qmax;
133  c.sigmaPadA[cidx] = sigmapad;
134  c.sigmaTimeA[cidx] = sigmatime;
135  c.flagsA[cidx] = orgCl.getFlags();
136  if (k && track.Filter(y, z - zOffset, hit.row)) {
137  break;
138  }
139  lastRow = hit.row;
140  lastSlice = hit.slice;
141  }
142  if (nClustersStored) {
143  CAMath::AtomicAdd(&compressor.mMemory->nStoredAttachedClusters, nClustersStored);
144  c.nTrackClusters[myTrack] = nClustersStored;
145  }
146  }
147 }

180 GPUdii() void GPUTPCCompressionKernels::Thread<GPUTPCCompressionKernels::step1unattached>(int nBlocks, int nThreads, int iBlock, int iThread, GPUsharedref() GPUSharedMemory& smem, processorType& GPUrestrict() processors)
181 {
182  const GPUTrackingInOutPointers& GPUrestrict() ioPtrs = processors.ioPtrs;
183  const o2::tpc::ClusterNativeAccess* GPUrestrict() clusters = ioPtrs.clustersNative;
184  GPUTPCCompression& GPUrestrict() compressor = processors.tpcCompressor;
185  GPUParam& GPUrestrict() param = processors.param;
186  unsigned int* sortBuffer = smem.sortBuffer;
187  for (int iSliceRow = iBlock; iSliceRow < GPUCA_NSLICES * GPUCA_ROW_COUNT; iSliceRow += nBlocks) {
188  const int iSlice = iSliceRow / GPUCA_ROW_COUNT;
189  const int iRow = iSliceRow % GPUCA_ROW_COUNT;
190  const int idOffset = clusters->clusterOffset[iSlice][iRow];
191  const int idOffsetOut = clusters->clusterOffset[iSlice][iRow] * compressor.mMaxClusterFactorBase1024 / 1024;
192  const int idOffsetOutMax = ((const unsigned int*)clusters->clusterOffset[iSlice])[iRow + 1] * compressor.mMaxClusterFactorBase1024 / 1024; // Array out of bounds access is ok, since it goes to the correct nClustersTotal
193  if (iThread == nThreads - 1) {
194  smem.nCount = 0;
195  }
196  unsigned int totalCount = 0;
197  GPUbarrier();
198 
199  CompressedClustersPtrs& GPUrestrict() c = compressor.mPtrs;
200 
201  const unsigned int nn = GPUCommonMath::nextMultipleOf<GPUCA_GET_THREAD_COUNT(GPUCA_LB_GPUTPCCompressionKernels_step1unattached)>(clusters->nClusters[iSlice][iRow]);
202  for (unsigned int i = iThread; i < nn + nThreads; i += nThreads) {
203  const int idx = idOffset + i;
204  int cidx = 0;
205  do {
206  if (i >= clusters->nClusters[iSlice][iRow]) {
207  break;
208  }
209  if (compressor.mClusterStatus[idx]) {
210  break;
211  }
212  int attach = ioPtrs.mergedTrackHitAttachment[idx];
213  bool unattached = attach == 0;
214 
215  if (unattached) {
216  if (processors.param.rec.tpc.rejectionStrategy >= GPUSettings::RejectionStrategyB) {
217  break;
218  }
219  } else if (processors.param.rec.tpc.rejectionStrategy >= GPUSettings::RejectionStrategyA) {
220  if (GPUTPCClusterRejection::GetIsRejected(attach)) {
221  break;
222  }
223  int id = attach & gputpcgmmergertypes::attachTrackMask;
224  auto& trk = ioPtrs.mergedTracks[id];
225  if (CAMath::Abs(trk.GetParam().GetQPt()) > processors.param.rec.tpc.rejectQPt || trk.MergedLooper()) {
226  break;
227  }
228  }
229  cidx = 1;
230  } while (false);
231 
232  GPUbarrier();
233  int myIndex = work_group_scan_inclusive_add(cidx);
234  int storeLater = -1;
235  if (cidx) {
236  if (smem.nCount + myIndex <= GPUCA_TPC_COMP_CHUNK_SIZE) {
237  sortBuffer[smem.nCount + myIndex - 1] = i;
238  } else {
239  storeLater = smem.nCount + myIndex - 1 - GPUCA_TPC_COMP_CHUNK_SIZE;
240  }
241  }
242  GPUbarrier();
243  if (iThread == nThreads - 1) {
244  smem.nCount += myIndex;
245  }
246  GPUbarrier();
247 
248  if (smem.nCount < GPUCA_TPC_COMP_CHUNK_SIZE && i < nn) {
249  continue;
250  }
251 
252  unsigned int count = CAMath::Min(smem.nCount, (unsigned int)GPUCA_TPC_COMP_CHUNK_SIZE);
253  if (param.rec.tpc.compressionTypeMask & GPUSettings::CompressionDifferences) {
254  if (param.rec.tpc.compressionSortOrder == GPUSettings::SortZPadTime) {
255  CAAlgo::sortInBlock(sortBuffer, sortBuffer + count, GPUTPCCompressionKernels_Compare<GPUSettings::SortZPadTime>(clusters->clusters[iSlice][iRow]));
256  } else if (param.rec.tpc.compressionSortOrder == GPUSettings::SortZTimePad) {
257  CAAlgo::sortInBlock(sortBuffer, sortBuffer + count, GPUTPCCompressionKernels_Compare<GPUSettings::SortZTimePad>(clusters->clusters[iSlice][iRow]));
258  } else if (param.rec.tpc.compressionSortOrder == GPUSettings::SortPad) {
259  CAAlgo::sortInBlock(sortBuffer, sortBuffer + count, GPUTPCCompressionKernels_Compare<GPUSettings::SortPad>(clusters->clusters[iSlice][iRow]));
260  } else if (param.rec.tpc.compressionSortOrder == GPUSettings::SortTime) {
261  CAAlgo::sortInBlock(sortBuffer, sortBuffer + count, GPUTPCCompressionKernels_Compare<GPUSettings::SortTime>(clusters->clusters[iSlice][iRow]));
262  }
263  GPUbarrier();
264  }
265 
266  for (unsigned int j = get_local_id(0); j < count; j += get_local_size(0)) {
267  int outidx = idOffsetOut + totalCount + j;
268  if (outidx >= idOffsetOutMax) {
269  compressor.raiseError(GPUErrors::ERROR_COMPRESSION_ROW_HIT_OVERFLOW, outidx, idOffsetOutMax);
270  count = 0;
271  break;
272  }
273  const ClusterNative& GPUrestrict() orgCl = clusters->clusters[iSlice][iRow][sortBuffer[j]];
274  unsigned int lastTime = 0;
275  unsigned int lastPad = 0;
276  if (param.rec.tpc.compressionTypeMask & GPUSettings::CompressionDifferences) {
277  if (j != 0) {
278  const ClusterNative& GPUrestrict() orgClPre = clusters->clusters[iSlice][iRow][sortBuffer[j - 1]];
279  lastPad = orgClPre.padPacked;
280  lastTime = orgClPre.getTimePacked();
281  } else if (totalCount != 0) {
282  const ClusterNative& GPUrestrict() orgClPre = clusters->clusters[iSlice][iRow][smem.lastIndex];
283  lastPad = orgClPre.padPacked;
284  lastTime = orgClPre.getTimePacked();
285  }
286 
287  c.padDiffU[outidx] = orgCl.padPacked - lastPad;
288  c.timeDiffU[outidx] = (orgCl.getTimePacked() - lastTime) & 0xFFFFFF;
289  } else {
290  c.padDiffU[outidx] = orgCl.padPacked;
291  c.timeDiffU[outidx] = orgCl.getTimePacked();
292  }
293 
294  unsigned short qtot = orgCl.qTot, qmax = orgCl.qMax;
295  unsigned char sigmapad = orgCl.sigmaPadPacked, sigmatime = orgCl.sigmaTimePacked;
296  if (param.rec.tpc.compressionTypeMask & GPUSettings::CompressionTruncate) {
297  compressor.truncateSignificantBitsChargeMax(qmax, param);
298  compressor.truncateSignificantBitsCharge(qtot, param);
299  compressor.truncateSignificantBitsWidth(sigmapad, param);
300  compressor.truncateSignificantBitsWidth(sigmatime, param);
301  }
302  c.qTotU[outidx] = qtot;
303  c.qMaxU[outidx] = qmax;
304  c.sigmaPadU[outidx] = sigmapad;
305  c.sigmaTimeU[outidx] = sigmatime;
306  c.flagsU[outidx] = orgCl.getFlags();
307  }
308 
309  GPUbarrier();
310  if (storeLater > 0) {
311  sortBuffer[storeLater] = i;
312  }
313  totalCount += count;
314  if (iThread == nThreads - 1 && count) {
315  smem.lastIndex = sortBuffer[count - 1];
316  smem.nCount -= count;
317  }
318  }
319 
320  if (iThread == nThreads - 1) {
321  c.nSliceRowClusters[iSlice * GPUCA_ROW_COUNT + iRow] = totalCount;
322  CAMath::AtomicAdd(&compressor.mMemory->nStoredUnattachedClusters, totalCount);
323  }
324  GPUbarrier();
325  }
326 }

414 {

416  size_t dsti = reinterpret_cast<size_t>(dst);
417  int offset = (alignof(BaseVector) - dsti % alignof(BaseVector)) / sizeof(Scalar);
418  compressorMemcpyBasic(dst, src, offset, nThreads, iThread);
419  src += offset;
420  dst += offset;
421  size -= offset;
422  }

456 template <typename V, typename T, typename S>
457 GPUdi() void GPUTPCCompressionGatherKernels::compressorMemcpyBuffered(V* buf, T* GPUrestrict() dst, const T* GPUrestrict() src, const S* GPUrestrict() nums, const unsigned int* GPUrestrict() srcOffsets, unsigned int nEntries, int nLanes, int iLane, int diff, size_t scaleBase1024)
458 {
459  int shmPos = 0;
460  unsigned int dstOffset = 0;
461  V* GPUrestrict() dstAligned = nullptr;
462 
463  T* bufT = reinterpret_cast<T*>(buf);
464  CONSTEXPR int bufSize = GPUCA_WARP_SIZE;
465  CONSTEXPR int bufTSize = bufSize * sizeof(V) / sizeof(T);
466 
467  for (unsigned int i = 0; i < nEntries; i++) {
468  unsigned int srcPos = 0;
469  unsigned int srcOffset = (srcOffsets[i] * scaleBase1024 / 1024) + diff;
470  unsigned int srcSize = nums[i] - diff;
471 
472  if (dstAligned == nullptr) {
473  if (not isAlignedTo<V>(dst)) {
474  size_t dsti = reinterpret_cast<size_t>(dst);
475  unsigned int offset = (alignof(V) - dsti % alignof(V)) / sizeof(T);
476  offset = CAMath::Min<unsigned int>(offset, srcSize);
477  compressorMemcpyBasic(dst, src + srcOffset, offset, nLanes, iLane);
478  dst += offset;
479  srcPos += offset;
480  }
481  if (isAlignedTo<V>(dst)) {
482  dstAligned = reinterpret_cast<V*>(dst);
483  }
484  }
485  while (srcPos < srcSize) {
