39 void Clusterer::process(gsl::span<const Digit> digits, gsl::span<const TriggerRecord> dtr,
40  const o2::dataformats::MCTruthContainer<MCLabel>* dmc,
41  std::vector<Cluster>& clusters, std::vector<CluElement>& cluelements, std::vector<TriggerRecord>& trigRec,
42  o2::dataformats::MCTruthContainer<MCLabel>& cluMC)
43 {
44  clusters.clear(); //final out list of clusters
45  cluelements.clear();
46  cluelements.reserve(digits.size());
47  trigRec.clear();
48  cluMC.clear();
49  mProcessMC = (dmc != nullptr);
50 
51  for (const auto& tr : dtr) {
52  int indexStart = clusters.size(); //final out list of clusters
53 
54  LOG(debug) << "Starting clusteriztion digits from " << mFirstElememtInEvent << " to " << mLastElementInEvent;
55  //Convert digits to cluelements
56  int firstDigitInEvent = tr.getFirstEntry();
57  int lastDigitInEvent = firstDigitInEvent + tr.getNumberOfObjects();
58  mFirstElememtInEvent = cluelements.size();
59  mCluEl.clear();
60  mTrigger.clear();
61  for (int i = firstDigitInEvent; i < lastDigitInEvent; i++) {
62  const Digit& digitSeed = digits[i];
63  short absId = digitSeed.getAbsId();
64  if (digitSeed.isTRU()) {
65  mTrigger.emplace_back(digitSeed);
66  continue;
67  }
68  if (isBadChannel(absId)) {
69  continue;
70  }
71  float energy = calibrate(digitSeed.getAmplitude(), absId, digitSeed.isHighGain());
72  if (energy < o2::phos::PHOSSimParams::Instance().mDigitMinEnergy) {
73  continue;
74  }
75  float x = 0., z = 0.;
76  Geometry::absIdToRelPosInModule(digits[i].getAbsId(), x, z);
77  mCluEl.emplace_back(absId, digitSeed.isHighGain(), energy, calibrateT(digitSeed.getTime(), absId, digitSeed.isHighGain()),
78  x, z, digitSeed.getLabel(), 1.);
79  }
80  mLastElementInEvent = cluelements.size();
81 
82  // Collect digits to clusters
83  makeClusters(clusters, cluelements);
84 
85  LOG(debug) << "Found clusters from " << indexStart << " to " << clusters.size();
86  trigRec.emplace_back(tr.getBCData(), indexStart, clusters.size() - indexStart);
87  }
88  if (mProcessMC) {
89  evalLabels(clusters, cluelements, dmc, cluMC);
90  }
91 }

93 void Clusterer::processCells(gsl::span<const Cell> cells, gsl::span<const TriggerRecord> ctr,
94  const o2::dataformats::MCTruthContainer<MCLabel>* dmc,
95  std::vector<Cluster>& clusters, std::vector<CluElement>& cluelements, std::vector<TriggerRecord>& trigRec,
96  o2::dataformats::MCTruthContainer<MCLabel>& cluMC)
97 {
98  // Transform input Cells to digits and run standard recontruction
99  clusters.clear(); //final out list of clusters
100  trigRec.clear();
101  cluelements.reserve(cells.size());
102  cluMC.clear();
103  mProcessMC = (dmc != nullptr);
104  miCellLabel = 0;
105  for (const auto& tr : ctr) {
106  int firstCellInEvent = tr.getFirstEntry();
107  int lastCellInEvent = firstCellInEvent + tr.getNumberOfObjects();
108  int indexStart = clusters.size(); //final out list of clusters
109  LOG(debug) << "Starting clusteriztion cells from " << firstCellInEvent << " to " << lastCellInEvent;
110  //convert cells to cluelements
111  mFirstElememtInEvent = cluelements.size();
112  mCluEl.clear();
113  mTrigger.clear();
114  for (int i = firstCellInEvent; i < lastCellInEvent; i++) {
115  const Cell c = cells[i];
116  short absId = c.getAbsId();
117  if (c.getTRU()) {
118  mTrigger.emplace_back(c.getTRUId(), c.getEnergy(), c.getTime(), 0);
119  continue;
120  }
121  if (isBadChannel(absId)) {
122  continue;
123  }
124  float energy = calibrate(c.getEnergy(), absId, c.getHighGain());
125  if (energy < o2::phos::PHOSSimParams::Instance().mDigitMinEnergy) {
126  continue;
127  }
128  float x = 0., z = 0.;
129  Geometry::absIdToRelPosInModule(absId, x, z);
130  mCluEl.emplace_back(absId, c.getHighGain(), energy, calibrateT(c.getTime(), absId, c.getHighGain()),
131  x, z, i, 1.);
132  }
133  mLastElementInEvent = cluelements.size();
134 
135  makeClusters(clusters, cluelements);
136 
137  LOG(debug) << "Found clusters from " << indexStart << " to " << clusters.size();
138  trigRec.emplace_back(tr.getBCData(), indexStart, clusters.size() - indexStart);
139  }
140  if (mProcessMC) {
141  evalLabels(clusters, cluelements, dmc, cluMC);
142  }

145 void Clusterer::makeClusters(std::vector<Cluster>& clusters, std::vector<CluElement>& cluelements)
146 {
147  // A cluster is defined as a list of neighbour digits (as defined in Geometry::areNeighbours)
148  // Cluster contains first and (next-to) last index of the combined list of clusterelements, so
149  // add elements to final list and mark element in internal list as used (zero energy)
150 
151  int iFirst = 0; // first index of digit which potentially can be a part of cluster
152  int n = mCluEl.size();
153  for (int i = iFirst; i < n; i++) {
154  if (mCluEl[i].energy == 0) { //already used
155  continue;
156  }
157 
158  CluElement& digitSeed = mCluEl[i];
159 
160  // is this digit so energetic that start cluster?
161  Cluster* clu = nullptr;
162  int iDigitInCluster = 0;
163  if (digitSeed.energy > o2::phos::PHOSSimParams::Instance().mClusteringThreshold) {
164  // start new cluster
165  clusters.emplace_back();
166  clu = &(clusters.back());
167  clu->setFirstCluEl(cluelements.size());
168  cluelements.emplace_back(digitSeed);
169  digitSeed.energy = 0;
170  iDigitInCluster = 1;
171  } else {
172  continue;
173  }
174  // Now scan remaining digits in list to find neigbours of our seed
175  int index = 0;
176  while (index < iDigitInCluster) { // scan over digits already in cluster
177  short digitSeedAbsId = cluelements.at(clu->getFirstCluEl() + index).absId;
178  index++;
179  for (int j = iFirst; j < n; j++) {
180  if (mCluEl[j].energy == 0) {
181  continue; // look through remaining digits
182  }
183  CluElement& digitN = mCluEl[j];
184 
185  // call (digit,digitN) in THAT oder !!!!!
186  Int_t ineb = Geometry::areNeighbours(digitSeedAbsId, digitN.absId);
187  switch (ineb) {
188  case -1: // too early (e.g. previous module), do not look before j at subsequent passes
189  iFirst = j;
190  break;
191  case 0: // not a neighbour
192  break;
193  case 1: // are neighbours
194  cluelements.emplace_back(digitN);
195  digitN.energy = 0;
196  iDigitInCluster++;
197  break;
198  case 2: // too far from each other
199  default:
200  break;
201  } // switch
202  }
203  } // loop over cluster
204  clu->setLastCluEl(cluelements.size());
205 
206  // Unfold overlapped clusters
207  // Split clusters with several local maxima if necessary
208  if (o2::phos::PHOSSimParams::Instance().mUnfoldClusters) {
209  makeUnfolding(*clu, clusters, cluelements);
210  } else {
211  evalAll(*clu, cluelements);
212  if (clu->getEnergy() < 1.e-4) { //remove cluster and belonging to it elements
213  for (int i = clu->getMultiplicity(); i--;) {
214  cluelements.pop_back();
215  }
216  clusters.pop_back();
217  }
218  }
219 
220  } // energy theshold
221 }

223 void Clusterer::makeUnfolding(Cluster& clu, std::vector<Cluster>& clusters, std::vector<CluElement>& cluelements)
224 {
225  //Split cluster if several local maxima are found
226  if (clu.getNExMax() > -1) { //already unfolded
227  return;
228  }
229 
230  char nMax = getNumberOfLocalMax(clu, cluelements);
231  if (nMax > 1) {
232  unfoldOneCluster(clu, nMax, clusters, cluelements);
233  } else {
234  clu.setNExMax(nMax); // Only one local maximum
235  evalAll(clu, cluelements);
236  if (clu.getEnergy() < 1.e-4) { //remove cluster and belonging to it elements
237  for (int i = clu.getMultiplicity(); i--;) {
238  cluelements.pop_back();
239  }
240  clusters.pop_back();
241  }
242  }

245 void Clusterer::unfoldOneCluster(Cluster& iniClu, char nMax, std::vector<Cluster>& clusters, std::vector<CluElement>& cluelements)
246 {
247  // Performs the unfolding of a cluster with nMax overlapping showers
248  // Parameters: iniClu cluster to be unfolded
249  // nMax number of local maxima found (this is the number of new clusters)
250  // digitId: index of digits, corresponding to local maxima
251  // maxAtEnergy: energies of digits, corresponding to local maxima
252 
253  // Take initial cluster and calculate local coordinates of digits
254  // To avoid multiple re-calculation of same parameters
255  short mult = iniClu.getMultiplicity();
256  std::vector<std::vector<float>> eInClusters(mult, std::vector<float>(nMax));
257  uint32_t firstCE = iniClu.getFirstCluEl();
258  uint32_t lastCE = iniClu.getLastCluEl();
259 
260  mProp.reserve(mult * nMax);
261 
262  for (int iclu = nMax; iclu--;) {
263  CluElement& ce = cluelements[mMaxAt[iclu]];
264  mxMax[iclu] = ce.localX;
265  mzMax[iclu] = ce.localZ;
266  meMax[iclu] = ce.energy;
267  mxMaxPrev[iclu] = mxMax[iclu];
268  mzMaxPrev[iclu] = mzMax[iclu];
269  }
270 
271  TMatrixDSym B(nMax);
272  TVectorD C(nMax);
273  TDecompBK bk(nMax);
274 
275  // Try to decompose cluster to contributions
276  int nIterations = 0;
277  bool insuficientAccuracy = true;
278  double chi2Previous = 1.e+6;
279  double step = 0.2;
280  while (insuficientAccuracy && nIterations < o2::phos::PHOSSimParams::Instance().mNMaxIterations) {
281  insuficientAccuracy = false; // will be true if at least one parameter changed too much
282  B.Zero();
283  C.Zero();
284  mProp.clear();
285  double chi2 = 0.;
286  for (int iclu = nMax; iclu--;) {
287  mA[iclu] = 0;
288  mxB[iclu] = 0;
289  mzB[iclu] = 0;
290  }
291  //Fill matrix and vector
292  for (int idig = firstCE; idig < lastCE; idig++) {
293  CluElement& ce = cluelements[idig];
294  double sumA = 0.;
295  for (int iclu = nMax; iclu--;) {
296  double lx = ce.localX - mxMax[iclu];
297  double lz = ce.localZ - mzMax[iclu];
298  double r2 = lx * lx + lz * lz;
299  double deriv = 0;
300  double ss = showerShape(r2, deriv);
301  mfij[iclu] = ss;
302  mfijr[iclu] = deriv;
303  mfijx[iclu] = deriv * ce.localX; //derivatives
304  mfijz[iclu] = deriv * ce.localZ;
305  sumA += ss * meMax[iclu];
306  C(iclu) += ce.energy * ss;
307  }
308  double dE = ce.energy - sumA;
309  chi2 += dE * dE;
310  for (int iclu = 0; iclu < nMax; iclu++) {
311  for (int jclu = iclu; jclu < nMax; jclu++) {
312  B(iclu, jclu) += mfij[iclu] * mfij[jclu];
313  }
314  mA[iclu] += mfijr[iclu] * dE;
315  mxB[iclu] += mfijx[iclu] * dE;
316  mzB[iclu] += mfijz[iclu] * dE;
317  mProp[(idig - firstCE) * nMax + iclu] = mfij[iclu] * meMax[iclu] / sumA;
318  }
319  }
320  if (nIterations > 0 && chi2 > chi2Previous) { //too big step
321  step = 0.5 * step;
322  for (int iclu = nMax; iclu--;) {
323  mxMax[iclu] = mxMaxPrev[iclu] + step * mdx[iclu];
324  mzMax[iclu] = mzMaxPrev[iclu] + step * mdz[iclu];
325  }
326  nIterations++;
327  insuficientAccuracy = true;
328  continue;
329  }
330  //Good iteration, move further
331  step = 0.2;
332  chi2Previous = chi2;
333  for (int iclu = nMax; iclu--;) {
334  mxMaxPrev[iclu] = mxMax[iclu];
335  mzMaxPrev[iclu] = mzMax[iclu];
336  }
337 
338  //calculate next step using derivative
339  //fill remaning part of B
340  for (int iclu = 1; iclu < nMax; iclu++) {
341  for (int jclu = 0; jclu < iclu; jclu++) {
342  B(iclu, jclu) = B(jclu, iclu);
343  }
344  }
345  for (int iclu = nMax; iclu--;) {
346  if (mA[iclu] != 0) {
347  mdx[iclu] = mxB[iclu] / mA[iclu] - mxMaxPrev[iclu];
348  mdz[iclu] = mzB[iclu] / mA[iclu] - mzMaxPrev[iclu];
349  }
350  }
351 
352  for (int iclu = nMax; iclu--;) {
353  //a-la Fletcher-Rivs algorithm
354  mdx[iclu] += 0.2 * mdxprev[iclu];
355  mdz[iclu] += 0.2 * mdzprev[iclu];
356  mdxprev[iclu] = mdx[iclu];
357  mdzprev[iclu] = mdz[iclu];
358  insuficientAccuracy |= fabs(step * mdx[iclu]) > o2::phos::PHOSSimParams::Instance().mUnfogingXZAccuracy;
359  insuficientAccuracy |= fabs(step * mdz[iclu]) > o2::phos::PHOSSimParams::Instance().mUnfogingXZAccuracy;
360  mxMax[iclu] = mxMaxPrev[iclu] + step * mdx[iclu];
361  mzMax[iclu] = mzMaxPrev[iclu] + step * mdz[iclu];
362  }
363  //now exact solution for amplitudes
364  bk.SetMatrix(B);
365  if (bk.Solve(C)) {
366  for (int iclu = 0; iclu < nMax; iclu++) {
367  double eOld = meMax[iclu];
368  meMax[iclu] = C(iclu);
369  // insuficientAccuracy|=fabs(meMax[iclu]-eOld)> meMax[iclu]*o2::phos::PHOSSimParams::Instance().mUnfogingEAccuracy ;
370  }
371  }
372  insuficientAccuracy &= (chi2 > o2::phos::PHOSSimParams::Instance().mUnfogingChi2Accuracy * nMax);
373  nIterations++;
374  }
375 
376  // Iterations finished, put first new cluster into place of mother one, others to the end of list
377  for (int iclu = 0; iclu < nMax; iclu++) {
378  //copy cluElements to the final list
379  int start = cluelements.size();
380  int nce = 0;
381  for (int idig = firstCE; idig < lastCE; idig++) {
382  float eDigit = eInClusters[idig - firstCE][iclu];
383  CluElement& el = cluelements[idig];
384  float ei = el.energy * mProp[(idig - firstCE) * nMax + iclu];
385  if (ei > o2::phos::PHOSSimParams::Instance().mDigitMinEnergy) {
386  cluelements.emplace_back(el);
387  cluelements.back().energy = ei;
388  cluelements.back().fraction = mProp[(idig - firstCE) * nMax + iclu];
389  nce++;
390  }
391  }
392  if (iclu == 0) { //replace parent
393  iniClu.setNExMax(nMax);
394  iniClu.setFirstCluEl(start);
395  iniClu.setLastCluEl(start + nce);
396  evalAll(iniClu, cluelements);
397  if (iniClu.getEnergy() < 1.e-4) { //remove cluster and belonging to it elements
398  for (int i = iniClu.getMultiplicity(); i--;) {
399  cluelements.pop_back();
400  }
401  clusters.pop_back();
402  }
403  } else {
404  clusters.emplace_back();
405  Cluster& clu = clusters.back();
406  clu.setNExMax(nMax);
407  clu.setFirstCluEl(start);
408  clu.setLastCluEl(start + nce);
409  evalAll(clu, cluelements);
410  if (clu.getEnergy() < 1.e-4) { //remove cluster and belonging to it elements
411  for (int i = clu.getMultiplicity(); i--;) {
412  cluelements.pop_back();
413  }
414  clusters.pop_back();
415  }
416  }
417  }
418 }

421 void Clusterer::evalLabels(std::vector<Cluster>& clusters, std::vector<CluElement>& cluElements,
422  const o2::dataformats::MCTruthContainer<MCLabel>* dmc,
423  o2::dataformats::MCTruthContainer<MCLabel>& cluMC)
424 {
425 
426  int labelIndex = cluMC.getIndexedSize();
427  auto clu = clusters.begin();
428 
429  while (clu != clusters.end()) {
430  //Calculate list of primaries
431  //loop over entries in digit MCTruthContainer
432  for (uint32_t id = clu->getFirstCluEl(); id < clu->getLastCluEl(); id++) {
433  CluElement& ll = cluElements[id];
434  int i = ll.label; //index
435  float sc = ll.fraction;
436  gsl::span<const MCLabel> spDigList = dmc->getLabels(i);
437  if (spDigList.size() == 0 || spDigList.begin()->isFake()) {
438  continue;
439  }
440  gsl::span<MCLabel> spCluList = cluMC.getLabels(labelIndex); //get updated list
441  auto digL = spDigList.begin();
442  while (digL != spDigList.end()) {
443  if (digL->isFake()) {
444  digL++;
445  continue;
446  }
447  bool merged = false;
448  auto cluL = spCluList.begin();
449  while (cluL != spCluList.end()) {
450  if (*digL == *cluL) {
451  (*cluL).add(*digL, sc);
452  merged = true;
453  break;
454  }
455  ++cluL;
456  }
457  if (!merged) { //just add label
458  if (sc == 1.) {
459  cluMC.addElement(labelIndex, (*digL));
460  } else { //rare case of unfolded clusters
461  MCLabel tmpL = (*digL);
462  tmpL.scale(sc);
463  cluMC.addElement(labelIndex, tmpL);
464  }
465  }
466  ++digL;
467  }
468  }
469  labelIndex++;
470  ++clu;
471  }
472 }

474 double Clusterer::showerShape(double r2, double& deriv)
475 {
476  // Shape of the shower (see PHOS TDR)
477  // we neglect dependence on the incident angle.
478 
479  // const float width = 1. / (2. * 2.32 * 2.32 * 2.32 * 2.32 * 2.32 * 2.32);
480  // const float width = 1. / (2. * 2.32 * 2.32 * 2.32 * 2.32 );
481  // return TMath::Exp(-r2 * r2 * width);
482  if (r2 == 0.) {
483  deriv = 0;
484  return 1.;
485  }
486  double r4 = r2 * r2;
487  double r295 = TMath::Power(r2, 2.95 / 2.);
488  double a = 2.32 + 0.26 * r4;
489  double b = 31.645570 + 2.0632911 * r295;
490  double s = TMath::Exp(-r4 * ((a + b) / (a * b)));
491  deriv = -2. * s * r2 * (2.32 / (a * a) + (0.54161392 * r295 + 31.645570) / (b * b));
492  return s;
493 }

496 void Clusterer::evalAll(Cluster& clu, std::vector<CluElement>& cluel) const
497 {
498  //position, energy, coreEnergy, dispersion, time,
499 
500  // Calculates the center of gravity in the local PHOS-module coordinates
501  // Note that correction for non-perpendicular incidence will be applied later
502  // when vertex will be known.
503  float fullEnergy = 0.;
504  float time = 0.;
505  float eMax = 0.;
506  uint32_t iFirst = clu.getFirstCluEl(), iLast = clu.getLastCluEl();
507  clu.setModule(Geometry::absIdToModule(cluel[iFirst].absId));
508  float eMin = o2::phos::PHOSSimParams::Instance().mDigitMinEnergy;
509  for (uint32_t i = iFirst; i < iLast; i++) {
510  float ei = cluel[i].energy;
511  if (ei < eMin) {
512  continue;
513  }
514  fullEnergy += ei;
515  if (ei > eMax) {
516  time = cluel[i].time;
517  eMax = ei;
518  }
519  }
520  clu.setEnergy(fullEnergy);
521  if (fullEnergy <= 0) {
522  return;
523  }
524  // Calculate time as time in the digit with maximal energy
525  clu.setTime(time);
526 
527  float localPosX = 0., localPosZ = 0.;
528  float wtot = 0.;
529  float invE = 1. / fullEnergy;
530  for (uint32_t i = iFirst; i < iLast; i++) {
531  CluElement& ce = cluel[i];
532  if (ce.energy < eMin) {
533  continue;
534  }
535  float w = std::max(float(0.), o2::phos::PHOSSimParams::Instance().mLogWeight + std::log(ce.energy * invE));
536  localPosX += ce.localX * w;
537  localPosZ += ce.localZ * w;
538  wtot += w;
539  }
540  if (wtot > 0) {
541  wtot = 1. / wtot;
542  localPosX *= wtot;
543  localPosZ *= wtot;
544  }
545  clu.setLocalPosition(localPosX, localPosZ);
546 
547  //Dispersion, core energy
548  float coreRadius2 = o2::phos::PHOSSimParams::Instance().mCoreR;
549  coreRadius2 *= coreRadius2;
550  float coreE = 0.;
551  float dispersion = 0.;
552  float dxx = 0., dxz = 0., dzz = 0., lambdaLong = 0., lambdaShort = 0.;
553  for (uint32_t i = iFirst; i < iLast; i++) {
554  CluElement& ce = cluel[i];
555  float ei = ce.energy;
556  if (ei < eMin) {
557  continue;
558  }
559  float x = ce.localX - localPosX;
560  float z = ce.localZ - localPosZ;
561  float distance = x * x + z * z;
562 
563  float w = std::max(float(0.), o2::phos::PHOSSimParams::Instance().mLogWeight + std::log(ei * invE));
564  dispersion += w * distance;
565  dxx += w * x * x;
566  dzz += w * z * z;
567  dxz += w * x * z;
568  if (distance < coreRadius2) {
569  coreE += ei;
570  }
571  }
572  clu.setCoreEnergy(coreE);
573  //dispersion
574  if (wtot > 0) {
575  wtot = 1. / wtot;
576  dispersion *= wtot;
577 
578  dxx *= wtot;
579  dzz *= wtot;
580  dxz *= wtot;
581

618 char Clusterer::getNumberOfLocalMax(Cluster& clu, std::vector<CluElement>& cluel)
619 {
620  // Calculates the number of local maxima in the cluster using LocalMaxCut as the minimum
621  // energy difference between maximum and surrounding digits
622 
623  float locMaxCut = o2::phos::PHOSSimParams::Instance().mLocalMaximumCut;
624  float cluSeed = o2::phos::PHOSSimParams::Instance().mClusteringThreshold;
625  mIsLocalMax.clear();
626  mIsLocalMax.reserve(clu.getMultiplicity());
627 
628  uint32_t iFirst = clu.getFirstCluEl(), iLast = clu.getLastCluEl();
629  for (uint32_t i = iFirst; i < iLast; i++) {
630  mIsLocalMax.push_back(cluel[i].energy > cluSeed);
631  }
632 
633  for (uint32_t i = iFirst; i < iLast - 1; i++) {
634  for (int j = i + 1; j < iLast; j++) {
635 
636  if (Geometry::areNeighbours(cluel[i].absId, cluel[j].absId) == 1) {
637  if (cluel[i].energy > cluel[j].energy) {
638  mIsLocalMax[j - iFirst] = false;
639  // but may be digit too is not local max ?
640  if (cluel[j].energy > cluel[i].energy - locMaxCut) {
641  mIsLocalMax[i - iFirst] = false;
642  }
643  } else {
644  mIsLocalMax[i - iFirst] = false;
645  // but may be digitN is not local max too?
646  if (cluel[i].energy > cluel[j].energy - locMaxCut) {
647  mIsLocalMax[j - iFirst] = false;
648  }
649  }
650  } // if areneighbours
651  } // digit j
652  } // digit i
653 
654  int iDigitN = 0;
655  for (int i = 0; i < mIsLocalMax.size(); i++) {
656  if (mIsLocalMax[i]) {
657  mMaxAt[iDigitN] = i + iFirst;
658  iDigitN++;
659  if (iDigitN >= NLOCMAX) { // Note that size of output arrays is limited:
660  LOG(error) << "Too many local maxima, cluster multiplicity " << mIsLocalMax.size();
661  return -2;
662  }
663  }
664  }
