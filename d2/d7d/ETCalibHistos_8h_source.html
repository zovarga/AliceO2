71  void merge(ETCalibHistos& other)
72  {
73  for (int i = nChannels; --i;) {
74  for (int j = nMass; --j;) {
75  mReInvMassPerCell[i][j] += other.mReInvMassPerCell[i][j];
76  mMiInvMassPerCell[i][j] += other.mMiInvMassPerCell[i][j];
77  }
78  for (int j = nTime; --j;) {
79  mTimeHGPerCell[i][j] += other.mTimeHGPerCell[i][j];
80  mTimeLGPerCell[i][j] += other.mTimeLGPerCell[i][j];
81  }
82  }
83  for (int i = npt; --i;) {
84  for (int j = nMass; --j;) {
85  mReInvMassNonlin[i][j] += other.mReInvMassNonlin[i][j];
86  mMiInvMassNonlin[i][j] += other.mMiInvMassNonlin[i][j];
87  }
88  for (int j = nTime; --j;) {
89  mTimeHGSlewing[i][j] += other.mTimeHGSlewing[i][j];
90  mTimeLGSlewing[i][j] += other.mTimeLGSlewing[i][j];
91  }
92  }
93  }

95  void fill(int kind, float x, float y)
96  {
97  if (kind == kReInvMassNonlin || kind == kMiInvMassNonlin) {
98  int i = int(x / dm);
99  int j = int(y / dpt);
100  if (i < nMass && j < npt) {
101  if (kind == kReInvMassNonlin) {
102  mReInvMassNonlin[i][j]++;
103  } else {
104  mMiInvMassNonlin[i][j]++;
105  }
106  }
107  }
108  if (kind == kTimeHGSlewing || kind == kTimeLGSlewing) {
109  int i = int((x - timeMin) / dt);
110  int j = int(y / dpt);
111  if (i >= 0 && i < nTime && j < npt) {
112  if (kind == kTimeHGSlewing) {
113  mTimeHGSlewing[i][j]++;
114  } else {
115  mTimeHGSlewing[i][j]++;
116  }
117  }
118  }
119  }
120  void fill(int kind, int x, float y)
121  {
122  if (kind == kReInvMassPerCell || kind == kMiInvMassPerCell) {
123  int j = int(y / dm);
124  if (j < nMass) {
125  if (kind == kReInvMassPerCell) {
126  mReInvMassPerCell[x - offset][j]++;
127  } else {
128  mMiInvMassPerCell[x - offset][j]++;
129  }
130  }
131  }
132  if (kind == kTimeHGPerCell || kind == kTimeLGPerCell) {
133  int j = int((y - timeMin) / dt);
134  if (j >= 0 && j < nTime) {
135  if (kind == kTimeHGPerCell) {
136  mTimeHGPerCell[x - offset][j]++;
137  } else {
138  mTimeLGPerCell[x - offset][j]++;
139  }
140  }
141  }
142  }
143  void reset()
144  {
145  memset(&mReInvMassPerCell, 0, sizeof(mReInvMassPerCell));
146  memset(&mMiInvMassPerCell, 0, sizeof(mMiInvMassPerCell));
147  memset(&mReInvMassNonlin, 0, sizeof(mReInvMassNonlin));
148  memset(&mMiInvMassNonlin, 0, sizeof(mMiInvMassNonlin));
149  memset(&mTimeHGPerCell, 0, sizeof(mTimeHGPerCell));
150  memset(&mTimeLGPerCell, 0, sizeof(mTimeLGPerCell));
151  memset(&mTimeHGSlewing, 0, sizeof(mTimeHGSlewing));
152  memset(&mTimeLGSlewing, 0, sizeof(mTimeLGSlewing));
153  }

158  std::array<std::array<float, npt>, nMass> mReInvMassNonlin;
