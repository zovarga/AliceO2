32 void MFTAssessmentSpec::init(InitContext& ic)
33 {
34  mMFTAssessment = std::make_unique<o2::mft::MFTAssessment>(mUseMC);
35  auto filename = o2::base::NameConf::getGRPFileName();
36  const auto grp = o2::parameters::GRPObject::loadFrom(filename.c_str());
37  if (grp) {
38  o2::base::Propagator::initFieldFromGRP(grp);
39  auto field = static_cast<o2::field::MagneticField*>(TGeoGlobalMagField::Instance()->GetField());
40  double centerMFT[3] = {0, 0, -61.4}; // Field at center of MFT
41  auto Bz = field->getBz(centerMFT);
42  mMFTAssessment->setBz(Bz);
43  }
44 
45  for (int sw = 0; sw < NStopWatches; sw++) {
46  mTimer[sw].Stop();
47  mTimer[sw].Reset();
48  }
49 
50  mTimer[SWTot].Start(false);
51  mMFTAssessment->init(mFinalizeAnalysis);
52 }

111 {

119  inputs.emplace_back("tracks", "MFT", "TRACKS", 0, Lifetime::Timeframe);
120  inputs.emplace_back("trackClIdx", "MFT", "TRACKCLSID", 0, Lifetime::Timeframe);
121 
122  if (useMC) {
123  inputs.emplace_back("clslabels", "MFT", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
124  inputs.emplace_back("trklabels", "MFT", "TRACKSMCTR", 0, Lifetime::Timeframe);
125  }
126 
127  outputs.emplace_back("MFT", "MFTASSESSMENT", 0, Lifetime::Sporadic);
128 
129  return DataProcessorSpec{
130  "mft-assessment",
131  inputs,
132  outputs,
133  AlgorithmSpec{adaptFromTask<o2::mft::MFTAssessmentSpec>(useMC, processGen, finalizeAnalysis)},
134  Options{{}}};
