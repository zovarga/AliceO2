27 void TOFCalibInfoSlot::fill(const gsl::span<const o2::dataformats::CalibInfoTOF> data)
28 {
29  // fill container
30  // we do not apply any calibration at this stage, it will be applied when we
31  // process the data before filling the CCDB in the separate process
32 
33  // we first order the data that arrived, to improve speed when filling
34  int nd = data.size();
35  LOG(debug) << "entries in incoming data = " << nd;
36  std::vector<int> ord(nd);
37  std::iota(ord.begin(), ord.end(), 0);
38  std::sort(ord.begin(), ord.end(), [&data](int i, int j) { return data[i].getTOFChIndex() < data[j].getTOFChIndex(); });
39  int chPrev = 0, offsPrev = 0;
40  for (int i = 0; i < nd; i++) {
41  const auto& dti = data[ord[i]];
42  auto ch = dti.getTOFChIndex();
43  auto offset = offsPrev;
44  if (ch > chPrev) {
45  offset += std::accumulate(mEntriesSlot.begin() + chPrev, mEntriesSlot.begin() + ch, 0);
46  }
47  offsPrev = offset;
48  chPrev = ch;
49  mTOFCollectedCalibInfoSlot.emplace(mTOFCollectedCalibInfoSlot.begin() + offset, data[ord[i]].getTimestamp(), data[ord[i]].getDeltaTimePi(), data[ord[i]].getTot(), data[ord[i]].getFlags());
50  mEntriesSlot[ch]++;
51  }
52 }

99 void TOFCalibInfoSlot::merge(const TOFCalibInfoSlot* prev)
100 {
101  // merge data of 2 slots
102 
103  LOG(debug) << "Merging two slots with entries: current slot -> " << mTOFCollectedCalibInfoSlot.size() << " , previous slot -> " << prev->mTOFCollectedCalibInfoSlot.size();
104 
105  int offset = 0, offsetPrev = 0;
106  std::vector<o2::dataformats::CalibInfoTOFshort> tmpVector;
107  for (int ch = 0; ch < Geo::NCHANNELS; ch++) {
108  if (mEntriesSlot[ch] != 0) {
109  for (int i = offset; i < offset + mEntriesSlot[ch]; i++) {
110  tmpVector.emplace_back(mTOFCollectedCalibInfoSlot[i]);
111  }
112  offset += mEntriesSlot[ch];
113  }
114  if (prev->mEntriesSlot[ch] != 0) {
115  for (int i = offsetPrev; i < offsetPrev + prev->mEntriesSlot[ch]; i++) {
116  tmpVector.emplace_back(prev->mTOFCollectedCalibInfoSlot[i]);
117  }
118  offsetPrev += prev->mEntriesSlot[ch];
119  mEntriesSlot[ch] += prev->mEntriesSlot[ch];
120  }
121  }
122  mTOFCollectedCalibInfoSlot.swap(tmpVector);
123  LOG(debug) << "After merging the size is " << mTOFCollectedCalibInfoSlot.size();
124  return;
125 }

168 bool TOFCalibCollector::hasEnoughData(const Slot& slot) const
169 {
170 
171  // We define that we have enough data if the tree is big enough.
172  // each CalibInfoTOFShort is composed of one int, two floats, one unsigned char --> 13 bytes
173  // E.g. supposing that we have 256 entries per channel (which is an upper limit ) --> ~523 MB
174  // we can check if we have at least 1 GB of data --> 500*o2::tof::Geo::NCHANNELS entries in the vector
175  // (see header file for the fact that mMaxNumOfHits = 500)
176  // The case in which mScaleMaxNumOfHits = false allows for a fast check
177 
178  if (mTest) {
179  return true;
180  }
181  const o2::tof::TOFCalibInfoSlot* c = slot.getContainer();
182  LOG(info) << "we have " << c->getCollectedCalibInfoSlot().size() << " entries";
183  int maxNumberOfHits = mAbsMaxNumOfHits ? mMaxNumOfHits : mMaxNumOfHits * o2::tof::Geo::NCHANNELS;
184  if (mTFsendingPolicy || c->getCollectedCalibInfoSlot().size() > maxNumberOfHits) {
185  return true;
186  }
187  return false;
188 }
