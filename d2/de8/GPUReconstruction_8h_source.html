424 template <class T>
425 inline T* GPUReconstruction::AllocateIOMemoryHelper(size_t n, const T*& ptr, std::unique_ptr<T[]>& u)
426 {
427  if (n == 0) {
428  u.reset(nullptr);
429  return nullptr;
430  }
431  T* retVal;
432  if (mInputControl.useExternal()) {
433  u.reset(nullptr);
434  mInputControl.checkCurrent();
435  GPUProcessor::computePointerWithAlignment(mInputControl.ptrCurrent, retVal, n);
436  if ((size_t)((char*)mInputControl.ptrCurrent - (char*)mInputControl.ptrBase) > mInputControl.size) {
437  throw std::bad_alloc();
438  }
439  } else {
440  u.reset(new T[n]);
441  retVal = u.get();
442  if (mProcessingSettings.registerStandaloneInputMemory) {
443  if (registerMemoryForGPU(u.get(), n * sizeof(T))) {
444  GPUError("Error registering memory for GPU: %p - %lld bytes\n", (void*)u.get(), (long long int)(n * sizeof(T)));
445  throw std::bad_alloc();
446  }
447  }
448  }
449  ptr = retVal;
450  return retVal;
451 }

460 template <class T>
461 inline short GPUReconstruction::RegisterMemoryAllocation(T* proc, void* (T::*setPtr)(void*), int type, const char* name, const GPUMemoryReuse& re)
462 {
463  if (!(type & (GPUMemoryResource::MEMORY_HOST | GPUMemoryResource::MEMORY_GPU))) {
464  if ((type & GPUMemoryResource::MEMORY_SCRATCH) && !mProcessingSettings.keepDisplayMemory) { // keepAllMemory --> keepDisplayMemory
465  type |= (proc->mGPUProcessorType == GPUProcessor::PROCESSOR_TYPE_CPU ? GPUMemoryResource::MEMORY_HOST : GPUMemoryResource::MEMORY_GPU);
466  } else {
467  type |= GPUMemoryResource::MEMORY_HOST | GPUMemoryResource::MEMORY_GPU;
468  }
469  }
470  if (proc->mGPUProcessorType == GPUProcessor::PROCESSOR_TYPE_CPU) {
471  type &= ~GPUMemoryResource::MEMORY_GPU;
472  }
473  mMemoryResources.emplace_back(proc, static_cast<void* (GPUProcessor::*)(void*)>(setPtr), (GPUMemoryResource::MemoryType)type, name);
474  if (mMemoryResources.size() >= 32768) {
475  throw std::bad_alloc();
476  }
477  unsigned short retVal = mMemoryResources.size() - 1;
478  if (re.type != GPUMemoryReuse::NONE && !mProcessingSettings.disableMemoryReuse) {
479  const auto& it = mMemoryReuse1to1.find(re.id);
480  if (it == mMemoryReuse1to1.end()) {
481  mMemoryReuse1to1[re.id] = {proc, retVal};
482  } else {
483  mMemoryResources[retVal].mReuse = it->second.res[0];
484  it->second.res.emplace_back(retVal);
485  }
486  }
487  return retVal;
488 }

516 template <class T, class S>
517 inline unsigned int GPUReconstruction::DumpData(FILE* fp, const T* const* entries, const S* num, InOutPointerType type)
518 {
519  int count = getNIOTypeMultiplicity(type);
520  unsigned int numTotal = 0;
521  for (int i = 0; i < count; i++) {
522  numTotal += num[i];
523  }
524  if (numTotal == 0) {
525  return 0;
526  }
527  fwrite(&type, sizeof(type), 1, fp);
528  for (int i = 0; i < count; i++) {
