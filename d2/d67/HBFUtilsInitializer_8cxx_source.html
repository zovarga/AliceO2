42 HBFUtilsInitializer::HBFUtilsInitializer(const o2f::ConfigContext& configcontext, o2f::WorkflowSpec& wf)
43 {
44  auto updateHBFUtils = [&configcontext]() -> std::string {
45  static bool done = false;
46  static std::string confTFInfo{};
47  if (!done) {
48  bool helpasked = configcontext.helpOnCommandLine(); // if help is asked, don't take for granted that the ini file is there, don't produce an error if it is not!
49  auto conf = configcontext.options().isSet(HBFConfOpt) ? configcontext.options().get<std::string>(HBFConfOpt) : "";
50  if (!conf.empty()) {
51  auto opt = getOptType(conf);
52  if ((opt == HBFOpt::INI || opt == HBFOpt::JSON) && (!(helpasked && !o2::utils::Str::pathExists(conf)))) {
53  o2::conf::ConfigurableParam::updateFromFile(conf, "HBFUtils", true); // update only those values which were not touched yet (provenance == kCODE)
54  const auto& hbfu = o2::raw::HBFUtils::Instance();
55  hbfu.checkConsistency();
56  } else if (opt == HBFOpt::ROOT) {
57  confTFInfo = conf;
58  }
59  }
60  done = true;
61  }
62  return confTFInfo;
63  };
64 
65  const auto& hbfu = o2::raw::HBFUtils::Instance();
66  for (auto& spec : wf) {
67  if (spec.inputs.empty()) {
68  auto conf = updateHBFUtils();
69  o2f::ConfigParamsHelper::addOptionIfMissing(spec.options, o2f::ConfigParamSpec{HBFTFInfoOpt, o2f::VariantType::String, conf, {"root file with per-TF info"}});
70  }
71  }
72 }

75 HBFUtilsInitializer::HBFOpt HBFUtilsInitializer::getOptType(const std::string& optString)
76 {
77  // return type of the file provided via HBFConfOpt
78  HBFOpt opt = HBFOpt::NONE;
79  if (!optString.empty()) {
80  if (o2::utils::Str::endsWith(optString, ".ini")) {
81  opt = HBFOpt::INI;
82  } else if (o2::utils::Str::endsWith(optString, ".json")) {
83  opt = HBFOpt::JSON;
84  } else if (o2::utils::Str::endsWith(optString, ".root")) {
85  opt = HBFOpt::ROOT;
86  } else if (optString != "none") {
87  throw std::runtime_error(fmt::format("invalid option {} for {}", optString, HBFConfOpt));
88  }
89  }
90  return opt;
91 }

117 void HBFUtilsInitializer::addNewTimeSliceCallback(std::vector<o2::framework::CallbacksPolicy>& policies)
118 {
119  policies.push_back(o2::framework::CallbacksPolicy{
120  [](o2::framework::DeviceSpec const& spec, o2::framework::ConfigContext const& context) -> bool {
121  return (!context.helpOnCommandLine() && o2f::ConfigParamsHelper::hasOption(spec.options, HBFTFInfoOpt));
122  },
123  [](o2::framework::CallbackService& service, o2::framework::InitContext& context) {
124  auto fname = context.options().get<std::string>(HBFTFInfoOpt);
125  if (!fname.empty()) {
126  if (!o2::utils::Str::pathExists(fname)) {
127  throw std::runtime_error(fmt::format("file {} does not exist", fname));
128  }
129  service.set(o2::framework::CallbackService::Id::NewTimeslice,
130  [tfidinfo = readTFIDInfoVector(fname)](o2::header::DataHeader& dh, o2::framework::DataProcessingHeader&) { assignDataHeader(tfidinfo, dh); });
131  } else { // simple linear enumeration from already updated HBFUtils
132  const auto& hbfu = o2::raw::HBFUtils::Instance();
133  service.set(o2::framework::CallbackService::Id::NewTimeslice,
134  [offset = int64_t(hbfu.getFirstIRofTF({0, hbfu.orbitFirstSampled}).orbit), increment = int64_t(hbfu.nHBFPerTF),
135  startTime = hbfu.startTime, orbitFirst = hbfu.orbitFirst](o2::header::DataHeader& dh, o2::framework::DataProcessingHeader& dph) {
136  dh.firstTForbit = offset + increment * dh.tfCounter;
137  dph.creation = startTime + (dh.firstTForbit - orbitFirst) * o2::constants::lhc::LHCOrbitMUS * 1.e-3;
138  });
139  }
140  }});
141 }
