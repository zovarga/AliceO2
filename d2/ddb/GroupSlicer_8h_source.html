47  template <typename Z>
48  std::string getLabelFromType()
49  {
50  if constexpr (soa::is_soa_index_table_t<std::decay_t<Z>>::value) {
51  using T = typename std::decay_t<Z>::first_t;
52  if constexpr (soa::is_type_with_originals_v<std::decay_t<T>>) {
53  using O = typename framework::pack_element_t<0, typename std::decay_t<Z>::originals>;
54  using groupingMetadata = typename aod::MetadataTrait<O>::metadata;
55  return groupingMetadata::tableLabel();
56  } else {
57  using groupingMetadata = typename aod::MetadataTrait<T>::metadata;
58  return groupingMetadata::tableLabel();
59  }
60  } else if constexpr (soa::is_type_with_originals_v<std::decay_t<Z>>) {

65  using groupingMetadata = typename aod::MetadataTrait<std::decay_t<Z>>::metadata;

70  template <typename T>
71  auto splittingFunction(T&& table)
72  {
73  constexpr auto index = framework::has_type_at_v<std::decay_t<T>>(associated_pack_t{});
74  if constexpr (relatedByIndex<std::decay_t<G>, std::decay_t<T>>()) {
75  auto name = getLabelFromType<std::decay_t<T>>();
76  if constexpr (!framework::is_specialization_v<std::decay_t<T>, soa::SmallGroups>) {
77  if (table.size() == 0) {
78  return;
79  }
80  // use presorted splitting approach
81  auto result = o2::framework::sliceByColumn(mIndexColumnName.c_str(),
82  name.c_str(),
83  table.asArrowTable(),
84  static_cast<int32_t>(mGt->tableSize()),
85  &groups[index],
86  &offsets[index],
87  &sizes[index]);
88  if (result.ok() == false) {
89  throw runtime_error("Cannot split collection");
90  }
91  if (groups[index].size() > mGt->tableSize()) {
92  throw runtime_error_f("Splitting collection %s resulted in a larger group number (%d) than there is rows in the grouping table (%d).", name.c_str(), groups[index].size(), mGt->tableSize());
93  };
94  } else {
95  if (table.tableSize() == 0) {
96  return;
97  }
98  // use generic splitting approach
99  o2::framework::sliceByColumnGeneric(mIndexColumnName.c_str(),
100  name.c_str(),
101  table.asArrowTable(),
102  static_cast<int32_t>(mGt->tableSize()),
103  &filterGroups[index]);
104  }
105  }
106  }

108  template <typename T>
109  auto extractingFunction(T&& table)
110  {
111  if constexpr (soa::is_soa_filtered_t<std::decay_t<T>>::value) {
112  constexpr auto index = framework::has_type_at_v<std::decay_t<T>>(associated_pack_t{});
113  selections[index] = &table.getSelectedRows();
114  starts[index] = selections[index]->begin();
115  offsets[index].push_back(table.tableSize());
116  }
117  }

146  template <typename B, typename C>
147  constexpr static bool isIndexTo()
148  {
149  if constexpr (soa::is_type_with_binding_v<C>) {
150  if constexpr (soa::is_soa_index_table_t<std::decay_t<B>>::value) {
151  using T = typename std::decay_t<B>::first_t;
152  if constexpr (soa::is_type_with_originals_v<std::decay_t<T>>) {
153  using TT = typename framework::pack_element_t<0, typename std::decay_t<T>::originals>;
154  return std::is_same_v<typename C::binding_t, TT>;
155  } else {
156  using TT = std::decay_t<T>;
157  return std::is_same_v<typename C::binding_t, TT>;
158  }
159  } else {
160  if constexpr (soa::is_type_with_originals_v<std::decay_t<B>>) {
161  using TT = typename framework::pack_element_t<0, typename std::decay_t<B>::originals>;
162  return std::is_same_v<typename C::binding_t, TT>;
163  } else {
164  using TT = std::decay_t<B>;
165  return std::is_same_v<typename C::binding_t, TT>;
166  }
167  }
168  }
169  return false;
170  }

172  template <typename B, typename C>

174  {
175  if constexpr (soa::is_type_with_binding_v<C>) {

190  using TT = std::decay_t<B>;

269  template <typename A1>
270  auto prepareArgument()
271  {
272  constexpr auto index = framework::has_type_at_v<A1>(associated_pack_t{});
273  auto& originalTable = std::get<A1>(*mAt);
274 
275  if constexpr (relatedByIndex<std::decay_t<G>, std::decay_t<A1>>()) {
276  uint64_t pos;
277  if constexpr (soa::is_soa_filtered_t<std::decay_t<G>>::value) {
278  pos = groupSelection[position];
279  } else {
280  pos = position;
281  }
282  if constexpr (!framework::is_specialization_v<std::decay_t<A1>, soa::SmallGroups>) {
283  if (originalTable.size() == 0) {
284  return originalTable;
285  }
286  // optimized split
287  if constexpr (soa::is_soa_filtered_t<std::decay_t<A1>>::value) {
288  auto groupedElementsTable = arrow::util::get<std::shared_ptr<arrow::Table>>(((groups[index])[pos]).value);
289 
290  // for each grouping element we need to slice the selection vector
291  auto start_iterator = std::lower_bound(starts[index], selections[index]->end(), (offsets[index])[pos]);
292  auto stop_iterator = std::lower_bound(start_iterator, selections[index]->end(), (offsets[index])[pos] + (sizes[index])[pos]);
293  starts[index] = stop_iterator;
294  soa::SelectionVector slicedSelection{start_iterator, stop_iterator};
295  std::transform(slicedSelection.begin(), slicedSelection.end(), slicedSelection.begin(),
296  [&](int64_t idx) {
297  return idx - static_cast<int64_t>((offsets[index])[pos]);
298  });
299 
300  std::decay_t<A1> typedTable{{groupedElementsTable}, std::move(slicedSelection), (offsets[index])[pos]};
301  typedTable.bindInternalIndicesTo(&originalTable);
302  return typedTable;
303  } else {
304  auto groupedElementsTable = arrow::util::get<std::shared_ptr<arrow::Table>>(((groups[index])[pos]).value);
305  std::decay_t<A1> typedTable{{groupedElementsTable}, (offsets[index])[pos]};
306  typedTable.bindInternalIndicesTo(&originalTable);
307  return typedTable;
308  }
309  } else {
310  //generic split
311  if constexpr (soa::is_soa_filtered_t<std::decay_t<A1>>::value) {
312  if (originalTable.tableSize() == 0) {
313  return originalTable;
314  }
315  // intersect selections
316  o2::soa::SelectionVector s;
317  if (selections[index]->empty()) {
318  std::copy((filterGroups[index])[pos].begin(), (filterGroups[index])[pos].end(), std::back_inserter(s));
319  } else {
320  std::set_intersection((filterGroups[index])[pos].begin(), (filterGroups[index])[pos].end(), selections[index]->begin(), selections[index]->end(), std::back_inserter(s));
321  }
322  std::decay_t<A1> typedTable{{originalTable.asArrowTable()}, std::move(s)};
323  typedTable.bindInternalIndicesTo(&originalTable);
324  return typedTable;
325  } else {
326  throw runtime_error("Unsorted grouped table needs to be used with soa::SmallGroups<>");
327  }
328  }
329  } else {
330  return std::get<A1>(*mAt);
331  }
332  }
333 
334  std::string mIndexColumnName;
335  G const* mGt;
336  std::tuple<A...>* mAt;
337  typename grouping_t::iterator mGroupingElement;
338  uint64_t position = 0;
339  gsl::span<int64_t const> groupSelection;
340  std::array<std::vector<arrow::Datum>, sizeof...(A)> groups;
