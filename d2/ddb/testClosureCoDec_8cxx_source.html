109 // Create a vector of SampaCluster from a string d
110 // where d is of the form ts-#-bc-#-cs-#-q-# or
111 // ts-#-bc-#-cs-#-q-#-#-# ...
112 // d is expected to be a valid SampaCluster string representation
113 // see the asString function in SampaCluster
114 std::vector<SampaCluster> getSampaClusters(const std::string& d)
115 {
116  std::vector<SampaCluster> clusters;
117 
118  //std::cout << "d: " << d << std::endl;
119 
120  auto index = d.find("ts-");
121  auto ts = std::stoi(d.substr(index + 3));
122  index = d.find("bc-");
123  auto bc = std::stoi(d.substr(index + 3));
124  index = d.find("cs-");
125  auto cs = std::stoi(d.substr(index + 3));
126  index = d.find("q-");
127  auto q = d.substr(index + 2);
128  index = q.find("-");
129  if (index != std::string::npos) {
130  std::vector<uint10_t> charges;
131  std::istringstream ss(q);
132  std::string adc;
133  while (std::getline(ss, adc, '-')) {
134  charges.emplace_back(std::stoi(adc) & 0x3FF);
135  }
136  clusters.emplace_back(SampaCluster(ts, bc, charges));
137  } else {
138  clusters.emplace_back(SampaCluster(ts, bc, std::stoi(q), cs));
139  }
140  return clusters;
141 }

157 // create a raw data buffer from a list of strings
158 // where each string is of the form
159 // S#-J#-DS#-CH#-ts-#-bc-#-q-#
160 //
161 // it's a two steps process :
162 //
163 // - first create a buffer of payloads using a PayloadEncoder
164 // - then create the raw data itself (with proper RDHs) from the payload buffer
165 //
166 template <typename ELECMAP, typename FORMAT, typename CHARGESUM, int VERSION>
167 std::vector<std::byte> createBuffer(gsl::span<std::string> data,
168  uint32_t orbit = 12345, uint16_t bc = 678)
169 {
170  const o2::raw::HBFUtils& hbfutils = o2::raw::HBFUtils::Instance();
171  o2::conf::ConfigurableParam::setValue<uint32_t>("HBFUtils", "orbitFirst", orbit);
172 
173  auto encoder = createPayloadEncoder(createSolar2FeeLinkMapper<ELECMAP>(),
174  isUserLogicFormat<FORMAT>::value,
175  VERSION,
176  isChargeSumMode<CHARGESUM>::value);
177  encoder->startHeartbeatFrame(orbit, bc);
178  std::set<DsElecId> dsElecIds = getDs(data);
179  encoder->addHeartbeatHeaders(dsElecIds);
180  for (auto d : data) {
181  auto dsElecId = decodeDsElecId(d);
182  if (!dsElecId) {
183  std::cout << "Could not get dsElecId for " << d << "\n";
184  }
185  auto channel = decodeChannelId(d);
186  if (!channel) {
187  std::cout << "Could not get channel for " << d << "\n";
188  }
189  auto sampaClusters = getSampaClusters(d);
190  encoder->addChannelData(dsElecId.value(), channel.value(), sampaClusters);
191  }
192  std::vector<std::byte> buffer;
193  encoder->moveToBuffer(buffer);
194 
195  o2::conf::ConfigurableParam::setValue<uint32_t>("HBFUtils", "orbitFirst", orbit);
196  o2::conf::ConfigurableParam::setValue<uint32_t>("HBFUtils", "orbitFirstSampled", orbit);
197 
198  std::vector<std::byte> out =
199  paginate<ELECMAP, FORMAT, CHARGESUM, VERSION>(buffer,
200  fmt::format("mch-closure-codec-{}-{}.raw",
201  orbit, bc));
202  return out;
203 }
