60 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
61 {
62  WorkflowSpec specs;
63 
64  GID::mask_t allowedSourcesPV = GID::getSourcesMask("ITS,ITS-TPC,ITS-TPC-TRD,ITS-TPC-TOF");
65  GID::mask_t allowedSourcesVT = GID::getSourcesMask("ITS,MFT,TPC,ITS-TPC,MCH,MFT-MCH,TPC-TOF,TPC-TRD,ITS-TPC-TRD,ITS-TPC-TOF");
66 
67  // Update the (declared) parameters if changed from the command line
68  o2::conf::ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
69  // write the configuration used for the workflow
70  o2::conf::ConfigurableParam::writeINI("o2primary-vertexing-workflow_configuration.ini");
71 
72  auto useMC = !configcontext.options().get<bool>("disable-mc");
73  auto disableRootOut = configcontext.options().get<bool>("disable-root-output");
74  auto validateWithFT0 = configcontext.options().get<bool>("validate-with-ft0");
75 
76  GID::mask_t srcPV = allowedSourcesPV & GID::getSourcesMask(configcontext.options().get<std::string>("vertexing-sources"));
77  GID::mask_t srcVT = allowedSourcesVT & GID::getSourcesMask(configcontext.options().get<std::string>("vertex-track-matching-sources"));
78  if (validateWithFT0) {
79  srcPV |= GID::getSourceMask(GID::FT0);
80  }
81  GID::mask_t srcComb = srcPV | srcVT;
82  GID::mask_t dummy, srcClus = GID::includesDet(DetID::TOF, srcComb) ? GID::getSourceMask(GID::TOF) : dummy;
83 
84  specs.emplace_back(o2::vertexing::getPrimaryVertexingSpec(srcPV, validateWithFT0, useMC));
85  if (!srcVT.none()) {
86  specs.emplace_back(o2::vertexing::getVertexTrackMatcherSpec(srcVT));
87  }
88 
89  auto srcMtc = srcComb & ~GID::getSourceMask(GID::MFTMCH); // Do not request MFTMCH matches
90 
91  // only TOF clusters are needed if TOF is involved, no clusters MC needed
92  o2::globaltracking::InputHelper::addInputSpecs(configcontext, specs, srcClus, srcMtc, srcComb, useMC, dummy);
93 
94  if (!disableRootOut) {
95  specs.emplace_back(o2::vertexing::getPrimaryVertexWriterSpec(srcVT.none(), useMC));
96  }
97 
98  // configure dpl timer to inject correct firstTFOrbit: start from the 1st orbit of TF containing 1st sampled orbit
99  o2::raw::HBFUtilsInitializer hbfIni(configcontext, specs);
100 
101  return std::move(specs);
102 }
