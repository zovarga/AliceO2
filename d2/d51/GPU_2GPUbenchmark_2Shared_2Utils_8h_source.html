169 inline float computeThroughput(Test test, float result, float chunkSizeGB, int ntests)
170 {
171  // https://docs.nvidia.com/cuda/cuda-c-best-practices-guide/index.html
172  // Eff_bandwidth (GB/s) = (B_r + B_w) / (~1e9 * Time (s))
173 
174  return 1e3 * chunkSizeGB * (float)ntests / result;
175 }

177 template <class chunk_t>
178 inline size_t getBufferCapacity(float chunkSizeGB, int prime)
179 {
180  auto chunkCapacity = (static_cast<size_t>(GB * chunkSizeGB) & 0xFFFFFFFFFFFFF000) / sizeof(chunk_t);
181  if (!prime) {
182  return chunkCapacity;
183  } else {
184  return (chunkCapacity % prime == 0) ? (chunkCapacity - 0x1000) : chunkCapacity;
185  }
186 }

188 inline bool is_prime(const int n)
189 {
190  bool isPrime = true;
191  if (n == 0 || n == 1) {
192  isPrime = false;
193  } else {
194  for (int i = 2; i <= sqrt(n); ++i) {
195  if (n % i == 0) {
196  isPrime = false;
197  break;
198  }
199  }
200  }
201 
202  return isPrime;
203 }

285 inline void ResultWriter::addBenchmarkEntry(const std::string bName, const std::string type, const int nChunks)
286 {
287  mTimeTrees.emplace_back(new TTree((bName + "_" + type).data(), (bName + "_" + type).data()));
288  mTimeResults.clear();
289  mTimeResults.resize(nChunks);
290  mTimeTrees.back()->Branch("elapsed", &mTimeResults);
291 
292  mThroughputTrees.emplace_back(new TTree((bName + "_" + type + "_TP").data(), (bName + "_" + type + "_TP").data()));
293  mThroughputResults.clear();
