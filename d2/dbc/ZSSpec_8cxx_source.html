61 {
62  std::string processorName = "tpc-zsEncoder";

64  constexpr static size_t NEndpoints = o2::gpu::GPUTrackingInOutZS::NENDPOINTS;

76  auto initFunction = [tpcSectors, outRaw, tpcSectorMask](InitContext& ic) {
77  auto processAttributes = std::make_shared<ProcessAttributes>();
78  auto& zsoutput = processAttributes->zsoutput;
79  processAttributes->tpcSectors = tpcSectors;
80  auto& verify = processAttributes->verify;
81  auto& sizes = processAttributes->sizes;
82  auto& verbosity = processAttributes->verbosity;
83 
84  auto processingFct = [processAttributes, outRaw, tpcSectorMask](
85  ProcessingContext& pc) {
86  if (processAttributes->finished) {
87  return;
88  }
89 
90  auto& zsoutput = processAttributes->zsoutput;
91  auto& verify = processAttributes->verify;
92  auto& sizes = processAttributes->sizes;
93  auto& verbosity = processAttributes->verbosity;
94 
95  GPUParam _GPUParam;
96  GPUO2InterfaceConfiguration config;
97  config.configGRP.solenoidBz = 5.00668;
98  config.ReadConfigurableParam();
99  _GPUParam.SetDefaults(&config.configGRP, &config.configReconstruction, &config.configProcessing, nullptr);
100 
101  const auto& inputs = getWorkflowTPCInput(pc, 0, false, false, tpcSectorMask, true);
102  sizes.resize(NSectors * NEndpoints);
103  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(pc.inputs().getFirstValid(true));
104  o2::InteractionRecord ir{0, dh->firstTForbit};
105  o2::gpu::GPUReconstructionConvert::RunZSEncoder<o2::tpc::Digit, DigitArray>(inputs->inputDigits, &zsoutput, sizes.data(), nullptr, &ir, _GPUParam, true, verify, config.configReconstruction.tpc.zsThreshold);
106  ZeroSuppressedContainer8kb* page = reinterpret_cast<ZeroSuppressedContainer8kb*>(zsoutput.get());
107  unsigned int offset = 0;
108  for (unsigned int i = 0; i < NSectors; i++) {
109  unsigned int pageSector = 0;
110  for (unsigned int j = 0; j < NEndpoints; j++) {
111  if (sizes[i * NEndpoints + j] != 0) {
112  pageSector += sizes[i * NEndpoints + j];
113  }
114  }
115  offset += pageSector;
116  }
117  o2::tpc::TPCSectorHeader sh{0};
118  gsl::span<ZeroSuppressedContainer8kb> outp(&page[0], offset);
119  pc.outputs().snapshot(Output{gDataOriginTPC, "TPCZS", 0, Lifetime::Timeframe, sh}, outp);
120  pc.outputs().snapshot(Output{gDataOriginTPC, "ZSSIZES", 0, Lifetime::Timeframe, sh}, sizes);
121 
122  if (outRaw) {
123  // ===| set up raw writer |===================================================
124  std::string inputGRP = o2::base::NameConf::getGRPFileName();
125  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
126  o2::raw::RawFileWriter writer{"TPC"}; // to set the RDHv6.sourceID if V6 is used
127  writer.setContinuousReadout(grp->isDetContinuousReadOut(o2::detectors::DetID::TPC)); // must be set explicitly
128  uint32_t rdhV = o2::raw::RDHUtils::getVersion<o2::header::RAWDataHeader>();
129  writer.useRDHVersion(rdhV);
130  writer.doLazinessCheck(false); // LazinessCheck is not thread-safe
131  std::string outDir = "./";
132  const unsigned int defaultLink = rdh_utils::UserLogicLinkID;
133  enum LinksGrouping { All,
134  Sector,
135  Link };
136  auto useGrouping = Sector;
137 
138  for (unsigned int i = 0; i < NSectors; i++) {
139  for (unsigned int j = 0; j < NEndpoints; j++) {
140  const unsigned int cruInSector = j / 2;
141  const unsigned int cruID = i * 10 + cruInSector;
142  const rdh_utils::FEEIDType feeid = rdh_utils::getFEEID(cruID, j & 1, defaultLink);
143  std::string outfname;
144  if (useGrouping == LinksGrouping::All) { // single file for all links
145  outfname = fmt::format("{}tpc_all.raw", outDir);
146  } else if (useGrouping == LinksGrouping::Sector) { // file per sector
147  outfname = fmt::format("{}tpc_sector{}.raw", outDir, i);
148  } else if (useGrouping == LinksGrouping::Link) { // file per link
149  outfname = fmt::format("{}cru{}_{}.raw", outDir, cruID, j & 1);
150  }
151  writer.registerLink(feeid, cruID, defaultLink, j & 1, outfname);
152  }
153  }
154  if (useGrouping != LinksGrouping::Link) {
155  writer.useCaching();
156  }
157  ir = o2::raw::HBFUtils::Instance().getFirstSampledTFIR();
158  o2::gpu::GPUReconstructionConvert::RunZSEncoder<o2::tpc::Digit>(inputs->inputDigits, nullptr, nullptr, &writer, &ir, _GPUParam, true, false, config.configReconstruction.tpc.zsThreshold);
159  writer.writeConfFile("TPC", "RAWDATA", fmt::format("{}tpcraw.cfg", outDir));
160  }
161  zsoutput.reset(nullptr);
162  sizes = {};
163  };
164  return processingFct;
165  };

198 DataProcessorSpec getZStoDigitsSpec(std::vector<int> const& tpcSectors)
199 {
200  std::string processorName = "tpc-zs-to-Digits";
201  constexpr static size_t NSectors = o2::tpc::Sector::MAXSECTOR;
202  constexpr static size_t NEndpoints = o2::gpu::GPUTrackingInOutZS::NENDPOINTS;
203 
204  struct ProcessAttributes {
205  std::array<std::vector<Digit>, NSectors> outDigits;
206  std::unique_ptr<unsigned long long int[]> zsinput;
207  std::vector<unsigned int> sizes;
208  std::unique_ptr<o2::tpc::ZeroSuppress> decoder;
209  std::vector<int> tpcSectors;
210  bool verify = false;
211  int verbosity = 1;
212  bool finished = false;
213 
215  void sortDigits()
216  {
217  // sort digits
218  for (auto& digits : outDigits) {
219  std::sort(digits.begin(), digits.end(), [](const auto& a, const auto& b) {
220  if (a.getTimeStamp() < b.getTimeStamp()) {
221  return true;
222  }
223  if ((a.getTimeStamp() == b.getTimeStamp()) && (a.getRow() < b.getRow())) {
224  return true;
225  }
226  return false;
227  });
228  }
229  }
230  };

232  auto initFunction = [tpcSectors](InitContext& ic) {
233  auto processAttributes = std::make_shared<ProcessAttributes>();
234  processAttributes->tpcSectors = tpcSectors;
235  auto& outDigits = processAttributes->outDigits;
236  auto& decoder = processAttributes->decoder;
237  decoder = std::make_unique<o2::tpc::ZeroSuppress>();
238  auto& verbosity = processAttributes->verbosity;
239 
240  auto processingFct = [processAttributes](
241  ProcessingContext& pc) {
242  if (processAttributes->finished) {
243  return;
244  }
245  std::array<unsigned int, NEndpoints * NSectors> sizes;
246  gsl::span<const ZeroSuppressedContainer8kb> inputZS;
247  std::array<gsl::span<const ZeroSuppressedContainer8kb>, NSectors> inputZSSector;
248  auto& outDigits = processAttributes->outDigits;
249  auto& decoder = processAttributes->decoder;
250  auto& verbosity = processAttributes->verbosity;
251  unsigned int firstOrbit = 0;
252 
253  for (unsigned int i = 0; i < NSectors; i++) {
254  outDigits[i].clear();
255  }
256  o2::InteractionRecord ir = o2::raw::HBFUtils::Instance().getFirstSampledTFIR();
257  firstOrbit = ir.orbit;
258  std::vector<InputSpec> filter = {{"check", ConcreteDataTypeMatcher{gDataOriginTPC, "RAWDATA"}, Lifetime::Timeframe}};
259  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
260  const o2::header::DataHeader* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
261  const gsl::span<const char> raw = pc.inputs().get<gsl::span<char>>(ref);
262  o2::framework::RawParser parser(raw.data(), raw.size());
263 
264  const unsigned char* ptr = nullptr;
265  int cruID = 0;
266  rdh_utils::FEEIDType FEEID = -1;
267  size_t totalSize = 0;
268  for (auto it = parser.begin(); it != parser.end(); it++) {
269  const unsigned char* current = it.raw();
270  const o2::header::RAWDataHeader* rdh = (const o2::header::RAWDataHeader*)current;
271  if (it.size() == 0) {
272  ptr = nullptr;
273  continue;
274  } else {
275  ptr = current;
276  FEEID = o2::raw::RDHUtils::getFEEID(*rdh);
277  cruID = int(o2::raw::RDHUtils::getCRUID(*rdh));
278  unsigned int sector = cruID / 10;
279  gsl::span<const ZeroSuppressedContainer8kb> z0in(reinterpret_cast<const ZeroSuppressedContainer8kb*>(ptr), 1);
280  decoder->DecodeZSPages(&z0in, &outDigits[sector], firstOrbit);
281  }
282  }
283  }
284  for (int i = 0; i < NSectors; i++) {
285  LOG(info) << "digits in sector " << i << " : " << outDigits[i].size();
286  o2::tpc::TPCSectorHeader sh{i};
287  pc.outputs().snapshot(Output{gDataOriginTPC, "DIGITS", (unsigned int)i, Lifetime::Timeframe, sh}, outDigits[i]);
288  }
289  };
290  return processingFct;
291  };
292 
293  auto createInputSpecs = []() {
294  Inputs inputs;
295  inputs.emplace_back(InputSpec{"zsraw", ConcreteDataTypeMatcher{"TPC", "RAWDATA"}, Lifetime::Timeframe});
296 
297  return std::move(inputs);
298  };
299 
300  auto createOutputSpecs = []() {
301  std::vector<OutputSpec> outputSpecs = {};
302  OutputLabel label{"tpcdigits"};
303  constexpr o2::header::DataDescription datadesc("DIGITS");
304  for (int i = 0; i < NSectors; i++) {
305  outputSpecs.emplace_back(gDataOriginTPC, "DIGITS", i, Lifetime::Timeframe);
306  }
307  return std::move(outputSpecs);
308  };
309 
310  return DataProcessorSpec{"decode-zs-to-digits",
311  {createInputSpecs()},
312  {createOutputSpecs()},
313  AlgorithmSpec(initFunction)};
