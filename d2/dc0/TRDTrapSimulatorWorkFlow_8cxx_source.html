87 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
88 {
89  // Reserve one entry which fill be filled with the SimReaderSpec
90  // at the end. This places the processor at the beginning of the
91  // workflow in the upper left corner of the GUI.
92  //
93  using namespace o2::conf;
94  ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
95  WorkflowSpec specs;
96  auto useMC = !configcontext.options().get<bool>("disable-mc");
97  auto disableRootInput = configcontext.options().get<bool>("disable-root-input");
98  auto disableRootOutput = configcontext.options().get<bool>("disable-root-output");
99  if (!disableRootInput) {
100  specs.emplace_back(o2::trd::getTRDDigitReaderSpec(useMC));
101  }
102  specs.emplace_back(o2::trd::getTRDTrapSimulatorSpec(useMC));
103  if (!disableRootOutput) {
104  specs.emplace_back(o2::trd::getTRDTrackletWriterSpec(useMC));
105  }
106  return std::move(specs);
107 }
