40 void TrackReader::run(ProcessingContext& pc)
41 {
42  auto ent = mTree->GetReadEntry() + 1;
43  accumulate(ent, 1); // to really accumulate all, use accumulate(ent,mTree->GetEntries());
44  assert(ent < mTree->GetEntries()); // this should not happen
45  mTree->GetEntry(ent);
46 
47  pc.outputs().snapshot(Output{"TPC", "TRACKS", 0, Lifetime::Timeframe}, mTracksOut);
48  pc.outputs().snapshot(Output{"TPC", "CLUSREFS", 0, Lifetime::Timeframe}, mCluRefVecOut);

50  pc.outputs().snapshot(Output{"TPC", "TRACKSMCLBL", 0, Lifetime::Timeframe}, mMCTruthOut);

54  pc.services().get<ControlService>().endOfStream();
55  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);

59 void TrackReader::accumulate(int from, int n)
60 {
61  assert(from + n <= mTree->GetEntries());
62  if (n == 1) {
63  mTree->GetEntry(from);
64  mTracksOut.swap(*mTracksInp);
65  mCluRefVecOut.swap(*mCluRefVecInp);
66  if (mUseMC) {
67  std::copy(mMCTruthInp->begin(), mMCTruthInp->end(), std::back_inserter(mMCTruthOut));
68  }
69  } else {
70  for (int iev = 0; iev < n; iev++) {
71  mTree->GetEntry(from + iev);
72  //
73  uint32_t shift = mCluRefVecOut.size(); // during accumulation clusters refs need to be shifted
74 
75  auto cl0 = mCluRefVecInp->begin();
76  auto cl1 = mCluRefVecInp->end();
77  std::copy(cl0, cl1, std::back_inserter(mCluRefVecOut));
78 
79  auto tr0 = mTracksInp->begin();
80  auto tr1 = mTracksInp->end();
81  // fix cluster references
82  if (shift) {
83  for (auto tr = tr0; tr != tr1; tr++) {
84  tr->shiftFirstClusterRef(shift);
85  }
86  }
87  std::copy(tr0, tr1, std::back_inserter(mTracksOut));
88  // MC
89  if (mUseMC) {
90  std::copy(mMCTruthInp->begin(), mMCTruthInp->end(), std::back_inserter(mMCTruthOut));
91  }
92  }
93  }
94  LOG(info) << "TPCTrackReader pushes " << mTracksOut.size() << " tracks from entries " << from << " : " << from + n - 1;
95 }

97 void TrackReader::connectTree(const std::string& filename)
98 {
99  mTree.reset(nullptr); // in case it was already loaded
100  mFile.reset(TFile::Open(filename.c_str()));
101  if (!(mFile && !mFile->IsZombie())) {
102  throw std::runtime_error("Error opening tree file");
103  }
104  mTree.reset((TTree*)mFile->Get(mTrackTreeName.c_str()));
105  if (!mTree) {
106  throw std::runtime_error("Error opening tree");
107  }
108 
109  mTree->SetBranchAddress(mTrackBranchName.c_str(), &mTracksInp);
110  if (!mSkipClusRefs) {
111  mTree->SetBranchAddress(mClusRefBranchName.c_str(), &mCluRefVecInp);
112  } else {
113  mCluRefVecInp = new std::vector<o2::tpc::TPCClRefElem>;
114  }
115  if (mUseMC) {
116  if (mTree->GetBranch(mTrackMCTruthBranchName.c_str())) {
117  mTree->SetBranchAddress(mTrackMCTruthBranchName.c_str(), &mMCTruthInp);
118  LOG(info) << "Will use MC-truth from " << mTrackMCTruthBranchName;
119  } else {
120  LOG(info) << "MC-truth is missing";
121  mUseMC = false;
122  }
123  }
124  LOG(info) << "Loaded tree from " << filename << " with " << mTree->GetEntries() << " entries";
125 }
