60 void PadParameters<T>::init(int iDet)

62  auto nRows = HelperMethods::getStack(iDet) == 2 ? constants::NROWC0 : constants::NROWC1;

67 template <class T>
68 int PadParameters<T>::getChannel(int col, int row) const
69 {
70  if (mData.empty()) {
71  LOG(error) << "Pad parameters not initialized";
72  } else if (mData.size() == constants::NROWC0 * constants::NCOLUMN) {
73  return row + col * constants::NROWC0;
74  } else if (mData.size() == constants::NROWC1 * constants::NCOLUMN) {
75  return row + col * constants::NROWC1;
76  } else {
77  LOG(error) << "Wrong number of channels set: " << mData.size();
78  }
79  return -1;
80 }
