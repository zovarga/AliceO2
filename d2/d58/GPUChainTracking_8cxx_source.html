161 bool GPUChainTracking::ValidateSteps()
162 {
163  if ((GetRecoSteps() & GPUDataTypes::RecoStep::TPCdEdx) && !(GetRecoSteps() & GPUDataTypes::RecoStep::TPCMerging)) {
164  GPUError("Invalid Reconstruction Step Setting: dEdx requires TPC Merger to be active");
165  return false;
166  }
167  if ((GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCdEdx) && !(GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCMerging)) {
168  GPUError("Invalid GPU Reconstruction Step Setting: dEdx requires TPC Merger to be active");
169  return false;
170  }
171  if (!param().par.earlyTpcTransform) {
172  if (((GetRecoSteps() & GPUDataTypes::RecoStep::TPCSliceTracking) || (GetRecoSteps() & GPUDataTypes::RecoStep::TPCMerging)) && !(GetRecoSteps() & GPUDataTypes::RecoStep::TPCConversion)) {
173  GPUError("Invalid Reconstruction Step Setting: Tracking without early transform requires TPC Conversion to be active");
174  return false;
175  }
176  }
177  if ((GetRecoSteps() & GPUDataTypes::RecoStep::TPCClusterFinding) && !(GetRecoStepsInputs() & GPUDataTypes::InOutType::TPCRaw)) {
178  GPUError("Invalid input, TPC Clusterizer needs TPC raw input");
179  return false;
180  }
181  if (param().rec.tpc.mergerReadFromTrackerDirectly && (GetRecoSteps() & GPUDataTypes::RecoStep::TPCMerging) && ((GetRecoStepsInputs() & GPUDataTypes::InOutType::TPCSectorTracks) || (GetRecoStepsOutputs() & GPUDataTypes::InOutType::TPCSectorTracks) || !(GetRecoSteps() & GPUDataTypes::RecoStep::TPCConversion))) {
182  GPUError("Invalid input / output / step, mergerReadFromTrackerDirectly cannot read/store sectors tracks and needs TPC conversion");
183  return false;
184  }
185  if (!GetProcessingSettings().fullMergerOnGPU && (param().rec.tpc.mergerReadFromTrackerDirectly || GetProcessingSettings().createO2Output) && (GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCMerging)) {
186  GPUError("createO2Output and mergerReadFromTrackerDirectly works only in combination with fullMergerOnGPU if the merger is to run on GPU");
187  return false;
188  }
189  bool tpcClustersAvail = (GetRecoStepsInputs() & GPUDataTypes::InOutType::TPCClusters) || (GetRecoSteps() & GPUDataTypes::RecoStep::TPCClusterFinding) || (GetRecoSteps() & GPUDataTypes::RecoStep::TPCDecompression);
190 #ifndef GPUCA_ALIROOT_LIB
191  if ((GetRecoSteps() & GPUDataTypes::RecoStep::TPCMerging) && !tpcClustersAvail) {
192  GPUError("Invalid Inputs, TPC Clusters required");
193  return false;
194  }
195 #endif
196 #ifndef GPUCA_TPC_GEOMETRY_O2
197  if (GetRecoSteps() & GPUDataTypes::RecoStep::TPCClusterFinding) {
198  GPUError("Can not run TPC GPU Cluster Finding with Run 2 Data");
199  return false;
200  }
201 #endif
202  if (((GetRecoSteps() & GPUDataTypes::RecoStep::TPCConversion) || (GetRecoSteps() & GPUDataTypes::RecoStep::TPCSliceTracking) || (GetRecoSteps() & GPUDataTypes::RecoStep::TPCCompression) || (GetRecoSteps() & GPUDataTypes::RecoStep::TPCdEdx)) && !tpcClustersAvail) {
203  GPUError("Invalid Inputs, TPC Clusters required");
204  return false;
205  }
206  if ((GetRecoSteps() & GPUDataTypes::RecoStep::TPCMerging) && !((GetRecoStepsInputs() & GPUDataTypes::InOutType::TPCSectorTracks) || (GetRecoSteps() & GPUDataTypes::RecoStep::TPCSliceTracking))) {
207  GPUError("Input for TPC merger missing");
208  return false;
209  }
210  if ((GetRecoSteps() & GPUDataTypes::RecoStep::TPCCompression) && !((GetRecoStepsInputs() & GPUDataTypes::InOutType::TPCMergedTracks) || (GetRecoSteps() & GPUDataTypes::RecoStep::TPCMerging))) {
211  GPUError("Input for TPC compressor missing");
212  return false;
213  }
214  if ((GetRecoSteps() & GPUDataTypes::RecoStep::TRDTracking) && (!((GetRecoStepsInputs() & GPUDataTypes::InOutType::TPCMergedTracks) || (GetRecoSteps() & GPUDataTypes::RecoStep::TPCMerging)) || !(GetRecoStepsInputs() & GPUDataTypes::InOutType::TRDTracklets))) {
215  GPUError("Input for TRD Tracker missing");
216  return false;
217  }
218  if ((GetRecoStepsOutputs() & GPUDataTypes::InOutType::TPCRaw) || (GetRecoStepsOutputs() & GPUDataTypes::InOutType::TRDTracklets)) {
219  GPUError("TPC Raw / TPC Clusters / TRD Tracklets cannot be output");
220  return false;
221  }
222  if ((GetRecoStepsOutputs() & GPUDataTypes::InOutType::TPCSectorTracks) && !(GetRecoSteps() & GPUDataTypes::RecoStep::TPCSliceTracking)) {
223  GPUError("No TPC Slice Tracker Output available");
224  return false;
225  }
226  if ((GetRecoStepsOutputs() & GPUDataTypes::InOutType::TPCMergedTracks) && !(GetRecoSteps() & GPUDataTypes::RecoStep::TPCMerging)) {
227  GPUError("No TPC Merged Track Output available");
228  return false;
229  }
230  if ((GetRecoStepsOutputs() & GPUDataTypes::InOutType::TPCCompressedClusters) && !(GetRecoSteps() & GPUDataTypes::RecoStep::TPCCompression)) {
231  GPUError("No TPC Compression Output available");
232  return false;
233  }
234  if ((GetRecoStepsOutputs() & GPUDataTypes::InOutType::TRDTracks) && !(GetRecoSteps() & GPUDataTypes::RecoStep::TRDTracking)) {
235  GPUError("No TRD Tracker Output available");
236  return false;
237  }
238  if ((GetRecoSteps() & GPUDataTypes::RecoStep::TPCdEdx) && (processors()->calibObjects.dEdxCalibContainer == nullptr)) {
239  GPUError("Cannot run dE/dx without dE/dx calibration container object");
240  return false;
241  }
242  if ((GetRecoSteps() & GPUDataTypes::RecoStep::TPCClusterFinding) && processors()->calibObjects.tpcPadGain == nullptr) {
243  GPUError("Cannot run gain calibration without calibration object");
244  return false;
245  }
246  if ((GetRecoSteps() & GPUDataTypes::RecoStep::Refit) && !param().rec.trackingRefitGPUModel && (processors()->calibObjects.o2Propagator == nullptr || processors()->calibObjects.matLUT == nullptr)) {
247  GPUError("Cannot run refit with o2 track model without o2 propagator");
248  return false;
249  }
250  return true;

253 bool GPUChainTracking::ValidateSettings()
254 {
255  if ((param().rec.tpc.nWays & 1) == 0) {
256  GPUError("nWay setting musst be odd number!");
257  return false;
258  }
259  if (param().rec.tpc.mergerInterpolateErrors && param().rec.tpc.nWays == 1) {
260  GPUError("Cannot do error interpolation with NWays = 1!");
261  return false;
262  }
263  if ((param().rec.tpc.mergerReadFromTrackerDirectly || !param().par.earlyTpcTransform) && param().rec.nonConsecutiveIDs) {
264  GPUError("incompatible settings for non consecutive ids");
265  return false;
266  }
267  if (!param().rec.tpc.mergerReadFromTrackerDirectly && GetProcessingSettings().ompKernels) {
268  GPUError("OMP Kernels require mergerReadFromTrackerDirectly");
269  return false;
270  }
271  if (param().par.continuousMaxTimeBin > (int)GPUSettings::TPC_MAX_TF_TIME_BIN) {
272  GPUError("configure max time bin exceeds 256 orbits");
273  return false;
274  }
275  if (mRec->IsGPU() && std::max(GetProcessingSettings().nTPCClustererLanes + 1, GetProcessingSettings().nTPCClustererLanes * 2) + (GetProcessingSettings().doublePipeline ? 1 : 0) > mRec->NStreams()) {
276  GPUError("NStreams must be > nTPCClustererLanes");
277  return false;
278  }
279  if (GetProcessingSettings().noGPUMemoryRegistration && GetProcessingSettings().tpcCompressionGatherMode != 3) {
280  GPUError("noGPUMemoryRegistration only possible with gather mode 3");
281  return false;
282  }
283  if (GetProcessingSettings().doublePipeline) {
284  if (!GetRecoStepsOutputs().isOnlySet(GPUDataTypes::InOutType::TPCMergedTracks, GPUDataTypes::InOutType::TPCCompressedClusters, GPUDataTypes::InOutType::TPCClusters)) {
285  GPUError("Invalid outputs for double pipeline mode 0x%x", (unsigned int)GetRecoStepsOutputs());
286  return false;
287  }
288  if (((GetRecoStepsOutputs().isSet(GPUDataTypes::InOutType::TPCCompressedClusters) && mSubOutputControls[GPUTrackingOutputs::getIndex(&GPUTrackingOutputs::compressedClusters)] == nullptr) ||
289  (GetRecoStepsOutputs().isSet(GPUDataTypes::InOutType::TPCClusters) && mSubOutputControls[GPUTrackingOutputs::getIndex(&GPUTrackingOutputs::clustersNative)] == nullptr) ||
290  (GetRecoStepsOutputs().isSet(GPUDataTypes::InOutType::TPCMergedTracks) && mSubOutputControls[GPUTrackingOutputs::getIndex(&GPUTrackingOutputs::tpcTracks)] == nullptr) ||
291  (GetProcessingSettings().outputSharedClusterMap && mSubOutputControls[GPUTrackingOutputs::getIndex(&GPUTrackingOutputs::sharedClusterMap)] == nullptr))) {
292  GPUError("Must use external output for double pipeline mode");
293  return false;
294  }
295  if (ProcessingSettings().tpcCompressionGatherMode == 1) {
296  GPUError("Double pipeline incompatible to compression mode 1");
297  return false;
298  }
299  if (!(GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCCompression) || !(GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCClusterFinding) || param().rec.fwdTPCDigitsAsClusters) {
300  GPUError("Invalid reconstruction settings for double pipeline");
301  return false;
302  }
303  }
304  if ((GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCCompression) && !(GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCCompression) && (ProcessingSettings().tpcCompressionGatherMode == 1 || ProcessingSettings().tpcCompressionGatherMode == 3)) {
305  GPUError("Invalid tpcCompressionGatherMode for compression on CPU");
306  return false;
307  }
308  if (GetRecoSteps() & RecoStep::TRDTracking) {
309  if (GetProcessingSettings().trdTrackModelO2 && (GetProcessingSettings().createO2Output == 0 || param().rec.tpc.nWaysOuter == 0 || GetMatLUT() == nullptr)) {
310  GPUError("TRD tracking can only run on O2 TPC tracks if createO2Output is enabled, nWaysOuter is set, and matBudLUT is available");
311  return false;
312  }
313  if ((GetRecoStepsGPU() & RecoStep::TRDTracking) && !GetProcessingSettings().trdTrackModelO2 && GetProcessingSettings().createO2Output > 1) {
314  GPUError("TRD tracking can only run on GPU TPC tracks if the createO2Output setting does not suppress them");
315  return false;
316  }
317  }
318  return true;

379  mFlatObjectsShadow.mCalibObjects.matLUT->clearInternalBufferPtr();

381  mFlatObjectsShadow.mCalibObjects.matLUT->setFutureBufferAddress(mFlatObjectsDevice.mMatLUTBuffer);
382  }
383  if (processors()->calibObjects.trdGeometry) {
384  memcpy((void*)mFlatObjectsShadow.mCalibObjects.trdGeometry, (const void*)processors()->calibObjects.trdGeometry, sizeof(*processors()->calibObjects.trdGeometry));
385  mFlatObjectsShadow.mCalibObjects.trdGeometry->clearInternalBufferPtr();
386  }

388  memcpy((void*)mFlatObjectsShadow.mCalibObjects.tpcPadGain, (const void*)processors()->calibObjects.tpcPadGain, sizeof(*processors()->calibObjects.tpcPadGain));
389  }
390  if (processors()->calibObjects.o2Propagator) {
391  memcpy((void*)mFlatObjectsShadow.mCalibObjects.o2Propagator, (const void*)processors()->calibObjects.o2Propagator, sizeof(*processors()->calibObjects.o2Propagator));

394  mFlatObjectsShadow.mCalibObjects.o2Propagator->setMatLUT(mFlatObjectsShadow.mCalibObjects.matLUT);
395  }
396 #endif
397  TransferMemoryResourceLinkToGPU(RecoStep::NoRecoStep, mFlatObjectsShadow.mMemoryResFlat);

399  WriteToConstantMemory(RecoStep::NoRecoStep, (char*)&processors()->calibObjects - (char*)processors(), &mFlatObjectsDevice.mCalibObjects, sizeof(mFlatObjectsDevice.mCalibObjects), -1); // First initialization, for users not using RunChain

401  WriteToConstantMemory(RecoStep::NoRecoStep, (char*)&processors()->errorCodes - (char*)processors(), &processorsShadow()->errorCodes, sizeof(processorsShadow()->errorCodes), -1);
402  TransferMemoryResourceLinkToGPU(RecoStep::NoRecoStep, mInputsHost->mResourceErrorCodes);
403  }
404 
405  if (GetProcessingSettings().debugLevel >= 6) {
406  mDebugFile->open(mRec->IsGPU() ? "GPU.out" : "CPU.out");
407  }
408 
409  return 0;

537 int GPUChainTracking::RunChain()
538 {
539  if (GetProcessingSettings().ompAutoNThreads && !mRec->IsGPU()) {
540  mRec->SetNOMPThreads(-1);
541  }
542  const auto threadContext = GetThreadContext();
543  if (GetProcessingSettings().runCompressionStatistics && mCompressionStatistics == nullptr) {
544  mCompressionStatistics.reset(new GPUTPCClusterStatistics);
545  }
546  const bool needQA = GPUQA::QAAvailable() && (GetProcessingSettings().runQA || (GetProcessingSettings().eventDisplay && (mIOPtrs.nMCInfosTPC || GetProcessingSettings().runMC)));
547  if (needQA && mQA->IsInitialized() == false) {
548  if (mQA->InitQA(GetProcessingSettings().runQA ? -GetProcessingSettings().runQA : -1)) {
549  return 1;
550  }
551  }
552  if (GetProcessingSettings().debugLevel >= 6) {
553  *mDebugFile << "\n\nProcessing event " << mRec->getNEventsProcessed() << std::endl;
554  }
555  if (mRec->slavesExist() && mRec->IsGPU()) {
556  WriteToConstantMemory(RecoStep::NoRecoStep, (char*)&processors()->calibObjects - (char*)processors(), &mFlatObjectsDevice.mCalibObjects, sizeof(mFlatObjectsDevice.mCalibObjects), 0); // Reinitialize
557  }
558 
559  mRec->getGeneralStepTimer(GeneralStep::Prepare).Start();
560  try {
561  mRec->PrepareEvent();
562  } catch (const std::bad_alloc& e) {
563  GPUError("Memory Allocation Error");
564  return (1);
565  }
566  mRec->getGeneralStepTimer(GeneralStep::Prepare).Stop();
567 
568  PrepareDebugOutput();
569 
570  SynchronizeStream(0); // Synchronize all init copies that might be ongoing
571 
572  if (mIOPtrs.tpcCompressedClusters) {
573  if (runRecoStep(RecoStep::TPCDecompression, &GPUChainTracking::RunTPCDecompression)) {
574  return 1;
575  }
576  } else if (mIOPtrs.tpcPackedDigits || mIOPtrs.tpcZS) {
577  if (runRecoStep(RecoStep::TPCClusterFinding, &GPUChainTracking::RunTPCClusterizer, false)) {
578  return 1;
579  }
580  }
581 
582  if (GetProcessingSettings().ompAutoNThreads && !mRec->IsGPU() && mIOPtrs.clustersNative) {
583  mRec->SetNOMPThreads(mIOPtrs.clustersNative->nClustersTotal / 5000);
584  }
585 
586  if (mIOPtrs.clustersNative && runRecoStep(RecoStep::TPCConversion, &GPUChainTracking::ConvertNativeToClusterData)) {
587  return 1;
588  }
589 
590  mRec->PushNonPersistentMemory(qStr2Tag("TPCSLCD1")); // 1st stack level for TPC tracking slice data
591  mTPCSliceScratchOnStack = true;
592  if (runRecoStep(RecoStep::TPCSliceTracking, &GPUChainTracking::RunTPCTrackingSlices)) {
593  return 1;
594  }
595 
596  for (unsigned int i = 0; i < NSLICES; i++) {
597  // GPUInfo("slice %d clusters %d tracks %d", i, mClusterData[i].NumberOfClusters(), processors()->tpcTrackers[i].Output()->NTracks());
598  processors()->tpcMerger.SetSliceData(i, param().rec.tpc.mergerReadFromTrackerDirectly ? nullptr : processors()->tpcTrackers[i].Output());
599  }
600  if (runRecoStep(RecoStep::TPCMerging, &GPUChainTracking::RunTPCTrackingMerger, false)) {
601  return 1;
602  }
603  if (mTPCSliceScratchOnStack) {
604  mRec->PopNonPersistentMemory(RecoStep::TPCSliceTracking, qStr2Tag("TPCSLCD1")); // Release 1st stack level, TPC slice data not needed after merger
605  mTPCSliceScratchOnStack = false;
606  }
607 
608  if (mIOPtrs.clustersNative) {
609  if (GetProcessingSettings().doublePipeline) {
610  GPUChainTracking* foreignChain = (GPUChainTracking*)GetNextChainInQueue();
611  if (foreignChain && foreignChain->mIOPtrs.tpcZS) {
612  if (GetProcessingSettings().debugLevel >= 3) {
613  GPUInfo("Preempting tpcZS input of foreign chain");
614  }
615  mPipelineFinalizationCtx.reset(new GPUChainTrackingFinalContext);
616  mPipelineFinalizationCtx->rec = this->mRec;
617  foreignChain->mPipelineNotifyCtx = mPipelineFinalizationCtx.get();
618  }
619  }
620  if (runRecoStep(RecoStep::TPCCompression, &GPUChainTracking::RunTPCCompression)) {
621  return 1;
622  }
623  }
624 
625  if (GetProcessingSettings().trdTrackModelO2 ? runRecoStep(RecoStep::TRDTracking, &GPUChainTracking::RunTRDTracking<GPUTRDTrackerKernels::o2Version>) : runRecoStep(RecoStep::TRDTracking, &GPUChainTracking::RunTRDTracking<GPUTRDTrackerKernels::gpuVersion>)) {
626  return 1;
627  }
628 
629  if (runRecoStep(RecoStep::Refit, &GPUChainTracking::RunRefit)) {
630  return 1;
631  }
632 
633  if (!GetProcessingSettings().doublePipeline) { // Synchronize with output copies running asynchronously
634  SynchronizeStream(mRec->NStreams() - 2);
635  }
636 
637  if (GetProcessingSettings().ompAutoNThreads && !mRec->IsGPU()) {
638  mRec->SetNOMPThreads(-1);
639  }
640 
641  if (CheckErrorCodes()) {
642  return 3;
643  }
644 
645  return GetProcessingSettings().doublePipeline ? 0 : RunChainFinalize();
646 }
