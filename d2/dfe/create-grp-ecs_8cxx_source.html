26 void createGRPECSObject(const std::string& dataPeriod,
27  int run,
28  int runType,
29  int nHBPerTF,
30  const std::string& detsReadout,
31  const std::string& detsContinuousRO,
32  const std::string& detsTrigger,
33  long tstart,
34  long tend,
35  std::string ccdbServer = "")
36 {
37  auto detMask = o2::detectors::DetID::getMask(detsReadout);
38  if (detMask.count() == 0) {
39  throw std::runtime_error("empty detectors list is provided");
40  }
41  if (runType < 0 || runType >= int(GRPECSObject::RunType::NRUNTYPES)) {
42  LOGP(warning, "run type {} is not recognized, consider updating GRPECSObject.h", runType);
43  }
44  auto detMaskCont = detMask & o2::detectors::DetID::getMask(detsContinuousRO);
45  auto detMaskTrig = detMask & o2::detectors::DetID::getMask(detsTrigger);
46  LOG(info) << tstart << " " << tend;
47  if (tstart == 0) {
48  tstart = std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count();
49  }
50  long tendVal = 0;
51  if (tend < tstart) {
52  tendVal = tstart + 2 * 24 * 3600 * 1000UL; // assume that we will never have run longer than 2 days
53  LOG(info) << tstart << " -> " << tend;
54  } else if (tendVal < tend) {
55  tendVal = tend + 3600 * 1000UL; // we want the version with EOR to fully override the version w/o EOR
56  }
57  GRPECSObject grpecs;
58  grpecs.setTimeStart(tstart);
59  grpecs.setTimeEnd(tend);
60 
61  grpecs.setNHBFPerTF(nHBPerTF);
62  grpecs.setDetsReadOut(detMask);
63  grpecs.setDetsContinuousReadOut(detMaskCont);
64  grpecs.setDetsTrigger(detMaskTrig);
65  grpecs.setRun(run);
66  grpecs.setRunType((GRPECSObject::RunType)runType);
67  grpecs.setDataPeriod(dataPeriod);
68 
69  grpecs.print();
70 
71  if (!ccdbServer.empty()) {
72  CcdbApi api;
73  api.init(ccdbServer);
74  std::map<std::string, std::string> metadata;
75  metadata["responsible"] = "ECS";
76  metadata[o2::base::NameConf::CCDBRunTag.data()] = std::to_string(run);
77  // long ts = std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count();
78  api.storeAsTFileAny(&grpecs, "GLO/Config/GRPECS", metadata, tstart, tendVal); // making it 1-year valid to be sure we have something
79  LOGP(info, "Uploaded to {}/{} with validity {}:{}", ccdbServer, "GLO/Config/GRPECS", tstart, tendVal);
80  } else { // write a local file
81  auto fname = o2::base::NameConf::getGRPECSFileName();
82  TFile grpF(fname.c_str(), "recreate");
83  grpF.WriteObjectAny(&grpecs, grpecs.Class(), o2::base::NameConf::CCDBOBJECT.data());
84  LOG(info) << "Stored to local file " << fname;
85  }
86 }

92  "Create GRP-ECS object and upload to the CCDB\n"

100  try {
101  auto add_option = opt_general.add_options();
102  add_option("help,h", "Print this help message");
103  add_option("period,p", bpo::value<std::string>(), "data taking period");
104  add_option("run,r", bpo::value<int>(), "run number");
105  add_option("run-type,t", bpo::value<int>()->default_value(int(GRPECSObject::RunType::NONE)), "run type");
106  add_option("hbf-per-tf,n", bpo::value<int>()->default_value(128), "number of HBFs per TF");
107  add_option("detectors,d", bpo::value<string>()->default_value("all"), "comma separated list of detectors");
108  add_option("continuous,c", bpo::value<string>()->default_value("ITS,TPC,TOF,MFT,MCH,MID,ZDC,FT0,FV0,FDD,CTP"), "comma separated list of detectors in continuous readout mode");
109  add_option("triggering,g", bpo::value<string>()->default_value("FT0,FV0"), "comma separated list of detectors providing a trigger");
110  add_option("start-time,s", bpo::value<long>()->default_value(0), "run start time in ms, now() if 0");
111  add_option("end-time,e", bpo::value<long>()->default_value(0), "run end time in ms, start-time+3days is used if 0");
112  add_option("ccdb-server", bpo::value<std::string>()->default_value("http://alice-ccdb.cern.ch"), "CCDB server for upload, local file if empty");
113 
114  opt_all.add(opt_general).add(opt_hidden);
115  bpo::store(bpo::command_line_parser(argc, argv).options(opt_all).positional(opt_pos).run(), vm);
116 
117  if (vm.count("help")) {
118  std::cout << opt_general << std::endl;
119  exit(0);
120  }
121 
122  bpo::notify(vm);
123  } catch (bpo::error& e) {
124  std::cerr << "ERROR: " << e.what() << std::endl
125  << std::endl;
126  std::cerr << opt_general << std::endl;
127  exit(1);
128  } catch (std::exception& e) {
129  std::cerr << e.what() << ", application will now exit" << std::endl;
130  exit(2);
131  }
132  if (vm.count("run") == 0) {
133  std::cerr << "ERROR: "
134  << "obligator run number is missing" << std::endl;
135  std::cerr << opt_general << std::endl;
136  exit(3);
137  }
138  if (vm.count("period") == 0) {
139  std::cerr << "ERROR: "
140  << "obligator data taking period name is missing" << std::endl;
141  std::cerr << opt_general << std::endl;
142  exit(3);
143  }
