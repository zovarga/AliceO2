50 void RawToCellConverterSpec::init(framework::InitContext& ctx)
51 {
52  auto& ilctx = ctx.services().get<AliceO2::InfoLogger::InfoLoggerContext>();
53  ilctx.setField(AliceO2::InfoLogger::InfoLoggerContext::FieldName::Detector, "EMC");
54 
55  LOG(debug) << "[EMCALRawToCellConverter - init] Initialize converter ";
56  if (!mGeometry) {
57  mGeometry = Geometry::GetInstanceFromRunNumber(223409);
58  }
59  if (!mGeometry) {
60  LOG(error) << "Failure accessing geometry";
61  }
62 
63  if (!mMapper) {
64  mMapper = std::unique_ptr<MappingHandler>(new o2::emcal::MappingHandler);
65  }
66  if (!mMapper) {
67  LOG(error) << "Failed to initialize mapper";
68  }
69 
70  auto fitmethod = ctx.options().get<std::string>("fitmethod");
71  if (fitmethod == "standard") {
72  LOG(info) << "Using standard raw fitter";
73  mRawFitter = std::unique_ptr<CaloRawFitter>(new o2::emcal::CaloRawFitterStandard);
74  } else if (fitmethod == "gamma2") {
75  LOG(info) << "Using gamma2 raw fitter";
76  mRawFitter = std::unique_ptr<CaloRawFitter>(new o2::emcal::CaloRawFitterGamma2);
77  } else {
78  LOG(fatal) << "Unknown fit method" << fitmethod;
79  }
80 
81  mPrintTrailer = ctx.options().get<bool>("printtrailer");
82 
83  mMaxErrorMessages = ctx.options().get<int>("maxmessage");
84  LOG(info) << "Suppressing error messages after " << mMaxErrorMessages << " messages";
85 
86  mMergeLGHG = !ctx.options().get<bool>("no-mergeHGLG");
87  mDisablePedestalEvaluation = ctx.options().get<bool>("no-evalpedestal");
88 
89  LOG(info) << "Running gain merging mode: " << (mMergeLGHG ? "yes" : "no") << std::endl;
90 
91  mRawFitter->setAmpCut(mNoiseThreshold);
92  mRawFitter->setL1Phase(0.);
93 }

95 void RawToCellConverterSpec::run(framework::ProcessingContext& ctx)
96 {
97  LOG(debug) << "[EMCALRawToCellConverter - run] called";
98  const double CONVADCGEV = 0.016; // Conversion from ADC counts to energy: E = 16 MeV / ADC
99  constexpr double timeshift = 600.; // subtract 600 ns in order to center the time peak around the nominal delay
100  constexpr auto originEMC = o2::header::gDataOriginEMC;
101  constexpr auto descRaw = o2::header::gDataDescriptionRawData;
102 
103  // reset message counter after 10 minutes
104  auto currenttime = std::chrono::system_clock::now();
105  std::chrono::duration<double> interval = mReferenceTime - currenttime;
106  if (interval.count() > 600) {
107  // Resetting error messages after 10 minutes
108  mNumErrorMessages = 0;
109  mReferenceTime = currenttime;
110  }
111 
112  mOutputCells.clear();
113  mOutputTriggerRecords.clear();
114  mOutputDecoderErrors.clear();
115 
116  if (isLostTimeframe(ctx)) {
117  sendData(ctx, mOutputCells, mOutputTriggerRecords, mOutputDecoderErrors);
118  return;
119  }
120 
121  // Cache cells from for bunch crossings as the component reads timeframes from many links consecutively
122  std::map<o2::InteractionRecord, std::shared_ptr<std::vector<RecCellInfo>>> cellBuffer; // Internal cell buffer
123  std::map<o2::InteractionRecord, uint32_t> triggerBuffer;
124 
125  std::vector<framework::InputSpec> filter{{"filter", framework::ConcreteDataTypeMatcher(originEMC, descRaw)}};
126  int firstEntry = 0;
127  for (const auto& rawData : framework::InputRecordWalker(ctx.inputs(), filter)) {
128 
129  // Skip SOX headers
130  auto rdhblock = reinterpret_cast<const o2::header::RDHAny*>(rawData.payload);
131  if (o2::raw::RDHUtils::getHeaderSize(rdhblock) == static_cast<int>(o2::framework::DataRefUtils::getPayloadSize(rawData))) {
132  continue;
133  }
134 
135  o2::emcal::RawReaderMemory rawreader(framework::DataRefUtils::as<const char>(rawData));
136  rawreader.setRangeSRUDDLs(0, 39);
137 
138  // loop over all the DMA pages
139  while (rawreader.hasNext()) {
140 
141  try {
142  rawreader.next();
143  } catch (RawDecodingError& e) {
144  mOutputDecoderErrors.emplace_back(e.getFECID(), ErrorTypeFEE::ErrorSource_t::PAGE_ERROR, RawDecodingError::ErrorTypeToInt(e.getErrorType()));
145  if (mNumErrorMessages < mMaxErrorMessages) {
146  LOG(error) << " EMCAL raw task: " << e.what() << " in FEC " << e.getFECID() << std::endl;
147  mNumErrorMessages++;
148  if (mNumErrorMessages == mMaxErrorMessages) {
149  LOG(error) << "Max. amount of error messages (" << mMaxErrorMessages << " reached, further messages will be suppressed";
150  }
151  } else {
152  mErrorMessagesSuppressed++;
153  }
154  }
155 
156  auto& header = rawreader.getRawHeader();
157  auto triggerBC = raw::RDHUtils::getTriggerBC(header);
158  auto triggerOrbit = raw::RDHUtils::getTriggerOrbit(header);
159  auto feeID = raw::RDHUtils::getFEEID(header);
160  auto triggerbits = raw::RDHUtils::getTriggerType(header);
161 
162  o2::InteractionRecord currentIR(triggerBC, triggerOrbit);
163  std::shared_ptr<std::vector<RecCellInfo>> currentCellContainer;
164  auto found = cellBuffer.find(currentIR);
165  if (found == cellBuffer.end()) {
166  currentCellContainer = std::make_shared<std::vector<RecCellInfo>>();
167  cellBuffer[currentIR] = currentCellContainer;
168  // also add trigger bits
169  triggerBuffer[currentIR] = triggerbits;
170  } else {
171  currentCellContainer = found->second;
172  }
173 
174  if (feeID >= 40) {
175  continue; // skip STU ddl
176  }
177 
178  // std::cout<<rawreader.getRawHeader()<<std::endl;
179 
180  // use the altro decoder to decode the raw data, and extract the RCU trailer
181  AltroDecoder decoder(rawreader);
182  // check the words of the payload exception in altrodecoder
183  try {
184  decoder.decode();
185  } catch (AltroDecoderError& e) {
186  ErrorTypeFEE errornum(feeID, ErrorTypeFEE::ErrorSource_t::ALTRO_ERROR, AltroDecoderError::errorTypeToInt(e.getErrorType()));
187  if (mNumErrorMessages < mMaxErrorMessages) {
188  std::string errormessage;
189  using AltroErrType = AltroDecoderError::ErrorType_t;
190  switch (e.getErrorType()) {
191  case AltroErrType::RCU_TRAILER_ERROR:
192  errormessage = " RCU Trailer Error ";
193  break;
194  case AltroErrType::RCU_VERSION_ERROR:
195  errormessage = " RCU Version Error ";
196  break;
197  case AltroErrType::RCU_TRAILER_SIZE_ERROR:
198  errormessage = " RCU Trailer Size Error ";
199  break;
200  case AltroErrType::ALTRO_BUNCH_HEADER_ERROR:
201  errormessage = " ALTRO Bunch Header Error ";
202  break;
203  case AltroErrType::ALTRO_BUNCH_LENGTH_ERROR:
204  errormessage = " ALTRO Bunch Length Error ";
205  break;
206  case AltroErrType::ALTRO_PAYLOAD_ERROR:
207  errormessage = " ALTRO Payload Error ";
208  break;
209  case AltroErrType::ALTRO_MAPPING_ERROR:
210  errormessage = " ALTRO Mapping Error ";
211  break;
212  case AltroErrType::CHANNEL_ERROR:
213  errormessage = " Channel Error ";
214  break;
215  default:
216  break;
217  };
218  LOG(error) << " EMCAL raw task: " << errormessage << " in DDL " << feeID << std::endl;
219  mNumErrorMessages++;
220  if (mNumErrorMessages == mMaxErrorMessages) {
221  LOG(error) << "Max. amount of error messages (" << mMaxErrorMessages << " reached, further messages will be suppressed";
222  }
223  } else {
224  mErrorMessagesSuppressed++;
225  }
226  // fill histograms with error types
227  mOutputDecoderErrors.push_back(errornum);
228  continue;
229  }
230  for (auto minorerror : decoder.getMinorDecodingErrors()) {
231  if (mNumErrorMessages < mMaxErrorMessages) {
232  LOG(error) << " EMCAL raw task - Minor error in DDL " << feeID << ": " << minorerror.what() << std::endl;
233  mNumErrorMessages++;
234  if (mNumErrorMessages == mMaxErrorMessages) {
235  LOG(error) << "Max. amount of error messages (" << mMaxErrorMessages << " reached, further messages will be suppressed";
236  }
237  } else {
238  mErrorMessagesSuppressed++;
239  }
240  ErrorTypeFEE errornum(feeID, ErrorTypeFEE::ErrorSource_t::ALTRO_ERROR, MinorAltroDecodingError::errorTypeToInt(minorerror.getErrorType()));
241  mOutputDecoderErrors.push_back(errornum);
242  }
243 
244  if (mPrintTrailer) {
245  // Can become very verbose, therefore must be switched on explicitly in addition
246  // to high debug level
247  LOG(debug4) << decoder.getRCUTrailer();
248  }
249  // Apply zero suppression only in case it was enabled
250  if (decoder.getRCUTrailer().hasZeroSuppression()) {
251  LOG(debug3) << "Zero suppression enabled";
252  } else {
253  LOG(debug3) << "Zero suppression disabled";
254  }
255  if (mDisablePedestalEvaluation) {
256  // auto-disable pedestal evaluation in the raw fitter
257  // treat all channels as zero-suppressed independent of
258  // what is provided from the RCU trailer
259  mRawFitter->setIsZeroSuppressed(true);
260  } else {
261  mRawFitter->setIsZeroSuppressed(decoder.getRCUTrailer().hasZeroSuppression());
262  }
263 
264  const auto& map = mMapper->getMappingForDDL(feeID);
265  int iSM = feeID / 2;
266 
267  // Loop over all the channels
268  int nBunchesNotOK = 0;
269  for (auto& chan : decoder.getChannels()) {
270 
271  int iRow, iCol;
272  ChannelType_t chantype;
273  try {
274  iRow = map.getRow(chan.getHardwareAddress());
275  iCol = map.getColumn(chan.getHardwareAddress());
276  chantype = map.getChannelType(chan.getHardwareAddress());
277  } catch (Mapper::AddressNotFoundException& ex) {
278  if (mNumErrorMessages < mMaxErrorMessages) {
279  LOG(error) << "Mapping error DDL " << feeID << ": " << ex.what();
280  mNumErrorMessages++;
281  if (mNumErrorMessages == mMaxErrorMessages) {
282  LOG(error) << "Max. amount of error messages (" << mMaxErrorMessages << " reached, further messages will be suppressed";
283  }
284  } else {
285  mErrorMessagesSuppressed++;
286  }
287  continue;
288  }
289 
290  if (!(chantype == o2::emcal::ChannelType_t::HIGH_GAIN || chantype == o2::emcal::ChannelType_t::LOW_GAIN)) {
291  continue;
292  }
293 
294  auto [phishift, etashift] = mGeometry->ShiftOnlineToOfflineCellIndexes(iSM, iRow, iCol);
295  int CellID = mGeometry->GetAbsCellIdFromCellIndexes(iSM, phishift, etashift);
296  if (CellID > 17664) {
297  if (mNumErrorMessages < mMaxErrorMessages) {
298  std::string celltypename;
299  switch (chantype) {
300  case o2::emcal::ChannelType_t::HIGH_GAIN:
301  celltypename = "high gain";
302  break;
303  case o2::emcal::ChannelType_t::LOW_GAIN:
304  celltypename = "low-gain";
305  break;
306  case o2::emcal::ChannelType_t::TRU:
307  celltypename = "TRU";
308  break;
309  case o2::emcal::ChannelType_t::LEDMON:
310  celltypename = "LEDMON";
311  break;
312  };
313  LOG(error) << "Sending invalid cell ID " << CellID << "(SM " << iSM << ", row " << iRow << " - shift " << phishift << ", col " << iCol << " - shift " << etashift << ") of type " << celltypename;
314  mNumErrorMessages++;
315  if (mNumErrorMessages == mMaxErrorMessages) {
316  LOG(error) << "Max. amount of error messages (" << mMaxErrorMessages << " reached, further messages will be suppressed";
317  }
318  } else {
319  mErrorMessagesSuppressed++;
320  }
321  mOutputDecoderErrors.emplace_back(feeID, ErrorTypeFEE::ErrorSource_t::GEOMETRY_ERROR, 0); // 0 -> Cell ID out of range
322  continue;
323  }
324  if (CellID < 0) {
325  if (mNumErrorMessages < mMaxErrorMessages) {
326  std::string celltypename;
327  switch (chantype) {
328  case o2::emcal::ChannelType_t::HIGH_GAIN:
329  celltypename = "high gain";
330  break;
331  case o2::emcal::ChannelType_t::LOW_GAIN:
332  celltypename = "low-gain";
333  break;
334  case o2::emcal::ChannelType_t::TRU:
335  celltypename = "TRU";
336  break;
337  case o2::emcal::ChannelType_t::LEDMON:
338  celltypename = "LEDMON";
339  break;
340  };
341  LOG(error) << "Sending negative cell ID " << CellID << "(SM " << iSM << ", row " << iRow << " - shift " << phishift << ", col " << iCol << " - shift " << etashift << ") of type " << celltypename;
342  mNumErrorMessages++;
343  if (mNumErrorMessages == mMaxErrorMessages) {
344  LOG(error) << "Max. amount of error messages (" << mMaxErrorMessages << " reached, further messages will be suppressed";
345  }
346  } else {
347  mErrorMessagesSuppressed++;
348  }
349  mOutputDecoderErrors.emplace_back(feeID, ErrorTypeFEE::ErrorSource_t::GEOMETRY_ERROR, -1); // Geometry error codes will start from 100
350  continue;
351  }
352 
353  // define the conatiner for the fit results, and perform the raw fitting using the stadnard raw fitter
354  CaloFitResults fitResults;
355  try {
356  fitResults = mRawFitter->evaluate(chan.getBunches());
357  // Prevent negative entries - we should no longer get here as the raw fit usually will end in an error state
358  if (fitResults.getAmp() < 0) {
359  fitResults.setAmp(0.);
360  }
361  if (fitResults.getTime() < 0) {
362  fitResults.setTime(0.);
363  }
364  double amp = fitResults.getAmp() * CONVADCGEV;
365  if (mMergeLGHG) {
366  // Handling of HG/LG for ceratin cells
367  // Keep the high gain if it is below the threshold, otherwise
368  // change to the low gain
369  auto res = std::find_if(currentCellContainer->begin(), currentCellContainer->end(), [CellID](const RecCellInfo& test) { return test.mCellData.getTower() == CellID; });
370  if (res != currentCellContainer->end()) {
371  // Cell already existing, store LG if HG is larger then the overflow cut
372  if (chantype == o2::emcal::ChannelType_t::LOW_GAIN) {
373  res->mHWAddressLG = chan.getHardwareAddress();
374  res->mHGOutOfRange = false; // LG is found so it can replace the HG if the HG is out of range
375  if (res->mCellData.getHighGain()) {
376  double ampOld = res->mCellData.getEnergy() / CONVADCGEV; // cut applied on ADC and not on energy
377  if (ampOld > o2::emcal::constants::OVERFLOWCUT) {
378  // High gain digit has energy above overflow cut, use low gain instead
379  res->mCellData.setEnergy(amp * o2::emcal::constants::EMCAL_HGLGFACTOR);
380  res->mCellData.setTimeStamp(fitResults.getTime() - timeshift);
381  res->mCellData.setLowGain();
382  }
383  res->mIsLGnoHG = false;
384  }
385  } else {
386  // new channel would be HG use that if it is belpw ADC cut
387  // as the channel existed before it must have been a LG channel,
389  res->mIsLGnoHG = false;
390  res->mHGOutOfRange = false;
391  res->mHWAddressHG = chan.getHardwareAddress();
392  if (amp / CONVADCGEV <= o2::emcal::constants::OVERFLOWCUT) {
393  res->mCellData.setEnergy(amp);
394  res->mCellData.setTimeStamp(fitResults.getTime() - timeshift);
395  res->mCellData.setHighGain();
396  }
397  }
398  } else {
399  // New cell
400  bool lgNoHG = false; // Flag for filter of cells which have only low gain but no high gain
401  bool hgOutOfRange = false; // Flag if only a HG is present which is out-of-range
402  int hwAddressLG = -1, // Hardware address of the LG of the tower (for monitoring)
403  hwAddressHG = -1; // Hardware address of the HG of the tower (for monitoring)
404  auto flagChanType = chantype;
405  if (chantype == o2::emcal::ChannelType_t::LOW_GAIN) {
406  lgNoHG = true;
407  amp *= o2::emcal::constants::EMCAL_HGLGFACTOR;
408  hwAddressLG = chan.getHardwareAddress();
409  } else {
410  // High gain cell: Flag as low gain if above threshold
411  if (amp / CONVADCGEV > o2::emcal::constants::OVERFLOWCUT) {
412  flagChanType = ChannelType_t::LOW_GAIN;
413  hgOutOfRange = true;
414  }
415  hwAddressHG = chan.getHardwareAddress();
416  }
417  int fecID = mMapper->getFEEForChannelInDDL(feeID, chan.getFECIndex(), chan.getBranchIndex());
418  currentCellContainer->push_back({o2::emcal::Cell(CellID, amp, fitResults.getTime() - timeshift, chantype),
419  lgNoHG,
420  hgOutOfRange,
421  fecID, feeID, hwAddressLG, hwAddressHG});
422  }
423  } else {
424  // No merge of HG/LG cells (usually MC where either
425  // of the two is simulated)
426  int hwAddressLG = chantype == ChannelType_t::LOW_GAIN ? chan.getHardwareAddress() : -1,
427  hwAddressHG = chantype == ChannelType_t::HIGH_GAIN ? chan.getHardwareAddress() : -1;
428  // New cell
429  if (chantype == o2::emcal::ChannelType_t::LOW_GAIN) {
430  amp *= o2::emcal::constants::EMCAL_HGLGFACTOR;
431  }
432  int fecID = mMapper->getFEEForChannelInDDL(feeID, chan.getFECIndex(), chan.getBranchIndex());
433  currentCellContainer->push_back({o2::emcal::Cell(CellID, amp, fitResults.getTime() - timeshift, chantype),
434  false,
435  false,
436  fecID, feeID, hwAddressLG, hwAddressHG});
437  }
438  } catch (CaloRawFitter::RawFitterError_t& fiterror) {
439  if (fiterror != CaloRawFitter::RawFitterError_t::BUNCH_NOT_OK) {
440  // Display
441  if (mNumErrorMessages < mMaxErrorMessages) {
442  LOG(error) << "Failure in raw fitting: " << CaloRawFitter::createErrorMessage(fiterror);
443  mNumErrorMessages++;
444  if (mNumErrorMessages == mMaxErrorMessages) {
445  LOG(error) << "Max. amount of error messages (" << mMaxErrorMessages << " reached, further messages will be suppressed";
446  }
447  } else {
448  mErrorMessagesSuppressed++;
449  }
450  } else {
451  LOG(debug2) << "Failure in raw fitting: " << CaloRawFitter::createErrorMessage(fiterror);
452  nBunchesNotOK++;
453  }
454  mOutputDecoderErrors.emplace_back(feeID, ErrorTypeFEE::ErrorSource_t::FIT_ERROR, CaloRawFitter::getErrorNumber(fiterror));
455  }
456  }
457  if (nBunchesNotOK) {
458  LOG(debug) << "Number of failed bunches: " << nBunchesNotOK;
459  }
460  }
461  }
462 
463  // Loop over BCs, sort cells with increasing tower ID and write to output containers
464  for (const auto& [bc, cells] : cellBuffer) {
465  int ncellsEvent = 0;
466  int eventstart = mOutputCells.size();
467  if (cells->size()) {
468  LOG(debug) << "Event has " << cells->size() << " cells";
469  // Sort cells according to cell ID
470  std::sort(cells->begin(), cells->end(), [](const RecCellInfo& lhs, const RecCellInfo& rhs) { return lhs.mCellData.getTower() < rhs.mCellData.getTower(); });
471  for (const auto& cell : *cells) {
472  if (cell.mIsLGnoHG) {
473  if (mNumErrorMessages < mMaxErrorMessages) {
474  LOG(error) << "FEC " << cell.mFecID << ": 0x" << std::hex << cell.mHWAddressLG << std::dec << " (DDL " << cell.mDDLID << ") has low gain but no high-gain";
475  mNumErrorMessages++;
476  if (mNumErrorMessages == mMaxErrorMessages) {
477  LOG(error) << "Max. amount of error messages (" << mMaxErrorMessages << " reached, further messages will be suppressed";
478  }
479  } else {
480  mErrorMessagesSuppressed++;
481  }
482  mOutputDecoderErrors.emplace_back(cell.mFecID, ErrorTypeFEE::GAIN_ERROR, 0);
483  continue;
484  }
485  if (cell.mHGOutOfRange) {
486  if (mNumErrorMessages < mMaxErrorMessages) {
487  LOG(error) << "FEC " << cell.mFecID << ": 0x" << std::hex << cell.mHWAddressHG << std::dec << " (DDL " << cell.mDDLID << ") has only high-gain out-of-range";
488  mNumErrorMessages++;
489  if (mNumErrorMessages == mMaxErrorMessages) {
490  LOG(error) << "Max. amount of error messages (" << mMaxErrorMessages << " reached, further messages will be suppressed";
491  }
492  } else {
493  mErrorMessagesSuppressed++;
494  }
495  mOutputDecoderErrors.emplace_back(cell.mFecID, ErrorTypeFEE::GAIN_ERROR, 1);
496  continue;
497  }
498  ncellsEvent++;
499  mOutputCells.push_back(cell.mCellData);
500  }
501  LOG(debug) << "Next event [Orbit " << bc.orbit << ", BC (" << bc.bc << "]: Accepted " << ncellsEvent << " cells";
502  }
503  mOutputTriggerRecords.emplace_back(bc, triggerBuffer[bc], eventstart, ncellsEvent);
504  }
505 
506  LOG(info) << "[EMCALRawToCellConverter - run] Writing " << mOutputCells.size() << " cells from " << mOutputTriggerRecords.size() << " events ...";
507  sendData(ctx, mOutputCells, mOutputTriggerRecords, mOutputDecoderErrors);
508 }

510 bool RawToCellConverterSpec::isLostTimeframe(framework::ProcessingContext& ctx) const
511 {
512  constexpr auto originEMC = header::gDataOriginEMC;
513  o2::framework::InputSpec dummy{"dummy",
514  framework::ConcreteDataMatcher{originEMC,
515  header::gDataDescriptionRawData,
516  0xDEADBEEF}};
517  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
518  for (const auto& ref : o2::framework::InputRecordWalker(ctx.inputs(), {dummy})) {
519  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
520  if (dh->payloadSize == 0) {
521  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
522  if (++contDeadBeef <= maxWarn) {
523  LOGP(warning, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
524  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize,
525  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
526  }
527  return true;
528  }
529  }
530  contDeadBeef = 0; // if good data, reset the counter
531  return false;
