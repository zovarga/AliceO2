114 DECLARE_SOA_DYNAMIC_COLUMN(Px, px,
115  [](float signed1Pt, float snp, float alpha) -> float {
116  auto pt = 1.f / std::abs(signed1Pt);
117  float cs, sn;
118  math_utils::sincos(alpha, sn, cs);
119  auto r = std::sqrt((1.f - snp) * (1.f + snp));
120  return pt * (r * cs - snp * sn);
121  });
122 DECLARE_SOA_DYNAMIC_COLUMN(Py, py,
123  [](float signed1Pt, float snp, float alpha) -> float {
124  auto pt = 1.f / std::abs(signed1Pt);
125  float cs, sn;
126  math_utils::sincos(alpha, sn, cs);
127  auto r = std::sqrt((1.f - snp) * (1.f + snp));
128  return pt * (snp * cs + r * sn);
129  });
