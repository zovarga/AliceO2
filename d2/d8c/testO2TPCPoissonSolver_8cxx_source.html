139 template <typename DataT>
140 void testAlmostEqualArray(o2::tpc::DataContainer3D<DataT>& analytical, o2::tpc::DataContainer3D<DataT>& numerical)
141 {
142  for (size_t iPhi = 0; iPhi < numerical.getNPhi(); ++iPhi) {
143  for (size_t iR = 0; iR < numerical.getNR(); ++iR) {
144  for (size_t iZ = 0; iZ < numerical.getNZ(); ++iZ) {
145  if (std::fabs(analytical(iZ, iR, iPhi)) < ABSTOLERANCE) {
146  BOOST_CHECK_SMALL(numerical(iZ, iR, iPhi) - analytical(iZ, iR, iPhi), ABSTOLERANCE);
147  } else {
148  BOOST_CHECK_CLOSE(numerical(iZ, iR, iPhi), analytical(iZ, iR, iPhi), TOLERANCE);
149  }
150  }
151  }
152  }
153 }

171 template <typename DataT>
172 void poissonSolver3D()
173 {
174  using GridProp = GridProperties<DataT>;
175  const o2::tpc::RegularGrid3D<DataT> grid3D{GridProp::ZMIN, GridProp::RMIN, GridProp::PHIMIN, GridProp::getGridSpacingZ(NZ), GridProp::getGridSpacingR(NR), GridProp::getGridSpacingPhi(NPHI)};
176 
177  using DataContainer = o2::tpc::DataContainer3D<DataT>;
178  DataContainer potentialNumerical(NZ, NR, NPHI);
179  DataContainer potentialAnalytical(NZ, NR, NPHI);
180  DataContainer charge(NZ, NR, NPHI);
181 
182  const o2::tpc::AnalyticalFields<DataT> analyticalFields;
183  // set the boudnary and charge for numerical poisson solver
184  setChargeDensityFromFormula<DataT>(analyticalFields, grid3D, charge);
185  setPotentialBoundaryFromFormula<DataT>(analyticalFields, grid3D, potentialNumerical);
186 
187  // set analytical potential
188  setPotentialFromFormula<DataT>(analyticalFields, grid3D, potentialAnalytical);
189 
190  //calculate numerical potential
191  PoissonSolver<DataT> poissonSolver(grid3D);
192  const int symmetry = 0;
193  poissonSolver.poissonSolver3D(potentialNumerical, charge, symmetry);
194 
195  // compare numerical with analytical solution of the potential
196  testAlmostEqualArray<DataT>(potentialAnalytical, potentialNumerical);
197 }
