34 o2::header::RDHAny RawReaderMemory::decodeRawHeader(const void* payloadwords)
35 {
36  auto headerversion = RDHDecoder::getVersion(payloadwords);
37  if (headerversion == 4) {
38  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV4*>(payloadwords));
39  } else if (headerversion == 5) {
40  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV5*>(payloadwords));
41  } else if (headerversion == 6) {
42  return o2::header::RDHAny(*reinterpret_cast<const o2::header::RAWDataHeaderV6*>(payloadwords));
43  }
44  LOG(error) << "Wrong header version " << int(headerversion);
45  throw RawDecodingError::ErrorType_t::HEADER_DECODING;
46 }

57 void RawReaderMemory::next()
58 {
59  //Reads pages till RCU trailer found
60  //Several 8kB pages can be concatenated
61  //RCU trailer expected at the end of payload
62  //but not at the end of each page
63  mRawPayload.reset();
64  mCurrentTrailer.reset();
65  bool isDataTerminated = false;
66  do {
67  nextPage(false);
68  if (hasNext()) {
69  try {
70  auto nextheader = decodeRawHeader(mRawMemoryBuffer.data() + mCurrentPosition);
71 
72  // check continuing payload based on the bc/orbit ID
73  auto currentbc = RDHDecoder::getTriggerBC(mRawHeader),
74  nextbc = RDHDecoder::getTriggerBC(nextheader);
75  auto currentorbit = RDHDecoder::getTriggerOrbit(mRawHeader),
76  nextorbit = RDHDecoder::getTriggerOrbit(nextheader);
77  if (currentbc != nextbc || currentorbit != nextorbit) {
78  isDataTerminated = true;
79  } else {
80  auto nextpagecounter = RDHDecoder::getPageCounter(nextheader);
81  if (nextpagecounter == 0) {
82  isDataTerminated = true;
83  } else {
84  isDataTerminated = false;
85  }
86  }
87  } catch (...) {
88  throw RawDecodingError::ErrorType_t::HEADER_DECODING;
89  }
90  } else {
91  isDataTerminated = true;
92  }
93  // Check if the data continues
94  } while (!isDataTerminated);
95  if (mRawBuffer.getNDataWords() > 0) {
96  try {
97  mCurrentTrailer.constructFromPayloadWords(mRawBuffer.getDataWords());
98  } catch (RCUTrailer::Error e) {
99  LOG(error) << "Trailer decoding error: " << e.what();
100  throw RawDecodingError::ErrorType_t::HEADER_DECODING;
101  }
102  }
103 }

105 void RawReaderMemory::nextPage(bool doResetPayload)
106 {
107  if (!hasNext()) {
108  return;
109  }
110  if (doResetPayload) {
111  mRawPayload.reset();
112  }
113  mRawHeaderInitialized = false;
114  mPayloadInitialized = false;
115 
116  // Read RDH header
117  try {
118  mRawHeader = decodeRawHeader(mRawMemoryBuffer.data() + mCurrentPosition);
119  while (RDHDecoder::getOffsetToNext(mRawHeader) == RDHDecoder::getHeaderSize(mRawHeader) &&
120  mCurrentPosition + RDHDecoder::getHeaderSize(mRawHeader) < mRawMemoryBuffer.size()) {
121  // No Payload - jump to next rawheader
122  // This will eventually move, depending on whether for events without payload in the SRU we send the RCU trailer
123  mCurrentPosition += RDHDecoder::getHeaderSize(mRawHeader);
124  mRawHeader = decodeRawHeader(mRawMemoryBuffer.data() + mCurrentPosition);
125  }
126  mRawHeaderInitialized = true;
127  } catch (...) {
128  throw RawDecodingError::ErrorType_t::HEADER_DECODING;
129  }
130  if (mCurrentPosition + RDHDecoder::getMemorySize(mRawHeader) > mRawMemoryBuffer.size()) {
131  // Payload incomplete
132  throw RawDecodingError::ErrorType_t::PAYLOAD_DECODING;
133  }
134 
135  mRawBuffer.readFromMemoryBuffer(gsl::span<const char>(mRawMemoryBuffer.data() + mCurrentPosition + RDHDecoder::getHeaderSize(mRawHeader),
136  RDHDecoder::getMemorySize(mRawHeader) - RDHDecoder::getHeaderSize(mRawHeader)));
137  gsl::span<const uint32_t> payloadWithoutTrailer(mRawBuffer.getDataWords().data(), mRawBuffer.getNDataWords());
138  mRawPayload.appendPayloadWords(payloadWithoutTrailer);
139  mRawPayload.increasePageCount();
140 
141  mCurrentPosition += RDHDecoder::getOffsetToNext(mRawHeader);
142 }
