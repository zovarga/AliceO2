37 void ROFTimeClusterFinder::initTimeBins()
38 {
39  // initialize the time bins vector
40  std::fill(mTimeBins.begin(), mTimeBins.end(), TimeBin());
41 
42  if (mInputROFs.empty()) {
43  return;
44  }
45 
46  o2::InteractionRecord mFirstIR = mInputROFs.front().getBCData();
47 
48  // store the number of digits in each bin
49  for (size_t iRof = 0; iRof < mInputROFs.size(); iRof++) {
50  const auto& rof = mInputROFs[iRof];
51  const auto& ir = rof.getBCData();
52  auto rofBc = ir.differenceInBC(mFirstIR);
53  auto binIdx = rofBc / mBinWidth;
54 
55  if (binIdx < 0) {
56  continue;
57  } else if (binIdx >= mNbinsInOneTF) {
58  break;
59  }
60 
61  auto& timeBin = mTimeBins[binIdx];
62 
63  if (timeBin.mFirstIdx < 0) {
64  timeBin.mFirstIdx = iRof;
65  }
66  timeBin.mLastIdx = iRof;
67  timeBin.mSize += 1;
68  timeBin.mNDigits += rof.getNEntries();
69  }
70 
71  if (mDebug) {
72  std::cout << "Peak search histogram:" << std::endl;
73  for (int32_t i = 0; i < mNbinsInOneTF; i++) {
74  if (mTimeBins[i].mNDigits > 0) {
75  std::cout << fmt::format("bin {}: {}", i, mTimeBins[i].mNDigits) << std::endl;
76  }
77  }
78  }
79 }

83 int32_t ROFTimeClusterFinder::getNextPeak()
84 {
85  int32_t sPadding = mNbinsInOneWindow / 2;
86 
87  if (mDebug) {
88  std::cout << "Searching peak from " << mLastSavedTimeBin + 1 << std::endl;
89  }
90 
91  // loop over the bins and search for local maxima
92  // a local maxima is defined as a bin tht is higher than all the surrounding 4 bins (2 below and 2 above)
93  for (int32_t i = mLastSavedTimeBin + sPadding + 1; i < mNbinsInOneTF; i++) {
94  auto& peak = mTimeBins[i];
95  if (peak.empty()) {
96  continue;
97  }
98 
99  bool found{true};
100  // the peak must be strictly higher than previous bins
101  for (int j = i - sPadding; j < i; j++) {
102  if (j >= 0 && peak <= mTimeBins[j]) {
103  found = false;
104  break;
105  }
106  }
107  // the peak must be higher than or equal to next bins
108  for (int j = i + 1; j <= i + sPadding; j++) {
109  if (j < mNbinsInOneTF && peak < mTimeBins[j]) {
110  found = false;
111  break;
112  }
113  }
114 
115  if (!found) {
116  continue;
117  }
118 
119  if (mDebug) {
120  std::cout << fmt::format("new peak found at bin {}, entries = {}", i, mTimeBins[i].mNDigits) << std::endl;
121  }
122 
123  return i;
124  }
125  return -1;
126 }

130 void ROFTimeClusterFinder::storeROF(int32_t firstBin, int32_t lastBin)
131 {
132  if (mDebug) {
133  std::cout << fmt::format("Storing ROF from bin range [{},{}]", firstBin, lastBin) << std::endl;
134  }
135 
136  if (firstBin < 0) {
137  firstBin = 0;
138  }
139  if (lastBin >= mNbinsInOneTF) {
140  lastBin = mNbinsInOneTF - 1;
141  }
142 
143  int32_t rofFirstIdx{-1};
144  int32_t rofLastIdx{-1};
145  uint32_t nDigits{0};
146  uint32_t nROFs{0};
147  for (int32_t j = firstBin; j <= lastBin; j++) {
148  auto& timeBin = mTimeBins[j];
149  if (timeBin.mFirstIdx < 0) {
150  continue;
151  }
152 
153  if (rofFirstIdx < 0) {
154  rofFirstIdx = timeBin.mFirstIdx;
155  };
156  rofLastIdx = timeBin.mLastIdx;
157 
158  nROFs += timeBin.mSize;
159  nDigits += timeBin.mNDigits;
160 
161  if (mDebug) {
162  std::cout << fmt::format(" bin {}: firstIdx={} size={} ndigits={}", j, timeBin.mFirstIdx, timeBin.mSize, timeBin.mNDigits) << std::endl;
163  }
164  }
165 
166  // a new time ROF is stored only if there are some digit ROFs in the interval
167  if (rofFirstIdx >= 0) {
168  // get the indexes of the first and last ROFs in this time cluster, and the corresponding interaction records
169  auto& firstRofInCluster = mInputROFs[rofFirstIdx];
170  auto& irFirst = firstRofInCluster.getBCData();
171  auto& lastRofInCluster = mInputROFs[rofLastIdx];
172  auto& irLast = lastRofInCluster.getBCData();
173 
174  // get the index of the first digit in this time cluster
175  auto firstDigitIdx = firstRofInCluster.getFirstIdx();
176 
177  // compute the width in BC units of the time-cluster ROF
178  auto bcDiff = irLast.differenceInBC(irFirst);
179  auto bcWidth = bcDiff + lastRofInCluster.getBCWidth();
180 
181  // create a ROF that includes all the digits in this time cluster
182  mOutputROFs.emplace_back(irFirst, firstDigitIdx, nDigits, bcWidth);
183 
184  if (mDebug) {
185  std::cout << fmt::format("new ROF stored: firstDigitIdx={} size={} bcWidth={}", firstDigitIdx, nDigits, bcWidth) << std::endl;
186  }
187  }
188 
189  mLastSavedTimeBin = lastBin;
190 }

194 void ROFTimeClusterFinder::process()
195 {
196  if (mDebug) {
197  std::cout << "\n\n==================\n[ROFTimeClusterFinder] processing new TF\n"
198  << std::endl;
199  }
200 
201  initTimeBins();
202 
203  mLastSavedTimeBin = -1;
204  int32_t peak{-1};
205  while ((peak = getNextPeak()) >= 0) {
206  int32_t peakStart = peak - mNbinsInOneWindow / 2;
207  int32_t peakEnd = peakStart + mNbinsInOneWindow - 1;
208 
209  // peak found, we add the corresponding rof(s)
210  // first we fill the gap between the last peak and the current one, if needed
211  if (mDebug) {
212  std::cout << fmt::format("peakStart={} mLastSavedTimeBin={}", peakStart, mLastSavedTimeBin) << std::endl;
213  }
214  while ((peakStart - mLastSavedTimeBin) > 1) {
215  int32_t firstBin = mLastSavedTimeBin + 1;
216  int32_t lastBin = firstBin + mNbinsInOneWindow - 1;
217  if (lastBin >= peakStart) {
218  lastBin = peakStart - 1;
219  }
220 
221  storeROF(firstBin, lastBin);
222  }
223  storeROF(peakStart, peakEnd);
224  }
225 }

229 char* ROFTimeClusterFinder::saveROFRsToBuffer(size_t& bufSize)
230 {
231  static constexpr size_t sizeOfROFRecord = sizeof(o2::mch::ROFRecord);
232 
233  if (mDebug) {
234  dumpOutputROFs();
235  }
236 
237  bufSize = mOutputROFs.size() * sizeOfROFRecord;
238  o2::mch::ROFRecord* buf = reinterpret_cast<o2::mch::ROFRecord*>(malloc(bufSize));
239  if (!buf) {
240  bufSize = 0;
241  return nullptr;
242  }
243 
244  o2::mch::ROFRecord* p = buf;
245  for (size_t i = 0; i < mOutputROFs.size(); i++) {
246  auto& rof = mOutputROFs[i];
247  memcpy(p, &(rof), sizeOfROFRecord);
248  p += 1;
249  }
250 
251  return reinterpret_cast<char*>(buf);
252 }
