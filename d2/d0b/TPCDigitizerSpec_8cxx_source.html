113  void initDigitizerTask(framework::InitContext& ic) override
114  {
115  LOG(info) << "Initializing TPC digitization";
116 
117  mLaneId = ic.services().get<const o2::framework::DeviceSpec>().rank;
118 
119  mWithMCTruth = o2::conf::DigiParams::Instance().mctruth;
120  auto useDistortions = ic.options().get<int>("distortionType");
121  auto triggeredMode = ic.options().get<bool>("TPCtriggered");
122 
123  if (useDistortions > 0) {
124  if (useDistortions == 1) {
125  LOG(info) << "Using realistic space-charge distortions.";
126  } else {
127  LOG(info) << "Using constant space-charge distortions.";
128  }
129  auto readSpaceChargeString = ic.options().get<std::string>("readSpaceCharge");
130  std::vector<std::string> readSpaceCharge;
131  std::stringstream ssSpaceCharge(readSpaceChargeString);
132  while (ssSpaceCharge.good()) {
133  std::string substr;
134  getline(ssSpaceCharge, substr, ',');
135  readSpaceCharge.push_back(substr);
136  }
137  if (readSpaceCharge[0].size() != 0) { // use pre-calculated space-charge object
138  if (std::filesystem::exists(readSpaceCharge[0])) {
139  TFile fileSC(readSpaceCharge[0].data(), "READ");
140  mDigitizer.setUseSCDistortions(fileSC);
141  } else {
142  LOG(error) << "Space-charge object or file not found!";
143  }
144  } else { // create new space-charge object either with empty TPC or an initial space-charge density provided by histogram
145  SC::SCDistortionType distortionType = useDistortions == 2 ? SC::SCDistortionType::SCDistortionsConstant : SC::SCDistortionType::SCDistortionsRealistic;
146  auto inputHistoString = ic.options().get<std::string>("initialSpaceChargeDensity");
147  std::vector<std::string> inputHisto;
148  std::stringstream ssHisto(inputHistoString);
149  while (ssHisto.good()) {
150  std::string substr;
151  getline(ssHisto, substr, ',');
152  inputHisto.push_back(substr);
153  }
154  std::unique_ptr<TH3> hisSCDensity;
155  if (std::filesystem::exists(inputHisto[0])) {
156  auto fileSCInput = std::unique_ptr<TFile>(TFile::Open(inputHisto[0].data()));
157  if (fileSCInput->FindKey(inputHisto[1].data())) {
158  hisSCDensity.reset((TH3*)fileSCInput->Get(inputHisto[1].data()));
159  hisSCDensity->SetDirectory(nullptr);
160  }
161  }
162  if (hisSCDensity.get() != nullptr) {
163  LOG(info) << "TPC: Providing initial space-charge density histogram: " << hisSCDensity->GetName();
164  mDigitizer.setUseSCDistortions(distortionType, hisSCDensity.get());
165  } else {
166  if (distortionType == SC::SCDistortionType::SCDistortionsConstant) {
167  LOG(error) << "Input space-charge density histogram or file not found!";
168  }
169  }
170  }
171  }
172  mDigitizer.setContinuousReadout(!triggeredMode);
173 
174  // we send the GRP data once if the corresponding output channel is available
175  // and set the flag to false after
176  mWriteGRP = true;
177 
178  // clean up (possibly) existing digit files
179  if (mInternalWriter) {
180  cleanDigitFile();
181  }
182  }

188  std::stringstream tmp;

191  std::filesystem::remove(tmp.str());
192  }

195  void writeToROOTFile()
196  {
197  if (!mInternalROOTFlushFile) {
198  std::stringstream tmp;
199  tmp << "tpc_driftime_digits_lane" << mLaneId << ".root";
200  mInternalROOTFlushFile = new TFile(tmp.str().c_str(), "UPDATE");
201  std::stringstream trname;
202  trname << mSector;
203  mInternalROOTFlushTTree = new TTree(trname.str().c_str(), "o2sim");
204  }
205  {
206  std::stringstream brname;
207  brname << "TPCDigit_" << mSector;
208  auto br = o2::base::getOrMakeBranch(*mInternalROOTFlushTTree, brname.str().c_str(), &mDigits);
209  br->Fill();
210  br->ResetAddress();
211  }
212  if (mWithMCTruth) {
213  // labels
214  std::stringstream brname;
215  brname << "TPCDigitMCTruth_" << mSector;
216  auto br = o2::base::getOrMakeBranch(*mInternalROOTFlushTTree, brname.str().c_str(), &mLabels);
217  br->Fill();
218  br->ResetAddress();
219  }
220  {
221  // common
222  std::stringstream brname;
223  brname << "TPCCommonMode_" << mSector;
224  auto br = o2::base::getOrMakeBranch(*mInternalROOTFlushTTree, brname.str().c_str(), &mCommonMode);
225  br->Fill();
226  br->ResetAddress();
227  }
228  }

230  void run(framework::ProcessingContext& pc)
231  {
232  LOG(info) << "Processing TPC digitization";
233 
235  auto& cdb = o2::tpc::CDBInterface::instance();
236  cdb.setUseDefaults();
237  if (std::filesystem::exists("GainMap.root")) {
238  LOG(info) << "TPC: Using gain map from 'GainMap.root'";
239  cdb.setGainMapFromFile("GainMap.root");
240  }
241 
242  for (auto it = pc.inputs().begin(), end = pc.inputs().end(); it != end; ++it) {
243  for (auto const& inputref : it) {
244  process(pc, inputref);
245  if (mInternalWriter) {
246  mInternalROOTFlushTTree->SetEntries(mFlushCounter);
247  mInternalROOTFlushFile->Write("", TObject::kOverwrite);
248  mInternalROOTFlushFile->Close();
249  // delete mInternalROOTFlushTTree; --> automatically done by ->Close()
250  delete mInternalROOTFlushFile;
251  mInternalROOTFlushFile = nullptr;
252  }
253  //TODO: make generic reset method?
254  mFlushCounter = 0;
255  mDigitCounter = 0;
256  }
257  }
258  }

260  // process one sector
261  void process(framework::ProcessingContext& pc, framework::DataRef const& inputref)
262  {
263  // read collision context from input
264  auto context = pc.inputs().get<o2::steer::DigitizationContext*>(inputref);
265  context->initSimChains(o2::detectors::DetID::TPC, mSimChains);
266  auto& irecords = context->getEventRecords();
267  LOG(info) << "TPC: Processing " << irecords.size() << " collisions";
268  if (irecords.size() == 0) {
269  return;
270  }
271  auto const* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(inputref);
272 
273  bool isContinuous = mDigitizer.isContinuousReadout();
274  // we publish the GRP data once if the output channel is there
275  if (mWriteGRP && pc.outputs().isAllowed({"TPC", "ROMode", 0})) {
276  auto roMode = isContinuous ? o2::parameters::GRPObject::CONTINUOUS : o2::parameters::GRPObject::PRESENT;
277  LOG(info) << "TPC: Sending ROMode= " << (mDigitizer.isContinuousReadout() ? "Continuous" : "Triggered")
278  << " to GRPUpdater from channel " << dh->subSpecification;
279  pc.outputs().snapshot(Output{"TPC", "ROMode", 0, Lifetime::Timeframe}, roMode);
280  }
281  mWriteGRP = false;
282 
283  // extract which sector to treat
284  auto const* sectorHeader = DataRefUtils::getHeader<TPCSectorHeader*>(inputref);
285  if (sectorHeader == nullptr) {
286  LOG(error) << "TPC: Sector header missing, skipping processing";
287  return;
288  }
289  auto sector = sectorHeader->sector();
290  mSector = sector;
291  mListOfSectors.push_back(sector);
292  LOG(info) << "TPC: Processing sector " << sector;
293  // the active sectors need to be propagated
294  uint64_t activeSectors = 0;
295  activeSectors = sectorHeader->activeSectors;
296 
297  // lambda that creates a DPL owned buffer to accumulate the digits (in shared memory)
298  auto makeDigitBuffer = [this, sector, &pc, activeSectors, &dh]() {
299  o2::tpc::TPCSectorHeader header{sector};
300  header.activeSectors = activeSectors;
301  if (mInternalWriter) {
302  using ContainerType = std::decay_t<decltype(pc.outputs().make<std::vector<o2::tpc::Digit>>(Output{"", "", 0}))>*;
303  return ContainerType(nullptr);
304  } else {
305  // default case
306  return &pc.outputs().make<std::vector<o2::tpc::Digit>>(Output{"TPC", "DIGITS", static_cast<SubSpecificationType>(dh->subSpecification), Lifetime::Timeframe, header});
307  }
308  };
309  // lambda that snapshots the common mode vector to be sent out; prepares and attaches header with sector information
310  auto snapshotCommonMode = [this, sector, &pc, activeSectors, &dh](std::vector<o2::tpc::CommonMode> const& commonMode) {
311  o2::tpc::TPCSectorHeader header{sector};
312  header.activeSectors = activeSectors;
313  if (!mInternalWriter) {
314  // note that snapshoting only works with non-const references (to be fixed?)
315  pc.outputs().snapshot(Output{"TPC", "COMMONMODE", static_cast<SubSpecificationType>(dh->subSpecification), Lifetime::Timeframe,
316  header},
317  const_cast<std::vector<o2::tpc::CommonMode>&>(commonMode));
318  }
319  };
320  // lambda that snapshots labels to be sent out; prepares and attaches header with sector information
321  auto snapshotLabels = [this, &sector, &pc, activeSectors, &dh](o2::dataformats::MCTruthContainer<o2::MCCompLabel> const& labels) {
322  o2::tpc::TPCSectorHeader header{sector};
323  header.activeSectors = activeSectors;
324  if (mWithMCTruth) {
325  if (!mInternalWriter) {
326  auto& sharedlabels = pc.outputs().make<o2::dataformats::ConstMCTruthContainer<o2::MCCompLabel>>(Output{"TPC", "DIGITSMCTR", static_cast<SubSpecificationType>(dh->subSpecification), Lifetime::Timeframe, header});
327  labels.flatten_to(sharedlabels);
328  }
329  }
330  };
331  // lambda that snapshots digits grouping (triggers) to be sent out; prepares and attaches header with sector information
332  auto snapshotEvents = [this, sector, &pc, activeSectors, &dh](const std::vector<DigiGroupRef>& events) {
333  o2::tpc::TPCSectorHeader header{sector};
334  header.activeSectors = activeSectors;
335  if (!mInternalWriter) {
336  LOG(info) << "TPC: Send TRIGGERS for sector " << sector << " channel " << dh->subSpecification << " | size " << events.size();
337  pc.outputs().snapshot(Output{"TPC", "DIGTRIGGERS", static_cast<SubSpecificationType>(dh->subSpecification), Lifetime::Timeframe,
338  header},
339  const_cast<std::vector<DigiGroupRef>&>(events));
340  }
341  };
342 
343  auto digitsAccum = makeDigitBuffer(); // accumulator for digits
344  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labelAccum; // timeframe accumulator for labels
345  std::vector<CommonMode> commonModeAccum;
346  std::vector<DigiGroupRef> eventAccum;
347 
348  // this should not happen any more, legacy condition when the sector variable was used
349  // to transport control information
350  if (sector < 0) {
351  throw std::runtime_error("Legacy control information is not expected any more");
352  }
353 
354  // the TPCSectorHeader now allows to transport information for more than one sector,
355  // e.g. for transporting clusters in one single data block. The digitization is however
356  // only on sector level
357  if (sector >= TPCSectorHeader::NSectors) {
358  throw std::runtime_error("Digitizer can only work on single sectors");
359  }
360 
361  mDigitizer.setSector(sector);
362  mDigitizer.init();
363 
364  auto& eventParts = context->getEventParts();
365 
366  auto flushDigitsAndLabels = [this, digitsAccum, &labelAccum, &commonModeAccum](bool finalFlush = false) {
367  mFlushCounter++;
368  // flush previous buffer
369  mDigits.clear();
370  mLabels.clear();
371  mCommonMode.clear();
372  mDigitizer.flush(mDigits, mLabels, mCommonMode, finalFlush);
373  LOG(info) << "TPC: Flushed " << mDigits.size() << " digits, " << mLabels.getNElements() << " labels and " << mCommonMode.size() << " common mode entries";
374 
375  if (mInternalWriter) {
376  // the natural place to write out this independent datachunk immediately ...
377  writeToROOTFile();
378  } else {
379  // ... or to accumulate and later forward to next DPL proc
380  std::copy(mDigits.begin(), mDigits.end(), std::back_inserter(*digitsAccum));
381  if (mWithMCTruth) {
382  labelAccum.mergeAtBack(mLabels);
383  }
384  std::copy(mCommonMode.begin(), mCommonMode.end(), std::back_inserter(commonModeAccum));
385  }
386  mDigitCounter += mDigits.size();
387  };
388 
389  if (isContinuous) {
390  auto& hbfu = o2::raw::HBFUtils::Instance();
391  double time = hbfu.getFirstIRofTF(o2::InteractionRecord(0, hbfu.orbitFirstSampled)).bc2ns() / 1000.;
392  mDigitizer.setOutputDigitTimeOffset(time);
393  mDigitizer.setStartTime(irecords[0].getTimeNS() / 1000.f);
394  }
395 
396  TStopwatch timer;
397  timer.Start();
398 
399  // loop over all composite collisions given from context
400  // (aka loop over all the interaction records)
401  for (int collID = 0; collID < irecords.size(); ++collID) {
402  const double eventTime = irecords[collID].getTimeNS() / 1000.f;
403  LOG(info) << "TPC: Event time " << eventTime << " us";
404  mDigitizer.setEventTime(eventTime);
405  if (!isContinuous) {
406  mDigitizer.setStartTime(eventTime);
407  }
408  size_t startSize = mDigitCounter; // digitsAccum->size();
409 
410  // for each collision, loop over the constituents event and source IDs
411  // (background signal merging is basically taking place here)
412  for (auto& part : eventParts[collID]) {
413  const int eventID = part.entryID;
414  const int sourceID = part.sourceID;
415 
416  // get the hits for this event and this source
417  std::vector<o2::tpc::HitGroup> hitsLeft;
418  std::vector<o2::tpc::HitGroup> hitsRight;
419  context->retrieveHits(mSimChains, getBranchNameLeft(sector).c_str(), part.sourceID, part.entryID, &hitsLeft);
420  context->retrieveHits(mSimChains, getBranchNameRight(sector).c_str(), part.sourceID, part.entryID, &hitsRight);
421  LOG(debug) << "TPC: Found " << hitsLeft.size() << " hit groups left and " << hitsRight.size() << " hit groups right in collision " << collID << " eventID " << part.entryID;
422 
423  mDigitizer.process(hitsLeft, eventID, sourceID);
424  mDigitizer.process(hitsRight, eventID, sourceID);
425 
426  flushDigitsAndLabels();
427 
428  if (!isContinuous) {
429  eventAccum.emplace_back(startSize, mDigits.size());
430  }
431  }
432  }
433 
434  // final flushing step; getting everything not yet written out
435  if (isContinuous) {
436  LOG(info) << "TPC: Final flush";
437  flushDigitsAndLabels(true);
438  eventAccum.emplace_back(0, mDigitCounter); // all digits are grouped to 1 super-event pseudo-triggered mode
439  }
440 
441  if (!mInternalWriter) {
442  // send out to next stage
443  snapshotEvents(eventAccum);
444  // snapshotDigits(digitsAccum); --> done automatically
445  snapshotCommonMode(commonModeAccum);
446  snapshotLabels(labelAccum);
447  }
448 
449  timer.Stop();
450  LOG(info) << "TPC: Digitization took " << timer.CpuTime() << "s";
451  }
452 
453  private:
454  o2::tpc::Digitizer mDigitizer;
455  std::vector<TChain*> mSimChains;
456  std::vector<o2::tpc::Digit> mDigits;
457  o2::dataformats::MCTruthContainer<o2::MCCompLabel> mLabels;
458  std::vector<o2::tpc::CommonMode> mCommonMode;
459  std::vector<int> mListOfSectors; // a list of sectors treated by this task
460  TFile* mInternalROOTFlushFile = nullptr;
461  TTree* mInternalROOTFlushTTree = nullptr;
462  size_t mDigitCounter = 0;
463  size_t mFlushCounter = 0;
464  int mLaneId = 0; // the id of the current process within the parallel pipeline
465  int mSector = 0;
466  bool mWriteGRP = false;
467  bool mWithMCTruth = true;
468  bool mInternalWriter = false;

471 o2::framework::DataProcessorSpec getTPCDigitizerSpec(int channel, bool writeGRP, bool mctruth, bool internalwriter)
472 {
473  // create the full data processor spec using
474  // a name identifier
475  // input description
476  // algorithmic description (here a lambda getting called once to setup the actual processing function)
477  // options that can be used for this processor (here: input file names where to take the hits)
478  std::stringstream id;
479  id << "TPCDigitizer" << channel;
480 
481  std::vector<OutputSpec> outputs; // define channel by triple of (origin, type id of data to be sent on this channel, subspecification)
482 
483  if (!internalwriter) {
484  outputs.emplace_back("TPC", "DIGITS", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
485  outputs.emplace_back("TPC", "DIGTRIGGERS", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
486  if (mctruth) {
487  outputs.emplace_back("TPC", "DIGITSMCTR", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
488  }
489  outputs.emplace_back("TPC", "COMMONMODE", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe);
490  }
491  if (writeGRP) {
492  outputs.emplace_back("TPC", "ROMode", 0, Lifetime::Timeframe);
493  LOG(debug) << "TPC: Channel " << channel << " will supply ROMode";
494  }
495 
496  return DataProcessorSpec{
497  id.str().c_str(),
498  Inputs{InputSpec{"collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe}},
499  outputs,
500  AlgorithmSpec{adaptFromTask<TPCDPLDigitizerTask>(internalwriter)},
501  Options{{"distortionType", VariantType::Int, 0, {"Distortion type to be used. 0 = no distortions (default), 1 = realistic distortions (not implemented yet), 2 = constant distortions"}},
502  {"initialSpaceChargeDensity", VariantType::String, "", {"Path to root file containing TH3 with initial space-charge density and name of the TH3 (comma separated)"}},
503  {"readSpaceCharge", VariantType::String, "", {"Path to root file containing pre-calculated space-charge object and name of the object (comma separated)"}},
504  {"TPCtriggered", VariantType::Bool, false, {"Impose triggered RO mode (default: continuous)"}}}};
505 }

507 o2::framework::WorkflowSpec getTPCDigitizerSpec(int nLanes, std::vector<int> const& sectors, bool mctruth, bool internalwriter)
508 {
509  // channel parameter is deprecated in the TPCDigitizer processor, all descendants
510  // are initialized not to publish GRP mode, but the channel will be added to the first
511  // processor after the pipelines have been created. The processor will decide upon
512  // the index in the ParallelContext whether to publish
513  WorkflowSpec pipelineTemplate{getTPCDigitizerSpec(0, false, mctruth, internalwriter)};
514  // override the predefined name, index will be added by parallelPipeline method
515  pipelineTemplate[0].name = "TPCDigitizer";
516  WorkflowSpec pipelines = parallelPipeline(
517  pipelineTemplate, nLanes, [size = sectors.size()]() { return size; }, [&sectors](size_t index) { return sectors[index]; });
518  // add the channel for the GRP information to the first processor
519  pipelines[0].outputs.emplace_back("TPC", "ROMode", 0, Lifetime::Timeframe);
520  return pipelines;
521 }
