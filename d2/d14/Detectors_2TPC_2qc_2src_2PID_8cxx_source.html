78 bool PID::processTrack(const o2::tpc::TrackTPC& track)
79 {
80  // ===| variables required for cutting and filling |===
81  const auto p = track.getP();
82  const auto dEdxTot = track.getdEdx().dEdxTotTPC;
83  const auto dEdxMax = track.getdEdx().dEdxMaxTPC;
84  const auto phi = track.getPhi();
85  const auto tgl = track.getTgl();
86  const auto snp = track.getSnp();
87  const auto nclusters = track.getNClusterReferences();
88  const auto eta = track.getEta();
89 
90  double absEta = TMath::Abs(eta);
91 
92  // ===| histogram filling before cuts |===
93  mHist1D[8].Fill(nclusters);
94  mHist2D[4].Fill(p, dEdxTot);
95  mHist2D[10].Fill(p, dEdxMax);
96 
97  // ===| histogram filling including cuts |===
98  if (absEta < 1. && nclusters > 60 && dEdxTot > 20) {
99  mHist1D[0].Fill(nclusters);
100  mHist1D[1].Fill(dEdxTot);
101  mHist1D[2].Fill(dEdxMax);
102  mHist1D[3].Fill(phi);
103  mHist1D[4].Fill(tgl);
104  mHist1D[5].Fill(snp);
105 
106  mHist2D[0].Fill(phi, dEdxTot);
107  mHist2D[1].Fill(tgl, dEdxTot);
108  mHist2D[2].Fill(nclusters, dEdxTot);
109  mHist2D[3].Fill(p, dEdxTot);
110  mHist2D[9].Fill(p, dEdxMax);
111  }
112 
113  // ===| cuts and histogram filling for MIPs |===
114  if (p > 0.4 && p < 0.55 && absEta < 1. && nclusters > 80 && dEdxTot > 35 && dEdxTot < 60) {
115  mHist1D[6].Fill(dEdxTot);
116  if (eta > 0.) {
117  mHist2D[5].Fill(phi, dEdxTot);
118  } else {
119  mHist2D[6].Fill(phi, dEdxTot);
120  }
121  }
122 
123  // ===| cuts and histogram filling for electrons |===
124  if (p > 0.32 && p < 0.38 && absEta < 1. && nclusters > 80 && dEdxTot > 70 && dEdxTot < 100) {
125  mHist1D[7].Fill(dEdxTot);
126  if (eta > 0.) {
127  mHist2D[7].Fill(phi, dEdxTot);
128  } else {
129  mHist2D[8].Fill(phi, dEdxTot);
130  }
131  }
132 
133  return true;
134 }
