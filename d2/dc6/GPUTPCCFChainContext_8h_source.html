59  nPagesTotal = nPagesSectorMax = nPagesFragmentMax = 0;
60  for (unsigned int i = 0; i < GPUCA_NSLICES; i++) {

64 
65  if (tpcZS) {

69  fragmentData.resize(nFragments);
70  }

72  for (unsigned int i = 0; i < nFragments; i++) {
73  for (unsigned int j = 0; j < GPUCA_NSLICES; j++) {

75  fragmentData[i].nDigits[j][k] = fragmentData[i].nPages[j][k] = 0;
