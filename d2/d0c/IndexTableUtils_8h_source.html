60 inline void IndexTableUtils::setTrackingParameters(const T& params)
61 {
62  mInversePhiBinSize = params.PhiBins / constants::math::TwoPi;
63  mInverseZBinSize.resize(params.LayerZ.size());
64  mNzBins = params.ZBins;
65  mNphiBins = params.PhiBins;
66  mLayerZ = params.LayerZ;
67  for (unsigned int iL{0}; iL < mInverseZBinSize.size(); ++iL) {
68  mInverseZBinSize[iL] = 0.5f * params.ZBins / params.LayerZ[iL];
69  }
70 }
