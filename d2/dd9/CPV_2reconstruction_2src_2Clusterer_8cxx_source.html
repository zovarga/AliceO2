76 void Clusterer::makeClusters(gsl::span<const Digit> digits)
77 {
78  // A cluster is defined as a list of neighbour digits
79 
80  // Mark all digits as unused yet
81  const int maxNDigits = 23040; // There is no digits more than in CPV modules ;)
82  std::bitset<maxNDigits> digitsUsed;
83  digitsUsed.reset();
84 
85  int iFirst = mFirstDigitInEvent; // first index of digit which potentially can be a part of cluster
86 
87  for (int i = iFirst; i < mLastDigitInEvent; i++) {
88  if (digitsUsed.test(i - mFirstDigitInEvent)) {
89  continue;
90  }
91 
92  const Digit& digitSeed = digits[i];
93  float digitSeedEnergy = digitSeed.getAmplitude(); //already calibrated digits
94  if (digitSeedEnergy < o2::cpv::CPVSimParams::Instance().mDigitMinEnergy) {
95  continue;
96  }
97 
98  // is this digit so energetic that start cluster?
99  if (digitSeedEnergy < o2::cpv::CPVSimParams::Instance().mClusteringThreshold) {
100  continue;
101  }
102  // start new cluster
103  mClusters.emplace_back(digitSeed.getAbsId(), digitSeedEnergy, digitSeed.getLabel());
104  FullCluster& clu = mClusters.back();
105  digitsUsed.set(i - mFirstDigitInEvent, true);
106  int iDigitInCluster = 1;
107 
108  // Now scan remaining digits in list to find neigbours of our seed
109  int index = 0;
110  while (index < iDigitInCluster) { // scan over digits already in cluster
111  short digitSeedAbsId = clu.getDigitAbsId(index);
112  index++;
113  bool runLoop = true;
114  for (Int_t j = iFirst; runLoop && (j < mLastDigitInEvent); j++) {
115  if (digitsUsed.test(j - mFirstDigitInEvent)) {
116  continue; // look through remaining digits
117  }
118  const Digit& digitN = digits[j];
119  float digitNEnergy = digitN.getAmplitude(); //Already calibrated digits!
120  if (digitNEnergy < o2::cpv::CPVSimParams::Instance().mDigitMinEnergy) {
121  continue;
122  }
123 
124  // call (digit,digitN) in THAT oder !!!!!
125  Int_t ineb = Geometry::areNeighbours(digitSeedAbsId, digitN.getAbsId());
126  switch (ineb) {
127  case -1: // too early (e.g. previous module), do not look before j at subsequent passes
128  iFirst = j;
129  break;
130  case 0: // not a neighbour
131  break;
132  case 1: // are neighbours
133  clu.addDigit(digitN.getAbsId(), digitNEnergy, digitN.getLabel());
134  iDigitInCluster++;
135  digitsUsed.set(j - mFirstDigitInEvent, true);
136  break;
137  case 2: // too far from each other
138  default:
139  runLoop = false;
140  break;
141  } // switch
142  }
143  } // loop over cluster
144  } // energy theshold
145 }

147 void Clusterer::makeUnfoldings(gsl::span<const Digit> digits)
148 {
149  //Split cluster if several local maxima are found
150 
151  std::array<int, NLMMax> maxAt; // NLMMax:Maximal number of local maxima
152 
153  int numberOfNotUnfolded = mClusters.size();
154 
155  for (int i = 0; i < numberOfNotUnfolded; i++) { //can not use iterator here as list can expand
156  FullCluster& clu = mClusters[i];
157  if (clu.getNExMax() > -1) { //already unfolded
158  continue;
159  }
160  char nMultipl = clu.getMultiplicity();
161  char nMax = clu.getNumberOfLocalMax(maxAt);
162  if (nMax > 1) {
163  unfoldOneCluster(clu, nMax, maxAt, digits);
164  clu.setEnergy(0); // will be skipped later
165  } else {
166  clu.setNExMax(nMax); // Only one local maximum
167  }
168  }

171 void Clusterer::unfoldOneCluster(FullCluster& iniClu, char nMax, gsl::span<int> digitId, gsl::span<const Digit> digits)
172 {
173  // Performs the unfolding of a cluster with nMax overlapping showers
174  // Parameters: iniClu cluster to be unfolded
175  // nMax number of local maxima found (this is the number of new clusters)
176  // digitId: index of digits, corresponding to local maxima
177  // maxAtEnergy: energies of digits, corresponding to local maxima
178 
179  // Take initial cluster and calculate local coordinates of digits
180  // To avoid multiple re-calculation of same parameters
181  char mult = iniClu.getMultiplicity();
182  if (meInClusters.capacity() < mult) {
183  meInClusters.reserve(mult);
184  mfij.reserve(mult);
185  }
186 
187  const std::vector<FullCluster::CluElement>* cluElist = iniClu.getElementList();
188 
189  // Coordinates of centers of clusters
190  std::array<float, NLMMax> xMax;
191  std::array<float, NLMMax> zMax;
192  std::array<float, NLMMax> eMax;
193  std::array<float, NLMMax> deNew;
194 
195  //transient variables
196  std::array<float, NLMMax> a;
197  std::array<float, NLMMax> b;
198  std::array<float, NLMMax> c;
199 
200  for (int iclu = 0; iclu < nMax; iclu++) {
201  xMax[iclu] = (*cluElist)[digitId[iclu]].localX;
202  zMax[iclu] = (*cluElist)[digitId[iclu]].localZ;
203  eMax[iclu] = 2. * (*cluElist)[digitId[iclu]].energy;
204  }
205 
206  std::array<float, NLMMax> prop; // proportion of clusters in the current digit
207 
208  // Try to decompose cluster to contributions
209  int nIterations = 0;
210  bool insuficientAccuracy = true;
211  while (insuficientAccuracy && nIterations < o2::cpv::CPVSimParams::Instance().mNMaxIterations) {
212  insuficientAccuracy = false; // will be true if at least one parameter changed too much
213  std::memset(&a, 0, sizeof a);
214  std::memset(&b, 0, sizeof b);
215  std::memset(&c, 0, sizeof c);
216  //First calculate shower shapes
217  for (int idig = 0; idig < mult; idig++) {
218  auto it = (*cluElist)[idig];
219  for (int iclu = 0; iclu < nMax; iclu++) {
220  mfij[idig][iclu] = responseShape(it.localX - xMax[iclu], it.localZ - zMax[iclu]);
221  }
222  }
223 
224  //Fit energies
225  for (int idig = 0; idig < mult; idig++) {
226  auto it = (*cluElist)[idig];
227  for (int iclu = 0; iclu < nMax; iclu++) {
228  a[iclu] += mfij[idig][iclu] * mfij[idig][iclu];
229  b[iclu] += it.energy * mfij[idig][iclu];
230  for (int kclu = 0; kclu < nMax; kclu++) {
231  if (iclu == kclu) {
232  continue;
233  }
234  c[iclu] += eMax[kclu] * mfij[idig][iclu] * mfij[idig][kclu];
235  }
236  }
237  }
238  //Evaluate new maximal energies
239  for (int iclu = 0; iclu < nMax; iclu++) {
240  if (a[iclu] != 0.) {
241  float eNew = (b[iclu] - c[iclu]) / a[iclu];
242  insuficientAccuracy += (std::abs(eMax[iclu] - eNew) > eNew * o2::cpv::CPVSimParams::Instance().mUnfogingEAccuracy);
243  eMax[iclu] = eNew;
244  }
245  } // otherwise keep old value
246 
247  // Loop over all digits of parent cluster and split their energies between daughter clusters
248  // according to shower shape
249  // then re-evaluate local position of clusters
250  for (int idig = 0; idig < mult; idig++) {
251  float eEstimated = 0;
252  for (int iclu = 0; iclu < nMax; iclu++) {
253  prop[iclu] = eMax[iclu] * mfij[idig][iclu];
254  eEstimated += prop[iclu];
255  }
256  if (eEstimated == 0.) { // numerical accuracy
257  continue;
258  }
259  // Split energy of digit according to contributions
260  for (int iclu = 0; iclu < nMax; iclu++) {
261  meInClusters[idig][iclu] = (*cluElist)[idig].energy * prop[iclu] / eEstimated;
262  }
263  }
264 
265  // Recalculate parameters of clusters and check relative variation of energy and absolute of position
266  for (int iclu = 0; iclu < nMax; iclu++) {
267  float oldX = xMax[iclu];
268  float oldZ = zMax[iclu];
269  // full energy, need for weight
270  float eTotNew = 0;
271  for (int idig = 0; idig < mult; idig++) {
272  eTotNew += meInClusters[idig][iclu];
273  }
274  xMax[iclu] = 0;
275  zMax[iclu] = 0.;
276  float wtot = 0.;
277  for (int idig = 0; idig < mult; idig++) {
278  if (meInClusters[idig][iclu] > 0) {
279  // In unfolding it is better to use linear weight to reduce contribution of unfolded tails
280  float w = meInClusters[idig][iclu] / eTotNew;
281  // float w = std::max(std::log(eInClusters[idig][iclu] / eTotNew) + o2::cpv::CPVSimParams::Instance().mLogWeight, float(0.));
282  xMax[iclu] += (*cluElist)[idig].localX * w;
283  zMax[iclu] += (*cluElist)[idig].localZ * w;
284  wtot += w;
285  }
286  }
287  if (wtot > 0.) {
288  wtot = 1. / wtot;
289  xMax[iclu] *= wtot;
290  zMax[iclu] *= wtot;
291  }
292  // Compare variation of parameters
293  insuficientAccuracy += (std::abs(xMax[iclu] - oldX) > o2::cpv::CPVSimParams::Instance().mUnfogingXZAccuracy);
294  insuficientAccuracy += (std::abs(zMax[iclu] - oldZ) > o2::cpv::CPVSimParams::Instance().mUnfogingXZAccuracy);
295  }
296  nIterations++;
297  }
298  // Iterations finished, add new clusters
299  for (int iclu = 0; iclu < nMax; iclu++) {
300  mClusters.emplace_back();
301  FullCluster& clu = mClusters.back();
302  clu.setNExMax(nMax);
303  int idig = 0;
304  for (int idig = 0; idig < mult; idig++) {
305  float eDigit = meInClusters[idig][iclu];
306  idig++;
307  if (eDigit < o2::cpv::CPVSimParams::Instance().mDigitMinEnergy) {
308  continue;
309  }
310  clu.addDigit((*cluElist)[idig].absId, eDigit, (*cluElist)[idig].label);
311  }
312  }
313 }

316 void Clusterer::evalCluProperties(gsl::span<const Digit> digits, std::vector<Cluster>* clusters,
317  const o2::dataformats::MCTruthContainer<o2::MCCompLabel>* dmc,
318  o2::dataformats::MCTruthContainer<o2::MCCompLabel>* cluMC)
319 {
320 
321  if (clusters->capacity() - clusters->size() < mClusters.size()) { //avoid expanding vector per element
322  clusters->reserve(clusters->size() + mClusters.size());
323  }
324 
325  int labelIndex = 0;
326  if (mRunMC) {
327  labelIndex = cluMC->getIndexedSize();
328  }
329 
330  auto clu = mClusters.begin();
331 
332  while (clu != mClusters.end()) {
333 
334  if (clu->getEnergy() < 1.e-4) { //Marked earlier for removal
335  ++clu;
336  continue;
337  }
338 
339  // may be soft digits remain after unfolding
340  clu->purify();
341 
342  // LOG(debug) << "Purify done";
343  clu->evalAll();
344 
345  if (clu->getEnergy() > 1.e-4) { //Non-empty cluster
346  clusters->emplace_back(*clu);
347 
348  if (mRunMC) { //Handle labels
349  //Calculate list of primaries
350  //loop over entries in digit MCTruthContainer
351  const std::vector<FullCluster::CluElement>* vl = clu->getElementList();
352  auto ll = vl->begin();
353  while (ll != vl->end()) {
354  int i = (*ll).label; //index
355  if (i < 0) {
356  ++ll;
357  continue;
358  }
359  gsl::span<const o2::MCCompLabel> spDigList = dmc->getLabels(i);
360  gsl::span<o2::MCCompLabel> spCluList = cluMC->getLabels(labelIndex); //get updated list
361  auto digL = spDigList.begin();
362  while (digL != spDigList.end()) {
363  bool exist = false;
364  auto cluL = spCluList.begin();
365  while (cluL != spCluList.end()) {
366  if (*digL == *cluL) { //exist
367  exist = true;
368  break;
369  }
370  ++cluL;
371  }
372  if (!exist) { //just add label
373  cluMC->addElement(labelIndex, (*digL));
374  }
375  ++digL;
376  }
377  ++ll;
378  }
379  labelIndex++;
380  } // Work with MC
381  }
382 
383  ++clu;
384  }
385 }
