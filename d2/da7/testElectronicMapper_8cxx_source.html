63 template <size_t N>
64 std::set<int> nofDualSampas(std::array<int, N> deIds)
65 {
66  std::set<int> ds;
67 
68  for (auto deId : deIds) {
69  for (auto dsid : dslist(deId)) {
70  ds.insert(encode(DsDetId{deId, dsid}));
71  }
72  }
73  return ds;
74 }
75 
76 template <typename T>
77 std::set<int> nofDualSampasFromMapper(gsl::span<int> deids)
78 {
79  std::set<int> ds;
80 
81  auto d2e = o2::mch::raw::createDet2ElecMapper<T>();
82 
83  for (auto deid : deids) {
84  size_t nref = dslist(deid).size();
85  std::set<int> dsForOneDE;
86  for (auto dsid : dslist(deid)) {
87  DsDetId id{static_cast<uint16_t>(deid), static_cast<uint16_t>(dsid)};
88  auto dsel = d2e(id);
89  if (dsel.has_value()) {
90  auto code = o2::mch::raw::encode(id); // encode to be sure we're counting unique pairs (deid,dsid)
91  ds.insert(code);
92  dsForOneDE.insert(code);
93  } else {
94  std::cout << "did not find matching dsel for id=" << id << "\n";
95  }
96  };
97  if (dsForOneDE.size() != nref) {
98  std::cout << fmt::format("ERROR : mapper has {:4d} DS while DE {:4d} should have {:4d}\n", dsForOneDE.size(), deid, nref);
99  }
100  }
101  return ds;
102 }
