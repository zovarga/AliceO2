197 inline void VertexerTraits::updateVertexingParameters(const VertexingParameters& vrtPar)
198 {
199  mVrtParams = vrtPar;
200  mIndexTableUtils.setTrackingParameters(vrtPar);
201  mVrtParams.phiSpan = static_cast<int>(std::ceil(mIndexTableUtils.getNphiBins() * mVrtParams.phiCut /
202  constants::math::TwoPi));
203  mVrtParams.zSpan = static_cast<int>(std::ceil(mVrtParams.zCut * mIndexTableUtils.getInverseZCoordinate(0)));
204  for (auto& table : mIndexTables) {
205  table.resize(mIndexTableUtils.getNphiBins() * mIndexTableUtils.getNzBins() + 1, 0);
206  }
207 }

220 GPUhdi() const int4 VertexerTraits::getBinsRect(const Cluster& currentCluster, const int layerIndex,
221  const float directionZIntersection, float maxdeltaz, float maxdeltaphi,
222  const IndexTableUtils& utils)
223 {
224  const float zRangeMin = directionZIntersection - 2 * maxdeltaz;
225  const float phiRangeMin = currentCluster.phi - maxdeltaphi;
226  const float zRangeMax = directionZIntersection + 2 * maxdeltaz;
227  const float phiRangeMax = currentCluster.phi + maxdeltaphi;
228 
229  if (zRangeMax < -utils.getLayerZ(layerIndex + 1) ||
230  zRangeMin > utils.getLayerZ(layerIndex + 1) || zRangeMin > zRangeMax) {
231 
232  return getEmptyBinsRect();
233  }
234 
235  return int4{o2::gpu::GPUCommonMath::Max(0, utils.getZBinIndex(layerIndex + 1, zRangeMin)),
236  utils.getPhiBinIndex(math_utils::getNormalizedPhi(phiRangeMin)),
237  o2::gpu::GPUCommonMath::Min(utils.getNzBins() - 1, utils.getZBinIndex(layerIndex + 1, zRangeMax)),
238  utils.getPhiBinIndex(math_utils::getNormalizedPhi(phiRangeMax))};
239 }
