64 bool GeneratorFromFile::rejectOrFixKinematics(TParticle& p)
65 {
66  const auto nominalmass = p.GetMass();
67  auto mom2 = p.Px() * p.Px() + p.Py() * p.Py() + p.Pz() * p.Pz();
68  auto calculatedmass = p.Energy() * p.Energy() - mom2;
69  calculatedmass = (calculatedmass >= 0.) ? std::sqrt(calculatedmass) : -std::sqrt(-calculatedmass);
70  const double tol = 1.E-4;
71  auto difference = std::abs(nominalmass - calculatedmass);
72  if (std::abs(nominalmass - calculatedmass) > tol) {
73  const auto asgmass = p.GetCalcMass();
74  bool fix = mFixOffShell && std::abs(nominalmass - asgmass) < tol;
75  LOG(warn) << "Particle " << p.GetPdgCode() << " has off-shell mass: M_PDG= " << nominalmass << " (assigned= " << asgmass
76  << ") calculated= " << calculatedmass << " -> diff= " << difference << " | " << (fix ? "fixing" : "skipping");
77  if (fix) {
78  double e = std::sqrt(nominalmass * nominalmass + mom2);
79  p.SetMomentum(p.Px(), p.Py(), p.Pz(), e);
80  p.SetCalcMass(nominalmass);
81  } else {
82  return false;
83  }
84  }
85  return true;
86 }

88 Bool_t GeneratorFromFile::ReadEvent(FairPrimaryGenerator* primGen)
89 {
90  if (mEventCounter < mEventsAvailable) {
91  int particlecounter = 0;
92 
93  // get the tree and the branch
94  std::stringstream treestringstr;
95  treestringstr << "Event" << mEventCounter << "/TreeK";
96  TTree* tree = (TTree*)mEventFile->Get(treestringstr.str().c_str());
97  if (tree == nullptr) {
98  return kFALSE;
99  }
100 
101  auto branch = tree->GetBranch("Particles");
102  TParticle* particle = nullptr;
103  branch->SetAddress(&particle);
104  LOG(info) << "Reading " << branch->GetEntries() << " particles from Kinematics file";
105 
106  // read the whole kinematics initially
107  std::vector<TParticle> particles;
108  for (int i = 0; i < branch->GetEntries(); ++i) {
109  branch->GetEntry(i);
110  particles.push_back(*particle);
111  }
112 
113  // filter the particles from Kinematics.root originally put by a generator
114  // and which are trackable
115  auto isFirstTrackableDescendant = [](TParticle const& p) {
116  const int kTransportBit = BIT(14);
117  // The particle should have not set kDone bit and its status should not exceed 1
118  if ((p.GetUniqueID() > 0 && p.GetUniqueID() != kPNoProcess) || !p.TestBit(kTransportBit)) {
119  return false;
120  }
121  return true;
122  };
123 
124  for (int i = 0; i < branch->GetEntries(); ++i) {
125  auto& p = particles[i];
126  if (!isFirstTrackableDescendant(p)) {
127  continue;
128  }
129 
130  bool wanttracking = true; // RS as far as I understand, if it reached this point, it is trackable
131  if (wanttracking || !mSkipNonTrackable) {
132  if (!rejectOrFixKinematics(p)) {
133  continue;
134  }
135  auto pdgid = p.GetPdgCode();
136  auto px = p.Px();
137  auto py = p.Py();
138  auto pz = p.Pz();
139  auto vx = p.Vx();
140  auto vy = p.Vy();
141  auto vz = p.Vz();
142  auto parent = -1;
143  auto e = p.Energy();
144  auto tof = p.T();
145  auto weight = p.GetWeight();
146  LOG(debug) << "Putting primary " << pdgid << " " << p.GetStatusCode() << " " << p.GetUniqueID();
147  primGen->AddTrack(pdgid, px, py, pz, vx, vy, vz, parent, wanttracking, e, tof, weight);
148  particlecounter++;
149  }
150  }
151  mEventCounter++;
152 
153  LOG(info) << "Event generator put " << particlecounter << " on stack";
154  return kTRUE;
155  } else {
156  LOG(error) << "GeneratorFromFile: Ran out of events\n";
157  }
158  return kFALSE;
159 }

163 GeneratorFromO2Kine::GeneratorFromO2Kine(const char* name)
164 {
165  mEventFile = TFile::Open(name);
166  if (mEventFile == nullptr) {
167  LOG(fatal) << "EventFile " << name << " not found";
168  return;
169  }
170  // the kinematics will be stored inside a branch MCTrack
171  // different events are stored inside different entries
172  auto tree = (TTree*)mEventFile->Get("o2sim");
173  if (tree) {
174  mEventBranch = tree->GetBranch("MCTrack");
175  if (mEventBranch) {
176  mEventsAvailable = mEventBranch->GetEntries();
177  LOG(info) << "Found " << mEventsAvailable << " events in this file";
178  return;
179  }
180  }
181  LOG(error) << "Problem reading events from file " << name;
182 }

206 bool GeneratorFromO2Kine::importParticles()
207 {
208  // NOTE: This should be usable with kinematics files without secondaries
209  // It might need some adjustment to make it work with secondaries or to continue
210  // from a kinematics snapshot
211 
212  if (mEventCounter < mEventsAvailable) {
213  int particlecounter = 0;
214 
215  std::vector<o2::MCTrack>* tracks = nullptr;
216  mEventBranch->SetAddress(&tracks);
217  mEventBranch->GetEntry(mEventCounter);
218 
219  for (auto& t : *tracks) {
220 
221  // in case we do not want to continue, take only primaries
222  if (!mContinueMode && !t.isPrimary()) {
223  continue;
224  }
225 
226  auto pdg = t.GetPdgCode();
227  auto px = t.Px();
228  auto py = t.Py();
229  auto pz = t.Pz();
230  auto vx = t.Vx();
231  auto vy = t.Vy();
232  auto vz = t.Vz();
233  auto m1 = t.getMotherTrackId();
234  auto m2 = t.getSecondMotherTrackId();
235  auto d1 = t.getFirstDaughterTrackId();
236  auto d2 = t.getLastDaughterTrackId();
237  auto e = t.GetEnergy();
238  auto vt = t.T();
239  auto weight = 1.; // p.GetWeight() ??
240  auto wanttracking = t.getToBeDone();
241 
242  if (mContinueMode) { // in case we want to continue, do only inhibited tracks
243  wanttracking &= t.getInhibited();
244  }
245 
246  LOG(debug) << "Putting primary " << pdg;
247 
248  mParticles.push_back(TParticle(pdg, wanttracking, m1, m2, d1, d2, px, py, pz, e, vx, vy, vz, vt));
249  mParticles.back().SetUniqueID((unsigned int)t.getProcess()); // we should propagate the process ID
250 
251  particlecounter++;
252  }
253  mEventCounter++;
254 
255  if (tracks) {
256  delete tracks;
257  }
258 
259  LOG(info) << "Event generator put " << particlecounter << " on stack";
260  return true;
261  } else {
262  LOG(error) << "GeneratorFromO2Kine: Ran out of events\n";
263  }
264  return false;
265 }
