18  24/09/2020: Added new data member for abs. charge. This is needed for uniform treatment of tracks with non-standard
19  charge: 0 (for V0s) and e.g. 2 for hypernuclei.
20  In the aliroot AliExternalTrackParam this was treated by derived classes using virtual methods, which we don't use in O2.
21  The meaning of mP[kQ2Pt] remains exactly the same, except for q=0 case: in this case the mP[kQ2Pt] is just an alias to
22  1/pT, regardless of its sign, and the getCurvature() will 0 (because mAbsCharge is 0).

450 template <typename value_T>
451 GPUdi() void TrackParametrization<value_T>::getCircleParamsLoc(value_t bz, o2::math_utils::CircleXY<value_t>& c) const
452 {
453  // get circle params in track local frame, for straight line just set to local coordinates
454  c.rC = getCurvature(bz);
455  // treat as straight track if sagitta between the vertex and middle of TPC is below 0.01 cm
456  constexpr value_t MinSagitta = 0.01f, TPCMidR = 160.f, MinCurv = 8 * MinSagitta / (TPCMidR * TPCMidR);
457  if (gpu::CAMath::Abs(c.rC) > MinCurv) {
458  c.rC = 1.f / getCurvature(bz);
459  value_t sn = getSnp(), cs = gpu::CAMath::Sqrt((1.f - sn) * (1.f + sn));
460  c.xC = getX() - sn * c.rC; // center in tracking
461  c.yC = getY() + cs * c.rC; // frame. Note: r is signed!!!
462  c.rC = gpu::CAMath::Abs(c.rC);
463  } else {
464  c.rC = 0.f; // signal straight line
465  c.xC = getX();
466  c.yC = getY();
467  }
468 }

471 template <typename value_T>
472 GPUdi() void TrackParametrization<value_T>::getCircleParams(value_t bz, o2::math_utils::CircleXY<value_t>& c, value_t& sna, value_t& csa) const
473 {
474  // get circle params in loc and lab frame, for straight line just set to global coordinates
475  getCircleParamsLoc(bz, c);
476  o2::math_utils::detail::sincos(getAlpha(), sna, csa);
477  o2::math_utils::detail::rotateZ<value_t>(c.xC, c.yC, c.xC, c.yC, sna, csa); // center in global frame
478 }

481 template <typename value_T>
482 GPUdi() void TrackParametrization<value_T>::getLineParams(o2::math_utils::IntervalXY<value_t>& ln, value_t& sna, value_t& csa) const
483 {
484  // get line parameterization as { x = x0 + xSlp*t, y = y0 + ySlp*t }
485  o2::math_utils::detail::sincos(getAlpha(), sna, csa);
486  o2::math_utils::detail::rotateZ<value_t>(getX(), getY(), ln.getX0(), ln.getY0(), sna, csa); // reference point in global frame
487  value_t snp = getSnp(), csp = gpu::CAMath::Sqrt((1.f - snp) * (1.f + snp));
488  ln.setDX(csp * csa - snp * sna);
489  ln.setDY(snp * csa + csp * sna);
490 }
