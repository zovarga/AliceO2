33 template <class T>
34 bool Clusters::processCluster(const T& cluster, const o2::tpc::Sector sector, const int row)
35 {
36  if (mIsNormalized) {
37  denormalize();
38  LOGP(warning, "calling denormalize() before filling");
39  }
40 
41  const int nROC = row < 63 ? int(sector) : int(sector) + 36;
42  const int rocRow = row < 63 ? row : row - 63;
43 
44  const float pad = cluster.getPad();
45 
46  const auto qMax = cluster.getQmax();
47  const auto qTot = cluster.getQtot();
48  const auto sigmaPad = cluster.getSigmaPad();
49  const auto sigmaTime = cluster.getSigmaTime();
50  const auto timeBin = cluster.getTime();
51 
52  float count = mNClusters.getCalArray(nROC).getValue(rocRow, pad);
53  mNClusters.getCalArray(nROC).setValue(rocRow, pad, count + 1);
54 
55  float charge = mQMax.getCalArray(nROC).getValue(rocRow, pad);
56  mQMax.getCalArray(nROC).setValue(rocRow, pad, charge + qMax);
57 
58  charge = mQTot.getCalArray(nROC).getValue(rocRow, pad);
59  mQTot.getCalArray(nROC).setValue(rocRow, pad, charge + qTot);
60 
61  count = mSigmaTime.getCalArray(nROC).getValue(rocRow, pad);
62  mSigmaTime.getCalArray(nROC).setValue(rocRow, pad, count + sigmaTime);
63 
64  count = mSigmaPad.getCalArray(nROC).getValue(rocRow, pad);
65  mSigmaPad.getCalArray(nROC).setValue(rocRow, pad, count + sigmaPad);
66 
67  count = mTimeBin.getCalArray(nROC).getValue(rocRow, pad);
68  mTimeBin.getCalArray(nROC).setValue(rocRow, pad, count + timeBin);
69 
70  return true;
71 }

85 
86  val = mQMax.getValue(cruID.sector(), rowInSector, padInRow);

88

90  mTimeBin.setValue(cruID.sector(), rowInSector, padInRow, val + timeBin);

137 void Clusters::merge(Clusters& clusters)
138 {
139  const bool isThisNormalized = mIsNormalized;
140  const bool isOtherNormalized = clusters.mIsNormalized;
141 
142  if (isThisNormalized) {
143  denormalize();
144  }
145  if (isOtherNormalized) {
146  clusters.denormalize();
147  }
148 
149  mNClusters += clusters.mNClusters;
150  mQMax += clusters.mQMax;
151  mQTot += clusters.mQTot;
152  mSigmaTime += clusters.mSigmaTime;
153  mSigmaPad += clusters.mSigmaPad;
154  mTimeBin += clusters.mTimeBin;
155 
156  if (isThisNormalized) {
157  normalize();
158  }
159  if (isOtherNormalized) {
160  clusters.normalize();
161  }
162 }
