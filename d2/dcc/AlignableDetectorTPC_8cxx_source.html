48 void AlignableDetectorTPC::defineVolumes()
49 {
50  // define TPC volumes
51  //
52  const int kNSect = 18, kAC = 2, kIOROC = 2;
53  const char* kSide[kAC] = {"A", "C"};
54  const char* kROC[kIOROC] = {"Inner", "Outer"};
55  // AlignableSensorTPC *chamb=0;
56  //
57  int labDet = getDetLabel();
58  AlignableVolume* volTPC = new AlignableVolume("ALIC_1/TPC_M_1", labDet);
59  addVolume(volTPC);
60  //
61 
62  for (int roc = 0; roc < kIOROC; roc++) { // inner/outer
63  for (int side = 0; side < kAC; side++) { // A/C
64  for (int isc = 0; isc < kNSect; isc++) { // sector ID
65  const char* symname = Form("TPC/Endcap%s/Sector%d/%sChamber", kSide[side], isc + 1, kROC[roc]);
66  if (!gGeoManager->GetAlignableEntry(symname)) {
67  AliErrorF("Did not find alignable %s", symname);
68  continue;
69  }
70  int iid = side * kNSect + isc;
71  uint16_t vid = AliGeomManager::LayerToVolUID(AliGeomManager::kTPC1 + roc, iid);
72  iid = labDet + (1 + side) * 10000 + (1 + isc) * 100 + (1 + roc);
73  AlignableSensorTPC* sens = new AlignableSensorTPC(symname, vid, iid, isc);
74  sens->setParent(volTPC);
75  addVolume(sens);
76  } // sector ID
77  } // A/C
78  } // inner/outer
79  //
80 }
