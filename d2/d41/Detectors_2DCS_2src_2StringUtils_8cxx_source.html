192 uint64_t exp(uint64_t base, uint8_t exp) noexcept
193 {
194  uint64_t result(1);
195  while (exp) {
196  if (exp & 1) {
197  result *= base;
198  }
199  exp >>= 1;
200  base *= base;
201  }
202  return result;

327 string o2::dcs::to_hex_big_endian(const char* const start, const size_t length) noexcept
328 {
329  string s;
330  s.reserve(length * 3);
331  // My machine is little endian:
332  size_t i = length - 1;
333  do {
334  const char next = start[i];
335  s.push_back(to_alpha_numeric(next & 0xF0));
336  s.push_back(to_alpha_numeric(next & 0x0F));
337  s.push_back(' ');
338  --i;
339  } while (i < length);
340  s.pop_back();
341  return s;
342 }
