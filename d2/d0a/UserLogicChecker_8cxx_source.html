28 {
30  if (loc1.statusWord == loc2.statusWord && loc1.triggerWord == loc2.triggerWord && loc1.firedChambers == loc2.firedChambers && loc1.boardId == loc2.boardId) {
31  for (int ich = 0; ich < 4; ++ich) {
32  if (loc1.patternsBP[ich] != loc2.patternsBP[ich] || loc1.patternsNBP[ich] != loc2.patternsNBP[ich]) {
33  return false;
34  }
35  }
36  return true;
37  }
38  return false;

57 void UserLogicChecker::fillBoards(gsl::span<const ROBoard> data, gsl::span<const ROFRecord> rofRecords, bool isUL)
58 {
60 
61  auto& boards = isUL ? mBoardsUL : mBoardsBare;
62 
63  // The UL rejects the events outside the SOX/EOX
64  // So we should do the same
65  for (auto rofIt = rofRecords.begin(); rofIt != rofRecords.end(); ++rofIt) {
66  auto& loc = data[rofIt->firstEntry];
67  auto id = getId(loc);
68  auto isInside = mInsideDataTaking.find(id);
69  if (isInside == mInsideDataTaking.end()) {
70  mInsideDataTaking[id] = false;
71  isInside = mInsideDataTaking.find(id);
72  }
73  if (loc.triggerWord & raw::sSOX) {
74  isInside->second = true;
75  } else if (loc.triggerWord & raw::sEOX) {
76  isInside->second = false;
77  }
78  if (isInside->second) {
79  boards[id].push_back({rofIt->interactionRecord, loc});
80  }
81  }
82 }

84 void UserLogicChecker::clearBoards()
85 {
87 
88  for (int itype = 0; itype < 2; ++itype) {
89  auto& boards = (itype == 0) ? mBoardsUL : mBoardsBare;
90  auto& lastChecked = (itype == 0) ? mLastCheckedUL : mLastCheckedBare;
91  std::unordered_map<uint32_t, std::vector<boardInfo>> newBoards;
92  for (auto& boardItem : boards) {
93  auto lastIdxItem = lastChecked.find(boardItem.first);
94  if (lastIdxItem->second != boardItem.second.size()) {
95  auto& vec = newBoards[boardItem.first];
96  vec.insert(vec.end(), boardItem.second.begin() + lastIdxItem->second, boardItem.second.end());
97  }
98  }
99  boards.swap(newBoards);
100  lastChecked.clear();
101  }
102 }

104 bool UserLogicChecker::checkBoards(gsl::span<const ROBoard> bareData, gsl::span<const ROFRecord> bareRofs, gsl::span<const ROBoard> ulData, gsl::span<const ROFRecord> ulRofs)
105 {
107  clearBoards();
108  fillBoards(bareData, bareRofs, false);
109  fillBoards(ulData, ulRofs, true);
110  bool isOk = true;
111 
112  for (auto& bareItem : mBoardsBare) {
113  std::string boardType = (bareItem.first < 0x10000) ? "Loc" : "Reg";
114  auto& lastCheckedBare = mLastCheckedBare[bareItem.first];
115  auto& stats = mStatistics[bareItem.first];
116  stats[0] += bareItem.second.size();
117  std::stringstream ss;
118  ss << "\n-----------" << std::endl;
119  uint16_t boardId = (bareItem.first & 0xFFFF);
120  ss << "Checking crate: " << static_cast<int>(raw::getCrateId(boardId))
121  << " " << boardType << " board: " << static_cast<int>(raw::getLocId(boardId)) << std::endl;
122  auto ulItem = mBoardsUL.find(bareItem.first);
123  if (ulItem == mBoardsUL.end()) {
124  ss << " cannot find " << printIRHex(bareItem.second.front().interactionRecord) << " in ul" << std::endl;
125  isOk = false;
126  ss << "-----------" << std::endl;
127  mDebugMsg += ss.str();
128  stats[1] += bareItem.second.size();
129  continue;
130  }
131  auto& lastCheckedUL = mLastCheckedUL[ulItem->first];
132  lastCheckedUL = 0;
133  auto lastOk = lastCheckedUL;
134  bool isCurrentOk = true;
135  for (lastCheckedBare = 0; lastCheckedBare < bareItem.second.size(); ++lastCheckedBare) {
136  if (lastCheckedUL == ulItem->second.size()) {
137  break;
138  } else if (!isSame(ulItem->second[lastCheckedUL].board, bareItem.second[lastCheckedBare].board) || ulItem->second[lastCheckedUL].interactionRecord != bareItem.second[lastCheckedBare].interactionRecord) {
139  ss << "\nFirst divergence at element " << lastCheckedBare + 1 << " / " << bareItem.second.size() << ":" << std::endl;
140  ss << "bare: " << printIRHex(bareItem.second[lastCheckedBare].interactionRecord) << std::endl;
141  ss << bareItem.second[lastCheckedBare].board << std::endl;
142  ss << "ul: " << printIRHex(ulItem->second[lastCheckedUL].interactionRecord) << std::endl;
143  ss << ulItem->second[lastCheckedUL].board << std::endl;
144  isCurrentOk = false;
145  }
146  if (!isCurrentOk) {
147  if (lastOk != lastCheckedUL) {
148  ss << "lastOk: " << printIRHex(ulItem->second[lastOk].interactionRecord) << std::endl;
149  ss << ulItem->second[lastOk].board << std::endl;
150  } else {
151  ss << "lastOk: none. This is the first event!" << std::endl;
152  }
153  ss << "-----------" << std::endl;
154  mDebugMsg += ss.str();
155  stats[1] += lastCheckedUL;
156  lastCheckedUL = ulItem->second.size();
157  lastCheckedBare = bareItem.second.size();
158  isOk = false;
159  break;
160  }
161  lastOk = lastCheckedUL;
162  ++lastCheckedUL;
163  } // loop on bare data for this board ID
164  } // loop on board IDs
165  return isOk;
166 }

169 {

179 bool UserLogicChecker::checkAll(gsl::span<const ROBoard> bareData, gsl::span<const ROFRecord> bareRofs, gsl::span<const ROBoard> ulData, gsl::span<const ROFRecord> ulRofs)
180 {
181  auto bareIndexes = getOrderedIndexes(bareRofs);
182  auto ulIndexes = getOrderedIndexes(ulRofs);
183 
184  bool isOk = true;
185  std::stringstream ss;
186  InteractionRecord ir;
187  for (auto& bareItem : bareIndexes) {
188  auto ulItem = ulIndexes.find(bareItem.first);
189  ir.setFromLong(bareItem.first);
190  if (ulItem == ulIndexes.end()) {
191  isOk = false;
192  ss << "\nCannot find: " << printIRHex(ir) << " in ul\n";
193  continue;
194  }
195  std::vector<size_t> auxVec = ulItem->second;
196  for (auto& idx1 : bareItem.second) {
197  bool found = false;
198  for (auto auxIt = auxVec.begin(); auxIt != auxVec.end(); ++auxIt) {
199  if (isSame(bareData[idx1], ulData[*auxIt])) {
200  auxVec.erase(auxIt);
201  found = true;
202  break;
203  }
204  }
205  if (!found) {
206  isOk = false;
207  ss << "\nOnly in bare: " << printIRHex(ir) << "\n";
208  ss << " " << bareData[idx1] << "\n";
209  }
210  }
211  for (auto& idx2 : auxVec) {
212  isOk = false;
213  ss << "\nOnly in ul: " << printIRHex(ir) << "\n";
214  ss << " " << ulData[idx2] << "\n";
215  }
216  }
217 
218  for (auto& ulItem : ulIndexes) {
219  auto bareItem = bareIndexes.find(ulItem.first);
220  if (bareItem == bareIndexes.end()) {
221  isOk = false;
222  ir.setFromLong(ulItem.first);
223  ss << "\nCannot find: " << printIRHex(ir) << " in bare\n";
224  }
225  }
226  mDebugMsg = ss.str();
227  return isOk;
228 }

244 std::string UserLogicChecker::getSummary() const
245 {
247  std::stringstream ss;
248  for (auto& statItem : mStatistics) {
249  std::string boardType = (statItem.first < 0x10000) ? "Loc" : "Reg";
250  double badFraction = (statItem.second[0] == 0) ? 0. : static_cast<double>(statItem.second[1]) / static_cast<double>(statItem.second[0]);
251  uint16_t boardId = (statItem.first & 0xFFFF);
252  ss << "Crate: " << static_cast<int>(raw::getCrateId(boardId)) << " " << boardType << " board: " << static_cast<int>(raw::getLocId(boardId)) << " fraction of events not in sync: " << statItem.second[1] << " / " << statItem.second[0] << " = " << badFraction << std::endl;
253  }
254  return ss.str();
255 }
