77  template <typename Predicate, typename Inserter>
78  void binary(Predicate pred, Inserter inserter)
79  {
80  for (auto const& ref : mInput) {
81  auto size = DataRefUtils::getPayloadSize(ref);
82  auto const pageSize = rawparser_type::max_size;
83  auto nPages = size / pageSize + (size % pageSize ? 1 : 0);
84  if (nPages == 0) {
85  continue;
86  }
87  // FIXME: automatic type from inserter/predicate?
88  const char* iterator = ref.payload;
89 
90  auto check = [&pred, &pageSize, payload = ref.payload](size_t left, size_t right) -> bool {
91  return pred(payload + left * pageSize, payload + right * pageSize);
92  };
93  auto insert = [&inserter, &pageSize, payload = ref.payload](size_t pos, size_t n) -> void {
94  inserter(payload + pos * pageSize, n);
95  };
96  // binary search the next different page based on the check predicate
97  auto search = [&check](size_t first, size_t n) -> size_t {
98  auto count = n;
99  auto pos = first;
100  while (count > 0) {
101  auto step = count / 2;
102  if (check(first, pos + step)) {
103  // still the same
104  pos += step;
105  count = n - (pos - first);
106  } else {
107  if (step == 1) {
108  pos += step;
109  break;
110  }
111  count = step;
112  }
113  }
114  return pos;
115  };
116 
117  size_t p = 0;
118  do {
119  // insert the full block if the last RDH matches the position
120  if (check(p, nPages - 1)) {
121  insert(p, nPages - p);
122  break;
123  }
124  auto q = search(p, nPages - p);
125  insert(p, q - p);
126  p = q;
127  } while (p < nPages);
128  // if payloads are consecutive in memory we could apply this algorithm even over
129  // O2 message boundaries
130  }
131  }
132 
133  template <typename Predicate, typename Inserter>
134  void forward(Predicate check, Inserter inserter)
135  {
136  for (auto const& ref : mInput) {
137  auto size = DataRefUtils::getPayloadSize(ref);
138  o2::framework::RawParser parser(ref.payload, size);
139  const char* ptr = nullptr;
140  int count = 0;
141  for (auto it = parser.begin(); it != parser.end(); it++) {
142  const char* current = reinterpret_cast<const char*>(it.raw());
143  if (ptr == nullptr) {
144  ptr = current;
145  } else if (check(ptr, current) == false) {
146  if (count) {
147  inserter(ptr, count);
148  }
149  count = 0;
150  ptr = current;
151  }
152  count++;
153  }
154  if (count) {
155  inserter(ptr, count);
156  }
157  }
158  }
159 
160  private:
161  InputRecordWalker mInput;
162 };
