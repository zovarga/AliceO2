54 Mapping::ErrorStatus Mapping::hwToAbsId(short ddl, short hwAddr, short& absId, CaloFlag& caloFlag) const
55 {
56 
57  if (!mInitialized) {
58  LOG(error) << "Mapping not initialized";
59  return kNotInitialized;
60  }
61 
62  if (ddl < 0 || ddl > 14) {
63  return kWrongDDL;
64  }
65  if ((hwAddr >= 112 && hwAddr < 128) || (hwAddr >= 2159 && hwAddr < 2176)) { //TRU flags
66  caloFlag = kTRU;
67  absId = -1;
68  return kOK;
69  }
70  if (hwAddr < 0 || hwAddr >= NMaxHWAddress) {
71  return kWrongHWAddress;
72  }
73 
74  //transform
75  absId = mAbsId[ddl][hwAddr];
76  caloFlag = mCaloFlag[ddl][hwAddr];
77 
78  if (absId > NCHANNELS || absId <= 1792) {
79  absId = 0;
80  return kWrongHWAddress;
81  }
82  return kOK;
83 }

85 Mapping::ErrorStatus Mapping::absIdTohw(short absId, short caloFlag, short& ddl, short& hwAddr) const
86 {
87 
88  if (caloFlag < 0 || caloFlag > 2) {
89  ddl = 0;
90  hwAddr = 0;
91  return kWrongCaloFlag;
92  }
93  if (caloFlag < 2) {
94  if (absId <= 1792 || absId > NCHANNELS) {
95  ddl = 0;
96  hwAddr = 0;
97  return kWrongAbsId;
98  }
99  } else {
100  if (absId < 0 || absId > NTRUReadoutChannels) {
101  ddl = 0;
102  hwAddr = 0;
103  return kWrongAbsId;
104  }
105  }
106 
107  if (!mInitialized) {
108  LOG(error) << "Mapping not initialized";
109  return kNotInitialized;
110  }
111 
112  ddl = mAbsToHW[absId - 1][caloFlag][0];
113  hwAddr = mAbsToHW[absId - 1][caloFlag][1];
114  return kOK;
115 }

117 Mapping::ErrorStatus Mapping::setMapping()
118 {
119  //Read mapping from data files a-la Run2
120 
121  std::string p;
122  if (mPath.empty()) { //use default path
123  p = gSystem->Getenv("O2_ROOT");
124  p += "/share/Detectors/PHOS/files";
125  } else {
126  p = mPath.data();
127  }
128 
129  for (short m = 0; m < 4; m++) { //modules
130  for (short i = 0; i < 4; i++) { //RCU
131  if (m == 0 && (i < 2)) {
132  continue; //half of module: only RCU 2,3
133  }
134 
135  short numberOfChannels = 0;
136  short maxHWAddress = 0;
137  std::string fname = fmt::format("{:s}/Mod{:d}RCU{:d}.data", p, m, i);
138  std::ifstream fIn(fname);
139  if (!fIn.is_open()) {
140  LOG(fatal) << "Missing mapping file " << p << "/Mod" << m << "RCU" << i << ".data";
141  return kNotInitialized;
142  }
143  if (!(fIn >> numberOfChannels)) {
144  LOG(fatal) << "Syntax of mapping file " << p << "/Mod" << m << "RCU" << i << ".data is wrong: no numberOfChannels";
145  return kNotInitialized;
146  }
147  if (numberOfChannels != NHWPERDDL) {
148  LOG(fatal) << "Unexpected number of channels: " << numberOfChannels << " expecting " << NHWPERDDL << " file " << p << "/Mod" << m << "RCU" << i << ".data is wrong: no numberOfChannels";
149  return kNotInitialized;
150  }
151  if (!(fIn >> maxHWAddress)) {
152  LOG(fatal) << "Syntax of mapping file " << p << "/Mod" << m << "RCU" << i << ".data is wrong: no maxHWAddress";
153  return kNotInitialized;
154  }
155  if (maxHWAddress > NMaxHWAddress) {
156  LOG(fatal) << "Maximal HW address in file " << maxHWAddress << "larger than array size " << NMaxHWAddress << "for /Mod" << m << "RCU" << i << ".data is wrong: no maxHWAddress";
157  return kNotInitialized;
158  }
159  for (short ich = 0; ich < numberOfChannels; ich++) { // 1792 = 2*896 channels connected to each RCU
160  int hwAddress;
161  if (!(fIn >> hwAddress)) {
162  LOG(fatal) << "Syntax of mapping file " << p << "/Mod" << m << "RCU" << i << ".data is wrong: no HWadd for ch " << ich;
163  return kNotInitialized;
164  }
165  if (hwAddress > maxHWAddress) {
166  LOG(fatal) << "Hardware (ALTRO) adress (" << hwAddress << ") outside the range (0 -> " << maxHWAddress << ") !";
167  return kNotInitialized;
168  }
169  int row, col, caloFlag;
170  if (!(fIn >> row >> col >> caloFlag)) {
171  LOG(fatal) << "Syntax of mapping file " << p << "/Mod" << m << "RCU" << i << ".data is wrong: no (raw col caloFlag)";
172  return kNotInitialized;
173  }
174 
175  if (caloFlag < 0 || caloFlag > 2) {
176  LOG(fatal) << "Wrong CaloFlag value found (" << caloFlag << "). Should be 0, 1, 2 !";
177  return kNotInitialized;
178  }
179 
180  //convert ddl, col,raw caloFlag to AbsId
181  // Converts the absolute numbering into the following array
182  // relid[0] = PHOS Module number
183  // relid[1] = Row number inside a PHOS module (Phi coordinate)
184  // relid[2] = Column number inside a PHOS module (Z coordinate)
185  short ddl = 4 * m + i - 2;
186  if (ddl < 0 || ddl >= NDDL) {
187  LOG(fatal) << "Wrong ddl address found (" << ddl << "). Module= " << m << " RCU =" << i;
188  return kNotInitialized;
189  }
190 
191  short absId;
192  if (caloFlag < 2) { //readout channels
193  char relid[3] = {static_cast<char>(m + 1), static_cast<char>(row + 1), static_cast<char>(col + 1)};
194  Geometry::relToAbsNumbering(relid, absId);
195  } else { //TRU channels
196  if (isTRUReadoutchannel(hwAddress)) {
197  if (hwAddress < 2048) { //branch 28<=z<56
198  absId = ddl * 2 * NTRUBranchReadoutChannels + hwAddress;
199  } else { //branch 0<=z<28
200  absId = (ddl * 2 + 1) * NTRUBranchReadoutChannels + hwAddress - 2048;
201  }
202  } else { //TRU flag channels, no absId
203  continue;
204  }
205  }
206 
207  mAbsId[ddl][hwAddress] = absId;
208  mCaloFlag[ddl][hwAddress] = (CaloFlag)caloFlag;
209  mAbsToHW[absId - 1][caloFlag][0] = ddl;
210  mAbsToHW[absId - 1][caloFlag][1] = hwAddress;
211  }
212  fIn.close();
213  } //RCU
214  } // module
215  mInitialized = true;
216  return kOK;
217 }
