99  // BC difference wrt previous if in the same orbit, otherwise the abs.value.
100  // For the very 1st entry return 0 (diff wrt 1st BC in the CTF header)
101  class Iter_bcIncTrig : public _Iter<Iter_bcIncTrig, CTPDigit, uint16_t>
102  {
103  public:
104  using _Iter<Iter_bcIncTrig, CTPDigit, uint16_t>::_Iter;
105  value_type operator*() const
106  {
107  if (mIndex) {
108  if (mData[mIndex].intRecord.orbit == mData[mIndex - 1].intRecord.orbit) {
109  return mData[mIndex].intRecord.bc - mData[mIndex - 1].intRecord.bc;
110  } else {
111  return mData[mIndex].intRecord.bc;
112  }
113  }
114  return 0;
