96  static int getORIinSuperModule(int detector, int readoutboard)
97  {
98  //given a detector and readoutboard
99  int ori = -1;
100  int trdstack = HelperMethods::getStack(detector);
101  int trdlayer = HelperMethods::getLayer(detector);
102  int side = HelperMethods::getROBSide(readoutboard);
103  //TODO ccdb lookup of detector/stack/layer/side for link id.
104  bool aside = false;
105  if (trdstack == 0 || trdstack == 1) {
106  aside = true; //aside
107  } else {
108  if (trdstack != 2) {
109  aside = false; //cside
110  } else {
111  if (side == 0) {
112  aside = true; //stack
113  } else {
114  aside = false; //stack2, halfchamber 1
115  }
116  }
117  }
118  if (aside) {
119  ori = trdstack * 12 + (5 - trdlayer + side * 5) + trdlayer / 6 + side; // <- that is correct for A side at least for now, probably not for very long LUT as that will come form CCDB ni anycase.
120  } else {
121  //cside
122  int newside = side;
123  if (trdstack == 2) {
124  newside = 0; // the last part of C side CRU is a special case.
125  }
126  ori = (4 - trdstack) * 12 + (5 - trdlayer + newside * 5) + trdlayer / 6 + newside;
127  ori += 30; // 30 to offset if from the a side link , 69 links in total
128  }
129  //see TDP for explanation of mapping TODO should probably come from CCDB
130  return ori;
131  }

133  static int getLinkIDfromHCID(int hcid)
134  {
135  //return a number in range [0:29] for the link related to this hcid with in its respective CRU
136  //lower 15 is endpoint 0 and upper 15 is endpoint 1
137  //a side has 30, c side has 30 to give 60 links for a supermodule
138  int detector = hcid / 2;
139  int supermodule = hcid / 60;
140  int chamberside = hcid % 2; // 0 for side 0, 1 for side 1;
141  int ori = -1;
142  // now offset for supermodule (+60*supermodule);
143  return HelperMethods::getORIinSuperModule(detector, chamberside) + 60 * supermodule; // it takes readoutboard but only cares if its odd or even hence side here.
144  }
