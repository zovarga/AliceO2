77 template <typename DigitType, typename TCMDataType>
78 auto ConvertDigit2TCMData(const DigitType& digit, TCMDataType& tcmData) -> std::enable_if_t<!IsFV0<DigitType>::value>
79 {
80  tcmData.orA = digit.mTriggers.getOrA();
81  tcmData.orC = digit.mTriggers.getOrC();
82  tcmData.sCen = digit.mTriggers.getSCen();
83  tcmData.cen = digit.mTriggers.getCen();
84  tcmData.vertex = digit.mTriggers.getVertex();
85  tcmData.laser = bool(digit.mTriggers.triggersignals & (1 << 5));
86  tcmData.outputsAreBlocked = bool(digit.mTriggers.triggersignals & (1 << 6));
87  tcmData.dataIsValid = bool(digit.mTriggers.triggersignals & (1 << 7));
88  tcmData.nChanA = digit.mTriggers.nChanA;
89  tcmData.nChanC = digit.mTriggers.nChanC;
90  const int64_t thresholdSignedInt17bit = 65535; //pow(2,17)/2-1
91  if (digit.mTriggers.amplA > thresholdSignedInt17bit) {
92  tcmData.amplA = thresholdSignedInt17bit;
93  } else {
94  tcmData.amplA = digit.mTriggers.amplA;
95  }
96  if (digit.mTriggers.amplC > thresholdSignedInt17bit) {
97  tcmData.amplC = thresholdSignedInt17bit;
98  } else {
99  tcmData.amplC = digit.mTriggers.amplC;
100  }
101  tcmData.timeA = digit.mTriggers.timeA;
102  tcmData.timeC = digit.mTriggers.timeC;
103 }

130 template <typename DigitType, typename TCMDataType>
131 auto ConvertTCMData2Digit(DigitType& digit, const TCMDataType& tcmData) -> std::enable_if_t<!IsFV0<DigitType>::value>
132 {
133  using TriggerType = decltype(digit.mTriggers);
134  auto& trg = digit.mTriggers;
135  trg.triggersignals = ((bool)tcmData.orA << TriggerType::bitA) |
136  ((bool)tcmData.orC << TriggerType::bitC) |
137  ((bool)tcmData.vertex << TriggerType::bitVertex) |
138  ((bool)tcmData.cen << TriggerType::bitCen) |
139  ((bool)tcmData.sCen << TriggerType::bitSCen) |
140  ((bool)tcmData.laser << 5) |
141  ((bool)tcmData.outputsAreBlocked << 6) |
142  ((bool)tcmData.dataIsValid << 7);
143  trg.nChanA = (int8_t)tcmData.nChanA;
144  trg.nChanC = (int8_t)tcmData.nChanC;
145  trg.amplA = (int32_t)tcmData.amplA;
146  trg.amplC = (int32_t)tcmData.amplC;
147  trg.timeA = (int16_t)tcmData.timeA;
148  trg.timeC = (int16_t)tcmData.timeC;
149 }

262  //Decompose digits into DataBlocks
263  //DataBlockPM
264  template <class DataBlockType>
265  auto decomposeDigits() const -> std::enable_if_t<DigitBlockHelper::IsSpecOfType<DataBlockPM, DataBlockType>::value, std::map<typename LookupTable_t::Topo_t, DataBlockType>>
266  {
267  using Topo_t = typename LookupTable_t::Topo_t;
268  std::map<Topo_t, DataBlockType> mapResult;
269  std::map<Topo_t, std::reference_wrapper<const ChannelDataType>> mapTopo2SortedCh;
270  std::map<Topo_t, std::size_t> mapTopoCounter;
271  //Preparing map "Topo to ChannelData refs" and map "Global Topo(FEE metadata) to number of ChannelData"
272  for (const auto& entry : DigitBlockBase_t::mSubDigit) {
273  auto topoPM = LookupTable_t::Instance().getTopoPM(static_cast<int>(entry.getChannelID()));
274  mapTopo2SortedCh.insert({topoPM, entry});
275  auto pairInserted = mapTopoCounter.insert({LookupTable_t::makeGlobalTopo(topoPM), 0});
276  pairInserted.first->second++;
277  }
278  //Preparing map of global Topo(related to PM module) to DataBlockPMs
279  for (const auto& entry : mapTopo2SortedCh) {
280  auto pairInserted = mapResult.insert({LookupTable_t::makeGlobalTopo(entry.first), {}});
281  auto& refDataBlock = pairInserted.first->second;
282  if (pairInserted.second) {
283  //Header preparation
284  refDataBlock.DataBlockWrapper<typename DataBlockType::RawHeaderPM>::mData[0].setIntRec(DigitBlockFIThelper::GetIntRecord(DigitBlockBase_t::mDigit));
285  refDataBlock.DataBlockWrapper<typename DataBlockType::RawHeaderPM>::mData[0].startDescriptor = 0xf;
286  std::size_t nElements = mapTopoCounter.find(pairInserted.first->first)->second;
287  std::size_t nWords = nElements / 2 + nElements % 2;
288  refDataBlock.DataBlockWrapper<typename DataBlockType::RawHeaderPM>::mData[0].nGBTWords = nWords;
289  refDataBlock.DataBlockWrapper<typename DataBlockType::RawHeaderPM>::mNelements = 1;
290  }
291  //Data preparation
292  auto& refPos = refDataBlock.DataBlockWrapper<typename DataBlockType::RawDataPM>::mNelements;
293  auto& refData = refDataBlock.DataBlockWrapper<typename DataBlockType::RawDataPM>::mData[refPos];
294  refPos++;
295  DigitBlockFIThelper::ConvertChData2EventData(entry.second.get(), refData, LookupTable_t::Instance().getLocalChannelID(entry.first));
296  }
297  return mapResult;
298  }

317  //Process DigitBlocks from TTree
318  template <typename DigitBlockProcType>
319  static void processDigitBlocks(TTree* inputTree, DigitBlockProcType& digitBlockProc)
320  {
321  assert(inputTree != nullptr);
322  std::vector<DigitBlockFIT_t> vecResult;
323  std::vector<typename DigitBlockBase_t::Digit_t> vecDigit;
324  std::vector<typename DigitBlockBase_t::Digit_t>* ptrVecDigit = &vecDigit;
325  typename DigitBlockBase_t::SubDigit_t vecChannelData;
326  typename DigitBlockBase_t::SubDigit_t* ptrVecChannelData = &vecChannelData;
327  inputTree->SetBranchAddress(decltype(vecDigit)::value_type::sDigitBranchName, &ptrVecDigit);
328  inputTree->SetBranchAddress(decltype(vecChannelData)::value_type::sDigitBranchName, &ptrVecChannelData);
329  for (int iEntry = 0; iEntry < inputTree->GetEntries(); iEntry++) {
330  inputTree->GetEntry(iEntry);
331  LOG(info) << "Processing TF " << iEntry;
332  digitBlockProc.processDigitBlockPerTF(DigitBlockBase_t::template makeDigitBlock<DigitBlockFIT_t>(vecDigit, vecChannelData));
333  }
334  }
