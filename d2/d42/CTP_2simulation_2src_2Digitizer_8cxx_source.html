25 std::vector<CTPDigit> Digitizer::process(const gsl::span<o2::ctp::CTPInputDigit> detinputs)
26 {
27  std::map<o2::detectors::DetID::ID, std::vector<CTPInput>> det2ctpinp = mCTPConfiguration->getDet2InputMap();
28  // To be taken from config database ?
29  std::map<std::string, uint64_t> detInputName2Mask =
30  {{"MFV0MB", 1}, {"MFV0MBInner", 2}, {"MFV0MBOuter", 4}, {"MFV0HM", 8}, {"MFT0A", 1}, {"MFT0C", 2}, {"MFT0Vertex", 4}, {"MFT0Cent", 8}, {"MFT0SemiCent", 0x10}};
31  std::map<o2::InteractionRecord, std::vector<const CTPInputDigit*>> predigits;
32  for (auto const& inp : detinputs) {
33  predigits[inp.intRecord].push_back(&inp);
34  }
35  std::vector<CTPDigit> digits;
36  for (auto const& hits : predigits) {
37  CTPDigit data;
38  data.intRecord = hits.first;
39  std::bitset<CTP_NINPUTS> inpmaskcoll = 0;
40  for (auto const inp : hits.second) {
41  switch (inp->detector) {
42  case o2::detectors::DetID::FT0: {
43  // see dummy database above
44  for (auto const& ctpinp : det2ctpinp[o2::detectors::DetID::FT0]) {
45  uint64_t mask = (inp->inputsMask).to_ullong() & detInputName2Mask[ctpinp.name];
46  inpmaskcoll |= std::bitset<CTP_NINPUTS>(mask);
47  }
48  break;
49  }
50  case o2::detectors::DetID::FV0: {
51  for (auto const& ctpinp : det2ctpinp[o2::detectors::DetID::FV0]) {
52  uint64_t mask = (inp->inputsMask).to_ullong() & detInputName2Mask[ctpinp.name];
53  inpmaskcoll |= std::bitset<CTP_NINPUTS>(mask);
54  }
55  break;
56  }
57  default:
58  // Error
59  LOG(fatal) << "CTP Digitizer: unknown detector:" << inp->detector;
60  break;
61  }
62  }
63  data.CTPInputMask = inpmaskcoll;
64  calculateClassMask(inpmaskcoll, data.CTPClassMask);
65  digits.emplace_back(data);
66  }
67  return std::move(digits);
68 }
69 void Digitizer::calculateClassMask(const std::bitset<CTP_NINPUTS> ctpinpmask, std::bitset<CTP_NCLASSES>& classmask)
70 {
71  classmask = 0;
72  for (auto const& tcl : mCTPConfiguration->getCTPClasses()) {
73  if (tcl.descriptor->getInputsMask() & ctpinpmask.to_ullong()) {
74  classmask |= (1 << tcl.classMask);
75  }
76  }
