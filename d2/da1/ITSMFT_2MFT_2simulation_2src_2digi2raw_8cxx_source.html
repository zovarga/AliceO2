104 void digi2raw(std::string_view inpName, std::string_view outDir, std::string_view fileFor, int verbosity, uint32_t rdhV, bool noEmptyHBF, int superPageSizeInB)
105 {
106  TStopwatch swTot;
107  swTot.Start();
108  using ROFR = o2::itsmft::ROFRecord;
109  using ROFRVEC = std::vector<o2::itsmft::ROFRecord>;
110  const uint8_t ruSWMin = 0, ruSWMax = 0xff; // seq.ID of 1st and last RU (stave) to convert
111 
112  LOG(info) << "HBFUtil settings:";
113  o2::raw::HBFUtils::Instance().print();
114 
115  // if needed, create output directory
116  if (!std::filesystem::exists(outDir)) {
117  if (!std::filesystem::create_directories(outDir)) {
118  LOG(fatal) << "could not create output directory " << outDir;
119  } else {
120  LOG(info) << "created output directory " << outDir;
121  }
122  }
123 
125  std::string digTreeName{o2::base::NameConf::MCTTREENAME.data()};
126  TChain digTree(digTreeName.c_str());
127  digTree.AddFile(inpName.data());
128  digTree.SetBranchStatus("*MCTruth*", 0); // ignore MC info
129 
130  std::vector<o2::itsmft::Digit> digiVec, *digiVecP = &digiVec;
131  std::string digBranchName = o2::utils::Str::concat_string(MAP::getName(), "Digit");
132  if (!digTree.GetBranch(digBranchName.c_str())) {
133  LOG(fatal) << "Failed to find the branch " << digBranchName << " in the tree " << digTreeName;
134  }
135  digTree.SetBranchAddress(digBranchName.c_str(), &digiVecP);
136 
137  // ROF record entries in the digit tree
138  ROFRVEC rofRecVec, *rofRecVecP = &rofRecVec;
139  std::string rofRecName = o2::utils::Str::concat_string(MAP::getName(), "DigitROF");
140  if (!digTree.GetBranch(rofRecName.c_str())) {
141  LOG(fatal) << "Failed to find the branch " << rofRecName << " in the tree " << digTreeName;
142  }
143  digTree.SetBranchAddress(rofRecName.c_str(), &rofRecVecP);
145  std::string inputGRP = o2::base::NameConf::getGRPFileName();
146  const auto grp = o2::parameters::GRPObject::loadFrom(inputGRP);
147 
148  o2::itsmft::MC2RawEncoder<MAP> m2r;
149  m2r.setVerbosity(verbosity);
150  m2r.setContinuousReadout(grp->isDetContinuousReadOut(MAP::getDetID())); // must be set explicitly
151  m2r.setDefaultSinkName(o2::utils::Str::concat_string(MAP::getName(), ".raw"));
152  m2r.setMinMaxRUSW(ruSWMin, ruSWMax);
153  m2r.getWriter().setSuperPageSize(superPageSizeInB);
154  m2r.getWriter().useRDHVersion(rdhV);
155  m2r.getWriter().setDontFillEmptyHBF(noEmptyHBF);
156 
157  m2r.setVerbosity(verbosity);
158  setupLinks(m2r, outDir, MAP::getName(), fileFor);
159  //-------------------------------------------------------------------------------<<<<
160  int lastTreeID = -1;
161  long offs = 0, nEntProc = 0;
162  for (int i = 0; i < digTree.GetEntries(); i++) {
163  digTree.GetEntry(i);
164  for (const auto& rofRec : rofRecVec) {
165  int nDigROF = rofRec.getNEntries();
166  if (verbosity) {
167  LOG(info) << "Processing ROF:" << rofRec.getROFrame() << " with " << nDigROF << " entries";
168  rofRec.print();
169  }
170  if (!nDigROF) {
171  if (verbosity) {
172  LOG(info) << "Frame is empty"; // ??
173  }
174  continue;
175  }
176  nEntProc++;
177  auto dgs = nDigROF ? gsl::span<const o2::itsmft::Digit>(&digiVec[rofRec.getFirstEntry()], nDigROF) : gsl::span<const o2::itsmft::Digit>();
178  m2r.digits2raw(dgs, rofRec.getBCData());
179  }
180  } // loop over multiple ROFvectors (in case of chaining)
181 
182  m2r.getWriter().writeConfFile(MAP::getName(), "RAWDATA", o2::utils::Str::concat_string(outDir, '/', MAP::getName(), "raw.cfg"));
183  m2r.finalize(); // finish TF and flush data
184  //
185  swTot.Stop();
186  swTot.Print();
187 }

280 void setupLinks(o2::itsmft::MC2RawEncoder<MAP>& m2r, std::string_view outDir, std::string_view outPrefix, std::string_view fileFor)
281 {
282  // see the same file from ITS
283  const auto& mp = m2r.getMapping();
284  std::string outFileLink;
285  o2::itsmft::ChipMappingMFT mftMapping;
286 
287  for (int ruID = 0; ruID < mftMapping.getNRUs(); ruID++) {
288 
289  if (ruID < m2r.getRUSWMin() || ruID > m2r.getRUSWMax()) { // ignored RUs ?
290  continue;
291  }
292 
293  auto FEEId = mftMapping.RUSW2FEEId(ruID);
294  uint16_t layer, ruOnLayer, linkId, zone = FEEId & 0x3;
295  mftMapping.expandFEEId(FEEId, layer, ruOnLayer, linkId);
296 
297  m2r.getCreateRUDecode(ruID); // create RU container
298 
299  auto& ru = *m2r.getRUDecode(ruID);
300  ru.links[0] = m2r.addGBTLink();
301  uint32_t lanes = mp.getCablesOnRUType(mp.getRUType(zone, layer)); // lanes pattern of this RU
302  auto link = m2r.getGBTLink(ru.links[0]);
303  link->lanes = lanes;
304  link->feeID = mftHWMap[ruID].feeID;
305  link->idInCRU = mftHWMap[ruID].idInCRU; // linkID
306  link->cruID = mftHWMap[ruID].cruHWID; // CRU Serial Number
307  link->endPointID = mftHWMap[ruID].endpoint; // endpoint = face
308  outFileLink = o2::utils::Str::concat_string(outDir, "/", outPrefix);
309  if (fileFor != "all") { // single file for all links
310  outFileLink += fmt::format("_{}", mftHWMap[ruID].flp);
311  if (fileFor != "flp") {
312  outFileLink += fmt::format("_cru{}_{}", mftHWMap[ruID].cruHWID, link->endPointID);
313  if (fileFor != "cru") {
314  outFileLink += fmt::format("_lnk{}_feeid{}", link->idInCRU, link->feeID);
315  if (fileFor != "link") {
316  throw std::runtime_error("invalid option provided for file grouping");
317  }
318  }
319  }
320  }
321  outFileLink += ".raw";
322  m2r.getWriter().registerLink(link->feeID, link->cruID, link->idInCRU, link->endPointID, outFileLink);
323  if (m2r.getVerbosity()) {
324  LOG(info) << "RU" << ruID << '(' << mftHWMap[ruID].cruHWID << " on idInCRU " << mftHWMap[ruID].idInCRU << ") " << link->describe()
325  << " -> " << outFileLink;
326  }
327  }
