28 void RawToDigitConverterSpec::run(framework::ProcessingContext& ctx)
29 {
30  mOutputDigits.clear();
31  std::map<o2::InteractionRecord, CTPDigit> digits;
32  const gbtword80_t bcidmask = 0xfff;
33  gbtword80_t pldmask;
34  using InputSpec = o2::framework::InputSpec;
35  using ConcreteDataTypeMatcher = o2::framework::ConcreteDataTypeMatcher;
36  using Lifetime = o2::framework::Lifetime;
37  //mOutputHWErrors.clear();
38  std::vector<InputSpec> filter{InputSpec{"filter", ConcreteDataTypeMatcher{"CTP", "RAWDATA"}, Lifetime::Timeframe}};
39  o2::framework::DPLRawParser parser(ctx.inputs(), filter);
40  //setUpDummyLink
41  auto& inputs = ctx.inputs();
42  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload this means that the "delayed message"
43  // mechanism created it in absence of real data from upstream. Processor should send empty output to not block the workflow
44  {
45  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
46  std::vector<InputSpec> dummy{InputSpec{"dummy", o2::framework::ConcreteDataMatcher{"CTP", "RAWDATA", 0xDEADBEEF}}};
47  for (const auto& ref : o2::framework::InputRecordWalker(inputs, dummy)) {
48  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
49  if (dh->payloadSize == 0) {
50  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
51  if (++contDeadBeef <= maxWarn) {
52  LOGP(warning, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
53  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize,
54  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
55  }
56  ctx.outputs().snapshot(o2::framework::Output{"CTP", "DIGITS", 0, o2::framework::Lifetime::Timeframe}, mOutputDigits);
57  return;
58  }
59  }
60  contDeadBeef = 0; // if good data, reset the counter
61  }
62  //
63  uint32_t payloadCTP;
64  uint32_t orbit0 = 0;
65  bool first = true;
66  gbtword80_t remnant = 0;
67  uint32_t size_gbt = 0;
68  for (auto it = parser.begin(); it != parser.end(); ++it) {
69  auto rdh = it.get_if<o2::header::RAWDataHeader>();
70  auto triggerOrbit = o2::raw::RDHUtils::getTriggerOrbit(rdh);
71  if (first) {
72  orbit0 = triggerOrbit;
73  first = false;
74  }
75  auto feeID = o2::raw::RDHUtils::getFEEID(rdh); // 0 = IR, 1 = TCR
76  auto linkCRU = (feeID & 0xf00) >> 8;
77  if (linkCRU == o2::ctp::GBTLinkIDIntRec) {
78  payloadCTP = o2::ctp::NIntRecPayload;
79  } else if (linkCRU == o2::ctp::GBTLinkIDClassRec) {
80  payloadCTP = o2::ctp::NClassPayload;
81  } else {
82  LOG(error) << "Unxpected CTP CRU link:" << linkCRU;
83  }
84  LOG(debug) << "RDH FEEid: " << feeID << " CTP CRU link:" << linkCRU << " Orbit:" << triggerOrbit;
85  pldmask = 0;
86  for (uint32_t i = 0; i < payloadCTP; i++) {
87  pldmask[12 + i] = 1;
88  }
89  //LOG(info) << "pldmask:" << pldmask;
90  // TF in 128 bits words
91  gsl::span<const uint8_t> payload(it.data(), it.size());
92  gbtword80_t gbtWord = 0;
93  int wordCount = 0;
94  std::vector<gbtword80_t> diglets;
95  if (orbit0 != triggerOrbit) {
96  remnant = 0;
97  size_gbt = 0;
98  orbit0 = triggerOrbit;
99  }
100  for (auto payloadWord : payload) {
101  //LOG(info) << wordCount << " payload:" << int(payloadWord);
102  if (wordCount == 15) {
103  wordCount = 0;
104  } else if (wordCount > 9) {
105  wordCount++;
106  } else if (wordCount == 9) {
107  for (int i = 0; i < 8; i++) {
108  gbtWord[wordCount * 8 + i] = bool(int(payloadWord) & (1 << i));
109  }
110  wordCount++;
111  diglets.clear();
112  //LOG(info) << " gbtword:" << gbtWord;
113  makeGBTWordInverse(diglets, gbtWord, remnant, size_gbt, payloadCTP);
114  // save digit in buffer recs
115  for (auto diglet : diglets) {
116  //LOG(info) << " diglet:" << diglet;
117  //LOG(info) << " pldmas:" << pldmask;
118  gbtword80_t pld = (diglet & pldmask);
119  if (pld.count() == 0) {
120  continue;
121  }
122  //LOG(info) << " pld:" << pld;
123  pld >>= 12;
124  CTPDigit digit;
125  uint32_t bcid = (diglet & bcidmask).to_ulong();
126  o2::InteractionRecord ir;
127  ir.orbit = triggerOrbit;
128  ir.bc = bcid;
129  digit.intRecord = ir;
130  if (linkCRU == o2::ctp::GBTLinkIDIntRec) {
131  LOG(debug) << "InputMaskCount:" << digits[ir].CTPInputMask.count();
132  if (digits.count(ir) == 0) {
133  digit.setInputMask(pld);
134  digits[ir] = digit;
135  LOG(debug) << bcid << " inputs case 0 bcid orbit " << triggerOrbit << " pld:" << pld;
136  } else if (digits.count(ir) == 1) {
137  if (digits[ir].CTPInputMask.count() == 0) {
138  digits[ir].setInputMask(pld);
139  LOG(debug) << bcid << " inputs bcid vase 1 orbit " << triggerOrbit << " pld:" << pld;
140  } else {
141  LOG(error) << "Two CTP IRs with the same timestamp.";
142  }
143  } else {
144  LOG(error) << "Two digits with the same rimestamp.";
145  }
146  } else if (linkCRU == o2::ctp::GBTLinkIDClassRec) {
147  if (digits.count(ir) == 0) {
148  digit.setClassMask(pld);
149  digits[ir] = digit;
150  LOG(debug) << bcid << " class bcid case 0 orbit " << triggerOrbit << " pld:" << pld;
151  } else if (digits.count(ir) == 1) {
152  if (digits[ir].CTPClassMask.count() == 0) {
153  digits[ir].setClassMask(pld);
154  LOG(debug) << bcid << " class bcid case 1 orbit " << triggerOrbit << " pld:" << pld;
155  } else {
156  LOG(error) << "Two CTP Class masks for same timestamp";
157  }
158  } else {
159  }
160  } else {
161  LOG(error) << "Unxpected CTP CRU link:" << linkCRU;
162  }
163  }
164  gbtWord = 0;
165  } else {
166  //std::cout << "wordCount:" << wordCount << std::endl;
167  for (int i = 0; i < 8; i++) {
168  gbtWord[wordCount * 8 + i] = bool(int(payloadWord) & (1 << i));
169  //gbtWord[(9-wordCount) * 8 + i] = bool(int(payloadWord) & (1 << i));
170  }
171  wordCount++;
172  }
173  }
174  }
175  for (auto const digmap : digits) {
176  mOutputDigits.push_back(digmap.second);
177  }
178 
179  LOG(info) << "[CTPRawToDigitConverter - run] Writing " << mOutputDigits.size() << " digits ...";
180  ctx.outputs().snapshot(o2::framework::Output{"CTP", "DIGITS", 0, o2::framework::Lifetime::Timeframe}, mOutputDigits);
181  //ctx.outputs().snapshot(o2::framework::Output{"CPV", "RAWHWERRORS", 0, o2::framework::Lifetime::Timeframe}, mOutputHWErrors);
182 }
183 // Inverse of Digits2Raw::makeGBTWord
184 void RawToDigitConverterSpec::makeGBTWordInverse(std::vector<gbtword80_t>& diglets, gbtword80_t& GBTWord, gbtword80_t& remnant, uint32_t& size_gbt, uint32_t Npld) const
185 {
186  gbtword80_t diglet = remnant;
187  uint32_t i = 0;
188  while (i < (NGBT - Npld)) {
189  std::bitset<NGBT> masksize = 0;
190  for (uint32_t j = 0; j < (Npld - size_gbt); j++) {
191  masksize[j] = 1;
192  }
193  diglet |= (GBTWord & masksize) << (size_gbt);
194  diglets.push_back(diglet);
195  diglet = 0;
196  i += Npld - size_gbt;
197  GBTWord = GBTWord >> (Npld - size_gbt);
198  size_gbt = 0;
199  }
200  size_gbt = NGBT - i;
201  remnant = GBTWord;
202 }
203 o2::framework::DataProcessorSpec o2::ctp::reco_workflow::getRawToDigitConverterSpec(bool askDISTSTF)
