94 bool RawWriter::processTrigger(const gsl::span<o2::phos::Digit> digitsbranch, const o2::phos::TriggerRecord& trg)
95 {
96  auto srucont = mSRUdata.begin();
97  while (srucont != mSRUdata.end()) {
98  srucont->mChannels.clear();
99  srucont++;
100  }
101  auto trucont = mTRUdata.begin();
102  while (trucont != mTRUdata.end()) {
103  trucont->mChannels.clear();
104  trucont++;
105  }
106  for (auto& dig : gsl::span(digitsbranch.data() + trg.getFirstEntry(), trg.getNumberOfObjects())) {
107  if (dig.isTRU()) {
108  short absId = dig.getTRUId();
109  short ddl, hwAddr;
110  // get ddl and High Gain hw addresses
111  if (Mapping::Instance()->absIdTohw(absId, Mapping::kTRU, ddl, hwAddr) != o2::phos::Mapping::kOK) {
112  LOG(error) << "Wrong truId=" << absId;
113  }
114  // Collect possible several digits (signal+pileup) into one map record
115  auto celldata = mTRUdata[ddl].mChannels.find(absId);
116  if (celldata == mTRUdata[ddl].mChannels.end()) {
117  const auto it = mTRUdata[ddl].mChannels.insert(celldata, {absId, std::vector<o2::phos::Digit*>()});
118  it->second.push_back(&dig);
119  } else {
120  celldata->second.push_back(&dig);
121  }
122  } else {
123  short absId = dig.getAbsId();
124  short ddl, hwAddr;
125  // get ddl and High Gain hw addresses
126  if (Mapping::Instance()->absIdTohw(absId, Mapping::kHighGain, ddl, hwAddr) != o2::phos::Mapping::kOK) {
127  LOG(error) << "Wrong AbsId" << absId;
128  }
129 
130  // Collect possible several digits (signal+pileup) into one map record
131  auto celldata = mSRUdata[ddl].mChannels.find(absId);
132  if (celldata == mSRUdata[ddl].mChannels.end()) {
133  const auto it = mSRUdata[ddl].mChannels.insert(celldata, {absId, std::vector<o2::phos::Digit*>()});
134  it->second.push_back(&dig);
135  } else {
136  celldata->second.push_back(&dig);
137  }
138  }
139  }
140  // Create and fill DMA pages for each channel
141  std::vector<uint32_t> rawbunches;
142  std::vector<char> payload;
143  std::vector<AltroBunch> rawbunchesTRU, rawbunchesHG, rawbunchesLG;
144 
145  for (short ddl = 0; ddl < o2::phos::Mapping::NDDL; ddl++) {
146  payload.clear();
147  // Create trigger
148  // Trigger mask
149  short trmask[2 * Mapping::NTRUBranchReadoutChannels] = {0}; // Time bin in which trigger was fired.
150  for (auto ch = mTRUdata[ddl].mChannels.cbegin(); ch != mTRUdata[ddl].mChannels.cend(); ch++) {
151  short truId = ch->first;
152  short hwAddr, iddl;
153  if ((Mapping::Instance()->absIdTohw(truId, Mapping::kTRU, iddl, hwAddr) != o2::phos::Mapping::kOK) || iddl != ddl) {
154  LOG(error) << "Wrong truId=" << truId << ", iDDL=" << iddl << "!=" << ddl;
155  }
156  rawbunchesTRU.clear();
157  createTRUBunches(truId, ch->second, rawbunchesTRU);
158  rawbunches.clear();
159  for (auto& bunch : rawbunchesTRU) {
160  rawbunches.push_back(bunch.mADCs.size() + 2);
161  rawbunches.push_back(bunch.mStarttime);
162  for (auto adc : bunch.mADCs) {
163  rawbunches.push_back(adc);
164  }
165  trmask[truId % (2 * Mapping::NTRUBranchReadoutChannels)] = bunch.mStarttime + 1; // need last tile (inverse time order)
166  }
167  if (rawbunches.size() == 0) {
168  continue;
169  }
170  auto encodedbunches = encodeBunchData(rawbunches);
171  ChannelHeader chanhead = {0};
172  chanhead.mHardwareAddress = hwAddr;
173  chanhead.mPayloadSize = rawbunches.size();
174  chanhead.mMark = 1; // mark channel header
175  char* chanheadwords = reinterpret_cast<char*>(&chanhead.mDataWord);
176  for (int iword = 0; iword < sizeof(ChannelHeader) / sizeof(char); iword++) {
177  payload.emplace_back(chanheadwords[iword]);
178  }
179  char* channelwords = reinterpret_cast<char*>(encodedbunches.data());
180  for (auto iword = 0; iword < encodedbunches.size() * sizeof(int) / sizeof(char); iword++) {
181  payload.emplace_back(channelwords[iword]);
182  }
183  }
184  if (mTRUdata[ddl].mChannels.size()) { // if there are TRU digits, fill trigger flags
185  std::vector<uint32_t> a;
186  for (short chan = 0; chan < Mapping::NTRUBranchReadoutChannels; chan++) {
187  if (trmask[chan] > 0) {
188  while (a.size() < trmask[chan]) {
189  a.push_back(0);
190  }
191  a[trmask[chan] - 1] |= (1 << (chan % 10)); // Fill mask for a given channel
192  }
193  if (chan % 10 == 9 || chan + 1 == Mapping::NTRUBranchReadoutChannels) {
194  auto encodedbunches = encodeBunchData(a);
195  ChannelHeader chanhead = {0};
196  chanhead.mHardwareAddress = 112 + chan / 10;
197  chanhead.mPayloadSize = a.size();
198  chanhead.mMark = 1; // mark channel header
199  char* chanheadwords = reinterpret_cast<char*>(&chanhead.mDataWord);
200  for (int iword = 0; iword < sizeof(ChannelHeader) / sizeof(char); iword++) {
201  payload.emplace_back(chanheadwords[iword]);
202  }
203  char* channelwords = reinterpret_cast<char*>(encodedbunches.data());
204  for (auto iword = 0; iword < encodedbunches.size() * sizeof(int) / sizeof(char); iword++) {
205  payload.emplace_back(channelwords[iword]);
206  }
207  a.clear();
208  }
209  }
210  // second branch
211  for (short i = 0; i < Mapping::NTRUBranchReadoutChannels; i++) {
212  short chan = i + Mapping::NTRUBranchReadoutChannels;
213  if (trmask[chan] > 0) {
214  while (a.size() < trmask[chan]) {
215  a.push_back(0);
216  }
217  a[trmask[chan] - 1] |= (1 << (i % 10)); // Fill mask for a given channel
218  }
219  if (i % 10 == 9 || i + 1 == Mapping::NTRUBranchReadoutChannels) {
220  auto encodedbunches = encodeBunchData(a);
221  ChannelHeader chanhead = {0};
222  chanhead.mHardwareAddress = 2048 + 112 + i / 10;
223  chanhead.mPayloadSize = a.size();
224  chanhead.mMark = 1; // mark channel header
225  char* chanheadwords = reinterpret_cast<char*>(&chanhead.mDataWord);
226  for (int iword = 0; iword < sizeof(ChannelHeader) / sizeof(char); iword++) {
227  payload.emplace_back(chanheadwords[iword]);
228  }
229  char* channelwords = reinterpret_cast<char*>(encodedbunches.data());
230  for (auto iword = 0; iword < encodedbunches.size() * sizeof(int) / sizeof(char); iword++) {
231  payload.emplace_back(channelwords[iword]);
232  }
233  a.clear();
234  }
235  }
236  }
237 
238  for (auto ch = mSRUdata[ddl].mChannels.cbegin(); ch != mSRUdata[ddl].mChannels.cend(); ch++) {
239  // Find out hardware address of the channel
240  bool isLGfilled = 0;
241  createRawBunches(ch->first, ch->second, rawbunchesHG, rawbunchesLG, isLGfilled);
242 
243  short hwAddrHG; // High gain always filled
244  if (Mapping::Instance()->absIdTohw(ch->first, Mapping::kHighGain, ddl, hwAddrHG) != o2::phos::Mapping::kOK) {
245  LOG(error) << "Wrong AbsId" << ch->first;
246  }
247  rawbunches.clear();
248  for (auto& bunch : rawbunchesHG) {
249  rawbunches.push_back(bunch.mADCs.size() + 2);
250  rawbunches.push_back(bunch.mStarttime);
251  for (auto adc : bunch.mADCs) {
252  rawbunches.push_back(adc);
253  }
254  }
255  if (rawbunches.size() == 0) {
256  continue;
257  }
258  auto encodedbunches = encodeBunchData(rawbunches);
259  ChannelHeader chanhead = {0};
260  chanhead.mHardwareAddress = hwAddrHG;
261  chanhead.mPayloadSize = rawbunches.size();
262  chanhead.mMark = 1; // mark channel header
263  char* chanheadwords = reinterpret_cast<char*>(&chanhead.mDataWord);
264  for (int iword = 0; iword < sizeof(ChannelHeader) / sizeof(char); iword++) {
265  payload.emplace_back(chanheadwords[iword]);
266  }
267 
268  char* channelwords = reinterpret_cast<char*>(encodedbunches.data());
269  for (auto iword = 0; iword < encodedbunches.size() * sizeof(int) / sizeof(char); iword++) {
270  payload.emplace_back(channelwords[iword]);
271  }
272 
273  if (isLGfilled) { // fill both HighGain, and LowGain channels in case of saturation
274  short hwAddrLG; // High gain always filled
275  if (Mapping::Instance()->absIdTohw(ch->first, 1, ddl, hwAddrLG) != o2::phos::Mapping::kOK) {
276  LOG(error) << "Wrong AbsId" << ch->first;
277  }
278 
279  rawbunches.clear();
280  for (auto& bunch : rawbunchesLG) {
281  rawbunches.push_back(bunch.mADCs.size() + 2);
282  rawbunches.push_back(bunch.mStarttime);
283  for (auto adc : bunch.mADCs) {
284  rawbunches.push_back(adc);
285  }
286  }
287 
288  encodedbunches = encodeBunchData(rawbunches);
289  ChannelHeader chanheadLG = {0};
290  chanheadLG.mHardwareAddress = hwAddrLG;
291  chanheadLG.mPayloadSize = rawbunches.size();
292  chanheadLG.mMark = 1; // mark channel header
293 
294  chanheadwords = reinterpret_cast<char*>(&chanheadLG.mDataWord);
295  for (int iword = 0; iword < sizeof(ChannelHeader) / sizeof(char); iword++) {
296  payload.emplace_back(chanheadwords[iword]);
297  }
298  channelwords = reinterpret_cast<char*>(encodedbunches.data());
299  for (auto iword = 0; iword < encodedbunches.size() * sizeof(int) / sizeof(char); iword++) {
300  payload.emplace_back(channelwords[iword]);
301  }
302  }
303  }
304 
305  // Create RCU trailer
306  auto trailerwords = createRCUTrailer(payload.size() / 4, 16, 16, 100., 0.);
307  for (auto word : trailerwords) {
308  payload.emplace_back(word);
309  }
310 
311  // register output data
312  LOG(debug1) << "Adding payload with size " << payload.size() << " (" << payload.size() / 4 << " ALTRO words)";
313 
314  short flp, crorc, link;
315  Mapping::ddlToCrorcLink(ddl, flp, crorc, link);
316  mRawWriter->addData(ddl, crorc, link, 0, trg.getBCData(), payload);
317  }
318  return true;

320 void RawWriter::createTRUBunches(short truId, const std::vector<o2::phos::Digit*>& channelDigits,
321  std::vector<o2::phos::AltroBunch>& bunchs)
322 {
323 
324  AltroBunch currentBunch;
325  std::vector<short> samples;
326  float maxAmp = 0;
327  for (auto dig : channelDigits) {
328  float ampADC = dig->getAmplitude(); // Digits amplitude already in ADC channels
329  short time = short(dig->getTime() / 25.); // digit time in nc, convert to bunch crossings (25ns), max readout time 3 mks
330  if (time > 120) {
331  time = 120;
332  }
333  if (time < 0) {
334  time = 0;
335  }
336  if (maxAmp < ampADC) {
337  currentBunch.mStarttime = time;
338  maxAmp = ampADC;
339  }
340  while (samples.size() <= time) {
341  samples.push_back(0);
342  }
343  samples[time] = ampADC;
344  }
345 
346  // Note reverse time order
347  for (int i = samples.size(); i--;) {
348  currentBunch.mADCs.emplace_back(samples[i]);
349  }
350  bunchs.push_back(currentBunch);
351 }

353 void RawWriter::createRawBunches(short absId, const std::vector<o2::phos::Digit*>& channelDigits, std::vector<o2::phos::AltroBunch>& bunchHG,
354  std::vector<o2::phos::AltroBunch>& bunchLG, bool& isLGFilled)
355 {
356 
357  isLGFilled = false;
358  short samples[kNPHOSSAMPLES] = {0};
359  float hglgratio = mCalibParams->getHGLGRatio(absId);
360  for (auto dig : channelDigits) {
361  // Convert energy and time to ADC counts and time ticks
362  float ampADC = dig->getAmplitude(); // Digits amplitude already in ADC channels
363  if (!dig->isHighGain() || ampADC > o2::phos::PHOSSimParams::Instance().mMCOverflow) { // High Gain in saturation, fill also Low Gain
364  isLGFilled = true;
365  }
366  float timeTicks = dig->getTime(); // time in ns
367  timeTicks /= o2::phos::PHOSSimParams::Instance().mTimeTick; // time in PHOS ticks
368  // Add to current sample contribution from digit
369  if (!dig->isHighGain()) {
370  ampADC *= hglgratio;
371  }
372  fillGamma2(ampADC, timeTicks, samples);
373  }
374 
375  // reduce samples below ZS and fill output
376  short zs = (short)o2::phos::PHOSSimParams::Instance().mZSthreshold;
377  bunchHG.clear();
378  AltroBunch currentBunch;
379  // Note reverse time order
380  for (int i = kNPHOSSAMPLES; i--;) {
381  if (samples[i] > zs) {
382  currentBunch.mADCs.emplace_back(std::min(o2::phos::PHOSSimParams::Instance().mMCOverflow, samples[i]));
383  } else { // end of sample?
384  if (currentBunch.mADCs.size()) {
385  currentBunch.mStarttime = i + 1;
386  bunchHG.push_back(currentBunch);
387  currentBunch.mADCs.clear();
388  }
389  }
390  }
391  if (currentBunch.mADCs.size()) {
392  bunchHG.push_back(currentBunch);
393  currentBunch.mADCs.clear();
394  }
395  if (isLGFilled) {
396  bunchLG.clear();
397  currentBunch.mADCs.clear();
398  for (int i = kNPHOSSAMPLES; i--;) {
399  if (samples[i] > zs * hglgratio) {
400  currentBunch.mADCs.emplace_back(std::min(o2::phos::PHOSSimParams::Instance().mMCOverflow, short(samples[i] / hglgratio)));
401  } else { // end of sample?
402  if (currentBunch.mADCs.size()) {
403  currentBunch.mStarttime = i + 1;
404  bunchLG.push_back(currentBunch);
405  currentBunch.mADCs.clear();
406  }
407  }
408  }
409  if (currentBunch.mADCs.size()) {
410  bunchLG.push_back(currentBunch);
411  }
412  }

415 void RawWriter::fillGamma2(float amp, float time, short* samples)
416 {
417  // Simulate Gamma2 signal added to current sample in PHOS
418  float alpha = o2::phos::PHOSSimParams::Instance().mSampleDecayTime;
419  amp += 0.5; // rounding err
420  for (int i = 0; i < kNPHOSSAMPLES; i++) {
421  if (i < time) {
422  continue;
423  }
424  float x = alpha * (i - time);
425  float y = 0.25 * amp * x * x * std::exp(2. - x); // 0.25*exp(-2) normalization to unity
426  samples[i] += short(y);
427  }
428 }

430 std::vector<uint32_t> RawWriter::encodeBunchData(const std::vector<uint32_t>& data)
431 {
432  std::vector<uint32_t> encoded;
433  CaloBunchWord currentword;
434  currentword.mDataWord = 0;
435  int wordnumber = 0;
436  for (auto adc : data) {
437  switch (wordnumber) {
438  case 0:
439  currentword.mWord0 = adc;
440  break;
441  case 1:
442  currentword.mWord1 = adc;
443  break;
444  case 2:
445  currentword.mWord2 = adc;
446  break;
447  };
448  wordnumber++;
449  if (wordnumber == 3) {
450  // start new word;
451  encoded.push_back(currentword.mDataWord);
452  currentword.mDataWord = 0;
453  wordnumber = 0;
454  }
455  }
456  if (wordnumber) {
457  encoded.push_back(currentword.mDataWord);
458  }
459  return encoded;
460 }

476 int RawWriter::carryOverMethod(const header::RDHAny* rdh, const gsl::span<char> data,
477  const char* ptr, int maxSize, int splitID,
478  std::vector<char>& trailer, std::vector<char>& header) const
479 {
480 
481  constexpr int phosTrailerSize = 36;
482  int offs = ptr - &data[0]; // offset wrt the head of the payload
483  assert(offs >= 0 && size_t(offs + maxSize) <= data.size()); // make sure ptr and end of the suggested block are within the payload
484  int leftBefore = data.size() - offs; // payload left before this splitting
485  int leftAfter = leftBefore - maxSize; // what would be left after the suggested splitting
486  int actualSize = maxSize;
487  if (leftAfter && leftAfter <= phosTrailerSize) { // avoid splitting the trailer or writing only it.
488  actualSize -= (phosTrailerSize - leftAfter) + 4; // (as we work with int, not char in decoding)
489  }
490  return actualSize;
491 }
