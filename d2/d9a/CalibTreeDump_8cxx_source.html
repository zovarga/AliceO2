71 void CalibTreeDump::addDefaultMapping(TTree* tree)
72 {
73  // loop over ROCs
74  //
75 
76  // ===| mapper |==============================================================
77  const auto& mapper = Mapper::instance();
78 
79  // ===| default mapping objects |=============================================
80  uint16_t rocNumber = 0;
81  // positions
82  std::vector<float> gx;
83  std::vector<float> gy;
84  std::vector<float> lx;
85  std::vector<float> ly;
86  // row and pad
87  std::vector<unsigned char> row;
88  std::vector<unsigned char> pad;
89  std::vector<short> cpad;
90 
91  // ===| add branches with default mappings |==================================
92  tree->Branch("roc", &rocNumber);
93  tree->Branch("gx", &gx);
94  tree->Branch("gy", &gy);
95  tree->Branch("lx", &lx);
96  tree->Branch("ly", &ly);
97  tree->Branch("row", &row);
98  tree->Branch("pad", &pad);
99  tree->Branch("cpad", &cpad);
100 
101  // ===| loop over readout chambers |==========================================
102  for (ROC roc; !roc.looped(); ++roc) {
103  rocNumber = roc;
104 
105  // ===| clear position vectors |============================================
106  gx.clear();
107  gy.clear();
108  lx.clear();
109  ly.clear();
110  row.clear();
111  pad.clear();
112  cpad.clear();
113 
114  // ===| loop over pad rows |================================================
115  const int numberOfRows = mapper.getNumberOfRowsROC(roc);
116  for (int irow = 0; irow < numberOfRows; ++irow) {
117 
118  // ===| loop over pads in row |===========================================
119  const int numberOfPadsInRow = mapper.getNumberOfPadsInRowROC(roc, irow);
120  for (int ipad = 0; ipad < numberOfPadsInRow; ++ipad) {
121  const PadROCPos padROCPos(rocNumber, irow, ipad);
122  const PadPos padPos = mapper.getGlobalPadPos(padROCPos); // pad and row in sector
123  const PadCentre& localPadXY = mapper.getPadCentre(padPos);
124  const LocalPosition2D globalPadXY = mapper.getPadCentre(padROCPos);
125 
126  gx.emplace_back(globalPadXY.X());
127  gy.emplace_back(globalPadXY.Y());
128  lx.emplace_back(localPadXY.X());
129  ly.emplace_back(localPadXY.Y());
130 
131  row.emplace_back(irow);
132  pad.emplace_back(ipad);
133  cpad.emplace_back(ipad - numberOfPadsInRow / 2);
134  }
135  }
136 
137  tree->Fill();
138  }
139 }

142 void CalibTreeDump::addFEEMapping(TTree* tree)
143 {
144  // loop over ROCs
145  //
146 
147  // ===| mapper |==============================================================
148  const auto& mapper = Mapper::instance();
149 
150  mTraceLengthIROC = mapper.getTraceLengthsIROC();
151  mTraceLengthOROC = mapper.getTraceLengthsOROC();
152 
153  // ===| default mapping objects |=============================================
154  // FEC
155  std::vector<unsigned char> fecInSector;
156  std::vector<unsigned char> sampaOnFEC;
157  std::vector<unsigned char> channelOnSampa;
158  // trace length
159  std::vector<float>* traceLength = nullptr;
160 
161  // ===| add branches with default mappings |==================================
162  auto brFecInSector = tree->Branch("fecInSector", &fecInSector);
163  auto brSampaOnFEC = tree->Branch("sampaOnFEC", &sampaOnFEC);
164  auto brChannelOnSampa = tree->Branch("channelOnSampa", &channelOnSampa);
165  auto brTraceLength = tree->Branch("traceLength", &traceLength);
166 
167  // ===| loop over readout chambers |==========================================
168  for (ROC roc; !roc.looped(); ++roc) {
169  int rocNumber = roc;
170  //tree->GetEntry(rocNumber);
171  traceLength = ((roc.rocType() == RocType::IROC) ? &mTraceLengthIROC : &mTraceLengthOROC);
172 
173  // ===| clear position vectors |============================================
174  fecInSector.clear();
175  sampaOnFEC.clear();
176  channelOnSampa.clear();
177 
178  const int rowOffset = (roc.rocType() == RocType::OROC) ? mapper.getPadsInIROC() : 0;
179  // ===| loop over pad rows |================================================
180  const int numberOfRows = mapper.getNumberOfRowsROC(roc);
181  for (int irow = 0; irow < numberOfRows; ++irow) {
182 
183  // ===| loop over pads in row |===========================================
184  const int numberOfPadsInRow = mapper.getNumberOfPadsInRowROC(roc, irow);
185  for (int ipad = 0; ipad < numberOfPadsInRow; ++ipad) {
186  const PadROCPos padROCPos(rocNumber, irow, ipad);
187  const auto& fecInfo = mapper.getFECInfo(padROCPos);
188  const auto globalPadNumber = mapper.globalPadNumber(fecInfo);
189 
190  const CRU cru(mapper.getCRU(roc.getSector(), globalPadNumber));
191 
192  fecInSector.emplace_back(fecInfo.getIndex());
193  sampaOnFEC.emplace_back(fecInfo.getSampaChip());
194  channelOnSampa.emplace_back(fecInfo.getSampaChannel());
195  }
196  }
197  brFecInSector->Fill();
198  brSampaOnFEC->Fill();
199  brChannelOnSampa->Fill();
200  brTraceLength->Fill();
201  }
202 }

205 void CalibTreeDump::addCalDetObjects(TTree* tree)
206 {
207 
208  int iter = 0;
209  for (auto pcalDet : mCalDetObjects) {
210  // ===| branch names |===
211  if (!pcalDet) {
212  continue;
213  }
214  auto& calDet = *pcalDet;
215  std::string name = calDet.getName();
216 
217  if (name == "PadCalibrationObject" || name.size() == 0) {
218  name = fmt::format("calDet_{%02d}", iter);
219  }
220 
221  std::string meanName = fmt::format("{}_mean", name);
222  std::string stdDevName = fmt::format("{}_stdDev", name);
223  std::string medianName = fmt::format("{}_median", name);
224  std::string median1Name = fmt::format("{}_median1", name);
225  std::string median2Name = fmt::format("{}_median2", name);
226  std::string median3Name = fmt::format("{}_median3", name);
227 
228  // ===| branch variables |===
229  std::vector<float>* data = nullptr;
230  float mean{};
231  float stdDev{};
232  float median[4]{};
233 
234  // ===| branch definitions |===
235  TBranch* brMean = tree->Branch(meanName.data(), &mean);
236  TBranch* brStdDev = tree->Branch(stdDevName.data(), &stdDev);
237  TBranch* brMedian = tree->Branch(medianName.data(), &median[0]);
238  TBranch* brMedian1 = tree->Branch(median1Name.data(), &median[1]);
239  TBranch* brMedian2 = tree->Branch(median2Name.data(), &median[2]);
240  TBranch* brMedian3 = tree->Branch(median3Name.data(), &median[3]);
241  TBranch* brData = tree->Branch(name.data(), &data);
242 
243  // ===| loop over ROCs and fill |===
244  int roc = 0;
245  for (auto& calArray : calDet.getData()) {
246  // ---| set data |---
247  data = &calArray.getData();
248 
249  // ---| statistics |---
250  mean = TMath::Mean(data->begin(), data->end());
251  stdDev = TMath::StdDev(data->begin(), data->end());
252  median[0] = median[1] = median[2] = median[3] = TMath::Median(data->size(), data->data());
253  if (roc > 35) {
254  median[1] = TMath::Median(Mapper::getPadsInOROC1(), data->data());
255  median[2] = TMath::Median(Mapper::getPadsInOROC2(), data->data() + Mapper::getPadsInOROC1());
256  median[3] = TMath::Median(Mapper::getPadsInOROC3(), data->data() + Mapper::getPadsInOROC1() + Mapper::getPadsInOROC2());
257  }
258 
259  // ---| filling |---
260  brData->Fill();
261  brMean->Fill();
262  brStdDev->Fill();
263  brMedian->Fill();
264  brMedian1->Fill();
265  brMedian2->Fill();
266  brMedian3->Fill();
267  ++roc;
268  }
269  }
270 }
