27 struct TPCVDTglContainer {
28  std::unique_ptr<o2::dataformats::FlatHisto2D_f> histo;
29  size_t entries = 0;
30 
31  TPCVDTglContainer(int ntgl, float tglMax, int ndtgl, float dtglMax)
32  {
33  histo = std::make_unique<o2::dataformats::FlatHisto2D_f>(ntgl, -tglMax, tglMax, ndtgl, -dtglMax, dtglMax);
34  }
35 
36  TPCVDTglContainer(const TPCVDTglContainer& src)
37  {
38  histo = std::make_unique<o2::dataformats::FlatHisto2D_f>(*(src.histo.get()));
39  entries = src.entries;
40  }
41 
42  void fill(const gsl::span<const o2::dataformats::Pair<float, float>> data)
43  {
44  for (auto p : data) {
45  histo->fill(p.first, p.first - p.second);
46  }
47  entries += data.size();
48  }
49 
50  void merge(const TPCVDTglContainer* other)
51  {
52  entries += other->entries;
53  histo->add(*(other->histo));
54  }
55 
56  void print() const
57  {
58  LOG(info) << "Nentries = " << entries;
59  }
60  ClassDefNV(TPCVDTglContainer, 1);
61 };
