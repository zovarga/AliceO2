38 {
40  ElectronicsDelay electronicsDelay;
41  std::ifstream inFile(filename);
42  if (inFile.is_open()) {
43  std::string line;
44  while (std::getline(inFile, line)) {
45  line.erase(std::remove_if(line.begin(), line.end(), [](unsigned char x) { return std::isspace(x); }), line.end());
46  auto pos = line.find(":");
47  if (pos != std::string::npos) {
48  std::string key = line.substr(0, pos);
49  uint16_t val = std::atoi(line.substr(pos + 1).c_str());
50  if (key == "calibToFET") {
51  electronicsDelay.calibToFET = val;
52  } else if (key == "BCToLocal") {
53  electronicsDelay.BCToLocal = val;
54  } else if (key == "regToLocal") {
55  electronicsDelay.regToLocal = val;
56  }
57  }
58  }
59  } else {
60  std::cout << "Error: cannot open file " << filename << std::endl;
61  }
