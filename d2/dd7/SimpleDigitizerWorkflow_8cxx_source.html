203 void customize(std::vector<o2::framework::CallbacksPolicy>& policies)
204 {
205  // we customize the time information sent in DPL headers
206  policies.push_back(o2::framework::CallbacksPolicy{
207  [](o2::framework::DeviceSpec const& spec, o2::framework::ConfigContext const& context) -> bool {
208  return true;
209  },
210  [](o2::framework::CallbackService& service, o2::framework::InitContext& context) {
211  // simple linear enumeration from already updated HBFUtils (set via config key values)
212  service.set(o2::framework::CallbackService::Id::NewTimeslice,
213  [](o2::header::DataHeader& dh, o2::framework::DataProcessingHeader& dph) {
214  const auto& hbfu = o2::raw::HBFUtils::Instance();
215  const auto offset = int64_t(hbfu.getFirstIRofTF({0, hbfu.orbitFirstSampled}).orbit);
216  const auto increment = int64_t(hbfu.nHBFPerTF);
217  const auto startTime = hbfu.startTime;
218  const auto orbitFirst = hbfu.orbitFirst;
219  dh.firstTForbit = offset + increment * dh.tfCounter;
220  LOG(info) << "Setting firstTFOrbit to " << dh.firstTForbit;
221  dph.creation = startTime + (dh.firstTForbit - orbitFirst) * o2::constants::lhc::LHCOrbitMUS * 1.e-3;
222  LOG(info) << "Setting timeframe creation time to " << dph.creation;
223  });
224  }} // end of struct
225  );
226 }

232 // extract num TPC lanes, a lane is a streaming line of processors (digitizer-clusterizer-etc)
233 // by default this will be std::max(the number of physical cores, numberofsectors)
234 // as a temporary means to fully use a machine and as a way to play with different topologies
235 int getNumTPCLanes(std::vector<int> const& sectors, ConfigContext const& configcontext)
236 {
237  auto lanes = configcontext.options().get<int>("tpc-lanes");
238  if (lanes < 0) {
239  if (gIsMaster) {
240  LOG(fatal) << "tpc-lanes needs to be positive\n";
241  }
242  return 0;
243  }
244  // crosscheck with sectors
245  return std::min(lanes, (int)sectors.size());
246 }

250 void initTPC()
251 {
252  // We only want to do this for the DPL master
253  // I am not aware of an easy way to query if "I am DPL master" so
254  // using for the moment a mechanism defining/setting an environment variable
255  // with the parent ID and query inside forks if this environment variable exists
256  // (it assumes fundamentally that the master executes this function first)
257  std::stringstream streamthis;
258  std::stringstream streamparent;
259 
260  streamthis << "TPCGEMINIT_PID" << getpid();
261  streamparent << "TPCGEMINIT_PID" << getppid();
262  if (getenv(streamparent.str().c_str())) {
263  LOG(debug) << "GEM ALREADY INITIALIZED ... SKIPPING HERE";
264  return;
265  }
266 
267  LOG(debug) << "INITIALIZING TPC GEMAmplification";
268  setenv(streamthis.str().c_str(), "ON", 1);
269 
270  auto& cdb = o2::tpc::CDBInterface::instance();
271  cdb.setUseDefaults();
272  // by invoking this constructor we make sure that a common file will be created
273  // in future we should take this from OCDB and just forward per message
274  const static auto& ampl = o2::tpc::GEMAmplification::instance();
275 }

294 // Split a given string on a separator character
295 std::vector<std::string> splitString(std::string src, char sep)
296 {
297  std::vector<std::string> fields;
298  std::string token;
299  std::istringstream ss(src);
300 
301  while (std::getline(ss, token, sep)) {
302  if (!token.empty()) {
303  fields.push_back(token);
304  }
305  }
306 
307  return fields;
308 }

311 // Filters detectors based on a white/black list provided via the onlyDet/skipDet CLI args
312 struct DetFilterer {
313  // detlist: A character-separated list of detectors
314  // unsetVal: The value when the option is unset
315  // separator: The character that separates the list of detectors defined in option
316  // mustContain: The nature of this DetFilterer. If true, it is a white lister
317  // i.e. option defines the list of allowed detectors. If false
318  // it is a black lister i.e defines the list of disallowed detectors.
319  DetFilterer(std::string detlist, std::string unsetVal, char separator, bool doWhiteListing)
320  {
321  // option is not set, nothing to do
322  if (detlist.compare(unsetVal) == 0) {
323  return;
324  }
325 
326  std::vector<std::string> tokens = splitString(detlist, separator);
327 
328  // Convert a vector of strings to one of o2::detectors::DetID
329  for (auto token : tokens) {
330  ids.emplace_back(token.c_str());
331  }
332 
333  isWhiteLister = doWhiteListing;
334  }
335 
336  // isSet determines if a detector list was provided
337  // against which to filter
338  bool isSet()
339  {
340  return ids.size() > 0;
341  }
342 
343  // accept determines if a given detector should be accepted
344  bool accept(o2::detectors::DetID id)
345  {
346  bool found = std::find(ids.begin(), ids.end(), id) != ids.end();
347  return found == isWhiteLister;
348  }
349 
350  private:
351  std::vector<o2::detectors::DetID> ids;
352  bool isWhiteLister; // true = accept only detectors in the ids vector
353 };
354 
355 // Helper function to define a white listing DetFilterer
356 DetFilterer whitelister(std::string optionVal, std::string unsetValue, char separator)
357 {
358  return DetFilterer(optionVal, unsetValue, separator, true);
359 }
360 
361 // Helper function to define a black listing DetFilterer
362 DetFilterer blacklister(std::string optionVal, std::string unsetValue, char separator)
363 {
364  return DetFilterer(optionVal, unsetValue, separator, false);
365 }

367 // Finding out if the current process is the master DPL driver process,
368 // first setting up the topology. Might be important to know when we write
369 // files (to prevent that multiple processes write the same file)
370 bool isMasterWorkflowDefinition(ConfigContext const& configcontext)
371 {
372  int argc = configcontext.argc();
373  auto argv = configcontext.argv();
374  bool ismaster = true;
375  for (int argi = 0; argi < argc; ++argi) {
376  // when channel-config is present it means that this is started as
377  // as FairMQDevice which means it is already a forked process
378  if (strcmp(argv[argi], "--channel-config") == 0) {
379  ismaster = false;
380  break;
381  }
382  }
383  return ismaster;
384 }

390 WorkflowSpec defineDataProcessing(ConfigContext const& configcontext)
391 {
392  // check if we merely construct the topology to create help options
393  // if this is the case we don't need to read from GRP
394  bool helpasked = configcontext.helpOnCommandLine();
395  bool ismaster = isMasterWorkflowDefinition(configcontext);
396  gIsMaster = ismaster;
397 
398  // Reserve one entry which fill be filled with the SimReaderSpec
399  // at the end. This places the processor at the beginning of the
400  // workflow in the upper left corner of the GUI.
401  WorkflowSpec specs(1);
402 
403  using namespace o2::conf;
404  ConfigurableParam::updateFromFile(configcontext.options().get<std::string>("configFile"));
405 
406  // Update the (declared) parameters if changed from the command line
407  // Note: In the future this should be done only on a dedicated processor managing
408  // the parameters and then propagated automatically to all devices
409  ConfigurableParam::updateFromString(configcontext.options().get<std::string>("configKeyValues"));
410  const auto& hbfu = o2::raw::HBFUtils::Instance();
411 
412  // which sim productions to overlay and digitize
413  auto simPrefixes = splitString(configcontext.options().get<std::string>("sims"), ',');
414  // First, read the GRP to detect which components need instantiations
415  std::shared_ptr<o2::parameters::GRPObject const> grp(nullptr);
416  if (!helpasked) {
417  grp = readGRP(simPrefixes[0]);
418  if (!grp) {
419  return WorkflowSpec{};
420  }
421  if (!hbfu.startTime) { // HBFUtils.startTime was not set from the command line, set it from GRP
422  hbfu.setValue("HBFUtils.startTime", std::to_string(grp->getTimeStart()));
423  }
424  }
425  auto grpfile = o2::base::NameConf::getGRPFileName(simPrefixes[0]);
426  // init on a high level, the time for the CCDB queries
427  // we expect that digitizers do not play with the manager themselves
428  // this will only be needed until digitizers take CCDB objects via DPL mechanism
429  o2::ccdb::BasicCCDBManager::instance().setTimestamp(hbfu.startTime);
430 
431  // update the digitization configuration with the right geometry file
432  // we take the geometry from the first simPrefix (could actually check if they are
433  // all compatible)
434  ConfigurableParam::setValue("DigiParams.digitizationgeometry_prefix", simPrefixes[0]);
435  ConfigurableParam::setValue("DigiParams.grpfile", grpfile);
436  LOG(info) << "MC-TRUTH " << !configcontext.options().get<bool>("disable-mc");
437  bool mctruth = !configcontext.options().get<bool>("disable-mc");
438  ConfigurableParam::setValue("DigiParams", "mctruth", mctruth);
439 
440  // write the configuration used for the digitizer workflow
441  // (In the case, in which we call multiple processes to do digitization,
442  // only one of them should write this file ... but take the complete configKeyValue line)
443  if (ismaster) {
444  if (!configcontext.options().get<bool>("disable-write-ini")) {
445  o2::conf::ConfigurableParam::writeINI(std::string(o2::base::NameConf::DIGITIZATIONCONFIGFILE));
446  }
447  }
448 
449  // onlyDet takes precedence on skipDet
450  DetFilterer filterers[2] = {
451  whitelister(configcontext.options().get<std::string>("onlyDet"), "none", ','),
452  blacklister(configcontext.options().get<std::string>("skipDet"), "none", ',')};
453 
454  auto accept = [&configcontext, &filterers](o2::detectors::DetID id) {
455  for (auto& f : filterers) {
456  if (f.isSet()) {
457  return f.accept(id);
458  }
459  }
460  // accept all if neither onlyDet/skipDet are provided
461  return true;
462  };
463 
464  // lambda to extract detectors which are enabled in the workflow
465  // will complain if user gave wrong input in construction of DetID
466  auto isEnabled = [&configcontext, &filterers, accept, grp, helpasked](o2::detectors::DetID id) {
467  if (helpasked) {
468  return true;
469  }
470  if (configcontext.options().get<bool>("only-context")) {
471  // no detector necessary if we are asked to produce only the digitization context
472  return false;
473  }
474  auto accepted = accept(id);
475  bool is_ingrp = grp->isDetReadOut(id);
476  if (gIsMaster) {
477  LOG(info) << id.getName()
478  << " is in grp? " << (is_ingrp ? "yes" : "no") << ";"
479  << " is skipped? " << (!accepted ? "yes" : "no");
480  }
481  return accepted && is_ingrp;
482  };
483 
484  std::vector<o2::detectors::DetID> detList; // list of participating detectors
485 
486  // keeps track of which tpc sectors to process
487  std::vector<int> tpcsectors;
488 
489  if (isEnabled(o2::detectors::DetID::TPC)) {
490  if (!helpasked && ismaster) {
491  initTPC();
492  }
493 
494  tpcsectors = o2::RangeTokenizer::tokenize<int>(configcontext.options().get<std::string>("tpc-sectors"));
495  // only one lane for the help printout
496  auto lanes = helpasked ? 1 : getNumTPCLanes(tpcsectors, configcontext);
497  detList.emplace_back(o2::detectors::DetID::TPC);
498 
499  auto internalwrite = configcontext.options().get<bool>("tpc-chunked-writer");
500  WorkflowSpec tpcPipelines = o2::tpc::getTPCDigitizerSpec(lanes, tpcsectors, mctruth, internalwrite);
501  specs.insert(specs.end(), tpcPipelines.begin(), tpcPipelines.end());
502 
503  if (configcontext.options().get<std::string>("tpc-reco-type").empty() == false) {
504  throw std::runtime_error("option 'tpc-reco-type' is deprecated, please connect workflows on the command line by pipe");
505  }
506  if (!internalwrite) {
507  // for writing digits to disc
508  specs.emplace_back(o2::tpc::getTPCDigitRootWriterSpec(tpcsectors, mctruth));
509  }
510  }
511 
512  // first 36 channels are reserved for the TPC
513  const int firstOtherChannel = 36;
514  int fanoutsize = firstOtherChannel;
515 
516  // the ITS part
517  if (isEnabled(o2::detectors::DetID::ITS)) {
518  detList.emplace_back(o2::detectors::DetID::ITS);
519  // connect the ITS digitization
520  specs.emplace_back(o2::itsmft::getITSDigitizerSpec(fanoutsize++, mctruth));
521  // connect ITS digit writer
522  specs.emplace_back(o2::itsmft::getITSDigitWriterSpec(mctruth));
523  }
524 
525 #ifdef ENABLE_UPGRADES
526  // the ITS3 part
527  if (isEnabled(o2::detectors::DetID::IT3)) {
528  detList.emplace_back(o2::detectors::DetID::IT3);
529  // connect the ITS digitization
530  specs.emplace_back(o2::its3::getITS3DigitizerSpec(fanoutsize++, mctruth));
531  // // connect ITS digit writer
532  specs.emplace_back(o2::its3::getITS3DigitWriterSpec(mctruth));
533  }
534 #endif
535 
536  // the MFT part
537  if (isEnabled(o2::detectors::DetID::MFT)) {
538  detList.emplace_back(o2::detectors::DetID::MFT);
539  // connect the MFT digitization
540  specs.emplace_back(o2::itsmft::getMFTDigitizerSpec(fanoutsize++, mctruth));
541  // connect MFT digit writer
542  specs.emplace_back(o2::itsmft::getMFTDigitWriterSpec(mctruth));
543  }
544 
545  // the TOF part
546  if (isEnabled(o2::detectors::DetID::TOF)) {
547  auto useCCDB = configcontext.options().get<bool>("use-ccdb-tof");
548  auto ccdb_url_tof = o2::base::NameConf::getCCDBServer();
549  auto timestamp = configcontext.options().get<int>("timestamp-tof");
550  detList.emplace_back(o2::detectors::DetID::TOF);
551  // connect the TOF digitization
552  specs.emplace_back(o2::tof::getTOFDigitizerSpec(fanoutsize++, useCCDB, mctruth, ccdb_url_tof.c_str(), timestamp));
553  // add TOF digit writer
554  specs.emplace_back(o2::tof::getTOFDigitWriterSpec(mctruth));
555  }
556 
557  // the FT0 part
558  if (isEnabled(o2::detectors::DetID::FT0)) {
559  detList.emplace_back(o2::detectors::DetID::FT0);
560  // connect the FIT digitization
