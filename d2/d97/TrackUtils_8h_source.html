42 template <typename value_T>
43 GPUd() void g3helx3(value_T qfield, value_T step, gpu::gpustd::array<value_T, 7>& vect)
44 {
45  /******************************************************************
46  * *
47  * GEANT3 tracking routine in a constant field oriented *
48  * along axis 3 *
49  * Tracking is performed with a conventional *
50  * helix step method *
51  * *
52  * Authors R.Brun, M.Hansroul ********* *
53  * Rewritten V.Perevoztchikov *
54  * *
55  * Rewritten in C++ by I.Belikov *
56  * *
57  * qfield (kG) - particle charge times magnetic field *
58  * step (cm) - step length along the helix *
59  * vect[7](cm,GeV/c) - input/output x, y, z, px/p, py/p ,pz/p, p *
60  * *
61  ******************************************************************/
62 #ifndef GPUCA_GPUCODE_DEVICE
63  static_assert(std::is_floating_point_v<value_T>);
64 #endif
65 
66  const int ix = 0, iy = 1, iz = 2, ipx = 3, ipy = 4, ipz = 5, ipp = 6;
67  constexpr value_T kOvSqSix = 0.408248f; // std::sqrt(1./6.);
68 
69  value_T cosx = vect[ipx], cosy = vect[ipy], cosz = vect[ipz];
70 
71  value_T rho = qfield * constants::math::B2C / vect[ipp];
72  value_T tet = rho * step;
73 
74  value_T tsint, sintt, sint, cos1t;
75  if (gpu::CAMath::Abs(tet) > 0.03f) {
76  sint = gpu::CAMath::Sin(tet);
77  sintt = sint / tet;
78  tsint = (tet - sint) / tet;
79  value_T t = gpu::CAMath::Sin(0.5f * tet);
80  cos1t = 2 * t * t / tet;
81  } else {
82  tsint = tet * tet / 6.f;
83  sintt = (1.f - tet * kOvSqSix) * (1.f + tet * kOvSqSix); // 1.- tsint;
84  sint = tet * sintt;
85  cos1t = 0.5f * tet;
86  }
87 
88  value_T f1 = step * sintt;
89  value_T f2 = step * cos1t;
90  value_T f3 = step * tsint * cosz;
91  value_T f4 = -tet * cos1t;
92  value_T f5 = sint;
93 
94  vect[ix] += f1 * cosx - f2 * cosy;
95  vect[iy] += f1 * cosy + f2 * cosx;
96  vect[iz] += f1 * cosz + f3;
97 
98  vect[ipx] += f4 * cosx - f5 * cosy;
99  vect[ipy] += f4 * cosy + f5 * cosx;
100 }

103 template <typename value_T>
104 GPUd() value_T BetheBlochSolid(value_T bg, value_T rho, value_T kp1, value_T kp2, value_T meanI,
105  value_T meanZA)
106 {
107  //
108  // This is the parameterization of the Bethe-Bloch formula inspired by Geant.
109  //
110  // bg - beta*gamma
111  // rho - density [g/cm^3]
112  // kp1 - density effect first junction point
113  // kp2 - density effect second junction point
114  // meanI - mean excitation energy [GeV]
115  // meanZA - mean Z/A
116  //
117  // The default values for the kp* parameters are for silicon.
118  // The returned value is in [GeV/(g/cm^2)].
119  //
120 #ifndef GPUCA_GPUCODE_DEVICE
121  static_assert(std::is_floating_point_v<value_T>);
122 #endif
123 
124  constexpr value_T mK = 0.307075e-3f; // [GeV*cm^2/g]
125  constexpr value_T me = 0.511e-3f; // [GeV/c^2]
126  kp1 *= 2.303f;
127  kp2 *= 2.303f;
128  value_T bg2 = bg * bg;
129  value_T maxT = 2.f * me * bg2; // neglecting the electron mass
130 
131  //*** Density effect
132  value_T d2 = 0.;
133  const value_T x = gpu::CAMath::Log(bg);
134  const value_T lhwI = gpu::CAMath::Log(28.816f * 1e-9f * gpu::CAMath::Sqrt(rho * meanZA) / meanI);
135  if (x > kp2) {
136  d2 = lhwI + x - 0.5f;
137  } else if (x > kp1) {
138  double r = (kp2 - x) / (kp2 - kp1);
139  d2 = lhwI + x - 0.5f + (0.5f - lhwI - kp1) * r * r * r;
140  }
141  return mK * meanZA * (1 + bg2) / bg2 * (0.5f * gpu::CAMath::Log(2 * me * bg2 * maxT / (meanI * meanI)) - bg2 / (1 + bg2) - d2);
142 }
