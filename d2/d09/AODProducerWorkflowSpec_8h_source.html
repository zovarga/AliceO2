156 struct TripletHash : std::unary_function<Triplet_t, std::size_t> {
157  std::size_t operator()(Triplet_t const& e) const
158  {
159  std::size_t seed = 0;
160  boost::hash_combine(seed, e.get<0>());
161  boost::hash_combine(seed, e.get<1>());
162  boost::hash_combine(seed, e.get<2>());
163  return seed;
164  }
165 };

167 struct TripletEqualTo : std::binary_function<Triplet_t, Triplet_t, bool> {
168  bool operator()(Triplet_t const& x, Triplet_t const& y) const
169  {
170  return (x.get<0>() == y.get<0>() &&
171  x.get<1>() == y.get<1>() &&
172  x.get<2>() == y.get<2>());
173  }
174 };
