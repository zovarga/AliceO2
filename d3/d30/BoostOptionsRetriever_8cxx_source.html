40 void BoostOptionsRetriever::update(std::vector<ConfigParamSpec> const& specs,
41  boost::property_tree::ptree& store,
42  boost::property_tree::ptree& provenance)
43 {
44  auto options = mDescription->add_options();
45  for (const auto& spec : specs) {
46  const char* name = spec.name.c_str();
47  const char* help = spec.help.c_str();
48  // FIXME: propagate default value?
49  switch (spec.type) {
50  case VariantType::Int:
51  options = options(name, bpo::value<int>()->default_value(spec.defaultValue.get<int>()), help);
52  break;
53  case VariantType::UInt8:
54  options = options(name, bpo::value<int>()->default_value(spec.defaultValue.get<uint8_t>()), help);
55  break;
56  case VariantType::UInt16:
57  options = options(name, bpo::value<int>()->default_value(spec.defaultValue.get<uint16_t>()), help);
58  break;
59  case VariantType::UInt32:
60  options = options(name, bpo::value<int>()->default_value(spec.defaultValue.get<uint32_t>()), help);
61  break;
62  case VariantType::UInt64:
63  options = options(name, bpo::value<int>()->default_value(spec.defaultValue.get<uint64_t>()), help);
64  break;
65  case VariantType::Int64:
66  options = options(name, bpo::value<int64_t>()->default_value(spec.defaultValue.get<int64_t>()), help);
67  break;
68  case VariantType::Float:
69  options = options(name, bpo::value<float>()->default_value(spec.defaultValue.get<float>()), help);
70  break;
71  case VariantType::Double:
72  options = options(name, bpo::value<double>()->default_value(spec.defaultValue.get<double>()), help);
73  break;
74  case VariantType::String:
75  options = options(name, bpo::value<std::string>()->default_value(spec.defaultValue.get<const char*>()), help);
76  break;
77  case VariantType::Bool:
78  options = options(name, bpo::value<bool>()->zero_tokens()->default_value(spec.defaultValue.get<bool>()), help);
79  break;
80  case VariantType::ArrayInt:
81  case VariantType::ArrayFloat:
82  case VariantType::ArrayDouble:
83  case VariantType::ArrayBool:
84  case VariantType::ArrayString:
85  case VariantType::Array2DInt:
86  case VariantType::Array2DFloat:
87  case VariantType::Array2DDouble:
88  options = options(name, bpo::value<std::string>()->default_value(spec.defaultValue.asString()), help);
89  break;
90  case VariantType::Dict:
91  case VariantType::LabeledArrayInt:
92  case VariantType::LabeledArrayFloat:
93  case VariantType::LabeledArrayDouble:
94  case VariantType::Unknown:
95  case VariantType::Empty:
96  break;
97  };
98  }
