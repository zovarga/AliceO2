60 template <class T>
61 void o2::globaltracking::RecoContainer::createTracksVariadic(T creator) const
62 {
63  // We go from most complete tracks to least complete ones, taking into account that some track times
64  // do not bear their own kinematics but just constrain the time
65  // As we get more track types functional, this method should be completed
66  // If user provided function creator returns true, then the track is considered as consumed and its contributing
67  // simpler tracks will not be provided to the creator. If it returns false, the creator will be called also
68  // with this simpler contrubutors.
69  // The creator function is called with track kinematics, track GlobalTrackID and track timing information as 2 floats
70  // which depends on the track time:
71  // 1) For track types containing TimeStampWithError ts it is ts.getTimeStamp(), getTimeStampError()
72  // 2) For tracks with asymmetric time uncertainty, e.g. TPC: as mean time of t0-errBwd,t+errFwd and 0.5(errBwd+errFwd), all in TPC time bins
73  // 3) For tracks whose timing is provided as RO frame: as time in \mus for RO frame start since the start of TF, half-duration of RO window.
74 
75  auto start_time = std::chrono::high_resolution_clock::now();
76  constexpr float PS2MUS = 1e-6;
77  std::array<std::vector<uint8_t>, GTrackID::NSources> usedData;
78  auto flagUsed2 = [&usedData](int idx, int src) {
79  if (!usedData[src].empty()) {
80  usedData[src][idx] = 1;
81  }
82  };
83  auto flagUsed = [&usedData, &flagUsed2](const GTrackID gidx) { flagUsed2(gidx.getIndex(), gidx.getSource()); };
84  auto isUsed2 = [&usedData](int idx, int src) { return (!usedData[src].empty()) && (usedData[src][idx] != 0); };
85  auto isUsed = [&usedData, isUsed2](const GTrackID gidx) { return isUsed2(gidx.getIndex(), gidx.getSource()); };
86 
87  // create only for those data types which are used
88  const auto tracksITS = getITSTracks();
89  const auto trkITABRefs = getITSABRefs();
90  const auto tracksMFT = getMFTTracks();
91  const auto tracksMCH = getMCHTracks();
92  const auto tracksMID = getMIDTracks();
93  const auto tracksTPC = getTPCTracks();
94  const auto tracksTPCITS = getTPCITSTracks();
95  const auto tracksMFTMCH = getGlobalFwdTracks();
96  const auto tracksTPCTOF = getTPCTOFTracks(); // TOF-TPC tracks with refit
97  const auto matchesTPCTOF = getTPCTOFMatches(); // and corresponding matches
98  const auto tracksTPCTRD = getTPCTRDTracks<o2::trd::TrackTRD>();
99  const auto matchesITSTPCTOF = getITSTPCTOFMatches(); // just matches, no refit done
100  const auto matchesTPCTRDTOF = getTPCTRDTOFMatches(); // just matches, no refit done
101  const auto matchesITSTPCTRDTOF = getITSTPCTRDTOFMatches(); // just matches, no refit done
102  const auto tofClusters = getTOFClusters();
103  const auto tracksITSTPCTRD = getITSTPCTRDTracks<o2::trd::TrackTRD>();
104 
105  const auto trigTPCTRD = getTPCTRDTriggers();
106 
107  usedData[GTrackID::ITS].resize(tracksITS.size()); // to flag used ITS tracks
108  usedData[GTrackID::MCH].resize(tracksMCH.size()); // to flag used MCH tracks
109  usedData[GTrackID::MFT].resize(tracksMFT.size()); // to flag used MFT tracks
110  usedData[GTrackID::MID].resize(tracksMID.size()); // to flag used MFT tracks
111  usedData[GTrackID::TPC].resize(tracksTPC.size()); // to flag used TPC tracks
112  usedData[GTrackID::ITSTPC].resize(tracksTPCITS.size()); // to flag used ITSTPC tracks
113  usedData[GTrackID::MFTMCH].resize(tracksMFTMCH.size()); // to flag used MFTMCH tracks
114  usedData[GTrackID::ITSTPCTRD].resize(tracksITSTPCTRD.size()); // to flag used ITSTPCTRD tracks
115  usedData[GTrackID::TPCTRD].resize(tracksTPCTRD.size()); // to flag used TPCTRD tracks
116  usedData[GTrackID::ITSTPCTRD].resize(tracksITSTPCTRD.size()); // to flag used TPCTRD tracks
117  usedData[GTrackID::ITSTPCTOF].resize(getITSTPCTOFMatches().size()); // to flag used ITSTPC-TOF matches
118  usedData[GTrackID::TPCTRDTOF].resize(getTPCTRDTOFMatches().size()); // to flag used ITSTPC-TOF matches
119  usedData[GTrackID::ITSTPCTRDTOF].resize(getITSTPCTRDTOFMatches().size()); // to flag used ITSTPC-TOF matches
120 
121  auto getBCDiff = [startIR = this->startIR](const o2::InteractionRecord& ir) {
122  static int BCDiffErrCount = 0;
123  constexpr int MAXBCDiffErrCount = 5;
124  auto bcd = ir.differenceInBC(startIR);
125  if (uint64_t(bcd) > o2::constants::lhc::LHCMaxBunches * 256 && BCDiffErrCount < MAXBCDiffErrCount) {
126  LOGP(error, "ATTENTION: wrong bunches diff. {} for current IR {} wrt 1st TF orbit {}", bcd, ir, startIR);
127  BCDiffErrCount++;
128  }
129  return bcd;
130  };
131 
132  // ITS-TPC-TRD-TOF
133  {
134 
135  if (matchesITSTPCTRDTOF.size() && (!tofClusters.size() || !tracksITSTPCTRD.size())) {
136  throw std::runtime_error(fmt::format("Global-TOF tracks ({}) require ITS-TPC-TRD tracks ({}) and TOF clusters ({})",
137  matchesITSTPCTRDTOF.size(), tracksITSTPCTRD.size(), tofClusters.size()));
138  }
139  for (unsigned i = 0; i < matchesITSTPCTRDTOF.size(); i++) {
140  const auto& match = matchesITSTPCTRDTOF[i];
141  auto gidx = match.getTrackRef(); // this should be corresponding ITS-TPC-TRD track
142  // no need to check isUsed: by construction this ITS-TPC-TRD was not used elsewhere
143  const auto& tofCl = tofClusters[match.getTOFClIndex()];
144  float timeTOFMUS = (tofCl.getTime() - match.getLTIntegralOut().getTOF(o2::track::PID::Pion)) * PS2MUS; // tof time in \mus, FIXME: account for time of flight to R TOF
145  const float timeErr = 0.010f; // assume 10 ns error FIXME
146  if (creator(tracksITSTPCTRD[gidx.getIndex()], {i, GTrackID::ITSTPCTRDTOF}, timeTOFMUS, timeErr)) {
147  // flagUsed2(i, GTrackID::TOF); // flag used TOF match // TODO might be not needed
148  flagUsed(gidx); // flag used ITS-TPC-TRD tracks
149  }
150  }
151  }
152 
153  // TPC-TRD-TOF
154  {
155 
156  if (matchesTPCTRDTOF.size() && (!tofClusters.size() || !tracksTPCTRD.size())) {
157  throw std::runtime_error(fmt::format("Global-TOF tracks ({}) require TPC-TRD tracks ({}) and TOF clusters ({})",
158  matchesTPCTRDTOF.size(), tracksTPCTRD.size(), tofClusters.size()));
159  }
160  for (unsigned i = 0; i < matchesTPCTRDTOF.size(); i++) {
161  const auto& match = matchesTPCTRDTOF[i];
162  auto gidx = match.getTrackRef(); // this should be corresponding ITS-TPC-TRD track
163  if (isUsed(gidx)) {
164  continue;
165  }
166  // no need to check isUsed: by construction this ITS-TPC-TRD was not used elsewhere
167  const auto& tofCl = tofClusters[match.getTOFClIndex()];
168  float timeTOFMUS = (tofCl.getTime() - match.getLTIntegralOut().getTOF(o2::track::PID::Pion)) * PS2MUS; // tof time in \mus, FIXME: account for time of flight to R TOF
169  const float timeErr = 0.010f; // assume 10 ns error FIXME
170  if (creator(tracksTPCTRD[gidx.getIndex()], {i, GTrackID::TPCTRDTOF}, timeTOFMUS, timeErr)) {
171  // flagUsed2(i, GTrackID::TOF); // flag used TOF match // TODO might be not needed
172  flagUsed(gidx); // flag used ITS-TPC tracks
173  }
174  }
175  }
176 
177  // ITS-TPC-TRD
178  {
179  const auto trigITSTPCTRD = getITSTPCTRDTriggers();
180  for (unsigned itr = 0; itr < trigITSTPCTRD.size(); itr++) {
181  const auto& trig = trigITSTPCTRD[itr];
182  auto bcdiff = getBCDiff(trig.getBCData());
183  float t0 = bcdiff * o2::constants::lhc::LHCBunchSpacingNS * 1e-3;
184  for (unsigned i = trig.getTrackRefs().getFirstEntry(); i < trig.getTrackRefs().getEntriesBound(); i++) {
185  const auto& trc = tracksITSTPCTRD[i];
186  if (isUsed2(i, GTrackID::ITSTPCTRD)) {
187  flagUsed(trc.getRefGlobalTrackId()); // flag seeding ITS-TPC track
188  continue;
189  }
190  if (creator(trc, {i, GTrackID::ITSTPCTRD}, t0, 1e-3)) { // assign 1ns error to BC
191  flagUsed2(i, GTrackID::ITSTPCTRD); // flag itself (is it needed?)
192  flagUsed(trc.getRefGlobalTrackId()); // flag seeding ITS-TPC track
193  }
194  }
195  }
196  }
197 
198  // ITS-TPC-TOF matches, thes are just MatchInfoTOF objects, pointing on ITS-TPC match and TOF cl.
199  {
200 
201  if (matchesITSTPCTOF.size() && (!tofClusters.size() || !tracksTPCITS.size())) {
202  throw std::runtime_error(fmt::format("Global-TOF tracks ({}) require ITS-TPC tracks ({}) and TOF clusters ({})",
203  matchesITSTPCTOF.size(), tracksTPCITS.size(), tofClusters.size()));
204  }
205  for (unsigned i = 0; i < matchesITSTPCTOF.size(); i++) {
206  const auto& match = matchesITSTPCTOF[i];
207  auto gidx = match.getTrackRef(); // this should be corresponding ITS-TPC track
208  if (isUsed(gidx)) { // RS FIXME: THIS IS TEMPORARY, until the TOF matching will use ITS-TPC-TRD as an input
209  continue;
210  }
211  // no need to check isUsed: by construction this ITS-TPC was not used elsewhere
212  const auto& tofCl = tofClusters[match.getTOFClIndex()];
213  float timeTOFMUS = (tofCl.getTime() - match.getLTIntegralOut().getTOF(o2::track::PID::Pion)) * PS2MUS; // tof time in \mus, FIXME: account for time of flight to R TOF
214  const float timeErr = 0.010f; // assume 10 ns error FIXME
215  if (creator(tracksTPCITS[gidx.getIndex()], {i, GTrackID::ITSTPCTOF}, timeTOFMUS, timeErr)) {
216  // flagUsed2(i, GTrackID::TOF); // flag used TOF match // TODO might be not needed
217  flagUsed(gidx); // flag used ITS-TPC tracks
218  }
219  }
220  }
221 
222  // TPC-TRD
223  {
224  const auto trigTPCTRD = getTPCTRDTriggers();
225  for (unsigned itr = 0; itr < trigTPCTRD.size(); itr++) {
226  const auto& trig = trigTPCTRD[itr];
227  auto bcdiff = getBCDiff(trig.getBCData());
228  float t0 = bcdiff * o2::constants::lhc::LHCBunchSpacingNS * 1e-3;
229  for (unsigned i = trig.getTrackRefs().getFirstEntry(); i < trig.getTrackRefs().getEntriesBound(); i++) {
230  const auto& trc = tracksTPCTRD[i];
231  if (isUsed2(i, GTrackID::TPCTRD)) {
232  flagUsed(trc.getRefGlobalTrackId()); // flag seeding TPC track
233  continue;
234  }
235  if (creator(trc, {i, GTrackID::TPCTRD}, t0, 1e-3)) { // assign 1ns error to BC
236  flagUsed2(i, GTrackID::TPCTRD); // flag itself (is it needed?)
237  flagUsed(trc.getRefGlobalTrackId()); // flag seeding TPC track
238  }
239  }
240  }
241  }
242 
243  // ITS-TPC matches, may refer to ITS, TPC (TODO: something else?) tracks
244  {
245  for (unsigned i = 0; i < tracksTPCITS.size(); i++) {
246  const auto& matchTr = tracksTPCITS[i];
247  if (isUsed2(i, GTrackID::ITSTPC)) {
248  flagUsed(matchTr.getRefITS()); // flag used ITS tracks or AB tracklets (though the latter is not really necessary)
249  flagUsed(matchTr.getRefTPC()); // flag used TPC tracks
250  continue;
251  }
252  if (creator(matchTr, {i, GTrackID::ITSTPC}, matchTr.getTimeMUS().getTimeStamp(), matchTr.getTimeMUS().getTimeStampError())) {
253  flagUsed2(i, GTrackID::ITSTPC);
254  flagUsed(matchTr.getRefITS()); // flag used ITS tracks or AB tracklets (though the latter is not really necessary)
255  flagUsed(matchTr.getRefTPC()); // flag used TPC tracks
256  }
257  }
258  }
259 
260  // TPC-TOF matches, may refer to TPC (TODO: something else?) tracks
261  {
262  if (matchesTPCTOF.size() && !tracksTPCTOF.size()) {
263  throw std::runtime_error(fmt::format("TPC-TOF matched tracks ({}) require TPCTOF matches ({}) and TPCTOF tracks ({})",
264  matchesTPCTOF.size(), tracksTPCTOF.size()));
265  }
266  for (unsigned i = 0; i < matchesTPCTOF.size(); i++) {
267  const auto& match = matchesTPCTOF[i];
268  const auto& gidx = match.getTrackRef(); // TPC track global idx
269  if (isUsed(gidx)) { // is TPC track already used
270  continue;
271  }
272  const auto& trc = tracksTPCTOF[i];
273  if (creator(trc, {i, GTrackID::TPCTOF}, trc.getTimeMUS().getTimeStamp(), trc.getTimeMUS().getTimeStampError())) {
274  flagUsed(gidx); // flag used TPC tracks
275  }
276  }
277  }
278 
279  // MFT-MCH tracks
280  {
281  for (unsigned i = 0; i < tracksMFTMCH.size(); i++) {
282  const auto& matchTr = tracksMFTMCH[i];
283  if (creator(matchTr, {i, GTrackID::MFTMCH}, matchTr.getTimeMUS().getTimeStamp(), matchTr.getTimeMUS().getTimeStampError())) {
284  flagUsed2(i, GTrackID::MFTMCH);
285  }
286  }
287  }
288 
289  // TPC only tracks
290  {
291  int nacc = 0, noffer = 0;
292  for (unsigned i = 0; i < tracksTPC.size(); i++) {
293  if (isUsed2(i, GTrackID::TPC)) { // skip used tracks
294  continue;
295  }
296  const auto& trc = tracksTPC[i];
297  if (creator(trc, {i, GTrackID::TPC}, trc.getTime0() + 0.5 * (trc.getDeltaTFwd() - trc.getDeltaTBwd()), 0.5 * (trc.getDeltaTFwd() + trc.getDeltaTBwd()))) {
298  flagUsed2(i, GTrackID::TPC); // flag used TPC tracks
299  }
300  }
301  }
302 
303  // ITS only tracks
304  {
305  const auto& rofrs = getITSTracksROFRecords();
306  for (unsigned irof = 0; irof < rofrs.size(); irof++) {
307  const auto& rofRec = rofrs[irof];
308  auto bcdiff = getBCDiff(rofRec.getBCData());
309  float t0 = bcdiff * o2::constants::lhc::LHCBunchSpacingNS * 1e-3;
310  int trlim = rofRec.getFirstEntry() + rofRec.getNEntries();
311  for (int it = rofRec.getFirstEntry(); it < trlim; it++) {
312  if (isUsed2(it, GTrackID::ITS)) { // skip used tracks
313  continue;
314  }
315  GTrackID gidITS(it, GTrackID::ITS);
316  const auto& trc = tracksITS[it];
317  if (creator(trc, gidITS, t0, 0.5)) {
318  flagUsed2(it, GTrackID::ITS);
319  }
320  }
321  }
322  }
323 
324  // MFT only tracks
325  {
326  const auto& rofrs = getMFTTracksROFRecords();
327  for (unsigned irof = 0; irof < rofrs.size(); irof++) {
328  const auto& rofRec = rofrs[irof];
329  auto bcdiff = getBCDiff(rofRec.getBCData());
330  float t0 = bcdiff * o2::constants::lhc::LHCBunchSpacingNS * 1e-3;
331  int trlim = rofRec.getFirstEntry() + rofRec.getNEntries();
332  for (int it = rofRec.getFirstEntry(); it < trlim; it++) {
333  if (isUsed2(it, GTrackID::MFT)) {
334  continue;
335  }
336  GTrackID gidMFT(it, GTrackID::MFT);
337  const auto& trc = tracksMFT[it];
338  if (creator(trc, gidMFT, t0, 0.5)) {
339  flagUsed2(it, GTrackID::MFT);
340  }
341  }
342  }
343  }
344 
345  // MCH standalone tracks
346  {
347  const auto& rofs = getMCHTracksROFRecords();
348  for (const auto& rof : rofs) {
349  auto bcWidth = 56;
350  // FIXME (LA): should really be rof.getBCWidth() once
351  // getBCWidth is actually set to a meaningfull value.
352  // For now we hard-code a 1.4 microseconds window for all tracks
353  auto bcdiff = getBCDiff(rof.getBCData());
354  auto rofMeanBC = bcdiff + bcWidth / 2;
355  float t0 = rofMeanBC * o2::constants::lhc::LHCBunchSpacingMUS;
356  float t0err = o2::constants::lhc::LHCBunchSpacingMUS * bcWidth / 2;
357  for (int idx = rof.getFirstIdx(); idx <= rof.getLastIdx(); ++idx) {
358  if (isUsed2(idx, GTrackID::MCH)) {
359  flagUsed2(idx, GTrackID::MCH);
360  continue;
361  }
362  GTrackID gidMCH(idx, GTrackID::MCH);
363  const auto& trc = tracksMCH[idx];
364  if (creator(trc, gidMCH, t0, t0err)) {
365  flagUsed2(idx, GTrackID::MCH);
366  }
367  }
368  }
369  }
370 
371  // MID standalone tracks
372  {
373  const auto& rofs = getMIDTracksROFRecords();
374  for (const auto& rof : rofs) {
375  float t0err = 0.0005;
376  auto bcdiff = getBCDiff(rof.interactionRecord);
377  float t0 = bcdiff * o2::constants::lhc::LHCBunchSpacingMUS;
378  for (int idx = rof.firstEntry; idx <= rof.getEndIndex(); ++idx) {
379  if (isUsed2(idx, GTrackID::MID)) {
380  continue;
381  }
382  GTrackID gidMID(idx, GTrackID::MID);
383  const auto& trc = tracksMID[idx];
384  if (creator(trc, gidMID, t0, t0err)) {
385  flagUsed2(idx, GTrackID::MID);
386  }
387  }
388  }
389  }
390 
391  auto current_time = std::chrono::high_resolution_clock::now();
392  LOG(info) << "RecoContainer::createTracks took " << std::chrono::duration_cast<std::chrono::microseconds>(current_time - start_time).count() * 1e-6 << " CPU s.";
393 }
