59 void CompressedClustersROOT::Streamer(TBuffer& R__b)
60 {
61  // the custom streamer for CompressedClustersROOT
62  if (R__b.IsReading()) {
63  R__b.ReadClassBuffer(CompressedClustersROOT::Class(), this);
64  gsl::span flatdata{this->flatdata, static_cast<std::size_t>(this->flatdataSize)};
65  CompressedClustersHelpers::restoreFrom(flatdata, *this);
66  } else {
67  std::vector<char> flatdata;
68  // member flatdata is nullptr unless it is an already extracted object which
69  // is streamed again. Overriding the existing flatbuffer is a potential
70  // memory leak
71  // Note: lots of cornercases here which are diffucult to catch with the design
72  // of CompClusters container (which is optimized to be used with GPU). Main concern
73  // is if an extracted object is not read-only and the counters are changed.
74  // TODO: we can add a consistency check whether the size of the flatbuffer
75  // and the pointers match the object.
76  bool isflat = this->flatdata != nullptr;
77  if (!isflat) {
78  CompressedClustersHelpers::flattenTo(flatdata, *this);
79  this->flatdataSize = flatdata.size();
80  this->flatdata = flatdata.data();
81  }
82  R__b.WriteClassBuffer(CompressedClustersROOT::Class(), this);
83  if (!isflat) {
84  this->flatdataSize = 0;
85  this->flatdata = nullptr;
86  }
87  }
88 }
