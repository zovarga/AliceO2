37 TransformationCreator transformationFromTGeoManager(const TGeoManager& geo)
38 {
39  return [&geo](int detElemId) -> o2::math_utils::Transform3D {
40  if (std::find(begin(allDeIds), end(allDeIds), detElemId) == end(allDeIds)) {
41  throw std::runtime_error("Wrong detection element Id");
42  }
43 
44  std::string volPathName = geo.GetTopVolume()->GetName();
45 
46  int nCh = detElemId / 100;
47 
48  if (nCh <= 4 && geo.GetVolume("YOUT1")) {

55  volPathName += "/";

70 TransformationCreator transformationFromJSON(std::istream& in)
71 {
72  rapidjson::IStreamWrapper isw(in);
73 
74  rapidjson::Document d;
75  d.ParseStream(isw);
76 
77  rapidjson::Value& alignables = d["alignables"];
78  assert(alignables.IsArray());
79 
80  std::map<int, std::tuple<double, double, double>> angles;
81  std::map<int, std::tuple<double, double, double>> translations;
82 
83  // loop over json document and extract Tait-Bryan angles (yaw,pitch,roll)
84  // as well as translation vector (tx,ty,tz)
85  // for each detection element
86 
87  constexpr double deg2rad = 3.14159265358979323846 / 180.0;
88 
89  for (auto& al : alignables.GetArray()) {
90  auto itr = al.FindMember("deid");
91  if (itr != al.MemberEnd()) {
92  int deid = itr->value.GetInt();
93  auto t = al["transform"].GetObject();
94  angles[deid] = {
95  deg2rad * t["yaw"].GetDouble(),
96  deg2rad * t["pitch"].GetDouble(),
97  deg2rad * t["roll"].GetDouble()};
98  translations[deid] = {
99  t["tx"].GetDouble(),
100  t["ty"].GetDouble(),
101  t["tz"].GetDouble()};
102  }
103  }
104 
105  return [angles, translations](int detElemId) -> o2::math_utils::Transform3D {
106  if (std::find(begin(allDeIds), end(allDeIds), detElemId) == end(allDeIds)) {
107  throw std::runtime_error("Wrong detection element Id");
108  }
109  auto [yaw, pitch, roll] = angles.at(detElemId);
110  auto [tx, ty, tz] = translations.at(detElemId);
111  double tr[3] = {tx, ty, tz};
112  // get the angles, convert them to a matrix and build a Transform3D
113  // from it
114  auto rot = o2::mch::geo::angles2matrix(yaw, pitch, roll);
115  TGeoHMatrix m;
116  m.SetRotation(&rot[0]);
117  m.SetTranslation(tr);
118  return o2::math_utils::Transform3D(m);
119  };
120 } // namespace o2::mch::geo

122 std::array<double, 9> angles2matrix(double yaw, double pitch, double roll)
123 {
124  std::array<double, 9> rot;
125 
126  double sinpsi = std::sin(roll);
127  double cospsi = std::cos(roll);
128  double sinthe = std::sin(pitch);
129  double costhe = std::cos(pitch);
130  double sinphi = std::sin(yaw);
131  double cosphi = std::cos(yaw);
132  rot[0] = costhe * cosphi;
133  rot[1] = -costhe * sinphi;
134  rot[2] = sinthe;
135  rot[3] = sinpsi * sinthe * cosphi + cospsi * sinphi;
136  rot[4] = -sinpsi * sinthe * sinphi + cospsi * cosphi;
137  rot[5] = -costhe * sinpsi;
138  rot[6] = -cospsi * sinthe * cosphi + sinpsi * sinphi;
139  rot[7] = cospsi * sinthe * sinphi + sinpsi * cosphi;
140  rot[8] = costhe * cospsi;
141  return rot;
142 }
