34 void TOFDiagnosticCalibrator::finalizeSlot(Slot& slot)
35 {
36  static const double TFlength = 1E-3 * o2::raw::HBFUtils::Instance().getNOrbitsPerTF() * o2::constants::lhc::LHCOrbitMUS; // in ms
37 
38  Diagnostic* diag = slot.getContainer();
39  LOG(info) << "Finalizing slot";
40  diag->print();
41  std::map<std::string, std::string> md;
42  auto clName = o2::utils::MemFileHelper::getClassName(*diag);
43  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
44 
45  uint64_t starting = slot.getTFStart() * TFlength;
46  uint64_t stopping = slot.getTFEnd() * TFlength;
47  LOG(info) << "starting = " << starting * 1E-3 << " - stopping = " << stopping * 1E-3;
48  mccdbInfoVector.emplace_back("TOF/Calib/Diagnostic", clName, flName, md, starting, stopping);
49  mDiagnosticVector.emplace_back(*diag);
50 }
