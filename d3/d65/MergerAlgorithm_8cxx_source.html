35 size_t estimateTreeSize(TTree* tree)
36 {
37  size_t totalSize = 0;
38  auto branchList = tree->GetListOfBranches();
39  for (const auto* branch : *branchList) {
40  totalSize += dynamic_cast<const TBranch*>(branch)->GetTotalSize();
41  }
42  return totalSize;
43 }

45 void merge(TObject* const target, TObject* const other)
46 {
47  if (target == nullptr) {
48  throw std::runtime_error("Merging target is nullptr");
49  }
50  if (other == nullptr) {
51  throw std::runtime_error("Object to be merged in is nullptr");
52  }
53  if (other == target) {
54  throw std::runtime_error("Merging target and the other object point to the same address");
55  }
56  // fixme: should we check if names match?
57 
58  // We expect that both objects follow the same structure, but we allow to add missing objects to TCollections.
59  // First we check if an object contains a MergeInterface, as it should overlap default Merge() methods of TObject.
60  if (auto custom = dynamic_cast<MergeInterface*>(target)) {
61 
62  custom->merge(dynamic_cast<MergeInterface* const>(other));
63 
64  } else if (auto targetCollection = dynamic_cast<TCollection*>(target)) {
65 
66  auto otherCollection = dynamic_cast<TCollection*>(other);
67  if (otherCollection == nullptr) {
68  throw std::runtime_error(std::string("The target object '") + target->GetName() +
69  "' is a TCollection, while the other object '" + other->GetName() + "' is not.");
70  }
71 
72  auto otherIterator = otherCollection->MakeIterator();
73  while (auto otherObject = otherIterator->Next()) {
74  TObject* targetObject = targetCollection->FindObject(otherObject->GetName());
75  if (targetObject) {
76  // That might be another collection or a concrete object to be merged, we walk on the collection recursively.
77  merge(targetObject, otherObject);
78  } else {
79  // We prefer to clone instead of passing the pointer in order to simplify deleting the `other`.
80  targetCollection->Add(otherObject->Clone());
81  }
82  }
83  delete otherIterator;
84  } else {
85  Long64_t errorCode = 0;
86  TObjArray otherCollection;
87  otherCollection.SetOwner(false);
88  otherCollection.Add(other);
89 
90  if (target->InheritsFrom(TH1::Class())) {
91  // this includes TH1, TH2, TH3
92  auto targetTH1 = reinterpret_cast<TH1*>(target);
93  if (targetTH1->TestBit(TH1::kIsAverage)) {
94  // Merge() does not support averages, we have to use Add()
95  // this will break if collection.size != 1
96  if (auto otherTH1 = dynamic_cast<TH1*>(otherCollection.First())) {
97  errorCode = targetTH1->Add(otherTH1);
98  }
99  } else {
100  // Add() does not support histograms with labels, thus we resort to Merge() by default
101  errorCode = targetTH1->Merge(&otherCollection);
102  }
103  } else if (target->InheritsFrom(THnBase::Class())) {
104  // this includes THn and THnSparse
105  errorCode = reinterpret_cast<THnBase*>(target)->Merge(&otherCollection);
106  } else if (target->InheritsFrom(TTree::Class())) {
107  auto targetTree = reinterpret_cast<TTree*>(target);
108  auto otherTree = reinterpret_cast<TTree*>(other);
109  auto targetTreeSize = estimateTreeSize(targetTree);
110  auto otherTreeSize = estimateTreeSize(otherTree);
111  if (auto totalSize = targetTreeSize + otherTreeSize; totalSize > 100000000) {
112  LOG(warn) << "The tree '" << targetTree << "' would be larger than 100MB (" << totalSize << "B) after merging, skipping to let the system survive";
113  errorCode = 0;
114  } else {
115  errorCode = targetTree->Merge(&otherCollection);
116  }
117  } else if (target->InheritsFrom(TGraph::Class())) {
118  errorCode = reinterpret_cast<TGraph*>(target)->Merge(&otherCollection);
119  } else if (target->InheritsFrom(TEfficiency::Class())) {
120  errorCode = reinterpret_cast<TEfficiency*>(target)->Merge(&otherCollection);
121  } else {
122  LOG(warn) << "Object with type '" + std::string(target->ClassName()) + "' is not one of the mergeable types, skipping";
123  }
124  if (errorCode == -1) {
125  throw std::runtime_error("Merging object of type '" + std::string(target->ClassName()) + "' failed.");
126  }
127  }
128 }
