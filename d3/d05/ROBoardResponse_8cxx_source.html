31 bool ROBoardResponse::isZeroSuppressed(const ROBoard& loc) const
32 {
33  auto cfg = mConfigHandler.getConfig(loc.boardId);
34  if (cfg.configWord & crateconfig::sXorY) {
35  return false;
36  }
37  for (int ich = 0; ich < 4; ++ich) {
38  if (loc.patternsBP[ich] && loc.patternsNBP[ich]) {
39  return false;
40  }
41  }
42  return true;
43 }

45 bool ROBoardResponse::applyZeroSuppression(std::vector<ROBoard>& locs) const
46 {
47  std::vector<ROBoard> zsLocs;
48  bool isSuppressed = false;
49  for (auto& loc : locs) {
50  if (!isZeroSuppressed(loc)) {
51  zsLocs.emplace_back(loc);
52  isSuppressed = true;
53  }
54  }
55  locs.swap(zsLocs);
56  return isSuppressed;
57 }

59 std::vector<ROBoard> ROBoardResponse::getTriggerResponse(uint8_t triggerWord) const
60 {
61  std::vector<ROBoard> locBoards;
62  auto& cfgMap = mConfigHandler.getConfigMap();
63  for (auto& item : cfgMap) {
64  locBoards.push_back({raw::sSTARTBIT | raw::sCARDTYPE, triggerWord, item.first, 0});
65  if (triggerWord & (raw::sSOX | raw::sEOX)) {
67  if (item.second.configWord & crateconfig::sMonmoff) {
68  locBoards.back().statusWord |= raw::sMASKED;
69  for (int ich = 0; ich < 4; ++ich) {
70  locBoards.back().patternsBP[ich] = item.second.masksBP[ich];
71  locBoards.back().patternsNBP[ich] = item.second.masksNBP[ich];
72  }
73  }
74  }
75  }
76  auto regBoards = getRegionalResponse(locBoards);
77  locBoards.insert(locBoards.begin(), regBoards.begin(), regBoards.end());
78  return locBoards;
79 }

81 std::vector<ROBoard> ROBoardResponse::getRegionalResponse(const std::vector<ROBoard>& locs) const
82 {
83  std::unordered_map<uint8_t, uint8_t> firedLocs;
84 
85  uint8_t triggerWord = 0;
86  for (auto& loc : locs) {
87  auto locId = raw::getLocId(loc.boardId);
88  auto regId = 8 * crateparams::getGBTIdFromBoardInCrate(locId) + (locId % 8) / 4;
89  auto uniqueRegId = raw::makeUniqueLocID(raw::getCrateId(loc.boardId), regId);
90  int locPos = locId % 4;
91  firedLocs[uniqueRegId] |= (1 << locPos);
92  triggerWord = loc.triggerWord;
93  }
94 
95  std::vector<ROBoard> regBoards;
96  for (auto& item : firedLocs) {
97  regBoards.push_back({raw::sSTARTBIT, triggerWord, item.first, item.second});
98  }
99 
100  return regBoards;
101 }
