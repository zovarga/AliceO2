152 struct MatchRecord {
153  float chi2 = -1.f;
154  int partnerID = MinusOne;
155  int nextRecID = MinusOne;
156  int matchedIC = MinusOne;
157  MatchRecord(int partID, float chi2match, int nxt = MinusOne, int candIC = MinusOne) : partnerID(partID), chi2(chi2match), nextRecID(nxt), matchedIC(candIC) {}
158  MatchRecord() = default;
159 
160  bool isBetter(float otherChi2, int otherIC = MinusOne) const
161  { // prefer record with matched IC candidate, otherwise, better chi2
162  if (otherIC == MinusOne) {
163  return matchedIC == MinusOne ? chi2 < otherChi2 : true;
164  } else {
165  return matchedIC == MinusOne ? false : chi2 < otherChi2;
166  }
167  }
168 
169  bool isBetter(const MatchRecord& other) const
170  {
171  return isBetter(other.chi2, other.matchedIC);
172  }
173 };

199 struct TPCABSeed {
200  static constexpr int8_t NeedAlternative = -3;
201  int tpcWID = MinusOne;
202  int ICCanID = MinusOne;
203  int winLinkID = MinusOne;
204  int8_t lowestLayer = o2::its::RecoGeomHelper::getNLayers();
205  int8_t status = MinusOne;
206  o2::track::TrackParCov track{};
207  std::array<int, o2::its::RecoGeomHelper::getNLayers()> firstInLr;
208  std::vector<ABTrackLink> trackLinks{};
209  TPCABSeed(int id, int ic, const o2::track::TrackParCov& trc) : tpcWID(id), ICCanID(ic), track(trc)
210  {
211  firstInLr.fill(MinusOne);
212  }
213  bool isDisabled() const { return status == MinusTen; }
214  void disable() { status = MinusTen; }
215  bool isValidated() const { return status == Validated; }
216  void validate(int lID)
217  {
218  winLinkID = lID;
219  status = Validated;
220  }
221  bool needAlteranative() const { return status == NeedAlternative; }
222  void setNeedAlternative() { status = NeedAlternative; }
223  ABTrackLink& getLink(int i) { return trackLinks[i]; }
224  const ABTrackLink& getLink(int i) const { return trackLinks[i]; }
225  auto getBestLinkID() const
226  {
227  return lowestLayer < o2::its::RecoGeomHelper::getNLayers() ? firstInLr[lowestLayer] : -1;
228  }
229  bool checkLinkHasUsedClusters(int linkID, const std::vector<int>& clStatus) const
230  {
231  // check if some clusters used by the link or its parents are forbidden (already used by validatet track)
232  while (linkID > MinusOne) {
233  const auto& link = trackLinks[linkID];
234  if (link.clID > MinusOne && clStatus[link.clID] != MinusOne) {
235  return true;
236  }
237  linkID = link.parentID;
238  }
239  return false;
240  }
241  void flagLinkUsedClusters(int linkID, std::vector<int>& clStatus) const
242  {
243  // check if some clusters used by the link or its parents are forbidden (already used by validated track)
244  while (linkID > MinusOne) {
245  const auto& link = trackLinks[linkID];
246  if (link.clID > MinusOne) {
247  clStatus[link.clID] = MinusTen;
248  }
249  linkID = link.parentID;
250  }
251  }
