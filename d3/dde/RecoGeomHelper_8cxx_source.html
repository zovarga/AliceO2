39 void RecoGeomHelper::RecoLadder::updateLimits(const o2::math_utils::Point3D<float>& pntGlo)
40 {
41  // update limits from the point in Global frame
42  float phi = pntGlo.phi(); // -pi:pi range
43  o2::math_utils::bringTo02Pi(phi); // temporary bring to 0:2pi range
44  o2::math_utils::bringTo02Pi(phiRange.getMin());
45  o2::math_utils::bringTo02Pi(phiRange.getMax());
46  phiRange.update(phi);
47  phiMean = phiRange.mean();
48  dphiH = 0.5 * phiRange.delta();
49  if (phiRange.delta() > o2::constants::math::PI) { // wrapping, swap
50  phiRange.set(phiRange.getMax(), phiRange.getMin()); // swap
51  phiMean -= o2::constants::math::PI;
52  dphiH = o2::constants::math::PI - dphiH;
53  }
54  o2::math_utils::bringToPMPi(phiRange.getMin()); // -pi:pi range
55  o2::math_utils::bringToPMPi(phiRange.getMax());
56  o2::math_utils::bringToPMPi(phiMean);
57  //
58  zRange.update(pntGlo.Z());
59 }

62 void RecoGeomHelper::RecoLadder::init()
63 {
64  auto& chip = chips[0].xyEdges;
65  float x0(chip.getX0()), y0(chip.getY0()), x1(chip.getX1()), y1(chip.getY1());
66  for (int i = 1; i < (int)chips.size(); i++) {
67  chip = chips[i].xyEdges;
68  x0 = chip.getDX() > 0 ? std::min(x0, chip.getX0()) : std::max(x0, chip.getX0());
69  x1 = chip.getDX() > 0 ? std::max(x1, chip.getX1()) : std::min(x1, chip.getX1());
70  y0 = chip.getDY() > 0 ? std::min(y0, chip.getY0()) : std::max(y0, chip.getY0());
71  y1 = chip.getDY() > 0 ? std::max(y1, chip.getY1()) : std::min(y1, chip.getY1());
72  }
73  xyEdges.setEdges(x0, y0, x1, y1);
74 }

90 void RecoGeomHelper::RecoLayer::init()
91 {
92  auto gm = o2::its::GeometryTGeo::Instance();
93  gm->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2GRot, o2::math_utils::TransformType::T2L)); // more matrices ?
94 
95  int nHStaves = gm->getNumberOfHalfStaves(id);
96  int nStaves = gm->getNumberOfStaves(id);
97  float dxH = o2::itsmft::SegmentationAlpide::SensorSizeRows / 2; // half width in rphi
98  float dzH = o2::itsmft::SegmentationAlpide::SensorSizeCols / 2; // half width in Z
99  int nCh = gm->getNumberOfChipsPerLayer(id), chip0 = gm->getFirstChipIndex(id);
100  int nChMod = gm->getNumberOfChipsPerModule(id), nChModH = nChMod / 2;
101  nLadders = nHStaves > 1 ? nStaves * nHStaves * 2 : nStaves; // 2 ladders per h-stave for OB
102  ladders.resize(nLadders);
103 
104  for (auto lad : ladders) {
105  lad.chips.reserve(gm->getNumberOfChipsPerHalfStave(id) / (id > 2 ? 2 : 1)); // 2 ladders per h-stave for OB
106  }
107  for (int ich = 0; ich < nCh; ich++) {
108  int chipID = chip0 + ich, lay, sta, ssta, mod, chipInMod;
109  gm->getChipId(chipID, lay, sta, ssta, mod, chipInMod);
110  int ladID = sta, chipInLadder = nChMod - chipInMod - 1; // count from negative to positive Z, contrary to official chips numbering
111  if (nHStaves > 1) { // OB
112  int modUpper = chipInMod / nChModH;
113  ladID = sta * 4 + ssta * 2 + modUpper; // OB module covers 2 "ladders"
114  }
115  auto& ladder = ladders[ladID];
116  auto& chip = ladder.chips.emplace_back();
117  chip.id = chipID;
118  gm->getSensorXAlphaRefPlane(chipID, chip.xRef, chip.alp);
119  o2::math_utils::sincos(chip.alp, chip.snAlp, chip.csAlp);
120 
121  o2::math_utils::Point3D<float> edgeLoc(-dxH, 0.f, -dzH);
122  auto edgeTra = gm->getMatrixT2L(chipID) ^ (edgeLoc); // edge in tracking frame
123  chip.updateLimits(edgeTra);
124  auto edgeGloM = gm->getMatrixT2GRot(chipID)(edgeTra); // edge in global frame
125  updateLimits(edgeGloM);
126  ladder.updateLimits(edgeGloM);
127 
128  edgeLoc.SetXYZ(dxH, 0.f, dzH);
129  edgeTra = gm->getMatrixT2L(chipID) ^ (edgeLoc); // edge in tracking frame
130  chip.updateLimits(edgeTra);
131  auto edgeGloP = gm->getMatrixT2GRot(chipID)(edgeTra); // edge in globalframe
132  updateLimits(edgeGloP);
133  ladder.updateLimits(edgeGloP);
134  chip.xyEdges.setEdges(edgeGloM.X(), edgeGloM.Y(), edgeGloP.X(), edgeGloP.Y());
135  }
136 
137  // sort according to mean phi (in -pi:pi range!!!)
138  std::sort(ladders.begin(), ladders.end(), [](auto& a, auto& b) {
139  float pha = a.phiMean, phb = b.phiMean;
140  o2::math_utils::bringTo02Pi(pha);
141  o2::math_utils::bringTo02Pi(phb);
142  return pha < phb;
143  });
144 
145  // make sure chips within the ladder are ordered in Z, renumber ladders
146  for (int i = nLadders; i--;) {
147  auto& lad = ladders[i];
148  std::sort(lad.chips.begin(), lad.chips.end(), [](auto& a, auto& b) { return a.zRange.getMin() < b.zRange.getMin(); });
149  lad.id = i;
150  lad.init();
151  }
152  // relate to its neighbour: is the egde at higher R than the edge of the next ladder and do we check for overlaps ?
153  for (int i = nLadders; i--;) {
154  auto& lad = ladders[i];
155  auto& plad = i > 0 ? ladders[i - 1] : ladders[nLadders - 1];
156  if (nHStaves == 2) { // on the OB the ladders of the same halfstave do not overlap
157  int tlay, tsta, tssta, tmod, tchipInMod;
158  int play, psta, pssta, pmod, pchipInMod;
159  gm->getChipId(lad.chips.front().id, tlay, tsta, tssta, tmod, tchipInMod);
160  gm->getChipId(plad.chips.front().id, play, psta, pssta, pmod, pchipInMod);
161  if (tsta == psta && tssta == pssta) { // these are 2 ladders of the same halfstave, no overlap
162  plad.overlapWithNext = RecoLadder::NoOverlap;
163  continue;
164  }
165  }
166  // need to compare the radius of the edge at lowest phi for this ladder with radius at highest phi of prev ladder
167  // find the radius of the edge at low phi
168  float phi0 = std::atan2(lad.xyEdges.getY0(), lad.xyEdges.getX0());
169  float phi1 = std::atan2(lad.xyEdges.getY1(), lad.xyEdges.getX1());
170  o2::math_utils::bringTo02Pi(phi0); // we don't know a priori if edge0/1 corresponds to low/high phi or vice versa
171  o2::math_utils::bringTo02Pi(phi1);
172  float r2This = (phi0 < phi1 && phi1 - phi0 < o2::constants::math::PI) ? // pick R of lowest angle
173  lad.xyEdges.getX0() * lad.xyEdges.getX0() + lad.xyEdges.getY0() * lad.xyEdges.getY0()
174  : lad.xyEdges.getX1() * lad.xyEdges.getX1() + lad.xyEdges.getY1() * lad.xyEdges.getY1();
175  //
176  phi0 = std::atan2(plad.xyEdges.getY0(), plad.xyEdges.getX0());
177  phi1 = std::atan2(plad.xyEdges.getY1(), plad.xyEdges.getX1());
178  o2::math_utils::bringTo02Pi(phi0); // we don't know a priori if edge0/1 corresponds to low/high phi or vice versa
179  o2::math_utils::bringTo02Pi(phi1);
180  float r2Prev = (phi0 < phi1 && phi1 - phi0 < o2::constants::math::PI) ? // pick R of highest angle
181  plad.xyEdges.getX1() * plad.xyEdges.getX1() + plad.xyEdges.getY1() * plad.xyEdges.getY1()
182  : plad.xyEdges.getX0() * plad.xyEdges.getX0() + plad.xyEdges.getY0() * plad.xyEdges.getY0();
183  //
184  plad.overlapWithNext = r2Prev > r2This ? RecoLadder::Above : RecoLadder::Below;
185  }
186 
187  int ndiv = nLadders * 3; // number of bins for mapping
188  phi2ladder.resize(ndiv);
189  float dphi = o2::constants::math::TwoPI / ndiv;
190  int laddId = 0;
191  for (int i = 0; i < ndiv; i++) {
192  float phi = (0.5 + i) * dphi;
193  o2::math_utils::bringToPMPi(phi);
194  while (laddId < nLadders) {
195  const auto& lad = ladders[laddId];
196  auto rel = lad.isPhiOutside(phi);
197  if (rel != RecoGeomHelper::Above) {
198  break;
199  }
200  laddId++; // laddId was below phi, catch up
201  }
202  phi2ladder[i] = laddId % nLadders;
203  }
204  lastChipInLadder = ladders[0].chips.size();
205  z2chipID = lastChipInLadder / zRange.delta();
206  lastChipInLadder--;
207  rInv = 1. / rRange.mean();
208 }
