40 void ClustererDPL::init(InitContext& ic)
41 {
42  mClusterer = std::make_unique<o2::its3::Clusterer>();
43  mClusterer->setNChips(o2::itsmft::ChipMappingITS::getNChips(o2::itsmft::ChipMappingITS::MB) + o2::itsmft::ChipMappingITS::getNChips(o2::itsmft::ChipMappingITS::OB) + SegmentationSuperAlpide::NLayers);
44 
45  auto filenameGRP = ic.options().get<std::string>("grp-file");
46  const auto grp = o2::parameters::GRPObject::loadFrom(filenameGRP.c_str());
47 
48  if (grp) {
49  mClusterer->setContinuousReadOut(grp->isDetContinuousReadOut("IT3"));
50  } else {
51  LOG(error) << "Cannot retrieve GRP from the " << filenameGRP.c_str() << " file !";
52  mState = 0;
53  return;
54  }
55 
56  mPatterns = !ic.options().get<bool>("no-patterns");
57  mNThreads = std::max(1, ic.options().get<int>("nthreads"));
58 
59  // settings for the fired pixel overflow masking
60  const auto& alpParams = o2::itsmft::DPLAlpideParam<o2::detectors::DetID::ITS>::Instance();
61  const auto& clParams = o2::itsmft::ClustererParam<o2::detectors::DetID::ITS>::Instance();
62  auto nbc = clParams.maxBCDiffToMaskBias;
63  nbc += mClusterer->isContinuousReadOut() ? alpParams.roFrameLengthInBC : (alpParams.roFrameLengthTrig / o2::constants::lhc::LHCBunchSpacingNS);
64  mClusterer->setMaxBCSeparationToMask(nbc);
65  mClusterer->setMaxRowColDiffToMask(clParams.maxRowColDiffToMask);
66 
67  std::string dictPath = ic.options().get<std::string>("its-dictionary-path");
68  std::string dictFile = o2::base::DetectorNameConf::getAlpideClusterDictionaryFileName(o2::detectors::DetID::ITS, dictPath, ".bin");
69  if (o2::utils::Str::pathExists(dictFile)) {
70  mClusterer->loadDictionary(dictFile);
71  LOG(info) << "ITS3Clusterer running with a provided dictionary: " << dictFile;
72  } else {
73  LOG(info) << "Dictionary " << dictFile << " is absent, ITSClusterer expects cluster patterns";
74  }
75  mState = 1;
76  mClusterer->print();
77 }

79 void ClustererDPL::run(ProcessingContext& pc)
80 {
81  auto digits = pc.inputs().get<gsl::span<o2::itsmft::Digit>>("digits");
82  auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
83 
84  gsl::span<const o2::itsmft::MC2ROFRecord> mc2rofs;
85  gsl::span<const char> labelbuffer;
86  if (mUseMC) {
87  labelbuffer = pc.inputs().get<gsl::span<char>>("labels");
88  mc2rofs = pc.inputs().get<gsl::span<o2::itsmft::MC2ROFRecord>>("MC2ROframes");
89  }
90  o2::dataformats::ConstMCTruthContainerView<o2::MCCompLabel> labels(labelbuffer);
91 
92  LOG(info) << "ITSClusterer pulled " << digits.size() << " digits, in "
93  << rofs.size() << " RO frames";
94  LOG(info) << "ITSClusterer pulled " << labels.getNElements() << " labels ";
95 
96  o2::itsmft::DigitPixelReader reader;
97  reader.setDigits(digits);
98  reader.setROFRecords(rofs);
99  if (mUseMC) {
100  reader.setMC2ROFRecords(mc2rofs);
101  reader.setDigitsMCTruth(labels.getIndexedSize() > 0 ? &labels : nullptr);
102  }
103  reader.init();
104  auto orig = o2::header::gDataOriginIT3;
105  std::vector<o2::its3::CompClusterExt> clusCompVec;
106  std::vector<o2::itsmft::ROFRecord> clusROFVec;
107  std::vector<unsigned char> clusPattVec;
108 
109  std::unique_ptr<o2::dataformats::MCTruthContainer<o2::MCCompLabel>> clusterLabels;
110  if (mUseMC) {
111  clusterLabels = std::make_unique<o2::dataformats::MCTruthContainer<o2::MCCompLabel>>();
112  }
113  mClusterer->process(mNThreads, reader, &clusCompVec, mPatterns ? &clusPattVec : nullptr, &clusROFVec, clusterLabels.get());
114  pc.outputs().snapshot(Output{orig, "COMPCLUSTERS", 0, Lifetime::Timeframe}, clusCompVec);
115  pc.outputs().snapshot(Output{orig, "CLUSTERSROF", 0, Lifetime::Timeframe}, clusROFVec);
116  pc.outputs().snapshot(Output{orig, "PATTERNS", 0, Lifetime::Timeframe}, clusPattVec);
117 
118  if (mUseMC) {
119  pc.outputs().snapshot(Output{orig, "CLUSTERSMCTR", 0, Lifetime::Timeframe}, *clusterLabels.get()); // at the moment requires snapshot
120  std::vector<o2::itsmft::MC2ROFRecord> clusterMC2ROframes(mc2rofs.size());
121  for (int i = mc2rofs.size(); i--;) {
122  clusterMC2ROframes[i] = mc2rofs[i]; // Simply, replicate it from digits ?
123  }
124  pc.outputs().snapshot(Output{orig, "CLUSTERSMC2ROF", 0, Lifetime::Timeframe}, clusterMC2ROframes);
125  }
126 
127  // TODO: in principle, after masking "overflow" pixels the MC2ROFRecord maxROF supposed to change, nominally to minROF
128  // -> consider recalculationg maxROF
129  LOG(info) << "ITSClusterer pushed " << clusCompVec.size() << " clusters, in " << clusROFVec.size() << " RO frames";
130 }

132 DataProcessorSpec getClustererSpec(bool useMC)
133 {
134  std::vector<InputSpec> inputs;
135  inputs.emplace_back("digits", "IT3", "DIGITS", 0, Lifetime::Timeframe);
136  inputs.emplace_back("ROframes", "IT3", "DIGITSROF", 0, Lifetime::Timeframe);
137 
138  std::vector<OutputSpec> outputs;
139  outputs.emplace_back("IT3", "COMPCLUSTERS", 0, Lifetime::Timeframe);
140  outputs.emplace_back("IT3", "PATTERNS", 0, Lifetime::Timeframe);
141  outputs.emplace_back("IT3", "CLUSTERSROF", 0, Lifetime::Timeframe);
142 
143  if (useMC) {
144  inputs.emplace_back("labels", "IT3", "DIGITSMCTR", 0, Lifetime::Timeframe);
145  inputs.emplace_back("MC2ROframes", "IT3", "DIGITSMC2ROF", 0, Lifetime::Timeframe);
146  outputs.emplace_back("IT3", "CLUSTERSMCTR", 0, Lifetime::Timeframe);
147  outputs.emplace_back("IT3", "CLUSTERSMC2ROF", 0, Lifetime::Timeframe);
148  }
149 
150  return DataProcessorSpec{
151  "its3-clusterer",
152  inputs,
153  outputs,
154  AlgorithmSpec{adaptFromTask<ClustererDPL>(useMC)},
155  Options{
156  {"its-dictionary-path", VariantType::String, "", {"Path of the cluster-topology dictionary file"}},
157  {"grp-file", VariantType::String, "o2sim_grp.root", {"Name of the grp file"}},
158  {"no-patterns", o2::framework::VariantType::Bool, false, {"Do not save rare cluster patterns"}},
159  {"nthreads", VariantType::Int, 1, {"Number of clustering threads"}}}};
160 }
