91  AxisSpec(int nBins_, double binMin_, double binMax_, std::optional<std::string> title_ = std::nullopt, std::optional<std::string> name_ = std::nullopt)
92  : nBins(nBins_),
93  binEdges({binMin_, binMax_}),
94  title(title_),
95  name(name_)
96  {
97  if (binMin_ > binMax_) {
98  LOG(fatal) << "Defined ill-defined axis";
99  }
100  }
101 
102  // first entry is assumed to be the number of bins; in case of variable size binning it must be set to zero
103  AxisSpec(ConfigurableAxis binEdges_, std::optional<std::string> title_ = std::nullopt, std::optional<std::string> name_ = std::nullopt)
104  : nBins(std::nullopt),
105  binEdges(std::vector<double>(binEdges_)),
106  title(title_),
107  name(name_)
108  {
109  if (binEdges.empty()) {
110  return;
111  }
112  if (binEdges[0] != VARIABLE_WIDTH) {
113  nBins = static_cast<int>(binEdges[0]);
114  binEdges.resize(3); // nBins, lowerBound, upperBound, disregard whatever else is stored in vecotr
115  }
116  binEdges.erase(binEdges.begin()); // remove first entry that we assume to be number of bins
117  }
