155 template <unsigned int Dim, unsigned int Degree>
156 void MultivariatePolynomial<Dim, Degree>::setFromContainer(const MultivariatePolynomialContainer& container)
157 {
158  if constexpr (Dim > 0 && Degree > 0) {
159  if (this->getDim() != container.mDim) {
160  LOGP(info, fmt::format("wrong number of dimensions! this {} container {}", this->getDim(), container.mDim));
161  return;
162  }
163  if (this->getDegree() != container.mDegree) {
164  LOGP(info, fmt::format("wrong number of degrees! this {} container {}", this->getDegree(), container.mDegree));
165  return;
166  }
167  setParams(container.mParams.data());
168  } else {
169  MultivariatePolynomial polTmp(container.mDim, container.mDegree);
170  polTmp.setParams(container.mParams.data());
171  this->cloneFromObject(polTmp, nullptr);
172  }
173 }

182 template <unsigned int Dim, unsigned int Degree>
183 void MultivariatePolynomial<Dim, Degree>::cloneFromObject(const MultivariatePolynomial<Dim, Degree>& obj, char* newFlatBufferPtr)
184 {
185  const char* oldFlatBufferPtr = obj.mFlatBufferPtr;
186  FlatObject::cloneFromObject(obj, newFlatBufferPtr);
187  mNParams = obj.mNParams;
188  if constexpr (Dim == 0 && Degree == 0) {
189  this->mDim = obj.mDim;
190  this->mDegree = obj.mDegree;
191  }
192  if (obj.mParams) {
193  mParams = FlatObject::relocatePointer(oldFlatBufferPtr, mFlatBufferPtr, obj.mParams);
194  }
195 }
