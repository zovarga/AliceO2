73 // customize dispatch policy, dispatch immediately what is ready
74 void customize(std::vector<o2::framework::DispatchPolicy>& policies)
75 {
76  using DispatchOp = o2::framework::DispatchPolicy::DispatchOp;
77  // we customize all devices to dispatch data immediately
78  auto readerMatcher = [](auto const& spec) {
79  return std::regex_match(spec.name.begin(), spec.name.end(), std::regex(".*-reader"));
80  };
81  auto triggerMatcher = [](auto const& query) {
82  // a bit of a hack but we want this to be configurable from the command line,
83  // however DispatchPolicy is inserted before all other setup. Triggering depending
84  // on the global variable set from the command line option. If scheduled messages
85  // are not triggered they are sent out at the end of the computation
86  return gDispatchTrigger.origin == query.origin && gDispatchTrigger.description == query.description;
87  };
88  policies.push_back({"prompt-for-reader", readerMatcher, DispatchOp::WhenReady, triggerMatcher});
89 }

91 // customize clusterers and cluster decoders to process immediately what comes in
92 void customize(std::vector<o2::framework::CompletionPolicy>& policies)
93 {
94  // we customize the pipeline processors to consume data as it comes
95  using CompletionPolicy = o2::framework::CompletionPolicy;
96  using CompletionPolicyHelpers = o2::framework::CompletionPolicyHelpers;
97  policies.push_back(CompletionPolicyHelpers::defineByName("tpc-cluster-decoder.*", CompletionPolicy::CompletionOp::Consume));
98  policies.push_back(CompletionPolicyHelpers::defineByName("tpc-clusterer.*", CompletionPolicy::CompletionOp::Consume));
99  // the custom completion policy for the tracker
100  policies.push_back(o2::tpc::TPCSectorCompletionPolicy("tpc-tracker.*", o2::tpc::TPCSectorCompletionPolicy::Config::RequireAll, &gPolicyData, &gTpcSectorMask)());
101 }

121 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
122 {
123  std::vector<int> tpcSectors = o2::RangeTokenizer::tokenize<int>(cfgc.options().get<std::string>("tpc-sectors"));
124  // the lane configuration defines the subspecification ids to be distributed among the lanes.
125  std::vector<int> laneConfiguration = tpcSectors; // Currently just a copy of the tpcSectors, why?
126  auto nLanes = cfgc.options().get<int>("tpc-lanes");
127  auto inputType = cfgc.options().get<std::string>("input-type");
128 
129  // depending on whether to dispatch early (prompt) and on the input type, we
130  // set the matcher. Note that this has to be in accordance with the OutputSpecs
131  // configured for the PublisherSpec
132  auto dispmode = cfgc.options().get<std::string>("dispatching-mode");
133  if (dispmode == "complete") {
134  // nothing to do we leave the matcher empty which will suppress the dispatch
135  // trigger and all messages will be sent out together at end of computation
136  } else if (inputType == "digits") {
137  gDispatchTrigger = o2::framework::Output{"TPC", "DIGITS"};
138  } else if (inputType == "clustershw") {
139  gDispatchTrigger = o2::framework::Output{"TPC", "CLUSTERHW"};
140  } else if (inputType == "clustersnative") {
141  gDispatchTrigger = o2::framework::Output{"TPC", "CLUSTERNATIVE"};
142  } else if (inputType == "zsraw") {
143  gDispatchTrigger = o2::framework::Output{"TPC", "RAWDATA"};
144  }
145  // set up configuration
146  o2::conf::ConfigurableParam::updateFromFile(cfgc.options().get<std::string>("configFile"));
147  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
148  o2::conf::ConfigurableParam::writeINI("o2tpcrecoworkflow_configuration.ini");
149 
150  gTpcSectorMask = 0;
151  for (auto s : tpcSectors) {
152  gTpcSectorMask |= (1ul << s);
153  }
154 
155  bool doMC = not cfgc.options().get<bool>("disable-mc");
156  auto wf = o2::tpc::reco_workflow::getWorkflow(&gPolicyData, //
157  tpcSectors, // sector configuration
158  gTpcSectorMask, // same as bitmask
159  laneConfiguration, // lane configuration
160  doMC, //
161  nLanes, //
162  inputType, //
163  cfgc.options().get<std::string>("output-type"), //
164  cfgc.options().get<bool>("disable-root-input"), //
165  !cfgc.options().get<bool>("no-ca-clusterer"), //
166  !cfgc.options().get<bool>("no-tpc-zs-on-the-fly"), //
167  !cfgc.options().get<bool>("ignore-dist-stf") //
168  );
169 
170  // configure dpl timer to inject correct firstTFOrbit: start from the 1st orbit of TF containing 1st sampled orbit
171  o2::raw::HBFUtilsInitializer hbfIni(cfgc, wf);
172 
173  return std::move(wf);
174 }
