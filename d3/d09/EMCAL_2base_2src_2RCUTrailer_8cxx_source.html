37 void RCUTrailer::constructFromRawPayload(const gsl::span<const uint32_t> payloadwords)
38 {
39  reset();
40  int index = payloadwords.size();
41  auto word = payloadwords[--index];
42  if ((word >> 30) != 3) {
43  throw Error(Error::ErrorType_t::DECODING_INVALID, "Last RCU trailer word not found!");
44  }
45  mFirmwareVersion = (word >> 16) & 0xFF;
46 
47  mRCUId = (int)((word >> 7) & 0x1FF);
48  int trailerSize = (word & 0x7F);
49 
50  if (trailerSize < 2) {
51  throw Error(Error::ErrorType_t::SIZE_INVALID, fmt::format("Invalid trailer size found (%d bytes) !", trailerSize * 4).data());
52  }
53  mTrailerSize = trailerSize;
54 
55  trailerSize -= 2; // Cut first and last trailer words as they are handled separately
56  for (; trailerSize > 0; trailerSize--) {
57  word = payloadwords[--index];
58  if ((word >> 30) != 2) {
59  std::cerr << "Missing RCU trailer identifier pattern!\n";
60  continue;
61  }
62  int parCode = (word >> 26) & 0xF;
63  int parData = word & 0x3FFFFFF;
64  // std::cout << "Found trailer word 0x" << std::hex << word << "(Par code: " << std::dec << parCode << ", Par data: 0x" << std::hex << parData << std::dec << ")";
65  switch (parCode) {
66  case 1:
67  // ERR_REG1
68  mFECERRA = ((parData >> 13) & 0x1FFF) << 7;
69  mFECERRB = ((parData & 0x1FFF)) << 7;
70  break;
71  case 2:
72  // ERR_REG2
73  mErrorCounter.mErrorRegister2 = parData & 0x1FF;
74  break;
75  case 3:
76  // ERR_REG3
77  mErrorCounter.mErrorRegister3 = parData & 0x1FFFFFF;
78  break;
79  case 4:
80  // FEC_RO_A
81  mActiveFECsA = parData & 0xFFFF;
82  break;
83  case 5:
84  // FEC_RO_B
85  mActiveFECsB = parData & 0xFFFF;
86  break;
87  case 6:
88  // RDO_CFG1
89  mAltroConfig.mWord1 = parData & 0xFFFFF;
90  break;
91  case 7:
92  // RDO_CFG2
93  mAltroConfig.mWord2 = parData & 0x1FFFFFF;
94  break;
95  default:
96  std::cerr << "Undefined parameter code " << parCode << ", ignore it !\n";
97  break;
98  }
99  }
100  mPayloadSize = payloadwords[--index] & 0x3FFFFFF;
101  mIsInitialized = true;
102 }

104 double RCUTrailer::getTimeSampleNS() const
105 {
106  uint8_t fq = mAltroConfig.mSampleTime;
107  double tSample;
108  switch (fq) {
109  case 0:
110  // 20 MHz
111  tSample = 2.0;
112  break;
113  case 1:
114  // 10 Mhz
115  tSample = 4.0;
116  break;
117  case 2:
118  // 5 MHz
119  tSample = 8.;
120  break;
121  default:
122  throw Error(Error::ErrorType_t::SAMPLINGFREQ_INVALID, fmt::format("Invalid sampling frequency value %d !", int(fq)).data());
123  }
124 
125  return tSample * o2::constants::lhc::LHCBunchSpacingNS;
126 }

128 void RCUTrailer::setTimeSamplePhaseNS(uint64_t triggertime, uint64_t timesample)
129 {
130  int sample = 0;
131  switch (timesample) {
132  case 50:
133  sample = 0;
134  break;
135  case 100:
136  sample = 1;
137  break;
138  case 200:
139  sample = 2;
140  break;
141  default:
142  throw Error(Error::ErrorType_t::SAMPLINGFREQ_INVALID, fmt::format("invalid time sample: %f", timesample).data());
143  };
144  mAltroConfig.mSampleTime = sample;
145  // calculate L1 phase
146  mAltroConfig.mL1Phase = (triggertime % timesample) / 25;

149 double RCUTrailer::getL1PhaseNS() const
150 {
151  double tSample = getTimeSampleNS(),
152  phase = static_cast<double>(mAltroConfig.mL1Phase) * o2::constants::lhc::LHCBunchSpacingNS;
153  if (phase >= tSample) {
154  throw Error(Error::ErrorType_t::L1PHASE_INVALID, fmt::format("Invalid L1 trigger phase (%e ns (phase) >= %e ns (sampling time)) !", phase, tSample).data());
155  }

159 std::vector<uint32_t> RCUTrailer::encode() const
160 {
161  std::vector<uint32_t> encoded;
162  encoded.emplace_back(mPayloadSize | 2 << 30);
163  encoded.emplace_back(mFECERRB >> 7 | (mFECERRA >> 7) << 13 | 1 << 26 | 2 << 30);
164  encoded.emplace_back(mErrorCounter.mErrorRegister2 | 2 << 26 | 2 << 30);
165  encoded.emplace_back(mErrorCounter.mErrorRegister3 | 3 << 26 | 2 << 30);
166  encoded.emplace_back(mActiveFECsA | 4 << 26 | 2 << 30);
167  encoded.emplace_back(mActiveFECsB | 5 << 26 | 2 << 30);
168  encoded.emplace_back(mAltroConfig.mWord1 | 6 << 26 | 2 << 30);
169  encoded.emplace_back(mAltroConfig.mWord2 | 7 << 26 | 2 << 30);
170 
171  uint32_t lasttrailerword = 3 << 30 | mFirmwareVersion << 16 | mRCUId << 7 | (encoded.size() + 1);
172  encoded.emplace_back(lasttrailerword);
173 
174  return encoded;
175 }

177 void RCUTrailer::printStream(std::ostream& stream) const
178 {
179  std::vector<std::string> errors;
180  double timesample = -1., l1phase = -1.;
181  try {
182  timesample = getTimeSampleNS();
183  } catch (Error& e) {
184  errors.push_back(e.what());
185  }
186  try {
187  l1phase = getL1PhaseNS();
188  } catch (Error& e) {
189  errors.push_back(e.what());
190  }
191 
192  stream << "RCU trailer (Format version 2):\n"
193  << "==================================================\n"
194  << "RCU ID: " << mRCUId << "\n"
195  << "Firmware version: " << int(mFirmwareVersion) << "\n"
196  << "Trailer size: " << mTrailerSize << "\n"
197  << "Payload size: " << mPayloadSize << "\n"
198  << "FECERRA: 0x" << std::hex << mFECERRA << "\n"
199  << "FECERRB: 0x" << std::hex << mFECERRB << "\n"
200  << "ERRREG2: 0x" << std::hex << mErrorCounter.mErrorRegister2 << "\n"
201  << "ERRREG3: 0x" << std::hex << mErrorCounter.mErrorRegister3 << "\n"
202  << "#channels skipped due to address mismatch: " << std::dec << getNumberOfChannelAddressMismatch() << "\n"
203  << "#channels skipped due to bad block length: " << std::dec << getNumberOfChannelLengthMismatch() << "\n"
204  << "Active FECs (branch A): 0x" << std::hex << mActiveFECsA << "\n"
205  << "Active FECs (branch B): 0x" << std::hex << mActiveFECsB << "\n"
206  << "Baseline corr: " << std::hex << getBaselineCorrection() << "\n"
207  << "Polarity: " << (getPolarity() ? "yes" : "no") << "\n"
208  << "Number of presamples: " << std::dec << getNumberOfPresamples() << "\n"
209  << "Number of postsamples: " << std::dec << getNumberOfPostsamples() << "\n"
210  << "Second baseline corr: " << (hasSecondBaselineCorr() ? "yes" : "no") << "\n"
211  << "Glitch filter: " << std::dec << getGlitchFilter() << "\n"
212  << "Number of non-ZS postsamples: " << std::dec << getNumberOfNonZeroSuppressedPostsamples() << "\n"
213  << "Number of non-ZS presamples: " << std::dec << getNumberOfNonZeroSuppressedPresamples() << "\n"
214  << "Zero suppression: " << (hasZeroSuppression() ? "yes" : "no") << "\n"
215  << "Number of ALTRO buffers: " << std::dec << getNumberOfAltroBuffers() << "\n"
216  << "Number of pretrigger samples: " << std::dec << getNumberOfPretriggerSamples() << "\n"
217  << "Number of samples per channel: " << std::dec << getNumberOfSamplesPerChannel() << "\n"
218  << "Sparse readout: " << (isSparseReadout() ? "yes" : "no") << "\n"
219  << "AltroCFG1: 0x" << std::hex << mAltroConfig.mWord1 << "\n"
220  << "AltroCFG2: 0x" << std::hex << mAltroConfig.mWord2 << "\n"
221  << "Sampling time: " << timesample << " ns\n"
222  << "L1 Phase: " << l1phase << " ns\n"
223  << std::dec << std::fixed;
224  if (errors.size()) {
225  stream << "Errors: \n"
226  << "-------------------------------------------------\n";
227  for (const auto& e : errors) {
228  stream << e << "\n";
229  }
230  }
231  stream << "==================================================\n";
232 }
