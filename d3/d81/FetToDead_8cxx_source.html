45 std::vector<ColumnData> FetToDead::process(gsl::span<const ColumnData> fetData)
46 {
48  std::vector<ColumnData> deadChannels;
49  std::unordered_map<uint16_t, bool> dataIds;
50  // First we loop on FET output and returns the empty strips
51  for (auto& col : fetData) {
52  invertPattern(col, deadChannels);
53  dataIds[getColumnDataUniqueId(col.deId, col.columnId)] = true;
54  }
55 
56  // Then we loop on the active channels and check if they where among the FET data.
57  // If they are not there, it means that non of the channels answered.
58  // So the full board was dead.
59  for (auto& item : mInvertedActive) {
60  auto found = dataIds.find(item.first);
61  if (found == dataIds.end()) {
62  deadChannels.emplace_back(item.second);
63  }
64  }
65  return deadChannels;
66 }

68 bool FetToDead::invertPattern(const ColumnData& col, std::vector<ColumnData>& invertedData)
69 {
71  ColumnData invertedCol{col.deId, col.columnId};
72  invertedCol.setNonBendPattern(~col.getNonBendPattern());
73  for (int iline = 0; iline < 4; ++iline) {
74  invertedCol.setBendPattern(~col.getBendPattern(iline), iline);
75  }
76  mMasksHandler.applyMask(invertedCol);
77  if (invertedCol.isEmpty()) {
78  return false;
79  }
80  invertedData.emplace_back(invertedCol);
81  return true;
82 }
