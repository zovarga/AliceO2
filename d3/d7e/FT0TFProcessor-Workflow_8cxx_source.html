31  {

35  auto digits = pc.inputs().get<gsl::span<o2::ft0::Digit>>("digits");

37  auto& calib_data = pc.outputs().make<std::vector<o2::ft0::FT0CalibrationInfoObject>>(o2::framework::OutputRef{"calib", 0});
38  calib_data.reserve(channels.size());
39  int nDig = digits.size();
40  LOG(debug) << " nDig " << nDig;
41  for (int id = 0; id < nDig; id++) {
42  const auto& digit = digits[id];
43  // auto chan = digit.getBunchChannelData(channels);
44  for (const auto& channel : channels) {
45  if (channel.QTCAmpl > 14 && std::abs(channel.CFDTime) < 100) {
46  calib_data.emplace_back(channel.ChId, channel.CFDTime, channel.QTCAmpl, uint64_t(creationTime));
47  }
48  }
