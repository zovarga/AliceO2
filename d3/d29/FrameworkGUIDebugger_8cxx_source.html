85 void displayHistory(const DeviceInfo& info, DeviceControl& control)
86 {
87  if (info.history.empty()) {
88  return;
89  }
90  int startPos = info.historyPos;
91  const int historySize = info.history.size();
92 
93  int triggerStartPos = startPos + 1 % historySize;
94  int triggerStopPos = startPos % historySize;
95 
96  int j = startPos;
97  // We look for a stop trigger, so that we know where to stop the search for
98  // out start search. If no stop trigger is found, we search until the end
99  if (control.logStopTrigger[0]) {
100  while ((j % historySize) != ((startPos + 1) % historySize)) {
101  assert(j >= 0);
102  assert(j < historySize);
103  auto& line = info.history[j];
104  if (strstr(line.c_str(), control.logStopTrigger)) {
105  triggerStopPos = (j + 1) % historySize;
106  break;
107  }
108  // Wrap in case we end up below 0
109  j = (j == 0) ? historySize - 1 : j - 1;
110  }
111  }
112 
113  // Look for the last instance of the start trigger before the
114  // last stop trigger.
115  j = startPos + 1;
116  if (control.logStartTrigger[0]) {
117  while ((j % historySize) != triggerStopPos) {
118  assert(historySize > j);
119  assert(historySize == 1000);
120  auto& line = info.history[j];
121  if (strstr(line.c_str(), control.logStartTrigger)) {
122  triggerStartPos = j;
123  }
124  j = (j + 1) % historySize;
125  }
126  }
127 
128  // We start from the last trigger found. Eventually this is the first
129  // line in the ring buffer, if no trigger is specified.
130  size_t ji = triggerStartPos % historySize;
131  size_t je = triggerStopPos % historySize;
132  size_t iterations = 0;
133  while (historySize && ((ji % historySize) != je)) {
134  assert(iterations < historySize);
135  iterations++;
136  assert(historySize == 1000);
137  assert(ji < historySize);
138  auto& line = info.history[ji];
139  auto logLevel = info.historyLevel[ji];
140 
141  // Skip empty lines
142  if (line.empty()) {
143  ji = (ji + 1) % historySize;
144  continue;
145  }
146  // Print matching lines
147  if (strstr(line.c_str(), control.logFilter) != nullptr) {
148  auto color = colorForLogLevel(logLevel);
149  // We filter twice, once on input, to reduce the
150  // stream, a second time at display time, to avoid
151  // showing unrelevant messages from past.
152  if (logLevel >= control.logLevel) {
153  if (line.find('%', 0) != std::string::npos) {
154  ImGui::TextUnformatted(line.c_str(), line.c_str() + line.size());
155  } else {
156  ImGui::TextColored(color, line.c_str(), line.c_str() + line.size());
157  }
158  }
159  }
160  ji = (ji + 1) % historySize;
161  }
162 }

323  size_t minDomain = std::numeric_limits<size_t>::max();

326  ImPlotFlags axisFlags = 0;
327

329  std::vector<DeviceMetricsInfo> const& metricsInfos = *metricStore.metrics[si];
330  auto const& specs = *metricStore.specs[si];

332  for (size_t mi = 0; mi < metricsInfos[di].metrics.size(); ++mi) {

346  data.legend = state[gmi].legend.c_str();
347  data.type = metric.type;

350  maxValue[data.axis] = std::max(maxValue[data.axis], metricsInfos[di].max[mi]);
351  minDomain = std::min(minDomain, metricsInfos[di].minDomain[mi]);
352  maxDomain = std::max(maxDomain, metricsInfos[di].maxDomain[mi]);
353  axisFlags |= data.axis == 1 ? ImPlotFlags_YAxis2 : ImPlotFlags_None;
354  axisFlags |= data.axis == 2 ? ImPlotFlags_YAxis3 : ImPlotFlags_None;
355  switch (metric.type) {
356  case MetricType::Int: {
357  data.Y = metricsInfos[di].intMetrics[metric.storeIdx].data();

359  case MetricType::Uint64: {

362  case MetricType::Float: {
363  data.Y = metricsInfos[di].floatMetrics[metric.storeIdx].data();

371  }
372 
373  userData.emplace_back(data);
374  gmi++;
375  }
376  }
377  }
378 
379  maxDomain = std::max(minDomain + 1024, maxDomain);
380  for (size_t ai = 0; ai < 3; ++ai) {
381  maxValue[ai] = std::max(minValue[ai] + 1.f, maxValue[ai]);

384  static size_t lastMinRange = minDomain;

402  } else if (histoData->type == MetricType::Uint64) {

409  };

411  ImGui::Checkbox("Log scale", &logScale);
412 
413  ImPlot::SetNextPlotLimitsX(minDomain, maxDomain, ImGuiCond_Once);

424
