101 inline void SegmentationAlpide::localToDetectorUnchecked(float xRow, float zCol, int& iRow, int& iCol)
102 {
103  // convert to row/col w/o over/underflow check
104  xRow = 0.5 * (ActiveMatrixSizeRows - PassiveEdgeTop + PassiveEdgeReadOut) - xRow; // coordinate wrt top edge of Active matrix
105  zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt left edge of Active matrix
106  iRow = int(xRow / PitchRow);
107  iCol = int(zCol / PitchCol);
108  if (xRow < 0) {
109  iRow -= 1;
110  }
111  if (zCol < 0) {
112  iCol -= 1;
113  }
114 }

116 //_________________________________________________________________________________________________
117 inline bool SegmentationAlpide::localToDetector(float xRow, float zCol, int& iRow, int& iCol)
118 {
119  // convert to row/col
120  xRow = 0.5 * (ActiveMatrixSizeRows - PassiveEdgeTop + PassiveEdgeReadOut) - xRow; // coordinate wrt left edge of Active matrix
121  zCol += 0.5 * ActiveMatrixSizeCols; // coordinate wrt bottom edge of Active matrix
122  if (xRow < 0 || xRow >= ActiveMatrixSizeRows || zCol < 0 || zCol >= ActiveMatrixSizeCols) {
123  iRow = iCol = -1;
124  return false;
125  }
126  iRow = int(xRow / PitchRow);
127  iCol = int(zCol / PitchCol);
128  return true;
129 }
