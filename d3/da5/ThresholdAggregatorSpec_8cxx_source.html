46 void ITSThresholdAggregator::run(ProcessingContext& pc)
47 {
48 
49  // take run type and scan type at the beginning
50  if (this->mRunType == -1) {
51  mRunType = pc.inputs().get<short int>("runtype");
52  mScanType = pc.inputs().get<char>("scantype");
53  mFitType = pc.inputs().get<char>("fittype");
54  updateRunID(pc); // run number
55  }
56 
57  if (mLHCPeriod.empty()) {
58  updateLHCPeriod(pc);
59  }
60 
61  // Read strings with tuning info
62  const auto tunString = pc.inputs().get<gsl::span<char>>("tunestring");
63  // Merge all strings coming from several sources (EPN)
64  std::copy(tunString.begin(), tunString.end(), std::back_inserter(tuningMerge));
65 
66  return;

70 void ITSThresholdAggregator::finalize(EndOfStreamContext* ec)
71 {
72  LOGF(info, "endOfStream report:", mSelfName);
73 
74  // Below is CCDB stuff
75  long tstart = o2::ccdb::getCurrentTimestamp();
76  long tend = tstart + 365L * 24 * 3600 * 1000;
77 
78  auto class_name = o2::utils::MemFileHelper::getClassName(tuningMerge);
79 
80  // Create metadata for database object
81  std::string ft = this->mFitType == 0 ? "derivative" : this->mFitType == 1 ? "fit"
82  : "hitcounting";
83  std::map<std::string, std::string> md = {
84  {"fittype", ft}, {"runtype", std::to_string(this->mRunType)}};
85  if (!(this->mLHCPeriod.empty())) {
86  md.insert({"LHC_period", this->mLHCPeriod});
87  }
88  if (!mCcdbUrl.empty()) { // add only if we write here otherwise ccdb-populator-wf add it already
89  md.insert({"runNumber", std::to_string(this->mRunNumber)});
90  }
91 
92  std::string path("ITS/Calib/");
93  std::string name_str = this->mScanType == 'V' ? "VCASN" : this->mScanType == 'I' ? "ITHR"
94  : "THR";
95  o2::ccdb::CcdbObjectInfo info((path + name_str), "threshold_map", "calib_scan.root", md, tstart, tend);
96  auto image = o2::ccdb::CcdbApi::createObjectImage(&tuningMerge, &info);
97  std::string file_name = "calib_scan_" + name_str + ".root";
98  info.setFileName(file_name);
99 
100  if (ec) { // send to ccdb-populator wf only if there is an EndOfStreamContext
101  LOG(info) << "Class Name: " << class_name << " | File Name: " << file_name
102  << "\nSending to ccdb-populator the object " << info.getPath() << "/" << info.getFileName()
103  << " of size " << image->size() << " bytes, valid for "
104  << info.getStartValidityTimestamp() << " : "
105  << info.getEndValidityTimestamp();
106 
107  if (this->mScanType == 'V') {
108  ec->outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "VCASN", 0}, *image);
109  ec->outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "VCASN", 0}, info);
110  } else if (this->mScanType == 'I') {
111  ec->outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "ITHR", 0}, *image);
112  ec->outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "ITHR", 0}, info);
113  } else if (this->mScanType == 'T') {
114  ec->outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "THR", 0}, *image);
115  ec->outputs().snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "THR", 0}, info);
116  }
117  }
118 
119  if (!mCcdbUrl.empty()) { // if url is specified, send object to ccdb from THIS wf
120 
121  LOG(info) << "Sending object " << info.getFileName() << " to " << mCcdbUrl << "/browse/" << info.getPath() << " from the ITS calib workflow";
122  o2::ccdb::CcdbApi mApi;
123  mApi.init(mCcdbUrl);
124  mApi.storeAsBinaryFile(&image->at(0), image->size(), info.getFileName(), info.getObjectType(), info.getPath(),
125  info.getMetaData(), info.getStartValidityTimestamp(), info.getEndValidityTimestamp());
126  }
127 
128  return;
129 }

155 // Search current month or LHCperiod
156 void ITSThresholdAggregator::updateLHCPeriod(ProcessingContext& pc)
157 {
158  auto conf = pc.services().get<RawDeviceService>().device()->fConfig;
159  const std::string LHCPeriodStr = conf->GetProperty<std::string>("LHCPeriod", "");
160  if (!(LHCPeriodStr.empty())) {
161  this->mLHCPeriod = LHCPeriodStr;
162  } else {
163  const char* months[12] = {"JAN", "FEB", "MAR", "APR", "MAY", "JUN",
164  "JUL", "AUG", "SEP", "OCT", "NOV", "DEC"};
165  std::time_t now = std::time(nullptr);
166  std::tm* ltm = std::gmtime(&now);
167  this->mLHCPeriod = (std::string)months[ltm->tm_mon];
168  LOG(warning) << "LHCPeriod is not available, using current month " << this->mLHCPeriod;
169  }
170  this->mLHCPeriod += "_ITS";
171 
172  return;
173 }

189 DataProcessorSpec getITSThresholdAggregatorSpec()
190 {
191  std::vector<InputSpec> inputs;
192  inputs.emplace_back("tunestring", "ITS", "TSTR", 0);
193  inputs.emplace_back("runtype", "ITS", "RUNT", 0);
194  inputs.emplace_back("scantype", "ITS", "SCANT", 0);
195  inputs.emplace_back("fittype", "ITS", "FITT", 0);
196 
197  std::vector<OutputSpec> outputs;
198  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "VCASN"});
199  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "VCASN"});
200 
201  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "ITHR"});
202  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "ITHR"});
203 
204  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "THR"});
205  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "THR"});
206 
207  return DataProcessorSpec{
208  "its-aggregator",
209  inputs,
210  outputs,
211  AlgorithmSpec{adaptFromTask<o2::its::ITSThresholdAggregator>()},
212  Options{
213  {"verbose", VariantType::Bool, false, {"Use verbose output mode"}},
214  {"ccdb-url", VariantType::String, "", {"CCDB url, default is empty (i.e. no upload to CCDB)"}}}};
215 }
