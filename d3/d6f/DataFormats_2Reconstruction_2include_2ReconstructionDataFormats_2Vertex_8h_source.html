107 // Base primary vertex class, with position, error, N candidates and flags field
108 // The Stamp template parameter allows to define vertex (time)stamp in different
109 // formats (ITS ROFrame ID, real time + error etc)
110 
111 template <typename Stamp>
112 class Vertex : public VertexBase
113 {
114  public:
115  using ushort = unsigned short;
116  enum Flags : ushort {
117  TimeValidated = 0x1 << 0, // Flag that the vertex was validated by external time measurement (e.g. FIT)
118  FlagsMask = 0xffff
119  };
120 
121  GPUdDefault() Vertex() = default;
122  GPUdDefault() ~Vertex() = default;
123  GPUd() Vertex(const math_utils::Point3D<float>& pos, const gpu::gpustd::array<float, kNCov>& cov, ushort nCont, float chi2)
124  : VertexBase(pos, cov), mNContributors(nCont), mChi2(chi2)
125  {
126  }
127 
128  GPUd() ushort getNContributors() const { return mNContributors; }
129  GPUd() void setNContributors(ushort v) { mNContributors = v; }
130  GPUd() void addContributor() { mNContributors++; }
131 
132  GPUd() ushort getFlags() const { return mBits; }
133  GPUd() bool isFlagSet(uint f) const { return mBits & (FlagsMask & f); }
134  GPUd() void setFlags(ushort f) { mBits |= FlagsMask & f; }
135  GPUd() void resetFrags(ushort f = FlagsMask) { mBits &= ~(FlagsMask & f); }
136 
137  GPUd() void setChi2(float v) { mChi2 = v; }
138  GPUd() float getChi2() const { return mChi2; }
139 
140  GPUd() const Stamp& getTimeStamp() const { return mTimeStamp; }
141  GPUd() Stamp& getTimeStamp() { return mTimeStamp; }
142  GPUd() void setTimeStamp(const Stamp& v) { mTimeStamp = v; }
143 
144  protected:
145  float mChi2 = 0;
146  ushort mNContributors = 0;
147  ushort mBits = 0;
148  Stamp mTimeStamp;
149 
150  ClassDefNV(Vertex, 3);
151 };
