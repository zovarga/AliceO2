28 struct ChannelRewriter : FairMQChannelConfigParser {
29  void beginChannel() override
30  {
31  names.push_back(-1);
32  isZMQ.push_back(false);
33  hasAddress.push_back(false);
34  isWrite.push_back(false);
35  propertiesBegin.push_back(propertyIndex);
36  }
37 
38  void endChannel() override
39  {
40  propertiesEnd.push_back(propertyIndex);
41  if (names.back() == -1) {
42  throw std::runtime_error("Channel does not have a name.");
43  }
44  // If we have a zmq channel which does not have an address,
45  // use a DDS property.
46  if (isZMQ.back() && (hasAddress.back() == false)) {
47  requiresProperties.push_back(channelIndex);
48  }
49  channelIndex++;
50  }
51 
52  void property(std::string_view key, std::string_view value) override
53  {
54  properties.push_back({key, value});
55  if (key == "address") {
56  hasAddress.back() = true;
57  }
58  if (key == "transport" && value == "zeromq") {
59  isZMQ.back() = true;
60  }
61  // Channels that bind need to write the bound address
62  if (key == "method" && value == "bind") {
63  isWrite.back() = true;
64  }
65  if (key == "name") {
66  names.back() = propertyIndex;
67  }
68  propertyIndex++;
69  }
70 
71  int propertyIndex = 0;
72  int channelIndex = 0;
73  std::vector<ChannelProperties> properties;
74  std::vector<int> names;
75  std::vector<int> requiresProperties;
76  std::vector<int> propertiesBegin;
77  std::vector<int> propertiesEnd;
78  std::vector<bool> isWrite;
79  std::vector<bool> isZMQ;
80  std::vector<bool> hasAddress;
81 };

83 void dumpDeviceSpec2DDS(std::ostream& out,
84  std::string const& workflowSuffix,
85  const std::vector<DeviceSpec>& specs,
86  const std::vector<DeviceExecution>& executions,
87  const CommandInfo& commandInfo)
88 {
89  out << R"(<topology name="o2-dataflow">)"
90  "\n";
91  assert(specs.size() == executions.size());
92  std::vector<ChannelRewriter> rewriters;
93  rewriters.resize(specs.size());
94 
95  // Find out if we need properties
96  // and a property for each zmq channel which does not have
97  // and address.
98  for (size_t di = 0; di < specs.size(); ++di) {
99  auto& rewriter = rewriters[di];
100  auto& spec = specs[di];
101  auto& execution = executions[di];
102  for (size_t cci = 0; cci < execution.args.size(); cci++) {
103  const char* arg = execution.args[cci];
104  if (!arg) {
105  break;
106  }
107  if (strcmp(arg, "--channel-config") == 0) {
108  if (cci + 1 == execution.args.size()) {
109  throw std::runtime_error("wrong channel config found");
110  }
111  ChannelSpecHelpers::parseChannelConfig(execution.args[cci + 1], rewriter);
112  }
113  }
114  for (int ci : rewriter.requiresProperties) {
115  out << " "
116  << fmt::format("<property name=\"fmqchan_{}\" />\n", rewriter.properties[rewriter.names[ci]].value);
117  }
118  }
119 
120  for (size_t di = 0; di < specs.size(); ++di) {
121  auto& spec = specs[di];
122  auto& execution = executions[di];
123  if (execution.args.empty()) {
124  continue;
125  }
126 
127  out << " "
128  << fmt::format("<decltask name=\"{}{}\">\n", spec.id, workflowSuffix);
129  out << " "
130  << R"(<exe reachable="true">)";
131  out << std::regex_replace(commandInfo.command, std::regex{"--dds(?!-)"}, "--dump") << " | ";
132  for (size_t ei = 0; ei < execution.environ.size(); ++ei) {
133  out << fmt::format(execution.environ[ei],
134  fmt::arg("timeslice0", spec.inputTimesliceId),
135  fmt::arg("timeslice1", spec.inputTimesliceId + 1),
136  fmt::arg("timeslice4", spec.inputTimesliceId + 4))
137  << " ";
138  }
139  std::string accumulatedChannelPrefix;
140  char* s = strdup(execution.args[0]);
141  out << basename(s) << " ";
142  free(s);
143  for (size_t ai = 1; ai < execution.args.size(); ++ai) {
144  const char* arg = execution.args[ai];
145  if (!arg) {
146  break;
147  }
148  if (strcmp(arg, "--id") == 0 && ai + 1 < execution.args.size()) {
149  out << fmt::format(R"(--id {}_dds%TaskIndex%_%CollectionIndex% )", execution.args[ai + 1]);
150  ai++;
151  continue;
152  }
153  // Do not print out the driver client explicitly
154  if (strcmp(arg, "--driver-client-backend") == 0) {
155  ai++;
156  continue;
157  }
158  if (strcmp(arg, "--control") == 0) {
159  ai++;
160  continue;
161  }
162  if (strcmp(arg, "--channel-prefix") == 0 &&
163  ai + 1 < execution.args.size() &&
164  *execution.args[ai + 1] == 0) {
165  ai++;
166  continue;
167  }
168  if (strpbrk(arg, "' ;@") != nullptr || arg[0] == 0) {
169  out << fmt::format(R"("{}" )", arg);
170  } else if (strpbrk(arg, "\"") != nullptr || arg[0] == 0) {
171  out << fmt::format(R"('{}' )", arg);
172  } else {
173  out << fmt::format(R"({} )", arg);
174  }
175  }
176  out << "--plugin odc";
177  if (accumulatedChannelPrefix.empty() == false) {
178  out << " --channel-config \"" << accumulatedChannelPrefix << "\"";
179  }
180  out << "</exe>\n";
181  auto& rewriter = rewriters[di];
182  if (rewriter.requiresProperties.empty() == false) {
183  out << " <properties>\n";
184  for (auto pi : rewriter.requiresProperties) {
185  out << fmt::format(
186  " <name access=\"{}\">fmqchan_{}</name>\n",
187  rewriter.isWrite[pi] ? "write" : "read",
188  rewriter.properties[rewriter.names[pi]].value);
189  }
190  out << " </properties>\n";
191  }
192  out << " </decltask>\n";
193  }
194  out << " <declcollection name=\"DPL\">\n <tasks>\n";
195  for (size_t di = 0; di < specs.size(); ++di) {
196  out << fmt::format(" <name>{}{}</name>\n", specs[di].id, workflowSuffix);
197  }
198  out << " </tasks>\n </declcollection>\n";
199  out << "</topology>\n";
200 }
