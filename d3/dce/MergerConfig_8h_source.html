27 enum class InputObjectsTimespan {
28  FullHistory, // Mergers expect objects with all data accumulated so far each time.
29  LastDifference // Mergers expect objects' differences (what has changed since the previous were sent).
30 };
31 
32 enum class MergedObjectTimespan {
33  // Merged object should be an sum of differences received since the beginning
34  // or a sum of latest versions of objects received on each input.
35  FullHistory,
36  // Merged object should be an sum of differences received after last publication.
37  // Merged object is reset after published. It won't produce meaningful results
38  // when InputObjectsTimespan::FullHistory is set.
39  LastDifference,
40  // Generalisation of the two above. Resets all objects in Mergers after n cycles (0 - infinite).
41  // The the above will be removed once we switch to NCycles in QC.
42  NCycles
43 };
44 
45 enum class PublicationDecision {
46  EachNSeconds, // Merged object is published each N seconds.
47 };
