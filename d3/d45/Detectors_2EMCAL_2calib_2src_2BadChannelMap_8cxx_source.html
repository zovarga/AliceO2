49 BadChannelMap::MaskType_t BadChannelMap::getChannelStatus(unsigned short channelID) const
50 {
51  auto status = MaskType_t::GOOD_CELL;
52  if (mDeadCells.test(channelID)) {
53  status = MaskType_t::DEAD_CELL;
54  } else if (mBadCells.test(channelID)) {
55  status = MaskType_t::BAD_CELL;
56  } else if (mWarmCells.test(channelID)) {
57  status = MaskType_t::WARM_CELL;
58  }
59  return status;
60 }

62 TH2* BadChannelMap::getHistogramRepresentation() const
63 {
64  const int MAXROWS = 208,
65  MAXCOLS = 96;
66  auto hist = new TH2S("badchannelmap", "Bad Channel Map", MAXCOLS, -0.5, double(MAXCOLS) - 0.5, MAXROWS, -0.5, double(MAXROWS) - 0.5);
67  hist->SetDirectory(nullptr);
68  auto geo = Geometry::GetInstance();
69  if (!geo) {
70  LOG(error) << "Geometry needs to be initialized";
71  return hist;
72  }
73 
74  for (size_t cellID = 0; cellID < mBadCells.size(); cellID++) {
75  int value(0);
76  if (mBadCells.test(cellID)) {
77  value = 1;
78  } else if (mDeadCells.test(cellID)) {
79  value = 2;
80  } else if (mWarmCells.test(cellID)) {
81  value = 3;
82  }
83  if (value) {
84  auto position = geo->GlobalRowColFromIndex(cellID);
85  hist->Fill(std::get<1>(position), std::get<0>(position), value);
86  }
87  }
88  return hist;
89 }

91 BadChannelMap& BadChannelMap::operator+=(const BadChannelMap& rhs)
92 {
93  for (size_t cellID = 0; cellID < mDeadCells.size(); cellID++) {
94  if (rhs.mDeadCells.test(cellID)) {
95  mDeadCells.set(cellID);
96  if (mBadCells.test(cellID)) {
97  mBadCells.reset(cellID);
98  }
99  if (mWarmCells.test(cellID)) {
100  mWarmCells.reset(cellID);
101  }
102  }
103  if (rhs.mBadCells.test(cellID) && !mDeadCells.test(cellID)) {
104  mBadCells.set(cellID);
105  if (mWarmCells.test(cellID)) {
106  mWarmCells.reset(cellID);
107  }
108  }
109  if (rhs.mWarmCells.test(cellID) && !(mBadCells.test(cellID) || mDeadCells.test(cellID))) {
110  mWarmCells.set(cellID);
111  }
112  }
113  return *this;
114 }
