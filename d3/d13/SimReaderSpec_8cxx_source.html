42 DataProcessorSpec getSimReaderSpec(SubspecRange range, const std::vector<std::string>& simprefixes, const std::vector<int>& tpcsectors)
43 {
44  uint64_t activeSectors = 0;
45  for (const auto& tpcsector : tpcsectors) {
46  activeSectors |= (uint64_t)0x1 << tpcsector;
47  }
48 
49  auto doit = [range, tpcsectors, activeSectors](ProcessingContext& pc) {
50  auto& mgr = steer::HitProcessingManager::instance();
51  auto eventrecords = mgr.getDigitizationContext().getEventRecords();
52  const auto& context = mgr.getDigitizationContext();
53 
54  for (auto const& sector : tpcsectors) {
55  // Note: the TPC sector header was serving the sector to lane mapping before
56  // now the only remaining purpose is to propagate the mask of valid sectors
57  // in principle even that is not necessary any more because every sector has
58  // a dedicated route bound to the sector numbers as subspecification and the
59  // merging can be done based on that. However if we in the future go over to
60  // a multipart scheme again, we again will need the information, so we keep it
61  // For the moment, sector member in the sector header is in sync with
62  // subspecification of the route
63  o2::tpc::TPCSectorHeader header{sector};
64  header.activeSectors = activeSectors;
65  pc.outputs().snapshot(OutputRef{"collisioncontext", static_cast<SubSpecificationType>(sector), {header}},
66  context);
67  }
68 
69  // the first 36 channel numbers are reserved for the TPC, now publish the remaining
70  // channels
71  for (int subchannel = range.min; subchannel < range.max; ++subchannel) {
72  LOG(info) << "SENDING SOMETHING TO OTHERS";
73  pc.outputs().snapshot(
74  OutputRef{"collisioncontext", static_cast<SubSpecificationType>(subchannel)},
75  context);
76  }
77 
78  // digitizer workflow runs only once
79  // send endOfData control event and mark the reader as ready to finish
80  pc.services().get<ControlService>().endOfStream();
81  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
82  };
83 
84  // init function return a lambda taking a ProcessingContext
85  auto initIt = [simprefixes, doit](InitContext& ctx) {
86  // initialize fundamental objects
87  auto& mgr = steer::HitProcessingManager::instance();
88 
89  // init gRandom to random start
90  // TODO: offer option to set seed
91  gRandom->SetSeed(0);
92 
93  if (simprefixes.size() == 0) {
94  LOG(error) << "No simulation prefix available";
95  } else {
96  LOG(info) << "adding " << simprefixes[0] << "\n";
97  mgr.addInputFile(simprefixes[0]);
98  for (int part = 1; part < simprefixes.size(); ++part) {
99  mgr.addInputSignalFile(simprefixes[part]);
100  }
101  }
102 
103  // do we start from an existing context
104  auto incontextstring = ctx.options().get<std::string>("incontext");
105  LOG(info) << "INCONTEXTSTRING " << incontextstring;
106  if (incontextstring.size() > 0) {
107  auto success = mgr.setupRunFromExistingContext(incontextstring.c_str());
108  if (!success) {
109  LOG(fatal) << "Could not read collision context from " << incontextstring;
110  }
111  } else {
112 
113  auto intRate = ctx.options().get<float>("interactionRate"); // is interaction rate requested?
114  if (intRate < 1.f) {
115  intRate = 1.f;
116  }
117  LOG(info) << "Imposing hadronic interaction rate " << intRate << "Hz";
118  mgr.getInteractionSampler().setInteractionRate(intRate);
119  o2::raw::HBFUtils::Instance().print();
120  o2::raw::HBFUtils::Instance().checkConsistency();
121  mgr.getInteractionSampler().setFirstIR({0, o2::raw::HBFUtils::Instance().orbitFirstSampled});
122  mgr.getDigitizationContext().setFirstOrbitForSampling(o2::raw::HBFUtils::Instance().orbitFirstSampled);
123 
124  auto bcPatternFile = ctx.options().get<std::string>("bcPatternFile");
125  if (!bcPatternFile.empty()) {
126  mgr.getInteractionSampler().setBunchFilling(bcPatternFile);
127  }
128 
129  mgr.getInteractionSampler().init();
130  mgr.getInteractionSampler().print();
131  // doing a random event selection/subsampling?
132  mgr.setRandomEventSequence(ctx.options().get<int>("randomsample") > 0);
133 
134  // finalize collisions (with number of collisions asked)
135  auto col = ctx.options().get<int>("ncollisions");
136  if (col != 0) {
137  mgr.setupRun(col);
138  } else {
139  mgr.setupRun();
140  }
141 
142  // --- we add QED contributions to the digitization context
143  // --- for now in between first and last real collision
144  auto qedprefix = ctx.options().get<std::string>("simPrefixQED");
145  if (qedprefix.size() > 0) {
146  o2::steer::InteractionSampler qedInteractionSampler;
147  if (!bcPatternFile.empty()) {
148  qedInteractionSampler.setBunchFilling(bcPatternFile);
149  }
150 
151  // get first and last "hadronic" interaction records and let
152  // QED events range from the first bunch crossing to the last bunch crossing
153  // in this range
154  auto first = mgr.getDigitizationContext().getEventRecords().front();
155  auto last = mgr.getDigitizationContext().getEventRecords().back();
156  first.bc = 0;
157  last.bc = o2::constants::lhc::LHCMaxBunches;
158 
159  const float ratio = ctx.options().get<float>("qed-x-section-ratio");
160  if (ratio <= 0.) {
161  throw std::runtime_error("no meaningful qed-x-section-ratio was provided");
162  }
163  const float hadronicrate = ctx.options().get<float>("interactionRate");
164  const float qedrate = ratio * hadronicrate;
165  LOG(info) << "QED RATE " << qedrate;
166  qedInteractionSampler.setInteractionRate(qedrate);
167  qedInteractionSampler.setFirstIR(first);
168  qedInteractionSampler.init();
169  qedInteractionSampler.print();
170  std::vector<o2::InteractionTimeRecord> qedinteractionrecords;
171  o2::InteractionTimeRecord t;
172  LOG(info) << "GENERATING COL TIMES";
173  t = qedInteractionSampler.generateCollisionTime();
174  while ((t = qedInteractionSampler.generateCollisionTime()) < last) {
175  qedinteractionrecords.push_back(t);
176  }
177  LOG(info) << "DONE GENERATING COL TIMES";
178 
179  // get digitization context and add QED stuff
180  mgr.getDigitizationContext().fillQED(qedprefix, qedinteractionrecords);
181  mgr.getDigitizationContext().printCollisionSummary(true, 2000); // print with QED but truncate output
182  }
183  // --- end addition of QED contributions
184 
185  LOG(info) << "Initializing Spec ... have " << mgr.getDigitizationContext().getEventRecords().size() << " times ";
186  LOG(info) << "Serializing Context for later reuse";
187  mgr.writeDigitizationContext(ctx.options().get<std::string>("outcontext").c_str());
188  }
189 
190  return doit;
191  };
192 
193  std::vector<OutputSpec> outputs;
194  for (auto const& tpcsector : tpcsectors) {
195  outputs.emplace_back(
196  OutputSpec{{"collisioncontext"}, "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(tpcsector), Lifetime::Timeframe});
197  }
198  for (int subchannel = range.min; subchannel < range.max; ++subchannel) {
199  outputs.emplace_back(
200  OutputSpec{{"collisioncontext"}, "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(subchannel), Lifetime::Timeframe});
201  }
202 
203  return DataProcessorSpec{
204  /*ID*/ "SimReader",
205  /*INPUT CHANNELS*/ Inputs{}, outputs,
206  /* ALGORITHM */
207  AlgorithmSpec{initIt},
208  /* OPTIONS */
209  Options{
210  {"interactionRate", VariantType::Float, 50000.0f, {"Total hadronic interaction rate (Hz)"}},
211  {"bcPatternFile", VariantType::String, "", {"Interacting BC pattern file (e.g. from CreateBCPattern.C)"}},
212  {"simPrefixQED", VariantType::String, "", {"Sim (QED) input prefix (example: path/o2qed). The prefix allows to find files like path/o2qed_Kine.root etc."}},
213  {"qed-x-section-ratio", VariantType::Float, -1.f, {"Ratio of cross sections QED/hadronic events. Determines QED interaction rate from hadronic interaction rate."}},
214  {"outcontext", VariantType::String, "collisioncontext.root", {"Output file for collision context"}},
215  {"incontext", VariantType::String, "", {"Take collision context from this file"}},
216  {"ncollisions,n",
217  VariantType::Int,
218  0,
219  {"number of collisions to sample (default is given by number of entries in chain"}},
220  {"randomsample", VariantType::Int, 0, {"Draw collisions random instead of linear sequence. (Default no = 0)"}}}};
221 }
