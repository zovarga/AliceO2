53 void DumpRaw::init()
54 {
55  gROOT->SetBatch();
56  auto& sopt = ZDCSimParam::Instance();
57  int nbx = (sopt.nBCAheadTrig + 1) * NTimeBinsPerBC;
58  double xmin = -sopt.nBCAheadTrig * NTimeBinsPerBC - 0.5;
59  double xmax = NTimeBinsPerBC - 0.5;
60  for (uint32_t i = 0; i < NDigiChannels; i++) {
61  uint32_t imod = i / NChPerModule;
62  uint32_t ich = i % NChPerModule;
63  if (mBaseline[i]) {
64  mBaseline[i]->Reset();
65  } else {
66  TString hname = TString::Format("hp%d%d", imod, ich);
67  TString htit = TString::Format("Baseline mod. %d ch. %d;Average orbit baseline", imod, ich);
68  //mBaseline[i]=new TH1F(hname,htit,ADCRange,ADCMin-0.5,ADCMax+0.5);
69  mBaseline[i] = new TH1F(hname, htit, 16378, -0.125, ADCMax + 0.125);
70  }
71  if (mCounts[i]) {
72  mCounts[i]->Reset();
73  } else {
74  TString hname = TString::Format("hc%d%d", imod, ich);
75  TString htit = TString::Format("Counts mod. %d ch. %d; Orbit hits", imod, ich);
76  mCounts[i] = new TH1F(hname, htit, o2::constants::lhc::LHCMaxBunches + 1, -0.5, o2::constants::lhc::LHCMaxBunches + 0.5);
77  }
78  if (mSignal[i]) {
79  mSignal[i]->Reset();
80  } else {
81  TString hname = TString::Format("hs%d%d", imod, ich);
82  TString htit = TString::Format("Signal mod. %d ch. %d; Sample; ADC", imod, ich);
83  mSignal[i] = new TH2F(hname, htit, nbx, xmin, xmax, ADCRange, ADCMin - 0.5, ADCMax + 0.5);
84  }
85  if (mBunch[i]) {
86  mBunch[i]->Reset();
87  } else {
88  TString hname = TString::Format("hb%d%d", imod, ich);
89  TString htit = TString::Format("Bunch mod. %d ch. %d; Sample; ADC", imod, ich);
90  mBunch[i] = new TH2F(hname, htit, 100, -0.5, 99.5, 36, -35.5, 0.5);
91  }
92  }
93  // Word id not present in payload
94  mCh.f.fixed_0 = Id_wn;
95  mCh.f.fixed_1 = Id_wn;
96  mCh.f.fixed_2 = Id_wn;
97 }

99 void DumpRaw::write()
100 {
101  TFile* f = new TFile("ZDCDumpRaw.root", "recreate");
102  if (f->IsZombie()) {
103  LOG(fatal) << "Cannot write to file " << f->GetName();
104  return;
105  }
106  for (uint32_t i = 0; i < NDigiChannels; i++) {
107  if (mBunch[i] && mBunch[i]->GetEntries() > 0) {
108  setStat(mBunch[i]);
109  mBunch[i]->Write();
110  }
111  }
112  for (uint32_t i = 0; i < NDigiChannels; i++) {
113  if (mBaseline[i] && mBaseline[i]->GetEntries() > 0) {
114  setStat(mBaseline[i]);
115  mBaseline[i]->Write();
116  }
117  }
118  for (uint32_t i = 0; i < NDigiChannels; i++) {
119  if (mCounts[i] && mCounts[i]->GetEntries() > 0) {
120  setStat(mCounts[i]);
121  mCounts[i]->Write();
122  }
123  }
124  for (uint32_t i = 0; i < NDigiChannels; i++) {
125  if (mSignal[i] && mSignal[i]->GetEntries() > 0) {
126  setStat(mSignal[i]);
127  mSignal[i]->Write();
128  }
129  }
130  f->Close();
131 }

144 int DumpRaw::processWord(const uint32_t* word)
145 {
146  if (word == nullptr) {
147  printf("NULL\n");
148  return 1;
149  }
150  if ((word[0] & 0x3) == Id_w0) {
151  for (int32_t iw = 0; iw < NWPerGBTW; iw++) {
152  mCh.w[0][iw] = word[iw];
153  }
154  } else if ((word[0] & 0x3) == Id_w1) {
155  if (mCh.f.fixed_0 == Id_w0) {
156  for (int32_t iw = 0; iw < NWPerGBTW; iw++) {
157  mCh.w[1][iw] = word[iw];
158  }
159  } else {
160  LOG(error) << "Wrong word sequence";
161  mCh.f.fixed_0 = Id_wn;
162  mCh.f.fixed_1 = Id_wn;
163  mCh.f.fixed_2 = Id_wn;
164  }
165  } else if ((word[0] & 0x3) == Id_w2) {
166  if (mCh.f.fixed_0 == Id_w0 && mCh.f.fixed_1 == Id_w1) {
167  for (int32_t iw = 0; iw < NWPerGBTW; iw++) {
168  mCh.w[2][iw] = word[iw];
169  }
170  process(mCh);
171  } else {
172  LOG(error) << "Wrong word sequence";
173  }
174  mCh.f.fixed_0 = Id_wn;
175  mCh.f.fixed_1 = Id_wn;
176  mCh.f.fixed_2 = Id_wn;
177  } else {
178  // Word not present in payload
179  LOG(fatal) << "Event format error";
180  return 1;
181  }
182  return 0;
183 }

185 int DumpRaw::process(const EventChData& ch)
186 {
187  static constexpr int last_bc = o2::constants::lhc::LHCMaxBunches - 1;
188  // Not empty event
189  auto f = ch.f;
190  int ih = getHPos(f.board, f.ch);
191  if (mVerbosity > 0) {
192  for (int32_t iw = 0; iw < NWPerBc; iw++) {
193  Digits2Raw::print_gbt_word(ch.w[iw]);
194  }
195  }
196  uint16_t us[12];
197  int16_t s[12];
198  us[0] = f.s00;
199  us[1] = f.s01;
200  us[2] = f.s02;
201  us[3] = f.s03;
202  us[4] = f.s04;
203  us[5] = f.s05;
204  us[6] = f.s06;
205  us[7] = f.s07;
206  us[8] = f.s08;
207  us[9] = f.s09;
208  us[10] = f.s10;
209  us[11] = f.s11;
210  for (int32_t i = 0; i < 12; i++) {
211  if (us[i] > ADCMax) {
212  s[i] = us[i] - ADCRange;
213  } else {
214  s[i] = us[i];
215  }
216  //printf("%d %u %d\n",i,us[i],s[i]);
217  }
218  if (f.Alice_3) {
219  for (int32_t i = 0; i < 12; i++) {
220  mSignal[ih]->Fill(i - 36., double(s[i]));
221  }
222  }
223  if (f.Alice_2) {
224  for (int32_t i = 0; i < 12; i++) {
225  mSignal[ih]->Fill(i - 24., double(s[i]));
226  }
227  }
228  if (f.Alice_1 || f.Auto_1) {
229  for (int32_t i = 0; i < 12; i++) {
230  mSignal[ih]->Fill(i - 12., double(s[i]));
231  }
232  }
233  if (f.Alice_0 || f.Auto_0) {
234  for (int32_t i = 0; i < 12; i++) {
235  mSignal[ih]->Fill(i + 0., double(s[i]));
236  }
237  double bc_d = uint32_t(f.bc / 100);
238  double bc_m = uint32_t(f.bc % 100);
239  mBunch[ih]->Fill(bc_m, -bc_d);
240  }
241  if (f.bc == last_bc) {
242  int32_t offset = f.offset - 32768;
243  double foffset = offset / 8.;
244  mBaseline[ih]->Fill(foffset);
245  mCounts[ih]->Fill(f.hits);
246  }
247  return 0;
248 }

250 int DumpRaw::process(const EventData& ev)
251 {
252  for (int32_t im = 0; im < NModules; im++) {
253  for (int32_t ic = 0; ic < NChPerModule; ic++) {
254  if (ev.data[im][ic].f.fixed_0 == Id_w0 && ev.data[im][ic].f.fixed_1 == Id_w1 && ev.data[im][ic].f.fixed_2 == Id_w2) {
255  process(ev.data[im][ic]);
256  } else if (ev.data[im][ic].f.fixed_0 == 0 && ev.data[im][ic].f.fixed_1 == 0 && ev.data[im][ic].f.fixed_2 == 0) {
257  // Empty channel
258  } else {
259  LOG(error) << "Data format error";
260  }
261  }
262  }
263  return 0;
264 }
