50 int InputHelper::addInputSpecs(const ConfigContext& configcontext, WorkflowSpec& specs,
51  GID::mask_t maskClusters, GID::mask_t maskMatches, GID::mask_t maskTracks,
52  bool useMC, GID::mask_t maskClustersMC, GID::mask_t maskTracksMC,
53  bool subSpecStrict)
54 {
55  if (configcontext.options().get<bool>("disable-root-input")) {
56  return 0;
57  }
58  if (useMC && configcontext.options().get<bool>("disable-mc")) {
59  useMC = false;
60  }
61  if (!useMC) {
62  maskClustersMC = GID::getSourcesMask(GID::NONE);
63  maskTracksMC = GID::getSourcesMask(GID::NONE);
64  }
65 
66  if (maskTracks[GID::ITS]) {
67  specs.emplace_back(o2::its::getITSTrackReaderSpec(maskTracksMC[GID::ITS]));
68  }
69  if (maskClusters[GID::ITS]) {
70  specs.emplace_back(o2::itsmft::getITSClusterReaderSpec(maskClustersMC[GID::ITS], true));
71  }
72  if (maskTracks[GID::MFT]) {
73  specs.emplace_back(o2::mft::getMFTTrackReaderSpec(maskTracksMC[GID::MFT]));
74  }
75  if (maskClusters[GID::MFT]) {
76  specs.emplace_back(o2::itsmft::getMFTClusterReaderSpec(maskClustersMC[GID::MFT], true));
77  }
78  if (maskTracks[GID::MCH]) {
79  specs.emplace_back(o2::mch::getTrackReaderSpec(maskTracksMC[GID::MCH]));
80  }
81  if (maskTracks[GID::MID]) {
82  specs.emplace_back(o2::mid::getTrackReaderSpec(maskTracksMC[GID::MID]));
83  }
84  if (maskTracks[GID::TPC]) {
85  specs.emplace_back(o2::tpc::getTPCTrackReaderSpec(maskTracksMC[GID::TPC]));
86  }
87  if (maskClusters[GID::TPC]) {
88  specs.emplace_back(o2::tpc::getClusterReaderSpec(maskClustersMC[GID::TPC]));
89  }
90  if (maskTracks[GID::TPC] && maskClusters[GID::TPC]) {
91  specs.emplace_back(o2::tpc::getClusterSharingMapSpec());
92  }
93  if (maskMatches[GID::ITSTPC] || maskMatches[GID::ITSTPCTOF] || maskTracks[GID::ITSTPC] || maskTracks[GID::ITSTPCTOF]) {
94  specs.emplace_back(o2::globaltracking::getTrackTPCITSReaderSpec(maskTracksMC[GID::ITSTPC] || maskTracksMC[GID::ITSTPCTOF]));
95  }
96  if (maskMatches[GID::ITSTPCTOF] || maskTracks[GID::ITSTPCTOF]) {
97  specs.emplace_back(o2::tof::getTOFMatchedReaderSpec(maskTracksMC[GID::ITSTPCTOF], 1, /*maskTracks[GID::ITSTPCTOF]*/ false)); // ITSTPCTOF does not provide tracks, only matchInfo
98  }
99  if (maskMatches[GID::MFTMCH]) {
100  specs.emplace_back(o2::globaltracking::getMFTMCHMatchedReaderSpec(maskTracksMC[GID::MFTMCH])); // MFTMCH matches does not provide tracks, only matchInfo
101  }
102  if (maskMatches[GID::MCHMID]) {
103  specs.emplace_back(o2::globaltracking::getMCHMIDMatchedReaderSpec(maskTracksMC[GID::MCHMID])); // MCHMID matches does not provide tracks, only matchInfo
104  }
105  if (maskMatches[GID::ITSTPCTRDTOF] || maskTracks[GID::ITSTPCTRDTOF]) {
106  specs.emplace_back(o2::tof::getTOFMatchedReaderSpec(maskTracksMC[GID::ITSTPCTRDTOF], 3, /*maskTracks[GID::ITSTPCTOF]*/ false)); // ITSTPCTOF does not provide tracks, only matchInfo
107  }
108  if (maskMatches[GID::TPCTRDTOF] || maskTracks[GID::TPCTRDTOF]) {
109  specs.emplace_back(o2::tof::getTOFMatchedReaderSpec(maskTracksMC[GID::TPCTRDTOF], 2, /*maskTracks[GID::ITSTPCTOF]*/ false)); // ITSTPCTOF does not provide tracks, only matchInfo
110  }
111  if (maskClusters[GID::TOF] || maskTracks[GID::ITSTPCTOF]) { // Note: maskTracks[GID::ITSTPCTOF] is only here to match the behavior of RecoContainer::requestTracks
112  specs.emplace_back(o2::tof::getClusterReaderSpec(maskClustersMC[GID::TOF]));
113  }
114  if (maskMatches[GID::TPCTOF]) {
115  specs.emplace_back(o2::tof::getTOFMatchedReaderSpec(maskTracksMC[GID::TPCTOF], 0, maskTracks[GID::TPCTOF], subSpecStrict));
116  }
117  if (maskTracks[GID::FT0] || maskClusters[GID::FT0]) {
118  specs.emplace_back(o2::ft0::getRecPointReaderSpec(maskTracksMC[GID::FT0] || maskClustersMC[GID::FT0]));
119  }
120  if (maskTracks[GID::FV0] || maskClusters[GID::FV0]) {
121  specs.emplace_back(o2::fv0::getRecPointReaderSpec(maskTracksMC[GID::FV0] || maskClustersMC[GID::FV0]));
122  }
123  if (maskTracks[GID::FDD] || maskClusters[GID::FDD]) {
124  specs.emplace_back(o2::fdd::getFDDRecPointReaderSpec(maskTracksMC[GID::FDD] || maskClustersMC[GID::FDD]));
125  }
126  if (maskTracks[GID::ZDC] || maskClusters[GID::ZDC]) {
127  specs.emplace_back(o2::zdc::getRecEventReaderSpec(maskTracksMC[GID::ZDC] || maskClustersMC[GID::ZDC]));
128  }
129 
130  if (maskClusters[GID::TRD]) {
131  specs.emplace_back(o2::trd::getTRDTrackletReaderSpec(maskClustersMC[GID::TRD], true));
132  }
133  if (maskTracks[GID::ITSTPCTRD] || maskTracks[GID::ITSTPCTRDTOF]) {
134  specs.emplace_back(o2::trd::getTRDGlobalTrackReaderSpec(maskTracksMC[GID::ITSTPCTRD]));
135  }
136  if (maskTracks[GID::TPCTRD] || maskTracks[GID::TPCTRDTOF]) {
137  specs.emplace_back(o2::trd::getTRDTPCTrackReaderSpec(maskTracksMC[GID::TPCTRD], subSpecStrict));
138  }
139  if (maskTracks[GID::MFTMCH]) {
140  specs.emplace_back(o2::globaltracking::getGlobalFwdTrackReaderSpec(maskTracksMC[GID::MFTMCH]));
141  }
142  if (maskTracks[GID::CTP] && maskClusters[GID::CTP]) {
143  specs.emplace_back(o2::ctp::getDigitsReaderSpec(maskTracksMC[GID::CTP] || maskClustersMC[GID::CTP]));
144  }
145 
146  if (maskTracks[GID::PHS] && maskClusters[GID::PHS]) {
147  specs.emplace_back(o2::phos::getCellReaderSpec(maskTracksMC[GID::PHS] || maskClustersMC[GID::PHS]));
148  }
149 
150  if (maskTracks[GID::CPV] && maskClusters[GID::CPV]) {
151  specs.emplace_back(o2::cpv::getClustersReaderSpec(maskTracksMC[GID::CPV] || maskClustersMC[GID::CPV]));
152  }
153 
154  if (maskTracks[GID::EMC] && maskClusters[GID::EMC]) {
155  specs.emplace_back(o2::emcal::getCellReaderSpec(maskTracksMC[GID::EMC] || maskClustersMC[GID::EMC]));
156  }
157 
158  return 0;
159 }
