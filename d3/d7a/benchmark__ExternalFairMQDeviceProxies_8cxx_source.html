100  struct BenchmarkState {
101  size_t logPeriod = 2;
102  size_t runningTime = 30;
103  size_t eventCount = 0;
104  size_t totalEventCount = 0;
105  size_t msgCount = 0;
106  size_t msgSize = 0;
107  size_t totalMsgCount = 0;
108  size_t totalMsgSize = 0;
109  benchclock::time_point startTime = benchclock::now();
110  benchclock::time_point idleTime = benchclock::now();
111  benchclock::time_point lastLogTime = benchclock::now();
112  float maxMsgPerSec = .0;
113  float maxDataRatePerSec = .0;
114  float totalIdleTime = .0;
115  float maxIdleTime = .0;
116  };

118  auto makeBenchmarkState = [&config]() -> std::shared_ptr<BenchmarkState> {
119  auto state = std::make_shared<BenchmarkState>();
120  state->runningTime = config.options().get<int>("runningTime");
121  return state;
122  };
123 
124  auto loggerInit = [](BenchmarkState& state) {
125  state.startTime = benchclock::now();
126  state.idleTime = benchclock::now();
127  state.lastLogTime = benchclock::now();
128  state.totalIdleTime = 0.;
129  };
130 
131  auto loggerCycle = [](BenchmarkState& state, size_t msgCount, size_t msgSize) {
132  ++state.eventCount;
133  state.msgCount += msgCount;
134  state.msgSize += msgSize;
135  auto secSinceLastLog = std::chrono::duration_cast<std::chrono::seconds>(benchclock::now() - state.lastLogTime);
136  if (secSinceLastLog.count() >= state.logPeriod) {
137  // TODO: introduce real counters for accumulated number of messages and message size
138  state.totalEventCount += state.eventCount;
139  state.totalMsgCount += state.msgCount;
140  state.totalMsgSize += state.msgSize;
141  float eventRate = state.eventCount / secSinceLastLog.count();
142  float msgPerSec = state.msgCount / secSinceLastLog.count();
143  float kbPerSec = state.msgSize / (1024 * secSinceLastLog.count());
144  auto elapsedTime = std::chrono::duration_cast<std::chrono::seconds>(benchclock::now() - state.startTime);
145  LOG(info) << fmt::format(
146  "{: 3d}s: Total messages: {} - Event rate {:.2f} Hz {:.2f} msg/s {:.2f} MB/s, "
147  "Accumulated idle time {:.2f} ms",
148  elapsedTime.count(), state.totalEventCount, eventRate, msgPerSec,
149  kbPerSec / 1024, state.totalIdleTime / 1000);
150  if (state.maxMsgPerSec < msgPerSec) {
151  state.maxMsgPerSec = msgPerSec;
152  }
153  if (state.maxDataRatePerSec < kbPerSec) {
154  state.maxDataRatePerSec = kbPerSec;
155  }
156  state.eventCount = 0;
157  state.msgCount = 0;
158  state.msgSize = 0;
159  state.lastLogTime = benchclock::now();
160  }
161  };
162 
163  struct ActiveGuard {
164  ActiveGuard(BenchmarkState& _state) : state(_state)
165  {
166  auto idleTime = std::chrono::duration_cast<std::chrono::microseconds>(benchclock::now() - state.idleTime);
167  state.totalIdleTime += idleTime.count();
168  }
169  ~ActiveGuard()
170  {
171  state.idleTime = benchclock::now();
172  }
173  BenchmarkState& state;
174  };
175 
176  auto loggerSummary = [](BenchmarkState& state) {
177  auto totalTime = std::chrono::duration_cast<std::chrono::seconds>(benchclock::now() - state.startTime);
178  if (totalTime.count() == 0 || state.totalEventCount == 0) {
179  return;
180  }
181  float eventRate = state.totalEventCount / totalTime.count();
182  float msgPerSec = state.totalMsgCount / totalTime.count();
183  float kbPerSec = state.totalMsgSize / (1024 * totalTime.count());
184  LOG(info) << fmt::format(
185  "Benchmarking "
186 #ifndef NDEBUG
187  "accumulated "
188 #endif
189  "for {} s: Avrg event rate {:.2f} Hz, "
190  "Avrg message rate {:.2f}/s (max {:.2f}/s), Avrg data rate {:.2f} MB/s (max {:.2f} MB/s), "
191  "Avrg idle time {:.2f} ms",
192  totalTime.count(), eventRate,
193  msgPerSec, state.maxMsgPerSec,
194  kbPerSec / 1024, state.maxDataRatePerSec / 1024,
195  state.totalIdleTime / (state.totalEventCount * 1000));
196  };

278  auto producerCallback = [pState, loggerCycle, loggerSummary, attributes](InputRecord& inputs, DataAllocator& outputs, ControlService& control, RawDeviceService& rds) {
279  auto& state = *pState;
280  ActiveGuard g(state);
281 
282  FairMQDevice& device = *(rds.device());
283  auto transport = device.GetChannel(attributes->channelName, 0).Transport();
284  auto channelAlloc = o2::pmr::getTransportAllocator(transport);
285 
286  DataProcessingHeader dph{attributes->iteration, 0};
287  FairMQParts messages;
288  size_t nHeaders = 0;
289  size_t totalPayload = 0;
290  size_t allocatedSize = 0;
291  auto createMessage = [&transport, &allocatedSize](size_t size) -> FairMQMessagePtr {
292  auto msg = transport->CreateMessage(size);
293  allocatedSize += size;
294  return msg;
295  };
296  auto insertHeader = [&dph, &createMessage, &messages, &nHeaders](DataHeader const& dh) -> void {
297  Stack stack{dh, dph};
298  FairMQMessagePtr header = createMessage(stack.size());
299  memcpy(header->GetData(), stack.data(), stack.size());
300  messages.AddPart(std::move(header));
301  ++nHeaders;
302  };
303  auto insertPayload = [&createMessage, &messages, &totalPayload](size_t size) -> void {
304  FairMQMessagePtr payload = createMessage(size);
305  messages.AddPart(std::move(payload));
306  totalPayload += size;
307  };
308  auto createSequence = [&attributes, &insertHeader, &insertPayload](size_t nPayloads, DataHeader dh) -> void {
309  // one header with index set to the number of split parts indicates sequence
310  // of payloads without additional headers
311  dh.payloadSize = attributes->msgSize;
312  dh.payloadSerializationMethod = o2::header::gSerializationMethodNone;
313  dh.splitPayloadIndex = nPayloads;
314  dh.splitPayloadParts = nPayloads;
315  insertHeader(dh);
316 
317  for (size_t i = 0; i < nPayloads; ++i) {
318  insertPayload(dh.payloadSize);
319  }
320  };
321 
322  auto createPairs = [&attributes, &insertHeader, &insertPayload](size_t nPayloads, DataHeader dh) -> void {
323  // one header with index set to the number of split parts indicates sequence
324  // of payloads without additional headers
325  dh.payloadSize = attributes->msgSize;
326  dh.payloadSerializationMethod = o2::header::gSerializationMethodNone;
327  dh.splitPayloadIndex = 0;
328  dh.splitPayloadParts = nPayloads;
329  for (size_t i = 0; i < nPayloads; ++i) {
330  dh.splitPayloadIndex = i;
331  insertHeader(dh);
332  insertPayload(dh.payloadSize);
333  }
334  };
335 
336  bool forcedTermination = false;
337  try {
338  if (attributes->mode == ProducerAttributes::Mode::Transport) {
339  for (unsigned int i = 0; i < attributes->nChannels; i++) {
340  createPairs(attributes->splitPayloadSize, DataHeader{"DATA", "TST", i});
341  }
342  // using utility from ExternalFairMQDeviceProxy
343  o2::framework::sendOnChannel(device, messages, attributes->channelName);
344  } else {
345  for (unsigned int i = 0; i < attributes->nChannels; i++) {
346  outputs.make<char>(OutputRef{"data", i}, attributes->msgSize);
347  }
348  }
349  } catch (const std::exception& e) {
350  // we cracefully handle if no shared memory can be allocated, that's simply
351  // a matter of configuration
352  if (std::string(e.what()).find("shmem: could not create a message of size") == std::string::npos) {
353  throw e;
354  }
355  LOG(error) << fmt::format("Exception {}\nallocated {} in cycle {} \nconsider increasing shared memory", e.what(), allocatedSize, attributes->iteration);
356  forcedTermination = true;
357  }
358  ++attributes->iteration;
359  loggerCycle(*pState, nHeaders, totalPayload);
360  auto elapsedTime = std::chrono::duration_cast<std::chrono::seconds>(benchclock::now() - state.startTime);
361  if (forcedTermination || elapsedTime.count() >= state.runningTime) {
362  loggerSummary(*pState);
363  if (forcedTermination) {
364  LOG(error) << "termination was forced by earlier error";
365  }
366  // send the end of stream signal, this is transferred by the proxies
367  // and allows to properly terminate downstream devices
368  control.endOfStream();
369  if (attributes->bypassProxies == ProxyBypass::Output) {
370  // since we are sending on the bare channel, also the EOS message needs to be created.
371  SourceInfoHeader sih;
372  sih.state = InputChannelState::Completed;
373  auto headerMessage = o2::pmr::getMessage(o2::header::Stack{channelAlloc, dph, sih});
374  FairMQParts out;
375  out.AddPart(std::move(headerMessage));
376  // add empty payload message
377  out.AddPart(std::move(device.NewMessageFor(attributes->channelName, 0, 0)));
378  o2::framework::sendOnChannel(device, out, attributes->channelName);
379  }
380  }
381  };
382 
383  return adaptStateless(producerCallback);
384  };
385

387  for (unsigned int i = 0; i < nChannels; i++) {
388  outputs.emplace_back(OutputSpec{{"data"}, "TST", "DATA", i, Lifetime::Timeframe});
389  }
390  workflow.emplace_back(DataProcessorSpec{"producer",

419  //
420  // the compute callback of the checker

423  ActiveGuard g(*cState);

433  };
434  auto checkerBenchInit = [cState, loggerInit]() {
435  loggerInit(*cState);
436  };

439  };

441  callbacks.set(CallbackService::Id::Start, checkerBenchInit);

443  return adaptStateless(checkerCallback);
444  };

447  Inputs checkerInputs;

465  // the input proxy process
466  // reads the messages from the output proxy via the out-of-band channel
467 
468  // converter callback for the external FairMQ device proxy ProcessorSpec generator
469  auto converter = [](FairMQDevice& device, FairMQParts& inputs, ChannelRetriever channelRetriever) {
470  ASSERT_ERROR(inputs.Size() >= 2);
471  if (inputs.Size() < 2) {
472  return;
473  }
474  int msgidx = 0;
475  auto dh = o2::header::get<o2::header::DataHeader*>(inputs.At(msgidx)->GetData());
476  if (!dh) {
477  LOG(error) << "data on input " << msgidx << " does not follow the O2 data model, DataHeader missing";
478  return;
479  }
480  auto dph = o2::header::get<DataProcessingHeader*>(inputs.At(msgidx)->GetData());
481  if (!dph) {
482  LOG(error) << "data on input " << msgidx << " does not follow the O2 data model, DataProcessingHeader missing";
483  return;
484  }
485  // Note: we want to run both the output and input proxy in the same workflow and thus we need
486  // different data identifiers and change the data origin in the forwarding
487  OutputSpec query{"PRX", dh->dataDescription, dh->subSpecification};
488  auto channelName = channelRetriever(query, dph->startTime);
489  bool isData = DataSpecUtils::match(OutputSpec{"TST", "DATA", 0}, dh->dataOrigin, dh->dataDescription, dh->subSpecification);
490  // for the configured data channel we require the channel name, the EOS message containing
491  // the forwarded SourceInfoHeader created by the output proxy will be skipped here since the
492  // input proxy handles this internally
493  ASSERT_ERROR(!isData || !channelName.empty());
494  LOG(debug) << "using channel '" << channelName << "' for " << DataSpecUtils::describe(OutputSpec{dh->dataOrigin, dh->dataDescription, dh->subSpecification});
495  if (channelName.empty()) {
496  return;
497  }
498  // make a copy of the header message, get the data header and change origin
499  auto outHeaderMessage = device.NewMessageFor(channelName, 0, inputs.At(msgidx)->GetSize());
500  memcpy(outHeaderMessage->GetData(), inputs.At(msgidx)->GetData(), inputs.At(msgidx)->GetSize());
501  // this we obviously need to fix in the get API, const'ness of the returned header pointer
502  // should depend on const'ness of the buffer
503  auto odh = const_cast<o2::header::DataHeader*>(o2::header::get<o2::header::DataHeader*>(outHeaderMessage->GetData()));
504  odh->dataOrigin = o2::header::DataOrigin("PRX");
505  FairMQParts output;
506  output.AddPart(std::move(outHeaderMessage));
507  output.AddPart(std::move(inputs.At(msgidx + 1)));
508  LOG(debug) << "sending " << DataSpecUtils::describe(OutputSpec{odh->dataOrigin, odh->dataDescription, odh->subSpecification});
509  o2::framework::sendOnChannel(device, output, channelName);
510  };

516  // skip the name in the configuration string as it is added in specifyExternalFairMQDeviceProxy
517  externalChannelSpec.name = "";
518  externalChannelSpec.type = ChannelType::Pull;
519  externalChannelSpec.method = ChannelMethod::Connect;
520  channelConfig = formatExternalChannelConfiguration(externalChannelSpec);
521  if (!defaultTransportConfig.empty() && defaultTransportConfig.find("transport=") == std::string::npos) {
522  channelConfig += ",transport=" + defaultTransportConfig;
523  }
524 
525  if (bypassProxies == ProxyBypass::None) {
526  // Note: in order to make the DPL output proxy and an input proxy working in the same
527  // workflow, we use different data description
528  Outputs inputProxyOutputs = {OutputSpec{ConcreteDataTypeMatcher{"PRX", "DATA"}, Lifetime::Timeframe}};
529  workflow.emplace_back(specifyExternalFairMQDeviceProxy(
530  "input-proxy",
531  std::move(inputProxyOutputs),
532  channelConfig.c_str(),
533  converter));
534  } else if (bypassProxies == ProxyBypass::Output) {
535  Outputs inputProxyOutputs = {OutputSpec{ConcreteDataTypeMatcher{"TST", "DATA"}, Lifetime::Timeframe}};
536  // we use the same specs as filters in the dpl adaptor
537  auto filterSpecs = inputProxyOutputs;
538  workflow.emplace_back(specifyExternalFairMQDeviceProxy(
539  "input-proxy",
540  std::move(inputProxyOutputs),
541  channelConfig.c_str(),
542  o2::framework::dplModelAdaptor(filterSpecs, true)));
543  }
544 
545  return workflow;
546 }

548 std::istream& operator>>(std::istream& in, enum benchmark_config::ProxyBypass& val)
549 {
550  std::string token;
551  in >> token;
552  if (token == "none") {
553  val = benchmark_config::ProxyBypass::None;
554  } else if (token == "all" || token == "both" || token == "a") {
555  val = benchmark_config::ProxyBypass::All;
556  } else if (token == "output" || token == "out" || token == "o") {
557  val = benchmark_config::ProxyBypass::Output;
558  } else {
559  in.setstate(std::ios_base::failbit);
560  }
561  return in;
562 }
