171 template <class T, int I, typename... Args>
172 int GPUReconstructionHIPBackend::runKernelBackend(krnlSetup& _xyz, const Args&... args)
173 {
174  auto& x = _xyz.x;
175  auto& z = _xyz.z;
176  if (z.evList) {
177  for (int k = 0; k < z.nEvents; k++) {
178  GPUFailedMsg(hipStreamWaitEvent(mInternals->Streams[x.stream], ((hipEvent_t*)z.evList)[k], 0));
179  }
180  }
181  runKernelBackendInternal<T, I>(_xyz, args...);
182  GPUFailedMsg(hipGetLastError());
183  if (z.ev) {
184  GPUFailedMsg(hipEventRecord(*(hipEvent_t*)z.ev, mInternals->Streams[x.stream]));
185  }
186  return 0;
187 }

222 int GPUReconstructionHIPBackend::InitDevice_Runtime()
223 {
224  if (mMaster == nullptr) {
225  hipDeviceProp_t hipDeviceProp;
226  int count, bestDevice = -1;
227  double bestDeviceSpeed = -1, deviceSpeed;
228  if (GPUFailedMsgI(hipGetDeviceCount(&count))) {
229  GPUError("Error getting HIP Device Count");
230  return (1);
231  }
232  if (mProcessingSettings.debugLevel >= 2) {
233  GPUInfo("Available HIP devices:");
234  }
235  std::vector<bool> devicesOK(count, false);
236  for (int i = 0; i < count; i++) {
237  if (mProcessingSettings.debugLevel >= 4) {
238  GPUInfo("Examining device %d", i);
239  }
240  if (mProcessingSettings.debugLevel >= 4) {
241  GPUInfo("Obtained current memory usage for device %d", i);
242  }
243  if (GPUFailedMsgI(hipGetDeviceProperties(&hipDeviceProp, i))) {
244  continue;
245  }
246  if (mProcessingSettings.debugLevel >= 4) {
247  GPUInfo("Obtained device properties for device %d", i);
248  }
249  int deviceOK = true;
250  const char* deviceFailure = "";
251 
252  deviceSpeed = (double)hipDeviceProp.multiProcessorCount * (double)hipDeviceProp.clockRate * (double)hipDeviceProp.warpSize * (double)hipDeviceProp.major * (double)hipDeviceProp.major;
253  if (mProcessingSettings.debugLevel >= 2) {
254  GPUImportant("Device %s%2d: %s (Rev: %d.%d - Mem %lld)%s %s", deviceOK ? " " : "[", i, hipDeviceProp.name, hipDeviceProp.major, hipDeviceProp.minor, (long long int)hipDeviceProp.totalGlobalMem, deviceOK ? " " : " ]", deviceOK ? "" : deviceFailure);
255  }
256  if (!deviceOK) {
257  continue;
258  }
259  devicesOK[i] = true;
260  if (deviceSpeed > bestDeviceSpeed) {
261  bestDevice = i;
262  bestDeviceSpeed = deviceSpeed;
263  } else {
264  if (mProcessingSettings.debugLevel >= 2 && mProcessingSettings.deviceNum < 0) {
265  GPUInfo("Skipping: Speed %f < %f\n", deviceSpeed, bestDeviceSpeed);
266  }
267  }
268  }
269  if (bestDevice == -1) {
270  GPUWarning("No %sHIP Device available, aborting HIP Initialisation (Required mem: %lld)", count ? "appropriate " : "", (long long int)mDeviceMemorySize);
271  return (1);
272  }
273 
274  if (mProcessingSettings.deviceNum > -1) {
275  if (mProcessingSettings.deviceNum >= (signed)count) {
276  GPUWarning("Requested device ID %d does not exist", mProcessingSettings.deviceNum);
277  return (1);
278  } else if (!devicesOK[mProcessingSettings.deviceNum]) {
279  GPUWarning("Unsupported device requested (%d)", mProcessingSettings.deviceNum);
280  return (1);
281  } else {
282  bestDevice = mProcessingSettings.deviceNum;
283  }
284  }
285  mDeviceId = bestDevice;
286 
287  GPUFailedMsgI(hipGetDeviceProperties(&hipDeviceProp, mDeviceId));
288 
289  if (mProcessingSettings.debugLevel >= 2) {
290  GPUInfo("Using HIP Device %s with Properties:", hipDeviceProp.name);
291  GPUInfo("\ttotalGlobalMem = %lld", (unsigned long long int)hipDeviceProp.totalGlobalMem);
292  GPUInfo("\tsharedMemPerBlock = %lld", (unsigned long long int)hipDeviceProp.sharedMemPerBlock);
293  GPUInfo("\tregsPerBlock = %d", hipDeviceProp.regsPerBlock);
294  GPUInfo("\twarpSize = %d", hipDeviceProp.warpSize);
295  GPUInfo("\tmaxThreadsPerBlock = %d", hipDeviceProp.maxThreadsPerBlock);
296  GPUInfo("\tmaxThreadsDim = %d %d %d", hipDeviceProp.maxThreadsDim[0], hipDeviceProp.maxThreadsDim[1], hipDeviceProp.maxThreadsDim[2]);
297  GPUInfo("\tmaxGridSize = %d %d %d", hipDeviceProp.maxGridSize[0], hipDeviceProp.maxGridSize[1], hipDeviceProp.maxGridSize[2]);
298  GPUInfo("\ttotalConstMem = %lld", (unsigned long long int)hipDeviceProp.totalConstMem);
299  GPUInfo("\tmajor = %d", hipDeviceProp.major);
300  GPUInfo("\tminor = %d", hipDeviceProp.minor);
301  GPUInfo("\tclockRate = %d", hipDeviceProp.clockRate);
302  GPUInfo("\tmemoryClockRate = %d", hipDeviceProp.memoryClockRate);
303  GPUInfo("\tmultiProcessorCount = %d", hipDeviceProp.multiProcessorCount);
304  GPUInfo(" ");
305  }
306  if (hipDeviceProp.warpSize != GPUCA_WARP_SIZE) {
307  throw std::runtime_error("Invalid warp size on GPU");
308  }
309  mBlockCount = hipDeviceProp.multiProcessorCount;
310  mMaxThreads = std::max<int>(mMaxThreads, hipDeviceProp.maxThreadsPerBlock * mBlockCount);
311  mWarpSize = 64;
312  mDeviceName = hipDeviceProp.name;
313  mDeviceName += " (HIP GPU)";
314 
315  if (hipDeviceProp.major < 3) {
316  GPUError("Unsupported HIP Device");
317  return (1);
318  }
319 #ifndef GPUCA_NO_CONSTANT_MEMORY
320  if (gGPUConstantMemBufferSize > hipDeviceProp.totalConstMem) {
321  GPUError("Insufficient constant memory available on GPU %d < %d!", (int)hipDeviceProp.totalConstMem, (int)gGPUConstantMemBufferSize);
322  return (1);
323  }
324 #endif
325 
326  if (GPUFailedMsgI(hipSetDevice(mDeviceId))) {
327  GPUError("Could not set HIP Device!");
328  return (1);
329  }
330  if (GPUFailedMsgI(hipSetDeviceFlags(hipDeviceScheduleBlockingSync))) {
331  GPUError("Could not set HIP Device!");
332  return (1);
333  }
334 
335  /*if (GPUFailedMsgI(hipDeviceSetLimit(hipLimitStackSize, GPUCA_GPU_STACK_SIZE)))
336  {
337  GPUError("Error setting HIP stack size");
338  GPUFailedMsgI(hipDeviceReset());
339  return(1);
340  }*/
341 
342  if (mDeviceMemorySize > hipDeviceProp.totalGlobalMem || GPUFailedMsgI(hipMalloc(&mDeviceMemoryBase, mDeviceMemorySize))) {
343  GPUError("HIP Memory Allocation Error (trying %lld bytes, %lld available)", (long long int)mDeviceMemorySize, (long long int)hipDeviceProp.totalGlobalMem);
344  GPUFailedMsgI(hipDeviceReset());
345  return (1);
346  }
347  if (GPUFailedMsgI(hipHostMalloc(&mHostMemoryBase, mHostMemorySize))) {
348  GPUError("Error allocating Page Locked Host Memory (trying %lld bytes)", (long long int)mHostMemorySize);
349  GPUFailedMsgI(hipDeviceReset());
350  return (1);
351  }
352  if (mProcessingSettings.debugLevel >= 1) {
353  GPUInfo("Memory ptrs: GPU (%lld bytes): %p - Host (%lld bytes): %p", (long long int)mDeviceMemorySize, mDeviceMemoryBase, (long long int)mHostMemorySize, mHostMemoryBase);
354  memset(mHostMemoryBase, 0, mHostMemorySize);
355  if (GPUFailedMsgI(hipMemset(mDeviceMemoryBase, 0xDD, mDeviceMemorySize))) {
356  GPUError("Error during HIP memset");
357  GPUFailedMsgI(hipDeviceReset());
358  return (1);
359  }
360  }
361 
362  for (int i = 0; i < mNStreams; i++) {
363  if (GPUFailedMsgI(hipStreamCreate(&mInternals->Streams[i]))) {
364  GPUError("Error creating HIP Stream");
365  GPUFailedMsgI(hipDeviceReset());
366  return (1);
367  }
368  }
369 
370  void* devPtrConstantMem;
371 #ifndef GPUCA_NO_CONSTANT_MEMORY
372  if (GPUFailedMsgI(hipGetSymbolAddress(&devPtrConstantMem, HIP_SYMBOL(gGPUConstantMemBuffer)))) {
373  GPUError("Error getting ptr to constant memory");
374  GPUFailedMsgI(hipDeviceReset());
375  return 1;
376  }
377 #else
378  if (GPUFailedMsgI(hipMalloc(&devPtrConstantMem, gGPUConstantMemBufferSize))) {
379  GPUError("HIP Memory Allocation Error");
380  GPUFailedMsgI(hipDeviceReset());
381  return (1);
382  }
383 #endif
384  mDeviceConstantMem = (GPUConstantMem*)devPtrConstantMem;
385 
386  hipLaunchKernelGGL(HIP_KERNEL_NAME(dummyInitKernel), dim3(mBlockCount), dim3(256), 0, 0, mDeviceMemoryBase);
387  GPUInfo("HIP Initialisation successfull (Device %d: %s (Frequency %d, Cores %d), %lld / %lld bytes host / global memory, Stack frame %d, Constant memory %lld)", mDeviceId, hipDeviceProp.name, hipDeviceProp.clockRate, hipDeviceProp.multiProcessorCount, (long long int)mHostMemorySize,
388  (long long int)mDeviceMemorySize, (int)GPUCA_GPU_STACK_SIZE, (long long int)gGPUConstantMemBufferSize);
389  } else {
390  GPUReconstructionHIPBackend* master = dynamic_cast<GPUReconstructionHIPBackend*>(mMaster);
391  mDeviceId = master->mDeviceId;
392  mBlockCount = master->mBlockCount;
393  mWarpSize = master->mWarpSize;
394  mMaxThreads = master->mMaxThreads;
395  mDeviceName = master->mDeviceName;
396  mDeviceConstantMem = master->mDeviceConstantMem;
397  mInternals = master->mInternals;
398  }
399  for (unsigned int i = 0; i < mEvents.size(); i++) {
400  hipEvent_t* events = (hipEvent_t*)mEvents[i].data();
401  for (unsigned int j = 0; j < mEvents[i].size(); j++) {
402  if (GPUFailedMsgI(hipEventCreateWithFlags(&events[j], hipEventBlockingSync))) {
403  GPUError("Error creating event");
404  GPUFailedMsgI(hipDeviceReset());
405  return 1;
406  }
407  }
408  }
409 
410  return (0);
411 }

450 size_t GPUReconstructionHIPBackend::GPUMemCpy(void* dst, const void* src, size_t size, int stream, int toGPU, deviceEvent* ev, deviceEvent* evList, int nEvents)
451 {
452  if (mProcessingSettings.debugLevel >= 3) {
453  stream = -1;
454  }
455  if (stream == -1) {
456  SynchronizeGPU();
457  GPUFailedMsg(hipMemcpy(dst, src, size, toGPU ? hipMemcpyHostToDevice : hipMemcpyDeviceToHost));
458  } else {
459  if (evList == nullptr) {
460  nEvents = 0;
461  }
462  for (int k = 0; k < nEvents; k++) {
463  GPUFailedMsg(hipStreamWaitEvent(mInternals->Streams[stream], ((hipEvent_t*)evList)[k], 0));
464  }
465  GPUFailedMsg(hipMemcpyAsync(dst, src, size, toGPU == -2 ? hipMemcpyDeviceToDevice : toGPU ? hipMemcpyHostToDevice : hipMemcpyDeviceToHost, mInternals->Streams[stream]));
466  }
467  if (ev) {
468  GPUFailedMsg(hipEventRecord(*(hipEvent_t*)ev, mInternals->Streams[stream == -1 ? 0 : stream]));
469  }
470  return size;
471 }
