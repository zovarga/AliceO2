71 void on_connect(uv_connect_t* connection, int status)
72 {
73  if (status < 0) {
74  LOG(error) << "Unable to connect to driver.";
75  return;
76  }
77  ConnectionContext* context = (ConnectionContext*)connection->data;
78  WSDriverClient* client = context->client;
79  context->state->loopReason |= DeviceState::WS_CONNECTED;
80  auto onHandshake = [client]() {
81  client->flushPending();
82  };
83  std::lock_guard<std::mutex> lock(client->mutex());
84  auto handler = std::make_unique<ClientWebSocketHandler>(*client);
85  client->observe("/ping", [](std::string_view) {
86  LOG(info) << "ping";
87  });
89  client->observe("/shm-offer", [state = context->state](std::string_view cmd) {
90  static constexpr int prefixSize = std::string_view{"/shm-offer "}.size();
91  if (prefixSize > cmd.size()) {
92  LOG(error) << "Malformed shared memory offer";
93  return;
94  }
95  cmd.remove_prefix(prefixSize);
96  size_t offerSize;
97  auto offerSizeError = std::from_chars(cmd.data(), cmd.data() + cmd.size(), offerSize);
98  if (offerSizeError.ec != std::errc()) {
99  LOG(error) << "Malformed shared memory offer";
100  return;
101  }
102  LOGP(info, "Received {}MB shared memory offer", offerSize);
103  ComputingQuotaOffer offer;
104  offer.cpu = 0;
105  offer.memory = 0;
106  offer.sharedMemory = offerSize * 1000000;
107  offer.runtime = 10000;
108  offer.user = -1;
109  offer.valid = true;
110 
111  state->pendingOffers.push_back(offer);
112  });
113 
114  client->observe("/quit", [state = context->state](std::string_view) {
115  state->quitRequested = true;
116  });
117 
118  client->observe("/restart", [state = context->state](std::string_view) {
119  state->nextFairMQState.emplace_back("RUN");
120  state->nextFairMQState.emplace_back("STOP");
121  });
122  auto clientContext = std::make_unique<o2::framework::DriverClientContext>(DriverClientContext{client->spec(), context->state});
123  client->setDPLClient(std::make_unique<WSDPLClient>(connection->handle, std::move(clientContext), onHandshake, std::move(handler)));
124  client->sendHandshake();
125 }

128 void connectToDriver(WSDriverClient* driver, DeviceState* state, char const* address, short port)
129 {
130  uv_tcp_t* socket = (uv_tcp_t*)malloc(sizeof(uv_tcp_t));
131  uv_tcp_init(state->loop, socket);
132  uv_connect_t* connection = (uv_connect_t*)malloc(sizeof(uv_connect_t));
133  ConnectionContext* context = new ConnectionContext;
134  context->client = driver;
135  context->state = state;
136  connection->data = context;
137 
138  struct sockaddr_in dest;
139  uv_ip4_addr(strdup(address), port, &dest);
140 
141  uv_tcp_connect(connection, socket, (const struct sockaddr*)&dest, on_connect);
142 }

150 WSDriverClient::WSDriverClient(ServiceRegistry& registry, DeviceState& state, char const* ip, unsigned short port)
151  : mSpec{registry.get<const DeviceSpec>()}
152 {
153  // Must connect the device to the server and send a websocket request.
154  // On successful connection we can then start to send commands to the driver.
155  // We keep a backlog to make sure we do not lose messages.
156  connectToDriver(this, &state, ip, port);
157  this->mAwakeMainThread = (uv_async_t*)malloc(sizeof(uv_async_t));
158  this->mAwakeMainThread->data = &state;
159  uv_async_init(state.loop, this->mAwakeMainThread, on_awake_main_thread);
160 }

199 void WSDriverClient::flushPending()
200 {
201  std::lock_guard<std::mutex> lock(mClientMutex);
202  static bool printed1 = false;
203  static bool printed2 = false;
204  if (!mClient) {
205  if (mBacklog.size() > 2000) {
206  if (!printed1) {
207  LOG(warning) << "Unable to communicate with driver because client does not exist. Continuing connection attempts.";
208  printed1 = true;
209  }
210  }
211  return;
212  }
213  if (!(mClient->isHandshaken())) {
214  if (mBacklog.size() > 2000) {
215  if (!printed2) {
216  LOG(warning) << "Unable to communicate with driver because client is not connected. Continuing connection attempts.";
217  printed2 = true;
218  }
219  }
220  return;
221  }
222  if (printed1 || printed2) {
223  LOGP(warning, "DriverClient connected successfully. Flushing message backlog of {} messages. All is good.", mBacklog.size());
224  printed1 = false;
225  printed2 = false;
226  }
227  mClient->write(mBacklog);
228  mBacklog.resize(0);
229 }
