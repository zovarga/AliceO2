64 template <typename Hist_IT>
65 HistogramView<Hist_IT> trim(const HistogramView<Hist_IT>& buffer)
66 {
67  auto isZero = [](count_t i) { return i == 0; };
68  auto nonZeroBegin = std::find_if_not(buffer.begin(), buffer.end(), isZero);
69  auto nonZeroEnd = std::find_if_not(buffer.rbegin(), buffer.rend(), isZero).base();
70 
71  if (std::distance(nonZeroBegin, nonZeroEnd) < 0) {
72  return {buffer.end(), buffer.end(), buffer.getOffset()};
73  } else {
74  const ptrdiff_t newOffset = buffer.getMin() + std::distance(buffer.begin(), nonZeroBegin);
75  return {nonZeroBegin, nonZeroEnd, newOffset};
76  }
77 };

91 template <typename HistA_IT, typename HistB_IT>
92 HistogramView<HistA_IT> intersection(const HistogramView<HistA_IT>& histA, const HistogramView<HistB_IT>& histB)
93 {
94  auto mkEmpty = [&histA]() { return HistogramView(histA.end(), histA.end(), 0); };
95 
96  if (histA.empty() || histB.empty()) {
97  // one is empty
98  return mkEmpty();
99  } else if (histA.getMin() > histB.getMax() || histA.getMax() < histB.getMin()) {
100  // disjoint
101  return mkEmpty();
102  } else {
103  // intersecting
104  const ptrdiff_t leftOffset = utils::leftOffset(histA, histB);
105  const ptrdiff_t rightOffset = utils::rightOffset(histA, histB);
106  HistA_IT begin = leftOffset > 0 ? internal::advanceIter(histA.begin(), leftOffset) : histA.begin();
107  HistA_IT end = rightOffset < 0 ? internal::advanceIter(histA.end(), rightOffset) : histA.end();
108  std::ptrdiff_t offset = leftOffset > 0 ? histA.getOffset() + leftOffset : histA.getOffset();
109  return {begin, end, offset};
110  }

113 template <typename HistA_IT, typename HistB_IT>
114 HistogramView<HistA_IT> leftTail(const HistogramView<HistA_IT>& histA, const HistogramView<HistB_IT>& histB)
115 {
116  if (histA.empty() || histB.empty()) {
117  return histA;
118  }
119 
120  const ptrdiff_t leftOffset = utils::leftOffset(histA, histB);
121  if (leftOffset <= 0) {
122  // case 1 no left difference
123  return {histA.end(), histA.end(), 0};
124  } else if (histA.getMin() > histB.getMax()) {
125  // case 2 disjoint
126  return histA;
127  } else {
128  // case 3 0 < leftOffset <= histA.size()
129  return {histA.begin(), internal::advanceIter(histA.begin(), leftOffset), histA.getOffset()};
130  }
131 };

133 template <typename HistA_IT, typename HistB_IT>
134 HistogramView<HistA_IT> rightTail(const HistogramView<HistA_IT>& histA, const HistogramView<HistB_IT>& histB)
135 {
136  if (histA.empty() || histB.empty()) {
137  return histA;
138  }
139 
140  const ptrdiff_t rightOffset = utils::rightOffset(histA, histB);
141 
142  if (rightOffset > 0) {
143  // case 1 no right tail
144  return {histA.end(), histA.end(), 0};
145  } else if (histA.getMax() < histB.getMin()) {
146  // case 2 disjoint
147  return histA;
148  } else {
149  // case 3 0 < -rightOffset <= histA.size()
150  auto newBegin = internal::advanceIter(histA.end(), rightOffset);
151  return {newBegin, histA.end(), *newBegin};
152  }
153 };
