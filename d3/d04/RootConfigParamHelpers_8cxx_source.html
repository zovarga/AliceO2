38 // a generic looper of data members of a TClass; calling a callback
39 // reused in various functions below
40 void loopOverMembers(TClass* cl, void* obj,
41  std::function<void(TDataMember*, int, int)>&& callback)
42 {
43  auto* memberlist = cl->GetListOfDataMembers();
44  for (int i = 0; i < memberlist->GetEntries(); ++i) {
45  auto* dm = (TDataMember*)memberlist->At(i);
46 
47  auto isValidComplex = [dm]() {
48  auto typehash = compile_time_hash(dm->GetTypeName());
49  return isString(*dm) || dm->IsEnum() || dm->IsSTLContainer() ||
50  (typehash == compile_time_hash("o2::framework::Array2D<int>")) ||
51  (typehash == compile_time_hash("o2::framework::Array2D<float>")) ||
52  (typehash == compile_time_hash("o2::framework::Array2D<double>"));
53  };
54 
55  // filter out static members for now
56  if (dm->Property() & kIsStatic) {
57  continue;
58  }
59 
60  if (dm->IsaPointer()) {
61  continue;
62  }
63  if (!dm->IsBasic() && !isValidComplex()) {
64  continue;
65  }
66 
67  const auto dim = dm->GetArrayDim();
68  // we support very simple vectored data in 1D for now
69  if (dim > 1) {
70  continue;
71  }
72 
73  const auto size = (dim == 1) ? dm->GetMaxIndex(dim - 1) : 1; // size of array (1 if scalar)
74  for (int index = 0; index < size; ++index) {
75  callback(dm, index, size);
76  }
77  }
78 }

91 void ptreeToMember(boost::property_tree::ptree const& value,
92  char const* tname,
93  TDataMember* dm,
94  void* ptr)
95 {
96  auto typehash = compile_time_hash(dm->GetTypeName());
97  if (dm->IsSTLContainer()) {
98  switch (typehash) {
99  case compile_time_hash("vector<int>"):
100  *static_cast<std::vector<int>*>(ptr) = vectorFromBranch<int>(value);
101  return;
102  case compile_time_hash("vector<float>"):
103  *static_cast<std::vector<float>*>(ptr) = vectorFromBranch<float>(value);
104  return;
105  case compile_time_hash("vector<double>"):
106  *static_cast<std::vector<double>*>(ptr) = vectorFromBranch<double>(value);
107  return;
108  case compile_time_hash("vector<bool>"):
109  throw std::runtime_error("Bool arrays are not implemented yet");
110  case compile_time_hash("vector<std::string>"):
111  case compile_time_hash("vector<string>"):
112  *static_cast<std::vector<std::string>*>(ptr) = vectorFromBranch<std::string>(value);
113  return;
114  default:
115  throw std::runtime_error("Not an int/float/double/bool vector");
116  }
117  } else {
118  switch (typehash) {
119  case compile_time_hash("o2::framework::Array2D<int>"):
120  *static_cast<Array2D<int>*>(ptr) = array2DFromBranch<int>(value);
121  return;
122  case compile_time_hash("o2::framework::Array2D<float>"):
123  *static_cast<Array2D<float>*>(ptr) = array2DFromBranch<float>(value);
124  return;
125  case compile_time_hash("o2::framework::Array2D<double>"):
126  *static_cast<Array2D<double>*>(ptr) = array2DFromBranch<double>(value);
127  return;
128  }
129  }
130  auto* dt = dm->GetDataType();
131  if (dt != nullptr) {
132  switch (dt->GetType()) {
133  case kChar_t: {
134  *(char*)ptr = value.get_value<char>();

160  }
161  case kULong_t: {

163  return;
164  }
165  case kFloat_t: {
166  *(float*)ptr = value.get_value<float>();
167  return;
168  }
169  case kDouble_t:
170  case kDouble32_t: {
171  *(double*)ptr = value.get_value<double>();
172  return;
173  }

189  }
190  }
191  // if we get here none of the above worked

193  *(std::string*)ptr = value.get_value<std::string>();

195  throw std::runtime_error("Unable to override value");
196 }

288 std::vector<ConfigParamSpec>
289  RootConfigParamHelpers::asConfigParamSpecsImpl(std::string const& mainKey, TClass* cl, void* obj)
290 {
291  std::vector<ConfigParamSpec> specs;
292 
293  auto toDataMember = [&mainKey, &specs, obj](TDataMember* dm, int index, int size) {
294  auto* dt = dm->GetDataType();
295  auto TS = dt != nullptr ? dt->Size() : 0;
296  char* ptr = ((char*)obj) + dm->GetOffset() + index * TS;
297  const std::string name = mainKey + "." + getName(dm, index, size);
298 
299  specs.push_back(memberToConfigParamSpec(name.c_str(), dm, ptr));
300  };
301 
302  loopOverMembers(cl, obj, toDataMember);
303  return specs;
304 }

308 void RootConfigParamHelpers::fillFromPtree(TClass* cl, void* obj, boost::property_tree::ptree const& pt)
309 {
310  auto toDataMember = [obj, &pt](TDataMember* dm, int index, int size) {
311  auto* dt = dm->GetDataType();
312  auto TS = dt != nullptr ? dt->Size() : 0;
313  char* ptr = ((char*)obj) + dm->GetOffset() + index * TS;
314  const std::string name = getName(dm, index, size);
315  auto it = pt.get_child_optional(name);
316  if (!it) {
317  return;
318  }
319  ptreeToMember(*it, dm->GetTrueTypeName(), dm, ptr);
320  };
321 
322  loopOverMembers(cl, obj, toDataMember);
323 }
