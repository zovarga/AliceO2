35 class CalibratorPadGainTracksDevice : public Task
36 {
37  public:
38  void init(framework::InitContext& ic) final
39  {
40  const int slotLength = ic.options().get<int>("tf-per-slot");
41  const int maxDelay = ic.options().get<int>("max-delay");
42  const int minEntries = ic.options().get<int>("min-entries");
43  const bool debug = ic.options().get<bool>("file-dump");
44  const auto lowTrunc = ic.options().get<float>("lowTrunc");
45  const auto upTrunc = ic.options().get<float>("upTrunc");
46 
47  mCalibrator = std::make_unique<CalibratorPadGainTracks>();
48  mCalibrator->setMinEntries(minEntries);
49  mCalibrator->setSlotLength(slotLength);
50  mCalibrator->setMaxSlotsDelay(maxDelay);
51  mCalibrator->setTruncationRange(lowTrunc, upTrunc);
52  mCalibrator->setWriteDebug(debug);
53 
54  // setting up the CCDB
55  mDBapi.init(ic.options().get<std::string>("ccdb-uri")); // or http://localhost:8080 for a local installation
56  mWriteToDB = mDBapi.isHostReachable() ? true : false;
57  }

59  void run(ProcessingContext& pc) final
60  {
61  const auto tfcounter = o2::header::get<DataProcessingHeader*>(pc.inputs().get("gainhistos").header)->startTime;
62  const auto histomaps = pc.inputs().get<CalibPadGainTracksBase::DataTHistos*>("gainhistos");
63  mCalibrator->process(tfcounter, *histomaps.get());
64  const auto& infoVec = mCalibrator->getTFinterval();
65  LOGP(info, "Created {} objects for TF {}", infoVec.size(), tfcounter);
66 
67  if (mCalibrator->hasCalibrationData()) {
68  mRunNumber = processing_helpers::getRunNumber(pc);
69  sendOutput(pc.outputs());
70  }
71  }
72 
73  void endOfStream(EndOfStreamContext& eos) final
74  {
75  LOGP(info, "Finalizing calibration");
76  constexpr calibration::TFType INFINITE_TF = 0xffffffffffffffff;
77  mCalibrator->checkSlotsToFinalize(INFINITE_TF);
78  sendOutput(eos.outputs());
79  }
80 
81  private:
82  void sendOutput(DataAllocator& output)
83  {
84  const auto& calibrations = mCalibrator->getCalibs();
85  for (uint32_t iCalib = 0; iCalib < calibrations.size(); ++iCalib) {
86  const auto& calib = calibrations[iCalib];
87  const auto& infoVec = mCalibrator->getTFinterval();
88  const auto firstTF = infoVec[iCalib].first;
89  const auto lastTF = infoVec[iCalib].second;
90  // store in CCDB
91  if (mWriteToDB) {
92  LOGP(info, "Writing pad-by-pad gain map to CCDB for TF {} to {}", firstTF, lastTF);
93  mMetadata["runNumber"] = std::to_string(mRunNumber);
94  mDBapi.storeAsTFileAny<o2::tpc::CalPad>(&calib, CDBTypeMap.at(CDBType::CalPadGainResidual), mMetadata, firstTF, lastTF + 1);
95  }
96  }
97  mCalibrator->initOutput(); // empty the outputs after they are send
98  }
99 
100  std::unique_ptr<CalibratorPadGainTracks> mCalibrator;
101  bool mWriteToDB{};
102  o2::ccdb::CcdbApi mDBapi;
103  std::map<std::string, std::string> mMetadata;
104  uint64_t mRunNumber{0};
105 };
