56 void websocket_server_callback(uv_stream_t* stream, ssize_t nread, const uv_buf_t* buf)
57 {
58  WSDPLHandler* server = (WSDPLHandler*)stream->data;
59  assert(server);
60  if (nread == 0) {
61  return;
62  }
63  if (nread == UV_EOF) {
64  LOG(debug) << "websocket_server_callback: communication with driver closed";
65  uv_close((uv_handle_t*)stream, websocket_server_close_callback);
66  return;
67  }
68  if (nread < 0) {
69  LOG(error) << "websocket_server_callback: Error while reading from websocket";
70  uv_close((uv_handle_t*)stream, websocket_server_close_callback);
71  return;
72  }
73  try {
74  parse_http_request(buf->base, nread, server);
75  free(buf->base);
76  } catch (RuntimeErrorRef& ref) {
77  auto& err = o2::framework::error_from_ref(ref);
78  LOG(error) << "Error while parsing request: " << err.what;
79  }
80 }

212 void remoteGuiCallback(uv_timer_s* ctx)
213 {
214  GuiRenderer* renderer = reinterpret_cast<GuiRenderer*>(ctx->data);
215 
216  void* frame = nullptr;
217  void* draw_data = nullptr;
218  int size;
219  uint64_t frameStart = uv_hrtime();
220  uint64_t frameLatency = frameStart - renderer->gui->frameLast;
221 
222  // if less than 15ms have passed reuse old frame
223  if (frameLatency / 1000000 > 15) {
224  renderer->gui->plugin->pollGUIPreRender(renderer->gui->window, (float)frameLatency / 1000000000.0f);
225  draw_data = renderer->gui->plugin->pollGUIRender(renderer->gui->callback);
226  } else {
227  draw_data = renderer->gui->lastFrame;
228  }
229 
230  renderer->gui->plugin->getFrameRaw(draw_data, &frame, &size);
231  std::vector<uv_buf_t> outputs;
232  encode_websocket_frames(outputs, (const char*)frame, size, WebSocketOpCode::Binary, 0);
233  renderer->handler->write(outputs);
234  free(frame);
235 
236  if (frameLatency / 1000000 > 15) {
237  uint64_t frameEnd = uv_hrtime();
238  *(renderer->gui->frameCost) = (frameEnd - frameStart) / 1000000;
239  *(renderer->gui->frameLatency) = frameLatency / 1000000;
240  renderer->gui->frameLast = frameStart;
241  renderer->gui->lastFrame = draw_data;
242  }
243 }

250 void WSDPLHandler::endHeaders()
251 {
253  if (mHeaders["upgrade"] != "websocket") {
254  throw WSError{400, "Bad Request: not a websocket upgrade"};
255  }
256 
257  if (mHeaders["connection"].find("upgrade") == std::string::npos) {
258  throw WSError{400, "Bad Request: connection not for upgrade"};
259  }
260  if (mHeaders["sec-websocket-protocol"] != "dpl") {
261  throw WSError{400, "Bad Request: websocket protocol not \"dpl\"."};
262  }
263  if (mHeaders.count("sec-websocket-key") == 0) {
264  throw WSError{400, "Bad Request: sec-websocket-key missing"};
265  }
266  if (mHeaders["sec-websocket-version"] != "13") {
267  throw WSError{400, "Bad Request: wrong protocol version"};
268  }
270  LOG(debug) << "Got upgrade request with nonce " << mHeaders["sec-websocket-key"].c_str();
271  std::string reply = encode_websocket_handshake_reply(mHeaders["sec-websocket-key"].c_str());
272  mHandshaken = true;
273 
274  uv_buf_t bfr = uv_buf_init(strdup(reply.data()), reply.size());
275  uv_write_t* info_req = (uv_write_t*)malloc(sizeof(uv_write_t));
276  uv_write(info_req, (uv_stream_t*)mStream, &bfr, 1, ws_handshake_done_callback);
277  auto header = mHeaders.find("x-dpl-pid");
278  if (header != mHeaders.end()) {
279  LOG(debug) << "Driver connected to PID : " << header->second;
280  mHandler = std::make_unique<ControlWebSocketHandler>(*mServerContext);
281  mHandler->headers(mHeaders);
282 
283  for (size_t i = 0; i < mServerContext->infos->size(); ++i) {
284  if (std::to_string((*mServerContext->infos)[i].pid) == header->second) {
285  (*mServerContext->controls)[i].controller = new DeviceController{this};
286  break;
287  }
288  }
289  } else {
290  LOG(info) << "Connection not bound to a PID";
291  }
292 }

368 {

370  context->state->loopReason |= DeviceState::WS_COMMUNICATION;
371  assert(context->client);
372  if (nread == 0) {
373  return;
374  }
375  if (nread == UV_EOF) {
376  LOG(debug) << "EOF received from server, closing.";
377  uv_read_stop(stream);
378  uv_close((uv_handle_t*)stream, close_client_websocket);
379  return;
380  }
381  if (nread < 0) {
382  // FIXME: improve error message
383  // FIXME: should I close?
384  LOG(error) << "Error while reading from websocket";
385  uv_read_stop(stream);

389  try {
390  LOG(debug) << "Data received from server";

447 void WSDPLClient::endHeaders()
448 {
450  if (mHeaders["upgrade"] != "websocket") {
451  throw runtime_error_f("No websocket upgrade");
452  }
453  // find is used to account for multiple options
454  if (mHeaders["connection"].find("upgrade") == std::string::npos) {
455  throw runtime_error_f("No connection upgrade");
456  }
457  if (mHeaders.count("sec-websocket-accept") == 0) {
458  throw runtime_error("sec-websocket-accept not found");
459  }
460 
461  std::string expectedAccept = HTTPParserHelpers::calculateAccept(mNonce.c_str());
462  if (mHeaders["sec-websocket-accept"] != expectedAccept) {
463  throw runtime_error_f(R"(Invalid accept received: "%s", expected "%s")", mHeaders["sec-websocket-accept"].c_str(), expectedAccept.c_str());
464  }
465 
466  LOG(info) << "Correctly handshaken websocket connection.";
468  mHandshaken = true;
469  mHandshake();
470 }
