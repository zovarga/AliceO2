271 constexpr const char* channelName(int channel)
272 {
273 
274  // sanity check
275  static_assert(NChannels == sizeof(ChannelNames) / sizeof(std::string_view), "Channels definition is not complete");
276 
277  if (channel >= 0 && channel < NChannels) {
278  return ChannelNames[channel].data();
279  } else if (channel == IdDummy) {
280  return DummyName.data();
281  }
282  return VoidName.data();
283 }

285 //< map channelID to detector/tower
286 constexpr int toDet(int channel, int& tower)
287 {
288  if (channel < toChannel(ZNC, 0)) {
289  tower = channel % NChannelsZN;
290  return DetIDOffs + channel / NChannelsZP;
291  } else {
292  channel -= toChannel(ZNC, 0);
293  tower = channel % NChannelsZN;
294  return ZNC + channel / NChannelsZP;
295  }
296 }
