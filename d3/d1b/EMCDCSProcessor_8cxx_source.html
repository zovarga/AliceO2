62 
63  if (mVerbose) {

92 int EMCDCSProcessor::processDP(const DPCOM& dp)
93 {
94 
95  auto& dpid = dp.id;
96  const auto& type = dpid.get_type();
97  auto& val = dp.data;
98 
99  if (mVerbose) {
100  if (type == RAW_DOUBLE) {
101  LOG(info) << "Processing DP = " << dp << ", with value = " << o2::dcs::getValue<double>(dp);
102  } else if (type == RAW_INT) {
103  LOG(info) << "Processing DP = " << dp << ", with value = " << o2::dcs::getValue<int32_t>(dp);
104  } else if (type == RAW_UINT) {
105  LOG(info) << "Processing DP = " << dp << ", with value = " << o2::dcs::getValue<uint32_t>(dp);
106  }
107  }
108 
109  if ((type == RAW_INT) || (type == RAW_UINT)) // FEE config params and STU_TRU error counters
110  {
111  auto& dpval_prev = mapFEEcfg[dpid];
112  if (dpval_prev.size() == 0 || val.get_epoch_time() != dpval_prev.back().get_epoch_time()) //compate the time stamps
113  {
114  dpval_prev.push_back(val); // do we need to archive them all?????
115  mUpdateFEEcfg = true;
116 
117  FillFeeDP(dp);
118  setTF(val.get_epoch_time()); // fix: this must not be here!
119  }
120  } else if (type == RAW_DOUBLE) { // ELMB data
121  FillElmbDP(dp);
122  }
123  //printPDCOM(dp);
124 
125  return 0;
126 }

128 void EMCDCSProcessor::FillElmbDP(const DPCOM& dpcom)
129 {
130  auto& dpid = dpcom.id;
131  const auto type = dpid.get_type();
132  std::string type_str = show(type);
133 
134  std::string alias(dpid.get_alias());
135 
136  auto& dpval = dpcom.data;
137  auto val = o2::dcs::getValue<double>(dpcom); //dpval.payload_pt1;
138 
139  std::size_t index;
140  int iPT = -1;
141 
142  if ((index = alias.find("EMC_PT")) != std::string::npos) {
143  std::sscanf(alias.data(), "EMC_PT_%d.Temperature", &iPT);
144  LOG(debug) << "alias=" << alias.data() << ": iPT=" << iPT << ", val=" << val;
145 
146  if (iPT < 0 || iPT > 159) {
147  LOG(error) << "Wrong Sensor Index iPT=" << iPT << " for DP " << alias.data();
148  return;
149  }
150  mELMB->addMeasurement(iPT, val);
151  } else {
152  LOG(info) << "EMC_PT pattern not found for DPype = RAW_DOUBLE: alias = " << alias.data();
153  }
154  return;

157 void EMCDCSProcessor::FillFeeDP(const DPCOM& dpcom)
158 {
159  auto& dpid = dpcom.id;
160  const auto type = dpid.get_type();
161  std::string type_str = show(type);
162 
163  std::string alias(dpid.get_alias());
164 
165  auto& dpval = dpcom.data;
166  auto ts = dpval.get_epoch_time();
167  auto val = dpval.payload_pt1;
168 
169  std::size_t index;
170  int iTRU = -1;
171  int iMask = -1;
172  int iSM = -1;
173 
174  // if (mVerbose) {
175  // LOG(info) << "EMCDCSProcessor::FillFeeDP called";
176  // }
177  //
178 
179  if ((index = alias.find("STU_ERROR_COUNT_TRU")) != std::string::npos) {
180  // processing of STU_TRU error counters not included yet
181  return;
182  } else if (alias.find("EMC_DDL_LIST0") != std::string::npos) {
183  mFEECFG->setDDLlist0(val);
184  } else if (alias.find("EMC_DDL_LIST1") != std::string::npos) {
185  mFEECFG->setDDLlist1(val);
186  } else if ((index = alias.find("SRU")) != std::string::npos) {
187  if ((index = alias.find("FMVER")) != std::string::npos) {
188  std::sscanf((std::string(alias.substr(index - 3, 2))).data(), "%02d", &iSM);
189  if (iSM < 0 || iSM >= 20) {
190  LOG(error) << "ERROR : iSM = " << iSM << " for" << alias.data();
191  return;
192  }
193  mFEECFG->setSRUFWversion(iSM, val);
194  } else if ((index = alias.find("CFG")) != std::string::npos) {
195  std::sscanf((std::string(alias.substr(index - 3, 2))).data(), "%02d", &iSM);
196  if (iSM < 0 || iSM >= 20) {
197  LOG(error) << "ERROR : iSM = " << iSM << " for" << alias.data();
198  return;
199  }
200  mFEECFG->setSRUconfig(iSM, val);
201  }
202  } else if ((index = alias.find("TRU")) != std::string::npos) {
203  std::sscanf((std::string(alias.substr(index + 3, 2))).data(), "%02d", &iTRU);
204 
205  if (iTRU < 0 || iTRU >= kNTRU) {
206  LOG(error) << "ERROR : iTRU = " << iTRU << " for" << alias.data();
207  return;
208  }
209 
210  mTRU = mFEECFG->getTRUDCS(iTRU);
211  if (alias.find("L0ALGSEL") != std::string::npos) {
212  mTRU.setL0SEL(val);
213  } else if (alias.find("PEAKFINDER") != std::string::npos) {
214  mTRU.setSELPF(val);
215  } else if (alias.find("GLOBALTHRESH") != std::string::npos) {
216  mTRU.setGTHRL0(val);
217  } else if (alias.find("COSMTHRESH") != std::string::npos) {
218  mTRU.setL0COSM(val);
219  } else if ((index = alias.find("MASK")) != std::string::npos) {
220  std::sscanf((std::string(alias.substr(index + 4, 1))).data(), "%02d", &iMask);
221  mTRU.setMaskReg(val, iMask);
222  if (iMask < 0 || iMask > 5) {
223  LOG(error) << "ERROR : iMask = " << iMask << " for" << alias.data();
224  return;
225  }
226  }
227 
228  mFEECFG->setTRUDCS(iTRU, mTRU);
229  } else if ((index = alias.find("_STU_")) != std::string::npos) {
230  bool kEMC = (std::string(alias.substr(index - 3, 3))).compare(0, 3, "DMC", 0, 3);
231  mSTU = kEMC ? mFEECFG->getSTUDCSEMCal() : mFEECFG->getSTUDCSDCal();
232 
233  if (alias.find("MEDIAN") != std::string::npos) {
234  mSTU.setMedianMode(val);
235  } else if (alias.find("GETRAW") != std::string::npos) {
236  mSTU.setRawData(val);
237  } else if (alias.find("REGION") != std::string::npos) {
238  mSTU.setRegion(val);
239  } else if (alias.find("FWVERS") != std::string::npos) {
240  mSTU.setFw(val);
241  } else if (alias.find("PATCHSIZE") != std::string::npos) {
242  mSTU.setPatchSize(val);
243  } else if ((index = alias.find("STU_G")) != std::string::npos) {
244  char par1;
245  int par2;
246  std::sscanf((std::string(alias.substr(index + 5, 2))).data(), "%c%d", &par1, &par2);
247  if (par2 == 0) {
248  mSTU.setGammaHigh((int)par1 - 65, val);
249  } else {
250  mSTU.setGammaLow((int)par1 - 65, val);
251  }
252  } else if ((index = alias.find("STU_J")) != std::string::npos) {
253  char par1;
254  int par2;
255  std::sscanf((std::string(alias.substr(index + 5, 2))).data(), "%c%d", &par1, &par2);
256  if (par2 == 0) {
257  mSTU.setJetHigh((int)par1 - 65, val);
258  } else {
259  mSTU.setJetLow((int)par1 - 65, val);
260  }
261  }
262 
263  if (kEMC) {
264  mFEECFG->setSTUEMCal(mSTU);
265  } else {
266  mFEECFG->setSTUDCal(mSTU);
267  }
268  }
