228 void SimpleEventDisplayGUI::drawPadSignal(int event, int x, int y, TObject* o)
229 {
230  //
231  // type: name of canvas
232  //
233 
234  if (!o) {
235  return;
236  }
237 
238  TString type;
239  if (std::string_view(o->GetName()) == "hMaxValsIROC") {
240  type = "SigI";
241  } else if (std::string_view(o->GetName()) == "hMaxValsOROC") {
242  type = "SigO";
243  } else {
244  return;
245  }
246  // check if an event was alreay loaded
247  if (!mEvDisp.getNumberOfProcessedEvents()) {
248  return;
249  }
250 
251  //return if mouse is pressed to allow looking at one pad
252  if (event != 51) {
253  return;
254  }
255 
256  int binx, biny;
257  float bincx, bincy;
258  TH1* h = getBinInfoXY(binx, biny, bincx, bincy);
259  if (!h) {
260  return;
261  }
262 
263  const int row = int(TMath::Floor(bincx));
264  const int cpad = int(TMath::Floor(bincy));
265  //find pad and channel
266  const int roc = h->GetUniqueID();
267  if (roc < 0 || roc >= (int)ROC::MaxROC) {
268  return;
269  }
270 
271  const auto& mapper = Mapper::instance();
272  if (row < 0 || row >= (int)mapper.getNumberOfRowsROC(roc)) {
273  return;
274  }
275 
276  const int nPads = mapper.getNumberOfPadsInRowROC(roc, row);
277  const int pad = cpad + nPads / 2;
278  if (pad < 0 || pad >= (int)nPads) {
279  return;
280  }
281 
282  //draw requested pad signal
283 
284  TH1*& hFFT = (roc < 36) ? mHFFTI : mHFFTO;
285 
286  TCanvas* c = (TCanvas*)gROOT->GetListOfCanvases()->FindObject(type);
287  TCanvas* cFFT = (TCanvas*)gROOT->GetListOfCanvases()->FindObject(Form("%sFFT", type.Data()));
288  if (c) {
289  c->Clear();
290  c->cd();
291  TH1D* h2 = mEvDisp.makePadSignals(roc, row, pad);
292  if (h2) {
293  h2->Draw();
294  h2->SetStats(0);
295 
296  if (cFFT) {
297  const bool init = (hFFT == nullptr);
298  const double maxTime = h2->GetNbinsX() * 200.e-6;
299  hFFT = h2->FFT(hFFT, "MAG M");
300  if (hFFT) {
301  hFFT->SetStats(0);
302  const auto nbinsx = hFFT->GetNbinsX();
303  auto xax = hFFT->GetXaxis();
304  xax->SetRange(2, nbinsx / 2);
305  if (init) {
306  xax->Set(nbinsx, xax->GetXmin() / maxTime, xax->GetXmax() / maxTime);
307  hFFT->SetNameTitle(Form("hFFT_%sROC", (roc < 36) ? "I" : "O"), "FFT magnitude;frequency (kHz);amplitude");
308  }
309  hFFT->Scale(2. / (nbinsx - 1));
310  cFFT->cd();
311  hFFT->Draw();
312  }
313  }
314  }
315  update(Form("%s;%sFFT", type.Data(), type.Data()));
316  }
317  // printf("bin=%03d.%03d(%03d)[%05d], name=%s, ROC=%02d content=%.1f, ev: %d\n",row,pad,cpad,chn,h->GetName(), roc, h->GetBinContent(binx,biny), event);
318 }

320 //__________________________________________________________________________
321 void SimpleEventDisplayGUI::fillMaxHists(int type)
322 {
323  //
324  // type: 0 fill side and sector, 1: fill sector only
325  //
326  const auto& mapper = Mapper::instance();
327 
328  float kEpsilon = 0.000000000001;
329  CalPad* pad = mEvDisp.getCalPadMax();
330  TH2F* hSide = nullptr;
331  TH2F* hROC = nullptr;
332  resetHists(type);
333  const int runNumber = TString(gSystem->Getenv("RUN_NUMBER")).Atoi();
334  //const int eventNumber = mEvDisp.getNumberOfProcessedEvents() - 1;
335  const int eventNumber = mEvDisp.getPresentEventNumber();
336  const bool eventComplete = mEvDisp.isPresentEventComplete();
337 
338  for (int iROC = 0; iROC < 72; iROC++) {
339  hROC = mHMaxOROC;
340  hSide = mHMaxC;
341  if (iROC < 36) {
342  hROC = mHMaxIROC;
343  }
344  if ((iROC % 36) < 18) {
345  hSide = mHMaxA;
346  }
347  if ((iROC % 36) == (mSelectedSector % 36)) {
348  TString title = Form("Max Values %cROC %c%02d (%02d) TF %s%d%s", (iROC < 36) ? 'I' : 'O', (iROC % 36 < 18) ? 'A' : 'C', iROC % 18, iROC, eventComplete ? "" : "(", eventNumber, eventComplete ? "" : ")");
349  //TString title = Form("Max Values Run %d Event %d", runNumber, eventNumber);
350  if (hROC) {
351  hROC->SetTitle(title.Data());
352  }
353  }
354  auto& calRoc = pad->getCalArray(iROC);
355  const int nRows = mapper.getNumberOfRowsROC(iROC);
356  for (int irow = 0; irow < nRows; irow++) {
357  const int nPads = mapper.getNumberOfPadsInRowROC(iROC, irow);
358  for (int ipad = 0; ipad < nPads; ipad++) {
359  float value = calRoc.getValue(irow, ipad);
360  //printf("iROC: %02d, sel: %02d, row %02d, pad: %02d, value: %.5f\n", iROC, mSelectedSector, irow, ipad, value);
361  if (TMath::Abs(value) > kEpsilon) {
362  if (!type && hSide) {
363  const GlobalPosition2D global2D = mapper.getPadCentre(PadSecPos(Sector(iROC % 36), PadPos(irow + (iROC >= 36) * mapper.getNumberOfRowsROC(0), ipad)));
364  int binx = 1 + TMath::Nint((global2D.X() + 250.) * hSide->GetNbinsX() / 500.);
365  int biny = 1 + TMath::Nint((global2D.Y() + 250.) * hSide->GetNbinsY() / 500.);
366  hSide->SetBinContent(binx, biny, value);
367  }
368  const int nPads = mapper.getNumberOfPadsInRowROC(iROC, irow);
369  const int cpad = ipad - nPads / 2;
370  if ((iROC % 36 == mSelectedSector % 36) && hROC) {
371  //printf(" ->>> Fill: iROC: %02d, sel: %02d, row %02d, pad: %02d, value: %.5f\n", iROC, mSelectedSector, irow, ipad, value);
372  hROC->Fill(irow, cpad, value);
373  }
374  }
375  if ((iROC % 36 == mSelectedSector % 36) && hROC) {
376  hROC->SetUniqueID(iROC);
377  }
378  }
379  }
380  }
381  if (!type) {
382  update("MaxValsA;MaxValsC");
383  }
384 
385  update("MaxValsI;MaxValsO");
386 }

396 //__________________________________________________________________________
397 int SimpleEventDisplayGUI::FindROCFromXY(const float x, const float y, const int side)
398 {
399  //
400  //
401  //
402 
403  const auto& mapper = Mapper::instance();
404  float r = TMath::Sqrt(x * x + y * y);
405  static const float innerWall = mapper.getPadCentre(PadPos(0, 0)).X() - 5.;
406  static const float outerWall = mapper.getPadCentre(PadPos(151, 0)).X() + 5.;
407  static const float outerIROC = mapper.getPadCentre(PadPos(62, 0)).X();
408  static const float innerOROC = mapper.getPadCentre(PadPos(63, 0)).X();
409  static const float betweenROC = (outerIROC + innerOROC) / 2.;
410 
411  //check radial boundary
412  if (r < innerWall || r > outerWall) {
413  return -1;
414  }
415 
416  //check for IROC or OROC
417  int type = 0;
418 
419  if (r > betweenROC) {
420  type = 1;
421  }
422 
423  int alpha = TMath::Nint(TMath::ATan2(y, x) / TMath::Pi() * 180);
424  // printf("%6.3f %6.3f %03d\n",x, y, alpha);
425  if (alpha < 0) {
426  alpha += 360;
427  }
428  const int roc = alpha / 20 + side * 18 + type * 36;
429  return roc;
430 }
