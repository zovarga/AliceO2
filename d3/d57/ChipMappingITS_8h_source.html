71  void expandFEEId(uint16_t feeID, uint16_t& lr, uint16_t& ruOnLr, uint16_t& link) const
72  {
73  lr = feeID >> 12;
74  ruOnLr = feeID & 0x3f;
75  link = (feeID >> 8) & 0x3;
76 #ifdef _OVERRIDE_RUID_HACK_
77  ruOnLr %= NStavesOnLr[lr];
78 #endif

85  uint16_t modifyLinkInFEEId(uint16_t feeID, uint16_t linkID) const

87  feeID &= ~(0x3 << 8);

111  void getChipInfoSW(int chipSW, ChipInfo& chInfo) const
112  {
113  chInfo.id = chipSW;
114  if (chipSW > NChipsSB[IB] + NChipsSB[MB] - 1) { // OB
115  chipSW -= NChipsSB[IB] + NChipsSB[MB];
116  chInfo.ruType = OB;
117  auto dvRU = std::div(chipSW, NChipsPerStaveSB[OB]);
118  chInfo.ru = NStavesSB[IB] + NStavesSB[MB] + dvRU.quot; // RU ID == stave ID
119  chInfo.chOnRU = getChipOnRUInfo(OB, dvRU.rem);
120  } else if (chipSW > NChipsSB[IB] - 1) { // MB
121  chipSW -= NChipsSB[IB];
122  chInfo.ruType = MB;
123  auto dvRU = std::div(chipSW, NChipsPerStaveSB[MB]);
124  chInfo.ru = NStavesSB[IB] + dvRU.quot; // RU ID == stave ID
125  chInfo.chOnRU = getChipOnRUInfo(MB, dvRU.rem);
126  } else { // IB
127  chInfo.ruType = IB;
128  auto dvRU = std::div(chipSW, NChipsPerStaveSB[IB]);
129  chInfo.ru = dvRU.quot; // RU ID == stave ID = module ID
130  chInfo.chOnRU = getChipOnRUInfo(IB, dvRU.rem);
131  }
132  }
