139 template <typename DataT>
140 bool DataContainer3D<DataT>::initFromFile(TFile& inpf, const char* name)
141 {
142  if (inpf.IsZombie()) {
143  LOGP(error, "Failed to read from file: {}", inpf.GetName());
144  return false;
145  }
146  DataContainer3D<DataT>* dataCont{nullptr};
147  dataCont = reinterpret_cast<DataContainer3D<DataT>*>(inpf.GetObjectChecked(name, DataContainer3D<DataT>::Class()));
148 
149  if (!dataCont) {
150  LOGP(error, "Failed to load {} from {}", name, inpf.GetName());
151  return false;
152  }
153 
154  if (mZVertices != dataCont->getNZ() || mRVertices != dataCont->getNR() || mPhiVertices != dataCont->getNPhi()) {
155  LOGP(error, "Data from input file has different definition of vertices!");
156  LOGP(error, "set vertices before creating the sc object to: SpaceCharge<>::setGrid({}, {}, {})", dataCont->getNZ(), dataCont->getNR(), dataCont->getNPhi());
157  delete dataCont;
158  return false;
159  }
160 
161  mData = dataCont->mData;
162  delete dataCont;
163  return true;
164 }
