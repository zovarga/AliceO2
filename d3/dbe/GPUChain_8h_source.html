256 template <bool Always, class T, class S, typename... Args>
257 inline void GPUChain::timeCpy(RecoStep step, int toGPU, S T::*func, Args... args)
258 {
259  if (!Always && step != RecoStep::NoRecoStep && !(GetRecoStepsGPU() & step)) {
260  return;
261  }
262  HighResTimer* timer = nullptr;
263  size_t* bytes = nullptr;
264  if (mRec->mProcessingSettings.debugLevel >= 1 && toGPU >= 0) { // Todo: time special cases toGPU < 0
265  int id = mRec->getRecoStepNum(step, false);
266  if (id != -1) {
267  auto& tmp = mRec->mTimersRecoSteps[id];
268  timer = toGPU ? &tmp.timerToGPU : &tmp.timerToHost;
269  bytes = toGPU ? &tmp.bytesToGPU : &tmp.bytesToHost;
270  (toGPU ? tmp.countToGPU : tmp.countToHost)++;
271  timer->Start();
272  }
273  }
274  size_t n = (mRec->*func)(args...);
275  if (timer) {
276  SynchronizeGPU();
277  timer->Stop();
278  *bytes += n;
279  }
280 }

283 bool GPUChain::DoDebugAndDump(GPUChain::RecoStep step, int mask, bool transfer, T& processor, S T::*func, Args&&... args)
284 {

287  TransferMemoryResourcesToHost(step, &processor, -1, true);
288  }
289  if (GetProcessingSettings().debugLevel >= 6 && (mask == 0 || (GetProcessingSettings().debugMask & mask))) {
290  (processor.*func)(args...);
291  return true;
292  }
293  }
294  return false;

297 template <class T, class S, typename... Args>
298 int GPUChain::runRecoStep(RecoStep step, S T::*func, Args... args)
299 {
300  if (GetRecoSteps().isSet(step)) {
301  if (GetProcessingSettings().debugLevel >= 1) {
302  mRec->getRecoStepTimer(step).Start();
303  }
304  int retVal = (reinterpret_cast<T*>(this)->*func)(args...);
305  if (GetProcessingSettings().debugLevel >= 1) {
306  mRec->getRecoStepTimer(step).Stop();
307  }
308  return retVal;
309  }
310  return false;
311 }
