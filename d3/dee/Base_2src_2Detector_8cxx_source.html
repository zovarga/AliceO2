120 void Detector::SetSpecialPhysicsCuts()
121 {
122  // default implementation for physics cuts setting (might still be overriden by detectors)
123  // we try to read an external text file supposed to be installed
124  // in a standard directory
125  // ${O2_ROOT}/share/Detectors/DETECTORNAME/simulation/data/simcuts.dat
126  LOG(info) << "Setting special cuts for " << GetName();
127  const char* aliceO2env = std::getenv("O2_ROOT");
128  std::string inputFile;
129  if (aliceO2env) {
130  inputFile = std::string(aliceO2env);
131  }
132  inputFile += "/share/Detectors/" + std::string(GetName()) + "/simulation/data/simcuts.dat";
133  auto& matmgr = o2::base::MaterialManager::Instance();
134  matmgr.loadCutsAndProcessesFromFile(GetName(), inputFile.c_str());

138  // via command line

141 void Detector::initFieldTrackingParams(int& integration, float& maxfield)
142 {
143  // set reasonable default values
144  integration = 2;
145  maxfield = 10;
146  auto vmc = TVirtualMC::GetMC();
147  if (vmc) {
148  auto field = vmc->GetMagField();
149  // see if we can query the o2 field
150  if (auto o2field = dynamic_cast<o2::field::MagneticField*>(field)) {
151  integration = o2field->Integral(); // default integration method?
152  maxfield = o2field->Max();
153  return;
154  }
155  }
156  LOG(info) << "No magnetic field found; using default tracking values " << integration << " " << maxfield
157  << " to initialize media\n";
158 }

172 int Detector::registerSensitiveVolumeAndGetVolID(TGeoVolume const* vol)
173 {
174  // register this volume with FairRoot
175  this->FairModule::AddSensitiveVolume(const_cast<TGeoVolume*>(vol));
176  // retrieve the VMC Monte Carlo ID for this volume
177  const int volid = TVirtualMC::GetMC()->VolId(vol->GetName());
178  if (volid <= 0) {
179  LOG(error) << "Could not retrieve VMC volume ID for " << vol->GetName();
180  }
181  return volid;
