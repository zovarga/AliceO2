29 void Digitizer::init()
30 {
31  LOG(info) << "CPVDigitizer::init() : CCDB Url = " << o2::base::NameConf::getCCDBServer();
32  if (o2::base::NameConf::getCCDBServer().compare("localtest") == 0) {
33  mCalibParams = new CalibParams(1); // test default calibration
34  mPedestals = new Pedestals(1); // test default pedestals
35  mBadMap = new BadChannelMap(1); // test default bad channels
36  LOG(info) << "[CPVDigitizer] No reading calibration from ccdb requested, set default";
37  } else {
38  auto& ccdbMgr = o2::ccdb::BasicCCDBManager::instance();
39  ccdbMgr.setCaching(true); //make local cache of remote objects
40  ccdbMgr.setLocalObjectValidityChecking(true); //query objects from remote site only when local one is not valid
41  // read calibration from ccdb (for now do it only at the beginning of dataprocessing)
42  // TODO: setup timestam according to anchors
43  // Do not set timestamp here: This should be set from the framework and is done via the digitizer workflow
44  // ccdbMgr.setTimestamp(o2::ccdb::getCurrentTimestamp());
45 
46  LOG(info) << "CCDB: Reading o2::cpv::CalibParams from CPV/Calib/Gains";
47  mCalibParams = ccdbMgr.get<o2::cpv::CalibParams>("CPV/Calib/Gains");
48  if (!mCalibParams) {
49  LOG(error) << "Cannot get o2::cpv::CalibParams from CCDB. using dummy calibration!";
50  mCalibParams = new CalibParams(1);
51  }
52 
53  LOG(info) << "CCDB: Reading o2::cpv::Pedestals from CPV/Calib/Pedestals";
54  mPedestals = ccdbMgr.get<o2::cpv::Pedestals>("CPV/Calib/Pedestals");
55  if (!mPedestals) {
56  LOG(error) << "Cannot get o2::cpv::Pedestals from CCDB. using dummy calibration!";
57  mPedestals = new Pedestals(1);
58  }
59 
60  LOG(info) << "CCDB: Reading o2::cpv::BadChannelMap from CPV/Calib/BadChannelMap";
61  mBadMap = ccdbMgr.get<o2::cpv::BadChannelMap>("CPV/Calib/BadChannelMap");
62  if (!mBadMap) {
63  LOG(error) << "Cannot get o2::cpv::BadChannelMap from CCDB. using dummy calibration!";
64  mBadMap = new BadChannelMap(1);
65  }
66 
67  LOG(info) << "Task configuration is done.";
68  }
69 
70  //signal thresolds for digits
71  //note that digits are calibrated objects
72  for (int i = 0; i < NCHANNELS; i++) {
73  mDigitThresholds[i] = o2::cpv::CPVSimParams::Instance().mZSnSigmas *
74  mPedestals->getPedSigma(i) * mCalibParams->getGain(i);
75  }
76 }

82 void Digitizer::processHits(const std::vector<Hit>* hits, const std::vector<Digit>& digitsBg,
83  std::vector<Digit>& digitsOut, o2::dataformats::MCTruthContainer<o2::MCCompLabel>& labels,
84  int collId, int source, double dt)
85 {
86  // Convert list of hits to digits:
87  // Add hits with ampl deposition in same pad and same time
88  // Add ampl corrections
89  // Apply time smearing
90  // Despite sorting in Detector::EndEvent(), hits still can be unsorted due to splitting of processing different bunches of primary
91  for (int i = NCHANNELS; i--;) {
92  mArrayD[i].reset();
93  }
94 
95  // First, add pedestal noise and BG digits
96  if (digitsBg.size() == 0) { // no digits provided: try simulate pedestal noise (do it only once)
97  for (int i = NCHANNELS; i--;) {
98  float amplitude = simulatePedestalNoise(i);
99  if (amplitude > mDigitThresholds[i]) { //add noise digit if its signal > threshold
100  mArrayD[i].setAmplitude(simulatePedestalNoise(i));
101  mArrayD[i].setAbsId(i);
102  // mArrayD[i].setLabel(-1); // noise marking (not needed to set as all mArrayD[i] elements are just resetted)
103  }
104  }
105  } else { //if digits exist, noise is already added
106  for (auto& dBg : digitsBg) { //digits are sorted and unique
107  mArrayD[dBg.getAbsId()] = dBg;
108  }
109  }
110 
111  //Second, add Hits
112  for (auto& h : *hits) {
113  int i = h.GetDetectorID();
114  if (mArrayD[i].getAmplitude() > 0) {
115  mArrayD[i].setAmplitude(mArrayD[i].getAmplitude() + h.GetEnergyLoss()); //if amplitude > 0 then pedestal noise is already added
116  } else {
117  mArrayD[i].setAbsId(i);
118  mArrayD[i].setAmplitude(h.GetEnergyLoss() + simulatePedestalNoise(i)); //if not then add pedestal noise to signal
119  }
120  if (mArrayD[i].getAmplitude() > mDigitThresholds[i]) {
121  int labelIndex = mArrayD[i].getLabel();
122  if (labelIndex == -1) { // noise
123  labelIndex = labels.getIndexedSize();
124  o2::MCCompLabel label(true); // noise label
125  labels.addElement(labelIndex, label);
126  mArrayD[i].setLabel(labelIndex);
127  } else { //check if lable already exist
128  gsl::span<MCCompLabel> sp = labels.getLabels(labelIndex);
129  bool found = false;
130  for (MCCompLabel& te : sp) {
131  if (te.getTrackID() == h.GetTrackID() && te.getEventID() == collId && te.getSourceID() == source) {
132  found = true;
133  break;
134  }
135  }
136  if (!found) {
137  o2::MCCompLabel label(h.GetTrackID(), collId, source);
138  //Highly inefficient management of Labels: commenting line below reeduces WHOLE digitization time by factor ~30
139  labels.addElementRandomAccess(labelIndex, label);
140  }
141  }
142  }
143  }
144 
145  //finalize output digits
146  for (int i = 0; i < NCHANNELS; i++) {
147  if (!mBadMap->isChannelGood(i)) {
148  continue; //bad channel -> skip this digit
149  }
150  if (mArrayD[i].getAmplitude() > mDigitThresholds[i]) {
151  digitsOut.push_back(mArrayD[i]);
152  }
153  }
154 }

156 float Digitizer::simulatePedestalNoise(int absId)
157 {
158  //this function is to simulate pedestal and its noise (ADC counts)
159  if (absId < 0 || absId >= NCHANNELS) {
160  return 0.;
161  }
162  return gRandom->Gaus(0, mPedestals->getPedSigma(absId) * mCalibParams->getGain(absId));
163 }
