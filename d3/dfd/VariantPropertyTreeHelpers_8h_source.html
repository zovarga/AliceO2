58 {

103 template <typename T>
104 auto basicArray2DFromBranch(boost::property_tree::ptree const& branch)
105 {
106  std::vector<T> cache;
107  uint32_t nrows = branch.size();
108  uint32_t ncols = 0;
109  bool first = true;
110  auto irow = 0u;
111  for (auto const& row : branch) {
112  if (first) {
113  ncols = row.second.size();
114  first = false;
115  }
116  auto icol = 0u;
117  for (auto const& entry : row.second) {
118  cache.push_back(entry.second.get_value<T>());
119  ++icol;
120  }
121  ++irow;
122  }
123  return Array2D<T>{cache, nrows, ncols};
124 }
