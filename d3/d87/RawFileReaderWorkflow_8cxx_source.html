109 void RawReaderSpecs::processDropTF(const std::string& dropTF)
110 {
111  static const std::regex delimDet(";");
112  if (dropTF.empty() || dropTF == "none") {
113  return;
114  }
115  std::sregex_token_iterator iter(dropTF.begin(), dropTF.end(), delimDet, -1), end;
116  for (; iter != end; ++iter) {
117  std::string sdet = iter->str();
118  if (sdet.length() < 5 || sdet[3] != ',') {
119  throw std::runtime_error(fmt::format("Wrong dropTF argument {} in {}", sdet, dropTF));
120  }
121  std::string detName = sdet.substr(0, 3);
122  o2::detectors::DetID det(detName.c_str()); // make sure this is a valid detector
123  std::string sdetArg = sdet.substr(4, sdet.length());
124  int modV = 0, rej = 0, posrej = sdetArg.find(',');
125  if (posrej != std::string::npos) {
126  modV = std::stoi(sdetArg.substr(0, posrej));
127  rej = std::stoi(sdetArg.substr(++posrej, sdetArg.length()));
128  } else {
129  modV = std::stoi(sdetArg);
130  }
131  if (modV < 1 || rej < 0 || rej >= modV) {
132  throw std::runtime_error(fmt::format("Wrong dropTF argument {}, 1st number must be > than 2nd", sdet));
133  }
134  mDropTFMap[detName] = {modV, rej};
135  LOG(info) << " Will drop TF for detector " << detName << " if (TF_ID%" << modV << ")==" << rej;
136  }
137 }

157 void RawReaderSpecs::run(o2f::ProcessingContext& ctx)
158 {
159  assert(mReader);
160  auto tTotStart = mTimer[TimerTotal].CpuTime(), tIOStart = mTimer[TimerIO].CpuTime();
161  mTimer[TimerTotal].Start(false);
162  auto device = ctx.services().get<o2f::RawDeviceService>().device();
163  assert(device);
164 
165  auto findOutputChannel = [&ctx, this](o2h::DataHeader& h) {
166  if (!this->mRawChannelName.empty()) {
167  return std::string{this->mRawChannelName};
168  } else {
169  auto outputRoutes = ctx.services().get<o2f::RawDeviceService>().spec().outputs;
170  for (auto& oroute : outputRoutes) {
171  LOG(debug) << "comparing with matcher to route " << oroute.matcher << " TSlice:" << oroute.timeslice;
172  if (o2f::DataSpecUtils::match(oroute.matcher, h.dataOrigin, h.dataDescription, h.subSpecification) && ((h.tfCounter % oroute.maxTimeslices) == oroute.timeslice)) {
173  LOG(debug) << "picking the route:" << o2f::DataSpecUtils::describe(oroute.matcher) << " channel " << oroute.channel;
174  return std::string{oroute.channel};
175  }
176  }
177  }
178  LOGP(error, "Failed to find output channel for {}/{}/{} @ timeslice {}", h.dataOrigin.str, h.dataDescription.str, h.subSpecification, h.tfCounter);
179  return std::string{};
180  };
181 
182  size_t tfNParts = 0, tfSize = 0;
183  std::unordered_map<std::string, std::unique_ptr<FairMQParts>> messagesPerRoute;
184 
185  auto addPart = [&messagesPerRoute, &tfNParts, &tfSize](FairMQMessagePtr hd, FairMQMessagePtr pl, const std::string& fairMQChannel) {
186  FairMQParts* parts = nullptr;
187  parts = messagesPerRoute[fairMQChannel].get(); // FairMQParts*
188  if (!parts) {
189  messagesPerRoute[fairMQChannel] = std::make_unique<FairMQParts>();
190  parts = messagesPerRoute[fairMQChannel].get();
191  }
192  tfSize += pl->GetSize();
193  tfNParts++;
194  parts->AddPart(std::move(hd));
195  parts->AddPart(std::move(pl));
196  };
197 
198  // clean-up before reading next TF
199  auto tfID = mReader->getNextTFToRead();
200  int nlinks = mReader->getNLinks();
201 
202  if (tfID > mMaxTFID) {
203  if (!mReader->isEmpty() && --mLoop) {
204  mLoopsDone++;
205  tfID = 0;
206  LOG(info) << "Starting new loop " << mLoopsDone << " from the beginning of data";
207  } else {
208  mTimer[TimerTotal].Stop();
209  LOGF(info, "Finished: payload of %zu bytes in %zu messages sent for %d TFs", mSentSize, mSentMessages, mTFCounter);
210  for (int i = 0; i < NTimers; i++) {
211  LOGF(info, "Timing for %15s: Cpu: %.3e Real: %.3e s in %d slots", TimerName[i], mTimer[i].CpuTime(), mTimer[i].RealTime(), mTimer[i].Counter() - 1);
212  }
213  if (!mRawChannelName.empty()) { // send endOfStream message to raw channel
214  o2f::SourceInfoHeader exitHdr;
215  exitHdr.state = o2::framework::InputChannelState::Completed;
216  const auto exitStack = o2::header::Stack(o2h::DataHeader(o2h::gDataDescriptionInfo, o2h::gDataOriginAny, 0, 0), o2f::DataProcessingHeader(), exitHdr);
217  auto fmqFactory = device->GetChannel(mRawChannelName, 0).Transport();
218  auto hdEOSMessage = fmqFactory->CreateMessage(exitStack.size(), fair::mq::Alignment{64});
219  auto plEOSMessage = fmqFactory->CreateMessage(0, fair::mq::Alignment{64});
220  memcpy(hdEOSMessage->GetData(), exitStack.data(), exitStack.size());
221  FairMQParts eosMsg;
222  eosMsg.AddPart(std::move(hdEOSMessage));
223  eosMsg.AddPart(std::move(plEOSMessage));
224  device->Send(eosMsg, mRawChannelName);
225  LOG(info) << "Sent EoS message to " << mRawChannelName;
226  } else {
227  ctx.services().get<o2f::ControlService>().endOfStream();
228  }
229  ctx.services().get<o2f::ControlService>().readyToQuit(o2f::QuitRequest::Me);
230  return;
231  }
232  }
233 
234  if (tfID < mMinTFID) {
235  tfID = mMinTFID;
236  }
237  mReader->setNextTFToRead(tfID);
238  std::vector<RawFileReader::PartStat> partsSP;
239 
240  // read next time frame
241  LOG(info) << "Reading TF#" << mTFCounter << " (" << tfID << " at iteration " << mLoopsDone << ')';
242  o2::header::Stack dummyStack{o2h::DataHeader{}, o2::framework::DataProcessingHeader{0}}; // dummy stack to just to get stack size
243  auto hstackSize = dummyStack.size();
244 
245  uint32_t firstOrbit = 0;
246  uint64_t creationTime = 0;
247  const auto& hbfU = HBFUtils::Instance();
248 
249  for (int il = 0; il < nlinks; il++) {
250  auto& link = mReader->getLink(il);
251 
252  if (!mDropTFMap.empty()) { // some TFs should be dropped
253  auto res = mDropTFMap.find(link.origin.str);
254  if (res != mDropTFMap.end() && (mTFCounter % res->second.first) == res->second.second) {
255  LOG(info) << "Dropping " << mTFCounter << " for " << link.origin.str << "/" << link.description.str << "/" << link.subspec;
256  continue; // drop the data
257  }
258  }
259  if (!link.rewindToTF(tfID)) {
260  continue; // this link has no data for wanted TF
261  }
262 
263  o2h::DataHeader hdrTmpl(link.description, link.origin, link.subspec); // template with 0 size
264  int nParts = mPartPerSP ? link.getNextTFSuperPagesStat(partsSP) : link.getNHBFinTF();
265  hdrTmpl.payloadSerializationMethod = o2h::gSerializationMethodNone;
266  hdrTmpl.splitPayloadParts = nParts;
267  hdrTmpl.tfCounter = mTFCounter;
268  if (mVerbosity > 1) {
269  LOG(info) << link.describe() << " will read " << nParts << " HBFs starting from block " << link.nextBlock2Read;
270  }
271  const auto fmqChannel = findOutputChannel(hdrTmpl);
272  if (fmqChannel.empty()) { // no output channel
273  continue;
274  }
275 
276  auto fmqFactory = device->GetChannel(fmqChannel, 0).Transport();
277  while (hdrTmpl.splitPayloadIndex < hdrTmpl.splitPayloadParts) {
278  hdrTmpl.payloadSize = mPartPerSP ? partsSP[hdrTmpl.splitPayloadIndex].size : link.getNextHBFSize();
279  auto hdMessage = fmqFactory->CreateMessage(hstackSize, fair::mq::Alignment{64});
280  auto plMessage = fmqFactory->CreateMessage(hdrTmpl.payloadSize, fair::mq::Alignment{64});
281  mTimer[TimerIO].Start(false);
282  auto bread = mPartPerSP ? link.readNextSuperPage(reinterpret_cast<char*>(plMessage->GetData()), &partsSP[hdrTmpl.splitPayloadIndex]) : link.readNextHBF(reinterpret_cast<char*>(plMessage->GetData()));
283  if (bread != hdrTmpl.payloadSize) {
284  LOG(error) << "Link " << il << " read " << bread << " bytes instead of " << hdrTmpl.payloadSize
285  << " expected in TF=" << mTFCounter << " part=" << hdrTmpl.splitPayloadIndex;
286  }
287  mTimer[TimerIO].Stop();
288  // check if the RDH to send corresponds to expected orbit
289  if (hdrTmpl.splitPayloadIndex == 0) {
290  auto ir = o2::raw::RDHUtils::getHeartBeatIR(plMessage->GetData());
291  auto tfid = hbfU.getTF(ir);
292  firstOrbit = hdrTmpl.firstTForbit = (mPreferCalcTF || !link.cruDetector) ? hbfU.getIRTF(tfid).orbit : ir.orbit; // will be picked for the following parts
293  creationTime = hbfU.getTFTimeStamp({0, firstOrbit});
294  }
295  o2::header::Stack headerStack{hdrTmpl, o2::framework::DataProcessingHeader{mTFCounter, 1, creationTime}};
296  memcpy(hdMessage->GetData(), headerStack.data(), headerStack.size());
297  hdrTmpl.splitPayloadIndex++; // prepare for next
298 
299  addPart(std::move(hdMessage), std::move(plMessage), fmqChannel);
300  }
301  LOGF(debug, "Added %d parts for TF#%d(%d in iteration %d) of %s/%s/0x%u", hdrTmpl.splitPayloadParts, mTFCounter, tfID,
302  mLoopsDone, link.origin.as<std::string>(), link.description.as<std::string>(), link.subspec);
303  }
304 
305  // send sTF acknowledge message
306  {
307  o2::header::STFHeader stfHeader{mTFCounter, firstOrbit, 0};
308  o2::header::DataHeader stfDistDataHeader(o2::header::gDataDescriptionDISTSTF, o2::header::gDataOriginFLP, 0, sizeof(o2::header::STFHeader), 0, 1);
309  stfDistDataHeader.payloadSerializationMethod = o2h::gSerializationMethodNone;
310  stfDistDataHeader.firstTForbit = stfHeader.firstOrbit;
311  stfDistDataHeader.tfCounter = mTFCounter;
312  const auto fmqChannel = findOutputChannel(stfDistDataHeader);
313  if (!fmqChannel.empty()) { // no output channel
314  auto fmqFactory = device->GetChannel(fmqChannel, 0).Transport();
315  o2::header::Stack headerStackSTF{stfDistDataHeader, o2::framework::DataProcessingHeader{mTFCounter, 1, creationTime}};
316  auto hdMessageSTF = fmqFactory->CreateMessage(hstackSize, fair::mq::Alignment{64});
317  auto plMessageSTF = fmqFactory->CreateMessage(stfDistDataHeader.payloadSize, fair::mq::Alignment{64});
318  memcpy(hdMessageSTF->GetData(), headerStackSTF.data(), headerStackSTF.size());
319  memcpy(plMessageSTF->GetData(), &stfHeader, sizeof(o2::header::STFHeader));
320  addPart(std::move(hdMessageSTF), std::move(plMessageSTF), fmqChannel);
321  }
322  }
323 
324  if (mTFCounter) { // delay sending
325  usleep(mDelayUSec);
326  }
327  for (auto& msgIt : messagesPerRoute) {
328  LOG(info) << "Sending " << msgIt.second->Size() / 2 << " parts to channel " << msgIt.first;
329  device->Send(*msgIt.second.get(), msgIt.first);
330  }
331  mTimer[TimerTotal].Stop();
332 
333  LOGP(info, "Sent payload of {} bytes in {} parts in {} messages for TF#{} firstTForbit={} timeStamp={} | Timing (total/IO): {} / {}", tfSize, tfNParts,
334  messagesPerRoute.size(), mTFCounter, firstOrbit, creationTime, mTimer[TimerTotal].CpuTime() - tTotStart, mTimer[TimerIO].CpuTime() - tIOStart);
335 
336  mSentSize += tfSize;
337  mSentMessages += tfNParts;
338 
339  mReader->setNextTFToRead(++tfID);
340  ++mTFCounter;
341 }

344 o2f::DataProcessorSpec getReaderSpec(ReaderInp rinp)
345 {
346  // check which inputs are present in files to read
347  o2f::DataProcessorSpec spec;
348  spec.name = "raw-file-reader";
349  std::string rawChannelName = "";
350  if (rinp.rawChannelConfig.empty()) {
351  if (!rinp.inifile.empty()) {
352  auto conf = o2::raw::RawFileReader::parseInput(rinp.inifile);
353  for (const auto& entry : conf) {
354  const auto& ordescard = entry.first;
355  if (!entry.second.empty()) { // origin and decription for files to process
356  spec.outputs.emplace_back(o2f::OutputSpec(o2f::ConcreteDataTypeMatcher{std::get<0>(ordescard), std::get<1>(ordescard)}));
357  }
358  }
359  }
360  // add output for DISTSUBTIMEFRAME
361  spec.outputs.emplace_back(o2f::OutputSpec{{"stfDist"}, o2::header::gDataOriginFLP, o2::header::gDataDescriptionDISTSTF, 0});
362  } else {
363  auto nameStart = rinp.rawChannelConfig.find("name=");
364  if (nameStart == std::string::npos) {
365  throw std::runtime_error("raw channel name is not provided");
366  }
367  nameStart += strlen("name=");
368  auto nameEnd = rinp.rawChannelConfig.find(",", nameStart + 1);
369  if (nameEnd == std::string::npos) {
370  nameEnd = rinp.rawChannelConfig.size();
371  }
372  spec.options = {o2f::ConfigParamSpec{"channel-config", o2f::VariantType::String, rinp.rawChannelConfig, {"Out-of-band channel config"}}};
373  rinp.rawChannelConfig = rinp.rawChannelConfig.substr(nameStart, nameEnd - nameStart);
374  LOG(info) << "Will send output to non-DPL channel " << rinp.rawChannelConfig;
375  }
376 
377  spec.algorithm = o2f::adaptFromTask<RawReaderSpecs>(rinp);
378 
379  return spec;
380 }
