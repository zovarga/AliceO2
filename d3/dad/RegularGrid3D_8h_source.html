165 template <typename DataT>
166 DataT RegularGrid3D<DataT>::clampToGrid(const DataT pos, const unsigned int dim) const
167 {
168  if (mMin[dim] < mMax[dim]) {
169  if (pos < mMin[dim]) {
170  return mMin[dim];
171  } else if (pos > mMax[dim]) {
172  return mMax[dim];
173  }
174  } else {
175  if (pos > mMin[dim]) {
176  return mMin[dim];
177  } else if (pos < mMax[dim]) {
178  return mMax[dim];
179  }
180  }
181  return pos;
182 }

184 template <typename DataT>
185 DataT RegularGrid3D<DataT>::clampToGridRel(const DataT pos, const unsigned int dim) const
186 {
187  if (pos < 0) {
188  return 0;
189  } else if (pos >= sMaxIndex[dim]) {
190  return sMaxIndex[dim] - 1; // -1 return second last index. otherwise two additional points have to be extrapolated for tricubic interpolation
191  }
192  return pos;
193 }

195 template <typename DataT>
196 DataT RegularGrid3D<DataT>::clampToGridCircular(DataT pos, const unsigned int dim) const
197 {
198  while (pos < mMin[dim]) {
199  pos += mMax[dim] - mMin[dim] + mSpacing[dim];
200  }
201  while (pos >= mMax[dim] + mSpacing[dim]) {
202  pos -= mMax[dim] + mSpacing[dim] - mMin[dim];
203  }
204  return pos;
205 }

207 template <typename DataT>
208 DataT RegularGrid3D<DataT>::clampToGridCircularRel(DataT pos, const unsigned int dim) const
209 {
210  while (pos < 0) {
211  pos += sNdim[dim];
212  }
213  while (pos > sNdim[dim]) {
214  pos -= sNdim[dim];
215  }
216  if (pos == sNdim[dim]) {
217  pos = 0;
218  }
219  return pos;
220 }
