25 double KrClusterFinder::LandauChi2Functor::operator()(const double* par) const
26 {
27  // provides chi2 estimate comparing y[] to par[0] * TMath::Landau(x[], par[1], par[2])
28  // par[0] : amplitude
29  // par[1] : location parameter (approximately most probable value)
30  // par[2] : sigma
31  double retVal = 0;
32  for (unsigned int i = xLowerBound; i <= xUpperBound; ++i) {
33  if (fabs(y[i]) < 1e-3f) {
34  // exclude bins with zero errors like in TH1::Fit
35  // the standard bin error is the square root of its content
36  continue;
37  }
38  retVal += TMath::Power(y[i] - par[0] * TMath::Landau(x[i], par[1], par[2]), 2) / y[i];
39  }
40  return retVal;
41 }

43 double KrClusterFinder::getRms(const std::vector<uint64_t>& adcIndices, int itTrunc, double nRmsTrunc, int minAdc, double& rmsTime, uint32_t& sumAdc) const
44 {
45  double rmsAdc = 1e7;
46  double meanAdc = -10.;
47  rmsTime = 1e7;
48  double meanTime = -10.f;
49 
50  for (int it = 0; it < itTrunc; ++it) {
51  // iterations for truncated mean
52  sumAdc = 0;
53  uint32_t sumAdc2 = 0;
54  uint32_t sumTime = 0;
55  uint32_t sumTime2 = 0;
56  uint32_t sumWeights = 0;
57  for (const auto& adcIdx : adcIndices) {
58  int iTb = adcIdx % TIMEBINS;
59  int iDigit = adcIdx / TIMEBINS;
60  int adc = mDigits[iDigit].getADC()[iTb] - mBaselineAdc;
61  if (adc < minAdc) {
62  continue;
63  }
64  if (fabs(adc - meanAdc) < nRmsTrunc * rmsAdc) {
65  sumAdc += adc;
66  sumAdc2 += adc * adc;
67  sumWeights += 1;
68  sumTime += iTb;
69  sumTime2 += iTb * iTb;
70  }
71  }
72  if (sumWeights > 0) {
73  auto sumWeightsInv = 1. / sumWeights;
74  meanAdc = sumAdc * sumWeightsInv;
75  rmsAdc = TMath::Sqrt(sumWeightsInv * (sumAdc2 - 2. * meanAdc * sumAdc + sumWeights * meanAdc * meanAdc));
76  meanTime = sumTime * sumWeightsInv;
77  rmsTime = TMath::Sqrt(sumWeightsInv * (sumTime2 - 2. * meanTime * sumTime + sumWeights * meanTime * meanTime));
78  }
79  }
80  return rmsAdc;
81 }

105 void KrClusterFinder::findClusters()
106 {
107  if (mDigits.size() == 0 || mTriggerRecords.size() == 0) {
108  return;
109  }
110  int nClsTotal = 0;
111  int nClsDropped = 0;
112  int nClsInvalidFit = 0;
113  /*
114  The input digits and trigger records are provided on a per time frame basis.
115  The cluster finding is performed for each trigger. In order to not copy the
116  input data we sort the digits for each trigger by their detector ID and keep
117  the sorted indices in the digitIdxArray.
118  */
119  std::vector<uint64_t> digitIdxArray(mDigits.size());
120  std::iota(digitIdxArray.begin(), digitIdxArray.end(), 0);
121  for (const auto& trig : mTriggerRecords) {
122  // sort the digits of given trigger record by detector ID
123  const auto& digits = mDigits; // this reference is only needed to be able to pass it to the lambda for sorting
124  std::stable_sort(std::begin(digitIdxArray) + trig.getFirstDigit(), std::begin(digitIdxArray) + trig.getNumberOfDigits() + trig.getFirstDigit(),
125  [&digits](uint64_t i, uint64_t j) { return digits[i].getDetector() < digits[j].getDetector(); });
126 
127  /*
128  Count number of digits per detector:
129  For each detector we need to know the number of digits.
130  We fill the array idxFirstDigitInDet with the total number of digits
131  up to given detector. So we start at 0 for detector 0 and end up with
132  the total number of digits in entry 540. Thus the number of digits in
133  detector iDet can be calculated with:
134  idxFirstDigitInDet[iDet+1] - idxFirstDigitInDet[iDet]
135  Note, that for a global index one needs to add trig.getFirstDigit()
136  */
137  int currDet = 0;
138  int nextDet = 0;
139  int digitCounter = 0;
140  std::array<unsigned int, MAXCHAMBER + 1> idxFirstDigitInDet{};
141  auto idxHelperPtr = &(idxFirstDigitInDet.data()[1]);
142  idxHelperPtr[-1] = 0;
143  for (int iDigit = trig.getFirstDigit(); iDigit < trig.getFirstDigit() + trig.getNumberOfDigits(); ++iDigit) {
144  if (mDigits[digitIdxArray[iDigit]].getDetector() > currDet) {
145  nextDet = mDigits[digitIdxArray[iDigit]].getDetector();
146  for (int iDet = currDet; iDet < nextDet; ++iDet) {
147  idxHelperPtr[iDet] = digitCounter;
148  }
149  currDet = nextDet;
150  }
151  ++digitCounter;
152  }
153  for (int iDet = currDet; iDet <= MAXCHAMBER; ++iDet) {
154  idxHelperPtr[iDet] = digitCounter;
155  }
156 
157  // the cluster search is done on a per detector basis
158  for (int iDet = 0; iDet < MAXCHAMBER; ++iDet) {
159  unsigned int nDigitsInDet = idxFirstDigitInDet[iDet + 1] - idxFirstDigitInDet[iDet];
160  std::vector<bool> isAdcUsed(nDigitsInDet * TIMEBINS); // keep track of the ADC values which have already been checked or added to a cluster
161  std::vector<bool> isDigitUsed(nDigitsInDet); // keep track of the digits which have already been processed
162  bool continueClusterSearch = true;
163  // the following loop searches for one cluster at a time in iDet
164  while (continueClusterSearch) {
165 
166  // start by finding the max ADC value in all digits of iDet
167  uint16_t adcMax = 0;
168  int tbMax = 0;
169  int rowMax = 0;
170  int colMax = 0;
171  unsigned int iDigitMax = 0;
172  for (int iDigit = 0; iDigit < nDigitsInDet; ++iDigit) {
173  uint64_t digitIdx = digitIdxArray[trig.getFirstDigit() + idxFirstDigitInDet[iDet] + iDigit]; // global index for array of all digits (mDigits)
174  if (mDigits[digitIdx].isSharedDigit()) {
175  // we need to skip the shared digits which are duplicates contained in the global digits array
176  continue;
177  }
178  if (isDigitUsed[iDigit]) {
179  // if a maximum has been found for this digit then all ADCs above threshold are already flagged as used
180  continue;
181  }
182  int tbMaxADC = -1;
183  auto maxAdcInDigit = mDigits[digitIdx].getADCmax(tbMaxADC);
184  if (maxAdcInDigit > adcMax) {
185  // potentially found a maximum
186  tbMax = tbMaxADC;
187  rowMax = mDigits[digitIdx].getPadRow();
188  colMax = mDigits[digitIdx].getPadCol();
189  iDigitMax = iDigit;
190  adcMax = maxAdcInDigit;
191  }
192  }
193  if (adcMax < mMinAdcForMax) {
194  // the maximum ADC value is below the threshold, go to next chamber
195  break;
196  }
197 
198  // cluster around max ADC value
199  int lowerTb = TIMEBINS;
200  int upperTb = 0;
201  int lowerCol = NCOLUMN;
202  int upperCol = 0;
203  int lowerRow = NROWC1;
204  int upperRow = 0;
205  int nUsedADCsInCl = 0;
206  std::vector<uint64_t> constituentAdcIndices;
207  for (unsigned int iDigit = 0; iDigit < nDigitsInDet; ++iDigit) {
208  uint64_t digitIdx = digitIdxArray[trig.getFirstDigit() + idxFirstDigitInDet[iDet] + iDigit]; // global index for array of all digits (mDigits)
209  if (mDigits[digitIdx].isSharedDigit()) {
210  // we need to skip the shared digits which are duplicates contained in the global digits array
211  continue;
212  }
213  int row = mDigits[digitIdx].getPadRow();
214  if (std::abs(row - rowMax) > 1) {
215  continue;
216  }
217  int col = mDigits[digitIdx].getPadCol();
218  if (std::abs(col - colMax) > 2) {
219  continue;
220  }
221  bool addedAdc = false;
222  for (int iTb = 0; iTb < TIMEBINS; ++iTb) {
223  if (isAdcUsed[iDigit * TIMEBINS + iTb]) {
224  continue;
225  }
226  // flag this ADC value as used, regardless of whether or not it is added to the cluster
227  // (if it is below the threshold it won't be used for another cluster anyway)
228  isAdcUsed[iDigit * TIMEBINS + iTb] = true;
229  if (mDigits[digitIdx].getADC()[iTb] > mMinAdcClContrib) {
230  addedAdc = true;
231  if (iTb < lowerTb) {
232  lowerTb = iTb;
233  }
234  if (iTb > upperTb) {
235  upperTb = iTb;
236  }
237  ++nUsedADCsInCl;
238  }
239  constituentAdcIndices.push_back(digitIdx * TIMEBINS + iTb); // also add ADC values below threshold here
240  }
241  if (addedAdc) {
242  isDigitUsed[iDigit] = true;
243  if (row < lowerRow) {
244  lowerRow = row;
245  }
246  if (row > upperRow) {
247  upperRow = row;
248  }
249  if (col < lowerCol) {
250  lowerCol = col;
251  }
252  if (col > upperCol) {
253  upperCol = col;
254  }
255  }
256  }
257 
258  // determine cluster size
259  int clSizeTime = upperTb - lowerTb;
260  int clSizeRow = upperRow - lowerRow;
261  int clSizeCol = upperCol - lowerCol;
262 
263  if (nUsedADCsInCl > 0) {
264  // after this cluster is processed, continue looking for another one in iDet
265  continueClusterSearch = true;
266  }
267 
268  // sum up ADC values (total sum, sum per time bin, total sum for ADCs above energy threshold)
269  std::array<int, TIMEBINS> sumPerTb{};
270  int sumOfAllTimeBins = 0;
271  int sumOfAllTimeBinsAboveThreshold = 0;
272  for (const auto idx : constituentAdcIndices) {
273  auto iTb = idx % TIMEBINS;
274  auto iDigit = idx / TIMEBINS;
275  sumOfAllTimeBins += mDigits[iDigit].getADC()[iTb] - mBaselineAdc;
276  sumPerTb[iTb] += mDigits[iDigit].getADC()[iTb] - mBaselineAdc;
277  if (mDigits[iDigit].getADC()[iTb] > mMinAdcClEoverT) {
278  sumOfAllTimeBinsAboveThreshold += mDigits[iDigit].getADC()[iTb] - mBaselineAdc;
279  }
280  }
281 
282  uint32_t sumOfAdcTrunc;
283  double rmsTimeTrunc;
284  auto rmsAdcClusterTrunc = getRms(constituentAdcIndices, 2, 3., static_cast<uint32_t>(mMinAdcClEoverT * .95), rmsTimeTrunc, sumOfAdcTrunc);
285 
286  // ADC value and time bin of first maximum
287  int maxAdcA = -1;
288  int maxTbA = -1;
289  mLandauChi2Functor.x.clear();
290  mLandauChi2Functor.y.clear();
291  for (int iTb = 0; iTb < TIMEBINS; ++iTb) {
292  mLandauChi2Functor.x.push_back((float)iTb);
293  mLandauChi2Functor.y.push_back(sumPerTb[iTb]);
294  if (sumPerTb[iTb] < mMinAdcForMax) {
295  continue;
296  }
297  if (sumPerTb[iTb] > maxAdcA) {
298  maxAdcA = sumPerTb[iTb];
299  maxTbA = iTb;
300  }
301  }
302  mLandauChi2Functor.xLowerBound = maxAdcA - 1;
303  mLandauChi2Functor.xUpperBound = maxAdcA + 2;
304 
305  // ADC value and time bin of second maximum (if there is one)
306  int maxAdcB = -1;
307  int maxTbB = -1;
308  for (int iTb = 2; iTb < TIMEBINS - 2; ++iTb) { // we need to check the neighbouring two bins, so ignore the outermost two bins in the search for a second maximum
309  if (std::abs(maxTbA - iTb) < 3 || sumPerTb[iTb] < mMinAdcForSecondMax) {
310  // we are too close to the first maximum (or the ADC value is not too small)
311  continue;
312  }
313  if (sumPerTb[iTb] > maxAdcB) {
314  // check neighbours
315  if (sumPerTb[iTb - 1] < sumPerTb[iTb] &&
316  sumPerTb[iTb - 2] < sumPerTb[iTb] &&
317  sumPerTb[iTb + 1] < sumPerTb[iTb] &&
318  sumPerTb[iTb + 2] < sumPerTb[iTb]) {
319  maxAdcB = sumPerTb[iTb];
320  maxTbB = iTb;
321  }
322  }
323  }
324 
325  // quality checks for maxima
326  bool isBadA = false;
327  bool isBadB = false;
328  if (maxAdcA < 0 || maxTbA <= 1 || maxTbA >= (TIMEBINS - 2)) {
329  isBadA = true;
330  }
331  if (maxAdcB < 0) {
332  // second maximum must be inside the time acceptance by construction
333  isBadB = true;
334  }
335  if (!isBadA) {
336  // we have a good first maximum, let's check its shape
337  if ((sumPerTb[maxTbA - 1] / static_cast<double>(maxAdcA)) < 0.1 || (sumPerTb[maxTbA + 1] / static_cast<double>(maxAdcA)) < 0.25) {
338  isBadA = true;
339  }
340  }
341  if (!isBadB) {
342  // we have a good second maximum, let's check its shape
343  if ((sumPerTb[maxTbB - 1] / static_cast<double>(maxAdcB)) < 0.1 || (sumPerTb[maxTbB + 1] / static_cast<double>(maxAdcB)) < 0.25) {
344  isBadB = true;
345  }
346  }
347  if (!isBadA && !isBadB) {
348  // we have two maxima, check order and size
349  if (maxTbA > maxTbB || maxAdcA <= maxAdcB) {
350  isBadA = true;
351  isBadB = true;
352  }
353  }
354 
355  if (clSizeCol > 0 && clSizeTime > 3 && clSizeTime < TIMEBINS) {
356  mFitter.Config().ParSettings(0).SetValue(maxAdcA);
357  mFitter.Config().ParSettings(1).SetValue(maxTbA);
358  mFitter.Config().ParSettings(2).SetValue(.5);
359  bool fitOK = mFitter.FitFCN();
360  if (!fitOK) {
361  ++nClsInvalidFit;
362  }
363  mFuncLandauFit->SetParameters(mFitResult->GetParams()[0], mFitResult->GetParams()[1], mFitResult->GetParams()[2]);
364  double integralLandauFit = fitOK ? mFuncLandauFit->Integral(0., static_cast<double>(TIMEBINS), 1.e-9) : 0.;
365 
366  double rmsTime;
367  uint32_t rmsSumAdc;
368  auto rmsAdc = getRms(constituentAdcIndices, 1, 2.5, mMinAdcClContrib, rmsTime, rmsSumAdc);
369 
370  double sumAdcA = 0.;
371  double sumAdcB = 0.;
372  if (!isBadA && !isBadB) {
373  // use the Landau fit to the first peak to extrapolate the energy of larger time bins (close time bins are evaluated from the array directly)
374  // for the second peak the Landau fit is used to subtract the energy from the first peak
375  for (int iTb = 0; iTb < TIMEBINS; ++iTb) {
376  if (iTb < (maxTbB - 2)) {
377  sumAdcA += mLandauChi2Functor.y[iTb];
378  } else {
379  sumAdcA += mFuncLandauFit->Eval(mLandauChi2Functor.x[iTb]);
380  sumAdcB += mLandauChi2Functor.y[iTb] - mFuncLandauFit->Eval(mLandauChi2Functor.x[iTb]);
381  }
382  }
383  }
384  // create Kr cluster
385  if (sumOfAllTimeBins <= std::numeric_limits<uint16_t>::max() &&
386  sumOfAllTimeBinsAboveThreshold <= std::numeric_limits<uint16_t>::max() &&
387  integralLandauFit <= std::numeric_limits<uint16_t>::max() &&
388  sumAdcA <= std::numeric_limits<uint16_t>::max() &&
389  sumAdcB <= std::numeric_limits<uint16_t>::max() &&
390  rmsAdc <= std::numeric_limits<uint16_t>::max() &&
391  rmsTime <= std::numeric_limits<uint8_t>::max() &&
392  nUsedADCsInCl <= std::numeric_limits<uint8_t>::max() &&
393  sumOfAdcTrunc <= std::numeric_limits<uint16_t>::max()) {
394  if (maxTbA < 0) {
395  maxTbA = tbMax;
396  }
397  if (maxTbB < 0) {
398  maxTbB = 2 * TIMEBINS + 5; // larger than any maximum time bin we can have
399  }
400  KrCluster cluster;
401  cluster.setGlobalPadID(iDet, rowMax, colMax);
402  cluster.setAdcData(sumOfAllTimeBins, (int)rmsAdc, (int)sumAdcA, (int)sumAdcB, sumOfAllTimeBinsAboveThreshold, (int)integralLandauFit, sumOfAdcTrunc);
403  cluster.setTimeData(maxTbA, maxTbB, (int)rmsTime);
404  cluster.setClusterSizeData(clSizeRow, clSizeCol, clSizeTime, nUsedADCsInCl);
405  mKrClusters.push_back(cluster);
406  ++nClsTotal;
407  } else {
408  //mFitResult->Print(std::cout);
409  ++nClsDropped;
410  LOG(debug) << "Kr cluster cannot be added because values are out of range";
411  LOGF(debug, "sumOfAllTimeBins(%i), sumAdcA(%f), sumAdcB(%f), clSizeRow(%i), clSizeCol(%i), clSizeTime(%i), maxTbA(%i), maxTbB(%i)", sumOfAllTimeBins, sumAdcA, sumAdcB, clSizeRow, clSizeCol, clSizeTime, maxTbA, maxTbB);
412  LOGF(debug, "rmsAdc(%f), rmsTime(%f), nUsedADCsInCl(%i), sumOfAllTimeBinsAboveThreshold(%i), integralLandauFit(%f), sumOfAdcTrunc(%u)", rmsAdc, rmsTime, nUsedADCsInCl, sumOfAllTimeBinsAboveThreshold, integralLandauFit, sumOfAdcTrunc);
413  }
414  }
415  } // end cluster search
416  } // end detector loop
417  } // end trigger loop
418 
419  // we don't need the exact BC time, just use first interaction record within this TF
420  mTrigRecs.emplace_back(mTriggerRecords[0].getBCData(), nClsTotal);
421  LOGF(info, "Number of Kr clusters with a) invalid fit (%i) b) out-of-range values which were dropped (%i)", nClsInvalidFit, nClsDropped);
