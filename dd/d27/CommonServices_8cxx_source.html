78 o2::framework::ServiceSpec CommonServices::monitoringSpec()
79 {
80  return ServiceSpec{
81  .name = "monitoring",
82  .init = [](ServiceRegistry& registry, DeviceState&, fair::mq::ProgOptions& options) -> ServiceHandle {
83  void* service = nullptr;
84  bool isWebsocket = strncmp(options.GetPropertyAsString("driver-client-backend").c_str(), "ws://", 4) == 0;
85  bool isDefault = options.GetPropertyAsString("monitoring-backend") == "default";
86  bool useDPL = (isWebsocket && isDefault) || options.GetPropertyAsString("monitoring-backend") == "dpl://";
87  o2::monitoring::Monitoring* monitoring;
88  if (useDPL) {
89  monitoring = new Monitoring();
90  auto dplBackend = std::make_unique<DPLMonitoringBackend>(registry);
91  (dynamic_cast<o2::monitoring::Backend*>(dplBackend.get()))->setVerbosity(o2::monitoring::Verbosity::Debug);
92  monitoring->addBackend(std::move(dplBackend));
93  } else {
94  auto backend = isDefault ? "infologger://" : options.GetPropertyAsString("monitoring-backend");
95  monitoring = MonitoringFactory::Get(backend).release();
96  }
97  service = monitoring;
98  monitoring->enableBuffering(MONITORING_QUEUE_SIZE);
99  assert(registry.get<DeviceSpec const>().name.empty() == false);
100  monitoring->addGlobalTag("dataprocessor_id", registry.get<DeviceSpec const>().id);
101  monitoring->addGlobalTag("dataprocessor_name", registry.get<DeviceSpec const>().name);
102  return ServiceHandle{TypeIdHelpers::uniqueId<Monitoring>(), service};
103  },
104  .configure = noConfiguration(),
105  .start = [](ServiceRegistry& services, void* service) {
106  o2::monitoring::Monitoring* monitoring = (o2::monitoring::Monitoring *) service;
107  auto& context = services.get<DataTakingContext>();
108 
109  try {
110  monitoring->setRunNumber(std::stoul(context.runNumber.c_str()));
111  } catch (...) {
112  } },
113  .exit = [](ServiceRegistry& registry, void* service) {
114  Monitoring* monitoring = reinterpret_cast<Monitoring*>(service);
115  delete monitoring; },
116  .kind = ServiceKind::Serial};
117 }

131 o2::framework::ServiceSpec CommonServices::datatakingContextSpec()
132 {
133  return ServiceSpec{
134  .name = "datataking-contex",
135  .init = simpleServiceInit<DataTakingContext, DataTakingContext>(),
136  .configure = noConfiguration(),
137  .preProcessing = [](ProcessingContext& processingContext, void* service) {
138  auto& context = processingContext.services().get<DataTakingContext>();
139  // Only on the first message
140  if (context.source == OrbitResetTimeSource::Data) {
141  return;
142  }
143  // Only if we do not have already the proper number from CTP
144  if (context.source == OrbitResetTimeSource::CTP) {
145  return;
146  }
147  context.source = OrbitResetTimeSource::Data;
148  context.orbitResetTime = -1;
149  for (auto const& ref : processingContext.inputs()) {
150  const o2::framework::DataProcessingHeader *dph = o2::header::get<DataProcessingHeader*>(ref.header);
151  const auto* dh = o2::header::get<o2::header::DataHeader*>(ref.header);
152  if (!dph || !dh) {
153  continue;
154  }
155  LOGP(debug, "Orbit reset time from data: {} ", dph->creation);
156  context.orbitResetTime = dph->creation;
157  context.runNumber = fmt::format("{}", dh->runNumber);
158  break;
159  } },
160  .start = [](ServiceRegistry& services, void* service) {
161  auto& context = services.get<DataTakingContext>();
162  auto extRunNumber = services.get<RawDeviceService>().device()->fConfig->GetProperty<std::string>("runNumber", "unspecified");
163  if (extRunNumber != "unspecified" || context.runNumber == "0") {
164  context.runNumber = extRunNumber;
165  }
166  // FIXME: we actually need to get the orbit, not only to know where it is
167  std::string orbitResetTimeUrl = services.get<RawDeviceService>().device()->fConfig->GetProperty<std::string>("orbit-reset-time", "ccdb://CTP/Calib/OrbitResetTime");
168  auto is_number = [](const std::string& s) -> bool {
169  return !s.empty() && std::all_of(s.begin(), s.end(), ::isdigit);
170  };
171 
172  if (orbitResetTimeUrl.rfind("file://") == 0) {
173  // FIXME: read it from a file
174  context.orbitResetTime = 490917600;
175  } else if (orbitResetTimeUrl.rfind("http://") == 0) {
176  // FIXME: read it from ccdb
177  context.orbitResetTime = 490917600;
178  } else if (is_number(orbitResetTimeUrl)) {
179  context.orbitResetTime = std::stoull(orbitResetTimeUrl.data());
180  // FIXME: specify it from the command line
181  } else {
182  context.orbitResetTime = 490917600;
183  }
184  context.nOrbitsPerTF = services.get<RawDeviceService>().device()->fConfig->GetProperty<uint64_t>("Norbits_per_TF", 128); },
185  .kind = ServiceKind::Serial};
186 }

204 // Creates the sink for FairLogger / InfoLogger integration
205 auto createInfoLoggerSinkHelper(InfoLogger* logger, InfoLoggerContext* ctx)
206 {
207  return [logger,
208  ctx](const std::string& content, const fair::LogMetaData& metadata) {
209  // translate FMQ metadata
210  InfoLogger::InfoLogger::Severity severity = InfoLogger::Severity::Undefined;
211  int level = InfoLogger::undefinedMessageOption.level;
212 
213  if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::nolog)) {
214  // discard
215  return;
216  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::fatal)) {
217  severity = InfoLogger::Severity::Fatal;
218  level = 1;
219  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::error)) {
220  severity = InfoLogger::Severity::Error;
221  level = 3;
222  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::alarm)) {
223  severity = InfoLogger::Severity::Warning;
224  level = 4;
225  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::important)) {
226  severity = InfoLogger::Severity::Info;
227  level = 5;
228  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::warn)) {
229  severity = InfoLogger::Severity::Warning;
230  level = 6;
231  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::state)) {
232  severity = InfoLogger::Severity::Info;
233  level = 8;
234  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::info)) {
235  severity = InfoLogger::Severity::Info;
236  level = 10;
237  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug)) {
238  severity = InfoLogger::Severity::Debug;
239  level = 11;
240  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug1)) {
241  severity = InfoLogger::Severity::Debug;
242  level = 12;
243  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug2)) {
244  severity = InfoLogger::Severity::Debug;
245  level = 13;
246  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug3)) {
247  severity = InfoLogger::Severity::Debug;
248  level = 14;
249  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::debug4)) {
250  severity = InfoLogger::Severity::Debug;
251  level = 15;
252  } else if (metadata.severity_name == fair::Logger::SeverityName(fair::Severity::trace)) {
253  severity = InfoLogger::Severity::Debug;
254  level = 50;
255  }
256 
257  InfoLogger::InfoLoggerMessageOption opt = {
258  severity,
259  level,
260  InfoLogger::undefinedMessageOption.errorCode,
261  metadata.file.c_str(),
262  atoi(metadata.line.c_str())};
263 
264  if (logger) {
265  logger->log(opt, *ctx, "%s", content.c_str());
266  }
267  };
268 };

273 o2::framework::ServiceSpec CommonServices::infologgerSpec()
274 {
275  return ServiceSpec{
276  .name = "infologger",
277  .init = [](ServiceRegistry& services, DeviceState&, fair::mq::ProgOptions& options) -> ServiceHandle {
278  auto infoLoggerMode = options.GetPropertyAsString("infologger-mode");
279  if (infoLoggerMode != "") {
280  setenv("O2_INFOLOGGER_MODE", infoLoggerMode.c_str(), 1);
281  }
282  char const* infoLoggerEnv = getenv("O2_INFOLOGGER_MODE");
283  if (infoLoggerEnv == nullptr || strcmp(infoLoggerEnv, "none") == 0) {
284  return ServiceHandle{TypeIdHelpers::uniqueId<MissingService>(), nullptr};
285  }
286  auto infoLoggerService = new InfoLogger;
287  auto infoLoggerContext = &services.get<InfoLoggerContext>();
288  infoLoggerContext->setField(InfoLoggerContext::FieldName::Facility, std::string("dpl/") + services.get<DeviceSpec const>().name);
289  infoLoggerContext->setField(InfoLoggerContext::FieldName::System, std::string("DPL"));
290  infoLoggerService->setContext(*infoLoggerContext);
291 
292  auto infoLoggerSeverity = options.GetPropertyAsString("infologger-severity");
293  if (infoLoggerSeverity != "") {
294  fair::Logger::AddCustomSink("infologger", infoLoggerSeverity, createInfoLoggerSinkHelper(infoLoggerService, infoLoggerContext));
295  }
296  return ServiceHandle{TypeIdHelpers::uniqueId<InfoLogger>(), infoLoggerService};
297  },
298  .configure = noConfiguration(),
299  .kind = ServiceKind::Serial};

478 auto sendRelayerMetrics(ServiceRegistry& registry, DataProcessingStats& stats) -> void
479 {
480  auto timeSinceLastUpdate = stats.beginIterationTimestamp - stats.lastSlowMetricSentTimestamp;
481  auto timeSinceLastLongUpdate = stats.beginIterationTimestamp - stats.lastVerySlowMetricSentTimestamp;
482  if (timeSinceLastUpdate < 5000) {
483  return;
484  }
485  // Derive the amount of shared memory used
486  auto& runningWorkflow = registry.get<RunningWorkflowInfo const>();
487  using namespace fair::mq::shmem;
488  auto& spec = registry.get<DeviceSpec const>();
489 
490  // FIXME: Ugly, but we do it only every 5 seconds...
491  if (spec.name == "readout-proxy") {
492  auto device = registry.get<RawDeviceService>().device();
493  try {
494  stats.availableManagedShm.store(Monitor::GetFreeMemory(SessionId{device->fConfig->GetProperty<std::string>("session")}, runningWorkflow.shmSegmentId));
495  } catch (...) {
496  }
497  }
498 
499  auto performedComputationsSinceLastUpdate = stats.performedComputations - stats.lastReportedPerformedComputations;
500 
501  ZoneScopedN("send metrics");
502  auto& relayerStats = registry.get<DataRelayer>().getStats();
503  auto& monitoring = registry.get<Monitoring>();
504 
505  O2_SIGNPOST_START(MonitoringStatus::ID, MonitoringStatus::SEND, 0, 0, O2_SIGNPOST_BLUE);
506 
507  monitoring.send(Metric{(int)relayerStats.malformedInputs, "malformed_inputs"}.addTag(Key::Subsystem, Value::DPL));
508  monitoring.send(Metric{(int)relayerStats.droppedComputations, "dropped_computations"}.addTag(Key::Subsystem, Value::DPL));
509  monitoring.send(Metric{(int)relayerStats.droppedIncomingMessages, "dropped_incoming_messages"}.addTag(Key::Subsystem, Value::DPL));
510  monitoring.send(Metric{(int)relayerStats.relayedMessages, "relayed_messages"}.addTag(Key::Subsystem, Value::DPL));
511 
512  monitoring.send(Metric{(int)stats.errorCount, "errors"}.addTag(Key::Subsystem, Value::DPL));
513  monitoring.send(Metric{(int)stats.exceptionCount, "exceptions"}.addTag(Key::Subsystem, Value::DPL));
514  monitoring.send(Metric{(int)stats.pendingInputs, "inputs/relayed/pending"}.addTag(Key::Subsystem, Value::DPL));
515  monitoring.send(Metric{(int)stats.incomplete, "inputs/relayed/incomplete"}.addTag(Key::Subsystem, Value::DPL));
516  monitoring.send(Metric{(int)stats.inputParts, "inputs/relayed/total"}.addTag(Key::Subsystem, Value::DPL));
517  monitoring.send(Metric{stats.lastElapsedTimeMs, "elapsed_time_ms"}.addTag(Key::Subsystem, Value::DPL));
518  monitoring.send(Metric{stats.lastProcessedSize, "last_processed_input_size_byte"}
519  .addTag(Key::Subsystem, Value::DPL));
520  monitoring.send(Metric{stats.totalProcessedSize, "total_processed_input_size_byte"}
521  .addTag(Key::Subsystem, Value::DPL));
522  monitoring.send(Metric{stats.totalSigusr1.load(), "total_sigusr1"}.addTag(Key::Subsystem, Value::DPL));
523  monitoring.send(Metric{(stats.lastProcessedSize.load() / (stats.lastElapsedTimeMs.load() ? stats.lastElapsedTimeMs.load() : 1) / 1000),
524  "processing_rate_mb_s"}
525  .addTag(Key::Subsystem, Value::DPL));
526  monitoring.send(Metric{stats.lastLatency.minLatency, "min_input_latency_ms"}
527  .addTag(Key::Subsystem, Value::DPL));
528  monitoring.send(Metric{stats.lastLatency.maxLatency, "max_input_latency_ms"}
529  .addTag(Key::Subsystem, Value::DPL));
530  monitoring.send(Metric{(stats.lastProcessedSize / (stats.lastLatency.maxLatency ? stats.lastLatency.maxLatency : 1) / 1000), "input_rate_mb_s"}
531  .addTag(Key::Subsystem, Value::DPL));
532  monitoring.send(Metric{((float)performedComputationsSinceLastUpdate / (float)timeSinceLastUpdate) * 1000, "processing_rate_hz"}.addTag(Key::Subsystem, Value::DPL));
533  monitoring.send(Metric{(uint64_t)stats.performedComputations, "performed_computations"}.addTag(Key::Subsystem, Value::DPL));
534 
535  if (stats.availableManagedShm) {
536  monitoring.send(Metric{(uint64_t)stats.availableManagedShm, fmt::format("available_managed_shm_{}", runningWorkflow.shmSegmentId)}.addTag(Key::Subsystem, Value::DPL));
537  }
538 
539  if (stats.consumedTimeframes) {
540  monitoring.send(Metric{(uint64_t)stats.consumedTimeframes, "consumed-timeframes"}.addTag(Key::Subsystem, Value::DPL));
541  }
542 
543  stats.lastSlowMetricSentTimestamp.store(stats.beginIterationTimestamp.load());
544  stats.lastReportedPerformedComputations.store(stats.performedComputations.load());
545  O2_SIGNPOST_END(MonitoringStatus::ID, MonitoringStatus::SEND, 0, 0, O2_SIGNPOST_BLUE);
546 
547  auto device = registry.get<RawDeviceService>().device();
548 
549  uint64_t lastTotalBytesIn = stats.totalBytesIn.exchange(0);
550  uint64_t lastTotalBytesOut = stats.totalBytesOut.exchange(0);
551  uint64_t totalBytesIn = 0;
552  uint64_t totalBytesOut = 0;
553 
554  for (auto& channel : device->fChannels) {
555  totalBytesIn += channel.second[0].GetBytesRx();
556  totalBytesOut += channel.second[0].GetBytesTx();
557  }
558 
559  monitoring.send(Metric{(float)(totalBytesOut - lastTotalBytesOut) / 1000000.f / (timeSinceLastUpdate / 1000.f), "total_rate_out_mb_s"}
560  .addTag(Key::Subsystem, Value::DPL));
561  monitoring.send(Metric{(float)(totalBytesIn - lastTotalBytesIn) / 1000000.f / (timeSinceLastUpdate / 1000.f), "total_rate_in_mb_s"}
562  .addTag(Key::Subsystem, Value::DPL));
563  stats.totalBytesIn.store(totalBytesIn);
564  stats.totalBytesOut.store(totalBytesOut);
565  // Things which we report every 30s
566  if (timeSinceLastLongUpdate < 30000) {
567  return;
568  }
569  stats.lastVerySlowMetricSentTimestamp.store(stats.beginIterationTimestamp.load());
570 };

573 auto flushMetrics(ServiceRegistry& registry, DataProcessingStats& stats) -> void
574 {
575  auto timeSinceLastUpdate = stats.beginIterationTimestamp - stats.lastMetricFlushedTimestamp;
576  static int counter = 0;
577  if (timeSinceLastUpdate < 1000) {
578  if (counter++ > 10) {
579  return;
580  }
581  } else {
582  counter = 0;
583  }
584 
585  ZoneScopedN("flush metrics");
586  auto& monitoring = registry.get<Monitoring>();
587  auto& relayer = registry.get<DataRelayer>();
588 
589  O2_SIGNPOST_START(MonitoringStatus::ID, MonitoringStatus::FLUSH, 0, 0, O2_SIGNPOST_RED);
590  // Send all the relevant metrics for the relayer to update the GUI
591  // FIXME: do a delta with the previous version if too many metrics are still
592  // sent...
593  for (size_t si = 0; si < stats.statesSize.load(); ++si) {
594  auto value = std::atomic_load_explicit(&stats.relayerState[si], std::memory_order_relaxed);
595  std::atomic_thread_fence(std::memory_order_acquire);
596  monitoring.send({value, fmt::format("data_relayer/{}", si), o2::monitoring::Verbosity::Debug});
597  }
598  relayer.sendContextState();
599  monitoring.flushBuffer();
600  stats.lastMetricFlushedTimestamp.store(stats.beginIterationTimestamp.load());
601  O2_SIGNPOST_END(MonitoringStatus::ID, MonitoringStatus::FLUSH, 0, 0, O2_SIGNPOST_RED);
602 };
