38 void TPCVDriftTglCalibration::finalizeSlot(Slot& slot)
39 {
40  // Extract results for the single slot
41  const auto* cont = slot.getContainer();
42  const auto* h = cont->histo.get();
43  auto nx = h->getNBinsX(), ny = h->getNBinsY();
44  std::array<double, 3> parg;
45  ROOT::Math::SMatrix<double, 3, 3, ROOT::Math::MatRepSym<double, 3>> mat;
46  double sS = 0, sX = 0, sY = 0, sXX = 0, sXY = 0;
47  int npval = 0;
48  for (auto ix = nx; ix--;) {
49  const auto sliceY = h->getSliceY(ix);
50  auto chi2 = o2::math_utils::fitGaus(ny, sliceY.data(), h->getYMin(), h->getYMax(), parg, &mat);
51  if (chi2 < -2) { // failed
52  continue;
53  }
54  double err2 = mat(1, 1);
55  if (err2 <= 0.f) {
56  continue;
57  }
58  double err2i = 1. / err2, xb = h->getBinXCenter(ix), xbw = xb * err2i, ybw = parg[1] * err2i;
59  sS += err2i;
60  sX += xbw;
61  sXX += xb * xbw;
62  sY += ybw;
63  sXY += xb * ybw;
64  npval++;
65  }
66  double det = sS * sXX - sX * sX;
67  if (!det || npval < 2) {
68  LOG(error) << "VDrift fit failed for slot " << slot.getTFStart() << " <= TF <= " << slot.getTFEnd()
69  << " with " << cont->entries << " entries";
70  }
71  det = 1. / det;
72  double offs = (sXX * sY - sX * sXY) * det, slope = (sS * sXY - sX * sY) * det;
73  double offsErr = sXX * det, slopErr = sS * det;
74  offsErr = offsErr > 0. ? std::sqrt(offsErr) : 0.;
75  slopErr = slopErr > 0. ? std::sqrt(slopErr) : 0.;
76  float corrFact = 1. / (1. - slope);
77  float corrFactErr = corrFact * corrFact * slopErr;
78  mVDPerSlot.emplace_back(corrFact, corrFactErr);
79  auto clName = o2::utils::MemFileHelper::getClassName(mVDPerSlot.back());
80  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
81  std::map<std::string, std::string> md;
82  mCCDBInfoPerSlot.emplace_back("TPC/Calib/VDriftTgl", clName, flName, md, slot.getTFStart(), 99999999999999);
83  LOG(info) << "Finalize slot " << slot.getTFStart() << " <= TF <= " << slot.getTFEnd() << " with " << cont->entries << " entries :"
84  << " dTgl vs Tgl_ITS offset:" << offs << "+-" << offsErr << " Slope: " << slope << "+-" << slopErr
85  << " -> VD corr factor = " << corrFact << "+-" << corrFactErr;
86 }
