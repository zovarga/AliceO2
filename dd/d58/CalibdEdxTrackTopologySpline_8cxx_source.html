184 void CalibdEdxTrackTopologySpline::setDefaultSplines()
185 {
186  FlatObject::startConstruction();
187 
188  int buffSize = 0;
189  int offsets1[FSplines];
190  int offsets2[FSplines];
191 
192  auto defaultF = [&](const double x[], double f[]) {
193  f[0] = 1.f;
194  };
195  double xMin[FDimX]{};
196  double xMax[FDimX]{};
197 
198  for (int iDimX = 0; iDimX < FDimX; ++iDimX) {
199  xMin[iDimX] = 0;
200  xMax[iDimX] = 1;
201  }
202 
203  for (unsigned int ireg = 0; ireg < FSplines; ++ireg) {
204  SplineType splineTmpqMax;
205  splineTmpqMax.approximateFunction(xMin, xMax, defaultF);
206  mCalibSplinesqMax[ireg] = splineTmpqMax;
207  buffSize = alignSize(buffSize, mCalibSplinesqMax[ireg].getBufferAlignmentBytes());
208  offsets1[ireg] = buffSize;
209  buffSize += mCalibSplinesqMax[ireg].getFlatBufferSize();
210  }
211 
212  for (unsigned int ireg = 0; ireg < FSplines; ++ireg) {
213  SplineType splineTmpqTot;
214  splineTmpqTot.approximateFunction(xMin, xMax, defaultF);
215  mCalibSplinesqTot[ireg] = splineTmpqTot;
216  buffSize = alignSize(buffSize, mCalibSplinesqTot[ireg].getBufferAlignmentBytes());
217  offsets2[ireg] = buffSize;
218  buffSize += mCalibSplinesqTot[ireg].getFlatBufferSize();
219  }
220 
221  FlatObject::finishConstruction(buffSize);
222 
223  for (unsigned int i = 0; i < FSplines; i++) {
224  mCalibSplinesqMax[i].moveBufferTo(mFlatBufferPtr + offsets1[i]);
225  }
226  for (unsigned int i = 0; i < FSplines; i++) {
227  mCalibSplinesqTot[i].moveBufferTo(mFlatBufferPtr + offsets2[i]);
228  }
229 }
