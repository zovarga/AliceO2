58 class DigitsSinkTask
59 {
60  public:
61  //_________________________________________________________________________________________________
62  void init(framework::InitContext& ic)
63  {
65  LOG(info) << "initializing digits sink";
66 
67  mText = ic.options().get<bool>("txt");
68 
69  auto outputFileName = ic.options().get<std::string>("outfile");
70  mOutputFile.open(outputFileName, (mText ? std::ios::out : (std::ios::out | std::ios::binary)));
71  if (!mOutputFile.is_open()) {
72  throw std::invalid_argument("Cannot open output file" + outputFileName);
73  }
74 
75  auto stop = [this]() {
77  LOG(info) << "stop file reader";
78  this->mOutputFile.close();
79  };
80  ic.services().get<CallbackService>().set(CallbackService::Id::Stop, stop);
81  }
82 
83  //_________________________________________________________________________________________________
84  void run(framework::ProcessingContext& pc)
85  {
86  // get the input digits
87  auto digits = pc.inputs().get<gsl::span<Digit>>("digits");
88  auto orbits = pc.inputs().get<gsl::span<OrbitInfo>>("orbits");
89  auto rofs = pc.inputs().get<gsl::span<ROFRecord>>("digitrofs");
90  std::set<OrbitInfo> ordered_orbits(orbits.begin(), orbits.end());
91 
92  if (mText) {
93  mOutputFile << std::endl
94  << "=======================" << std::endl;
95  for (const auto& o : ordered_orbits) {
96  mOutputFile << " FEEID " << o.getFeeID() << " LINK " << (int)o.getLinkID() << " ORBIT " << o.getOrbit() << std::endl;
97  }
98  mOutputFile << "---------------" << std::endl;
99  for (const auto& d : digits) {
100  mOutputFile << " DE# " << d.getDetID() << " PadId " << d.getPadID() << " ADC " << d.getADC() << std::endl;
101  }
102  mOutputFile << "---------------" << std::endl;
103  for (const auto& rof : rofs) {
104  mOutputFile << " IR " << rof.getBCData() << " first " << rof.getFirstIdx() << " size " << rof.getNEntries() << std::endl;
105  }
106  } else {
107  int nDigits = digits.size();
108  mOutputFile.write(reinterpret_cast<char*>(&nDigits), sizeof(int));
109  mOutputFile.write(reinterpret_cast<const char*>(digits.data()), digits.size_bytes());
110  }
111  }
112 
113  private:
114  std::ofstream mOutputFile{};
115  bool mText = false;
116 };
