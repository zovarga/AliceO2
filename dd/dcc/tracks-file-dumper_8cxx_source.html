37 template <typename T>
38 bool readBinaryStruct(std::istream& in, int nitems, std::vector<T>& items, const char* itemName)
39 {
40  if (in.peek() == EOF) {
41  return false;
42  }
43  // get the items if any
44  if (nitems > 0) {
45  auto offset = items.size();
46  items.resize(offset + nitems);
47  in.read(reinterpret_cast<char*>(&items[offset]), nitems * sizeof(T));
48  if (in.fail()) {
49  throw std::length_error(fmt::format("invalid input : cannot read {} {}", nitems, itemName));
50  }
51  }
52  return true;
53 }

80 int dumpBinary(std::string inputFile)
81 {
82  std::ifstream in(inputFile.c_str());
83  if (!in.is_open()) {
84  std::cerr << "cannot open input file " << inputFile << "\n";
85  return 3;
86  }
87 
88  while (in.good()) {
89  int nofTracksAtVertex{-1};
90  int nofTracks{-1};
91  int nofAttachedClusters{-1};
92  // read the number of tracks at vertex, MCH tracks and attached clusters
93  if (!in.read(reinterpret_cast<char*>(&nofTracksAtVertex), sizeof(int))) {
94  return -1;
95  }
96  if (!in.read(reinterpret_cast<char*>(&nofTracks), sizeof(int))) {
97  return -1;
98  }
99  if (!in.read(reinterpret_cast<char*>(&nofAttachedClusters), sizeof(int))) {
100  return -1;
101  }
102  std::cout << fmt::format("=== nof MCH tracks: {:2d} at vertex: {:2d} w/ {:4d} attached clusters\n",
103  nofTracks, nofTracksAtVertex, nofAttachedClusters);
104  std::vector<TrackAtVtxStruct> tracksAtVertex;
105  std::vector<TrackMCH> tracks;
106  std::vector<Cluster> clusters;
107  // read the tracks, tracks at vertex and clusters (mind the reverse order of tracks
108  // compared to the numbers above)
109  if (!readBinaryStruct<TrackAtVtxStruct>(in, nofTracksAtVertex, tracksAtVertex, "TracksAtVertex")) {
110  return -1;
111  }
112  if (!readBinaryStruct<TrackMCH>(in, nofTracks, tracks, "Tracks")) {
113  return -1;
114  }
115  if (!readBinaryStruct<Cluster>(in, nofAttachedClusters, clusters, "AttachedClusters")) {
116  return -1;
117  }
118 
119  dump(std::cout, tracksAtVertex);
120  }
121  return 0;
122 }

124 int dumpRoot(std::string inputFile)
125 {
126  std::unique_ptr<TFile> fin(TFile::Open(inputFile.c_str()));
127  TTree* tree = static_cast<TTree*>(fin->Get("o2sim"));
128 
129  TrackTreeReader tr(tree);
130 
131  ROFRecord rof;
132  std::vector<TrackMCH> tracks;
133  std::vector<Cluster> clusters;
134  std::vector<o2::MCCompLabel> labels;
135 
136  while (tr.next(rof, tracks, clusters, labels)) {
137  std::cout << rof << "\n";
138  if (tr.hasLabels() && labels.size() != tracks.size()) {
139  std::cerr << "the number of labels do not match the number of tracks\n";
140  return -1;
141  }
142  int it(0);
143  for (const auto& t : tracks) {
144  std::cout << " ";
145  dump(std::cout, t);
146  if (tr.hasLabels()) {
147  std::cout << " MC label: ";
148  labels[it++].print();
149  }
150  }
151  }
152  return 0;
153 }
