56 int check(const Hit& hit, const Digit& digit, Segmentation& seg, o2::mch::geo::TransformationCreator transformation)
57 {
58  int padid = digit.getPadID();
59  bool check = seg.isValid(padid);
60  if (!check) {
61  BOOST_FAIL(" digit-pad not belonging to hit det-element-ID ");
62  return 0;
63  }
64  double padposX = seg.padPositionX(padid);
65  double padsizeX = seg.padSizeX(padid);
66  double padposY = seg.padPositionY(padid);
67  double padsizeY = seg.padSizeY(padid);
68  auto t = transformation(digit.getDetID());
69 
70  o2::math_utils::Point3D<float> pos(hit.GetX(), hit.GetY(), hit.GetZ());
71  o2::math_utils::Point3D<float> lpos;
72  t.MasterToLocal(pos, lpos);
73 
74  // very loose check : check that digit position is within 10 pads of the
75  // hit center in both directions
76  BOOST_CHECK(std::abs(lpos.x() - padposX) < padsizeX * 10);
77  BOOST_CHECK(std::abs(lpos.y() - padposY) < padsizeY * 10);
78  return 1;
79 }

88 BOOST_AUTO_TEST_CASE(DigitizerTest)
89 {
90  // FIXME: must set a (global) seed here to get reproducible results
91 
92  auto transformation = o2::mch::geo::transformationFromTGeoManager(*gGeoManager);
93 
94  o2::mch::Digitizer digitizer(transformation);
95  int trackId1 = 0;
96  int trackId2 = 1;
97  short detElemId1 = 101;
98  short detElemId2 = 1012;
99 
100  float eloss1 = 1e-6;
101  float eloss2 = 1e-6;
102  float length = 0.f;
103  float tof = 0.0;
104 
105  std::vector<o2::mch::Hit> hits(2);
106  hits.at(0) = o2::mch::Hit(trackId1, detElemId1, entrancePoint1, exitPoint1, eloss1, length, tof);
107  hits.at(1) = o2::mch::Hit(trackId2, detElemId2, entrancePoint2, exitPoint2, eloss2, length, tof);
108 
109  Segmentation seg1{detElemId1};
110  Segmentation seg2{detElemId2};
111 
112  digitizer.startCollision({0, 0});
113 
114  o2::conf::ConfigurableParam::setValue("MCHDigitizer", "noiseProba", 0.0);
115  digitizer.processHits(hits, 0, 0);
116 
117  std::vector<Digit> digits;
118  o2::dataformats::MCTruthContainer<o2::MCCompLabel> labels;
119 
120  digitizer.extractDigitsAndLabels(digits, labels);
121 
122  int digitcounter1 = 0;
123  int digitcounter2 = 0;
124  int count = 0;
125 
126  for (const auto& digit : digits) {
127  auto label = (labels.getLabels(count))[0];
128  int trackID = label.getTrackID();
129  ++count;
130 
131  if (trackID == trackId1) {
132  digitcounter1 += check(hits.at(0), digit, seg1, transformation);
133  } else if (trackID == trackId2) {
134  digitcounter2 += check(hits.at(1), digit, seg2, transformation);
135  } else {
136  BOOST_FAIL(" MC-labels not matching between hit and digit ");
137  }
138  }
139  BOOST_TEST(digitcounter1 > 0);
140  BOOST_TEST(digitcounter1 < 20);
141  BOOST_TEST(digitcounter2 > 0);
142  BOOST_TEST(digitcounter2 < 10);
143 }
