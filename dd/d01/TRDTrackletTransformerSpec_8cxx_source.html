35 void TRDTrackletTransformerSpec::run(o2::framework::ProcessingContext& pc)
36 {
37  LOG(info) << "Running tracklet transformer";
38  if (!mTransformer.hasCalibration()) {
39  // ccdb object was not found for specified timestamp
40  return;
41  }
42 
43  o2::globaltracking::RecoContainer inputData;
44  inputData.collectData(pc, *mDataRequest);
45 
46  //auto tracklets = inputData.getTRDTracklets();
47  //auto trigRecs = inputData.getTRDTriggerRecords();
48 
49  auto tracklets = pc.inputs().get<gsl::span<Tracklet64>>("trdtracklets");
50  auto trigRecs = pc.inputs().get<gsl::span<TriggerRecord>>("trdtriggerrec");
51 
52  std::vector<CalibratedTracklet> calibratedTracklets(tracklets.size());
53 
54  std::vector<char> trigRecBitfield(trigRecs.size()); // flag TRD IR with ITS data (std::vector<bool> platform dependend)
55  int nTrackletsTransformed = 0;
56 
57  if (mTrigRecFilterActive) {
58  const auto irFrames = inputData.getIRFramesITS();
59  int lastMatchedIdx = 0; // ITS IR are sorted in time and do not overlap
60  for (const auto& irFrame : irFrames) {
61  for (int j = lastMatchedIdx; j < trigRecs.size(); ++j) {
62  const auto& trigRec = trigRecs[j];
63  if (trigRec.getBCData() >= irFrame.getMin()) {
64  if (trigRec.getBCData() <= irFrame.getMax()) {
65  // TRD interaction record inside ITS frame
66  trigRecBitfield[j] = 1;
67  lastMatchedIdx = j;
68  } else {
69  // too late, also the higher trigger records won't match
70  break;
71  }
72  }
73  }
74  LOGF(debug, "ITS IR Frame start: %li, end: %li", irFrame.getMin().toLong(), irFrame.getMax().toLong());
75  }
76  /*
77  // for debugging: print TRD trigger times which are accepted and which are filtered out
78  for (int j = 0; j < trigRecs.size(); ++j) {
79  const auto& trigRec = trigRecs[j];
80  if (!trigRecBitfield[j]) {
81  LOGF(debug, "Could not find ITS info for TRD trigger %i: %li", j, trigRec.getBCData().toLong());
82  } else {
83  LOGF(debug, "Found ITS info for TRD trigger %i: %li", j, trigRec.getBCData().toLong());
84  }
85  }
86  */
87  } else {
88  // fill bitmask with 1
89  std::fill(trigRecBitfield.begin(), trigRecBitfield.end(), 1);
90  }
91 
92  if (mTrigRecFilterActive) {
93  // skip tracklets from TRD triggers without ITS data
94  for (int iTrig = 0; iTrig < trigRecs.size(); ++iTrig) {
95  if (!trigRecBitfield[iTrig]) {
96  continue;
97  } else {
98  const auto& trigRec = trigRecs[iTrig];
99  for (int iTrklt = trigRec.getFirstTracklet(); iTrklt < trigRec.getFirstTracklet() + trigRec.getNumberOfTracklets(); ++iTrklt) {
100  calibratedTracklets[iTrklt] = mTransformer.transformTracklet(tracklets[iTrklt]);
101  ++nTrackletsTransformed;
102  }
103  }
104  }
105  } else {
106  // transform all tracklets
107  for (int iTrklt = 0; iTrklt < tracklets.size(); ++iTrklt) {
108  calibratedTracklets[iTrklt] = mTransformer.transformTracklet(tracklets[iTrklt]);
109  ++nTrackletsTransformed;
110  }
111  }
112 
113  LOGF(info, "Found %lu tracklets. Applied filter for ITS IR frames: %i. Transformed %i tracklets.", tracklets.size(), mTrigRecFilterActive, nTrackletsTransformed);
114 
115  pc.outputs().snapshot(Output{"TRD", "CTRACKLETS", 0, Lifetime::Timeframe}, calibratedTracklets);
116  pc.outputs().snapshot(Output{"TRD", "TRIGRECMASK", 0, Lifetime::Timeframe}, trigRecBitfield);
117 }

119 o2::framework::DataProcessorSpec getTRDTrackletTransformerSpec(bool trigRecFilterActive, int timestamp)
120 {
121  std::shared_ptr<DataRequest> dataRequest = std::make_shared<DataRequest>();
122  if (trigRecFilterActive) {
123  dataRequest->requestIRFramesITS();
124  }
125  auto& inputs = dataRequest->inputs;
126  inputs.emplace_back("trdtracklets", "TRD", "TRACKLETS", 0, Lifetime::Timeframe);
127  inputs.emplace_back("trdtriggerrec", "TRD", "TRKTRGRD", 0, Lifetime::Timeframe);
128 
129  std::vector<OutputSpec> outputs;
130  outputs.emplace_back("TRD", "CTRACKLETS", 0, Lifetime::Timeframe);
131  outputs.emplace_back("TRD", "TRIGRECMASK", 0, Lifetime::Timeframe);
132 
133  return DataProcessorSpec{
134  "TRDTRACKLETTRANSFORMER",
135  inputs,
136  outputs,
137  AlgorithmSpec{adaptFromTask<TRDTrackletTransformerSpec>(dataRequest, trigRecFilterActive, timestamp)},
138  Options{}};
