52  BOOST_CHECK_EQUAL(std::string(dataType.description.as<std::string>()), "FOOO");
53 
54  BOOST_CHECK(DataSpecUtils::match(spec, ConcreteDataMatcher{"TEST", "FOOO", 1}));
55  BOOST_CHECK(DataSpecUtils::match(spec, ConcreteDataMatcher{"TEST", "FOOO", 0}) == false);
56  DataSpecUtils::updateMatchingSubspec(spec, 0);
57  BOOST_CHECK(DataSpecUtils::match(spec, ConcreteDataMatcher{"TEST", "FOOO", 0}) == true);
58  BOOST_CHECK(DataSpecUtils::match(inputSpec, ConcreteDataMatcher{"TEST", "FOOO", 1}));
59  BOOST_CHECK(DataSpecUtils::match(inputSpec, ConcreteDataMatcher{"TEST", "FOOO", 0}) == false);

136 BOOST_AUTO_TEST_CASE(MatchingOutputs)
137 {
138  OutputSpec output1{
139  "TST", "A1", 0, Lifetime::Timeframe};
140 
141  OutputSpec output2{
142  "TST", "B1", 0, Lifetime::Timeframe};
143 
144  OutputSpec output3{
145  {"TST", "A1"}, Lifetime::Timeframe};
146 
147  InputSpec input1{
148  "binding", "TST", "A1", 0, Lifetime::Timeframe};
149 
150  InputSpec input2{
151  "binding", "TST", "A1", 1, Lifetime::Timeframe};
152 
153  InputSpec input3{
154  "binding", "TST", "B1", 0, Lifetime::Timeframe};
155 
156  InputSpec input4{
157  "binding", {"TST", "A1"}, Lifetime::Timeframe};
158 
159  // matching inputs to outputs
160  BOOST_CHECK(DataSpecUtils::match(input1, output1) == true);
161  BOOST_CHECK(DataSpecUtils::match(input1, output2) == false);
162  BOOST_CHECK(DataSpecUtils::match(input1, output3) == false); // Wildcard on output!
163  BOOST_CHECK(DataSpecUtils::match(input2, output1) == false);
164  BOOST_CHECK(DataSpecUtils::match(input2, output2) == false);
165  BOOST_CHECK(DataSpecUtils::match(input2, output3) == false); // Wildcard on output!
166  BOOST_CHECK(DataSpecUtils::match(input3, output1) == false);
167  BOOST_CHECK(DataSpecUtils::match(input3, output2) == true);
168  BOOST_CHECK(DataSpecUtils::match(input3, output3) == false); // Wildcard on output!
169  BOOST_CHECK(DataSpecUtils::match(input4, output1) == true); // Wildcard in input!
170  BOOST_CHECK(DataSpecUtils::match(input4, output2) == false);
171  BOOST_CHECK(DataSpecUtils::match(input4, output3) == true); // Wildcard on both!
172 
173  // matching outputs to output definitions
174  // ConcreteDataMatcher on both sides
175  BOOST_CHECK(DataSpecUtils::match(output1, OutputSpec{"TST", "A1", 0}) == true);
176  BOOST_CHECK(DataSpecUtils::match(output1, OutputSpec{"TST", "A1", 1}) == false);
177 
178  // ConcreteDataMatcher left, ConcreteDataTypeMatcher right (subspec ignored)
179  BOOST_CHECK(DataSpecUtils::match(output1, OutputSpec{"TST", "A1"}) == true);
180 
181  // ConcreteDataTypeMatcher left (subspec ignored), ConcreteDataMatcher right
182  BOOST_CHECK(DataSpecUtils::match(output3, OutputSpec{"TST", "A1", 0}) == true);
183  BOOST_CHECK(DataSpecUtils::match(output3, OutputSpec{"TST", "A1", 1}) == true);
184 
185  // ConcreteDataTypeMatcher on both sides
186  BOOST_CHECK(DataSpecUtils::match(output3, OutputSpec{"TST", "A1"}) == true);
187 }
