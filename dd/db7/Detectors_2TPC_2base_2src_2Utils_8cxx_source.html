45 TH1* utils::getBinInfoXY(int& binx, int& biny, float& bincx, float& bincy)
46 {
47  TObject* select = gPad->GetSelected();
48  if (!select) {
49  return nullptr;
50  }
51  if (!select->InheritsFrom("TH2")) {
52  gPad->SetUniqueID(0);
53  return nullptr;
54  }
55 
56  TH1* h = (TH1*)select;
57  gPad->GetCanvas()->FeedbackMode(kTRUE);
58 
59  const int px = gPad->GetEventX();
60  const int py = gPad->GetEventY();
61  const float xx = gPad->AbsPixeltoX(px);
62  const float x = gPad->PadtoX(xx);
63  const float yy = gPad->AbsPixeltoY(py);
64  const float y = gPad->PadtoX(yy);
65  binx = h->GetXaxis()->FindBin(x);
66  biny = h->GetYaxis()->FindBin(y);
67  bincx = h->GetXaxis()->GetBinCenter(binx);
68  bincy = h->GetYaxis()->GetBinCenter(biny);
69  //printf("binx, biny: %d %d\n",binx,biny);
70 
71  return h;
72 }

74 void utils::addFECInfo()
75 {
76  using namespace o2::tpc;
77  const int event = gPad->GetEvent();
78  if (event != 51) {
79  return;
80  }
81 
82  int binx, biny;
83  float bincx, bincy;
84  TH1* h = utils::getBinInfoXY(binx, biny, bincx, bincy);
85  if (!h) {
86  return;
87  }
88 
89  const float binValue = h->GetBinContent(binx, biny);
90  const int row = int(std::floor(bincx));
91  const int cpad = int(std::floor(bincy));
92 
93  const auto& mapper = Mapper::instance();
94 
95  const int roc = h->GetUniqueID();
96  if (roc < 0 || roc >= (int)ROC::MaxROC) {
97  return;
98  }
99  if (row < 0 || row >= (int)mapper.getNumberOfRowsROC(roc)) {
100  return;
101  }
102  const int nPads = mapper.getNumberOfPadsInRowROC(roc, row);
103  const int pad = cpad + nPads / 2;
104  //printf("row %d, cpad %d, pad %d, nPads %d\n", row, cpad, pad, nPads);
105  if (pad < 0 || pad >= (int)nPads) {
106  return;
107  }
108  const int channel = mapper.getPadNumberInROC(PadROCPos(roc, row, pad));
109 
110  const auto& fecInfo = mapper.getFECInfo(PadROCPos(roc, row, pad));
111 
112  const std::string title = fmt::format(
113  "#splitline{{#lower[.1]{{#scale[.5]{{"
114  "{}{:02d} ({:02d}) row: {:02d}, pad: {:03d}, globalpad: {:05d} (in roc)"
115  "}}}}}}{{#scale[.5]{{FEC: "
116  "{:02d}, Chip: {:02d}, Chn: {:02d}, Value: {:.3f}"
117  "}}}}",
118  (roc / 18 % 2 == 0) ? "A" : "C", roc % 18, roc, row, pad, channel, fecInfo.getIndex(), fecInfo.getSampaChip(), fecInfo.getSampaChannel(), binValue);
119 
120  h->SetTitle(title.data());
121 }

182 void utils::mergeCalPads(std::string_view outputFileName, std::string_view inputFileNames, std::string_view calPadNames, bool average)
183 {
184  using namespace o2::tpc;
185 
186  const auto calPadNamesVec = utils::tokenize(calPadNames, ",");
187 
188  std::string_view cmd = "ls";
189  if (inputFileNames.rfind(".root") == std::string_view::npos) {
190  cmd = "cat";
191  }
192  auto files = gSystem->GetFromPipe(TString::Format("%s %s", cmd.data(), inputFileNames.data()));
193  std::unique_ptr<TObjArray> arrFiles(files.Tokenize("\n"));
194 
195  std::vector<CalPad*> mergedCalPads;
196 
197  for (auto ofile : *arrFiles) {
198  auto calPads = utils::readCalPads(ofile->GetName(), calPadNamesVec);
199  if (!calPads.size()) {
200  continue;
201  }
202  if (!mergedCalPads.size()) {
203  mergedCalPads = calPads;
204  } else {
205  for (size_t iCalPad = 0; iCalPad < calPads.size(); ++iCalPad) {
206  auto calPadName = calPadNamesVec[iCalPad];
207  auto calPadMerged = mergedCalPads[iCalPad];
208  calPadMerged->setName(calPadName);
209  auto calPadToMerge = calPads[iCalPad];
210 
211  *calPadMerged += *calPadToMerge;
212 
213  delete calPadToMerge;
214  }
215  }
216  }
217 
218  if (average) {
219  const auto n = arrFiles->GetEntriesFast();
220  if (n > 0) {
221  for (auto calPad : mergedCalPads) {
222  *calPad /= n;
223  }
224  }
225  }
226 
227  std::unique_ptr<TFile> outFile(TFile::Open(outputFileName.data(), "recreate"));
228  for (auto calPad : mergedCalPads) {
229  outFile->WriteObject(calPad, calPad->getName().data());
230  }
231 }
