47 std::vector<painter::PadCoordinates> painter::getPadCoordinatesSector()
48 {
49  std::vector<painter::PadCoordinates> padCoords;
50 
51  const auto& regInf = Mapper::instance().getMapPadRegionInfo();
52 
53  for (const auto& padReg : regInf) {
54  const auto npr = padReg.getNumberOfPadRows();
55  const auto ro = padReg.getRowOffset();
56  const auto xm = padReg.getXhelper();
57  const auto ph = padReg.getPadHeight();
58  const auto pw = padReg.getPadWidth();
59  const auto yro = padReg.getRadiusFirstRow();
60  const auto ks = ph / pw * std::tan(1.74532925199432948e-01);
61 
62  for (int irow = 0; irow < npr; ++irow) {
63  const auto npads = std::floor(ks * (irow + ro) + xm);
64  for (int ipad = -npads; ipad < npads; ++ipad) {
65  const auto xPadBottomRight = yro + ph * irow;
66  const auto xPadTopRight = yro + ph * (irow + 1);
67  const auto ri = xPadBottomRight;
68  const auto yPadBottomRight = pw * ipad * xPadBottomRight / (ri + ph / 2);
69  const auto yPadTopRight = pw * ipad * xPadTopRight / (ri + ph / 2);
70  const auto yPadBottomLeft = pw * (ipad + 1) * xPadBottomRight / (ri + ph / 2);
71  const auto yPadTopLeft = pw * (ipad + 1) * xPadTopRight / (ri + ph / 2);
72  auto& padCoord = padCoords.emplace_back();
73  padCoord.xVals = {xPadBottomRight, xPadTopRight, xPadTopRight, xPadBottomRight};
74  padCoord.yVals = {yPadBottomRight, yPadTopRight, yPadTopLeft, yPadBottomLeft};
75  }
76  }
77  }
78 
79  return padCoords;
80 }

89 template <class T>
90 TCanvas* painter::draw(const CalDet<T>& calDet, int nbins1D, float xMin1D, float xMax1D, TCanvas* outputCanvas)
91 {
92  using DetType = CalDet<T>;
93  using CalType = CalArray<T>;
94 
95  static const Mapper& mapper = Mapper::instance();
96 
97  // ===| name and title |======================================================
98  std::string title = calDet.getName();
99  std::string name = calDet.getName();
100  std::replace(name.begin(), name.end(), ' ', '_');
101  std::replace(title.begin(), title.end(), '_', ' ');
102 
103  // ===| define histograms |===================================================
104  // TODO: auto scaling of ranges based on mean and variance?
105  // for the moment use roots auto scaling
106 
107  // set buffer size such that autoscaling uses the full range. This is about 2MB per histogram!
108  const int bufferSize = TH1::GetDefaultBufferSize();
109  TH1::SetDefaultBufferSize(Sector::MAXSECTOR * mapper.getPadsInSector());
110 
111  auto hAside1D = new TH1F(fmt::format("h_Aside_1D_{}", name).data(), fmt::format("{} (A-Side)", title).data(),
112  nbins1D, xMin1D, xMax1D); // TODO: modify ranges
113 
114  auto hCside1D = new TH1F(fmt::format("h_Cside_1D_{}", name).data(), fmt::format("{} (C-Side)", title).data(),
115  nbins1D, xMin1D, xMax1D); // TODO: modify ranges
116 
117  auto hAside2D = new TH2F(fmt::format("h_Aside_2D_{}", name).data(), fmt::format("{} (A-Side);#it{{x}} (cm);#it{{y}} (cm)", title).data(),
118  330, -270, 270, 330, -270, 270);
119 
120  auto hCside2D = new TH2F(fmt::format("h_Cside_2D_{}", name).data(), fmt::format("{} (C-Side);#it{{x}} (cm);#it{{y}} (cm)", title).data(),
121  330, -270, 270, 330, -270, 270);
122 
123  for (ROC roc; !roc.looped(); ++roc) {
124 
125  auto hist2D = hAside2D;
126  auto hist1D = hAside1D;
127  if (roc.side() == Side::C) {
128  hist2D = hCside2D;
129  hist1D = hCside1D;
130  }
131 
132  const int nrows = mapper.getNumberOfRowsROC(roc);
133  for (int irow = 0; irow < nrows; ++irow) {
134  const int npads = mapper.getNumberOfPadsInRowROC(roc, irow);
135  for (int ipad = 0; ipad < npads; ++ipad) {
136  const auto val = calDet.getValue(roc, irow, ipad);
137  const GlobalPosition2D pos = mapper.getPadCentre(PadROCPos(roc, irow, ipad));
138  const int bin = hist2D->FindBin(pos.X(), pos.Y());
139  if (!hist2D->GetBinContent(bin)) {
140  hist2D->SetBinContent(bin, val);
141  }
142  hist1D->Fill(val);
143  }
144  }
145  }
146 
147  if (xMax1D > xMin1D) {
148  hAside2D->SetMinimum(xMin1D);
149  hAside2D->SetMaximum(xMax1D);
150  hCside2D->SetMinimum(xMin1D);
151  hCside2D->SetMaximum(xMax1D);
152  }
153 
154  // ===| Draw histograms |=====================================================
155  gStyle->SetOptStat("mr");
156  auto c = outputCanvas;
157  if (!c) {
158  c = new TCanvas(fmt::format("c_{}", name).data(), title.data(), 1000, 1000);
159  }
160  gStyle->SetStatX(1. - gPad->GetRightMargin());
161  gStyle->SetStatY(1. - gPad->GetTopMargin());
162 
163  c->Clear();
164  c->Divide(2, 2);
165 
166  c->cd(1);
167  hAside2D->Draw("colz");
168  hAside2D->SetStats(0);
169  hAside2D->SetTitleOffset(1.05, "XY");
170  hAside2D->SetTitleSize(0.05, "XY");
171  drawSectorsXY(Side::A);
172 
173  c->cd(2);
174  hCside2D->Draw("colz");
175  hCside2D->SetStats(0);
176  hCside2D->SetTitleOffset(1.05, "XY");
177  hCside2D->SetTitleSize(0.05, "XY");
178  drawSectorsXY(Side::C);
179 
180  c->cd(3);
181  hAside1D->Draw();
182 
183  c->cd(4);
184  hCside1D->Draw();
185 
186  // reset the buffer size
187  TH1::SetDefaultBufferSize(bufferSize);
188 
189  // associate histograms to canvas
190  hAside1D->SetBit(TObject::kCanDelete);
191  hCside1D->SetBit(TObject::kCanDelete);
192  hAside2D->SetBit(TObject::kCanDelete);
193  hCside2D->SetBit(TObject::kCanDelete);
194 
195  return c;
196 }

217 void painter::fillHistogram2D(TH2& h2D, const CalDet<T>& calDet, Side side)
218 {
219  static const Mapper& mapper = Mapper::instance();
220 
221  for (ROC roc; !roc.looped(); ++roc) {
222  if (roc.side() != side) {
223  continue;
224  }
225 
226  const int nrows = mapper.getNumberOfRowsROC(roc);
227  for (int irow = 0; irow < nrows; ++irow) {
228  const int npads = mapper.getNumberOfPadsInRowROC(roc, irow);
229  for (int ipad = 0; ipad < npads; ++ipad) {
230  const auto val = calDet.getValue(roc, irow, ipad);
231  const GlobalPosition2D pos = mapper.getPadCentre(PadROCPos(roc, irow, ipad));
232  const int bin = h2D.FindBin(pos.X(), pos.Y());
233  if (!h2D.GetBinContent(bin)) {
234  h2D.SetBinContent(bin, val);
235  }
236  }
237  }
238  }
239 }

242 template <class T>
243 void painter::fillHistogram2D(TH2& h2D, const CalArray<T>& calArray)
244 {
245  static const Mapper& mapper = Mapper::instance();
246 
247  const size_t position = calArray.getPadSubsetNumber();
248  const PadSubset padSubset = calArray.getPadSubset();
249  const int nrows = mapper.getNumberOfPadRows(padSubset, position);
250 
251  // ===| fill hist |===========================================================
252  for (int irow = 0; irow < nrows; ++irow) {
253  const int padsInRow = mapper.getNumberOfPadsInRow(padSubset, position, irow);
254  for (int ipad = 0; ipad < padsInRow; ++ipad) {
255  const GlobalPadNumber pad = mapper.getPadNumber(padSubset, position, irow, ipad);
256  const auto val = calArray.getValue(pad);
257  const int cpad = ipad - padsInRow / 2;
258  h2D.Fill(irow, cpad, val);
259  }
260  }
261 }

283 template <class T>
284 TH2* painter::getHistogram2D(const CalArray<T>& calArray)
285 {
286  static const Mapper& mapper = Mapper::instance();
287 
288  const size_t position = calArray.getPadSubsetNumber();
289  const PadSubset padSubset = calArray.getPadSubset();
290 
291  // ===| maximum number of rows and pads |=====================================
292  const int nrows = mapper.getNumberOfPadRows(padSubset, position);
293  const int npads = mapper.getNumberOfPadsInRow(padSubset, position, nrows - 1) + 6;
294 
295  // ===| create histogram |====================================================
296  std::string title = calArray.getName();
297  std::string name = calArray.getName();
298  std::replace(title.begin(), title.end(), '_', ' ');
299  std::replace(name.begin(), name.end(), ' ', '_');
300 
301  if (padSubset == PadSubset::ROC) {
302  title += fmt::format(" ({})", getROCTitle(position));
303  }
304 
305  auto hist = new TH2F(fmt::format("h_{}", name).data(),
306  fmt::format("{};pad row;pad", title).data(),
307  nrows, 0., nrows,
308  npads, -npads / 2., npads / 2.);
309 
310  fillHistogram2D(*hist, calArray);
311 
312  return hist;
313 }

327 template <class T>
328 std::vector<TCanvas*> painter::makeSummaryCanvases(const CalDet<T>& calDet, int nbins1D, float xMin1D, float xMax1D, bool onlyFilled, std::vector<TCanvas*>* outputCanvases)
329 {
330 
331  std::vector<TCanvas*> vecCanvases;
332 
333  auto nROCs = calDet.getData().size();
334 
335  if (onlyFilled) {
336  nROCs = 0;
337  for (size_t iroc = 0; iroc < calDet.getData().size(); ++iroc) {
338  const auto& roc = calDet.getCalArray(iroc);
339 
340  if (hasData(roc)) {
341  ++nROCs;
342  }
343  }
344 
345  if (!nROCs) {
346  return vecCanvases;
347  }
348  }
349 
350  // ===| set up canvases |===
351  TCanvas* cSides = nullptr;
352  TCanvas* cROCs1D = nullptr;
353  TCanvas* cROCs2D = nullptr;
354  const std::string_view calName = calDet.getName();
355 
356  if (outputCanvases) {
357  if (outputCanvases->size() < 3) {
358  LOGP(error, "At least 3 canvases are needed to fill the output, only {} given", outputCanvases->size());
359  return vecCanvases;
360  }
361 
362  cSides = outputCanvases->at(0);
363  cROCs1D = outputCanvases->at(1);
364  cROCs2D = outputCanvases->at(2);
365  cSides->Clear();
366  cROCs1D->Clear();
367  cROCs2D->Clear();
368  } else {
369 
370  cROCs1D = new TCanvas(fmt::format("c_ROCs_{}_1D", calName).data(), fmt::format("{} values for each ROC", calName).data(), 1400, 1000);
371  cROCs2D = new TCanvas(fmt::format("c_ROCs_{}_2D", calName).data(), fmt::format("{} values for each ROC", calName).data(), 1400, 1000);
372  }
373  cSides = draw(calDet, nbins1D, xMin1D, xMax1D, cSides);
374  cROCs1D->DivideSquare(nROCs);
375  cROCs2D->DivideSquare(nROCs);
376 
377  vecCanvases.emplace_back(cSides);
378  vecCanvases.emplace_back(cROCs1D);
379  vecCanvases.emplace_back(cROCs2D);
380 
381  // ===| produce plots for each ROC |===
382  size_t pad = 1;
383  for (size_t iroc = 0; iroc < calDet.getData().size(); ++iroc) {
384  const auto& roc = calDet.getCalArray(iroc);
385 
386  if (onlyFilled && !hasData(roc)) {
387  continue;
388  }
389 
390  // ===| 1D histogram |===
391  auto h1D = new TH1F(fmt::format("h1_{}_{:02d}", calName, iroc).data(), fmt::format("{} distribution ROC {:02d} ({});ADC value", calName, iroc, getROCTitle(iroc)).data(), nbins1D, xMin1D, xMax1D);
392  for (const auto& val : roc.getData()) {
393  h1D->Fill(val);
394  }
395 
396  // ===| 2D histogram |===
397  auto h2D = painter::getHistogram2D(roc);
398  h2D->SetStats(0);
399  if (xMax1D > xMin1D) {
400  h2D->SetMinimum(xMin1D);
401  h2D->SetMaximum(xMax1D);
402  }
403  h2D->SetUniqueID(iroc);
404 
405  cROCs1D->cd(pad);
406  h1D->Draw();
407 
408  cROCs2D->cd(pad);
409  h2D->Draw("colz");
410 
411  ++pad;
412 
413  // associate histograms to canvas
414  h1D->SetBit(TObject::kCanDelete);
415  h2D->SetBit(TObject::kCanDelete);
416  }
417 
418  return vecCanvases;
419 }

496 //______________________________________________________________________________
497 void painter::drawSectorsXY(Side side, int sectorLineColor, int sectorTextColor)
498 {
499  TLine l;
500  l.SetLineColor(sectorLineColor);
501 
502  TLine l2;
503  l2.SetLineColor(sectorLineColor);
504  l2.SetLineStyle(kDotted);
505 
506  TLatex latSide;
507  latSide.SetTextColor(sectorTextColor);
508  latSide.SetTextAlign(22);
509  latSide.SetTextSize(0.08);
510  latSide.DrawLatex(0, 0, (side == Side::C) ? "C" : "A");
511 
512  TLatex lat;
513  lat.SetTextAlign(22);
514  lat.SetTextSize(0.03);
515  lat.SetTextColor(sectorLineColor);
516 
517  constexpr float phiWidth = float(SECPHIWIDTH);
518  const float rFactor = std::cos(phiWidth / 2.);
519  const float rLow = 83.65 / rFactor;
520  const float rIROCup = 133.3 / rFactor;
521  const float rOROClow = 133.5 / rFactor;
522  const float rOROC12 = 169.75 / rFactor;
523  const float rOROC23 = 207.85 / rFactor;
524  const float rOut = 247.7 / rFactor;
525  const float rText = rLow * rFactor * 3. / 4.;
526 
527  for (Int_t isector = 0; isector < 18; ++isector) {
528  const float sinR = std::sin(phiWidth * isector);
529  const float cosR = std::cos(phiWidth * isector);
530 
531  const float sinL = std::sin(phiWidth * ((isector + 1) % 18));
532  const float cosL = std::cos(phiWidth * ((isector + 1) % 18));
533 
534  const float sinText = std::sin(phiWidth * (isector + 0.5));
535  const float cosText = std::cos(phiWidth * (isector + 0.5));
536 
537  const float xR1 = rLow * cosR;
538  const float yR1 = rLow * sinR;
539  const float xR2 = rOut * cosR;
540  const float yR2 = rOut * sinR;
541 
542  const float xL1 = rLow * cosL;
543  const float yL1 = rLow * sinL;
544  const float xL2 = rOut * cosL;
545  const float yL2 = rOut * sinL;
546 
547  const float xOROCmup1 = rOROClow * cosR;
548  const float yOROCmup1 = rOROClow * sinR;
549  const float xOROCmup2 = rOROClow * cosL;
550  const float yOROCmup2 = rOROClow * sinL;
551 
552  const float xIROCmup1 = rIROCup * cosR;
553  const float yIROCmup1 = rIROCup * sinR;
554  const float xIROCmup2 = rIROCup * cosL;
555  const float yIROCmup2 = rIROCup * sinL;
556 
557  const float xO121 = rOROC12 * cosR;
558  const float yO121 = rOROC12 * sinR;
559  const float xO122 = rOROC12 * cosL;
560  const float yO122 = rOROC12 * sinL;
561 
562  const float xO231 = rOROC23 * cosR;
563  const float yO231 = rOROC23 * sinR;
564  const float xO232 = rOROC23 * cosL;
565  const float yO232 = rOROC23 * sinL;
566 
567  const float xText = rText * cosText;
568  const float yText = rText * sinText;
569 
570  // left side line
571  l.DrawLine(xR1, yR1, xR2, yR2);
572 
573  // IROC inner line
574  l.DrawLine(xR1, yR1, xL1, yL1);
575 
576  // IROC end line
577  l.DrawLine(xIROCmup1, yIROCmup1, xIROCmup2, yIROCmup2);
578 
579  // OROC start line
580  l.DrawLine(xOROCmup1, yOROCmup1, xOROCmup2, yOROCmup2);

670 template <typename DataT>
671 TH3F painter::convertCalDetToTH3(const std::vector<CalDet<DataT>>& calDet, const bool norm, const int nRBins, const float rMin, const float rMax, const int nPhiBins, const float zMax)
672 {
673  const int nZBins = calDet.size();
674  TH3F histConvSum("hisCalDet", "hisCalDet", nPhiBins, 0, o2::constants::math::TwoPI, nRBins, rMin, rMax, 2 * nZBins, -zMax, zMax); // final converted histogram
675  TH3F histConvWeight("histConvWeight", "histConvWeight", nPhiBins, 0, o2::constants::math::TwoPI, nRBins, rMin, rMax, 2 * nZBins, -zMax, zMax);
676 
677  typedef boost::geometry::model::polygon<boost::geometry::model::d2::point_xy<float>> polygon;
678  const auto coords = o2::tpc::painter::getPadCoordinatesSector(); // coordinates of the pads in one sector
679 
680  // create polygons per pad of the input histogram
681  std::vector<polygon> geoBin;
682  const int nGeoBins = nPhiBins * nRBins;
683  geoBin.reserve(nGeoBins);
684  for (int iphi = 1; iphi <= nPhiBins; ++iphi) {
685  const double phiLow = histConvSum.GetXaxis()->GetBinLowEdge(iphi);
686  const double phiUp = histConvSum.GetXaxis()->GetBinUpEdge(iphi);
687  for (int ir = 1; ir <= nRBins; ++ir) {
688  const double rLow = histConvSum.GetYaxis()->GetBinLowEdge(ir);
689  const double rUp = histConvSum.GetYaxis()->GetBinUpEdge(ir);
690  const double xPos1 = rLow * std::cos(phiLow);
691  const double yPos1 = rLow * std::sin(phiLow);
692  const double xPos2 = rLow * std::cos(phiUp);
693  const double yPos2 = rLow * std::sin(phiUp);
694  const double xPos4 = rUp * std::cos(phiLow);
695  const double yPos4 = rUp * std::sin(phiLow);
696  const double xPos3 = rUp * std::cos(phiUp);
697  const double yPos3 = rUp * std::sin(phiUp);
698  // round the values due to problems in intersection in boost with polygons close to each other
699  boost::geometry::read_wkt(Form("POLYGON((%.4f %.4f, %.4f %.4f, %.4f %.4f, %.4f %.4f, %.4f %.4f))", xPos1, yPos1, xPos2, yPos2, xPos3, yPos3, xPos4, yPos4, xPos1, yPos1), geoBin.emplace_back());
700  boost::geometry::correct(geoBin.back());
701  }
702  }
703 
704  for (unsigned int sector = 0; sector < Mapper::NSECTORS / 2; ++sector) {
705  for (unsigned int region = 0; region < Mapper::NREGIONS; ++region) {
706  const unsigned int rowsRegion = o2::tpc::Mapper::ROWSPERREGION[region];
707  for (unsigned int iRow = 0; iRow < rowsRegion; ++iRow) {
708  const unsigned int padsInRow = o2::tpc::Mapper::PADSPERROW[region][iRow] - 1;
709  for (unsigned int iPad = 0; iPad <= padsInRow; ++iPad) {
710  const GlobalPadNumber padNum = Mapper::getGlobalPadNumber(iRow, iPad, region);
711 
712  const float angDeg = 10.f + sector * 20;
713  auto coordinate = coords[padNum];
714  coordinate.rotate(angDeg);
715 
716  const std::array<double, 2> radiusPadCoord{
717  std::sqrt(coordinate.xVals[0] * coordinate.xVals[0] + coordinate.yVals[0] * coordinate.yVals[0]),
718  std::sqrt(coordinate.xVals[2] * coordinate.xVals[2] + coordinate.yVals[2] * coordinate.yVals[2]),
719  };
720 
721  std::array<float, 4> phiPadCoord{
722  static_cast<float>(std::atan2(coordinate.yVals[0], coordinate.xVals[0])),
723  static_cast<float>(std::atan2(coordinate.yVals[1], coordinate.xVals[1])),
724  static_cast<float>(std::atan2(coordinate.yVals[2], coordinate.xVals[2])),
725  static_cast<float>(std::atan2(coordinate.yVals[3], coordinate.xVals[3]))};
726 
727  for (auto& phi : phiPadCoord) {
728  o2::math_utils::bringTo02PiGen(phi);
729  }
730 
731  // bins of the histogram of the edges of the pad
732  const int binRBottomStart = std::clamp(histConvSum.GetYaxis()->FindBin(radiusPadCoord[0]) - 1, 1, nRBins);
733  const int binRTopEnd = std::clamp(histConvSum.GetYaxis()->FindBin(radiusPadCoord[1]) + 1, 1, nRBins);
734  int binPhiStart = std::min(histConvSum.GetXaxis()->FindBin(phiPadCoord[0]), histConvSum.GetXaxis()->FindBin(phiPadCoord[1]));
735  std::clamp(binPhiStart - 1, 1, nPhiBins);
736  int binPhiEnd = std::max(histConvSum.GetXaxis()->FindBin(phiPadCoord[2]), histConvSum.GetXaxis()->FindBin(phiPadCoord[3]));
737  std::clamp(binPhiEnd + 1, 1, nPhiBins);
738 
739  // define boost geoemtry object
740  polygon geoPad;
741  boost::geometry::read_wkt(Form("POLYGON((%f %f, %f %f, %f %f, %f %f, %f %f))", coordinate.xVals[0], coordinate.yVals[0], coordinate.xVals[1], coordinate.yVals[1], coordinate.xVals[2], coordinate.yVals[2], coordinate.xVals[3], coordinate.yVals[3], coordinate.xVals[0], coordinate.yVals[0]), geoPad);
742  boost::geometry::correct(geoPad);
743 
744  for (int binR = binRBottomStart; binR <= binRTopEnd; ++binR) {
745  for (int binPhi = binPhiStart; binPhi <= binPhiEnd; ++binPhi) {
746  const int ind = (binPhi - 1) * nRBins + binR - 1;
747 
748  std::deque<polygon> output;
749  boost::geometry::intersection(geoPad, geoBin[ind], output);
750  if (output.empty()) {
751  continue;
752  }
753  const double area = boost::geometry::area(output.front());
754  const double fac = area * Mapper::INVPADAREA[region];
755 
756  for (int iSide = 0; iSide < 2; ++iSide) {
757  const Side side = iSide == 0 ? Side::C : Side::A;
758  const unsigned int iCRU = side == Side::A ? (sector * Mapper::NREGIONS + region) : ((sector + Mapper::NSECTORS / 2) * Mapper::NREGIONS + region);
759  const CRU cru(iCRU);
760 
761  for (int iz = 0; iz < nZBins; ++iz) {
762  const auto val = calDet[iz].getValue(cru, iRow, (side == Side::A) ? iPad : padsInRow - iPad);
763  const int zBin = side == Side::A ? (nZBins + iz + 1) : (nZBins - iz);
764  const auto globBin = histConvSum.GetBin(binPhi, binR, zBin);
765  histConvSum.AddBinContent(globBin, val * fac);
766  if (norm) {
767  histConvWeight.AddBinContent(globBin, fac);
768  }
769  }
770  }
771  }
772  }
773  }
774  }
775  }
776  }
777 
778  if (norm) {
779  histConvSum.Divide(&histConvWeight);
780  }
781 
782  return histConvSum;
783 }
