248 template <class S, int I, int J, typename... Args>
249 inline int GPUReconstructionCPU::runKernel(const krnlExec& x, const krnlRunRange& y, const krnlEvent& z, const Args&... args)
250 {
251  HighResTimer* t = nullptr;
252  GPUCA_RECO_STEP myStep = S::GetRecoStep() == GPUCA_RECO_STEP::NoRecoStep ? x.step : S::GetRecoStep();
253  if (myStep == GPUCA_RECO_STEP::NoRecoStep) {
254  throw std::runtime_error("Failure running general kernel without defining RecoStep");
255  }
256  int cpuFallback = IsGPU() ? (x.device == krnlDeviceType::CPU ? 2 : (mRecoStepsGPU & myStep) != myStep) : 0;
257  unsigned int nThreads = x.nThreads;
258  unsigned int nBlocks = x.nBlocks;
259  auto prop = getKernelProperties<S, I>();
260  const int autoThreads = cpuFallback ? 1 : prop.nThreads;
261  const int autoBlocks = cpuFallback ? 1 : (prop.forceBlocks ? prop.forceBlocks : (prop.minBlocks * mBlockCount));
262  if (nBlocks == (unsigned int)-1) {
263  nBlocks = (nThreads + autoThreads - 1) / autoThreads;
264  nThreads = autoThreads;
265  } else if (nBlocks == (unsigned int)-2) {
266  nBlocks = nThreads;
267  nThreads = autoThreads;
268  } else if (nBlocks == (unsigned int)-3) {
269  nBlocks = autoBlocks;
270  nThreads = autoThreads;
271  } else if ((int)nThreads < 0) {
272  nThreads = cpuFallback ? 1 : -nThreads;
273  }
274  if (nThreads > GPUCA_MAX_THREADS) {
275  throw std::runtime_error("GPUCA_MAX_THREADS exceeded");
276  }
277  if (mProcessingSettings.debugLevel >= 3) {
278  GPUInfo("Running kernel %s (Stream %d, Range %d/%d, Grid %d/%d) on %s", GetKernelName<S, I>(), x.stream, y.start, y.num, nBlocks, nThreads, cpuFallback == 2 ? "CPU (forced)" : cpuFallback ? "CPU (fallback)" : mDeviceName.c_str());
279  }
280  if (nThreads == 0 || nBlocks == 0) {
281  return 0;
282  }
283  if (mProcessingSettings.debugLevel >= 1) {
284  t = &getKernelTimer<S, I, J>(myStep, !IsGPU() || cpuFallback ? getOMPThreadNum() : x.stream);
285  if ((!mProcessingSettings.deviceTimers || !IsGPU() || cpuFallback) && (mNestedLoopOmpFactor < 2 || getOMPThreadNum() == 0)) {
286  t->Start();
287  }
288  }
289  krnlSetup setup{{nBlocks, nThreads, x.stream, x.device, x.step}, y, z, 0.};
290  if (cpuFallback) {
291  if (GPUReconstructionCPU::runKernelImpl(classArgument<S, I>(), setup, args...)) {
292  return 1;
293  }
294  } else {
295  if (runKernelImpl(classArgument<S, I>(), setup, args...)) {
296  return 1;
297  }
298  }
299  if (GPUDebug(GetKernelName<S, I>(), x.stream)) {
300  throw std::runtime_error("kernel failure");
301  }
302  if (mProcessingSettings.debugLevel >= 1) {
303  if (t) {
304  if (!(!mProcessingSettings.deviceTimers || !IsGPU() || cpuFallback)) {
305  t->AddTime(setup.t);
306  } else if (mNestedLoopOmpFactor < 2 || getOMPThreadNum() == 0) {
307  t->Stop();
308  }
309  }
310  if (CheckErrorCodes(cpuFallback) && !mProcessingSettings.ignoreNonFatalGPUErrors) {
311  throw std::runtime_error("kernel error code");
312  }
313  }
314  return 0;
315 }

333 template <class T, int I, int J>
334 HighResTimer& GPUReconstructionCPU::getKernelTimer(RecoStep step, int num, size_t addMemorySize)
335 {
336  static int id = getNextTimerId();
337  timerMeta* timer = getTimerById(id);
338  if (timer == nullptr) {
339  timer = insertTimer(id, GetKernelName<T, I>(), J, NSLICES, 0, step);
340  }
341  if (addMemorySize) {
342  timer->memSize += addMemorySize;
343  }
344  if (num < 0 || num >= timer->num) {
345  throw std::runtime_error("Invalid timer requested");
346  }
347  return timer->timer[num];
348 }

350 template <class T, int J>
351 HighResTimer& GPUReconstructionCPU::getTimer(const char* name, int num)
352 {
353  static int id = getNextTimerId();
354  timerMeta* timer = getTimerById(id);
355  if (timer == nullptr) {
356  int max = std::max<int>({getOMPMaxThreads(), mProcessingSettings.nDeviceHelperThreads + 1, mProcessingSettings.nStreams});
357  timer = insertTimer(id, name, J, max, 1, RecoStep::NoRecoStep);
358  }
359  if (num == -1) {
360  num = getOMPThreadNum();
361  }
362  if (num < 0 || num >= timer->num) {
363  throw std::runtime_error("Invalid timer requested");
364  }
365  return timer->timer[num];
366 }
