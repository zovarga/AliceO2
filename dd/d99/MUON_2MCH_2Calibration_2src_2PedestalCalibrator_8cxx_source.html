74 //_____________________________________________
75 void PedestalCalibrator::finalizeSlot(Slot& slot)
76 {
77  // Extract results for the single slot
78  o2::mch::calibration::PedestalData* c = slot.getContainer();
79  LOG(info) << "Finalize slot " << slot.getTFStart() << " <= TF <= " << slot.getTFEnd();
80 
81  // keep track of first TimeFrame
82  if (slot.getTFStart() < mTFStart) {
83  mTFStart = slot.getTFStart();
84  }
85 
86  auto pedestals = c->getPedestals();
87  for (auto& p : pedestals) {
88  auto& pMat = p.second;
89  for (size_t dsId = 0; dsId < pMat.size(); dsId++) {
90  auto& pRow = pMat[dsId];
91  for (size_t ch = 0; ch < pRow.size(); ch++) {
92  auto& pRecord = pRow[ch];
93 
94  if (pRecord.mEntries == 0) {
95  continue;
96  }
97 
98  mPedestalsVector.emplace_back(DsChannelId::make(p.first, dsId, ch), pRecord.mPedestal, pRecord.getRms());
99 
100  bool bad = true;
101  if (pRecord.mPedestal < mPedestalThreshold) {
102  if (pRecord.getRms() < mNoiseThreshold) {
103  bad = false;
104  }
105  }
106 
107  if (bad) {
108  LOG(info) << "S " << p.first << " DS " << dsId << " CH " << ch
109  << " ENTRIES " << pRecord.mEntries << " PED " << pRecord.mPedestal << " RMS " << pRecord.getRms();
110  mBadChannelsVector.getChannels().emplace_back(p.first, dsId, ch);
111  }
112  }
113  }
114  }
115 }
