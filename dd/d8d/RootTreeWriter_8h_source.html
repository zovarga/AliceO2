242  template <typename Arg, typename... Args>
243  void init(Arg&& arg, Args&&... args)
244  {
245  using Type = std::decay_t<Arg>;
246  if constexpr (can_assign<Type, IndexExtractor>::value) {
247  getIndex = arg;
248  } else if constexpr (can_assign<Type, BranchNameMapper>::value) {
249  getName = arg;
250  } else if constexpr (can_assign<Type, Fill>::value) {
251  callback = arg;
252  } else if constexpr (can_assign<Type, FillExt>::value) {
253  callback = arg;
254  } else if constexpr (can_assign<Type, Spectator>::value) {
255  callback = arg;
256  } else if constexpr (can_assign<Type, SpectatorExt>::value) {
257  callback = arg;
258  } else if constexpr (std::is_integral<Type>::value) {
259  nofBranches = arg;
260  } else {
261  assertNoMatchingType(std::forward<Arg>(arg));
262  }
263  if constexpr (sizeof...(args) > 0) {
264  init(std::forward<Args>(args)...);
265  }

395  template <typename T>
396  static TBranch* remapBranch(TBranch& branchRef, T** newdata)
397  {
398  auto tree = branchRef.GetTree();
399  auto name = branchRef.GetName();
400  auto branch = tree->GetBranch(name); // the input branch might actually no belong to the tree but to TreeWriter cache
401  assert(branch);
402  if (branch->GetEntries()) { // if it has entries, then it was already remapped/filled at prevous event
403  branch->SetAddress(newdata);
404  return branch;
405  }
406  auto branchleaves = branch->GetListOfLeaves();
407  branch->DropBaskets("all");
408  branch->DeleteBaskets("all");
409  tree->GetListOfBranches()->Remove(branch);
410  for (auto entry : *branchleaves) {
411  tree->GetListOfLeaves()->Remove(entry);
412  }
413  // ROOT leaves holes in the arrays so we need to compress as well
414  tree->GetListOfBranches()->Compress();
415  tree->GetListOfLeaves()->Compress();
416  // create a new branch with the same original name but attached to the new data
417  return tree->Branch(name, newdata);
418  }

463  // type trait to determine the type of the storage
464  // the storage type is different depending on the type, because
465  // - binary branches are supported
466  // - the ROOT TTree API makes a difference between fundamental types where the pointer to variable is
467  // passed and complex objects with dictionary where the pointer to a pointer variable needs to be
468  // passed. The former case will allways include a copy of the value, the latter uses the deserialized
469  // object by pointer
470  // It appears, that also vectors of fundamental and also messageable tyapes can be treated in the same
471  // way as fundamental types, as long as there is a corresponding TClass implementation. ROOT creates those
472  // for vectors of fundamental types, for all other types, ClassDefNV macro is needed in the class
473  // declaration together with entry in LinkDef.
474  template <typename T, typename _ = void>
475  struct StructureElementTypeTrait {
476  };

567  // specialization for binary buffers indicated by const char*
568  // a variable binary data branch is written, the size of each entry is stored in a size branch
569  // specialized on specialization_id
570  template <typename S, typename std::enable_if_t<std::is_same<S, BinaryBranchSpecialization>::value, int> = 0>
571  TBranch* createBranch(TTree* tree, const char* name, size_t branchIdx)
572  {
573  // size variable to write the size of the data branch
574  std::get<2>(mStore.at(branchIdx)) = 1;
575  // branch name of the size branch
576  std::string sizeBranchName = std::string(name) + "Size";
577  // leaf list: one leaf of type unsinged int
578  std::string leafList = sizeBranchName + "/i";
579  std::get<1>(mStore.at(branchIdx)) = tree->Branch(sizeBranchName.c_str(), &(std::get<2>(mStore.at(branchIdx))), leafList.c_str());
580  return tree->Branch(name, &(std::get<0>(mStore.at(branchIdx))));
581  }

584  void setupInstance(std::vector<BranchSpec>& specs, TTree* tree)
585  {
586  // recursing through the tree structure by simply using method of the previous type,
587  // i.e. the base class method.
588  PrevT::setupInstance(specs, tree);
589  constexpr size_t SpecIndex = STAGE - 1;
590  if (specs[SpecIndex].branches.size() == 0) {
591  // this definition is disabled
592  return;
593  }
594  specs[SpecIndex].classinfo = TClass::GetClass(typeid(value_type));
595  if (std::is_same<value_type, const char*>::value == false && std::is_fundamental<value_type>::value == false &&
596  specs[SpecIndex].classinfo == nullptr) {
597  // for all non-fundamental types but the special case for binary chunks, a dictionary is required
598  // FIXME: find a reliable way to check that the type has been specified in the LinkDef
599  // Only then the required functionality for streaming the type to the branch is available.
600  // If e.g. a standard container of some ROOT serializable type has not been specified in
601  // LinkDef, the functionality is not available and the attempt to stream will simply crash.
602  // Unfortunately, a class info object can be extracted for the type, so this check does not help
603  throw std::runtime_error(std::to_string(SpecIndex) + ": no dictionary available for non-fundamental type " + typeid(value_type).name());
604  }
605  size_t branchIdx = 0;
606  mStore.resize(specs[SpecIndex].names.size());
607  for (auto const& name : specs[SpecIndex].names) {
608  specs[SpecIndex].branches.at(branchIdx) = createBranch<specialization_id>(tree, name.c_str(), branchIdx);
609  if (specs[SpecIndex].branches.at(branchIdx) == nullptr) {
610  throw std::runtime_error(std::to_string(SpecIndex) + ": can not create branch " + name + " for type " + typeid(value_type).name() + " - LinkDef entry missing?");
611  }
612  LOG(info) << SpecIndex << ": branch " << name << " set up";
613  branchIdx++;
614  }
615  }

619  template <typename DataType>
620  bool runCallback(TBranch* branch, DataType const& data, DataRef const& ref)
621  {
622  if (std::holds_alternative<typename BranchDef<value_type>::Spectator>(mCallback)) {
623  std::get<typename BranchDef<value_type>::Spectator>(mCallback)(data);
624  }
625  if (std::holds_alternative<typename BranchDef<value_type>::SpectatorExt>(mCallback)) {
626  std::get<typename BranchDef<value_type>::SpectatorExt>(mCallback)(data, ref);
627  }
628  if (std::holds_alternative<typename BranchDef<value_type>::Fill>(mCallback)) {
629  std::get<typename BranchDef<value_type>::Fill>(mCallback)(*branch, data);
630  return true;
631  }
632  if (std::holds_alternative<typename BranchDef<value_type>::FillExt>(mCallback)) {
633  std::get<typename BranchDef<value_type>::FillExt>(mCallback)(*branch, data, ref);
634  return true;
635  }
636  return false;
637  }

667  // specialization for binary buffers using const char*
668  // this writes both the data branch and a size branch
669  template <typename S, typename std::enable_if_t<std::is_same<S, BinaryBranchSpecialization>::value, int> = 0>

673  std::get<2>(mStore.at(branchIdx)) = data.size();
674  std::get<1>(mStore.at(branchIdx))->Fill();
675  std::get<0>(mStore.at(branchIdx)).resize(data.size());

680  // specialization for vectors of messageable types
681  template <typename S, typename std::enable_if_t<std::is_same<S, MessageableVectorSpecialization>::value, int> = 0>
682  void fillData(InputContext& context, DataRef const& ref, TBranch* branch, size_t branchIdx)
683  {
684  using ElementType = typename value_type::value_type;
685  static_assert(is_messageable<ElementType>::value, "logical error: should be correctly selected by StructureElementTypeTrait");
686 
687  // A helper struct mimicking data layout of std::vector containers
688  // We assume a standard layout of begin, end, end_capacity
689  struct VecBase {
690  VecBase() = default;
691  const ElementType* start = nullptr;
692  const ElementType* end = nullptr;
693  const ElementType* cap = nullptr;
694  };
695 
696  // a low level hack to make a gsl::span appear as a std::vector so that ROOT serializes the correct type
697  // but without the need for an extra copy
698  auto adopt = [](auto const& data, value_type& v) {
699  static_assert(sizeof(v) == 24);
700  if (data.size() == 0) {
701  return;
702  }
703  VecBase impl;
704  impl.start = &(data[0]);
705  impl.end = &(data[data.size() - 1]) + 1; // end pointer (beyond last element)
706  impl.cap = impl.end;
707  std::memcpy(&v, &impl, sizeof(VecBase));
708  };
709 
710  // if the value type is messagable and has a ROOT dictionary, two serialization methods are possible
711  // for the moment, the InputRecord API can not handle both with the same call
712  try {
713  // try extracting from message with serialization method NONE, throw runtime error
714  // if message is serialized
715  auto data = context.get<gsl::span<ElementType>>(ref);
716  // take an ordinary std::vector "view" on the data
717  auto* dataview = new value_type;
718  adopt(data, *dataview);
719  if (!runCallback(branch, *dataview, ref)) {
720  mStore[branchIdx] = dataview;
721  branch->Fill();
722  }
723  // we delete JUST the view without deleting the data (which is handled by DPL)
724  auto ptr = (VecBase*)dataview;
725  if (ptr) {
726  delete ptr;
727  }
728  } catch (RuntimeErrorRef e) {
729  if constexpr (has_root_dictionary<value_type>::value == true) {
730  // try extracting from message with serialization method ROOT
731  auto data = context.get<typename std::add_pointer<value_type>::type>(ref);
732  if (!runCallback(branch, *data, ref)) {
733  mStore[branchIdx] = const_cast<value_type*>(data.get());
734  branch->Fill();
735  }
736  } else {
737  // the type has no ROOT dictionary, re-throw exception
738  throw e;
739  }
740  }
741  }

743  // process previous stage and this stage
744  void process(InputContext& context, std::vector<BranchSpec>& specs)
745  {
746  // recursing through the tree structure by simply using method of the previous type,
747  // i.e. the base class method.
748  PrevT::process(context, specs);
749  constexpr size_t SpecIndex = STAGE - 1;
750  BranchSpec const& spec = specs[SpecIndex];
751  if (spec.branches.size() == 0) {
752  // this definition is disabled
753  return;
754  }
755  // loop over all defined inputs
756  for (auto const& key : spec.keys) {
757  auto keypos = context.getPos(key.c_str());
758  auto parts = context.getNofParts(keypos);
759  for (decltype(parts) part = 0; part < parts; part++) {

786  template <typename... Args>
787  void setCallback(Args&&... args)
788  {
789  // recursing through the tree structure by simply using method of the previous type,
790  // i.e. the base class method.
791  if constexpr (STAGE > 1) {
792  PrevT::setCallback(std::forward<Args>(args)...);
793  }
794

796  auto arg = getArg<STAGE - 1>(std::forward<Args>(args)...);

798  if (not std::holds_alternative<std::monostate>(arg.callback)) {
799  mCallback = std::move(arg.callback);

828  template <size_t N, typename BASE, typename T, typename... Args>
829  auto createTreeStructure(T&& def, Args&&... args)
830  {
831  // Note: a branch definition can be disabled by setting nofBranches to zero
832  // an entry of the definition is kept in the registry, but the branches vector
833  // is of size zero, which will make the writer to always skip this definition
834  mBranchSpecs.push_back({{}, {def.branchName}});
835  auto& spec = mBranchSpecs.back();
836 
837  // extract the internal keys for the list of provided input definitions
838  size_t idx = 0;
839  spec.keys.resize(def.keys.size());
840  std::generate(spec.keys.begin(), spec.keys.end(), [&def, &idx] { return T::key_extractor::asString(def.keys[idx++]); });
841  mBranchSpecs.back().branches.resize(def.nofBranches, nullptr);
842  // a branch definition might be disabled by setting number of branches to 0; if enabled,
843  // the number of branches has to match the number of inputs but can be larger depending
844  // on the exact functionality provided with the getIndex callback. In any case, the
845  // callbacks only need to be propagated if multiple branches are defined
846  assert(def.nofBranches == 0 || def.nofBranches >= spec.keys.size());
847  // a getIndex function makes only sense if there are multiple branches
848  assert(def.nofBranches <= 1 || def.getIndex);
849  if (def.nofBranches > 1) {
850  // FIXME: should that be an exception since assert is disabled in the normal build?
851  assert(def.getIndex && def.getName);
852  mBranchSpecs.back().getIndex = def.getIndex;
853  mBranchSpecs.back().names.resize(def.nofBranches);
854  }
855 
856  // always amend the branch name(s) if the callback is configured
857  if (def.getName) {
858  mBranchSpecs.back().getName = def.getName;
859  idx = 0;
860  std::generate(mBranchSpecs.back().names.begin(), mBranchSpecs.back().names.end(),
861  [&def, &idx]() { return def.getName(def.branchName, idx++); });
862  }
863  using type = TreeStructureElement<typename T::type, BASE>;
864  if constexpr (N == 0) {
865  // in the upper most call we have the concrete type of the mixin object
866  // and call its method to set the callbacks from the argument pack, it
867  // recurses to all folded types
868  auto instance = createTreeStructure<N + 1, type>(std::forward<Args>(args)...);
869  instance->setCallback(std::move(def), std::forward<Args>(args)...);
870  // the upper most call returns the unique pointer of the virtual interface
871  std::unique_ptr<TreeStructureInterface> ret(instance.release());
872  return ret;
873  } else {
874  return std::move(createTreeStructure<N + 1, type>(std::forward<Args>(args)...));
875  }
876  }
877 
878  // last iteration, returning unique pointer of the concrete mixin
879  template <size_t N, typename T>
880  std::unique_ptr<T> createTreeStructure()
881  {
882  static_assert(N > 0, "The writer does not make sense without branch definitions");
883  return std::make_unique<T>();
884  }
