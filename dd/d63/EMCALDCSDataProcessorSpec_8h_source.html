129  void run(o2::framework::ProcessingContext& pc) final
130  {
131  auto grp = checkRunStartEnd();
132 
133  auto tfid = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime;
134  auto dps = pc.inputs().get<gsl::span<DPCOM>>("input");
135  mProcessor->setTF(tfid);
136  mProcessor->process(dps);
137  auto timeNow = HighResClock::now();
138  Duration elapsedTime = timeNow - mTimer; // in seconds
139  if (elapsedTime.count() >= mDPsUpdateInterval) {
140  sendELMButput(pc.outputs());
141  mTimer = timeNow;
142  }
143  sendCFGoutput(pc.outputs());
144  }
145 
146  void endOfStream(o2::framework::EndOfStreamContext& ec) final
147  {
148  sendELMButput(ec.outputs());
149  sendCFGoutput(ec.outputs());
150  }

197  //________________________________________________________________
198  const o2::parameters::GRPECSObject* checkRunStartEnd()
199  {
200  // check if EMC run is going on
201  if (!mCheckRunStartStop) {
202  return nullptr;
203  }
204  auto& mgr = CcdbManager::instance();
205  bool fatalOn = mgr.getFatalWhenNull();
206  mgr.setFatalWhenNull(false);
207  if (mRunStatus == EMCRunStatus::STOP) { // the STOP was detected at previous check
208  mRunFollowed = -1;
209  mRunStatus = EMCRunStatus::NONE;
210  }
211  std::map<std::string, std::string> md;
212  if (mRunFollowed > 0) { // run start was seen
213  md["runNumber"] = std::to_string(mRunFollowed);
214  }
215  long ts = std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::system_clock::now().time_since_epoch()).count();
216  const auto* grp = mgr.getSpecific<o2::parameters::GRPECSObject>("GLO/Config/GRPECS", ts, md);
217  if (grp) { // some object was returned
218  if (mRunFollowed > 0) {
219  if (grp->getTimeEnd() > grp->getTimeStart()) { // this means that the EOR was registered
220  mRunStatus = EMCRunStatus::STOP;
221  } else { // run still continues
222  mRunStatus = EMCRunStatus::ONGOING;
223  }
224  } else { // we were not following EMC run, check if the current one has an EMC
225  if (grp->getDetsReadOut()[o2::detectors::DetID::EMC]) { // we start following EMC run
226  mRunStatus = grp->getTimeEnd() > grp->getTimeStart() ? EMCRunStatus::STOP : EMCRunStatus::START;
227  mRunFollowed = grp->getRun();
228  }
229  }
230  } else { // query did not return any GRP -> we are certainly not in the EMC run
231  if (mRunFollowed > 0) { // normally this should not happen
232  LOGP(warning, "We were following EMC run {} but the query at {} did not return any GRP, problem with EOR?", mRunFollowed, ts);
233  mRunStatus = EMCRunStatus::STOP;
234  }
235  }
236  mgr.setFatalWhenNull(fatalOn);
237  return grp;
238  }

266  {"follow-emcal-run", VariantType::Bool, false, {"Check EMCAL runs SOR/EOR"}},
267  {"DPs-update-interval", VariantType::Int64, 600ll, {"Interval (in s) after which to update the DPs CCDB entry"}}}};
