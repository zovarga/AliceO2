44 void doQueryHVLV(const std::string ccdbUrl, uint64_t timestamp, bool hv, bool lv)
45 {
46  std::vector<std::string> what;
47  if (hv) {
48  what.emplace_back(fmt::format("{}/HV", o2::muon::subsysname()));
49  }
50  if (lv) {
51  what.emplace_back(fmt::format("{}/LV", o2::muon::subsysname()));
52  }
53 
54  auto sum =
55  [](float s, o2::dcs::DataPointValue v) {
56  union Converter {
57  uint64_t raw_data;
58  double value;
59  } converter;
60  converter.raw_data = v.payload_pt1;
61  return s + converter.value;
62  };
63 
64  o2::ccdb::CcdbApi api;
65  api.init(ccdbUrl);
66  for (auto w : what) {
67  std::map<std::string, std::string> metadata;
68  auto* m = api.retrieveFromTFileAny<DPMAP>(w, metadata, timestamp);
69  std::cout << "size of " << w << " map = " << m->size() << std::endl;
70  if (verbose) {
71  for (auto& i : *m) {
72  auto v = i.second;
73  auto mean = std::accumulate(v.begin(), v.end(), 0.0, sum);
74  if (v.size()) {
75  mean /= v.size();
76  }
77  std::cout << fmt::format("{:64s} {:4d} values of mean {:7.2f}\n", i.first.get_alias(), v.size(),
78  mean);
79  }
80  }
81  }
82 }

100 void makeCCDBEntryForDCS(const std::string ccdbUrl, uint64_t timestamp)
101 {
102  std::unordered_map<DPID, std::string> dpid2DataDesc;
103 #if defined(MUON_SUBSYSTEM_MCH)
104  auto aliases = o2::mch::dcs::aliases();
105 #elif defined(MUON_SUBSYSTEM_MID)
106  auto aliases = o2::mid::dcs::aliases();
107 #endif
108 
109  DPID dpidtmp;
110  for (const auto& a : aliases) {
111  auto legitName = o2::muon::replaceDotByUnderscore(a);
112  DPID::FILL(dpidtmp, legitName, o2::dcs::DeliveryType::RAW_DOUBLE);
113  dpid2DataDesc[dpidtmp] = fmt::format("{}DATAPOINTS", o2::muon ::subsysname());
114  }
115 
116  o2::ccdb::CcdbApi api;
117  api.init(ccdbUrl);
118  std::map<std::string, std::string> md;
119  std::cout << "storing config of " << dpid2DataDesc.size()
120  << o2::muon::subsysname() << " data points to "
121  << CcdbDpConfName() << "\n";
122 
123  api.storeAsTFileAny(&dpid2DataDesc, CcdbDpConfName(), md, timestamp);
124 }

176

182  lv = match(query, ".*(lv)");
183 #else
184  lv = false;
185 #endif
186  dpconf = match(query, ".*(dpconf)");
187 
188  if (!hv && !lv && !dpconf) {
189  std::cout << "Must specify at least one of dpconf,hv";
190 #if defined(MUON_SUBSYSTEM_MCH)
191  std::cout << ",lv";
192 #endif
193  std::cout << " parameter to --query option\n";
194  std::cout
195  << usage << "\n";
196  return 3;
197  }
198 
199  if (hv || lv) {
200  doQueryHVLV(ccdbUrl, timestamp, hv, lv);
201  }
202 
203  if (dpconf) {
204  doQueryDataPointConfig(ccdbUrl, timestamp, dpConfName);
205  }
206  }
207 
208  if (put) {
209  makeCCDBEntryForDCS(ccdbUrl, timestamp);
210  }
211  return 0;
