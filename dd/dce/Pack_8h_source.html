165 template <typename T, typename T1, typename... Ts>
166 constexpr size_t has_type_at(pack<T1, Ts...> const&)
167 {
168  if constexpr (std::is_same_v<T, T1>) {
169  return 0;
170  }
171  if constexpr (has_type_v<T, pack<T1, Ts...>>) {
172  return 1 + has_type_at<T>(pack<Ts...>{});
173  }
174  return sizeof...(Ts) + 2;
