38 /* |63|62|61|60|59|58|57|56|55|54|53|52|51|50|49|48|47|46|45|44|43|42|41|40|39|38|37|36|35|34|33|32|
39  -------------------------------------------------------------------------------------------------
40 Word 0 | Format | HCID | padrow | col | position |
41  -------------------------------------------------------------------------------------------------
42  |31|30|29|28|27|26|25|24|23|22|21|20|19|18|17|16|15|14|13|12|11|10|09|08|07|06|05|04|03|02|01|00|
43  -------------------------------------------------------------------------------------------------
44 Word 0 | slope | Q2 | Q1 | Q0 |
45  -------------------------------------------------------------------------------------------------
46 
47  Note: In the tracklet word both the position and the slope have one bit inverted. This avoids the
48  misinterpretation of the word as tracklet end marker. So before the position and slope are
49  written into the tracklet word they are XOR'd with 0x80, i.e. bits 31 and 39 are inverted.
50 */
51 class Tracklet64
52 {
53 
54  public:
55  GPUdDefault() Tracklet64() = default;
56  GPUd() Tracklet64(uint64_t trackletword) { mtrackletWord = trackletword; }
57  GPUdDefault() Tracklet64(const Tracklet64&) = default;
58  GPUd() Tracklet64(uint64_t format, uint64_t hcid, uint64_t padrow, uint64_t col, uint64_t position,
59  uint64_t slope, uint64_t Q0, uint64_t Q1, uint64_t Q2)
60  {
61  mtrackletWord = ((format << formatbs) & formatmask) |
62  ((hcid << hcidbs) & hcidmask) |
63  ((padrow << padrowbs) & padrowmask) |
64  ((col << colbs) & colmask) |
65  ((position << posbs) & posmask) |
66  ((slope << slopebs) & slopemask) |
67  ((Q2 << Q2bs) & Q2mask) |
68  ((Q1 << Q1bs) & Q1mask) |
69  ((Q0 << Q0bs) & Q0mask);
70  mtrackletWord ^= 0x8080000000UL;
71  }
72 
73  Tracklet64(uint64_t format, uint64_t hcid, uint64_t padrow, uint64_t col, uint64_t position,
74  uint64_t slope, uint64_t pid)
75  {
76  mtrackletWord = ((format << formatbs) & formatmask) |
77  ((hcid << hcidbs) & hcidmask) |
78  ((padrow << padrowbs) & padrowmask) |
79  ((col << colbs) & colmask) |
80  ((position << posbs) & posmask) |
81  ((slope << slopebs) & slopemask) |
82  (pid & PIDmask);
83  mtrackletWord ^= 0x8080000000UL;
84  }

89  // ----- Getters for contents of tracklet word -----
90  GPUd() int getFormat() const { return ((mtrackletWord & formatmask) >> formatbs); }; // no units 0..15
91  GPUd() int getHCID() const { return ((mtrackletWord & hcidmask) >> hcidbs); }; // no units 0..1079
92  GPUd() int getPadRow() const { return ((mtrackletWord & padrowmask) >> padrowbs); }; // pad row number [0..15]
93  GPUd() int getColumn() const { return ((mtrackletWord & colmask) >> colbs); }; // column refers to MCM position in column direction on readout board [0..3]
94  GPUd() int getPosition() const { return ((mtrackletWord & posmask) >> posbs) ^ 0x80; }; // in units of 1/40 pads, 11 bit granularity
95  GPUd() int getSlope() const { return ((mtrackletWord & slopemask) >> slopebs) ^ 0x80; }; // in units of 1/1000 pads/timebin, 8 bit granularity
96  GPUd() int getPID() const { return ((mtrackletWord & PIDmask)); }; // no unit, all 3 charge windows combined
97  GPUd() int getQ0() const { return ((mtrackletWord & Q0mask) >> Q0bs); }; // no unit
98  GPUd() int getQ1() const { return ((mtrackletWord & Q1mask) >> Q1bs); }; // no unit
99  GPUd() int getQ2() const { return ((mtrackletWord & Q2mask) >> Q2bs); }; // no unit
100 
101  GPUd() void setTrackletWord(uint64_t trackletword) { mtrackletWord = trackletword; }
102 
103  // ----- Getters for tracklet information -----
104  GPUd() int getMCM() const { return 4 * (getPadRow() % 4) + getColumn(); } // returns MCM position on ROB [0..15]
105  GPUd() int getROB() const { return (getHCID() % 2) ? (getPadRow() / 4) * 2 + 1 : (getPadRow() / 4) * 2; } // returns ROB number [0..5] for C0 chamber and [0..7] for C1 chamber
106  GPUd() int getPositionBinSigned() const;
107  GPUd() int getSlopeBinSigned() const;
108  GPUd() float getUncalibratedY() const; // translate local position into global y (in cm) not taking into account calibrations (ExB, vDrift, t0)
109  GPUd() float getUncalibratedDy(float nTbDrift = 19.4f) const; // translate local slope into dy/dx with dx=3m (drift length) and default drift time in time bins (19.4 timebins / 3cm)

184 GPUdi() int Tracklet64::getPositionBinSigned() const
185 {
186  int padLocalBin = getPosition();
187  int padLocal = 0;
188  if (padLocalBin & (1 << (constants::NBITSTRKLPOS - 1))) {
189  padLocal = -((~(padLocalBin - 1)) & ((1 << constants::NBITSTRKLPOS) - 1));
190  } else {
191  padLocal = padLocalBin & ((1 << constants::NBITSTRKLPOS) - 1);
192  }
193  return padLocal;
194 }

196 GPUdi() float Tracklet64::getUncalibratedY() const
197 {
198  int padLocal = getPositionBinSigned();
199  int mcmCol = (getMCM() % constants::NMCMROBINCOL) + constants::NMCMROBINCOL * (getROB() % 2);
200  // one pad column has 144 pads, the offset of -63 is the center of the first MCM in that column
201  // which is connected to the pads -63 - 9 = -72 to -63 + 9 = -54
202  float offset = -63.f + ((float)constants::NCOLMCM) * mcmCol;
203  float padWidth = 0.635f + 0.03f * (getDetector() % constants::NLAYER);
204  return (offset + padLocal * constants::GRANULARITYTRKLPOS) * padWidth;
205 }

207 GPUdi() int Tracklet64::getSlopeBinSigned() const
208 {
209  int slopeBin = getSlope();
210  int slope = 0;
211  if (slopeBin & (1 << (constants::NBITSTRKLSLOPE - 1))) {
212  slope = -((~(slopeBin - 1)) & ((1 << constants::NBITSTRKLSLOPE) - 1));
213  } else {
214  slope = slopeBin & ((1 << constants::NBITSTRKLSLOPE) - 1);
215  }
216  return -slope;
217 }
218 
219 GPUdi() float Tracklet64::getUncalibratedDy(float nTbDrift) const
220 {
221  int slope = getSlopeBinSigned();
222  float padWidth = 0.635f + 0.03f * (getDetector() % constants::NLAYER);
223  return slope * constants::GRANULARITYTRKLSLOPE * padWidth * nTbDrift / constants::ADDBITSHIFTSLOPE;
224 }
