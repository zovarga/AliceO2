33 void AlpideSimResponse::initData(int tableNumber)
34 {
35  /*
36  * read grid parameters and load data
37  */
38  if (tableNumber == 0) // 0V back bias
39  {
40  const std::string newDataPath = mDataPath + "Vbb-3.0V"; // temporary solution (Vbb-0.0V will be called in the end)
41  setDataPath(newDataPath); // setting the new data path
42  } else if (tableNumber == 1) // -3V back bias
43  {
44  const std::string newDataPath = mDataPath + "Vbb-3.0V";
45  setDataPath(newDataPath); // setting the new data path
46  }
47 
48  if (mData.size()) {
49  cout << "Object already initialized" << endl;
50  print();
51  return;
52  }
53 
54  const float kTiny = 1e-6; // to check 0 values
55 
56  // if needed, append path with slash
57  if (mDataPath.length() && mDataPath.back() != '/') {
58  mDataPath.push_back('/');
59  }
60  mDataPath = gSystem->ExpandPathName(mDataPath.data());
61  string inpfname = mDataPath + mGridColName;
62  std::ifstream inpGrid;
63 
64  // read X grid
65  inpGrid.open(inpfname, std::ifstream::in);
66  if (inpGrid.fail()) {
67  LOG(fatal) << "Failed to open file " << inpfname;
68  }
69 
70  while (inpGrid >> mStepInvCol && inpGrid.good()) {
71  mNBinCol++;
72  }
73 
74  if (!mNBinCol || mStepInvCol < kTiny) {
75  LOG(fatal) << "Failed to read X(col) binning from " << inpfname;
76  }
77  mMaxBinCol = mNBinCol - 1;
78  mStepInvCol = mMaxBinCol / mStepInvCol; // inverse of the X bin width
79  inpGrid.close();
80 
81  // read Y grid
82  inpfname = mDataPath + mGridRowName;
83  inpGrid.open(inpfname, std::ifstream::in);
84  if (inpGrid.fail()) {
85  LOG(fatal) << "Failed to open file " << inpfname;
86  }
87 
88  while (inpGrid >> mStepInvRow && inpGrid.good()) {
89  mNBinRow++;
90  }
91  if (!mNBinRow || mStepInvRow < kTiny) {
92  LOG(fatal) << "Failed to read Y(row) binning from " << inpfname;
93  }
94  mMaxBinRow = mNBinRow - 1;
95  mStepInvRow = mMaxBinRow / mStepInvRow; // inverse of the Row bin width
96  inpGrid.close();
97 
98  // load response data
99  int nz = 0;
100  size_t cnt = 0;
101  float val, gx, gy, gz;
102  int lost, untrck, dead, nele;
103  size_t dataSize = 0;
104  mDptMax = -2.e9;
105  mDptMin = 2.e9;
106  const int npix = AlpideRespSimMat::getNPix();
107 
108  for (int ix = 0; ix < mNBinCol; ix++) {
109  for (int iy = 0; iy < mNBinRow; iy++) {
110  inpfname = composeDataName(ix, iy);
111  inpGrid.open(inpfname, std::ifstream::in);
112  if (inpGrid.fail()) {
113  LOG(fatal) << "Failed to open file " << inpfname;
114  }
115  inpGrid >> nz;
116  if (cnt == 0) {
117  mNBinDpt = nz;
118  dataSize = mNBinCol * mNBinRow * mNBinDpt;
119  mData.reserve(dataSize); // reserve space for data
120  } else if (nz != mNBinDpt) {
121  LOG(fatal) << "Mismatch in Nz slices of bin X(col): " << ix << " Y(row): " << iy
122  << " wrt bin 0,0. File " << inpfname;
123  }
124 
125  // load data
126  for (int iz = 0; iz < nz; iz++) {
127  AlpideRespSimMat mat;
128 
129  std::array<float, AlpideRespSimMat::MatSize>* arr = mat.getArray();
130  for (int ip = 0; ip < npix * npix; ip++) {
131  inpGrid >> val;
132  (*arr)[ip] = val;
133  cnt++;
134  }
135  inpGrid >> lost >> dead >> untrck >> nele >> gx >> gy >> gz;
136 
137  if (inpGrid.bad()) {
138  LOG(fatal) << "Failed reading data for depth(Z) slice " << iz << " from "
139  << inpfname;
140  }
141  if (!nele) {
142  LOG(fatal) << "Wrong normalization Nele=" << nele << "for depth(Z) slice "
143  << iz << " from " << inpfname;
144  }
145 
146  if (mDptMax < -1e9) {
147  mDptMax = gz;
148  }
149  if (mDptMin > gz) {
150  mDptMin = gz;
151  }
152 
153  // normalize
154  float norm = 1.f / nele;
155  for (int ip = npix * npix; ip--;) {
156  (*arr)[ip] *= norm;
157  }
158  mData.push_back(mat); // store in the final container
159  } // loop over z
160 
161  inpGrid.close();
162 
163  } // loop over y
164  } // loop over x
165 
166  // final check
167  if (dataSize != mData.size()) {
168  LOG(fatal) << "Mismatch between expected " << dataSize << " and loaded " << mData.size()
169  << " number of bins";
170  }
171 
172  // normalize Dpt boundaries
173 
174  mStepInvCol /= micron2cm;
175  mStepInvRow /= micron2cm;
176 
177  mDptMin *= micron2cm;
178  mDptMax *= micron2cm;
179  mStepInvDpt = (mNBinDpt - 1) / (mDptMax - mDptMin);
180  mDptMin -= 0.5 / mStepInvDpt;
181  mDptMax += 0.5 / mStepInvDpt;
182  mDptShift = 0.5 * (mDptMax + mDptMin);
183  print();
184 }

200 string AlpideSimResponse::composeDataName(int colBin, int rowBin)
201 {
202  /*
203  * compose the file-name to read data for bin colBin,rowBin
204  */
205 
206  // ugly but safe way to compose the file name
207  float vcol = colBin / mStepInvCol, vrow = rowBin / mStepInvRow;
208  size_t size = snprintf(nullptr, 0, mColRowDataFmt.data(), vcol, vrow) + 1;
209  unique_ptr<char[]> tmp(new char[size]);
210  snprintf(tmp.get(), size, mColRowDataFmt.data(), vcol, vrow);
211  return mDataPath + string(tmp.get(), tmp.get() + size - 1);
212 }

215 bool AlpideSimResponse::getResponse(float vRow, float vCol, float vDepth, AlpideRespSimMat& dest) const
216 {
217  /*
218  * get linearized NPix*NPix matrix for response at point vRow(sensor local X, along row)
219  * vCol(sensor local Z, along columns) and vDepth (sensor local Y, i.e. depth)
220  */
221  if (!mNBinDpt) {
222  LOG(fatal) << "response object is not initialized";
223  }
224  bool flipCol = false, flipRow = true;
225  if (vDepth < mDptMin || vDepth > mDptMax) {
226  return false;
227  }
228  if (vCol < 0) {
229  vCol = -vCol;
230  flipCol = true;
231  }
232  if (vCol > mColMax) {
233  return false;
234  }
235  if (vRow < 0) {
236  vRow = -vRow;
237  flipRow = false;
238  }
239  if (vRow > mRowMax) {
240  return false;
241  }
242 
243  size_t bin = getDepthBin(vDepth) + mNBinDpt * (getRowBin(vRow) + mNBinRow * getColBin(vCol));
244  if (bin >= mData.size()) {
245  // this should not happen
246  LOG(fatal) << "requested bin " << bin << "row/col/depth: " << getRowBin(vRow) << ":" << getColBin(vCol)
247  << ":" << getDepthBin(vDepth) << ")"
248  << ">= maxBin " << mData.size()
249  << " for X(row)=" << vRow << " Z(col)=" << vCol << " Y(depth)=" << vDepth;
250  }
251  // printf("bin %d %d %d\n",getColBin(vCol),getRowBin(vRow),getDepthBin(vDepth));
252  // return &mData[bin];
253  dest.adopt(mData[bin], flipRow, flipCol);
254  return true;
255 }

264 const AlpideRespSimMat* AlpideSimResponse::getResponse(float vRow, float vCol, float vDepth, bool& flipRow, bool& flipCol, float rowMax, float colMax) const
265 {
266  /*
267  * get linearized NPix*NPix matrix for response at point vRow(sensor local X, along row)
268  * vCol(sensor local Z, along columns) and vDepth (sensor local Y, i.e. depth)
269  */
270  if (!mNBinDpt) {
271  LOG(fatal) << "response object is not initialized";
272  }
273  if (vDepth < mDptMin || vDepth > mDptMax) {
274  return nullptr;
275  }
276  if (vCol < 0) {
277  vCol = -vCol;
278  flipCol = true;
279  } else {
280  flipCol = false;
281  }
282  if (vCol > colMax) {
283  return nullptr;
284  }
285  if (vRow < 0) {
286  vRow = -vRow;
287  flipRow = false;
288  } else {
289  flipRow = true;
290  }
291  if (vRow > rowMax) {
292  return nullptr;
293  }
294 
295  size_t bin = getDepthBin(vDepth) + mNBinDpt * (getRowBin(vRow) + mNBinRow * getColBin(vCol));
296  if (bin >= mData.size()) {
297  // this should not happen
298  LOG(fatal) << "requested bin " << bin << "row/col/depth: " << getRowBin(vRow) << ":" << getColBin(vCol)
299  << ":" << getDepthBin(vDepth) << ")"
300  << ">= maxBin " << mData.size()
301  << " for X(row)=" << vRow << " Z(col)=" << vCol << " Y(depth)=" << vDepth;
302  }
303  return &mData[bin];
304 }
