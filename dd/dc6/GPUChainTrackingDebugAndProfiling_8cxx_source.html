40 int GPUChainTracking::DoProfile()
41 {
42 #ifdef GPUCA_TRACKLET_CONSTRUCTOR_DO_PROFILE
43  std::unique_ptr<char[]> stageAtSync{new char[PROFILE_MAX_SIZE]};
44  mRec->GPUMemCpy(stageAtSync.get(), processorsShadow()->tpcTrackers[0].mStageAtSync, PROFILE_MAX_SIZE, -1, false);
45 
46  FILE* fp = fopen("profile.txt", "w+");
47  FILE* fp2 = fopen("profile.bmp", "w+b");
48 
49  const int bmpheight = 8192;
50  BITMAPFILEHEADER bmpFH;
51  BITMAPINFOHEADER bmpIH;
52  memset(&bmpFH, 0, sizeof(bmpFH));
53  memset(&bmpIH, 0, sizeof(bmpIH));
54 
55  bmpFH.bfType = 19778; //"BM"
56  bmpFH.bfSize = sizeof(bmpFH) + sizeof(bmpIH) + (ConstructorBlockCount() * ConstructorThreadCount() / 32 * 33 - 1) * bmpheight;
57  bmpFH.bfOffBits = sizeof(bmpFH) + sizeof(bmpIH);
58 
59  bmpIH.biSize = sizeof(bmpIH);
60  bmpIH.biWidth = ConstructorBlockCount() * ConstructorThreadCount() / 32 * 33 - 1;
61  bmpIH.biHeight = bmpheight;
62  bmpIH.biPlanes = 1;
63  bmpIH.biBitCount = 32;
64 
65  fwrite(&bmpFH, 1, sizeof(bmpFH), fp2);
66  fwrite(&bmpIH, 1, sizeof(bmpIH), fp2);
67 
68  int nEmptySync = 0;
69  for (unsigned int i = 0; i < bmpheight * ConstructorBlockCount() * ConstructorThreadCount(); i += ConstructorBlockCount() * ConstructorThreadCount()) {
70  int fEmpty = 1;
71  for (unsigned int j = 0; j < ConstructorBlockCount() * ConstructorThreadCount(); j++) {
72  fprintf(fp, "%d\t", stageAtSync[i + j]);
73  int color = 0;
74  if (stageAtSync[i + j] == 1) {
75  color = RGB(255, 0, 0);
76  }
77  if (stageAtSync[i + j] == 2) {
78  color = RGB(0, 255, 0);
79  }
80  if (stageAtSync[i + j] == 3) {
81  color = RGB(0, 0, 255);
82  }
83  if (stageAtSync[i + j] == 4) {
84  color = RGB(255, 255, 0);
85  }
86  fwrite(&color, 1, sizeof(int), fp2);
87  if (j > 0 && j % 32 == 0) {
88  color = RGB(255, 255, 255);
89  fwrite(&color, 1, 4, fp2);
90  }
91  if (stageAtSync[i + j]) {
92  fEmpty = 0;
93  }
94  }
95  fprintf(fp, "\n");
96  if (fEmpty) {
97  nEmptySync++;
98  } else {
99  nEmptySync = 0;
100  }
101  (void)nEmptySync;
102  // if (nEmptySync == GPUCA_SCHED_ROW_STEP + 2) break;
103  }
104 
105  fclose(fp);
106  fclose(fp2);
107 #endif
108  return 0;
109 }

205 void GPUChainTracking::PrintOutputStat()
206 {
207  int nTracks = 0, nAttachedClusters = 0, nAttachedClustersFitted = 0, nAdjacentClusters = 0;
208  unsigned int nCls = GetProcessingSettings().doublePipeline ? mIOPtrs.clustersNative->nClustersTotal : GetTPCMerger().NMaxClusters();
209  if (ProcessingSettings().createO2Output > 1) {
210  nTracks = mIOPtrs.nOutputTracksTPCO2;
211  nAttachedClusters = mIOPtrs.nMergedTrackHits;
212  } else {
213  for (unsigned int k = 0; k < mIOPtrs.nMergedTracks; k++) {
214  if (mIOPtrs.mergedTracks[k].OK()) {
215  nTracks++;
216  nAttachedClusters += mIOPtrs.mergedTracks[k].NClusters();
217  nAttachedClustersFitted += mIOPtrs.mergedTracks[k].NClustersFitted();
218  }
219  }
220  for (unsigned int k = 0; k < nCls; k++) {
221  int attach = mIOPtrs.mergedTrackHitAttachment[k];
222  if (attach & gputpcgmmergertypes::attachFlagMask) {
223  nAdjacentClusters++;
224  }
225  }
226  }
227 
228  char trdText[1024] = "";
229  if (GetRecoSteps() & GPUDataTypes::RecoStep::TRDTracking) {
230  int nTRDTracks = 0;
231  int nTRDTracklets = 0;
232  for (unsigned int k = 0; k < mIOPtrs.nTRDTracks; k++) {
233  if (mIOPtrs.trdTracksO2) {
234  auto& trk = mIOPtrs.trdTracksO2[k];
235  nTRDTracklets += trk.getNtracklets();
236  nTRDTracks += trk.getNtracklets() != 0;
237  } else {
238  auto& trk = mIOPtrs.trdTracks[k];
239  nTRDTracklets += trk.getNtracklets();
240  nTRDTracks += trk.getNtracklets() != 0;
241  }
242  }
243  snprintf(trdText, 1024, " - TRD Tracker reconstructed %d tracks (%d tracklets)", nTRDTracks, nTRDTracklets);
244  }
245  printf("Output Tracks: %d (%d / %d / %d / %d clusters (fitted / attached / adjacent / total))%s\n", nTracks, nAttachedClustersFitted, nAttachedClusters, nAdjacentClusters, nCls, trdText);
246 }
