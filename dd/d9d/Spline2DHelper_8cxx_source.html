71 template <typename DataT>
72 void Spline2DHelper<DataT>::approximateFunctionBatch(
73  DataT* Fparameters, double x1Min, double x1Max, double x2Min, double x2Max,
74  std::function<void(const std::vector<double>& x1, const std::vector<double>& x2, std::vector<double> f[/*mFdimensions*/])> F,
75  unsigned int batchsize) const

80 
81  std::vector<double> dataPointF(getNumberOfDataPoints() * mFdimensions);

83  double scaleX1 = (x1Max - x1Min) / ((double)mHelperU1.getSpline().getUmax());
84  double scaleX2 = (x2Max - x2Min) / ((double)mHelperU2.getSpline().getUmax());
85 
86  std::vector<double> x1;
87  x1.reserve(batchsize);
88 
89  std::vector<double> x2;
90  x2.reserve(batchsize);
91 
92  std::vector<int> index;
93  index.reserve(batchsize);
94 
95  std::vector<double> dataPointFTmp[mFdimensions];
96  for (unsigned int iDim = 0; iDim < mFdimensions; ++iDim) {
97  dataPointFTmp[iDim].reserve(batchsize);
98  }
99 
100  unsigned int counter = 0;
101  for (int iv = 0; iv < getNumberOfDataPointsU2(); iv++) {
102  double x2Tmp = x2Min + mHelperU2.getDataPoint(iv).u * scaleX2;
103  for (int iu = 0; iu < getNumberOfDataPointsU1(); iu++) {
104  double x1Tmp = x1Min + mHelperU1.getDataPoint(iu).u * scaleX1;
105  x1.emplace_back(x1Tmp);
106  x2.emplace_back(x2Tmp);
107  index.emplace_back((iv * getNumberOfDataPointsU1() + iu) * mFdimensions);
108  ++counter;
109 
110  if (counter == batchsize || (iu == (getNumberOfDataPointsU1() - 1) && (iv == (getNumberOfDataPointsU2() - 1)))) {
111  counter = 0;
112  F(x1, x2, dataPointFTmp);
113  unsigned int entries = index.size();
114 
115  for (unsigned int i = 0; i < entries; ++i) {
116  const unsigned int indexTmp = index[i];
117  for (unsigned int iDim = 0; iDim < mFdimensions; ++iDim) {
118  dataPointF[indexTmp + iDim] = dataPointFTmp[iDim][i];
119  }
120  }
121 
122  x1.clear();
123  x2.clear();
124  index.clear();
125  for (unsigned int iDim = 0; iDim < mFdimensions; ++iDim) {
126  dataPointFTmp[iDim].clear();
127  }
128  }
129  }
130  }
131  approximateFunction(Fparameters, dataPointF.data());
132 }

134 template <typename DataT>
135 void Spline2DHelper<DataT>::approximateFunction(
136  DataT* Fparameters, const double DataPointF[/*getNumberOfDataPoints() x nFdim*/]) const
137 {
139 
140  const int Ndim = mFdimensions;
141  const int Ndim2 = 2 * Ndim;
142  const int Ndim3 = 3 * Ndim;
143  const int Ndim4 = 4 * Ndim;
144 
145  int nDataPointsU = getNumberOfDataPointsU1();
146  int nDataPointsV = getNumberOfDataPointsU2();
147 
148  int nKnotsU = mHelperU1.getSpline().getNumberOfKnots();
149  int nKnotsV = mHelperU2.getSpline().getNumberOfKnots();
150 
151  std::unique_ptr<double[]> rotDataPointF(new double[nDataPointsU * nDataPointsV * Ndim]); // U DataPoints x V DataPoints : rotated DataPointF for one output dimension
152  std::unique_ptr<double[]> Dv(new double[nKnotsV * nDataPointsU * Ndim]); // V knots x U DataPoints
153 
154  std::unique_ptr<DataT[]> parU(new DataT[mHelperU1.getSpline().calcNumberOfParameters(Ndim)]);
155  std::unique_ptr<DataT[]> parV(new DataT[mHelperU2.getSpline().calcNumberOfParameters(Ndim)]);
156 
157  std::unique_ptr<double[]> parUdbl(new double[mHelperU1.getSpline().calcNumberOfParameters(Ndim)]);
158  std::unique_ptr<double[]> parVdbl(new double[mHelperU2.getSpline().calcNumberOfParameters(Ndim)]);
159 
160  // rotated data points (u,v)->(v,u)
161 
162  for (int ipu = 0; ipu < nDataPointsU; ipu++) {
163  for (int ipv = 0; ipv < nDataPointsV; ipv++) {
164  for (int dim = 0; dim < Ndim; dim++) {
165  rotDataPointF[Ndim * (ipu * nDataPointsV + ipv) + dim] = DataPointF[Ndim * (ipv * nDataPointsU + ipu) + dim];
166  }
167  }
168  }
169 
170  // get S and S'u at all the knots by interpolating along the U axis
171 
172  for (int iKnotV = 0; iKnotV < nKnotsV; ++iKnotV) {
173  int ipv = mHelperU2.getKnotDataPoint(iKnotV);
174  const double* DataPointFrow = &(DataPointF[Ndim * ipv * nDataPointsU]);
175  mHelperU1.approximateFunctionGradually(parU.get(), DataPointFrow);
176 
177  for (int i = 0; i < mHelperU1.getSpline().calcNumberOfParameters(Ndim); i++) {
178  parUdbl[i] = parU[i];
179  }
180  for (int iKnotU = 0; iKnotU < nKnotsU; ++iKnotU) {
181  DataT* knotPar = &Fparameters[Ndim4 * (iKnotV * nKnotsU + iKnotU)];
182  for (int dim = 0; dim < Ndim; ++dim) {
183  knotPar[dim] = parU[Ndim * (2 * iKnotU) + dim]; // store S for all the knots
184  knotPar[Ndim2 + dim] = parU[Ndim * (2 * iKnotU) + Ndim + dim]; // store S'u for all the knots //SG!!!
185  }
186  }
187 
188  // recalculate F values for all ipu DataPoints at V = ipv
189  for (int ipu = 0; ipu < nDataPointsU; ipu++) {
190  double splineF[Ndim];
191  double u = mHelperU1.getDataPoint(ipu).u;
192  mHelperU1.getSpline().interpolateU(Ndim, parUdbl.get(), u, splineF);
193  for (int dim = 0; dim < Ndim; dim++) {
194  rotDataPointF[(ipu * nDataPointsV + ipv) * Ndim + dim] = splineF[dim];
195  }
196  }
197  }
198 
199  // calculate S'v at all data points with V == V of a knot
200 
201  for (int ipu = 0; ipu < nDataPointsU; ipu++) {
202  const double* DataPointFcol = &(rotDataPointF[ipu * nDataPointsV * Ndim]);
203  mHelperU2.approximateFunctionGradually(parV.get(), DataPointFcol);
204  for (int iKnotV = 0; iKnotV < nKnotsV; iKnotV++) {
205  for (int dim = 0; dim < Ndim; dim++) {
206  double dv = parV[(iKnotV * 2 + 1) * Ndim + dim];
207  Dv[(iKnotV * nDataPointsU + ipu) * Ndim + dim] = dv;
208  }
209  }
210  }
211 
212  // fit S'v and S''_vu at all the knots
213 
214  for (int iKnotV = 0; iKnotV < nKnotsV; ++iKnotV) {
215  const double* Dvrow = &(Dv[iKnotV * nDataPointsU * Ndim]);
216  mHelperU1.approximateFunction(parU.get(), Dvrow);
217  for (int iKnotU = 0; iKnotU < nKnotsU; ++iKnotU) {
218  for (int dim = 0; dim < Ndim; ++dim) {
219  Fparameters[Ndim4 * (iKnotV * nKnotsU + iKnotU) + Ndim + dim] = parU[Ndim * 2 * iKnotU + dim]; // store S'v for all the knots
220  Fparameters[Ndim4 * (iKnotV * nKnotsU + iKnotU) + Ndim3 + dim] = parU[Ndim * 2 * iKnotU + Ndim + dim]; // store S''vu for all the knots
221  }
222  }
223  }
224 }

243  const double scale = TMath::Pi() / uMax;
244  double uu = u * scale;
245  double vv = v * scale;
246  double cosu[Fdegree + 1], sinu[Fdegree + 1], cosv[Fdegree + 1], sinv[Fdegree + 1];
247  double ui = 0, vi = 0;
248  for (int i = 0; i <= Fdegree; i++, ui += uu, vi += vv) {
249  GPUCommonMath::SinCosd(ui, sinu[i], cosu[i]);
250  GPUCommonMath::SinCosd(vi, sinv[i], cosv[i]);
251  }
252  for (int dim = 0; dim < Ndim; dim++) {
253  double f = 0; // Fcoeff[dim][0]/2;
254  for (int i = 1; i <= Fdegree; i++) {
255  for (int j = 1; j <= Fdegree; j++) {
256  double* c = &(Fcoeff[dim][4 * (i * Fdegree + j)]);
257  f += c[0] * cosu[i] * cosv[j];
258  f += c[1] * cosu[i] * sinv[j];
259  f += c[2] * sinu[i] * cosv[j];
260  f += c[3] * sinu[i] * sinv[j];
