156 inline FrequencyTable& FrequencyTable::addSamples(Source_IT begin, Source_IT end, bool extendTable)
157 {
158  if (begin != end) {
159  const auto& [minIter, maxIter] = std::minmax_element(begin, end);
160  addSamples(begin, end, *minIter, *maxIter, extendTable);
161  } else {
162  LOG(warning) << "Passed empty message to " << __func__; // RS this is ok for empty columns
163  }
164  return *this;
165 }

167 template <typename Source_IT, std::enable_if_t<internal::isIntegralIter_v<Source_IT>, bool>>
168 FrequencyTable& FrequencyTable::addSamples(Source_IT begin, Source_IT end, symbol_t min, symbol_t max, bool extendTable)
169 {
170  LOG(trace) << "start adding samples";
171  internal::RANSTimer t;
172  t.start();
173 
174  if (begin == end) {
175  LOG(warning) << "Passed empty message to " << __func__; // RS this is ok for empty columns
176  } else {
177  if (extendTable) {
178  this->resize(min, max);
179  // add new symbols
180  std::for_each(begin, end, [this](symbol_t symbol) { ++this->getSymbol(this->sampleCountingDecorator(symbol)); });
181  } else {
182  // add new symbols but all that are out of range are set to incompressible
183  std::for_each(begin, end, [this](symbol_t symbol) {
184  if (this->empty() || symbol < this->getMinSymbol() || symbol > this->getMaxSymbol()) {
185  ++this->mIncompressibleSymbolFrequency;
186  } else {
187  ++this->getSymbol(this->sampleCountingDecorator(symbol));
188  }
189  });
190  }
191  }
192 
193  t.stop();
194  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
195 
196 #if !defined(NDEBUG)
197  LOG(debug2) << *this;
198 #endif
199  LOG(trace) << "done adding samples";
200  return *this;
201 }

203 template <typename Freq_IT, std::enable_if_t<internal::isIntegralIter_v<Freq_IT>, bool>>
204 FrequencyTable& FrequencyTable::addFrequencies(Freq_IT begin, Freq_IT end, symbol_t min, bool extendTable)
205 {
206  LOG(trace) << "start adding frequencies";
207  internal::RANSTimer t;
208  t.start();
209 
210  auto thisHistogram = utils::HistogramView{mFrequencyTable.begin(), mFrequencyTable.end(), mOffset};
211  auto addedHistogram = utils::trim(utils::HistogramView{begin, end, min});
212  if (addedHistogram.empty()) {
213  LOG(warning) << "Passed empty FrequencyTable to " << __func__; // RS this is ok for empty columns
214  } else {
215 
216  const symbol_t newMin = std::min(thisHistogram.getMin(), addedHistogram.getMin());
217  const symbol_t newMax = std::max(thisHistogram.getMax(), addedHistogram.getMax());
218 
219  // resize table
220  const bool needsExtend = thisHistogram.empty() || (newMin < thisHistogram.getMin()) || (newMax > thisHistogram.getMax());
221 
222  if (needsExtend && extendTable) {
223 
224  if (thisHistogram.empty()) {
225  mFrequencyTable = histogram_t(addedHistogram.size());
226  std::transform(addedHistogram.begin(), addedHistogram.end(), mFrequencyTable.begin(), [this](count_t frequency) {
227  return this->frequencyCountingDecorator(frequency);
228  });
229  mOffset = addedHistogram.getOffset();
230  } else {
231  const symbol_t newSize = newMax - newMin + 1;
232  histogram_t newFreequencyTable(newSize, 0);
233  auto newHistogram = utils::HistogramView{newFreequencyTable.begin(), newFreequencyTable.end(), newMin};
234  auto histogramOverlap = utils::intersection(newHistogram, thisHistogram);
235  assert(!histogramOverlap.empty());
236  assert(histogramOverlap.size() == thisHistogram.size());
237  std::copy(thisHistogram.begin(), thisHistogram.end(), histogramOverlap.begin());
238 
239  histogramOverlap = utils::intersection(newHistogram, addedHistogram);
240  assert(!histogramOverlap.empty());
241  assert(histogramOverlap.size() == addedHistogram.size());
242  std::transform(addedHistogram.begin(), addedHistogram.end(), histogramOverlap.begin(), histogramOverlap.begin(), [this](const count_t& a, const count_t& b) { return this->frequencyCountingDecorator(a) + b; });
243 
244  this->mFrequencyTable = std::move(newFreequencyTable);
245  this->mOffset = newHistogram.getOffset();
246  }
247  } else {
248  thisHistogram = utils::HistogramView{mFrequencyTable.begin(), mFrequencyTable.end(), mOffset};
249 
250  // left incompressible tail
251  auto leftTail = utils::leftTail(addedHistogram, thisHistogram);
252  if (!leftTail.empty()) {
253  mIncompressibleSymbolFrequency += std::accumulate(leftTail.begin(), leftTail.end(), 0);
254  }
255 
256  // intersection
257  auto overlapAdded = utils::intersection(addedHistogram, thisHistogram);
258  auto overlapThis = utils::intersection(thisHistogram, addedHistogram);
259  if (!overlapAdded.empty()) {
260  assert(overlapAdded.getMin() == overlapThis.getMin());
261  assert(overlapAdded.size() == overlapThis.size());
262  std::transform(overlapAdded.begin(), overlapAdded.end(), overlapThis.begin(), overlapThis.begin(), [this](const count_t& a, const count_t& b) { return this->frequencyCountingDecorator(a) + b; });
263  }
264 
265  // right incompressible tail
266  auto rightTail = utils::rightTail(addedHistogram, thisHistogram);
267  if (!rightTail.empty()) {
268  mIncompressibleSymbolFrequency += std::accumulate(rightTail.begin(), rightTail.end(), 0);
269  }
270  }
271  }
272  t.stop();
273  LOG(debug1) << __func__ << " inclusive time (ms): " << t.getDurationMS();
274 
275 #if !defined(NDEBUG)
276  LOG(debug2) << *this;
277 #endif
278 
279  LOG(trace) << "done adding frequencies";
280  return *this;
281 }

333 template <typename IT>
334 double_t computeEntropy(IT begin, IT end, symbol_t min)
335 {
336  double_t numSamples = std::accumulate(begin, end, 0);
337  return std::accumulate(begin, end, 0, [numSamples](double_t entropy, count_t frequency) {
338  const double_t p = static_cast<double_t>(frequency) / static_cast<double_t>(numSamples);
339  const double_t length = p == 0 ? 0 : std::log2(p);
340  return entropy -= p * length;
341  });
342 };
