127 template <typename T>
128 inline const Float_t Tracker<T>::getDistanceToSeed(const Cluster& cluster1, const Cluster& cluster2, const Cluster& cluster) const
129 {
130  // the seed is between "cluster1" and "cluster2" and cuts the plane
131  // of the "cluster" at a distance dR from it
132  Float_t dxSeed, dySeed, dzSeed, invdzSeed, dz, dR2, xSeed, ySeed;
133  dxSeed = cluster2.getX() - cluster1.getX();
134  dySeed = cluster2.getY() - cluster1.getY();
135  dzSeed = cluster2.getZ() - cluster1.getZ();
136  dz = cluster.getZ() - cluster1.getZ();
137  invdzSeed = dz / dzSeed;
138  xSeed = cluster1.getX() + dxSeed * invdzSeed;
139  ySeed = cluster1.getY() + dySeed * invdzSeed;
140  dR2 = (cluster.getX() - xSeed) * (cluster.getX() - xSeed) + (cluster.getY() - ySeed) * (cluster.getY() - ySeed);
141  return dR2;
142 }

154 template <typename T>
155 inline const Float_t Tracker<T>::getCellDeviation(const Cell& cell1, const Cell& cell2) const
156 {
157  Float_t cell1dx = cell1.getX2() - cell1.getX1();
158  Float_t cell1dy = cell1.getY2() - cell1.getY1();
159  Float_t cell1dz = cell1.getZ2() - cell1.getZ1();
160 
161  Float_t cell2dx = cell2.getX2() - cell2.getX1();
162  Float_t cell2dy = cell2.getY2() - cell2.getY1();
163  Float_t cell2dz = cell2.getZ2() - cell2.getZ1();
164 
165  Float_t cell1mod = std::sqrt(cell1dx * cell1dx + cell1dy * cell1dy + cell1dz * cell1dz);
166  Float_t cell2mod = std::sqrt(cell2dx * cell2dx + cell2dy * cell2dy + cell2dz * cell2dz);
167 
168  Float_t cosAngle = (cell1dx * cell2dx + cell1dy * cell2dy + cell1dz * cell2dz) / (cell1mod * cell2mod);
169 
170  return std::acos(cosAngle);
171 }

174 template <typename T>
175 inline const Bool_t Tracker<T>::getCellsConnect(const Cell& cell1, const Cell& cell2) const
176 {
177  Float_t cell1x2 = cell1.getX2();
178  Float_t cell1y2 = cell1.getY2();
179  Float_t cell2x1 = cell2.getX1();
180  Float_t cell2y1 = cell2.getY1();
181  Float_t dx = cell1x2 - cell2x1;
182  Float_t dy = cell1y2 - cell2y1;
183  Float_t dr2 = dx * dx + dy * dy;
184 
185  if (dr2 > (constants::mft::Resolution * constants::mft::Resolution)) {
186  return kFALSE;
187  }
188  return kTRUE;
189 }

192 template <typename T>
193 inline void Tracker<T>::computeTracksMClabels(const std::vector<T>& tracks)
194 {
196  for (auto& track : tracks) {
197  MCCompLabel maxOccurrencesValue{-1, -1, -1, false};
198  int count{0};
199  bool isFakeTrack{false};
200  auto nClusters = track.getNumberOfPoints();
201  for (int iCluster = 0; iCluster < nClusters; ++iCluster) {
202  const MCCompLabel& currentLabel = track.getMCCompLabels()[iCluster];
203  if (currentLabel == maxOccurrencesValue) {
204  ++count;
205  } else {
206  if (count != 0) { // only in the first iteration count can be 0 at this point
207  --count;
208  }
209  if (count == 0) {
210  maxOccurrencesValue = currentLabel;
211  count = 1;
212  }
213  }
214  }
215  count = 0;
216  for (int iCluster = 0; iCluster < nClusters; ++iCluster) {
217  if (track.getMCCompLabels()[iCluster] == maxOccurrencesValue) {
218  count++;
219  }
220  }
221 
222  auto labelratio = 1.0 * count / nClusters;
223  if (labelratio >= 0.8) {
224  } else {
225  isFakeTrack = true;
226  maxOccurrencesValue.setFakeFlag();
227  }
228  mTrackLabels.emplace_back(maxOccurrencesValue);
229  }
230 }
