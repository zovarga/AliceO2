31 std::shared_ptr<arrow::Table> ArrowHelpers::joinTables(std::vector<std::shared_ptr<arrow::Table>>&& tables)
32 {
33  if (tables.size() == 1) {
34  return tables[0];
35  }
36  for (auto i = 0u; i < tables.size() - 1; ++i) {
37  if (tables[i]->num_rows() != tables[i + 1]->num_rows()) {
38  throw o2::framework::runtime_error_f("Tables %s and %s have different sizes (%d vs %d) and cannot be joined!",
39  tables[i]->schema()->metadata()->Get("label").ValueOrDie().c_str(),
40  tables[i + 1]->schema()->metadata()->Get("label").ValueOrDie().c_str(),
41  tables[i]->num_rows(),
42  tables[i + 1]->num_rows());
43  }
44  }
45  std::vector<std::shared_ptr<arrow::Field>> fields;
46  std::vector<std::shared_ptr<arrow::ChunkedArray>> columns;
47 
48  for (auto& t : tables) {
49  auto tf = t->fields();
50  std::copy(tf.begin(), tf.end(), std::back_inserter(fields));
51  }
52 
53  auto schema = std::make_shared<arrow::Schema>(fields);
54 
55  if (tables[0]->num_rows() != 0) {
56  for (auto& t : tables) {
57  auto tc = t->columns();
58  std::copy(tc.begin(), tc.end(), std::back_inserter(columns));
59  }
60  }
61  return arrow::Table::Make(schema, columns);
62 }

64 std::shared_ptr<arrow::Table> ArrowHelpers::concatTables(std::vector<std::shared_ptr<arrow::Table>>&& tables)
65 {
66  if (tables.size() == 1) {
67  return tables[0];
68  }
69  std::vector<std::shared_ptr<arrow::ChunkedArray>> columns;
70  assert(tables.size() > 1);
71  std::vector<std::shared_ptr<arrow::Field>> resultFields = tables[0]->schema()->fields();
72  auto compareFields = [](std::shared_ptr<arrow::Field> const& f1, std::shared_ptr<arrow::Field> const& f2) {
73  // Let's do this with stable sorting.
74  return (!f1->Equals(f2)) && (f1->name() < f2->name());
75  };
76  for (size_t i = 1; i < tables.size(); ++i) {
77  auto& fields = tables[i]->schema()->fields();
78  std::vector<std::shared_ptr<arrow::Field>> intersection;
79 
80  std::set_intersection(resultFields.begin(), resultFields.end(),
81  fields.begin(), fields.end(),
82  std::back_inserter(intersection), compareFields);
83  resultFields.swap(intersection);
84  }
85 
86  for (auto& field : resultFields) {
87  arrow::ArrayVector chunks;
88  for (auto& table : tables) {
89  auto ci = table->schema()->GetFieldIndex(field->name());
90  if (ci == -1) {
91  throw std::runtime_error("Unable to find field " + field->name());
92  }
93  auto column = table->column(ci);
94  auto otherChunks = column->chunks();
95  chunks.insert(chunks.end(), otherChunks.begin(), otherChunks.end());
96  }
97  columns.push_back(std::make_shared<arrow::ChunkedArray>(chunks));
98  }
99 
100  auto result = arrow::Table::Make(std::make_shared<arrow::Schema>(resultFields), columns);
101  return result;
102 }
