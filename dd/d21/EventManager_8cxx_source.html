59 void EventManager::displayCurrentEvent()
60 {
61  if (getDataSource()->getEventCount() > 0) {
62  MultiView::getInstance()->destroyAllEvents();

64 
65  for (int i = 0; i < EVisualisationDataType::NdataTypes; ++i) {
66  dataTypeLists[i] = new TEveElementList(gDataTypeNames[i].c_str());

69  auto displayList = getDataSource()->getVisualisationList(no);

73

75  MultiView::getInstance()->registerElement(dataTypeLists[i]);
76  }
77  }
78  MultiView::getInstance()->redraw3D();

147 void EventManager::displayVisualisationEvent(VisualisationEvent& event, const std::string& detectorName)
148 {
149  double eta = 0.1;
150  size_t trackCount = event.getTrackCount();
151  LOG(info) << "displayVisualisationEvent: " << trackCount << " detector: " << detectorName;
152  // tracks
153  auto* list = new TEveTrackList(detectorName.c_str());
154  list->IncDenyDestroy();
155  // clusters
156  size_t clusterCount = 0;
157  auto* point_list = new TEvePointSet(detectorName.c_str());
158  point_list->IncDenyDestroy(); // don't delete if zero parent
159  point_list->SetMarkerColor(kBlue);
160 
161  for (size_t i = 0; i < trackCount; ++i) {
162  VisualisationTrack track = event.getTrack(i);
163  TEveRecTrackD t;
164  //double* p = track.getMomentum();
165  //t.fP = {p[0], p[1], p[2]};
166  t.fSign = track.getCharge() > 0 ? 1 : -1;
167  auto* vistrack = new TEveTrack(&t, &TEveTrackPropagator::fgDefault);
168  vistrack->SetLineColor(kMagenta);
169  //vistrack->SetName(detectorName + " track: " + i);
170  vistrack->SetName(track.getGIDAsString().c_str());
171  size_t pointCount = track.getPointCount();
172  vistrack->Reset(pointCount);
173 
174  int points = 0;
175  for (size_t j = 0; j < pointCount; ++j) {
176  auto point = track.getPoint(j);
177  if (point[2] > eta || point[2] < -1 * eta) {
178  vistrack->SetNextPoint(point[0], point[1], point[2]);
179  points++;
180  }
181  }
182  if (points > 0) {
183  list->AddElement(vistrack);
184  }
185 
186  // clusters connected with track
187  for (size_t i = 0; i < track.getClusterCount(); ++i) {
188  VisualisationCluster cluster = track.getCluster(i);
189  if (cluster.Z() > eta || cluster.Z() < -1 * eta) { // temporary remove eta=0 artefacts
190  point_list->SetNextPoint(cluster.X(), cluster.Y(), cluster.Z());
191  clusterCount++;
192  }
193  }
194  }
195 
196  if (trackCount != 0) {
197  dataTypeLists[EVisualisationDataType::Tracks]->AddElement(list);
198  }
199 
200  // global clusters (with no connection information)
201  for (size_t i = 0; i < event.getClusterCount(); ++i) {
202  VisualisationCluster cluster = event.getCluster(i);
203  if (cluster.Z() > eta || cluster.Z() < -1 * eta) { // temporary remove eta=0 artefacts
204  point_list->SetNextPoint(cluster.X(), cluster.Y(), cluster.Z());
205  clusterCount++;
206  }
207  }
208 
209  if (clusterCount != 0) {
210  dataTypeLists[EVisualisationDataType::Clusters]->AddElement(point_list);
211  }
212  LOG(info) << "tracks: " << trackCount << " detector: " << detectorName << ":" << dataTypeLists[EVisualisationDataType::Tracks]->NumChildren();
213  LOG(info) << "clusters: " << clusterCount << " detector: " << detectorName << ":" << dataTypeLists[EVisualisationDataType::Clusters]->NumChildren();
214 }
