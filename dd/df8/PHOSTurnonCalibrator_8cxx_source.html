85 void PHOSTurnonSlot::scanClusters(const gsl::span<const Cell>& cells, const TriggerRecord& celltr,
86  const gsl::span<const Cluster>& clusters, const TriggerRecord& clutr)
87 {
88  //First fill map of expected tiles from TRU cells
89  mNoisyTiles.reset();
90  int firstCellInEvent = celltr.getFirstEntry();
91  int lastCellInEvent = firstCellInEvent + celltr.getNumberOfObjects();
92  for (int i = firstCellInEvent; i < lastCellInEvent; i++) {
93  const Cell& c = cells[i];
94  if (c.getTRU()) {
95  mNoisyTiles.set(c.getTRUId());
96  }
97  }
98 
99  //Copy to have good and noisy map
100 
101  mFiredTiles.reset();
102  char mod;
103  float x, z;
104  short ddl;
105  int firstCluInEvent = clutr.getFirstEntry();
106  int lastCluInEvent = firstCluInEvent + clutr.getNumberOfObjects();
107  for (int i = firstCluInEvent; i < lastCluInEvent; i++) {
108  const Cluster& clu = clusters[i];
109  if (clu.getEnergy() < 1.e-4) {
110  continue;
111  }
112  mod = clu.module();
113  clu.getLocalPosition(x, z);
114  short truId = Geometry::relPosToTruId(mod, x, z, ddl);
115  //TODO!!! if(map.isGood2x2(truId)){
116  mTurnOnHistos->fillTotSp(ddl, clu.getEnergy());
117  if (clu.firedTrigger() & 1) { //Bit 1: 2x2, bit 2 4x4 //TODO: do we need separate 2x2 and 4x4 spectra? Switch?
118  mTurnOnHistos->fillFiredSp(ddl, clu.getEnergy());
119  //Fill trigger map
120  mFiredTiles.set(truId);
121  }
122  // }
123  }
124  //Fill final good and noisy maps
125  mTurnOnHistos->fillFiredMap(mFiredTiles);
126  mNoisyTiles ^= mFiredTiles;
127  mTurnOnHistos->fillNoisyMap(mFiredTiles);
128 }

131 void PHOSTurnonCalibrator::finalizeSlot(Slot& slot)
132 {
133  // Extract results for the single slot

140  //Add histos

142  mTurnOnHistos->merge(c->getCollectedHistos());
