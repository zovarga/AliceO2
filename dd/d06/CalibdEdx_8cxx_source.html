57 void CalibdEdx::fill(const TrackTPC& track)
58 {
59  // applying cuts
60  if (track.hasBothSidesClusters() || (mApplyCuts && !mCuts.goodTrack(track))) {
61  return;
62  }
63 
64  const auto& dEdx = track.getdEdx();
65  const auto sideOffset = track.hasASideClustersOnly() ? 0 : SECTORSPERSIDE;
66  const std::array<float, 4> dEdxMax{dEdx.dEdxMaxIROC, dEdx.dEdxMaxOROC1, dEdx.dEdxMaxOROC2, dEdx.dEdxMaxOROC3};
67  const std::array<float, 4> dEdxTot{dEdx.dEdxTotIROC, dEdx.dEdxTotOROC1, dEdx.dEdxTotOROC2, dEdx.dEdxTotOROC3};
68  // We need a copy of the track to perform propagations
69  auto cpTrack = track;
70 
71  for (const GEMstack stack : {IROCgem, OROC1gem, OROC2gem, OROC3gem}) {
72  // These are the x value in cm of the center of the stacks (IROC, OROC1, ...) in the local frame.
73  constexpr std::array<float, 4> xks{108.475f, 151.7f, 188.8f, 227.65f};
74 
75  bool ok = cpTrack.propagateTo(xks[stack], mField);
76 
77  // Ignore stack if we are not able to find its sector
78  if (!ok) {
79  continue;
80  }
81 
82  const float z = abs(cpTrack.getZ());
83  const float tgl = cpTrack.getTgl();
84  // const float snp = cpTrack.getSnp();
85  const float alpha = o2::math_utils::to02PiGen(cpTrack.getAlpha());
86  const auto sector = static_cast<int>(alpha / SECPHIWIDTH) + sideOffset;
87 
88  mHist(dEdxMax[stack] * mipScale, z, tgl, /* snp ,*/ sector, stack, ChargeType::Max);
89  mHist(dEdxTot[stack] * mipScale, z, tgl, /* snp ,*/ sector, stack, ChargeType::Tot);
90  }
91 }

107 template <typename Hist>
108 void fitHist(const Hist& hist, CalibdEdxCorrection& corr, TLinearFitter& fitter, const CalibdEdxCorrection* stackMean = nullptr)
109 {
110  using ax = CalibdEdx::Axis;
111 
112  // number of bins per stack
113  int stackBins = 1;
114  for (int i = 0; i < ax::Sector; ++i) {
115  stackBins *= hist.axis(i).size();
116  }
117 
118  const bool projSectors = stackMean != nullptr;
119 
120  constexpr int sectors = SECTORSPERSIDE * SIDES;
121  constexpr int stackCount = 144;
122  // number of fits to perform
123  const int fitCount = projSectors ? GEMSTACKSPERSECTOR * CHARGETYPES : stackCount * CHARGETYPES;
124  // number of GEM stacks per fit
125  const int fitStacks = projSectors ? sectors : 1;
126 
127  auto entry = bh::indexed(hist).begin();
128 
129  for (int fit = 0; fit < fitCount; ++fit) {
130  StackID id{};
131  id.type = static_cast<GEMstack>(entry->bin(ax::Stack).center());
132  const auto charge = static_cast<ChargeType>(entry->bin(ax::Charge).center());
133 
134  for (int stack = 0; stack < fitStacks; ++stack) {
135  id.sector = static_cast<int>(entry->bin(ax::Sector).center());
136 
137  for (int bin = 0; bin < stackBins; ++bin, ++entry) {
138  const float counts = *entry;
139  if (counts == 0) {
140  continue;
141  }
142  std::array<double, 2> values{entry->bin(1).center(),
143  entry->bin(2).center()};
144 
145  double dEdx = entry->bin(ax::dEdx).center();
146  // scale fit using the stacks mean
147  if (stackMean != nullptr) {
148  dEdx /= stackMean->getCorrection(id, charge);
149  }
150 
151  // constexpr float dEdxResolution = 0.05;
152  // const double error = dEdx * dEdxResolution / sqrt(counts);
153  const double error = 1. / sqrt(counts);
154 
155  fitter.AddPoint(values.data(), dEdx, error);
156  }
157  }
158  fitter.Eval();
159 
160  constexpr auto paramSize = CalibdEdxCorrection::paramSize;
161  float params[paramSize] = {0};
162  for (int param = 0; param < fitter.GetNumberFreeParameters(); ++param) {
163  params[param] = fitter.GetParameter(param);
164  }
165 
166  // for projected hist, copy the fit to every sector
167  if (projSectors) {
168  for (int i = 0; i < sectors; ++i) {
169  id.sector = i;
170  const float mean = stackMean->getCorrection(id, charge);
171 
172  // rescale the params to get the true correction
173  float scaledParams[paramSize];
174  for (int i = 0; i < paramSize; ++i) {
175  scaledParams[i] = params[i] * mean;
176  }
177  corr.setParams(id, charge, scaledParams);
178  corr.setChi2(id, charge, fitter.GetChisquare());
179  }
180  } else {
181  corr.setParams(id, charge, params);
182  corr.setChi2(id, charge, fitter.GetChisquare());
183  }
184  fitter.ClearPoints();
185  }
186 }

188 void CalibdEdx::finalize()
189 {
190  const float entries = minStackEntries();
191  mCalib.clear();
192 
193  TLinearFitter fitter(2);
194 
195  // Choose the fit dimension based on the available statistics
196  if (entries >= mFitCuts[2]) {
197  fitter.SetFormula("1 ++ x ++ x*x ++ y ++ x*y ++ y*y");
198  mCalib.setDims(2);
199  } else if (entries >= mFitCuts[1]) {
200  fitter.SetFormula("1 ++ x ++ x*x");
201  mCalib.setDims(1);
202  } else {
203  fitter.SetFormula("1");
204  mCalib.setDims(0);
205  }
206  LOGP(info, "Fitting {}D dE/dx correction for GEM stacks", mCalib.getDims());
207 
208  // if entries bellow minimum threshold, integrate all sectors
209  if (mCalib.getDims() == 0 || entries >= mFitCuts[0]) {
210  fitHist(mHist, mCalib, fitter);
211  } else {
212  LOGP(info, "Integrating GEM stacks sectors in dE/dx correction due to low statistics");
213 
214  // get mean of each GEM stack
215  CalibdEdxCorrection meanCorr{};
216  meanCorr.setDims(0);
217  TLinearFitter meanFitter(0);
218  meanFitter.SetFormula("1");
219  fitHist(mHist, meanCorr, meanFitter);
220 
221  // get highier dimension corrections with projected sectors
222  fitHist(mHist, mCalib, fitter, &meanCorr);
223  }
224 }

226 float CalibdEdx::minStackEntries() const
227 {
228  // sum over the dEdx bins to get the number of entries per stack
229  auto projection = bh::algorithm::project(mHist, std::vector<int>{Axis::Sector, Axis::Stack});
230  auto dEdxCounts = bh::indexed(projection);
231  // find the stack with the least number of entries
232  auto min_it = std::min_element(dEdxCounts.begin(), dEdxCounts.end());
233  // the count is doubled since we sum qMax and qTot entries
234  return static_cast<float>(*min_it / 2);
235 }

242 TH2F CalibdEdx::getRootHist(const std::vector<int>& projected_axis) const
243 {
244  const float lower = mHist.axis(0).begin()->lower();
245  const float upper = mHist.axis(0).end()->lower();
246 
247  auto projectedHist = getHist(projected_axis);
248 
249  const int nBins = mHist.axis(Axis::dEdx).size();
250  const int nHists = projectedHist.size() / nBins;
251 
252  TH2F rootHist("hdEdxMIP", "MIP dEdx per GEM stack", nHists, 0, nHists, nBins, lower, upper);
253 
254  int stack = 0;
255  float last_center = -1;
256  // fill TH2
257  for (auto&& x : bh::indexed(projectedHist)) {
258  const auto y = x.bin(0).center(); // current bin interval along dEdx axis
259  const auto w = *x; // "dereference" to get the bin value
260  rootHist.Fill(stack, y, w);
261 
262  if (y < last_center) {
263  stack++;
264  }
265  last_center = y;
266  }
267 
268  return rootHist;
269 }
