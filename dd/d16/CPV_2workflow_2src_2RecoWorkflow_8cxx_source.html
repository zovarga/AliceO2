50  {"raw", InputType::Raw},

55  {"digits", OutputType::Digits},
56  {"clusters", OutputType::Clusters}};

58 o2::framework::WorkflowSpec getWorkflow(bool disableRootInp,
59  bool disableRootOut,
60  bool propagateMC,
61  bool askSTFDist,
62  std::string const& cfgInput,
63  std::string const& cfgOutput)
64 {
65  InputType inputType;
66 
67  try {
68  inputType = InputMap.at(cfgInput);
69  } catch (std::out_of_range&) {
70  throw std::invalid_argument(std::string("invalid input type: ") + cfgInput);
71  }
72  std::vector<OutputType> outputTypes;
73  try {
74  outputTypes = RangeTokenizer::tokenize<OutputType>(cfgOutput, [](std::string const& token) { return OutputMap.at(token); });
75  } catch (std::out_of_range&) {
76  throw std::invalid_argument(std::string("invalid output type: ") + cfgOutput);
77  }
78  auto isEnabled = [&outputTypes](OutputType type) {
79  return std::find(outputTypes.begin(), outputTypes.end(), type) != outputTypes.end();
80  };
81 
82  o2::framework::WorkflowSpec specs;
83 
84  // //Raw to ....
85  if (inputType == InputType::Raw) {
86  //no explicit raw reader
87 
88  if (isEnabled(OutputType::Digits)) {
89  specs.emplace_back(o2::cpv::reco_workflow::getRawToDigitConverterSpec(askSTFDist));
90  if (!disableRootOut) {
91  specs.emplace_back(o2::cpv::getDigitWriterSpec(false));
92  }
93  }
94  if (isEnabled(OutputType::Clusters)) {
95  // add clusterizer
96  specs.emplace_back(o2::cpv::reco_workflow::getRawToDigitConverterSpec(askSTFDist));
97  specs.emplace_back(o2::cpv::reco_workflow::getClusterizerSpec(false));
98  if (!disableRootOut) {
99  specs.emplace_back(o2::cpv::getClusterWriterSpec(false));
100  }
101  }
102  }
103 
104  // Digits to ....
105  if (inputType == InputType::Digits) {
106  if (!disableRootInp) {
107  specs.emplace_back(o2::cpv::getDigitsReaderSpec(propagateMC));
108  }
109 
110  if (isEnabled(OutputType::Clusters)) {
111  // add clusterizer
112  specs.emplace_back(o2::cpv::reco_workflow::getClusterizerSpec(propagateMC));
113  if (!disableRootOut) {
114  specs.emplace_back(o2::cpv::getClusterWriterSpec(propagateMC));
115  }
116  }
117  }
118 
119  return std::move(specs);
120 }
