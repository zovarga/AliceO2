31 void GRPLHCIFData::setBeamAZ(beamDirection beam)
32 {
33  // set both A and Z of the beam in direction 'beam'
34  if (beam == beamDirection::BeamC) {
35  auto atomicNum = mZtoA.find(getAtomicNumberB1());
36  if (atomicNum == mZtoA.end()) {
37  LOG(fatal) << "We don't know the Mass Number for Z = " << getAtomicNumberB1();
38  } else {
39  mBeamAZ[static_cast<int>(beam)] = (atomicNum->second << 16) + getAtomicNumberB1();
40  }
41  } else {
42  auto atomicNum = mZtoA.find(getAtomicNumberB2());
43  if (atomicNum == mZtoA.end()) {
44  LOG(fatal) << "We don't know the Mass Number for Z = " << getAtomicNumberB2();
45  } else {
46  mBeamAZ[static_cast<int>(beam)] = (atomicNum->second << 16) + getAtomicNumberB2();
47  }
48  }
49 }

61 float GRPLHCIFData::getSqrtS() const
62 {
63  // get center of mass energy
64  double e0 = getBeamEnergyPerNucleon(BeamC);
65  double e1 = getBeamEnergyPerNucleon(BeamA);
66  if (e0 <= MassProton || e1 <= MassProton) {
67  return 0.f;
68  }
69  double beta0 = 1. - MassProton * MassProton / (e0 * e0);
70  double beta1 = 1. - MassProton * MassProton / (e1 * e1);
71  beta0 = beta0 > 0 ? sqrt(beta0) : 0.;
72  beta1 = beta1 > 0 ? sqrt(beta1) : 0.;
73  double ss = 2. * (MassProton * MassProton + e0 * e1 * (1. + beta0 * beta1 * cos(getCrossingAngle())));
74  return ss > 0. ? sqrt(ss) : 0.;
75 }

79 void GRPLHCIFData::translateBucketsToBCNumbers(std::vector<int32_t>& bcNb, std::vector<int32_t>& buckets, int beam)
80 {
81  // to translate the vector of bucket numbers to BC numbers
82  for (auto i : buckets) {
83  bcNb.push_back(i = 0 ? 0 : (i / 10 + o2::constants::lhc::BunchOffsetsP2[beam]) % o2::constants::lhc::LHCMaxBunches);
84  }
