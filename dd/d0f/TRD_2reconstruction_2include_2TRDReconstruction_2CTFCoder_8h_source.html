58 template <typename VEC>
59 void CTFCoder::encode(VEC& buff, const gsl::span<const TriggerRecord>& trigData, const gsl::span<const Tracklet64>& trkData, const gsl::span<const Digit>& digData)
60 {
61  using MD = o2::ctf::Metadata::OptStore;
62  // what to do which each field: see o2::ctd::Metadata explanation
63  constexpr MD optField[CTF::getNBlocks()] = {
64  MD::EENCODE, // BLC_bcIncTrig
65  MD::EENCODE, // BLC_orbitIncTrig
66  MD::EENCODE, // BLC_entriesTrk
67  MD::EENCODE, // BLC_entriesDig
68  MD::EENCODE, // BLC_HCIDTrk
69  MD::EENCODE, // BLC_padrowTrk
70  MD::EENCODE, // BLC_colTrk
71  MD::EENCODE, // BLC_posTrk
72  MD::EENCODE, // BLC_slopeTrk
73  MD::EENCODE, // BLC_pidTrk
74  MD::EENCODE, // BLC_CIDDig
75  MD::EENCODE, // BLC_ROBDig
76  MD::EENCODE, // BLC_MCMDig
77  MD::EENCODE, // BLC_chanDig
78  MD::EENCODE, // BLC_ADCDig
79  };
80 
81  CTFHelper helper(trigData, trkData, digData);
82 
83  // book output size with some margin
84  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
85  buff.resize(szIni);
86 
87  auto ec = CTF::create(buff);
88  using ECB = CTF::base;
89 
90  ec->setHeader(helper.createHeader());
91  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
92  ec->getANSHeader().majorVersion = 0;
93  ec->getANSHeader().minorVersion = 1;
94  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
95 #define ENCODETRD(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
96  // clang-format off
97  ENCODETRD(helper.begin_bcIncTrig(), helper.end_bcIncTrig(), CTF::BLC_bcIncTrig, 0);
98  ENCODETRD(helper.begin_orbitIncTrig(), helper.end_orbitIncTrig(), CTF::BLC_orbitIncTrig, 0);
99  ENCODETRD(helper.begin_entriesTrk(), helper.end_entriesTrk(), CTF::BLC_entriesTrk, 0);
100  ENCODETRD(helper.begin_entriesDig(), helper.end_entriesDig(), CTF::BLC_entriesDig, 0);
101 
102  ENCODETRD(helper.begin_HCIDTrk(), helper.end_HCIDTrk(), CTF::BLC_HCIDTrk, 0);
103  ENCODETRD(helper.begin_padrowTrk(), helper.end_padrowTrk(), CTF::BLC_padrowTrk, 0);
104  ENCODETRD(helper.begin_colTrk(), helper.end_colTrk(), CTF::BLC_colTrk, 0);
105  ENCODETRD(helper.begin_posTrk(), helper.end_posTrk(), CTF::BLC_posTrk, 0);
106  ENCODETRD(helper.begin_slopeTrk(), helper.end_slopeTrk(), CTF::BLC_slopeTrk, 0);
107  ENCODETRD(helper.begin_pidTrk(), helper.end_pidTrk(), CTF::BLC_pidTrk, 0);
108 
109  ENCODETRD(helper.begin_CIDDig(), helper.end_CIDDig(), CTF::BLC_CIDDig, 0);
110  ENCODETRD(helper.begin_ROBDig(), helper.end_ROBDig(), CTF::BLC_ROBDig, 0);
111  ENCODETRD(helper.begin_MCMDig(), helper.end_MCMDig(), CTF::BLC_MCMDig, 0);
112  ENCODETRD(helper.begin_chanDig(), helper.end_chanDig(), CTF::BLC_chanDig, 0);
113  ENCODETRD(helper.begin_ADCDig(), helper.end_ADCDig(), CTF::BLC_ADCDig, 0);
114 
115  // clang-format on
116  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
117 }

120 template <typename VTRG, typename VTRK, typename VDIG>
121 void CTFCoder::decode(const CTF::base& ec, VTRG& trigVec, VTRK& trkVec, VDIG& digVec)
122 {
123  auto header = ec.getHeader();
124  checkDictVersion(static_cast<const o2::ctf::CTFDictHeader&>(header));
125  ec.print(getPrefix(), mVerbosity);
126  std::vector<uint16_t> bcInc, HCIDTrk, posTrk, CIDDig, ADCDig;
127  std::vector<uint32_t> orbitInc, entriesTrk, entriesDig, pidTrk;
128  std::vector<uint8_t> padrowTrk, colTrk, slopeTrk, ROBDig, MCMDig, chanDig;
129 
130 #define DECODETRD(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
131  // clang-format off
132  DECODETRD(bcInc, CTF::BLC_bcIncTrig);
133  DECODETRD(orbitInc, CTF::BLC_orbitIncTrig);
134  DECODETRD(entriesTrk, CTF::BLC_entriesTrk);
135  DECODETRD(entriesDig, CTF::BLC_entriesDig);
136 
137  DECODETRD(HCIDTrk, CTF::BLC_HCIDTrk);
138  DECODETRD(padrowTrk, CTF::BLC_padrowTrk);
139  DECODETRD(colTrk, CTF::BLC_colTrk);
140  DECODETRD(posTrk, CTF::BLC_posTrk);
141  DECODETRD(slopeTrk, CTF::BLC_slopeTrk);
142  DECODETRD(pidTrk, CTF::BLC_pidTrk);
143 
144  DECODETRD(CIDDig, CTF::BLC_CIDDig);
145  DECODETRD(ROBDig, CTF::BLC_ROBDig);
146  DECODETRD(MCMDig, CTF::BLC_MCMDig);
147  DECODETRD(chanDig, CTF::BLC_chanDig);
148  DECODETRD(ADCDig, CTF::BLC_ADCDig);
149  // clang-format on
150  //
151  trigVec.clear();
152  trkVec.clear();
153  digVec.clear();
154  trigVec.reserve(header.nTriggers);
155  trkVec.reserve(header.nTracklets);
156  digVec.reserve(header.nDigits);
157 
158  uint32_t trkCount = 0, digCount = 0, adcCount = 0;
159  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
160 
161  for (uint32_t itrig = 0; itrig < header.nTriggers; itrig++) {
162  // restore TrigRecord
163  if (orbitInc[itrig]) { // non-0 increment => new orbit
164  ir.bc = bcInc[itrig]; // bcInc has absolute meaning
165  ir.orbit += orbitInc[itrig];
166  } else {
167  ir.bc += bcInc[itrig];
168  }
169 
170  uint32_t firstEntryTrk = trkVec.size();
171  uint16_t hcid = 0;
172  for (uint32_t it = 0; it < entriesTrk[itrig]; it++) {
173  hcid += HCIDTrk[trkCount]; // 1st tracklet of trigger was encoded with abs HCID, then increments
174  trkVec.emplace_back(header.format, hcid, padrowTrk[trkCount], colTrk[trkCount], posTrk[trkCount], slopeTrk[trkCount], pidTrk[trkCount]);
175  trkCount++;
176  }
177 
178  uint32_t firstEntryDig = digVec.size();
179  int16_t cid = 0;
180  for (uint32_t id = 0; id < entriesDig[itrig]; id++) {
181  cid += CIDDig[digCount]; // 1st digit of trigger was encoded with abs CID, then increments
182  auto& dig = digVec.emplace_back(cid, ROBDig[digCount], MCMDig[digCount], chanDig[digCount]);
183  dig.setADC({&ADCDig[adcCount], constants::TIMEBINS});
184  digCount++;
185  adcCount += constants::TIMEBINS;
186  }
187 
188  trigVec.emplace_back(ir, firstEntryDig, entriesDig[itrig], firstEntryTrk, entriesTrk[itrig]);
189  }
190  assert(digCount == header.nDigits && trkCount == header.nTracklets && adcCount == (int)ADCDig.size());
191 }
