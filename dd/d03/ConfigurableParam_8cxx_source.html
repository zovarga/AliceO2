143 // getIntValue takes a string value which is supposed to be
144 // a legal enum value and tries to cast it to an int.
145 // If it succeeds, and if the int value is legal, it is returned.
146 // If it fails, and if it is a legal string enum value, we look up
147 // and return the equivalent int value. In any case, if it is not
148 // a legal value we return -1 to indicate this fact.
149 int EnumLegalValues::getIntValue(const std::string& value) const
150 {
151  try {
152  int val = boost::lexical_cast<int>(value);
153  if (isLegal(val)) {
154  return val;
155  }
156  } catch (const boost::bad_lexical_cast& e) {
157  if (isLegal(value)) {
158  for (auto& pair : vvalues) {
159  if (pair.first == value) {
160  return pair.second;
161  }
162  }
163  }
164  }
165 
166  return -1;
167 }

386 // Update the storage map of params from the given configuration file.
387 // It can be in JSON or INI format.
388 // If nonempty comma-separated paramsList is provided, only those params will
389 // be updated, absence of data for any of requested params will lead to fatal
390 // If unchangedOnly is true, then only those parameters whose provenance is kCODE will be updated
391 // (to allow prefernce of run-time settings)
392 void ConfigurableParam::updateFromFile(std::string const& configFile, std::string const& paramsList, bool unchangedOnly)
393 {
394  if (!sIsFullyInitialized) {
395  initialize();
396  }
397 
398  auto cfgfile = o2::utils::Str::trim_copy(configFile);
399 
400  if (cfgfile.length() == 0) {
401  return;
402  }
403 
404  boost::property_tree::ptree pt = readConfigFile(cfgfile);
405 
406  std::vector<std::pair<std::string, std::string>> keyValPairs;
407  auto request = o2::utils::Str::tokenize(paramsList, ',', true);
408  std::unordered_map<std::string, int> requestMap;
409  for (const auto& par : request) {
410  if (!par.empty()) {
411  requestMap[par] = 0;
412  }
413  }
414 
415  try {
416  for (auto& section : pt) {
417  std::string mainKey = section.first;
418  if (requestMap.size()) {
419  if (requestMap.find(mainKey) == requestMap.end()) {
420  continue; // if something was requested, ignore everything else
421  } else {
422  requestMap[mainKey] = 1;
423  }
424  }
425  for (auto& subKey : section.second) {
426  auto name = subKey.first;
427  auto value = subKey.second.get_value<std::string>();
428  std::string key = mainKey + "." + name;
429  if (!unchangedOnly || getProvenance(key) == kCODE) {
430  std::pair<std::string, std::string> pair = std::make_pair(key, o2::utils::Str::trim_copy(value));
431  keyValPairs.push_back(pair);
432  }
433  }
434  }
435  } catch (std::exception const& error) {
436  LOG(error) << "Error while updating params " << error.what();
437  } catch (...) {
438  LOG(error) << "Unknown while updating params ";
439  }
440 
441  // make sure all requested params were retrieved
442  for (const auto& req : requestMap) {
443  if (req.second == 0) {
444  throw std::runtime_error(fmt::format("Param {:s} was not found in {:s}", req.first, configFile));
445  }
446  }
447 
448  try {
449  setValues(keyValPairs);
450  } catch (std::exception const& error) {
451  LOG(error) << "Error while setting values " << error.what();
452  }
453 }

458 void ConfigurableParam::updateFromString(std::string const& configString)
459 {
460  if (!sIsFullyInitialized) {
461  initialize();
462  }
463 
464  auto cfgStr = o2::utils::Str::trim_copy(configString);
465  if (cfgStr.length() == 0) {
466  return;
467  }
468 
469  // Take a vector of strings with elements of form a=b, and
470  // return a vector of pairs with each pair of form <a, b>
471  auto toKeyValPairs = [](std::vector<std::string>& tokens) {
472  std::vector<std::pair<std::string, std::string>> pairs;
473 
474  for (auto& token : tokens) {
475  auto keyval = o2::utils::Str::tokenize(token, '=');
476  if (keyval.size() != 2) {
477  LOG(fatal) << "Illegal command-line key/value string: " << token;
478  continue;
479  }
480 
481  std::pair<std::string, std::string> pair = std::make_pair(keyval[0], o2::utils::Str::trim_copy(keyval[1]));
482  pairs.push_back(pair);
483  }
484 
485  return pairs;
486  };
487 
488  // Simple check that the string starts/ends with an open square bracket
489  // Find the maximum index of a given key with array value.
490  // We store string keys for arrays as a[0]...a[size_of_array]
491  /*
492  auto maxIndex = [](std::string baseName) {
493  bool isFound = true;
494  int index = -1;
495  do {
496  index++;
497  std::string key = baseName + "[" + std::to_string(index) + "]";
498  isFound = keyInTree(sPtree, key);
499  } while (isFound);
500 
501  return index;
502  };
503 */
504 
505  // ---- end of helper functions --------------------
506 
507  // Command-line string is a ;-separated list of key=value params
508  auto params = o2::utils::Str::tokenize(configString, ';', true);
509 
510  // Now split each key=value string into its std::pair<key, value> parts
511  auto keyValues = toKeyValPairs(params);
512 
513  setValues(keyValues);
514 
515  const auto& kv = o2::conf::KeyValParam::Instance();
516  if (getProvenance("keyval.input_dir") != kCODE) {
517  sInputDir = o2::utils::Str::concat_string(o2::utils::Str::rectifyDirectory(kv.input_dir));
518  }
519  if (getProvenance("keyval.output_dir") != kCODE) {
520  if (kv.output_dir == "/dev/null") {
521  sOutputDir = kv.output_dir;
522  } else {
523  sOutputDir = o2::utils::Str::concat_string(o2::utils::Str::rectifyDirectory(kv.output_dir));
524  }
525  }
526 }

528 // setValues takes a vector of pairs where each pair is a key and value
529 // to be set in the storage map
530 void ConfigurableParam::setValues(std::vector<std::pair<std::string, std::string>> const& keyValues)
531 {
532  auto isArray = [](std::string& el) {
533  return el.size() > 0 && (el.at(0) == '[') && (el.at(el.size() - 1) == ']');
534  };
535 
536  // Take a vector of param key/value pairs
537  // and update the storage map for each of them by calling setValue.
538  // 1. For string/scalar types this is simple.
539  // 2. For array values we need to iterate over each array element
540  // and call setValue on the element, using an appropriately constructed key.
541  // 3. For enum types we check for the existence of the key in the enum registry
542  // and also confirm that the value is in the list of legal values
543  for (auto& keyValue : keyValues) {
544  std::string key = keyValue.first;
545  std::string value = o2::utils::Str::trim_copy(keyValue.second);
546 
547  if (!keyInTree(sPtree, key)) {
548  LOG(fatal) << "Inexistant ConfigurableParam key: " << key;
549  }
550 
551  if (sEnumRegistry->contains(key)) {
552  setEnumValue(key, value);
553  } else if (isArray(value)) {
554  setArrayValue(key, value);
555  } else {
556  assert(sKeyToStorageMap->find(key) != sKeyToStorageMap->end());
557 
558  // If the value is given as a boolean true|false, change to 1|0 int equivalent
559  if (value == "true") {
560  value = "1";
561  } else if (value == "false") {
562  value = "0";
563  }
564 
565  // TODO: this will trap complex types like maps and structs.
566  // These need to be broken into their own cases, so that we only
567  // get strings and scalars here.
568  setValue(key, value);
569  }
570  }
571 }

576  // value.erase(0, 1).pop_back();
577  auto elems = o2::utils::Str::tokenize(value.substr(1, value.length() - 2), ',', true);
578 
579  // TODO:
580  // 1. Should not assume each array element is a scalar/string. We may need to recurse.
581  // 2. Should not assume each array element - even if not complex - is correctly written. Validate.
582  // 3. Validation should include finding same types as in provided defaults.
583  for (int i = 0; i < elems.size(); ++i) {
584  std::string indexKey = key + "[" + std::to_string(i) + "]";
585  setValue(indexKey, elems[i]);

628 ConfigurableParam::EParamUpdateStatus ConfigurableParam::updateThroughStorageMap(std::string mainkey, std::string subkey, std::type_info const& tinfo,
629  void* addr)
630 {
631  // check if key_exists
632  auto key = mainkey + "." + subkey;
633  auto iter = sKeyToStorageMap->find(key);
634  if (iter == sKeyToStorageMap->end()) {
635  LOG(warn) << "Cannot update parameter " << key << " not found";
636  return ConfigurableParam::EParamUpdateStatus::Failed;
637  }
638 
639  // the type we need to convert to
640  int type = TDataType::GetType(tinfo);
641 
642  // check that type matches
643  if (iter->second.first != tinfo) {
644  LOG(warn) << "Types do not match; cannot update value";
645  return ConfigurableParam::EParamUpdateStatus::Failed;
646  }
647 
648  auto targetaddress = iter->second.second;
649  switch (type) {
650  case kChar_t: {
651  return Copy<char>(addr, targetaddress);
652  break;
653  }
654  case kUChar_t: {
655  return Copy<unsigned char>(addr, targetaddress);
656  break;
657  }
658  case kShort_t: {
659  return Copy<short>(addr, targetaddress);
660  break;
661  }
662  case kUShort_t: {
663  return Copy<unsigned short>(addr, targetaddress);
664  break;
665  }
666  case kInt_t: {
667  return Copy<int>(addr, targetaddress);
668  break;
669  }
670  case kUInt_t: {
671  return Copy<unsigned int>(addr, targetaddress);
672  break;
673  }
674  case kLong_t: {
675  return Copy<long>(addr, targetaddress);
676  break;
677  }
678  case kULong_t: {
679  return Copy<unsigned long>(addr, targetaddress);
680  break;
681  }
682  case kFloat_t: {
683  return Copy<float>(addr, targetaddress);
684  break;
685  }
686  case kDouble_t: {
687  return Copy<double>(addr, targetaddress);
688  break;
689  }
690  case kDouble32_t: {
691  return Copy<double>(addr, targetaddress);
692  break;
