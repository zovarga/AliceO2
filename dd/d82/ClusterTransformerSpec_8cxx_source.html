41 // convert all clusters from local to global reference frames
42 void local2global(geo::TransformationCreator transformation,
43  gsl::span<const Cluster> localClusters,
44  std::vector<Cluster, o2::pmr::polymorphic_allocator<Cluster>>& globalClusters)
45 {
46  int i{0};
47  globalClusters.insert(globalClusters.end(), localClusters.begin(), localClusters.end());
48  for (auto& c : localClusters) {
49  auto deId = c.getDEId();
50  o2::math_utils::Point3D<float> local{c.x, c.y, c.z};
51  auto t = transformation(deId);
52  auto global = t(local);
53  auto& gcluster = globalClusters[i];
54  gcluster.x = global.x();
55  gcluster.y = global.y();
56  gcluster.z = global.z();
57  i++;
58  }
59 }

61 class ClusterTransformerTask
62 {
63  public:
64  void init(InitContext& ic)
65  {
66  auto geoFile = ic.options().get<std::string>("geometry");
67  std::string ext = fs::path(geoFile).extension();
68  std::transform(ext.begin(), ext.begin(), ext.end(), [](unsigned char c) { return std::tolower(c); });
69 
70  if (ext == ".json") {
71  std::ifstream in(geoFile);
72  if (!in.is_open()) {
73  throw std::invalid_argument("cannot open geometry file" + geoFile);
74  }
75  transformation = o2::mch::geo::transformationFromJSON(in);
76  } else if (ext == ".root") {
77  o2::base::GeometryManager::loadGeometry(geoFile);
78  transformation = o2::mch::geo::transformationFromTGeoManager(*gGeoManager);
79  } else {
80  throw std::invalid_argument("Geometry can only be in JSON or Root format");
81  }
82  }
83 
84  // read the clusters (assumed to be in local reference frame) and
85  // tranform them into master reference frame.
86  void run(ProcessingContext& pc)
87  {
88  // get the input clusters
89  auto localClusters = pc.inputs().get<gsl::span<Cluster>>("clusters");
90 
91  // create the output message
92  auto& globalClusters = pc.outputs().make<std::vector<Cluster>>(OutputRef{"globalclusters"});
93 
94  local2global(transformation, localClusters, globalClusters);
95  }
96 
97  public:
98  o2::mch::geo::TransformationCreator transformation;
99 };
