42 RawErrorType_t RawDecoder::readChannels()
43 {
44  mChannelsInitialized = false;
45 
46  auto& payloadWords = mRawReader.getPayload();
47  uint32_t wordCountFromLastHeader = 1; //header word is included
48  int nDigitsAddedFromLastHeader = 0;
49  bool isHeaderExpected = true; //true if we expect to read header, false otherwise
50  bool skipUntilNextHeader = true; //true if something wrong with data format, try to read next header
51  uint16_t currentBC;
52  uint32_t currentOrbit = mRawReader.getCurrentHBFOrbit();
53  auto b = payloadWords.cbegin();
54  auto e = payloadWords.cend();
55  while (b != e) { //payload must start with cpvheader folowed by cpvwords and finished with cpvtrailer
56  CpvHeader header(b, e);
57  if (header.isOK()) {
58  LOG(debug) << "RawDecoder::readChannels() : "
59  << "I read cpv header for orbit = " << header.orbit()
60  << " and BC = " << header.bc();
61  if (!isHeaderExpected) { //actually, header was not expected
62  LOG(error) << "RawDecoder::readChannels() : "
63  << "header was not expected";
64  removeLastNDigits(nDigitsAddedFromLastHeader); //remove previously added digits as they are bad
65  mErrors.emplace_back(5, 0, 0, 0, kNO_CPVTRAILER);
66  }
67  skipUntilNextHeader = false;
68  currentBC = header.bc();
69  wordCountFromLastHeader = 0;
70  nDigitsAddedFromLastHeader = 0;
71  if (currentOrbit != header.orbit()) { //bad cpvheader
72  LOG(error) << "RawDecoder::readChannels() : "
73  << "currentOrbit(=" << currentOrbit
74  << ") != header.orbit()(=" << header.orbit() << ")";
75  mErrors.emplace_back(5, 0, 0, 0, kCPVHEADER_INVALID); //5 is non-existing link with general errors
76  skipUntilNextHeader = true;
77  }
78  } else {
79  if (skipUntilNextHeader) {
80  b += 16;
81  continue; //continue while'ing until it's not header
82  }
83  CpvWord word(b, e);
84  if (word.isOK()) {
85  wordCountFromLastHeader++;
86  for (int i = 0; i < 3; i++) {
87  PadWord pw = {word.cpvPadWord(i)};
88  if (pw.zero == 0) { //cpv pad word, not control or empty
89  if (addDigit(pw.mDataWord, word.ccId(), currentBC)) {
90  nDigitsAddedFromLastHeader++;
91  } else {
92  LOG(debug) << "RawDecoder::readChannels() : "
93  << "read pad word with non-valid pad address";
94  unsigned int dil = pw.dil, gas = pw.gas, address = pw.address;
95  mErrors.emplace_back(word.ccId(), dil, gas, address, kPadAddress);
96  }
97  }
98  }
99  } else { //this may be trailer
100  CpvTrailer trailer(b, e);
101  if (trailer.isOK()) {
102  int diffInCount = wordCountFromLastHeader - trailer.wordCounter();
103  if (diffInCount > 1 ||
104  diffInCount < -1) {
105  //some words lost?
106  LOG(error) << "RawDecoder::readChannels() : "
107  << "Read " << wordCountFromLastHeader << " words, expected " << trailer.wordCounter();
108  mErrors.emplace_back(5, 0, 0, 0, kCPVTRAILER_INVALID);
109  //throw all previous data and go to next header
110  removeLastNDigits(nDigitsAddedFromLastHeader);
111  skipUntilNextHeader = true;
112  }
113  if (trailer.bc() != currentBC) {
114  //trailer does not fit header
115  LOG(error) << "RawDecoder::readChannels() : "
116  << "CPVHeader BC is " << currentBC << " but CPVTrailer BC is " << trailer.bc();
117  mErrors.emplace_back(5, 0, 0, 0, kCPVTRAILER_INVALID);
118  removeLastNDigits(nDigitsAddedFromLastHeader);
119  skipUntilNextHeader = true;
120  }
121  isHeaderExpected = true;
122  } else {
123  wordCountFromLastHeader++;
124  //error
125  LOG(error) << "RawDecoder::readChannels() : "
126  << "Read unknown word";
127  mErrors.emplace_back(5, 0, 0, 0, kUNKNOWN_WORD); //add error for non-existing row
128  //what to do?
129  }
130  }
131  }
132  b += 16;
133  }
134  mChannelsInitialized = true;
135  return kOK;
136 }

138 bool RawDecoder::addDigit(uint32_t w, short ccId, uint16_t bc)
139 {
140  //add digit
141  PadWord pad = {w};
142  unsigned short absId;
143  if (!o2::cpv::Geometry::hwaddressToAbsId(ccId, pad.dil, pad.gas, pad.address, absId)) {
144  return false;
145  }
146 
147  //new bc -> add bc reference
148  if (mBCRecords.empty() || (mBCRecords.back().bc != bc)) {
149  mBCRecords.push_back(BCRecord(bc, mDigits.size(), mDigits.size()));
150  } else {
151  mBCRecords.back().lastDigit++;
152  }
153 
154  AddressCharge ac = {0};
155  ac.Address = absId;
156  ac.Charge = pad.charge;
157  mDigits.push_back(ac.mDataWord);
158 
159  return true;
160 }

162 void RawDecoder::removeLastNDigits(int n)
163 {
164  if (n < 0) {
165  return;
166  }
167  int nRemoved = 0;
168  while (nRemoved < n) {
169  if (mDigits.size() > 0) { // still has digits to remove
170  mDigits.pop_back();
171  if (mBCRecords.back().lastDigit == mBCRecords.back().firstDigit) {
172  mBCRecords.pop_back();
173  } else {
174  mBCRecords.back().lastDigit--;
175  }
176  nRemoved++;
177  } else { // has nothing to remove already
178  break;
179  }
180  }
