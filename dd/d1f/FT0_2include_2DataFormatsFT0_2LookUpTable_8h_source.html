85 inline o2::ft0::Topo read_Topo(std::string_view str)
86 {
87  assert(str.substr(0, 2) == "PM"); // && str[4] == '/' && str[5] == 'C' && str[6] == 'h');
88  char side = str[2];
89  char* ptr;
90  uint8_t pm_num = std::strtol(str.data() + 3, &ptr, 10); // = str[3] - '0';
91  /* auto res = std::from_chars(str.data()+3, str.data()+3+str.size(), pm_num); */
92  /* if (res.ec != std::errc() || res.ptr[0] != '/') */
93  if (errno || ptr[0] != '/') {
94  throw std::invalid_argument("Cannot read pm_num");
95  }
96  if (ptr[1] != 'C' || ptr[2] != 'h') {
97  throw std::invalid_argument("Expected 'Ch'");
98  }
99  uint8_t pm_ch = std::strtol(ptr + 3, &ptr, 10);
100  uint8_t ep = side == 'C' ? 1 : 0;
101  if (errno) {
102  throw std::invalid_argument("Cannot read pm_ch");
103  }
104  assert(side == 'A' || side == 'C');
105  return Topo{pm_num, pm_ch, ep};
106 }

183  static o2::ft0::LookUpTable readTableFile()
184  {
185  std::string inputDir;
186  const char* aliceO2env = std::getenv("O2_ROOT");
187  if (aliceO2env) {
188  inputDir = aliceO2env;
189  }
190  inputDir += "/share/Detectors/FT0/files/";
191 
192  std::string lutPath = inputDir + "FT0ChannelsTable.txt";
193  lutPath = gSystem->ExpandPathName(lutPath.data()); // Expand $(ALICE_ROOT) into real system path
194 
195  std::ifstream infile;
196  infile.open(lutPath.c_str());
197 
198  std::vector<o2::ft0::Topo> lut_data(NUMBER_OF_MCPs * NUMBER_OF_PMs - 8);
199  std::string comment; // dummy, used just to read 4 first lines and move the cursor to the 5th, otherwise unused
200  if (!getline(infile, comment)) { // first comment line
201  /* std::cout << "Error opening ascii file (it is probably a folder!): " << filename.c_str() << std::endl; */
202  throw std::runtime_error("Error reading lookup table");
203  }
204  int channel;
205  std::string pm, pm_channel, hv_board, hv_channel, mcp_sn, hv_cable, signal_cable;
206  std::getline(infile, pm); // skip one line
207  std::string line;
208  while (std::getline(infile, line) && std::istringstream(line) >> channel >> pm >> pm_channel >> hv_board >> hv_channel >> mcp_sn >> hv_cable >> signal_cable) {
209  lut_data[channel] = read_Topo(pm_channel);
210  }
211  return o2::ft0::LookUpTable{lut_data};
212  }
213  static o2::ft0::LookUpTable readTable()
214  {
215 
216  std::vector<o2::ft0::Topo> lut_data;
217  auto& mgr = o2::ccdb::BasicCCDBManager::instance();
218  mgr.setURL(o2::base::NameConf::getCCDBServer());
219  auto hvch = mgr.get<std::vector<o2::ft0::HVchannel>>("FT0/LookUpTable");
220  size_t max = 0;
221  for (auto const& chan : *hvch) {
222  if (max < chan.channel) {
223  max = chan.channel;
224  }
225  }
226  lut_data.resize(max + 1);
227  for (auto const& chan : *hvch) {
228  o2::ft0::Topo topo = chan.pm;
229  lut_data[chan.channel] = topo;
230  }
231  LOG(info) << "lut_data.size " << lut_data.size();
232  return o2::ft0::LookUpTable{lut_data};
233  }
