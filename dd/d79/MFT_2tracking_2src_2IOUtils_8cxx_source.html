40 int ioutils::loadROFrameData(const o2::itsmft::ROFRecord& rof, ROframe<T>& event, gsl::span<const itsmft::CompClusterExt> clusters, gsl::span<const unsigned char>::iterator& pattIt, const itsmft::TopologyDictionary& dict, const dataformats::MCTruthContainer<MCCompLabel>* mcLabels, const o2::mft::Tracker<T>* tracker)
41 {
42  event.clear();
43  GeometryTGeo* geom = GeometryTGeo::Instance();
44  geom->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L, o2::math_utils::TransformType::L2G));
45  int clusterId{0};
46  auto first = rof.getFirstEntry();
47  auto clusters_in_frame = rof.getROFData(clusters);
48  for (auto& c : clusters_in_frame) {
49  auto sensorID = c.getSensorID();
50  int layer = geom->getLayer(sensorID);
51  auto pattID = c.getPatternID();
52  o2::math_utils::Point3D<float> locXYZ;
53  float sigmaX2 = ioutils::DefClusError2Row, sigmaY2 = ioutils::DefClusError2Col; //Dummy COG errors (about half pixel size)
54  if (pattID != itsmft::CompCluster::InvalidPatternID) {
55  sigmaX2 = dict.getErr2X(pattID); // ALPIDE local X coordinate => MFT global X coordinate (ALPIDE rows)
56  sigmaY2 = dict.getErr2Z(pattID); // ALPIDE local Z coordinate => MFT global Y coordinate (ALPIDE columns)
57  if (!dict.isGroup(pattID)) {
58  locXYZ = dict.getClusterCoordinates(c);
59  } else {
60  o2::itsmft::ClusterPattern patt(pattIt);
61  locXYZ = dict.getClusterCoordinates(c, patt);
62  }
63  } else {
64  o2::itsmft::ClusterPattern patt(pattIt);
65  locXYZ = dict.getClusterCoordinates(c, patt, false);
66  }
67  // Transformation to the local --> global
68  auto gloXYZ = geom->getMatrixL2G(sensorID) * locXYZ;
69 
70  auto clsPoint2D = math_utils::Point2D<Float_t>(gloXYZ.x(), gloXYZ.y());
71  Float_t rCoord = clsPoint2D.R();
72  Float_t phiCoord = clsPoint2D.Phi();
73  o2::math_utils::bringTo02PiGen(phiCoord);
74  int rBinIndex = tracker->getRBinIndex(rCoord);
75  int phiBinIndex = tracker->getPhiBinIndex(phiCoord);
76  int binIndex = tracker->getBinIndex(rBinIndex, phiBinIndex);
77  event.addClusterToLayer(layer, gloXYZ.x(), gloXYZ.y(), gloXYZ.z(), phiCoord, rCoord, event.getClustersInLayer(layer).size(), binIndex, sigmaX2, sigmaY2, sensorID);
78  if (mcLabels) {
79  event.addClusterLabelToLayer(layer, *(mcLabels->getLabels(first + clusterId).begin()));
80  }
81  event.addClusterExternalIndexToLayer(layer, first + clusterId);
82  clusterId++;
83  }
84  return clusters_in_frame.size();
85 }
