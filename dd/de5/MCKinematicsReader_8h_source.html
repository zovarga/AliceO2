153 inline std::vector<MCTrack> const& MCKinematicsReader::getTracks(int source, int event) const
154 {
155  if (mTracks[source].size() == 0) {
156  initTracksForSource(source);
157  }
158  if (mTracks[source][event] == nullptr) {
159  loadTracksForSourceAndEvent(source, event);
160  }
161  return *mTracks[source][event];
162 }
