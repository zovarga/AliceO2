47 void AlignableSensor::dPosTraDParGeomLOC(const AlignmentPoint* pnt, double* deriv) const
48 {
49  // Jacobian of position in sensor tracking frame (tra) vs sensor LOCAL frame
50  // parameters in TGeoHMatrix convention.
51  // Result is stored in array deriv as linearized matrix 6x3
52  const double kDelta[kNDOFGeom] = {0.1, 0.1, 0.1, 0.5, 0.5, 0.5};
53  double delta[kNDOFGeom], pos0[3], pos1[3], pos2[3], pos3[3];
54  TGeoHMatrix matMod;
55  //
56  memset(delta, 0, kNDOFGeom * sizeof(double));
57  memset(deriv, 0, kNDOFGeom * 3 * sizeof(double));
58  const double* tra = pnt->getXYZTracking();
59  //
60  for (int ip = kNDOFGeom; ip--;) {
61  //
62  if (!isFreeDOF(ip)) {
63  continue;
64  }
65  //
66  double var = kDelta[ip];
67  delta[ip] -= var;
68  // variation matrix in tracking frame for variation in sensor LOCAL frame
69  getDeltaT2LmodLOC(matMod, delta);
70  matMod.LocalToMaster(tra, pos0); // varied position in tracking frame
71  //
72  delta[ip] += 0.5 * var;
73  getDeltaT2LmodLOC(matMod, delta);
74  matMod.LocalToMaster(tra, pos1); // varied position in tracking frame
75  //
76  delta[ip] += var;
77  getDeltaT2LmodLOC(matMod, delta);
78  matMod.LocalToMaster(tra, pos2); // varied position in tracking frame
79  //
80  delta[ip] += 0.5 * var;
81  getDeltaT2LmodLOC(matMod, delta);
82  matMod.LocalToMaster(tra, pos3); // varied position in tracking frame
83  //
84  delta[ip] = 0;
85  double* curd = deriv + ip * 3;
86  for (int i = 3; i--;) {
87  curd[i] = (8. * (pos2[i] - pos1[i]) - (pos3[i] - pos0[i])) / 6. / var;
88  }
89  }
90  //
91 }

191 void AlignableSensor::dPosTraDParGeomTRA(const AlignmentPoint* pnt, double* deriv, const AlignableVolume* parent) const
192 {
193  // Jacobian of position in sensor tracking frame (tra) vs sensor TRACKING
194  // frame parameters in TGeoHMatrix convention, i.e. the modified parameter is
195  // tra' = tau*tra
196  //
197  // Result is stored in array deriv as linearized matrix 6x3
198  const double kDelta[kNDOFGeom] = {0.1, 0.1, 0.1, 0.5, 0.5, 0.5};
199  double delta[kNDOFGeom], pos0[3], pos1[3], pos2[3], pos3[3];
200  TGeoHMatrix matMod;
201  //
202  // 1st we need a matrix for transition between child and parent TRACKING frames
203  // Let TRA,LOC are positions in tracking and local frame of parent, linked as LOC=T2L*TRA
204  // and tra,loc are positions in tracking and local frame of child, linked as loc=t2l*tra
205  // The loc and LOC are linked as LOC=R*loc, where R = L2G^-1*l2g, with L2G and l2g
206  // local2global matrices for parent and child
207  //
208  // Then, TRA = T2L^-1*LOC = T2L^-1*R*loc = T2L^-1*R*t2l*tra
209  // -> TRA = matRel*tra, with matRel = T2L^-1*L2G^-1 * l2g*t2l
210  // Note that l2g*t2l are tracking to global matrices
211  TGeoHMatrix matRel, t2gP;
212  getMatrixT2G(matRel); // t2g matrix of child
213  parent->getMatrixT2G(t2gP); // t2g matrix of parent
214  const TGeoHMatrix& t2gpi = t2gP.Inverse();
215  matRel.MultiplyLeft(&t2gpi);
216  //
217  memset(delta, 0, kNDOFGeom * sizeof(double));
218  memset(deriv, 0, kNDOFGeom * 3 * sizeof(double));
219  const double* tra = pnt->getXYZTracking();
220  //
221  for (int ip = kNDOFGeom; ip--;) {
222  //
223  if (!isFreeDOF(ip)) {
224  continue;
225  }
226  //
227  double var = kDelta[ip];
228  delta[ip] -= var;
229  getDeltaT2LmodTRA(matMod, delta, matRel);
230  matMod.LocalToMaster(tra, pos0); // varied position in tracking frame
231  //
232  delta[ip] += 0.5 * var;
233  getDeltaT2LmodTRA(matMod, delta, matRel);
234  matMod.LocalToMaster(tra, pos1); // varied position in tracking frame
235  //
236  delta[ip] += var;
237  getDeltaT2LmodTRA(matMod, delta, matRel);
238  matMod.LocalToMaster(tra, pos2); // varied position in tracking frame
239  //
240  delta[ip] += 0.5 * var;
241  getDeltaT2LmodTRA(matMod, delta, matRel);
242  matMod.LocalToMaster(tra, pos3); // varied position in tracking frame
243  //
244  delta[ip] = 0;
245  double* curd = deriv + ip * 3;
246  for (int i = 3; i--;) {
247  curd[i] = (8. * (pos2[i] - pos1[i]) - (pos3[i] - pos0[i])) / 6. / var;
248  }
249  }
250  //
251 }

254 void AlignableSensor::dPosTraDParGeom(const AlignmentPoint* pnt, double* deriv, const AlignableVolume* parent) const
255 {
256  // calculate point position derivatives in tracking frame of sensor
257  // vs standard geometrical DOFs of its parent volume (if parent!=0) or sensor itself
258  Frame_t frame = parent ? parent->getVarFrame() : getVarFrame();
259  switch (frame) {
260  case kLOC:
261  parent ? dPosTraDParGeomLOC(pnt, deriv, parent) : dPosTraDParGeomLOC(pnt, deriv);
262  break;
263  case kTRA:
264  parent ? dPosTraDParGeomTRA(pnt, deriv, parent) : dPosTraDParGeomTRA(pnt, deriv);
265  break;
266  default:
267  LOG(error) << "Alignment frame " << parent->getVarFrame() << " is not implemented";
268  break;
269  }
270 }

274 {

461 void AlignableSensor::applyAlignmentFromMPSol()
462 {
463  // apply to the tracking coordinates in the sensor frame the full chain
464  // of alignments found by MP for this sensor and its parents
465  //
466  const AlignableVolume* vol = this;
467  TGeoHMatrix deltaG;
468  // create global combined delta:
469  // DeltaG = deltaG_0*...*deltaG_j, where delta_i is global delta of each member of hierarchy
470  while (vol) {
471  TGeoHMatrix deltaGJ;
472  vol->createAlignmenMatrix(deltaGJ);
473  deltaG.MultiplyLeft(&deltaGJ);
474  vol = vol->getParent();
475  }
476  //
477  // update misaligned L2G matrix
478  deltaG *= getMatrixL2GIdeal();
479  setMatrixL2G(deltaG);
480  //
481  // update local misalignment matrix
482  prepareMatrixClAlg();
483  //
484 }
