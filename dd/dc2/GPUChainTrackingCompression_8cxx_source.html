107  break;
108  case 2:
109  runKernel<GPUTPCCompressionGatherKernels, GPUTPCCompressionGatherKernels::buffered64>(GetGridBlkStep(nBlocksDefault, outputStream, RecoStep::TPCCompression), krnlRunRangeNone, krnlEventNone);
110  getKernelTimer<GPUTPCCompressionGatherKernels, GPUTPCCompressionGatherKernels::buffered64>(RecoStep::TPCCompression, 0, outputSize);

126  if (ProcessingSettings().tpcCompressionGatherMode == 3) {
127  RecordMarker(&mEvents->stream[outputStream], outputStream);
128  char* deviceFlatPts = (char*)Compressor.mOutput->qTotU;
129  if (GetProcessingSettings().doublePipeline) {

132  for (unsigned int i = 0; i < n; i++) {

137  }
138  }
139  } else {
140  char direction = 0;
141  if (ProcessingSettings().tpcCompressionGatherMode == 0) {
142  P = &CompressorShadow.mPtrs;
143  } else if (ProcessingSettings().tpcCompressionGatherMode == 1) {
144  P = &Compressor.mPtrs;
145  direction = -1;
146  gatherTimer = &getTimer<GPUTPCCompressionKernels>("GPUTPCCompression_GatherOnCPU", 0);
147  gatherTimer->Start();
148  }

157  GPUMemCpyAlways(myStep, O->qMaxU + offset, P->qMaxU + srcOffset, O->nSliceRowClusters[i * GPUCA_ROW_COUNT + j] * sizeof(O->qMaxU[0]), outputStream, direction);
158  GPUMemCpyAlways(myStep, O->flagsU + offset, P->flagsU + srcOffset, O->nSliceRowClusters[i * GPUCA_ROW_COUNT + j] * sizeof(O->flagsU[0]), outputStream, direction);

168  GPUMemCpyAlways(myStep, O->qTotA + offset, P->qTotA + Compressor.mAttachedClusterFirstIndex[i], O->nTrackClusters[i] * sizeof(O->qTotA[0]), outputStream, direction);
169  GPUMemCpyAlways(myStep, O->qMaxA + offset, P->qMaxA + Compressor.mAttachedClusterFirstIndex[i], O->nTrackClusters[i] * sizeof(O->qMaxA[0]), outputStream, direction);

172  GPUMemCpyAlways(myStep, O->sigmaTimeA + offset, P->sigmaTimeA + Compressor.mAttachedClusterFirstIndex[i], O->nTrackClusters[i] * sizeof(O->sigmaTimeA[0]), outputStream, direction);

174  // First index stored with track
175  GPUMemCpyAlways(myStep, O->rowDiffA + offset - i, P->rowDiffA + Compressor.mAttachedClusterFirstIndex[i] + 1, (O->nTrackClusters[i] - 1) * sizeof(O->rowDiffA[0]), outputStream, direction);

178  GPUMemCpyAlways(myStep, O->timeResA + offset - i, P->timeResA + Compressor.mAttachedClusterFirstIndex[i] + 1, (O->nTrackClusters[i] - 1) * sizeof(O->timeResA[0]), outputStream, direction);

182  GPUMemCpyAlways(myStep, O->rowA, P->rowA, O->nTracks * sizeof(O->rowA[0]), outputStream, direction);
183  GPUMemCpyAlways(myStep, O->sliceA, P->sliceA, O->nTracks * sizeof(O->sliceA[0]), outputStream, direction);
184  GPUMemCpyAlways(myStep, O->timeA, P->timeA, O->nTracks * sizeof(O->timeA[0]), outputStream, direction);

188  gatherTimer->Stop();
189  }
190  mIOPtrs.tpcCompressedClusters = Compressor.mOutputFlat;
191  if (ProcessingSettings().tpcCompressionGatherMode == 3) {

193  mRec->ReturnVolatileDeviceMemory();

201  mRec->PopNonPersistentMemory(RecoStep::TPCCompression, qStr2Tag("TPCCOMPR"));

206 int GPUChainTracking::RunTPCDecompression()
207 {
208 #ifdef GPUCA_HAVE_O2HEADERS
209  const auto& threadContext = GetThreadContext();
210  TPCClusterDecompressor decomp;
211  auto allocator = [this](size_t size) {
212  this->mInputsHost->mNClusterNative = this->mInputsShadow->mNClusterNative = size;
213  this->AllocateRegisteredMemory(this->mInputsHost->mResourceClusterNativeOutput, this->mSubOutputControls[GPUTrackingOutputs::getIndex(&GPUTrackingOutputs::clustersNative)]);
214  return this->mInputsHost->mPclusterNativeOutput;
215  };
216  auto& gatherTimer = getTimer<TPCClusterDecompressor>("TPCDecompression", 0);
217  gatherTimer.Start();
218  if (decomp.decompress(mIOPtrs.tpcCompressedClusters, *mClusterNativeAccess, allocator, param())) {
219  GPUError("Error decompressing clusters");
220  return 1;
221  }
222  gatherTimer.Stop();
223  mIOPtrs.clustersNative = mClusterNativeAccess.get();
224  if (mRec->IsGPU()) {
225  AllocateRegisteredMemory(mInputsHost->mResourceClusterNativeBuffer);
226  processorsShadow()->ioPtrs.clustersNative = mInputsShadow->mPclusterNativeAccess;
227  WriteToConstantMemory(RecoStep::TPCDecompression, (char*)&processors()->ioPtrs - (char*)processors(), &processorsShadow()->ioPtrs, sizeof(processorsShadow()->ioPtrs), 0);
228  *mInputsHost->mPclusterNativeAccess = *mIOPtrs.clustersNative;
229  mInputsHost->mPclusterNativeAccess->clustersLinear = mInputsShadow->mPclusterNativeBuffer;
230  mInputsHost->mPclusterNativeAccess->setOffsetPtrs();
231  GPUMemCpy(RecoStep::TPCDecompression, mInputsShadow->mPclusterNativeBuffer, mIOPtrs.clustersNative->clustersLinear, sizeof(mIOPtrs.clustersNative->clustersLinear[0]) * mIOPtrs.clustersNative->nClustersTotal, 0, true);
232  TransferMemoryResourceLinkToGPU(RecoStep::TPCDecompression, mInputsHost->mResourceClusterNativeAccess, 0);
233  SynchronizeStream(0);
234  }
235 #endif
236  return 0;
