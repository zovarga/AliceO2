70 void O2DPLDisplaySpec::run(ProcessingContext& pc)
71 {
72  if (!this->mEveHostNameMatch) {
73  return;
74  }
75  LOG(info) << "------------------------ O2DPLDisplay::run version " << this->mWorkflowVersion << " ------------------------------------";
76  // filtering out any run which occur before reaching next time interval
77  auto currentTime = std::chrono::high_resolution_clock::now();
78  std::chrono::duration<double> elapsed = currentTime - this->mTimeStamp;
79  if (elapsed < this->mTimeInteval) {
80  return; // skip this run - it is too often
81  }
82  this->mTimeStamp = currentTime;
83 
84  EveWorkflowHelper helper;
85  helper.getRecoContainer().collectData(pc, *mDataRequest);
86  helper.selectTracks(&(mData.mConfig->configCalib), mClMask, mTrkMask, mTrkMask);
87 
88  helper.prepareITSClusters(mData.mITSDict);
89  helper.prepareMFTClusters(mData.mMFTDict);
90 
91  const auto& ref = pc.inputs().getFirstValid(true);
92  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
93  const auto* dph = DataRefUtils::getHeader<DataProcessingHeader*>(ref);
94 
95  helper.draw(this->mJsonPath, this->mNumberOfFiles, this->mNumberOfTracks, this->mTrkMask, this->mClMask, dh->runNumber, dph->creation, this->mWorkflowVersion);
96  auto endTime = std::chrono::high_resolution_clock::now();
97  LOGP(info, "Visualization of TF:{} at orbit {} took {} s.", dh->tfCounter, dh->firstTForbit, std::chrono::duration_cast<std::chrono::microseconds>(endTime - currentTime).count() * 1e-6);
98 }

104 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
105 {
106  LOG(info) << "------------------------ defineDataProcessing " << O2DPLDisplaySpec::mWorkflowVersion << " ------------------------------------";
107 
108  WorkflowSpec specs;
109 
110  std::string jsonFolder = cfgc.options().get<std::string>("jsons-folder");
111  std::string eveHostName = cfgc.options().get<std::string>("eve-hostname");
112  o2::conf::ConfigurableParam::updateFromString(cfgc.options().get<std::string>("configKeyValues"));
113  bool useMC = cfgc.options().get<bool>("enable-mc") && !cfgc.options().get<bool>("disable-mc");
114 
115  char hostname[HOST_NAME_MAX];
116  gethostname(hostname, HOST_NAME_MAX);
117  bool eveHostNameMatch = eveHostName.empty() || eveHostName == hostname;
118 
119  int eveDDSColIdx = cfgc.options().get<int>("eve-dds-collection-index");
120  if (eveDDSColIdx != -1) {
121  char* colIdx = getenv("DDS_COLLECTION_INDEX");
122  int myIdx = colIdx ? atoi(colIdx) : -1;
123  if (myIdx == eveDDSColIdx) {
124  LOG(important) << "Restricting DPL Display to collection index, my index " << myIdx << ", enabled " << int(myIdx == eveDDSColIdx);
125  } else {
126  LOG(info) << "Restricting DPL Display to collection index, my index " << myIdx << ", enabled " << int(myIdx == eveDDSColIdx);
127  }
128  eveHostNameMatch &= myIdx == eveDDSColIdx;
129  }
130 
131  std::chrono::milliseconds timeInterval(cfgc.options().get<int>("time-interval"));
132  int numberOfFiles = cfgc.options().get<int>("number-of_files");
133  int numberOfTracks = cfgc.options().get<int>("number-of_tracks");
134 
135  GID::mask_t allowedTracks = GID::getSourcesMask("ITS,TPC,MFT,MCH,ITS-TPC,ITS-TPC-TOF,TPC-TRD,ITS-TPC-TRD,MID");
136  GID::mask_t allowedClusters = GID::getSourcesMask("ITS,TPC,MFT,MCH,TRD,TOF,MID,TRD");
137 
138  GlobalTrackID::mask_t srcTrk = GlobalTrackID::getSourcesMask(cfgc.options().get<std::string>("display-tracks")) & allowedTracks;
139  GlobalTrackID::mask_t srcCl = GlobalTrackID::getSourcesMask(cfgc.options().get<std::string>("display-clusters")) & allowedClusters;
140  if (!srcTrk.any() && !srcCl.any()) {
141  throw std::runtime_error("No input configured");
142  }
143  std::shared_ptr<DataRequest> dataRequest = std::make_shared<DataRequest>();
144  dataRequest->requestTracks(srcTrk, useMC);
145  dataRequest->requestClusters(srcCl, useMC);
146 
147  if (cfgc.options().get<bool>("read-from-files")) {
148  InputHelper::addInputSpecs(cfgc, specs, srcCl, srcTrk, srcTrk, useMC);
149  }
150 
151  specs.emplace_back(DataProcessorSpec{
152  "o2-eve-display",
153  dataRequest->inputs,
154  {},
155  AlgorithmSpec{adaptFromTask<O2DPLDisplaySpec>(useMC, srcTrk, srcCl, dataRequest, jsonFolder, timeInterval, numberOfFiles, numberOfTracks, eveHostNameMatch)}});
156 
157  return std::move(specs);
158 }
