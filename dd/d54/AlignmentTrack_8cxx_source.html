88 void AlignmentTrack::defineDOFs()
89 {
90  // define varied DOF's (local parameters) for the track:
91  // 1) kinematic params (5 or 4 depending on Bfield)
92  // 2) mult. scattering angles (2)
93  // 3) if requested by point: energy loss
94  //
95  mNLocPar = mNLocExtPar = getFieldON() ? kNKinParBON : kNKinParBOFF;
96  int np = getNPoints();
97  //
98  // the points are sorted in order opposite to track direction -> outer points come 1st,
99  // but for the 2-leg cosmic track the innermost points are in the middle (1st lower leg, then upper one)
100  //
101  // start along track direction, i.e. last point in the ordered array
102  int minPar = mNLocPar;
103  for (int ip = getInnerPointID() + 1; ip--;) { // collision track or cosmic lower leg
104  AlignmentPoint* pnt = getPoint(ip);
105  pnt->setMinLocVarID(minPar);
106  if (pnt->containsMaterial()) {
107  mNLocPar += pnt->getNMatPar();
108  }
109  pnt->setMaxLocVarID(mNLocPar); // flag up to which parameted ID this points depends on
110  }
111  //
112  if (isCosmic()) {
113  minPar = mNLocPar;
114  for (int ip = getInnerPointID() + 1; ip < np; ip++) { // collision track or cosmic lower leg
115  AlignmentPoint* pnt = getPoint(ip);
116  pnt->setMinLocVarID(minPar);
117  if (pnt->containsMaterial()) {
118  mNLocPar += pnt->getNMatPar();
119  }
120  pnt->setMaxLocVarID(mNLocPar); // flag up to which parameted ID this points depends on
121  }
122  }
123  //
124  if (mLocPar.GetSize() < mNLocPar) {
125  mLocPar.Set(mNLocPar);
126  }
127  mLocPar.Reset();
128  mLocParA = mLocPar.GetArray();
129  //
130  if (mResid[0].GetSize() < np) {
131  mResid[0].Set(np);
132  mResid[1].Set(np);
133  }
134  if (mDResDLoc[0].GetSize() < mNLocPar * np) {
135  mDResDLoc[0].Set(mNLocPar * np);
136  mDResDLoc[1].Set(mNLocPar * np);
137  }
138  for (int i = 2; i--;) {
139  mResid[i].Reset();
140  mDResDLoc[i].Reset();
141  mResidA[i] = mResid[i].GetArray();
142  mDResDLocA[i] = mDResDLoc[i].GetArray();
143  }
144  //
145  // memcpy(mLocParA,GetParameter(),mNLocExtPar*sizeof(double));
146  memset(mLocParA, 0, mNLocExtPar * sizeof(double));
147 }

152  // Propagate for given local params and calculate residuals and their derivatives.
153  // The 1st 4 or 5 elements of params vector should be the reference trackParam_t
154  // Then parameters of material corrections for each point
155  // marked as having materials should come (4 or 5 dependending if ELoss is varied or fixed).
156  // They correspond to kink parameters
157  // (trackParam_t_after_material - trackParam_t_before_material)
158  // rotated to frame where they error matrix is diagonal. Their conversion to trackParam_t
159  // increment will be done locally in the applyMatCorr routine.

162  //
163  if (!params) {
164  params = mLocParA;
165  }
166  //
167  if (!getResidDone()) {
168  calcResiduals(params);
169  }
170  //

172  //
173  // collision track or cosmic lower leg
174  if (!calcResidDeriv(params, mNeedInv[0], getInnerPointID(), 0)) {

176  LOG(warn) << "Failed on derivatives calculation 0";
177 #endif
178  return false;
179  }

194 bool AlignmentTrack::calcResidDeriv(double* extendedParams, bool invert, int pFrom, int pTo)
195 {
196  // Calculate derivatives of residuals vs params for points pFrom to pT. For cosmic upper leg
197  // track parameter may require inversion.
198  // The 1st 4 or 5 elements of params vector should be the reference trackParam_t
199  // Then parameters of material corrections for each point
200  // marked as having materials should come (4 or 5 dependending if ELoss is varied or fixed).
201  // They correspond to kink parameters
202  // (trackParam_t_after_material - trackParam_t_before_material)
203  // rotated to frame where they error matrix is diagonal. Their conversion to trackParam_t
204  // increment will be done locally in the applyMatCorr routine.
205  //
206  // The derivatives are calculated using Richardson extrapolation
207  // (like http://root.cern.ch/root/html/ROOT__Math__RichardsonDerivator.html)
208  //
209  trackParam_t probD[kNRDClones]; // use this to vary supplied param for derivative calculation
210  double varDelta[kRichardsonN];
211  const int kInvElem[kNKinParBON] = {-1, 1, 1, -1, -1};
212  //
213  const double kDelta[kNKinParBON] = {0.02, 0.02, 0.001, 0.001, 0.01}; // variations for ExtTrackParam and material effects
214  //
215  double delta[kNKinParBON]; // variations of curvature term are relative
216  for (int i = kNKinParBOFF; i--;) {
217  delta[i] = kDelta[i];
218  }
219  if (getFieldON()) {
220  delta[kParQ2Pt] = kDelta[kParQ2Pt] * Abs(getQ2Pt());
221  }
222  //
223  int pinc;
224  if (pTo > pFrom) { // fit in points decreasing order: cosmics upper leg
225  pTo++;
226  pinc = 1;
227  } else { // fit in points increasing order: collision track or cosmics lower leg
228  pTo--;
229  pinc = -1;
230  }
231 
232  // 1) derivative wrt trackParam_t parameters
233  for (int ipar = mNLocExtPar; ipar--;) {
234 
235  setParams(probD, kNRDClones, getX(), getAlpha(), extendedParams, true);
236  if (invert) {
237  for (int ic = kNRDClones; ic--;) {
238  probD[ic].invert();
239  }
240  }
241  double del = delta[ipar];
242  //
243  for (int icl = 0; icl < kRichardsonN; icl++) { // calculate kRichardsonN variations with del, del/2, del/4...
244  varDelta[icl] = del;
245  modParam(probD[(icl << 1) + 0], ipar, del);
246  modParam(probD[(icl << 1) + 1], ipar, -del);
247  del *= 0.5;
248  }
249  // propagate varied tracks to each point
250  for (int ip = pFrom; ip != pTo; ip += pinc) { // points are ordered against track direction
251  AlignmentPoint* pnt = getPoint(ip);
252  if (!propagateParamToPoint(probD, kNRDClones, pnt)) {
253  return false;
254  }
255  // if (pnt->containsMaterial()) { // apply material corrections
256  if (!applyMatCorr(probD, kNRDClones, extendedParams, pnt)) {
257  return false;
258  }
259  // }
260  //
261  if (pnt->containsMeasurement()) {
262  int offsDer = ip * mNLocPar + ipar;
263  richardsonDeriv(probD, varDelta, pnt, mDResDLocA[0][offsDer], mDResDLocA[1][offsDer]); // calculate derivatives
264  if (invert && kInvElem[ipar] < 0) {
265  mDResDLocA[0][offsDer] = -mDResDLocA[0][offsDer];
266  mDResDLocA[1][offsDer] = -mDResDLocA[1][offsDer];
267  }
268  }
269  } // loop over points
270  } // loop over ExtTrackParam parameters
271  //
272  // 2) now vary material effect related parameters: MS and eventually ELoss
273  //
274  for (int ip = pFrom; ip != pTo; ip += pinc) { // points are ordered against track direction
275  AlignmentPoint* pnt = getPoint(ip);
276  //
277  // global derivatives at this point
278  if (pnt->containsMeasurement() && !calcResidDerivGlo(pnt)) {
279 #if DEBUG > 3
280  AliWarningF("Failed on global derivatives calculation at point %d", ip);
281  pnt->print("meas");
282 #endif
283  return false;
284  }
285  //
286  if (!pnt->containsMaterial()) {
287  continue;
288  }
289  //
290  int nParFreeI = pnt->getNMatPar();
291  //
292  // array delta gives desired variation of parameters in trackParam_t definition,
293  // while the variation should be done for parameters in the frame where the vector
294  // of material corrections has diagonal cov. matrix -> rotate the delta to this frame
295  double deltaMatD[kNKinParBON];
296  pnt->diagMatCorr(delta, deltaMatD);
297  //
298  // printf("Vary %d [%+.3e %+.3e %+.3e %+.3e] ",ip,deltaMatD[0],deltaMatD[1],deltaMatD[2],deltaMatD[3]); pnt->print();
299 
300  int offsI = pnt->getMaxLocVarID() - nParFreeI; // the parameters for this point start with this offset
301  // they are irrelevant for the points upstream
302  for (int ipar = 0; ipar < nParFreeI; ipar++) { // loop over DOFs related to MS and ELoss are point ip
303  double del = deltaMatD[ipar];
304  //
305  // We will vary the tracks starting from the original parameters propagated to given point
306  // and stored there (before applying material corrections for this point)
307  //
308  setParams(probD, kNRDClones, pnt->getXPoint(), pnt->getAlphaSens(), pnt->getTrParamWSB(), false);
309  // no need for eventual track inversion here: if needed, this is already done in ParamWSB
310  //
311  int offsIP = offsI + ipar; // parameter entry in the extendedParams array
312  // printf(" Var:%d (%d) %e\n",ipar,offsIP, del);
313 
314  for (int icl = 0; icl < kRichardsonN; icl++) { // calculate kRichardsonN variations with del, del/2, del/4...
315  varDelta[icl] = del;
316  double parOrig = extendedParams[offsIP];
317  extendedParams[offsIP] += del;
318  //
319  // apply varied material effects : incremented by delta
320  if (!applyMatCorr(probD[(icl << 1) + 0], extendedParams, pnt)) {
321  return false;
322  }
323  //
324  // apply varied material effects : decremented by delta
325  extendedParams[offsIP] = parOrig - del;
326  if (!applyMatCorr(probD[(icl << 1) + 1], extendedParams, pnt)) {
327  return false;
328  }
329  //
330  extendedParams[offsIP] = parOrig;
331  del *= 0.5;
332  }
333  if (pnt->containsMeasurement()) { // calculate derivatives at the scattering point itself
334  int offsDerIP = ip * mNLocPar + offsIP;
335  richardsonDeriv(probD, varDelta, pnt, mDResDLocA[0][offsDerIP], mDResDLocA[1][offsDerIP]); // calculate derivatives for ip
336  // printf("DR SELF: %e %e at %d (%d)\n",mDResDLocA[0][offsDerIP], mDResDLocA[1][offsDerIP],offsI, offsDerIP);
337  }
338  //
339  // loop over points whose residuals can be affected by the material effects on point ip
340  for (int jp = ip + pinc; jp != pTo; jp += pinc) {
341  AlignmentPoint* pntJ = getPoint(jp);
342 
343  // printf(" DerFor:%d ",jp); pntJ->print();
344 
345  if (!propagateParamToPoint(probD, kNRDClones, pntJ)) {
346  return false;
347  }
348  //
349  if (pntJ->containsMaterial()) { // apply material corrections
350  if (!applyMatCorr(probD, kNRDClones, extendedParams, pntJ)) {
351  return false;
352  }
353  }
354  //
355  if (pntJ->containsMeasurement()) {
356  int offsDerJ = jp * mNLocPar + offsIP;
357  // calculate derivatives
358  richardsonDeriv(probD, varDelta, pntJ, mDResDLocA[0][offsDerJ], mDResDLocA[1][offsDerJ]);
359  }
360  //
361  } // << loop over points whose residuals can be affected by the material effects on point ip
362  } // << loop over DOFs related to MS and ELoss are point ip
363  } // << loop over all points of the track
364  //
365  return true;
366 }

369 bool AlignmentTrack::calcResidDerivGlo(AlignmentPoint* pnt)
370 {
371  // calculate residuals derivatives over point's sensor and its parents global params
372  double deriv[AlignableVolume::kNDOFGeom * 3];
373  //
374  const AlignableSensor* sens = pnt->getSensor();
375  const AlignableVolume* vol = sens;
376  // precalculated track parameters
377  double snp = pnt->getTrParamWSA(kParSnp), tgl = pnt->getTrParamWSA(kParTgl);
378  // precalculate track slopes to account tracking X veriation
379  // these are coeffs to translate deltaX of the point to deltaY and deltaZ of track
380  double cspi = 1. / Sqrt((1 - snp) * (1 + snp)), slpY = snp * cspi, slpZ = tgl * cspi;
381  //
382  pnt->setDGloOffs(mNGloPar); // mark 1st entry of derivatives
383  do {
384  // measurement residuals
385  int nfree = vol->getNDOFsFree();
386  if (!nfree) {
387  continue;
388  } // no free parameters?
389  sens->dPosTraDParGeom(pnt, deriv, vol == sens ? nullptr : vol);
390  //
391  checkExpandDerGloBuffer(mNGloPar + nfree); // if needed, expand derivatives buffer
392  //
393  for (int ip = 0; ip < AlignableVolume::kNDOFGeom; ip++) { // we need only free parameters
394  if (!vol->isFreeDOF(ip)) {
395  continue;
396  }
397  double* dXYZ = &deriv[ip * 3]; // tracking XYZ derivatives over this parameter
398  // residual is defined as diagonalized track_estimate - measured Y,Z in tracking frame
399  // where the track is evaluated at measured X!
400  // -> take into account modified X using track parameterization at the point (paramWSA)
401  // Attention: small simplifications(to be checked if it is ok!!!):
402  // effect of changing X is accounted neglecting track curvature to preserve linearity
403  //
404  // store diagonalized residuals in track buffer
405  pnt->diagonalizeResiduals((dXYZ[AlignmentPoint::kX] * slpY - dXYZ[AlignmentPoint::kY]),
406  (dXYZ[AlignmentPoint::kX] * slpZ - dXYZ[AlignmentPoint::kZ]),
407  mDResDGloA[0][mNGloPar], mDResDGloA[1][mNGloPar]);
408  // and register global ID of varied parameter
409  mGloParIDA[mNGloPar] = vol->getParGloID(ip);
410  mNGloPar++;
411  }
412  //
413  } while ((vol = vol->getParent()));
414  //
415  // eventual detector calibration parameters
416  const AlignableDetector* det = sens->getDetector();
417  int ndof = 0;
418  if (det && (ndof = det->getNCalibDOFs())) {
419  // if needed, expand derivatives buffer
420  checkExpandDerGloBuffer(mNGloPar + det->getNCalibDOFsFree());
421  for (int idf = 0; idf < ndof; idf++) {
422  if (!det->isFreeDOF(idf)) {
423  continue;
424  }
425  sens->dPosTraDParCalib(pnt, deriv, idf, nullptr);
426  pnt->diagonalizeResiduals((deriv[AlignmentPoint::kX] * slpY - deriv[AlignmentPoint::kY]),
427  (deriv[AlignmentPoint::kX] * slpZ - deriv[AlignmentPoint::kZ]),
428  mDResDGloA[0][mNGloPar], mDResDGloA[1][mNGloPar]);
429  // and register global ID of varied parameter
430  mGloParIDA[mNGloPar] = det->getParGloID(idf);
431  mNGloPar++;
432  }
433  }
434  //
435  pnt->setNGloDOFs(mNGloPar - pnt->getDGloOffs()); // mark number of global derivatives filled
436  //
437  return true;
438 }

484 bool AlignmentTrack::calcResiduals(const double* extendedParams, bool invert, int pFrom, int pTo)
485 {
486  // Calculate residuals for the single leg from points pFrom to pT
487  // The 1st 4 or 5 elements of extendedParams vector should be corrections to
488  // the reference trackParam_t
489  // Then parameters of material corrections for each point
490  // marked as having materials should come (4 or 5 dependending if ELoss is varied or fixed).
491  // They correspond to kink parameters
492  // (trackParam_t_after_material - trackParam_t_before_material)
493  // rotated to frame where they error matrix is diagonal. Their conversion to trackParam_t
494  // increment will be done locally in the applyMatCorr routine.
495  //
496  trackParam_t probe;
497  setParams(probe, getX(), getAlpha(), extendedParams, true);
498  if (invert) {
499  probe.invert();
500  }
501  int pinc;
502  if (pTo > pFrom) { // fit in points decreasing order: cosmics upper leg
503  pTo++;
504  pinc = 1;
505  } else { // fit in points increasing order: collision track or cosmics lower leg
506  pTo--;
507  pinc = -1;
508  }
509  //
510  for (int ip = pFrom; ip != pTo; ip += pinc) { // points are ordered against track direction
511  AlignmentPoint* pnt = getPoint(ip);
512  if (!propagateParamToPoint(probe, pnt)) {
513  return false;
514  }
515  //
516  // store the current track kinematics at the point BEFORE applying eventual material
517  // corrections. This kinematics will be later varied around supplied parameters (in the calcResidDeriv)
518  pnt->setTrParamWSB(probe.getParams());
519  //
520  // account for materials
521  // if (pnt->ContainsMaterial()) { // apply material corrections
522  if (!applyMatCorr(probe, extendedParams, pnt)) {
523  return false;
524  }
525  // }
526  pnt->setTrParamWSA(probe.getParams());
527  //
528  if (pnt->containsMeasurement()) { // need to calculate residuals in the frame where errors are orthogonal
529  pnt->getResidualsDiag(probe.getParams(), mResidA[0][ip], mResidA[1][ip]);
530  mChi2 += mResidA[0][ip] * mResidA[0][ip] / pnt->getErrDiag(0);
531  mChi2 += mResidA[1][ip] * mResidA[1][ip] / pnt->getErrDiag(1);
532  mNDF += 2;
533  }
534  //
535  if (pnt->containsMaterial()) {
536  // material degrees of freedom do not contribute to NDF since they are constrained by 0 expectation
537  int nCorrPar = pnt->getNMatPar();
538  const double* corrDiag = &mLocParA[pnt->getMaxLocVarID() - nCorrPar]; // corrections in diagonalized frame
539  float* corCov = pnt->getMatCorrCov(); // correction diagonalized covariance
540  for (int i = 0; i < nCorrPar; i++) {
541  mChi2 += corrDiag[i] * corrDiag[i] / corCov[i];
542  }
543  }
544  }
545  return true;
546 }

575 bool AlignmentTrack::propagateToPoint(trackParam_t& tr, const AlignmentPoint* pnt, double maxStep, double maxSnp, MatCorrType mt, track::TrackLTIntegral* tLT)

950 bool AlignmentTrack::combineTracks(trackParam_t& trcL, const trackParam_t& trcU)
951 {

953  // The trcL and trcU MUST be defined at same X,Alpha
954  //
955  // Update equations: tracks described by vectors vL and vU and coviriances CL and CU resp.
956  // then the gain matrix K = CL*(CL+CU)^-1
957  // Updated vector and its covariance:
958  // CL' = CL - K*CL
959  // vL' = vL + K(vU-vL)
960  //
961  if (Abs(trcL.getX() - trcU.getX()) > TinyDist || Abs(trcL.getAlpha() - trcU.getAlpha()) > TinyDist) {
962  LOG(error) << "Tracks must be defined at same reference X and Alpha";

966  }

968  // const covMat_t& covU = trcU.getCov();
969  // const covMat_t& covL = trcL.getCov();
970  //
971  int mtSize = getFieldON() ? kNKinParBON : kNKinParBOFF;
972  TMatrixD matCL(mtSize, mtSize), matCLplCU(mtSize, mtSize);
973  TVectorD vl(mtSize), vUmnvL(mtSize);
974  //
975  // trcL.print();
976  // trcU.print();
977  //
978  for (int i = mtSize; i--;) {
979  vUmnvL[i] = trcU.getParam(i) - trcL.getParam(i); // y = residual of 2 tracks
980  vl[i] = trcL.getParam(i);
981  for (int j = i + 1; j--;) {
982  int indIJ = ((i * (i + 1)) >> 1) + j; // position of IJ cov element in the trackParam_t covariance array
983  matCL(i, j) = matCL(j, i) = trcL.getCovarElem(i, j);

985  }
986  }

992 #endif
993  return false; // inversion failed
994  }

1003  }

1008  for (int j = mtSize; j--;) {
1009  chi2 += matCLplCU(i, j) * vUmnvL[i] * vUmnvL[j];
1010  }
1011  }
1012  mChi2 += chi2;
1013  //
1014  // printf("Combined: Chi2Tot:%.2f ChiUp:%.2f ChiDn:%.2f ChiCmb:%.2f\n",mChi2,mChi2CosmUp,mChi2CosmDn, chi2);
1015 
1016  return true;
