35 uint32_t ChipStat::addErrors(uint32_t mask, uint16_t chID, int verbosity)
36 {
37  uint32_t res = 0;
38  if (mask) {
39  for (int i = NErrorsDefined; i--;) {
40  if (mask & (0x1 << i)) {
41  res |= ErrActions[i] & ErrActPropagate;
42  if (verbosity > -1 && (!errorCounts[i] || verbosity > 1)) {
43  LOGP(important, "New error registered on the FEEID:{:#04x}: chip#{}: {}", feeID, chID, ErrNames[i]);
44  res |= ErrActions[i] & ErrActDump;
45  }
46  errorCounts[i]++;
47  }
48  }
49  }
50  return res;
51 }

55 uint32_t ChipStat::addErrors(const ChipPixelData& d, int verbosity)
56 {
57  uint32_t res = 0;

59  for (int i = NErrorsDefined; i--;) {
60  if (d.getErrorFlags() & (0x1 << i)) {
61  res |= ErrActions[i] & ErrActPropagate;
62  if (verbosity > -1 && (!errorCounts[i] || verbosity > 1)) {
63  LOGP(important, "New error registered at bc/orbit {}/{} on the FEEID:{:#04x} chip#{}: {}{}",
64  d.getInteractionRecord().bc, d.getInteractionRecord().orbit,
65  feeID, int16_t(d.getChipID()), ErrNames[i], d.getErrorDetails(i));
66  res |= ErrActions[i] & ErrActDump;
67  }
68  errorCounts[i]++;
69  }
70  }
71  }
