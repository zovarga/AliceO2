76 template <class Mapping>
77 int RawPixelDecoder<Mapping>::decodeNextTrigger()
78 {
79  mTimerDecode.Start(false);
80  mNChipsFiredROF = 0;
81  mNPixelsFiredROF = 0;
82  mInteractionRecord.clear();
83  int nLinksWithData = 0, nru = mRUDecodeVec.size();
84  do {
85 #ifdef WITH_OPENMP
86  omp_set_num_threads(mNThreads);
87 #pragma omp parallel for schedule(dynamic) reduction(+ \
88  : nLinksWithData, mNChipsFiredROF, mNPixelsFiredROF)
89 #endif
90  for (int iru = 0; iru < nru; iru++) {
91  nLinksWithData += decodeNextTrigger(iru);
92  mNChipsFiredROF += mRUDecodeVec[iru].nChipsFired;
93  int npix = 0;
94  for (int ic = mRUDecodeVec[iru].nChipsFired; ic--;) {
95  npix += mRUDecodeVec[iru].chipsData[ic].getData().size();
96  }
97  mNPixelsFiredROF += npix;
98  }
99 
100  if (nLinksWithData) { // fill some statistics
101  mROFCounter++;
102  mNChipsFired += mNChipsFiredROF;
103  mNPixelsFired += mNPixelsFiredROF;
104  mCurRUDecodeID = 0; // getNextChipData will start from here
105  mLastReadChipID = -1;
106  // set IR and trigger from the 1st non empty link
107  for (const auto& link : mGBTLinks) {
108  if (link.status == GBTLink::DataSeen) {
109  mInteractionRecord = link.ir;
110  mInteractionRecordHB = o2::raw::RDHUtils::getHeartBeatIR(*link.lastRDH);
111  mTrigger = link.trigger;
112  break;
113  }
114  }
115  break;
116  }
117 
118  } while (mNLinksDone < mGBTLinks.size());
119 
120  ensureChipOrdering();
121  mTimerDecode.Stop();
122  return nLinksWithData;
123 }

146 template <class Mapping>
147 int RawPixelDecoder<Mapping>::decodeNextTrigger(int iru)
148 {
149  auto& ru = mRUDecodeVec[iru];
150  ru.clear();
151  int ndec = 0; // number of yet non-empty links
152  for (int il = 0; il < RUDecodeData::MaxLinksPerRU; il++) {
153  auto* link = getGBTLink(ru.links[il]);
154  if (link) {
155  auto res = link->collectROFCableData(mMAP);
156  if (res == GBTLink::DataSeen) { // at the moment process only DataSeen
157  ndec++;
158  } else if (res == GBTLink::StoppedOnEndOfData || res == GBTLink::AbortedOnError) { // this link has exhausted its data or it has to be discarded due to the error
159  mNLinksDone++;
160  }
161  }
162  }
163  if (ndec) {
164  ru.decodeROF(mMAP);
165  }
166  return ndec;
167 }

171 template <class Mapping>
172 void RawPixelDecoder<Mapping>::setupLinks(InputRecord& inputs)
173 {
174  mCurRUDecodeID = NORUDECODED;
175  auto nLinks = mGBTLinks.size();
176  auto origin = (mUserDataOrigin == o2::header::gDataOriginInvalid) ? mMAP.getOrigin() : mUserDataOrigin;
177  auto datadesc = (mUserDataDescription == o2::header::gDataDescriptionInvalid) ? o2::header::gDataDescriptionRawData : mUserDataDescription;
178  std::vector<InputSpec> filter{InputSpec{"filter", ConcreteDataTypeMatcher{origin, datadesc}, Lifetime::Timeframe}};
179 
180  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload this means that the "delayed message"
181  // mechanism created it in absence of real data from upstream. Processor should send empty output to not block the workflow
182  {
183  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
184  std::vector<InputSpec> dummy{InputSpec{"dummy", ConcreteDataMatcher{origin, datadesc, 0xDEADBEEF}}};
185  for (const auto& ref : InputRecordWalker(inputs, dummy)) {
186  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
187  if (dh->payloadSize == 0) {
188  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
189  if (++contDeadBeef <= maxWarn) {
190  LOGP(warning, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
191  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize,
192  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
193  }
194  return;
195  }
196  }
197  contDeadBeef = 0; // if good data, reset the counter
198  }
199 
200  DPLRawParser parser(inputs, filter);
201  uint32_t currSSpec = 0xffffffff; // dummy starting subspec
202  int linksAdded = 0;
203  for (auto it = parser.begin(); it != parser.end(); ++it) {
204  auto const* dh = it.o2DataHeader();
205  auto& lnkref = mSubsSpec2LinkID[dh->subSpecification];
206  const auto& rdh = *reinterpret_cast<const header::RDHAny*>(it.raw()); // RSTODO this is a hack in absence of generic header getter
207 
208  if (lnkref.entry == -1) { // new link needs to be added
209  lnkref.entry = int(mGBTLinks.size());
210  auto& lnk = mGBTLinks.emplace_back(RDHUtils::getCRUID(rdh), RDHUtils::getFEEID(rdh), RDHUtils::getEndPointID(rdh), RDHUtils::getLinkID(rdh), lnkref.entry);
211  lnk.subSpec = dh->subSpecification;
212  getCreateRUDecode(mMAP.FEEId2RUSW(RDHUtils::getFEEID(rdh))); // make sure there is a RU for this link
213  lnk.verbosity = GBTLink::Verbosity(mVerbosity);
214  lnk.format = mFormat;
215  LOG(info) << mSelfName << " registered new link " << lnk.describe() << " RUSW=" << int(mMAP.FEEId2RUSW(lnk.feeID));
216  linksAdded++;
217  }
218  auto& link = mGBTLinks[lnkref.entry];
219  if (currSSpec != dh->subSpecification) { // this is the 1st part for this link in this TF, next parts must follow contiguously!!!
220  link.clear(false, true); // clear link data except statistics
221  currSSpec = dh->subSpecification;
222  }
223  link.cacheData(it.raw(), RDHUtils::getMemorySize(rdh));
224  }
225 
226  if (linksAdded) { // new links were added, update link<->RU mapping, usually is done for 1st TF only
227  if (nLinks) {
228  LOG(alarm) << mSelfName << " New links appeared although the initialization was already done";
229  for (auto& ru : mRUDecodeVec) { // reset RU->link references since they may have been changed
230  memset(&ru.links[0], -1, RUDecodeData::MaxLinksPerRU * sizeof(int));
231  }
232  }
233  // sort RUs in stave increasing order
234  std::sort(mRUDecodeVec.begin(), mRUDecodeVec.end(), [](const RUDecodeData& ruA, const RUDecodeData& ruB) -> bool { return ruA.ruSWID < ruB.ruSWID; });
235  for (auto i = 0; i < mRUDecodeVec.size(); i++) {
236  mRUEntry[mRUDecodeVec[i].ruSWID] = i;
237  }
238  nLinks = mGBTLinks.size();
239  // attach link to corresponding RU: this can be done once all RUs are created, to make sure their pointers don't change
240  for (int il = 0; il < nLinks; il++) {
241  auto& link = mGBTLinks[il];
242  bool newLinkAdded = (link.ruPtr == nullptr);
243  link.ruPtr = getRUDecode(mMAP.FEEId2RUSW(link.feeID)); // link to RU reference, reattach even it was already set before
244  uint16_t lr, ruOnLr, linkInRU;
245  mMAP.expandFEEId(link.feeID, lr, ruOnLr, linkInRU);
246  if (newLinkAdded) {
247  LOG(info) << mSelfName << " Attaching " << link.describe() << " to RU#" << int(mMAP.FEEId2RUSW(link.feeID))
248  << " (stave " << ruOnLr << " of layer " << lr << ')';
249  }
250  link.idInRU = linkInRU;
251  link.ruPtr->links[linkInRU] = il; // RU to link reference
252  }
253  }
254 }

275 template <class Mapping>
276 ChipPixelData* RawPixelDecoder<Mapping>::getNextChipData(std::vector<ChipPixelData>& chipDataVec)
277 {
278  // decode new RU if no cached non-empty chips
279  for (; mCurRUDecodeID < mRUDecodeVec.size(); mCurRUDecodeID++) {
280  auto& ru = mRUDecodeVec[mCurRUDecodeID];
281  if (ru.lastChipChecked < ru.nChipsFired) {
282  auto& chipData = ru.chipsData[ru.lastChipChecked++];
283  assert(mLastReadChipID < chipData.getChipID());
284  mLastReadChipID = chipData.getChipID();
285  chipDataVec[mLastReadChipID].swap(chipData);
286  return &chipDataVec[mLastReadChipID];
287  }
288  }
289  // will need to decode new trigger
290  if (!mDecodeNextAuto || !decodeNextTrigger()) { // no more data to decode
291  return nullptr;
292  }
293  return getNextChipData(chipDataVec);
294 }

319 template <>
320 void RawPixelDecoder<ChipMappingMFT>::ensureChipOrdering()
321 {
322  mOrderedChipsPtr.clear();
323  // define looping order, if mCurRUDecodeID < mRUDecodeVec.size(), this means that decodeNextTrigger() was called before
324  if (mCurRUDecodeID < mRUDecodeVec.size()) { // define sort order
325  for (; mCurRUDecodeID < mRUDecodeVec.size(); mCurRUDecodeID++) {
326  auto& ru = mRUDecodeVec[mCurRUDecodeID];
327  while (ru.lastChipChecked < ru.nChipsFired) {
328  mOrderedChipsPtr.push_back(&ru.chipsData[ru.lastChipChecked++]);
329  }
330  }
331  // sort in decreasing order
332  std::sort(mOrderedChipsPtr.begin(), mOrderedChipsPtr.end(), [](const ChipPixelData* a, const ChipPixelData* b) { return a->getChipID() > b->getChipID(); });
333  }
334 }

415 template <class Mapping>
416 void RawPixelDecoder<Mapping>::produceRawDataDumps(int dump, const o2::header::DataHeader* dh)
417 {
418  bool dumpFullTF = false;
419  for (auto& ru : mRUDecodeVec) {
420  if (ru.linkHBFToDump.size()) {
421  if (dump == int(GBTLink::RawDataDumps::DUMP_TF)) {
422  dumpFullTF = true;
423  break;
424  }
425  for (auto it : ru.linkHBFToDump) {
426  if (dump == int(GBTLink::RawDataDumps::DUMP_HBF)) {
427  const auto& lnk = mGBTLinks[mSubsSpec2LinkID[it.first >> 32].entry];
428  int entry = it.first & 0xffffffff;
429  bool allHBFs = false;
430  std::string fnm;
431  if (entry >= lnk.rawData.getNPieces()) {
432  allHBFs = true;
433  entry = 0;
434  fnm = fmt::format("rawdump_{}_run{}_tf_orb{}_full_feeID{:#06x}.raw",
435  Mapping::getName(), dh->runNumber, dh->firstTForbit, lnk.feeID);
436  } else {
437  fnm = fmt::format("rawdump_{}_run{}_tf_orb{}_hbf_orb{}_feeID{:#06x}.raw",
438  Mapping::getName(), dh->runNumber, dh->firstTForbit, it.second, lnk.feeID);
439  }
440  std::ofstream ostrm(fnm, std::ios::binary);
441  if (!ostrm.good()) {
442  LOG(error) << "failed to open " << fnm;
443  continue;
444  }
445  while (entry < lnk.rawData.getNPieces()) {
446  const auto* piece = lnk.rawData.getPiece(entry);
447  if (!allHBFs && RDHUtils::getHeartBeatOrbit(reinterpret_cast<const RDH*>(piece->data)) != it.second) {
448  break;
449  }
450  ostrm.write(reinterpret_cast<const char*>(piece->data), piece->size);
451  entry++;
452  }
453  LOG(important) << "produced " << std::filesystem::current_path().c_str() << '/' << fnm;
454  }
455  }
456  }
457  }
458  while (dumpFullTF) {
459  std::string fnm = fmt::format("rawdump_{}_run{}_tf_orb{}_full.raw",
460  Mapping::getName(), dh->runNumber, dh->firstTForbit);
461  std::ofstream ostrm(fnm, std::ios::binary);
462  if (!ostrm.good()) {
463  LOG(error) << "failed to open " << fnm;
464  break;
465  }
466  for (const auto& lnk : mGBTLinks) {
467  for (size_t i = 0; i < lnk.rawData.getNPieces(); i++) {
468  const auto* piece = lnk.rawData.getPiece(i);
469  ostrm.write(reinterpret_cast<const char*>(piece->data), piece->size);
470  }
471  }
472  LOG(important) << "produced " << std::filesystem::current_path().c_str() << '/' << fnm;
473  break;
474  }
475 }
