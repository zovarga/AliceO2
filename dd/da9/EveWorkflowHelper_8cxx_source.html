31 void EveWorkflowHelper::selectTracks(const CalibObjectsConst* calib,
32  GID::mask_t maskCl, GID::mask_t maskTrk, GID::mask_t maskMatch)
33 {
34  auto creator = [maskTrk, this](auto& trk, GID gid, float time, float) {
35  if (!maskTrk[gid.getSource()]) {
36  return true;
37  }
38  if constexpr (isTPCTrack<decltype(trk)>()) { // unconstrained TPC track, with t0 = TrackTPC.getTime0+0.5*(DeltaFwd-DeltaBwd) and terr = 0.5*(DeltaFwd+DeltaBwd) in TimeBins
39  time = trk.getTime0(); // for TPC we need internal time, not the center of the possible interval
40  }
41  mTrackSet.trackGID.push_back(gid);
42  mTrackSet.trackTime.push_back(time);
43  return true;
44  };
45  this->mRecoCont.createTracksVariadic(creator);
46 }

48 void EveWorkflowHelper::draw(const std::string& jsonPath,
49  int numberOfFiles,
50  int numberOfTracks,
51  o2::dataformats::GlobalTrackID::mask_t trkMask,
52  o2::dataformats::GlobalTrackID::mask_t clMask,
53  o2::header::DataHeader::RunNumberType runNumber,
54  o2::framework::DataProcessingHeader::CreationTime creation,
55  float workflowVersion)
56 {
57  size_t nTracks = mTrackSet.trackGID.size();
58  if (numberOfTracks != -1 && numberOfTracks < nTracks) {
59  nTracks = numberOfTracks; // less than available
60  }
61  for (size_t it = 0; it < nTracks; it++) {
62  const auto& gid = mTrackSet.trackGID[it];
63  auto tim = mTrackSet.trackTime[it];
64  // LOG(info) << "EveWorkflowHelper::draw " << gid.asString();
65  switch (gid.getSource()) {
66  case GID::TPC:
67  drawTPC(gid, tim);
68  break;
69  case GID::MFT:
70  drawMFT(gid, tim);
71  break;
72  case GID::MCH:
73  drawMCH(gid, tim);
74  break;
75  case GID::MID:
76  drawMID(gid, tim);
77  break;
78  case GID::ITS:
79  drawITS(gid, tim);
80  break;
81  case GID::ITSTPC:
82  drawITSTPC(gid, tim);
83  break;
84  case GID::ITSTPCTOF:
85  drawITSTPCTOF(gid, tim);
86  break;
87  case GID::TPCTRD:
88  drawTPCTRD(gid, tim);
89  break;
90  case GID::TPCTOF:
91  drawTPCTOF(gid, tim);
92  break;
93  case GID::TPCTRDTOF:
94  drawTPCTRDTOF(gid, tim);
95  break;
96  case GID::ITSTPCTRD:
97  drawITSTPCTRD(gid, tim);
98  break;
99  case GID::ITSTPCTRDTOF:
100  drawITSTPCTRDTOF(gid, tim);
101  break;
102  default:
103  LOG(info) << "Track type " << gid.getSource() << " not handled";
104  }
105  }
106 
107  // save json file
108  save(jsonPath, numberOfFiles, trkMask, clMask, workflowVersion, runNumber, creation);
109 }

131 std::vector<PNT> EveWorkflowHelper::getTrackPoints(const o2::track::TrackPar& trc, float minR, float maxR, float maxStep, float minZ, float maxZ)
132 {
133  // adjust minR according to real track start from track starting point
134  auto maxR2 = maxR * maxR;
135  float rMin = std::sqrt(trc.getX() * trc.getX() + trc.getY() * trc.getY());
136  if (rMin > minR) {
137  minR = rMin;
138  }
139  // prepare space points from the track param
140  std::vector<PNT> pnts;
141  int nSteps = std::max(2, int((maxR - minR) / maxStep));
142  const auto prop = o2::base::Propagator::Instance();
143  float xMin = trc.getX(), xMax = maxR * maxR - trc.getY() * trc.getY();
144  if (xMax > 0) {
145  xMax = std::sqrt(xMax);
146  }
147 
148  float dx = (xMax - xMin) / nSteps;
149  auto tp = trc;
150  float dxmin = std::abs(xMin - tp.getX()), dxmax = std::abs(xMax - tp.getX());
151 
152  if (dxmin > dxmax) { // start from closest end
153  std::swap(xMin, xMax);
154  dx = -dx;
155  }
156  if (!prop->propagateTo(tp, xMin, false, 0.99, maxStep, o2::base::PropagatorF::MatCorrType::USEMatCorrNONE)) {
157  return pnts;
158  }
159  auto xyz = tp.getXYZGlo();
160  pnts.emplace_back(PNT{xyz.X(), xyz.Y(), xyz.Z()});
161  for (int is = 0; is < nSteps; is++) {
162  if (!prop->propagateTo(tp, tp.getX() + dx, false, 0.99, 999., o2::base::PropagatorF::MatCorrType::USEMatCorrNONE)) {
163  return pnts;
164  }
165  xyz = tp.getXYZGlo();
166  if (xyz.Z() < minZ) {
167  return pnts;
168  }
169  if (xyz.Z() > maxZ) {
170  return pnts;
171  }
172  if (xyz.X() * xyz.X() + xyz.Y() * xyz.Y() > maxR2) {
173  return pnts;
174  }
175  pnts.emplace_back(PNT{xyz.X(), xyz.Y(), xyz.Z()});
176  }
177  return pnts;
178 }

180 void EveWorkflowHelper::addTrackToEvent(const o2::track::TrackParCov& tr, GID gid, float trackTime, float dz, GID::Source source)
181 {
182  if (source == GID::NSources) {
183  source = (o2::dataformats::GlobalTrackID::Source)gid.getSource();
184  }
185  auto vTrack = mEvent.addTrack({.time = trackTime,
186  .charge = tr.getCharge(),
187  .PID = tr.getPID(),
188  .startXYZ = {tr.getX(), tr.getY(), tr.getZ()},
189  .phi = tr.getPhi(),
190  .theta = tr.getTheta(),
191  .eta = tr.getEta(),
192  .gid = gid.asString(),
193  .source = source});
194  auto pnts = getTrackPoints(tr, minmaxR[source].first, minmaxR[source].second, 4, minmaxZ[source].first, minmaxZ[source].second);
195 
196  for (size_t ip = 0; ip < pnts.size(); ip++) {
197  vTrack->addPolyPoint(pnts[ip][0], pnts[ip][1], pnts[ip][2] + dz);
198  }
199 }

304 void EveWorkflowHelper::drawTOFClusters(GID gid, float trackTime)
305 {
306  auto tOFClustersArray = mRecoCont.getTOFClusters();
307  if (!gid.includesDet(o2::dataformats::GlobalTrackID::Source::TOF)) {
308  return;
309  }
310  const o2::dataformats::MatchInfoTOF& match = mRecoCont.getTOFMatch(gid);
311  int tofcl = match.getIdxTOFCl();
312  int sector = tOFClustersArray[tofcl].getSector();
313  float x = tOFClustersArray[tofcl].getX();
314  float y = tOFClustersArray[tofcl].getY();
315  float z = tOFClustersArray[tofcl].getZ();
316 
317  // rotation
318  float alpha = o2::math_utils::sector2Angle(sector);
319  float xGlb = x * cos(alpha) - y * sin(alpha);
320  float yGlb = y * cos(alpha) + x * sin(alpha);
321  float zGlb = z;
322  drawPoint(xGlb, yGlb, zGlb, trackTime);
323 }

325 void EveWorkflowHelper::drawITSClusters(GID gid, float trackTime)
326 {
327  // LOG(info) << "EveWorkflowHelper::drawITSClusters" << gid;
328  if (gid.getSource() == GID::ITS) { // this is for for full standalone tracks
329  // LOG(info) << "EveWorkflowHelper::drawITSClusters ITS " << gid;
330  const auto& trc = mRecoCont.getITSTrack(gid);
331  auto refs = mRecoCont.getITSTracksClusterRefs();
332  int ncl = trc.getNumberOfClusters();
333  int offset = trc.getFirstClusterEntry();
334  for (int icl = 0; icl < ncl; icl++) {
335  const auto& pnt = mITSClustersArray[refs[icl + offset]];
336  const auto glo = mITSGeom->getMatrixT2G(pnt.getSensorID()) * pnt.getXYZ();
337  drawPoint(glo.X(), glo.Y(), glo.Z(), trackTime);
338  }
339  } else if (gid.getSource() == GID::ITSAB) { // this is for ITS tracklets from ITS-TPC afterburner
340  // LOG(info) << "EveWorkflowHelper::drawITSClusters ITSAB " << gid;
341  const auto& trc = mRecoCont.getITSABRef(gid);
342  const auto& refs = mRecoCont.getITSABClusterRefs();
343  int ncl = trc.getNClusters();
344  int offset = trc.getFirstEntry();
345  for (int icl = 0; icl < ncl; icl++) {
346  const auto& pnt = mITSClustersArray[refs[icl + offset]];
347  const auto glo = mITSGeom->getMatrixT2G(pnt.getSensorID()) * pnt.getXYZ();
348  drawPoint(glo.X(), glo.Y(), glo.Z(), trackTime);
349  }
350  }
351 }

353 // TPC cluseters for given TPC track (gid)
354 void EveWorkflowHelper::drawTPCClusters(GID gid, float trackTimeTB)
355 {
356  const auto& trc = mRecoCont.getTPCTrack(gid);
357  auto mTPCTracksClusIdx = mRecoCont.getTPCTracksClusterRefs();
358  auto mTPCClusterIdxStruct = &mRecoCont.getTPCClusters();
359 
360  // store the TPC cluster positions
361  for (int iCl = trc.getNClusterReferences(); iCl--;) {
362  uint8_t sector, row;
363  const auto& clTPC = trc.getCluster(mTPCTracksClusIdx, iCl, *mTPCClusterIdxStruct, sector, row);
364 
365  std::array<float, 3> xyz;
366  this->mTPCFastTransform->TransformIdeal(sector, row, clTPC.getPad(), clTPC.getTime(), xyz[0], xyz[1], xyz[2], trackTimeTB); // in sector coordinate
367  o2::math_utils::rotateZ(xyz, o2::math_utils::sector2Angle(sector % o2::tpc::SECTORSPERSIDE)); // lab coordinate (global)
368  mEvent.addCluster(xyz[0], xyz[1], xyz[2], trackTimeTB / mMUS2TPCTimeBins);
369  }

540 
541  for (int iLayer = 0; iLayer < 6; ++iLayer) {
542  if (tpcTrdTrack.getTrackletIndex(iLayer) >= 0) {
543  // there is a TRD space point in this layer
544  const auto& tracklet = trdTracklets[tpcTrdTrack.getTrackletIndex(iLayer)];
545  const auto& spacePoint = trdCalibratedTracklets[tpcTrdTrack.getTrackletIndex(iLayer)];
546  // get position in sector coordinates from the spacePoint
547  float x = spacePoint.getX(), y = spacePoint.getY(), z = spacePoint.getZ();

553  // now the rotation is simply
554  float xGlb = x * cos(alpha) - y * sin(alpha);
555  float yGlb = y * cos(alpha) + x * sin(alpha);
556  float zGlb = z;
557  drawPoint(xGlb, yGlb, zGlb, trackTime);
558  }
