92 template <typename I, typename Cmp>
93 GPUdi() void GPUCommonAlgorithm::InsertionSort(I f, I l, Cmp cmp) noexcept
94 {
95  auto it0{f};
96  while (it0 != l) {
97  auto tmp{*it0};
98 
99  auto it1{it0};
100  while (it1 != f && cmp(tmp, it1[-1])) {
101  it1[0] = it1[-1];
102  --it1;
103  }
104  it1[0] = tmp;
105 
106  ++it0;
107  }
108 }

110 template <typename I, typename Cmp>
111 GPUdi() I GPUCommonAlgorithm::MedianOf3Select(I f, I l, Cmp cmp) noexcept
112 {
113  auto m = f + (l - f) / 2;
114 
115  --l;
116 
117  if (cmp(*f, *m)) {
118  if (cmp(*m, *l)) {
119  return m;
120  } else if (cmp(*f, *l)) {
121  return l;
122  } else {
123  return f;
124  }
125  } else if (cmp(*f, *l)) {
126  return f;
127  } else if (cmp(*m, *l)) {
128  return l;
129  } else {
130  return m;
131  }
132 }

134 template <typename I, typename T, typename Cmp>
135 GPUdi() I GPUCommonAlgorithm::UnguardedPartition(I f, I l, T piv, Cmp cmp) noexcept
136 {
137  do {
138  while (cmp(*f, piv)) {
139  ++f;
140  }
141  --l;
142  while (cmp(piv, *l)) {
143  --l;
144  }
145 
146  if (l <= f) {
147  return f;
148  }
149  IterSwap(f, l);
150  ++f;
151  } while (true);
152 }

154 template <typename I, typename Cmp>
155 GPUdi() void GPUCommonAlgorithm::QuickSort(I f, I l, Cmp cmp) noexcept
156 {
157  if (f == l) {
158  return;
159  }
160  using IndexType = unsigned short;
161 
162  struct pair {
163  IndexType first;
164  IndexType second;
165  };
166 
167  struct Stack {
168  pair data[11];
169  unsigned char n{0};
170 
171  GPUd() void emplace(IndexType x, IndexType y)
172  {
173  data[n++] = {x, y};
174  }
175  GPUd() bool empty() const { return n == 0; }
176  GPUd() pair& top() { return data[n - 1]; }
177  GPUd() void pop() { --n; }
178  };
179 
180  Stack s;
181  s.emplace(0, l - f);
182  while (!s.empty()) {
183  const auto it0 = f + s.top().first;
184  const auto it1 = f + s.top().second;
185  s.pop();
186 
187  const auto piv = *MedianOf3Select(it0, it1, cmp);
188  const auto pp = UnguardedPartition(it0, it1, piv, cmp);
189 
190  constexpr auto cutoff = 50u;
191  const auto lsz = pp - it0;
192  const auto rsz = it1 - pp;
193  if (lsz < rsz) {
194  if (rsz > cutoff) {
195  s.emplace(pp - f, it1 - f);
196  }
197  if (lsz > cutoff) {
198  s.emplace(it0 - f, pp - f);
199  }
200  } else {
201  if (lsz > cutoff) {
202  s.emplace(it0 - f, pp - f);
203  }
204  if (rsz > cutoff) {
205  s.emplace(pp - f, it1 - f);
206  }
207  }
208  }
209  InsertionSort(f, l, cmp);
210 }

296 template <class T, class S>
297 GPUdi() void GPUCommonAlgorithm::sortInBlock(T* begin, T* end, const S& comp)
298 {
299 #ifndef GPUCA_GPUCODE
300  GPUCommonAlgorithm::sort(begin, end, comp);
301 #else
302  int n = end - begin;
303  for (int i = 0; i < n; i++) {
304  for (int tIdx = get_local_id(0); tIdx < n; tIdx += get_local_size(0)) {
305  int offset = i % 2;
306  int curPos = 2 * tIdx + offset;
307  int nextPos = curPos + 1;
308 
309  if (nextPos < n) {
310  if (!comp(begin[curPos], begin[nextPos])) {
311  IterSwap(&begin[curPos], &begin[nextPos]);
312  }
313  }
314  }
315  GPUbarrier();
316  }
317 #endif
318 }
