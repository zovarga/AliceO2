38 void DataProcessor::doSend(DataSender& sender, MessageContext& context, ServiceRegistry&)
39 {
40  std::unordered_map<std::string const*, FairMQParts> outputs;
41  auto contextMessages = context.getMessagesForSending();
42  for (auto& message : contextMessages) {
43  // monitoringService.send({ message->parts.Size(), "outputs/total" });
44  FairMQParts parts = std::move(message->finalize());
45  assert(message->empty());
46  assert(parts.Size() == 2);
47  for (auto& part : parts) {
48  outputs[&(message->channel())].AddPart(std::move(part));
49  }
50  }
51  for (auto& [channel, parts] : outputs) {
52  sender.send(parts, *channel);
53  }
54 }

79  using o2::monitoring::tags::Key;

82 
83  for (auto& messageRef : context) {
84  FairMQParts parts;
85  // Depending on how the arrow table is constructed, we finalize
86  // the writing here.
87  messageRef.finalize(messageRef.buffer);
88 
89  std::unique_ptr<FairMQMessage> payload = messageRef.buffer->Finalise();
90  // FIXME: for the moment we simply send empty bodies.
91  const DataHeader* cdh = o2::header::get<DataHeader*>(messageRef.header->GetData());
92  // sigh... See if we can avoid having it const by not
93  // exposing it to the user in the first place.
94  DataHeader* dh = const_cast<DataHeader*>(cdh);
95  dh->payloadSize = payload->GetSize();
96  dh->serialization = o2::header::gSerializationMethodArrow;
97  monitoring.send(Metric{(uint64_t)payload->GetSize(), fmt::format("table-bytes-{}-{}-created", dh->dataOrigin.as<std::string>(), dh->dataDescription.as<std::string>())}.addTag(Key::Subsystem, Value::DPL));
98  LOGP(info, "Creating {}MB for table {}/{}.", payload->GetSize() / 1000000., dh->dataOrigin, dh->dataDescription);
99  context.updateBytesSent(payload->GetSize());
100  context.updateMessagesSent(1);
101  parts.AddPart(std::move(messageRef.header));
102  parts.AddPart(std::move(payload));
103  sender.send(parts, messageRef.channel);
104  }
105  static int64_t previousBytesSent = 0;
106  auto disposeResources = [bs = context.bytesSent() - previousBytesSent](int taskId,
107  std::array<ComputingQuotaOffer, 16>& offers,
108  ComputingQuotaStats& stats,
109  std::function<void(ComputingQuotaOffer const&, ComputingQuotaStats&)> accountDisposed) {
110  ComputingQuotaOffer disposed;
111  disposed.sharedMemory = 0;
112  int64_t bytesSent = bs;
113  for (size_t oi = 0; oi < offers.size(); oi++) {
114  auto& offer = offers[oi];
115  if (offer.user != taskId) {
116  continue;
117  }
118  int64_t toRemove = std::min((int64_t)bytesSent, offer.sharedMemory);
119  offer.sharedMemory -= toRemove;
120  bytesSent -= toRemove;
121  disposed.sharedMemory += toRemove;
122  if (bytesSent <= 0) {
123  break;
124  }
125  }
126  return accountDisposed(disposed, stats);
127  };
128  registry.get<DeviceState>().offerConsumers.push_back(disposeResources);
129  previousBytesSent = context.bytesSent();
130  monitoring.send(Metric{(uint64_t)context.bytesSent(), "arrow-bytes-created"}.addTag(Key::Subsystem, Value::DPL));
131  monitoring.send(Metric{(uint64_t)context.messagesCreated(), "arrow-messages-created"}.addTag(Key::Subsystem, Value::DPL));
132  monitoring.flushBuffer();
133 }
