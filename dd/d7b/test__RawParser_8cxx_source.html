28 template <typename RDH, typename Container>
29 void fillPages(Container& buffer)
30 {
31  RDH defaultHeader;
32  size_t const NofPages = buffer.size() / PageSize;
33  memset(buffer.data(), 0, buffer.size());
34  for (int pageNo = 0; pageNo < NofPages; pageNo++) {
35  auto* rdh = reinterpret_cast<RDH*>(buffer.data() + pageNo * PageSize);
36  rdh->version = defaultHeader.version;
37  rdh->headerSize = sizeof(RDH);
38  rdh->offsetToNext = PageSize;
39  rdh->memorySize = PageSize;
40  rdh->pageCnt = NofPages;
41  rdh->packetCounter = pageNo;
42  rdh->stop = pageNo + 1 == NofPages;
43  auto* data = reinterpret_cast<size_t*>(buffer.data() + pageNo * PageSize + rdh->headerSize);
44  *data = pageNo;
45  }
46 }
