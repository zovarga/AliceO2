185 inline gsl::span<const Vertex> TimeFrame::getPrimaryVertices(int tf) const
186 {
187  const int start = mROframesPV[tf];
188  const int stop = tf >= mNrof - 1 ? mNrof : tf + 1;
189  return {&mPrimaryVertices[start], static_cast<gsl::span<const Vertex>::size_type>(mROframesPV[stop] - mROframesPV[tf])};
190 }
