60 template <typename VEC>
61 void CTFCoder::encode(VEC& buff, const CTFHelper::TFData& tfData)
62 {
63  using MD = o2::ctf::Metadata::OptStore;
64  // what to do which each field: see o2::ctd::Metadata explanation
65  constexpr MD optField[CTF::getNBlocks()] = {
66  MD::EENCODE, // BLC_bcIncROF
67  MD::EENCODE, // BLC_orbitIncROF
68  MD::EENCODE, // BLC_entriesROF
69  MD::EENCODE, // BLC_evtypeROF
70  MD::EENCODE, // BLC_pattern
71  MD::EENCODE, // BLC_deId
72  MD::EENCODE // BLC_colId
73  };
74  CTFHelper helper(tfData);
75 
76  // book output size with some margin
77  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
78  buff.resize(szIni);
79 
80  auto ec = CTF::create(buff);
81  using ECB = CTF::base;
82 
83  ec->setHeader(helper.createHeader());
84  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
85  ec->getANSHeader().majorVersion = 0;
86  ec->getANSHeader().minorVersion = 1;
87  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
88 #define ENCODEMID(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
89  // clang-format off
90  ENCODEMID(helper.begin_bcIncROF(), helper.end_bcIncROF(), CTF::BLC_bcIncROF, 0);
91  ENCODEMID(helper.begin_orbitIncROF(), helper.end_orbitIncROF(), CTF::BLC_orbitIncROF, 0);
92  ENCODEMID(helper.begin_entriesROF(), helper.end_entriesROF(), CTF::BLC_entriesROF, 0);
93  ENCODEMID(helper.begin_evtypeROF(), helper.end_evtypeROF(), CTF::BLC_evtypeROF, 0);
94 
95  ENCODEMID(helper.begin_pattern(), helper.end_pattern(), CTF::BLC_pattern, 0);
96  ENCODEMID(helper.begin_deId(), helper.end_deId(), CTF::BLC_deId, 0);
97  ENCODEMID(helper.begin_colId(), helper.end_colId(), CTF::BLC_colId, 0);
98  // clang-format on
99  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
100 }

103 template <typename VROF, typename VCOL>
104 void CTFCoder::decode(const CTF::base& ec, std::array<VROF, NEvTypes>& rofVec, std::array<VCOL, NEvTypes>& colVec)
105 {
106  auto header = ec.getHeader();
107  checkDictVersion(static_cast<const o2::ctf::CTFDictHeader&>(header));
108  ec.print(getPrefix(), mVerbosity);
109  std::vector<uint16_t> bcInc, entries, pattern;
110  std::vector<uint32_t> orbitInc;
111  std::vector<uint8_t> evType, deId, colId;
112 
113 #define DECODEMID(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
114  // clang-format off
115  DECODEMID(bcInc, CTF::BLC_bcIncROF);
116  DECODEMID(orbitInc, CTF::BLC_orbitIncROF);
117  DECODEMID(entries, CTF::BLC_entriesROF);
118  DECODEMID(evType, CTF::BLC_evtypeROF);
119 
120  DECODEMID(pattern, CTF::BLC_pattern);
121  DECODEMID(deId, CTF::BLC_deId);
122  DECODEMID(colId, CTF::BLC_colId);
123  // clang-format on
124  //
125  for (uint32_t i = 0; i < NEvTypes; i++) {
126  rofVec[i].clear();
127  colVec[i].clear();
128  rofVec[i].reserve(header.nROFs);
129  colVec[i].reserve(header.nColumns);
130  }
131 
132  uint32_t firstEntry = 0, rofCount = 0, colCount = 0, pCount = 0;
133  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
134 
135  for (uint32_t irof = 0; irof < header.nROFs; irof++) {
136  // restore ROFRecord
137  if (orbitInc[irof]) { // non-0 increment => new orbit
138  ir.bc = bcInc[irof]; // bcInc has absolute meaning
139  ir.orbit += orbitInc[irof];
140  } else {
141  ir.bc += bcInc[irof];
142  }
143  auto& cv = colVec[evType[irof]];
144  firstEntry = cv.size();
145  for (uint8_t ic = 0; ic < entries[irof]; ic++) {
146  cv.emplace_back(ColumnData{deId[colCount], colId[colCount], std::array{pattern[pCount], pattern[pCount + 1], pattern[pCount + 2], pattern[pCount + 3], pattern[pCount + 4]}});
147  pCount += 5;
148  colCount++;
149  }
150  rofVec[evType[irof]].emplace_back(ROFRecord{ir, EventType(evType[irof]), firstEntry, entries[irof]});
151  }
152  assert(colCount == header.nColumns);
153 }
