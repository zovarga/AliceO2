86 template <typename VEC>
87 void CTFCoder::encode(VEC& buff, const gsl::span<const ROFRecord>& rofRecVec, const gsl::span<const CompClusterExt>& cclusVec, const gsl::span<const unsigned char>& pattVec)
88 {
89  using MD = o2::ctf::Metadata::OptStore;
90  // what to do which each field: see o2::ctd::Metadata explanation
91  constexpr MD optField[CTF::getNBlocks()] = {
92  MD::EENCODE, //BLCfirstChipROF
93  MD::EENCODE, //BLCbcIncROF
94  MD::EENCODE, //BLCorbitIncROF
95  MD::EENCODE, //BLCnclusROF
96  MD::EENCODE, //BLCchipInc
97  MD::EENCODE, //BLCchipMul
98  MD::EENCODE, //BLCrow
99  MD::EENCODE, //BLCcolInc
100  MD::EENCODE, //BLCpattID
101  MD::EENCODE //BLCpattMap
102  };
103  CompressedClusters compCl;
104  compress(compCl, rofRecVec, cclusVec, pattVec);
105  // book output size with some margin
106  auto szIni = estimateCompressedSize(compCl);
107  buff.resize(szIni);
108 
109  auto ec = CTF::create(buff);
110  using ECB = CTF::base;
111 
112  ec->setHeader(compCl.header);
113  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
114  ec->getANSHeader().majorVersion = 0;
115  ec->getANSHeader().minorVersion = 1;
116  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
117 #define ENCODEITSMFT(part, slot, bits) CTF::get(buff.data())->encode(part, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
118  // clang-format off
119  ENCODEITSMFT(compCl.firstChipROF, CTF::BLCfirstChipROF, 0);
120  ENCODEITSMFT(compCl.bcIncROF, CTF::BLCbcIncROF, 0);
121  ENCODEITSMFT(compCl.orbitIncROF, CTF::BLCorbitIncROF, 0);
122  ENCODEITSMFT(compCl.nclusROF, CTF::BLCnclusROF, 0);
123  //
124  ENCODEITSMFT(compCl.chipInc, CTF::BLCchipInc, 0);
125  ENCODEITSMFT(compCl.chipMul, CTF::BLCchipMul, 0);
126  ENCODEITSMFT(compCl.row, CTF::BLCrow, 0);
127  ENCODEITSMFT(compCl.colInc, CTF::BLCcolInc, 0);
128  ENCODEITSMFT(compCl.pattID, CTF::BLCpattID, 0);
129  ENCODEITSMFT(compCl.pattMap, CTF::BLCpattMap, 0);
130  // clang-format on
131  //CTF::get(buff.data())->print(getPrefix());
132 }

151 template <typename VROF, typename VCLUS, typename VPAT>
152 void CTFCoder::decompress(const CompressedClusters& compCl, VROF& rofRecVec, VCLUS& cclusVec, VPAT& pattVec, const NoiseMap* noiseMap, const LookUp& clPattLookup)
153 {
154  PMatrix pmat{};
155  RowColBuff firedPixBuff{}, maskedPixBuff{};
156  rofRecVec.resize(compCl.header.nROFs);
157  cclusVec.clear();
158  cclusVec.reserve(compCl.header.nClusters);
159  pattVec.clear();
160  pattVec.reserve(compCl.header.nPatternBytes);
161  o2::InteractionRecord prevIR(compCl.header.firstBC, compCl.header.firstOrbit);
162  uint32_t clCount = 0, chipCount = 0;
163  auto pattIt = compCl.pattMap.begin();
164  auto pattItStored = pattIt;
165 
166  // >> ====== Helper functions for reclusterization after masking some pixels in decoded clusters ======
167  // clusterize the pmat matrix holding pixels of the single cluster after masking the noisy ones
168 
169  auto clusterize = [&](uint16_t chipID, int16_t row, int16_t col, int leftFired) {
170 #ifdef _ALLOW_DIAGONAL_ALPIDE_CLUSTERS_
171  const std::array<int16_t, 8> walkRow = {1, -1, 0, 0, 1, 1, -1, -1};
172  const std::array<int16_t, 8> walkCol = {0, 0, -1, 1, 1, -1, 1, 1};
173 #else
174  const std::array<int16_t, 4> walkRow = {1, -1, 0, 0};
175  const std::array<int16_t, 4> walkCol = {0, 0, -1, 1};
176 #endif
177  Clusterer::BBox bbox(chipID);
178  // check and add to new cluster seed fired pixels around ir1, ic1, return true if there are still fired pixels left
179  std::function<bool(int16_t, int16_t)> checkPixelAndNeighbours = [&](int16_t ir1, int16_t ic1) {
180  // if pixel in pmat is fired, add it to new cluster seed and adjust the BBox, decreasing the number of fired pixels left
181  auto checkPixel = [&](int16_t ir1, int16_t ic1) {
182  if (pmat[ir1][ic1]) {
183  pmat[ir1][ic1] = false;
184  uint16_t r = row + ir1 - 1, c = col + ic1 - 1;
185  firedPixBuff.emplace_back(r, c);
186  bbox.adjust(r, c);
187  leftFired--;
188  return true;
189  }
190  return false;
191  };
192  // check and add to new cluster seed fired pixels at and around ir1, ic1, return true if there are still fired pixels left
193  if (checkPixel(ir1, ic1) && leftFired) {
194  uint16_t iw = 0;
195  while (checkPixelAndNeighbours(ir1 + walkRow[iw], ic1 + walkCol[iw]) && ++iw < walkRow.size()) {
196  }
197  }
198  return leftFired;
199  };
200  // true will be returned if after incremental check of neighbours fired pixels are still left
201 
202  firedPixBuff.clear(); // start new cluster seed
203  for (auto s : maskedPixBuff) { // we start checking from the holes remaining from the masked pixels
204  uint16_t iw = 0;
205  do {
206  checkPixelAndNeighbours(s.getRowDirect() + walkRow[iw], s.getCol() + walkCol[iw]);
207  if (!firedPixBuff.empty()) {
208  bbox.chipID = chipID;
209  Clusterer::streamCluster(firedPixBuff, nullptr, bbox, clPattLookup, &cclusVec, &pattVec, nullptr, 0);
210  firedPixBuff.clear();
211  bbox.clear();
212  }
213  } while (leftFired && ++iw < walkRow.size());
214  if (!leftFired) {
215  break;
216  }
217  }
218  };
219 
220  auto reclusterize = [&]() {
221  auto clus = cclusVec.back(); // original newly added cluster
222  // acquire pattern
223  o2::itsmft::ClusterPattern patt;
224  auto pattItPrev = pattIt;
225  maskedPixBuff.clear();
226  int rowRef = clus.getRow(), colRef = clus.getCol();
227  if (clPattLookup.size() == 0 && clus.getPatternID() != o2::itsmft::CompCluster::InvalidPatternID) {
228  throw std::runtime_error("Clusters contain pattern IDs, but no dictionary is provided...");
229  }
230  if (clus.getPatternID() == o2::itsmft::CompCluster::InvalidPatternID) {
231  patt.acquirePattern(pattIt);
232  } else if (clPattLookup.isGroup(clus.getPatternID())) {
233  patt.acquirePattern(pattIt);
234  float xCOG = 0, zCOG = 0;
235  patt.getCOG(xCOG, zCOG); // for grouped patterns the reference pixel is at COG
236  rowRef -= round(xCOG);
237  colRef -= round(zCOG);
238  } else {
239  patt = clPattLookup.getPattern(clus.getPatternID());
240  }
241  int rowSpan = patt.getRowSpan(), colSpan = patt.getColumnSpan(), nMasked = 0;
242  if (rowSpan == 1 && colSpan == 1) { // easy case: 1 pixel cluster
243  if (noiseMap->isNoisy(clus.getChipID(), rowRef, colRef)) { // just kill the cluster
244  std::copy(pattItStored, pattItPrev, back_inserter(pattVec)); // save patterns from after last saved to the one before killing this
245  pattItStored = pattIt; // advance to the head of the pattern iterator
246  cclusVec.pop_back();
247  }
248  // otherwise do nothing: cluster was already added, eventual patterns will be copied in large block at next modified cluster writing
249  } else {
250  int rowSpan = patt.getRowSpan(), colSpan = patt.getColumnSpan(), nMasked = 0, nPixels = 0; // apply noise and fill hits matrix
251  for (int ir = 0; ir < rowSpan; ir++) {
252  int row = rowRef + ir;
253  for (int ic = 0; ic < colSpan; ic++) {
254  if (patt.isSet(ir, ic)) {
255  if (noiseMap->isNoisy(clus.getChipID(), row, colRef + ic)) {
256  maskedPixBuff.emplace_back(ir + 1, ic + 1);
257  pmat[ir + 1][ic + 1] = false; // reset since might be left from prev cluster
258  nMasked++;
259  } else {
260  pmat[ir + 1][ic + 1] = true;
261  nPixels++;
262  }
263  } else {
264  pmat[ir + 1][ic + 1] = false; // reset since might be left from prev cluster
265  }
266  }
267  }
268 
269  if (nMasked) {
270  cclusVec.pop_back(); // remove added cluster
271  std::copy(pattItStored, pattItPrev, back_inserter(pattVec)); // save patterns from after last saved to the one before killing this
272  pattItStored = pattIt; // advance to the head of the pattern iterator
273  if (nPixels) { // need to reclusterize remaining pixels
274  clusterize(clus.getChipID(), rowRef, colRef, nPixels);
275  }
276  }
277  }
278  };
279  // << ====== Helper functions for reclusterization after masking some pixels in decoded clusters ======
280 
281  for (uint32_t irof = 0; irof < compCl.header.nROFs; irof++) {
282  // restore ROFRecord
283  auto& rofRec = rofRecVec[irof];
284  if (compCl.orbitIncROF[irof]) { // new orbit
285  prevIR.bc = compCl.bcIncROF[irof]; // bcInc has absolute meaning
286  prevIR.orbit += compCl.orbitIncROF[irof];
287  } else {
288  prevIR.bc += compCl.bcIncROF[irof];
289  }
290  rofRec.setBCData(prevIR);
291  rofRec.setFirstEntry(cclusVec.size());
292 
293  // resrore chips data
294  auto chipID = compCl.firstChipROF[irof];
295  uint16_t col = 0;
296  int inChip = 0;
297  for (uint32_t icl = 0; icl < compCl.nclusROF[irof]; icl++) {
298  auto& clus = cclusVec.emplace_back();
299  if (inChip++ < compCl.chipMul[chipCount]) { // still the same chip
300  clus.setCol((col += compCl.colInc[clCount]));
301  } else { // new chip starts
302  chipID += compCl.chipInc[++chipCount];
303  inChip = 1;
304  clus.setCol((col = compCl.colInc[clCount])); // colInc has abs. col meaning
305  }
306  clus.setRow(compCl.row[clCount]);
307  clus.setPatternID(compCl.pattID[clCount]);
308  clus.setChipID(chipID);
309  if (noiseMap) { // noise masking was requested
310  reclusterize();
311  }
312  clCount++;
313  }
314  if (compCl.nclusROF[irof]) {
315  chipCount++; // since next chip for sure will be new and inChip will be 0...
316  }
317  rofRec.setNEntries(cclusVec.size() - rofRec.getFirstEntry());
318  }
319  if (noiseMap) { // reclusterization was requested
320  if (pattItStored != pattIt) { // copy unsaved patterns
321  std::copy(pattItStored, pattIt, back_inserter(pattVec));
322  }
323  } else { // copy decoded patterns as they are
324  pattVec.resize(compCl.header.nPatternBytes);
325  memcpy(pattVec.data(), compCl.pattMap.data(), compCl.header.nPatternBytes);
326  }
327  assert(chipCount == compCl.header.nChips);
328 
329  if (clCount != compCl.header.nClusters) {
330  LOG(error) << "expected " << compCl.header.nClusters << " but counted " << clCount << " in ROFRecords";
331  throw std::runtime_error("mismatch between expected and counter number of clusters");
332  }
333 }

336 template <typename VROF, typename VDIG>
337 void CTFCoder::decompress(const CompressedClusters& compCl, VROF& rofRecVec, VDIG& digVec, const NoiseMap* noiseMap, const LookUp& clPattLookup)
338 {
339  rofRecVec.resize(compCl.header.nROFs);
340  digVec.reserve(compCl.header.nClusters * 2);
341  o2::InteractionRecord prevIR(compCl.header.firstBC, compCl.header.firstOrbit);
342  uint32_t clCount = 0, chipCount = 0;
343  auto pattIt = compCl.pattMap.begin();
344  o2::itsmft::ClusterPattern patt;
345  for (uint32_t irof = 0; irof < compCl.header.nROFs; irof++) {
346  size_t chipStartNDig = digVec.size();
347  // restore ROFRecord
348  auto& rofRec = rofRecVec[irof];
349  if (compCl.orbitIncROF[irof]) { // new orbit
350  prevIR.bc = compCl.bcIncROF[irof]; // bcInc has absolute meaning
351  prevIR.orbit += compCl.orbitIncROF[irof];
352  } else {
353  prevIR.bc += compCl.bcIncROF[irof];
354  }
355  rofRec.setBCData(prevIR);
356  rofRec.setFirstEntry(digVec.size());
357 
358  // resrore chips data
359  uint16_t chipID = compCl.firstChipROF[irof], col = 0;
360  int inChip = 0;
361  for (uint32_t icl = 0; icl < compCl.nclusROF[irof]; icl++) {
362  if (inChip++ < compCl.chipMul[chipCount]) { // still the same chip
363  col += compCl.colInc[clCount];
364 #ifdef _CHECK_INCREMENTES_ // RS FIXME with current clusterization column increment can be slightly negative
365 // if (int16_t(compCl.colInc[clCount])<0) {
366 // LOG(warning) << "Negative column increment " << int16_t(compCl.colInc[clCount]) << " -> " << col << " in chip " << chipID;
367 // }
368 #endif
369  } else { // new chip starts
370  // sort digits of previous chip in col/row
371  auto added = digVec.size() - chipStartNDig;
372  if (added > 1) { // we need to sort digits in colums and in rows within a column
373  std::sort(digVec.end() - added, digVec.end(),
374  [](Digit& a, Digit& b) { return a.getColumn() < b.getColumn() || (a.getColumn() == b.getColumn() && a.getRow() < b.getRow()); });
375  }
376  chipStartNDig = digVec.size();
377  chipID += compCl.chipInc[++chipCount];
378 #ifdef _CHECK_INCREMENTES_
379  if (int16_t(compCl.chipInc[chipCount]) < 0) {
380  LOG(warning) << "Negative chip increment " << int16_t(compCl.chipInc[chipCount]) << " -> " << chipID;
381  }
382 #endif
383  inChip = 1;
384  col = compCl.colInc[clCount]; // colInc has abs. col meaning
385  }
386  uint16_t rowRef = compCl.row[clCount], colRef = col;
387  auto pattID = compCl.pattID[clCount];
388  if (pattID == o2::itsmft::CompCluster::InvalidPatternID) {
389  patt.acquirePattern(pattIt);
390  } else {
391  if (clPattLookup.size() == 0) {
392  throw std::runtime_error("Clusters contain pattern IDs, but no dictionary is provided...");
393  }
394  if (pattID == o2::itsmft::CompCluster::InvalidPatternID) {
395  patt.acquirePattern(pattIt);
396  } else if (clPattLookup.isGroup(pattID)) {
397  patt.acquirePattern(pattIt);
398  float xCOG = 0., zCOG = 0.;
399  patt.getCOG(xCOG, zCOG); // for grouped patterns the reference pixel is at COG
400  rowRef -= round(xCOG);
401  colRef -= round(zCOG);
402  } else {
403  patt = clPattLookup.getPattern(pattID);
404  }
405  }
406  clCount++;
407 
408  auto fillRowCol = [&digVec, chipID, rowRef, colRef, noiseMap](int r, int c) {
409  r += rowRef;
410  c += colRef;
411  if (noiseMap && noiseMap->isNoisy(chipID, r, c)) {
412  return;
413  }
414  digVec.emplace_back(chipID, uint16_t(r), uint16_t(c));
415  };
416  patt.process(fillRowCol);
417  }
418  auto added = digVec.size() - chipStartNDig;
419  if (added > 1) { // Last chip of the ROF: we need to sort digits in colums and in rows within a column
420  std::sort(digVec.end() - added, digVec.end(),
421  [](Digit& a, Digit& b) { return a.getColumn() < b.getColumn() || (a.getColumn() == b.getColumn() && a.getRow() < b.getRow()); });
422  }
423 
424  if (compCl.nclusROF[irof]) {
425  chipCount++; // since next chip for sure will be new and incChip will be 0...
426  }
427  rofRec.setNEntries(digVec.size() - rofRec.getFirstEntry());
428  }
429  // explicit patterns
430  assert(pattIt == compCl.pattMap.end());
431  assert(chipCount == compCl.header.nChips);
432 
433  if (clCount != compCl.header.nClusters) {
434  LOG(error) << "expected " << compCl.header.nClusters << " but counted " << clCount << " in ROFRecords";
435  throw std::runtime_error("mismatch between expected and counter number of clusters");
436  }
437 }
