60 template <typename VEC>
61 void CTFCoder::encode(VEC& buff, const gsl::span<const ROFRecord>& rofData, const gsl::span<const Digit>& digData)
62 {
63  using MD = o2::ctf::Metadata::OptStore;
64  // what to do which each field: see o2::ctd::Metadata explanation
65  constexpr MD optField[CTF::getNBlocks()] = {
66  MD::EENCODE, // BLC_bcIncROF
67  MD::EENCODE, // BLC_orbitIncROF
68  MD::EENCODE, // BLC_nDigitsROF
69  MD::EENCODE, // BLC_tfTime
70  MD::EENCODE, // BLC_nSamples
71  MD::EENCODE, // BLC_isSaturated
72  MD::EENCODE, // BLC_detID
73  MD::EENCODE, // BLC_padID
74  MD::EENCODE // BLC_ADC
75  };
76  CTFHelper helper(rofData, digData);
77 
78  // book output size with some margin
79  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
80  buff.resize(szIni);
81 
82  auto ec = CTF::create(buff);
83  using ECB = CTF::base;
84 
85  ec->setHeader(helper.createHeader());
86  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
87  ec->getANSHeader().majorVersion = 0;
88  ec->getANSHeader().minorVersion = 1;
89  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
90 #define ENCODEMCH(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
91  // clang-format off
92  ENCODEMCH(helper.begin_bcIncROF(), helper.end_bcIncROF(), CTF::BLC_bcIncROF, 0);
93  ENCODEMCH(helper.begin_orbitIncROF(), helper.end_orbitIncROF(), CTF::BLC_orbitIncROF, 0);
94  ENCODEMCH(helper.begin_nDigitsROF(), helper.end_nDigitsROF(), CTF::BLC_nDigitsROF, 0);
95 
96  ENCODEMCH(helper.begin_tfTime(), helper.end_tfTime(), CTF::BLC_tfTime, 0);
97  ENCODEMCH(helper.begin_nSamples(), helper.end_nSamples(), CTF::BLC_nSamples, 0);
98  ENCODEMCH(helper.begin_isSaturated(), helper.end_isSaturated(), CTF::BLC_isSaturated, 0);
99  ENCODEMCH(helper.begin_detID(), helper.end_detID(), CTF::BLC_detID, 0);
100  ENCODEMCH(helper.begin_padID(), helper.end_padID(), CTF::BLC_padID, 0);
101  ENCODEMCH(helper.begin_ADC() , helper.end_ADC(), CTF::BLC_ADC, 0);
102  // clang-format on
103  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
104 }

107 template <typename VROF, typename VCOL>
108 void CTFCoder::decode(const CTF::base& ec, VROF& rofVec, VCOL& digVec)
109 {
110  auto header = ec.getHeader();
111  checkDictVersion(static_cast<const o2::ctf::CTFDictHeader&>(header));
112  ec.print(getPrefix(), mVerbosity);
113 
114  std::vector<uint16_t> bcInc, nSamples;
115  std::vector<uint32_t> orbitInc, ADC, nDigits;
116  std::vector<int32_t> tfTime;
117  std::vector<int16_t> detID, padID;
118  std::vector<uint8_t> isSaturated;
119 
120 #define DECODEMCH(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
121  // clang-format off
122  DECODEMCH(bcInc, CTF::BLC_bcIncROF);
123  DECODEMCH(orbitInc, CTF::BLC_orbitIncROF);
124  DECODEMCH(nDigits, CTF::BLC_nDigitsROF);
125 
126  DECODEMCH(tfTime, CTF::BLC_tfTime);
127  DECODEMCH(nSamples, CTF::BLC_nSamples);
128  DECODEMCH(isSaturated, CTF::BLC_isSaturated);
129  DECODEMCH(detID, CTF::BLC_detID);
130  DECODEMCH(padID, CTF::BLC_padID);
131  DECODEMCH(ADC, CTF::BLC_ADC);
132  // clang-format on
133  //
134  rofVec.clear();
135  digVec.clear();
136  rofVec.reserve(header.nROFs);
137  digVec.reserve(header.nDigits);
138 
139  uint32_t firstEntry = 0, rofCount = 0, digCount = 0;
140  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
141 
142  for (uint32_t irof = 0; irof < header.nROFs; irof++) {
143  // restore ROFRecord
144  if (orbitInc[irof]) { // non-0 increment => new orbit
145  ir.bc = bcInc[irof]; // bcInc has absolute meaning
146  ir.orbit += orbitInc[irof];
147  } else {
148  ir.bc += bcInc[irof];
149  }
150 
151  firstEntry = digVec.size();
152  for (auto ic = 0; ic < nDigits[irof]; ic++) {
153  digVec.emplace_back(Digit{detID[digCount], padID[digCount], ADC[digCount], tfTime[digCount], nSamples[digCount]});
154  digVec.back().setSaturated(isSaturated[digCount]);
155  digCount++;
156  }
157  rofVec.emplace_back(ROFRecord{ir, int(firstEntry), static_cast<int>(nDigits[irof])});
158  }
159  assert(rofVec.size() == header.nROFs);
160  assert(digCount == header.nDigits);
161 }
