26 //__________________________________________________________
27 void MFTAssessment::init(bool finalizeAnalysis)
28 {
29  mFinalizeAnalysis = finalizeAnalysis;
30  createHistos();
31  //get geometry
32  o2::base::GeometryManager::loadGeometry("", true);
33 
34  //load the cluster dictionary
35  std::string dictPath = o2::itsmft::ClustererParam<o2::detectors::DetID::MFT>::Instance().dictFilePath;
36  std::string dictFile = o2::base::DetectorNameConf::getAlpideClusterDictionaryFileName(o2::detectors::DetID::MFT, dictPath);
37  if (o2::utils::Str::pathExists(dictFile)) {
38  mDictionary.readBinaryFile(dictFile);
39  LOG(info) << "MFTAssessment running with a provided dictionary: " << dictFile;
40  printf("DictPath is %s\n", dictPath.c_str());
41  } else {
42  LOG(fatal) << "Dictionary " << dictFile << " is absent, MFTAssessment expects cluster patterns";
43  }
44 
45  mUnusedChips.fill(true);
46 }

49 void MFTAssessment::reset()
50 {
51 
52  mTrackNumberOfClusters->Reset();
53  mCATrackNumberOfClusters->Reset();
54  mLTFTrackNumberOfClusters->Reset();
55  mTrackOnvQPt->Reset();
56  mTrackChi2->Reset();
57  mTrackCharge->Reset();
58  mTrackPhi->Reset();
59  mPositiveTrackPhi->Reset();
60  mNegativeTrackPhi->Reset();
61  mTrackEta->Reset();
62 
63  mMFTClsZ->Reset();
64  mMFTClsOfTracksZ->Reset();
65 
66  mUnusedChips.fill(true);
67  mNumberTFs = 0;
68 
69  for (auto nMFTLayer = 0; nMFTLayer < 10; nMFTLayer++) {
70  mMFTClsXYinLayer[nMFTLayer]->Reset();
71  mMFTClsOfTracksXYinLayer[nMFTLayer]->Reset();
72  }
73 
74  for (auto nMFTDisk = 0; nMFTDisk < 5; nMFTDisk++) {
75  mMFTClsXYRedundantInDisk[nMFTDisk]->Reset();
76  }
77 
78  for (auto minNClusters : sMinNClustersList) {
79  auto nHisto = minNClusters - sMinNClustersList[0];
80  mTrackEtaNCls[nHisto]->Reset();
81  mTrackPhiNCls[nHisto]->Reset();
82  mTrackXYNCls[nHisto]->Reset();
83  mTrackEtaPhiNCls[nHisto]->Reset();
84  }

86  mLTFTrackEta->Reset();
87  mTrackTanl->Reset();
88 
89  mTrackROFNEntries->Reset();
90  mClusterROFNEntries->Reset();
91 
92  mClusterSensorIndex->Reset();
93  mClusterPatternIndex->Reset();
94 
95  if (mUseMC) {
96  mMFTTrackables.clear();
97 
98  mHistPhiRecVsPhiGen->Reset();
99  mHistEtaRecVsEtaGen->Reset();
100  for (int trackType = 0; trackType < kNumberOfTrackTypes; trackType++) {
101  mHistPhiVsEta[trackType]->Reset();
102  mHistPtVsEta[trackType]->Reset();
103  mHistPhiVsPt[trackType]->Reset();
104  mHistZvtxVsEta[trackType]->Reset();
105  if (trackType == kGen || trackType == kTrackable) {
106  mHistRVsZ[trackType]->Reset();
107  }
108  }
109 
110  auto h = mChargeMatchEff->GetCopyTotalHisto();
111  h->Reset();
112  mChargeMatchEff->SetTotalHistogram(*h, "");
113  mChargeMatchEff->SetPassedHistogram(*h, "");
114 
115  for (auto& h : mTH3Histos) {
116  h->Reset();
117  }
118  }

292 //__________________________________________________________
293 void MFTAssessment::runASyncQC(o2::framework::ProcessingContext& ctx)
294 {
295  mNumberTFs++; //TF Counter
296 
297  // get tracks
298  mMFTTracks = ctx.inputs().get<gsl::span<o2::mft::TrackMFT>>("tracks");
299  mMFTTracksROF = ctx.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("tracksrofs");
300  mMFTTrackClusIdx = ctx.inputs().get<gsl::span<int>>("trackClIdx");
301 
302  // get clusters
303  mMFTClusters = ctx.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
304  mMFTClustersROF = ctx.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("clustersrofs");
305  mMFTClusterPatterns = ctx.inputs().get<gsl::span<unsigned char>>("patterns");
306  pattIt = mMFTClusterPatterns.begin();
307  mMFTClustersGlobal.clear();
308  mMFTClustersGlobal.reserve(mMFTClusters.size());
309  o2::mft::ioutils::convertCompactClusters(mMFTClusters, pattIt, mMFTClustersGlobal, mDictionary);
310 
311  if (mUseMC) {
312  // get labels
313  mMFTClusterLabels = ctx.inputs().get<const dataformats::MCTruthContainer<MCCompLabel>*>("clslabels");
314  mMFTTrackLabels = ctx.inputs().get<gsl::span<MCCompLabel>>("trklabels");
315  }
316 
317  // Fill the clusters histograms
318  for (const auto& rof : mMFTClustersROF) {
319  mClusterROFNEntries->Fill(rof.getNEntries());
320  float seconds = orbitToSeconds(rof.getBCData().orbit, mRefOrbit) + rof.getBCData().bc * o2::constants::lhc::LHCBunchSpacingNS * 1e-9;
321  mNOfClustersTime->Fill(seconds, rof.getNEntries());
322  }
323 
324  for (int icls = 0; icls < mMFTClusters.size(); ++icls) {
325  auto const oneCluster = mMFTClusters[icls];
326  auto const globalCluster = mMFTClustersGlobal[icls];
327 
328  mClusterSensorIndex->Fill(oneCluster.getSensorID());
329  mClusterPatternIndex->Fill(oneCluster.getPatternID());
330 
331  mMFTClsZ->Fill(globalCluster.getZ());
332 
333  auto clsLayer = mMFTChipMapper.chip2Layer(oneCluster.getChipID());
334  mMFTClsXYinLayer[clsLayer]->Fill(globalCluster.getX(), globalCluster.getY());
335 
336  mUnusedChips[oneCluster.getChipID()] = false; //this chipID is used
337  }
338 
339  // fill the tracks histogram
340 
341  for (const auto& rof : mMFTTracksROF) {
342  mTrackROFNEntries->Fill(rof.getNEntries());
343  mTracksBC->Fill(rof.getBCData().bc, rof.getNEntries());
344  float seconds = orbitToSeconds(rof.getBCData().orbit, mRefOrbit) + rof.getBCData().bc * o2::constants::lhc::LHCBunchSpacingNS * 1e-9;
345  mNOfTracksTime->Fill(seconds, rof.getNEntries());
346  }
347 
348  std::array<std::array<int, 2>, 5> clsEntriesForRedundancy;
349 
350  for (auto& oneTrack : mMFTTracks) {
351  mTrackNumberOfClusters->Fill(oneTrack.getNumberOfPoints());
352  mTrackChi2->Fill(oneTrack.getTrackChi2());
353  mTrackCharge->Fill(oneTrack.getCharge());
354  mTrackPhi->Fill(oneTrack.getPhi());
355  mTrackEta->Fill(oneTrack.getEta());
356  mTrackTanl->Fill(oneTrack.getTanl());
357 
358  for (auto idisk = 0; idisk < 5; idisk++) {
359  clsEntriesForRedundancy[idisk] = {-1, -1};
360  }
361 
362  auto ncls = oneTrack.getNumberOfPoints();
363  auto offset = oneTrack.getExternalClusterIndexOffset();
364 
365  for (int icls = 0; icls < ncls; ++icls) //cluster loop
366  {
367 
368  auto clsEntry = mMFTTrackClusIdx[offset + icls];
369  auto globalCluster = mMFTClustersGlobal[clsEntry];
370 
371  mMFTClsOfTracksZ->Fill(globalCluster.getZ());
372 
373  auto layer = mMFTMapping.ChipID2Layer[globalCluster.getSensorID()];
374 
375  mMFTClsOfTracksXYinLayer[layer]->Fill(globalCluster.getX(), globalCluster.getY());
376 
377  int clsMFTdiskID = layer / 2;
378 
379  if (clsEntriesForRedundancy[clsMFTdiskID][0] != -1) {
380  clsEntriesForRedundancy[clsMFTdiskID][1] = clsEntry;
381  } else {
382  clsEntriesForRedundancy[clsMFTdiskID][0] = clsEntry;
383  }
384  }
385 
386  for (auto idisk = 0; idisk < 5; idisk++) {
387  if ((clsEntriesForRedundancy[idisk][0] != -1) && (clsEntriesForRedundancy[idisk][1] != -1)) {
388  auto globalCluster1 = mMFTClustersGlobal[clsEntriesForRedundancy[idisk][0]];
389 
390  mMFTClsXYRedundantInDisk[idisk]->Fill(globalCluster1.getX(), globalCluster1.getY());
391  }
392  }
393 
394  for (auto minNClusters : sMinNClustersList) {
395  if (oneTrack.getNumberOfPoints() >= minNClusters) {
396  mTrackEtaNCls[minNClusters - sMinNClustersList[0]]->Fill(oneTrack.getEta());
397  mTrackPhiNCls[minNClusters - sMinNClustersList[0]]->Fill(oneTrack.getPhi());
398  mTrackXYNCls[minNClusters - sMinNClustersList[0]]->Fill(oneTrack.getX(), oneTrack.getY());
399  mTrackEtaPhiNCls[minNClusters - sMinNClustersList[0]]->Fill(oneTrack.getEta(), oneTrack.getPhi());
400  }
401  }
402 
403  if (oneTrack.getCharge() == +1) {
404  mPositiveTrackPhi->Fill(oneTrack.getPhi());
405  mTrackOnvQPt->Fill(1 / oneTrack.getPt());
406  }
407 
408  if (oneTrack.getCharge() == -1) {
409  mNegativeTrackPhi->Fill(oneTrack.getPhi());
410  mTrackOnvQPt->Fill(-1 / oneTrack.getPt());
411  }
412 
413  if (oneTrack.isCA()) {
414  mCATrackNumberOfClusters->Fill(oneTrack.getNumberOfPoints());
415  mCATrackEta->Fill(oneTrack.getEta());
416  }
417  if (oneTrack.isLTF()) {
418  mLTFTrackNumberOfClusters->Fill(oneTrack.getNumberOfPoints());
419  mLTFTrackEta->Fill(oneTrack.getEta());
420  }
421  }
422 }

424 //__________________________________________________________
425 void MFTAssessment::processGeneratedTracks()
426 {
427  for (auto src = 0; src < mcReader.getNSources(); src++) {
428  for (Int_t event = 0; event < mcReader.getNEvents(src); event++) {
429  const auto& mcTracks = mcReader.getTracks(src, event);
430  for (const auto& mcParticle : mcTracks) {
431  auto evh = mcReader.getMCEventHeader(src, event);

500 //__________________________________________________________
501 void MFTAssessment::processRecoAndTrueTracks()
502 {
503  auto trkId = 0;
504  for (auto mftTrack : mMFTTracks) {
505  auto pt_Rec = mftTrack.getPt();
506  auto invQPt_Rec = mftTrack.getInvQPt();
507  auto invQPt_Seed = mftTrack.getInvQPtSeed();
508  auto eta_Rec = std::abs(mftTrack.getEta());
509  auto phi_Rec = mftTrack.getPhi();
510  auto nClusters = mftTrack.getNumberOfPoints();
511  auto Chi2_Rec = mftTrack.getTrackChi2();
512  int Q_Rec = mftTrack.getCharge();
513 
514  auto TrackType = kReco;
515  mHistPtVsEta[TrackType]->Fill(eta_Rec, pt_Rec);
516  mHistPhiVsEta[TrackType]->Fill(eta_Rec, phi_Rec);
517  mHistPhiVsPt[TrackType]->Fill(pt_Rec, phi_Rec);
518 
519  auto trackLabel = mMFTTrackLabels[trkId];
520  if (trackLabel.isCorrect()) {
521  TrackType = kRecoTrue;
522  auto evH = mcReader.getMCEventHeader(trackLabel.getSourceID(), trackLabel.getEventID());
523  auto zVtx = evH.GetZ();
524 
525  auto const* mcParticle = mcReader.getTrack(trackLabel);
526  auto etaGen = std::abs(mcParticle->GetEta());
527  auto phiGen = TMath::ATan2(mcParticle->Py(), mcParticle->Px());
528  auto ptGen = mcParticle->GetPt();
529  auto vxGen = mcParticle->GetStartVertexCoordinatesX();
530  auto vyGen = mcParticle->GetStartVertexCoordinatesY();
531  auto vzGen = mcParticle->GetStartVertexCoordinatesZ();
532  auto tanlGen = mcParticle->Pz() / mcParticle->GetPt();
533 
534  auto pdgcode_MC = mcParticle->GetPdgCode();
535  int Q_Gen;
536  if (TDatabasePDG::Instance()->GetParticle(pdgcode_MC)) {
537  Q_Gen = TDatabasePDG::Instance()->GetParticle(pdgcode_MC)->Charge() / 3;
538  } else {
539  continue;
540  }
541  auto invQPtGen = 1.0 * Q_Gen / ptGen;
542  mftTrack.propagateToZ(vzGen, mBz);
543 
544  // Residuals at vertex
545  auto x_res = mftTrack.getX() - vxGen;
546  auto y_res = mftTrack.getY() - vyGen;
547  auto eta_res = mftTrack.getEta() - etaGen;
548  auto phi_res = mftTrack.getPhi() - phiGen;
549  auto tanl_res = mftTrack.getTanl() - tanlGen;
550  auto invQPt_res = invQPt_Rec - invQPtGen;
551  mHistPtVsEta[TrackType]->Fill(eta_Rec, pt_Rec);
552  mHistPhiVsEta[TrackType]->Fill(eta_Rec, phi_Rec);
553  mHistPhiVsPt[TrackType]->Fill(pt_Rec, phi_Rec);
554  mHistZvtxVsEta[TrackType]->Fill(eta_Rec, zVtx);
555 
556  mHistPhiRecVsPhiGen->Fill(phiGen, phi_Rec);
557  mHistEtaRecVsEtaGen->Fill(etaGen, eta_Rec);
558 
560  auto d_Charge = Q_Rec - Q_Gen;
561  mChargeMatchEff->Fill(!d_Charge, ptGen);
562 
563  mTH3Histos[kTH3TrackDeltaXVertexPtEta]->Fill(ptGen, etaGen, 1e4 * x_res);
564  mTH3Histos[kTH3TrackDeltaYVertexPtEta]->Fill(ptGen, etaGen, 1e4 * y_res);
565  mTH3Histos[kTH3TrackDeltaXDeltaYEta]->Fill(etaGen, 1e4 * x_res, 1e4 * y_res);
566  mTH3Histos[kTH3TrackDeltaXDeltaYPt]->Fill(ptGen, 1e4 * x_res, 1e4 * y_res);
567  mTH3Histos[kTH3TrackXPullPtEta]->Fill(ptGen, etaGen, x_res / sqrt(mftTrack.getCovariances()(0, 0)));
568  mTH3Histos[kTH3TrackYPullPtEta]->Fill(ptGen, etaGen, y_res / sqrt(mftTrack.getCovariances()(1, 1)));
569  mTH3Histos[kTH3TrackPhiPullPtEta]->Fill(ptGen, etaGen, phi_res / sqrt(mftTrack.getCovariances()(2, 2)));
570  mTH3Histos[kTH3TrackTanlPullPtEta]->Fill(ptGen, etaGen, tanl_res / sqrt(mftTrack.getCovariances()(3, 3)));
571  mTH3Histos[kTH3TrackInvQPtPullPtEta]->Fill(ptGen, etaGen, invQPt_res / sqrt(mftTrack.getCovariances()(4, 4)));
572  mTH3Histos[kTH3TrackInvQPtResolutionPtEta]->Fill(ptGen, etaGen, (invQPt_Rec - invQPtGen) / invQPtGen);
573  mTH3Histos[kTH3TrackInvQPtResSeedPtEta]->Fill(ptGen, etaGen, (invQPt_Seed - invQPtGen) / invQPtGen);
574  mTH3Histos[kTH3TrackReducedChi2PtEta]->Fill(ptGen, etaGen, Chi2_Rec / (2 * nClusters - 5)); // 5: number of fitting parameters
575  }
576  trkId++;
577  }
578 }
