190 inline void PVertexer::applyConstraint(VertexSeed& vtxSeed) const
191 {
192  // impose meanVertex constraint, i.e. account terms
193  // (V_i-Constrain_i)^2/sig2constr_i for i=X,Y in the fit chi2 definition
194  vtxSeed.cxx += mXYConstraintInvErr[0];
195  vtxSeed.cxy += mXYConstraintInvErr[1];
196  vtxSeed.cyy += mXYConstraintInvErr[2];
197  vtxSeed.cx0 += mXYConstraintInvErr[0] * mMeanVertex.getX() + mXYConstraintInvErr[1] * mMeanVertex.getY();
198  vtxSeed.cy0 += mXYConstraintInvErr[1] * mMeanVertex.getX() + mXYConstraintInvErr[2] * mMeanVertex.getY();
199 }

202 inline bool PVertexer::upscaleSigma(VertexSeed& vtxSeed) const
203 {
204  // scale upward the scaleSigma2 if needes
205  if (vtxSeed.scaleSigma2 < mPVParams->maxScale2) {
206  auto s = vtxSeed.scaleSigma2 * mPVParams->upscaleFactor;
207  vtxSeed.setScale(s > mPVParams->maxScale2 ? mPVParams->maxScale2 : s, mTukey2I);
208  return true;
209  }
210  return false;
211 }

214 template <typename TR>
215 void PVertexer::createTracksPool(const TR& tracks, gsl::span<const o2d::GlobalTrackID> gids)
216 {
217  // create pull of all candidate tracks in a global array ordered in time
218  mTracksPool.clear();
219  auto ntGlo = tracks.size();
220  std::vector<int> sortedTrackID(ntGlo);
221  mTracksPool.reserve(ntGlo);
222  std::iota(sortedTrackID.begin(), sortedTrackID.end(), 0);
223  std::sort(sortedTrackID.begin(), sortedTrackID.end(), [&tracks](int i, int j) {
224  return tracks[i].timeEst.getTimeStamp() < tracks[j].timeEst.getTimeStamp();
225  });
226 
227  // check all containers
228  float vtxErr2 = 0.5 * (mMeanVertex.getSigmaX2() + mMeanVertex.getSigmaY2());
229  o2d::DCA dca;
230 
231  for (uint32_t i = 0; i < ntGlo; i++) {
232  int id = sortedTrackID[i];
233  o2::track::TrackParCov trc = tracks[id];
234  if (!relateTrackToMeanVertex(trc, vtxErr2)) {
235  continue;
236  }
237  auto& tvf = mTracksPool.emplace_back(trc, tracks[id].getTimeMUS(), id, gids[id], mPVParams->addTimeSigma2, mPVParams->addZSigma2);
238  }
239 
240  if (mTracksPool.empty()) {
241  return;
242  }
243  //
244  auto tMin = mTracksPool.front().timeEst.getTimeStamp();
245  auto tMax = mTracksPool.back().timeEst.getTimeStamp();
246 }

259 template <typename TR>
260 bool PVertexer::prepareVertexRefit(const TR& tracks, const o2d::VertexBase& vtxSeed)
261 {
262  // Create pull of all tracks for the refitting starting from the tracks of already found vertex.
263  // This method should be called once before calling refitVertex method for given vtxSeed (which can be
264  // called multiple time with different useTrack request)
265  mTracksPool.clear();
266  mTracksPool.reserve(tracks.size());
267  for (uint32_t i = 0; i < tracks.size(); i++) {
268  o2::track::TrackParCov trc = tracks[i];
269  if (!relateTrackToVertex(trc, vtxSeed)) {
270  continue;
271  }
272  auto& tvf = mTracksPool.emplace_back(trc, TimeEst{0.f, 1.f}, i, GTrackID{}, 1., mPVParams->addZSigma2);
273  tvf.entry = i;
274  }
275  if (int(mTracksPool.size()) < mPVParams->minTracksPerVtx) {
276  return false;
277  }
278  mRefitTrackIDs.resize(tracks.size());
279  std::iota(mRefitTrackIDs.begin(), mRefitTrackIDs.end(), 0);
280  mVtxRefitOrig = vtxSeed;
281  return true;
282 }
