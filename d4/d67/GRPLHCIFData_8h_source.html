114 inline float GRPLHCIFData::getBeamZoverA(beamDirection b) const
115 {
116  // Z/A of beam 0 or 1
117  int a = getBeamA(b);
118  return a ? getBeamZ(b) / static_cast<float>(a) : 0.f;
