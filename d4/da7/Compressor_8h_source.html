40  inline bool run()
41  {
42  rewind();
43  if (mDecoderCONET) {
44  mDecoderPointerMax = reinterpret_cast<const uint32_t*>(mDecoderBuffer + mDecoderBufferSize);
45  while (mDecoderPointer < mDecoderPointerMax) {
46  mEventCounter++;
47  processDRM();
48  if (mDecoderFatal) {
49  mFatalCounter++;
50  }
51  if (mDecoderError) {
52  mErrorCounter++;
53  }
54  }
55  return false;
56  }
57  while (!processHBF()) {
58  ;
59  }
60  return false;
61  };
