46 auto Mapping::addPad(MpDE& de, const mapping::Segmentation& segmentation)
47 {
49  return [&de, &segmentation](int padID) {
50  MpPad& pad = de.pads[padID];
51  double padX = segmentation.padPositionX(padID);
52  double padY = segmentation.padPositionY(padID);
53  double padSizeX = segmentation.padSizeX(padID);
54  double padSizeY = segmentation.padSizeY(padID);
55  pad.area[0][0] = padX - padSizeX / 2.;
56  pad.area[0][1] = padX + padSizeX / 2.;
57  pad.area[1][0] = padY - padSizeY / 2.;
58  pad.area[1][1] = padY + padSizeY / 2.;
59  pad.nNeighbours = 0;
60  segmentation.forEachNeighbouringPad(padID, addNeighbour(pad));
61  };
62 }

65 std::vector<std::unique_ptr<Mapping::MpDE>> Mapping::createMapping()
66 {
68 
69  std::vector<std::unique_ptr<MpDE>> detectionElements{};
70 
71  // create the internal mapping for each DE
72  mapping::forEachDetectionElement([&detectionElements](int deID) {
73  auto& segmentation = mapping::segmentation(deID);
74  detectionElements.push_back(std::make_unique<MpDE>());
75  MpDE& de(*(detectionElements.back()));
76  de.uid = deID;
77  de.nPads[0] = segmentation.bending().nofPads();
78  de.nPads[1] = segmentation.nonBending().nofPads();
79  de.pads = std::make_unique<MpPad[]>(de.nPads[0] + de.nPads[1]);
80  segmentation.forEachPad(addPad(de, segmentation));
81  });
82 
83  return detectionElements;
84 }

109 bool Mapping::areOverlappingExcludeCorners(float area1[2][2], float area2[2][2])
110 {
112 
113  // precision in cm: positive = increase pad size / negative = decrease pad size
114  constexpr float precision = 1.e-4;
115 
116  if (areOverlapping(area1, area2, precision)) {
117  for (int ip1 = 0; ip1 < 2; ++ip1) {
118  for (int ip2 = 0; ip2 < 2; ++ip2) {
119  if (TMath::Abs(area1[0][ip1] - area2[0][1 - ip1]) < precision &&
120  TMath::Abs(area1[1][ip2] - area2[1][1 - ip2]) < precision) {
121  return false;
122  }
123  }
124  }
125  return true;
126  }
127 
128  return false;
