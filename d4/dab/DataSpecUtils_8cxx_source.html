256 bool DataSpecUtils::match(OutputSpec const& left, OutputSpec const& right)
257 {
258  if (auto leftConcrete = std::get_if<ConcreteDataMatcher>(&left.matcher)) {
259  return match(right, *leftConcrete);
260  } else if (auto rightConcrete = std::get_if<ConcreteDataMatcher>(&right.matcher)) {
261  return match(left, *rightConcrete);
262  } else {
263  // both sides are ConcreteDataTypeMatcher without subspecification, we simply specify 0
264  // this is ignored in the mathing since also left hand object is ConcreteDataTypeMatcher
265  ConcreteDataTypeMatcher dataType = DataSpecUtils::asConcreteDataTypeMatcher(right);
266  return match(left, dataType.origin, dataType.description, 0);
267  }
268 }

307 struct MatcherInfo {
308  header::DataOrigin origin = header::gDataOriginInvalid; // Whether or not we found an origins (should be a bad query!)
309  header::DataDescription description = header::gDataDescriptionInvalid; // Whether or not we found a description
310  header::DataHeader::SubSpecificationType subSpec = 0; // Whether or not we found a description
311  bool hasOrigin = false;
312  bool hasDescription = false;
313  bool hasSubSpec = false;
314  bool hasUniqueOrigin = false; // Whether the matcher involves a unique origin
315  bool hasUniqueDescription = false; // Whether the matcher involves a unique origin
316  bool hasUniqueSubSpec = false;
317  bool hasError = false;
318 };
319 
320 MatcherInfo extractMatcherInfo(DataDescriptorMatcher const& top)
321 {
322  using namespace data_matcher;
323  using ops = DataDescriptorMatcher::Op;
324 
325  MatcherInfo state;
326  auto nodeWalker = overloaded{
327  [&state](EdgeActions::EnterNode action) {
328  if (state.hasError) {
329  return VisitNone;
330  }
331  // For now we do not support extracting a type from things
332  // which have an OR, so we reset all the uniqueness attributes.
333  if (action.node->getOp() == ops::Or || action.node->getOp() == ops::Xor) {
334  state.hasError = true;
335  return VisitNone;
336  }
337  if (action.node->getOp() == ops::Just) {
338  return VisitLeft;
339  }
340  return VisitBoth;
341  },
342  [](auto) { return VisitNone; }};
343 
344  auto leafWalker = overloaded{
345  [&state](OriginValueMatcher const& valueMatcher) {
346  // FIXME: If we found already more than one data origin, it means
347  // we are ANDing two incompatible origins. Until we support OR,
348  // this is an error.
349  // In case we find a ContextRef, it means we have
350  // a wildcard, so there is not a unique origin.
351  if (state.hasOrigin) {
352  state.hasError = false;
353  return;
354  }
355  state.hasOrigin = true;
356 
357  valueMatcher.visit(overloaded{
358  [&state](std::string const& s) {
359  strncpy(state.origin.str, s.data(), 4);
360  state.hasUniqueOrigin = true;
361  },
362  [&state](auto) { state.hasUniqueOrigin = false; }});
363  },
364  [&state](DescriptionValueMatcher const& valueMatcher) {
365  if (state.hasDescription) {
366  state.hasError = true;
367  return;
368  }
369  state.hasDescription = true;
370  valueMatcher.visit(overloaded{
371  [&state](std::string const& s) {
372  strncpy(state.description.str, s.data(), 16);
373  state.hasUniqueDescription = true;
374  },
375  [&state](auto) { state.hasUniqueDescription = false; }});
376  },
377  [&state](SubSpecificationTypeValueMatcher const& valueMatcher) {
378  if (state.hasSubSpec) {
379  state.hasError = true;
380  return;
381  }
382  state.hasSubSpec = true;
383  valueMatcher.visit(overloaded{
384  [&state](uint32_t const& data) {
385  state.subSpec = data;
386  state.hasUniqueSubSpec = true;
387  },
388  [&state](auto) { state.hasUniqueSubSpec = false; }});
389  },
390  [](auto t) {}};
391  DataMatcherWalker::walk(top, nodeWalker, leafWalker);
392  return state;
393 }

395 ConcreteDataMatcher DataSpecUtils::asConcreteDataMatcher(InputSpec const& spec)
396 {
397  return std::visit(overloaded{[](ConcreteDataMatcher const& concrete) {
398  return concrete;
399  },
400  [&binding = spec.binding](DataDescriptorMatcher const& matcher) {
401  auto info = extractMatcherInfo(matcher);
402  if (info.hasOrigin && info.hasUniqueOrigin &&
403  info.hasDescription && info.hasDescription &&
404  info.hasSubSpec && info.hasUniqueSubSpec) {
405  return ConcreteDataMatcher{info.origin, info.description, info.subSpec};
406  }
407  throw std::runtime_error("Cannot convert " + binding + " to ConcreteDataMatcher");
408  }},
409  spec.matcher);
410 }

577 InputSpec DataSpecUtils::fromMetadataString(std::string s)
578 {
579  std::regex word_regex("(\\w+)");
580  auto words = std::sregex_iterator(s.begin(), s.end(), word_regex);
581  if (std::distance(words, std::sregex_iterator()) != 3) {
582  throw runtime_error_f("Malformed input spec metadata: %s", s.c_str());
583  }
584  std::vector<std::string> data;
585  for (auto i = words; i != std::sregex_iterator(); ++i) {
586  data.emplace_back(i->str());
587  }
588  char origin[4];
589  char description[16];
590  std::memcpy(&origin, data[1].c_str(), 4);
591  std::memcpy(&description, data[2].c_str(), 16);
592  return InputSpec{data[0], header::DataOrigin{origin}, header::DataDescription{description}};
593 }

692 {

695  // amend entry
696  auto& entryMetadata = locate->metadata;
697  entryMetadata.insert(entryMetadata.end(), input.metadata.begin(), input.metadata.end());
698  std::sort(entryMetadata.begin(), entryMetadata.end(), [](ConfigParamSpec const& a, ConfigParamSpec const& b) { return a.name < b.name; });
699  auto new_end = std::unique(entryMetadata.begin(), entryMetadata.end(), [](ConfigParamSpec const& a, ConfigParamSpec const& b) { return a.name == b.name; });
700  entryMetadata.erase(new_end, entryMetadata.end());
701  } else {
