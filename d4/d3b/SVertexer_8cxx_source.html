32 void SVertexer::process(const o2::globaltracking::RecoContainer& recoData) // accessor to various reconstrucred data types
33 {
34  updateTimeDependentParams(); // TODO RS: strictly speaking, one should do this only in case of the CCDB objects update
35  mPVertices = recoData.getPrimaryVertices();
36  buildT2V(recoData); // build track->vertex refs from vertex->track (if other workflow will need this, consider producing a message in the VertexTrackMatcher)
37  int ntrP = mTracksPool[POS].size(), ntrN = mTracksPool[NEG].size(), iThread = 0;
38  mV0sTmp[0].clear();
39  mCascadesTmp[0].clear();
40 
41 #ifdef WITH_OPENMP
42  omp_set_num_threads(mNThreads);
43  int dynGrp = std::min(4, std::max(1, mNThreads / 2));
44 #pragma omp parallel for schedule(dynamic, dynGrp)
45 #endif
46  for (int itp = 0; itp < ntrP; itp++) {
47  auto& seedP = mTracksPool[POS][itp];
48  int firstN = mVtxFirstTrack[NEG][seedP.vBracket.getMin()];
49  if (firstN < 0) {
50  LOG(debug) << "No partner is found for pos.track " << itp << " out of " << ntrP;
51  continue;
52  }
53  for (int itn = firstN; itn < ntrN; itn++) { // start from the 1st negative track of lowest-ID vertex of positive
54  auto& seedN = mTracksPool[NEG][itn];
55  if (seedN.vBracket > seedP.vBracket) { // all vertices compatible with seedN are in future wrt that of seedP
56  LOG(debug) << "Brackets do not match";
57  break;
58  }
59 #ifdef WITH_OPENMP
60  iThread = omp_get_thread_num();
61 #endif
62  if (mSVParams->maxPVContributors < 2 && seedP.gid.isPVContributor() + seedN.gid.isPVContributor() > mSVParams->maxPVContributors) {
63  continue;
64  }
65  checkV0(seedP, seedN, itp, itn, iThread);
66  }
67  }
68 
69  // sort V0s and Cascades in vertex id
70  struct vid {
71  int thrID;
72  int entry;
73  int vtxID;
74  };
75  size_t nv0 = 0, ncsc = 0;
76  for (int i = 0; i < mNThreads; i++) {
77  nv0 += mV0sTmp[0].size();
78  ncsc += mCascadesTmp[i].size();
79  }
80  std::vector<vid> v0SortID, cascSortID;
81  v0SortID.reserve(nv0);
82  cascSortID.reserve(ncsc);
83  for (int i = 0; i < mNThreads; i++) {
84  for (int j = 0; j < (int)mV0sTmp[i].size(); j++) {
85  v0SortID.emplace_back(vid{i, j, mV0sTmp[i][j].getVertexID()});
86  }
87  for (int j = 0; j < (int)mCascadesTmp[i].size(); j++) {
88  cascSortID.emplace_back(vid{i, j, mCascadesTmp[i][j].getVertexID()});
89  }
90  }
91  std::sort(v0SortID.begin(), v0SortID.end(), [](const vid& a, const vid& b) { return a.vtxID > b.vtxID; });
92  std::sort(cascSortID.begin(), cascSortID.end(), [](const vid& a, const vid& b) { return a.vtxID > b.vtxID; });
93  // sorted V0s
94  std::vector<V0> bufV0;
95  bufV0.reserve(nv0);
96  for (const auto& id : v0SortID) {
97  auto& v0 = mV0sTmp[id.thrID][id.entry];
98  int pos = bufV0.size();
99  bufV0.push_back(v0);
100  v0.setVertexID(pos); // this v0 copy will be discarded, use its vertexID to store the new position of final V0
101  }
102  // since V0s were reshuffled, we need to correct the cascade -> V0 reference indices
103  for (int i = 0; i < mNThreads; i++) { // merge results of all threads
104  for (auto& casc : mCascadesTmp[i]) { // before merging fix cascades references on v0
105  casc.setV0ID(mV0sTmp[i][casc.getV0ID()].getVertexID());
106  }
107  }
108  // sorted Cascades
109  std::vector<Cascade> bufCasc;
110  bufCasc.reserve(ncsc);
111  for (const auto& id : cascSortID) {
112  bufCasc.push_back(mCascadesTmp[id.thrID][id.entry]);
113  }
114  //
115  mV0sTmp[0].swap(bufV0); // the final result is fetched from here
116  mCascadesTmp[0].swap(bufCasc); // the final result is fetched from here
117  for (int i = 1; i < mNThreads; i++) { // clean unneeded s.vertices
118  mV0sTmp[i].clear();
119  mCascadesTmp[i].clear();
120  }
121  LOG(debug) << "DONE : " << mV0sTmp[0].size() << " " << mCascadesTmp[0].size();
122 }

130 void SVertexer::updateTimeDependentParams()
131 {
132  // TODO RS: strictly speaking, one should do this only in case of the CCDB objects update
133  mSVParams = &SVertexerParams::Instance();
134  // precalculated selection cuts
135  mMinR2ToMeanVertex = mSVParams->minRToMeanVertex * mSVParams->minRToMeanVertex;
136  mMaxR2ToMeanVertexCascV0 = mSVParams->maxRToMeanVertexCascV0 * mSVParams->maxRToMeanVertexCascV0;
137  mMaxDCAXY2ToMeanVertex = mSVParams->maxDCAXYToMeanVertex * mSVParams->maxDCAXYToMeanVertex;
138  mMaxDCAXY2ToMeanVertexV0Casc = mSVParams->maxDCAXYToMeanVertexV0Casc * mSVParams->maxDCAXYToMeanVertexV0Casc;
139  mMinR2DiffV0Casc = mSVParams->minRDiffV0Casc * mSVParams->minRDiffV0Casc;
140  mMinPt2V0 = mSVParams->minPtV0 * mSVParams->minPtV0;
141  mMaxTgl2V0 = mSVParams->maxTglV0 * mSVParams->maxTglV0;
142  mMinPt2Casc = mSVParams->minPtCasc * mSVParams->minPtCasc;
143  mMaxTgl2Casc = mSVParams->maxTglCasc * mSVParams->maxTglCasc;
144 
145  auto bz = o2::base::Propagator::Instance()->getNominalBz();
146 
147  mV0Hyps[HypV0::Photon].set(PID::Photon, PID::Electron, PID::Electron, mSVParams->pidCutsPhoton, bz);
148  mV0Hyps[HypV0::K0].set(PID::K0, PID::Pion, PID::Pion, mSVParams->pidCutsK0, bz);
149  mV0Hyps[HypV0::Lambda].set(PID::Lambda, PID::Proton, PID::Pion, mSVParams->pidCutsLambda, bz);
150  mV0Hyps[HypV0::AntiLambda].set(PID::Lambda, PID::Pion, PID::Proton, mSVParams->pidCutsLambda, bz);
151  mV0Hyps[HypV0::HyperTriton].set(PID::HyperTriton, PID::Helium3, PID::Pion, mSVParams->pidCutsHTriton, bz);
152  mV0Hyps[HypV0::AntiHyperTriton].set(PID::HyperTriton, PID::Pion, PID::Helium3, mSVParams->pidCutsHTriton, bz);
153 
154  mCascHyps[HypCascade::XiMinus].set(PID::XiMinus, PID::Lambda, PID::Pion, mSVParams->pidCutsXiMinus, bz);
155  mCascHyps[HypCascade::OmegaMinus].set(PID::OmegaMinus, PID::Lambda, PID::Kaon, mSVParams->pidCutsOmegaMinus, bz);
156 
157  setupThreads();
158 
159  for (auto& ft : mFitterV0) {
160  ft.setBz(bz);
161  }
162  for (auto& ft : mFitterCasc) {
163  ft.setBz(bz);
164  }
165 
166  auto& gasParam = o2::tpc::ParameterGas::Instance();
167  mTPCBin2Z = gasParam.DriftV / mMUS2TPCBin;
168 }

171 void SVertexer::setupThreads()
172 {

174  return;
175  }
176  mV0sTmp.resize(mNThreads);
177  mCascadesTmp.resize(mNThreads);

179  auto bz = o2::base::Propagator::Instance()->getNominalBz();
180  for (auto& fitter : mFitterV0) {
181  fitter.setBz(bz);
182  fitter.setUseAbsDCA(mSVParams->useAbsDCA);
183  fitter.setPropagateToPCA(false);
184  fitter.setMaxR(mSVParams->maxRIni);
185  fitter.setMinParamChange(mSVParams->minParamChange);
186  fitter.setMinRelChi2Change(mSVParams->minRelChi2Change);
187  fitter.setMaxDZIni(mSVParams->maxDZIni);
188  fitter.setMaxChi2(mSVParams->maxChi2);
189  fitter.setMatCorrType(o2::base::Propagator::MatCorrType(mSVParams->matCorr));
190  fitter.setUsePropagator(mSVParams->usePropagator);
191  fitter.setRefitWithMatCorr(mSVParams->refitWithMatCorr);
192  fitter.setMaxStep(mSVParams->maxStep);
193  fitter.setMaxSnp(mSVParams->maxSnp);
194  fitter.setMinXSeed(mSVParams->minXSeed);
195  }
196  mFitterCasc.resize(mNThreads);
197  for (auto& fitter : mFitterCasc) {
198  fitter.setBz(bz);
199  fitter.setUseAbsDCA(mSVParams->useAbsDCA);
200  fitter.setPropagateToPCA(false);
201  fitter.setMaxR(mSVParams->maxRIniCasc);
202  fitter.setMinParamChange(mSVParams->minParamChange);
203  fitter.setMinRelChi2Change(mSVParams->minRelChi2Change);
204  fitter.setMaxDZIni(mSVParams->maxDZIni);
205  fitter.setMaxChi2(mSVParams->maxChi2);
206  fitter.setMatCorrType(o2::base::Propagator::MatCorrType(mSVParams->matCorr));
207  fitter.setUsePropagator(mSVParams->usePropagator);
208  fitter.setRefitWithMatCorr(mSVParams->refitWithMatCorr);
209  fitter.setMaxStep(mSVParams->maxStep);
210  fitter.setMaxSnp(mSVParams->maxSnp);
211  fitter.setMinXSeed(mSVParams->minXSeed);
212  }

216 bool SVertexer::acceptTrack(GIndex gid, const o2::track::TrackParCov& trc) const
217 {
218  if (gid.isPVContributor() && mSVParams->maxPVContributors < 1) {
219  return false;
220  }
221  // DCA to mean vertex
222  if (mSVParams->minDCAToPV > 0.f) {
223  o2::track::TrackPar trp(trc);
224  std::array<float, 2> dca;
225  auto* prop = o2::base::Propagator::Instance();
226  if (mSVParams->usePropagator) {
227  if (trp.getX() > mSVParams->minRFor3DField && !prop->PropagateToXBxByBz(trp, mSVParams->minRFor3DField, mSVParams->maxSnp, mSVParams->maxStep, o2::base::Propagator::MatCorrType(mSVParams->matCorr))) {
228  return true; // we don't need actually to propagate to the beam-line
229  }
230  if (!prop->propagateToDCA(mMeanVertex.getXYZ(), trp, prop->getNominalBz(), mSVParams->maxStep, o2::base::Propagator::MatCorrType(mSVParams->matCorr), &dca)) {
231  return true;
232  }
233  } else {
234  if (!trp.propagateParamToDCA(mMeanVertex.getXYZ(), prop->getNominalBz(), &dca)) {
235  return true;
236  }
237  }
238  if (std::abs(dca[0]) < mSVParams->minDCAToPV) {
239  return false;
240  }
241  }
242  return true;
243 }

245 //__________________________________________________________________
246 void SVertexer::buildT2V(const o2::globaltracking::RecoContainer& recoData) // accessor to various tracks
247 {
248  // build track->vertices from vertices->tracks, rejecting vertex contributors if requested
249  auto trackIndex = recoData.getPrimaryVertexMatchedTracks(); // Global ID's for associated tracks
250  auto vtxRefs = recoData.getPrimaryVertexMatchedTrackRefs(); // references from vertex to these track IDs
251 
252  std::unordered_map<GIndex, std::pair<int, int>> tmap;
253  std::unordered_map<GIndex, bool> rejmap;
254  int nv = vtxRefs.size() - 1; // The last entry is for unassigned tracks, ignore them
255  for (int i = 0; i < 2; i++) {
256  mTracksPool[i].clear();
257  mVtxFirstTrack[i].clear();
258  mVtxFirstTrack[i].resize(nv, -1);
259  }
260  for (int iv = 0; iv < nv; iv++) {
261  const auto& vtref = vtxRefs[iv];
262  int it = vtref.getFirstEntry(), itLim = it + vtref.getEntries();
263  for (; it < itLim; it++) {
264  auto tvid = trackIndex[it];
265  if (!recoData.isTrackSourceLoaded(tvid.getSource())) {
266  continue;
267  }
268  // unconstrained TPC tracks require special treatment: there is no point in checking DCA to mean vertex since it is not precise,
269  // but we need to create a clone of TPC track constrained to this particular vertex time.
270  if (tvid.getSource() == GIndex::TPC && processTPCTrack(recoData.getTPCTrack(tvid), tvid, iv)) { // processTPCTrack may decide that this track does not need special treatment (e.g. it is constrained...)
271  continue;
272  }
273  std::decay_t<decltype(tmap.find(tvid))> tref{};
274  if (tvid.isAmbiguous()) { // was this track already processed?
275  auto tref = tmap.find(tvid);
276  if (tref != tmap.end()) {
277  mTracksPool[tref->second.second][tref->second.first].vBracket.setMax(iv); // this track was already processed with other vertex, account the latter
278  continue;
279  }
280  // was it already rejected?
281  if (rejmap.find(tvid) != rejmap.end()) {
282  continue;
283  }
284  }
285  const auto& trc = recoData.getTrackParam(tvid);
286  if (!acceptTrack(tvid, trc)) {
287  if (tvid.isAmbiguous()) {
288  rejmap[tvid] = true;
289  }
290  continue;
291  }
292  int posneg = trc.getSign() < 0 ? 1 : 0;
293  float r = std::sqrt(trc.getX() * trc.getX() + trc.getY() * trc.getY());
294  mTracksPool[posneg].emplace_back(TrackCand{trc, tvid, {iv, iv}, r});
295  if (tvid.isAmbiguous()) { // track attached to >1 vertex, remember that it was already processed
296  tmap[tvid] = {mTracksPool[posneg].size() - 1, posneg};
297  }
298  }
299  }
300  // register 1st track of each charge for each vertex
301 
302  for (int pn = 0; pn < 2; pn++) {
303  auto& vtxFirstT = mVtxFirstTrack[pn];
304  const auto& tracksPool = mTracksPool[pn];
305  for (unsigned i = 0; i < tracksPool.size(); i++) {
306  const auto& t = tracksPool[i];
307  if (vtxFirstT[t.vBracket.getMin()] == -1) {
308  vtxFirstT[t.vBracket.getMin()] = i;
309  }
310  }
311  }
312 
313  LOG(info) << "Collected " << mTracksPool[POS].size() << " positive and " << mTracksPool[NEG].size() << " negative seeds";
314 }

317 bool SVertexer::checkV0(const TrackCand& seedP, const TrackCand& seedN, int iP, int iN, int ithread)
318 {
319  auto& fitterV0 = mFitterV0[ithread];
320  int nCand = fitterV0.process(seedP, seedN);
321  if (nCand == 0) { // discard this pair
322  return false;
323  }
324  const auto& v0XYZ = fitterV0.getPCACandidate();
325  // validate V0 radial position
326  // check closeness to the beam-line
327  float dxv0 = v0XYZ[0] - mMeanVertex.getX(), dyv0 = v0XYZ[1] - mMeanVertex.getY(), r2v0 = dxv0 * dxv0 + dyv0 * dyv0;
328  if (r2v0 < mMinR2ToMeanVertex) {
329  return false;
330  }
331  float rv0 = std::sqrt(r2v0), drv0P = rv0 - seedP.minR, drv0N = rv0 - seedN.minR;
332  if (drv0P > mSVParams->causalityRTolerance || drv0P < -mSVParams->maxV0ToProngsRDiff ||
333  drv0N > mSVParams->causalityRTolerance || drv0N < -mSVParams->maxV0ToProngsRDiff) {
334  LOG(debug) << "RejCausality " << drv0P << " " << drv0N;
335  return false;
336  }
337 
338  if (!fitterV0.isPropagateTracksToVertexDone() && !fitterV0.propagateTracksToVertex()) {
339  return false;
340  }
341  int cand = 0;
342  auto& trPProp = fitterV0.getTrack(0, cand);
343  auto& trNProp = fitterV0.getTrack(1, cand);
344  std::array<float, 3> pP, pN;
345  trPProp.getPxPyPzGlo(pP);
346  trNProp.getPxPyPzGlo(pN);
347  // estimate DCA of neutral V0 track to beamline: straight line with parametric equation
348  // x = X0 + pV0[0]*t, y = Y0 + pV0[1]*t reaches DCA to beamline (Xv, Yv) at
349  // t = -[ (x0-Xv)*pV0[0] + (y0-Yv)*pV0[1]) ] / ( pT(pV0)^2 )
350  // Similar equation for 3D distance involving pV0[2]
351  std::array<float, 3> pV0 = {pP[0] + pN[0], pP[1] + pN[1], pP[2] + pN[2]};
352  float pt2V0 = pV0[0] * pV0[0] + pV0[1] * pV0[1], prodXYv0 = dxv0 * pV0[0] + dyv0 * pV0[1], tDCAXY = prodXYv0 / pt2V0;
353  if (pt2V0 < mMinPt2V0) { // pt cut
354  LOG(debug) << "RejPt2 " << pt2V0;
355  return false;
356  }
357  if (pV0[2] * pV0[2] / pt2V0 > mMaxTgl2V0) { // tgLambda cut
358  LOG(debug) << "RejTgL " << pV0[2] * pV0[2] / pt2V0;
359  return false;
360  }
361  float p2V0 = pt2V0 + pV0[2] * pV0[2], ptV0 = std::sqrt(pt2V0);
362  // apply mass selections
363  float p2Pos = pP[0] * pP[0] + pP[1] * pP[1] + pP[2] * pP[2], p2Neg = pN[0] * pN[0] + pN[1] * pN[1] + pN[2] * pN[2];
364 
365  bool goodHyp = false;
366  std::array<bool, NHypV0> hypCheckStatus{};
367  for (int ipid = 0; ipid < NHypV0; ipid++) {
368  if (mV0Hyps[ipid].check(p2Pos, p2Neg, p2V0, ptV0)) {
369  goodHyp = hypCheckStatus[ipid] = true;
370  }
371  }
372  if (!goodHyp && mSVParams->checkV0Hypothesis) {
373  LOG(debug) << "RejHypo";
374  return false;
375  }
376 
377  bool checkForCascade = mEnableCascades && r2v0 < mMaxR2ToMeanVertexCascV0 && (!mSVParams->checkV0Hypothesis || (hypCheckStatus[HypV0::Lambda] || hypCheckStatus[HypV0::AntiLambda]));
378  bool rejectIfNotCascade = false;
379  float dcaX = dxv0 - pV0[0] * tDCAXY, dcaY = dyv0 - pV0[1] * tDCAXY, dca2 = dcaX * dcaX + dcaY * dcaY;
380  float cosPAXY = prodXYv0 / std::sqrt(r2v0 * pt2V0);
381 
382  if (checkForCascade) { // use loser cuts for cascade v0 candidates
383  if (dca2 > mMaxDCAXY2ToMeanVertexV0Casc || cosPAXY < mSVParams->minCosPAXYMeanVertexCascV0) {
384  LOG(debug) << "Rej for cascade DCAXY2: " << dca2 << " << cosPAXY: " << cosPAXY;
385  return false;
386  }
387  }
388  if (dca2 > mMaxDCAXY2ToMeanVertex || cosPAXY < mSVParams->minCosPAXYMeanVertex) {
389  if (checkForCascade) {
390  rejectIfNotCascade = true;
391  } else {
392  return false;
393  }
394  }
395 
396  auto vlist = seedP.vBracket.getOverlap(seedN.vBracket); // indices of vertices shared by both seeds
397  bool added = false;
398  auto bestCosPA = checkForCascade ? mSVParams->minCosPACascV0 : mSVParams->minCosPA;
399  for (int iv = vlist.getMin(); iv <= vlist.getMax(); iv++) {
400  const auto& pv = mPVertices[iv];
401  const auto v0XYZ = fitterV0.getPCACandidatePos(cand);
402  // check cos of pointing angle
403  float dx = v0XYZ[0] - pv.getX(), dy = v0XYZ[1] - pv.getY(), dz = v0XYZ[2] - pv.getZ(), prodXYZv0 = dx * pV0[0] + dy * pV0[1] + dz * pV0[2];
404  float cosPA = prodXYZv0 / std::sqrt((dx * dx + dy * dy + dz * dz) * p2V0);
405  if (cosPA < bestCosPA) {
406  LOG(debug) << "Rej. cosPA: " << cosPA;
407  continue;
408  }
409  if (!added) {
410  auto& v0new = mV0sTmp[ithread].emplace_back(v0XYZ, pV0, fitterV0.calcPCACovMatrixFlat(cand), trPProp, trNProp, seedP.gid, seedN.gid);
411  v0new.setDCA(fitterV0.getChi2AtPCACandidate());
412  added = true;
413  }
414  auto& v0 = mV0sTmp[ithread].back();
415  v0.setCosPA(cosPA);
416  v0.setVertexID(iv);
417  bestCosPA = cosPA;
418  }
419  if (!added) {
420  return false;
421  }
422  auto& v0 = mV0sTmp[ithread].back();
423 
424  // check cascades
425  if (checkForCascade) {
426  int nCascAdded = 0;
427  if (hypCheckStatus[HypV0::Lambda] || !mSVParams->checkCascadeHypothesis) {
428  nCascAdded += checkCascades(rv0, pV0, p2V0, iN, NEG, ithread);
429  }
430  if (hypCheckStatus[HypV0::AntiLambda] || !mSVParams->checkCascadeHypothesis) {
431  nCascAdded += checkCascades(rv0, pV0, p2V0, iP, POS, ithread);
432  }
433  if (!nCascAdded && rejectIfNotCascade) { // v0 would be accepted only if it creates a cascade
434  mV0sTmp[ithread].pop_back();
435  return false;
436  }
437  }
438 
439  return true;
440 }

442 //__________________________________________________________________
443 int SVertexer::checkCascades(float rv0, std::array<float, 3> pV0, float p2V0, int avoidTrackID, int posneg, int ithread)
444 {
445  // check last added V0 for belonging to cascade
446  auto& fitterCasc = mFitterCasc[ithread];
447  const auto& v0 = mV0sTmp[ithread].back();
448  auto& tracks = mTracksPool[posneg];
449  const auto& pv = mPVertices[v0.getVertexID()];
450  int nCascIni = mCascadesTmp[ithread].size();
451  // start from the 1st track compatible with V0's primary vertex
452  int firstTr = mVtxFirstTrack[posneg][v0.getVertexID()], nTr = tracks.size();
453  if (firstTr < 0) {
454  firstTr = nTr;
455  }
456  for (int it = firstTr; it < nTr; it++) {
457  if (it == avoidTrackID) {
458  continue; // skip the track used by V0
459  }
460  auto& bach = tracks[it];
461  if (bach.vBracket > v0.getVertexID()) {
462  break; // all other bachelor candidates will be also not compatible with this PV
463  }
464  if (bach.vBracket.isOutside(v0.getVertexID())) {
465  LOG(error) << "Incompatible bachelor: PV " << bach.vBracket.asString() << " vs V0 " << v0.getVertexID();
466  }
467  if (bach.minR > rv0 + mSVParams->causalityRTolerance) {
468  continue;
469  }
470  int nCandC = fitterCasc.process(v0, bach);
471  if (nCandC == 0) { // discard this pair
472  continue;
473  }
474  int candC = 0;
475  const auto& cascXYZ = fitterCasc.getPCACandidatePos(candC);
476  // make sure the cascade radius is smaller than that of the vertex
477  float dxc = cascXYZ[0] - pv.getX(), dyc = cascXYZ[1] - pv.getY(), dzc = cascXYZ[2] - pv.getZ(), r2casc = dxc * dxc + dyc * dyc;
478  if (rv0 * rv0 - r2casc < mMinR2DiffV0Casc || r2casc < mMinR2ToMeanVertex) {
479  continue;
480  }
481  // do we want to apply mass cut ?
482  //
483  if (!fitterCasc.isPropagateTracksToVertexDone() && !fitterCasc.propagateTracksToVertex()) {
484  continue;
485  }
486  auto& trNeut = fitterCasc.getTrack(0, candC);
487  auto& trBach = fitterCasc.getTrack(1, candC);
488  trNeut.setPID(o2::track::PID::Lambda);
489  trBach.setPID(o2::track::PID::Pion);
490  std::array<float, 3> pNeut, pBach;
491  trNeut.getPxPyPzGlo(pNeut);
492  trBach.getPxPyPzGlo(pBach);
493  std::array<float, 3> pCasc = {pNeut[0] + pBach[0], pNeut[1] + pBach[1], pNeut[2] + pBach[2]};
494  auto prodPPos = pV0[0] * dxc + pV0[1] * dyc + pV0[2] * dzc;
495  if (prodPPos < 0.) { // causality cut
496  continue;
497  }
498  float pt2Casc = pCasc[0] * pCasc[0] + pCasc[1] * pCasc[1], p2Casc = pt2Casc + pCasc[2] * pCasc[2];
499  if (pt2Casc < mMinPt2Casc) { // pt cut
500  continue;
501  }
502  if (pCasc[2] * pCasc[2] / pt2Casc > mMaxTgl2Casc) { // tgLambda cut
503  continue;
504  }
505  // LOG(info) << "ptcasc2 " << pt2Casc << " tglcasc2 " << pCasc[2]*pCasc[2] / pt2Casc << " cut " << mMaxTgl2Casc;
506  float cosPA = (pCasc[0] * dxc + pCasc[1] * dyc + pCasc[2] * dzc) / std::sqrt(p2Casc * (r2casc + dzc * dzc));
507  if (cosPA < mSVParams->minCosPACasc) {
508  continue;
509  }
510  float p2Bach = pBach[0] * pBach[0] + pBach[1] * pBach[1] + pBach[2] * pBach[2];
511  float ptCasc = std::sqrt(pt2Casc);
512  bool goodHyp = false;
513  for (int ipid = 0; ipid < NHypCascade; ipid++) {
514  if (mCascHyps[ipid].check(p2V0, p2Bach, p2Casc, ptCasc)) {
515  goodHyp = true;
516  break;
517  }
518  }
519  if (!goodHyp) {
520  continue;
521  }
522  auto& casc = mCascadesTmp[ithread].emplace_back(cascXYZ, pCasc, fitterCasc.calcPCACovMatrixFlat(candC), trNeut, trBach, mV0sTmp[ithread].size() - 1, bach.gid);
523  o2::track::TrackParCov trc = casc;
524  o2::dataformats::DCA dca;
525  if (!trc.propagateToDCA(pv, fitterCasc.getBz(), &dca, 5.) ||
526  std::abs(dca.getY()) > mSVParams->maxDCAXYCasc || std::abs(dca.getZ()) > mSVParams->maxDCAZCasc) {
527  mCascadesTmp[ithread].pop_back();
528  continue;
529  }
530  casc.setCosPA(cosPA);
531  casc.setVertexID(v0.getVertexID());
532  casc.setDCA(fitterCasc.getChi2AtPCACandidate());
533  }
534  return mCascadesTmp[ithread].size() - nCascIni;
535 }

548 bool SVertexer::processTPCTrack(const o2::tpc::TrackTPC& trTPC, GIndex gid, int vtxid)
549 {
550  // if TPC trackis unconstrained, try to create in the tracks pool a clone constrained to vtxid vertex time.
551  if (trTPC.hasBothSidesClusters()) { // this is effectively constrained track
552  return false; // let it be processed as such
553  }
554  const auto& vtx = mPVertices[vtxid];
555  auto twe = vtx.getTimeStamp();
556  int posneg = trTPC.getSign() < 0 ? 1 : 0;
557  auto trLoc = mTracksPool[posneg].emplace_back(TrackCand{trTPC, gid, {vtxid, vtxid}, 0.});
558  auto err = correctTPCTrack(trLoc, trTPC, twe.getTimeStamp(), twe.getTimeStampError());
559  if (err < 0) {
560  mTracksPool[posneg].pop_back(); // discard
561  }
562  trLoc.minR = std::sqrt(trLoc.getX() * trLoc.getX() + trLoc.getY() * trLoc.getY());
563  return true;
564 }

567 float SVertexer::correctTPCTrack(o2::track::TrackParCov& trc, const o2::tpc::TrackTPC tTPC, float tmus, float tmusErr) const
568 {
569  // Correct the track copy trc of the TPC track for the assumed interaction time
570  // return extra uncertainty in Z due to the interaction time uncertainty
571  // TODO: at the moment, apply simple shift, but with Z-dependent calibration we may
572  // need to do corrections on TPC cluster level and refit
573  // This is a clone of MatchTPCITS::correctTPCTrack
574  float dDrift = (tmus * mMUS2TPCBin - tTPC.getTime0()) * mTPCBin2Z;
575  float driftErr = tmusErr * mMUS2TPCBin * mTPCBin2Z;
576  // eventually should be refitted, at the moment we simply shift...
577  trc.setZ(tTPC.getZ() + (tTPC.hasASideClustersOnly() ? dDrift : -dDrift));
578  trc.setCov(trc.getSigmaZ2() + driftErr * driftErr, o2::track::kSigZ2);
579 
580  return driftErr;
581 }
