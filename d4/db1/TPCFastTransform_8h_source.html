265 GPUdi() void TPCFastTransform::convPadTimeToUV(int slice, int row, float pad, float time, float& u, float& v, float vertexTime) const
266 {
267  bool sideC = (slice >= getGeometry().getNumberOfSlicesA());
268 
269  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
270  const TPCFastTransformGeo::SliceInfo& sliceInfo = getGeometry().getSliceInfo(slice);
271 
272  float x = rowInfo.x;
273  u = (pad - 0.5 * rowInfo.maxPad) * rowInfo.padWidth;
274 
275  float y = sideC ? -u : u; // pads are mirrorred on C-side
276  float yLab = y * sliceInfo.cosAlpha + x * sliceInfo.sinAlpha;
277 
278  v = (time - mT0 - vertexTime) * (mVdrift + mVdriftCorrY * yLab) + mLdriftCorr; // drift length cm
279 }

311 GPUdi() void TPCFastTransform::convUVtoPadTime(int slice, int row, float u, float v, float& pad, float& time, float vertexTime) const
312 {
313  bool sideC = (slice >= getGeometry().getNumberOfSlicesA());
314 
315  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
316  const TPCFastTransformGeo::SliceInfo& sliceInfo = getGeometry().getSliceInfo(slice);
317 
318  pad = u / rowInfo.padWidth + 0.5 * rowInfo.maxPad;
319 
320  float x = rowInfo.x;
321  float y = sideC ? -u : u; // pads are mirrorred on C-side
322  float yLab = y * sliceInfo.cosAlpha + x * sliceInfo.sinAlpha;
323  time = mT0 + vertexTime + (v - mLdriftCorr) / (mVdrift + mVdriftCorrY * yLab);
324 }

344 GPUdi() void TPCFastTransform::getTOFcorrection(int slice, int /*row*/, float x, float y, float z, float& dz) const
345 {
346  // calculate time of flight correction for z coordinate
347 
348  bool sideC = (slice >= getGeometry().getNumberOfSlicesA());
349  float distZ = z - mPrimVtxZ;
350  float dv = -GPUCommonMath::Sqrt(x * x + y * y + distZ * distZ) * mTOFcorr;
351  dz = sideC ? dv : -dv;
352 }

354 GPUdi() void TPCFastTransform::Transform(int slice, int row, float pad, float time, float& x, float& y, float& z, float vertexTime) const
355 {
361 
362  const TPCFastTransformGeo::RowInfo& rowInfo = getGeometry().getRowInfo(row);
363 
364  // const SliceInfo &sliceInfo = getSliceInfo( slice );
365  // bool sideC = ( slice >= NumberOfSlices / 2 );

380

382  getTOFcorrection(slice, row, x, y, z, dzTOF);
