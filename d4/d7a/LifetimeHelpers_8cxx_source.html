62 ExpirationHandler::Creator LifetimeHelpers::enumDrivenCreation(size_t start, size_t end, size_t step, size_t inputTimeslice, size_t maxInputTimeslices, size_t maxRepetitions)
63 {
64  auto last = std::make_shared<size_t>(start + inputTimeslice * step);
65  auto repetition = std::make_shared<size_t>(0);
66 
67  return [start, end, step, last, inputTimeslice, maxInputTimeslices, maxRepetitions, repetition](TimesliceIndex& index) -> TimesliceSlot {
68  for (size_t si = 0; si < index.size(); si++) {
69  if (*last > end) {
70  return TimesliceSlot{TimesliceSlot::INVALID};
71  }
72  auto slot = TimesliceSlot{si};
73  if (index.isValid(slot) == false) {
74  TimesliceId timestamp{*last};
75  *repetition += 1;
76  if (*repetition % maxRepetitions == 0) {
77  *last += step * maxInputTimeslices;
78  }
79  index.associate(timestamp, slot);
80  return slot;
81  }
82  }
83 
84  return TimesliceSlot{TimesliceSlot::INVALID};
85  };
86 }

88 ExpirationHandler::Creator LifetimeHelpers::timeDrivenCreation(std::chrono::microseconds period)
89 {
90  auto start = getCurrentTime();
91  auto last = std::make_shared<decltype(start)>(start);
92  // FIXME: should create timeslices when period expires....
93  return [last, period](TimesliceIndex& index) -> TimesliceSlot {
94  // Nothing to do if the time has not expired yet.
95  auto current = getCurrentTime();
96  auto delta = current - *last;
97  if (delta < period.count()) {
98  return TimesliceSlot{TimesliceSlot::INVALID};
99  }
100  // We first check if the current time is not already present
101  // FIXME: this should really be done by query matching? Ok
102  // for now to avoid duplicate entries.
103  for (size_t i = 0; i < index.size(); ++i) {
104  TimesliceSlot slot{i};
105  if (index.isValid(slot) == false) {
106  continue;
107  }
108  auto& variables = index.getVariablesForSlot(slot);
109  if (VariableContextHelpers::getTimeslice(variables).value == current) {
110  return TimesliceSlot{TimesliceSlot::INVALID};
111  }
112  }
113  // If we are here the timer has expired and a new slice needs
114  // to be created.
115  *last = current;
116  data_matcher::VariableContext newContext;
117  newContext.put({0, static_cast<uint64_t>(current)});
118  newContext.commit();
119  auto [action, slot] = index.replaceLRUWith(newContext, TimesliceId{current});
120  switch (action) {
121  case TimesliceIndex::ActionTaken::ReplaceObsolete:
122  case TimesliceIndex::ActionTaken::ReplaceUnused:
123  index.associate(TimesliceId{current}, slot);
124  break;
125  case TimesliceIndex::ActionTaken::DropInvalid:
126  case TimesliceIndex::ActionTaken::DropObsolete:
127  case TimesliceIndex::ActionTaken::Wait:
128  break;
129  }
130  return slot;
131  };
132 }

144 ExpirationHandler::Checker LifetimeHelpers::expireTimed(std::chrono::microseconds period)
145 {
146  auto start = getCurrentTime();
147  auto last = std::make_shared<decltype(start)>(start);
148  return [last, period](ServiceRegistry&, int64_t) -> bool {
149  auto current = getCurrentTime();
150  auto delta = current - *last;
151  if (delta > period.count()) {
152  *last = current;
153  return true;
154  }
155  return false;

207 ExpirationHandler::Handler
208  LifetimeHelpers::fetchFromCCDBCache(InputSpec const& spec,
209  std::string const& prefix,
210  std::string const& overrideTimestamp,
211  std::string const& sourceChannel)
212 {
213  char* err;
214  uint64_t overrideTimestampMilliseconds = strtoll(overrideTimestamp.c_str(), &err, 10);
215  if (*err != 0) {
216  throw runtime_error("fetchFromCCDBCache: Unable to parse forced timestamp for conditions");
217  }
218  if (overrideTimestampMilliseconds) {
219  LOGP(info, "fetchFromCCDBCache: forcing timestamp for conditions to {} milliseconds from epoch UTC", overrideTimestampMilliseconds);
220  }
221  auto matcher = std::get_if<ConcreteDataMatcher>(&spec.matcher);
222  if (matcher == nullptr) {
223  throw runtime_error("InputSpec for Conditions must be fully qualified");
224  }
225  return [spec, matcher, sourceChannel, serverUrl = prefix, overrideTimestampMilliseconds](ServiceRegistry& services, PartRef& ref, data_matcher::VariableContext& variables) -> void {
226  // We should invoke the handler only once.
227  assert(!ref.header);
228  assert(!ref.payload);
229 
230  auto& rawDeviceService = services.get<RawDeviceService>();
231  auto& dataTakingContext = services.get<DataTakingContext>();
232 
233  auto&& transport = rawDeviceService.device()->GetChannel(sourceChannel, 0).Transport();
234  auto channelAlloc = o2::pmr::getTransportAllocator(transport);
235  o2::pmr::vector<char> payloadBuffer{transport->GetMemoryResource()};
236  payloadBuffer.reserve(10000); // we begin with messages of 10KB
237 
238  CURL* curl = curl_easy_init();
239  if (curl == nullptr) {
240  throw runtime_error("fetchFromCCDBCache: Unable to initialise CURL");
241  }
242  CURLcode res;
243 
244  // * By default we use the time when the data was created.
245  // * If an override is specified, we use it.
246  // * If the orbit reset time comes from CTP, we use it for precise
247  // timestamp evaluation via the firstTFOrbit
248  uint64_t timestamp = -1;
249  if (overrideTimestampMilliseconds) {
250  timestamp = overrideTimestampMilliseconds;
251  } else if (dataTakingContext.source == OrbitResetTimeSource::CTP) {
252  // Orbit reset time is in microseconds, LHCOrbitNS is in nanoseconds, CCDB uses milliseconds
253  timestamp = ceilf((VariableContextHelpers::getFirstTFOrbit(variables) * o2::constants::lhc::LHCOrbitNS / 1000 + dataTakingContext.orbitResetTime) / 1000);
254  } else {
255  // The timestamp used by DPL is in nanoseconds
256  timestamp = ceilf(VariableContextHelpers::getTimeslice(variables).value / 1000);
257  }
258 
259  std::string path = "";
260  bool runDependent = false;
261  for (auto& meta : spec.metadata) {
262  if (meta.name == "ccdb-path") {
263  path = meta.defaultValue.get<std::string>();
264  }
265  if (meta.name == "ccdb-run-dependent") {
266  runDependent = meta.defaultValue.get<bool>();
267  }
268  }
269  if (path.empty()) {
270  path = fmt::format("{}/{}", matcher->origin, matcher->description);
271  }
272  std::string url;
273  if (runDependent == false) {
274  url = fmt::format("{}/{}/{}", serverUrl, path, timestamp);
275  } else {
276  url = fmt::format("{}/{}/{}/runNumber={}", serverUrl, path, timestamp, dataTakingContext.runNumber);
277  }
278  LOG(debug) << "fetchFromCCDBCache: Fetching " << url;
279 
280  curl_easy_setopt(curl, CURLOPT_URL, url.c_str());
281  curl_easy_setopt(curl, CURLOPT_WRITEDATA, &payloadBuffer);
282  curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, readToMessage);
283  curl_easy_setopt(curl, CURLOPT_FOLLOWLOCATION, true);
284 
285  res = curl_easy_perform(curl);
286  if (res != CURLE_OK) {
287  throw runtime_error_f("fetchFromCCDBCache: Unable to fetch %s from CCDB", url.c_str());
288  }
289  long responseCode;
290  curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &responseCode);
291 
292  if (responseCode != 200) {
293  throw runtime_error_f("fetchFromCCDBCache: HTTP error %d while fetching %s from CCDB", responseCode, url.c_str());
294  }
295 
296  curl_easy_cleanup(curl);
297 
298  DataHeader dh;
299  dh.dataOrigin = matcher->origin;
300  dh.dataDescription = matcher->description;
301  dh.subSpecification = matcher->subSpec;
302  // FIXME: should use curl_off_t and CURLINFO_SIZE_DOWNLOAD_T, but
303  // apparently not there on some platforms.
304  double dl;
305  res = curl_easy_getinfo(curl, CURLINFO_SIZE_DOWNLOAD, &dl);
306  dh.payloadSize = payloadBuffer.size();
307  dh.payloadSerializationMethod = gSerializationMethodCCDB;
308 
309  DataProcessingHeader dph{timestamp, 1};
310  auto header = o2::pmr::getMessage(o2::header::Stack{channelAlloc, dh, dph});
311  auto payload = o2::pmr::getMessage(std::forward<o2::pmr::vector<char>>(payloadBuffer), transport->GetMemoryResource());
312 
313  ref.header = std::move(header);
314  ref.payload = std::move(payload);
315 
316  return;
317  };
318 }

361 ExpirationHandler::Handler LifetimeHelpers::enumerate(ConcreteDataMatcher const& matcher, std::string const& sourceChannel,
362  int64_t orbitOffset, int64_t orbitMultiplier)
363 {
364  using counter_t = int64_t;
365  auto counter = std::make_shared<counter_t>(0);
366  return [matcher, counter, sourceChannel, orbitOffset, orbitMultiplier](ServiceRegistry& services, PartRef& ref, data_matcher::VariableContext& variables) -> void {
367  // We should invoke the handler only once.
368  assert(!ref.header);
369  assert(!ref.payload);
370  auto& rawDeviceService = services.get<RawDeviceService>();
371 
372  auto timestamp = VariableContextHelpers::getTimeslice(variables).value;
373  DataHeader dh;
374  dh.dataOrigin = matcher.origin;
375  dh.dataDescription = matcher.description;
376  dh.subSpecification = matcher.subSpec;
377  dh.payloadSize = sizeof(counter_t);
378  dh.payloadSerializationMethod = gSerializationMethodNone;
379  dh.tfCounter = timestamp;
380  dh.firstTForbit = timestamp * orbitMultiplier + orbitOffset;
381  DataProcessingHeader dph{timestamp, 1};
382  services.get<CallbackService>()(CallbackService::Id::NewTimeslice, dh, dph);
383 
384  variables.put({data_matcher::FIRSTTFORBIT_POS, dh.firstTForbit});
385  variables.put({data_matcher::TFCOUNTER_POS, dh.tfCounter});
386 
387  auto&& transport = rawDeviceService.device()->GetChannel(sourceChannel, 0).Transport();
388  auto channelAlloc = o2::pmr::getTransportAllocator(transport);
389  auto header = o2::pmr::getMessage(o2::header::Stack{channelAlloc, dh, dph});
390  ref.header = std::move(header);
391 
392  auto payload = rawDeviceService.device()->NewMessage(sizeof(counter_t));
393  *(counter_t*)payload->GetData() = *counter;
394  ref.payload = std::move(payload);
395  (*counter)++;
396  };
397 }
