79 Status FairMQOutputStream::Write(const void* data, int64_t nbytes)
80 {
81  if (ARROW_PREDICT_FALSE(!is_open_)) {
82  return Status::IOError("OutputStream is closed");
83  }
84  if (ARROW_PREDICT_TRUE(nbytes > 0)) {
85  if (ARROW_PREDICT_FALSE(position_ + nbytes >= capacity_)) {
86  RETURN_NOT_OK(Reserve(nbytes));
87  }
88  memcpy(mutable_data_ + position_, data, nbytes);
89  position_ += nbytes;
90  }
91  return Status::OK();
92 }

94 Status FairMQOutputStream::Reserve(int64_t nbytes)
95 {
96  // Always overallocate by doubling. It seems that it is a better growth
97  // strategy, at least for memory_benchmark.cc.
98  // This may be because it helps match the allocator's allocation buckets
99  // more exactly. Or perhaps it hits a sweet spot in jemalloc.
100  int64_t new_capacity = std::max(kBufferMinimumSize, capacity_);
101  new_capacity = position_ + nbytes;
102  if (new_capacity > capacity_) {
103  RETURN_NOT_OK(buffer_->Resize(new_capacity));
104  capacity_ = new_capacity;
105  mutable_data_ = buffer_->mutable_data();
106  }
107  return Status::OK();
108 }

123 arrow::Status FairMQResizableBuffer::Resize(const int64_t newSize, bool shrink_to_fit)
124 {
125  // NOTE: we ignore "shrink_to_fit" because in any case we
126  // invoke SetUsedSize when we send the message. This
127  // way we avoid unneeded copies at the arrow level.
128  if (newSize > this->capacity_) {
129  auto status = this->Reserve(newSize);
130  if (status.ok() == false) {
131  return status;
132  }
133  }
134  assert(newSize <= this->capacity_);
135 
136  this->size_ = newSize;
137  assert(this->size_ <= mMessage->GetSize());
138  return arrow::Status::OK();
139 }

141 arrow::Status FairMQResizableBuffer::Reserve(const int64_t capacity)
142 {
143  assert(!mMessage || this->size_ <= mMessage->GetSize());
144  assert(!mMessage || this->capacity_ == mMessage->GetSize());
145  if (capacity <= this->capacity_) {
146  return arrow::Status::OK();
147  }
148  auto newMessage = mCreator(capacity);
149  assert(!mMessage || capacity > mMessage->GetSize());
150  if (mMessage) {
151  memcpy(newMessage->GetData(), mMessage->GetData(), mMessage->GetSize());
152  }
153  mMessage = std::move(newMessage);
154  assert(mMessage);
155  this->data_ = reinterpret_cast<uint8_t*>(mMessage->GetData());
156  this->capacity_ = static_cast<int64_t>(mMessage->GetSize());
157  assert(this->data_);
158  return arrow::Status::OK();
159 }
