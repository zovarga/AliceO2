144  void deserialize(const gsl::span<const uint8_t> inputBytes, size_t nWords, size_t& srcPos)
145  {
146  mNelements = 0;
147  mNwords = 0;
148  if (nWords < MinNwords || nWords > MaxNwords || inputBytes.size() - srcPos < nWords * SIZE_WORD) {
149  //in case of bad fields responsible for deserialization logic, byte position will be pushed to the end of binary sequence

159  for (int iStep = 0; iStep < nSteps; iStep++) {
160  memcpy(destAddress + std::get<kDESTBYTEPOS>(sByteLookupTable[iStep]), inputBytes.data() + std::get<kSRCBYTEPOS>(sByteLookupTable[iStep]) + srcPos, std::get<kNBYTES>(sByteLookupTable[iStep]));

172  //get number of byte reading steps
173  static constexpr size_t getNsteps()
174  {
175  int count = 0;
176  size_t payloadFull = Data_t::MaxNelements * Data_t::PayloadSize;
177  size_t payloadInWord = Data_t::PayloadPerGBTword;
178  size_t payloadPerElem = Data_t::PayloadSize;
179  while (payloadFull > 0) {
180  if (payloadPerElem < payloadInWord) {
181  count++;
182  payloadFull -= payloadPerElem;
183  payloadInWord -= payloadPerElem;
184  payloadPerElem = 0;
185  } else {
186  count++;
187  payloadFull -= payloadInWord;
188  payloadPerElem -= payloadInWord;
189  payloadInWord = 0;
190  }
191  if (payloadInWord == 0) {
192  payloadInWord = Data_t::PayloadPerGBTword;
193  }
194  if (payloadPerElem == 0) {
195  payloadPerElem = Data_t::PayloadSize;
196  }
197  }
198  return count;
199  }
200  //enumerator for tuple access:
201  //[Index] is index of step to read bytes
202  //kNBYTES - number of bytes to read from source and to write into destination
203  //kSRCBYTEPOS - Byte position in the source(binary raw data sequence)
204  //kDESTBYTEPOS - Byte position to write at destionation(memory allocation of T-element array)
205  //kELEMENTINDEX - element index at current step
206  //kWORDINDEX - word index at current step
207  //
208  enum AccessByteLUT { kNBYTES,
209  kSRCBYTEPOS,
210  kDESTBYTEPOS,
211  kELEMENTINDEX,
212  kWORDINDEX };
213  static constexpr std::array<std::tuple<size_t, size_t, size_t, int, int>, getNsteps()> GetByteLookupTable()
214  {
215  std::array<std::tuple<size_t, size_t, size_t, int, int>, getNsteps()> seqBytes{};
216  int count = 0;
217  int countElement = 0;
218  int countWord = 0;
219  size_t destBytePosPerElem = 0;
220  size_t srcBytePos = 0;
221  size_t payloadFull = Data_t::MaxNelements * Data_t::PayloadSize;
222 
223  size_t bytesInWord = SIZE_WORD;
224  size_t payloadInWord = Data_t::PayloadPerGBTword;
225 
226  size_t payloadPerElem = Data_t::PayloadSize;
227 
228  uint64_t indexElem = 0;
229  uint64_t indexLastElem = Data_t::MaxNelements - 1;
230 
231  while (payloadFull > 0) {
232  if (payloadPerElem < payloadInWord) { //new element
233  std::get<kNBYTES>(seqBytes[count]) = payloadPerElem;
234  std::get<kSRCBYTEPOS>(seqBytes[count]) = srcBytePos;
235  std::get<kDESTBYTEPOS>(seqBytes[count]) = destBytePosPerElem;
236  std::get<kELEMENTINDEX>(seqBytes[count]) = countElement;
237  std::get<kWORDINDEX>(seqBytes[count]) = countWord;
238  srcBytePos += payloadPerElem;
239  count++;
240  payloadFull -= payloadPerElem;
241  payloadInWord -= payloadPerElem;
242  bytesInWord -= payloadPerElem;
243  payloadPerElem = 0;
244 
245  } else {
246  std::get<kNBYTES>(seqBytes[count]) = payloadInWord;
247  std::get<kSRCBYTEPOS>(seqBytes[count]) = srcBytePos;
248  std::get<kDESTBYTEPOS>(seqBytes[count]) = destBytePosPerElem;
249  std::get<kELEMENTINDEX>(seqBytes[count]) = countElement;
250  std::get<kWORDINDEX>(seqBytes[count]) = countWord;
251  srcBytePos += bytesInWord;
252  count++;
253  destBytePosPerElem += payloadInWord;
254 
255  payloadFull -= payloadInWord;
256  payloadPerElem -= payloadInWord;
257  payloadInWord = 0;
258  bytesInWord = 0;
259  }
260 
261  if (payloadInWord == 0) {
262  payloadInWord = Data_t::PayloadPerGBTword;
263  }
264  if (payloadPerElem == 0) {
265  payloadPerElem = Data_t::PayloadSize;
266  countElement++;
267  destBytePosPerElem = countElement * sizeof(T);
268  }
269  if (bytesInWord == 0) {
270  bytesInWord = SIZE_WORD;
271  countWord++;
272  }
273  }
274  return seqBytes;
275  }
276  static constexpr std::array<std::tuple<size_t, size_t, size_t, int, int>, getNsteps()> sByteLookupTable = GetByteLookupTable();

278  //enumerator for tuple access:
279  //[Index] is word index position, i.e. "Index" number of words will be deserialized
280  //kNELEMENTS - number of T elements will be fully deserialized in "Index+1" words
281  //kNSTEPS - number of steps for reading "Index" words
282  //kISPARTED - if one T-element is parted at current word,i.e. current word contains partially deserialized T element at the end of the word
283  enum AccessReadingLUT { kNELEMENTS,
284  kNSTEPS,

286  static constexpr std::array<std::tuple<unsigned int, unsigned int, bool>, MaxNwords + 1> GetReadingLookupTable()
287  {
288  std::array<std::tuple<unsigned int, unsigned int, bool>, MaxNwords + 1> readingScheme{};
289  size_t payloadPerElem = Data_t::PayloadSize;
290  std::get<kNSTEPS>(readingScheme[0]) = 0;
291  std::get<kNELEMENTS>(readingScheme[0]) = 0;
292  std::get<kISPARTED>(readingScheme[0]) = false;
293  int countWord = 1;
294  for (int iStep = 0; iStep < getNsteps(); iStep++) {
295  if (countWord - 1 < std::get<kWORDINDEX>((GetByteLookupTable())[iStep])) { //new word
296  std::get<kNSTEPS>(readingScheme[countWord]) = iStep;
297  std::get<kNELEMENTS>(readingScheme[countWord]) = std::get<kELEMENTINDEX>((GetByteLookupTable())[iStep]);
298  if (payloadPerElem > 0) {
299  std::get<kISPARTED>(readingScheme[countWord]) = true;
300  } else {
301  std::get<kISPARTED>(readingScheme[countWord]) = false;
302  }
303  countWord++;
304  }
305  if (payloadPerElem == 0) {
306  payloadPerElem = Data_t::PayloadSize;
307  }
308  payloadPerElem -= std::get<kNBYTES>((GetByteLookupTable())[iStep]);
309  }
310  //Last step checking
311  std::get<kNSTEPS>(readingScheme[countWord]) = getNsteps();
312  if (payloadPerElem > 0) {
313  std::get<kISPARTED>(readingScheme[countWord]) = true;
314  std::get<kNELEMENTS>(readingScheme[countWord]) = std::get<kELEMENTINDEX>((GetByteLookupTable())[getNsteps() - 1]);
315  } else {
316  std::get<kISPARTED>(readingScheme[countWord]) = false;
317  std::get<kNELEMENTS>(readingScheme[countWord]) = std::get<kELEMENTINDEX>((GetByteLookupTable())[getNsteps() - 1]) + 1;
318  }
319  return readingScheme;
320  }
