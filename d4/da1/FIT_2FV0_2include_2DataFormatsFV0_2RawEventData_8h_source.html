63  std::vector<char> to_vector(bool tcm)
64  {
65  constexpr int CRUWordSize = 16;
66 
67  std::vector<char> result(size() * CRUWordSize);
68  char* out = result.data();
69  if (!tcm) {
70  std::memcpy(out, &mEventHeader, sPayloadSize);
71  out += sPayloadSize;
72  LOG(debug) << " Write PM header: nWords: " << (int)mEventHeader.nGBTWords
73  << " orbit: " << int(mEventHeader.orbit)
74  << " BC: " << int(mEventHeader.bc)
75  << " size: " << result.size();
76  printHexEventHeader();
77  out += CRUWordSize - sPayloadSize; // Padding enabled
78 
79  for (uint64_t i = 0; i < mEventHeader.nGBTWords; ++i) {
80  std::memcpy(out, &mEventData[2 * i], sPayloadSizeFirstWord);
81  out += sPayloadSizeFirstWord;
82  LOG(debug) << " 1st word: Ch: " << std::setw(2) << mEventData[2 * i].channelID
83  << " charge: " << std::setw(4) << mEventData[2 * i].charge
84  << " time: " << std::setw(4) << mEventData[2 * i].time;
85  std::memcpy(out, &mEventData[2 * i + 1], sPayloadSizeSecondWord);
86  out += sPayloadSizeSecondWord;
87  LOG(debug) << " 2nd word: Ch: " << std::setw(2) << mEventData[2 * i + 1].channelID
88  << " charge: " << std::setw(4) << mEventData[2 * i + 1].charge
89  << " time: " << std::setw(4) << mEventData[2 * i + 1].time;
90 
91  out += CRUWordSize - sPayloadSizeSecondWord - sPayloadSizeFirstWord;
92  printHexEventData(i);
93  }
94  } else {
95  // TCM data
96  std::memcpy(out, &mEventHeader, sPayloadSize);
97  out += sPayloadSize;
98  LOG(debug) << " Write TCM header: nWords: " << (int)mEventHeader.nGBTWords
99  << " orbit: " << int(mEventHeader.orbit)
100  << " BC: " << int(mEventHeader.bc)
101  << " size: " << result.size();
102  std::memcpy(out, &mTCMdata, sizeof(TCMdata));
103  out += sizeof(TCMdata);
104  // TODO: No TCM payload printing until the meaning of trigger bits and other flags is clarified
105  }
106  return result;
107  }
