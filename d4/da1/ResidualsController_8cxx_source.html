149 void ResidualsController::Print(const Option_t* opt) const
150 {
151  // print info
152  TString opts = opt;
153  opts.ToLower();
154  bool lab = opts.Contains("l");
155  printf("%5sTr.", isCosmic() ? "Cosm." : "Coll.");
156  if (isCosmic()) {
157  printf("%2d/%2d ", mTrackID >> 16, mTrackID & 0xffff);
158  } else {
159  printf("%5d ", mTrackID);
160  }
161  printf("Run:%6d Bz:%+4.1f Np: %3d q/Pt:%+.4f | Chi2: Ini: %6.1f LinSol:%6.1f Kalm:%6.1f |Vtx:%3s| TStamp:%d\n",
162  mRun, mBz, mNPoints, mQ2Pt, mChi2Ini, mChi2, mChi2K, hasVertex() ? "ON" : "OFF", mTimeStamp);
163  if (opts.Contains("r")) {
164  bool ers = opts.Contains("e");
165  printf("%5s %7s %s %7s %7s %7s %5s %5s %9s %9s",
166  " VID ", " Label ", " Alp ", " X ", " Y ", " Z ", " Snp ", " Tgl ", " DY ", " DZ ");
167  if (ers) {
168  printf(" %8s %8s %8s", " pSgYY ", " pSgYZ ", " pSgZZ ");
169  } // cluster errors
170  if (getKalmanDone()) {
171  printf(" %9s %9s", " DYK ", " DZK ");
172  if (ers) {
173  printf(" %8s %8s %8s", " tSgYY ", " tSgYZ ", " tSgZZ ");
174  } // track errors
175  }
176  printf("\n");
177  for (int i = 0; i < mNPoints; i++) {
178  float x = mX[i], y = mY[i], z = mZ[i];
179  if (lab) {
180  x = getXLab(i);
181  y = getYLab(i);
182  z = getZLab(i);
183  }
184  printf("%5d %7d %+5.2f %+7.2f %+7.2f %+7.2f %+5.2f %+5.2f %+9.2e %+9.2e",
185  mVolID[i], mLabel[i], mAlpha[i], x, y, z, mSnp[i], mTgl[i], mDY[i], mDZ[i]);
186  if (ers) {
187  printf(" %.2e %+.1e %.2e", mSigY2[i], mSigYZ[i], mSigZ2[i]);
188  }
189  if (getKalmanDone()) {
190  printf(" %+9.2e %+9.2e", mDYK[i], mDZK[i]);
191  if (ers) {
192  printf(" %.2e %+.1e %.2e", mSigY2K[i], mSigYZK[i], mSigZ2K[i]);
193  }
194  }
195  printf("\n");
196  }
197  }

201 bool ResidualsController::fillTrack(AlignmentTrack* trc, bool doKalman)
202 {
203  // fill tracks residuals info
204  int nps, np = trc->getNPoints();
205  if (trc->getInnerPoint()->containsMeasurement()) {
206  setHasVertex();
207  nps = np;
208  } else {
209  nps = np - 1;
210  } // ref point is dummy?
211  if (nps < 0) {
212  return true;
213  }
214  setCosmic(trc->isCosmic());
215  //
216  setNPoints(nps);
217  mQ2Pt = trc->getQ2Pt();
218  mChi2 = trc->getChi2();
219  mChi2Ini = trc->getChi2Ini();
220  int nfill = 0;
221  for (int i = 0; i < np; i++) {
222  AlignmentPoint* pnt = trc->getPoint(i);
223  int inv = pnt->isInvDir() ? -1 : 1; // Flag invertion for cosmic upper leg
224  if (!pnt->containsMeasurement()) {
225  continue;
226  }
227  if (!pnt->isStatOK()) {
228  pnt->incrementStat();
229  }
230  mVolID[nfill] = pnt->getVolID();
231  mLabel[nfill] = pnt->getSensor()->getInternalID();
232  mAlpha[nfill] = pnt->getAlphaSens();
233  mX[nfill] = pnt->getXPoint() * inv;
234  mY[nfill] = pnt->getYTracking();
235  mZ[nfill] = pnt->getZTracking();
236  mDY[nfill] = pnt->getResidY();
237  mDZ[nfill] = pnt->getResidZ();
238  mSigY2[nfill] = pnt->getYZErrTracking()[0];
239  mSigYZ[nfill] = pnt->getYZErrTracking()[1];
240  mSigZ2[nfill] = pnt->getYZErrTracking()[2];
241  //
242  mSnp[nfill] = pnt->getTrParamWSA()[AlignmentPoint::kParSnp];
243  mTgl[nfill] = pnt->getTrParamWSA()[AlignmentPoint::kParTgl];
244  //
245  nfill++;
246  }
247  if (nfill != nps) {
248  trc->Print("p");
249  LOG(fatal) << nfill << " residuals were stored instead of " << nps;
250  }
251  //
252  setKalmanDone(false);
253  int nfilk = 0;
254  if (doKalman && trc->residKalman()) {
255  for (int i = 0; i < np; i++) {
256  AlignmentPoint* pnt = trc->getPoint(i);
257  if (!pnt->containsMeasurement()) {
258  continue;
259  }
260  if (mVolID[nfilk] != int(pnt->getVolID())) {
261  LOG(fatal) << "Mismatch in Kalman filling for point " << i << ": filled VID:" << mVolID[nfilk] << ", point VID:" << pnt->getVolID();
262  }
263  const double* wsA = pnt->getTrParamWSA();
264  mDYK[nfilk] = pnt->getResidY();
265  mDZK[nfilk] = pnt->getResidZ();
266  mSigY2K[nfilk] = wsA[2];
267  mSigYZK[nfilk] = wsA[3];
268  mSigZ2K[nfilk] = wsA[4];
269  //
270  nfilk++;
271  }
272  //
273  mChi2K = trc->getChi2();
274  setKalmanDone(true);
275  }
276 
277  return true;
278 }

281 float ResidualsController::getXLab(int i) const
282 {
283  // cluster lab X
284  return Abs(mX[i]) * Cos(mAlpha[i]) - mY[i] * Sin(mAlpha[i]);
285 }

287 //_________________________________________________
288 float ResidualsController::getYLab(int i) const
289 {
290  // cluster lab Y
291  return Abs(mX[i]) * Sin(mAlpha[i]) + mY[i] * Cos(mAlpha[i]);
292 }
