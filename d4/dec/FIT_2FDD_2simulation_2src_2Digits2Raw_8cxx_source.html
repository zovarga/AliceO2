24 void Digits2Raw::readDigits(const std::string& outDir, const std::string& fileDigitsName)
25 {
26  LOG(info) << "==============FDD: Digits2Raw::convertDigits" << std::endl;
27  mWriter.setCarryOverCallBack(this);
28  LookUpTable lut(true);
29 
30  std::string outd = outDir;
31  if (outd.back() != '/') {
32  outd += '/';
33  }
34 
35  // Register PM links linearly
36  for (int iPmLink = 0; iPmLink < Nmodules; ++iPmLink) {
37  uint16_t feeId = uint16_t(iPmLink);
38  uint8_t linkId = uint8_t(iPmLink);
39  std::string outFileLink = mOutputPerLink ? (outd + "fdd_link" + std::to_string(iPmLink) + ".raw") : (outd + "fdd.raw");
40  LOG(info) << " Register PM link: " << iPmLink << " to file: " << outFileLink;
41  mWriter.registerLink(feeId, sCruId, linkId, sEndPointId, outFileLink);
42  }
43 
44  // Register TCM link separately
45  std::string outFileLink = mOutputPerLink ? (outd + "fdd_link" + std::to_string(sTcmLink) + ".raw") : (outd + "fdd.raw");
46  LOG(info) << " Register TCM link: " << outFileLink;
47  mWriter.registerLink(uint16_t(sTcmLink), sCruId, sTcmLink, sEndPointId, outFileLink);
48 
49  TFile* fdig = TFile::Open(fileDigitsName.data());
50  assert(fdig != nullptr);
51  LOG(info) << "Open digits file: " << fileDigitsName.data();
52 
53  TTree* digTree = (TTree*)fdig->Get("o2sim");
54  std::vector<o2::fdd::Digit> digitsBC, *fddDigitPtr = &digitsBC;
55  std::vector<o2::fdd::ChannelData> digitsCh, *fddChDataPtr = &digitsCh;
56  digTree->SetBranchAddress("FDDDigit", &fddDigitPtr);
57  digTree->SetBranchAddress("FDDDigitCh", &fddChDataPtr);
58 
59  for (int ient = 0; ient < digTree->GetEntries(); ient++) {
60  digTree->GetEntry(ient);
61  int nbc = digitsBC.size();
62  for (int ibc = 0; ibc < nbc; ibc++) {
63  auto& bcd = digitsBC[ibc];
64  auto channels = bcd.getBunchChannelData(digitsCh);
65 
66  if (!channels.empty()) {
67  LOG(debug) << "o2::fdd::Digits2Raw::readDigits(): Start to convertDigits() at ibc = " << ibc << " " << bcd.mIntRecord
68  << " iCh0:" << bcd.ref.getFirstEntry() << " nentries:" << bcd.ref.getEntries();
69  convertDigits(bcd, channels, lut);
70  }
71  }
72  }
73 }

75 void Digits2Raw::convertDigits(o2::fdd::Digit bcdigits, gsl::span<const ChannelData> pmchannels,
76  const o2::fdd::LookUpTable& lut)
77 {
78  const o2::InteractionRecord intRecord = bcdigits.getIntRecord();
79  int prevPmLink = -1;
80  int iChannelPerLink = 0;
81  int nch = pmchannels.size();
82 
83  std::stringstream ss;
84  ss << " Number of channels: " << nch << " (Ch, PMT, Q, T)\n";
85  for (int ich = 0; ich < nch; ich++) {
86  if (pmchannels[ich].mChargeADC != 0) {
87  ss << " " << std::setw(2) << ich
88  << std::setw(3) << pmchannels[ich].mPMNumber
89  << std::setw(5) << pmchannels[ich].mChargeADC
90  << std::setw(7) << std::setprecision(3) << pmchannels[ich].mTime << "\n";
91  }
92  }
93  LOG(debug) << ss.str().substr(0, ss.str().size() - 1);
94 
95  for (int ich = 0; ich < nch; ich++) {
96  int nLinkPm = lut.getLink(pmchannels[ich].mPMNumber);
97  if (nLinkPm != prevPmLink) {
98  if (prevPmLink >= 0) {
99  fillSecondHalfWordAndAddData(iChannelPerLink, prevPmLink, intRecord);
100  }
101  makeGBTHeader(mRawEventData.mEventHeader, nLinkPm, intRecord);
102  iChannelPerLink = 0;
103  prevPmLink = nLinkPm;
104  }
105  LOG(debug) << " Store data for channel: " << ich << " PmLink = " << nLinkPm << " ";
106  auto& newData = mRawEventData.mEventData[iChannelPerLink];
107  newData.charge = pmchannels[ich].mChargeADC;
108  newData.time = pmchannels[ich].mTime;
109 
110  newData.numberADC = bool(pmchannels[ich].mFEEBits & ChannelData::kNumberADC);
111  newData.isDoubleEvent = bool(pmchannels[ich].mFEEBits & ChannelData::kIsDoubleEvent);
112  newData.isTimeInfoNOTvalid = bool(pmchannels[ich].mFEEBits & ChannelData::kIsTimeInfoNOTvalid);
113  newData.isCFDinADCgate = bool(pmchannels[ich].mFEEBits & ChannelData::kIsCFDinADCgate);
114  newData.isTimeInfoLate = bool(pmchannels[ich].mFEEBits & ChannelData::kIsTimeInfoLate);
115  newData.isAmpHigh = bool(pmchannels[ich].mFEEBits & ChannelData::kIsAmpHigh);
116  newData.isEventInTVDC = bool(pmchannels[ich].mFEEBits & ChannelData::kIsEventInTVDC);
117  newData.isTimeInfoLost = bool(pmchannels[ich].mFEEBits & ChannelData::kIsTimeInfoLost);
118 
119  newData.channelID = lut.getModChannel(pmchannels[ich].mPMNumber);
120  iChannelPerLink++;
121  if (ich == nch - 1) {
122  fillSecondHalfWordAndAddData(iChannelPerLink, prevPmLink, intRecord);
123  }
124  }
125 
126  // TCM
127  makeGBTHeader(mRawEventData.mEventHeader, sTcmLink, intRecord);
128  mRawEventData.mEventHeader.nGBTWords = 1;
129  auto& tcmdata = mRawEventData.mTCMdata;
130  mTriggers = bcdigits.mTriggers;
131 
132  float ampA = mTriggers.amplA;
133  float ampC = mTriggers.amplC;
134  if (ampA > 131071) {
135  ampA = 131071; //2^17
136  }
137  if (ampC > 131071) {
138  ampC = 131071; //2^17
139  }
140  tcmdata.vertex = mTriggers.getVertex();
141  tcmdata.orA = mTriggers.getOrA();
142  tcmdata.orC = mTriggers.getOrC();
143  tcmdata.sCen = mTriggers.getSCen();
144  tcmdata.cen = mTriggers.getCen();
145  tcmdata.nChanA = mTriggers.nChanA;
146  tcmdata.nChanC = mTriggers.nChanC;
147  tcmdata.amplA = ampA;
148  tcmdata.amplC = ampC;
149  tcmdata.timeA = mTriggers.timeA;
150  tcmdata.timeC = mTriggers.timeC;
151  LOG(debug) << " TCM triggers read "
152  << " time A " << mTriggers.timeA << " time C " << mTriggers.timeC
153  << " amp A " << ampA << " amp C " << ampC
154  << " N A " << int(mTriggers.nChanA) << " N C " << int(mTriggers.nChanC)
155  << " trig "
156  << " ver " << mTriggers.getVertex() << " A " << mTriggers.getOrA() << " C " << mTriggers.getOrC();
157 
158  LOG(debug) << "TCMdata"
159  << " time A " << tcmdata.timeA << " time C " << tcmdata.timeC
160  << " amp A " << tcmdata.amplA << " amp C " << tcmdata.amplC
161  << " N A " << int(tcmdata.nChanA) << " N C " << int(tcmdata.nChanC)
162  << " trig "
163  << " ver " << tcmdata.vertex << " A " << tcmdata.orA << " C " << tcmdata.orC
164  << " size " << sizeof(tcmdata);
165 
166  auto data = mRawEventData.to_vector(kTRUE); //for tcm module
167  uint32_t linkId = uint32_t(sTcmLink);
168  uint64_t feeId = uint64_t(sTcmLink);
169  mWriter.addData(feeId, sCruId, linkId, sEndPointId, intRecord, data);
170 
171  // fill mEventData[iChannelPerLink] with 0s to flag that this is a dummy data
172  uint nGBTWords = uint((iChannelPerLink + 1) / 2);
173  if ((iChannelPerLink % 2) == 1) {
174  mRawEventData.mEventData[iChannelPerLink] = {};
175  }
176  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
177  // LOG(debug) << " last link: " << prevPmLink;
178 }

180 void Digits2Raw::makeGBTHeader(EventHeader& eventHeader, int link, o2::InteractionRecord const& mIntRecord)
181 {
182  eventHeader.startDescriptor = 0xf;
183  eventHeader.reservedField1 = 0;
184  eventHeader.reservedField2 = 0;
185  eventHeader.reservedField3 = 0;
186  eventHeader.bc = mIntRecord.bc;
187  eventHeader.orbit = mIntRecord.orbit;
188  LOG(debug) << " makeGBTHeader for link: " << link;
189 }

191 void Digits2Raw::fillSecondHalfWordAndAddData(int iChannelPerLink, int prevPmLink, const o2::InteractionRecord& ir)
192 {
193  uint nGBTWords = uint((iChannelPerLink + 1) / 2);
194  if ((iChannelPerLink % 2) == 1) {
195  mRawEventData.mEventData[iChannelPerLink] = {};
196  LOG(debug) << " Fill up empty second half-word.";
197  }
198  mRawEventData.mEventHeader.nGBTWords = nGBTWords;
199  auto data = mRawEventData.to_vector(false);
200  uint32_t linkId = uint32_t(prevPmLink);
201  uint64_t feeId = uint64_t(prevPmLink);
202  mWriter.addData(feeId, sCruId, linkId, sEndPointId, ir, data);
203  LOG(debug) << " Switch prevPmLink: " << prevPmLink << ". Save data with nGBTWords="
204  << nGBTWords << " in header. Last channel: " << iChannelPerLink;
205 }
