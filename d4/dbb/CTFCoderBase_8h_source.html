58  template <typename S>
59  void createCoder(OpType op, const o2::rans::RenormedFrequencyTable& renormedFrequencyTable, int slot)
60  {
61  if (renormedFrequencyTable.empty()) {
62  LOG(warning) << "Empty dictionary provided for slot " << slot << ", " << (op == OpType::Encoder ? "encoding" : "decoding") << " will assume literal symbols only";
63  }
64 
65  switch (op) {
66  case OpType::Encoder:
67  mCoders[slot].reset(new o2::rans::LiteralEncoder64<S>(renormedFrequencyTable));
68  break;
69  case OpType::Decoder:
70  mCoders[slot].reset(new o2::rans::LiteralDecoder64<S>(renormedFrequencyTable));
71  break;
72  }
73  }
74 
75  void clear()
76  {
77  for (auto c : mCoders) {
78  c.reset();
79  }
80  }

110 template <typename T>
111 bool CTFCoderBase::readFromTree(TTree& tree, const std::string brname, T& dest, int ev)
112 {
113  auto* br = tree.GetBranch(brname.c_str());
114  if (br && br->GetEntries() > ev) {
115  auto* ptr = &dest;
116  br->SetAddress(&ptr);
117  br->GetEntry(ev);
118  br->ResetAddress();
119  return true;
120  }
121  return false;
122 }

140 template <typename CTF>
141 std::vector<char> CTFCoderBase::readDictionaryFromFile(const std::string& dictPath, bool mayFail)
142 {
143  std::vector<char> bufVec;
144  std::unique_ptr<TFile> fileDict;
145  if (std::filesystem::exists(dictPath)) {
146  fileDict.reset(TFile::Open(dictPath.c_str()));
147  }
148  if (!fileDict || fileDict->IsZombie()) {
149  std::string errstr = fmt::format("CTF dictionary file {} for detector {} is absent", dictPath, mDet.getName());
150  if (mayFail) {
151  LOGP(info, "{}, will assume dictionary stored in CTF", errstr);
152  } else {
153  throw std::runtime_error(errstr);
154  }
155  return bufVec;
156  }
157  std::unique_ptr<TTree> tree((TTree*)fileDict->Get(std::string(o2::base::NameConf::CTFDICT).c_str()));
158  std::unique_ptr<std::vector<char>> bv((std::vector<char>*)fileDict->GetObjectChecked(o2::base::NameConf::CCDBOBJECT.data(), "std::vector<char>"));
159  if (tree) {
160  CTFHeader ctfHeader;
161  if (!readFromTree(*tree.get(), "CTFHeader", ctfHeader) || !ctfHeader.detectors[mDet]) {
162  std::string errstr = fmt::format("CTF dictionary file for detector {} is absent in the tree from file {}", mDet.getName(), dictPath);
163  if (mayFail) {
164  LOGP(info, "{}, will assume dictionary stored in CTF", errstr);
165  } else {
166  throw std::runtime_error(errstr);
167  }
168  return bufVec;
169  }
170  CTF::readFromTree(bufVec, *tree.get(), mDet.getName());
171  } else if (bv) {
172  bufVec.swap(*bv);
173  if (bufVec.size()) {
174  auto dictHeader = static_cast<o2::ctf::CTFDictHeader&>(CTF::get(bufVec.data())->getHeader());
175  if (dictHeader.det != mDet) {
176  throw std::runtime_error(fmt::format("{} contains dictionary vector for {}, expected {}", dictPath, dictHeader.det.getName(), mDet.getName()));
177  }
178  }
179  }
180  if (bufVec.size()) {
181  mExtHeader = static_cast<CTFDictHeader&>(CTF::get(bufVec.data())->getHeader());
182  LOGP(info, "Found {} in {}", mExtHeader.asString(), dictPath);
183  } else {
184  std::string errstr = fmt::format("CTF dictionary file for detector {} is empty", mDet.getName());
185  if (mayFail) {
186  LOGP(info, "{}, will assume dictionary stored in CTF", errstr);
187  } else {
188  throw std::runtime_error(errstr);
189  }
190  }
191  return bufVec;
192 }
