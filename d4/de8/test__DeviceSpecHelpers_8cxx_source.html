36 // search for an option in the device execution
37 bool search(DeviceExecution const& execution, std::string const& option, std::string const& argument)
38 {
39  bool foundOption = false;
40  for (auto const& execArg : execution.args) {
41  if (execArg == nullptr) {
42  break;
43  }
44  if (!foundOption) {
45  foundOption = option == execArg;
46  } else if (argument == execArg) {
47  return true;
48  } else {
49  // the required argument to the option is not found
50  foundOption = false;
51  }
52  }
53  return false;
54 }

56 // create the device execution from the device specs and process the command line arguments
57 // the check matrix contains a map of options to be founf per processor
58 void check(const std::vector<std::string>& arguments,
59  const std::vector<ConfigParamSpec>& workflowOptions,
60  const std::vector<DeviceSpec>& deviceSpecs,
61  CheckMatrix& matrix)
62 {
63  std::stringstream output;
64  for (auto const& arg : arguments) {
65  output << " " << arg;
66  }
67 
68  std::vector<DeviceExecution> deviceExecutions(deviceSpecs.size());
69  std::vector<DeviceControl> deviceControls(deviceSpecs.size());
70  std::vector<DataProcessorInfo> dataProcessorInfos;
71  for (auto& [name, _] : matrix) {
72  dataProcessorInfos.push_back(DataProcessorInfo{
73  name,
74  "executable-name",
75  arguments,
76  workflowOptions,
77  });
78  }
79  DeviceSpecHelpers::prepareArguments(true, true, false, 8080,
80  dataProcessorInfos,
81  deviceSpecs,
82  deviceExecutions,
83  deviceControls,
84  "workflow-id");
85 
86  for (size_t index = 0; index < deviceSpecs.size(); index++) {
87  const auto& deviceSpec = deviceSpecs[index];
88  const auto& deviceExecution = deviceExecutions[index];
89  std::stringstream execArgs;
90  for (const auto& arg : deviceExecution.args) {
91  if (arg == nullptr) {
92  // the nullptr terminates the argument list
93  break;
94  }
95  execArgs << " " << arg;
96  }
97  for (auto const& testCase : matrix[deviceSpec.name]) {
98  BOOST_TEST_INFO(std::string("can not find option: ") + testCase.first + " " + testCase.second);
99  BOOST_CHECK(search(deviceExecution, testCase.first, testCase.second));
100  }
101  }
102 }

104 BOOST_AUTO_TEST_CASE(test_prepareArguments)
105 {
106  std::vector<ConfigParamSpec> workflowOptions{
107  {"foo", VariantType::String, "bar", {"the famous foo option"}},
108  {"depth", VariantType::Int, 1, {"number of processors"}},
109  };
110 
111  auto algorithm = [](ProcessingContext& ctx) {};
112 
113  WorkflowSpec workflow{
114  {"processor0",
115  {},
116  {OutputSpec{{"output"}, "TST", "DUMMYDATA", 0, Lifetime::Timeframe}},
117  AlgorithmSpec(algorithm),
118  Options{
119  {"mode", VariantType::String, "default", {"The Mode"}},
120  }},
121  {"processor1",
122  {InputSpec{"input", "TST", "DUMMYDATA", 0, Lifetime::Timeframe}},
123  {},
124  AlgorithmSpec(algorithm),
125  Options{
126  {"mode", VariantType::String, "default", {"The Mode"}},
127  }},
128  };
129 
130  std::vector<DeviceSpec> deviceSpecs;
131 
132  std::vector<ComputingResource> resources = {ComputingResourceHelpers::getLocalhostResource()};
133  auto rm = std::make_unique<SimpleResourceManager>(resources);
134 
135  auto configContext = makeEmptyConfigContext();
136  auto channelPolicies = ChannelConfigurationPolicy::createDefaultPolicies(*configContext);
137  auto callbacksPolicies = CallbacksPolicy::createDefaultPolicies();
138 
139  DeviceSpecHelpers::dataProcessorSpecs2DeviceSpecs(workflow,
140  channelPolicies,
141  CompletionPolicy::createDefaultPolicies(),
142  callbacksPolicies,
143  deviceSpecs,
144  *rm, "workflow-id", *configContext);
145 
146  // Now doing the test cases
147  CheckMatrix matrix;
148 
149  // checking with empty command line arguments, all processors must have the options with
150  // default arguments
151  matrix["processor0"] = {{"--depth", "1"}, {"--foo", "bar"}, {"--mode", "default"}};
152  matrix["processor1"] = matrix["processor0"];
153  check({}, workflowOptions, deviceSpecs, matrix);
154 
155  // checking with unknown arguments, silently ignored, same test matrix
156  check({"--unknown", "option"}, workflowOptions, deviceSpecs, matrix);
157 
158  // configuring mode, both devices must have the option set
159  matrix["processor0"] = {{"--depth", "1"}, {"--foo", "bar"}, {"--mode", "silly"}};
160  matrix["processor1"] = matrix["processor0"];
161  check({"--mode", "silly"}, workflowOptions, deviceSpecs, matrix);
162 
163  // configuring option group, only processor0 must have the option set, processor1 default
164  matrix["processor0"] = {{"--depth", "1"}, {"--foo", "bar"}, {"--mode", "silly"}};
165  matrix["processor1"] = {{"--depth", "1"}, {"--foo", "bar"}, {"--mode", "default"}};
166  check({"--processor0", "--mode silly"}, workflowOptions, deviceSpecs, matrix);
167 
168  // processor0 must have the mode set to silly via option group, processor1 advanced from the argument
169  matrix["processor0"] = {{"--depth", "1"}, {"--foo", "bar"}, {"--mode", "silly"}};
170  matrix["processor1"] = {{"--depth", "1"}, {"--foo", "bar"}, {"--mode", "advanced"}};
171  check({"--mode", "advanced", "--processor0", "--mode silly"}, workflowOptions, deviceSpecs, matrix);
172 
173  // both devices have the workflow option propagated, others defaulted
174  matrix["processor0"] = {{"--depth", "2"}, {"--foo", "bar"}, {"--mode", "default"}};
175  matrix["processor1"] = matrix["processor0"];
176  check({"--depth", "2"}, workflowOptions, deviceSpecs, matrix);
177 
178  // both devices have the workflow option propagated, processor0 mode silly via option group
179  matrix["processor0"] = {{"--depth", "2"}, {"--foo", "bar"}, {"--mode", "silly"}};
180  matrix["processor1"] = {{"--depth", "2"}, {"--foo", "bar"}, {"--mode", "default"}};
181  check({"--depth", "2", "--processor0", "--mode silly"}, workflowOptions, deviceSpecs, matrix);
182 }
