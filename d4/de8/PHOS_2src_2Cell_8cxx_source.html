18 // split 40 bits as following:

32 void Cell::setAbsId(short absId)
33 {
34  //14 bits available
35  if (absId < kOffset) {
36  absId = kNmaxCell;
37  }
38  ULong_t t = (ULong_t)(absId - kOffset);

40  ULong_t b = getLong() & 0xffffffc000; // 1111 1111 1111 1111 1111 1111 1100 0000 0000 0000

65 void Cell::setTime(float time)
66 {
67  //10 bits available for time
68  ULong_t t = 0;
69  //Convert time to long
70  t = ULong_t((time - kTime0) / kTimeAccuracy);
71  if (t > 0x1fff) {
72  t = 0x1fff;
73  } else {
74  if (t < 0) {
75  t = 0;
76  }
77  }
78 
79  t <<= 14;
80  ULong_t b = getLong() & 0xfff8003fff; // 1111 1111 1111 1000 0000 0000 0011 1111 1111 1111
81  mBits = b + t;
82 }
83 float Cell::getTime() const
84 {
85  ULong_t t = getLong();
86  t >>= 14;
87  t &= 0x1fff;
88  //Convert back long to float
89 
90  return float(t * kTimeAccuracy) + kTime0;

93 void Cell::setEnergy(float amp)
94 {
95  //12 bits
96  ULong_t a;
97  if (getType() == HIGH_GAIN) {
98  a = static_cast<ULong_t>(amp * 4);
99  } else {
100  a = static_cast<ULong_t>(amp);
101  }
102  a = a & 0xfff; //12 bits
103 
104  a <<= 27;
105  ULong_t b = getLong() & 0x8007ffffff; // 1000 0000 0000 0111 1111 1111 1111 1111 1111 1111
106  mBits = b + a;
107 }
