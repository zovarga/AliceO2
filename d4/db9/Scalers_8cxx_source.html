100 int CTPRunScalers::readScalers(const std::string& rawscalers)
101 {
102  LOG(info) << "Loading CTP scalers.";
103  std::istringstream iss(rawscalers);
104  int ret = 0;
105  int level = 0;
106  std::string line;
107  int nclasses = 0;
108  int nlines = 0;
109  while (std::getline(iss, line)) {
110  o2::utils::Str::trim(line);
111  if ((ret = processScalerLine(line, level, nclasses)) != 0) {
112  return ret;
113  }
114  nlines++;
115  }
116  if (nlines < 4) {
117  LOG(error) << "Input string seems too small:\n"
118  << rawscalers;
119  return 6;
120  }
121  if (nclasses != 0) {
122  LOG(error) << "Wrong number of classes in final record";
123  return 6;
124  }
125  return 0;

127 int CTPRunScalers::processScalerLine(const std::string& line, int& level, int& nclasses)
128 {
129  //std::cout << "Processing line" << std::endl;
130  if (line.size() == 0) {
131  return 0;
132  }
133  if (line.at(0) == '#') {
134  return 0;
135  }
136  std::vector<std::string> tokens = o2::utils::Str::tokenize(line, ' ');
137  size_t ntokens = tokens.size();
138  if (ntokens == 0) {
139  return 0;
140  }
141  //std::cout << line << " level in::" << level << std::endl;
142  // Version
143  if (level == 0) {
144  if (ntokens != 1) {
145  LOG(error) << "Expected version in the first line";
146  return 1;
147  } else {
148  mVersion = std::stoi(tokens[0]);
149  level = 1;
150  return 0;
151  }
152  }
153  // Run Number N Classes [class indexes]
154  if (level == 1) {
155  if (ntokens < 3) {
156  LOG(error) << "Wrong syntax of second line in CTP scalers";
157  return 2;
158  } else {
159  mRunNumber = std::stol(tokens[0]);
160  int numofclasses = std::stoi(tokens[1]);
161  if ((numofclasses + 2) != ntokens) {
162  LOG(error) << "Wrong syntax of second line in CTP scalers";
163  return 3;
164  }
165  mClassMask.reset();
166  for (int i = 0; i < numofclasses; i++) {
167  int index = std::stoi(tokens[i + 2]);
168  mClassMask[index] = 1;
169  }
170  level = 2;
171  nclasses = 0;
172  return 0;
173  }
174  }
175  // Time stamp: orbit bcid linuxtime
176  if (level == 2) {
177  if (ntokens != 4) {
178  LOG(error) << "Wrong syntax of time stamp line in CTP scalers";
179  return 4;
180  } else {
181  CTPScalerRecordRaw rec;
182  rec.intRecord.orbit = std::stol(tokens[0]);
183  rec.intRecord.bc = std::stol(tokens[1]);
184  rec.seconds = std::stol(tokens[2]);
185  rec.microSeconds = std::stol(tokens[3]);
186  mScalerRecordRaw.push_back(rec);
187  level = 3;
188  return 0;
189  }
190  }
191  if (level == 3) {
192  if (ntokens != 6) {
193  LOG(error) << "Wrong syntax of counters line in CTP scalers";
194  return 5;
195  } else {
196  //std::cout << "nclasses:" << nclasses << std::endl;
197  CTPScalerRaw scaler;
198  scaler.classIndex = getClassIndexes()[nclasses];
199  scaler.lmBefore = std::stol(tokens[0]);
200  scaler.lmAfter = std::stol(tokens[1]);
201  scaler.l0Before = std::stol(tokens[2]);
202  scaler.l0After = std::stol(tokens[3]);
203  scaler.l1Before = std::stol(tokens[4]);
204  scaler.l1After = std::stol(tokens[5]);
205  (mScalerRecordRaw.back()).scalers.push_back(scaler);
206  nclasses++;
207  if (nclasses >= mClassMask.count()) {
208  level = 2;
209  nclasses = 0;
210  }
211  return 0;
212  }
213  }
214  return 0;
215 }
218 int CTPRunScalers::convertRawToO2()
219 {
220  //struct classScalersOverflows {uint32_t overflows[6];};
221  overflows_t overflows;
222  for (uint32_t i = 0; i < mClassMask.size(); i++) {
223  if (mClassMask[i]) {
224  overflows[i] = {0, 0, 0, 0, 0, 0};
225  }
226  }
227  // 1st o2 rec is just copy
228  CTPScalerRecordO2 o2rec;
229  copyRawToO2ScalerRecord(mScalerRecordRaw[0], o2rec, overflows);
230  mScalerRecordO2.push_back(o2rec);
231  for (uint32_t i = 1; i < mScalerRecordRaw.size(); i++) {
232  //update overflows
233  updateOverflows(mScalerRecordRaw[i - 1], mScalerRecordRaw[i], overflows);
234  //
235  CTPScalerRecordO2 o2rec;
236  copyRawToO2ScalerRecord(mScalerRecordRaw[i], o2rec, overflows);
237  mScalerRecordO2.push_back(o2rec);
238  // Check consistency
239  checkConsistency(mScalerRecordO2[i - 1], mScalerRecordO2[i]);
240  }
241  return 0;

243 int CTPRunScalers::copyRawToO2ScalerRecord(const CTPScalerRecordRaw& rawrec, CTPScalerRecordO2& o2rec, overflows_t& classesoverflows)
244 {
245  if (rawrec.scalers.size() != (mClassMask.count())) {
246  LOG(error) << "Inconsistent scaler record size:" << rawrec.scalers.size() << " Expected:" << mClassMask.count();
247  return 1;
248  }
249  o2rec.scalers.clear();
250  o2rec.intRecord = rawrec.intRecord;
251  o2rec.seconds = rawrec.seconds;
252  o2rec.microSeconds = rawrec.microSeconds;
253  for (uint32_t i = 0; i < rawrec.scalers.size(); i++) {
254  CTPScalerRaw rawscal = rawrec.scalers[i];
255  CTPScalerO2 o2scal;
256  int k = (getClassIndexes())[i];
257  o2scal.createCTPScalerO2FromRaw(rawscal, classesoverflows[k]);
258  o2rec.scalers.push_back(o2scal);
259  }
260  return 0;

262 int CTPRunScalers::checkConsistency(const CTPScalerO2& scal0, const CTPScalerO2& scal1) const
263 {
264  int ret = 0;
265  // Scaler should never decrease
266  if (scal0.lmBefore > scal1.lmBefore) {
267  LOG(error) << "Scaler decreasing: Class:" << scal0.classIndex << " lmBefore 0:" << scal0.lmBefore << " lmBefore :" << scal1.lmBefore;
268  ret++;
269  }
270  if (scal0.l0Before > scal1.l0Before) {
271  LOG(error) << "Scaler decreasing: Class:" << scal0.classIndex << " lmBefore 0:" << scal0.l0Before << " lmBefore :" << scal1.l0Before;
272  ret++;
273  }
274  if (scal0.l1Before > scal1.l1Before) {
275  LOG(error) << "Scaler decreasing: Class:" << scal0.classIndex << " lmBefore 0:" << scal0.l1Before << " lmBefore :" << scal1.l1Before;
276  ret++;
277  }
278  if (scal0.lmAfter > scal1.lmAfter) {
279  LOG(error) << "Scaler decreasing: Class:" << scal0.classIndex << " lmAfter 0:" << scal0.lmAfter << " lmAfter :" << scal1.lmAfter;
280  ret++;
281  }
282  if (scal0.l0After > scal1.l0After) {
283  LOG(error) << "Scaler decreasing: Class:" << scal0.classIndex << " lmAfter 0:" << scal0.l0After << " lmAfter :" << scal1.l0After;
284  ret++;
285  }
286  if (scal0.l1After > scal1.l1After) {
287  LOG(error) << "Scaler decreasing: Class:" << scal0.classIndex << " lmAfter 0:" << scal0.l1After << " lmAfter :" << scal1.l1After;
288  ret++;
289  }
290  //
291  // LMB >= LMA >= L0B >= L0A >= L1B >= L1A: 5 relations
292  //
293  if ((scal1.lmAfter - scal0.lmAfter) > (scal1.lmBefore - scal0.lmBefore)) {
294  LOG(error) << "LMA > LMB eerror";
295  ret++;
296  }
297  if ((scal1.l0After - scal0.l0After) > (scal1.l0Before - scal0.l0Before)) {
298  LOG(error) << "L0A > L0B error";
299  ret++;
300  }
301  if ((scal1.l1After - scal0.l1After) > (scal1.l1Before - scal0.l1Before)) {
302  LOG(error) << "L1A > L1B error";
303  ret++;
304  }
305  if ((scal1.l0Before - scal0.l0Before) > (scal1.lmAfter - scal0.lmAfter)) {
306  LOG(error) << "L0B > LMA error.";
307  ret++;
308  }
309  if ((scal1.l1Before - scal0.l1Before) > (scal1.l0After - scal0.l0After)) {
310  LOG(error) << "L1B > L0A Before error.";
311  ret++;
312  }
313  //
314  if (ret) {
315  scal0.printStream(std::cout);
316  scal1.printStream(std::cout);
317  }
318  return ret;
319 }
