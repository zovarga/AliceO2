56 void TrackInterpolation::process(const o2::globaltracking::RecoContainer& inp, const std::vector<GTrackID>& gids, const std::vector<o2::globaltracking::RecoContainer::GlobalIDSet>& gidTables, std::vector<o2::track::TrackParCov>& seeds, const std::vector<float>& trkTimes)
57 {
58  // main processing function
59 
60  if (!mInitDone) {
61  LOG(error) << "Initialization not yet done. Aborting...";
62  return;
63  }
64 
65  // reset output vectors
66  reset();
67 
68  // set the input containers
69  mRecoCont = &inp;
70  mGIDs = &gids;
71  mGIDtables = &gidTables;
72  mSeeds = &seeds;
73  mTrackTimes = &trkTimes;
74  mTPCTracksClusIdx = mRecoCont->getTPCTracksClusterRefs();
75  mTPCClusterIdxStruct = &mRecoCont->getTPCClusters();
76 
77  int nSeeds = mSeeds->size();
78  mTrackData.reserve(nSeeds);
79  mClRes.reserve(nSeeds * param::NPadRows);
80 
81  for (int iSeed = 0; iSeed < nSeeds; ++iSeed) {
82  if (gids[iSeed].includesDet(DetID::TRD) || gids[iSeed].includesDet(DetID::TOF)) {
83  interpolateTrack(iSeed);
84  } else {
85  extrapolateTrack(iSeed);
86  }
87  }
88 
89  LOG(info) << "Could process " << mTrackData.size() << " tracks successfully";
90 }

92 void TrackInterpolation::interpolateTrack(int iSeed)
93 {
94  TrackData trackData;
95  auto propagator = o2::base::Propagator::Instance();
96  const auto& gidTable = (*mGIDtables)[iSeed];
97  const auto& trkTPC = mRecoCont->getTPCTrack(gidTable[GTrackID::TPC]);
98  const auto& trkITS = mRecoCont->getITSTrack(gidTable[GTrackID::ITS]);
99  auto& trkWork = (*mSeeds)[iSeed];
100  // reset the cache array (sufficient to set cluster available to zero)
101  for (auto& elem : mCache) {
102  elem.clAvailable = 0;
103  }
104  trackData.clIdx.setFirstEntry(mClRes.size()); // reference the first cluster residual belonging to this track
105  float clusterTimeBinOffset = (*mTrackTimes)[iSeed] / mTPCTimeBinMUS;
106 
107  // store the TPC cluster positions in the cache
108  for (int iCl = trkTPC.getNClusterReferences(); iCl--;) {
109  uint8_t sector, row;
110  uint32_t clusterIndexInRow;
111  const auto& clTPC = trkTPC.getCluster(mTPCTracksClusIdx, iCl, *mTPCClusterIdxStruct, sector, row);
112  float clTPCX;
113  std::array<float, 2> clTPCYZ;
114  mFastTransform->TransformIdeal(sector, row, clTPC.getPad(), clTPC.getTime(), clTPCX, clTPCYZ[0], clTPCYZ[1], clusterTimeBinOffset);
115  mCache[row].clSec = sector;
116  mCache[row].clAvailable = 1;
117  mCache[row].clY = clTPCYZ[0];
118  mCache[row].clZ = clTPCYZ[1];
119  mCache[row].clAngle = o2::math_utils::sector2Angle(sector);
120  }
121 
122  // extrapolate seed through TPC and store track position at each pad row
123  for (int iRow = 0; iRow < param::NPadRows; ++iRow) {
124  if (!mCache[iRow].clAvailable) {
125  continue;
126  }
127  if (!trkWork.rotate(mCache[iRow].clAngle)) {
128  LOG(debug) << "Failed to rotate track during first extrapolation";
129  return;
130  }
131  if (!propagator->PropagateToXBxByBz(trkWork, param::RowX[iRow], mMaxSnp, mMaxStep, mMatCorr)) {
132  LOG(debug) << "Failed on first extrapolation";
133  return;
134  }
135  mCache[iRow].y[ExtOut] = trkWork.getY();
136  mCache[iRow].z[ExtOut] = trkWork.getZ();
137  mCache[iRow].sy2[ExtOut] = trkWork.getSigmaY2();
138  mCache[iRow].szy[ExtOut] = trkWork.getSigmaZY();
139  mCache[iRow].sz2[ExtOut] = trkWork.getSigmaZ2();
140  mCache[iRow].phi[ExtOut] = trkWork.getSnp();
141  mCache[iRow].tgl[ExtOut] = trkWork.getTgl();
142  //printf("Track alpha at row %i: %.2f, Y(%.2f), Z(%.2f)\n", iRow, trkWork.getAlpha(), trkWork.getY(), trkWork.getZ());
143  }
144 
145  // start from outermost cluster with outer refit and back propagation
146  if (gidTable[GTrackID::TOF].isIndexSet()) {
147  LOG(debug) << "TOF point available";
148  const auto& clTOF = mRecoCont->getTOFClusters()[gidTable[GTrackID::TOF]];
149  const int clTOFSec = clTOF.getCount();
150  const float clTOFAlpha = o2::math_utils::sector2Angle(clTOFSec);
151  if (!trkWork.rotate(clTOFAlpha)) {
152  LOG(debug) << "Failed to rotate into TOF cluster sector frame";
153  return;
154  }
155  float clTOFX = clTOF.getX();
156  std::array<float, 2> clTOFYZ{clTOF.getY(), clTOF.getZ()};
157  std::array<float, 3> clTOFCov{mSigYZ2TOF, 0.f, mSigYZ2TOF}; // assume no correlation between y and z and equal cluster error sigma^2 = (3cm)^2 / 12
158  if (!propagator->PropagateToXBxByBz(trkWork, clTOFX, mMaxSnp, mMaxStep, mMatCorr)) {
159  LOG(debug) << "Failed final propagation to TOF radius";
160  return;
161  }
162  // TODO: check if reset of covariance matrix is needed here (or, in case TOF point is not available at outermost TRD layer)
163  if (!trkWork.update(clTOFYZ, clTOFCov)) {
164  LOG(debug) << "Failed to update extrapolated ITS track with TOF cluster";
165  //LOGF(info, "trkWork.y=%f, cl.y=%f, trkWork.z=%f, cl.z=%f", trkWork.getY(), clTOFYZ[0], trkWork.getZ(), clTOFYZ[1]);
166  return;
167  }
168  }
169  if (gidTable[GTrackID::TRD].isIndexSet()) {
170  LOG(debug) << "TRD available";
171  const auto& trkTRD = mRecoCont->getITSTPCTRDTrack<o2::trd::TrackTRD>(gidTable[GTrackID::TRD]);
172  for (int iLayer = o2::trd::constants::NLAYER - 1; iLayer >= 0; --iLayer) {
173  int trkltIdx = trkTRD.getTrackletIndex(iLayer);
174  if (trkltIdx < 0) {
175  // no TRD tracklet in this layer
176  continue;
177  }
178  const auto& trdSP = mRecoCont->getTRDCalibratedTracklets()[trkltIdx];
179  const auto& trdTrklt = mRecoCont->getTRDTracklets()[trkltIdx];
180  auto trkltDet = trdTrklt.getDetector();
181  auto trkltSec = trkltDet / (o2::trd::constants::NLAYER * o2::trd::constants::NSTACK);
182  if (trkltSec != o2::math_utils::angle2Sector(trkWork.getAlpha())) {
183  if (!trkWork.rotate(o2::math_utils::sector2Angle(trkltSec))) {
184  LOG(debug) << "Track could not be rotated in TRD tracklet coordinate system in layer " << iLayer;
185  return;
186  }
187  }
188  if (!propagator->PropagateToXBxByBz(trkWork, trdSP.getX(), mMaxSnp, mMaxStep, mMatCorr)) {
189  LOG(debug) << "Failed propagation to TRD layer " << iLayer;
190  return;
191  }
192 
193  const auto* pad = mGeoTRD->getPadPlane(trkltDet);
194  float tilt = tan(TMath::DegToRad() * pad->getTiltingAngle()); // tilt is signed! and returned in degrees
195  float tiltCorrUp = tilt * (trdSP.getZ() - trkWork.getZ());
196  float zPosCorrUp = trdSP.getZ() + mRecoParam.getZCorrCoeffNRC() * trkWork.getTgl(); // maybe Z can be corrected on avarage already by the tracklet transformer?
197  float padLength = pad->getRowSize(trdTrklt.getPadRow());
198  if (!((trkWork.getSigmaZ2() < (padLength * padLength / 12.f)) && (std::fabs(trdSP.getZ() - trkWork.getZ()) < padLength))) {
199  tiltCorrUp = 0.f;
200  }
201  std::array<float, 2> trkltTRDYZ{trdSP.getY() - tiltCorrUp, zPosCorrUp};
202  std::array<float, 3> trkltTRDCov;
203  mRecoParam.recalcTrkltCov(tilt, trkWork.getSnp(), pad->getRowSize(trdTrklt.getPadRow()), trkltTRDCov);
204  if (!trkWork.update(trkltTRDYZ, trkltTRDCov)) {
205  LOG(debug) << "Failed to update track at TRD layer " << iLayer;
206  return;
207  }
208  }
209  }
210 
211  // go back through the TPC and store updated track positions
212  for (int iRow = param::NPadRows; iRow--;) {
213  if (!mCache[iRow].clAvailable) {
214  continue;
215  }
216  if (!trkWork.rotate(mCache[iRow].clAngle)) {
217  LOG(debug) << "Failed to rotate track during back propagation";
218  return;
219  }
220  if (!propagator->PropagateToXBxByBz(trkWork, param::RowX[iRow], mMaxSnp, mMaxStep, mMatCorr)) {
221  LOG(debug) << "Failed on back propagation";
222  //printf("trkX(%.2f), clX(%.2f), clY(%.2f), clZ(%.2f), alphaTOF(%.2f)\n", trkWork.getX(), param::RowX[iRow], clTOFYZ[0], clTOFYZ[1], clTOFAlpha);
223  return;
224  }
225  mCache[iRow].y[ExtIn] = trkWork.getY();
226  mCache[iRow].z[ExtIn] = trkWork.getZ();
227  mCache[iRow].sy2[ExtIn] = trkWork.getSigmaY2();
228  mCache[iRow].szy[ExtIn] = trkWork.getSigmaZY();
229  mCache[iRow].sz2[ExtIn] = trkWork.getSigmaZ2();
230  mCache[iRow].phi[ExtIn] = trkWork.getSnp();
231  mCache[iRow].tgl[ExtIn] = trkWork.getTgl();
232  }
233 
234  // calculate weighted mean at each pad row (assume for now y and z are uncorrelated) and store residuals to TPC clusters
235  unsigned short deltaRow = 0;
236  unsigned short nMeasurements = 0;
237  for (int iRow = 0; iRow < param::NPadRows; ++iRow) {
238  if (!mCache[iRow].clAvailable) {
239  ++deltaRow;
240  continue;
241  }
242  float wTotY = 1.f / mCache[iRow].sy2[ExtOut] + 1.f / mCache[iRow].sy2[ExtIn];
243  float wTotZ = 1.f / mCache[iRow].sz2[ExtOut] + 1.f / mCache[iRow].sz2[ExtIn];
244  mCache[iRow].y[Int] = (mCache[iRow].y[ExtOut] / mCache[iRow].sy2[ExtOut] + mCache[iRow].y[ExtIn] / mCache[iRow].sy2[ExtIn]) / wTotY;
245  mCache[iRow].z[Int] = (mCache[iRow].z[ExtOut] / mCache[iRow].sz2[ExtOut] + mCache[iRow].z[ExtIn] / mCache[iRow].sz2[ExtIn]) / wTotZ;
246 
247  // simple average w/o weighting for angles
248  mCache[iRow].phi[Int] = (mCache[iRow].phi[ExtOut] + mCache[iRow].phi[ExtIn]) / 2.f;
249  mCache[iRow].tgl[Int] = (mCache[iRow].tgl[ExtOut] + mCache[iRow].tgl[ExtIn]) / 2.f;
250 
251  TPCClusterResiduals res;
252  res.setDY(mCache[iRow].clY - mCache[iRow].y[Int]);
253  res.setDZ(mCache[iRow].clZ - mCache[iRow].z[Int]);
254  res.setY(mCache[iRow].y[Int]);
255  res.setZ(mCache[iRow].z[Int]);
256  res.setPhi(mCache[iRow].phi[Int]);
257  res.setTgl(mCache[iRow].tgl[Int]);
258  res.sec = mCache[iRow].clSec;
259  res.dRow = deltaRow;
260  res.row = iRow;
261  mClRes.push_back(std::move(res));
262  ++nMeasurements;
263  deltaRow = 1;
264  }
265 
266  trackData.gid = (*mGIDs)[iSeed];
267  trackData.eta = trkTPC.getEta();
268  trackData.phi = trkTPC.getSnp();
269  trackData.qPt = trkTPC.getQ2Pt();
270  trackData.chi2TPC = trkTPC.getChi2();
271  trackData.chi2ITS = trkITS.getChi2();
272  trackData.nClsTPC = trkTPC.getNClusterReferences();
273  trackData.nClsITS = trkITS.getNumberOfClusters();
274  trackData.nTrkltsTRD = gidTable[GTrackID::TRD].isIndexSet() ? mRecoCont->getITSTPCTRDTrack<o2::trd::TrackTRD>(gidTable[GTrackID::TRD]).getNtracklets() : 0;
275  trackData.clAvailTOF = gidTable[GTrackID::TOF].isIndexSet() ? 1 : 0;
276  trackData.clIdx.setEntries(nMeasurements);
277 
278  mTrackData.push_back(std::move(trackData));
279  mGIDsSuccess.push_back((*mGIDs)[iSeed]);
280 }

282 void TrackInterpolation::extrapolateTrack(int iSeed)
283 {
284  // extrapolate ITS-only track through TPC and store residuals to TPC clusters in the output vectors
285  const auto& gidTable = (*mGIDtables)[iSeed];
286  TrackData trackData;
287  trackData.clIdx.setFirstEntry(mClRes.size());
288  const auto& trkITS = mRecoCont->getITSTrack(gidTable[GTrackID::ITS]);
289  const auto& trkTPC = mRecoCont->getTPCTrack(gidTable[GTrackID::TPC]);
290  auto& trkWork = (*mSeeds)[iSeed];
291  float clusterTimeBinOffset = (*mTrackTimes)[iSeed] / mTPCTimeBinMUS;
292  auto propagator = o2::base::Propagator::Instance();
293  unsigned short rowPrev = 0;
294  unsigned short nMeasurements = 0;
295  for (int iCl = trkTPC.getNClusterReferences(); iCl--;) {
296  uint8_t sector, row;
297  uint32_t clusterIndexInRow;
298  const auto& cl = trkTPC.getCluster(mTPCTracksClusIdx, iCl, *mTPCClusterIdxStruct, sector, row);
299  float x = 0, y = 0, z = 0;
300  mFastTransform->TransformIdeal(sector, row, cl.getPad(), cl.getTime(), x, y, z, clusterTimeBinOffset);
301  if (!trkWork.rotate(o2::math_utils::sector2Angle(sector))) {
302  return;
303  }
304  if (!propagator->PropagateToXBxByBz(trkWork, x, mMaxSnp, mMaxStep, mMatCorr)) {
305  return;
306  }
307  TPCClusterResiduals res;
308  res.setDY(y - trkWork.getY());
309  res.setDY(z - trkWork.getZ());
310  res.setY(trkWork.getY());
311  res.setZ(trkWork.getZ());
312  res.setPhi(trkWork.getSnp());
313  res.setTgl(trkWork.getTgl());
314  res.sec = sector;
315  res.dRow = row - rowPrev;
316  res.row = row;
317  rowPrev = row;
318  mClRes.push_back(std::move(res));
319  ++nMeasurements;
320  }
321  trackData.gid = (*mGIDs)[iSeed];
322  trackData.eta = trkTPC.getEta();
323  trackData.phi = trkTPC.getSnp();
324  trackData.qPt = trkTPC.getQ2Pt();
325  trackData.chi2TPC = trkTPC.getChi2();
326  trackData.chi2ITS = trkITS.getChi2();
327  trackData.nClsTPC = trkTPC.getNClusterReferences();
328  trackData.nClsITS = trkITS.getNumberOfClusters();
329  trackData.clIdx.setEntries(nMeasurements);
330 
331  mTrackData.push_back(std::move(trackData));
332  mGIDsSuccess.push_back((*mGIDs)[iSeed]);
333 }
