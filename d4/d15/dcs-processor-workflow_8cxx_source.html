85 /*
86 * Send a DPMAP to the output.
87 *
88 * @param dpmap a map of string to vector of DataPointValue
89 * @param output a DPL data allocator
90 * @param info a CCDB object info describing the dpmap
91 * @param reason (optional, can be empty) a string description why the dpmap
92 * was ready to be shipped (e.g. big enough, long enough, end of process, etc...)
93 */
94 void sendOutput(const DPMAP& dpmap, o2::framework::DataAllocator& output, o2::ccdb::CcdbObjectInfo info, const std::string& reason)
95 {
96  if (dpmap.empty()) {
97  // we do _not_ write empty objects
98  return;
99  }
100  auto md = info.getMetaData();
101  md["upload reason"] = reason;
102  info.setMetaData(md);
103  auto image = o2::ccdb::CcdbApi::createObjectImage(&dpmap, &info);
104  LOG(info) << "Sending object " << info.getPath() << "/"
105  << info.getFileName() << " of size " << image->size()
106  << " bytes, valid for " << info.getStartValidityTimestamp()
107  << " : " << info.getEndValidityTimestamp()
108  << " | reason: " << reason;
109  output.snapshot(o2::framework::Output{Utils::gDataOriginCDBPayload, CCDBOBJ, 0}, *image.get());
110  output.snapshot(o2::framework::Output{Utils::gDataOriginCDBWrapper, CCDBOBJ, 0}, info);
111 }

126 /*
127 * Compute the (approximate) size (in KB) of a dpmap.
128 */
129 size_t computeSize(const DPMAP& dpmap)
130 {
131  constexpr int itemSize = 64; // DataPointIdentifier or DataPointValue have the same size = 64 bytes
132  constexpr float byte2KB = 1.0 / 1024;
133 
134  size_t nofItems = 0;
135  for (auto did : dpmap) {
136  nofItems++; // +1 for the DataPointIdentifier itself
137  nofItems += did.second.size(); // +number of DataPointValues
138  }
139  return static_cast<size_t>(std::floor(nofItems * itemSize * byte2KB));
140 }

142 /*
143 * Compute the duration (in seconds) span by the datapoints in the dpmap.
144 */
145 int computeDuration(const DPMAP& dpmap)
146 {
147  uint64_t minTime{std::numeric_limits<uint64_t>::max()};
148  uint64_t maxTime{0};
149 
150  for (auto did : dpmap) {
151  for (auto d : did.second) {
152  minTime = std::min(minTime, d.get_epoch_time());
153  maxTime = std::max(maxTime, d.get_epoch_time());
154  }
155  }
156  return static_cast<int>((maxTime - minTime) / 1000);
157 }

159 /*
160 * Decides whether or not the dpmap should be sent to the output.
161 *
162 * @param maxSize if the dpmap size is above this size,
163 * then it should go to output
164 * @param maxDuration if the dpmap spans more than this duration,
165 * then it should go to output
166 *
167 * @returns a boolean stating if the dpmap should be output and a string
168 * describing why it should be output.
169 */
170 std::tuple<bool, std::string> needOutput(const DPMAP& dpmap, int maxSize, int maxDuration)
171 {
172  std::string reason;
173 
174  if (dpmap.empty()) {
175  return {false, reason};
176  }
177 
178  bool bigEnough{false};
179  bool longEnough{false};
180  bool complete{true}; // FIXME: should check here that we indeed have all our dataPoints
181 
182  if (maxSize && (computeSize(dpmap) > maxSize)) {
183  bigEnough = true;
184  reason += "[big enough]";
185  }
186 
187  if (maxDuration) {
188  auto seconds = computeDuration(dpmap);
189  if (seconds > maxDuration) {
190  longEnough = true;
191  reason += fmt::format("[long enough ({} s)]", seconds);
192  }
193  }
194 
195  return {complete && (bigEnough || longEnough), reason};
196 }

209 /*
210 * Process the datapoints received.
211 *
212 * The datapoints are accumulated into one (MID) or two (MCH) DPMAPs (map from
213 * alias names to vector of DataPointValue) : one for HV values (MID and MCH)
214 * and one for LV values (MCH only).
215 *
216 * If the DPMAPs satisfy certain conditions (@see needOutput) they are sent to
217 * the output.
218 *
219 * @param aliases an array of one or two vectors of aliases (one for HV values,
220 * one for LV values)
221 * @param maxSize an array of one or two values for the
222 * maxsizes of the HV and LV values respectively
223 * @param maxDuration an array of
224 * one or two values for the max durations of the HV and LV values respectively
225 */
226 void processDataPoints(o2::framework::ProcessingContext& pc,
227  std::array<std::vector<std::string>, NOBJECTS> aliases,
228  std::array<int, NOBJECTS> maxSize,
229  std::array<int, NOBJECTS> maxDuration)
230 {
231 
232  auto tfid = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("input").header)->startTime;
233  if (t0 < 0) {
234  t0 = tfid;
235  }
236  auto dps = pc.inputs().get<gsl::span<o2::dcs::DataPointCompositeObject>>("input");
237  for (auto dp : dps) {
238  //FIXME: check we're not adding twice the same dp (i.e. check timestamp ?)
239  for (auto i = 0; i < NOBJECTS; i++) {
240  if (std::find(aliases[i].begin(), aliases[i].end(), dp.id.get_alias()) != aliases[i].end()) {
241  dataPoints[i][dp.id].emplace_back(dp.data);
242  }
243  }
244  }
245  for (auto i = 0; i < NOBJECTS; i++) {
246  auto [shouldOutput, reason] = needOutput(dataPoints[i], maxSize[i], maxDuration[i]);
247  if (shouldOutput) {
248  auto inf = addTFInfo(info[i], t0, tfid);
249  sendOutput(dataPoints[i], pc.outputs(), inf, reason);
250  t0 = tfid;
251  dataPoints[i].clear(); //FIXME: here the clear should be more clever and keep at least one value per dp ?
252  }
253  }
254 }

265 /*
266 * Creates the main processing function.
267 *
268 * @param ic InitContext which is used to get the options and set the end of
269 * stream callback
270 */
271 o2::framework::AlgorithmSpec::ProcessCallback createProcessFunction(o2::framework::InitContext& ic)
272 {
273  auto& callbacks = ic.services().get<o2::framework::CallbackService>();
274  callbacks.set(o2::framework::CallbackService::Id::EndOfStream, endOfStream);
275 
276  // the aliases arrays contain all the names of the MCH or MID data points
277  // we are interested to transit to the CCDB
278 #if defined(MUON_SUBSYSTEM_MCH)
279  std::array<std::vector<std::string>, NOBJECTS> aliases = {
280  replaceDotByUnderscore(o2::mch::dcs::aliases({o2::mch::dcs::MeasurementType::HV_V,
281  o2::mch::dcs::MeasurementType::HV_I})),
282  replaceDotByUnderscore(o2::mch::dcs::aliases({o2::mch::dcs::MeasurementType::LV_V_FEE_ANALOG,
283  o2::mch::dcs::MeasurementType::LV_V_FEE_DIGITAL,
284  o2::mch::dcs::MeasurementType::LV_V_SOLAR}))};
285  std::array<int, 2> maxSize{
286  ic.options().get<int>("hv-max-size"),
287  ic.options().get<int>("lv-max-size")};
288 
289  std::array<int, 2> maxDuration{
290  ic.options().get<int>("hv-max-duration"),
291  ic.options().get<int>("lv-max-duration")};
292 #elif defined(MUON_SUBSYSTEM_MID)
293  std::array<std::vector<std::string>, NOBJECTS> aliases = {
294  replaceDotByUnderscore(o2::mid::dcs::aliases({o2::mid::dcs::MeasurementType::HV_V,
295  o2::mid::dcs::MeasurementType::HV_I}))};
296  std::array<int, NOBJECTS> maxSize{ic.options().get<int>("hv-max-size")};
297  std::array<int, NOBJECTS> maxDuration{ic.options().get<int>("hv-max-duration")};
298 #endif
299 
300  for (auto i = 0; i < NOBJECTS; i++) {
301  dataPoints[i].clear();
302  }
303 
304  return [aliases, maxSize, maxDuration](o2::framework::ProcessingContext& pc) {
305  processDataPoints(pc, aliases, maxSize, maxDuration);
306  };
307 }
