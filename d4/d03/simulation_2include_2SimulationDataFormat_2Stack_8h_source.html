191  {
192  Reset();
193  for (auto p : primaries) {
194  Int_t doTrack = 0;
195  if (p.TestBit(ParticleStatus::kToBeDone)) {
196  doTrack = 1;
197  }
198  PushTrack(doTrack, p);
199  }
200  mNumberOfPrimaryParticles = primaries.size();
201  mNumberOfEntriesInParticles = mNumberOfPrimaryParticles;
202  }

321 inline void Stack::addTrackReference(const o2::TrackReference& ref)
322 {
323  if (mIndexOfCurrentTrack >= mNumberOfPrimaryParticles) {
324  Int_t iTrack = mTrackIDtoParticlesEntry[mIndexOfCurrentTrack];
325  auto& part = mParticles[iTrack];
326  part.setStore(true);
327  }
328  mTrackRefs->push_back(ref);
