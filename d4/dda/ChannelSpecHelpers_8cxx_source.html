67 std::string composeIPCName(std::string const& prefix, std::string const& hostname, short port)
68 {
69  if (prefix == "@") {
70  return fmt::format("ipc://@{}_{},transport=shmem", hostname, port);
71  }
72  if (prefix.back() == '/') {
73  return fmt::format("ipc://{}{}_{},transport=shmem", prefix, hostname, port);
74  }
75  return fmt::format("ipc://{}/{}_{},transport=shmem", prefix, hostname, port);

126 void ChannelSpecHelpers::parseChannelConfig(char const* config, FairMQChannelConfigParser& handler)
127 {
128  ChannelConfigParserState state = ChannelConfigParserState::BEGIN;
129  char const* cur = config;
130  char const* next = config;
131  std::string_view key;
132  std::string_view value;
133  char const* nameKey = "name";
134 
135  while (true) {
136  switch (state) {
137  case ChannelConfigParserState::BEGIN: {
138  if (*cur == '\0') {
139  state = ChannelConfigParserState::ERROR;
140  } else if (!isalpha(*cur)) {
141  state = ChannelConfigParserState::ERROR;
142  } else {
143  state = ChannelConfigParserState::BEGIN_CHANNEL;
144  }
145  break;
146  }
147  case ChannelConfigParserState::BEGIN_CHANNEL: {
148  next = strpbrk(cur, ":=;,");
149  if (*next == ';' || *next == ',') {
150  state = ChannelConfigParserState::ERROR;
151  break;
152  } else if (*next == ':') {
153  handler.beginChannel();
154  key = std::string_view(nameKey, 4);
155  value = std::string_view(cur, next - cur);
156  handler.property(key, value);
157  state = ChannelConfigParserState::BEGIN_KEY;
158  cur = next + 1;
159  break;
160  }
161  handler.beginChannel();
162  state = ChannelConfigParserState::BEGIN_KEY;
163  break;
164  }
165  case ChannelConfigParserState::BEGIN_KEY: {
166  next = strchr(cur, '=');
167  if (next == nullptr) {
168  state = ChannelConfigParserState::ERROR;
169  } else {
170  key = std::string_view(cur, next - cur);
171  state = ChannelConfigParserState::BEGIN_VALUE;
172  cur = next + 1;
173  }
174  break;
175  }
176  case ChannelConfigParserState::BEGIN_VALUE: {
177  next = strpbrk(cur, ";,");
178  if (next == nullptr) {
179  size_t l = strlen(cur);
180  value = std::string_view(cur, l);
181  state = ChannelConfigParserState::END_VALUE;
182  cur = cur + l;
183  } else if (*next == ';') {
184  value = std::string_view(cur, next - cur);
185  state = ChannelConfigParserState::END_CHANNEL;
186  cur = next;
187  } else if (*next == ',') {
188  value = std::string_view(cur, next - cur);
189  state = ChannelConfigParserState::END_VALUE;
190  cur = next;
191  }
192  handler.property(key, value);
193  break;
194  }
195  case ChannelConfigParserState::END_VALUE: {
196  if (*cur == '\0') {
197  state = ChannelConfigParserState::END_CHANNEL;
198  } else if (*cur == ',') {
199  state = ChannelConfigParserState::BEGIN_KEY;
200  cur++;
201  } else if (*cur == ';') {
202  state = ChannelConfigParserState::END_CHANNEL;
203  cur++;
204  }
205  break;
206  }
207  case ChannelConfigParserState::END_CHANNEL: {
208  handler.endChannel();
209  if (*cur == '\0') {
210  state = ChannelConfigParserState::END;
211  } else if (*cur == ';') {
212  state = ChannelConfigParserState::BEGIN_CHANNEL;
213  cur++;
214  } else {
215  state = ChannelConfigParserState::ERROR;
216  }
217  break;
218  }
219  case ChannelConfigParserState::END: {
220  return;
221  }
222  case ChannelConfigParserState::ERROR: {
223  throw runtime_error("Unable to parse channel config");
224  }
225  }
226  }
227 }
