65 static inline auto doExtractTypedOriginal(framework::pack<Ts...>, ProcessingContext& pc)
66 {

68  return pc.inputs().get<TableConsumer>(aod::MetadataTrait<framework::pack_element_t<0, framework::pack<Ts...>>>::metadata::tableLabel())->asArrowTable();

70  return std::vector{pc.inputs().get<TableConsumer>(aod::MetadataTrait<Ts>::metadata::tableLabel())->asArrowTable()...};

86 AlgorithmSpec AODReaderHelpers::indexBuilderCallback(std::vector<InputSpec>& requested)
87 {
88  return AlgorithmSpec::InitCallback{[requested](InitContext& ic) {
89  return [requested](ProcessingContext& pc) {
90  auto outputs = pc.outputs();
91  // spawn tables
92  for (auto& input : requested) {
93  auto&& [origin, description] = DataSpecUtils::asConcreteDataTypeMatcher(input);
94  auto maker = [&](auto metadata) {
95  using metadata_t = decltype(metadata);
96  using Key = typename metadata_t::Key;
97  using index_pack_t = typename metadata_t::index_pack_t;
98  using sources = typename metadata_t::originals;
99  if constexpr (metadata_t::exclusive == true) {
100  return o2::framework::IndexExclusive::indexBuilder(input.binding.c_str(), index_pack_t{},
101  extractTypedOriginal<Key>(pc),
102  extractOriginalsTuple(sources{}, pc));
103  } else {
104  return o2::framework::IndexSparse::indexBuilder(input.binding.c_str(), index_pack_t{},
105  extractTypedOriginal<Key>(pc),
106  extractOriginalsTuple(sources{}, pc));
107  }
108  };
109 
110  if (description == header::DataDescription{"MA_RN2_EX"}) {
111  outputs.adopt(Output{origin, description}, maker(o2::aod::Run2MatchedExclusiveMetadata{}));
112  } else if (description == header::DataDescription{"MA_RN2_SP"}) {
113  outputs.adopt(Output{origin, description}, maker(o2::aod::Run2MatchedSparseMetadata{}));
114  } else if (description == header::DataDescription{"MA_RN3_EX"}) {
115  outputs.adopt(Output{origin, description}, maker(o2::aod::Run3MatchedExclusiveMetadata{}));
116  } else if (description == header::DataDescription{"MA_RN3_SP"}) {
117  outputs.adopt(Output{origin, description}, maker(o2::aod::Run3MatchedSparseMetadata{}));
118  } else if (description == header::DataDescription{"MA_BCCOL_EX"}) {
119  outputs.adopt(Output{origin, description}, maker(o2::aod::MatchedBCCollisionsExclusiveMetadata{}));
120  } else if (description == header::DataDescription{"MA_BCCOL_SP"}) {
121  outputs.adopt(Output{origin, description}, maker(o2::aod::MatchedBCCollisionsSparseMetadata{}));
122  } else if (description == header::DataDescription{"MA_RN3_BC_SP"}) {
123  outputs.adopt(Output{origin, description}, maker(o2::aod::Run3MatchedToBCSparseMetadata{}));
124  } else if (description == header::DataDescription{"MA_RN3_BC_EX"}) {
125  outputs.adopt(Output{origin, description}, maker(o2::aod::Run3MatchedToBCExclusiveMetadata{}));
126  } else if (description == header::DataDescription{"MA_RN2_BC_SP"}) {
127  outputs.adopt(Output{origin, description}, maker(o2::aod::Run2MatchedToBCSparseMetadata{}));
128  } else {
129  throw std::runtime_error("Not an index table");
130  }
131  }
132  };
133  }};
134 }

136 AlgorithmSpec AODReaderHelpers::aodSpawnerCallback(std::vector<InputSpec>& requested)
137 {
138  return AlgorithmSpec::InitCallback{[requested](InitContext& ic) {
139  return [requested](ProcessingContext& pc) {
140  auto outputs = pc.outputs();
141  // spawn tables
142  for (auto& input : requested) {
143  auto&& [origin, description] = DataSpecUtils::asConcreteDataTypeMatcher(input);
144 
145  auto maker = [&](auto metadata) {
146  using metadata_t = decltype(metadata);
147  using expressions = typename metadata_t::expression_pack_t;
148  std::vector<std::shared_ptr<arrow::Table>> originalTables;
149  for (auto& i : input.metadata) {
150  if ((i.type == VariantType::String) && (i.name.find("input:") != std::string::npos)) {
151  auto spec = DataSpecUtils::fromMetadataString(i.defaultValue.get<std::string>());
152  originalTables.push_back(pc.inputs().get<TableConsumer>(spec.binding)->asArrowTable());
153  }
154  }
155  return o2::framework::spawner(expressions{}, std::move(originalTables), input.binding.c_str());
156  };
157 
158  if (description == header::DataDescription{"TRACK"}) {
159  outputs.adopt(Output{origin, description}, maker(o2::aod::TracksExtensionMetadata{}));
160  } else if (description == header::DataDescription{"TRACKCOV"}) {
161  outputs.adopt(Output{origin, description}, maker(o2::aod::TracksCovExtensionMetadata{}));
162  } else if (description == header::DataDescription{"TRACKEXTRA"}) {
163  outputs.adopt(Output{origin, description}, maker(o2::aod::TracksExtraExtensionMetadata{}));
164  } else if (description == header::DataDescription{"MFTTRACK"}) {
165  outputs.adopt(Output{origin, description}, maker(o2::aod::MFTTracksExtensionMetadata{}));
166  } else if (description == header::DataDescription{"FWDTRACK"}) {
167  outputs.adopt(Output{origin, description}, maker(o2::aod::FwdTracksExtensionMetadata{}));
168  } else if (description == header::DataDescription{"FWDTRACKCOV"}) {
169  outputs.adopt(Output{origin, description}, maker(o2::aod::FwdTracksCovExtensionMetadata{}));
170  } else if (description == header::DataDescription{"MCPARTICLE"}) {
171  outputs.adopt(Output{origin, description}, maker(o2::aod::McParticles_000ExtensionMetadata{}));
172  } else if (description == header::DataDescription{"MCPARTICLE_001"}) {
173  outputs.adopt(Output{origin, description}, maker(o2::aod::McParticles_001ExtensionMetadata{}));
174  } else {
175  throw runtime_error("Not an extended table");
176  }
177  }
178  };
179  }};
180 }
