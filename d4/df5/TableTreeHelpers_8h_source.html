24 // TableToTree allows to save the contents of a given arrow::Table into
25 // a TTree
26 // ColumnToBranch is used by GenericTableToTree
27 //
28 // To write the contents of a table ta to a tree tr on file f do:
29 // . GenericTableToTree t2t(ta, f,treename);
30 // . t2t.addBranches();
31 // OR t2t.addBranch(column.get(), field.get()), ...;
32 // . t2t.process();
33 //
34 // .............................................................................
35 // -----------------------------------------------------------------------------
36 // TreeToTable allows to fill the contents of a given TTree to an arrow::Table
37 // ColumnIterator is used by TreeToTable
38 //
39 // To copy the contents of a tree tr to a table ta do:
40 // . TreeToTable t2t(tr);
41 // . t2t.addColumn(columnname1); t2t.addColumn(columnname2); ...
42 // OR
43 // t2t.addAllColumns();
44 // . auto ta = t2t.process();
45 //
46 // .............................................................................
