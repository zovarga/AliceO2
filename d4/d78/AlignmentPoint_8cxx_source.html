49 void AlignmentPoint::init()
50 {
51  // compute aux info
52  const double kCorrToler = 1e-6;
53  const double kDiagToler = 1e-14;
54  //
55  // compute parameters of tranformation to diagonal error matrix
56  if (!isZeroPos(mErrYZTracking[0] + mErrYZTracking[2])) {
57  //
58  // is there a correlation?
59  if (smallerAbs(mErrYZTracking[1] * mErrYZTracking[1] / (mErrYZTracking[0] * mErrYZTracking[2]), kCorrToler)) {
60  mCosDiagErr = 1.;
61  mSinDiagErr = 0.;
62  mErrDiag[0] = mErrYZTracking[0];
63  mErrDiag[1] = mErrYZTracking[2];
64  } else {
65  double dfd = 0.5 * (mErrYZTracking[2] - mErrYZTracking[0]);
66  double phi = 0;
67  // special treatment if errors are equal
68  if (Abs(dfd) < kDiagToler) {
69  phi = mErrYZTracking[1] > 0 ? (Pi() * 0.25) : (Pi() * 0.75);
70  } else {
71  phi = 0.5 * ATan2(mErrYZTracking[1], dfd);
72  }
73  //
74  mCosDiagErr = Cos(phi);
75  mSinDiagErr = Sin(phi);
76  //
77  // double det = dfd*dfd + mErrYZTracking[1]*mErrYZTracking[1];
78  // det = det>0 ? Sqrt(det) : 0;
79  // double smd = 0.5*(mErrYZTracking[0] + mErrYZTracking[2]);
80  // mErrDiag[0] = smd + det;
81  // mErrDiag[1] = smd - det;
82  double xterm = 2 * mCosDiagErr * mSinDiagErr * mErrYZTracking[1];
83  double cc = mCosDiagErr * mCosDiagErr;
84  double ss = mSinDiagErr * mSinDiagErr;
85  mErrDiag[0] = mErrYZTracking[0] * cc + mErrYZTracking[2] * ss - xterm;
86  mErrDiag[1] = mErrYZTracking[0] * ss + mErrYZTracking[2] * cc + xterm;
87  }
88  }
89  //
90 }

232 int AlignmentPoint::Compare(const TObject* b) const
233 {
234  // sort points in direction opposite to track propagation, i.e.
235  // 1) for tracks from collision: range in decreasing tracking X
236  // 2) for cosmic tracks: upper leg (pnt->isInvDir()==true) ranged in increasing X
237  // lower leg - in decreasing X
238  AlignmentPoint* pnt = (AlignmentPoint*)b;
239  double x = getXPoint();
240  double xp = pnt->getXPoint();
241  if (!isInvDir()) { // track propagates from low to large X via this point
242  if (!pnt->isInvDir()) { // via this one also
243  return x > xp ? -1 : 1;
244  } else {
245  return -1;
246  } // range points of lower leg 1st
247  } else { // this point is from upper cosmic leg: track propagates from large to low X
248  if (pnt->isInvDir()) { // this one also
249  return x > xp ? 1 : -1;
250  } else {
251  return 1;
252  } // other point is from lower leg
253  }
254  //
255 }

258 void AlignmentPoint::getXYZGlo(double r[3]) const
259 {
260  // position in lab frame
261  double cs = TMath::Cos(mAlphaSens);
262  double sn = TMath::Sin(mAlphaSens);
263  double x = getXPoint();
264  r[0] = x * cs - getYTracking() * sn;
265  r[1] = x * sn + getYTracking() * cs;
266  r[2] = getZTracking();
