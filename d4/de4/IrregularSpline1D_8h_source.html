274 template <typename T>
275 GPUdi() T IrregularSpline1D::getSpline(const IrregularSpline1D::Knot& knot1, T f0, T f1, T f2, T f3, float u)
276 {
279 
280  f0 -= f1;
281  f2 -= f1;
282  f3 -= f1;
283 
284  T x = T((u - knot1.u) * knot1.scale); // scaled u
285  T z1 = T(f0 * knot1.scaleL0 + f2 * knot1.scaleL2); // scaled u derivative at the knot 1
286  T z2 = T(f2 * knot1.scaleR2 + f3 * knot1.scaleR3); // scaled u derivative at the knot 2
287 
288  T x2 = x * x;
289 
290  // f(x) = ax^3 + bx^2 + cx + d
291  //
292  // f(0) = f1
293  // f(1) = f2
294  // f'(0) = z1
295  // f'(1) = z2
296  //
297 
298  // T d = f1;
299  // T c = z1;
300  T a = -f2 - f2 + z1 + z2;
301  T b = f2 - z1 - a;
302  return a * x * x2 + b * x2 + z1 * x + f1;
303 }

328 GPUdi() void IrregularSpline1D::getEdgeCorrectionCoefficients(double u0, double u1, double u2, double u3, double& c0, double& c1, double& c2, double& c3)
329 {
333 
334  double du = u2 - u1;
335  double x0 = (u0 - u1) / du;
336  // double x1 = ( u1 - u1 )/du = 0
337  // double x2 = ( u2 - u1 )/du = 1
338  double x3 = (u3 - u1) / du;
339 
340  double cL0 = -1. / (x0 * (x0 - 1.));
341  double cL2 = x0 / (x0 - 1.);
342  double cR2 = (x3 - 2.) / (x3 - 1.);
343  double cR3 = 1. / (x3 * (x3 - 1.));
344 
345  // Fi = fi - f1
346 
347  // z1 = F0corr*cL0 + F2*cL2; // scaled u derivative at the knot 1
348  // z2 = F2*cR2 + F3*cR3; // scaled u derivative at the knot 2
349 
350  // a = -2F2 + z1 + z2
351  // b = 3F2 - 2*z1 - z2 ;
352  // c = z1
353  // d = 0
354 
355  // F(x) = ax^3 + bx^2 + cx + d
356  // F(x) = (-2*x^3+3*x^2)*F2 + (x^3-2*x^2+x)*z1 + (x^3-x^2)*z2
357  // F(x) = (-2*x^3+3*x^2)*F2 + (x^3-2*x^2+x)*z1 + (x^3-x^2)*(cR2*F2 + cR3*F3)
358  // F(x) = (-2*x^3+3*x^2+(x^3-x^2)*cR2)*F2 + (x^3-2*x^2+x)*z1 + (x^3-x^2)*cR3*F3
359  // F(x) = x^2(-2*x+3+(x-1)*cR2)*F2 + x*(x-1)^2*z1 + x^2(x-1)*cR3*F3
360 
361  // z1*x*(x-1)^2 = F(x) - x^2(-2*x+3+(x-1)*cR2)*F2 - x^2(x-1)*cR3*F3
362  // z1*x0*(x0-1)^2 = F0 - x0^2(-2*x0+3+(x0-1)*cR2)*F2 - x0^2(x0-1)*cR3*F3
363 
364  double x01 = x0 - 1.;
365 
366  // coeff. for z1 at F0, F1, F2, F3:
367 
368  c0 = 1. / (x0 * x01 * x01);
369  c1 = 0;
370  c2 = -(-2 * x0 + 3. + x01 * cR2) * x0 / (x01 * x01);
371  c3 = -x0 * cR3 / x01;
372 
373  // F0corr = (z1 - F2*cL2)/cL0;
374 
375  c0 = c0 / cL0;
376  c1 = c1 / cL0;
377  c2 = (c2 - cL2) / cL0;
378  c3 = c3 / cL0;
379 
380  // F0corr = c0*F0 + c1*F1 + C2*F2 + c3*F3;
381  // f0corr-f1 = c0*(f0-f1) + c1*(f1-f1) + c2*(f2-f1) + c3*(f3-f1);
382  // f0corr-f1 = c0*(f0-f1) + c2*(f2-f1) + c3*(f3-f1);
383  // f0corr = c0*f0 + c2*f2 + c3*f3 + (1-c0-c2-c3)*f1;
384 
385  c1 = 1. - c0 - c2 - c3;
386 }

388 template <typename T>
389 GPUdi() void IrregularSpline1D::correctEdges(T* data) const
390 {
391  const IrregularSpline1D::Knot* s = getKnots();
392  double c0, c1, c2, c3;
393  getEdgeCorrectionCoefficients(s[0].u, s[1].u, s[2].u, s[3].u, c0, c1, c2, c3);
394  data[0] = c0 * data[0] + c1 * data[1] + c2 * data[2] + c3 * data[3];
395  int i = mNumberOfKnots - 1;
396  getEdgeCorrectionCoefficients(s[i - 0].u, s[i - 1].u, s[i - 2].u, s[i - 3].u, c0, c1, c2, c3);
397  data[i] = c0 * data[i - 0] + c1 * data[i - 1] + c2 * data[i - 2] + c3 * data[i - 3];
398 }
