64 template <typename T>
65 size_t appendToTree(TTree& tree, const std::string brname, T& ptr)
66 {
67  size_t s = 0;
68  auto* br = tree.GetBranch(brname.c_str());
69  auto* pptr = &ptr;
70  if (br) {
71  br->SetAddress(&pptr);
72  } else {
73  br = tree.Branch(brname.c_str(), &pptr);
74  }
75  int res = br->Fill();
76  if (res < 0) {
77  throw std::runtime_error(fmt::format("Failed to fill CTF branch {}", brname));
78  }
79  s += res;
80  br->ResetAddress();
81  return s;
82 }

180 //___________________________________________________________________
181 void CTFWriterSpec::init(InitContext& ic)
182 {
183  mDictPerDetector = ic.options().get<bool>("dict-per-det");
184  // auto outmode = ic.options().get<std::string>("output-type"); // RS FIXME once global/local options clash is solved, --output-type will become device option
185  auto outmode = mOutputType;
186  if (outmode == "ctf") {
187  mWriteCTF = true;
188  mCreateDict = false;
189  } else if (outmode == "dict") {
190  mWriteCTF = false;
191  mCreateDict = true;
192  } else if (outmode == "both") {
193  mWriteCTF = true;
194  mCreateDict = true;
195  } else if (outmode == "none") {
196  mWriteCTF = false;
197  mCreateDict = false;
198  } else {
199  throw std::invalid_argument("Invalid output-type");
200  }
201 
202  mSaveDictAfter = ic.options().get<int>("save-dict-after");
203  mCTFAutoSave = ic.options().get<int>("save-ctf-after");
204  mDictDir = o2::utils::Str::rectifyDirectory(ic.options().get<std::string>("ctf-dict-dir"));
205  mCTFDir = o2::utils::Str::rectifyDirectory(ic.options().get<std::string>("output-dir"));
206  mCTFDirFallBack = ic.options().get<std::string>("output-dir-alt");
207  if (mCTFDirFallBack != "/dev/null") {
208  mCTFDirFallBack = o2::utils::Str::rectifyDirectory(mCTFDirFallBack);
209  }
210  mCTFMetaFileDir = ic.options().get<std::string>("meta-output-dir");
211  if (mCTFMetaFileDir != "/dev/null") {
212  mCTFMetaFileDir = o2::utils::Str::rectifyDirectory(mCTFMetaFileDir);
213  mStoreMetaFile = true;
214  }
215  mFlagMinDet = ic.options().get<int>("append-det-to-period");
216  mCreateRunEnvDir = !ic.options().get<bool>("ignore-partition-run-dir");
217  mMinSize = ic.options().get<int64_t>("min-file-size");
218  mMaxSize = ic.options().get<int64_t>("max-file-size");
219  mMaxCTFPerFile = ic.options().get<int>("max-ctf-per-file");
220  if (mWriteCTF) {
221  if (mMinSize > 0) {
222  LOG(info) << "Multiple CTFs will be accumulated in the tree/file until its size exceeds " << mMinSize << " bytes";
223  if (mMaxSize > mMinSize) {
224  LOG(info) << "but does not exceed " << mMaxSize << " bytes";
225  }
226  }
227  }
228  mChkSize = std::max(size_t(mMinSize * 1.1), mMaxSize);
229  if (!std::filesystem::exists(LOCKFileDir)) {
230  if (!std::filesystem::create_directories(LOCKFileDir)) {
231  usleep(10); // protection in case the directory was created by other process at the time of query
232  if (std::filesystem::exists(LOCKFileDir)) {
233  throw std::runtime_error(fmt::format("Failed to create {} directory", LOCKFileDir));
234  }
235  }
236  }
237 
238  if (mCreateDict) { // make sure that there is no local dictonary
239  for (int id = 0; id < DetID::nDetectors; id++) {
240  DetID det(id);
241  if (isPresent(det)) {
242  auto dictName = dictionaryFileName(det.getName());
243  if (std::filesystem::exists(dictName)) {
244  throw std::runtime_error(o2::utils::Str::concat_string("CTF dictionary creation is requested but ", dictName, " already exists, remove it!"));
245  }
246  if (!mDictPerDetector) {
247  break; // no point in checking further
248  }
249  }
250  }
251  }
252 }

255 // process data of particular detector
256 template <typename C>
257 size_t CTFWriterSpec::processDet(o2::framework::ProcessingContext& pc, DetID det, CTFHeader& header, TTree* tree)
258 {
259  size_t sz = 0;
260  if (!isPresent(det) || !pc.inputs().isValid(det.getName())) {
261  return sz;
262  }
263  auto ctfBuffer = pc.inputs().get<gsl::span<o2::ctf::BufferType>>(det.getName());
264  const auto ctfImage = C::getImage(ctfBuffer.data());
265  ctfImage.print(o2::utils::Str::concat_string(det.getName(), ": "), mVerbosity);
266  if (mWriteCTF) {
267  sz += ctfImage.appendToTree(*tree, det.getName());
268  header.detectors.set(det);
269  }
270  if (mCreateDict) {
271  if (!mFreqsAccumulation[det].size()) {
272  mFreqsAccumulation[det].resize(C::getNBlocks());
273  mFreqsMetaData[det].resize(C::getNBlocks());
274  }
275  if (!mHeaders[det]) { // store 1st header
276  mHeaders[det] = ctfImage.cloneHeader();
277  auto& hb = *static_cast<o2::ctf::CTFDictHeader*>(mHeaders[det].get());
278  hb.dictTimeStamp = uint32_t(std::time(nullptr));
279  hb.det = det;
280  }
281  for (int ib = 0; ib < C::getNBlocks(); ib++) {
282  const auto& bl = ctfImage.getBlock(ib);
283  if (bl.getNDict()) {
284  auto& freq = mFreqsAccumulation[det][ib];
285  auto& mdSave = mFreqsMetaData[det][ib];
286  const auto& md = ctfImage.getMetadata(ib);
287  freq.addFrequencies(bl.getDict(), bl.getDict() + bl.getNDict(), md.min);
288  mdSave = o2::ctf::Metadata{0, 0, md.messageWordSize, md.coderType, md.streamSize, md.probabilityBits, md.opt, freq.getMinSymbol(), freq.getMaxSymbol(), (int)freq.size(), 0, 0};
289  }
290  }
291  }
292  return sz;
293 }

296 // store dictionary of a particular detector
297 template <typename C>
298 void CTFWriterSpec::storeDictionary(DetID det, CTFHeader& header)
299 {
300  if (!isPresent(det) || !mFreqsAccumulation[det].size()) {
301  return;
302  }
303  prepareDictionaryTreeAndFile(det);
304  // create vector whose data contains dictionary in CTF format (EncodedBlock)
305  auto dictBlocks = C::createDictionaryBlocks(mFreqsAccumulation[det], mFreqsMetaData[det]);
306  auto& h = C::get(dictBlocks.data())->getHeader();
307  h = *reinterpret_cast<typename std::remove_reference<decltype(h)>::type*>(mHeaders[det].get());
308  auto& hb = static_cast<o2::ctf::CTFDictHeader&>(h);
309  hb = *static_cast<const o2::ctf::CTFDictHeader*>(mHeaders[det].get());
310 
311  C::get(dictBlocks.data())->print(o2::utils::Str::concat_string("Storing dictionary for ", det.getName(), ": "));
312  C::get(dictBlocks.data())->appendToTree(*mDictTreeOut.get(), det.getName()); // cast to EncodedBlock
313  // mFreqsAccumulation[det].clear();
314  // mFreqsMetaData[det].clear();
315  if (mDictPerDetector) {
316  header.detectors.reset();
317  }
318  header.detectors.set(det);
319  if (mDictPerDetector) {
320  closeDictionaryTreeAndFile(header);
321  }
322 }

338 //___________________________________________________________________
339 void CTFWriterSpec::run(ProcessingContext& pc)
340 {
341  const std::string NAStr = "NA";
342  auto cput = mTimer.CpuTime();
343  mTimer.Start(false);
344  const auto ref = pc.inputs().getFirstValid(true);
345  const auto dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
346  const auto dph = DataRefUtils::getHeader<DataProcessingHeader*>(ref);
347  auto oldRun = mRun;
348  if (dh->runNumber != 0) {
349  mRun = dh->runNumber;
350  }
351  // check runNumber with FMQ property, if set, override DH number
352  {
353  auto runNStr = pc.services().get<RawDeviceService>().device()->fConfig->GetProperty<std::string>("runNumber", NAStr);
354  if (runNStr != NAStr) {
355  size_t nc = 0;
356  auto runNProp = std::stol(runNStr, &nc);
357  if (nc != runNStr.size()) {
358  LOGP(error, "Property runNumber={} is provided but is not a number, ignoring", runNStr);
359  } else {
360  mRun = runNProp;
361  }
362  }
363  }
364  auto oldEnv = mEnvironmentID;
365  {
366  auto envN = pc.services().get<RawDeviceService>().device()->fConfig->GetProperty<std::string>("environment_id", NAStr);
367  if (envN != NAStr) {
368  mEnvironmentID = envN;
369  }
370  }
371  if ((oldRun != 0 && oldRun != mRun) || (!oldEnv.empty() && oldEnv != mEnvironmentID)) {
372  LOGP(warning, "RunNumber/Environment changed from {}/{} to {}/{}", oldRun, oldEnv, mRun, mEnvironmentID);
373  closeTFTreeAndFile();
374  }
375  // check for the LHCPeriod
376  if (mLHCPeriod.empty()) {
377  auto LHCPeriodStr = pc.services().get<RawDeviceService>().device()->fConfig->GetProperty<std::string>("LHCPeriod", NAStr);
378  if (LHCPeriodStr != NAStr) {
379  mLHCPeriod = LHCPeriodStr;
380  } else {
381  const char* months[12] = {"JAN", "FEB", "MAR", "APR", "MAY", "JUN", "JUL", "AUG", "SEP", "OCT", "NOV", "DEC"};
382  time_t now = time(nullptr);
383  auto ltm = gmtime(&now);
384  mLHCPeriod = months[ltm->tm_mon];
385  LOG(warning) << "LHCPeriod is not available, using current month " << mLHCPeriod;
386  }
387  if (mDets.count() <= mFlagMinDet) { // flag participating detectors
388  for (auto id = DetID::First; id <= DetID::Last; id++) {
389  if (isPresent(id)) {
390  mLHCPeriod += fmt::format("_{}", DetID::getName(id));
391  }
392  }
393  }
394  }
395 
396  mCurrCTFSize = estimateCTFSize(pc);
397  if (mWriteCTF) {
398  prepareTFTreeAndFile(dh);
399  }
400 
401  // create header
402  CTFHeader header{mRun, dph->creation, dh->firstTForbit};
403  size_t szCTF = 0;
404  szCTF += processDet<o2::itsmft::CTF>(pc, DetID::ITS, header, mCTFTreeOut.get());
405  szCTF += processDet<o2::itsmft::CTF>(pc, DetID::MFT, header, mCTFTreeOut.get());
406  szCTF += processDet<o2::tpc::CTF>(pc, DetID::TPC, header, mCTFTreeOut.get());
407  szCTF += processDet<o2::trd::CTF>(pc, DetID::TRD, header, mCTFTreeOut.get());
408  szCTF += processDet<o2::tof::CTF>(pc, DetID::TOF, header, mCTFTreeOut.get());
409  szCTF += processDet<o2::ft0::CTF>(pc, DetID::FT0, header, mCTFTreeOut.get());
410  szCTF += processDet<o2::fv0::CTF>(pc, DetID::FV0, header, mCTFTreeOut.get());
411  szCTF += processDet<o2::fdd::CTF>(pc, DetID::FDD, header, mCTFTreeOut.get());
412  szCTF += processDet<o2::mid::CTF>(pc, DetID::MID, header, mCTFTreeOut.get());
413  szCTF += processDet<o2::mch::CTF>(pc, DetID::MCH, header, mCTFTreeOut.get());
414  szCTF += processDet<o2::emcal::CTF>(pc, DetID::EMC, header, mCTFTreeOut.get());
415  szCTF += processDet<o2::phos::CTF>(pc, DetID::PHS, header, mCTFTreeOut.get());
416  szCTF += processDet<o2::cpv::CTF>(pc, DetID::CPV, header, mCTFTreeOut.get());
417  szCTF += processDet<o2::zdc::CTF>(pc, DetID::ZDC, header, mCTFTreeOut.get());
418  szCTF += processDet<o2::hmpid::CTF>(pc, DetID::HMP, header, mCTFTreeOut.get());
419  szCTF += processDet<o2::ctp::CTF>(pc, DetID::CTP, header, mCTFTreeOut.get());
420 
421  mTimer.Stop();
422 
423  if (mWriteCTF) {
424  szCTF += appendToTree(*mCTFTreeOut.get(), "CTFHeader", header);
425  mAccCTFSize += szCTF;
426  mCTFTreeOut->SetEntries(++mNAccCTF);
427  mTFOrbits.push_back(dh->firstTForbit);
428  LOG(info) << "TF#" << mNCTF << ": wrote CTF{" << header << "} of size " << szCTF << " to " << mCurrentCTFFileNameFull << " in " << mTimer.CpuTime() - cput << " s";
429  if (mNAccCTF > 1) {
430  LOG(info) << "Current CTF tree has " << mNAccCTF << " entries with total size of " << mAccCTFSize << " bytes";
431  }
432  if (mLockFD != -1) {
433  lseek(mLockFD, 0, SEEK_SET);
434  auto nwr = write(mLockFD, &mAccCTFSize, sizeof(size_t));
435  if (nwr != sizeof(size_t)) {
436  LOG(error) << "Failed to write current CTF size " << mAccCTFSize << " to lock file, bytes written: " << nwr;
437  }
438  }
439 
440  if (mAccCTFSize >= mMinSize || (mMaxCTFPerFile > 0 && mNAccCTF >= mMaxCTFPerFile)) {
441  closeTFTreeAndFile();
442  } else if (mCTFAutoSave > 0 && mNAccCTF % mCTFAutoSave == 0) {
443  mCTFTreeOut->AutoSave("override");
444  }
445  } else {
446  LOG(info) << "TF#" << mNCTF << " CTF writing is disabled, size was " << szCTF << " bytes";
447  }
448 
449  mNCTF++;
450  if (mCreateDict && mSaveDictAfter > 0 && (mNCTF % mSaveDictAfter) == 0) {
451  storeDictionaries();
452  }
453 }

455 //___________________________________________________________________
456 void CTFWriterSpec::finalize()
457 {
458  if (mFinalized) {
459  return;
460  }
461  if (mCreateDict) {
462  storeDictionaries();
463  }
464  if (mWriteCTF) {
465  closeTFTreeAndFile();
466  }
467  LOGF(info, "CTF writing total timing: Cpu: %.3e Real: %.3e s in %d slots",
468  mTimer.CpuTime(), mTimer.RealTime(), mTimer.Counter() - 1);
469  mFinalized = true;
470 }

472 //___________________________________________________________________
473 void CTFWriterSpec::prepareTFTreeAndFile(const o2::header::DataHeader* dh)
474 {
475  if (!mWriteCTF) {
476  return;
477  }
478  bool needToOpen = false;
479  if (!mCTFTreeOut) {
480  needToOpen = true;
481  } else {
482  if ((mAccCTFSize >= mMinSize) || // min size exceeded, may close the file.
483  (mAccCTFSize && mMaxSize > mMinSize && ((mAccCTFSize + mCurrCTFSize) > mMaxSize))) { // this is not the 1st CTF in the file and the new size will exceed allowed max
484  needToOpen = true;
485  } else {
486  LOGP(info, "Will add new CTF of estimated size {} to existing file of size {}", mCurrCTFSize, mAccCTFSize);
487  }
488  }
489  if (needToOpen) {
490  closeTFTreeAndFile();
491  auto fname = o2::base::NameConf::getCTFFileName(mRun, dh->firstTForbit, dh->tfCounter);
492  auto ctfDir = mCTFDir.empty() ? o2::utils::Str::rectifyDirectory("./") : mCTFDir;
493  if (mChkSize > 0 && (mCTFDirFallBack != "/dev/null")) {
494  createLockFile(dh, 0);
495  auto sz = getAvailableDiskSpace(ctfDir, 0); // check main storage
496  if (sz < mChkSize) {
497  removeLockFile();
498  LOG(warning) << "Primary CTF output device has available size " << sz << " while " << mChkSize << " is requested: will write on secondary one";
499  ctfDir = mCTFDirFallBack;
500  }
501  }
502  if (mCreateRunEnvDir && !mEnvironmentID.empty()) {
503  ctfDir += fmt::format("{}_{}/", mEnvironmentID, mRun);
504  if (!std::filesystem::exists(ctfDir)) {
505  if (!std::filesystem::create_directories(ctfDir)) {
506  throw std::runtime_error(fmt::format("Failed to create {} directory", ctfDir));
507  } else {

643 void CTFWriterSpec::createLockFile(const o2::header::DataHeader* dh, int level)
644 {
645  // create lock file for the CTF to be written to the storage of given level
646  while (1) {
647  mLockFileName = fmt::format("{}/ctfs{}-{}_{}_{}_{}.lock", LOCKFileDir, level, o2::utils::Str::getRandomString(8), mRun, dh->firstTForbit, dh->tfCounter);
648  if (!std::filesystem::exists(mLockFileName)) {
649  break;
650  }
651  }
652  mLockFD = open(mLockFileName.c_str(), O_RDWR | O_CREAT | O_CLOEXEC, S_IRUSR | S_IWUSR | S_IRGRP | S_IWGRP | S_IROTH | S_IWOTH);
653  if (mLockFD == -1) {
654  throw std::runtime_error(fmt::format("Error opening lock file {}", mLockFileName));
655  }
656  if (lockf(mLockFD, F_LOCK, 0)) {
657  throw std::runtime_error(fmt::format("Error locking file {}", mLockFileName));
658  }
659 }

662 void CTFWriterSpec::removeLockFile()
663 {
664  // remove CTF lock file
665  if (mLockFD != -1) {
666  if (lockf(mLockFD, F_ULOCK, 0)) {
667  throw std::runtime_error(fmt::format("Error unlocking file {}", mLockFileName));
668  }
669  mLockFD = -1;
670  std::error_code ec;
671  std::filesystem::remove(mLockFileName, ec); // use non-throwing version
672  }

675 //___________________________________________________________________
676 size_t CTFWriterSpec::getAvailableDiskSpace(const std::string& path, int level)
677 {
678  // count number of CTF files in processing (written to storage at given level) from their lock files
679  std::regex pat{fmt::format("({}/ctfs{}-[[:alnum:]_]+\\.lock$)", LOCKFileDir, level)};
680  int nLocked = 0;
681  size_t written = 0;
682  std::error_code ec;
683  for (const auto& entry : std::filesystem::directory_iterator(LOCKFileDir)) {
684  const auto& entryName = entry.path().native();
685  if (std::regex_search(entryName, pat) && (mLockFD < 0 || entryName != mLockFileName)) {
686  int fdt = open(entryName.c_str(), O_RDONLY);
687  if (fdt != -1) {
688  bool locked = lockf(fdt, F_TEST, 0) != 0;
689  if (locked) {
690  nLocked++;
691  size_t sz = 0;
692  auto nrd = read(fdt, &sz, sizeof(size_t));
693  if (nrd == sizeof(size_t)) {
694  written += sz;
695  }
696  }
697  close(fdt);
698  // unlocked file is either leftover from crached job or a file from concurent job which was being locked
699  // or just unlocked but not yet removed. In the former case remove it
700  if (!locked) {
701  struct stat statbuf;
702  if (stat(entryName.c_str(), &statbuf) != -1) { // if we fail to stat, the file was already removed
703 #ifdef __APPLE__
704  auto ftime = statbuf.st_mtimespec.tv_sec; // last write time
705 #else
706  auto ftime = statbuf.st_mtim.tv_sec; // last write time
707 #endif
708  auto ctime = std::chrono::duration_cast<std::chrono::seconds>(std::chrono::system_clock::now().time_since_epoch()).count();
709  if (ftime + 60 < ctime) { // this is an old file, remove it
710  std::filesystem::remove(entryName, ec); // use non-throwing version
711  }
712  }
713  }
714  }
715  }
716  }
717  const auto si = std::filesystem::space(path, ec);
718  int64_t avail = int64_t(si.available) - nLocked * mChkSize + written; // account already written part of unfinished files
719  LOGP(debug, "{} CTF files open (curr.size: {}) -> can use {} of {} bytes", nLocked, written, avail, si.available);
720  return avail > 0 ? avail : 0;
721 }
