57  void run(framework::ProcessingContext& pc)
58  {
59 
60  if (mFinished) {
61  return;
62  }
63 
64  // read collision context from input
65  auto context = pc.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
66  context->initSimChains(o2::detectors::DetID::FT0, mSimChains);
67  const bool withQED = context->isQEDProvided() && !mDisableQED;
68  auto& timesview = context->getEventRecords(withQED);
69 
70  // if there is nothing to do ... return
71  if (timesview.size() == 0) {
72  return;
73  }
74 
75  TStopwatch timer;
76  timer.Start();
77 
78  LOG(info) << "CALLING FT0 DIGITIZATION";
79 
80  static std::vector<o2::ft0::HitType> hits;
81  // o2::dataformats::MCTruthContainer<o2::ft0::MCLabel> labelAccum;
82  o2::dataformats::MCTruthContainer<o2::ft0::MCLabel> labels;
83 
84  // mDigitizer.setMCLabels(&labels);
85  auto& eventParts = context->getEventParts(withQED);
86  // loop over all composite collisions given from context
87  // (aka loop over all the interaction records)
88  for (int collID = 0; collID < timesview.size(); ++collID) {
89  mDigitizer.setInteractionRecord(timesview[collID]);
90  LOG(debug) << " setInteractionRecord " << timesview[collID] << " bc " << mDigitizer.getBC() << " orbit " << mDigitizer.getOrbit();
91  // for each collision, loop over the constituents event and source IDs
92  // (background signal merging is basically taking place here)
93  for (auto& part : eventParts[collID]) {
94  // get the hits for this event and this source
95  hits.clear();
96  context->retrieveHits(mSimChains, "FT0Hit", part.sourceID, part.entryID, &hits);
97  LOG(debug) << "For collision " << collID << " eventID " << part.entryID << " source ID " << part.sourceID << " found " << hits.size() << " hits ";
98  if (hits.size() > 0) {
99  // call actual digitization procedure
100  mDigitizer.setEventID(part.entryID);
101  mDigitizer.setSrcID(part.sourceID);
102  mDigitizer.process(&hits, mDigitsBC, mDigitsCh, mDigitsTrig, labels);
103  }
104  }
105  }
106  mDigitizer.flush_all(mDigitsBC, mDigitsCh, mDigitsTrig, labels);
107 
108  // send out to next stage
109  pc.outputs().snapshot(Output{"FT0", "DIGITSBC", 0, Lifetime::Timeframe}, mDigitsBC);
110  pc.outputs().snapshot(Output{"FT0", "DIGITSCH", 0, Lifetime::Timeframe}, mDigitsCh);
111  pc.outputs().snapshot(Output{"FT0", "TRIGGERINPUT", 0, Lifetime::Timeframe}, mDigitsTrig);
112  if (pc.outputs().isAllowed({"FT0", "DIGITSMCTR", 0})) {
113  pc.outputs().snapshot(Output{"FT0", "DIGITSMCTR", 0, Lifetime::Timeframe}, labels);
114  }
115  LOG(info) << "FT0: Sending ROMode= " << mROMode << " to GRPUpdater";
116  pc.outputs().snapshot(Output{"FT0", "ROMode", 0, Lifetime::Timeframe}, mROMode);
117 
118  timer.Stop();
119  LOG(info) << "Digitization took " << timer.CpuTime() << "s";
120 
121  // we should be only called once; tell DPL that this process is ready to exit
122  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
123  mFinished = true;
124  }
125 
126  protected:
127  bool mFinished = false;
128  std::vector<o2::ft0::ChannelData> mDigitsCh;
129  std::vector<o2::ft0::Digit> mDigitsBC;
130  std::vector<o2::ft0::DetTrigInput> mDigitsTrig;
131 
132  Bool_t mContinuous = kFALSE;
133  double mFairTimeUnitInNS = 1;
134  o2::ft0::Digitizer mDigitizer;
135 
136  // RS: at the moment using hardcoded flag for continuos readout
137  o2::parameters::GRPObject::ROMode mROMode = o2::parameters::GRPObject::CONTINUOUS; // readout mode
138 
139  //
140  bool mDisableQED = false;
141 
142  std::vector<TChain*> mSimChains;
143 };

145 o2::framework::DataProcessorSpec getFT0DigitizerSpec(int channel, bool mctruth)
146 {
147  // create the full data processor spec using
148  // a name identifier
149  // input description
150  // algorithmic description (here a lambda getting called once to setup the actual processing function)
151  // options that can be used for this processor (here: input file names where to take the hits)
152 
153  std::vector<OutputSpec> outputs;
154  outputs.emplace_back("FT0", "DIGITSBC", 0, Lifetime::Timeframe);
155  outputs.emplace_back("FT0", "DIGITSCH", 0, Lifetime::Timeframe);
156  outputs.emplace_back("FT0", "TRIGGERINPUT", 0, Lifetime::Timeframe);

161

167  Options{{"pileup", VariantType::Int, 1, {"whether to run in continuous time mode"}},
168  {"disable-qed", o2::framework::VariantType::Bool, false, {"disable QED handling"}}}};
