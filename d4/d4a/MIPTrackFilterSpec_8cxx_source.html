50 void MIPTrackFilterDevice::init(framework::InitContext& ic)
51 {
52  const double minP = ic.options().get<double>("min-momentum");
53  const double maxP = ic.options().get<double>("max-momentum");
54  assert(minP < maxP);
55  const int minClusters = std::max(10, ic.options().get<int>("min-clusters"));
56 
57  mCuts.setPMin(minP);
58  mCuts.setPMax(maxP);
59  mCuts.setNClusMin(minClusters);

62 void MIPTrackFilterDevice::run(ProcessingContext& pc)
63 {
64  const auto tracks = pc.inputs().get<gsl::span<TrackTPC>>("tracks");
65 
66  std::copy_if(tracks.begin(), tracks.end(), std::back_inserter(mMIPTracks),
67  [this](const auto& track) { return this->mCuts.goodTrack(track); });
68 
69  LOG(info) << mMIPTracks.size() << " MIP tracks in a total of " << tracks.size() << " tracks";
70 
71  pc.outputs().snapshot(Output{"TPC", "MIPS", 0, Lifetime::Timeframe}, mMIPTracks);
72  mMIPTracks.clear();
73 }
