37 void sendAnswer(const std::string& what, const std::string& ack_chan, FairMQDevice& device)
38 {
39  if (!ack_chan.empty()) {
40  auto fmqFactory = device.GetChannel(ack_chan).Transport();
41  auto msg = fmqFactory->CreateMessage(what.size(), fair::mq::Alignment{64});
42  memcpy(msg->GetData(), what.c_str(), what.size());
43  FairMQParts outParts;
44  outParts.AddPart(std::move(msg));
45  sendOnChannel(device, outParts, ack_chan);
46  }
47 }

49 auto getDataOriginFromFilename(const std::string& filename)
50 {
51  // assume the filename start with detector name
52  auto dIDStr = filename.substr(0, 3);
53  auto dID = DetID::nameToID(dIDStr.c_str(), DetID::First);
54  o2::header::DataOrigin dataOrigin;
55  if (dID < 0) {
56  for (auto& el : exceptionsDetID) {
57  if (el.as<std::string>() == dIDStr) {
58  return el;
59  }
60  }
61  return o2::header::gDataOriginInvalid;
62  }
63  return DetID(dID).getDataOrigin();
64 }

66 InjectorFunction dcs2dpl(const std::string& acknowledge)
67 {
68 
69  auto timesliceId = std::make_shared<size_t>(0);
70 
71  return [acknowledge, timesliceId](FairMQDevice& device, FairMQParts& parts, ChannelRetriever channelRetriever) {
72  // make sure just 2 messages received
73  if (parts.Size() != 2) {
74  LOG(error) << "received " << parts.Size() << " instead of 2 expected";
75  sendAnswer("error0: wrong number of messages", acknowledge, device);
76  return;
77  }
78  std::string filename{static_cast<const char*>(parts.At(0)->GetData()), parts.At(0)->GetSize()};
79  size_t filesize = parts.At(1)->GetSize();
80  LOG(info) << "received file " << filename << " of size " << filesize;
81  o2::header::DataOrigin dataOrigin = getDataOriginFromFilename(filename);
82  if (dataOrigin == o2::header::gDataOriginInvalid) {
83  LOG(error) << "unknown detector for " << filename;
84  sendAnswer(fmt::format("{}:error1: unrecognized filename", filename), acknowledge, device);
85  return;
86  }
87 
88  o2::header::DataHeader hdrF("DCS_CONFIG_FILE", dataOrigin, 0);
89  o2::header::DataHeader hdrN("DCS_CONFIG_NAME", dataOrigin, 0);
90  OutputSpec outsp{hdrF.dataOrigin, hdrF.dataDescription, hdrF.subSpecification};
91  auto channel = channelRetriever(outsp, *timesliceId);
92  if (channel.empty()) {
93  LOG(error) << "No output channel found for OutputSpec " << outsp;
94  sendAnswer(fmt::format("{}:error2: no channel to send", filename), acknowledge, device);
95  return;
96  }
97 
98  hdrF.tfCounter = *timesliceId;
99  hdrF.payloadSerializationMethod = o2::header::gSerializationMethodNone;
100  hdrF.splitPayloadParts = 1;
101  hdrF.splitPayloadIndex = 0;
102  hdrF.payloadSize = filesize;
103  hdrF.firstTForbit = 0; // this should be irrelevant for DCS
104 
105  hdrN.tfCounter = *timesliceId;
106  hdrN.payloadSerializationMethod = o2::header::gSerializationMethodNone;
107  hdrN.splitPayloadParts = 1;
108  hdrN.splitPayloadIndex = 0;
109  hdrN.payloadSize = parts.At(0)->GetSize();
110  hdrN.firstTForbit = 0; // this should be irrelevant for DCS
111 
112  auto fmqFactory = device.GetChannel(channel).Transport();
113  std::uint64_t creation = std::chrono::time_point_cast<std::chrono::milliseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
114 
115  o2::header::Stack headerStackF{hdrF, DataProcessingHeader{*timesliceId, 1, creation}};
116  auto hdMessageF = fmqFactory->CreateMessage(headerStackF.size(), fair::mq::Alignment{64});
117  auto plMessageF = fmqFactory->CreateMessage(hdrF.payloadSize, fair::mq::Alignment{64});
118  memcpy(hdMessageF->GetData(), headerStackF.data(), headerStackF.size());
119  memcpy(plMessageF->GetData(), parts.At(1)->GetData(), hdrF.payloadSize);
120 
121  o2::header::Stack headerStackN{hdrN, DataProcessingHeader{*timesliceId, 1, creation}};
122  auto hdMessageN = fmqFactory->CreateMessage(headerStackN.size(), fair::mq::Alignment{64});
123  auto plMessageN = fmqFactory->CreateMessage(hdrN.payloadSize, fair::mq::Alignment{64});
124  memcpy(hdMessageN->GetData(), headerStackN.data(), headerStackN.size());
125  memcpy(plMessageN->GetData(), parts.At(0)->GetData(), hdrN.payloadSize);
126 
127  FairMQParts outPartsF;
128  outPartsF.AddPart(std::move(hdMessageF));
129  outPartsF.AddPart(std::move(plMessageF));
130  sendOnChannel(device, outPartsF, channel);
131 
132  FairMQParts outPartsN;
133  outPartsN.AddPart(std::move(hdMessageN));
134  outPartsN.AddPart(std::move(plMessageN));
135  sendOnChannel(device, outPartsN, channel);
136 
137  sendAnswer("OK", acknowledge, device);
138  LOG(info) << "Sent DPL message and acknowledgment for file " << filename;
139  (*timesliceId)++;
140  };
141 }

152 WorkflowSpec defineDataProcessing(ConfigContext const& config)
153 {
154  auto setChanName = [](const std::string& chan, const std::string& name) {
155  size_t n = 0;
156  if (std::string(chan).find("name=") != std::string::npos) {
157  n = std::string(chan).find(",");
158  if (n == std::string::npos) {
159  throw std::runtime_error(fmt::format("wrongly formatted channel: {}", chan));
160  }
161  n++;
162  }
163  return o2::utils::Str::concat_string("name=", name, ",", chan.substr(n, chan.size()));
164  };
165 
166  const std::string devName = "dcs-config-proxy";
167  auto chan = config.options().get<std::string>("subscribe-to");
168  if (chan.empty()) {
169  throw std::runtime_error("input channel is not provided");
170  }
171  chan = setChanName(chan, devName);
172 
173  auto chanTo = config.options().get<std::string>("acknowlege-to");
174  std::string ackChan{};
175  if (!chanTo.empty()) {
176  ackChan = "ackChan";
177  chan = o2::utils::Str::concat_string(chan, ";", setChanName(chanTo, ackChan));
178  }
179  LOG(info) << "Channels setup: " << chan;
180  Outputs dcsOutputs;
181 
182  for (int id = DetID::First; id <= DetID::Last; id++) {
183  dcsOutputs.emplace_back(DetID(id).getDataOrigin(), "DCS_CONFIG_FILE", 0, Lifetime::Sporadic);
184  dcsOutputs.emplace_back(DetID(id).getDataOrigin(), "DCS_CONFIG_NAME", 0, Lifetime::Sporadic);
185  }
186  for (auto& el : exceptionsDetID) {
187  dcsOutputs.emplace_back(el, "DCS_CONFIG_FILE", 0, Lifetime::Sporadic);
188  dcsOutputs.emplace_back(el, "DCS_CONFIG_NAME", 0, Lifetime::Sporadic);
189  }
190 
191  DataProcessorSpec dcsConfigProxy = specifyExternalFairMQDeviceProxy(
192  devName.c_str(),
193  std::move(dcsOutputs),
194  // this is just default, can be overriden by --dcs-config-proxy '--channel-config..'
195  chan.c_str(),
196  dcs2dpl(ackChan));
197 
198  WorkflowSpec workflow;
199  workflow.emplace_back(dcsConfigProxy);
200  return workflow;
201 }
