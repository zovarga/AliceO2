42 struct TOFDCSinfo {
43  std::pair<uint64_t, double> firstValue; // first value seen by the TOF DCS processor
44  std::pair<uint64_t, double> lastValue; // last value seen by the TOF DCS processor
45  std::pair<uint64_t, double> midValue; // mid value seen by the TOF DCS processor
46  std::pair<uint64_t, double> maxChange; // maximum variation seen by the TOF DCS processor
47  TOFDCSinfo()
48  {
49  firstValue = std::make_pair(0, -999999999);
50  lastValue = std::make_pair(0, -999999999);
51  midValue = std::make_pair(0, -999999999);
52  maxChange = std::make_pair(0, -999999999);
53  }
54  void makeEmpty()
55  {
56  firstValue.first = lastValue.first = midValue.first = maxChange.first = 0;
57  firstValue.second = lastValue.second = midValue.second = maxChange.second = -999999999;
58  }
59  void print() const;
60 
61  ClassDefNV(TOFDCSinfo, 1);
62 };
