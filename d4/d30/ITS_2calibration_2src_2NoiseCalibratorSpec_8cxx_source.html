34 void NoiseCalibratorSpec::init(InitContext& ic)
35 {
36  auto onepix = ic.options().get<bool>("1pix-only");
37  LOG(info) << "Fast 1=pixel calibration: " << onepix;
38  auto probT = ic.options().get<float>("prob-threshold");
39  LOG(info) << "Setting the probability threshold to " << probT;
40 
41  mCalibrator = std::make_unique<CALIBRATOR>(onepix, probT);
42  std::string dictPath = o2::itsmft::ClustererParam<o2::detectors::DetID::ITS>::Instance().dictFilePath;
43  std::string dictFile = o2::base::DetectorNameConf::getAlpideClusterDictionaryFileName(o2::detectors::DetID::ITS, dictPath);
44  if (o2::utils::Str::pathExists(dictFile)) {
45  mCalibrator->loadDictionary(dictFile);
46  LOG(info) << "ITS NoiseCalibrator is running with a provided dictionary: " << dictFile;
47  } else {
48  LOG(info) << "Dictionary " << dictFile
49  << " is absent, ITS NoiseCalibrator expects cluster patterns for all clusters";
50  }
51 }

53 void NoiseCalibratorSpec::run(ProcessingContext& pc)
54 {
55  const auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
56  gsl::span<const unsigned char> patterns = pc.inputs().get<gsl::span<unsigned char>>("patterns");
57  const auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
58 
59  if (mCalibrator->processTimeFrame(compClusters, patterns, rofs)) {
60  LOG(info) << "Minimum number of noise counts has been reached !";
61  sendOutput(pc.outputs());
62  pc.services().get<ControlService>().readyToQuit(QuitRequest::All);
63  }
64 }

66 void NoiseCalibratorSpec::sendOutput(DataAllocator& output)
67 {
68  mCalibrator->finalize();
69 
70  long tstart = 0, tend = 9999999;
71 #ifdef TIME_SLOT_CALIBRATION
72  const auto& payload = mCalibrator->getNoiseMap(tstart, tend);
73 #else
74  const auto& payload = mCalibrator->getNoiseMap();
75 #endif
76  std::map<std::string, std::string> md;
77  o2::ccdb::CcdbObjectInfo info("ITS/Noise", "NoiseMap", "noise.root", md, tstart, tend);
78 
79  auto image = o2::ccdb::CcdbApi::createObjectImage(&payload, &info);
80  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName()
81  << " of size " << image->size()
82  << " bytes, valid for " << info.getStartValidityTimestamp()
83  << " : " << info.getEndValidityTimestamp();
84 
85  using clbUtils = o2::calibration::Utils;
86  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "ITS_NOISE", 0}, *image.get());
87  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "ITS_NOISE", 0}, info);
88  LOG(info) << "sending done";
89 }

96 DataProcessorSpec getNoiseCalibratorSpec()
97 {
98  std::vector<InputSpec> inputs;
99  inputs.emplace_back("compClusters", "ITS", "COMPCLUSTERS", 0, Lifetime::Timeframe);
100  inputs.emplace_back("patterns", "ITS", "PATTERNS", 0, Lifetime::Timeframe);
101  inputs.emplace_back("ROframes", "ITS", "CLUSTERSROF", 0, Lifetime::Timeframe);
102 
103  using clbUtils = o2::calibration::Utils;
104  std::vector<OutputSpec> outputs;
105  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBPayload, "ITS_NOISE"}, Lifetime::Sporadic);
106  outputs.emplace_back(ConcreteDataTypeMatcher{o2::calibration::Utils::gDataOriginCDBWrapper, "ITS_NOISE"}, Lifetime::Sporadic);
107 
108  return DataProcessorSpec{
109  "its-noise-calibrator",
110  inputs,
111  outputs,
112  AlgorithmSpec{adaptFromTask<NoiseCalibratorSpec>()},
113  Options{
114  {"1pix-only", VariantType::Bool, false, {"Fast 1-pixel calibration only"}},
115  {"prob-threshold", VariantType::Float, 3.e-6f, {"Probability threshold for noisy pixels"}}}};
116 }
