47  GPUd() float getCorrection(const StackID& stack, ChargeType charge, float z = 0, float tgl = 0) const
48  {
49  // by default return 1 if no correction was loaded
50  if (mDims < 0) {
51  return 1;
52  }
53 
54  const auto& p = mParams[stackIndex(stack, charge)];
55  float corr = p[0];
56 
57  if (mDims > 0) {
58  corr += p[1] * z + p[2] * z * z;
59  if (mDims > 1) {
60  corr += p[3] * tgl + p[4] * z * tgl + p[5] * tgl * tgl;
61  }
62  }
63 
64  return corr;
65  }
66
