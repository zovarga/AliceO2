78  template <typename T>
79  static auto getNumericMetricCursor(size_t metricIndex)
80  {
81  return [metricIndex](DeviceMetricsInfo& metrics, T value, size_t timestamp) {
82  MetricInfo& metric = metrics.metrics[metricIndex];
83 
84  metrics.minDomain[metricIndex] = std::min(metrics.minDomain[metricIndex], timestamp);
85  metrics.maxDomain[metricIndex] = std::max(metrics.maxDomain[metricIndex], timestamp);
86  metrics.max[metricIndex] = std::max(metrics.max[metricIndex], (float)value);
87  metrics.min[metricIndex] = std::min(metrics.min[metricIndex], (float)value);
88  metrics.changed.at(metricIndex) = true;
89  auto& store = getMetricsStore<T>(metrics);
90  size_t pos = metric.pos++ % store[metric.storeIdx].size();
91  metrics.timestamps[metricIndex][pos] = timestamp;
92  store[metric.storeIdx][pos] = value;
93  metric.filledMetrics++;
94  };
95  }

100  template <typename T>
101  static size_t
102  bookNumericMetric(DeviceMetricsInfo& metrics,
103  char const* name,
104  NewMetricCallback newMetricsCallback = nullptr)
105  {
106  static_assert(std::is_same_v<T, int> || std::is_same_v<T, uint64_t> || std::is_same_v<T, float>, "Unsupported metric type");
107  size_t metricIndex = bookMetricInfo(metrics, name);
108  auto& metricInfo = metrics.metrics[metricIndex];
109  metricInfo.type = getMetricType<T>();
110  metricInfo.storeIdx = getMetricsStore<T>(metrics).size();
111  getMetricsStore<T>(metrics).emplace_back(std::array<T, 1024>{});
112  if (newMetricsCallback != nullptr) {
113  newMetricsCallback(name, metricInfo, 0, metricIndex);
114  }
115  return metricIndex;
116  }
