149 template <typename T>
150 std::vector<std::string> solar2FeeLinkConsistencyCheck()
151 {
152  std::vector<std::string> errors;
153 
154  // All solars must have a FeeLinkId
155  std::set<uint16_t> solarIds = getSolarUIDs<T>();
156  auto solar2feeLink = createSolar2FeeLinkMapper<T>();
157  std::vector<o2::mch::raw::FeeLinkId> feeLinkIds;
158  for (auto s : solarIds) {
159  auto p = solar2feeLink(s);
160  if (!p.has_value()) {
161  errors.push_back(fmt::format("Got no feelinkId for solarId {}", s));
162  } else {
163  feeLinkIds.push_back(p.value());
164  }
165  }
166 
167  // All FeeLinkId must have a SolarId
168  auto feeLinkId2SolarId = createFeeLink2SolarMapper<T>();
169  for (auto f : feeLinkIds) {
170  auto p = feeLinkId2SolarId(f);
171  if (!p.has_value()) {
172  errors.push_back(fmt::format("Got no solarId for FeeLinkId {}", asString(f)));
173  }
174  }
175  return errors;
176 }
