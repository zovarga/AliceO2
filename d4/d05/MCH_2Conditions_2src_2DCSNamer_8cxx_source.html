91 uint8_t quadrantLVGroup(int deId, bool bending)
92 {
93  uint8_t group{invalidGroup};
94 
95  // For Chamber 1 to 4 Left the relationship between DCS GUI names and groups is:
96  // Quad2B --> Group3 = DE x01 Non Bending
97  // Quad2F --> Group1 = DE x01 Bending
98  // Quad3B --> Group4 = DE x02 Bending
99  // Quad3F --> Group2 = DE x02 Non Bending
100  // for Chamber 1 to 4 Right the relationship is:
101  // Quad1B --> Group3 = DE x00 Bending
102  // Quad1F --> Group1 = DE x00 Non Bending
103  // Quad4B --> Group4 = DE x03 Non Bending
104  // Quad4F --> Group2 = DE x03 Bending
105  // where x = 1,2,3,4
106  // and Quad#B = Back = towards IP = cath1
107  // while Quad#F = Front = towards muon trigger = cath0
108  //
109  int remnant = deId % 100;
110  switch (remnant) {
111  case 0: // DE x00
112  group = bending ? 3 : 1;
113  break;
114  case 1: // DE x01
115  group = bending ? 1 : 3;
116  break;
117  case 2: // DE x02
118  group = bending ? 4 : 2;
119  break;
120  case 3: // DE x03
121  group = bending ? 2 : 4;
122  break;
123  default:
124  break;
125  }
126  return group;
127 }

244 std::vector<std::string> measurement(const std::vector<std::string>& patterns,
245  o2::mch::dcs::MeasurementType m)
246 {
247  std::vector<std::string> result;
248 
249  result.resize(patterns.size());
250 
251  std::transform(patterns.begin(), patterns.end(), result.begin(), [&m](std::string s) {
252  auto name = fmt::sprintf(s, measurementName(m));

254  });

368 std::optional<ID> detElemId2DCS(int deId)
369 {
370  if (std::find(detElemIds.begin(), detElemIds.end(), deId) == detElemIds.end()) {
371  return std::nullopt;
372  }
373  int chamberId = deId / 100 - 1;
374  int id = deId - (chamberId + 1) * 100;
375 
376  Side side = Side::Left;
377 
378  if (isQuadrant(deId)) {
379  if (id == 0 || id == 3) {
380  side = Side::Right;
381  } else {
382  side = Side::Left;
383  }
384  } else {
385  int nofDe = nofDetectionElementsInChamber(chamberId);
386  int quarter = nofDe / 4;
387  int half = nofDe / 2;
388  int threeQuarter = quarter + half;
389  if (id <= quarter) {
390  id += quarter + 1;
391  side = Side::Right;
392  } else if (id <= threeQuarter) {
393  id = (threeQuarter - id + 1);
394  side = Side::Left;
395  } else {
396  id -= threeQuarter;
397  side = Side::Right;
398  }
399  // dcs convention change : numbering from top, not from bottom
400  id = half - id;
401  }
402 
403  return std::optional<ID>{{id, side, chamberId}};
404 }
