49 std::pair<unsigned int, unsigned int> GPUChainTracking::TPCClusterizerDecodeZSCountUpdate(unsigned int iSlice, const CfFragment& fragment)
50 {
51  bool doGPU = mRec->GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCClusterFinding;
52  GPUTPCClusterFinder& clusterer = processors()->tpcClusterer[iSlice];
53  GPUTPCClusterFinder::ZSOffset* o = processors()->tpcClusterer[iSlice].mPzsOffsets;
54  unsigned int digits = 0;
55  unsigned int pages = 0;
56  for (unsigned short j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
57  clusterer.mMinMaxCN[j] = mCFContext->fragmentData[fragment.index].minMaxCN[iSlice][j];
58  if (doGPU) {
59  unsigned short posInEndpoint = 0;
60  unsigned short pagesEndpoint = 0;
61  for (unsigned int k = clusterer.mMinMaxCN[j].minC; k < clusterer.mMinMaxCN[j].maxC; k++) {
62  const unsigned int minL = (k == clusterer.mMinMaxCN[j].minC) ? clusterer.mMinMaxCN[j].minN : 0;
63  const unsigned int maxL = (k + 1 == clusterer.mMinMaxCN[j].maxC) ? clusterer.mMinMaxCN[j].maxN : mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k];
64  for (unsigned int l = minL; l < maxL; l++) {
65  unsigned short pageDigits = mCFContext->fragmentData[fragment.index].pageDigits[iSlice][j][posInEndpoint++];
66  if (pageDigits) {
67  *(o++) = GPUTPCClusterFinder::ZSOffset{digits, j, pagesEndpoint};
68  digits += pageDigits;
69  }
70  pagesEndpoint++;
71  }
72  }
73  if (pagesEndpoint != mCFContext->fragmentData[fragment.index].pageDigits[iSlice][j].size()) {
74  GPUFatal("TPC raw page count mismatch in TPCClusterizerDecodeZSCountUpdate: expected %d / buffered %lu", pagesEndpoint, mCFContext->fragmentData[fragment.index].pageDigits[iSlice][j].size());
75  }
76  } else {
77  clusterer.mPzsOffsets[j] = GPUTPCClusterFinder::ZSOffset{digits, j, 0};
78  digits += mCFContext->fragmentData[fragment.index].nDigits[iSlice][j];
79  pages += mCFContext->fragmentData[fragment.index].nPages[iSlice][j];
80  }
81  }
82  if (doGPU) {
83  pages = o - processors()->tpcClusterer[iSlice].mPzsOffsets;
84  }
85  return {digits, pages};
86 }

88 std::pair<unsigned int, unsigned int> GPUChainTracking::TPCClusterizerDecodeZSCount(unsigned int iSlice, const CfFragment& fragment)
89 {
90  mRec->getGeneralStepTimer(GeneralStep::Prepare).Start();
91  unsigned int nDigits = 0;
92  unsigned int nPages = 0;
93  bool doGPU = mRec->GetRecoStepsGPU() & GPUDataTypes::RecoStep::TPCClusterFinding;
94  int firstHBF = (mIOPtrs.settingsTF && mIOPtrs.settingsTF->hasTfStartOrbit) ? mIOPtrs.settingsTF->tfStartOrbit : (mIOPtrs.tpcZS->slice[iSlice].count[0] && mIOPtrs.tpcZS->slice[iSlice].nZSPtr[0][0]) ? o2::raw::RDHUtils::getHeartBeatOrbit(*(const o2::header::RAWDataHeader*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[0][0]) : 0;
95 
96  for (unsigned short j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
97 #ifndef GPUCA_NO_VC
98  if (GetProcessingSettings().prefetchTPCpageScan >= 3 && j < GPUTrackingInOutZS::NENDPOINTS - 1) {
99  for (unsigned int k = 0; k < mIOPtrs.tpcZS->slice[iSlice].count[j + 1]; k++) {
100  for (unsigned int l = 0; l < mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j + 1][k]; l++) {
101  Vc::Common::prefetchMid(((const unsigned char*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j + 1][k]) + l * TPCZSHDR::TPC_ZS_PAGE_SIZE);
102  Vc::Common::prefetchMid(((const unsigned char*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j + 1][k]) + l * TPCZSHDR::TPC_ZS_PAGE_SIZE + sizeof(o2::header::RAWDataHeader));
103  }
104  }
105  }
106 #endif
107 
108  std::vector<std::pair<CfFragment, std::array<int, 5>>> fragments;
109  fragments.reserve(mCFContext->nFragments);
110  fragments.emplace_back(std::pair<CfFragment, std::array<int, 5>>{fragment, {0, 0, 0, 0, 0}});
111  for (unsigned int i = 1; i < mCFContext->nFragments; i++) {
112  fragments.emplace_back(std::pair<CfFragment, std::array<int, 5>>{fragments.back().first.next(), {0, 0, 0, 0, 0}});
113  }
114 
115  unsigned int emptyPages = 0;
116  unsigned int firstPossibleFragment = 0;
117  for (unsigned int k = 0; k < mIOPtrs.tpcZS->slice[iSlice].count[j]; k++) {
118  nPages += mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k];
119  for (unsigned int l = 0; l < mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k]; l++) {
120 #ifndef GPUCA_NO_VC
121  if (GetProcessingSettings().prefetchTPCpageScan >= 2 && l + 1 < mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k]) {
122  Vc::Common::prefetchForOneRead(((const unsigned char*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j][k]) + (l + 1) * TPCZSHDR::TPC_ZS_PAGE_SIZE);
123  Vc::Common::prefetchForOneRead(((const unsigned char*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j][k]) + (l + 1) * TPCZSHDR::TPC_ZS_PAGE_SIZE + sizeof(o2::header::RAWDataHeader));
124  }
125 #endif
126  const unsigned char* const page = ((const unsigned char*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j][k]) + l * TPCZSHDR::TPC_ZS_PAGE_SIZE;
127  const o2::header::RAWDataHeader* rdh = (const o2::header::RAWDataHeader*)page;
128  if (o2::raw::RDHUtils::getMemorySize(*rdh) == sizeof(o2::header::RAWDataHeader)) {
129  emptyPages++;
130  continue;
131  }
132  const TPCZSHDR* const hdr = (const TPCZSHDR*)(page + sizeof(o2::header::RAWDataHeader));
133  nDigits += hdr->nADCsamples;
134  unsigned int timeBin = (hdr->timeOffset + (o2::raw::RDHUtils::getHeartBeatOrbit(*rdh) - firstHBF) * o2::constants::lhc::LHCMaxBunches) / LHCBCPERTIMEBIN;
135  if (timeBin + hdr->nTimeBins > mCFContext->tpcMaxTimeBin) {
136  mCFContext->tpcMaxTimeBin = timeBin + hdr->nTimeBins;
137  }
138  for (unsigned int f = firstPossibleFragment; f < mCFContext->nFragments; f++) {
139  if (timeBin < (unsigned int)fragments[f].first.last()) {
140  if ((unsigned int)fragments[f].first.first() <= timeBin + hdr->nTimeBins) {
141  fragments[f].second[2] = k + 1;
142  fragments[f].second[3] = l + 1;
143  if (!fragments[f].second[4]) {
144  fragments[f].second[4] = 1;
145  fragments[f].second[0] = k;
146  fragments[f].second[1] = l;
147  } else if (emptyPages) {
148  mCFContext->fragmentData[f].nPages[iSlice][j] += emptyPages;
149  for (unsigned int m = 0; m < emptyPages; m++) {
150  mCFContext->fragmentData[f].pageDigits[iSlice][j].emplace_back(0);
151  }
152  }
153  mCFContext->fragmentData[f].nPages[iSlice][j]++;
154  mCFContext->fragmentData[f].nDigits[iSlice][j] += hdr->nADCsamples;
155  if (doGPU) {
156  mCFContext->fragmentData[f].pageDigits[iSlice][j].emplace_back(hdr->nADCsamples);
157  }
158  }
159  } else {
160  firstPossibleFragment = f + 1;
161  }
162  }
163  emptyPages = 0;
164  }
165  }
166  for (unsigned int f = 0; f < mCFContext->nFragments; f++) {
167  mCFContext->fragmentData[f].minMaxCN[iSlice][j].maxC = fragments[f].second[2];
168  mCFContext->fragmentData[f].minMaxCN[iSlice][j].minC = fragments[f].second[0];
169  mCFContext->fragmentData[f].minMaxCN[iSlice][j].maxN = fragments[f].second[3];
170  mCFContext->fragmentData[f].minMaxCN[iSlice][j].minN = fragments[f].second[1];
171  }
172  }
173  mCFContext->nPagesTotal += nPages;
174  mCFContext->nPagesSector[iSlice] = nPages;
175  mCFContext->nPagesSectorMax = std::max(mCFContext->nPagesSectorMax, nPages);
176 
177  unsigned int digitsFragment = 0;
178  for (unsigned int i = 0; i < mCFContext->nFragments; i++) {
179  unsigned int pages = 0;
180  unsigned int digits = 0;
181  for (unsigned short j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
182  pages += mCFContext->fragmentData[i].nPages[iSlice][j];
183  digits += mCFContext->fragmentData[i].nDigits[iSlice][j];
184  }
185  mCFContext->nPagesFragmentMax = std::max(mCFContext->nPagesSectorMax, pages);
186  digitsFragment = std::max(digitsFragment, digits);
187  }
188  mRec->getGeneralStepTimer(GeneralStep::Prepare).Stop();
189  return {nDigits, digitsFragment};
190 }

192 void GPUChainTracking::RunTPCClusterizer_compactPeaks(GPUTPCClusterFinder& clusterer, GPUTPCClusterFinder& clustererShadow, int stage, bool doGPU, int lane)
193 {
194  auto& in = stage ? clustererShadow.mPpeakPositions : clustererShadow.mPpositions;
195  auto& out = stage ? clustererShadow.mPfilteredPeakPositions : clustererShadow.mPpeakPositions;
196  if (doGPU) {
197  const unsigned int iSlice = clusterer.mISlice;
198  auto& count = stage ? clusterer.mPmemory->counters.nPeaks : clusterer.mPmemory->counters.nPositions;
199 
200  std::vector<size_t> counts;
201 
202  unsigned int nSteps = clusterer.getNSteps(count);
203  if (nSteps > clusterer.mNBufs) {
204  GPUError("Clusterer buffers exceeded (%u > %u)", nSteps, (int)clusterer.mNBufs);
205  exit(1);
206  }
207 
208  size_t tmpCount = count;
209  if (nSteps > 1) {
210  for (unsigned int i = 1; i < nSteps; i++) {
211  counts.push_back(tmpCount);
212  if (i == 1) {
213  runKernel<GPUTPCCFStreamCompaction, GPUTPCCFStreamCompaction::scanStart>(GetGrid(tmpCount, clusterer.mScanWorkGroupSize, lane), {iSlice}, {}, i, stage);
214  } else {
215  runKernel<GPUTPCCFStreamCompaction, GPUTPCCFStreamCompaction::scanUp>(GetGrid(tmpCount, clusterer.mScanWorkGroupSize, lane), {iSlice}, {}, i, tmpCount);
216  }
217  tmpCount = (tmpCount + clusterer.mScanWorkGroupSize - 1) / clusterer.mScanWorkGroupSize;
218  }
219 
220  runKernel<GPUTPCCFStreamCompaction, GPUTPCCFStreamCompaction::scanTop>(GetGrid(tmpCount, clusterer.mScanWorkGroupSize, lane), {iSlice}, {}, nSteps, tmpCount);
221 
222  for (unsigned int i = nSteps - 1; i > 1; i--) {
223  tmpCount = counts[i - 1];
224  runKernel<GPUTPCCFStreamCompaction, GPUTPCCFStreamCompaction::scanDown>(GetGrid(tmpCount - clusterer.mScanWorkGroupSize, clusterer.mScanWorkGroupSize, lane), {iSlice}, {}, i, clusterer.mScanWorkGroupSize, tmpCount);
225  }
226  }
227 
228  runKernel<GPUTPCCFStreamCompaction, GPUTPCCFStreamCompaction::compactDigits>(GetGrid(count, clusterer.mScanWorkGroupSize, lane), {iSlice}, {}, 1, stage, in, out);
229  } else {
230  auto& nOut = stage ? clusterer.mPmemory->counters.nClusters : clusterer.mPmemory->counters.nPeaks;
231  auto& nIn = stage ? clusterer.mPmemory->counters.nPeaks : clusterer.mPmemory->counters.nPositions;
232  size_t count = 0;
233  for (size_t i = 0; i < nIn; i++) {
234  if (clusterer.mPisPeak[i]) {
235  out[count++] = in[i];
236  }
237  }
238  nOut = count;
239  }
240 }

242 std::pair<unsigned int, unsigned int> GPUChainTracking::RunTPCClusterizer_transferZS(int iSlice, const CfFragment& fragment, int lane)
243 {
244  bool doGPU = GetRecoStepsGPU() & RecoStep::TPCClusterFinding;
245  const auto& retVal = TPCClusterizerDecodeZSCountUpdate(iSlice, fragment);
246  if (doGPU) {
247  GPUTPCClusterFinder& clusterer = processors()->tpcClusterer[iSlice];
248  GPUTPCClusterFinder& clustererShadow = doGPU ? processorsShadow()->tpcClusterer[iSlice] : clusterer;
249  unsigned int nPagesSector = 0;
250  for (unsigned int j = 0; j < GPUTrackingInOutZS::NENDPOINTS; j++) {
251  unsigned int nPages = 0;
252  mInputsHost->mPzsMeta->slice[iSlice].zsPtr[j] = &mInputsShadow->mPzsPtrs[iSlice * GPUTrackingInOutZS::NENDPOINTS + j];
253  mInputsHost->mPzsPtrs[iSlice * GPUTrackingInOutZS::NENDPOINTS + j] = clustererShadow.mPzs + (nPagesSector + nPages) * TPCZSHDR::TPC_ZS_PAGE_SIZE;
254  for (unsigned int k = clusterer.mMinMaxCN[j].minC; k < clusterer.mMinMaxCN[j].maxC; k++) {
255  const unsigned int min = (k == clusterer.mMinMaxCN[j].minC) ? clusterer.mMinMaxCN[j].minN : 0;
256  const unsigned int max = (k + 1 == clusterer.mMinMaxCN[j].maxC) ? clusterer.mMinMaxCN[j].maxN : mIOPtrs.tpcZS->slice[iSlice].nZSPtr[j][k];
257  if (max > min) {
258  char* src = (char*)mIOPtrs.tpcZS->slice[iSlice].zsPtr[j][k] + min * TPCZSHDR::TPC_ZS_PAGE_SIZE;
259  size_t size = o2::raw::RDHUtils::getMemorySize(*(const o2::header::RAWDataHeader*)src);
260  size = (max - min - 1) * TPCZSHDR::TPC_ZS_PAGE_SIZE + (size ? TPCZSHDR::TPC_ZS_PAGE_SIZE : size);
261  GPUMemCpy(RecoStep::TPCClusterFinding, clustererShadow.mPzs + (nPagesSector + nPages) * TPCZSHDR::TPC_ZS_PAGE_SIZE, src, size, lane, true);
262  }
263  nPages += max - min;
264  }
265  mInputsHost->mPzsMeta->slice[iSlice].nZSPtr[j] = &mInputsShadow->mPzsSizes[iSlice * GPUTrackingInOutZS::NENDPOINTS + j];
266  mInputsHost->mPzsSizes[iSlice * GPUTrackingInOutZS::NENDPOINTS + j] = nPages;
267  mInputsHost->mPzsMeta->slice[iSlice].count[j] = 1;
268  nPagesSector += nPages;
269  }
270  GPUMemCpy(RecoStep::TPCClusterFinding, clustererShadow.mPzsOffsets, clusterer.mPzsOffsets, clusterer.mNMaxPages * sizeof(*clusterer.mPzsOffsets), lane, true);
271  }
272  return retVal;
273 }

320  }

322  processors()->tpcClusterer[iSlice].SetNMaxDigits(processors()->tpcClusterer[iSlice].mPmemory->counters.nDigits, mCFContext->nPagesFragmentMax, nDigitsFragment[iSlice]);

325  }
326  if (mPipelineNotifyCtx && GetProcessingSettings().doublePipelineClusterizer) {
327  mPipelineNotifyCtx->rec->AllocateRegisteredForeignMemory(processors()->tpcClusterer[iSlice].mZSOffsetId, mRec);
328  mPipelineNotifyCtx->rec->AllocateRegisteredForeignMemory(processors()->tpcClusterer[iSlice].mZSId, mRec);
329  } else {
330  AllocateRegisteredMemory(processors()->tpcClusterer[iSlice].mZSOffsetId);
331  AllocateRegisteredMemory(processors()->tpcClusterer[iSlice].mZSId);
332  }
333  }
334  } else {
335  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
336  unsigned int nDigits = mIOPtrs.tpcPackedDigits->nTPCDigits[iSlice];
337  mRec->MemoryScalers()->nTPCdigits += nDigits;
338  processors()->tpcClusterer[iSlice].SetNMaxDigits(nDigits, mCFContext->nPagesFragmentMax, nDigits);
339  }
340  }
341  if (mIOPtrs.tpcZS) {
342  GPUInfo("Event has %u 8kb TPC ZS pages, %lld digits", mCFContext->nPagesTotal, (long long int)mRec->MemoryScalers()->nTPCdigits);
343  } else {
344  GPUInfo("Event has %lld TPC Digits", (long long int)mRec->MemoryScalers()->nTPCdigits);
345  }
346  mCFContext->fragmentFirst = CfFragment{std::max<int>(mCFContext->tpcMaxTimeBin + 1, TPC_MAX_FRAGMENT_LEN), TPC_MAX_FRAGMENT_LEN};
347  for (int iSlice = 0; iSlice < GetProcessingSettings().nTPCClustererLanes && iSlice < NSLICES; iSlice++) {
348  if (mIOPtrs.tpcZS && mCFContext->nPagesSector[iSlice]) {
349  mCFContext->nextPos[iSlice] = RunTPCClusterizer_transferZS(iSlice, mCFContext->fragmentFirst, GetProcessingSettings().nTPCClustererLanes + iSlice);
350  }
351  }
352

354  for (unsigned int iSlice = 0; iSlice < NSLICES; iSlice++) {
355  mCFContext->ptrSave[iSlice].zsOffsetHost = processors()->tpcClusterer[iSlice].mPzsOffsets;
356  mCFContext->ptrSave[iSlice].zsOffsetDevice = processorsShadow()->tpcClusterer[iSlice].mPzsOffsets;
357  mCFContext->ptrSave[iSlice].zsDevice = processorsShadow()->tpcClusterer[iSlice].mPzs;
358  }
359  }
360  return 0;
361 }

365 int GPUChainTracking::RunTPCClusterizer(bool synchronizeOutput)
366 {
367  if (param().rec.fwdTPCDigitsAsClusters) {
368  return ForwardTPCDigits();
369  }
370 #ifdef GPUCA_TPC_GEOMETRY_O2
371  mRec->PushNonPersistentMemory(qStr2Tag("TPCCLUST"));
372  const auto& threadContext = GetThreadContext();
373  bool doGPU = GetRecoStepsGPU() & RecoStep::TPCClusterFinding;
374  if (RunTPCClusterizer_prepare(mPipelineNotifyCtx && GetProcessingSettings().doublePipelineClusterizer)) {
375  return 1;
376  }
377  if (GetProcessingSettings().ompAutoNThreads && !mRec->IsGPU()) {
378  mRec->SetNOMPThreads(mRec->MemoryScalers()->nTPCdigits / 20000);
379  }
380 
381  mRec->MemoryScalers()->nTPCHits = mRec->MemoryScalers()->NTPCClusters(mRec->MemoryScalers()->nTPCdigits);
382  float tpcHitLowOccupancyScalingFactor = 1.f;
383  if (mIOPtrs.settingsTF && mIOPtrs.settingsTF->hasNHBFPerTF) {
