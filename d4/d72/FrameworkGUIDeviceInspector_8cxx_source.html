100 void optionsTable(const char* label, std::vector<ConfigParamSpec> const& options, const DeviceControl& control)
101 {
102  if (options.empty()) {
103  return;
104  }
105  if (ImGui::CollapsingHeader(label, ImGuiTreeNodeFlags_DefaultOpen)) {
106  ImGui::Columns(2);
107  auto labels = {"Name", "Value"};
108  for (auto& label : labels) {
109  ImGui::TextUnformatted(label);
110  ImGui::NextColumn();
111  }
112  for (auto& option : options) {
113  ImGui::TextUnformatted(option.name.c_str());
114  ImGui::NextColumn();
115  auto currentValueIt = control.options.find(option.name);
116 
117  // Did not find the option
118  if (currentValueIt == control.options.end()) {
119  switch (option.type) {
120  case VariantType::String:
121  ImGui::Text(R"("%s" (default))", option.defaultValue.get<const char*>());
122  break;
123  case VariantType::Int:
124  ImGui::Text("%d (default)", option.defaultValue.get<int>());
125  break;
126  case VariantType::Int64:
127  ImGui::Text("%" PRId64 " (default)", option.defaultValue.get<int64_t>());
128  break;
129  case VariantType::UInt8:
130  ImGui::Text("%d (default)", option.defaultValue.get<uint8_t>());
131  break;
132  case VariantType::UInt16:
133  ImGui::Text("%d (default)", option.defaultValue.get<uint16_t>());
134  break;
135  case VariantType::UInt32:
136  ImGui::Text("%d (default)", option.defaultValue.get<uint32_t>());
137  break;
138  case VariantType::UInt64:
139  ImGui::Text("%" PRIu64 " (default)", option.defaultValue.get<uint64_t>());
140  break;
141  case VariantType::Float:
142  ImGui::Text("%f (default)", option.defaultValue.get<float>());
143  break;
144  case VariantType::Double:
145  ImGui::Text("%f (default)", option.defaultValue.get<double>());
146  break;
147  case VariantType::Empty:
148  ImGui::TextUnformatted(""); // no default value
149  break;
150  default:
151  ImGui::TextUnformatted("unknown");
152  }
153  } else {
154  ImGui::TextUnformatted(currentValueIt->second.c_str());
155  }
156  ImGui::NextColumn();
157  }
158  }
159 
160  ImGui::Columns(1);
161 }

163 void servicesTable(const char* label, std::vector<ServiceSpec> const& services)
164 {
165  if (services.empty()) {
166  return;
167  }
168  if (ImGui::CollapsingHeader(label, ImGuiTreeNodeFlags_DefaultOpen)) {
169  ImGui::Columns(2);
170  auto labels = {"Service", "Kind"};
171  for (auto& label : labels) {
172  ImGui::TextUnformatted(label);
173  ImGui::NextColumn();
174  }
175  for (auto& service : services) {
176  if (!service.name.empty()) {
177  ImGui::TextUnformatted(service.name.c_str());
178  } else {
179  ImGui::TextUnformatted("unknown");
180  }
181  ImGui::NextColumn();
182  switch (service.kind) {
183  case ServiceKind::Serial:
184  ImGui::TextUnformatted("Serial");
185  break;
186  case ServiceKind::Global:
187  ImGui::TextUnformatted("Global");
188  break;
189  case ServiceKind::Stream:
190  ImGui::TextUnformatted("Stream");
191  break;
192  default:
193  ImGui::TextUnformatted("unknown");
194  }
195  ImGui::NextColumn();
196  }
197  }
198 
199  ImGui::Columns(1);
200 }

202 void displayDeviceInspector(DeviceSpec const& spec,
203  DeviceInfo const& info,
204  DeviceMetricsInfo const& metrics,
205  DataProcessorInfo const& metadata,
206  DeviceControl& control)
207 {
208  ImGui::Text("Name: %s", spec.name.c_str());
209  ImGui::Text("Executable: %s", metadata.executable.c_str());
210  if (info.active) {
211  ImGui::Text("Pid: %d", info.pid);
212  } else {
213  ImGui::Text("Pid: %d (exit status: %d)", info.pid, info.exitStatus);
214  }
215 #ifdef DPL_ENABLE_TRACING
216  ImGui::Text("Tracy Port: %d", info.tracyPort);
217 #endif
218  ImGui::Text("Rank: %zu/%zu%%%zu/%zu", spec.rank, spec.nSlots, spec.inputTimesliceId, spec.maxInputTimeslices);
219 
220  if (ImGui::Button("Attach debugger")) {
221  std::string pid = std::to_string(info.pid);
222  setenv("O2DEBUGGEDPID", pid.c_str(), 1);
223 #ifdef __APPLE__
224  std::string defaultAppleDebugCommand =
225  "osascript -e 'tell application \"Terminal\" to activate'"
226  " -e 'tell application \"Terminal\" to do script \"lldb -p \" & (system attribute \"O2DEBUGGEDPID\") & \"; exit\"'";
227  setenv("O2DPLDEBUG", defaultAppleDebugCommand.c_str(), 0);
228 #else
229  setenv("O2DPLDEBUG", "xterm -hold -e gdb attach $O2DEBUGGEDPID &", 0);
230 #endif
231  int retVal = system(getenv("O2DPLDEBUG"));
232  (void)retVal;
233  }
234 
235  ImGui::SameLine();
236  if (ImGui::Button("Profile 30s")) {
237  std::string pid = std::to_string(info.pid);
238  setenv("O2PROFILEDPID", pid.c_str(), 1);
239 #ifdef __APPLE__
240  auto defaultAppleProfileCommand = fmt::format(
241  "osascript -e 'tell application \"Terminal\"'"
242  " -e 'activate'"
243  " -e 'do script \"xcrun xctrace record --output dpl-profile-{0}.trace"
244  " --time-limit 30s --template Time\\\\ Profiler --attach {0} "
245  " && open dpl-profile-{0}.trace && exit\"'"
246  " -e 'end tell'",
247  pid);
248 
249  setenv("O2DPLPROFILE", defaultAppleProfileCommand.c_str(), 0);
250 #else
251  setenv("O2DPLPROFILE", "xterm -hold -e perf record -a -g -p $O2PROFILEDPID > perf-$O2PROFILEDPID.data &", 0);
252 #endif
253  LOG(error) << getenv("O2DPLPROFILE");
254  int retVal = system(getenv("O2DPLPROFILE"));
255  (void)retVal;
256  }
257 
258 #if DPL_ENABLE_TRACING
259  ImGui::SameLine();
260  if (ImGui::Button("Tracy")) {
261  std::string tracyPort = std::to_string(info.tracyPort);
262  auto cmd = fmt::format("tracy-profiler -p {} -a 127.0.0.1 &", info.tracyPort);
263  LOG(debug) << cmd;
264  int retVal = system(cmd.c_str());
265  (void)retVal;
266  }
267 #endif
268  if (control.controller) {
269  if (ImGui::Button("Offer SHM")) {
270  control.controller->write("/shm-offer 1000", strlen("/shm-offer 1000"));
271  }
272 
273  if (ImGui::Button("Restart")) {
274  control.controller->write("/restart", strlen("/restart"));
275  }
276  }
277 
278  deviceInfoTable("Inputs:", info.queriesViewIndex, metrics);
279  deviceInfoTable("Outputs:", info.outputsViewIndex, metrics);
280  configurationTable(info.currentConfig, info.currentProvenance);
281  optionsTable("Workflow Options", metadata.workflowOptions, control);
282  servicesTable("Services", spec.services);
283  if (ImGui::CollapsingHeader("Command line arguments", ImGuiTreeNodeFlags_DefaultOpen)) {
284  for (auto& arg : metadata.cmdLineArgs) {
285  ImGui::Text("%s", arg.c_str());
286  }
287  }
288 
289  if (ImGui::CollapsingHeader("Channels", ImGuiTreeNodeFlags_DefaultOpen)) {
290  ImGui::Text("# channels: %lu", spec.inputChannels.size() + spec.outputChannels.size());
291  ChannelsTableHelper::channelsTable("Inputs:", spec.inputChannels);
292  ChannelsTableHelper::channelsTable("Outputs:", spec.outputChannels);
293  }
294  if (ImGui::CollapsingHeader("Data relayer")) {
295  ImGui::Text("Completion policy: %s", spec.completionPolicy.name.c_str());
296  }
297  if (ImGui::CollapsingHeader("Signals", ImGuiTreeNodeFlags_DefaultOpen)) {
298  if (ImGui::Button("SIGSTOP")) {
299  kill(info.pid, SIGSTOP);
300  }
301  ImGui::SameLine();
302  if (ImGui::Button("SIGTERM")) {
303  kill(info.pid, SIGTERM);
304  }
305  ImGui::SameLine();
306  if (ImGui::Button("SIGKILL")) {
307  kill(info.pid, SIGKILL);
308  }
309  if (ImGui::Button("SIGCONT")) {
310  kill(info.pid, SIGCONT);
311  }
312  ImGui::SameLine();
313  if (ImGui::Button("SIGUSR1")) {
314  kill(info.pid, SIGUSR1);
315  }
316  ImGui::SameLine();
317  if (ImGui::Button("SIGUSR2")) {
318  kill(info.pid, SIGUSR2);
319  }
320  }
321 }
