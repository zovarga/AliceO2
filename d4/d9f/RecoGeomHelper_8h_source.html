112 inline RecoGeomHelper::Relation RecoGeomHelper::RecoLadder::isPhiOutside(float phi, float toler) const
113 {
114  // check if phi+-toler is out of the limits of the ladder phi.
115  // return -1 or +1 if phi is above or below the region. If inside, return 0
116  float dif = phi - phiMean, difa = fabs(dif);
117  if (difa < dphiH + toler) {
118  return RecoGeomHelper::Inside;
119  }
120  if (difa > o2::constants::math::PI) { // wraps?
121  difa = o2::constants::math::TwoPI - difa;
122  if (difa < dphiH + toler) {
123  return RecoGeomHelper::Inside;
124  }
125  return dif < 0 ? RecoGeomHelper::Above : RecoGeomHelper::Below;
126  }
127  return dif < 0 ? RecoGeomHelper::Below : RecoGeomHelper::Above;
128 }

131 inline int RecoGeomHelper::RecoLayer::getChipID(float z) const
132 {
133  // Get chip ID within the ladder corresponding to this phi
134  // Note: this is an approximate method, one should check also the neighbouring ladders +/-1
135  int ic = (z - zRange.getMin()) * z2chipID;
136  return ic < 0 ? 0 : (ic < lastChipInLadder ? ic : lastChipInLadder);
137 }

140 inline int RecoGeomHelper::RecoLayer::getLadderID(float phi) const
141 {
142  // Get ladder ID corresponding to phi.
143  // Note: this is an approximate method, precise within 1/3 of average ladder width,
144  // one should check also the neighbouring ladders +/-1
145  o2::math_utils::bringTo02Pi(phi);
146  constexpr float PI2Inv = 1.f / o2::constants::math::TwoPI;
147  return phi2ladder[int(phi * PI2Inv * phi2ladder.size())];
148 }
