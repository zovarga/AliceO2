87 bool Millepede2Record::fillTrack(const AlignmentTrack* trc, const int* id2Lab)
88 {
89  // fill track info, optionally substitutind glopar par ID by label
90  //
91  if (!trc->getDerivDone()) {
92  LOG(error) << "Track derivatives are not yet evaluated";
93  return false;
94  }
95  mNVarLoc = trc->getNLocPar(); // number of local degrees of freedom in the track
96  mNResid = 0;
97  mNDLocTot = 0;
98  mNDGloTot = 0;
99  mChi2Ini = trc->getChi2Ini();
100  mQ2Pt = trc->getQ2Pt();
101  mTgl = trc->getTgl();
102  mNMeas = 0;
103  setCosmic(trc->isCosmic());
104  // 1) check sizes for buffers, expand if needed
105  int np = trc->getNPoints();
106  int nres = 0;
107  int nlocd = 0;
108  int nglod = 0;
109  for (int ip = np; ip--;) {
110  AlignmentPoint* pnt = trc->getPoint(ip);
111  int ngl = pnt->getNGloDOFs(); // number of DOF's this point depends on
112  if (pnt->containsMeasurement()) {
113  nres += 2; // every point has 2 residuals
114  nlocd += mNVarLoc + mNVarLoc; // each residual has max mNVarLoc local derivatives
115  nglod += ngl + ngl; // number of global derivatives
116  mNMeas++;
117  }
118  if (pnt->containsMaterial()) {
119  int nmatpar = pnt->getNMatPar();
120  nres += nmatpar; // each point with materials has nmatpar fake residuals
121  nlocd += nmatpar; // and nmatpar non-0 local derivatives (orthogonal)
122  }
123  }
124  //
125  resize(nres, nlocd, nglod);
126  int nParETP = trc->getNLocExtPar(); // numnber of local parameters for reference track param
127  //
128  const int* gloParID = trc->getGloParID(); // IDs of global DOFs this track depends on
129  for (int ip = 0; ip < np; ip++) {
130  AlignmentPoint* pnt = trc->getPoint(ip);
131  if (pnt->containsMeasurement()) {
132  int gloOffs = pnt->getDGloOffs(); // 1st entry of global derivatives for this point
133  int nDGlo = pnt->getNGloDOFs(); // number of global derivatives (number of DOFs it depends on)
134  if (!pnt->isStatOK()) {
135  pnt->incrementStat();
136  }
137  //
138  for (int idim = 0; idim < 2; idim++) { // 2 dimensional orthogonal measurement
139  mNDGlo[mNResid] = 0;
140  mVolID[mNResid] = pnt->getSensor()->getVolID() + 1;
141  //
142  // measured residual/error
143  mResid[mNResid] = trc->getResidual(idim, ip);
144  mResErr[mNResid] = Sqrt(pnt->getErrDiag(idim));
145  //
146  // derivatives over local params
147  const double* deriv = trc->getDResDLoc(idim, ip); // array of Dresidual/Dparams_loc
148  int nnon0 = 0;
149  for (int j = 0; j < nParETP; j++) { // derivatives over reference track parameters
150  if (isZeroAbs(deriv[j])) {
151  continue;
152  }
153  nnon0++;
154  mDLoc[mNDLocTot] = deriv[j]; // store non-0 derivative
155  mIDLoc[mNDLocTot] = j; // and variable id
156  mNDLocTot++;
157  }
158  int lp0 = pnt->getMinLocVarID(); // point may depend on material variables starting from this one
159  int lp1 = pnt->getMaxLocVarID(); // and up to this one (exclusive)
160  for (int j = lp0; j < lp1; j++) { // derivatives over material variables
161  if (isZeroAbs(deriv[j])) {
162  continue;
163  }
164  nnon0++;
165  mDLoc[mNDLocTot] = deriv[j]; // store non-0 derivative
166  mIDLoc[mNDLocTot] = j; // and variable id
167  mNDLocTot++;
168  }
169  //
170  mNDLoc[mNResid] = nnon0; // local derivatives done, store their number for this residual
171  //
172  // derivatives over global params
173  nnon0 = 0;
174  deriv = trc->getDResDGlo(idim, gloOffs);
175  const int* gloIDP = gloParID + gloOffs;
176  for (int j = 0; j < nDGlo; j++) {
177  if (isZeroAbs(deriv[j])) {
178  continue;
179  }
180  nnon0++;
181  mDGlo[mNDGloTot] = deriv[j]; // value of derivative
182  mIDGlo[mNDGloTot] = id2Lab ? id2Lab[gloIDP[j]] : gloIDP[j] + 1; // global DOF ID
183  mNDGloTot++;
184  }
185  mNDGlo[mNResid] = nnon0;
186  //
187  mNResid++;
188  }
189  }
190  if (pnt->containsMaterial()) { // material point can add 4 or 5 otrhogonal pseudo-measurements
191  int nmatpar = pnt->getNMatPar(); // residuals (correction expectation value)
192  // const float* expMatCorr = pnt->getMatCorrExp(); // expected corrections (diagonalized)
193  const float* expMatCov = pnt->getMatCorrCov(); // their diagonalized error matrix
194  int offs = pnt->getMaxLocVarID() - nmatpar; // start of material variables
195  // here all derivatives are 1 = dx/dx
196  for (int j = 0; j < nmatpar; j++) {
197  mNDGlo[mNResid] = 0; // mat corrections don't depend on global params
198  mVolID[mNResid] = 0; // not associated to global parameter
199  mResid[mNResid] = 0; // expectation for MS effects is 0
200  mResErr[mNResid] = Sqrt(expMatCov[j]);
201  mNDLoc[mNResid] = 1; // only 1 non-0 derivative
202  mDLoc[mNDLocTot] = 1.0;
203  mIDLoc[mNDLocTot] = offs + j; // variable id
204  mNDLocTot++;
205  mNResid++;
206  }
207  }
208  }
209  //
210  if (!mNDGloTot) {
211  LOG(info) << "Track does not depend on free global parameters, discard";
212  return false;
213  }
214  return true;
215 }

218 void Millepede2Record::resize(int nresid, int nloc, int nglo)
219 {
220  // resize container
221  if (nresid > mNResidBook) {
222  delete[] mNDLoc;
223  delete[] mNDGlo;
224  delete[] mVolID;
225  delete[] mResid;
226  delete[] mResErr;
227  mNDLoc = new int16_t[nresid];
228  mNDGlo = new int[nresid];
229  mVolID = new int[nresid];
230  mResid = new float[nresid];
231  mResErr = new float[nresid];
232  mNResidBook = nresid;
233  memset(mNDLoc, 0, nresid * sizeof(int16_t));
234  memset(mNDGlo, 0, nresid * sizeof(int));
235  memset(mVolID, 0, nresid * sizeof(int));
236  memset(mResid, 0, nresid * sizeof(float));
237  memset(mResErr, 0, nresid * sizeof(float));
238  }
239  if (nloc > mNDLocTotBook) {
240  delete[] mIDLoc;
241  delete[] mDLoc;
242  mIDLoc = new int16_t[nloc];
243  mDLoc = new float[nloc];
244  mNDLocTotBook = nloc;
245  memset(mIDLoc, 0, nloc * sizeof(int16_t));
246  memset(mDLoc, 0, nloc * sizeof(float));
247  }
248  if (nglo > mNDGloTotBook) {
249  delete[] mIDGlo;
250  delete[] mDGlo;
251  mIDGlo = new int[nglo];
252  mDGlo = new float[nglo];
253  mNDGloTotBook = nglo;
254  memset(mIDGlo, 0, nglo * sizeof(int));
255  memset(mDGlo, 0, nglo * sizeof(float));
256  }
257  //
258 }

275 {

277  //

323  }
324  // if (!eolOK) printf("\n");
