69 void DigitReader::run(ProcessingContext& pc)
70 {
71  auto ent = mTree->GetReadEntry() + 1;
72  assert(ent < mTree->GetEntries()); // this should not happen
73 
74  mTree->GetEntry(ent);
75  LOG(info) << "DigitReader pushes " << mDigits.size() << " digits at entry " << ent;
76  pc.outputs().snapshot(Output{"CTP", "DIGITS", 0, Lifetime::Timeframe}, mDigits);
77  if (mTree->GetReadEntry() + 1 >= mTree->GetEntries()) {
78  pc.services().get<ControlService>().endOfStream();
79  pc.services().get<ControlService>().readyToQuit(QuitRequest::Me);
80  }
81 }

83 void DigitReader::connectTree(const std::string& filename)
84 {
85  mTree.reset(nullptr); // in case it was already loaded
86  mFile.reset(TFile::Open(filename.c_str()));
87  assert(mFile && !mFile->IsZombie());
88  mTree.reset((TTree*)mFile->Get(mDigTreeName.c_str()));
89  assert(mTree);
90  mTree->SetBranchAddress(mDigitBranchName.c_str(), &mDigitsPtr);
91  LOG(info) << "Loaded tree from " << filename << " with " << mTree->GetEntries() << " entries";
92 }
