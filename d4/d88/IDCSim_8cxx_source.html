25 void o2::tpc::IDCSim::integrateDigitsForOneTF(const gsl::span<const o2::tpc::Digit>& digits)
26 {
27  resetIDCs();
28 
29  // loop over digits from one sector for ALL Time Frames
30  const unsigned int switchAfterTB = getLastTimeBinForSwitch();
31 
32  if (mAddInterval) {
33  // decrease the size of the vector if the last integration intervall is empty
34  if (switchAfterTB == (mIntegrationIntervalsPerTF - 1) * mTimeStampsPerIntegrationInterval) {
35  for (unsigned int ireg = 0; ireg < Mapper::NREGIONS; ++ireg) {
36  mIDCs[mBufferIndex][ireg].resize(mMaxIDCs[ireg] - Mapper::PADSPERREGION[ireg]);
37  }
38  } else {
39  for (auto& idcs : mIDCs[mBufferIndex]) {
40  idcs.resize(idcs.capacity());
41  }
42  }
43  }
44 
45  for (const auto& digit : digits) {
46  const o2::tpc::CRU cru(digit.getCRU());
47  const unsigned int region = cru.region();
48  const int timeStamp = digit.getTimeStamp();
49  if (timeStamp < switchAfterTB) {
50  const unsigned int indexPad = getIndex(timeStamp, region, digit.getRow(), digit.getPad());
51  mIDCs[mBufferIndex][region][indexPad] += digit.getChargeFloat();
52  } else {
53  const unsigned int indexPad = getIndex(timeStamp - switchAfterTB, region, digit.getRow(), digit.getPad());
54  mIDCs[!mBufferIndex][region][indexPad] += digit.getChargeFloat();
55  }
56  }
57 
58  mBufferIndex = !mBufferIndex; // switch buffer index
59  setNewOffset(); // set offset
60 }

62 unsigned int o2::tpc::IDCSim::getLastTimeBinForSwitch() const
63 {
64  const int totaloffs = mTimeBinsOff + static_cast<int>(mTimeStampsRemainder);
65  return (totaloffs >= mTimeStampsPerIntegrationInterval) ? mIntegrationIntervalsPerTF * mTimeStampsPerIntegrationInterval - mTimeBinsOff : (mIntegrationIntervalsPerTF - mAddInterval) * mTimeStampsPerIntegrationInterval - mTimeBinsOff;
66 }

69 {
70  const int totaloffs = mTimeBinsOff + static_cast<int>(mTimeStampsRemainder);
71  mTimeBinsOff = (totaloffs >= mTimeStampsPerIntegrationInterval) ? (totaloffs - static_cast<int>(mTimeStampsPerIntegrationInterval)) : totaloffs;

119 void o2::tpc::IDCSim::createDebugTree(const IDCSim& idcsim, o2::utils::TreeStreamRedirector& pcstream)
120 {
121  const Mapper& mapper = Mapper::instance();
122  const unsigned int sector = idcsim.getSector();
123  unsigned int cru = sector * Mapper::NREGIONS;
124 
125  // loop over data from regions
126  for (const auto& idcs : idcsim.get()) {
127  int sectorTmp = sector;
128  const o2::tpc::CRU cruTmp(cru);
129  unsigned int region = cruTmp.region();
130  const unsigned long padsPerCRU = Mapper::PADSPERREGION[region];
131  std::vector<int> vRow(padsPerCRU);
132  std::vector<int> vPad(padsPerCRU);
133  std::vector<float> vXPos(padsPerCRU);
134  std::vector<float> vYPos(padsPerCRU);
135  std::vector<float> vGlobalXPos(padsPerCRU);
136  std::vector<float> vGlobalYPos(padsPerCRU);
137  std::vector<float> idcsPerTimeBin(padsPerCRU); // idcs for one time bin
138 
139  for (unsigned int iPad = 0; iPad < padsPerCRU; ++iPad) {
140  const GlobalPadNumber globalNum = Mapper::GLOBALPADOFFSET[region] + iPad;
141  const auto& padPosLocal = mapper.padPos(globalNum);
142  vRow[iPad] = padPosLocal.getRow();
143  vPad[iPad] = padPosLocal.getPad();
144  vXPos[iPad] = mapper.getPadCentre(padPosLocal).X();
145  vYPos[iPad] = mapper.getPadCentre(padPosLocal).Y();
146  const GlobalPosition2D globalPos = mapper.LocalToGlobal(LocalPosition2D(vXPos[iPad], vYPos[iPad]), cruTmp.sector());
147  vGlobalXPos[iPad] = globalPos.X();
148  vGlobalYPos[iPad] = globalPos.Y();
149  }
150 
151  for (unsigned int integrationInterval = 0; integrationInterval < idcsim.getNIntegrationIntervalsPerTF(); ++integrationInterval) {
152  for (unsigned int iPad = 0; iPad < padsPerCRU; ++iPad) {
153  idcsPerTimeBin[iPad] = (idcs)[iPad + integrationInterval * Mapper::PADSPERREGION[region]];
154  }
155 
156  pcstream << "tree"
157  << "cru=" << cru
158  << "sector=" << sectorTmp
159  << "region=" << region
160  << "integrationInterval=" << integrationInterval
161  << "IDC.=" << idcsPerTimeBin
162  << "pad.=" << vPad
163  << "row.=" << vRow
164  << "lx.=" << vXPos
165  << "ly.=" << vYPos
166  << "gx.=" << vGlobalXPos
167  << "gy.=" << vGlobalYPos
168  << "\n";
169  }
170  ++cru;
171  }
172 }

174 void o2::tpc::IDCSim::drawIDCs(const unsigned int integrationInterval, const std::string filename) const
175 {
176  const auto coords = o2::tpc::painter::getPadCoordinatesSector();

178  poly->SetContour(255);
179  poly->SetTitle(nullptr);
180  poly->GetYaxis()->SetTickSize(0.002f);
181  poly->GetYaxis()->SetTitleOffset(0.7f);
182  poly->GetZaxis()->SetTitleOffset(1.3f);
183  poly->SetStats(0);
184 
185  TCanvas* can = new TCanvas("can", "can", 2000, 1400);
186  can->SetRightMargin(0.14f);
187  can->SetLeftMargin(0.06f);
188  can->SetTopMargin(0.04f);
189 
190  TLatex lat;
191  lat.SetTextFont(63);
192  lat.SetTextSize(2);
193 
194  poly->Draw("colz");
195  for (unsigned int region = 0; region < Mapper::NREGIONS; ++region) {
196  for (unsigned int irow = 0; irow < Mapper::ROWSPERREGION[region]; ++irow) {
197  for (unsigned int ipad = 0; ipad < Mapper::PADSPERROW[region][irow]; ++ipad) {
198  const auto padNum = Mapper::getGlobalPadNumber(irow, ipad, region);
199  const auto coordinate = coords[padNum];
200  const float yPos = -0.5 * (coordinate.yVals[0] + coordinate.yVals[2]); // local coordinate system is mirrored
201  const float xPos = 0.5 * (coordinate.xVals[0] + coordinate.xVals[2]);
202  const unsigned int indexIDC = integrationInterval * Mapper::PADSPERREGION[region] + Mapper::OFFSETCRULOCAL[region][irow] + ipad;
203  const float idc = mIDCs[!mBufferIndex][region][indexIDC];
204  poly->Fill(xPos, yPos, idc);
205  lat.SetTextAlign(12);
206  lat.DrawLatex(xPos, yPos, Form("%i", ipad));
207  }
208  }
209  }
210 
211  if (!filename.empty()) {
212  can->SaveAs(filename.data());
213  delete poly;
214  delete can;
215  }
216 }
