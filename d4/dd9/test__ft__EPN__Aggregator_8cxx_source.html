68 WorkflowSpec defineDataProcessing(ConfigContext const& config)
69 {
70  const auto tpcCRUs = o2::RangeTokenizer::tokenize<int>(config.options().get<std::string>("crus"));
71  const auto nCRUs = tpcCRUs.size();
72  const auto first = tpcCRUs.begin();
73  const auto last = std::min(tpcCRUs.end(), first + nCRUs);
74  const std::vector<uint32_t> crus(first, last);
75  const auto timeframes = static_cast<unsigned int>(config.options().get<int>("timeframes"));
76  const auto iondrifttime = static_cast<unsigned int>(config.options().get<int>("ion-drift-time"));
77  const auto nFourierCoefficients = std::clamp(static_cast<unsigned int>(config.options().get<int>("nFourierCoeff")), static_cast<unsigned int>(0), iondrifttime + 2);
78  const auto nthreads = static_cast<unsigned int>(config.options().get<int>("nthreads"));
79  const auto seed = static_cast<unsigned int>(config.options().get<int>("seed"));
80  const auto idcgenlanes = static_cast<unsigned int>(config.options().get<int>("idc-gen-lanes"));
81  const auto idcgentimelanes = static_cast<unsigned int>(config.options().get<int>("idc-gen-time-lanes"));
82 
83  const bool fft = config.options().get<bool>("use-naive-fft");
84  const bool onlyIDCGen = config.options().get<bool>("only-idc-gen");
85  const bool fastgen = config.options().get<bool>("fast-gen");
86  const bool debugFT = config.options().get<bool>("debug");
87  const float sigma = 5;
88  const unsigned int firstTF = 0;
89  const unsigned int nLanes = 1;
90  const bool loadFromFile = false;
91  gRandom->SetSeed(seed);
92 
93  WorkflowSpec workflow;
94  for (int ilane = 0; ilane < idcgenlanes; ++ilane) {
95  const auto crusPerLane = nCRUs / idcgenlanes + ((nCRUs % idcgenlanes) != 0);
96  const auto first = tpcCRUs.begin() + ilane * crusPerLane;
97  if (first >= tpcCRUs.end()) {
98  break;
99  }
100  const auto last = std::min(tpcCRUs.end(), first + crusPerLane);
101  const std::vector<uint32_t> rangeCRUs(first, last);
102  workflow.emplace_back(timePipeline(generateIDCsCRU(ilane, timeframes, rangeCRUs, fastgen), idcgentimelanes));
103  }
104 
105  if (onlyIDCGen == true) {
106  return workflow;
107  }
108 
109  auto workflowTmp = WorkflowSpec{getTPCFLPIDCSpec<TPCFLPIDCDeviceNoGroup>(0, crus, iondrifttime, debugFT, false), getTPCFourierTransformEPNSpec(crus, iondrifttime, nFourierCoefficients, debugFT), getTPCDistributeIDCSpec(crus, timeframes, nLanes, firstTF, loadFromFile), ftAggregatorIDC(nFourierCoefficients, iondrifttime, timeframes, debugFT, crus), receiveFourierCoeffEPN(timeframes, nFourierCoefficients), compare_EPN_AGG()};
110  for (auto& spec : workflowTmp) {
111  workflow.emplace_back(spec);
112  }
113 
114  IDCAverageGroup<IDCAverageGroupTPC>::setNThreads(nthreads);
115  TPCFourierTransformEPNSpec::IDCFType::setFFT(!fft);
116  TPCFourierTransformAggregatorSpec::IDCFType::setNThreads(nthreads);
117  TPCFourierTransformAggregatorSpec::IDCFType::setFFT(!fft);
118  return workflow;
119 }

121 DataProcessorSpec generateIDCsCRU(int lane, const unsigned int maxTFs, const std::vector<uint32_t>& crus, const bool fastgen)
122 {
123  // generate random IDCs per CRU
124  const int nOrbitsPerTF = 128; // number of orbits per TF
125  const int nOrbitsPerIDC = 12; // integration interval in units of orbits per IDC
126  const int nIDCs = nOrbitsPerTF / nOrbitsPerIDC;
127 
128  std::vector<OutputSpec> outputSpecs;
129  outputSpecs.reserve(crus.size());
130  for (const auto& cru : crus) {
131  const o2::header::DataHeader::SubSpecificationType subSpec{cru << 7};
132  outputSpecs.emplace_back(ConcreteDataMatcher{gDataOriginTPC, TPCIntegrateIDCDevice::getDataDescription(TPCIntegrateIDCDevice::IDCFormat::Sim), subSpec});
133  }
134 
135  return DataProcessorSpec{
136  fmt::format("idc-generator-{:02}", lane).data(),
137  Inputs{},
138  outputSpecs,
139  AlgorithmSpec{
140  [maxTFs, fastgen, nIDCs, cruStart = crus.front(), cruEnde = crus.back()](ProcessingContext& ctx) {
141  const auto tf = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ctx.inputs().getFirstValid(true))->tfCounter;
142 
143  for (uint32_t icru = cruStart; icru <= cruEnde; ++icru) {
144  const o2::tpc::CRU cruTmp(icru);
145  const unsigned int nPads = o2::tpc::Mapper::PADSPERREGION[cruTmp.region()];
146  const unsigned int additionalInterval = (tf % 3) ? 1 : 0; // in each integration inerval are either 10 or 11 values when having 128 orbits per TF and 12 orbits integration length
147  const unsigned int nTotIDCs = (nIDCs + additionalInterval) * nPads;
148 
149  // generate random IDCs per CRU
150  if (!fastgen) {
151  o2::pmr::vector<float> idcs;
152  idcs.reserve(nTotIDCs);
153  for (int i = 0; i < nTotIDCs; ++i) {
154  idcs.emplace_back(gRandom->Gaus(1000, 20));
155  }
156  ctx.outputs().adoptContainer(Output{gDataOriginTPC, TPCIntegrateIDCDevice::getDataDescription(TPCIntegrateIDCDevice::IDCFormat::Sim), o2::header::DataHeader::SubSpecificationType{icru << 7}, Lifetime::Timeframe}, std::move(idcs));
157  } else {
158  o2::pmr::vector<float> idcs(nTotIDCs, 1000.);
159  ctx.outputs().adoptContainer(Output{gDataOriginTPC, TPCIntegrateIDCDevice::getDataDescription(TPCIntegrateIDCDevice::IDCFormat::Sim), o2::header::DataHeader::SubSpecificationType{icru << 7}, Lifetime::Timeframe}, std::move(idcs));
160  }
161  }
162 
163  if (tf >= (maxTFs - 1)) {
164  ctx.services().get<ControlService>().readyToQuit(QuitRequest::Me);
165  }
166  }}};
167 }

208 class TPCCompareFourierCoeffSpec : public o2::framework::Task
209 {
211  public:
212  void run(o2::framework::ProcessingContext& ctx) final
213  {
214  LOGP(info, "==== Comparing Fourier coefficients for EPN and Aggregator... ====");
215  const auto fourierCoeffAgg = ctx.inputs().get<FourierCoeff*>(ctx.inputs().get("coeffAgg"));
216  const std::vector<o2::tpc::FourierCoeff> fourierCoeffEPN = ctx.inputs().get<std::vector<o2::tpc::FourierCoeff>>(ctx.inputs().get("coeffEPN"));
217  for (int iside = 0; iside < o2::tpc::SIDES; ++iside) {
218  const o2::tpc::Side side = iside == 0 ? o2::tpc::Side::A : o2::tpc::Side::C;
219  for (int tf = 0; tf < fourierCoeffEPN.size(); ++tf) {
220  for (int i = 0; i < fourierCoeffEPN[tf].getNCoefficientsPerTF(); ++i) {
221  const float epnval = fourierCoeffEPN[tf](side, i);
222  const float aggval = (*fourierCoeffAgg)(side, fourierCoeffAgg->getIndex(tf, i));
223  ASSERT_ERROR((std::abs(std::min(epnval, aggval)) < 1.f) ? isSameZero(epnval, aggval) : isSame(epnval, aggval));
224  }
225  }
226  }
227  LOGP(info, "==== Test finished successfull! Fourier coefficients for EPN and Aggregator are equal. ====");
228  ctx.services().get<ControlService>().endOfStream();
229  ctx.services().get<ControlService>().readyToQuit(QuitRequest::All);
230  }
231 
232  private:
233  const float mTolerance = 1e-4;
234 
235  bool isSameZero(const float epnval, const float aggval) const { return (epnval - aggval) * (epnval - aggval) < mTolerance; }
236  bool isSame(const float epnval, const float aggval) const { return std::abs((epnval - aggval) / std::min(epnval, aggval)) < mTolerance; }
237 };
