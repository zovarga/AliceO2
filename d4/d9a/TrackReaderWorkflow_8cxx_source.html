55 // customize dispatch policy, dispatch immediately what is ready
56 void customize(std::vector<o2::framework::DispatchPolicy>& policies)
57 {
58  using DispatchOp = o2::framework::DispatchPolicy::DispatchOp;
59  // we customize all devices to dispatch data immediately
60  auto readerMatcher = [](auto const& spec) {
61  return std::regex_match(spec.name.begin(), spec.name.end(), std::regex(".*-reader"));
62  };
63  auto triggerMatcher = [](auto const& query) {
64  // a bit of a hack but we want this to be configurable from the command line,
65  // however DispatchPolicy is inserted before all other setup. Triggering depending
66  // on the global variable set from the command line option. If scheduled messages
67  // are not triggered they are sent out at the end of the computation
68  return gDispatchTrigger.origin == query.origin && gDispatchTrigger.description == query.description;
69  };
70  policies.push_back({"prompt-for-reader", readerMatcher, DispatchOp::WhenReady, triggerMatcher});
71 }

80 WorkflowSpec defineDataProcessing(ConfigContext const& cfgc)
81 {
82  WorkflowSpec specs;
83 
84  auto inputType = cfgc.options().get<std::string>("input-type");
85  auto dispmode = cfgc.options().get<std::string>("dispatching-mode");
86  if (dispmode == "complete") {
87  // nothing to do we leave the matcher empty which will suppress the dispatch
88  // trigger and all messages will be sent out together at end of computation
89  } else if (inputType == "tracks") {
90  gDispatchTrigger = o2::framework::Output{"TPC", "TRACKS"};
91  }
92 
93  bool doMC = not cfgc.options().get<bool>("disable-mc");
94 
95  specs.push_back(o2::tpc::getTPCTrackReaderSpec(doMC));
96 
97  return std::move(specs);
