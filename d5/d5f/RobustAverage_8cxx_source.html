27 float o2::tpc::RobustAverage::getStdDev(const float mean)
28 {
29  const auto size = mValues.size();
30  mTmpValues.resize(size);
31  std::transform(mValues.begin(), mValues.end(), mTmpValues.begin(), [mean](const float val) { return val - mean; });
32  const float sqsum = std::inner_product(mTmpValues.begin(), mTmpValues.end(), mTmpValues.begin(), decltype(mTmpValues)::value_type(0));
33  const float stdev = std::sqrt(sqsum / size);
34  return stdev;
35 }

37 float o2::tpc::RobustAverage::getFilteredMean(const float mean, const float stdev, const float sigma)
38 {
39  std::sort(mValues.begin(), mValues.end());
40  const float sigmastddev = sigma * stdev;
41  const float minVal = mean - sigmastddev;
42  const float maxVal = mean + sigmastddev;
43  const auto upper = std::upper_bound(mValues.begin(), mValues.end(), maxVal);
44  const auto lower = std::lower_bound(mValues.begin(), mValues.end(), minVal);
45  return getMean(lower, upper);
46 }
