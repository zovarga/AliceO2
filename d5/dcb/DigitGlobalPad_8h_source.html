120 inline bool DigitGlobalPad::compareMClabels(const MCCompLabel& label1, const MCCompLabel& label2) const
121 {
122  // we compare directly on the bare label (in which eventID, labelID etc. are encoded)
123  // this avoids any logical operator on the label; optimization motivated from an Intel VTune analysis
124  // (note that this is also faster than using the operator= of MCCompLabel)
125  return label1.getRawValue() == label2.getRawValue();
126 }

128 template <DigitzationMode MODE>
129 inline void DigitGlobalPad::fillOutputContainer(std::vector<Digit>& output,
130  dataformats::MCTruthContainer<MCCompLabel>& mcTruth,
131  const CRU& cru, TimeBin timeBin,
132  GlobalPadNumber globalPad,
133  o2::dataformats::LabelContainer<std::pair<MCCompLabel, int>, false>& labels,
134  float commonMode)
135 {
136  const static Mapper& mapper = Mapper::instance();
137  static SAMPAProcessing& sampaProcessing = SAMPAProcessing::instance();
138  const PadPos pad = mapper.padPos(globalPad);
139  static std::vector<std::pair<MCCompLabel, int>> labelCollector; // static workspace container for sorting
140 
143 
144  float noise, pedestal;
145  const float mADC = sampaProcessing.makeSignal<MODE>(mChargePad, cru.sector(), globalPad, commonMode, pedestal, noise);
146 
148  if (mADC > 0 && mChargePad > 0) {
149  auto labelview = labels.getLabels(mID);
150 
152  const auto digiPos = output.size();
153  output.emplace_back(cru, mADC, pad.getRow(), pad.getPad(), timeBin);
154 
155  labelCollector.clear();
156  for (auto& mcLabel : labelview) {
157  labelCollector.push_back(mcLabel);
158  }
159  if (labelview.size() > 1) {
161  using P = std::pair<MCCompLabel, int>;
162  std::sort(labelCollector.begin(), labelCollector.end(), [](const P& a, const P& b) { return a.second > b.second; });
163  }
164  for (auto& mcLabel : labelCollector) {
165  mcTruth.addElement(digiPos, mcLabel.first);
166  }
167  }
168 }
