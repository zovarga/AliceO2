102 template <typename DataT>
103 void o2::tpc::IDCCCDBHelper<DataT>::dumpToTree(int integrationIntervals, const char* outFileName) const
104 {
105  const Mapper& mapper = Mapper::instance();
106  o2::utils::TreeStreamRedirector pcstream(outFileName, "RECREATE");
107  pcstream.GetFile()->cd();
108 
109  if (integrationIntervals <= 0) {
110  integrationIntervals = std::min(getNIntegrationIntervalsIDCDelta(Side::A), getNIntegrationIntervalsIDCDelta(Side::C));
111  }
112 
113  for (unsigned int integrationInterval = 0; integrationInterval < integrationIntervals; ++integrationInterval) {
114  const unsigned int nIDCsSector = Mapper::getPadsInSector() * Mapper::NSECTORS;
115  std::vector<int> vRow(nIDCsSector);
116  std::vector<int> vPad(nIDCsSector);
117  std::vector<float> vXPos(nIDCsSector);
118  std::vector<float> vYPos(nIDCsSector);
119  std::vector<float> vGlobalXPos(nIDCsSector);
120  std::vector<float> vGlobalYPos(nIDCsSector);
121  std::vector<float> idcs(nIDCsSector);
122  std::vector<float> idcsZero(nIDCsSector);
123  std::vector<float> idcsDelta(nIDCsSector);
124  std::vector<unsigned int> sectorv(nIDCsSector);
125 
126  unsigned int index = 0;
127  for (unsigned int sector = 0; sector < Mapper::NSECTORS; ++sector) {
128  for (unsigned int region = 0; region < Mapper::NREGIONS; ++region) {
129  for (unsigned int irow = 0; irow < Mapper::ROWSPERREGION[region]; ++irow) {
130  for (unsigned int ipad = 0; ipad < Mapper::PADSPERROW[region][irow]; ++ipad) {
131  const auto padNum = Mapper::getGlobalPadNumber(irow, ipad, region);
132  const auto padTmp = (sector < SECTORSPERSIDE) ? ipad : (Mapper::PADSPERROW[region][irow] - ipad - 1); // C-Side is mirrored
133  const auto& padPosLocal = mapper.padPos(padNum);
134  vRow[index] = padPosLocal.getRow();
135  vPad[index] = padPosLocal.getPad();
136  vXPos[index] = mapper.getPadCentre(padPosLocal).X();
137  vYPos[index] = mapper.getPadCentre(padPosLocal).Y();
138  const GlobalPosition2D globalPos = mapper.LocalToGlobal(LocalPosition2D(vXPos[index], vYPos[index]), Sector(sector));
139  vGlobalXPos[index] = globalPos.X();
140  vGlobalYPos[index] = globalPos.Y();
141  idcs[index] = getIDCVal(sector, region, irow, padTmp, integrationInterval);
142  idcsZero[index] = getIDCZeroVal(sector, region, irow, padTmp);
143  idcsDelta[index] = getIDCDeltaVal(sector, region, irow, padTmp, integrationInterval);
144  sectorv[index++] = sector;
145  }
146  }
147  }
148  }
149  float idcOneA = getIDCOneVal(Side::A, integrationInterval);
150  float idcOneC = getIDCOneVal(Side::C, integrationInterval);
151 
152  pcstream << "tree"
153  << "integrationInterval=" << integrationInterval
154  << "IDC.=" << idcs
155  << "IDC0.=" << idcsZero
156  << "IDC1A=" << idcOneA
157  << "IDC1C=" << idcOneC
158  << "IDCDelta.=" << idcsDelta
159  << "pad.=" << vPad
160  << "row.=" << vRow
161  << "lx.=" << vXPos
162  << "ly.=" << vYPos
163  << "gx.=" << vGlobalXPos
164  << "gy.=" << vGlobalYPos
165  << "sector.=" << sectorv
166  << "\n";
167  }
168  pcstream.Close();
169 }
