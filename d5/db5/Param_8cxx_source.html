189 void Param::idealPosition(Int_t iCh, TGeoHMatrix* pMatrix)
190 {
191  // Construct ideal position matrix for a given chamber
192  // Arguments: iCh- chamber ID; pMatrix- pointer to precreated unity matrix where to store the results
193  // Returns: none
194  const double kAngHor = 19.5; // horizontal angle between chambers 19.5 grad
195  const double kAngVer = 20; // vertical angle between chambers 20 grad
196  const double kAngCom = 30; // common HMPID rotation with respect to x axis 30 grad
197  const double kTrans[3] = {490, 0, 0}; // center of the chamber is on window-gap surface
198  pMatrix->RotateY(90); // rotate around y since initial position is in XY plane -> now in YZ plane
199  pMatrix->SetTranslation(kTrans); // now plane in YZ is shifted along x
200  switch (iCh) {
201  case 0:
202  pMatrix->RotateY(kAngHor);
203  pMatrix->RotateZ(-kAngVer);
204  break; //right and down
205  case 1:
206  pMatrix->RotateZ(-kAngVer);
207  break; //down
208  case 2:
209  pMatrix->RotateY(kAngHor);
210  break; //right
211  case 3:
212  break; //no rotation
213  case 4:
214  pMatrix->RotateY(-kAngHor);
215  break; //left
216  case 5:
217  pMatrix->RotateZ(kAngVer);
218  break; //up
219  case 6:
220  pMatrix->RotateY(-kAngHor);
221  pMatrix->RotateZ(kAngVer);
222  break; //left and up
223  }
224  pMatrix->RotateZ(kAngCom); //apply common rotation in XY plane
225 }

278 //+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
279 double Param::sigma2(double trkTheta, double trkPhi, double ckovTh, double ckovPh)
280 {
281  // Analithical calculation of total error (as a sum of localization, geometrical and chromatic errors)
282  // on Cerenkov angle for a given Cerenkov photon
283  // created by a given MIP. Fromulae according to CERN-EP-2000-058
284  // Arguments: Cerenkov and azimuthal angles for Cerenkov photon, [radians]
285  // dip and azimuthal angles for MIP taken at the entrance to radiator, [radians]
286  // MIP beta
287  // Returns: absolute error on Cerenkov angle, [radians]
288 
289  TVector3 v(-999, -999, -999);
290  double trkBeta = 1. / (TMath::Cos(ckovTh) * getRefIdx());
291 
292  if (trkBeta > 1) {
293  trkBeta = 1; //protection against bad measured thetaCer
294  }
295  if (trkBeta < 0) {
296  trkBeta = 0.0001; //
297  }
298 
299  v.SetX(sigLoc(trkTheta, trkPhi, ckovTh, ckovPh, trkBeta));
300  v.SetY(sigGeom(trkTheta, trkPhi, ckovTh, ckovPh, trkBeta));
301  v.SetZ(sigCrom(trkTheta, trkPhi, ckovTh, ckovPh, trkBeta));
302 
303  return v.Mag2();
304 }
305 //++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
306 double Param::sigLoc(double trkTheta, double trkPhi, double thetaC, double phiC, double betaM)
307 {
308  // Analitical calculation of localization error (due to finite segmentation of PC) on Cerenkov angle for a given
309  // Cerenkov photon
310  // created by a given MIP. Fromulae according to CERN-EP-2000-058
311  // Arguments: Cerenkov and azimuthal angles for Cerenkov photon, [radians]
312  // dip and azimuthal angles for MIP taken at the entrance to radiator, [radians]
313  // MIP beta
314  // Returns: absolute error on Cerenkov angle, [radians]
315 
316  double phiDelta = phiC - trkPhi;
317 
318  double sint = TMath::Sin(trkTheta);
319  double cost = TMath::Cos(trkTheta);
320  double sinf = TMath::Sin(trkPhi);
321  double cosf = TMath::Cos(trkPhi);
322  double sinfd = TMath::Sin(phiDelta);
323  double cosfd = TMath::Cos(phiDelta);
324  double tantheta = TMath::Tan(thetaC);
325 
326  double alpha = cost - tantheta * cosfd * sint; // formula (11)
327  double k = 1. - getRefIdx() * getRefIdx() + alpha * alpha / (betaM * betaM); // formula (after 8 in the text)
328  if (k < 0) {
329  return 1e10;
330  }
331  double mu = sint * sinf + tantheta * (cost * cosfd * sinf + sinfd * cosf); // formula (10)
332  double e = sint * cosf + tantheta * (cost * cosfd * cosf - sinfd * sinf); // formula (9)
333 
334  double kk = betaM * TMath::Sqrt(k) / (gapThick() * alpha); // formula (6) and (7)
335  // formula (6)
336  double dtdxc = kk * (k * (cosfd * cosf - cost * sinfd * sinf) - (alpha * mu / (betaM * betaM)) * sint * sinfd);
337  // formula (7) pag.4
338  double dtdyc = kk * (k * (cosfd * sinf + cost * sinfd * cosf) + (alpha * e / (betaM * betaM)) * sint * sinfd);
339 
340  double errX = 0.2, errY = 0.25; //end of page 7
341  return TMath::Sqrt(errX * errX * dtdxc * dtdxc + errY * errY * dtdyc * dtdyc);
342 }
343 //+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
344 double Param::sigCrom(double trkTheta, double trkPhi, double thetaC, double phiC, double betaM)
345 {
346  // Analitical calculation of chromatic error (due to lack of knowledge of Cerenkov photon energy)
347  // on Cerenkov angle for a given Cerenkov photon
348  // created by a given MIP. Fromulae according to CERN-EP-2000-058
349  // Arguments: Cerenkov and azimuthal angles for Cerenkov photon, [radians]
350  // dip and azimuthal angles for MIP taken at the entrance to radiator, [radians]
351  // MIP beta
352  // Returns: absolute error on Cerenkov angle, [radians]
353 
354  double phiDelta = phiC - trkPhi;
355 
356  double sint = TMath::Sin(trkTheta);
357  double cost = TMath::Cos(trkTheta);
358  double cosfd = TMath::Cos(phiDelta);
359  double tantheta = TMath::Tan(thetaC);
360 
361  double alpha = cost - tantheta * cosfd * sint; // formula (11)
362  double dtdn = cost * getRefIdx() * betaM * betaM / (alpha * tantheta); // formula (12)
363 
364  // double f = 0.00928*(7.75-5.635)/TMath::Sqrt(12.);
365  double f = 0.0172 * (7.75 - 5.635) / TMath::Sqrt(24.);
366 
367  return f * dtdn;
368 } //SigCrom()
369 //+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
370 double Param::sigGeom(double trkTheta, double trkPhi, double thetaC, double phiC, double betaM)
371 {
372  // Analitical calculation of geometric error (due to lack of knowledge of creation point in radiator)
373  // on Cerenkov angle for a given Cerenkov photon
374  // created by a given MIP. Formulae according to CERN-EP-2000-058
375  // Arguments: Cerenkov and azimuthal angles for Cerenkov photon, [radians]
376  // dip and azimuthal angles for MIP taken at the entrance to radiator, [radians]
377  // MIP beta
378  // Returns: absolute error on Cerenkov angle, [radians]
379 
380  double phiDelta = phiC - trkPhi;
381 
382  double sint = TMath::Sin(trkTheta);
383  double cost = TMath::Cos(trkTheta);
384  double sinf = TMath::Sin(trkPhi);
385  double cosfd = TMath::Cos(phiDelta);
386  double costheta = TMath::Cos(thetaC);
387  double tantheta = TMath::Tan(thetaC);
388 
389  double alpha = cost - tantheta * cosfd * sint; // formula (11)
390 
391  double k = 1. - getRefIdx() * getRefIdx() + alpha * alpha / (betaM * betaM); // formula (after 8 in the text)
392  if (k < 0) {
393  return 1e10;
394  }
395 
396  double eTr = 0.5 * radThick() * betaM * TMath::Sqrt(k) / (gapThick() * alpha); // formula (14)
397  double lambda = (1. - sint * sinf) * (1. + sint * sinf); // formula (15)
398 
399  double c1 = 1. / (1. + eTr * k / (alpha * alpha * costheta * costheta)); // formula (13.a)
400  double c2 = betaM * TMath::Power(k, 1.5) * tantheta * lambda / (gapThick() * alpha * alpha); // formula (13.b)
401  double c3 = (1. + eTr * k * betaM * betaM) / ((1 + eTr) * alpha * alpha); // formula (13.c)
402  double c4 = TMath::Sqrt(k) * tantheta * (1 - lambda) / (gapThick() * betaM); // formula (13.d)
403  double dtdT = c1 * (c2 + c3 * c4);
404  double trErr = radThick() / (TMath::Sqrt(12.) * cost);
405 
406  return trErr * dtdT;
407 } //SigGeom()

484 void Param::lors2Pad(float x, float y, Int_t& pc, Int_t& px, Int_t& py)
485 {
486  // Check the pad of given position
487  // Arguments: x,y- position [cm] in LORS; pc,px,py- pad where to store the result
488  // Returns: none
489  pc = px = py = -1;
490  if (x > fgkMinPcX[0] && x < fgkMaxPcX[0]) {
491  pc = 0;
492  px = Int_t(x / sizePadX());
493  } //PC 0 or 2 or 4
494  else if (x > fgkMinPcX[1] && x < fgkMaxPcX[1]) {
495  pc = 1;
496  px = Int_t((x - fgkMinPcX[1]) / sizePadX());
497  } //PC 1 or 3 or 5
498  else {
499  return;
500  }
501  if (y > fgkMinPcY[0] && y < fgkMaxPcY[0]) {
