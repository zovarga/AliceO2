54 void DigitizerSpec::run(framework::ProcessingContext& ctx)
55 {
56  if (mFinished) {
57  return;
58  }
59 
60  // read collision context from input
61  auto context = ctx.inputs().get<o2::steer::DigitizationContext*>("collisioncontext");
62  context->initSimChains(o2::detectors::DetID::EMC, mSimChains);
63  auto& timesview = context->getEventRecords();
64  LOG(debug) << "GOT " << timesview.size() << " COLLISSION TIMES";
65 
66  // if there is nothing to do ... return
67  if (timesview.size() == 0) {
68  return;
69  }
70 
71  TStopwatch timer;
72  timer.Start();
73 
74  LOG(info) << " CALLING EMCAL DIGITIZATION ";
75  o2::dataformats::MCTruthContainer<o2::emcal::MCLabel> labelAccum;
76  std::vector<TriggerRecord> triggers;
77 
78  auto& eventParts = context->getEventParts();
79  mDigitizer.clear();
80  mAccumulatedDigits.clear();
81  int trigID = 0;
82  int indexStart = mAccumulatedDigits.size();
83 
84  bool preTriggerColl = false;
85  // loop over all composite collisions given from context
86  // (aka loop over all the interaction records)
87  for (int collID = 0; collID < timesview.size(); ++collID) {
88 
89  mDigitizer.setEventTime(timesview[collID].getTimeNS());
90 
91  if (!mDigitizer.isEmpty() && (o2::emcal::SimParam::Instance().isDisablePileup() || !mDigitizer.isLive()) && !preTriggerColl) {
92  // copy digits into accumulator
93  mDigits.clear();
94  mLabels.clear();
95  mDigitizer.fillOutputContainer(mDigits, mLabels);
96  std::copy(mDigits.begin(), mDigits.end(), std::back_inserter(mAccumulatedDigits));
97  labelAccum.mergeAtBack(mLabels);
98  LOG(info) << "Have " << mAccumulatedDigits.size() << " digits ";
99  triggers.emplace_back(timesview[trigID], o2::trigger::PhT, indexStart, mDigits.size());
100  indexStart = mAccumulatedDigits.size();
101  mDigits.clear();
102  mLabels.clear();
103  }
104 
105  if (!mDigitizer.isLive()) {
106  // Take collisions that occurs 600 ns before the readout window is open
107  if (mDigitizer.preTriggerCollision()) {
108  for (auto& part : eventParts[collID]) {
109  mSumDigitizer.setCurrEvID(part.entryID);
110  mSumDigitizer.setCurrSrcID(part.sourceID);
111 
112  // get the hits for this event and this source
113  mHits.clear();
114  context->retrieveHits(mSimChains, "EMCHit", part.sourceID, part.entryID, &mHits);
115 
116  LOG(info) << "For collision " << collID << " eventID " << part.entryID << " found " << mHits.size() << " hits ";
117 
118  std::vector<o2::emcal::LabeledDigit> summedDigits = mSumDigitizer.process(mHits);
119 
120  // call actual digitization procedure
121  mDigitizer.process(summedDigits);
122 
123  preTriggerColl = true;
124  }
125  }
126  continue;
127  } else {
128  preTriggerColl = false;
129  }
130 
131  if (mDigitizer.isEmpty()) {
132  mDigitizer.initCycle();
133  trigID = collID;
134  }
135 
136  // for each collision, loop over the constituents event and source IDs
137  // (background signal merging is basically taking place here)
138  for (auto& part : eventParts[collID]) {
139 
140  mSumDigitizer.setCurrEvID(part.entryID);
141  mSumDigitizer.setCurrSrcID(part.sourceID);
142 
143  // get the hits for this event and this source
144  mHits.clear();
145  context->retrieveHits(mSimChains, "EMCHit", part.sourceID, part.entryID, &mHits);
146 
147  LOG(info) << "For collision " << collID << " eventID " << part.entryID << " found " << mHits.size() << " hits ";
148 
149  std::vector<o2::emcal::LabeledDigit> summedDigits = mSumDigitizer.process(mHits);
150 
151  // call actual digitization procedure
152  mDigitizer.process(summedDigits);
153  }
154  }
155 
156  if (!mDigitizer.isEmpty()) {
157  // copy digits into accumulator
158  mDigits.clear();
159  mLabels.clear();
160  mDigitizer.fillOutputContainer(mDigits, mLabels);
161  std::copy(mDigits.begin(), mDigits.end(), std::back_inserter(mAccumulatedDigits));
162  labelAccum.mergeAtBack(mLabels);
163  LOG(info) << "Have " << mAccumulatedDigits.size() << " digits ";
164  triggers.emplace_back(timesview[trigID], o2::trigger::PhT, indexStart, mDigits.size());
165  indexStart = mAccumulatedDigits.size();
166  mDigits.clear();
167  mLabels.clear();
168  }
169 
170  LOG(info) << "Have " << labelAccum.getNElements() << " EMCAL labels ";
171  // here we have all digits and we can send them to consumer (aka snapshot it onto output)
172  ctx.outputs().snapshot(Output{"EMC", "DIGITS", 0, Lifetime::Timeframe}, mAccumulatedDigits);
173  ctx.outputs().snapshot(Output{"EMC", "TRGRDIG", 0, Lifetime::Timeframe}, triggers);
174  if (ctx.outputs().isAllowed({"EMC", "DIGITSMCTR", 0})) {
175  ctx.outputs().snapshot(Output{"EMC", "DIGITSMCTR", 0, Lifetime::Timeframe}, labelAccum);
176  }
177  // EMCAL is always a triggering detector
178  const o2::parameters::GRPObject::ROMode roMode = o2::parameters::GRPObject::TRIGGERING;
179  LOG(info) << "EMCAL: Sending ROMode= " << roMode << " to GRPUpdater";
180  ctx.outputs().snapshot(Output{"EMC", "ROMode", 0, Lifetime::Timeframe}, roMode);
181 
182  timer.Stop();
183  LOG(info) << "Digitization took " << timer.CpuTime() << "s";
184  // we should be only called once; tell DPL that this process is ready to exit
185  ctx.services().get<ControlService>().readyToQuit(QuitRequest::Me);
186  mFinished = true;
187 }

189 o2::framework::DataProcessorSpec getEMCALDigitizerSpec(int channel, bool mctruth)
190 {
191  // create the full data processor spec using
192  // a name identifier
193  // input description
194  // algorithmic description (here a lambda getting called once to setup the actual processing function)
195  // options that can be used for this processor (here: input file names where to take the hits)
196  std::vector<OutputSpec> outputs;
197  outputs.emplace_back("EMC", "DIGITS", 0, Lifetime::Timeframe);
198  outputs.emplace_back("EMC", "TRGRDIG", 0, Lifetime::Timeframe);
199  if (mctruth) {
200  outputs.emplace_back("EMC", "DIGITSMCTR", 0, Lifetime::Timeframe);
201  }
202  outputs.emplace_back("EMC", "ROMode", 0, Lifetime::Timeframe);
203 
204  return DataProcessorSpec{
205  "EMCALDigitizer", Inputs{InputSpec{"collisioncontext", "SIM", "COLLISIONCONTEXT", static_cast<SubSpecificationType>(channel), Lifetime::Timeframe}},
206  outputs,
207  AlgorithmSpec{o2::framework::adaptFromTask<DigitizerSpec>()},
208  Options{{"pileup", VariantType::Int, 1, {"whether to run in continuous time mode"}}}
209  // I can't use VariantType::Bool as it seems to have a problem
210  };
