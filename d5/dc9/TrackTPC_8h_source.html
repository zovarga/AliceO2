82  uint8_t& sectorIndex, uint8_t& rowIndex, uint32_t& clusterIndex, const ClusRef& ref)
83  {
84  // data for given tracks starts at clinfo[ ref.getFirstEntry() ],
85  // 1st ref.getEntries() cluster indices are stored as uint32_t
