65 WorkflowSpec defineDataProcessing(ConfigContext const& config)
66 {
67  using namespace o2::tpc;
68  o2::conf::ConfigurableParam::updateFromString(config.options().get<std::string>("configKeyValues"));
69  const auto tpcCRUs = o2::RangeTokenizer::tokenize<int>(config.options().get<std::string>("crus"));
70  const auto nCRUs = tpcCRUs.size();
71  const auto nLanes = std::min(static_cast<unsigned long>(config.options().get<int>("lanes")), nCRUs);
72  const auto time_lanes = static_cast<unsigned int>(config.options().get<int>("time-lanes"));
73  const auto crusPerLane = nCRUs / nLanes + ((nCRUs % nLanes) != 0);
74  const auto debug = config.options().get<bool>("debug");
75  const auto propagateIDCs = config.options().get<bool>("propagateIDCs");
76  const auto nthreads = static_cast<unsigned long>(config.options().get<int>("nthreads"));
77  IDCAverageGroup<IDCAverageGroupCRU>::setNThreads(nthreads);
78  const auto rangeIDC = static_cast<unsigned int>(config.options().get<int>("rangeIDC"));
79  const auto minIDCsPerTF = static_cast<unsigned int>(config.options().get<int>("minIDCsPerTF"));
80  propagateIDCs ? TPCFLPIDCDevice<TPCFLPIDCDeviceNoGroup>::setMinIDCsPerTF(minIDCsPerTF) : TPCFLPIDCDevice<TPCFLPIDCDeviceGroup>::setMinIDCsPerTF(minIDCsPerTF);
81  const auto loadFromFile = config.options().get<bool>("load-from-file");
82 
83  const std::string sgroupPads = config.options().get<std::string>("groupPads");
84  const std::string sgroupRows = config.options().get<std::string>("groupRows");
85  const std::string sgroupLastRowsThreshold = config.options().get<std::string>("groupLastRowsThreshold");
86  const std::string sgroupLastPadsThreshold = config.options().get<std::string>("groupLastPadsThreshold");
87  ParameterIDCGroup::setGroupingParameterFromString(sgroupPads, sgroupRows, sgroupLastRowsThreshold, sgroupLastPadsThreshold);
88 
89  // set up configuration
90  o2::conf::ConfigurableParam::updateFromFile(config.options().get<std::string>("configFile"));
91  o2::conf::ConfigurableParam::writeINI("o2tpcaveragegroupidc_configuration.ini");
92 
93  WorkflowSpec workflow;
94  if (nLanes <= 0) {
95  return workflow;
96  }
97 
98  for (int ilane = 0; ilane < nLanes; ++ilane) {
99  const auto first = tpcCRUs.begin() + ilane * crusPerLane;
100  if (first >= tpcCRUs.end()) {
101  break;
102  }
103  const auto last = std::min(tpcCRUs.end(), first + crusPerLane);
104  const std::vector<uint32_t> rangeCRUs(first, last);
105  propagateIDCs ? workflow.emplace_back(timePipeline(getTPCFLPIDCSpec<TPCFLPIDCDeviceNoGroup>(ilane, rangeCRUs, rangeIDC, debug, loadFromFile), time_lanes)) : workflow.emplace_back(timePipeline(getTPCFLPIDCSpec<TPCFLPIDCDeviceGroup>(ilane, rangeCRUs, rangeIDC, debug, loadFromFile), time_lanes));
106  }
107 
108  return workflow;
109 }
