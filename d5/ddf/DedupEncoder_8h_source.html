56 template <typename coder_T, typename stream_T, typename source_T>
57 template <typename stream_IT, typename source_IT, std::enable_if_t<internal::isCompatibleIter_v<source_T, source_IT>, bool>>
58 stream_IT DedupEncoder<coder_T, stream_T, source_T>::process(source_IT inputBegin, source_IT inputEnd, stream_IT outputBegin, duplicatesMap_t& duplicates) const
59 {
60  using namespace internal;
61  LOG(trace) << "start encoding";
62  RANSTimer t;
63  t.start();
64 
65  if (inputBegin == inputEnd) {
66  LOG(warning) << "passed empty message to encoder, skip encoding";
67  return outputBegin;
68  }
69  ransCoder_t rans{this->mSymbolTable.getPrecision()};
70 
71  stream_IT outputIter = outputBegin;
72  source_IT inputIT = inputEnd;
73 
74  const auto inputBufferSize = std::distance(inputBegin, inputEnd);
75 
76  auto encode = [&inputBegin, &duplicates, this](source_IT symbolIter, stream_IT outputIter, ransCoder_t& coder) {
77  const source_T symbol = *symbolIter;
78  const auto& encoderSymbol = (this->mSymbolTable)[symbol];
79 
80  // dedup step:
81  auto dedupIT = symbolIter;
82  // advance on source by one.
83  --dedupIT;
84  size_t numDuplicates = 0;
85 
86  // find out how many duplicates we have
87  while (*dedupIT == symbol) {
88  --dedupIT;
89  ++numDuplicates;
90  }
91 
92  // if we have a duplicate treat it.
93  if (numDuplicates > 0) {
94  const auto pos = std::distance(inputBegin, symbolIter) - 1;
95  LOG(trace) << "pos[" << pos << "]: found " << numDuplicates << " duplicates of symbol " << (char)symbol;
96  duplicates.emplace(pos, numDuplicates);
97  }
98 
99  return std::pair(++dedupIT, coder.putSymbol(outputIter, encoderSymbol));
100  };
101 
102  while (inputIT != inputBegin) { // NB: working in reverse!
103  std::tie(inputIT, outputIter) = encode(--inputIT, outputIter, rans);
104  }
105  outputIter = rans.flush(outputIter);
106  // first iterator past the range so that sizes, distances and iterators work correctly.
107  ++outputIter;
108 
109  t.stop();
110  LOG(debug1) << "Encoder::" << __func__ << " {ProcessedBytes: " << inputBufferSize * sizeof(source_T) << ","
111  << " inclusiveTimeMS: " << t.getDurationMS() << ","
112  << " BandwidthMiBPS: " << std::fixed << std::setprecision(2) << (inputBufferSize * sizeof(source_T) * 1.0) / (t.getDurationS() * 1.0 * (1 << 20)) << "}";
113 
114 // advanced diagnostics for debug builds
115 #if !defined(NDEBUG)
116 
117  const auto inputBufferSizeB = inputBufferSize * sizeof(source_T);
118 
119  LOG(debug2)
120  << "EncoderProperties: {"
121  << "sourceTypeB: " << sizeof(source_T) << ", "
122  << "streamTypeB: " << sizeof(stream_T) << ", "
123  << "coderTypeB: " << sizeof(coder_T) << ", "
124  << "symbolTablePrecision: " << this->mSymbolTable.getPrecision() << ", "
125  << "inputBufferSizeB: " << inputBufferSizeB << "}";
126 #endif
127 
128  LOG(trace) << "done encoding";
129 
130  return outputIter;
131 };
