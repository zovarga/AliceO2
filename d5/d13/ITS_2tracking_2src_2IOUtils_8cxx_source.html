53 void ioutils::convertCompactClusters(gsl::span<const itsmft::CompClusterExt> clusters,
54  gsl::span<const unsigned char>::iterator& pattIt,
55  std::vector<o2::BaseCluster<float>>& output,
56  const itsmft::TopologyDictionary& dict)
57 {
58  GeometryTGeo* geom = GeometryTGeo::Instance();
59  bool applyMisalignment = false;
60  const auto& conf = TrackerParamConfig::Instance();
61  const auto& chmap = getChipMappingITS();
62  for (int il = 0; il < chmap.NLayers; il++) {
63  if (conf.sysErrY2[il] > 0.f || conf.sysErrZ2[il] > 0.f) {
64  applyMisalignment = true;
65  break;
66  }
67  }
68 
69  for (auto& c : clusters) {
70  auto pattID = c.getPatternID();
71  o2::math_utils::Point3D<float> locXYZ;
72  float sigmaY2 = ioutils::DefClusError2Row, sigmaZ2 = ioutils::DefClusError2Col, sigmaYZ = 0; //Dummy COG errors (about half pixel size)
73  if (pattID != itsmft::CompCluster::InvalidPatternID) {
74  sigmaY2 = dict.getErr2X(pattID);
75  sigmaZ2 = dict.getErr2Z(pattID);
76  if (!dict.isGroup(pattID)) {
77  locXYZ = dict.getClusterCoordinates(c);
78  } else {
79  o2::itsmft::ClusterPattern patt(pattIt);
80  locXYZ = dict.getClusterCoordinates(c, patt);
81  }
82  } else {
83  o2::itsmft::ClusterPattern patt(pattIt);
84  locXYZ = dict.getClusterCoordinates(c, patt, false);
85  }
86  auto& cl3d = output.emplace_back(c.getSensorID(), geom->getMatrixT2L(c.getSensorID()) ^ locXYZ); // local --> tracking
87  if (applyMisalignment) {
88  auto lrID = chmap.getLayer(c.getSensorID());
89  sigmaY2 += conf.sysErrY2[lrID];
90  sigmaZ2 += conf.sysErrZ2[lrID];
91  }
92  cl3d.setErrors(sigmaY2, sigmaZ2, sigmaYZ);
93  }
94 }

96 void ioutils::loadEventData(ROframe& event, gsl::span<const itsmft::CompClusterExt> clusters,
97  gsl::span<const unsigned char>::iterator& pattIt, const itsmft::TopologyDictionary& dict,
98  const dataformats::MCTruthContainer<MCCompLabel>* clsLabels)
99 {

106  geom->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L, o2::math_utils::TransformType::L2G));
107  int clusterId{0};
108 
109  for (auto& c : clusters) {
110  int layer = geom->getLayer(c.getSensorID());
111 
112  auto pattID = c.getPatternID();
113  o2::math_utils::Point3D<float> locXYZ;
114  float sigmaY2 = ioutils::DefClusError2Row, sigmaZ2 = ioutils::DefClusError2Col, sigmaYZ = 0; //Dummy COG errors (about half pixel size)
115  if (pattID != itsmft::CompCluster::InvalidPatternID) {
116  sigmaY2 = dict.getErr2X(pattID);
117  sigmaZ2 = dict.getErr2Z(pattID);
118  if (!dict.isGroup(pattID)) {
119  locXYZ = dict.getClusterCoordinates(c);
120  } else {
121  o2::itsmft::ClusterPattern patt(pattIt);
122  locXYZ = dict.getClusterCoordinates(c, patt);
123  }
124  } else {
125  o2::itsmft::ClusterPattern patt(pattIt);
126  locXYZ = dict.getClusterCoordinates(c, patt, false);
127  }
128  auto sensorID = c.getSensorID();
129  // Inverse transformation to the local --> tracking
130  auto trkXYZ = geom->getMatrixT2L(sensorID) ^ locXYZ;
131  // Transformation to the local --> global
132  auto gloXYZ = geom->getMatrixL2G(sensorID) * locXYZ;
133 
134  event.addTrackingFrameInfoToLayer(layer, gloXYZ.x(), gloXYZ.y(), gloXYZ.z(), trkXYZ.x(), geom->getSensorRefAlpha(sensorID),
135  std::array<float, 2>{trkXYZ.y(), trkXYZ.z()},
136  std::array<float, 3>{sigmaY2, sigmaYZ, sigmaZ2});
137 
139  event.addClusterToLayer(layer, gloXYZ.x(), gloXYZ.y(), gloXYZ.z(), event.getClustersOnLayer(layer).size());
140  if (clsLabels) {
141  // event.addClusterLabelToLayer(layer, *(clsLabels->getLabels(clusterId).begin()));
142  event.setMClabelsContainer(clsLabels);
143  }
144  event.addClusterExternalIndexToLayer(layer, clusterId);
145  clusterId++;
146  }

200 std::vector<std::unordered_map<int, Label>> ioutils::loadLabels(const int eventsNum, const std::string& fileName)
201 {
202  std::vector<std::unordered_map<int, Label>> labelsMap{};
203  std::unordered_map<int, Label> currentEventLabelsMap{};
204  std::ifstream inputStream{};
205  std::string line{};
206  int monteCarloId{}, pdgCode{}, numberOfClusters{};
207  float transverseMomentum{}, phi{}, pseudorapidity{};
208 
209  labelsMap.reserve(eventsNum);
210 
211  inputStream.open(fileName);
212  std::getline(inputStream, line);
213 
214  while (std::getline(inputStream, line)) {
215 
216  std::istringstream inputStringStream(line);
217 
218  if (inputStringStream >> monteCarloId) {
219 
220  if (monteCarloId == EventLabelsSeparator) {
221 
222  labelsMap.emplace_back(currentEventLabelsMap);
223  currentEventLabelsMap.clear();
224 
225  } else {
226 
227  if (inputStringStream >> transverseMomentum >> phi >> pseudorapidity >> pdgCode >> numberOfClusters) {
228 
229  if (std::abs(pdgCode) == constants::pdgcodes::PionCode && numberOfClusters == 7) {
230 
231  currentEventLabelsMap.emplace(std::piecewise_construct, std::forward_as_tuple(monteCarloId),
232  std::forward_as_tuple(monteCarloId, transverseMomentum, phi,
233  pseudorapidity, pdgCode, numberOfClusters));
234  }
235  }
236  }
237  }
238  }
239 
240  labelsMap.emplace_back(currentEventLabelsMap);
241 
242  return labelsMap;
243 }

246  std::ofstream& fakeRoadsOutputStream, const std::vector<std::vector<Road>>& roads,
247  const std::unordered_map<int, Label>& labelsMap)

249  const int numVertices{static_cast<int>(roads.size())};

251 
252  correctRoadsOutputStream << EventLabelsSeparator << std::endl;
253  fakeRoadsOutputStream << EventLabelsSeparator << std::endl;
254 
255  for (int iVertex{0}; iVertex < numVertices; ++iVertex) {
256 
257  const std::vector<Road>& currentVertexRoads{roads[iVertex]};
258  const int numRoads{static_cast<int>(currentVertexRoads.size())};
259 
260  for (int iRoad{0}; iRoad < numRoads; ++iRoad) {
261 
262  const Road& currentRoad{currentVertexRoads[iRoad]};
263  const int currentRoadLabel{currentRoad.getLabel()};
264 
265  if (!labelsMap.count(currentRoadLabel)) {
266 
267  continue;
268  }
269 
270  const Label& currentLabel{labelsMap.at(currentRoadLabel)};
271 
272  if (currentRoad.isFakeRoad()) {
273 
274  fakeRoadsOutputStream << currentLabel << std::endl;
275 
276  } else {
277 
278  if (foundMonteCarloIds.count(currentLabel.monteCarloId)) {
279 
280  duplicateRoadsOutputStream << currentLabel << std::endl;
281 
282  } else {
283 
284  correctRoadsOutputStream << currentLabel << std::endl;
285  foundMonteCarloIds.emplace(currentLabel.monteCarloId);
286  }
287  }
288  }
289  }
