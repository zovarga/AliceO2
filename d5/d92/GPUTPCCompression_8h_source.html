106 template <class T>
107 GPUdi() void GPUTPCCompression::truncateSignificantBits(T& v, unsigned int nBits, unsigned int max)
108 {
109  if (nBits == 0) {
110  return;
111  }
112 
113  unsigned int val = v;
114  unsigned int ldz = sizeof(unsigned int) * 8 - CAMath::Clz(val);
115  if (val && ldz > nBits) {
116  if (val & (1 << (ldz - nBits - 1))) {
117  val += (1 << (ldz - nBits - 1));
118  ldz = sizeof(unsigned int) * 8 - CAMath::Clz(val);
119  }
120  val &= ((1 << ldz) - 1) ^ ((1 << (ldz - nBits)) - 1);
121  if (val >= max) {
122  val = max - 1;
123  }
124  // GPUInfo("CHANGING X %x --> %x", (unsigned int) v, val);
125  v = val;
126  }
127 }
