97 std::string VisualisationEvent::toJson()
98 {
99  Document tree(kObjectType);
100  Document::AllocatorType& allocator = tree.GetAllocator();
101 
102  // compatibility verification
103  tree.AddMember("fileVersion", rapidjson::Value().SetInt(JSON_FILE_VERSION), allocator);
104  tree.AddMember("runNumber", rapidjson::Value().SetInt(this->mRunNumber), allocator);
105  tree.AddMember("collisionTime", rapidjson::Value().SetString(this->mCollisionTime.c_str(), this->mCollisionTime.size()), allocator);
106  tree.AddMember("workflowVersion", rapidjson::Value().SetFloat(this->mWorkflowVersion), allocator);
107  tree.AddMember("workflowParameters", rapidjson::Value().SetString(this->mWorkflowParameters.c_str(), this->mWorkflowParameters.size()), allocator);
108  // Tracks
109  tree.AddMember("trackCount", rapidjson::Value().SetInt(this->getTrackCount()), allocator);
110 
111  Value jsonTracks(kArrayType);
112  for (size_t i = 0; i < this->getTrackCount(); i++) {
113  jsonTracks.PushBack(this->mTracks[i].jsonTree(allocator), allocator);
114  }
115  tree.AddMember("mTracks", jsonTracks, allocator);
116 
117  // Clusters
118  rapidjson::Value clusterCount(rapidjson::kNumberType);
119  clusterCount.SetInt(this->getClusterCount());
120  tree.AddMember("clusterCount", clusterCount, allocator);
121  Value jsonClusters(kArrayType);
122  for (size_t i = 0; i < this->getClusterCount(); i++) {
123  jsonClusters.PushBack(this->mClusters[i].jsonTree(allocator), allocator);
124  }
125  tree.AddMember("mClusters", jsonClusters, allocator);
126 
127  // stringify
128  rapidjson::StringBuffer buffer;
129  rapidjson::PrettyWriter<rapidjson::StringBuffer> writer(buffer);
130  tree.Accept(writer);
131  std::string json_str = std::string(buffer.GetString(), buffer.GetSize());
132  return json_str;
133 }

154 void VisualisationEvent::fromJson(std::string json)
155 {
156  mTracks.clear();
157  mClusters.clear();
158 
159  rapidjson::Document tree;
160  tree.Parse(json.c_str());
161 
162  auto version = 1;
163  if (tree.HasMember("fileVersion")) {
164  rapidjson::Value& jsonFileVersion = tree["fileVersion"];
165  version = jsonFileVersion.GetInt();
166  }
167 
168  o2::header::DataHeader::RunNumberType runNumber = 0;
169  if (tree.HasMember("runNumber")) {
170  rapidjson::Value& jsonRunNumber = tree["runNumber"];
171  runNumber = jsonRunNumber.GetInt();
172  }
173  this->setRunNumber(runNumber);
174 
175  auto collisionTime = "not specified";
176  if (tree.HasMember("collisionTime")) {
177  rapidjson::Value& jsonCollisionTime = tree["collisionTime"];
178  collisionTime = jsonCollisionTime.GetString();
179  }
180  this->setCollisionTime(collisionTime);
181 
182  rapidjson::Value& trackCount = tree["trackCount"];
183  this->mTracks.reserve(trackCount.GetInt());
184  rapidjson::Value& jsonTracks = tree["mTracks"];
185  for (auto& v : jsonTracks.GetArray()) {
186  mTracks.emplace_back(v);
187  }
188 
189  rapidjson::Value& clusterCount = tree["clusterCount"];
190  this->mClusters.reserve(clusterCount.GetInt());
191  rapidjson::Value& jsonClusters = tree["mClusters"];
192  for (auto& v : jsonClusters.GetArray()) {
193  mClusters.emplace_back(v);
194  }
195 }
