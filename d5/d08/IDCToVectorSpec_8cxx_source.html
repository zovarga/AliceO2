78  void run(o2::framework::ProcessingContext& pc) final
79  {
80  std::vector<InputSpec> filter = {{"check", ConcreteDataTypeMatcher{o2::header::gDataOriginTPC, "RAWDATA"}, Lifetime::Timeframe}}; // TODO: Change to IDC when changed in DD
81  const auto& mapper = Mapper::instance();
82 
83  uint32_t heartbeatOrbit = 0;
84  uint32_t heartbeatBC = 0;
85  uint32_t tfCounter = 0;
86  bool first = true;
87 
88  CalPad* pedestals = nullptr;
89  if (mNoisePedestal.size() && mNoisePedestal[0]) {
90  pedestals = mNoisePedestal[0].get();
91  }
92 
93  for (auto const& ref : InputRecordWalker(pc.inputs(), filter)) {
94  const auto* dh = DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
95  // ---| extract hardware information to do the processing |---
96  const auto feeId = (FEEIDType)dh->subSpecification;
97  const auto link = rdh_utils::getLink(feeId);
98  const uint32_t cruID = rdh_utils::getCRU(feeId);
99  const auto endPoint = rdh_utils::getEndPoint(feeId);
100  tfCounter = dh->tfCounter;
101 
102  // only select IDCs
103  // ToDo: cleanup once IDCs will be propagated not as RAWDATA, but IDC.
104  if (link != rdh_utils::IDCLinkID) {
105  continue;
106  }
107  LOGP(info, "IDC Processing firstTForbit {:9}, tfCounter {:5}, run {:6}, feeId {:6} ({:3}/{}/{:2})", dh->firstTForbit, dh->tfCounter, dh->runNumber, feeId, cruID, endPoint, link);
108 
109  if (std::find(mCRUs.begin(), mCRUs.end(), cruID) == mCRUs.end()) {
110  LOGP(error, "IDC CRU {:3} not configured in CRUs, skipping", cruID);
111  continue;
112  }
113 
114  const CRU cru(cruID);
115  const int sector = cru.sector();
116  const auto& partInfo = mapper.getPartitionInfo(cru.partition());
117  const int fecLinkOffsetCRU = (partInfo.getNumberOfFECs() + 1) / 2;
118  const int fecSectorOffset = partInfo.getSectorFECOffset();
119  const GlobalPadNumber regionPadOffset = Mapper::GLOBALPADOFFSET[cru.region()];
120  const GlobalPadNumber numberPads = Mapper::PADSPERREGION[cru.region()];
121  int sampaOnFEC{}, channelOnSAMPA{};
122  auto& idcVec = mIDCvectors[cruID];
123  auto& infoVec = mIDCInfos[cruID];
124 
125  // ---| data loop |---
126  const gsl::span<const char> raw = pc.inputs().get<gsl::span<char>>(ref);
127  o2::framework::RawParser parser(raw.data(), raw.size());
128  for (auto it = parser.begin(), end = parser.end(); it != end; ++it) {
129  const auto size = it.size();
130  assert(size == sizeof(idc::Container));
131  auto data = it.data();
132  auto& idcs = *((idc::Container*)(data));
133  const uint32_t orbit = idcs.header.heartbeatOrbit;
134  const uint32_t bc = idcs.header.heartbeatBC;
135  //LOGP(info, "IDC Procssing orbit/BC: {:9}/{:4}", orbit, bc);
136 
137  auto infoIt = std::find(infoVec.begin(), infoVec.end(), orbit);
138  if (!infoVec.size()) {
139  infoVec.emplace_back(orbit, bc);
140  infoIt = infoVec.end() - 1;
141  //} else if (!infoVec.back().matches(orbit, bc)) {
142  } else if (infoIt == infoVec.end()) {
143  auto& lastInfo = infoVec.back();
144  if ((orbit - lastInfo.heartbeatOrbit) != mNOrbitsIDC) {
145  LOGP(error, "received packet with invalid jump in idc orbit ({} - {} == {} != {})", orbit, lastInfo.heartbeatOrbit, orbit - lastInfo.heartbeatOrbit, mNOrbitsIDC);
146  }
147  infoVec.emplace_back(orbit, bc);
148  infoIt = infoVec.end() - 1;
149  }
150 
151  // check if end poit was already processed
152  auto& lastInfo = *infoIt;
153  if (lastInfo.wasEPseen(endPoint)) {
154  LOGP(info, "Already received another data packet for CRU {}, ep {}, orbit {}, bc {}", cruID, endPoint, orbit, bc);
155  continue;
156  }
157 
158  lastInfo.setEPseen(endPoint);
159  // idc value offset in present time frame
160  const size_t idcOffset = std::distance(infoVec.begin(), infoIt);
161 
162  // TODO: for debugging, remove later
163  /*
164  auto* rdhPtr = it.get_if<o2::header::RAWDataHeaderV6>();
165  const auto feeId2 = (FEEIDType)RDHUtils::getFEEID(*rdhPtr);
166  const auto link2 = rdh_utils::getLink(feeId2);
167  const uint32_t cruID2 = rdh_utils::getCRU(feeId2);
168  const auto endPoint2 = rdh_utils::getEndPoint(feeId2);
169  const auto detField = RDHUtils::getDetectorField(*rdhPtr);
170  LOGP(info, "processing IDCs for CRU {}, ep {}, feeId {:6} ({:3}/{}/{:2}), detField: {}, orbit {}, bc {}, idcOffset {}, idcVec size {}, epSeen {:02b}", cruID, endPoint, feeId2, cruID2, endPoint2, link2, detField, orbit, bc, idcOffset, idcVec.size(), lastInfo.epSeen);
171  */
172 
173  const float norm = 1. / float(mTimeStampsPerIntegrationInterval);
174  for (uint32_t iLink = 0; iLink < idc::Links; ++iLink) {
175  if (!idcs.hasLink(iLink)) {
176  continue;
177  }
178  const int fecInSector = iLink + endPoint * fecLinkOffsetCRU + fecSectorOffset;
179 
180  for (uint32_t iChannel = 0; iChannel < idc::Channels; ++iChannel) {
181  auto val = idcs.getChannelValueFloat(iLink, iChannel);
182  Mapper::getSampaAndChannelOnFEC(cruID, iChannel, sampaOnFEC, channelOnSAMPA);
183  const GlobalPadNumber padInSector = mapper.globalPadNumber(fecInSector, sampaOnFEC, channelOnSAMPA);
184  if (pedestals) {
185  val -= pedestals->getValue(sector, padInSector) * mTimeStampsPerIntegrationInterval;
186  val *= norm;
187  }
188  const GlobalPadNumber padInRegion = padInSector - regionPadOffset;
189  const GlobalPadNumber vectorPosition = padInRegion + idcOffset * numberPads;
190  // TODO: for debugging, remove later
191  //auto rawVal = idcs.getChannelValue(iLink, iChannel);
192  //auto rawValF = idcs.getChannelValueFloat(iLink, iChannel);
193  //LOGP(info, "filling channel {}, link {}, fecLinkOffsetCRU {:2}, fecSectorOffset {:3}, fecInSector {:3}, idcVec[{} ({})] = {} ({} / {})", iChannel, iLink, fecLinkOffsetCRU, fecSectorOffset, fecInSector, vectorPosition, padInRegion, val, rawVal, rawValF);
194  idcVec[vectorPosition] = val;
195  }
196  }
197  }
198  }
199 
200  if (mDebugStream) {
201  writeDebugOutput(tfCounter);
202  }
203  snapshotIDCs(pc.outputs());
204  }

217  struct IDCInfo {
218  IDCInfo() = default;
219  IDCInfo(const IDCInfo&) = default;
220  IDCInfo(uint32_t orbit, uint16_t bc) : heartbeatOrbit(orbit), heartbeatBC(bc) {}
221 
222  uint32_t heartbeatOrbit{0};
223  uint16_t heartbeatBC{0};
224  uint16_t epSeen{0};
225 
226  bool operator==(const uint32_t orbit) const { return (heartbeatOrbit == orbit); }
227  bool operator==(const IDCInfo& inf) const { return (inf.heartbeatOrbit == heartbeatOrbit) && (inf.heartbeatBC == heartbeatBC) && (inf.epSeen == epSeen); }
228  void setEPseen(uint32_t ep) { epSeen |= uint16_t(1 << ep); }
229  bool wasEPseen(uint32_t ep) const { return epSeen & uint16_t(1 << ep); }
230  bool matches(uint32_t orbit, int16_t bc) const { return ((heartbeatOrbit == orbit) && (heartbeatBC == bc)); }
231  bool hasBothEPs() const { return epSeen == 3; }
232  };

244  void snapshotIDCs(DataAllocator& output)
245  {
246  LOGP(info, "snapshotIDCs");
247 
248  // check integrety of data between CRUs
249  size_t orbitsInTF = 0;
250  std::vector<IDCInfo> const* infVecComp = nullptr;
251  std::vector<uint64_t> orbitBCInfo;
252 
253  for (const auto& [cru, infVec] : mIDCInfos) {
254 
255  for (const auto& inf : infVec) {
256  if (!inf.hasBothEPs()) {
257  LOGP(fatal, "IDC CRU {:3}: data missing at ({:8}, {:4}) for one or both end points {:02b}", cru, inf.heartbeatOrbit, inf.heartbeatBC, inf.epSeen);
258  }
259  }
260 
261  if (!infVecComp) {
262  infVecComp = &infVec;
263  orbitsInTF = infVec.size();
264  std::for_each(infVec.begin(), infVec.end(), [&orbitBCInfo](const auto& inf) { orbitBCInfo.emplace_back((uint64_t(inf.heartbeatOrbit) << 32) + uint64_t(inf.heartbeatBC)); });
265  continue;
266  }
267 
268  if (orbitsInTF != infVec.size()) {
269  LOGP(fatal, "IDC CRU {:3}: unequal number of IDC values {} != {}", cru, orbitsInTF, infVec.size());
270  }
271 
272  if (!std::equal(infVecComp->begin(), infVecComp->end(), infVec.begin())) {
273  LOGP(fatal, "IDC CRU {:3}: mismatch in orbits");
274  }
275  }
276 
277  // send data
278  for (auto& [cru, idcVec] : mIDCvectors) {
279  idcVec.resize(Mapper::PADSPERREGION[CRU(cru).region()] * orbitsInTF);
280  const header::DataHeader::SubSpecificationType subSpec{cru << 7};
281  output.snapshot(Output{gDataOriginTPC, "IDCVECTOR", subSpec}, idcVec);
282  output.snapshot(Output{gDataOriginTPC, "IDCORBITS", subSpec}, orbitBCInfo);
283  }
284 
285  // clear output
286  initIDC();
287  }

289  //____________________________________________________________________________
290  void initIDC()
291  {
292  for (const auto cruID : mCRUs) {
293  const CRU cru(cruID);
294  const GlobalPadNumber numberPads = Mapper::PADSPERREGION[cru.region()] * mMaxIDCPerTF;
295  auto& idcVec = mIDCvectors[cruID];
296  idcVec.resize(numberPads);
297  std::fill(idcVec.begin(), idcVec.end(), -1.f);
298 
299  auto& infosCRU = mIDCInfos[cruID];
300  infosCRU.clear();
301  }
302  }

305  void writeDebugOutput(uint32_t tfCounter)
306  {
307  const auto& mapper = Mapper::instance();
308 
309  mDebugStream->GetFile()->cd();
310  auto& stream = (*mDebugStream) << "idcs";
311  uint32_t seen = 0;
312  for (auto cru : mCRUs) {
313  if (mIDCInfos.find(cru) == mIDCInfos.end()) {
314  continue;
315  }
316  auto& infos = mIDCInfos[cru];
317  auto& idcVec = mIDCvectors[cru];
318 
319  for (int i = 0; i < infos.size(); ++i) {
320  auto& info = infos[i];
321 
322  auto idcFirst = idcVec.begin() + i * Mapper::PADSPERREGION[cru % Mapper::NREGIONS];
323  auto idcLast = idcFirst + Mapper::PADSPERREGION[cru % Mapper::NREGIONS];
324  std::vector<float> idcs(idcFirst, idcLast);
325  std::vector<short> cpad(idcs.size());
326  std::vector<short> row(idcs.size());
327  for (int ipad = 0; ipad < idcs.size(); ++ipad) {
328  const auto& padPos = mapper.padPos(ipad + Mapper::GLOBALPADOFFSET[cru % Mapper::NREGIONS]);
329  row[ipad] = (short)padPos.getRow();
330  const short pads = (short)mapper.getNumberOfPadsInRowSector(row[ipad]);
331  cpad[ipad] = (short)padPos.getPad() - pads / 2;
332  }
333  float mean = std::accumulate(idcs.begin(), idcs.end(), 0.f) / float(idcs.size());
334 
335  stream << "cru=" << cru
336  << "epSeen=" << info.epSeen
337  << "tfCounter=" << tfCounter
338  << "orbit=" << info.heartbeatOrbit
339  << "bc=" << info.heartbeatBC
340  << "idcs=" << idcs
341  << "cpad=" << cpad
342  << "row=" << row
343  << "idc_mean=" << mean
344  << "\n";
345  }
346  }
