38 void MeanVertexCalibrator::finalizeSlot(Slot& slot)
39 {
40  // Extract results for the single slot
41  o2::calibration::MeanVertexData* c = slot.getContainer();
42  LOG(info) << "Finalize slot " << slot.getTFStart() << " <= TF <= " << slot.getTFEnd() << " with "
43  << c->getEntries() << " entries";
44  mTmpMVobjDqTime.emplace_back(slot.getTFStart(), slot.getTFEnd());
45 
46  if (mUseFit) {
47  MeanVertexObject mvo;
48  // x coordinate
49  std::vector<float> fitValues;
50  float* array = &c->histoX[0];
51  double fitres = fitGaus(c->nbinsX, array, -(c->rangeX), c->rangeX, fitValues);
52  if (fitres >= 0) {
53  LOG(info) << "X: Fit result (of single Slot) => " << fitres << ". Mean = " << fitValues[1] << " Sigma = " << fitValues[2];
54  } else {
55  LOG(error) << "X: Fit failed with result = " << fitres;
56  }
57  mvo.setX(fitValues[1]);
58  mvo.setSigmaX(fitValues[2]);
59 
60  // y coordinate
61  array = &c->histoY[0];
62  fitres = fitGaus(c->nbinsY, array, -(c->rangeY), c->rangeY, fitValues);
63  if (fitres >= 0) {
64  LOG(info) << "Y: Fit result (of single Slot) => " << fitres << ". Mean = " << fitValues[1] << " Sigma = " << fitValues[2];
65  } else {
66  LOG(error) << "Y: Fit failed with result = " << fitres;
67  }
68  mvo.setY(fitValues[1]);
69  mvo.setSigmaY(fitValues[2]);
70 
71  // z coordinate
72  array = &c->histoZ[0];
73  fitres = fitGaus(c->nbinsZ, array, -(c->rangeZ), c->rangeZ, fitValues);
74  if (fitres >= 0) {
75  LOG(info) << "Z: Fit result (of single Slot) => " << fitres << ". Mean = " << fitValues[1] << " Sigma = " << fitValues[2];
76  } else {
77  LOG(error) << "Z: Fit failed with result = " << fitres;
78  }
79  mvo.setZ(fitValues[1]);
80  mvo.setSigmaZ(fitValues[2]);
81 
82  // now we add the object to the deque
83  mTmpMVobjDq.push_back(std::move(mvo));
84  } else {
85  mTmpMVdataDq.push_back(std::move(*c));
86  mSMAdata.merge(&mTmpMVdataDq.back());
87  if (mTmpMVobjDqTime.size() > mSMAslots) {
88  mSMAdata.subtract(&mTmpMVdataDq.front());
89  mTmpMVdataDq.pop_front();
90  }
91  }
92 
93  // output object
94  MeanVertexObject mvo;
95 
96  if (mUseFit) {
97  doSimpleMovingAverage(mTmpMVobjDq, mSMAMVobj);
98  } else {
99  // now we need to fit, on the merged data
100  LOG(debug) << "**** Printing content of SMA MVData object for x coordinate";
101  for (int i = 0; i < mSMAdata.nbinsX; i++) {
102  LOG(debug) << "i = " << i << ", content of histogram = " << mSMAdata.histoX[i];
103  }
104  std::vector<float> fitValues;
105  float* array = &mSMAdata.histoX[0];
106  double fitres = fitGaus(mSMAdata.nbinsX, array, -(mSMAdata.rangeX), mSMAdata.rangeX, fitValues);
107  if (fitres >= 0) {
108  LOG(info) << "X: Fit result (of merged Slots) => " << fitres << ". Mean = " << fitValues[1] << " Sigma = " << fitValues[2];
109  } else {
110  LOG(error) << "X: Fit failed with result = " << fitres;
111  }
112  mSMAMVobj.setX(fitValues[1]);
113  mSMAMVobj.setSigmaX(fitValues[2]);
114 
115  // y coordinate
116  array = &mSMAdata.histoY[0];
117  fitres = fitGaus(mSMAdata.nbinsY, array, -(mSMAdata.rangeY), mSMAdata.rangeY, fitValues);
118  if (fitres >= 0) {
119  LOG(info) << "Y: Fit result (of merged Slots) => " << fitres << ". Mean = " << fitValues[1] << " Sigma = " << fitValues[2];
120  } else {
121  LOG(error) << "Y: Fit failed with result = " << fitres;
122  }
123  mSMAMVobj.setY(fitValues[1]);
124  mSMAMVobj.setSigmaY(fitValues[2]);
125 
126  // z coordinate
127  array = &mSMAdata.histoZ[0];
128  fitres = fitGaus(mSMAdata.nbinsZ, array, -(mSMAdata.rangeZ), mSMAdata.rangeZ, fitValues);
129  if (fitres >= 0) {
130  LOG(info) << "Z: Fit result (of merged Slots) => " << fitres << ". Mean = " << fitValues[1] << " Sigma = " << fitValues[2];
131  } else {
132  LOG(error) << "Z: Fit failed with result = " << fitres;
133  }
134  mSMAMVobj.setZ(fitValues[1]);
135  mSMAMVobj.setSigmaZ(fitValues[2]);
136  }
137 
138  // TODO: the timestamp is now given with the TF index, but it will have
139  // to become an absolute time. This is true both for the lhc phase object itself
140  // and the CCDB entry
141  if (mTmpMVobjDqTime.size() > mSMAslots) {
142  mTmpMVobjDqTime.pop_front();
143  }
144  TFType startValidity = (mTmpMVobjDqTime.front().getMin() + mTmpMVobjDqTime.back().getMax()) / 2; // will be rounded to uint64_t
145  LOG(info) << "start validity = " << startValidity;
146  std::map<std::string, std::string> md;
147  auto clName = o2::utils::MemFileHelper::getClassName(mSMAMVobj);
148  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
149  mInfoVector.emplace_back("GLO/Calib/MeanVertex", clName, flName, md, startValidity, 99999999999999);
150  mMeanVertexVector.emplace_back(mSMAMVobj);
151 
152  slot.print();
153 }

156 void MeanVertexCalibrator::doSimpleMovingAverage(std::deque<float>& dq, float& sma)
157 {
158 
159  // doing simple moving average
160 
161  if (dq.size() <= mSMAslots) {
162  sma = std::accumulate(dq.begin(), dq.end(), 0.0) / dq.size();
163  //avg = (avg * (vect.size() - 1) + vect.back()) / vect.size();
164  return;
165  }
166 
167  // if the vector has size > mSMAslots, we calculate the SMA, and then we drop 1 element
168  // (note that it can have a size > mSMAslots only of 1 element!)
169  sma += (dq[dq.size() - 1] - dq[0]) / mSMAslots;
170  dq.pop_front();
171 
172  return;
173 }

176 void MeanVertexCalibrator::doSimpleMovingAverage(std::deque<MVObject>& dq, MVObject& sma)
177 {
178 
179  // doing simple moving average
180 
181  if (dq.size() <= mSMAslots) {
182  sma.setX((sma.getX() * (dq.size() - 1) + dq.back().getX()) / dq.size());
183  sma.setY((sma.getY() * (dq.size() - 1) + dq.back().getY()) / dq.size());
184  sma.setZ((sma.getZ() * (dq.size() - 1) + dq.back().getZ()) / dq.size());
185  sma.setSigmaX((sma.getSigmaX() * (dq.size() - 1) + dq.back().getSigmaX()) / dq.size());
186  sma.setSigmaY((sma.getSigmaY() * (dq.size() - 1) + dq.back().getSigmaY()) / dq.size());
187  sma.setSigmaZ((sma.getSigmaZ() * (dq.size() - 1) + dq.back().getSigmaZ()) / dq.size());
188  return;
189  }
190 
191  // if the vector has size > mSMAslots, we calculate the SMA, and then we drop 1 element
192  // (note that it can have a size > mSMAslots only of 1 element!)
193  sma.setX(sma.getX() + (dq[dq.size() - 1].getX() - dq[0].getX()) / mSMAslots);
194  sma.setY(sma.getY() + (dq[dq.size() - 1].getY() - dq[0].getY()) / mSMAslots);
195  sma.setZ(sma.getZ() + (dq[dq.size() - 1].getZ() - dq[0].getZ()) / mSMAslots);
196  sma.setSigmaX(sma.getSigmaX() + (dq[dq.size() - 1].getSigmaX() - dq[0].getSigmaX()) / mSMAslots);
197  sma.setSigmaY(sma.getSigmaY() + (dq[dq.size() - 1].getSigmaY() - dq[0].getSigmaY()) / mSMAslots);
198  sma.setSigmaZ(sma.getSigmaZ() + (dq[dq.size() - 1].getSigmaZ() - dq[0].getSigmaZ()) / mSMAslots);
199 
200  dq.pop_front();
201 
202  return;
203 }
