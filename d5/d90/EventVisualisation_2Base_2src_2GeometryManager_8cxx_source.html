42 TEveGeoShape* GeometryManager::getGeometryForDetector(string detectorName)
43 {
44  TEnv settings;
45  ConfigurationManager::getInstance().getConfig(settings);
46 
47  // read geometry path from config file
48  string geomPath = settings.GetValue("simple.geom.R3.path", "");
49 
50  // load ROOT file with geometry
51  TFile* f = TFile::Open(Form("%s/simple_geom_%s.root", geomPath.c_str(), detectorName.c_str()));
52  if (!f) {
53  LOG(error) << "GeometryManager::GetSimpleGeom -- no file with geometry found for: " << detectorName << "!";
54  return nullptr;
55  }
56  LOG(info) << "GeometryManager::GetSimpleGeom for: " << detectorName << " from " << Form("%s/simple_geom_%s.root", geomPath.c_str(), detectorName.c_str());
57 
58  TEveGeoShapeExtract* geomShapreExtract = static_cast<TEveGeoShapeExtract*>(f->Get(detectorName.c_str()));
59  TEveGeoShape* geomShape = TEveGeoShape::ImportShapeExtract(geomShapreExtract);
60  f->Close();
61 
62  geomShape->SetName(detectorName.c_str());
63  // tricks for different R-Phi geom of TPC:
64  if (detectorName == "RPH") { // use all other parameters of regular TPC geom
65  detectorName = "TPC";
66  }
67 
68  // prepare geometry to be drawn including all children
69  drawDeep(geomShape,
70  settings.GetValue(Form("%s.color", detectorName.c_str()), -1),
71  settings.GetValue(Form("%s.trans", detectorName.c_str()), -1),
72  settings.GetValue(Form("%s.line.color", detectorName.c_str()), -1));
73 
74  gEve->GetDefaultGLViewer()->UpdateScene();
75 
76  return geomShape;
77 }

79 void GeometryManager::drawDeep(TEveGeoShape* geomShape, Color_t color, Char_t transparency, Color_t lineColor)
80 {
81  if (geomShape->HasChildren()) {
82  geomShape->SetRnrSelf(false);
83 
84  if (strcmp(geomShape->GetElementName(), "TPC_Drift_1") == 0) { // hack for TPC drift chamber
85  geomShape->SetRnrSelf(kTRUE);
86  if (color >= 0) {
87  geomShape->SetMainColor(color);
88  }
89  if (lineColor >= 0) {
90  geomShape->SetLineColor(lineColor);
91  geomShape->SetLineWidth(0.1);
92  geomShape->SetDrawFrame(true);
93  } else {
94  geomShape->SetDrawFrame(false);
95  }
96  if (transparency >= 0) {
97  geomShape->SetMainTransparency(transparency);
98  }
99  }
100 
101  for (TEveElement::List_i i = geomShape->BeginChildren(); i != geomShape->EndChildren(); ++i) {
102  drawDeep(static_cast<TEveGeoShape*>(*i), color, transparency, lineColor);
103  }
104  } else {
105  geomShape->SetRnrSelf(true);
106  if (color >= 0) {
107  geomShape->SetMainColor(color);
108  }
109  if (lineColor >= 0) {
110  geomShape->SetLineColor(lineColor);
111  geomShape->SetLineWidth(0.1);
112  geomShape->SetDrawFrame(true);
113  } else {
114  geomShape->SetDrawFrame(false);
115  }
116  if (transparency >= 0) {
117  geomShape->SetMainTransparency(transparency);
118  }
119 
120  if (strcmp(geomShape->GetElementName(), "PHOS_5") == 0) { // hack for PHOS module which is not installed
121  geomShape->SetRnrSelf(false);
122  }
123  }
124 }
