28 // This defines an output region. ptrBase points to a memory buffer, which should have a proper alignment.
29 // Since DPL does not respect the alignment of data types, we do not impose anything specic but just use void*, but it should be >= 64 bytes ideally.
30 // The size defines the maximum possible buffer size when GPUReconstruction is called, and returns the number of filled bytes when it returns.
31 // If the buffer size is exceeded, size is set to 1
32 // ptrCurrent must equal ptr if set (or nullptr), and can be incremented by GPUReconstruction step by step if multiple buffers are used.
33 // If ptr == nullptr, there is no region defined and GPUReconstruction will write its output to an internal buffer.
34 // If allocator is set, it is called as a callback to provide a ptr to the memory.
35 
36 struct GPUOutputControl {
37  GPUOutputControl() = default;
38  void set(void* p, size_t s)
39  {
40  reset();
41  ptrBase = ptrCurrent = p;
42  size = s;
43  }
44  void set(const std::function<void*(size_t)>& a)
45  {
46  reset();
47  allocator = a;
48  }
49  void reset()
50  {
51  new (this) GPUOutputControl;
52  }
53  bool useExternal() { return size || allocator; }
54  bool useInternal() { return !useExternal(); }
55  void checkCurrent()
56  {
57  if (ptrBase && ptrCurrent == nullptr) {
58  ptrCurrent = ptrBase;
59  }
60  }
61 
62  void* ptrBase = nullptr; // Base ptr to memory pool, occupied size is ptrCurrent - ptr
63  void* ptrCurrent = nullptr; // Pointer to free Output Space
64  size_t size = 0; // Max Size of Output Data if Pointer to output space is given
65  std::function<void*(size_t)> allocator = nullptr; // Allocator callback
66 };
