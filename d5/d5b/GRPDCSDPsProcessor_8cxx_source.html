37 int GRPDCSDPsProcessor::process(const gsl::span<const DPCOM> dps)
38 {
39 
40  // first we check which DPs are missing - if some are, it means that
41  // the delta map was sent
42  if (mVerbose) {
43  LOG(info) << "\n\n\nProcessing new TF\n-----------------";
44  }
45  if (!mFirstTimeSet) {
46  mFirstTime = mStartValidity;
47  mFirstTimeSet = true;
48  }
49  std::unordered_map<DPID, DPVAL> mapin;
50  for (auto& it : dps) {
51  mapin[it.id] = it.data;
52  }
53  for (auto& it : mPids) {
54  const auto& el = mapin.find(it.first);
55  if (el == mapin.end()) {
56  LOG(debug) << "DP " << it.first << " not found in list of DPs expected for GRP";
57  } else {
58  LOG(debug) << "DP " << it.first << " found in list of DPs expected for GRP";
59  }
60  }
61 
62  mUpdateMagField = false; // by default, we do not foresee a new entry in the CCDB for the B field
63  mUpdateEnvVars = false; // by default, we do not foresee a new entry in the CCDB for the Env Var
64  mUpdateCollimators = false; // by default, we do not foresee a new entry in the CCDB for the Collimators
65  mUpdateLHCIFInfo = false; // by default, we do not foresee a new entry in the CCDB for the LHCIF DPs
66 
67  // now we process all DPs, one by one
68  for (const auto& it : dps) {
69  // we process only the DPs defined in the configuration
70  const auto& el = mPids.find(it.id);
71  if (el == mPids.end()) {
72  LOG(info) << "DP " << it.id << " not found in GRPDCSProcessor, we will not process it";
73  continue;
74  }
75  processDP(it);
76  mPids[it.id] = true;
77  }
78 
79  if (mUpdateMagField) {
80  updateMagFieldCCDB();
81  }
82  if (mUpdateEnvVars) {
83  updateEnvVarsCCDB();
84  }
85  if (mUpdateCollimators) {
86  updateCollimatorsCCDB();
87  }
88  if (mUpdateLHCIFInfo) {
89  updateLHCIFInfoCCDB();
90  }
91 
92  return 0;
93 }

97 int GRPDCSDPsProcessor::processDP(const DPCOM& dpcom)
98 {
99 
100  // processing single DP
101 
102  auto& dpid = dpcom.id;
103  const auto& type = dpid.get_type();
104  auto& val = dpcom.data;
105  if (mVerbose) {
106  if (type == RAW_DOUBLE) {
107  LOG(info);
108  LOG(info) << "Processing DP = " << dpcom << ", with double value = " << o2::dcs::getValue<double>(dpcom);
109  } else if (type == RAW_BOOL) {
110  LOG(info);
111  LOG(info) << "Processing DP = " << dpcom << ", with bool value = " << o2::dcs::getValue<bool>(dpcom);
112  } else if (type == RAW_STRING) {
113  LOG(info);
114  LOG(info) << "Processing DP = " << dpcom << ", with string value = " << o2::dcs::getValue<std::string>(dpcom);
115  }
116  }
117  auto flags = val.get_flags();
118  if (processFlags(flags, dpid.get_alias()) == 0) {
119 
120  // now I need to access the correct element
121  std::string aliasStr(dpid.get_alias());
122  // LOG(info) << "alias 0 = " << aliasStr;
123  // B-field DPs
124  if (aliasStr.find("Current") != string::npos || aliasStr.find("Polarity") != string::npos) { // B-field DPs
125  mUpdateMagField = true;
126  if (aliasStr == "L3Current") {
127  mMagField.setL3Current(std::signbit(mMagField.getL3Current()) ? -static_cast<float>(o2::dcs::getValue<double>(dpcom)) : static_cast<float>(o2::dcs::getValue<double>(dpcom))); // true is negative field
128  if (mVerbose) {
129  LOG(info) << "Updating L3 current with value " << mMagField.getL3Current();
130  }
131  } else if (aliasStr == "L3Polarity") {
132  mMagField.setL3Current(o2::dcs::getValue<bool>(dpcom) ? -mMagField.getL3Current() : mMagField.getL3Current()); // true is negative field
133  if (mVerbose) {
134  LOG(info) << "Updating L3 polarity with value " << std::signbit(mMagField.getL3Current()); // poisitive is false, negative is true
135  }
136  } else if (aliasStr == "DipoleCurrent") {
137  mMagField.setDipoleCurrent(std::signbit(mMagField.getDipoleCurrent()) ? -static_cast<float>(o2::dcs::getValue<double>(dpcom)) : static_cast<float>(o2::dcs::getValue<double>(dpcom))); // true is negative field
138  if (mVerbose) {
139  LOG(info) << "Updating Dipole current with value " << mMagField.getDipoleCurrent();
140  }
141  } else if (aliasStr == "DipolePolarity") {
142  mMagField.setDipoleCurrent(o2::dcs::getValue<bool>(dpcom) ? -mMagField.getDipoleCurrent() : mMagField.getDipoleCurrent()); // true is negative field
143  if (mVerbose) {
144  LOG(info) << "Updating Dipole polarity with value " << std::signbit(mMagField.getDipoleCurrent()); // poisitive is false, negative is true
145  }
146  } else {
147  if (mVerbose) {
148  LOG(info) << "Alias " << aliasStr << " seemd from B field, but it is not recognized";
149  }
150  mUpdateMagField = false;
151  }
152  } else {
153  // environment variables
154  if (aliasStr.find("Cavern") != string::npos || aliasStr.find("Surface") != string::npos) {
155  if (mVerbose) {
156  LOG(info) << "Alias " << aliasStr << " seems from Env Variables";
157  }
158  processEnvVar(dpcom);
159  }
160 
161  else {
162  // Collimators
163  if (aliasStr.find("Collimator") != string::npos) {
164  // this is a collimator
165  if (mVerbose) {
166  LOG(info) << "Alias " << aliasStr << " seems from Collimators";
167  }
168  processCollimators(dpcom);
169  }
170 
171  else {
172  // the rest should all be LHCIF DPs
173  if (mVerbose) {
174  LOG(info) << "Alias " << aliasStr << " should be related to LHCIF";
175  }
176  processLHCIFDPs(dpcom);
177  }
178  }
179  }
180  }
181  return 0;
182 }

218 void GRPDCSDPsProcessor::processPair(const DPCOM& dpcom, const std::string& alias, std::pair<uint64_t, double>& p, bool& flag)
219 {
220 
221  // function to process Data Points that is stored in a pair
222 
223  auto& dpcomdata = dpcom.data;
224  auto& dpid = dpcom.id;
225  std::string aliasStr(dpid.get_alias());
226 
227  if (mVerbose) {
228  LOG(info) << "Processing alias " << aliasStr;
229  }
230 
231  if (aliasStr == alias) {
232  if (mVerbose) {
233  LOG(info) << "It matches the requested string " << alias << ", let's check if the value needs to be updated";
234  }
235  flag = compareAndUpdate(p, dpcom);
236  }
237  return;

242 bool GRPDCSDPsProcessor::compareAndUpdate(std::pair<uint64_t, double>& p, const DPCOM& dpcom)
243 {
244 
245  // check if the content of the pair should be updated
246 
247  double val = o2::dcs::getValue<double>(dpcom);
248  auto& dpcomdata = dpcom.data;
249  if (mVerbose) {
250  LOG(info) << "old value = " << p.second << ", new value = " << val << ", absolute difference = " << std::abs(p.second - val);
251  }
252  if (std::abs(p.second - val) > 0.5e-7 * (std::abs(p.second) + std::abs(val))) {
253  p.first = dpcomdata.get_epoch_time();
254  p.second = val;
255  if (mVerbose) {
256  LOG(info) << "value will be updated";
257  }
258  return true;
259  }
260  if (mVerbose) {
261  LOG(info) << "value will not be updated";
262  }
263  return false;
264 }

268 bool GRPDCSDPsProcessor::processLHCIFDPs(const DPCOM& dpcom)
269 {
270 
271  // function to process the remaining LHCIF DPs
272 
273  auto& dpcomdata = dpcom.data;
274  auto& dpid = dpcom.id;
275  std::string aliasStr(dpid.get_alias());
276  if (mVerbose) {
277  LOG(info) << "Processing LHCIF DP " << aliasStr;
278  }
279  for (int ibeam = 1; ibeam <= 2; ++ibeam) {
280  if (aliasStr.find(fmt::format("{}{}", "LHC_IntensityBeam", ibeam)) != string::npos) {
281  double val = o2::dcs::getValue<double>(dpcom);
282  mLHCInfo.mIntensityBeam[ibeam - 1].emplace_back(dpcomdata.get_epoch_time(), val);
283  if (mVerbose) {
284  LOG(info) << "Adding value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " to mIntensityBeam[" << ibeam - 1 << "]";
285  }
286  return true;
287  }
288  }
289 
290  if (aliasStr.find("BPTX") != string::npos) {
291  double val = o2::dcs::getValue<double>(dpcom);
292  if (aliasStr == "BPTX_deltaT_B1_B2") {
293  mLHCInfo.mBPTXdeltaT.emplace_back(dpcomdata.get_epoch_time(), val);
294  if (mVerbose) {
295  LOG(info) << "Adding value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " to BPTX_deltaT_B1_B2";
296  }
297  return true;
298  }
299  if (aliasStr == "BPTX_deltaTRMS_B1_B2") {
300  mLHCInfo.mBPTXdeltaTRMS.emplace_back(dpcomdata.get_epoch_time(), val);
301  if (mVerbose) {
302  LOG(info) << "Adding value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " to BPTX_deltaTRMS_B1_B2";
303  }
304  return true;
305  }
306  for (int ibeam = 1; ibeam <= 2; ++ibeam) {
307  if (aliasStr == fmt::format("{}{}", "BPTX_Phase_B", ibeam)) {
308  mLHCInfo.mBPTXPhase[ibeam - 1].emplace_back(dpcomdata.get_epoch_time(), val);
309  if (mVerbose) {
310  LOG(info) << "Adding value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " to mBPTXPhase[" << ibeam - 1 << "]";
311  }
312  return true;
313  } else if (aliasStr == fmt::format("{}{}", "BPTX_PhaseRMS_B", ibeam)) {
314  mLHCInfo.mBPTXPhaseRMS[ibeam - 1].emplace_back(dpcomdata.get_epoch_time(), val);
315  if (mVerbose) {
316  LOG(info) << "Adding value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " to mBPTXPhaseRMS[" << ibeam - 1 << "]";
317  }
318  return true;
319  } else if (aliasStr == fmt::format("{}{}", "BPTX_Phase_Shift_B", ibeam)) {
320  mLHCInfo.mBPTXPhaseShift[ibeam - 1].emplace_back(dpcomdata.get_epoch_time(), val);
321  if (mVerbose) {
322  LOG(info) << "Adding value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " to mBPTXPhaseShift[" << ibeam - 1 << "]";
323  }
324  return true;
325  }
326  }
327  return true;
328  }
329  if (aliasStr == "ALI_Lumi_Total_Inst") {
330  double val = o2::dcs::getValue<double>(dpcom);
331  mLHCInfo.mInstLumi.emplace_back(dpcomdata.get_epoch_time(), val);
332  if (mVerbose) {
333  LOG(info) << "Adding value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " to mInstLumi";
334  }
335  return true;
336  }
337 
338  for (int ibkg = 1; ibkg <= 3; ++ibkg) {
339  if (aliasStr.find(fmt::format("{}{}", "ALI_Background", ibkg)) != string::npos) {
340  double val = o2::dcs::getValue<double>(dpcom);
341  mLHCInfo.mBackground[ibkg - 1].emplace_back(dpcomdata.get_epoch_time(), val);
342  LOG(info) << "Adding value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " to mBackground[" << ibkg - 1 << "]";
343  if (mVerbose) {
344  LOG(info) << "Adding value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " to mBackground[" << ibkg - 1 << "]";
345  }
346  return true;
347  }
348  }
349  if (aliasStr == "ALI_Lumi_Source_Name") {
350  std::string val = o2::dcs::getValue<std::string>(dpcom);
351  mLHCInfo.mLumiSource.first = dpcomdata.get_epoch_time();
352  mLHCInfo.mLumiSource.second = val;
353  LOG(info) << "Updating value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " for mLumiSource";
354  if (mVerbose) {
355  LOG(info) << "Updating value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " for mLumiSource";
356  }
357  mUpdateLHCIFInfo = true; // we force the update of the LHCIF if the lumi source changes
358  return true;
359  }
360 
361  if (aliasStr == "MACHINE_MODE") {
362  std::string val = o2::dcs::getValue<std::string>(dpcom);
363  mLHCInfo.mMachineMode.first = dpcomdata.get_epoch_time();
364  mLHCInfo.mMachineMode.second = val;
365  LOG(info) << "Updating value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " for mMachineMode";
366  if (mVerbose) {
367  LOG(info) << "Updating value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " for mMachineMode";
368  }
369  mUpdateLHCIFInfo = true; // we force the update of the LHCIF if the machine mode (ION PHYSICS, PROTON PHYSICS..., see https://lhc-commissioning.web.cern.ch/systems/data-exchange/doc/LHC-OP-ES-0005-10-00.pdf) changes
370  return true;
371  }
372  if (aliasStr == "BEAM_MODE") {
373  std::string val = o2::dcs::getValue<std::string>(dpcom);
374  LOG(info) << "Updating value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " for mBeamMode";
375  mLHCInfo.mBeamMode.first = dpcomdata.get_epoch_time();
376  mLHCInfo.mBeamMode.second = val;
377  if (mVerbose) {
378  LOG(info) << "Updating value " << val << " with timestamp " << dpcomdata.get_epoch_time() << " for mBeamMode";
379  }
380 
381  mUpdateLHCIFInfo = true; // we force the update of the LHCIF if the beam mode (SETUP, RAMP, STABLE BEAMS..., see https://lhc-commissioning.web.cern.ch/systems/data-exchange/doc/LHC-OP-ES-0005-10-00.pdf) changes
382  return true;
383  }
384 
385  // LOG(error) << "DP " << aliasStr << " not known for GRP, please check";
386  return false;
387 }
