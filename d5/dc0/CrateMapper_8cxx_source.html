158 uint8_t CrateMapper::deLocalBoardToRO(uint8_t deId, uint8_t columnId, uint8_t lineId) const
159 {
161  auto item = mDEToROMap.find(deBoardId(detparams::getRPCLine(deId), columnId, lineId));
162  if (item == mDEToROMap.end()) {
163  throw std::runtime_error(fmt::format("Non-existent deId: {:d} columnId: {:d} lineId: {:d}", deId, columnId, lineId));
164  }
165  return detparams::isRightSide(deId) ? item->second : item->second + (crateparams::sNCratesPerSide << 4);
166 }

168 uint16_t CrateMapper::roLocalBoardToDE(uint8_t uniqueLocId) const
169 {
171  auto item = mROToDEMap.find(getROBoardIdRight(uniqueLocId));
172  if (item == mROToDEMap.end()) {
173  throw std::runtime_error(fmt::format("Non-existent crateId: {:d} boardId: {:d}", raw::getCrateId(uniqueLocId), raw::getLocId(uniqueLocId)));
174  }
175  return item->second;
176 }

178 std::vector<uint8_t> CrateMapper::getROBoardIds(uint16_t gbtUniqueId) const
179 {
180  std::vector<uint8_t> roBoardIds;
181  std::array<uint8_t, 2> offsets{0, 0x80};
182  for (auto& item : mROToDEMap) {
183  // For simplicity, the map only contains one side
184  // So we loop on the two sides
185  for (auto& off : offsets) {
186  auto locId = item.first + off;
187  if (gbtUniqueId != 0xFFFF) {
188  if (!gbtmapper::isBoardInGBT(locId, gbtUniqueId)) {
189  continue;
190  }
191  }
192  roBoardIds.emplace_back(locId);
193  }
194  }
195  std::sort(roBoardIds.begin(), roBoardIds.end());
196  return roBoardIds;
197 }
