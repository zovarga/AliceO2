61 bool ChannelMasksHandler::applyMask(ColumnData& data) const
62 {
65  auto uniqueId = getColumnDataUniqueId(data.deId, data.columnId);
66  auto maskIt = mMasks.find(uniqueId);
67  if (maskIt == mMasks.end()) {
68  return true;
69  }
70  uint16_t allPatterns = 0;
71  data.setNonBendPattern(data.getNonBendPattern() & maskIt->second.getNonBendPattern());
72  allPatterns |= data.getNonBendPattern();
73  for (int iline = 0; iline < 4; ++iline) {
74  data.setBendPattern(data.getBendPattern(iline) & maskIt->second.getBendPattern(iline), iline);
75  allPatterns |= data.getBendPattern(iline);
76  }
77  return (allPatterns != 0);
78 }
