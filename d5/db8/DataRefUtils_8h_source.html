34  template <typename T>
35  static auto as(DataRef const& ref)
36  {
37  // SFINAE makes this available only for the case we are using
38  // trivially copyable type, this is to distinguish it from the
39  // alternative below, which works for TObject (which are serialised).
40  if constexpr (is_messageable<T>::value == true) {
41  using DataHeader = o2::header::DataHeader;
42  auto header = o2::header::get<const DataHeader*>(ref.header);
43  if (header->payloadSerializationMethod != o2::header::gSerializationMethodNone) {
44  throw runtime_error("Attempt to extract a POD from a wrong message kind");
45  }
46  if ((header->payloadSize % sizeof(T)) != 0) {
47  throw runtime_error("Cannot extract POD from message as size do not match");
48  }
49  //FIXME: provide a const collection
50  return gsl::span<T>(reinterpret_cast<T*>(const_cast<char*>(ref.payload)), header->payloadSize / sizeof(T));
51  } else if constexpr (has_root_dictionary<T>::value == true &&
52  is_messageable<T>::value == false) {
53  std::unique_ptr<T> result;
54  static_assert(is_type_complete_v<struct RootSerializationSupport>, "Framework/RootSerializationSupport.h not included");
55  call_if_defined<struct RootSerializationSupport>([&](auto* p) {
56  // Classes using the ROOT ClassDef/ClassImp macros can be detected automatically
57  // and the serialization method can be deduced. If the class is also
58  // messageable, gSerializationMethodNone is used by default. This substitution
59  // does not apply for such types, the serialization method needs to be specified
60  // explicitely using type wrapper @a ROOTSerialized.
61  using RSS = std::decay_t<decltype(*p)>;
62  using DataHeader = o2::header::DataHeader;
63  auto header = o2::header::get<const DataHeader*>(ref.header);
64  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
65  throw runtime_error("Attempt to extract a TMessage from non-ROOT serialised message");
66  }
67 
68  typename RSS::FairTMessage ftm(const_cast<char*>(ref.payload), header->payloadSize);
69  auto* storedClass = ftm.GetClass();
70  auto* requestedClass = RSS::TClass::GetClass(typeid(T));
71  // should always have the class description if has_root_dictionary is true
72  assert(requestedClass != nullptr);
73 
74  auto* object = ftm.ReadObjectAny(storedClass);
75  if (object == nullptr) {
76  throw runtime_error_f("Failed to read object with name %s from message using ROOT serialization.",
77  (storedClass != nullptr ? storedClass->GetName() : "<unknown>"));
78  }
79 
80  if constexpr (std::is_base_of<typename RSS::TObject, T>::value) { // compile time switch
81  // if the type to be extracted inherits from TObject, the class descriptions
82  // do not need to match exactly, only the dynamic_cast has to work
83  // FIXME: could probably try to extend this to arbitrary types T being a base
84  // to the stored type, but this would require to cast the extracted object to
85  // the actual type. This requires this information to be available as a type
86  // in the ROOT dictionary, check if this is the case or if TClass::DynamicCast
87  // can be used for the test. Right now, this case was and is not covered and
88  // not yet checked in the unit test either.
89  auto* r = dynamic_cast<T*>(static_cast<typename RSS::TObject*>(object));
90  if (r) {
91  result.reset(r);
92  }
93  // This check includes the case: storedClass == requestedClass
94  } else if (storedClass->InheritsFrom(requestedClass)) {
95  result.reset(static_cast<T*>(object));
96  }
97  if (result == nullptr) {
98  // did not manage to cast the pointer to result
99  // delete object via the class info if available, not the case for all types,
100  // e.g. for standard containers of ROOT objects apparently this is not always
101  // the case
102  auto* delfunc = storedClass->GetDelete();
103  if (delfunc) {
104  (*delfunc)(object);
105  }
106 
107  throw runtime_error_f("Attempting to extract a %s but a %s is actually stored which cannot be casted to the requested one.",
108  (requestedClass != nullptr ? requestedClass->GetName() : "<unknown>"),
109  (storedClass != nullptr ? storedClass->GetName() : "<unknown>"));
110  }
111  // collections in ROOT can be non-owning or owning and the proper cleanup depends on
112  // this flag. Be it a bug or a feature in ROOT, but the owning flag of the extracted
113  // object only depends on the state at serialization of the original object. However,
114  // all objects created during deserialization are new and must be owned by the collection
115  // to avoid memory leak. So we call SetOwner if it is available for the type.
116  if constexpr (has_root_setowner<T>::value) {
117  result->SetOwner(true);
118  }
119  });
120 
121  return std::move(result);
122  } else if constexpr (is_specialization_v<T, ROOTSerialized> == true) {
123  // See above. SFINAE allows us to use this to extract a ROOT-serialized object
124  // with a somewhat uniform API. ROOT serialization method is enforced by using
125  // type wrapper @a ROOTSerialized
126  static_assert(is_type_complete_v<struct RootSerializationSupport>, "Framework/RootSerializationSupport.h not included");
127  using wrapped = typename T::wrapped_type;
128  using DataHeader = o2::header::DataHeader;
129  std::unique_ptr<wrapped> result;
130 
131  call_if_defined<struct RootSerializationSupport>([&](auto* p) {
132  using RSS = std::decay_t<decltype(*p)>;
133 
134  auto header = o2::header::get<const DataHeader*>(ref.header);
135  if (header->payloadSerializationMethod != o2::header::gSerializationMethodROOT) {
136  throw runtime_error("Attempt to extract a TMessage from non-ROOT serialised message");
137  }
138  auto* cl = RSS::TClass::GetClass(typeid(wrapped));
139  if (has_root_dictionary<wrapped>::value == false && cl == nullptr) {
140  throw runtime_error("ROOT serialization not supported, dictionary not found for data type");
141  }
142 
143  typename RSS::FairTMessage ftm(const_cast<char*>(ref.payload), header->payloadSize);
144  result.reset(static_cast<wrapped*>(ftm.ReadObjectAny(cl)));
145  if (result.get() == nullptr) {
146  throw runtime_error_f("Unable to extract class %s", cl == nullptr ? "<name not available>" : cl->GetName());
147  }
148  // workaround for ROOT feature, see above
149  if constexpr (has_root_setowner<T>::value) {
150  result->SetOwner(true);
151  }
152  });
153  return std::move(result);
154  } else if constexpr (is_specialization_v<T, CCDBSerialized> == true) {
155  using wrapped = typename T::wrapped_type;
156  using DataHeader = o2::header::DataHeader;
157  std::unique_ptr<wrapped> result(static_cast<wrapped*>(DataRefUtils::decodeCCDB(ref, typeid(wrapped))));
158  return std::move(result);
159  }
160  }

164  static o2::header::DataHeader::PayloadSizeType getPayloadSize(const DataRef& ref)
165  {
166  using DataHeader = o2::header::DataHeader;
167  auto* header = o2::header::get<const DataHeader*>(ref.header);
168  if (!header) {
169  return 0;
170  }
171  // in case of an O2 message with multiple payloads, the size of the message stored
172  // in DataRef is returned,
173  // as a prototype solution we are using splitPayloadIndex == splitPayloadParts to
174  // indicate that there are splitPayloadParts payloads following the header
175  if (header->splitPayloadParts > 1 && header->splitPayloadIndex == header->splitPayloadParts) {
176  return ref.payloadSize;
177  }
178  return header->payloadSize < ref.payloadSize || ref.payloadSize == 0 ? header->payloadSize : ref.payloadSize;
179  }

182  static auto getHeader(const DataRef& ref)
183  {
184  using HeaderT = typename std::remove_pointer<T>::type;
185  static_assert(std::is_pointer<T>::value && std::is_base_of<o2::header::BaseHeader, HeaderT>::value,
186  "pointer to BaseHeader-derived type required");
187  return o2::header::get<T>(ref.header);
