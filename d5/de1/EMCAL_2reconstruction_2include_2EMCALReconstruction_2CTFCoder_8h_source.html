58 template <typename VEC>
59 void CTFCoder::encode(VEC& buff, const gsl::span<const TriggerRecord>& trigData, const gsl::span<const Cell>& cellData)
60 {
61  using MD = o2::ctf::Metadata::OptStore;
62  // what to do which each field: see o2::ctd::Metadata explanation
63  constexpr MD optField[CTF::getNBlocks()] = {
64  MD::EENCODE, // BLC_bcIncTrig
65  MD::EENCODE, // BLC_orbitIncTrig
66  MD::EENCODE, // BLC_entriesTrig
67  MD::EENCODE, // BLC_towerID
68  MD::EENCODE, // BLC_time
69  MD::EENCODE, // BLC_energy
70  MD::EENCODE, // BLC_status
71  // extra slot was added in the end
72  MD::EENCODE // BLC_trigger
73  };
74 
75  CTFHelper helper(trigData, cellData);
76 
77  // book output size with some margin
78  auto szIni = sizeof(CTFHeader) + helper.getSize() * 2. / 3; // will be autoexpanded if needed
79  buff.resize(szIni);
80 
81  auto ec = CTF::create(buff);
82  using ECB = CTF::base;
83 
84  ec->setHeader(helper.createHeader());
85  assignDictVersion(static_cast<o2::ctf::CTFDictHeader&>(ec->getHeader()));
86  ec->getANSHeader().majorVersion = 0;
87  ec->getANSHeader().minorVersion = 1;
88  // at every encoding the buffer might be autoexpanded, so we don't work with fixed pointer ec
89 #define ENCODEEMC(beg, end, slot, bits) CTF::get(buff.data())->encode(beg, end, int(slot), bits, optField[int(slot)], &buff, mCoders[int(slot)].get(), getMemMarginFactor());
90  // clang-format off
91  ENCODEEMC(helper.begin_bcIncTrig(), helper.end_bcIncTrig(), CTF::BLC_bcIncTrig, 0);
92  ENCODEEMC(helper.begin_orbitIncTrig(), helper.end_orbitIncTrig(), CTF::BLC_orbitIncTrig, 0);
93  ENCODEEMC(helper.begin_entriesTrig(), helper.end_entriesTrig(), CTF::BLC_entriesTrig, 0);
94 
95  ENCODEEMC(helper.begin_towerID(), helper.end_towerID(), CTF::BLC_towerID, 0);
96  ENCODEEMC(helper.begin_time(), helper.end_time(), CTF::BLC_time, 0);
97  ENCODEEMC(helper.begin_energy(), helper.end_energy(), CTF::BLC_energy, 0);
98  ENCODEEMC(helper.begin_status(), helper.end_status(), CTF::BLC_status, 0);
99  // extra slot was added in the end
100  ENCODEEMC(helper.begin_trigger(), helper.end_trigger(), CTF::BLC_trigger, 0);
101  // clang-format on
102  CTF::get(buff.data())->print(getPrefix(), mVerbosity);
103 }

106 template <typename VTRG, typename VCELL>
107 void CTFCoder::decode(const CTF::base& ec, VTRG& trigVec, VCELL& cellVec)
108 {
109  const auto& header = ec.getHeader();
110  checkDictVersion(static_cast<const o2::ctf::CTFDictHeader&>(header));
111  ec.print(getPrefix(), mVerbosity);
112  std::vector<uint16_t> bcInc, entries, energy, cellTime, tower, trigger;
113  std::vector<uint32_t> orbitInc;
114  std::vector<uint8_t> status;
115 
116 #define DECODEEMCAL(part, slot) ec.decode(part, int(slot), mCoders[int(slot)].get())
117  // clang-format off
118  DECODEEMCAL(bcInc, CTF::BLC_bcIncTrig);
119  DECODEEMCAL(orbitInc, CTF::BLC_orbitIncTrig);
120  DECODEEMCAL(entries, CTF::BLC_entriesTrig);
121  DECODEEMCAL(tower, CTF::BLC_towerID);
122 
123  DECODEEMCAL(cellTime, CTF::BLC_time);
124  DECODEEMCAL(energy, CTF::BLC_energy);
125  DECODEEMCAL(status, CTF::BLC_status);
126  // extra slot was added in the end
127  DECODEEMCAL(trigger, CTF::BLC_trigger);
128  // triggers were added later, in old data they are absent:
129  if (trigger.empty()) {
130  trigger.resize(header.nTriggers);
131  }
132  //
133  // clang-format on
134  //
135  trigVec.clear();
136  cellVec.clear();
137  trigVec.reserve(header.nTriggers);
138  status.reserve(header.nCells);
139 
140  uint32_t firstEntry = 0, cellCount = 0;
141  o2::InteractionRecord ir(header.firstBC, header.firstOrbit);
142 
143  Cell cell;
144  TriggerRecord trg;
145  for (uint32_t itrig = 0; itrig < header.nTriggers; itrig++) {
146  // restore TrigRecord
147  if (orbitInc[itrig]) { // non-0 increment => new orbit
148  ir.bc = bcInc[itrig]; // bcInc has absolute meaning
149  ir.orbit += orbitInc[itrig];
150  } else {
151  ir.bc += bcInc[itrig];
152  }
153 
154  firstEntry = cellVec.size();
155  for (uint16_t ic = 0; ic < entries[itrig]; ic++) {
156  cell.setPacked(tower[cellCount], cellTime[cellCount], energy[cellCount], status[cellCount]);
157  cellVec.emplace_back(cell);
158  cellCount++;
159  }
160  trg.setBCData(ir);
161  trg.setDataRange(firstEntry, entries[itrig]);
162  trg.setTriggerBitsCompressed(trigger[itrig]);
163  trigVec.emplace_back(trg);
164  }
165  assert(cellCount == header.nCells);
166 }
