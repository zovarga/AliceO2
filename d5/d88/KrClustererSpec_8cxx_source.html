48 void TRDKrClustererDevice::run(ProcessingContext& pc)
49 {
50  TStopwatch timer;
51 
52  const auto digits = pc.inputs().get<gsl::span<Digit>>("digits");
53  const auto triggerRecords = pc.inputs().get<gsl::span<TriggerRecord>>("triggerRecords");
54 
55  mKrClFinder.reset();
56  mKrClFinder.setInput(digits, triggerRecords);
57  timer.Start();
58  mKrClFinder.findClusters();
59  timer.Stop();
60 
61  LOGF(info, "TRD Krypton cluster finder total timing: Cpu: %.3e Real: %.3e s", timer.CpuTime(), timer.RealTime());
62  LOGF(info, "Found %lu Kr clusters in %lu input trigger records.", mKrClFinder.getKrClusters().size(), triggerRecords.size());
63 
64  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "KRCLUSTER", 0, Lifetime::Timeframe}, mKrClFinder.getKrClusters());
65  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "TRGKRCLS", 0, Lifetime::Timeframe}, mKrClFinder.getKrTrigRecs());
66 }

73 DataProcessorSpec getKrClustererSpec(bool digitTrigRec)
74 {
75  std::vector<InputSpec> inputs;
76  inputs.emplace_back("digits", o2::header::gDataOriginTRD, "DIGITS", 0, Lifetime::Timeframe);
77  if (digitTrigRec) {
78  inputs.emplace_back("triggerRecords", o2::header::gDataOriginTRD, "TRGRDIG", 0, Lifetime::Timeframe);
79  } else {
80  inputs.emplace_back("triggerRecords", o2::header::gDataOriginTRD, "TRKTRGRD", 0, Lifetime::Timeframe);
81  }
82  std::vector<OutputSpec> outputs;
83  outputs.emplace_back(o2::header::gDataOriginTRD, "KRCLUSTER", 0, Lifetime::Timeframe);
84  outputs.emplace_back(o2::header::gDataOriginTRD, "TRGKRCLS", 0, Lifetime::Timeframe);
85 
86  return DataProcessorSpec{
87  "trd-kr-clusterer",
88  inputs,
89  outputs,
90  AlgorithmSpec{adaptFromTask<TRDKrClustererDevice>()},
91  Options{}};
