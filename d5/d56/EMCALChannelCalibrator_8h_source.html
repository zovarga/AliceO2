112 template <typename DataInput, typename DataOutput, typename HistContainer>
113 void EMCALChannelCalibrator<DataInput, DataOutput, HistContainer>::finalizeSlot(o2::calibration::TimeSlot<DataInput>& slot)
114 {
115  // Extract results for the single slot
116  DataInput* c = slot.getContainer();
117  LOG(info) << "Finalize slot " << slot.getTFStart() << " <= TF <= " << slot.getTFEnd();
118 
119  std::map<std::string, std::string> md;
120  if constexpr (std::is_same<DataInput, o2::emcal::EMCALChannelData>::value) {
121  auto bcm = mCalibrator->calibrateBadChannels(c->getHisto());
122  // for the CCDB entry
123  auto clName = o2::utils::MemFileHelper::getClassName(bcm);
124  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
125  mInfoVector.emplace_back(CalibDB::getCDBPathBadChannelMap(), clName, flName, md, slot.getTFStart(), 99999999999999);
126  mCalibObjectVector.push_back(bcm);
127  } else if constexpr (std::is_same<DataInput, o2::emcal::EMCALTimeCalibData>::value) {
128  auto tcd = mCalibrator->calibrateTime(c->getHisto());
129  // for the CCDB entry
130  auto clName = o2::utils::MemFileHelper::getClassName(slot);
131  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
132  mInfoVector.emplace_back(CalibDB::getCDBPathTimeCalibrationParams(), clName, flName, md, slot.getTFStart(), 99999999999999);
133  mCalibObjectVector.push_back(tcd);
134  }
135 }

138 o2::calibration::TimeSlot<DataInput>& EMCALChannelCalibrator<DataInput, DataOutput, HistContainer>::emplaceNewSlot(bool front, TFType tstart, TFType tend)
139 {
140  auto& cont = o2::calibration::TimeSlotCalibration<o2::emcal::Cell, DataInput>::getSlots();
141  auto& slot = front ? cont.emplace_front(tstart, tend) : cont.emplace_back(tstart, tend);
142  HistContainer histcont; // initialize struct with (default) ranges for time or channel calibration.
143  slot.setContainer(std::make_unique<DataInput>(histcont));
144  return slot;
145 }
