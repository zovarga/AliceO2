88 void GPUDisplayBackendOpenGL::createFB(GLfb& fb, bool tex, bool withDepth, bool msaa)
89 {
90  fb.tex = tex;
91  fb.depth = withDepth;
92  fb.msaa = msaa;
93  GLint drawFboId = 0, readFboId = 0;
94  glGetIntegerv(GL_DRAW_FRAMEBUFFER_BINDING, &drawFboId);
95  glGetIntegerv(GL_READ_FRAMEBUFFER_BINDING, &readFboId);
96  CHKERR(glGenFramebuffers(1, &fb.fb_id));
97  CHKERR(glBindFramebuffer(GL_FRAMEBUFFER, fb.fb_id));
98 
99  auto createFB_texture = [&](GLuint& id, GLenum storage, GLenum attachment) {
100  GLenum textureType = fb.msaa ? GL_TEXTURE_2D_MULTISAMPLE : GL_TEXTURE_2D;
101  CHKERR(glGenTextures(1, &id));
102  CHKERR(glBindTexture(textureType, id));
103  if (fb.msaa) {
104  CHKERR(glTexImage2DMultisample(GL_TEXTURE_2D_MULTISAMPLE, mDisplay->cfgR().drawQualityMSAA, storage, mDisplay->renderWidth(), mDisplay->renderHeight(), false));
105  } else {
106  CHKERR(glTexImage2D(GL_TEXTURE_2D, 0, storage, mDisplay->renderWidth(), mDisplay->renderHeight(), 0, storage, GL_UNSIGNED_BYTE, nullptr));
107  CHKERR(glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR));
108  CHKERR(glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR));
109  }
110  CHKERR(glFramebufferTexture2D(GL_FRAMEBUFFER, attachment, textureType, id, 0));
111  };
112 
113  auto createFB_renderbuffer = [&](GLuint& id, GLenum storage, GLenum attachment) {
114  CHKERR(glGenRenderbuffers(1, &id));
115  CHKERR(glBindRenderbuffer(GL_RENDERBUFFER, id));
116  if (fb.msaa) {
117  CHKERR(glRenderbufferStorageMultisample(GL_RENDERBUFFER, mDisplay->cfgR().drawQualityMSAA, storage, mDisplay->renderWidth(), mDisplay->renderHeight()));
118  } else {
119  CHKERR(glRenderbufferStorage(GL_RENDERBUFFER, storage, mDisplay->renderWidth(), mDisplay->renderHeight()));
120  }
121  CHKERR(glFramebufferRenderbuffer(GL_FRAMEBUFFER, attachment, GL_RENDERBUFFER, id));
122  };
123 
124  if (tex) {
125  createFB_texture(fb.fbCol_id, GL_RGBA, GL_COLOR_ATTACHMENT0);
126  } else {
127  createFB_renderbuffer(fb.fbCol_id, GL_RGBA, GL_COLOR_ATTACHMENT0);
128  }
129 
130  if (withDepth) {
131  if (tex && fb.msaa) {
132  createFB_texture(fb.fbDepth_id, GL_DEPTH_COMPONENT24, GL_DEPTH_ATTACHMENT);
133  } else {
134  createFB_renderbuffer(fb.fbDepth_id, GL_DEPTH_COMPONENT24, GL_DEPTH_ATTACHMENT);
135  }
136  }
137 
138  GLenum status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
139  if (status != GL_FRAMEBUFFER_COMPLETE) {
140  GPUError("Error creating framebuffer (tex %d) - incomplete (%d)", (int)tex, status);
141  exit(1);
142  }
143  CHKERR(glBindFramebuffer(GL_DRAW_FRAMEBUFFER, drawFboId));
144  CHKERR(glBindFramebuffer(GL_READ_FRAMEBUFFER, readFboId));
145  fb.created = true;
146 }

166 unsigned int GPUDisplayBackendOpenGL::drawVertices(const vboList& v, const drawType tt)
167 {
168  static constexpr GLenum types[3] = {GL_POINTS, GL_LINES, GL_LINE_STRIP};
169  GLenum t = types[tt];
170  auto first = std::get<0>(v);
171  auto count = std::get<1>(v);
172  auto iSlice = std::get<2>(v);
173  if (count == 0) {
174  return 0;
175  }
176 
177  if (mDisplay->useMultiVBO()) {
178  if (mDisplay->cfgR().openGLCore) {
179  CHKERR(glBindVertexArray(mVertexArray));
180  }
181  CHKERR(glBindBuffer(GL_ARRAY_BUFFER, mVBOId[iSlice]));
182 #ifndef GPUCA_DISPLAY_OPENGL_CORE
183  if (!mDisplay->cfgR().openGLCore) {
184  CHKERR(glVertexPointer(3, GL_FLOAT, 0, nullptr));
185  } else
186 #endif
187  {
188  CHKERR(glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 0, nullptr));
189  glEnableVertexAttribArray(0);
190  }
191  }
192 
193  if (mDisplay->cfgR().useGLIndirectDraw) {
194  CHKERR(glMultiDrawArraysIndirect(t, (void*)(size_t)((mIndirectSliceOffset[iSlice] + first) * sizeof(DrawArraysIndirectCommand)), count, 0));
195  } else if (OPENGL_EMULATE_MULTI_DRAW) {
196  for (unsigned int k = 0; k < count; k++) {
197  CHKERR(glDrawArrays(t, mDisplay->vertexBufferStart()[iSlice][first + k], mDisplay->vertexBufferCount()[iSlice][first + k]));
198  }
199  } else {
200  static_assert(sizeof(GLsizei) == sizeof(*mDisplay->vertexBufferCount()[iSlice].data()), "Invalid counter size does not match GLsizei");
201  CHKERR(glMultiDrawArrays(t, mDisplay->vertexBufferStart()[iSlice].data() + first, ((const GLsizei*)mDisplay->vertexBufferCount()[iSlice].data()) + first, count));
202  }
203  return count;
204 }

250 int GPUDisplayBackendOpenGL::InitBackend()
251 {
252  int glVersion[2] = {0, 0};
253  glGetIntegerv(GL_MAJOR_VERSION, &glVersion[0]);
254  glGetIntegerv(GL_MINOR_VERSION, &glVersion[1]);
255  if (glVersion[0] < GPUDisplayFrontend::GL_MIN_VERSION_MAJOR || (glVersion[0] == GPUDisplayFrontend::GL_MIN_VERSION_MAJOR && glVersion[1] < GPUDisplayFrontend::GL_MIN_VERSION_MINOR)) {
256  GPUError("Unsupported OpenGL runtime %d.%d < %d.%d", glVersion[0], glVersion[1], GPUDisplayFrontend::GL_MIN_VERSION_MAJOR, GPUDisplayFrontend::GL_MIN_VERSION_MINOR);
257  return (1);
258  }
259  mIndirectSliceOffset.resize(GPUCA_NSLICES);
260  mVBOId.resize(GPUCA_NSLICES);
261  CHKERR(glCreateBuffers(mVBOId.size(), mVBOId.data()));
262  CHKERR(glBindBuffer(GL_ARRAY_BUFFER, mVBOId[0]));
263  CHKERR(glGenBuffers(1, &mIndirectId));
264  CHKERR(glBindBuffer(GL_DRAW_INDIRECT_BUFFER, mIndirectId));
265 #ifndef GPUCA_DISPLAY_OPENGL_CORE
266  CHKERR(glShadeModel(GL_SMOOTH)); // Enable Smooth Shading
267 #endif
268  setDepthBuffer();
269  setQuality();
270  CHKERR(mVertexShader = glCreateShader(GL_VERTEX_SHADER));
271  CHKERR(glShaderSource(mVertexShader, 1, &GPUDisplayShaders::vertexShader, nullptr));
272  CHKERR(glCompileShader(mVertexShader));
273  CHKERR(mFragmentShader = glCreateShader(GL_FRAGMENT_SHADER));
274  CHKERR(glShaderSource(mFragmentShader, 1, &GPUDisplayShaders::fragmentShader, nullptr));
275  CHKERR(glCompileShader(mFragmentShader));
276  CHKERR(mShaderProgram = glCreateProgram());
277  CHKERR(glAttachShader(mShaderProgram, mVertexShader));
278  CHKERR(glAttachShader(mShaderProgram, mFragmentShader));
279  CHKERR(glLinkProgram(mShaderProgram));
280  CHKERR(glGenVertexArrays(1, &mVertexArray));
281  CHKERR(mModelViewProjId = glGetUniformLocation(mShaderProgram, "ModelViewProj"));
282  CHKERR(mColorId = glGetUniformLocation(mShaderProgram, "color"));
283  return (0); // Initialization Went OK
284 }

331 void GPUDisplayBackendOpenGL::loadDataToGPU(size_t totalVertizes)
332 {
333  // TODO: Check if this can be parallelized
334  if (mDisplay->useMultiVBO()) {
335  for (int i = 0; i < GPUCA_NSLICES; i++) {
336  CHKERR(glNamedBufferData(mVBOId[i], mDisplay->vertexBuffer()[i].size() * sizeof(mDisplay->vertexBuffer()[i][0]), mDisplay->vertexBuffer()[i].data(), GL_STATIC_DRAW));
337  mDisplay->vertexBuffer()[i].clear();
338  }
339  } else {
340  size_t totalYet = mDisplay->vertexBuffer()[0].size();
341  mDisplay->vertexBuffer()[0].resize(totalVertizes);
342  for (int i = 1; i < GPUCA_NSLICES; i++) {
343  for (unsigned int j = 0; j < mDisplay->vertexBufferStart()[i].size(); j++) {
344  mDisplay->vertexBufferStart()[i][j] += totalYet;
345  }
346  memcpy(&mDisplay->vertexBuffer()[0][totalYet], &mDisplay->vertexBuffer()[i][0], mDisplay->vertexBuffer()[i].size() * sizeof(mDisplay->vertexBuffer()[i][0]));
347  totalYet += mDisplay->vertexBuffer()[i].size();
348  mDisplay->vertexBuffer()[i].clear();
349  }
350  CHKERR(glBindBuffer(GL_ARRAY_BUFFER, mVBOId[0])); // Bind ahead of time, since it is not going to change
351  CHKERR(glNamedBufferData(mVBOId[0], totalVertizes * sizeof(mDisplay->vertexBuffer()[0][0]), mDisplay->vertexBuffer()[0].data(), GL_STATIC_DRAW));
352  mDisplay->vertexBuffer()[0].clear();
353  }
354 
355  if (mDisplay->cfgR().useGLIndirectDraw) {
356  mCmdBuffer.clear();
357  // TODO: Check if this can be parallelized
358  for (int iSlice = 0; iSlice < GPUCA_NSLICES; iSlice++) {
359  mIndirectSliceOffset[iSlice] = mCmdBuffer.size();
360  for (unsigned int k = 0; k < mDisplay->vertexBufferStart()[iSlice].size(); k++) {
361  mCmdBuffer.emplace_back(mDisplay->vertexBufferCount()[iSlice][k], 1, mDisplay->vertexBufferStart()[iSlice][k], 0);
362  }
363  }
364  CHKERR(glBufferData(GL_DRAW_INDIRECT_BUFFER, mCmdBuffer.size() * sizeof(mCmdBuffer[0]), mCmdBuffer.data(), GL_STATIC_DRAW));
365  mCmdBuffer.clear();
366  }
367 }

440 {

445  glMatrixMode(GL_PROJECTION);
