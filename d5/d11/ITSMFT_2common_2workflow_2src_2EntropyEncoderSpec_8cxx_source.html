46 void EntropyEncoderSpec::run(ProcessingContext& pc)
47 {
48  auto cput = mTimer.CpuTime();
49  mTimer.Start(false);
50  auto compClusters = pc.inputs().get<gsl::span<o2::itsmft::CompClusterExt>>("compClusters");
51  auto pspan = pc.inputs().get<gsl::span<unsigned char>>("patterns");
52  auto rofs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("ROframes");
53 
54  auto& buffer = pc.outputs().make<std::vector<o2::ctf::BufferType>>(Output{mOrigin, "CTFDATA", 0, Lifetime::Timeframe});
55  mCTFCoder.encode(buffer, rofs, compClusters, pspan);
56  auto eeb = CTF::get(buffer.data()); // cast to container pointer
57  eeb->compactify(); // eliminate unnecessary padding
58  buffer.resize(eeb->size()); // shrink buffer to strictly necessary size
59  // eeb->print();
60  mTimer.Stop();
61  LOG(info) << "Created encoded data of size " << eeb->size() << " for " << mOrigin.as<std::string>() << " in " << mTimer.CpuTime() - cput << " s";
62 }

70 DataProcessorSpec getEntropyEncoderSpec(o2::header::DataOrigin orig)
71 {
72  std::vector<InputSpec> inputs;
73  inputs.emplace_back("compClusters", orig, "COMPCLUSTERS", 0, Lifetime::Timeframe);
74  inputs.emplace_back("patterns", orig, "PATTERNS", 0, Lifetime::Timeframe);
75  inputs.emplace_back("ROframes", orig, "CLUSTERSROF", 0, Lifetime::Timeframe);
76 
77  return DataProcessorSpec{
78  orig == o2::header::gDataOriginITS ? "its-entropy-encoder" : "mft-entropy-encoder",
79  inputs,
80  Outputs{{orig, "CTFDATA", 0, Lifetime::Timeframe}},
81  AlgorithmSpec{adaptFromTask<EntropyEncoderSpec>(orig)},
82  Options{{"ctf-dict", VariantType::String, o2::base::NameConf::getCTFDictFileName(), {"File of CTF encoding dictionary"}},
83  {"mem-factor", VariantType::Float, 1.f, {"Memory allocation margin factor"}}}};
84 }
