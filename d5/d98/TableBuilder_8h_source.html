108  static arrow::Status appendToList(std::unique_ptr<arrow::FixedSizeListBuilder>& builder, T* data, int size = 1)
109  {
110  using ArrowType = typename detail::ConversionTraits<std::decay_t<T>>::ArrowType;
111  using BuilderType = typename arrow::TypeTraits<ArrowType>::BuilderType;
112  size_t numElements = static_cast<const arrow::FixedSizeListType*>(builder->type().get())->list_size();
113 
114  auto status = builder->AppendValues(size);
115  auto ValueBuilder = static_cast<BuilderType*>(builder->value_builder());
116  status &= ValueBuilder->AppendValues(data, numElements * size, nullptr);
117 
118  return status;
119  }

204  template <typename HolderType, typename PTR>
205  static arrow::Status bulkAppendChunked(HolderType& holder, BulkInfo<PTR> info)
206  {
207  // Appending nullptr is a no-op.
208  if (info.ptr == nullptr) {
209  return arrow::Status::OK();
210  }
211  if constexpr (std::is_same_v<decltype(holder.builder), std::unique_ptr<arrow::FixedSizeListBuilder>>) {
212  if (appendToList<std::remove_pointer_t<decltype(info.ptr)>>(holder.builder, info.ptr, info.size).ok() == false) {
213  throw runtime_error("Unable to append to column");
214  } else {
215  return arrow::Status::OK();
216  }
217  } else {
218  if (holder.builder->AppendValues(info.ptr, info.size, nullptr).ok() == false) {
219  throw runtime_error("Unable to append to column");
220  } else {
221  return arrow::Status::OK();
222  }
223  }
224  }

227  static arrow::Status append(HolderType& holder, std::pair<ITERATOR, ITERATOR> ip)
228  {
229  using ArrowType = typename detail::ConversionTraits<typename ITERATOR::value_type>::ArrowType;
230  using ValueBuilderType = typename arrow::TypeTraits<ArrowType>::BuilderType;
231  // FIXME: for the moment we do not fill things.
232  auto status = holder.builder->Append();
233  auto valueBuilder = reinterpret_cast<ValueBuilderType*>(holder.builder->value_builder());
234  return status & valueBuilder->AppendValues(&*ip.first, std::distance(ip.first, ip.second));
235  }

238  template <typename HolderType, typename ITERATOR>
239  static void unsafeAppend(HolderType& holder, std::pair<ITERATOR, ITERATOR> ip)
240  {
241  using ArrowType = typename detail::ConversionTraits<typename ITERATOR::value_type>::ArrowType;
242  using ValueBuilderType = typename arrow::TypeTraits<ArrowType>::BuilderType;

247  if (!status.ok()) {
248  throw runtime_error("Unable to append values to valueBuilder!");
249  }
250  return;

440 template <typename T>
441 struct CachedInsertion {
442  static constexpr int CHUNK_SIZE = 256;
443 
444  template <typename BUILDER>
445  arrow::Status append(BUILDER& builder, T value)
446  {
447  cache[pos % CHUNK_SIZE] = value;
448  ++pos;
449  if (pos % CHUNK_SIZE == 0) {
450  return builder->AppendValues(cache, CHUNK_SIZE, nullptr);
451  }
452  return arrow::Status::OK();
453  }
454 
455  template <typename BUILDER>
456  arrow::Status flush(BUILDER& builder)
457  {
458  if (pos % CHUNK_SIZE != 0) {
459  return builder->AppendValues(cache, pos % CHUNK_SIZE, nullptr);
460  }
461  return arrow::Status::OK();
462  }
463  T cache[CHUNK_SIZE];
464  int pos = 0;
465 };

615 template <class T>
616 auto constexpr to_tuple(T&& object) noexcept
617 {
618  using type = std::decay_t<T>;
619  if constexpr (is_braces_constructible<type, any_type, any_type, any_type, any_type>{}) {
620  auto&& [p0, p1, p2, p3] = object;
621  return std::make_tuple(p0, p1, p2, p3);
622  } else if constexpr (is_braces_constructible<type, any_type, any_type, any_type>{}) {
623  auto&& [p0, p1, p2] = object;
624  return std::make_tuple(p0, p1, p2);
625  } else if constexpr (is_braces_constructible<type, any_type, any_type>{}) {
626  auto&& [p0, p1] = object;
627  return std::make_tuple(p0, p1);
628  } else if constexpr (is_braces_constructible<type, any_type>{}) {
629  auto&& [p0] = object;
630  return std::make_tuple(p0);
631  } else {
632  return std::make_tuple();
633  }

857 template <typename... C>
858 auto spawner(framework::pack<C...> columns, std::vector<std::shared_ptr<arrow::Table>>&& tables, const char* name)
859 {
860  auto fullTable = soa::ArrowHelpers::joinTables(std::move(tables));
861  if (fullTable->num_rows() == 0) {
862  return makeEmptyTable<soa::Table<C...>>(name);
863  }
864  static auto new_schema = o2::soa::createSchemaFromColumns(columns);
865  auto projectors = framework::expressions::createProjectors(columns, fullTable->schema());
866 
867  arrow::TableBatchReader reader(*fullTable);
868  std::shared_ptr<arrow::RecordBatch> batch;
869  arrow::ArrayVector v;
870  std::array<arrow::ArrayVector, sizeof...(C)> chunks;
871  std::vector<std::shared_ptr<arrow::ChunkedArray>> arrays;
872 
873  while (true) {
874  auto s = reader.ReadNext(&batch);
875  if (!s.ok()) {
876  throw runtime_error_f("Cannot read batches from source table to spawn %s: %s", name, s.ToString().c_str());
877  }
878  if (batch == nullptr) {
879  break;
880  }
881  try {
882  s = projectors->Evaluate(*batch, arrow::default_memory_pool(), &v);
883  if (!s.ok()) {
884  throw runtime_error_f("Cannot apply projector to source table of %s: %s", name, s.ToString().c_str());
885  }
886  } catch (std::exception& e) {
887  throw runtime_error_f("Cannot apply projector to source table of %s: exception caught: %s", name, e.what());
888  }
889 
890  for (auto i = 0u; i < sizeof...(C); ++i) {
891  chunks[i].emplace_back(v.at(i));
892  }
893  }
894 
895  for (auto i = 0u; i < sizeof...(C); ++i) {
896  arrays.push_back(std::make_shared<arrow::ChunkedArray>(chunks[i]));
897  }
898 
899  addLabelToSchema(new_schema, name);
900  return arrow::Table::Make(new_schema, arrays);
901 }
