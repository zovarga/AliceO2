82 int TrackBasedCalib::doTrdOnlyTrackFits(gsl::span<const TrackTRD>& tracks)
83 {
84  int nTracksSuccess = 0;
85  for (const auto& trkIn : tracks) {
86  if (trkIn.getNtracklets() < 3) {
87  // with less than 3 tracklets the TRD-only refit not meaningful
88  continue;
89  }
90  auto trkWork = trkIn; // input is const, so we need to create a copy
91  bool trackFailed = false;
92 
93  trkWork.setChi2(0.f);
94  trkWork.resetCovariance(20);
95 
96  // first inward propagation (TRD track fit)
97  int currLayer = NLAYER;
98  for (int iLayer = NLAYER - 1; iLayer >= 0; --iLayer) {
99  if (trkWork.getTrackletIndex(iLayer) == -1) {
100  continue;
101  }
102  if (mNoiseCalib->isTrackletFromNoisyMCM(mTrackletsRaw[trkWork.getTrackletIndex(iLayer)])) {
103  // ignore tracklets which originate from noisy MCMs
104  continue;
105  }
106  if (propagateAndUpdate(trkWork, iLayer, true)) {
107  trackFailed = true;
108  break;
109  }
110  currLayer = iLayer;
111  }
112  if (trackFailed) {
113  continue;
114  }
115 
116  // outward propagation (smoothing)
117  for (int iLayer = currLayer + 1; iLayer < NLAYER; ++iLayer) {
118  if (trkWork.getTrackletIndex(iLayer) == -1) {
119  continue;
120  }
121  if (mNoiseCalib->isTrackletFromNoisyMCM(mTrackletsRaw[trkWork.getTrackletIndex(iLayer)])) {
122  // ignore tracklets which originate from noisy MCMs
123  continue;
124  }
125  if (propagateAndUpdate(trkWork, iLayer, true)) {
126  trackFailed = true;
127  break;
128  }
129  currLayer = iLayer;
130  }
131  if (trackFailed) {
132  continue;
133  }
134 
135  // second inward propagation (collect angular differences between tracklets + TRD track)
136  for (int iLayer = currLayer; iLayer >= 0; --iLayer) {
137  if (trkWork.getTrackletIndex(iLayer) == -1) {
138  continue;
139  }
140  if (mNoiseCalib->isTrackletFromNoisyMCM(mTrackletsRaw[trkWork.getTrackletIndex(iLayer)])) {
141  // ignore tracklets which originate from noisy MCMs
142  continue;
143  }
144  if (propagateAndUpdate(trkWork, iLayer, false)) {
145  trackFailed = true;
146  break;
147  }
148 
149  float trkAngle = o2::math_utils::asin(trkWork.getSnp()) * TMath::RadToDeg();
150  float trkltAngle = o2::math_utils::atan(mTrackletsCalib[trkWork.getTrackletIndex(iLayer)].getDy() / Geometry::cdrHght()) * TMath::RadToDeg();
151  float angleDeviation = trkltAngle - trkAngle;
152  if (mAngResHistos.addEntry(angleDeviation, trkAngle, mTrackletsRaw[trkWork.getTrackletIndex(iLayer)].getDetector())) {
153  // track impact angle out of histogram range
154  continue;
155  }
156  }
157 
158  // here we can count the number of successfully processed tracks
159  ++nTracksSuccess;
160  } // end of track loop
161  return nTracksSuccess;
162 }

164 bool TrackBasedCalib::propagateAndUpdate(TrackTRD& trk, int iLayer, bool doUpdate) const
165 {
166  // Propagates the track to TRD layer iLayer and updates the track
167  // parameters (if requested)
168  // returns 0 in case of success
169 
170  auto propagator = o2::base::Propagator::Instance();
171 
172  int trkltId = trk.getTrackletIndex(iLayer);
173  int trkltDet = mTrackletsRaw[trkltId].getDetector();
174  int trkltSec = trkltDet / (NLAYER * NSTACK);
175 
176  if (trkltSec != o2::math_utils::angle2Sector(trk.getAlpha())) {
177  if (!trk.rotate(o2::math_utils::sector2Angle(trkltSec))) {
178  LOGF(debug, "Track could not be rotated in tracklet coordinate system");
179  return 1;
180  }
181  }
182 
183  if (!propagator->PropagateToXBxByBz(trk, mTrackletsCalib[trkltId].getX(), mMaxSnp, mMaxStep, mMatCorr)) {
184  LOGF(debug, "Track propagation failed in layer %i (pt=%f, xTrk=%f, xToGo=%f)", iLayer, trk.getPt(), trk.getX(), mTrackletsCalib[trkltId].getX());
185  return 1;
186  }
187 
188  if (!doUpdate) {
189  // nothing more to be done
190  return 0;
191  }
192 
193  const PadPlane* pad = Geometry::instance()->getPadPlane(trkltDet);
194  float tilt = tan(TMath::DegToRad() * pad->getTiltingAngle()); // tilt is signed! and returned in degrees
195  float tiltCorrUp = tilt * (mTrackletsCalib[trkltId].getZ() - trk.getZ());
196  float zPosCorrUp = mTrackletsCalib[trkltId].getZ() + mRecoParam.getZCorrCoeffNRC() * trk.getTgl();
197  float padLength = pad->getRowSize(mTrackletsRaw[trkltId].getPadRow());
198  if (!((trk.getSigmaZ2() < (padLength * padLength / 12.f)) && (std::fabs(mTrackletsCalib[trkltId].getZ() - trk.getZ()) < padLength))) {
199  tiltCorrUp = 0.f;
200  }
201 
202  std::array<float, 2> trkltPosUp{mTrackletsCalib[trkltId].getY() - tiltCorrUp, zPosCorrUp};
203  std::array<float, 3> trkltCovUp;
204  mRecoParam.recalcTrkltCov(tilt, trk.getSnp(), pad->getRowSize(mTrackletsRaw[trkltId].getPadRow()), trkltCovUp);
205 
206  if (!trk.update(trkltPosUp, trkltCovUp)) {
207  LOGF(info, "Failed to update track with space point in layer %i", iLayer);
208  return 1;
209  }
210  return 0;
211 }
