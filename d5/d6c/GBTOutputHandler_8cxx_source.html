50 EventType GBTOutputHandler::processSelfTriggered(size_t ilink, uint16_t localClock, uint16_t& correctedClock)
51 {
53  correctedClock = localClock - mElectronicsDelay.BCToLocal;
54  uint16_t linkMask = 1 << ilink;
55  if ((mReceivedCalibration & linkMask) && (localClock == mExpectedFETClock[ilink])) {
56  // Reset the calibration flag for this e-link
57  mReceivedCalibration &= ~linkMask;
58  return EventType::FET;
59  }
60  return EventType::Standard;
61 }

71 void GBTOutputHandler::processOrbitTrigger(size_t ilink, uint16_t localClock, uint8_t triggerWord)
72 {
74 
75  // The local clock is reset: we are now in synch with the new HB
76  // We have to way to account for the orbit change:
77  // - increase the orbit counter by 1 for this e-link
78  // (CAVEAT: synch is lost if we lose some orbit)
79  // - set the orbit to the one found in RDH
80  // (CAVEAT: synch is lost if we have lot of data, spanning over two orbits)
81  // FIXME: pick one of the two
82  ++mIRs[ilink].orbit; // orbit increase
83  // mIRs[ilink].orbit = mOrbit; // reset to RDH
84  if ((triggerWord & raw::sSOX) == 0) {
85  mLastClock[ilink] = localClock;
86  }
87  // The orbit trigger resets the clock.
88  // If we received a calibration trigger, we need to change the value of the expected clock accordingly
89  if (mReceivedCalibration & (1 << ilink)) {
90  mExpectedFETClock[ilink] -= (localClock + 1);
91  }
92 }

94 bool GBTOutputHandler::processTrigger(size_t ilink, const ELinkDecoder& decoder, EventType& eventType, uint16_t& correctedClock)
95 {
99  uint16_t linkMask = 1 << ilink;
100  auto localClock = decoder.getCounter();
101 
102  if (decoder.getTriggerWord() == 0) {
103  // This is a self-triggered event
104  eventType = processSelfTriggered(ilink, localClock, correctedClock);
105  return true;
106  }
107 
108  // From here we treat triggered events
109  bool goOn = false;
110  correctedClock = localClock;
111  if (decoder.getTriggerWord() & raw::sCALIBRATE) {
112  // This is an answer to a calibration trigger
113  eventType = processCalibrationTrigger(ilink, localClock);
114  goOn = true;
115  }
116 
117  if (decoder.getTriggerWord() & raw::sORB) {
118  // This is the answer to an orbit trigger
119  processOrbitTrigger(ilink, localClock, decoder.getTriggerWord());
120  eventType = EventType::Standard;
121  }
122 
123  return goOn;
124 }

126 void GBTOutputHandler::addLoc(size_t ilink, const ELinkDecoder& decoder, EventType eventType, uint16_t correctedClock)
127 {
129  auto firstEntry = mData->size();
130  mData->push_back({decoder.getStatusWord(), decoder.getTriggerWord(), raw::makeUniqueLocID(crateparams::getCrateIdFromGBTUniqueId(mFeeId), decoder.getId()), decoder.getInputs()});
131  InteractionRecord intRec(mIRs[ilink].bc + correctedClock, mIRs[ilink].orbit);
132  mROFRecords->emplace_back(intRec, eventType, firstEntry, 1);
133  for (int ich = 0; ich < 4; ++ich) {
134  if ((mData->back().firedChambers & (1 << ich))) {
135  mData->back().patternsBP[ich] = decoder.getPattern(0, ich);
136  mData->back().patternsNBP[ich] = decoder.getPattern(1, ich);
137  }
138  }

166 void GBTOutputHandler::onDoneRegDebug(size_t ilink, const ELinkDecoder& decoder)
167 {
169  EventType eventType;
170  uint16_t correctedClock;
171  processTrigger(ilink, decoder, eventType, correctedClock);
172  // If we want to distinguish the two regional e-links, we can use the link ID instead
173  auto firstEntry = mData->size();
174  mData->push_back({decoder.getStatusWord(), decoder.getTriggerWord(), raw::makeUniqueLocID(crateparams::getCrateIdFromGBTUniqueId(mFeeId), ilink + 8 * (crateparams::getGBTIdInCrate(mFeeId) - 1)), decoder.getInputs()});
175 
176  auto orbit = (decoder.getTriggerWord() & raw::sORB) ? mIRs[ilink].orbit - 1 : mIRs[ilink].orbit;
177 
178  InteractionRecord intRec(mIRs[ilink].bc + correctedClock, orbit);
179  if (decoder.getTriggerWord() == 0) {
180  if (intRec.bc < mElectronicsDelay.regToLocal) {
181  // In the tests, the HB does not really correspond to a change of orbit
182  // So we need to keep track of the last clock at which the HB was received
183  // and come back to that value
184  // FIXME: Remove this part as well as mLastClock when tests are no more needed
185  intRec -= (constants::lhc::LHCMaxBunches - mLastClock[ilink] - 1);
186  }
187  // This is a self-triggered event.
188  // In this case the regional card needs to wait to receive the tracklet decision of each local
189  // which result in a delay that needs to be subtracted if we want to be able to synchronize
190  // local and regional cards for the checks
191  intRec -= mElectronicsDelay.regToLocal;
192  }
193  mROFRecords->emplace_back(intRec, eventType, firstEntry, 1);
194 }
