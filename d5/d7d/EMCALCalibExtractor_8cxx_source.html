21 boostHisto EMCALCalibExtractor::buildHitAndEnergyMean(double emin, double emax, boostHisto cellAmplitude)
22 {
23  // create the output histo
24  auto eSumHisto = boost::histogram::make_histogram(boost::histogram::axis::regular<>(100, 0, 100, "t-texp"), boost::histogram::axis::integer<>(0, 17665, "CELL ID"));
25  // The outline for this function goes as follows
26  // (1) loop over the total number of cells
27  // (2) slice the existing histogram for one cell ranging from emin to emax
28  // (3) calculate the mean of that sliced histogram using BoostHistoramUtils.h
29  // (4) fill the next histogram with this value
30  for (int cellID = 0; cellID < NCELLS; cellID++) {
31  // create a slice for each cell with energies ranging from emin to emax
32  auto tempSlice = boost::histogram::algorithm::reduce(cellAmplitude, boost::histogram::algorithm::shrink(cellID, cellID), boost::histogram::algorithm::shrink(emin, emax));
33  // calculate the mean of the slice
34  std::vector<double> result = o2::utils::fitBoostHistoWithGaus<double>(tempSlice);
35  double meanVal = result.at(1);
36  double sumVal = result.at(3);
37  //..Set the values only for cells that are not yet marked as bad
38  if (sumVal > 0.) {
39  eSumHisto(cellID, meanVal / (sumVal)); //..average energy per hit
40  }
41  }
42  return eSumHisto;
43 }

47 // This function builds the scaled hit distribution
48 // It normalizes the hits/cell to the mean value of the row and the column of the cell
49 // this is done in an iterative procedure (about 5 iterations are needed)
50 // with this procedure, the effects of the TRD and SM structures on the EMCal can be minimized
51 // The output is a boost histogram with the hits/cell as a function of cell ID.

56 boostHisto EMCALCalibExtractor::buildHitAndEnergyMeanScaled(double emin, double emax, boostHisto cellAmplitude)
57 {
58  // create the output histogram
59  auto eSumHistoScaled = boost::histogram::make_histogram(boost::histogram::axis::regular<>(100, 0, 100, "t-texp"), boost::histogram::axis::integer<>(0, 17665, "CELL ID"));
60 
61  // create a slice for each cell with energies ranging from emin to emax
62  auto hEnergyCol = boost::histogram::make_histogram(boost::histogram::axis::regular<>(100, 0, 100., "t-texp"));
63  auto hEnergyRow = boost::histogram::make_histogram(boost::histogram::axis::regular<>(250, 0, 250., "t-texp"));
64  // temp histogram used to get the scaled energies
65  auto hEnergyScaled = boost::histogram::make_histogram(boost::histogram::axis::regular<>(100, 0, 100, "t-texp"), boost::histogram::axis::integer<>(0, NCELLS, "CELL ID"));
66 
67  //...........................................
68  //start iterative process of scaling of cells
69  //...........................................
70  for (int iter = 1; iter < 5; iter++) {
71  // array of vectors for calculating the mean hits per col/row
72  std::vector<double> vecCol[100];
73  std::vector<double> vecRow[250];
74 
75  for (int cellID = 0; cellID < NCELLS; cellID++) {
76  auto tempSlice = boost::histogram::algorithm::reduce(cellAmplitude, boost::histogram::algorithm::shrink(cellID, cellID), boost::histogram::algorithm::shrink(emin, emax));
77  auto geo = Geometry::GetInstance();
78  // (0 - row, 1 - column)
79  auto position = geo->GlobalRowColFromIndex(cellID);
80  int row = std::get<0>(position);
81  int col = std::get<1>(position);
82 
83  double dCellEnergy = 0.;
84  double dNumOfHits = 0.;
85 
86  // will need to change this from 100 to a normal value for the energy axis
87  auto energyAxis = boost::histogram::axis::regular<>(100, 0, 100, "t-texp");
88  auto eMinIndex = energyAxis.index(emin);
89  auto eMaxIndex = energyAxis.index(emax);
90  for (int EBin = eMinIndex; EBin < eMaxIndex; EBin++) {
91  dCellEnergy += hEnergyScaled.at(EBin, cellID) * energyAxis.value(EBin);
92  dNumOfHits += hEnergyScaled.at(EBin, cellID);
93  }
94 
95  if (dCellEnergy > 0.) {
96  hEnergyCol(col + 0.5, dCellEnergy);
97  hEnergyRow(row + 0.5, dCellEnergy);
98  vecCol[col].push_back(dCellEnergy);
99  vecRow[row].push_back(dCellEnergy);
100  }
101  } // end loop over cells
102 
103  // Fill the histogram: mean hit per column
104  for (int iCol = 1; iCol <= 100; iCol++) {
105  if (vecCol[iCol - 1].size() > 0.) {
106  auto colAxis = boost::histogram::axis::regular<>(100, 0, 100., "t-texp");
107  auto indexCol = colAxis.index(iCol - 0.5);
108  hEnergyCol(indexCol, hEnergyCol.at(indexCol) / vecCol[iCol - 1].size());
109  }
110  }
111 
112  // Fill the histogram: mean hit per row
113  for (int iRow = 1; iRow <= 250; iRow++) {
114  if (vecRow[iRow - 1].size() > 0.) {
115  auto rowAxis = boost::histogram::axis::regular<>(250, 0, 250., "t-texp");
116  auto indexRow = rowAxis.index(iRow - 0.5);
117  hEnergyRow(indexRow, hEnergyRow.at(indexRow) / vecRow[iRow - 1].size());
118  }
119  }
120 
121  // in run2 there was now a global fit to hits per row
122  // could we just do the mean?
123  auto rowResult = o2::utils::fitBoostHistoWithGaus<double>(hEnergyRow);
124  double meanValRow = rowResult.at(1);
125 
126  auto colResult = o2::utils::fitBoostHistoWithGaus<double>(hEnergyCol);
127  double meanValCol = colResult.at(1);
128 
129  //Scale each cell by the deviation of the mean of the column and the global mean
130  for (int iCell = 0; iCell < NCELLS; iCell++) {
131  auto geo = Geometry::GetInstance();
132  // (0 - row, 1 - column)
133  auto position = geo->GlobalRowColFromIndex(iCell);
134  int col = std::get<1>(position);
135  if (hEnergyCol.at(col) > 0.) {
136  // will need to change the 100 depending on the number of energy bins we end up having
137  for (int EBin = 1; EBin < 100; EBin++) {
138  hEnergyScaled(EBin, iCell, hEnergyScaled.at(EBin, iCell) * (meanValCol / hEnergyCol.at(col)));
139  }
140  }
141  }
142 
143  //Scale each cell by the deviation of the mean of the row and the global mean
144  for (int iCell = 0; iCell < NCELLS; iCell++) {
145  auto geo = Geometry::GetInstance();
146  // (0 - row, 1 - column)
147  auto position = geo->GlobalRowColFromIndex(iCell);
148  int row = std::get<0>(position);
149  if (hEnergyRow.at(row) > 0.) {
150  // will need to change the 100 depending on the number of energy bins we end up having
151  for (int EBin = 1; EBin < 100; EBin++) {
152  hEnergyScaled(EBin, iCell, hEnergyScaled.at(EBin, iCell) * (meanValRow / hEnergyRow.at(row)));
153  }
154  }
155  }
156 
157  //....................
158  // iteration ends here
159  //....................
160 
161  } // end loop iters
162 
163  //............................................................................................
164  //..here the average hit per event and the average energy per hit is caluclated for each cell.
165  //............................................................................................
166  for (Int_t cell = 0; cell < NCELLS; cell++) {
167  Double_t Esum = 0;
168  Double_t Nsum = 0;
169 
170  for (Int_t j = 1; j <= 100; j++) {
171  auto energyAxis = boost::histogram::axis::regular<>(100, 0, 100, "t-texp");
172  Double_t E = energyAxis.value(j);
173  Double_t N = hEnergyScaled.at(j, cell);
174  if (E < emin || E > emax) {
175  continue;
176  }
177  Esum += E * N;
178  Nsum += N;
179  }
180  if (Nsum > 0.) {
181  eSumHistoScaled(cell, Esum / (Nsum)); //..average energy per hit
182  }
183  }
184 
185  return eSumHistoScaled;
186 }
