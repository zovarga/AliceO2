32 static ROFVector makeROFs(std::vector<int> binEntries, uint32_t winSize, uint32_t nBinsInOneWindow)
33 {
34  uint32_t binWidth = winSize / nBinsInOneWindow;
35  uint32_t orbit = 1;
36  uint32_t tfTime = 0;
37  size_t nBins = binEntries.size();
38  ROFVector rofRecords;
39 
40  int firstDigitIdx = 0;
41  for (int bin = 0; bin < nBins; bin++) {
42  // one ROF in each peak search bin, for simplicity
43  // skip empty bins
44  if (binEntries[bin] == 0) {
45  continue;
46  }
47  o2::InteractionRecord ir(tfTime + bin * binWidth, orbit);
48  rofRecords.emplace_back(ir, firstDigitIdx, binEntries[bin], 4);
49  firstDigitIdx += binEntries[bin];
50  }
51  return rofRecords;
52 }

55 static ROFVector makeTimeROFs(std::vector<int> binEntries, uint32_t winSize, uint32_t nBinsInOneWindow)
56 {
57  const auto& rofRecords = makeROFs(binEntries, winSize, nBinsInOneWindow);
58 
59  o2::mch::ROFTimeClusterFinder rofProcessor(rofRecords, winSize, nBinsInOneWindow, 1);
60  rofProcessor.process();
61 
62  const auto& rofTimeRecords = rofProcessor.getROFRecords();
63 
64  return rofTimeRecords;
65 }

78 //----------------------------------------------------------------------------
79 static void checkROF(const ROFRecord rof, std::vector<int> binEntries, uint32_t winSize, uint32_t nBinsInOneWindow, int start, int width)
80 {
81  uint32_t binWidth = winSize / nBinsInOneWindow;
82  uint32_t orbit = 1;
83  uint32_t tfTime = 0;
84  std::vector<int> binIntegral = getBinIntegral(binEntries);
85 
86  // checks of indexes and sizes
87  int firstIdx = binIntegral[start] - binEntries[start];
88  int size = binIntegral[start + width - 1] - firstIdx;
89  BOOST_CHECK_EQUAL(rof.getFirstIdx(), firstIdx);
90  BOOST_CHECK_EQUAL(rof.getNEntries(), size);
91 
92  // checks of interaction records and BC widths
93  o2::InteractionRecord irStart(tfTime + start * binWidth, orbit);
94  BOOST_CHECK_EQUAL(rof.getBCData(), irStart);
95  int bcWidth = (width - 1) * binWidth + 4;
96  BOOST_CHECK_EQUAL(rof.getBCWidth(), bcWidth);
97 }
