56 float TrackletTransformer::calculateY(int hcid, int column, int position)
57 {
58  double padWidth = mPadPlane->getWidthIPad();
59  int side = hcid % 2;
60  position += 1 << (NBITSTRKLPOS - 1); // shift such that position = 1 << (NBITSTRKLPOS - 1) corresponds to the MCM center
61  // slightly modified TDP eq 16.1 (appended -1 to the end to account for MCM shared pads)
62  double pad = float(position - (1 << (NBITSTRKLPOS - 1))) * GRANULARITYTRKLPOS + NCOLMCM * (4 * side + column) + 10. - 1.;
63  float y = padWidth * (pad - 72);
64 
65  return y;
66 }

77 float TrackletTransformer::calculateDy(int detector, int slope)
78 {
79  double padWidth = mPadPlane->getWidthIPad();
80 
81  float vDrift = mCalibration->getVdrift(detector);
82  float exb = mCalibration->getExB(detector);
83 
84  // dy = slope * nTimeBins * padWidth * GRANULARITYTRKLSLOPE;
85  // nTimeBins should be number of timebins in drift region. 1 timebin is 100 nanosecond
86  double rawDy = slope * ((mXCathode / vDrift) * 10.) * padWidth * GRANULARITYTRKLSLOPE / ADDBITSHIFTSLOPE;
87 
88  // NOTE: check what drift height is used in calibration code to ensure consistency
89  // NOTE: check sign convention of Lorentz angle
90  // NOTE: confirm the direction in which vDrift is measured/determined. Is it in x or in direction of drift?
91  double lorentzCorrection = TMath::Tan(exb) * mXAnode;
92 
93  // assuming angle in Bailhache, fig. 4.17 would be positive in our calibration code
94  double calibratedDy = rawDy - lorentzCorrection;
95 
96  return calibratedDy;
97 }

116 CalibratedTracklet TrackletTransformer::transformTracklet(Tracklet64 tracklet)
117 {
118  auto detector = tracklet.getDetector();
119  auto hcid = tracklet.getHCID();
120  auto padrow = tracklet.getPadRow();
121  auto column = tracklet.getColumn();
122  auto position = tracklet.getPositionBinSigned();
123  auto slope = tracklet.getSlopeBinSigned();
124 
125  // calculate raw local chamber space point
126  loadPadPlane(detector);
127  float x = getXDrift();
128  float y = calculateY(hcid, column, position);
129  float z = calculateZ(padrow);
130 
131  float dy = calculateDy(detector, slope);
132 
133  float calibratedX = calibrateX(x);
134 
135  // NOTE: Correction to y position based on x calibration NOT YET implemented. Need t0.
136 
137  std::array<float, 3> sectorSpacePoint = transformL2T(detector, std::array<double, 3>{calibratedX, y, z});
138 
139  LOG(debug) << "x: " << sectorSpacePoint[0] << " | "
140  << "y: " << sectorSpacePoint[1] << " | "
141  << "z: " << sectorSpacePoint[2];
142 
143  return CalibratedTracklet(sectorSpacePoint[0], sectorSpacePoint[1], sectorSpacePoint[2], dy);
144 }

146 double TrackletTransformer::getTimebin(int detector, double x)
147 {
148  // calculate timebin from x position within chamber
149  float vDrift = mCalibration->getVdrift(detector);
150  double t0 = 4.0; // time (in timebins) of start of drift region
151 
152  double timebin;
153  // x = 0 at anode plane and points toward pad plane.
154  if (x < -mGeo->camHght() / 2) {
155  // drift region
156  timebin = t0 - (x + mGeo->camHght() / 2) / (vDrift * 0.1);
157  } else {
158  // anode region: very rough guess
159  timebin = t0 - 1.0 + fabs(x);
160  }
161 
162  return timebin;
163 }
