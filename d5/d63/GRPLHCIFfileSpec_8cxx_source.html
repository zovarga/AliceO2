46 void GRPLHCIFfileProcessor::run(o2::framework::ProcessingContext& pc)
47 {
48  auto configBuff = pc.inputs().get<gsl::span<char>>("inputConfig");
49  auto configFileName = pc.inputs().get<std::string>("inputConfigFileName");
50  auto timer = std::chrono::duration_cast<std::chrono::milliseconds>(HighResClock::now().time_since_epoch()).count();
51  LOG(info) << "got input file " << configFileName << " of size " << configBuff.size();
52  mReader.loadLHCIFfile(configBuff);
53  std::vector<std::pair<long, std::vector<int32_t>>> beamEnergy;
54  std::vector<std::pair<long, std::vector<std::string>>> injScheme;
55  std::vector<std::pair<long, std::vector<std::string>>> fillNb;
56  std::vector<std::pair<long, std::vector<int32_t>>> atomicNbB1;
57  std::vector<std::pair<long, std::vector<int32_t>>> atomicNbB2;
58  std::vector<std::pair<long, std::vector<o2::units::AngleRad_t>>> crossAngle;
59  std::vector<std::pair<long, std::vector<int32_t>>> bunchConfigB1;
60  std::vector<std::pair<long, std::vector<int32_t>>> bunchConfigB2;
61 
62  int nEle = 0;
63  int nMeas = 0;
64  std::string type{""};
65 
66  GRPLHCIFData lhcifdata;
67 
68  // Beam Energy
69  mReader.readValue<int32_t>("BEAM_ENERGY", type, nEle, nMeas, beamEnergy);
70  if (nMeas == 0) {
71  LOG(fatal) << "Beam energy not present";
72  }
73  if (nEle != 1 || nMeas != 1) {
74  LOG(error) << "More than one value/measurement found for Beam Energy, keeping the last one";
75  }
76  LOG(info) << "beam energy size = " << beamEnergy.size();
77  lhcifdata.setBeamEnergyPerZWithTime(beamEnergy.back().first, beamEnergy.back().second.back());
78 
79  // Injection scheme
80  mReader.readValue<std::string>("INJECTION_SCHEME", type, nEle, nMeas, injScheme);
81  if (nMeas == 0) {
82  LOG(fatal) << "Injection scheme not present";
83  }
84  if (nEle != 1 || nMeas != 1) {
85  LOG(error) << "More than one value/measurement found for Injection Scheme, keeping the last one";
86  }
87  lhcifdata.setInjectionSchemeWithTime(injScheme.back().first, injScheme.back().second.back());
88 
89  // fill number
90  mReader.readValue<std::string>("FILL_NUMBER", type, nEle, nMeas, fillNb);
91  if (nMeas == 0) {
92  LOG(fatal) << "Fill Number not present";
93  }
94  if (nEle != 1 || nMeas != 1) {
95  LOG(error) << "More than one value/measurement found for Fill Number, keeping the last one";
96  }
97  lhcifdata.setFillNumberWithTime(fillNb.back().first, atoi(fillNb.back().second.back().c_str()));
98 
99  // Atomic Number (Z) for B1
100  mReader.readValue<int32_t>("ATOMIC_NUMBER_B1", type, nEle, nMeas, atomicNbB1);
101  if (nMeas == 0) {
102  LOG(fatal) << "Atomic Number Beam 1 (clockwise) not present";
103  }
104  if (nEle != 1 || nMeas != 1) {
105  LOG(error) << "More than one value/measurement found for Atomic Number Beam 1 (clockwise), keeping the last one";
106  }
107  lhcifdata.setAtomicNumberB1WithTime(atomicNbB1.back().first, atomicNbB1.back().second.back());
108 
109  // Atomic Number (Z) for B2
110  mReader.readValue<int32_t>("ATOMIC_NUMBER_B2", type, nEle, nMeas, atomicNbB2);
111  if (nMeas == 0) {
112  LOG(fatal) << "Atomic Number Beam 2 (anticlockwise) not present";
113  }
114  if (nEle != 1 || nMeas != 1) {
115  LOG(error) << "More than one value/measurement found for Atomic Number Beam 2 (anticlockwise), keeping the last one";
116  }
117  lhcifdata.setAtomicNumberB2WithTime(atomicNbB2.back().first, atomicNbB2.back().second.back());
118 
119  // Crossing Angle
120  mReader.readValue<float>("IP2_XING_V_MURAD", type, nEle, nMeas, crossAngle);
121  if (nMeas == 0) {
122  LOG(fatal) << "Crossing Angle not present";
123  }
124  if (nEle != 1 || nMeas != 1) {
125  LOG(error) << "More than one value/measurement found for Crossing Angle, keeping the last one";
126  }
127  lhcifdata.setCrossingAngleWithTime(crossAngle.back().first, crossAngle.back().second.back());
128 
129  // Bunch Config for B1
130  mReader.readValue<int>("CIRCULATING_BUNCH_CONFIG_BEAM1", type, nEle, nMeas, bunchConfigB1);
131  if (nMeas == 0) {
132  LOG(fatal) << "Bunch Config Beam 1 not present";
133  }
134  if (nEle != 1 || nMeas != 1) {
135  LOG(error) << "More than one value/measurement found for Bunch Config Beam 1, keeping the last one";
136  }
137 
138  // Bunch Config for B2
139  mReader.readValue<int>("CIRCULATING_BUNCH_CONFIG_BEAM2", type, nEle, nMeas, bunchConfigB2);
140  if (nMeas == 0) {
141  LOG(fatal) << "Bunch Config Beam 2 not present";
142  }
143  if (nEle != 1 || nMeas != 1) {
144  LOG(error) << "More than one value/measurement found for Bunch Config Beam 2, keeping the last one";
145  }
146  if (nMeas != o2::constants::lhc::LHCMaxBunches) {
147  LOG(error) << "We don't have the right number of bunches information for Beam 2";
148  }
149 
150  // Building Bunch Filling
151  std::vector<int32_t> bcNumbersB1, bcNumbersB2;
152  lhcifdata.translateBucketsToBCNumbers(bcNumbersB1, bunchConfigB1.back().second, 1);
153  lhcifdata.translateBucketsToBCNumbers(bcNumbersB2, bunchConfigB2.back().second, 2);
154  o2::BunchFilling bunchFilling;
155  for (int i = 0; i < bcNumbersB1.size(); ++i) {
156  int bc = bcNumbersB1[i];
157  int val = (bc == 0 ? 0 : 1);
158  bunchFilling.setBC(bc, val, 0);
159  }
160  for (int i = 0; i < bcNumbersB2.size(); ++i) {
161  int bc = bcNumbersB2[i];
162  int val = (bc == 0 ? 0 : 1);
163  bunchFilling.setBC(bc, val, 1);
164  }
165  bunchFilling.setInteractingBCsFromBeams();
166 
167  lhcifdata.setBunchFillingWithTime((bunchConfigB1.back().first + bunchConfigB2.back().first) / 2, bunchFilling);
168 
169  if (mVerbose) {
170  LOG(info) << " **** Beam Energy ****";
171  for (auto& el : beamEnergy) {
172  for (auto elVect : el.second) {
173  std::cout << el.first << " --> " << elVect << std::endl;
174  }
175  }
176  LOG(info) << " **** Injection Scheme ****";
177  for (auto& el : injScheme) {
178  for (auto elVect : el.second) {
179  std::cout << el.first << " --> " << elVect << std::endl;
180  }
181  }
182  LOG(info) << " **** Fill Number ****";
183  for (auto& el : fillNb) {
184  for (auto elVect : el.second) {
185  std::cout << el.first << " --> " << elVect << std::endl;
186  }
187  }
188  LOG(info) << " **** Atomic Number Beam 1 (clockwise) ****";
189  for (auto& el : atomicNbB1) {
190  for (auto elVect : el.second) {
191  std::cout << el.first << " --> " << elVect << std::endl;
192  }
193  }
194  LOG(info) << " **** Atomic Number B2 (anticlockwise) ****";
195  for (auto& el : atomicNbB2) {
196  for (auto elVect : el.second) {
197  std::cout << el.first << " --> " << elVect << std::endl;
198  }
199  }
200  LOG(info) << " **** Crossing Angle ****";
201  for (auto& el : crossAngle) {
202  for (auto elVect : el.second) {
203  std::cout << el.first << " --> " << elVect << std::endl;
204  }
205  }
206  }
207 
208  sendOutput(pc.outputs(), timer, lhcifdata);
209 }

219 void GRPLHCIFfileProcessor::sendOutput(DataAllocator& output, long start, const GRPLHCIFData& lhcifdata)
220 {
221  // sending output to CCDB
222 
223  using clbUtils = o2::calibration::Utils;
224  auto clName = o2::utils::MemFileHelper::getClassName(lhcifdata);
225  auto flName = o2::ccdb::CcdbApi::generateFileName(clName);
226  std::map<std::string, std::string> md;
227  md.emplace("created by", "dpl");
228  o2::ccdb::CcdbObjectInfo info("GLO/Config/GRPLHCIFData", clName, flName, md, start, o2::calibration::Utils::INFINITE_TIME);
229  auto image = o2::ccdb::CcdbApi::createObjectImage(&lhcifdata, &info);
230  LOG(info) << "Sending object " << info.getPath() << "/" << info.getFileName() << " of size " << image->size()
231  << " bytes, valid for " << info.getStartValidityTimestamp() << " : " << info.getEndValidityTimestamp();
232  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "GRP_LHCIFData", 0}, *image.get()); // vector<char>
233  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "GRP_LHCIFData", 0}, info); // root-serialized
234 }

249  outputs,
