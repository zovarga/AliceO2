54 void DPLMonitoringBackend::send(o2::monitoring::Metric const& metric)
55 {
56  std::array<char, 4096> buffer;
57  auto mStream = fmt::format_to(buffer.begin(), "[METRIC] {}", metric.getName());
58  for (auto& value : metric.getValues()) {
59  auto stringValue = std::visit(overloaded{
60  [](const std::string& value) -> std::string { return value; },
61  [](auto value) -> std::string { return std::to_string(value); }},
62  value.second);
63  if (metric.getValuesSize() == 1) {
64  mStream = fmt::format_to(mStream, ",{} {}", metric.getFirstValueType(), stringValue);
65  } else {
66  mStream = fmt::format_to(mStream, " {}={}", value.first, stringValue);
67  }
68  }
69  // FIXME: tags are ignored by the DPL backend in any case...
70  mStream = fmt::format_to(mStream, " {} {}", convertTimestamp(metric.getTimestamp()), mTagString);
71 
72  bool first = mTagString.empty();
73  for (const auto& [key, value] : metric.getTags()) {
74  if (!first) {
75  mStream = fmt::format_to(mStream, ",");
76  }
77  first = true;
78  mStream = fmt::format_to(mStream, "{}={}", o2::monitoring::tags::TAG_KEY[key], o2::monitoring::tags::GetValue(value));
79  }
80  mStream = fmt::format_to(mStream, "\n");
81  auto size = std::distance(buffer.begin(), mStream);
82  if (size + 1 >= 4096) {
83  throw runtime_error_f("Metric too long");
84  }
85  buffer[size] = '\0';
86  mRegistry.get<framework::DriverClient>().tell(buffer.data(), size);
87 }
