39 
40  if (integrationIntervals <= 0) {

43 
44  std::vector<float> idcOneA = mIDCOne.mIDCOne[0];

123 void o2::tpc::IDCFactorization::calcIDCZero(const bool norm)
124 {
125  const unsigned int nIDCsSide = mNIDCsPerSector * o2::tpc::SECTORSPERSIDE;
126  mIDCZero.clear();
127  mIDCZero.resize(nIDCsSide);
128 
129 #pragma omp parallel for num_threads(sNThreads)
130  for (unsigned int cru = 0; cru < mIDCs.size(); ++cru) {
131  const o2::tpc::CRU cruTmp(cru);
132  const unsigned int region = cruTmp.region();
133  const auto factorIndexGlob = mRegionOffs[region] + mNIDCsPerSector * cruTmp.sector();
134  for (unsigned int timeframe = 0; timeframe < mTimeFrames; ++timeframe) {
135  for (unsigned int idcs = 0; idcs < mIDCs[cru][timeframe].size(); ++idcs) {
136  const unsigned int indexGlob = (idcs % mNIDCsPerCRU[region]) + factorIndexGlob;
137  if (norm) {
138  mIDCs[cru][timeframe][idcs] *= Mapper::INVPADAREA[region];
139  }
140  mIDCZero.fillValueIDCZero(mIDCs[cru][timeframe][idcs], cruTmp.side(), indexGlob % nIDCsSide);
141  }
142  }
143  }
144  std::transform(mIDCZero.mIDCZero[Side::A].begin(), mIDCZero.mIDCZero[Side::A].end(), mIDCZero.mIDCZero[Side::A].begin(), [normVal = getNIntegrationIntervals()](auto& val) { return val / normVal; });
145  std::transform(mIDCZero.mIDCZero[Side::C].begin(), mIDCZero.mIDCZero[Side::C].end(), mIDCZero.mIDCZero[Side::C].begin(), [normVal = getNIntegrationIntervals()](auto& val) { return val / normVal; });
146 }

148 void o2::tpc::IDCFactorization::calcIDCOne()
149 {
150  const unsigned int nIDCsSide = mNIDCsPerSector * SECTORSPERSIDE;
151  const unsigned int integrationIntervals = getNIntegrationIntervals();
152  mIDCOne.clear();
153  mIDCOne.resize(integrationIntervals);
154  const unsigned int crusPerSide = Mapper::NREGIONS * SECTORSPERSIDE;
155 
156 #pragma omp parallel for num_threads(sNThreads)
157  for (unsigned int cru = 0; cru < mIDCs.size(); ++cru) {
158  const o2::tpc::CRU cruTmp(cru);
159  const unsigned int region = cruTmp.region();
160  const auto factorIDCOne = crusPerSide * mNIDCsPerCRU[region];
161  const auto side = cruTmp.side();
162  const auto factorIndexGlob = mRegionOffs[region] + mNIDCsPerSector * cruTmp.sector();
163  unsigned int integrationIntervallast = 0;
164  for (unsigned int timeframe = 0; timeframe < mTimeFrames; ++timeframe) {
165  for (unsigned int idcs = 0; idcs < mIDCs[cru][timeframe].size(); ++idcs) {
166  const unsigned int integrationInterval = idcs / mNIDCsPerCRU[region] + integrationIntervallast;
167  const unsigned int indexGlob = (idcs % mNIDCsPerCRU[region]) + factorIndexGlob;
168  const auto idcZeroVal = mIDCZero.mIDCZero[side][indexGlob % nIDCsSide];
169  if (idcZeroVal) {
170  mIDCOne.mIDCOne[side][integrationInterval] += mIDCs[cru][timeframe][idcs] / (factorIDCOne * idcZeroVal);
171  }
172  }
173  integrationIntervallast += mIDCs[cru][timeframe].size() / mNIDCsPerCRU[region];
174  }
175  }
176 }

178 void o2::tpc::IDCFactorization::calcIDCDelta()
179 {
180  const unsigned int nIDCsSide = mNIDCsPerSector * SECTORSPERSIDE;
181  for (unsigned int i = 0; i < getNChunks(); ++i) {
182  const auto idcsSide = nIDCsSide * getNIntegrationIntervals(i);
183  mIDCDelta[i].getIDCDelta(Side::A).resize(idcsSide);
184  mIDCDelta[i].getIDCDelta(Side::C).resize(idcsSide);
185  }
186 
187 #pragma omp parallel for num_threads(sNThreads)
188  for (unsigned int cru = 0; cru < mIDCs.size(); ++cru) {
189  const o2::tpc::CRU cruTmp(cru);
190  const unsigned int region = cruTmp.region();
191  const auto side = cruTmp.side();
192  const auto factorIndexGlob = mRegionOffs[region] + mNIDCsPerSector * cruTmp.sector();
193  unsigned int integrationIntervallast = 0;
194  unsigned int integrationIntervallastLocal = 0;
195  unsigned int lastChunk = 0;
196 
197  for (unsigned int timeframe = 0; timeframe < mTimeFrames; ++timeframe) {
198  const unsigned int chunk = getChunk(timeframe);
199  if (lastChunk != chunk) {
200  integrationIntervallastLocal = 0;
201  }
202 
203  for (unsigned int idcs = 0; idcs < mIDCs[cru][timeframe].size(); ++idcs) {
204  const unsigned int intervallocal = idcs / mNIDCsPerCRU[region];
205  const unsigned int integrationIntervalGlobal = intervallocal + integrationIntervallast;
206  const unsigned int integrationIntervalLocal = intervallocal + integrationIntervallastLocal;
207  const unsigned int indexGlob = (idcs % mNIDCsPerCRU[region]) + factorIndexGlob;
208  const unsigned int indexGlobMod = indexGlob % nIDCsSide;
209  const auto idcZero = mIDCZero.mIDCZero[side][indexGlobMod];
210  const auto idcOne = mIDCOne.mIDCOne[side][integrationIntervalGlobal];
211  const auto mult = idcZero * idcOne;
212  const auto val = (mult > 0) ? mIDCs[cru][timeframe][idcs] / mult : 0;
213  mIDCDelta[chunk].setIDCDelta(side, indexGlobMod + integrationIntervalLocal * nIDCsSide, val - 1.f);
214  }
215 
216  const unsigned int intervals = mIDCs[cru][timeframe].size() / mNIDCsPerCRU[region];
217  integrationIntervallast += intervals;
218  integrationIntervallastLocal += intervals;
219  lastChunk = chunk;
220  }
221  }
222 }

225 {
226  unsigned int timeFrame = 0;
227  unsigned int interval = 0;
228  getTF(region, integrationInterval, timeFrame, interval);
229  if (mIDCs[sector * Mapper::NREGIONS + region][timeFrame].empty()) {
230  return 0.f;
231  }
232 
233  const int index = interval * mNIDCsPerCRU[region] + mOffsRow[region][getGroupedRow(region, urow)] + getGroupedPad(region, urow, upad) + getOffsetForEdgePad(upad, urow, region);
234  return mIDCs[sector * Mapper::NREGIONS + region][timeFrame][index];

238 {

279 void o2::tpc::IDCFactorization::getLocalIntegrationInterval(const unsigned int region, const unsigned int integrationInterval, unsigned int& chunk, unsigned int& localintegrationInterval) const
280 {
281  unsigned int nintervals = 0;
282  unsigned int nitervalsChunk = 0;
283  unsigned int globalTF = 0;
284  for (unsigned int ichunk = 0; ichunk < getNChunks(); ++ichunk) {
285  const auto nTFsPerChunk = getNTFsPerChunk(ichunk);
286  for (unsigned int tf = 0; tf < nTFsPerChunk; ++tf) {
287  nintervals += mIDCs[region][globalTF].size() / mNIDCsPerCRU[region];
288  if (integrationInterval < nintervals) {
289  chunk = getChunk(globalTF);
290  localintegrationInterval = integrationInterval - nitervalsChunk;
291  return;
292  }
293  ++globalTF;
294  }
295  nitervalsChunk = nintervals;
296  }
