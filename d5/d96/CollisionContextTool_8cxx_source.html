49 struct InteractionSpec {

51  float interactionRate;

59 InteractionSpec parseInteractionSpec(std::string const& specifier, std::vector<InteractionSpec> const& existingPatterns)
60 {
61  // An interaction specification is a command-separated string
62  // of the following form:
63  // SPEC=NAMESTRING,INTERACTIONSTRING[,MCNUMBERSTRING]
64  //
65  // where
66  //
67  // NAMESTRING : a simple named specifier for the interaction; matching to a simulation prefix used by o2-sim
68  //
69  // INTERACTIONSTRING: irate | @ID:[ed]FLOATVALUE
70  // - either: a simple number irate specifying the interaction rate in kHz
71  // - or: a string such as @0:e5, saying that this interaction should match/sync
72  // with collisions of the 0-th interaction, but inject only every 5 collisions.
73  // Alternatively @0:d10000 means to inject but leaving a timedistance of at least 10000ns between signals
74  //
75  // MCNUMBERSTRING: NUMBER1:r?NUMBER2 can specify how many collisions NUMBER1 to produce, taking from a sample of NUMBER2 available collisions
76  // - this option is only supported on the first interaction which is supposed to be the background interaction
77  // - if the 'r' character is present we randomize the order of the MC events
78 
79  // tokens are separated by comma
80  std::vector<std::string> tokens = o2::RangeTokenizer::tokenize<std::string>(specifier);
81 
82  float rate = -1.;
83  std::pair<int, float> synconto(-1, 1);
84 
85  // extract name
86  std::string name = tokens[0];
87 
88  // extract the MC number spec if given
89  int collisionsasked = -1;
90  int collisionsavail = -1;
91  bool randomizeorder = false;
92  if (tokens.size() > 2) {
93  auto mctoken = tokens[2];
94  std::regex re("([0-9]*):(r?)([0-9]*)$", std::regex_constants::extended);
95 
96  std::cmatch m;
97  if (std::regex_match(mctoken.c_str(), m, re)) {
98  collisionsasked = std::atoi(m[1].str().c_str());
99  if (m[2].str().compare("r") == 0) {
100  randomizeorder = true;
101  }
102  collisionsavail = std::atoi(m[3].str().c_str());
103  } else {
104  LOG(error) << "Could not parse " << mctoken << " as MCNUMBERSTRING";
105  exit(1);
106  }
107  }
108 
109  // extract interaction rate ... or locking
110  auto& interactionToken = tokens[1];
111  if (interactionToken[0] == '@') {
112  try {
113  // locking onto some other interaction
114  std::regex re("@([0-9]*):([ed])([0-9]*[.]?[0-9]?)$", std::regex_constants::extended);
115 
116  std::cmatch m;
117  if (std::regex_match(interactionToken.c_str(), m, re)) {
118  auto crossindex = std::atoi(m[1].str().c_str());
119  auto mode = m[2].str();
120  auto modevalue = std::atof(m[3].str().c_str());
121 
122  if (crossindex > existingPatterns.size()) {
123  LOG(error) << "Reference to non-existent interaction spec";
124  exit(1);
125  }
126  synconto = std::pair<int, float>(crossindex, modevalue);
127 
128  InteractionLockMode lockMode;
129  if (mode.compare("e") == 0) {
130  lockMode = InteractionLockMode::EVERYN;
131  }
132  if (mode.compare("d") == 0) {
133  lockMode = InteractionLockMode::MINTIMEDISTANCE;
134  }
135  return InteractionSpec{name, rate, synconto, lockMode, collisionsasked, collisionsavail, randomizeorder};
136  } else {
137  LOG(error) << "Could not parse " << interactionToken << " as INTERACTIONSTRING";
138  exit(1);
139  }
140  } catch (std::regex_error e) {
141  LOG(error) << "Exception during regular expression match " << e.what();
142  exit(1);
143  }
144  } else {
145  rate = std::atof(interactionToken.c_str());
146  return InteractionSpec{name, rate, synconto, InteractionLockMode::NOLOCK, collisionsasked, collisionsavail, randomizeorder};
147  }
148 }

150 bool parseOptions(int argc, char* argv[], Options& optvalues)
151 {
152  namespace bpo = boost::program_options;
153  bpo::options_description options(
154  "A utility to create and manipulate digitization contexts (MC collision structure within a timeframe).\n\n"
155  "Allowed options");
156 
157  options.add_options()(
158  "interactions,i", bpo::value<std::vector<std::string>>(&optvalues.interactionRates)->multitoken(), "name,IRate|LockSpecifier")(
159  "outfile,o", bpo::value<std::string>(&optvalues.outfilename)->default_value("collisioncontext.root"), "Outfile of collision context")(
160  "tflength", bpo::value<double>(&optvalues.timeframelengthinMS)->default_value(-1.),
161  "Length of timeframe to generate in ms (if given). "
162  "Otherwise, the context will be generated by using collision numbers from the interaction specification.")(
163  "seed", bpo::value<long>(&optvalues.seed)->default_value(0L), "Seed for random number generator (for time sampling etc). Default 0: Random")(
164  "show-context", "Print generated collision context to terminal.")(
165  "bcPatternFile", bpo::value<std::string>(&optvalues.bcpatternfile)->default_value(""), "Interacting BC pattern file (e.g. from CreateBCPattern.C)")(
166  "orbitsPerTF", bpo::value<int>(&optvalues.orbitsPerTF)->default_value(256), "Orbits per timeframes")(
167  "timeframeID", bpo::value<int>(&optvalues.tfid)->default_value(0), "Timeframe id of this context. Allows to generate contexts for different start orbits");
168 
169  options.add_options()("help,h", "Produce help message.");
170 
171  bpo::variables_map vm;
172  try {
173  bpo::store(bpo::command_line_parser(argc, argv).options(options).run(), vm);
174  bpo::notify(vm);
175 
176  // help
177  if (vm.count("help")) {
178  std::cout << options << std::endl;
179  return false;
180  }
181  if (vm.count("show-context")) {
182  optvalues.printContext = true;
183  }
184 
185  } catch (const bpo::error& e) {
186  std::cerr << e.what() << "\n\n";
187  std::cerr << "Error parsing options; Available options:\n";
188  std::cerr << options << std::endl;
189  return false;
190  }
191  return true;
192 }

194 int main(int argc, char* argv[])
195 {
196  Options options;
197  if (!parseOptions(argc, argv, options)) {
198  exit(1);
199  }
200 
201  // init random generator
202  gRandom->SetSeed(options.seed);
203 
204  std::vector<InteractionSpec> ispecs;
205  // building the interaction spec
206  for (auto& i : options.interactionRates) {
207  // this is created as output from
208  ispecs.push_back(parseInteractionSpec(i, ispecs));
209  }
210 
211  std::vector<std::pair<o2::InteractionTimeRecord, std::vector<o2::steer::EventPart>>> collisions;
212  std::vector<o2::BunchFilling> bunchFillings; // vector of bunch filling objects; generated by interaction samplers
213 
214  // now we generate the collision structure (interaction type by interaction type)
215  bool usetimeframelength = options.timeframelengthinMS > 0;
216 
217  for (int id = 0; id < ispecs.size(); ++id) {
218  auto mode = ispecs[id].syncmode;
219  if (mode == InteractionLockMode::NOLOCK) {
220  o2::steer::InteractionSampler sampler;
221  sampler.setInteractionRate(ispecs[id].interactionRate);
222  if (!options.bcpatternfile.empty()) {
223  sampler.setBunchFilling(options.bcpatternfile);
224  }
225  sampler.setFirstIR(o2::InteractionRecord(0, options.tfid * options.orbitsPerTF));
226  sampler.init();
227  o2::InteractionTimeRecord record;
228  int count = 0;
229  do {
230  record = sampler.generateCollisionTime();
231  std::vector<o2::steer::EventPart> parts;
232  parts.emplace_back(id, count);
233 
234  std::pair<o2::InteractionTimeRecord, std::vector<o2::steer::EventPart>> insertvalue(record, parts);
235  auto iter = std::lower_bound(collisions.begin(), collisions.end(), insertvalue, [](std::pair<o2::InteractionTimeRecord, std::vector<o2::steer::EventPart>> const& a, std::pair<o2::InteractionTimeRecord, std::vector<o2::steer::EventPart>> const& b) { return a.first < b.first; });
236  collisions.insert(iter, insertvalue);
237  count++;
238  } while ((usetimeframelength && record.getTimeNS() < options.timeframelengthinMS * 1000 * 1000) || (ispecs[id].mcnumberasked > 0 && count < ispecs[id].mcnumberasked));
239 
240  // we support randomization etc on non-injected/embedded interactions
241  // and we can apply them here
242  auto random_shuffle = [](auto first, auto last) {
243  auto n = last - first;
244  for (auto i = n - 1; i > 0; --i) {
245  using std::swap;
246  swap(first[i], first[(int)(gRandom->Rndm() * n)]);
247  }
248  };
249  std::vector<int> eventindices(count);
250  std::iota(eventindices.begin(), eventindices.end(), 0);
251  // apply randomization of order if any
252  if (ispecs[id].randomizeorder) {
253  random_shuffle(eventindices.begin(), eventindices.end());
254  }
255  if (ispecs[id].mcnumberavail > 0) {
256  // apply cutting to number of available entries
257  for (auto& e : eventindices) {
258  e = e % ispecs[id].mcnumberavail;
259  }
260  }
261  // make these transformations final:
262  for (auto& col : collisions) {
263  for (auto& part : col.second) {
264  if (part.sourceID == id) {
265  part.entryID = eventindices[part.entryID];
266  }
267  }
268  }
269 
270  // keep bunch filling information produced by these samplers
271  bunchFillings.push_back(sampler.getBunchFilling());
272 
273  } else {
274  // we are in some lock/sync mode and modify existing collisions
275  int lastcol = -1;
276  double lastcoltime = -1.;
277  auto distanceval = ispecs[id].synconto.second;
278  auto lockonto = ispecs[id].synconto.first;
279  int eventcount = 0;
280 
281  for (int colid = 0; colid < collisions.size(); ++colid) {
282  auto& col = collisions[colid];
283  auto coltime = col.first.getTimeNS();
284 
285  bool rightinteraction = false;
286  // we are locking only on collisions which have the referenced interaction present
287  // --> there must be an EventPart with the right sourceID
288  for (auto& eventPart : col.second) {
289  if (eventPart.sourceID == lockonto) {
290  rightinteraction = true;
291  break;
292  }
293  }
294  if (!rightinteraction) {
295  continue;
296  }
297 
298  bool inject = false;
299  // we always start with first one
300  if (lastcol == -1) {
301  inject = true;
302  }
303  if (mode == InteractionLockMode::EVERYN && (colid - lastcol) >= distanceval) {
304  inject = true;
305  }
306  if (mode == InteractionLockMode::MINTIMEDISTANCE && (coltime - lastcoltime) >= distanceval) {
307  inject = true;
308  }
309 
310  if (inject) {
311  col.second.emplace_back(id, eventcount++);
312  lastcol = colid;
313  lastcoltime = coltime;
314  }
315  }
316  }
317  }
318 
319  // create DigitizationContext
320  o2::steer::DigitizationContext digicontext;
321  // we can fill this container
322  auto& parts = digicontext.getEventParts();
323  // we can fill this container
324  auto& records = digicontext.getEventRecords();
325  // copy over information
326  size_t maxParts = 0;
327  for (auto& p : collisions) {
328  records.push_back(p.first);
329  parts.push_back(p.second);
330  maxParts = std::max(p.second.size(), maxParts);
331  }
332  digicontext.setNCollisions(collisions.size());
333  digicontext.setMaxNumberParts(maxParts);
334  // merge bunch filling info
335  for (int i = 1; i < bunchFillings.size(); ++i) {
336  bunchFillings[0].mergeWith(bunchFillings[i]);
337  }
338  digicontext.setBunchFilling(bunchFillings[0]);
339  std::vector<std::string> prefixes;
340  for (auto& p : ispecs) {
341  prefixes.push_back(p.name);
342  }
343  digicontext.setSimPrefixes(prefixes);
344  if (options.printContext) {
345  digicontext.printCollisionSummary();
346  }
347  digicontext.saveToFile(options.outfilename);
348 
349  return 0;
350 }
