26 // Define error function for ROOT fitting
27 double erf(double* xx, double* par)
28 {
29  return (N_INJ / 2) * TMath::Erf((xx[0] - par[0]) / (sqrt(2) * par[1])) + (N_INJ / 2);
30 }
31 
32 // ITHR erf is reversed
33 double erf_ithr(double* xx, double* par)
34 {
35  return (N_INJ / 2) * (1 - TMath::Erf((xx[0] - par[0]) / (sqrt(2) * par[1])));
36 }

170 // Returns upper / lower limits for threshold determination.
171 // data is the number of trigger counts per charge injected;
172 // x is the array of charge injected values;
173 // NPoints is the length of both arrays.
174 bool ITSThresholdCalibrator::findUpperLower(
175  const char* data, const short int* x, const short int& NPoints,
176  short int& lower, short int& upper, bool flip)
177 {
178  // Initialize (or re-initialize) upper and lower
179  upper = -1;
180  lower = -1;
181 
182  if (flip) { // ITHR case. lower is at large mX[i], upper is at small mX[i]
183 
184  for (int i = 0; i < NPoints; i++) {
185  if (data[i] == 0) {
186  upper = i;
187  break;
188  }
189  }
190 
191  if (upper == -1) {
192  return false;
193  }
194  for (int i = upper; i > 0; i--) {
195  if (data[i] >= N_INJ) {
196  lower = i;
197  break;
198  }
199  }
200 
201  } else { // not flipped
202 
203  for (int i = 0; i < NPoints; i++) {
204  if (data[i] >= N_INJ) {
205  upper = i;
206  break;
207  }
208  }
209 
210  if (upper == -1) {
211  return false;
212  }
213  for (int i = upper; i > 0; i--) {
214  if (data[i] == 0) {
215  lower = i;
216  break;
217  }
218  }
219  }
220 
221  // If search was successful, return central x value
222  if ((lower == -1) || (upper < lower)) {
223  return false;
224  }
225  return true;
226 }

229 // Main findThreshold function which calls one of the three methods
230 bool ITSThresholdCalibrator::findThreshold(
231  const char* data, const short int* x, const short int& NPoints,
232  float& thresh, float& noise)
233 {
234  bool success = false;
235 
236  switch (this->mFitType) {
237  case DERIVATIVE: // Derivative method
238  success = this->findThresholdDerivative(data, x, NPoints, thresh, noise);
239  break;
240 
241  case FIT: // Fit method
242  success = this->findThresholdFit(data, x, NPoints, thresh, noise);
243  break;
244 
245  case HITCOUNTING: // Hit-counting method
246  success = this->findThresholdHitcounting(data, x, NPoints, thresh);
247  // noise = 0;
248  break;
249  }
250 
251  return success;
252 }

255 // Use ROOT to find the threshold and noise via S-curve fit
256 // data is the number of trigger counts per charge injected;
257 // x is the array of charge injected values;
258 // NPoints is the length of both arrays.
259 // thresh, noise, chi2 pointers are updated with results from the fit
260 bool ITSThresholdCalibrator::findThresholdFit(
261  const char* data, const short int* x, const short int& NPoints,
262  float& thresh, float& noise)
263 {
264  // Find lower & upper values of the S-curve region
265  short int lower, upper;
266  bool flip = (this->mScanType == 'I');
267  if (!this->findUpperLower(data, x, NPoints, lower, upper, flip) || lower == upper) {
268  if (this->mVerboseOutput) {
269  LOG(warning) << "Start-finding unsuccessful: (lower, upper) = ("
270  << lower << ", " << upper << ")";
271  }
272  return false;
273  }
274  float start = (this->mX[upper] + this->mX[lower]) / 2;
275 
276  if (start < 0) {
277  if (this->mVerboseOutput) {
278  LOG(warning) << "Start-finding unsuccessful: Start = " << start;
279  }
280  return false;
281  }
282 
283  for (int i = 0; i < NPoints; i++) {
284  this->mFitHist->SetBinContent(i + 1, data[i]);
285  }
286 
287  // Initialize starting parameters
288  this->mFitFunction->SetParameter(0, start);
289  this->mFitFunction->SetParameter(1, 8);
290 
291  this->mFitHist->Fit("mFitFunction", "QL");
292 
293  noise = this->mFitFunction->GetParameter(1);
294  thresh = this->mFitFunction->GetParameter(0);
295  float chi2 = this->mFitFunction->GetChisquare() / this->mFitFunction->GetNDF();
296 
297  // Clean up histogram for next time it is used
298  this->mFitHist->Reset();
299 
300  return (chi2 < 5 && noise < 15);
301 }

304 // Use ROOT to find the threshold and noise via derivative method
305 // data is the number of trigger counts per charge injected;
306 // x is the array of charge injected values;
307 // NPoints is the length of both arrays.
308 bool ITSThresholdCalibrator::findThresholdDerivative(
309  const char* data, const short int* x, const short int& NPoints,
310  float& thresh, float& noise)
311 {
312  // Find lower & upper values of the S-curve region
313  short int lower, upper;
314  bool flip = (this->mScanType == 'I');
315  if (!this->findUpperLower(data, x, NPoints, lower, upper, flip) || lower == upper) {
316  if (this->mVerboseOutput) {
317  LOG(warning) << "Start-finding unsuccessful: (lower, upper) = ("
318  << lower << ", " << upper << ")";
319  }
320  return false;
321  }
322 
323  int deriv_size = upper - lower;
324  float deriv[deriv_size];
325  float xfx = 0, fx = 0;
326 
327  // Fill array with derivatives
328  for (int i = lower; i < upper; i++) {
329  deriv[i - lower] = (data[i + 1] - data[i]) / float(this->mX[i + 1] - mX[i]);
330  if (deriv[i - lower] < 0) {
331  deriv[i - lower] = 0.;
332  }
333  xfx += this->mX[i] * deriv[i - lower];
334  fx += deriv[i - lower];
335  }
336 
337  thresh = xfx / fx;
338  float stddev = 0;
339  for (int i = lower; i < upper; i++) {
340  stddev += std::pow(this->mX[i] - thresh, 2) * deriv[i - lower];
341  }
342 
343  stddev /= fx;
344  noise = std::sqrt(stddev);
345 
346  return (noise < 15);
347 }

350 // Use ROOT to find the threshold and noise via derivative method
351 // data is the number of trigger counts per charge injected;
352 // x is the array of charge injected values;
353 // NPoints is the length of both arrays.
354 bool ITSThresholdCalibrator::findThresholdHitcounting(
355  const char* data, const short int* x, const short int& NPoints, float& thresh)
356 {
357  unsigned short int numberOfHits = 0;
358  bool is50 = false;
359  for (unsigned short int i = 0; i < NPoints; i++) {
360  numberOfHits += data[i];
361  if (!is50 && data[i] == N_INJ) {
362  is50 = true;
363  }
364  }
365 
366  // If not enough counts return a failure
367  // if (numberOfHits < N_INJ) { return false; }
368  if (!is50) {
369  if (this->mVerboseOutput) {
370  LOG(warning) << "Calculation unsuccessful: too few hits. Skipping this pixel";
371  }
372  return false;
373  }
374 
375  if (this->mScanType == 'T') {
376  thresh = this->mX[*(this->N_RANGE) - 1] - numberOfHits / float(N_INJ);
377  } else if (this->mScanType == 'V') {
378  thresh = (this->mX[*(this->N_RANGE) - 1] * N_INJ - numberOfHits) / float(N_INJ);
379  } else if (this->mScanType == 'I') {
380  thresh = (numberOfHits + N_INJ * this->mX[0]) / float(N_INJ);
381  } else {
382  LOG(error) << "Unexpected runtype encountered in findThresholdHitcounting()";
383  return false;
384  }
385 
386  return true;
387 }

390 // Run threshold extraction on completed row and update memory
391 void ITSThresholdCalibrator::extractThresholdRow(const short int& chipID, const short int& row)
392 {
393 #ifdef WITH_OPENMP
394  omp_set_num_threads(mNThreads);
395 #pragma omp parallel for schedule(dynamic)
396 #endif
397  // Loop over all columns (pixels) in the row
398  for (short int col_i = 0; col_i < this->N_COL; col_i++) {
399 
400  // Do the threshold fit
401  float thresh = 0., noise = 0.;
402  bool success = false;
403  success = this->findThreshold(&(this->mPixelHits[chipID][col_i][0]),
404  this->mX, *(this->N_RANGE), thresh, noise);
405 
406  vChipid[col_i] = chipID;
407  vRow[col_i] = row;
408  vThreshold[col_i] = this->mScanType == 'T' ? (short int)(thresh * 10.) : (short int)(thresh);
409  vNoise[col_i] = this->mScanType == 'T' ? (unsigned char)(noise * 10.) : (short int)(noise);
410  vSuccess[col_i] = success;
411  }
412 
413  // Saves threshold information to internal memory
414  this->saveThreshold();
415 }

418 void ITSThresholdCalibrator::saveThreshold()
419 {
420  // In the case of a full threshold scan, write to TTree
421  if (this->mScanType == 'T') {
422  this->mThresholdTree->Fill();
423  }
424 
425  // Save info in a map for later averaging
426  int sumT = 0, sumSqT = 0, sumN = 0, sumSqN = 0;
427  int countSuccess = 0;
428 #ifdef WITH_OPENMP
429  omp_set_num_threads(mNThreads);
430 #pragma omp parallel for default(shared) reduction(+ \
431  : sumT, sumSqT, sumN, sumSqN, countSuccess)
432 #endif
433  for (int i = 0; i < this->N_COL; i++) {
434  if (vSuccess[i]) {
435  sumT += vThreshold[i];
436  sumN += (int)vNoise[i];
437  sumSqT += (vThreshold[i]) * (vThreshold[i]);
438  sumSqN += ((int)vNoise[i]) * ((int)vNoise[i]);
439  countSuccess++;
440  }
441  }
442  short int chipID = vChipid[0];
443  std::array<int, 5> dataSum{{sumT, sumSqT, sumN, sumSqN, countSuccess}};
444  if (!(this->mThresholds.count(chipID))) {
445  this->mThresholds[chipID] = dataSum;
446  } else {
447  std::array<int, 5> dataAll{{this->mThresholds[chipID][0] + dataSum[0], this->mThresholds[chipID][1] + dataSum[1], this->mThresholds[chipID][2] + dataSum[2], this->mThresholds[chipID][3] + dataSum[3], this->mThresholds[chipID][4] + dataSum[4]}};
448  this->mThresholds[chipID] = dataAll;
449  }
450 
451  return;
452 }

455 // Perform final operations on output objects. In the case of a full threshold
456 // scan, rename ROOT file and create metadata file for writing to EOS
457 void ITSThresholdCalibrator::finalizeOutput()
458 {
459  // Check that objects actually exist in memory
460  if (!(this->mRootOutfile) || !(this->mThresholdTree)) {
461  return;
462  }
463 
464  // Ensure that everything has been written to the ROOT file
465  this->mRootOutfile->cd();
466  this->mThresholdTree->Write(nullptr, TObject::kOverwrite);
467 
468  // Clean up the mThresholdTree and ROOT output file
469  delete this->mThresholdTree;
470  this->mThresholdTree = nullptr;
471 
472  this->mRootOutfile->Close();
473  delete this->mRootOutfile;
474  this->mRootOutfile = nullptr;
475 
476  // Check that expected output directory exists
477  std::string dir = this->mOutputDir + fmt::format("{}_{}/", this->mEnvironmentID, this->mRunNumber);
478  if (!std::filesystem::exists(dir)) {
479  LOG(error) << "Cannot find expected output directory " << dir;
480  return;
481  }
482 
483  // Expected ROOT output filename
484  std::string filename = std::to_string(this->mRunNumber) + '_' +
485  std::to_string(this->mFileNumber) + '_' + this->mHostname;
486  std::string filenameFull = dir + filename;
487  try {
488  std::rename((filenameFull + ".root.part").c_str(),
489  (filenameFull + ".root").c_str());
490  } catch (std::exception const& e) {
491  LOG(error) << "Failed to rename ROOT file " << filenameFull
492  << ".root.part, reason: " << e.what();
493  }
494 
495  // Create metadata file
496  o2::dataformats::FileMetaData* mdFile = new o2::dataformats::FileMetaData();
497  mdFile->fillFileData(filenameFull + ".root");
498  mdFile->run = this->mRunNumber;
499  mdFile->LHCPeriod = (this->mLHCPeriod.find("_ITS") == std::string::npos)
500  ? this->mLHCPeriod + "_ITS"
501  : this->mLHCPeriod;
502  if (!(this->mMetaType.empty())) {
503  mdFile->type = this->mMetaType;
504  }
505  mdFile->priority = "high";
506  mdFile->lurl = filenameFull + ".root";
507  auto metaFileNameTmp = fmt::format("{}{}.tmp", this->mMetafileDir, filename);
508  auto metaFileName = fmt::format("{}{}.done", this->mMetafileDir, filename);
509  try {
510  std::ofstream metaFileOut(metaFileNameTmp);
511  metaFileOut << mdFile->asString() << '\n';
512  metaFileOut.close();
513  std::filesystem::rename(metaFileNameTmp, metaFileName);
514  } catch (std::exception const& e) {
515  LOG(error) << "Failed to create threshold metadata file "
516  << metaFileName << ", reason: " << e.what();
517  }
518  delete mdFile;
519 
520  // Next time a file is created, use a larger number
521  this->mFileNumber++;
522 
523  return;
524 
525 } // finalizeOutput

528 // Set the run_type for this run
529 // Initialize the memory needed for this specific type of run
530 void ITSThresholdCalibrator::setRunType(const short int& runtype)
531 {
532 
533  // Save run type info for future evaluation
534  this->mRunType = runtype;
535 
536  if (runtype == THR_SCAN) {
537  // full_threshold-scan -- just extract thresholds for each pixel and write to TTree
538  // 512 rows per chip
539  this->initThresholdTree();
540  this->N_RANGE = &(this->N_CHARGE);
541  this->mScanType = 'T';
542  this->mMin = 0;
543  this->mMax = 50;
544 
545  } else if (runtype == THR_SCAN_SHORT || runtype == THR_SCAN_SHORT_100HZ ||
546  runtype == THR_SCAN_SHORT_200HZ) {
547  // threshold_scan_short -- just extract thresholds for each pixel and write to TTree
548  // 10 rows per chip
549  this->initThresholdTree();
550  this->N_RANGE = &(this->N_CHARGE);
551  this->mScanType = 'T';
552  this->mMin = 0;
553  this->mMax = 50;
554 
555  } else if (runtype == VCASN150 || runtype == VCASN100 || runtype == VCASN100_100HZ) {
556  // VCASN tuning for different target thresholds
557  // Store average VCASN for each chip into CCDB
558  // 4 rows per chip
559  this->N_RANGE = &(this->N_VCASN);
560  this->mScanType = 'V';
561  this->mMin = 30;
562  this->mMax = 80;
563 
564  } else if (runtype == ITHR150 || runtype == ITHR100 || runtype == ITHR100_100HZ) {
565  // ITHR tuning -- average ITHR per chip
566  // S-curve is backwards from VCASN case, otherwise same
567  // 4 rows per chip
568  this->N_RANGE = &(this->N_ITHR);
569  this->mScanType = 'I';
570  this->mMin = 30;
571  this->mMax = 100;
572 
573  } else if (runtype == END_RUN) {
574  // Trigger end-of-stream method to do averaging and save data to CCDB
575  // TODO :: test this. In theory end-of-stream should be called automatically
576  // but we could use a check on this->mRunType
577 
578  } else {
579  // No other run type recognized by this workflow
580  LOG(error) << "runtype " << runtype << " not recognized by threshold scan workflow.";
581  throw runtype;
582  }
583 
584  this->mX = new short int[*(this->N_RANGE)];
585  for (short int i = this->mMin; i <= this->mMax; i++) {
586  this->mX[i - this->mMin] = i;
587  }
588 
589  // Initialize objects for doing the threshold fits
590  if (this->mFitType == FIT) {
591  // Initialize the histogram used for error function fits
592  // Will initialize the TF1 in setRunType (it is different for different runs)
593  this->mFitHist = new TH1F(
594  "mFitHist", "mFitHist", *(this->N_RANGE), this->mX[0], this->mX[*(this->N_RANGE) - 1]);
595 
596  // Initialize correct fit function for the scan type
597  this->mFitFunction = (this->mScanType == 'I')
598  ? new TF1("mFitFunction", erf_ithr, 0, 1500, 2)
599  : new TF1("mFitFunction", erf, 0, 1500, 2);
600  this->mFitFunction->SetParName(0, "Threshold");
601  this->mFitFunction->SetParName(1, "Noise");
602  }
603 
604  return;
605 }

681 // Main running function
682 // Get info from previous stf decoder workflow, then loop over readout frames
683 // (ROFs) to count hits and extract thresholds
684 void ITSThresholdCalibrator::run(ProcessingContext& pc)
685 {
686  // Save environment ID and run number if needed
687  if (this->mEnvironmentID.empty()) {
688  this->updateEnvironmentID(pc);
689  }
690  if (this->mRunNumber == -1) {
691  this->updateRunID(pc);
692  }
693  if (this->mLHCPeriod.empty()) {
694  this->updateLHCPeriod(pc);
695  }
696 
697  // Calibration vector
698  const auto calibs = pc.inputs().get<gsl::span<o2::itsmft::GBTCalibData>>("calib");
699  const auto digits = pc.inputs().get<gsl::span<o2::itsmft::Digit>>("digits");
700  const auto ROFs = pc.inputs().get<gsl::span<o2::itsmft::ROFRecord>>("digitsROF");
701 
702  // Store some lengths for convenient looping
703  const unsigned int nROF = (unsigned int)ROFs.size();
704 
705  // Loop over readout frames (usually only 1, sometimes 2)
706  for (unsigned int iROF = 0; iROF < nROF; iROF++) {
707 
708  unsigned int rofIndex = ROFs[iROF].getFirstEntry();
709  unsigned int rofNEntries = ROFs[iROF].getNEntries();
710 
711  // Find the correct charge and row values for this ROF
712  short int charge = -1;
713  short int row = -1;
714  for (short int iRU = 0; iRU < this->N_RU; iRU++) {
715  const auto& calib = calibs[iROF * this->N_RU + iRU];
716  if (calib.calibUserField != 0) {
717  if (charge >= 0) {
718  LOG(warning) << "More than one charge detected!";
719  }
720 
721  if (this->mRunType == -1) {
722  short int runtype = (short int)(calib.calibUserField >> 24);
723  this->setRunType(runtype);
724  }
725  this->mRunTypeUp = (short int)(calib.calibUserField >> 24);
726  // Divide calibration word (24-bit) by 2^16 to get the first 8 bits
727  if (this->mScanType == 'T') {
728  // For threshold scan have to subtract from 170 to get charge value
729  charge = (short int)(170 - (calib.calibUserField >> 16) & 0xff);
730  } else { // VCASN or ITHR tuning
731  charge = (short int)((calib.calibUserField >> 16) & 0xff);
732  }
733 
734  // Last 16 bits should be the row (only uses up to 9 bits)
735  row = (short int)(calib.calibUserField & 0xffff);
736 
737  break;
738  }
739  }
740 
741  // If a charge was not found, skip this ROF
742  if (charge < 0) {
743  if (this->mVerboseOutput) {
744  LOG(warning) << "Charge not updated";
745  }
746  } else {
747 
748  for (unsigned int idig = rofIndex; idig < rofIndex + rofNEntries; idig++) {
749  auto& d = digits[idig];
750  short int chipID = (short int)d.getChipIndex();
751  short int col = (short int)d.getColumn();
752  if (!this->mCheckEos && !this->mRunTypeUp) {
753  if (!this->mIsChipDone[chipID]) {
754  this->mRunTypeChip[chipID]++;
755  this->mIsChipDone[chipID] = true;
756  }
757  }
758  // Row should be the same for the whole ROF
759  // Check if chip hasn't appeared before
760  if (!(this->mCurrentRow.count(chipID))) {
761  // Update current row and initialize hit map
762  this->mCurrentRow[chipID] = row;
763  this->mPixelHits[chipID] = std::vector<std::vector<char>>(
764  this->N_COL, std::vector<char>(*(this->N_RANGE), 0));
765 
766  // Check if we have a new row on an already existing chip
767  } else if (this->mCurrentRow[chipID] != row) {
768 
769  this->extractAndUpdate(chipID);
770  // Reset row & hit map for the new row
771  this->mCurrentRow[chipID] = row;
772  for (std::vector<char>& v : this->mPixelHits[chipID]) {
773  std::fill(v.begin(), v.end(), 0);
774  }
775  }
776 
777  // Increment the number of counts for this pixel
778  if (charge > this->mMax || charge < this->mMin) {
779  LOG(error) << "charge value " << charge << " out of range for min " << this->mMin
780  << " and max " << this->mMax << " (range: " << *(this->N_RANGE) << ")";
781  throw charge;
782  }
783 
784  this->mPixelHits[chipID][col][charge - this->mMin]++;
785 
786  } // for (digits)
787 
788  } // if (charge)
789 
790  // reset mIsChipDone before the next ROF
791  memset(this->mIsChipDone, 0, sizeof(this->mIsChipDone));
792 
793  } // for (ROFs)
794 
795  if (!(this->mCheckEos) && !(this->mRunTypeUp)) {
796  LOG(info) << "Shipping DCSconfigObject_t, run type, scan type and fit type to aggregator from run function (no endOfStream will be used!)";
797  this->finalize(nullptr);
798  pc.outputs().snapshot(Output{"ITS", "TSTR", 0}, this->mTuning);
799  pc.outputs().snapshot(Output{"ITS", "RUNT", 0}, this->mRunType);
800  pc.outputs().snapshot(Output{"ITS", "SCANT", 0}, this->mScanType);
801  pc.outputs().snapshot(Output{"ITS", "FITT", 0}, this->mFitType);
802  // reset the DCSconfigObject_t before next ship out
803  this->mTuning.clear();
804  }
805 
806  return;
807 }

810 // Calculate the average threshold given a vector of threshold objects
811 void ITSThresholdCalibrator::findAverage(const std::array<int, 5>& data, float& avgT, float& rmsT, float& avgN, float& rmsN)
812 {
813  avgT = (!data[4]) ? 0. : (float)data[0] / (float)data[4];
814  rmsT = (!data[4]) ? 0. : std::sqrt((float)data[1] / (float)data[4] - avgT * avgT);
815  avgN = (!data[4]) ? 0. : (float)data[2] / (float)data[4];
816  rmsN = (!data[4]) ? 0. : std::sqrt((float)data[3] / (float)data[4] - avgN * avgN);
817  return;
818 }

862 void ITSThresholdCalibrator::finalize(EndOfStreamContext* ec)
863 {
864 
865  for (auto const& [chipID, hits_vec] : this->mPixelHits) {
866  // Check that we have received all the data for this row
867  // Require that the last charge value has at least half counts
868  if (this->isScanFinished(chipID)) {
869  this->mCounter++;
870  this->extractAndUpdate(chipID);
871  if (!this->mCheckEos) {
872  this->mPixelHits[chipID][0][*(this->N_RANGE) - 1] = 0; // so that at next call, if does not pass isScanFinished again
873  }
874  }
875  }
876 
877  if (!this->mCheckEos && this->mCounter > 0) {
878  this->finalizeOutput();
879  this->initThresholdTree();
880  this->mCounter = 0;
881  } else if (this->mCheckEos) {
882  this->finalizeOutput();
883  }
884 
885  // Add configuration item to output strings for CCDB
886  const char* name = nullptr;
887  if (this->mScanType == 'V') {
888  // Loop over each chip and calculate avg and rms
889  name = "VCASN";
890  auto it = this->mThresholds.cbegin();
891  while (it != this->mThresholds.cend()) {
892  if (!this->mCheckEos && this->mRunTypeChip[it->first] < N_INJ - 2) { // TODO: -2 is a safety factor, to be modified once THR scan is stable enough
893  ++it;
894  continue;
895  }
896  float avgT, rmsT, avgN, rmsN;
897  this->findAverage(it->second, avgT, rmsT, avgN, rmsN);
898  bool status = (this->mX[0] < avgT && avgT < this->mX[*(this->N_RANGE) - 1]);
899  this->addDatabaseEntry(it->first, name, (short int)avgT, rmsT, (short int)avgN, rmsN, status);
900  if (!this->mCheckEos) {
901  this->mRunTypeChip[it->first] = 0; // so that this chip will never appear again in the DCSconfigObject_t
902  it = this->mThresholds.erase(it);
903  } else {
904  ++it;
905  }
906  }
907 
908  } else if (this->mScanType == 'I') {
909  // Loop over each chip and calculate avg and rms
910  name = "ITHR";
911  auto it = this->mThresholds.cbegin();
912  while (it != this->mThresholds.cend()) {
913  if (!this->mCheckEos && this->mRunTypeChip[it->first] < N_INJ - 2) { // TODO: -2 is a safety factor, to be modified once THR scan is stable enough
914  ++it;
915  continue;

917  float avgT, rmsT, avgN, rmsN;
918  this->findAverage(it->second, avgT, rmsT, avgN, rmsN);
919  bool status = (this->mX[0] < avgT && avgT < this->mX[*(this->N_RANGE) - 1]);
920  this->addDatabaseEntry(it->first, name, (short int)avgT, rmsT, (short int)avgN, rmsN, status);
921  if (!this->mCheckEos) {
922  this->mRunTypeChip[it->first] = 0; // so that this chip will never appear again in the DCSconfigObject_t
923  it = this->mThresholds.erase(it);
924  } else {
925  ++it;
926  }
927  }
928 
929  } else if (this->mScanType == 'T') {
930  // Loop over each chip and calculate avg and rms
931  name = "THR";
