33 class LaserTracksCalibratorDevice : public o2::framework::Task
34 {
35  public:
36  void init(o2::framework::InitContext& ic) final
37  {
38  const int minTFs = ic.options().get<int>("min-tfs");
39  const int slotL = ic.options().get<int>("tf-per-slot");
40  const int delay = ic.options().get<int>("max-delay");
41  const bool debug = ic.options().get<bool>("write-debug");
42 
43  mCalibrator = std::make_unique<LaserTracksCalibrator>(minTFs);
44  mCalibrator->setSlotLength(slotL);
45  mCalibrator->setMaxSlotsDelay(delay);
46  mCalibrator->setWriteDebug(debug);
47  }
48 
49  void run(o2::framework::ProcessingContext& pc) final
50  {
51  const auto dph = o2::header::get<o2::framework::DataProcessingHeader*>(pc.inputs().get("laserTracks").header);
52  const auto startTime = dph->startTime;
53  const auto endTime = dph->startTime + dph->duration - 1;
54 
55  auto data = pc.inputs().get<gsl::span<TrackTPC>>("laserTracks");
56  LOGP(info, "Processing TF with start time {} and {} tracks", startTime, data.size());
57 
58  mCalibrator->getSlotForTF(startTime).getContainer()->setTFtimes(startTime, endTime);
59  mCalibrator->process(startTime, data);
60 
61  if (mCalibrator->hasCalibrationData()) {
62  sendOutput(pc.outputs());
63  }
64  }
65 
66  void endOfStream(o2::framework::EndOfStreamContext& ec) final
67  {
68  LOGP(info, "LaserTracksCalibratorDevice::endOfStream: Finalizing calibration");
69  constexpr uint64_t INFINITE_TF = 0xffffffffffffffff;
70  mCalibrator->checkSlotsToFinalize(INFINITE_TF);
71  sendOutput(ec.outputs());
72  }
73 
74  private:
75  std::unique_ptr<LaserTracksCalibrator> mCalibrator;
76 
77  //________________________________________________________________
78  void sendOutput(DataAllocator& output)
79  {
80  using clbUtils = o2::calibration::Utils;
81  const auto& calibrations = mCalibrator->getCalibPerSlot();
82  const long timeEnd = 99999999999999;
83  for (uint32_t iCalib = 0; iCalib < calibrations.size(); ++iCalib) {
84  const auto& object = calibrations[iCalib];
85  o2::ccdb::CcdbObjectInfo w;
86  auto image = o2::ccdb::CcdbApi::createObjectImage(&object, &w);
87 
88  w.setPath("TPC/Calib/LaserTracks");
89  w.setStartValidityTimestamp(object.firstTime);
90  //w.setEndValidityTimestamp(object.lastTime);
91  w.setEndValidityTimestamp(timeEnd);
92 
93  LOGP(info, "Sending object {} / {} of size {} bytes, valid for {} : {} ", w.getPath(), w.getFileName(), image->size(), w.getStartValidityTimestamp(), w.getEndValidityTimestamp());
94  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBPayload, "TPC_CalibLtr", iCalib}, *image.get());
95  output.snapshot(Output{o2::calibration::Utils::gDataOriginCDBWrapper, "TPC_CalibLtr", iCalib}, w);
96  }
97  mCalibrator->initOutput();
98  }
99 };
