31 void RawToDigitConverterSpec::init(framework::InitContext& ctx)
32 {
33  LOG(debug) << "Initializing RawToDigitConverterSpec...";
34 
35  //Read command-line options
36  //Pedestal flag true/false
37  mIsPedestalData = false;
38  if (ctx.options().isSet("pedestal")) {
39  mIsPedestalData = ctx.options().get<bool>("pedestal");
40  }
41  LOG(info) << "Pedestal data: " << mIsPedestalData;
42  if (mIsPedestalData) { //no calibration for pedestal runs needed
43  return; //skip CCDB initialization for pedestal runs
44  }
45 
46  // is no-gain-calibration flag setted?
47  mIsUsingGainCalibration = true;
48  if (ctx.options().isSet("no-gain-calibration")) {
49  mIsUsingGainCalibration = !ctx.options().get<bool>("no-gain-calibration");
50  LOG(info) << "no-gain-calibration is swithed ON";
51  }
52 
53  // is no-bad-channel-map flag setted?
54  mIsUsingBadMap = true;
55  if (ctx.options().isSet("no-bad-channel-map")) {
56  mIsUsingBadMap = !ctx.options().get<bool>("no-bad-channel-map");
57  LOG(info) << "no-bad-channel-map is swithed ON";
58  }
59 
60  //CCDB Url
61  std::string ccdbUrl = "localtest";
62  if (ctx.options().isSet("ccdb-url")) {
63  ccdbUrl = ctx.options().get<std::string>("ccdb-url");
64  }
65  LOG(info) << "CCDB Url: " << ccdbUrl;
66 
67  //dummy calibration objects
68  if (ccdbUrl.compare("localtest") == 0) { // test default calibration
69  mIsUsingCcdbMgr = false;
70  mCalibParams = new o2::cpv::CalibParams(1);
71  mBadMap = new o2::cpv::BadChannelMap(1);
72  mPedestals = new o2::cpv::Pedestals(1);
73  LOG(info) << "No reading calibration from ccdb requested, using dummy calibration for testing";
74  LOG(info) << "Task configuration is done.";
75  return; //localtest = no reading ccdb
76  }
77 
78  //init CCDB
79  auto& ccdbMgr = o2::ccdb::BasicCCDBManager::instance();
80  ccdbMgr.setURL(ccdbUrl);
81  mIsUsingCcdbMgr = ccdbMgr.isHostReachable(); //if host is not reachable we can use only dummy calibration
82  if (!mIsUsingCcdbMgr) {
83  LOG(error) << "Host " << ccdbUrl << " is not reachable!!!";
84  LOG(error) << "Using dummy calibration";
85  mCalibParams = new o2::cpv::CalibParams(1);
86  mBadMap = new o2::cpv::BadChannelMap(1);
87  mPedestals = new o2::cpv::Pedestals(1);
88 
89  } else {
90  ccdbMgr.setCaching(true); //make local cache of remote objects
91  ccdbMgr.setLocalObjectValidityChecking(true); //query objects from remote site only when local one is not valid
92  LOG(info) << "Successfully initializated BasicCCDBManager with caching option";
93 
94  //read calibration from ccdb (for now do it only at the beginning of dataprocessing)
95  //probably later we can check bad channel map more oftenly
96  mCurrentTimeStamp = o2::ccdb::getCurrentTimestamp();
97  ccdbMgr.setTimestamp(mCurrentTimeStamp);
98 
99  if (mIsUsingGainCalibration) {
100  mCalibParams = ccdbMgr.get<o2::cpv::CalibParams>("CPV/Calib/Gains");
101  if (!mCalibParams) {
102  LOG(error) << "Cannot get o2::cpv::CalibParams from CCDB. Using dummy calibration!";
103  mCalibParams = new o2::cpv::CalibParams(1);
104  }
105  } else {
106  LOG(info) << "Using dummy gain calibration (all coeffs = 1.)";
107  mCalibParams = new o2::cpv::CalibParams(1);
108  }
109 
110  if (mIsUsingBadMap) {
111  mBadMap = ccdbMgr.get<o2::cpv::BadChannelMap>("CPV/Calib/BadChannelMap");
112  if (!mBadMap) {
113  LOG(error) << "Cannot get o2::cpv::BadChannelMap from CCDB. Using dummy calibration!";
114  mBadMap = new o2::cpv::BadChannelMap(1);
115  }
116  } else {
117  mBadMap = new o2::cpv::BadChannelMap(1);
118  LOG(info) << "Using dummy bad map (all channels are good)";
119  }
120 
121  mPedestals = ccdbMgr.get<o2::cpv::Pedestals>("CPV/Calib/Pedestals");
122  if (!mPedestals) {
123  LOG(error) << "Cannot get o2::cpv::Pedestals from CCDB. Using dummy calibration!";
124  mPedestals = new o2::cpv::Pedestals(1);
125  }
126  LOG(info) << "Task configuration is done.";
127  }

130 void RawToDigitConverterSpec::run(framework::ProcessingContext& ctx)
131 {
132  // Cache digits from bunch crossings as the component reads timeframes from many links consecutively
133  std::map<o2::InteractionRecord, std::shared_ptr<std::vector<o2::cpv::Digit>>> digitBuffer; // Internal digit buffer
134  int firstEntry = 0;
135  mOutputHWErrors.clear();
136 
137  // if we see requested data type input with 0xDEADBEEF subspec and 0 payload this means that the "delayed message"
138  // mechanism created it in absence of real data from upstream. Processor should send empty output to not block the workflow
139  static size_t contDeadBeef = 0; // number of times 0xDEADBEEF was seen continuously
140  std::vector<o2::framework::InputSpec> dummy{o2::framework::InputSpec{"dummy", o2::framework::ConcreteDataMatcher{"CPV", o2::header::gDataDescriptionRawData, 0xDEADBEEF}}};
141  for (const auto& ref : framework::InputRecordWalker(ctx.inputs(), dummy)) {
142  const auto dh = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
143  if (dh->payloadSize == 0) { // send empty output
144  auto maxWarn = o2::conf::VerbosityConfig::Instance().maxWarnDeadBeef;
145  if (++contDeadBeef <= maxWarn) {
146  LOGP(warning, "Found input [{}/{}/{:#x}] TF#{} 1st_orbit:{} Payload {} : assuming no payload for all links in this TF{}",
147  dh->dataOrigin.str, dh->dataDescription.str, dh->subSpecification, dh->tfCounter, dh->firstTForbit, dh->payloadSize,
148  contDeadBeef == maxWarn ? fmt::format(". {} such inputs in row received, stopping reporting", contDeadBeef) : "");
149  }
150  mOutputDigits.clear();
151  ctx.outputs().snapshot(o2::framework::Output{"CPV", "DIGITS", 0, o2::framework::Lifetime::Timeframe}, mOutputDigits);
152  mOutputTriggerRecords.clear();
153  ctx.outputs().snapshot(o2::framework::Output{"CPV", "DIGITTRIGREC", 0, o2::framework::Lifetime::Timeframe}, mOutputTriggerRecords);
154  mOutputHWErrors.clear();
155  ctx.outputs().snapshot(o2::framework::Output{"CPV", "RAWHWERRORS", 0, o2::framework::Lifetime::Timeframe}, mOutputHWErrors);
156  return; //empty TF, nothing to process
157  }
158  }
159  contDeadBeef = 0; // if good data, reset the counter
160 
161  std::vector<o2::framework::InputSpec> rawFilter{
162  {"RAWDATA", o2::framework::ConcreteDataTypeMatcher{"CPV", "RAWDATA"}, o2::framework::Lifetime::Timeframe},
163  };
164  for (const auto& rawData : framework::InputRecordWalker(ctx.inputs(), rawFilter)) {
165  o2::cpv::RawReaderMemory rawreader(o2::framework::DataRefUtils::as<const char>(rawData));
166  // loop over all the DMA pages
167  while (rawreader.hasNext()) {
168  try {
169  rawreader.next();
170  } catch (RawErrorType_t e) {
171  LOG(error) << "Raw decoding error " << (int)e;
172  //add error list
173  //RawErrorType_t is defined in O2/Detectors/CPV/reconstruction/include/CPVReconstruction/RawReaderMemory.h
174  //RawDecoderError(short c, short d, short g, short p, RawErrorType_t e)
175  mOutputHWErrors.emplace_back(25, 0, 0, 0, e); //Put general errors to non-existing ccId 25
176  //if problem in header, abandon this page
177  if (e == RawErrorType_t::kRDH_DECODING) {
178  break;
179  }
180  //if problem in payload, try to continue
181  continue;
182  }
183  auto& rdh = rawreader.getRawHeader();
184  auto triggerOrbit = o2::raw::RDHUtils::getTriggerOrbit(rdh);
185  auto mod = o2::raw::RDHUtils::getLinkID(rdh) + 2; //link=0,1,2 -> mod=2,3,4
186  //for now all modules are written to one LinkID
187  if (mod > o2::cpv::Geometry::kNMod || mod < 2) { //only 3 correct modules:2,3,4
188  LOG(error) << "module=" << mod << "do not exist";
189  mOutputHWErrors.emplace_back(25, mod, 0, 0, kRDH_INVALID); //Add non-existing modules to non-existing ccId 25 and dilogic = mod
190  continue; //skip STU mod
191  }
192  o2::cpv::RawDecoder decoder(rawreader);
193  RawErrorType_t err = decoder.decode();
194 
195  if (!(err == kOK || err == kOK_NO_PAYLOAD)) {
196  //TODO handle severe errors
197  //TODO: probably careful conversion of decoder errors to Fitter errors?
198  mOutputHWErrors.emplace_back(25, mod, 0, 0, err); //assign general RDH errors to non-existing ccId 25 and dilogic = mod
199  }
200 
201  std::shared_ptr<std::vector<o2::cpv::Digit>> currentDigitContainer;
202  auto digilets = decoder.getDigits();
203  if (digilets.empty()) { //no digits -> continue to next pages
204  continue;
205  }
206  o2::InteractionRecord currentIR(0, triggerOrbit); //(bc, orbit)
207  // Loop over all the BCs
208  for (auto itBCRecords : decoder.getBCRecords()) {
209  currentIR.bc = itBCRecords.bc;
210  for (int iDig = itBCRecords.firstDigit; iDig <= itBCRecords.lastDigit; iDig++) {
211  auto adch = digilets[iDig];
212  auto found = digitBuffer.find(currentIR);
213  if (found == digitBuffer.end()) {
214  currentDigitContainer = std::make_shared<std::vector<o2::cpv::Digit>>();
215  digitBuffer[currentIR] = currentDigitContainer;
216  } else {
217  currentDigitContainer = found->second;
218  }
219 
220  AddressCharge ac = {adch};
221  unsigned short absId = ac.Address;
222  //if we deal with non-pedestal data?
223  if (!mIsPedestalData) { //not a pedestal data
224  //test bad map
225  if (mBadMap->isChannelGood(absId)) {
226  //we need to subtract pedestal from amplidute and calibrate it
227  float amp = mCalibParams->getGain(absId) * (ac.Charge - mPedestals->getPedestal(absId));
228  if (amp > 0) {
229  currentDigitContainer->emplace_back(absId, amp, -1);
230  }
231  }
232  } else { //pedestal data, no calibration needed.
233  currentDigitContainer->emplace_back(absId, (float)ac.Charge, -1);
234  }
235  }
236  }
237  //Check and send list of hwErrors
238  for (auto& er : decoder.getErrors()) {
239  mOutputHWErrors.push_back(er);
240  }
241  } //RawReader::hasNext
242  }
243 
244  // Loop over BCs, sort digits with increasing digit ID and write to output containers
245  mOutputDigits.clear();
246  mOutputTriggerRecords.clear();
247  for (auto [bc, digits] : digitBuffer) {
248  int prevDigitSize = mOutputDigits.size();
249  if (digits->size()) {
250  // Sort digits according to digit ID
251  std::sort(digits->begin(), digits->end(), [](o2::cpv::Digit& lhs, o2::cpv::Digit& rhs) { return lhs.getAbsId() < rhs.getAbsId(); });
252 
253  for (auto digit : *digits) {
254  mOutputDigits.push_back(digit);
255  }
256  }
257 
258  mOutputTriggerRecords.emplace_back(bc, prevDigitSize, mOutputDigits.size() - prevDigitSize);
259  }
260  digitBuffer.clear();
261 
262  LOG(info) << "[CPVRawToDigitConverter - run] Sending " << mOutputDigits.size() << " digits in " << mOutputTriggerRecords.size() << "trigger records.";
263  ctx.outputs().snapshot(o2::framework::Output{"CPV", "DIGITS", 0, o2::framework::Lifetime::Timeframe}, mOutputDigits);
264  ctx.outputs().snapshot(o2::framework::Output{"CPV", "DIGITTRIGREC", 0, o2::framework::Lifetime::Timeframe}, mOutputTriggerRecords);
265  ctx.outputs().snapshot(o2::framework::Output{"CPV", "RAWHWERRORS", 0, o2::framework::Lifetime::Timeframe}, mOutputHWErrors);
266 }

268 o2::framework::DataProcessorSpec o2::cpv::reco_workflow::getRawToDigitConverterSpec(bool askDISTSTF)
269 {
270  std::vector<o2::framework::InputSpec> inputs;
271  inputs.emplace_back("RAWDATA", o2::framework::ConcreteDataTypeMatcher{"CPV", "RAWDATA"}, o2::framework::Lifetime::Optional);
272  //receive at least 1 guaranteed input (which will allow to acknowledge the TF)
273  if (askDISTSTF) {
274  inputs.emplace_back("STFDist", "FLP", "DISTSUBTIMEFRAME", 0, o2::framework::Lifetime::Timeframe);
275  }
276  std::vector<o2::framework::OutputSpec> outputs;
277  outputs.emplace_back("CPV", "DIGITS", 0, o2::framework::Lifetime::Timeframe);
278  outputs.emplace_back("CPV", "DIGITTRIGREC", 0, o2::framework::Lifetime::Timeframe);
279  outputs.emplace_back("CPV", "RAWHWERRORS", 0, o2::framework::Lifetime::Timeframe);
280  //note that for cpv we always have stream #0 (i.e. CPV/DIGITS/0)
281 
282  return o2::framework::DataProcessorSpec{"CPVRawToDigitConverterSpec",
283  inputs, // o2::framework::select("A:CPV/RAWDATA"),
284  outputs,
285  o2::framework::adaptFromTask<o2::cpv::reco_workflow::RawToDigitConverterSpec>(),
286  o2::framework::Options{
287  {"pedestal", o2::framework::VariantType::Bool, false, {"do not subtract pedestals from digits"}},
288  {"ccdb-url", o2::framework::VariantType::String, o2::base::NameConf::getCCDBServer(), {"CCDB Url"}},
289  {"no-gain-calibration", o2::framework::VariantType::Bool, false, {"do not apply gain calibration"}},
290  {"no-bad-channel-map", o2::framework::VariantType::Bool, false, {"do not mask bad channels"}},
291  }};
