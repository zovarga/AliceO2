64 void TRDGlobalTracking::init(InitContext& ic)
65 {
66 
67  //-------- init geometry and field --------//
68  o2::base::GeometryManager::loadGeometry();
69  o2::base::Propagator::initFieldFromGRP();
70  auto geo = Geometry::instance();
71  o2::its::GeometryTGeo::Instance()->fillMatrixCache(o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2GRot) | o2::math_utils::bit2Mask(o2::math_utils::TransformType::T2L));
72  geo->createPadPlaneArray();
73  geo->createClusterMatrixArray();
74  mFlatGeo = std::make_unique<GeometryFlat>(*geo);
75 
76  // this is a hack to provide Mat.LUT from the local file, in general will be provided by the framework from CCDB
77  std::string matLUTPath = ic.options().get<std::string>("material-lut-path");
78  std::string matLUTFile = o2::base::NameConf::getMatLUTFileName(matLUTPath);
79  if (o2::utils::Str::pathExists(matLUTFile)) {
80  auto* lut = o2::base::MatLayerCylSet::loadFromFile(matLUTFile);
81  o2::base::Propagator::Instance()->setMatLUT(lut);
82  LOG(info) << "Loaded material LUT from " << matLUTFile;
83  } else {
84  LOG(info) << "Material LUT " << matLUTFile << " file is absent, only TGeo can be used";
85  }
86 
87  // this is a hack to provide ITS dictionary from the local file, in general will be provided by the framework from CCDB
88  auto dictFile = o2::itsmft::ClustererParam<o2::detectors::DetID::ITS>::Instance().dictFilePath;
89  dictFile = o2::base::DetectorNameConf::getAlpideClusterDictionaryFileName(o2::detectors::DetID::ITS, dictFile);
90  if (o2::utils::Str::pathExists(dictFile)) {
91  mITSDict.readFromFile(dictFile);
92  LOG(info) << "Matching is running with a provided ITS dictionary: " << dictFile;
93  } else {
94  LOG(info) << "Dictionary " << dictFile << " is absent, Matching expects ITS cluster patterns";
95  }
96 
97  //-------- init GPU reconstruction --------//
98  GPURecoStepConfiguration cfgRecoStep;
99  cfgRecoStep.steps = GPUDataTypes::RecoStep::NoRecoStep;
100  cfgRecoStep.inputs.clear();
101  cfgRecoStep.outputs.clear();
102  mRec = GPUReconstruction::CreateInstance("CPU", true);
103  mRec->SetSettings(o2::base::Propagator::Instance()->getNominalBz(), &cfgRecoStep);
104 
105  mChainTracking = mRec->AddChain<GPUChainTracking>();
106 
107  mTracker = new GPUTRDTracker();
108  mTracker->SetNCandidates(mRec->GetProcessingSettings().trdNCandidates); // must be set before initialization
109  if (mStrict && mRec->GetProcessingSettings().trdNCandidates == 1) {
110  LOG(error) << "Strict matching mode requested, but tracks with another close hypothesis will not be rejected. Please set trdNCandidates to at least 3.";
111  }
112  mTracker->SetProcessPerTimeFrame(true);
113  mTracker->SetGenerateSpacePoints(false); // set to true to force space point calculation by the TRD tracker itself
114 
115  mRec->RegisterGPUProcessor(mTracker, false);
116  mChainTracking->SetTRDGeometry(std::move(mFlatGeo));
117  if (mRec->Init()) {
118  LOG(fatal) << "GPUReconstruction could not be initialized";
119  }
120 
121  std::unique_ptr<o2::gpu::TPCFastTransform> fastTransform = (o2::tpc::TPCFastTransformHelperO2::instance()->create(0));
122  mTPCTransform = std::move(fastTransform);
123  mRecoParam.setBfield(o2::base::Propagator::Instance()->getNominalBz());
124 
125  mTracker->PrintSettings();
126  LOG(info) << "Strict matching mode is " << ((mStrict) ? "ON" : "OFF");
127 
128  mTimer.Stop();
129  mTimer.Reset();
130 }

144 void TRDGlobalTracking::fillMCTruthInfo(const TrackTRD& trk, o2::MCCompLabel lblSeed, std::vector<o2::MCCompLabel>& lblContainerTrd, std::vector<o2::MCCompLabel>& lblContainerMatch, const o2::dataformats::MCTruthContainer<o2::MCCompLabel>* trkltLabels) const
145 {
146  // Check MC labels of the TRD tracklets attached to the track seed.
147  // Set TRD track label to the most frequent tracklet label.
148  // Fake flag is set if either not all TRD tracklets have most frequent label
149  // or if the seeding label is different from the most frequent TRD label.
150  // In case multiple tracklet labels occur most often we choose the one which matches the label of the seed, or,
151  // if that is not the case one of the most frequent labels is chosen arbitrarily
152  LOG(debug) << "Checking seed with label: " << lblSeed;
153  std::unordered_map<o2::MCCompLabel, unsigned int> labelCounter;
154  int nTracklets = 0;
155  unsigned int maxOccurences = 0;
156  for (int iLy = 0; iLy < constants::NLAYER; ++iLy) {
157  auto trkltIndex = trk.getTrackletIndex(iLy);
158  if (trkltIndex == -1) {
159  // no tracklet in this layer
160  continue;
161  }
162  const auto& lblsTrklt = trkltLabels->getLabels(trkltIndex);
163  for (const auto lblTrklt : lblsTrklt) {
164  int nOcc = ++labelCounter[lblTrklt];
165  if (nOcc > maxOccurences) {
166  maxOccurences = nOcc;
167  }
168  }
169  }
170  o2::MCCompLabel mostFrequentLabel;
171  for (const auto& [lbl, count] : labelCounter) {
172  LOG(debug) << "Label " << lbl << " occured " << count << " times.";
173  if (count == maxOccurences) {
174  if (lblSeed == lbl) {
175  // most frequent label matches seed label
176  mostFrequentLabel = lbl;
177  mostFrequentLabel.setFakeFlag(maxOccurences != trk.getNtracklets());
178  lblContainerTrd.push_back(mostFrequentLabel);
179  lblContainerMatch.push_back(lblSeed); // is not fake by definition, since the seed label matches the TRD track label
180  return;
181  } else {
182  // maybe multiple labels occur with the same frequency and the next one might match the seed?
183  mostFrequentLabel = lbl;
184  }
185  }
186  }
187  mostFrequentLabel.setFakeFlag(maxOccurences != trk.getNtracklets());
188  lblContainerTrd.push_back(mostFrequentLabel);
189  lblSeed.setFakeFlag(lblSeed != mostFrequentLabel);
190  lblContainerMatch.push_back(lblSeed);
191 }

193 void TRDGlobalTracking::fillTrackTriggerRecord(const std::vector<TrackTRD>& tracks, std::vector<TrackTriggerRecord>& trigRec, const gsl::span<const o2::trd::TriggerRecord>& trackletTrigRec) const
194 {
195  // after the tracking is done we assemble here a TrackTriggerRecord similar to the TriggerRecord
196  // which for each TRD trigger stored the found tracks
197  int currTrigRec = 0;
198  int nTracksCurr = 0;
199  int iTrackFirst = 0;
200  for (const auto& trk : tracks) {
201  if (trk.getCollisionId() != currTrigRec) {
202  // new collision ID, create new track trigger record
203  trigRec.emplace_back(trackletTrigRec[currTrigRec].getBCData(), iTrackFirst, nTracksCurr);
204  currTrigRec = trk.getCollisionId();
205  iTrackFirst += nTracksCurr;
206  nTracksCurr = 0;
207  }
208  ++nTracksCurr;
209  }
210  if (nTracksCurr > 0) {
211  // create track trigger record for remaining track range
212  trigRec.emplace_back(trackletTrigRec[currTrigRec].getBCData(), iTrackFirst, nTracksCurr);
213  }
214 }

216 void TRDGlobalTracking::run(ProcessingContext& pc)
217 {
218  mTimer.Start(false);
219  mChainTracking->ClearIOPointers();
220  o2::globaltracking::RecoContainer inputTracks;
221  inputTracks.collectData(pc, *mDataRequest);
222  mTPCClusterIdxStruct = &inputTracks.inputsTPCclusters->clusterIndex;
223  mTPCRefitter = std::make_unique<o2::gpu::GPUO2InterfaceRefit>(mTPCClusterIdxStruct, mTPCTransform.get(), o2::base::Propagator::Instance()->getNominalBz(), inputTracks.getTPCTracksClusterRefs().data(), inputTracks.clusterShMapTPC.data(), nullptr, o2::base::Propagator::Instance());
224  auto tmpInputContainer = getRecoInputContainer(pc, &mChainTracking->mIOPtrs, &inputTracks, mUseMC);
225  auto tmpContainer = GPUWorkflowHelper::fillIOPtr(mChainTracking->mIOPtrs, inputTracks, mUseMC, nullptr, GTrackID::getSourcesMask("TRD"), mTrkMask, GTrackID::mask_t{GTrackID::MASK_NONE});
226  mTrackletsRaw = inputTracks.getTRDTracklets();
227  mTrackletsCalib = inputTracks.getTRDCalibratedTracklets();
228  mTPCTracksArray = inputTracks.getTPCTracks();
229  if (GTrackID::includesDet(GTrackID::DetID::ITS, mTrkMask)) {
230  // load ITS tracks and clusters needed for the refit
231  mITSTracksArray = inputTracks.getITSTracks();
232  mITSTrackClusIdx = inputTracks.getITSTracksClusterRefs();
233  mITSABRefsArray = inputTracks.getITSABRefs();
234  mITSABTrackClusIdx = inputTracks.getITSABClusterRefs();
235  const auto clusITS = inputTracks.getITSClusters();
236  const auto patterns = inputTracks.getITSClustersPatterns();
237  auto pattIt = patterns.begin();
238  mITSClustersArray.clear();
239  mITSClustersArray.reserve(clusITS.size());
240  o2::its::ioutils::convertCompactClusters(clusITS, pattIt, mITSClustersArray, mITSDict);
241  }
242 
243  LOGF(info, "There are %i tracklets in total from %i trigger records", mChainTracking->mIOPtrs.nTRDTracklets, mChainTracking->mIOPtrs.nTRDTriggerRecords);
244  LOGF(info, "As input seeds are available: %i ITS-TPC matched tracks and %i TPC tracks", mChainTracking->mIOPtrs.nTracksTPCITSO2, mChainTracking->mIOPtrs.nOutputTracksTPCO2);
245 
246  std::vector<o2::MCCompLabel> matchLabelsITSTPC;
247  std::vector<o2::MCCompLabel> trdLabelsITSTPC;
248  std::vector<o2::MCCompLabel> matchLabelsTPC;
249  std::vector<o2::MCCompLabel> trdLabelsTPC;
250  gsl::span<const o2::MCCompLabel> tpcTrackLabels;
251  gsl::span<const o2::MCCompLabel> itstpcTrackLabels;
252  if (mUseMC) {
253  if (GTrackID::includesSource(GTrackID::Source::ITSTPC, mTrkMask)) {
254  itstpcTrackLabels = inputTracks.getTPCITSTracksMCLabels();
255  }
256  if (GTrackID::includesSource(GTrackID::Source::TPC, mTrkMask)) {
257  tpcTrackLabels = inputTracks.getTPCTracksMCLabels();
258  }
259  }
260 
261  mTracker->Reset();
262  updateTimeDependentParams();
263  mRec->PrepareEvent();
264  mRec->SetupGPUProcessor(mTracker, true);
265 
266  // check trigger record filter setting
267  bool foundFilteredTrigger = false;
268  for (int iTrig = 0; iTrig < mChainTracking->mIOPtrs.nTRDTriggerRecords; ++iTrig) {
269  if (mChainTracking->mIOPtrs.trdTrigRecMask[iTrig] == 0) {
270  foundFilteredTrigger = true;
271  }
272  LOGF(debug, "TRD trigger %i added with time %f", iTrig, mChainTracking->mIOPtrs.trdTriggerTimes[iTrig]);
273  }
274  if (!foundFilteredTrigger && mTrigRecFilter) {
275  static bool warningSent = false;
276  if (!warningSent) {
277  LOG(warning) << "Trigger filtering requested, but no TRD trigger is actually masked. Can be that none needed to be masked or that the setting was not active for the tracklet transformer";
278  warningSent = true;
279  }
280  } else if (foundFilteredTrigger && !mTrigRecFilter) {
281  LOG(error) << "Trigger filtering is not requested, but masked TRD triggers are found. Rerun tracklet transformer without trigger filtering";
282  }
283 
284  // load input tracks
285  LOG(debug) << "Start loading input seeds into TRD tracker";
286  int nTracksLoadedITSTPC = 0;
287  int nTracksLoadedTPC = 0;
288  // load ITS-TPC matched tracks
289  for (int iTrk = 0; iTrk < mChainTracking->mIOPtrs.nTracksTPCITSO2; ++iTrk) {
290  const auto& trkITSTPC = mChainTracking->mIOPtrs.tracksTPCITSO2[iTrk];
291  GPUTRDTracker::HelperTrackAttributes trkAttribs;
292  trkAttribs.mTime = trkITSTPC.getTimeMUS().getTimeStamp();
293  trkAttribs.mTimeAddMax = trkITSTPC.getTimeMUS().getTimeStampError() * mRec->GetParam().rec.trd.nSigmaTerrITSTPC;
294  trkAttribs.mTimeSubMax = trkITSTPC.getTimeMUS().getTimeStampError() * mRec->GetParam().rec.trd.nSigmaTerrITSTPC;
295  GPUTRDTrack trkLoad(trkITSTPC);
296  auto trackGID = GTrackID(iTrk, GTrackID::ITSTPC);
297  if (mTracker->LoadTrack(trkLoad, trackGID.getRaw(), true, &trkAttribs)) {
298  continue;
299  }
300  ++nTracksLoadedITSTPC;
301  LOGF(debug, "Loaded ITS-TPC track %i with time %f. Window from %f to %f", nTracksLoadedITSTPC, trkAttribs.mTime, trkAttribs.mTime - trkAttribs.mTimeSubMax, trkAttribs.mTime + trkAttribs.mTimeAddMax);
302  }
303  // load TPC-only tracks
304  for (int iTrk = 0; iTrk < mChainTracking->mIOPtrs.nOutputTracksTPCO2; ++iTrk) {
305  if (mChainTracking->mIOPtrs.tpcLinkITS && mChainTracking->mIOPtrs.tpcLinkITS[iTrk] != -1) {
306  // this TPC tracks has already been matched to ITS and the ITS-TPC track has already been loaded in the tracker
307  continue;
308  }
309  const auto& trkTpc = mChainTracking->mIOPtrs.outputTracksTPCO2[iTrk];
310  GPUTRDTracker::HelperTrackAttributes trkAttribs;
311  trkAttribs.mTime = trkTpc.getTime0() * mTPCTBinMUS;
312  trkAttribs.mTimeAddMax = trkTpc.getDeltaTFwd() * mTPCTBinMUS;
313  trkAttribs.mTimeSubMax = trkTpc.getDeltaTBwd() * mTPCTBinMUS;
314  if (trkTpc.hasASideClustersOnly()) {
315  trkAttribs.mSide = -1;
316  } else if (trkTpc.hasCSideClustersOnly()) {
317  trkAttribs.mSide = 1;
318  }
319  GPUTRDTrack trkLoad(trkTpc);
320  auto trackGID = GTrackID(iTrk, GTrackID::TPC);
321  if (mTracker->LoadTrack(trkLoad, trackGID.getRaw(), true, &trkAttribs)) {
322  continue;
323  }
324  ++nTracksLoadedTPC;
325  LOGF(debug, "Loaded TPC track %i with time %f. Window from %f to %f", nTracksLoadedTPC, trkAttribs.mTime, trkAttribs.mTime - trkAttribs.mTimeSubMax, trkAttribs.mTime + trkAttribs.mTimeAddMax);
326  }
327  LOGF(info, "%i tracks are loaded into the TRD tracker. Out of those %i ITS-TPC tracks and %i TPC tracks", nTracksLoadedITSTPC + nTracksLoadedTPC, nTracksLoadedITSTPC, nTracksLoadedTPC);
328 
329  // start the tracking
330  //mTracker->DumpTracks();
331  mChainTracking->DoTRDGPUTracking<GPUTRDTrackerKernels::o2Version>(mTracker);
332  //mTracker->DumpTracks();
333 
334  // finished tracking, now collect the output
335  std::vector<TrackTRD> tracksOutITSTPC;
336  std::vector<TrackTRD> tracksOutTPC;
337  std::vector<TrackTriggerRecord> trackTrigRecITSTPC;
338  std::vector<TrackTriggerRecord> trackTrigRecTPC;
339  GPUTRDTrack* tracksOutRaw = mTracker->Tracks();
340  std::vector<unsigned int> trackIdxArray(mTracker->NTracks()); // track indices sorted by trigger record index
341  std::iota(trackIdxArray.begin(), trackIdxArray.end(), 0);
342  std::sort(trackIdxArray.begin(), trackIdxArray.end(), [tracksOutRaw](int lhs, int rhs) { return tracksOutRaw[lhs].getCollisionId() < tracksOutRaw[rhs].getCollisionId(); });
343 
344  int nTrackletsAttached = 0; // only used for debug information
345  int nTracksFailedTPCTRDRefit = 0;
346  int nTracksFailedITSTPCTRDRefit = 0;
347  for (int iTrk = 0; iTrk < mTracker->NTracks(); ++iTrk) {
348  const auto& trdTrack = mTracker->Tracks()[trackIdxArray[iTrk]];
349  if (trdTrack.getCollisionId() < 0) {
350  // skip tracks without TRD tracklets (the collision ID for the TRD tracks is initialized to -1 and only changed if a tracklet is attached to the track)
351  continue;
352  }
353  if (mStrict && (trdTrack.getIsAmbiguous() || trdTrack.getReducedChi2() > mTracker->Param().rec.trd.chi2StrictCut)) {
354  // skip tracks which have another hypothesis close to the best one or which do are above strict chi2 threshold
355  continue;
356  }
357  nTrackletsAttached += trdTrack.getNtracklets();
358  auto trackGID = trdTrack.getRefGlobalTrackId();
359  if (trackGID.includesDet(GTrackID::Source::ITS)) {
360  // this track is from an ITS-TPC seed
361  tracksOutITSTPC.push_back(trdTrack);
362  if (!refitITSTPCTRDTrack(tracksOutITSTPC.back(), mChainTracking->mIOPtrs.trdTriggerTimes[trdTrack.getCollisionId()], &inputTracks)) {
363  tracksOutITSTPC.pop_back();
364  ++nTracksFailedITSTPCTRDRefit;
365  continue;
366  }
367  if (mUseMC) {
368  fillMCTruthInfo(trdTrack, itstpcTrackLabels[trackGID], trdLabelsITSTPC, matchLabelsITSTPC, inputTracks.getTRDTrackletsMCLabels());
369  }
370  } else {
371  // this track is from a TPC-only seed
372  tracksOutTPC.push_back(trdTrack);
373  if (!refitTPCTRDTrack(tracksOutTPC.back(), mChainTracking->mIOPtrs.trdTriggerTimes[trdTrack.getCollisionId()], &inputTracks)) {
374  tracksOutTPC.pop_back();
375  ++nTracksFailedTPCTRDRefit;
376  continue;
377  }
378  if (mUseMC) {
379  fillMCTruthInfo(trdTrack, tpcTrackLabels[trackGID], trdLabelsTPC, matchLabelsTPC, inputTracks.getTRDTrackletsMCLabels());
380  }
381  }
382  }
383 
384  fillTrackTriggerRecord(tracksOutITSTPC, trackTrigRecITSTPC, tmpInputContainer->mTriggerRecords);
385  fillTrackTriggerRecord(tracksOutTPC, trackTrigRecTPC, tmpInputContainer->mTriggerRecords);
386 
387  LOGF(info, "The TRD tracker found %lu tracks from TPC seeds and %lu tracks from ITS-TPC seeds and attached in total %i tracklets out of %i",
388  tracksOutTPC.size(), tracksOutITSTPC.size(), nTrackletsAttached, mChainTracking->mIOPtrs.nTRDTracklets);
389  LOGF(info, "Number of tracks failed in the refit: TPC-TRD (%i), ITS-TPC-TRD (%i)", nTracksFailedTPCTRDRefit, nTracksFailedITSTPCTRDRefit);
390 
391  uint32_t ss = o2::globaltracking::getSubSpec(mStrict ? o2::globaltracking::MatchingType::Strict : o2::globaltracking::MatchingType::Standard);
392  if (GTrackID::includesSource(GTrackID::Source::ITSTPC, mTrkMask)) {
393  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MATCH_ITSTPC", 0, Lifetime::Timeframe}, tracksOutITSTPC);
394  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "TRGREC_ITSTPC", 0, Lifetime::Timeframe}, trackTrigRecITSTPC);
395  if (mUseMC) {
396  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MCLB_ITSTPC", 0, Lifetime::Timeframe}, matchLabelsITSTPC);
397  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MCLB_ITSTPC_TRD", 0, Lifetime::Timeframe}, trdLabelsITSTPC);
398  }
399  }
400  if (GTrackID::includesSource(GTrackID::Source::TPC, mTrkMask)) {
401  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MATCH_TPC", ss, Lifetime::Timeframe}, tracksOutTPC);
402  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "TRGREC_TPC", ss, Lifetime::Timeframe}, trackTrigRecTPC);
403  if (mUseMC) {
404  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MCLB_TPC", ss, Lifetime::Timeframe}, matchLabelsTPC);
405  pc.outputs().snapshot(Output{o2::header::gDataOriginTRD, "MCLB_TPC_TRD", ss, Lifetime::Timeframe}, trdLabelsTPC);
406  }
407  }
408 
409  mTimer.Stop();
410 }

412 bool TRDGlobalTracking::refitITSTPCTRDTrack(TrackTRD& trk, float timeTRD, o2::globaltracking::RecoContainer* recoCont)
413 {
414  auto propagator = o2::base::Propagator::Instance();
415 
416  // refit ITS-TPC-TRD track outwards to outermost TRD space point (start with ITS outer track parameters)
417  auto& outerParam = trk.getOuterParam();
418  auto detRefs = recoCont->getSingleDetectorRefs(trk.getRefGlobalTrackId());
419  int nCl = -1, clEntry = -1, nClRefit = 0, clRefs[14];
420  float chi2Out = 0;
421  auto geom = o2::its::GeometryTGeo::Instance();
422 
423  if (detRefs[GTrackID::ITS].isIndexSet()) { // this is ITS track
424  const auto& trkITS = mITSTracksArray[detRefs[GTrackID::ITS]];
425  outerParam = trkITS.getParamOut();
426  nCl = trkITS.getNumberOfClusters();
427  clEntry = trkITS.getFirstClusterEntry();
428  chi2Out = trkITS.getChi2();
429  for (int icl = 0; icl < nCl; icl++) { // clusters are stored from outer to inner layers
430  clRefs[icl] = mITSTrackClusIdx[clEntry + icl]; // from outer to inner layer
431  }

437  // refit

439  const auto& clus = mITSClustersArray[clRefs[nCl - icl - 1] = mITSABTrackClusIdx[clEntry + icl]]; // register in clRefs from outer to inner layer
440  if (!outerParam.rotate(geom->getSensorRefAlpha(clus.getSensorID())) ||
441  !propagator->propagateToX(outerParam, clus.getX(), propagator->getNominalBz(), o2::base::Propagator::MAX_SIN_PHI, o2::base::Propagator::MAX_STEP, o2::base::Propagator::MatCorrType::USEMatCorrLUT)) {
442  break;
443  }
444  chi2Out += outerParam.getPredictedChi2(clus);
445  if (!outerParam.update(clus)) {
446  break;
447  }
448  nClRefit++;
449  }
450  if (nClRefit != nCl) {
451  LOG(debug) << "ITS-AB refit outward failed";
452  return false;
453  }
454  }
455 
456  int retVal = mTPCRefitter->RefitTrackAsTrackParCov(outerParam, mTPCTracksArray[detRefs[GTrackID::TPC]].getClusterRef(), timeTRD * mTPCTBinMUSInv, &chi2Out, true, false); // outward refit
457  if (retVal < 0) {
458  LOG(debug) << "TPC refit outwards failed";
459  return false;
460  }
461  if (!refitTRDTrack(trk, chi2Out, false)) {
462  LOG(debug) << "TRD refit outwards failed";
463  return false;
464  }
465 
466  // refit ITS-TPC-TRD track inwards to innermost ITS cluster
467  // here we also calculate the LT integral for matching to TOF
468  float chi2In = 0.f;
469  if (!refitTRDTrack(trk, chi2In, true)) {
470  LOG(debug) << "TRD refit inwards failed";
471  return false;
472  }
473  auto posStart = trk.getXYZGlo();
474  retVal = mTPCRefitter->RefitTrackAsTrackParCov(trk, mTPCTracksArray[detRefs[GTrackID::TPC]].getClusterRef(), timeTRD * mTPCTBinMUSInv, &chi2In, false, false); // inward refit
475  if (retVal < 0) {
476  LOG(debug) << "TPC refit inwards failed";
477  return false;
478  }
479  auto posEnd = trk.getXYZGlo();
480  // account path integrals
481  float dX = posEnd.x() - posStart.x(), dY = posEnd.y() - posStart.y(), dZ = posEnd.z() - posStart.z(), d2XY = dX * dX + dY * dY;
482  if (std::abs(o2::base::Propagator::Instance()->getNominalBz()) > 0.01) { // circular arc = 2*R*asin(dXY/2R)
483  float b[3];
484  o2::math_utils::Point3D<float> posAv(0.5 * (posEnd.x() + posStart.x()), 0.5 * (posEnd.y() + posStart.y()), 0.5 * (posEnd.z() + posStart.z()));

486  float curvH = std::abs(0.5f * trk.getCurvature(b[2])), arcXY = 1. / curvH * std::asin(curvH * std::sqrt(d2XY));
487  d2XY = arcXY * arcXY;
488  }
489  auto lInt = std::sqrt(d2XY + dZ * dZ);
490  trk.getLTIntegralOut().addStep(lInt, trk.getP2Inv());
491  // trk.getLTIntegralOut().addX2X0(lInt * mTPCmeanX0Inv); // do we need to account for the material budget here? probably
492 
493  for (int icl = 0; icl < nCl; icl++) {
494  const auto& clus = mITSClustersArray[clRefs[icl]];
495  if (!trk.rotate(geom->getSensorRefAlpha(clus.getSensorID())) ||
496  // note: here we also calculate the L,T integral (in the inward direction, but this is irrelevant)
497  // note: we should eventually use TPC pid in the refit (TODO)
498  // note: since we are at small R, we can use field BZ component at origin rather than 3D field
499  !propagator->propagateToX(trk, clus.getX(), propagator->getNominalBz(), o2::base::Propagator::MAX_SIN_PHI, o2::base::Propagator::MAX_STEP, o2::base::Propagator::MatCorrType::USEMatCorrLUT, &trk.getLTIntegralOut())) {
500  break;
501  }
502  chi2In += trk.getPredictedChi2(clus);
503  if (!trk.update(clus)) {
504  break;
505  }
506  nClRefit++;
507  }
508  if (nClRefit != nCl) {
509  LOG(debug) << "ITS refit inwards failed";
510  return false;
511  }
512  // We need to update the LTOF integral by the distance to the "primary vertex"
513  // We want to leave the track at the the position of its last update, so we do a fast propagation on the TrackPar copy of trfit,
514  // and since for the LTOF calculation the material effects are irrelevant, we skip material corrections
515  const o2::dataformats::VertexBase vtxDummy; // at the moment using dummy vertex: TODO use MeanVertex constraint instead
516  o2::track::TrackPar trkPar(trk);
517  if (!propagator->propagateToDCA(vtxDummy.getXYZ(), trkPar, propagator->getNominalBz(), o2::base::Propagator::MAX_STEP, o2::base::Propagator::MatCorrType::USEMatCorrNONE, nullptr, &trk.getLTIntegralOut())) {
518  LOG(error) << "LTOF integral might be incorrect";
519  }
520  return true;

575 bool TRDGlobalTracking::refitTRDTrack(TrackTRD& trk, float& chi2, bool inwards)
576 {
577  auto propagator = o2::base::Propagator::Instance();
578  int lyStart = inwards ? NLAYER - 1 : 0;
579  int direction = inwards ? -1 : 1;
580  int lyEnd = inwards ? -1 : NLAYER;
581  o2::track::TrackParCov* trkParam = inwards ? &trk : &trk.getOuterParam();
582  o2::track::TrackLTIntegral* tofL = inwards ? &trk.getLTIntegralOut() : nullptr;
583  for (int iLy = lyStart; iLy != lyEnd; iLy += direction) {
584  int trkltId = trk.getTrackletIndex(iLy);
585  if (trkltId < 0) {
586  continue;
587  }
588  int trkltDet = mTrackletsRaw[trkltId].getDetector();
589  int trkltSec = trkltDet / (NLAYER * NSTACK);
590  if (trkltSec != o2::math_utils::angle2Sector(trkParam->getAlpha())) {
591  if (!trkParam->rotate(o2::math_utils::sector2Angle(trkltSec))) {
592  LOGF(debug, "Track at alpha=%.2f could not be rotated in tracklet coordinate system with alpha=%.2f", trkParam->getAlpha(), o2::math_utils::sector2Angle(trkltSec));
593  return false;
594  }
595  }
596  if (!propagator->PropagateToXBxByBz(*trkParam, mTrackletsCalib[trkltId].getX(), o2::base::Propagator::MAX_SIN_PHI, o2::base::Propagator::MAX_STEP, o2::base::Propagator::MatCorrType::USEMatCorrNONE, tofL)) {
597  LOGF(debug, "Track propagation failed in layer %i (pt=%f, xTrk=%f, xToGo=%f)", iLy, trkParam->getPt(), trkParam->getX(), mTrackletsCalib[trkltId].getX());
598  return false;
599  }
600  const PadPlane* pad = Geometry::instance()->getPadPlane(trkltDet);
601  float tilt = tan(TMath::DegToRad() * pad->getTiltingAngle()); // tilt is signed! and returned in degrees
602  float tiltCorrUp = tilt * (mTrackletsCalib[trkltId].getZ() - trkParam->getZ());
603  float zPosCorrUp = mTrackletsCalib[trkltId].getZ() + mRecoParam.getZCorrCoeffNRC() * trkParam->getTgl();
604  float padLength = pad->getRowSize(mTrackletsRaw[trkltId].getPadRow());
605  if (!((trkParam->getSigmaZ2() < (padLength * padLength / 12.f)) && (std::fabs(mTrackletsCalib[trkltId].getZ() - trkParam->getZ()) < padLength))) {
606  tiltCorrUp = 0.f;
607  }
608 
609  std::array<float, 2> trkltPosUp{mTrackletsCalib[trkltId].getY() - tiltCorrUp, zPosCorrUp};
610  std::array<float, 3> trkltCovUp;
611  mRecoParam.recalcTrkltCov(tilt, trkParam->getSnp(), pad->getRowSize(mTrackletsRaw[trkltId].getPadRow()), trkltCovUp);
612 
613  chi2 += trkParam->getPredictedChi2(trkltPosUp, trkltCovUp);
614  if (!trkParam->update(trkltPosUp, trkltCovUp)) {
615  LOGF(debug, "Failed to update track with space point in layer %i", iLy);
616  return false;
617  }
618  }
619  return true;
620 }
