42 DataProcessorSpec getDigitsReaderSpec(bool propagateMC)
43 {
44 
45  auto initFunction = [propagateMC](InitContext& ic) {
46  // get the option from the init context
47  auto filename = o2::utils::Str::concat_string(o2::utils::Str::rectifyDirectory(ic.options().get<std::string>("input-dir")),
48  ic.options().get<std::string>("infile"));
49  auto treename = ic.options().get<std::string>("treename");
50  auto nofEvents = ic.options().get<int>("nevents");
51  auto publishingMode = nofEvents == -1 ? RootTreeReader::PublishingMode::Single : RootTreeReader::PublishingMode::Loop;
52 
53  auto processAttributes = std::make_shared<ProcessAttributes>();
54  {
55  processAttributes->terminateOnEod = ic.options().get<bool>("terminate-on-eod");
56  processAttributes->finished = false;
57  processAttributes->datatype = "PHOSDigit";
58  constexpr auto persistency = Lifetime::Timeframe;
59  o2::header::DataHeader::SubSpecificationType subSpec = 0;
60  if (propagateMC) {
61  processAttributes->reader = std::make_shared<RootTreeReader>(treename.c_str(), // tree name
62  filename.c_str(), // input file name
63  nofEvents, // number of entries to publish
64  publishingMode,
65  Output{"PHS", "DIGITS", subSpec, persistency},
66  "PHOSDigit", // name of data branch
67  Output{"PHS", "DIGITTRIGREC", subSpec, persistency},
68  "PHOSDigitTrigRecords", // name of data triggerrecords branch
69  Output{"PHS", "DIGITSMCTR", subSpec, persistency},
70  "PHOSDigitMCTruth"); // name of mc label branch
71  } else {
72  processAttributes->reader = std::make_shared<RootTreeReader>(treename.c_str(), // tree name
73  filename.c_str(), // input file name
74  nofEvents, // number of entries to publish
75  publishingMode,
76  Output{"PHS", "DIGITS", subSpec, persistency},
77  "PHOSDigit", // name of data branch
78  Output{"PHS", "DIGITTRIGREC", subSpec, persistency},
79  "PHOSDigitTrigRecords"); // name of data triggerrecords branch
80  }
81  }
82 
83  auto processFunction = [processAttributes, propagateMC](ProcessingContext& pc) {
84  if (processAttributes->finished) {
85  return;
86  }
87 
88  auto publish = [&processAttributes, &pc, propagateMC]() {
89  o2::phos::PHOSBlockHeader phosheader(true);
90  if (processAttributes->reader->next()) {
91  (*processAttributes->reader)(pc, phosheader);
92  } else {
93  processAttributes->reader.reset();
94  return false;
95  }
96  return true;
97  };
98 
99  bool active(true);
100  if (!publish()) {
101  active = false;
102  // Send dummy header with no payload option
103  o2::phos::PHOSBlockHeader dummyheader(false);
104  pc.outputs().snapshot(OutputRef{"output", 0, {dummyheader}}, 0);
105  pc.outputs().snapshot(OutputRef{"outputTR", 0, {dummyheader}}, 0);
106  if (propagateMC) {
107  pc.outputs().snapshot(OutputRef{"outputMC", 0, {dummyheader}}, 0);
108  }
109  }
110  if ((processAttributes->finished = (active == false)) && processAttributes->terminateOnEod) {
111  pc.services().get<ControlService>().endOfStream();
112  pc.services().get<ControlService>().readyToQuit(framework::QuitRequest::Me);
113  }
114  };
115  return processFunction;
116  };
117 
118  std::vector<OutputSpec> outputSpecs;
119  outputSpecs.emplace_back(OutputSpec{{"output"}, "PHS", "DIGITS", 0, Lifetime::Timeframe});
120  outputSpecs.emplace_back(OutputSpec{{"outputTR"}, "PHS", "DIGITTRIGREC", 0, Lifetime::Timeframe});
121  if (propagateMC) {
122  outputSpecs.emplace_back(OutputSpec{{"outputMC"}, "PHS", "DIGITSMCTR", 0, Lifetime::Timeframe});
123  }
124 
125  return DataProcessorSpec{
126  "phos-digit-reader",
127  Inputs{}, // no inputs
128  outputSpecs,
129  AlgorithmSpec(initFunction),
130  Options{
131  {"infile", VariantType::String, "phosdigits.root", {"Name of the input file"}},
132  {"input-dir", VariantType::String, "none", {"Input directory"}},
133  {"treename", VariantType::String, "o2sim", {"Name of input tree"}},
134  {"nevents", VariantType::Int, -1, {"number of events to run, -1: inf loop"}},
135  {"terminate-on-eod", VariantType::Bool, true, {"terminate on end-of-data"}},
136  }};
137 }
