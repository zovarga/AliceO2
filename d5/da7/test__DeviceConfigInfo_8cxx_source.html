24 template <typename T>
25 std::string arrayPrinter(boost::property_tree::ptree const& tree)
26 {
27  std::stringstream ss;
28  int size = tree.size();
29  int count = 0;
30  ss << o2::framework::variant_array_symbol<T>::symbol << "[";
31  for (auto& element : tree) {
32  ss << element.second.get_value<T>();
33  if (count < size - 1) {
34  ss << ",";
35  }
36  ++count;
37  }
38  ss << "]";
39  return ss.str();
40 }
