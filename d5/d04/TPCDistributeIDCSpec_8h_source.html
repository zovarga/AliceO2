60  void init(o2::framework::InitContext& ic) final
61  {
62  if (mLoadFromFile) {
63  TFile fInp("IDCGroup.root", "READ");
64  for (TObject* keyAsObj : *fInp.GetListOfKeys()) {
65  const auto key = dynamic_cast<TKey*>(keyAsObj);
66  LOGP(info, "Key name: {} Type: {}", key->GetName(), key->GetClassName());
67 
68  if (std::strcmp(o2::tpc::IDCAverageGroup<o2::tpc::IDCAverageGroupCRU>::Class()->GetName(), key->GetClassName()) != 0) {
69  LOGP(info, "skipping object. wrong class.");
70  continue;
71  }
72  IDCAverageGroup<IDCAverageGroupCRU>* idcavg = (IDCAverageGroup<IDCAverageGroupCRU>*)fInp.Get(key->GetName());
73  unsigned int cru = idcavg->getSector() * Mapper::NREGIONS + idcavg->getRegion();
74  const std::vector<float>& idcData = idcavg->getIDCGroup().getData();
75  const std::vector<float>& idc1D = idcavg->getIDCGroup().get1DIDCs();
76 
77  for (auto& idcbuffer : mIDCs) {
78  const auto nIDCS = idcData.size();
79  // TODO use memcyp etc here
80  for (auto& relTF : idcbuffer[cru]) {
81  relTF.reserve(nIDCS);
82  for (int i = 0; i < nIDCS; ++i) {
83  relTF.emplace_back(idcData[i]);
84  }
85  }
86  }
87 
88  for (auto& idcbuffer : mOneDIDCs) {
89  const auto nIDCS = idc1D.size();
90  for (auto& relTF : idcbuffer[cru]) {
91  relTF.reserve(nIDCS);
92  for (int i = 0; i < nIDCS; ++i) {
93  relTF.emplace_back(idc1D[i]);
94  }
95  }
96  }
97  delete idcavg;
98  }
99  }
100  }

102  void run(o2::framework::ProcessingContext& pc) final
103  {
104  // check which buffer to use for current incoming data
105  const auto tf = getCurrentTF(pc);
106  const bool currentBuffer = (tf > mTFEnd[mBuffer]) ? !mBuffer : mBuffer;
107  const unsigned int currentOutLane = (tf > mTFEnd[mBuffer]) ? (mCurrentOutLane + 1) % mOutLanes : mCurrentOutLane;
108  const auto relTF = tf - mTFStart[currentBuffer];
109 
110  if (!mLoadFromFile) {
111  for (auto& ref : InputRecordWalker(pc.inputs(), mFilter)) {
112  ++mProcessedCRU[currentBuffer][relTF];
113  auto const* tpcCRUHeader = o2::framework::DataRefUtils::getHeader<o2::header::DataHeader*>(ref);
114  const int cru = tpcCRUHeader->subSpecification >> 7;
115  const auto descr = tpcCRUHeader->dataDescription;
116  if (TPCFLPIDCDevice<TPCFLPIDCDeviceGroup>::getDataDescriptionIDCGroup() == descr) {
117  mIDCs[currentBuffer][cru][relTF] = std::move(pc.inputs().get<pmr::vector<float>>(ref));
118  } else {
119  mOneDIDCs[currentBuffer][cru][relTF] = std::move(pc.inputs().get<pmr::vector<float>>(ref));
120  }
121  }
122  }
123 
124  // check if all CRUs for current TF are already aggregated and send data
125  if (mProcessedCRU[currentBuffer][relTF] == 2 * mCRUs.size() || mLoadFromFile) {
126  ++mProcessedTFs[currentBuffer];
127  sendOutput(pc, currentOutLane, currentBuffer, relTF);
128  }
129 
130  if (mProcessedTFs[currentBuffer] == mTimeFrames) {
131  mProcessedTFs[currentBuffer] = 0; // reset processed TFs for next aggregation interval
132  std::fill(mProcessedCRU[currentBuffer].begin(), mProcessedCRU[currentBuffer].end(), 0);
133 
134  // set integration range for next integration interval
135  mTFStart[mBuffer] = mTFEnd[!mBuffer] + 1;
136  mTFEnd[mBuffer] = mTFStart[mBuffer] + mTimeFrames - 1;
137 
138  // switch buffer
139  mBuffer = !mBuffer;
140 
141  // set output lane
142  mCurrentOutLane = ++mCurrentOutLane % mOutLanes;
143  }
144  }

174  void sendOutput(o2::framework::ProcessingContext& pc, const unsigned int currentOutLane, const bool currentBuffer, const unsigned int relTF)
175  {
176  // send output data for one TF for all CRUs
177  if (!mLoadFromFile) {
178  for (unsigned int i = 0; i < mCRUs.size(); ++i) {
179  pc.outputs().adoptContainer(Output{gDataOriginTPC, TPCDistributeIDCSpec::getDataDescriptionIDC(), header::DataHeader::SubSpecificationType{mCRUs[i] + currentOutLane * CRU::MaxCRU}}, std::move(mIDCs[currentBuffer][mCRUs[i]][relTF]));
180  pc.outputs().adoptContainer(Output{gDataOriginTPC, TPCDistributeIDCSpec::getDataDescription1DIDC(), header::DataHeader::SubSpecificationType{mCRUs[i]}}, std::move(mOneDIDCs[currentBuffer][mCRUs[i]][relTF]));
181  }
182  } else {
183  for (unsigned int i = 0; i < mCRUs.size(); ++i) {
184  pc.outputs().snapshot(Output{gDataOriginTPC, TPCDistributeIDCSpec::getDataDescriptionIDC(), header::DataHeader::SubSpecificationType{mCRUs[i] + currentOutLane * CRU::MaxCRU}}, mIDCs[currentBuffer][mCRUs[i]][relTF]);
185  pc.outputs().snapshot(Output{gDataOriginTPC, TPCDistributeIDCSpec::getDataDescription1DIDC(), header::DataHeader::SubSpecificationType{mCRUs[i]}}, mOneDIDCs[currentBuffer][mCRUs[i]][relTF]);
186  }
187  }
188  }

191 DataProcessorSpec getTPCDistributeIDCSpec(const std::vector<uint32_t>& crus, const unsigned int timeframes, const unsigned int outlanes, const unsigned int firstTF, const bool loadFromFile)
192 {
193  std::vector<InputSpec> inputSpecs;
194  if (!loadFromFile) {
195  inputSpecs.emplace_back(InputSpec{"idcsgroup", ConcreteDataTypeMatcher{gDataOriginTPC, TPCFLPIDCDevice<TPCFLPIDCDeviceGroup>::getDataDescriptionIDCGroup()}, Lifetime::Timeframe});
196  inputSpecs.emplace_back(InputSpec{"1didc", ConcreteDataTypeMatcher{gDataOriginTPC, TPCFLPIDCDevice<TPCFLPIDCDeviceGroup>::getDataDescription1DIDC()}, Lifetime::Timeframe});
197  }
198 
199  std::vector<OutputSpec> outputSpecs;
200  outputSpecs.reserve((outlanes + 1) * crus.size());
201  for (int lane = 0; lane < outlanes; ++lane) {
202  for (const auto cru : crus) {
203  const header::DataHeader::SubSpecificationType subSpec{cru + lane * CRU::MaxCRU};
204  outputSpecs.emplace_back(ConcreteDataMatcher{gDataOriginTPC, TPCDistributeIDCSpec::getDataDescriptionIDC(), subSpec});
205  }
206  }
207 
208  for (const auto cru : crus) {
209  outputSpecs.emplace_back(ConcreteDataMatcher{gDataOriginTPC, TPCDistributeIDCSpec::getDataDescription1DIDC(), header::DataHeader::SubSpecificationType{cru}});
210  }
211 
212  return DataProcessorSpec{
213  "tpc-distribute-idc",
214  inputSpecs,
215  outputSpecs,
216  AlgorithmSpec{adaptFromTask<TPCDistributeIDCSpec>(crus, timeframes, outlanes, loadFromFile, firstTF)}};
217 }
